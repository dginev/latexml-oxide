1//! latexmlpost_oxide — Post-process LaTeXML XML output.
2//!
3//! Usage: latexmlpost_oxide [--pmml] [--keepXMath] [--noscan] [--nocrossref]
4//!        [--stylesheet path.xsl] [--dest output.xml] input.xml
5
6use latexml_post::Post;
7use latexml_post::document::{PostDocument, PostDocumentOptions};
8use latexml_post::mathml::MathML;
9use latexml_post::processor::Processor;
10use latexml_post::xslt::XSLT;
11use std::collections::HashMap;
12
13fn main() {
14  let args: Vec<String> = std::env::args().collect();
15
16  let mut input_path = None;
17  let mut dest_path = None;
18  let mut pmml = false;
19  let mut keep_xmath = false;
20  let mut stylesheet = None;
21
22  let mut i = 1;
23  while i < args.len() {
24    match args[i].as_str() {
25      "--pmml" => pmml = true,
26      "--keepXMath" | "--xmath" => keep_xmath = true,
27      "--noscan" | "--nocrossref" => {}, // accepted but no-op for now
28      "--stylesheet" => {
29        i += 1;
30        if i < args.len() {
31          stylesheet = Some(args[i].clone());
32        }
33      },
34      "--dest" => {
35        i += 1;
36        if i < args.len() {
37          dest_path = Some(args[i].clone());
38        }
39      },
40      arg if !arg.starts_with('-') => input_path = Some(arg.to_string()),
41      other => eprintln!("Warning: unknown option '{}'", other),
42    }
43    i += 1;
44  }
45
46  let input_path = input_path.unwrap_or_else(|| {
47    eprintln!("Usage: latexmlpost_oxide [--pmml] [--keepXMath] [--stylesheet path.xsl] [--dest output] input.xml");
48    std::process::exit(1);
49  });
50
51  // Default to pmml if nothing specified
52  if !pmml && stylesheet.is_none() {
53    pmml = true;
54  }
55
56  let input = std::fs::read_to_string(&input_path).unwrap_or_else(|e| {
57    eprintln!("Failed to read {}: {}", input_path, e);
58    std::process::exit(1);
59  });
60
61  let doc =
62    PostDocument::new_from_string(&input, PostDocumentOptions::default()).unwrap_or_else(|e| {
63      eprintln!("Failed to parse {}: {}", input_path, e);
64      std::process::exit(1);
65    });
66
67  let mut post = Post::new();
68  let mut processors: Vec<Box<dyn Processor>> = Vec::new();
69
70  if pmml {
71    let mathml = MathML::new_presentation().with_keep_xmath(keep_xmath);
72    processors.push(Box::new(mathml));
73  }
74
75  if let Some(ref xsl_path) = stylesheet {
76    let searchpaths = vec!["resources/XSLT".to_string(), ".".to_string()];
77    let xslt = XSLT::new(xsl_path, HashMap::new(), false, None, searchpaths).unwrap_or_else(|e| {
78      eprintln!("Failed to create XSLT processor: {}", e);
79      std::process::exit(1);
80    });
81    processors.push(Box::new(xslt));
82  }
83
84  let results = post
85    .process_chain(doc, &mut processors)
86    .unwrap_or_else(|e| {
87      eprintln!("Post-processing failed: {}", e);
88      std::process::exit(1);
89    });
90
91  let output = results[0].to_xml_string();
92
93  if let Some(dest) = dest_path {
94    std::fs::write(&dest, &output).unwrap_or_else(|e| {
95      eprintln!("Failed to write {}: {}", dest, e);
96      std::process::exit(1);
97    });
98    eprintln!("Wrote {}", dest);
99  } else {
100    print!("{}", output);
101  }
102}
