376/// Allocation-failure hook — emits a `Fatal:` line in the project's
377/// logging convention so aggregation tooling records the failure, then
378/// exits with code 137. See `cortex_worker.rs::custom_alloc_error_hook`
379/// for full rationale + witness paper.
380fn custom_alloc_error_hook(layout: Layout) {
381  eprintln!(
382    "Fatal:oom:alloc_failed allocation of {} bytes (align {}) failed; \
383     likely runaway macro expansion (gullet pushback Vec growth past \
384     worker memory budget). Exiting with code 137.",
385    layout.size(),
386    layout.align()
387  );
388  process::exit(137);
389}
390
391fn main() -> Result<(), Box<dyn Error>> {
392  set_alloc_error_hook(custom_alloc_error_hook);
393
394  // Run all work on a worker thread with a 256 MB stack so deeply
395  // nested math trees don't overflow the OS-default 8 MB main-thread
396  // stack during finalize/post-processing. See cortex_worker.rs for
397  // full rationale (sandbox 0711.4787 et al, #17).
398  std::thread::Builder::new()
399    .stack_size(256 * 1024 * 1024)
400    .spawn(|| real_main().map_err(|e| e.to_string()))
401    .expect("spawn worker thread")
402    .join()
403    .expect("worker thread panicked")
404    .map_err(|s| s.into())
405}

407fn real_main() -> Result<(), Box<dyn Error>> {
408  // Heap profiler (`--features dhat-heap`). Held for the whole conversion, which
409  // runs on this (worker) thread. The success/fatal exits below go through
410  // `process::exit`, which skips destructors, so the profile is flushed
411  // explicitly via `_dhat.take()` before those exits (writing `dhat-heap.json`);
412  // a normal `return` still drops it as a fallback.
413  #[cfg(feature = "dhat-heap")]
414  let mut _dhat = Some(dhat::Profiler::new_heap());
415
416  let wall_start = std::time::Instant::now();
417  let cli = Cli::parse();
418
419  // Kick off kpathsea pre-init in a background thread. Force-runs
420  // `kpathsea_init_db` + per-format `kpathsea_init_format` so the
421  // first real `find_file` from digest sees the fast post-init path
422  // instead of paying ~30-40 ms of setup on its first lookup. The
423  // worker briefly holds the `KPSE` Mutex while it probes — a main-
424  // thread `kpsewhich(...)` racing in early would block briefly, but
425  // dump load + arg parsing run for >50 ms before any digest-time
426  // package resolution, so the warm-up usually completes before its
427  // first real consumer arrives. Disable with
428  // `LATEXML_NO_KPATHSEA_PREWARM=1` for A/B benchmarking.
429  //
430  // This is now purely a *latency* pre-warm (overlap the init cost with dump
431  // load). The CORRECTNESS invariant — tables warm before the first `find_file`
432  // — is enforced by the shared `Converter::initialize_session`, the single path
433  // both this binary and the library (`latexml::api`, tests) funnel through, so
434  // the library can no longer drift from the binaries the way it did (the flaky
435  // spurious "1 warning" root-caused 2026-07-16).
436  let _kpse_warmup_handle = if std::env::var("LATEXML_NO_KPATHSEA_PREWARM").is_err() {
437    Some(std::thread::spawn(
438      latexml_core::util::pathname::prewarm_kpathsea,
439    ))
440  } else {
441    None
442  };
443
444  // Initialize logger with verbosity level
445  let verbosity: i32 = if cli.quiet {
446    -1
447  } else if cli.verbose {
448    1
449  } else {
450    0
451  };
452  let log_level = if !cli.debug.is_empty() {
453    // --debug NAME implies debug-level logging (Perl: Debug() output is
454    // emitted whenever the feature flag is set).
455    log::LevelFilter::Debug
456  } else {
457    match verbosity {
458      v if v < 0 => log::LevelFilter::Warn,
459      0 => log::LevelFilter::Info,
460      _ => log::LevelFilter::Debug,
461    }
462  };
463  latexml_core::util::logger::init(log_level).ok();
464  // Perl: --debug=NAME sets $LaTeXML::DEBUG{NAME}; gates DebugFeature! sites.
465  for feature in &cli.debug {
466    latexml_core::common::error::enable_debug_feature(feature);
467  }
468
469  // Dump-model mode — load the embedded LaTeXML schema, serialise to
470  // stdout in `.model` format, exit. Mirrors Perl
471  // `LaTeXML::Common::Model::compileSchema` (Model.pm L121-136). Used
472  // by tools/compileschema.sh stage 2 to regenerate `LaTeXML.model`
473  // from the same source the runtime sees.
474  if cli.dump_model {
475    print!("{}", latexml::dump_compiled_latexml_model());
476    process::exit(0);
477  }
478
479  // Codegen mode — handle early, no source file needed
480  if let Some(dump_path) = cli.codegen {
481    let output = cli.dest.unwrap_or_else(|| "latex_dump.rs".to_string());
482    match latexml::ini_tex::codegen_from_dump(&dump_path, &output) {
483      Ok(count) => {
484        eprintln!("Codegen complete: {} entries → {}", count, output);
485        process::exit(0);
486      },
487      Err(e) => {
488        eprintln!("Codegen failed: {}", e);
489        process::exit(1);
490      },
491    }
492  }
493
494  // Persistent LSP Server mode — handle early before source file checks
495  if cli.server {
496    log::info!("Starting persistent LSP server...");
497    latexml::lsp_server::run_lsp_server(cli.timeout, cli.max_memory)?;
498    process::exit(0);
499  }
500
501  // Determine source: --source > --init > positional
502  let source = if let Some(ref init) = cli.init {
503    init.clone()
504  } else if let Some(ref src) = cli.source {
505    src.clone()
506  } else {
507    match cli.source_positional {
508      Some(ref s) => s.clone(),
509      None => {
510        eprintln!("Error: no source file specified. Use: latexml_oxide [OPTIONS] <SOURCE>");
511        process::exit(1);
512      },
513    }
514  };
515  let target = cli.dest.clone();
516
517  // Resolve `--whatsout <mode>` (Perl Pack.pm `whatsout` option +
518  // Config.pm L421-439). Explicit `--whatsout` wins; otherwise a `.zip`
519  // destination extension implies `archive` (Config.pm L421-426).
520  // Unknown explicit values fall back to `document`, like Perl
521  // `pack_collection`. Hoisted here (rather than inside the post block)
522  // so both the post stage and the post-run `--log` guard can see it.
523  let dest_ext_is_zip = target
524    .as_deref()
525    .is_some_and(|t| t.to_ascii_lowercase().ends_with(".zip"));
526  let whatsout_mode = match cli.whatsout.as_deref() {
527    Some(s) => latexml_post::extract::Whatsout::from_cli(s).unwrap_or_default(),
528    None if dest_ext_is_zip => latexml_post::extract::Whatsout::Archive,
529    // `--embed` is Perl's shortcut for `--whatsout=fragment` (Config.pm L72).
530    None if cli.embed => latexml_post::extract::Whatsout::Fragment,
531    None => latexml_post::extract::Whatsout::Document,
532  };
533  let is_archive_out = whatsout_mode.is_archive();
534
535  // --whatsin=archive: extract archive to temp directory, find main .tex file
536  let mut path_flags = cli.search_paths.clone();
537  let _archive_tempdir; // hold tempdir alive for the duration of processing
538  let is_archive_mode = cli.whatsin.as_deref() == Some("archive")
539    || source.ends_with(".tar.gz")
540    || source.ends_with(".tgz")
541    || source.ends_with(".zip")
542    || source.ends_with(".tar");
543  let source = if is_archive_mode {
544    let (tempdir, main_tex) = match unpack_archive(&source) {
545      Ok(r) => r,
546      Err(e) => {
547        eprintln!("Failed to unpack archive '{}': {}", source, e);
548        process::exit(1);
549      },
550    };
551    let dir_str = tempdir.path().to_string_lossy().to_string();
552    path_flags.push(dir_str);
553    _archive_tempdir = Some(tempdir);
554    main_tex
555  } else {
556    _archive_tempdir = None;
557    source
558  };
559
560  // --whatsin=directory: auto-detect from trailing / or explicit flag
561  let is_directory_mode = cli.whatsin.as_deref() == Some("directory") || source.ends_with('/');
562  let source = if is_directory_mode {
563    let dir_path = Path::new(&source);
564    if let Ok(abs_source) = std::fs::canonicalize(dir_path) {
565      path_flags.push(abs_source.to_string_lossy().to_string());
566    } else {
567      path_flags.push(source.clone());
568    }
569    // Find the main .tex file in the directory, matching Perl's behavior
570    match latexml::main_tex::find_main_tex(dir_path) {
571      Ok(main_tex) => main_tex.to_string_lossy().to_string(),
572      Err(e) => {
573        eprintln!("Failed to find main .tex file in '{}': {}", source, e);
574        process::exit(1);
575      },
576    }
577  } else {
578    source
579  };
580
581  // Some arXiv source archives ship a PDF mis-named with a `.tex` extension
582  // (e.g. 2301.04210.tex). Perl LaTeXML detects the `%PDF-` magic and bails
583  // with a single Fatal; without this guard the binary catcode-tokenizes
584  // the PDF stream and emits ~100 Error:undefined / Error:unexpected lines.
585  if Path::new(&source).is_file() && latexml::main_tex::is_pdf_magic(Path::new(&source)) {
586    eprintln!(
587      "Fatal:invalid:not_tex_source PDF magic detected in source file '{}'",
588      source
589    );
590    process::exit(1);
591  }
592
593  // Stash a copy of the resolved main-tex path for end-of-run telemetry,
594  // since `source` itself is moved into `converter.convert(...)`.
595  let telemetry_source = source.clone();

599    None
600  } else {
601    Some(cli.preload_files.clone())
602  };

604    None
605  } else {

607  };

609  // Perl `Common/Config.pm:24,216`: `$is_bibtex = qr/(^literal:\s*@)|(\.bib$)/`.
610  // `--bibtex` forces the type; otherwise auto-detect when the
611  // source ends in `.bib` or begins with `literal:@`.

614    trimmed.trim_start().starts_with('@') && trimmed.len() < source.len()
615  };

617    Some(DigestionMode::BibTeX)
618  } else {
619    None
620  };

622  // Map `--whatsin` to the core input-chunk size (Perl Config.pm
623  // L399-404 + LaTeXML.pm:165-194). `archive`/`directory` have already
624  // been resolved to a concrete main `.tex` above, so the core digests
625  // them as a plain document; only `math`/`fragment` change the core's
626  // preamble/postamble wrapping. When `--whatsin` is unset, a supplied
627  // `--preamble`/`--postamble` implies a `fragment` input.
628  let whatsin_size = match cli.whatsin.as_deref() {
629    Some("math") => DataSize::Math,
630    Some("fragment") => DataSize::Fragment,
631    Some("document") | Some("archive") | Some("directory") => DataSize::Document,
632    None if cli.preamble.is_some() || cli.postamble.is_some() => DataSize::Fragment,
633    _ => DataSize::Document,
634  };

642    postamble: cli.postamble.clone(),

645    extra_bindings_dispatch: Some(Rc::new(latexml_contrib::dispatch)),
646    preload,

650    include_comments: if cli.nocomments {
651      Some(false)

656    },

659    include_styles: if cli.includestyles { Some(true) } else { None },
660    // `--nomathparse` disables; `--mathparse` explicitly enables (the default).

662      Some(true)
663    } else if cli.mathparse {
664      Some(false)
665    } else {
666      None
667    },

669    // tracking + emission; otherwise leave unset (off). The env reads
670    // once here, off the hot path. See `docs/performance/SOURCE_PROVENANCE.md`.
671    source_map: if cli.source_map || std::env::var_os("LATEXML_SOURCE_MAP").is_some() {
672      Some(true)
673    } else {
674      None
675    },
676    // Perl Config.pm L57 / Core.pm L60-61: --inputencoding seeds State
677    // PERL_INPUT_ENCODING, which the Mouth reads to decode source bytes
678    // (default utf-8 when unset).
679    inputencoding: cli.inputencoding.clone(),
680  };
681  // CRITICAL: must be set BEFORE `prepare_session`. `tex.rs` /
682  // `latex.rs`'s LoadFormat split (plain_bootstrap → plain_dump|base
683  // → plain_constructs and the latex equivalent) reads
684  // `LATEXML_INI_MODE` to decide whether to fully load the format
685  // or stop after the bootstrap pool. If it's not set yet,
686  // `prepare_session` pre-loads `plain_base` / `latex_base`, which
687  // pollutes the snapshot taken later in `ini_tex::dump_format` and
688  // silences the diff for everything raw plain.tex / latex.ltx defines
689  // (the `\countdef\allocationnumber=21` → `Stored::Register{...}`
690  // problem from 2026-04-26).

692    // SAFETY: setting the var before any thread is spawned. `prepare_session`
693    // and `ini_tex::dump_format` both read it but neither mutates env.
694    unsafe {
695      std::env::set_var("LATEXML_INI_MODE", "1");
696    }

699  let mut converter = Converter::from_config(opts.clone());
700  // Skip engine init for already-converted XML input: post-processing is pure
701  // libxml2 and never touches the TeX engine or its dump, so loading
702  // TeX.pool/latex + the format dump (~85–160 ms and a chunk of RSS) is wasted
703  // work. Init mode (`--init`) and every real TeX conversion still need it.
704  if (cli.init.is_some() || !is_xml_input(&source))
705    && let Err(e) = converter.prepare_session(&opts)
706  {
707    eprintln!("Could not prepare converter session: {}", e);
708    process::exit(1);
709  }

712  if cli.nobibtex {
713    // Set BIB_CONFIG to ['bbl'] — skip BibTeX, use pre-existing .bbl file
714    latexml_core::state::assign_value(
715      "BIB_CONFIG",
716      latexml_core::common::store::Stored::Strings(Rc::new([latexml_core::common::arena::pin(
717        "bbl",
718      )])),
719      Some(latexml_core::state::Scope::Global),
720    );
721  }
722  if cli.nonumbersections {
723    latexml_core::state::assign_value(
724      "no_number_sections",
725      true,
726      Some(latexml_core::state::Scope::Global),
727    );
728  } else if cli.numbersections {
729    // Positive complement (Perl `numbersections!`, default on): explicitly
730    // restore section numbering if a profile/package turned it off.
731    latexml_core::state::assign_value(
732      "no_number_sections",
733      false,
734      Some(latexml_core::state::Scope::Global),
735    );
736  }
737  // Perl Core.pm L48: DOCUMENTID value
738  if let Some(ref docid) = cli.documentid {
739    latexml_core::state::assign_value(
740      "DOCUMENTID",
741      latexml_core::common::store::Stored::String(latexml_core::common::arena::pin(docid)),
742      Some(latexml_core::state::Scope::Global),
743    );
744  }
745
746  if cli.init.is_some() {
747    // Init mode: process file and dump state
748    match latexml::ini_tex::dump_format(&mut converter, &source, target.as_deref()) {
749      Ok(count) => eprintln!("Format dump complete: {} entries written", count),
750      Err(e) => {
751        eprintln!("Format dump failed: {}", e);
752        process::exit(1);
753      },
754    }
755  } else {
756    // Normal mode: convert document
757    //
758    // Two-layer timeout: the cooperative stomach::check_timeout gives a graceful
759    // Err(Fatal) when the digestion loop can poll it, and the Watchdog forcibly
760    // aborts the process if the deadline is reached without cooperation (e.g. a
761    // tight native loop in Marpa / libxml2 / libxslt). The Watchdog cancels
762    // automatically on drop at end of main. `--max-memory` rides the same
763    // Watchdog (it was previously a silent no-op outside `--server`).
764    let _watchdog = latexml_core::watchdog::Watchdog::with_limits(
765      cli.timeout,
766      cli.max_memory.saturating_mul(1024),
767    );
768    if cli.timeout > 0 {
769      latexml_core::stomach::set_timeout(cli.timeout);
770    }
771    if let Some(limit) = cli.token_limit {
772      latexml_core::gullet::set_token_limit(Some(limit));
773    }
774
775    let source_for_post = source.clone();
776    // XML-input mode: when the source is already-converted LaTeXML XML
777    // (file extension `.xml`/`.xhtml` or content starts with `<?xml`
778    // / `<document xmlns="…">`), skip the TeX → XML converter and feed
779    // the file straight to post-processing. Mirrors what
780    // `latexmlpost_oxide` did as a separate binary (per the
781    // retirement plan in `docs/SYNC_STATUS.md`).
782    let response = if is_xml_input(&source) {
783      // Do NOT slurp the file into a String — a large already-converted
784      // document (the reporter's index.xml is 614 MB) would sit resident on
785      // top of the ~11× libxml2 DOM. Post-processing streams it from disk via
786      // `PostDocument::new_from_file` (see the `run_post_processing_from_file*`
787      // call below). We only need a non-empty `result` sentinel so the post
788      // gate fires; the real input is the source path.
789      latexml::converter::ConversionResponse {
790        result:      Some(String::new()),
791        log:         String::new(),
792        status:      String::from("Status:conversion:0"),
793        status_code: 0,
794      }
795    } else {
796      converter.convert(source)
797    };
798    let _ = &source_for_post; // keep alive for post-processing
799    // Post-phase log (Graphics/MathML/XSLT) captured by
800    // `run_post_processing_logged`; written after the core log into --log / the
801    // archive log so BOTH conversion phases reach the persisted log (SYNC_STATUS
802    // task 5; Perl LaTeXML.pm flushes once after convert_post). Declared out here
803    // (not in the `Some(xml)` arm) so the post-if-let --log write can still see
804    // it; stays empty when post-processing is skipped, keeping that --log
805    // byte-identical to before.
806    let mut post_log = String::new();
807    if let Some(xml) = response.result {
808      // Infer format from --dest extension if --format not specified (Perl Config.pm L408-441)
809      let inferred_format: Option<String> = cli
810        .format
811        .clone()
812        // `--xml` is Perl's shortcut for `--format=xml` (Config.pm L59).
813        .or(if cli.xml {
814          Some("xml".to_string())
815        } else {
816          None
817        })
818        .or_else(|| {
819          target.as_ref().and_then(|dest| {
820            Path::new(dest)
821              .extension()
822              .and_then(|ext| ext.to_str())
823              .map(|ext| {
824                match ext.to_lowercase().as_str() {
825                  "html" | "htm" => "html5".to_string(), // Perl L435: html → html5
826                  "xhtml" => "xhtml".to_string(),
827                  "xml" => "xml".to_string(),
828                  "zip" => "html5".to_string(), // Perl L431: zip → html5
829                  "epub" | "mobi" => "epub".to_string(),
830                  other => other.to_string(),
831                }
832              })
833          })
834        })
835        // `--whatsout=archive` with no `--dest`/`--format` still wants a
836        // web bundle — default it to html5, matching the `--dest *.zip`
837        // inference above (a `.zip` dest already maps to html5).
838        .or_else(|| {
839          if is_archive_out {
840            Some("html5".to_string())
841          } else {
842            None
843          }
844        });
845
846      // Auto-select stylesheet from format (Perl Config.pm L543-551)
847      // Shared with the library entrypoint via `post::default_stylesheet` so
848      // the CLI and `latexml::api` never disagree on the per-format sheet.
849      let effective_stylesheet = cli.stylesheet.clone().or_else(|| {
850        latexml::post::default_stylesheet(inferred_format.as_deref()).map(String::from)
851      });
852
853      // Auto-enable post-processing when dest implies HTML (Perl Config.pm L448-455)
854      let is_html_format = matches!(
855        inferred_format.as_deref(),
856        Some("html5") | Some("html") | Some("xhtml") | Some("epub") | Some("epub3")
857      );
858      // XML-input mode implies post-processing — there's nothing to
859      // convert (the file is already converted XML), so the only
860      // meaningful action is to run the post-pipeline on it.
861      // Matches the always-on post-processing behaviour of the now-
862      // retired `latexmlpost_oxide` binary.
863      let xml_input_mode = is_xml_input(&source_for_post);
864      // `--nopost` (Perl `post!` negated) force-skips post-processing so the
865      // raw LaTeXML XML is emitted even for an HTML-implying destination.
866      let do_post = !cli.nopost
867        && (cli.post
868          || cli.pmml
869          || cli.cmml
870          || effective_stylesheet.is_some()
871          || is_html_format
872          || cli.split
873          || cli.splitat.is_some()
874          || xml_input_mode
875          // Perl Config.pm L454: any non-`document` whatsout forces post.
876          || whatsout_mode.requires_post());
877
878      // Build split XPath from --splitat
879      // `--nosplit` (Perl `split!` negated) forces splitting off even when a
880      // `--splitat`/`--splitpath`/`--splitnaming` would otherwise enable it.
881      let split_enabled = !cli.nosplit
882        && (cli.split
883          || cli.splitat.is_some()
884          || cli.splitnaming.is_some()
885          || cli.splitpath.is_some());
886      let split_xpath = if split_enabled {
887        cli.splitpath.clone().or_else(|| {
888          let splitat = cli.splitat.as_deref().unwrap_or("section");
889          Some(make_splitpaths(splitat))
890        })
891      } else {
892        None
893      };
894
895      if do_post {
896        // Perl LaTeXML.pm:429 passes opts{sourcedirectory} to Post as
897        // `sourceDirectory`; when omitted, Post derives it from the source
898        // filename (Post.pm:727-729). Mirror that: honour `--sourcedirectory`
899        // if given, else fall back to the source file's own directory.
900        let source_dir = cli.sourcedirectory.clone().unwrap_or_else(|| {
901          Path::new(&source_for_post)
902            .parent()
903            .map(|p| p.to_string_lossy().to_string())
904            .unwrap_or_else(|| ".".to_string())
905        });
906
907        // `--whatsout=archive` (or a `.zip` destination) bundles into a
908        // zip. When `--dest` is omitted, Perl LaTeXML.pm:185-187 invents
909        // a placeholder `<source-name>.zip`; mirror that so an archive
910        // job always lands a file rather than dumping HTML to stdout.
911        let zip_dest: Option<String> = if is_archive_out {
912          Some(match &target {
913            Some(t) if t.to_ascii_lowercase().ends_with(".zip") => t.clone(),
914            Some(t) => format!(
915              "{}.zip",
916              t.trim_end_matches(".html").trim_end_matches(".xml")
917            ),
918            None => {
919              let stem = Path::new(&source_for_post)
920                .file_stem()
921                .and_then(|s| s.to_str())
922                .unwrap_or("document");
923              format!("{stem}.zip")
924            },
925          })
926        } else {
927          None
928        };
929
930        // For zip output, route graphics conversions through a TempDir so
931        // the converted PNG/SVG files can be collected and bundled into
932        // the output zip (mirroring `cortex_worker::pack_output_zip_with_resources`).
933        // Without this, the Graphics post-processor wrote PNGs next to
934        // `target` on the filesystem but the zip only carried HTML+log+status —
935        // confirmed-bug 2026-05-18 on 1910.01256.
936        let resource_tempdir: Option<tempfile::TempDir> = if is_archive_out {
937          Some(tempfile::tempdir()?)
938        } else {
939          None
940        };
941        let dest_for_post: Option<String> = if let Some(tmp) = resource_tempdir.as_ref() {
942          // Use a stable HTML filename derived from the zip stem so the
943          // Graphics processor's relative paths resolve naturally.
944          let stem = zip_dest
945            .as_deref()
946            .and_then(|z| Path::new(z).file_stem())
947            .and_then(|s| s.to_str())
948            .unwrap_or("document");
949          Some(
950            tmp
951              .path()
952              .join(format!("{stem}.html"))
953              .to_string_lossy()
954              .to_string(),
955          )
956        } else {
957          target.clone()
958        };
959
960        // latexmlpost_oxide's default was "if no --pmml AND no
961        // --stylesheet, default pmml = true". Apply the same rule for
962        // XML-input mode so `latexml_oxide foo.xml --dest out.html`
963        // does something useful out of the box.
964        let default_pmml_for_xml_input =
965          xml_input_mode && !cli.pmml && effective_stylesheet.is_none();
966        let post_opts = PostOptions {
967          // The `no*` forms (Perl `removeMathFormat`) suppress a rep that a
968          // format would otherwise default on.
969          pmml: (cli.pmml || cli.post || is_html_format || default_pmml_for_xml_input)
970            && !cli.nopmml,
971          cmml: cli.cmml && !cli.nocmml,
972          keep_xmath: cli.keep_xmath && !cli.noxmath,
973          stylesheet: effective_stylesheet.as_deref(),
974          destination: dest_for_post.as_deref(),
975          source_directory: Some(&source_dir),
976          // Perl LaTeXML.pm:430 `siteDirectory`; None ⇒ Post defaults it to the
977          // destination's directory (document.rs / Perl Config.pm L466-469).
978          site_directory: cli.sitedirectory.as_deref(),
979          search_paths: &cli.search_paths,
980          nodefaultresources: cli.nodefaultresources && !cli.defaultresources,
981          css_files: &cli.css_files,
982          js_files: &cli.js_files,
983          noinvisibletimes: cli.noinvisibletimes && !cli.invisibletimes,
984          mathtex: cli.mathtex && !cli.nomathtex,
985          navigationtoc: cli.navigationtoc.as_deref(),
986          schemadocs: cli.schemadocs,
987          split: split_enabled,
988          split_xpath,
989          split_naming: cli.splitnaming.as_deref(),
990          xslt_parameters: &cli.xslt_parameters,
991          graphics_svg_threshold_kb: cli.graphics_svg_threshold_kb,
992          graphicimages: cli.graphicimages || !cli.nographicimages,
993          // Perl `if ($timestamp)`: "0" (and empty) means "omit the timestamp".
994          timestamp: cli
995            .timestamp
996            .as_deref()
997            .filter(|t| !t.is_empty() && *t != "0"),
998          icon: cli.icon.as_deref(),
999          whatsout: whatsout_mode,
1000        };
1001        // XML-input mode parses the (possibly huge) source straight from disk
1002        // via the streaming file reader; TeX-conversion output is already in
1003        // memory as `xml`.
1004        let post = if xml_input_mode {
1005          latexml::post::run_post_processing_from_file_logged(&source_for_post, &post_opts)
1006        } else {
1007          latexml::post::run_post_processing_logged(&xml, &post_opts)
1008        };
1009        let output = post.html;
1010        post_log = post.log;
1011        if let Some(zip_dest) = zip_dest {
1012          // whatsout=archive: pack the full document + resources into a ZIP.
1013          latexml_post::writer::ensure_parent_dir(&zip_dest)?;
1014          let resource_dir = resource_tempdir.as_ref().map(|t| t.path());
1015          let stem = Path::new(&zip_dest)
1016            .file_stem()
1017            .and_then(|s| s.to_str())
1018            .unwrap_or("document");
1019          let html_name = format!("{stem}.html");
1020          let log_name = format!("{stem}.log");
1021          // Reproducible-build support: honour SOURCE_DATE_EPOCH for the
1022          // zip member timestamps (Perl Pack/Zip.pm L113-115).
1023          let source_date_epoch = std::env::var("SOURCE_DATE_EPOCH")
1024            .ok()
1025            .and_then(|s| s.trim().parse::<u64>().ok());
1026          latexml_post::pack::pack_archive(&latexml_post::pack::PackOptions {
1027            zip_path: &zip_dest,
1028            html_filename: &html_name,
1029            html: &output,
1030            log_filename: Some(&log_name),
1031            log: &assemble_conversion_log(&response.log, &post_log),
1032            status: &response.status,
1033            resource_dir,
1034            telemetry_json: None,
1035            source_date_epoch,
1036          })?;
1037          eprintln!("Output written to {}", zip_dest);
1038        } else {
1039          latexml_post::writer::write_output(&output, target.as_deref())?;
1040        }
1041        // resource_tempdir is dropped here (after pack_archive has copied
1042        // every file in), cleaning up the converted-PNG staging directory.
1043      } else {
1044        latexml_post::writer::write_output(&xml, target.as_deref())?;
1045      }
1046    }
1047
1048    // --log: write conversion log to file (skip if already packed into
1049    // the ZIP by the archive output stage).
1050    if let Some(ref log_path) = cli.log
1051      && !is_archive_out
1052    {
1053      // Write the core log and the post-phase log sequentially rather than
1054      // concatenating — both are already-allocated and large for real
1055      // articles, so a merged `format!` would allocate a third copy of their
1056      // combined size on the conversion path.
1057      if post_log.is_empty() {
1058        latexml_post::writer::write_output_segments(&[response.log.as_str()], Some(log_path))?;
1059      } else {
1060        latexml_post::writer::write_output_segments(
1061          &[response.log.as_str(), "\n", post_log.as_str()],
1062          Some(log_path),
1063        )?;
1064      }
1065      eprintln!("Log written to {}", log_path);
1066    }
1067  }
1068
1069  // Perl bin/latexml:151 — `if ($exit_message) { exit(1); }`: a Fatal
1070  // (status_code 3) conversion exits non-zero. cortex_worker already carries the
1071  // identical guard (`if final_status >= 3 { process::exit(...) }`); the standalone
1072  // CLI was missing it, so a 0-byte "complete" run (e.g. the plain-TeX
1073  // `$\displaylines{...}$` runaway that trips the memory-budget Fatal — shared with
1074  // Perl, which terminates at the same line) exited 0 and masqueraded as success.
1075  // Read the global status (thread-local REPORT, as cortex_worker does) — `response`
1076  // is scoped to the conversion branch. Match bin/latexml's exit(1) exactly;
1077  // status_code 2 ("errors but recoverable") stays a 0 exit, as in Perl.
1078  let final_status_code = latexml_core::common::error::get_status_code();
1079  if final_status_code >= 3 {
1080    write_telemetry_record(
1081      cli.telemetry_out.as_deref(),
1082      &telemetry_source,
1083      wall_start,
1084      "fatal",
1085      final_status_code as i32,
1086    );
1087    #[cfg(feature = "dhat-heap")]
1088    drop(_dhat.take());
1089    process::exit(1);
1090  }
1091  write_telemetry_record(
1092    cli.telemetry_out.as_deref(),
1093    &telemetry_source,
1094    wall_start,
1095    "ok",
1096    0,
1097  );
1098  #[cfg(feature = "dhat-heap")]
1099  drop(_dhat.take());
1100  process::exit(0);
1101}

1103/// Emit a single-line JSON telemetry record. No-op when neither
1104/// `--telemetry-out` nor `LATEXML_TELEMETRY_OUT` is set. Errors writing
1105/// the file are swallowed (the conversion already succeeded; logging
1106/// the failure on stderr would be noise for batch runs).
1107fn write_telemetry_record(
1108  cli_path: Option<&str>,
1109  source: &str,
1110  wall_start: std::time::Instant,
1111  category: &str,
1112  exit_code: i32,
1113) {
1114  use latexml_core::telemetry;
1115  let path = cli_path
1116    .map(|s| s.to_string())
1117    .or_else(|| std::env::var("LATEXML_TELEMETRY_OUT").ok());
1118  let Some(path) = path else { return };
1119
1120  // paper_id ≈ source basename without extension; cortex_worker
1121  // overrides this when it knows the arxiv id. Keep the binary's
1122  // best-effort default for direct CLI users.
1123  let paper_id = Path::new(source)
1124    .file_stem()
1125    .and_then(|s| s.to_str())
1126    .unwrap_or("")
1127    .to_string();
1128  telemetry::set_paper_id(&paper_id);
1129  telemetry::set_cmdline(&std::env::args().collect::<Vec<_>>().join(" "));
1130  if let Ok(host) = std::env::var("HOSTNAME").or_else(|_| {
1131    // Linux fallback: read /etc/hostname
1132    std::fs::read_to_string("/etc/hostname").map(|s| s.trim().to_string())
1133  }) {
1134    telemetry::set_host(&host);
1135  }
1136  telemetry::set_wall_us(wall_start.elapsed().as_micros() as u64);
1137  telemetry::set_category(category);
1138  telemetry::set_exit_code(exit_code);
1139  telemetry::set_max_rss_kb(read_max_rss_kb());
1140  let (cu, cs) = read_child_rusage_us();
1141  telemetry::set_child_rusage_us(cu, cs);
1142
1143  let record = telemetry::take();
1144  let line = record.to_json_line();
1145  if let Some(parent) = Path::new(&path).parent()
1146    && !parent.as_os_str().is_empty()
1147  {
1148    let _ = std::fs::create_dir_all(parent);
1149  }
1150  if let Ok(mut fh) = File::create(&path) {
1151    let _ = writeln!(fh, "{line}");
1152  }
1153}

1155/// Read peak resident-set size from `/proc/self/status` (`VmHWM`).
1156/// Returns 0 on non-Linux or read failure.
1157fn read_max_rss_kb() -> u64 {
1158  std::fs::read_to_string("/proc/self/status")
1159    .ok()
1160    .and_then(|content| {
1161      content
1162        .lines()
1163        .find(|l| l.starts_with("VmHWM:"))
1164        .and_then(|l| l.split_whitespace().nth(1))
1165        .and_then(|n| n.parse::<u64>().ok())
1166    })
1167    .unwrap_or(0)
1168}

1170/// Read accumulated child user/sys CPU time in microseconds via getrusage(2).
1171/// Returns (0, 0) on failure or non-unix.
1172#[cfg(unix)]
1173fn read_child_rusage_us() -> (u64, u64) {
1174  // SAFETY: getrusage(RUSAGE_CHILDREN, &ru) is async-signal-safe and
1175  // populates the struct unconditionally on success.
1176  unsafe {
1177    let mut ru: libc::rusage = std::mem::zeroed();
1178    if libc::getrusage(libc::RUSAGE_CHILDREN, &mut ru) == 0 {
1179      let user = (ru.ru_utime.tv_sec as u64) * 1_000_000 + (ru.ru_utime.tv_usec as u64);
1180      let sys = (ru.ru_stime.tv_sec as u64) * 1_000_000 + (ru.ru_stime.tv_usec as u64);
1181      (user, sys)
1182    } else {
1183      (0, 0)
1184    }
1185  }
1186}
1187
1188#[cfg(not(unix))]
1189fn read_child_rusage_us() -> (u64, u64) { (0, 0) }

1210fn make_splitpaths(splitat: &str) -> String {
1211  let ancestors: &[&str] = match splitat {
1212    "part" => &[],
1213    "chapter" => &["part"],
1214    "section" => &["part", "chapter"],
1215    "subsection" => &["part", "chapter", "section"],
1216    "subsubsection" => &["part", "chapter", "section", "subsection"],
1217    _ => &["part", "chapter"],
1218  };
1219  let back = ["bibliography", "appendix", "index"];
1220  let mut paths = Vec::new();
1221  let all_units: Vec<&str> = std::iter::once(splitat)
1222    .chain(ancestors.iter().copied())
1223    .collect();
1224  for unit in &all_units {
1225    paths.push(format!("//ltx:{}", unit));
1226    for b in &back {
1227      let mut conditions = vec![format!("preceding-sibling::ltx:{}", unit)];
1228      let unit_ancestors: &[&str] = match *unit {
1229        "part" => &[],
1230        "chapter" => &["part"],
1231        "section" => &["part", "chapter"],
1232        "subsection" => &["part", "chapter", "section"],
1233        "subsubsection" => &["part", "chapter", "section", "subsection"],
1234        _ => &[],
1235      };
1236      for anc in unit_ancestors {
1237        conditions.push(format!("parent::ltx:{}", anc));
1238      }
1239      paths.push(format!("//ltx:{}[{}]", b, conditions.join(" or ")));
1240    }
1241  }
1242  paths.join(" | ")
1243}

1245/// Unpack a ZIP (primary) or tar.gz archive into a temp directory.
1246/// Returns (TempDir, main_tex_path).
1247///
1248/// Port of Perl LaTeXML::Util::Pack::unpack_source.
1249/// Detect whether `source` is already-converted LaTeXML XML — i.e. a
1250/// `.xml` file — so the TeX → XML converter front-end can be skipped
1251/// and the file fed straight to post-processing. Matches what Perl
1252/// `latexmlpost` accepts and replaces the separate (now retired)
1253/// `latexmlpost_oxide` binary.
1254fn is_xml_input(source: &str) -> bool {

1256    .extension()
1257    .and_then(|e| e.to_str())
1258    .is_some_and(|ext| ext.eq_ignore_ascii_case("xml"))
1259}

1261fn unpack_archive(archive_path: &str) -> Result<(tempfile::TempDir, String), Box<dyn Error>> {
1262  let tempdir = tempfile::tempdir()?;
1263  let dest = tempdir.path();
1264
1265  if archive_path.ends_with(".zip") {
1266    // Primary format: ZIP
1267    let file = File::open(archive_path)?;
1268    let mut archive = zip::ZipArchive::new(file)?;
1269    for i in 0..archive.len() {
1270      let mut entry = archive.by_index(i)?;
1271      let outpath = dest.join(entry.mangled_name());
1272      if entry.is_dir() {
1273        std::fs::create_dir_all(&outpath)?;
1274      } else {
1275        if let Some(parent) = outpath.parent() {
1276          std::fs::create_dir_all(parent)?;
1277        }
1278        let mut outfile = File::create(&outpath)?;
1279        std::io::copy(&mut entry, &mut outfile)?;
1280      }
1281    }
1282  } else if archive_path.ends_with(".tar.gz") || archive_path.ends_with(".tgz") {
1283    let file = File::open(archive_path)?;
1284    let gz = flate2::read::GzDecoder::new(file);
1285    let mut archive = tar::Archive::new(gz);
1286    archive.unpack(dest)?;
1287  } else if archive_path.ends_with(".tar") {
1288    let file = File::open(archive_path)?;
1289    let mut archive = tar::Archive::new(file);
1290    archive.unpack(dest)?;
1291  } else {
1292    return Err(format!("Unsupported archive format: {}", archive_path).into());
1293  }
1294
1295  // Find main .tex file (Perl: LaTeXML::Util::Pack looks for the largest .tex file
1296  // or one containing \documentclass)
1297  let main_tex =
1298    latexml::main_tex::find_main_tex(dest).map_err(|e| -> Box<dyn Error> { e.into() })?;
1299  Ok((tempdir, main_tex.to_string_lossy().to_string()))
1300}
