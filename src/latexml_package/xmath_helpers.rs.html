1//! XMath helper functions for building XMDual token streams.
2//! Port of Perl's I_dual, I_arg, I_xmarg, I_xmref, I_wrap from Base_XMath.pool.ltxml.
3//!
4//! These functions generate token lists that expand into semantic math markup.
5//! Used by physics.sty, diffcoeff.sty, and other packages that need XMDual structures.
6
7use crate::prelude::*;
8
9/// Perl: getXMArgID — step @lx@xmarg counter, return its value as a string.
10pub fn get_xm_arg_id() -> Result<String> {
11  step_counter("@lx@xmarg", false)?;
12  let val = state::lookup_register("\\c@@lx@xmarg", Vec::new())?
13    .map(|rv| rv.value_of())
14    .unwrap_or(0);
15  Ok(val.to_string())
16}
17
18/// Perl: I_arg(n) — create a parameter token #n (cc ARG).
19pub fn i_arg(n: &str) -> Token {
20  let idx = n.parse::<u8>().unwrap_or(1);
21  // Create a CC_ARG token with the given index
22  CharToken!(
23    char::from_digit(idx as u32, 10).unwrap_or('1'),
24    Catcode::ARG
25  )
26}
27
28/// Perl: I_xmarg(id, arg) — generates `\lx@xmarg{id}{arg}` tokens.
29pub fn i_xmarg(id: &str, arg: Tokens) -> Tokens {
30  let mut tks = vec![T_CS!("\\lx@xmarg"), T_BEGIN!()];
31  tks.extend(ExplodeText!(id));
32  tks.push(T_END!());
33  tks.push(T_BEGIN!());
34  tks.extend(arg.unlist());
35  tks.push(T_END!());
36  Tokens::new(tks)
37}

39/// Perl: I_xmref(id) — generates `\lx@xmref{id}` tokens.
40pub fn i_xmref(id: &str) -> Tokens {
41  let mut tks = vec![T_CS!("\\lx@xmref"), T_BEGIN!()];
42  tks.extend(ExplodeText!(id));
43  tks.push(T_END!());
44  Tokens::new(tks)
45}

47/// Perl: I_wrap(keyvals, content) — generates `\lx@wrap[keyvals]{content}` tokens.
48pub fn i_wrap(keyvals: Option<Tokens>, content: Tokens) -> Tokens {
49  let mut tks = vec![T_CS!("\\lx@wrap")];
50  if let Some(kv) = keyvals {
51    tks.push(T_OTHER!("["));
52    tks.extend(kv.unlist());
53    tks.push(T_OTHER!("]"));
54  }
55  tks.push(T_BEGIN!());
56  tks.extend(content.unlist());
57  tks.push(T_END!());
58  Tokens::new(tks)
59}

61/// Perl: I_dual(keyvals, content, presentation, args) — generates full XMDual token stream.
62///
63/// Creates `\lx@dual[options]{content_with_xmrefs}{wrapped_presentation_with_xmargs}`.
64///
65/// # Arguments
66/// - `keyvals`: Optional HashMap of key-value attributes (reversion, meaning, role, etc.)
67/// - `content`: Token template for content branch (with #1, #2, ... parameter refs)
68/// - `presentation`: Token template for presentation branch (with #1, #2, ... parameter refs)
69/// - `args`: The actual argument token lists to be shared between branches
70pub fn i_dual(
71  keyvals: Option<&std::collections::HashMap<String, String>>,
72  content: Tokens,
73  presentation: Tokens,
74  args: Vec<Tokens>,
75) -> Result<Tokens> {
76  let mut revargs: Vec<Tokens> = Vec::new();
77  let mut pargs: Vec<Tokens> = Vec::new();
78  let mut cargs: Vec<Tokens> = Vec::new();
79
80  for arg in args {
81    let id = get_xm_arg_id()?;
82    revargs.push(Tokens::new(vec![i_arg(&id)]));
83    pargs.push(i_xmarg(&id, arg));
84    cargs.push(i_xmref(&id));
85  }
86
87  // Build optional keyvals string
88  let optional = if let Some(kv) = keyvals {
89    let mut opts = Vec::new();
90    for (key, value) in kv {
91      if !opts.is_empty() {
92        opts.push(T_OTHER!(","));
93      }
94      opts.extend(ExplodeText!(key));
95      opts.push(T_OTHER!("="));
96      opts.push(T_BEGIN!());
97      // For reversion keys, substitute parameter refs
98      let value_toks = Tokenize!(value);
99      if key.ends_with("reversion") {
100        let rev_opt: Vec<Option<Cow<Tokens>>> = revargs.iter().map(|t| Some(Cow::Borrowed(t))).collect();
101        opts.extend(value_toks.substitute_parameters(&rev_opt).unlist());
102      } else {
103        opts.extend(value_toks.unlist());
104      }
105      opts.push(T_END!());
106    }
107    Some(Tokens::new(opts))
108  } else {
109    None
110  };
111
112  // Build content with xmrefs substituted
113  let cargs_opt: Vec<Option<Cow<Tokens>>> = cargs.into_iter().map(|t| Some(Cow::Owned(t))).collect();
114  let content_subst = content.substitute_parameters(&cargs_opt);
115
116  // Build presentation with xmargs substituted, wrapped in \lx@wrap
117  let pargs_opt: Vec<Option<Cow<Tokens>>> = pargs.into_iter().map(|t| Some(Cow::Owned(t))).collect();
118  let pres_subst = presentation.substitute_parameters(&pargs_opt);
119  let wrapped_pres = i_wrap(None, pres_subst);
120
121  // Assemble: \lx@dual[options]{content}{presentation}
122  let mut tks = vec![T_CS!("\\lx@dual")];
123  if let Some(opts) = optional {
124    tks.push(T_OTHER!("["));
125    tks.extend(opts.unlist());
126    tks.push(T_OTHER!("]"));
127  }
128  tks.push(T_BEGIN!());
129  tks.extend(content_subst.unlist());
130  tks.push(T_END!());
131  tks.push(T_BEGIN!());
132  tks.extend(wrapped_pres.unlist());
133  tks.push(T_END!());
134
135  Ok(Tokens::new(tks))
136}
