3/// Perl floatflt.sty.ltxml L40-42: float direction from optional arg or
4/// `floatfltpos` state: `v` / `r` prefix → right, else left. Prior Rust
5/// hardcoded `float='right'`, ignoring both the option and `[l]` / `[p]`
6/// optional-arg forms.
7fn floatflt_float_direction(whatsit: &Whatsit) -> &'static str {
8  let pos_arg = whatsit
9    .get_arg(1)
10    .map(|a| a.to_string())
11    .unwrap_or_default();
12  let pos_arg = pos_arg.trim().to_string();
13  let pos = if !pos_arg.is_empty() {
14    pos_arg
15  } else {
16    match lookup_value("floatfltpos") {
17      Some(Stored::String(sym)) => with(sym, |s| s.to_string()),
18      _ => "v".to_string(),
19    }
20  };
21  if pos.starts_with('v') || pos.starts_with('r') {
22    "right"
23  } else {
24    "left"
25  }
26}

28/// Perl `toPercent($_[2])` (floatflt.sty.ltxml L55-57):
29/// `int(100 * $dimen->valueOf / LookupValue('\textwidth')->valueOf) . '%'`.
30/// Reads the mandatory `{Dimension}` arg (#2) and `\textwidth` via the VALUE
31/// table — `lookup_dimension` mirrors Perl's `LookupValue('\textwidth')`, which
32/// never warns about register-ness (a `\def`-ized `\textwidth` is read silently,
33/// unlike the old `lookup_register` path which emitted a spurious
34/// `expected:register`). Must be called where the env args exist (after_digest_begin).
35fn floatflt_pct_width(whatsit: &Whatsit) -> String {
36  let Some(dim_arg) = whatsit.get_arg(2) else {
37    return String::new();
38  };
39  let Some(tw) = lookup_dimension("\\textwidth") else {
40    return String::new();
41  };
42  let tw_sp = tw.value_of();
43  if tw_sp == 0 {
44    return String::new();
45  }
46  let pct = (100 * dim_arg.value_of()) / tw_sp;
47  s!("{pct}%")
48}
