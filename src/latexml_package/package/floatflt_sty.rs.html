4/// Perl floatflt.sty.ltxml L40-42: float direction from optional arg or
5/// `floatfltpos` state: `v` / `r` prefix → right, else left. Prior Rust
6/// hardcoded `float='right'`, ignoring both the option and `[l]` / `[p]`
7/// optional-arg forms.
8fn floatflt_float_direction(whatsit: &Whatsit) -> &'static str {
9  let pos_arg = whatsit
10    .get_arg(1)
11    .map(|a| a.to_string())
12    .unwrap_or_default();
13  let pos_arg = pos_arg.trim().to_string();
14  let pos = if !pos_arg.is_empty() {
15    pos_arg
16  } else if let Some(Stored::String(sym)) = state::lookup_value("floatfltpos") {
17    arena::with(sym, |s| s.to_string())
18  } else {
19    "v".to_string()
20  };
21  if pos.starts_with('v') || pos.starts_with('r') {
22    "right"
23  } else {
24    "left"
25  }
26}

28/// Perl `toPercent($_[2])` — 100 * dimen / \textwidth, formatted as "NN%".
29/// Called on the mandatory Dimension arg to populate the `width` attribute.
30fn floatflt_pct_width(whatsit: &Whatsit) -> String {
31  let dim_arg = whatsit
32    .get_arg(2)
33    .map(|a| a.to_attribute())
34    .unwrap_or_default();
35  let Ok(dim) = Dimension::from_str(&dim_arg) else {
36    return String::new();
37  };
38  let tw = match state::lookup_register("\\textwidth", Vec::new()) {
39    Ok(Some(RegisterValue::Dimension(d))) => d.value_of(),
40    _ => return String::new(),
41  };
42  if tw == 0 {
43    return String::new();
44  }
45  let pct = (100 * dim.value_of()) / tw;
46  s!("{pct}%")
47}
