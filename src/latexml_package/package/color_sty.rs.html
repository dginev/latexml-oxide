4/// Parse a color from an optional model name and a spec string.
5/// If model is given, constructs the Color directly.
6/// If no model, looks up the named color from state.
7/// Perl: ParseColor($model, $spec) in color.sty.ltxml
8pub fn parse_color(model: Option<&str>, spec: &str) -> Color {
9  let spec = spec.trim().trim_matches(|c| c == '{' || c == '}').trim();
10  if let Some(model) = model {
11    let model_lc = model.to_lowercase();
12    if model_lc == "named" {
13      return lookup_color_obj(&format!("named_{spec}"));
14    }
15    color_from_model_spec(&model_lc, spec)
16  } else {
17    lookup_color_obj(spec)
18  }
19}

21/// Look up a named color from state, returning a Color object.
22/// Perl: LookupColor($name) in Package.pm
23pub fn lookup_color_obj(name: &str) -> Color {
24  // Empty or whitespace-only name: silently return BLACK (expression-decode
25  // paths invoke us with `""` or `" "` when parsing malformed input; no
26  // error needed here because the decoder already surfaces its own).
27  if name.trim().is_empty() {
28    return color::BLACK;
29  }
30  let key = s!("color_{name}");
31  match state::lookup_value(&key) {
32    Some(Stored::String(sym)) => {
33      let stored_str = arena::with(sym, |s| s.to_string());
34      Color::from_stored(&stored_str).unwrap_or_else(|| {
35        Info!("undefined", name, &s!("color '{}' is undefined...", name));
36        color::BLACK
37      })
38    },
39    _ => {
40      // Perl color.sty.ltxml L50-53:
41      //   AssignValue('color_'.$spec => Black);
42      //   Error('unexpected', $spec, $STATE->getStomach,
43      //     "Can't find color named '$spec'; assuming Black");
44      // Persist Black under this name so subsequent lookups resolve it
45      // without repeating the diagnostic, then surface an Error-status
46      // diagnostic. We inline the bookkeeping from the `Error!` macro
47      // because that macro is return-based (`Fatal!` at threshold) and
48      // `lookup_color_obj` returns `Color`, not `Result<Color>`.
49      assign_value(
50        &s!("color_{name}"),
51        Stored::String(arena::pin(color::BLACK.to_stored())),
52        None,
53      );
54      latexml_core::common::error::note_status(latexml_core::common::error::LogStatus::Error, None);
55      if !latexml_core::common::error::is_log_output_suppressed() {
56        log::error!(
57          target: &format!("unexpected:{}", name),
58          "Can't find color named '{}'; assuming Black",
59          name
60        );
61      }
62      color::BLACK
63    },
64  }
65}
