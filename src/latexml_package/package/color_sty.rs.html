4/// Parse a color from an optional model name and a spec string.
5/// If model is given, constructs the Color directly.
6/// If no model, looks up the named color from state.
7/// Perl: ParseColor($model, $spec) in color.sty.ltxml
8fn parse_color(model: Option<&str>, spec: &str) -> Color {
9  let spec = spec.trim().trim_matches(|c| c == '{' || c == '}').trim();
10  if let Some(model) = model {
11    let model_lc = model.to_lowercase();
12    if model_lc == "named" {
13      return lookup_color_obj(&format!("named_{spec}"));
14    }
15    color_from_model_spec(&model_lc, spec)
16  } else {
17    lookup_color_obj(spec)
18  }
19}

21/// Look up a named color from state, returning a Color object.
22/// Perl: LookupColor($name) in Package.pm
23pub fn lookup_color_obj(name: &str) -> Color {
24  let key = s!("color_{name}");
25  match state::lookup_value(&key) {
26    Some(Stored::String(sym)) => {
27      let stored_str = arena::with(sym, |s| s.to_string());
28      Color::from_stored(&stored_str).unwrap_or_else(|| {
29        Info!("undefined", name, &s!("color '{}' is undefined...", name));
30        color::BLACK
31      })
32    },
33    _ => {
34      Info!("undefined", name, &s!("color '{}' is undefined...", name));
35      color::BLACK
36    },
37  }
38}

40/// Look up a named color from state. Returns hex string.
41/// Perl: LookupColor($name) in Package.pm
42pub fn lookup_color(name: &str) -> String {
43  lookup_color_obj(name).to_attribute()
44}

64  // \definecolor{name}{model}{spec}
65  // Perl: DefColor(ToString($name), ParseColor($model, $spec))
66  DefPrimitive!("\\definecolor{}{}{}", sub[(name, model, spec)] {
67    let name_str = do_expand(name)?.to_string();
68    let model_str = do_expand(model)?.to_string();
69    let spec_str = do_expand(spec)?.to_string();
70    // Use parse_color to handle all models including "named" lookups
71    let color = parse_color(Some(&model_str), &spec_str);
72    def_color(&name_str, &color, None)?;
73    Ok(Vec::new())
74  });

76  // \DefineNamedColor{dmodel}{name}{model}{spec}
77  // Perl: DefColor('named_'.$name, ParseColor($model, $spec))
78  DefPrimitive!("\\DefineNamedColor{}{}{}{}", sub[(dmodel, name, model, spec)] {
79    let name_str = do_expand(name)?.to_string();
80    let model_str = do_expand(model)?.to_string();
81    let spec_str = do_expand(spec)?.to_string();
82    let color = parse_color(Some(&model_str), &spec_str);
83    let named_key = format!("named_{}", name_str);
84    def_color(&named_key, &color, None)?;
85    Ok(Vec::new())

88  // \color[model]{spec} or \color{name}
89  // Perl: returns Box(undef,undef,undef, Invocation(\color, T_OTHER('rgb'), T_OTHER(components)))
90  DefPrimitive!("\\color[]{}", sub[(model_opt, spec)] {
91    let model_str = model_opt.map(|m| do_expand(m).ok()).flatten().map(|t| t.to_string());
92    let spec_str = do_expand(spec)?.to_string();
93    let color = parse_color(model_str.as_deref(), &spec_str);
94
95    // If in preamble, store for \normalcolor
96    if lookup_bool("inPreamble") {
97      assign_value("preambleTextcolor", Stored::String(arena::pin(color.to_stored())), None);
98    }
99    merge_font(fontmap!(color => color.clone()));
100
101    // Perl: Box(undef,undef,undef, Invocation(\color, T_OTHER('rgb'), T_OTHER(comps)))
102    // Return an empty Tbox whose reversion produces \color[rgb]{r,g,b} for the tex attribute.
103    let rgb = color.to_rgb();
104    let comps = rgb.components().iter()
105      .map(|c| {
106        let v = (*c * 10000.0).round() / 10000.0;
107        if v == v.floor() { format!("{}", v as i64) } else { format!("{v}") }
108      })
109      .collect::<Vec<_>>().join(",");
110    let reversion_tokens = Invocation!("\\color",
111      vec![Some(Tokens::from(T_OTHER!(&*"rgb"))),
112           Some(Tokens::from(T_OTHER!(&*comps)))]);
113    Ok(vec![Digested::from(Tbox::new(*EMPTY_SYM, None, None,
114      reversion_tokens, arena::SymHashMap::default()))])
115  });

121    let spec_str = do_expand(spec)?.to_string();
122    let color = parse_color(model_str.as_deref(), &spec_str);
