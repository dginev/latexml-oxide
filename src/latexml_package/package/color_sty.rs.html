5/// Parse a color from an optional model name and a spec string.
6/// If model is given, constructs the Color directly.
7/// If no model, looks up the named color from state.
8/// Perl: ParseColor($model, $spec) in color.sty.ltxml
9pub fn parse_color(model: Option<&str>, spec: &str) -> Color {
10  let spec = spec.trim().trim_matches(|c| c == '{' || c == '}').trim();
11  if let Some(model) = model {
12    let model_lc = model.to_lowercase();
13    if model_lc == "named" {
14      let key = format!("named_{spec}");
15      // The `named` color model's values are the dvips predefined colors
16      // (Red, Green, RubineRed, …) defined in dvipsnam.def via
17      // `\DefineNamedColor`. color.sty only loads dvipsnam.def when its
18      // `dvips`/`dvipsnames` *option* is processed — but if `color` was
19      // already pulled in earlier without that option (e.g. our hyperref
20      // binding's unconditional `RequirePackage("color")`), a later
21      // `\usepackage[dvips]{color}` is a no-op and the option never fires,
22      // leaving `named_Red` undefined. Since the `named` model inherently
23      // means a dvips named color, lazily load dvipsnam.def on first miss.
24      // Perl reaches the same end-state (it loads dvipsnam for this paper);
25      // witness 2005.01533 (revtex4-2 → hyperref preloads color →
26      // `\definecolor{red}{named}{Red}` → `named_Red` undefined).
27      if lookup_value(&format!("color_{key}")).is_none() {
28        let _ = input_definitions("dvipsnam", InputDefinitionOptions {
29          extension: Some(Cow::Borrowed("def")),
30          noerror: true,
31          ..InputDefinitionOptions::default()
32        });
33      }
34      return lookup_color_obj(&key);
35    }
36    color_from_model_spec(&model_lc, spec)
37  } else {
38    lookup_color_obj(spec)
39  }
40}

42/// Parse xcolor's color-algebra `!` operator. Supports:
43///   * `<color>!<pct>`            → mix(<color>, pct%, white)
44///   * `<color>!<pct>!<color2>`   → mix(<color>, pct%, <color2>)
45///   * chained `c1!p1!c2!p2!c3`   → left-to-right reduction
46///
47/// Returns Some(Color) if `name` contains `!` and parses cleanly,
48/// else None (caller treats `name` as a plain named color).
49fn try_color_algebra(name: &str) -> Option<Color> {
50  if !name.contains('!') {
51    return None;
52  }
53  let parts: Vec<&str> = name.split('!').map(str::trim).collect();
54  if parts.len() < 2 || parts[0].is_empty() {
55    return None;
56  }
57  let mut current = lookup_color_obj_no_algebra(parts[0]);
58  let mut i = 1;
59  while i < parts.len() {
60    let pct_str = parts[i];
61    let pct: f64 = pct_str.parse().ok()?;
62    let pct_frac = (pct / 100.0).clamp(0.0, 1.0);
63    let other =
64      if i + 1 < parts.len() && !parts[i + 1].is_empty() && parts[i + 1].parse::<f64>().is_err() {
65        let o = lookup_color_obj_no_algebra(parts[i + 1]);
66        i += 2;
67        o
68      } else {
69        i += 1;
70        color::WHITE
71      };
72    // xcolor `c!p!c2`: mix c at p% with c2. mix(self, other, fraction)
73    // expects fraction = weight of `other`; xcolor's `p` is weight of `self`.
74    current = current.mix(&other, 1.0 - pct_frac);
75  }
76  Some(current)
77}

79/// Look up a named color from state WITHOUT trying xcolor's `!` algebra.
80/// Used as the recursive base case from `try_color_algebra` to avoid
81/// infinite descent when neither half of a `!` expression is a base
82/// named color.
83fn lookup_color_obj_no_algebra(name: &str) -> Color {
84  let name = name.trim();
85  if name.is_empty() {
86    return color::BLACK;
87  }
88  let key = s!("color_{name}");
89  match lookup_value(&key) {
90    Some(Stored::String(sym)) => {
91      let stored_str = with(sym, |s| s.to_string());
92      Color::from_stored(&stored_str).unwrap_or(color::BLACK)
93    },
94    _ => color::BLACK,
95  }
96}

98/// Look up a named color from state, returning a Color object.
99/// Perl: LookupColor($name) in Package.pm
100pub fn lookup_color_obj(name: &str) -> Color {
101  // Empty or whitespace-only name: silently return BLACK (expression-decode
102  // paths invoke us with `""` or `" "` when parsing malformed input; no
103  // error needed here because the decoder already surfaces its own).
104  if name.trim().is_empty() {
105    return color::BLACK;
106  }
107  // xcolor color algebra: `red!20`, `red!20!blue`, etc. Try this first
108  // before falling through to the named-color table. Witness 2311.09262
109  // (\sethlcolor{red!20}\hl{...} pattern in tables).
110  if let Some(mixed) = try_color_algebra(name) {
111    return mixed;
112  }
113  let key = s!("color_{name}");
114  match lookup_value(&key) {
115    Some(Stored::String(sym)) => {
116      let stored_str = with(sym, |s| s.to_string());
117      Color::from_stored(&stored_str).unwrap_or_else(|| {
118        Info!("undefined", name, &s!("color '{}' is undefined...", name));
119        color::BLACK
120      })
121    },
122    _ => {
123      // Lazy-load the dvips named colors (Blue, Purple, Red, RubineRed, …) on
124      // first miss. The user's `\usepackage[usenames,dvipsnames]{color}` can be
125      // a no-op when `color` was already pulled in WITHOUT those options
126      // (hyperref's eager `RequirePackage("color")` for the colorlinks cluster
127      // loads it first), so the options never fire and the dvips names stay
128      // undefined → `\color{Blue}` inside e.g. `\lstset{keywordstyle=…}` errors.
129      // Activate `usenames` (so `\DefineNamedColor` ALSO exposes the plain name)
130      // + load dvipsnam.def ONCE, then re-look-up. Same end-state Perl reaches:
131      // its hyperref does NOT preload color, so the user's option-bearing load
132      // wins. Witness 1705.06183 (revtex4-1 + hyperref + `\lstset{…\color{Blue}}`).
133      // The `named`-model path (parse_color above) already does the analogous
134      // lazy-load; this covers the plain `\color{Blue}` (no model) path.
135      if !lookup_bool("color_dvipsnam_lazy_loaded") {
136        assign_value("color_dvipsnam_lazy_loaded", true, Some(Scope::Global));
137        assign_value("color_usenames_active", true, Some(Scope::Global));
138        // Force the 68 dvips colors GLOBAL: we are inside a grouped digestion
139        // (e.g. `\textcolor{Blue}{…}`), so without this they'd be local and
140        // revert before the next color is looked up (see `def_color`).
141        assign_value("color_force_global", true, Some(Scope::Global));
142        let _ = input_definitions("dvipsnam", InputDefinitionOptions {
143          extension: Some(Cow::Borrowed("def")),
144          noerror: true,
145          ..InputDefinitionOptions::default()
146        });
147        assign_value("color_force_global", false, Some(Scope::Global));
148        if let Some(Stored::String(sym)) = lookup_value(&key) {
149          let stored_str = with(sym, |s| s.to_string());
150          if let Some(c) = Color::from_stored(&stored_str) {
151            return c;
152          }
153        }
154      }
155      // Perl color.sty.ltxml L50-53:
156      //   AssignValue('color_'.$spec => Black);
157      //   Error('unexpected', $spec, $STATE->getStomach,
158      //     "Can't find color named '$spec'; assuming Black");
159      // Persist Black under this name so subsequent lookups resolve it
160      // without repeating the diagnostic, then surface an Error-status
161      // diagnostic. We inline the bookkeeping from the `Error!` macro
162      // because that macro is return-based (`Fatal!` at threshold) and
163      // `lookup_color_obj` returns `Color`, not `Result<Color>`.
164      assign_value(
165        &s!("color_{name}"),
166        Stored::String(pin(color::BLACK.to_stored())),
167        None,
168      );
169      note_status(LogStatus::Error, None);
170      if !is_log_output_suppressed() {
171        log::error!(
172          target: &format!("unexpected:{}", name),
173          "Can't find color named '{}'; assuming Black",
174          name
175        );
176      }
177      color::BLACK
178    },
179  }
180}

237  //   DefColor(ToString($name), ParseColor($model, $spec));
