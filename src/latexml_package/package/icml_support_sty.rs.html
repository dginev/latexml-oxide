25  // Perl gobbles \icmltitlerunning; surpass: it's the running-head

28    "\\@add@frontmatter{ltx:toctitle}{#1}");
29  // \icmlsetsymbol{name}{symbol} — creates `\<name>` macro expanding
30  // to <symbol> for use in author lists. Real icml2024.sty L100-ish:
31  //   `\def\icmlsetsymbol#1#2{\expandafter\def\csname #1\endcsname{#2}}`
32  // Previous stub gobbled both args without defining the CS, breaking
33  // `\icmlauthor{...}{equal,affil,icmlWorkDone}` which later references
34  // `\icmlWorkDone` in `\printAffiliationsAndNotice`. Witness 2310.06430.
35  DefMacro!("\\icmlsetsymbol{}{}",

43  // ICML: \icmlaffiliation{shortname}{full text} maps a short id to
44  // an affiliation string used in author list. Preserve only the
45  // full-text — the shortname (#1) is an internal identifier that
46  // commonly contains `_` characters (e.g. `mit_idss`, `osu_ece`),
47  // which our frontmatter pipeline tokenizes as math-mode subscript
48  // and errors with "Script _ can only appear in math mode" when
49  // rendered into ltx:note text. Witness 2404.08592.
50  DefMacro!("\\icmlaffiliation{}{}",
51    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}");
52  // \icmlcorrespondingauthor{name}{email} — preserve as ltx:note.
53  // The email arg often contains `_` (e.g. `m_smith@apple.com`) which would
54  // otherwise be tokenized as subscript-mode at digest time, triggering
55  // "Script _ can only appear in math mode" cascades. Semiverbatim
56  // neutralizes `_`/`#`/`&`/`%`/`^`/`~`/`$`/`{`/`}` in the email arg.
57  // Perl's icml_support binding gobbles both args (empty body); we surpass
58  // by preserving the contact text as a frontmatter note. Witness 2312.09299.
59  DefMacro!("\\icmlcorrespondingauthor{} Semiverbatim",
60    "\\@add@frontmatter{ltx:note}[role=corresponding-author]{#2 <#1>}");
