13  // graphics.sty's body. Perl's `graphicx.sty.ltxml` L22 only requires
14  // `graphics` (relying on Perl's keyval binding being preloaded by
15  // some other path), but Rust's graphics_sty.rs doesn't require keyval
16  // either. Downstream `\RequirePackage{graphbox}` (which raw-loads and
17  // calls `\define@key`) then errors with `\define@key` undefined
18  // because keyval never loaded — graphics.sty's hand-port handles its
19  // own keyvals via `DefKeyVal!`/`DefParameterType!` Rust-side, but raw-
20  // loaded sibling sty files that call `\define@key` directly need the
21  // real CS. Witness: arXiv:2504.13697 (IEEEtran + graphbox).
22  //
23  // GUARD: only do this when the LaTeX kernel is already initialized
24  // (proxy: `\@onefilewithoptions` defined). Without the guard, old
25  // LaTeX 2.09 papers (e.g. astro-ph9501095 with `\input psfig` BEFORE
26  // `\documentstyle`) trigger graphicx via ar5iv preload BEFORE LaTeX.pool
27  // loads, and the keyval raw-load tries to use kernel hooks that aren't
28  // ready yet (`Extra \PopDefaultHookLabel` + `\@nil` undefined errors).

124  DefConstructor!(
125    "\\@includegraphicx OptionalMatch:* OptionalKeyVals:Gin Semiverbatim",
126    "<ltx:graphics graphic='#path' candidates='#candidates' options='#options'/>",
127    // Perl L72: mode => 'restricted_horizontal', enterHorizontal => 1.
128    mode => "restricted_horizontal",
129    enter_horizontal => true,
130    // Perl L54: alias => '\includegraphics' so the reversion `tex=`
131    // attribute serializes back to the author-facing name rather than

134    // Perl L56: scope => 'global'.
135    scope => Some(Scope::Global),
136    // Perl L63-71: properties callback.
137    properties => sub[args] {
138      // arg 0: starred, arg 1: keyvals, arg 2: graphic path
139      let path = args[2].as_ref().map(|a| a.to_attribute()).unwrap_or_default();
140      let path = path.trim().to_string();
141      // Perl: ($path, @candidates) = image_candidates(ToString($graphic))
142      let candidates = image_candidates(&path);
143      // Perl: $options = graphicX_options($starred, $kv)
144      let starred = args[0].is_some();
145      let mut options_vec: Vec<String> = Vec::new();
146      if starred {
147        options_vec.push(s!("clip=true"));
148      }
149      let mut saw_w = false;
150      let mut saw_h = false;
151      let mut has_keepaspectratio = false;
152      // Perl extracts `alt` separately as a semantic property — it
153      // becomes the `description` attribute via afterConstruct, not
154      // a graphics option.
155      let mut alt_value: Option<String> = None;
156      let mut actualtext_value: Option<String> = None;
157      let mut artifact_value: Option<String> = None;
158      if let Some(ref kv_digested) = args[1]
159        && let DigestedData::KeyVals(kv) = kv_digested.data() {
160          for (key, value) in kv.get_pairs() {
161            if key == "alt" {
162              alt_value = Some(value.to_string());
163              continue;
164            }
165            if key == "actualtext" {
166              actualtext_value = Some(value.to_string());
167              continue;
168            }
169            if key == "artifact" {
170              artifact_value = Some(value.to_string());
171              continue;
172            }
173            if key.ends_with("width") { saw_w = true; }
174            if key.ends_with("height") { saw_h = true; }
175            if key == "keepaspectratio" { has_keepaspectratio = true; }
176            let val_str = value.to_string();
177            let val_str = val_str.replace(',', "\\,");
178            options_vec.push(format!("{key}={val_str}"));
179          }
180        }
181      // Auto-add keepaspectratio if only width or height (not both) specified
182      if (saw_w ^ saw_h) && !has_keepaspectratio {
183        options_vec.push(s!("keepaspectratio=true"));
184      }
185      let options = options_vec.join(",");
186      let mut props = stored_map!("path" => path, "candidates" => candidates, "options" => options);
187      if let Some(alt) = alt_value {
188        props.insert("alt", Stored::from(alt));
189      }
190      if let Some(at) = actualtext_value {
191        props.insert("actualtext", Stored::from(at));
192      }
193      if let Some(af) = artifact_value {
194        props.insert("artifact", Stored::from(af));
195      }
196      Ok(props)
197    },
198    // Perl L55: sizer => \&image_graphicx_sizer. We port it via

204    // Perl L57-62: afterConstruct — emit `description` attribute from

208    after_construct => sub[document, whatsit] {
209      // arXiv-fork e2b200fb: description = actualtext // alt // '' (set
210      // even when empty); when BOTH actualtext and alt are given, alt goes
211      // to aria:description; artifact (any value but "false") hides the
212      // image from assistive tech via aria:hidden.
213      let alt = whatsit.get_property("alt").map(|v| v.to_string());
214      let actualtext = whatsit.get_property("actualtext").map(|v| v.to_string());
215      let artifact = whatsit
216        .get_property("artifact")
217        .map(|v| v.to_string())
218        .is_some_and(|v| v != "false");
219      if (artifact || actualtext.is_some() || alt.is_some())
220        && let Some(mut last_child) = document.get_node().get_last_child() {
221          let description = actualtext
222            .clone()
223            .or_else(|| alt.clone())
224            .unwrap_or_default();
225          document.set_attribute(&mut last_child, "description", &description)?;
226          if let (Some(alt_str), Some(_)) = (&alt, &actualtext) {
227            document.set_attribute(&mut last_child, "aria:description", alt_str)?;
228          }
229          if artifact {
230            document.set_attribute(&mut last_child, "aria:hidden", "true")?;
231          }
232        }
233    }
234  );
