3/// Perl: image_candidates($path) from LaTeXML::Util::Image
4/// Searches for files matching `path` (possibly with extensions) in graphics/search paths.
5/// Returns comma-separated list of candidate paths, relative to source directory.
6pub fn image_candidates(path: &str) -> String {
7  use std::path::{Path, PathBuf};
8  let path = path.trim().trim_matches('"');
9  if path.is_empty() {
10    return String::new();
11  }
12  let mut search_dirs: Vec<String> = state::get_graphics_paths();
13  search_dirs.extend(state::get_search_paths());
14  let source_dir = state::lookup_string("SOURCEDIRECTORY");
15  if !source_dir.is_empty() {
16    search_dirs.push(source_dir.clone());
17  }
18  if search_dirs.is_empty() {
19    search_dirs.push(".".to_string());
20  }
21
22  let mut candidates: Vec<String> = Vec::new();
23  let path_obj = Path::new(path);
24  let has_extension = path_obj.extension().is_some();
25  let source_path = if source_dir.is_empty() { None } else { Some(PathBuf::from(&source_dir)) };
26
27  for dir in &search_dirs {
28    let base = PathBuf::from(dir).join(path);
29    if has_extension {
30      if base.exists() {
31        let rel = match &source_path {
32          Some(sp) => base.strip_prefix(sp).unwrap_or(&base).to_string_lossy().to_string(),
33          None => base.to_string_lossy().to_string(),
34        };
35        candidates.push(rel);
36      }
37    } else {
38      // Search for path with any extension
39      let parent = base.parent().unwrap_or(Path::new("."));
40      let stem = base.file_name().map(|s| s.to_string_lossy().to_string()).unwrap_or_default();
41      if let Ok(entries) = std::fs::read_dir(parent) {
42        for entry in entries.flatten() {
43          let fname = entry.file_name().to_string_lossy().to_string();
44          if let Some(dot_pos) = fname.find('.') {
45            if fname[..dot_pos] == stem {
46              let full = entry.path();
47              let rel = match &source_path {
48                Some(sp) => full.strip_prefix(sp).unwrap_or(&full).to_string_lossy().to_string(),
49                None => full.to_string_lossy().to_string(),
50              };
51              candidates.push(rel);
52            }
53          }
54        }
55      }
56    }
57  }
58
59  // If no candidates found and path has no extension, try common image extensions
60  // (matching Perl's pathname_findall with types => ['*'] which tries all known types)
61  if candidates.is_empty() && !has_extension {
62    // Perl typically returns just the first match (png)
63    if let Some(ext) = ["png", "jpg", "jpeg", "gif", "pdf", "eps", "svg"].first() {
64      let with_ext = format!("{path}.{ext}");
65      candidates.push(with_ext);
66    }
67  }
68
69  // Deduplicate while preserving order
70  let mut seen = std::collections::HashSet::new();
71  candidates.retain(|c| seen.insert(c.clone()));
72
73  if candidates.is_empty() {
74    path.to_string()
75  } else {
76    candidates.join(",")
77  }
78}

80/// Perl: image_graphicx_sizer from LaTeXML::Util::Image
81/// Reads image dimensions from candidates, applies graphicx options (height/width/scale),
82/// and sets the whatsit's cached size properties so PGF/tikz get correct box dimensions.
83fn image_graphicx_sizer(whatsit: &mut Whatsit) {
84  use std::path::{Path, PathBuf};
85
86  let dpi_val = state::lookup_int("DPI");
87  let dpi = if dpi_val > 0 { dpi_val as f64 } else { 100.0 }; // Perl: our $DPI = 100
88  let candidates = whatsit.get_property("candidates")
89    .map(|c| c.to_string()).unwrap_or_default();
90  let options = whatsit.get_property("options")
91    .map(|c| c.to_string()).unwrap_or_default();
92
93  // Try to read actual image dimensions from file
94  let mut img_w: f64 = 0.0;
95  let mut img_h: f64 = 0.0;
96  let source_dir = state::lookup_string("SOURCEDIRECTORY");
97  for candidate in candidates.split(',') {
98    let candidate = candidate.trim();
99    if candidate.is_empty() { continue; }
100    let full_path = if Path::new(candidate).is_absolute() {
101      PathBuf::from(candidate)
102    } else if !source_dir.is_empty() {
103      PathBuf::from(&source_dir).join(candidate)
104    } else {
105      PathBuf::from(candidate)
106    };
107    if let Some((w, h)) = read_image_dimensions(&full_path) {
108      img_w = w as f64;
109      img_h = h as f64;
110      break;
111    }
112  }
113
114  if img_w <= 0.0 || img_h <= 0.0 { return; }
115
116  // Apply graphicx options (height, width, scale, keepaspectratio)
117  // Perl: image_graphicx_size applies parsed transformations
118  let dppt = dpi / 72.27; // dots per point
119  let mut w = img_w;
120  let mut h = img_h;
121
122  // Parse options string for simple cases
123  // Perl: image_graphicx_parse uses to_bp() to convert dimensions to big points (1/72 inch)
124  let mut req_w: Option<f64> = None; // in bp (big points)
125  let mut req_h: Option<f64> = None; // in bp
126  let mut keep_ratio = false;
127  let mut scale: Option<f64> = None;
128
129  for opt in options.split(',') {
130    let opt = opt.trim();
131    if let Some(val) = opt.strip_prefix("width=") {
132      if let Ok(dim) = Dimension::from_str(val.trim()) {
133        // to_bp: convert pt to bp (1bp = 1/72 inch, 1pt = 1/72.27 inch)
134        req_w = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
135      }
136    } else if let Some(val) = opt.strip_prefix("height=") {
137      if let Ok(dim) = Dimension::from_str(val.trim()) {
138        req_h = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
139      }
140    } else if let Some(val) = opt.strip_prefix("totalheight=") {
141      if let Ok(dim) = Dimension::from_str(val.trim()) {
142        req_h = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
143      }
144    } else if opt.starts_with("keepaspectratio") {
145      keep_ratio = true;
146    } else if let Some(val) = opt.strip_prefix("scale=") {
147      scale = val.trim().parse::<f64>().ok();
148    }
149  }
150
151  // Apply transformations (matching Perl image_graphicx_size logic)
152  if let Some(s) = scale {
153    w = (w * s).ceil();
154    h = (h * s).ceil();
155  }
156  if req_w.is_some() || req_h.is_some() {
157    let target_w = req_w.map(|rw| rw * dppt);
158    let target_h = req_h.map(|rh| rh * dppt);
159    if keep_ratio {
160      match (target_w, target_h) {
161        (Some(tw), Some(th)) => {
162          // Both specified with keepaspectratio: use the more restrictive
163          if w > 0.0 && h > 0.0 {
164            if tw / w < th / h {
165              let th2 = h * tw / w;
166              w = tw;
167              h = th2;
168            } else {
169              let tw2 = w * th / h;
170              w = tw2;
171              h = th;
172            }
173          }
174        },
175        (Some(tw), None) => {
176          if w > 0.0 { h = h * tw / w; w = tw; }
177        },
178        (None, Some(th)) => {
179          if h > 0.0 { w = w * th / h; h = th; }
180        },
181        (None, None) => {},
182      }
183    } else {
184      if let Some(tw) = target_w { w = tw; }
185      if let Some(th) = target_h { h = th; }
186    }
187  }
188  // Perl: ceil pixel dimensions after applying transforms
189  w = w.ceil();
190  h = h.ceil();
191
192  // Convert pixel dimensions back to points, then to scaled points (sp)
193  let width_pt = w * 72.27 / dpi;
194  let height_pt = h * 72.27 / dpi;
195
196  // Perl: Dimension($w * 72.27 / $dpi . 'pt') — parses via TeX fixed-point arithmetic
197  let w_dim = Dimension::from_str(&format!("{width_pt}pt")).unwrap_or_default();
198  let h_dim = Dimension::from_str(&format!("{height_pt}pt")).unwrap_or_default();
199  whatsit.set_property("cached_width", Stored::Dimension(w_dim));
200  whatsit.set_property("cached_height", Stored::Dimension(h_dim));
201  whatsit.set_property("cached_depth", Stored::Dimension(Dimension::default()));
202}

204/// Read image dimensions (width, height) in pixels from a file.
205/// Supports PNG and JPEG formats.
206fn read_image_dimensions(path: &std::path::Path) -> Option<(u32, u32)> {
207  use std::io::Read;
208  let mut file = std::fs::File::open(path).ok()?;
209  let mut header = [0u8; 32];
210  file.read_exact(&mut header).ok()?;
211
212  // PNG: signature + IHDR chunk
213  if &header[0..8] == b"\x89PNG\r\n\x1a\n" {
214    let width = u32::from_be_bytes([header[16], header[17], header[18], header[19]]);
215    let height = u32::from_be_bytes([header[20], header[21], header[22], header[23]]);
216    return Some((width, height));
217  }
218
219  // JPEG: look for SOF marker
220  if header[0] == 0xFF && header[1] == 0xD8 {
221    // Read the full file for JPEG parsing
222    let mut data = header.to_vec();
223    file.read_to_end(&mut data).ok()?;
224    let mut i = 2;
225    while i + 9 < data.len() {
226      if data[i] != 0xFF { break; }
227      let marker = data[i + 1];
228      // SOF markers: 0xC0-0xCF (except 0xC4 DHT, 0xC8 JPG, 0xCC DAC)
229      if (0xC0..=0xCF).contains(&marker) && marker != 0xC4 && marker != 0xC8 && marker != 0xCC {
230        let height = u16::from_be_bytes([data[i + 5], data[i + 6]]) as u32;
231        let width = u16::from_be_bytes([data[i + 7], data[i + 8]]) as u32;
232        return Some((width, height));
233      }
234      let len = u16::from_be_bytes([data[i + 2], data[i + 3]]) as usize;
235      i += 2 + len;
236    }
237  }
238
239  None
240}

242LoadDefinitions!({
243  // graphicx.sty provides alternative argument syntax for graphics inclusion.
244  // (See LaTeXML::Post::Graphics for suggested postprocessing)
245
246  // Load the base graphics package
247  RequirePackage!("graphics");
248
249  // Raw TeX graphicx.sty redefines \rotatebox with \protected\def which overrides
250  // our DefConstructor from graphics_sty.rs. Re-register using Let! to restore
251  // the Rust definition. The Rust definition (from graphics_sty.rs) is stored
252  // as \rotatebox's meaning before the raw TeX override happens, so we can
253  // restore it by re-defining here after graphics is loaded.
254  // Perl: graphicx.sty.ltxml doesn't need this because .ltxml bindings prevent
255  // raw TeX loading entirely.
256  DefConstructor!("\\rotatebox OptionalKeyVals:Grot {Float} {}",
257    "<ltx:inline-block angle='#angle' width='#width' height='#height' depth='#depth' innerwidth='#innerwidth' innerheight='#innerheight' innerdepth='#innerdepth' xtranslate='#xtranslate' ytranslate='#ytranslate'>#3</ltx:inline-block>",
258    mode => "restricted_horizontal", enter_horizontal => true,
259    after_digest => sub[whatsit] {
260      let angle = whatsit.get_arg(2).map(|a| a.to_attribute().parse::<f64>().unwrap_or(0.0)).unwrap_or(0.0);
261      if let Some(body) = whatsit.get_arg(3) {
262        if let Ok(props) = crate::package::graphics_sty::rotated_properties(body.clone(), angle, false) {
263          for (k, v) in props {
264            whatsit.set_property(k, v);
265          }
266        }
267      }
268    });
269
270  // Internal macros for graphicx sizing
271  DefMacro!("\\Gin@ewidth", "");
272  DefMacro!("\\Gin@eheight", "");
273  DefMacro!("\\Gin@eresize", "");
274  DefMacro!("\\Gin@esetsize", "");
275
276  // KeyVal options for the Gin family
277  // NOTE: GraphixDimension and GraphixDimensions are custom parameter types
278  // defined in graphics.sty.ltxml. We use "Dimension" as a closest approximation
279  // for GraphixDimension, and "" (plain text) for GraphixDimensions (sequence of 4 dims).
280  DefKeyVal!("Gin", "width", "Dimension");
281  DefKeyVal!("Gin", "height", "Dimension");
282  DefKeyVal!("Gin", "totalheight", "Dimension");
283  DefKeyVal!("Gin", "keepaspectratio", "", "true");
284  DefKeyVal!("Gin", "clip", "", "true");
285  DefKeyVal!("Gin", "scale", "");
286  DefKeyVal!("Gin", "angle", "");
287  DefKeyVal!("Gin", "alt", "");
288  DefKeyVal!("Gin", "trim", "");
289  DefKeyVal!("Gin", "viewport", "");
290
291  // LaTeXML extensions:
292  DefKeyVal!("Gin", "vrml", "Semiverbatim");
293  DefKeyVal!("Gin", "magnifiable", "", "true");
294
295  // Redefine \includegraphics to dispatch based on bracket syntax:
296  // If a second [] follows, use the old graphics.sty-style \@includegraphics,
297  // otherwise use the graphicx keyval-style \@includegraphicx.
298  DefMacro!(
299    "\\includegraphics OptionalMatch:* []",
300    "\\@ifnextchar[{\\@includegraphics#1[#2]}{\\@includegraphicx#1[#2]}"
301  );
302
303  // The graphicx-style \includegraphics with keyval options.
304  // Perl: properties callback computes path, candidates, options from keyval args.
305  // Perl also has: sizer => \&image_graphicx_sizer — computes box dimensions from image file.
306  DefConstructor!(
307    "\\@includegraphicx OptionalMatch:* OptionalKeyVals:Gin Semiverbatim",
308    "<ltx:graphics graphic='#path' candidates='#candidates' options='#options'/>",
309    enter_horizontal => true,
310    properties => sub[args] {
311      // arg 0: starred, arg 1: keyvals, arg 2: graphic path
312      let path = args[2].as_ref().map(|a| a.to_attribute()).unwrap_or_default();
313      let path = path.trim().to_string();
314      // Perl: image_candidates searches filesystem for files matching path with any extension
315      let candidates = image_candidates(&path);
316      // Build options string from keyval pairs, matching Perl's graphicX_options
317      let starred = args[0].is_some();
318      let mut options_vec: Vec<String> = Vec::new();
319      if starred {
320        options_vec.push(s!("clip=true"));
321      }
322      let mut saw_w = false;
323      let mut saw_h = false;
324      let mut has_keepaspectratio = false;
325      if let Some(ref kv_digested) = args[1] {
326        if let DigestedData::KeyVals(ref kv) = kv_digested.data() {
327          for (key, value) in kv.get_pairs() {
328            if key.ends_with("width") { saw_w = true; }
329            if key.ends_with("height") { saw_h = true; }
330            if key == "keepaspectratio" { has_keepaspectratio = true; }
331            let val_str = value.to_string();
332            let val_str = val_str.replace(',', "\\,");
333            options_vec.push(format!("{key}={val_str}"));
334          }
335        }
336      }
337      // Auto-add keepaspectratio if only width or height (not both) specified
338      if (saw_w ^ saw_h) && !has_keepaspectratio {
339        options_vec.push(s!("keepaspectratio=true"));
340      }
341      let options = options_vec.join(",");
342      Ok(stored_map!("path" => path, "candidates" => candidates, "options" => options))
343    },
344    after_digest => sub[whatsit] {
345      // Perl: sizer => \&image_graphicx_sizer
346      // Compute box dimensions from image file + graphicx options
347      image_graphicx_sizer(whatsit);
348    }
349  );
