11  // Real TL graphicx.sty L31: `\RequirePackage{keyval,graphics}` —
12  // keyval FIRST so `\define@key`, `\setkeys` are available before
13  // graphics.sty's body. Perl's `graphicx.sty.ltxml` L22 only requires
14  // `graphics` (relying on Perl's keyval binding being preloaded by
15  // some other path), but Rust's graphics_sty.rs doesn't require keyval
16  // either. Downstream `\RequirePackage{graphbox}` (which raw-loads and
17  // calls `\define@key`) then errors with `\define@key` undefined
18  // because keyval never loaded — graphics.sty's hand-port handles its
19  // own keyvals via `DefKeyVal!`/`DefParameterType!` Rust-side, but raw-
20  // loaded sibling sty files that call `\define@key` directly need the
21  // real CS. Witness: arXiv:2504.13697 (IEEEtran + graphbox).
22  //
23  // GUARD: only do this when the LaTeX kernel is already initialized
24  // (proxy: `\@onefilewithoptions` defined). Without the guard, old
25  // LaTeX 2.09 papers (e.g. astro-ph9501095 with `\input psfig` BEFORE
26  // `\documentstyle`) trigger graphicx via ar5iv preload BEFORE LaTeX.pool
27  // loads, and the keyval raw-load tries to use kernel hooks that aren't
28  // ready yet (`Extra \PopDefaultHookLabel` + `\@nil` undefined errors).

122    // Perl L72: mode => 'restricted_horizontal', enterHorizontal => 1.

124    enter_horizontal => true,

134      let path = args[2].as_ref().map(|a| a.to_attribute()).unwrap_or_default();
135      let path = path.trim().to_string();
136      // Perl: ($path, @candidates) = image_candidates(ToString($graphic))

143      }
144      let mut saw_w = false;
145      let mut saw_h = false;
146      let mut has_keepaspectratio = false;
147      // Perl extracts `alt` separately as a semantic property — it

156              continue;
157            }
158            if key.ends_with("width") { saw_w = true; }
159            if key.ends_with("height") { saw_h = true; }

161            let val_str = value.to_string();
162            let val_str = val_str.replace(',', "\\,");
163            options_vec.push(format!("{key}={val_str}"));
164          }
165        }
166      // Auto-add keepaspectratio if only width or height (not both) specified

168        options_vec.push(s!("keepaspectratio=true"));
169      }
170      let options = options_vec.join(",");
171      let mut props = stored_map!("path" => path, "candidates" => candidates, "options" => options);
172      if let Some(alt) = alt_value {
173        props.insert("alt", Stored::from(alt));
174      }
175      Ok(props)
176    },
