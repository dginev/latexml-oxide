3/// Perl: image_candidates($path) from LaTeXML::Util::Image
4/// Searches for files matching `path` (possibly with extensions) in graphics/search paths.
5/// Returns comma-separated list of candidate paths, relative to source directory.
6pub fn image_candidates(path: &str) -> String {
7  use std::path::{Path, PathBuf};
8  let path = path.trim().trim_matches('"');
9  if path.is_empty() {
10    return String::new();
11  }
12  let mut search_dirs: Vec<String> = state::get_graphics_paths();
13  search_dirs.extend(state::get_search_paths());
14  let source_dir = state::lookup_string("SOURCEDIRECTORY");
15  if !source_dir.is_empty() {
16    search_dirs.push(source_dir.clone());
17  }
18  if search_dirs.is_empty() {
19    search_dirs.push(".".to_string());
20  }
21
22  let mut candidates: Vec<String> = Vec::new();
23  let path_obj = Path::new(path);
24  let has_extension = path_obj.extension().is_some();
25  let source_path = if source_dir.is_empty() { None } else { Some(PathBuf::from(&source_dir)) };
26
27  for dir in &search_dirs {
28    let base = PathBuf::from(dir).join(path);
29    if has_extension {
30      if base.exists() {
31        let rel = match &source_path {
32          Some(sp) => base.strip_prefix(sp).unwrap_or(&base).to_string_lossy().to_string(),
33          None => base.to_string_lossy().to_string(),
34        };
35        candidates.push(rel);
36      }
37    } else {
38      // Search for path with any extension
39      let parent = base.parent().unwrap_or(Path::new("."));
40      let stem = base.file_name().map(|s| s.to_string_lossy().to_string()).unwrap_or_default();
41      if let Ok(entries) = std::fs::read_dir(parent) {
42        for entry in entries.flatten() {
43          let fname = entry.file_name().to_string_lossy().to_string();
44          if let Some(dot_pos) = fname.find('.') {
45            if fname[..dot_pos] == stem {
46              let full = entry.path();
47              let rel = match &source_path {
48                Some(sp) => full.strip_prefix(sp).unwrap_or(&full).to_string_lossy().to_string(),
49                None => full.to_string_lossy().to_string(),
50              };
51              candidates.push(rel);
52            }
53          }
54        }
55      }
56    }
57  }
58
59  // If no candidates found and path has no extension, try common image extensions
60  // (matching Perl's pathname_findall with types => ['*'] which tries all known types)
61  if candidates.is_empty() && !has_extension {
62    // Perl typically returns just the first match (png)
63    if let Some(ext) = ["png", "jpg", "jpeg", "gif", "pdf", "eps", "svg"].first() {
64      let with_ext = format!("{path}.{ext}");
65      candidates.push(with_ext);
66    }
67  }
68
69  // Deduplicate while preserving order
70  let mut seen = std::collections::HashSet::new();
71  candidates.retain(|c| seen.insert(c.clone()));
72
73  if candidates.is_empty() {
74    path.to_string()
75  } else {
76    candidates.join(",")
77  }
78}

80LoadDefinitions!({

82  // (See LaTeXML::Post::Graphics for suggested postprocessing)
83
84  // Load the base graphics package

87  // Raw TeX graphicx.sty redefines \rotatebox with \protected\def which overrides
88  // our DefConstructor from graphics_sty.rs. Re-register using Let! to restore
89  // the Rust definition. The Rust definition (from graphics_sty.rs) is stored
90  // as \rotatebox's meaning before the raw TeX override happens, so we can
91  // restore it by re-defining here after graphics is loaded.
92  // Perl: graphicx.sty.ltxml doesn't need this because .ltxml bindings prevent
93  // raw TeX loading entirely.
94  DefConstructor!("\\rotatebox OptionalKeyVals:Grot {Float} {}",
95    "<ltx:inline-block angle='#angle' width='#width' height='#height' depth='#depth' innerwidth='#innerwidth' innerheight='#innerheight' innerdepth='#innerdepth' xtranslate='#xtranslate' ytranslate='#ytranslate'>#3</ltx:inline-block>",
96    mode => "restricted_horizontal", enter_horizontal => true,
97    after_digest => sub[whatsit] {
98      let angle = whatsit.get_arg(2).map(|a| a.to_attribute().parse::<f64>().unwrap_or(0.0)).unwrap_or(0.0);
99      if let Some(body) = whatsit.get_arg(3) {
100        if let Ok(props) = crate::package::graphics_sty::rotated_properties(body.clone(), angle, false) {
101          for (k, v) in props {
102            whatsit.set_property(k, v);
103          }
104        }
105      }
106    });
107
108  // Internal macros for graphicx sizing
109  DefMacro!("\\Gin@ewidth", "");
110  DefMacro!("\\Gin@eheight", "");
111  DefMacro!("\\Gin@eresize", "");
112  DefMacro!("\\Gin@esetsize", "");
113
114  // KeyVal options for the Gin family
115  // NOTE: GraphixDimension and GraphixDimensions are custom parameter types
116  // defined in graphics.sty.ltxml. We use "Dimension" as a closest approximation
117  // for GraphixDimension, and "" (plain text) for GraphixDimensions (sequence of 4 dims).
118  DefKeyVal!("Gin", "width", "Dimension");
119  DefKeyVal!("Gin", "height", "Dimension");
120  DefKeyVal!("Gin", "totalheight", "Dimension");
121  DefKeyVal!("Gin", "keepaspectratio", "", "true");
122  DefKeyVal!("Gin", "clip", "", "true");
123  DefKeyVal!("Gin", "scale", "");
124  DefKeyVal!("Gin", "angle", "");
125  DefKeyVal!("Gin", "alt", "");
126  DefKeyVal!("Gin", "trim", "");
127  DefKeyVal!("Gin", "viewport", "");
128
129  // LaTeXML extensions:
130  DefKeyVal!("Gin", "vrml", "Semiverbatim");
131  DefKeyVal!("Gin", "magnifiable", "", "true");
132
133  // Redefine \includegraphics to dispatch based on bracket syntax:
134  // If a second [] follows, use the old graphics.sty-style \@includegraphics,
135  // otherwise use the graphicx keyval-style \@includegraphicx.
136  DefMacro!(
137    "\\includegraphics OptionalMatch:* []",
138    "\\@ifnextchar[{\\@includegraphics#1[#2]}{\\@includegraphicx#1[#2]}"
139  );
140
141  // The graphicx-style \includegraphics with keyval options.
142  // Perl: properties callback computes path, candidates, options from keyval args.
143  DefConstructor!(
144    "\\@includegraphicx OptionalMatch:* OptionalKeyVals:Gin Semiverbatim",
145    "<ltx:graphics graphic='#path' candidates='#candidates' options='#options'/>",
146    enter_horizontal => true,
147    properties => sub[args] {
148      // arg 0: starred, arg 1: keyvals, arg 2: graphic path
149      let path = args[2].as_ref().map(|a| a.to_attribute()).unwrap_or_default();
150      let path = path.trim().to_string();
151      // Perl: image_candidates searches filesystem for files matching path with any extension
152      let candidates = image_candidates(&path);
153      // Build options string from keyval pairs, matching Perl's graphicX_options
154      let starred = args[0].is_some();
155      let mut options_vec: Vec<String> = Vec::new();
156      if starred {
157        options_vec.push(s!("clip=true"));
158      }
159      let mut saw_w = false;
160      let mut saw_h = false;
161      let mut has_keepaspectratio = false;
162      if let Some(ref kv_digested) = args[1] {
163        if let DigestedData::KeyVals(ref kv) = kv_digested.data() {
164          for (key, value) in kv.get_pairs() {
165            if key.ends_with("width") { saw_w = true; }
166            if key.ends_with("height") { saw_h = true; }
167            if key == "keepaspectratio" { has_keepaspectratio = true; }
168            let val_str = value.to_string();
169            let val_str = val_str.replace(',', "\\,");
170            options_vec.push(format!("{key}={val_str}"));
171          }
172        }
173      }
174      // Auto-add keepaspectratio if only width or height (not both) specified
175      if (saw_w ^ saw_h) && !has_keepaspectratio {
176        options_vec.push(s!("keepaspectratio=true"));
177      }
178      let options = options_vec.join(",");
179      Ok(stored_map!("path" => path, "candidates" => candidates, "options" => options))
180    }
181  );
