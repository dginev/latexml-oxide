4LoadDefinitions!({
5  // Perl: multirow.sty.ltxml
6
7  DefPrimitive!("\\multirowsetup", None);
8  DefPrimitive!("\\multirow[]{Float}[Number]{}[Dimension]{}", sub[(attachment, nrows, _bigstruts, _width, _fixup, content)] {
9    if let Some(alignment) = lookup_alignment() {
10      if let Some(data) = alignment.alignment_cell() {
11        let mut data_lock = data.borrow_mut();
12        if let Some(colspec) = data_lock.current_column() {
13          let rowspan_f = nrows.value_f64();
14          let rowspan = if rowspan_f < 0.0 {
15            Warn!("unsupported", "multirow",
16              "Negative row sizes for \\multirow are not yet supported.");
17            1usize
18          } else if rowspan_f != rowspan_f.floor() {
19            Warn!("unsupported", "multirow",
20              "Fractional row sizes for \\multirow are not yet supported.");
21            rowspan_f as usize
22          } else {
23            rowspan_f as usize
24          };
25          colspec.rowspan = Some(rowspan);
26          if let Some(att) = attachment {
27            colspec.vattach = Some(translate_attachment(att).to_string());
28          }
29        }
30      }
31    }
32    let rev: Vec<Token> = content.revert();
33    let mut toks: Vec<Token> = vec![T_CS!("\\hbox"), T_BEGIN!(), T_CS!("\\multirowsetup")];
34    toks.extend(rev);
35    toks.push(T_END!());
36    stomach::digest(Tokens::new(toks))?;
37    Ok(())
38  });
39});
