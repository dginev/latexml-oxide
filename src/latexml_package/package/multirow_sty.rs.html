4LoadDefinitions!({
5  // Perl: multirow.sty.ltxml
6
7  DefPrimitive!("\\multirowsetup", None);
8  // \multirow: structural split into DefMacro wrapper + internal DefPrimitive
9  // setup (2-layer pattern). Perl's single DefPrimitive digests all args;
10  // Rust separates the alignment-cell state mutation (primitive, below)
11  // from the \hbox-wrapped content flow (DefMacro at :52). The split lets
12  // content flow naturally through alignment cell boxes and enables the
13  // text-mode \hbox wrap used by 1004.2626 Table 6. (Not WISDOM #41 —
14  // that entry is about math-mode ParameterType adaptations.)
15  //
16  // \lx@multirow@setup: internal primitive that sets rowspan/vattach on current cell.
17  // Separated from content so that content flows naturally through alignment cell boxes.
18  DefPrimitive!("\\lx@multirow@setup{Float}[]{}", sub[(nrows, attachment, _width)] {
19    if let Some(alignment) = lookup_alignment() {
20      if let Some(data) = alignment.alignment_cell() {
21        let mut data_lock = data.borrow_mut();
22        if let Some(colspec) = data_lock.current_column() {
23          let rowspan_f = nrows.value_f64();
24          let rowspan = if rowspan_f < 0.0 {
25            Warn!("unsupported", "multirow",
26              "Negative row sizes for \\multirow are not yet supported.");
27            1usize
28          } else if rowspan_f != rowspan_f.floor() {
29            Warn!("unsupported", "multirow",
30              "Fractional row sizes for \\multirow are not yet supported.");
31            rowspan_f as usize
32          } else {
33            rowspan_f as usize
34          };
35          colspec.rowspan = Some(rowspan);
36          // Only set vattach when optional [] is explicitly provided and non-empty
37          if let Some(ref att) = attachment {
38            let att_str = att.to_string();
39            if !att_str.trim().is_empty() {
40              colspec.vattach = Some(translate_attachment(att).to_string());
41            }
42          }
43        }
44      }
45    }
46    Ok(())
47  });
48  // \multirow[vpos]{nrows}[bigstruts]{width}[fixup]{content}
49  //
50  // Perl multirow.sty.ltxml L19 defines `\multirow` as DefPrimitive
51  // whose sub body reads the Alignment column, computes rowspan, and
52  // digests the content inline. Rust uses DefMacro instead so that
53  // the content can be wrapped in `\hbox{\multirowsetup #6}` at gullet
54  // expansion time — the \hbox forces text mode so nested `$…$` cleanly
55  // switches into math; without it, inside array-cell (outer math)
56  // context the inner `$` toggles math OFF, landing content in text
57  // mode with script errors (arxiv 1004.2626 Table 6 was the witness).
58  //
59  // Intentional DefPrimitive → DefMacro kind divergence (WISDOM #44).
60  // The rowspan/colspec computation moved into `\lx@multirow@setup`,
61  // a paired DefPrimitive that runs after gullet expansion. Observable
62  // XML remains identical to the Perl port for well-formed input, and
63  // strictly better (no script-mode bleed) for malformed input.
64  // Inside the hbox, `\\` is still bound to the surrounding tabular's
65  // `\lx@alignment@newline` which fires `\lx@begin@alignment` — invalid
66  // in restricted_horizontal mode. multirow's content allows `\\` as a
67  // soft line break (the package stacks rows visually). Rebind `\\` to
68  // `\lx@newline` (horizontal-mode break) at the start of the hbox so
69  // nested `\\` survives. Witness: arXiv:1504.01713 line 694
70  // `\multirow{6}{17pt}{$\alpha_\mathrm{exp}$\\$\alpha_\mathrm{C}$}`.
71  DefMacro!("\\multirow[]{Float}[Number]{}[Dimension]{}",
72    "\\lx@multirow@setup{#2}[#1]{#4}\\hbox{\\let\\\\\\lx@newline\\multirowsetup #6}");
73});
