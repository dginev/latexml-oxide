1//! babel-language stubs for `.ldf` files that aren't installed in
2//! minimal TeXLive environments.
3//!
4//! When babel-italian (or babel-spanish, babel-portuges, …) is
5//! missing on disk, babel.sty L4175's `\InputIfFileExists{<lang>.ldf}`
6//! fails and babel errors with "Unknown option '<lang>'. Either you
7//! misspelled it or the language definition file <lang>.ldf was not
8//! found".
9//!
10//! Each stub here:
11//!   - registers via `lib.rs` as `<lang>.ldf` so find_file resolves
12//!     to our binding rather than the missing file;
13//!   - allocates `\l@<lang>` and defines empty `\captions<lang>` /
14//!     `\extras<lang>` / `\noextras<lang>` / `\date<lang>` hooks so
15//!     babel's `\selectlanguage` chain runs cleanly.
16//!
17//! The actual ISO-639 → xml:lang mapping happens at
18//! `\selectlanguage` time via
19//! `babel_support_sty::babel_language_to_iso`. We don't reproduce
20//! the per-language typographic conventions (punctuation, dates,
21//! etc.) — they're a typesetting-only concern, irrelevant for
22//! XML/HTML output.
23//!
24//! Cluster (recent stages): italian 7, spanish 5, brazil 5, vietnamese
25//! 3, portuguese 3, brazilian 2, polish 2, romanian 2, icelandic 2,
26//! czech 2, turkish 1, slovene 1, portuges 1, farsi 1, dutch 1,
27//! arabic 1 ≈ 38 papers.
28use crate::prelude::*;

30fn install_lang_stub(lang: &str) -> Result<()> {
31  // `\newlanguage` is a TeX macro of the form
32  // `\alloc@9\language\chardef\@cclvi`, which delegates to a 5-arg
33  // `\alloc@{}{}{}{}{}` macro. So a *raw* `\newlanguage\csname
34  // l@<lang>\endcsname` makes `\alloc@` grab `\csname` as the 5th
35  // argument (parameter-text token-grabbing does NOT expand `\csname`)
36  // and leaves `l@<lang>\endcsname` orphaned in the input stream — the
37  // unmatched `\endcsname` then cascades through every following
38  // package-load → 100 errors → fatal TooManyErrors abort.
39  //
40  // Force `\csname...\endcsname` to expand *first* with
41  // `\expandafter\newlanguage\csname...`, so `\newlanguage` receives
42  // the resolved `\l@<lang>` token directly.
43  //
44  // (The `\providecommand\captions{lang}{{}}` lines below are correct
45  // — Rust `format!` substitutes `{lang}` inline, giving
46  // `\providecommand\captionsbrazil{}` etc.)
47  let body = format!(
48    r"\expandafter\ifx\csname l@{lang}\endcsname\relax
49      \expandafter\newlanguage\csname l@{lang}\endcsname
50    \fi
51    \providecommand\captions{lang}{{}}%
52    \providecommand\extras{lang}{{}}%
53    \providecommand\noextras{lang}{{}}%
54    \providecommand\date{lang}{{}}",
55    lang = lang
56  );
57  latexml_core::stomach::raw_tex(&body)?;
58  Ok(())
59}
60
61pub fn load_italian() -> Result<()>    { install_lang_stub("italian") }
62pub fn load_spanish() -> Result<()>    {
63  install_lang_stub("spanish")?;
64  // babel-spanish-specific `\decimalpoint` — switches decimal separator
65  // from `,` (Spanish default) to `.`. We don't render locale-aware
66  // numerics; HTML uses `.` by default. No-op preserves intent.
67  // Driver 2511.19353 (`\usepackage[spanish]{babel}\decimalpoint`).
68  // Also `\decimalcomma` for the reverse direction.
69  latexml_core::stomach::raw_tex(
70    r"\providecommand\decimalpoint{}\providecommand\decimalcomma{}"
71  )?;
72  Ok(())
73}
