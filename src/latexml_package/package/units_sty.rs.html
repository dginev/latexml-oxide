4/// Perl: units_assert_units — walks DOM and sets role=ID, class=ltx_unit on UNKNOWN XMToks
5fn units_assert_units(document: &mut Document, node: &Node) -> Result<()> {
6  let qname = get_node_qname(node);
7  let tag = arena::to_string(qname);
8  if tag == "ltx:XMTok" {
9    let role = node.get_attribute("role").unwrap_or_default();
10    if role == "UNKNOWN" || role.is_empty() {
11      let mut n = node.clone();
12      n.set_attribute("role", "ID")?;
13      document.add_class(&mut n, "ltx_unit")?;
14    }
15  } else if !tag.is_empty() {
16    for child in document.findnodes("*", Some(node)) {
17      units_assert_units(document, &child)?;
18    }
19  }
20  Ok(())
21}
