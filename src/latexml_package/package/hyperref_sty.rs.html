331  // \href{url}{text}
332  //
333  // `protected => true`: in real hyperref `\href` is a *robust* command
334  // (`\DeclareRobustCommand`/`\protected`), so an `\edef`/`\xdef` over it
335  // leaves the literal `\href{url}{text}` in the macro body rather than
336  // expanding it. LaTeXML's expansion re-emits `\href` itself (for the
337  // `\lx@hyper@url@` reversion `#1`), so WITHOUT the protected flag a
338  // partial-expansion context re-expands that re-emitted `\href` forever:
339  // `\href` → `\lx@hyper@url@\href{}{}…` → (constructor kept) → `\href` →
340  // … an unbounded expansion loop. Perl LaTeXML omits the flag and so
341  // *hangs* on `\xdef\x{\href{u}{t}}` (rc=124); marking it protected is
342  // both faithful to real-TeX robustness and fixes the loop. At top-level
343  // digestion `fully_expand` is true, so normal `\href{…}{…}` still
344  // expands exactly as before — only edef-family contexts defer it.
345  // Witness 2110.10227 (ems-journal.sty `\Emsaffil`→`\build@ffil`'s
346  // `\xdef\ems@temp{…\href{mailto:…}{\mbox{…}}…}`).

349    "\\lx@hyper@url@\\href{}{}{#1}{#2}",

359  // Redefine \url{url} from url.sty...
360  // It's slightly different in that it expands the argument
361  // Redefine \@url to sanitize the argument less
362  DefMacro!("\\lx@hyper@url Token", sub[(cmd)] {
363    // `\url` requires a following delimiter/argument token; on input-exhaustion
364    // `read_token_required` emits the parity "file ended while scanning use of
365    // \url" error (real TeX "Emergency stop"). Don't panic — close the
366    // `\begingroup` `\url` opened (see the `\url` def below) and emit nothing.
367    let Some(open) = read_token_required(&cmd.to_string())? else {
368      return Ok(Tokens::new(vec![T_CS!("\\endgroup")]));
369    };
370    begin_semiverbatim(Some(&['%']));
371    let_i(&T_ACTIVE!('~'), &T_OTHER!("~"), None); // Needs special protection?
372    // URLs are verbatim: any character a shorthand package made ACTIVE must
373    // appear literally, not expand. We read the arg with partial expansion
374    // ("expand as we go", below) to allow `\macro`s in URLs, but that also
375    // expands active chars — and an active char whose meaning is a `\newif`
376    // conditional (e.g. `"` made active by bosisio quotes.sty →
377    // `\@VIRGOLETTE` = `\if@virgolette…\else…\fi`) leaks its `\fi` through
378    // `read_balanced`, so the conditional "falls off end". Neutralise the
379    // common shorthand-active chars to OTHER (same `\let`-to-literal trick as
380    // `~` above). `:` is doubly important — French babel makes it active and
381    // it is ubiquitous in `http://…` URLs. Witness 1910.09629 (revtex4 →
382    // hyperref `\url`, quotes.sty active `"`, `.bbl` `\url{"http://…"}`).
383    for ch in ['"', ':', ';', '!', '?', '\'', '`'] {
384      if lookup_catcode(ch) == Some(Catcode::ACTIVE) {
385        assign_catcode(ch, Catcode::OTHER, Some(Scope::Local));
386      }
387    }
388    let (open,close,url) = if open.get_catcode() == Catcode::BEGIN {
389      ( T_OTHER!("{"), T_OTHER!("}"),
390        read_balanced(ExpansionLevel::Partial,false,false)?.unwrap_or_default()) // Expand as we go!
391    } else {
392      ( T_OTHER!("{"), T_OTHER!("}"),
393        Tokens!(open.as_other()) )
394    };
395    end_semiverbatim()?;
396    let toks : Vec<Token> = url.unlist().into_iter()
397      .filter(|t| t.get_catcode() != Catcode::SPACE)
398      // Identical with url's \@Url except, let CS's through!
399      .map(|t| if t.get_catcode() == Catcode::CS { t } else { t.as_other() })
400      .collect();
401    let mut url_wrapped = vec![T_CS!("\\UrlFont"), T_CS!("\\UrlLeft")];
402    url_wrapped.extend(toks.clone());
403    url_wrapped.push(T_CS!("\\UrlRight"));
404    let mut invocation_tokens = Invocation!(T_CS!("\\lx@hyper@url@"),vec![
405        Tokens!(cmd.as_other()),
406        Tokens!(open),
407        Tokens!(close),
408        Tokens::new(toks),
409        Tokens::new(url_wrapped)]).unlist();
410    invocation_tokens.push(T_CS!("\\endgroup"));
411    Tokens::new(invocation_tokens)
412  });

1105// Perl: hyperref.sty.ltxml `localized_anchor`. DFS walks the current node's
1106// subtree and wraps the first descendant that ltx:anchor is allowed to
1107// contain. The traversal mirrors Perl's pop+unshift order: candidates are
1108// popped from the back and child nodes are unshifted to the front, so we
1109// visit the rightmost element of each level before descending.
1110fn localized_anchor(document: &mut Document, whatsit: &Whatsit) -> CoreResult<()> {
1111  let id = match whatsit.get_property("id") {
1112    Some(v) => v.to_string(),
1113    None => return Ok(()),
1114  };
1115  let mut candidates: Vec<Node> = vec![document.get_node().clone()];
1116  let mut found: Option<Node> = None;
1117  while let Some(candidate) = candidates.pop() {
1118    match candidate.get_type() {
1119      Some(NodeType::ElementNode) => {
1120        let qname = get_node_qname(&candidate);
1121        if can_contain_qsym(pin!("ltx:anchor"), qname) {
1122          found = Some(candidate);
1123          break;
1124        }
1125        // Perl: unshift(@candidates, $candidate->childNodes); pushes the child
1126        // list to the front, so the rightmost child is popped next.
1127        let children = candidate.get_child_nodes();
1128        for child in children.into_iter().rev() {
1129          candidates.insert(0, child);
1130        }
1131      },
1132      // Perl: any non-element node short-circuits; the candidate (text node)
1133      // is then wrapped, which works because anchor.model = Inline allows text.
1134      _ => {
1135        found = Some(candidate);
1136        break;
1137      },
1138    }
1139  }
1140  if let Some(target) = found {
1141    match document.wrap_nodes("ltx:anchor", vec![target])? {
1142      Some(mut anchor) => {
1143        document.set_attribute(&mut anchor, "xml:id", &id)?;
1144        if document.is_open(&anchor) {
1145          document.close_node(&anchor)?;
1146        }
1147      },
1148      _ => {
1149        Warn!(
1150          "malformed",
1151          "ltx:anchor",
1152          &s!(
1153            "No available insertion point for ltx:anchor, failing \\hypertarget to {}",
1154            id
1155          )
1156        );
1157      },
1158    }
1159  } else {
1160    Warn!(
1161      "malformed",
1162      "ltx:anchor",
1163      &s!(
1164        "No available insertion point for ltx:anchor, failing \\hypertarget to {}",
1165        id
1166      )
1167    );
1168  }
1169  Ok(())
1170}
