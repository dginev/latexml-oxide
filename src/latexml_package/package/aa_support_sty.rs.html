74  // Perl L88-93: `\abstract{#1}` reads one arg, peeks for T_BEGIN; if present,
75  // dispatches to \abstract@new (4 more args), else to \abstract@old. Our
76  // previous always-1-arg stub dumped the remaining 4 `{…}` groups into the
77  // document body, reordering the frontmatter/body (arxiv 1209.2771:
78  // abstract paragraphs showed up BEFORE <title> and no <abstract> emitted).
79  DefMacro!("\\abstract{}", sub[(arg1)] {
80    gullet::skip_spaces()?;
81    let next_is_begin = gullet::if_next(T_BEGIN!())?;
82    let target = if next_is_begin { "\\abstract@new" } else { "\\abstract@old" };
83    let mut out = vec![T_CS!(target), T_BEGIN!()];
84    out.extend(arg1.unlist());
85    out.push(T_END!());
86    Ok(Tokens::new(out))
87  });

200    mode => "display_math",

202      prepare_equation_counter(stored_map!("numbered" => true, "preset" => true));
