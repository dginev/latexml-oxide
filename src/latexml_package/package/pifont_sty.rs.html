8  // \Pisymbol{font}{code} — decode a codepoint from a Pi font
9  DefPrimitive!("\\Pisymbol{}{Number}", sub[(pifont, code)] {
10    let font_name = pifont.unwrap().to_string();
11    let code_val = code.value_of();
12    let (glyph, font) = font_decode(code_val as i32, Some(&font_name), None);
13    // Perl: Box($glyph, $font, ...) — undef glyph produces empty box
14    let sym = match glyph {
15      Some(ch) => arena::pin(ch.to_string()),
16      None => arena::pin_static(""),
17    };
18    Ok(Digested::from(Tbox::new(
19      sym,
20      font,
21      None,
22      Tokens::new(vec![]),
23      SymHashMap::default(),
24    )))
25  });
26
27  // \lx@Picountersymbol{font}{counter}{codebase} — decode counter-offset symbol
28  DefPrimitive!("\\lx@Picountersymbol{}{}{Number}", sub[(pifont, counter, codebase)] {
29    let font_name = pifont.unwrap().to_string();
30    let counter_name = counter.unwrap().to_string();
31    let base = codebase.value_of();
32    let counter_val = lookup_register(&s!("\\c@{counter_name}"), vec![])?
33      .map(|rv| rv.value_of())
34      .unwrap_or(0);
35    let code = base + counter_val - 1;
36    let (glyph, font) = font_decode(code as i32, Some(&font_name), None);
37    // Perl: Box($glyph, $font, ...) — undef glyph produces empty box
38    let sym = match glyph {
39      Some(ch) => arena::pin(ch.to_string()),
40      None => arena::pin_static(""),
41    };
42    Ok(Digested::from(Tbox::new(
43      sym,
44      font,
45      None,
46      Tokens::new(vec![]),
47      SymHashMap::default(),
48    )))
49  });
50
51  DefMacro!("\\Pilist{}{}", "\\list{\\Pisymbol{#1}{#2}}{}");
52  DefMacro!("\\endPilist", "\\endlist");
53
54  // \lx@defpiautolabel{font}{base} — define pi font auto-labels for enumerate
55  DefMacro!("\\lx@defpiautolabel{}{}", sub[(font, base)] {
56    let font_str = font.unwrap().to_string();
57    let base_str = base.unwrap().to_string();
58    let level = (lookup_int("enumlevel").max(0) + 1) as i64;
59    let postfix = roman_aux(level);
60    // DefMacroI for \theenumX, \p@enumX, \labelenumX
61    let the_body = s!("\\lx@Picountersymbol{{{font_str}}}{{enum{postfix}}}{{{base_str}}}");
62    let the_tokens = latexml_core::mouth::tokenize_internal(&the_body);
63    def_macro(T_CS!(s!("\\theenum{postfix}")), None, the_tokens, None)?;
64    let empty_tokens = Tokens::new(vec![]);
65    def_macro(T_CS!(s!("\\p@enum{postfix}")), None, empty_tokens, None)?;
66    let label_body = s!("\\theenum{postfix}");
67    let label_tokens = latexml_core::mouth::tokenize_internal(&label_body);
68    def_macro(T_CS!(s!("\\labelenum{postfix}")), None, label_tokens, None)?;
69    Ok(Tokens::new(vec![]))
70  });
71
72  DefMacro!("\\Piautolist{}{}", "\\lx@defpiautolabel{#1}{#2}\\enumerate");
73  DefMacro!("\\endPiautolist", "\\endenumerate");
