27/// Perl: amsRearrangeableBindings — creates equationgroup/equation/_Capture_ alignment
28/// amsmath.sty.ltxml lines 121-147
29fn ams_rearrangeable_bindings(
30  template: Template,
31  xml_attributes: HashMap<String, String>,
32) -> Result<()> {
33  let properties = SymHashMap::default();
34  // Create alignment with equationgroup/equation/_Capture_ hooks
35  let alignment = Alignment::new(AlignmentConfig {
36    template: Some(template),
37    open_container: Rc::new(move |document, mut props| {
38      if let Ok(id_props) = ref_step_id("@equationgroup") {
39        if let Some(id) = id_props.get("id") {
40          props.insert(String::from("xml:id"), id.to_string());
41        }
42      }
43      // Merge xml_attributes into props
44      // (attributes passed at creation time)
45      document
46        .open_element("ltx:equationgroup", Some(props), None)
47        .map(Option::Some)
48    }),
49    close_container: Rc::new(|document| document.close_element("ltx:equationgroup")),
50    open_row: Rc::new(|document, mut props| {
51      // Read equation tags from state
52      if let Some(Stored::HashStored(eq_props)) =
53        state::remove_value("EQUATIONROW_PROPS")
54      {
55        if let Some(id) = eq_props.get("id") {
56          props.insert(String::from("xml:id"), Stored::from(id.to_string()));
57        }
58      }
59      // Extract tags (Digested) before converting to string props
60      let tags_digested = props.remove("tags");
61      let str_props: HashMap<String, String> = props.into_iter()
62        .map(|(k, v)| (k, v.to_string()))
63        .collect();
64      document
65        .open_element("ltx:equation", Some(str_props), None)?;
66      // If we have digested tags, absorb them into the opened element
67      if let Some(Stored::Digested(d)) = tags_digested {
68        document.absorb(&d, None)?;
69      }
70      Ok(())
71    }),
72    close_row: Rc::new(|document| document.close_element("ltx:equation")),
73    open_column: Rc::new(|document, props| {
74      document
75        .open_element("ltx:_Capture_", Some(props), None)
76        .map(Option::Some)
77    }),
78    close_column: Rc::new(|document| document.close_element("ltx:_Capture_")),
79    is_math: true,
80    properties,
81    xml_attributes,
82  });
83  assign_alignment(alignment, None);
84  state::let_i(&T_MATH!(), &T_CS!("\\lx@dollar@in@mathmode"), None);
85  state::let_i(
86    &T_CS!("\\\\"),
87    &T_CS!("\\lx@alignment@newline@noskip"),
88    None,
89  );
90  state::let_i(
91    &T_CS!("\\lx@alignment@row@before"),
92    &T_CS!("\\eqnarray@row@before"),
93    None,
94  );
95  state::let_i(
96    &T_CS!("\\lx@alignment@row@after"),
97    &T_CS!("\\eqnarray@row@after"),
98    None,
99  );
100  // Perl: Let('\intertext', '\@ams@intertext');
101  state::let_i(&T_CS!("\\intertext"), &T_CS!("\\@ams@intertext"), None);
102  Ok(())
103}

855/// Perl: rearrangeAMSGather (amsmath.sty.ltxml L400-415)
856/// Each equation row consists of single equation. Pull math content up past _Capture_.
857pub fn rearrange_ams_gather(
858  document: &mut Document,
859  equationgroup: &mut Node,
860) -> Result<()> {
861  let equations: Vec<Node> = document.findnodes("ltx:equation", Some(equationgroup));
862  for mut equation in equations {
863    let cells: Vec<Node> = document.findnodes("ltx:_Capture_", Some(&equation));
864    if cells.is_empty() {
865      continue;
866    }
867    let cell1_children: Vec<Node> = cells[0].get_child_elements();
868    // Check if this equation is really an intertext
869    if cells.len() == 1 && cell1_children.len() == 1 {
870      let class = cell1_children[0].get_attribute("class").unwrap_or_default();
871      if class.contains("ltx_intertext") {
872        // Replace equation with the block
873        let mut block = cell1_children[0].clone();
874        block.unlink_node();
875        equation.add_prev_sibling(&mut block).ok();
876        equation.unlink_node();
877        continue;
878      }
879    }
880    if cells.len() == 1 && cell1_children.is_empty() {
881      // Empty row — remove it
882      equation.unlink_node();
883      continue;
884    }
885    // Unwrap _Capture_ elements, set Math mode to display
886    let children: Vec<Node> = equation.get_child_elements();
887    for child in children {
888      let qname = document::get_node_qname(&child);
889      if qname == arena::pin_static("ltx:_Capture_") {
890        document.unwrap_nodes(child)?;
891      }
892    }
893    // Set mode='display' on Math elements
894    let maths: Vec<Node> = document.findnodes("ltx:Math", Some(&equation));
895    for mut math in maths {
896      document.set_attribute(&mut math, "mode", "display")?;
897    }
898  }
899  Ok(())
900}
