1//! standalone.sty — compile standalone sub-documents
2//! Perl: standalone.sty.ltxml (40 lines).
3//! NOTE: standalone.cls is handled separately; this is the .sty package.
4use crate::prelude::*;
5
6/// The `standalone.cls` options that load a same-named package
7/// (standalone.cls L171/193/237/249/255, resolved at L562 and L611-620).
8/// Every other option — `crop`, `multi`, `math`, `beamer`, `float`, `png`,
9/// `border=`, `class=`, `10pt`/`11pt`/`12pt`, … — the class handles itself.
10const CLASS_OPTION_PACKAGES: [&str; 5] = ["tikz", "pstricks", "preview", "varwidth", "multido"];
11
12#[rustfmt::skip]
13LoadDefinitions!({
14  // BEYOND PERL (the Perl standalone.sty.ltxml omits these): the real
15  // standalone.sty has exactly TWO *unconditional* `\RequirePackage`s —
16  // `xkeyval` (L107) and `currfile` (L305). (Every other require is guarded:
17  // engine probes `ifpdf`/`ifluatex`/`ifxetex`/`shellesc`, and the
18  // `\IfFileExists`/option-gated `varwidth`/`trimclip`/`adjustbox`/`gincltex`/
19  // `filemod-expmin`.) Restore just those two, which real LaTeX always provides:
20  //   * `xkeyval` defines `\define@key` (and the wider keyval family);
21  //   * `currfile` `\RequirePackage{filehook}` (currfile.sty L30), and filehook
22  //     defines the package-file hooks `\AtEndOfPackageFile`/`\AtBeginOfPackageFile`/…
23  // sTeX 3.x leans on both: `\AtEndOfPackageFile{graphicx}{\define@key{Gin}
24  // {archive}{…}}` (stex.sty L2134) — without them the hook is undefined and its
25  // deferred body runs prematurely (`\define@key` undefined). Rust ships bindings
26  // for all three. Witness: raw stex.sty under ar5iv. (Unconditional — a binding
27  // emulates the package identically regardless of INCLUDE_STYLES; both requires
28  // resolve to always-available package bindings.)
29  RequirePackage!("xkeyval");
30  RequirePackage!("currfile");
31
32  DefMacro!("\\@standalone@end@input", "\\egroup\\endinput");
33
34  // Perl L21-23: DefPrimitiveI \@standalone@start@input — sets inPreamble = 0.
35  DefPrimitive!("\\@standalone@start@input", {
36    assign_value("inPreamble", false, None);
37  });
38
39  // Perl L24-33: DefPrimitive \@standalone@documentclass[]{} — open a
40  // group, mark inPreamble = 1, RequirePackage each comma-separated entry
41  // of the OPTIONAL `[]` argument (Perl binds `$packages = $_[1]`, the
42  // optional), and alias \begin{document}/\end{document} to the start/end
43  // input primitives so the sub-document is injected as a bounded scope
44  // inside the outer document.
45  //
46  // OXIDIZED_DESIGN #63: NEITHER argument is a package list, so both are
47  // gated. The mandatory class name is ignored outright — the parent already
48  // loaded a class, and requiring it warned `missing_file:article` on a
49  // `\documentclass{article}` child (#293). The optional list holds class
50  // OPTIONS; Perl requires all of them for every class, so
51  // `\documentclass[12pt]{article}` warns `missing_file:12pt` (#309).
52  // standalone.sty L604-614 consults a subfile's options only when the
53  // subfile's class is literally `standalone`, so we require them only there
54  // — and only the ones standalone.cls turns into a package load, which is
55  // what makes `\documentclass[tikz]{standalone}` work (upstream LaTeXML#1432,
56  // the reason this loop exists).
57  // The optional argument is read as `OptionalKeyVals`, NOT as a raw string we
58  // comma-split ourselves: a class option list IS a keyval list, and every
59  // option here has a valued form — `\sa@boolorvalue` accepts `varwidth=5cm`
60  // and `tikz=true` exactly as it accepts bare `varwidth`/`tikz`
61  // (standalone.sty L815-824), and `border={1pt 2pt}` puts a brace group in the
62  // list. Splitting on `,` and matching the whole item missed every valued form
63  // — `[varwidth=5cm]{standalone}` then lost the package and reported
64  // `Error:undefined:{varwidth}` where pdflatex is clean. Reusing the engine's
65  // keyval reader gets brace-aware splitting and key/value separation for free,
66  // and keeps this on the same parser `\documentclass`/`\usepackage` options
67  // already flow through instead of a second, weaker one.
68  DefPrimitive!("\\@standalone@documentclass OptionalKeyVals {}", sub[(options_kv, class_tks)] {
69    bgroup();
70    assign_value("inPreamble", true, None);
71    if class_tks.to_string().trim() == "standalone"
72      && let Some(kv) = options_kv.as_ref()
73    {
74      // Match on the KEY, so `varwidth` and `varwidth=5cm` behave alike. An
75      // absent optional yields no pairs ⇒ nothing required.
76      for (key, _value) in kv.get_pairs() {
77        if CLASS_OPTION_PACKAGES.contains(&key.trim()) {
78          RequirePackage!(key.trim());
79        }
80      }
81    }
82    Let!(T_CS!("\\begin{document}"), T_CS!("\\@standalone@start@input"));
83    Let!(T_CS!("\\end{document}"),   T_CS!("\\@standalone@end@input"));
84  });
85
86  // Perl L35-36: AtBeginDocument — swap \documentclass to the intercept.
87  // Native push to @at@begin@document so the hook fires at the same
88  // lifecycle point Perl uses.
89  at_begin_document(TokenizeInternal!(r"\let\documentclass\@standalone@documentclass"))?;
90
91  // standalone.sty L1014: \includestandalone[opts]{file}. Treat as
92  // \includegraphics{file} so the figure surfaces in the XML output.
93  // Witness 2406.02722.
94  DefMacro!("\\includestandalone[]{}", "\\includegraphics{#2}");
95});
