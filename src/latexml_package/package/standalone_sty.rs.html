1//! standalone.sty — compile standalone sub-documents
2//! Perl: standalone.sty.ltxml (40 lines).
3//! NOTE: standalone.cls is handled separately; this is the .sty package.
4use crate::prelude::*;
5
6#[rustfmt::skip]
7LoadDefinitions!({
8  DefMacro!("\\@standalone@end@input", "\\egroup\\endinput");
9
10  // Perl L21-23: DefPrimitiveI \@standalone@start@input — sets inPreamble = 0.
11  DefPrimitive!("\\@standalone@start@input", {
12    assign_value("inPreamble", false, None);
13  });
14
15  // Perl L24-33: DefPrimitive \@standalone@documentclass[]{} — open a
16  // group, mark inPreamble = 1, RequirePackage each comma-separated entry
17  // of the OPTIONAL `[]` argument (Perl binds `$packages = $_[1]`, the
18  // optional), and alias \begin{document}/\end{document} to the start/end
19  // input primitives so the sub-document is injected as a bounded scope
20  // inside the outer document.
21  // NOTE: the mandatory `{}` class name is intentionally IGNORED (the parent
22  // already loaded a class). Requiring it as a package — the earlier port
23  // used the mandatory arg — warned `missing_file:article` on a
24  // `\documentclass{article}` standalone child (issue #293).
25  DefPrimitive!("\\@standalone@documentclass[]{}", sub[(packages_tks, _class_tks)] {
26    bgroup();
27    assign_value("inPreamble", true, None);
28    // Perl `split(",", ToString($packages))`: an absent optional stringifies
29    // to "" (⇒ no packages required), matching the `\documentclass{article}`
30    // no-options case.
31    let packages_str = packages_tks.map(|t| t.to_string()).unwrap_or_default();
32    for pkg in packages_str.split(',') {
33      let pkg = pkg.trim();
34      if !pkg.is_empty() {
35        RequirePackage!(pkg);
36      }
37    }
38    Let!(T_CS!("\\begin{document}"), T_CS!("\\@standalone@start@input"));
39    Let!(T_CS!("\\end{document}"),   T_CS!("\\@standalone@end@input"));
40  });
41
42  // Perl L35-36: AtBeginDocument — swap \documentclass to the intercept.
43  // Native push to @at@begin@document so the hook fires at the same
44  // lifecycle point Perl uses.
45  at_begin_document(TokenizeInternal!(r"\let\documentclass\@standalone@documentclass"))?;
46
47  // standalone.sty L1014: \includestandalone[opts]{file}. Treat as
48  // \includegraphics{file} so the figure surfaces in the XML output.
49  // Witness 2406.02722.
50  DefMacro!("\\includestandalone[]{}", "\\includegraphics{#2}");
51});
