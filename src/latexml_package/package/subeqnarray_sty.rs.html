4LoadDefinitions!({
5  // Perl: subeqnarray.sty.ltxml L21-22 — both \subeqnarray and
6  // \endsubeqnarray carry `locked => 1`. The lock matters because the
7  // immediately-following InputDefinitions pulls the raw subeqnarray.sty,
8  // which redefines both names; without the lock the raw-TeX version
9  // overwrites our subnumbering trampolines. See arXiv:hep-th/0002165.
10  DefMacro!("\\subeqnarray",
11    "\\lx@equationgroup@subnumbering@begin\\bgroup\\lx@begin@display@math",
12    locked => true);
13  DefMacro!("\\endsubeqnarray",
14    "\\lx@end@display@math\\egroup\\lx@equationgroup@subnumbering@end",
15    locked => true);
16
17  InputDefinitions!("subeqnarray", noltxml => true, extension => Some(Cow::Borrowed("sty")));
18});
