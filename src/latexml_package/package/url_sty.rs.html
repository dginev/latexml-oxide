37  // This is an extended version of \Url that takes an extra token as 1st arg.
38  // That token is the cs that invoked it, so that it can be reflected in the generated XML,
39  // as well as used to generate the reversion.
40  // In any case, we read the verbatim arg, and build a Whatsit for @@Url
41  DefMacro!("\\lx@url@url Token", sub[(cmd)] {
42    let perc = vec!['%'];
43   begin_semiverbatim(Some(&perc));
44    // `\url`/`\path` requires a following delimiter/argument token; on
45    // input-exhaustion `read_token_required` emits the parity "file ended while
46    // scanning use of \url" error. Don't panic — unwind the semiverbatim frame
47    // and close the `\begingroup` `\url`/`\path` opened, emitting nothing.
48    let Some(mut open) = read_token_required(&cmd.to_string())? else {
49      end_semiverbatim()?;
50      return Ok(Tokens::new(vec![T_CS!("\\endgroup")]));
51    };
52    let close;
53    let url = if open.get_catcode() == Catcode::BEGIN {
54      open = T_OTHER!("{");
55      close = T_OTHER!("}");
56      read_balanced(ExpansionLevel::Off,false,false)?.unwrap_or_default()
57    } else {
58      // The ORIGINAL catcode of the delimiter token distinguishes two
59      // very different `\url`/`\path` invocation shapes:
60      let delim_is_explicit = open.get_catcode() == Catcode::OTHER;
61      open = open.as_other();
62      close = open;
63      // EXPLICIT punctuation delimiter (`\url|...|`, `\path!...!`): the
64      // chars between the two delimiter tokens — INCLUDING any `{`/`}` —
65      // are LITERAL url content. Demote `{`/`}` to OTHER so
66      // read_until_token doesn't engage its balanced-read branch on `{`
67      // and overshoot the closing delimiter. Real users write
68      // `\\path|{partial,| ... |partial}|` across multiple `\\urldef`
69      // calls, where `{`/`}` are literal chars in separate `|...|`
70      // paths. Driver 1906.08946 — without this demotion read_until('|')
71      // on `\\path|{...,|` reads PAST the closing `|` looking for a
72      // matching `}`, swallowing intervening `\\urldef` declarations.
73      // (Perl OVERSHOOTS here too — this is a deliberate surpass.)
74      //
75      // SPACE-FORM (`\url www...`, delimiter is the first content char =
76      // a LETTER): a degenerate/misused invocation. The verbatim scan
77      // can never match (OTHER-delimiter vs LETTER content), so — exactly
78      // as in Perl — it runs to the end and unreads, yielding an empty
79      // url + the rest as leftover text. Here we must NOT demote `{`/`}`:
80      // the surrounding `{\url www...pdf}` group's closing `}` must keep
81      // its END catcode so it still closes the group after the unread.
82      // Demoting it froze `}` as OTHER → the `{` group stayed open → the
83      // trailing `\endgroup` from `\url` hit the boxing group and errored
84      // (`Attempt to close non-boxing group`). Perl never demotes `{`/`}`
85      // (semiverbatim SPECIALS = ^_~&$#'). Witness 1503.07894 (bibitem
86      // `{\url www.maths...pdf}`).
87      if delim_is_explicit {
88        assign_catcode('{', Catcode::OTHER, Some(Scope::Local));
89        assign_catcode('}', Catcode::OTHER, Some(Scope::Local));
90      }
91      read_until_token(close)?
92    };
93    end_semiverbatim()?;
94    let toks : Vec<Token> = url.unlist().into_iter().filter(|t| t.get_catcode() != Catcode::SPACE)
95      .map(|t| t.as_other()).collect();
96
97    let mut url_wrapped = vec![T_CS!("\\UrlFont"), T_CS!("\\UrlLeft")];
98    url_wrapped.extend(toks.clone());
99    url_wrapped.push(T_CS!("\\UrlRight"));
100    let mut invocation_tokens = Invocation!(T_CS!("\\lx@url@url@nolink"),vec![
101        Tokens!(cmd.as_other()),
102        Tokens!(open),
103        Tokens!(close),
104        Tokens::new(toks),
105        Tokens::new(url_wrapped)]).unlist();
106    invocation_tokens.push(T_CS!("\\endgroup"));
107    Ok(Tokens::new(invocation_tokens))
108  });
109
110  // Define \Url, in case its used; won't be represented as nicely
111  DefMacro!("\\Url", {
112    unread_one(T_OTHER!("\\Url"));
113    Ok(Tokens!(T_CS!("\\lx@url@url")))
114  });
115
116  DefConstructor!("\\lx@url@url@nolink Undigested {}{} Semiverbatim {}",// Allow this to work in Math!
117    "?#isMath(<ltx:XMWrap class='ltx_nolink #class' href='#href'>#5</ltx:XMWrap>)(<ltx:ref href='#href' class='ltx_nolink #class'>#5</ltx:ref>)",
118    properties => sub[args] {
119      unref!(args => cmd, _open, _close, url, _formattedurl);
120      let ltx_cmd = s!("ltx_{}", LEADING_BACKSLASH_RE.replace(&cmd.to_string(),""));
121      Ok(stored_map!(
122        "href" => compose_url(&lookup_string("BASE_URL"), &url.to_string(), None),
123        // TODO: why was class realized in Perl as "sub" closure here?
124        "class"=> ltx_cmd
125      ))
126    },
127    sizer     => "#5",
128    enter_horizontal => true,
129    reversion => "#1#2#4#3");

135  // \urldef{newcmd}\cmd{arg}
136  // Kinda tricky, since we need to get the expansion of \cmd as the value of \newcmd
137  // Along with the annoying \endgroup that must balance the one always preceding \Url!
138  DefRegister!("\\Urlmuskip", MuGlue::default());
139
140  // Perl L91-99: \urldef{newcmd}{arg} — takes TWO brace args. The second
141  // `{}` lets TeX parse a balanced brace group without expansion, then
142  // unreads it back so digest_next_body can digest up through \endgroup.
143  // Prior Rust had only one arg `{}` and no unread, so `\urldef\foo{...}`
144  // digested whatever followed in the input stream instead of the braced arg.
145  DefPrimitive!("\\urldef{}{}", sub[(cmd, start)] {
146    let cmd = cmd.to_string();
147    unread(start);
148    let expansion : Vec<Digested> = digest_next_body(Some(T_CS!("\\endgroup")))?;
149    DefPrimitive!(&cmd, { Ok(expansion.clone()) });
150    Ok(vec![])
151  });
