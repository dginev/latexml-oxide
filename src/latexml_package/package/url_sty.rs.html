37  // This is an extended version of \Url that takes an extra token as 1st arg.
38  // That token is the cs that invoked it, so that it can be reflected in the generated XML,
39  // as well as used to generate the reversion.
40  // In any case, we read the verbatim arg, and build a Whatsit for @@Url
41  DefMacro!("\\lx@url@url Token", sub[(cmd)] {
42    let perc = vec!['%'];
43   begin_semiverbatim(Some(&perc));
44    let mut open = gullet::read_token()?.unwrap();
45    let close;
46    let url = if open.get_catcode() == Catcode::BEGIN {
47      open = T_OTHER!("{");
48      close = T_OTHER!("}");
49      gullet::read_balanced(ExpansionLevel::Off,false,false)?.unwrap_or_default()
50    } else {
51      open = open.as_other();
52      close = open;
53      gullet::read_until_token(close)?
54    };
55    end_semiverbatim()?;
56    let toks : Vec<Token> = url.unlist().into_iter().filter(|t| t.get_catcode() != Catcode::SPACE)
57      .map(|t| t.as_other()).collect();
58
59    let mut url_wrapped = vec![T_CS!("\\UrlFont"), T_CS!("\\UrlLeft")];
60    url_wrapped.extend(toks.clone());
61    url_wrapped.push(T_CS!("\\UrlRight"));
62    let mut invocation_tokens = Invocation!(T_CS!("\\lx@url@url@nolink"),vec![
63        Tokens!(cmd.as_other()),
64        Tokens!(open),
65        Tokens!(close),
66        Tokens::new(toks),
67        Tokens::new(url_wrapped)]).unlist();
68    invocation_tokens.push(T_CS!("\\endgroup"));
69    Ok(Tokens::new(invocation_tokens))
70  });

102  // Perl L91-99: \urldef{newcmd}{arg} — takes TWO brace args. The second
103  // `{}` lets TeX parse a balanced brace group without expansion, then
104  // unreads it back so digest_next_body can digest up through \endgroup.
105  // Prior Rust had only one arg `{}` and no unread, so `\urldef\foo{...}`
106  // digested whatever followed in the input stream instead of the braced arg.
107  DefPrimitive!("\\urldef{}{}", sub[(cmd, start)] {
108    let cmd = cmd.to_string();
109    gullet::unread(start);
110    let expansion : Vec<Digested> = stomach::digest_next_body(Some(T_CS!("\\endgroup")))?;
111    DefPrimitive!(&cmd, { Ok(expansion.clone()) });
112    Ok(vec![])
113  });
