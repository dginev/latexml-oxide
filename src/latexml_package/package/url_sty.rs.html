38  // This is an extended version of \Url that takes an extra token as 1st arg.
39  // That token is the cs that invoked it, so that it can be reflected in the generated XML,
40  // as well as used to generate the reversion.
41  // In any case, we read the verbatim arg, and build a Whatsit for @@Url
42  DefMacro!("\\lx@url@url Token", sub[(cmd)] {
43    let perc = vec!['%'];
44   begin_semiverbatim(Some(&perc));
45    let mut open = gullet::read_token()?.unwrap();
46    let close;
47    let url = if open.get_catcode() == Catcode::BEGIN {
48      open = T_OTHER!("{");
49      close = T_OTHER!("}");
50      gullet::read_balanced(ExpansionLevel::Off,false,false)?.unwrap_or_default()
51    } else {
52      open = open.as_other();
53      close = open;
54      // Once we've determined the delimiter is non-brace (e.g. `|`),
55      // demote `{` and `}` to OTHER so read_until_token doesn't engage
56      // its balanced-read branch on `{`. Real users write
57      // `\\path|{partial,| ... |partial}|` across multiple
58      // \\urldef calls, where the `{` and `}` are LITERAL chars in
59      // separate `|...|` paths, not balanced TeX groups. Driver:
60      // 1906.08946 — without this demotion our read_until('|') on
61      // line 25's `\\path|{...,|` reads PAST the closing `|`,
62      // through line 26 and into line 27 looking for the matching
63      // `}` — which it finds in `lncs}@springer.com|`, swallowing
64      // the intervening `\\urldef{\\mailsb}` and `\\urldef{\\mailsc}`
65      // declarations entirely.
66      latexml_core::state::assign_catcode('{', Catcode::OTHER, Some(Scope::Local));
67      latexml_core::state::assign_catcode('}', Catcode::OTHER, Some(Scope::Local));
68      gullet::read_until_token(close)?
69    };
70    end_semiverbatim()?;
71    let toks : Vec<Token> = url.unlist().into_iter().filter(|t| t.get_catcode() != Catcode::SPACE)
72      .map(|t| t.as_other()).collect();
73
74    let mut url_wrapped = vec![T_CS!("\\UrlFont"), T_CS!("\\UrlLeft")];
75    url_wrapped.extend(toks.clone());
76    url_wrapped.push(T_CS!("\\UrlRight"));
77    let mut invocation_tokens = Invocation!(T_CS!("\\lx@url@url@nolink"),vec![
78        Tokens!(cmd.as_other()),
79        Tokens!(open),
80        Tokens!(close),
81        Tokens::new(toks),
82        Tokens::new(url_wrapped)]).unlist();
83    invocation_tokens.push(T_CS!("\\endgroup"));
84    Ok(Tokens::new(invocation_tokens))
85  });
86
87  // Define \Url, in case its used; won't be represented as nicely
88  DefMacro!("\\Url", {
89    unread_one(T_OTHER!("\\Url"));
90    Ok(Tokens!(T_CS!("\\lx@url@url")))
91  });

103    },

112  // \urldef{newcmd}\cmd{arg}
113  // Kinda tricky, since we need to get the expansion of \cmd as the value of \newcmd
114  // Along with the annoying \endgroup that must balance the one always preceding \Url!
115  DefRegister!("\\Urlmuskip", MuGlue::default());
116
117  // Perl L91-99: \urldef{newcmd}{arg} — takes TWO brace args. The second
118  // `{}` lets TeX parse a balanced brace group without expansion, then
119  // unreads it back so digest_next_body can digest up through \endgroup.
120  // Prior Rust had only one arg `{}` and no unread, so `\urldef\foo{...}`
121  // digested whatever followed in the input stream instead of the braced arg.
122  DefPrimitive!("\\urldef{}{}", sub[(cmd, start)] {
123    let cmd = cmd.to_string();
124    gullet::unread(start);
125    let expansion : Vec<Digested> = stomach::digest_next_body(Some(T_CS!("\\endgroup")))?;
126    DefPrimitive!(&cmd, { Ok(expansion.clone()) });
127    Ok(vec![])
128  });
