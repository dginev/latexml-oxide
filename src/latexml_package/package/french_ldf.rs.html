189  // `\DecimalMathComma` / `\StandardMathComma` — french.ldf L815-877.
190  // French writes decimals with a comma (`3,14`); the real macros toggle
191  // the math comma's `\mathcode` between *punctuation* (class 6, small
192  // trailing space → list "1, 5") and *ordinary* (class 0, no space →
193  // decimal "1,5") via `\dec@math@comma`/`\std@math@comma` (L838-841).
194  // This is a purely *visual* spacing nuance: LaTeXML's number tokenizer
195  // already recognizes a digit-comma-digit run as a single decimal NUMBER
196  // independent of the mathcode, so the toggle has NO effect on LaTeXML's
197  // semantic output. Verified against Perl (which raw-loads french.ldf via
198  // `InputDefinitions('french', noltxml=>1)` and runs the real macros):

232  // French active-punctuation dispatch primitives for :;!? (frenchb.ldf's
233  // \extrasfrench inserts a thin space before these chars). The catcode
234  // flip + meaning attachment happens in babel's \select@language path
235  // (babel_sty.rs \lx@babel@activate@mainlang, babel_support_sty.rs
236  // \ltx@bbl@select@language). The primitives check current font language
237  // and fall back to bare punctuation in non-French groups (needed because
238  // `\foreignlanguage{english}{…!}` re-uses already-tokenized ACTIVE tokens).
239  //
240  //   ':'  → " :" (regular space, espace insécable visual)
241  //   ';!?' → "\u{2006}X" (thin space, SIX-PER-EM SPACE)
242  fn in_french() -> bool {
243    lookup_font()
244      .and_then(|f| f.get_language().map(|l| l.as_ref() == "fr" || l.as_ref() == "fr-CA"))
245      .unwrap_or(false)
246  }
247  DefPrimitive!("\\lx@french@punct@colon", {
248    enter_horizontal();
249    let s = if in_french() { " :" } else { ":" };
250    Tbox::new(pin_static(s), None, None, Tokens!(), stored_map!())
251  });
252  DefPrimitive!("\\lx@french@punct@semi", {
253    enter_horizontal();
254    let s = if in_french() { "\u{2006};" } else { ";" };
255    Tbox::new(pin_static(s), None, None, Tokens!(), stored_map!())
256  });
257  DefPrimitive!("\\lx@french@punct@exclam", {
258    enter_horizontal();
259    let s = if in_french() { "\u{2006}!" } else { "!" };
260    Tbox::new(pin_static(s), None, None, Tokens!(), stored_map!())
261  });
262  DefPrimitive!("\\lx@french@punct@question", {
263    enter_horizontal();
264    let s = if in_french() { "\u{2006}?" } else { "?" };
265    Tbox::new(pin_static(s), None, None, Tokens!(), stored_map!())
266  });
