1//! french.ldf / frenchb.ldf — French language support for babel
2//! Perl: french.ldf.ltxml + frenchb.ldf.ltxml (~35 lines each)
3//!
4//! Provides: French superscript commands, ordinals, guillemets,
5//! degree symbol, number formatting delegation.
6use crate::prelude::*;
7
8#[rustfmt::skip]
9LoadDefinitions!({
10  // Perl: InputDefinitions('french', type => 'ldf', noltxml => 1)
11  // We skip raw loading (it fails on babel 3.x \SetString commands)
12  // and provide the essential definitions directly.
13
14  // French superscript (Perl french.ldf.ltxml L24-26)
15  DefConstructor!("\\up{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
16  DefConstructor!("\\fup{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
17  DefConstructor!("\\FB@up@fake{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
18
19  // Ordinal suffixes (from raw frenchb.ldf)
20  DefMacro!("\\ier", "\\up{er}");
21  DefMacro!("\\iers", "\\up{ers}");
22  DefMacro!("\\iere", "\\up{re}");
23  DefMacro!("\\ieres", "\\up{res}");
24  DefMacro!("\\ieme", "\\up{e}");
25  DefMacro!("\\iemes", "\\up{es}");
26
27  // French enumeration (from raw frenchb.ldf)
28  DefMacro!("\\FrenchEnumerate{}", "#1\\up{o}");
29  DefMacro!("\\FrenchPopularEnumerate{}", "#1\\up{o})");
30  DefMacro!("\\primo", "1\\up{o}");
31  DefMacro!("\\secundo", "2\\up{o}");
32  DefMacro!("\\tertio", "3\\up{o}");
33  DefMacro!("\\quarto", "4\\up{o}");
34  DefMacro!("\\fprimo)", "1\\up{o})");
35  DefMacro!("\\fsecundo)", "2\\up{o})");
36  DefMacro!("\\ftertio)", "3\\up{o})");
37  DefMacro!("\\fquarto)", "4\\up{o})");
38
39  // \No, \no, \Nos, \nos — French abbreviations for "Numéro"
40  DefMacro!("\\No", "N\\up{o}");
41  DefMacro!("\\no", "n\\up{o}");
42  DefMacro!("\\Nos", "N\\up{os}");
43  DefMacro!("\\nos", "n\\up{os}");
44
45  // \bsc — small caps (from raw frenchb.ldf)
46  DefMacro!("\\bsc{}", "{\\scshape #1}");
47
48  // French quotes: \og and \fg (guillemets)
49  DefMacro!("\\og", "\u{00AB}\u{00A0}");
50  DefMacro!("\\fg", "\u{00A0}\u{00BB}");
51
52  // Symbols (Perl french.ldf.ltxml L32-35, AtBeginDocument)
53  DefMacro!("\\degre", "\\textdegree ");
54  DefMacro!("\\degres", "\\hbox to 0.3em{\\degre}");
55  DefMacro!("\\tild", "\\textasciitilde ");
56  DefMacro!("\\circonflexe", "\\textasciicircum ");
57  DefMacro!("\\at", "@");
58  DefMacro!("\\boi", "\\textbackslash ");
59
60  // \nombre — delegates to numprint if loaded (Perl french.ldf.ltxml L30)
61  DefMacro!("\\nombre{}", "\\numprint{#1}");
62
63  Let!("\\xspace", "\\relax");
64});
