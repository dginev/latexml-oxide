1//! french.ldf / frenchb.ldf — French language support for babel
2//! Perl: french.ldf.ltxml + frenchb.ldf.ltxml (~35 lines each)
3//!
4//! Provides: French superscript commands, ordinals, guillemets,
5//! degree symbol, number formatting delegation.
6use crate::prelude::*;
7
8#[rustfmt::skip]
9LoadDefinitions!({
10  // Perl: InputDefinitions('french', type => 'ldf', noltxml => 1)
11  // We skip raw loading (it fails on babel 3.x \SetString commands)
12  // and provide the essential definitions directly.
13
14  // frenchb's ordinal/abbreviation macros trail with \xspace — load
15  // xspace_sty up front so those expansions resolve, matching what
16  // the raw french.ldf implicitly assumes.
17  RequirePackage!("xspace");
18
19  // \captionsfrench — the French caption strings, equivalent to what
20  // babel's frenchb.ldf defines. Use \providecommand so the raw load
21  // (if it reaches this point) doesn't clobber our strings.
22  RawTeX!(r"\providecommand\captionsfrench{%
23    \def\prefacename{Pr\'eface}\def\refname{R\'ef\'erences}%
24    \def\abstractname{R\'esum\'e}\def\bibname{Bibliographie}%
25    \def\chaptername{Chapitre}\def\appendixname{Annexe}%
26    \def\contentsname{Table des mati\`eres}%
27    \def\listfigurename{Table des figures}%
28    \def\listtablename{Liste des tableaux}%
29    \def\indexname{Index}\def\figurename{Figure}%
30    \def\tablename{Table}\def\partname{partie}%
31    \def\pagename{page}\def\seename{voir}%
32    \def\alsoname{voir aussi}\def\proofname{D\'emonstration}}");
33  RawTeX!(r"\providecommand\datefrench{}");
34
35  // French superscript (Perl french.ldf.ltxml L24-26)
36  DefConstructor!("\\up{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
37  DefConstructor!("\\fup{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
38  DefConstructor!("\\FB@up@fake{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
39
40  // \FBthickkern — frenchb.ldf thick kern between ordinal and next token.
41  // Rendered as \thinspace in our port.
42  DefMacro!("\\FBthickkern", "\\thinspace");
43
44  // Ordinal suffixes (from raw frenchb.ldf) — trail with \xspace so a
45  // following punctuation/word gets proper spacing, matching babel's
46  // frenchb behavior.
47  DefMacro!("\\ier", "\\up{er}\\xspace");
48  DefMacro!("\\iers", "\\up{ers}\\xspace");
49  DefMacro!("\\iere", "\\up{re}\\xspace");
50  DefMacro!("\\ieres", "\\up{res}\\xspace");
51  DefMacro!("\\ieme", "\\up{e}\\xspace");
52  DefMacro!("\\iemes", "\\up{es}\\xspace");
53
54  // French enumeration (from raw frenchb.ldf) — use \FBthickkern between
55  // number and following content.
56  DefMacro!("\\FrenchEnumerate{}", "#1\\up{o}\\FBthickkern");
57  DefMacro!("\\FrenchPopularEnumerate{}", "#1\\up{o})\\FBthickkern");
58  DefMacro!("\\primo", "\\FrenchEnumerate1");
59  DefMacro!("\\secundo", "\\FrenchEnumerate2");
60  DefMacro!("\\tertio", "\\FrenchEnumerate3");
61  DefMacro!("\\quarto", "\\FrenchEnumerate4");
62  DefMacro!("\\fprimo)", "\\FrenchPopularEnumerate1");
63  DefMacro!("\\fsecundo)", "\\FrenchPopularEnumerate2");
64  DefMacro!("\\ftertio)", "\\FrenchPopularEnumerate3");
65  DefMacro!("\\fquarto)", "\\FrenchPopularEnumerate4");
66
67  // \No, \no, \Nos, \nos — French abbreviations for "Numéro".
68  // frenchb.ldf trails these with \xspace for consistency with a following
69  // number / punctuation.
70  DefMacro!("\\No", "N\\up{o}\\xspace");
71  DefMacro!("\\no", "n\\up{o}\\xspace");
72  DefMacro!("\\Nos", "N\\up{os}\\xspace");
73  DefMacro!("\\nos", "n\\up{os}\\xspace");
74
75  // \bsc — small caps (from raw frenchb.ldf)
76  DefMacro!("\\bsc{}", "{\\scshape #1}");
77
78  // French quotes: \og and \fg (guillemets).
79  // frenchb.ldf's \og ends with \nobreakspace; \fg starts with one.
80  DefMacro!("\\og", "\\guillemotleft\\nobreakspace");
81  DefMacro!("\\fg", "\\nobreakspace\\guillemotright\\xspace");
82
83  // Symbols (Perl french.ldf.ltxml L32-35, AtBeginDocument)
84  DefMacro!("\\degre", "\\textdegree");
85  DefMacro!("\\degres", "\\hbox to 0.3em{\\degre}");
86  Let!("\\tild", "\\textasciitilde");
87  Let!("\\circonflexe", "\\textasciicircum");
88  DefMacro!("\\at", "@");
89  DefMacro!("\\boi", "\\textbackslash");
90
91  // \nombre — delegates to numprint if loaded (Perl french.ldf.ltxml L30)
92  DefMacro!("\\nombre{}", "\\numprint{#1}");
93
94  // French active-punctuation dispatch primitives for :;!? (frenchb.ldf's
95  // \extrasfrench inserts a thin space before these chars). The catcode
96  // flip + meaning attachment happens in babel's \select@language path
97  // (babel_sty.rs \lx@babel@activate@mainlang, babel_support_sty.rs
98  // \ltx@bbl@select@language). The primitives check current font language
99  // and fall back to bare punctuation in non-French groups (needed because
100  // `\foreignlanguage{english}{…!}` re-uses already-tokenized ACTIVE tokens).
101  //
102  //   ':'  → " :" (regular space, espace insécable visual)
103  //   ';!?' → "\u{2006}X" (thin space, SIX-PER-EM SPACE)
104  fn in_french() -> bool {
105    lookup_font()
106      .and_then(|f| f.get_language().map(|l| l.as_ref() == "fr" || l.as_ref() == "fr-CA"))
107      .unwrap_or(false)
108  }
109  DefPrimitive!("\\lx@french@punct@colon", {
110    enter_horizontal();
111    let s = if in_french() { " :" } else { ":" };
112    Tbox::new(arena::pin_static(s), None, None, Tokens!(), stored_map!())
113  });
114  DefPrimitive!("\\lx@french@punct@semi", {
115    enter_horizontal();
116    let s = if in_french() { "\u{2006};" } else { ";" };
117    Tbox::new(arena::pin_static(s), None, None, Tokens!(), stored_map!())
118  });
119  DefPrimitive!("\\lx@french@punct@exclam", {
120    enter_horizontal();
121    let s = if in_french() { "\u{2006}!" } else { "!" };
122    Tbox::new(arena::pin_static(s), None, None, Tokens!(), stored_map!())
123  });
124  DefPrimitive!("\\lx@french@punct@question", {
125    enter_horizontal();
126    let s = if in_french() { "\u{2006}?" } else { "?" };
127    Tbox::new(arena::pin_static(s), None, None, Tokens!(), stored_map!())
128  });
129});
