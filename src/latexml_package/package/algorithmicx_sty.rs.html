1use crate::prelude::*;
2
3LoadDefinitions!({
4  // Perl: algorithmicx.sty.ltxml
5  // Was algorithmic.sty loaded? If so: BAIL immediately. (deeply incompatible)
6  if state::has_meaning(&T_CS!("\\algorithmic")) {
7    Warn!(
8      "unexpected",
9      "\\algorithmic",
10      "Another package has already defined \\algorithmic, will not load algorithmicx.sty"
11    );
12    return Ok(());
13  }
14
15  // Load core, make a few redefinitions
16  InputDefinitions!("algorithmicx", noltxml => true, extension => Some(Cow::Borrowed("sty")));
17
18  state::let_i(
19    &T_CS!("\\lx@orig@algorithmic"),
20    &T_CS!("\\algorithmic"),
21    None,
22  );
23  DefMacro!(
24    "\\algorithmic",
25    "\\lx@setup@algorithmicx\\lx@orig@algorithmic"
26  );
27
28  DefPrimitive!("\\lx@setup@algorithmicx", sub [_args] {
29    ResetCounter!("ALG@line");
30    // If we are not within an algorithm environment, step the counter for its id's
31    let in_algorithm = state::with_stacked_values_sym(pin!("current_environment"), |vals| {
32      vals.iter().any(|v| {
33        matches!(v, Stored::String(s) if arena::with(*s, |v| v == "algorithm"))
34      })
35    });
36    if !in_algorithm {
37      ref_step_id("algorithm")?;
38    }
39    state::let_i(&T_CS!("\\list"), &T_CS!("\\lx@algorithmicx@beginlist"), None);
40    state::let_i(&T_CS!("\\endlist"), &T_CS!("\\lx@algorithmicx@endlist"), None);
41    state::let_i(&T_CS!("\\item"), &T_CS!("\\lx@algorithmicx@item"), None);
42    state::let_i(&T_CS!("\\hfill"), &T_CS!("\\lx@algorithmicx@hfill"), None);
43  });
44
45  // IGNORE \list 1st arg (we'll handle counter stepping in \item)
46  DefMacro!(
47    "\\lx@algorithmicx@beginlist{}{}",
48    "\\lx@algorithmicx@beginlist@{#2}"
49  );
50  DefConstructor!("\\lx@algorithmicx@beginlist@{}", "<ltx:listing>");
51
52  DefConstructor!("\\lx@algorithmicx@endlist", "</ltx:listing>",
53    before_construct => sub[document] {
54      document.maybe_close_element("ltx:listingline")?;
55    }
56  );
57
58  // Empty lines still get an \item, but they're followed by \nointerlineskip!
59  // We do NOT want to generate a listingline in those cases.
60  DefMacro!(
61    "\\lx@algorithmicx@item[]",
62    "\\@ifnextchar\\nointerlineskip{}{\\lx@algorithmicx@@item}"
63  );
64
65  // This imitates \item; just opens the ltx:listingline, but somebody's got to close it.
66  DefConstructor!("\\lx@algorithmicx@@item",
67    "<ltx:listingline xml:id='#id' itemsep='#itemsep'>#tags",
68    properties => sub[_args] {
69      // Perl: my $step = Digest(T_BEGIN, T_CS('\ALG@step'), T_END);
70      let step = Digest!(Tokens::new(vec![
71        T_BEGIN!(), T_CS!("\\ALG@step"), T_END!()
72      ]))?;
73      // Perl: my $id = Digest(T_CS('\theALG@line@ID'));
74      let id = Digest!(Tokens::new(vec![T_CS!("\\theALG@line@ID")]))?;
75      // Perl: my $tags = Digest(T_BEGIN,
76      //   T_CS('\def'), T_CS('\fnum@ALG@line'), T_BEGIN, Tokens(Revert($step)), T_END,
77      //   Invocation(T_CS('\lx@make@tags'), T_OTHER('ALG@line')), T_END);
78      let step_revert = step.revert()?;
79      let invocation = Invocation!("\\lx@make@tags", vec![Some(Tokens::new(vec![T_OTHER!("ALG@line")]))]);
80      let mut tag_tokens = vec![T_BEGIN!(), T_CS!("\\def"), T_CS!("\\fnum@ALG@line"), T_BEGIN!()];
81      tag_tokens.extend(step_revert.unlist());
82      tag_tokens.push(T_END!());
83      tag_tokens.extend(invocation.unlist());
84      tag_tokens.push(T_END!());
85      let tags = Digest!(Tokens::new(tag_tokens))?;
86
87      Ok(stored_map!("id" => id, "tags" => tags))
88    },
89    before_construct => sub[document] {
90      document.maybe_close_element("ltx:listingline")?;
91    }
92  );
93
94  // Ideally, these appear within an algorithm environment, and we'd like to number lines within it.
95  // BUT algorithm package isn't required, so define it here!
96  NewCounter!("algorithm", "", idprefix => "alg");
97  NewCounter!("ALG@line", "algorithm", idprefix => "l");
98
99  // Hopefully this will only get used for right justifying a comment;
100  // the ltx:text should autoclose at end of line?
101  DefConstructor!(
102    "\\lx@algorithmicx@hfill",
103    "<ltx:text cssstyle='float:right'>"
104  );
105
106  // Protect against obsolete versions of algorithmicx source
107  DefMacro!("\\ALG@g{}", "");
108  DefMacro!("\\endALG@g", "");
109});
