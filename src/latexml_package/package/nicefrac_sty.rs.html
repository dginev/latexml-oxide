1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: nicefrac.sty.ltxml
6  RequirePackage!("ifthen");
7
8  // Handy for cases where macros want to use math, but track the current text font
9  DefPrimitive!("\\nf@mathcopytextfont", sub[_args] {
10    use latexml_core::binding::content::merge_font;
11    use latexml_core::common::font::Font;
12    if let Some(Stored::Font(ref f)) = lookup_value("savedfont") {
13      merge_font(Font {
14        family: f.family.clone(),
15        series: f.series.clone(),
16        shape: f.shape.clone(),
17        ..Font::default()
18      });
19    }
20    Ok(())
21  });
22
23  // Inline nicefrac: up-shifted numerator with /
24  DefConstructor!("\\ltx@nicefrac@inline InFractionStyle InFractionStyle",
25    "<ltx:XMApp>\
26       <ltx:XMTok stretchy='true' meaning='divide' role='MULOP'\
27         xoffset='-0.1em' width='-0.15em'>/</ltx:XMTok>\
28       <ltx:XMArg yoffset='0.3em' rpadding='-0.5em'>#1</ltx:XMArg>\
29       <ltx:XMArg>#2</ltx:XMArg>\
30     </ltx:XMApp>",
31    alias => "\\nicefrac",
32    after_construct => sub[document] {
33      // Set specialized font on the slash XMTok to prevent italic inheritance.
34      // Perl: _font='#slashfont' with font.specialize("/") — returns upright default font.
35      let node = document.get_node().clone();
36      let slash_toks = document.findnodes("descendant::ltx:XMTok[@meaning='divide']", Some(&node));
37      for mut tok in slash_toks {
38        let current_font = document.get_node_font(&tok);
39        let slash_font = current_font.specialize("/");
40        document.set_node_font(&mut tok, &slash_font)?;
41      }
42    }
43  );
44  // Bevelled version: MathML mfrac with bevelled='true'
45  DefConstructor!("\\ltx@nicefrac@bevelled InFractionStyle InFractionStyle",
46    "<ltx:XMApp>\
47       <ltx:XMTok meaning='divide' role='FRACOP' mathstyle='#mathstyle' class='ltx_bevelled'/>\
48       <ltx:XMArg>#1</ltx:XMArg>\
49       <ltx:XMArg>#2</ltx:XMArg>\
50     </ltx:XMApp>",
51    alias => "\\nicefrac",
52    after_digest => sub[whatsit] {
53      let style = lookup_font()
54        .and_then(|f| f.get_mathstyle().map(|s| s.to_string()))
55        .unwrap_or_default();
56      if !style.is_empty() {
57        whatsit.set_property("mathstyle", style);
58      }
59      Ok(Vec::new())
60    }
61  );
62
63  // Note: use math mode, but inherit text font when in text mode
64  DefMacro!("\\@UnitsNiceFrac Optional {}{}",
65    "\\ifmmode\\ltx@nicefrac@inline{#1{#2}}{#1{#3}}\\else\\if.#1.$\\ltx@nicefrac@inline{\\nf@mathcopytextfont{#2}}{\\nf@mathcopytextfont{#3}}$\\else$\\ltx@nicefrac@inline{#1{#2}}{#1{#3}}$\\fi\\fi");
66  DefMacro!("\\@UnitsNiceFrac@bevelled Optional {}{}",
67    "\\ifmmode\\ltx@nicefrac@bevelled{#1{#2}}{#1{#3}}\\else\\if.#1.$\\ltx@nicefrac@bevelled{\\nf@mathcopytextfont{#2}}{\\nf@mathcopytextfont{#3}}$\\else$\\ltx@nicefrac@bevelled{#1{#2}}{#1{#3}}$\\fi\\fi");
68
69  Let!("\\@UnitsUglyFrac", "\\@UnitsNiceFrac");
70
71  // Default: nice style
72  Let!("\\nicefrac", "\\@UnitsNiceFrac");
73
74  // Perl nicefrac.sty.ltxml L70-73: three options letting the user pick
75  // which \nicefrac variant is active. All three were unported, so
76  // `\usepackage[bevelled]{nicefrac}` fell through to the default (inline
77  // nice fraction) instead of switching to bevelled output. Pure additive
78  // — no test exercises these options so zero golden risk.
79  DeclareOption!("nice",     { Let!("\\nicefrac", "\\@UnitsNiceFrac"); });
80  DeclareOption!("ugly",     { Let!("\\nicefrac", "\\@UnitsUglyFrac"); });
81  DeclareOption!("bevelled", { Let!("\\nicefrac", "\\@UnitsNiceFrac@bevelled"); });
82  ProcessOptions!();
83});
