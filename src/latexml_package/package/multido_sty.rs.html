35  //   Dimensions are always printed in scaled points (sp)
36  //   Number are fixed point (and print that way!)
37  // concievably variables can be redefined in middle of loop?
38  DefMacro!("\\multido@@initvars@@{}", sub[(variables)] {
39    let reader_mouth = Mouth::new("", None)?;
40    let read_result : Result<Vec<Token>> =
41    gullet::reading_from_mouth(reader_mouth, || {
42
43      gullet::unread(variables);
44      let mut inits : Vec<Token> = Vec::new();
45      let mut steps = Vec::new();
46      gullet::skip_spaces()?;
47      while let Some(var) = gullet::read_token()? {
48        // TODO: this defeats the point of the performance optimization
49        // but it is so much *simpler* to allocate...
50        let csname = var.with_cs_name(ToString::to_string);
51        if let Some(cap) = DNIR_REX.captures(&csname) {
52          let vtype = cap.get(1).map_or(String::new(), |m| m.as_str().to_lowercase());
53          if gullet::read_keyword(&["="])?.is_none() {
54            Error!("expected", "=", "Missing = in multido variables");
55          }
56          let init = match vtype.as_str() {
57            "d" => Tokens!(Explode!(s!("{}sp", gullet::read_dimension()?.value_of()))),
58            "n" => gullet::read_float()?.revert()?,
59            "i" => gullet::read_number()?.revert()?,
60            "r" => gullet::read_float()?.revert()?,
61            _ => panic!("This voids the regex condition (d|n|i|r).")
62          };
63          inits.push(T_CS!("\\def"));
64          inits.push(var);
65          inits.push(T_BEGIN!());
66          inits.extend(init.unlist());
67          inits.push(T_END!());
68          if gullet::read_keyword(&["+"])?.is_none() {
69            Error!("expected", "+", "Missing + in multido variables");
70          }
71          let needs_negate = lookup_int("\\multido@count") < 0;
72          let step = match vtype.as_str() {
73            "d" => {
74              let mut stepv = gullet::read_dimension()?;
75              if needs_negate { stepv = stepv.negate(); }
76              stepv.revert()?
77            },
78            "n" => {
79              let mut stepv = gullet::read_float()?;
80              if needs_negate { stepv = stepv.negate(); }
81              stepv.revert()?
82            },
83            "i" => {
84              let mut stepv = gullet::read_number()?;
85              if needs_negate { stepv = stepv.negate(); }
86              stepv.revert()?
87            },
88            "r" => {
89              let mut stepv = gullet::read_float()?;
90              if needs_negate { stepv = stepv.negate(); }
91              stepv.revert()?
92            },
93            _ => panic!("This voids the regex condition (d|n|i|r).")
94          };
95          steps.push(T_CS!(s!("\\multido@step@{vtype}")));
96          steps.push(var);
97          steps.push(T_BEGIN!());
98          steps.extend(step.unlist());
99          steps.push(T_END!());
100          if gullet::read_keyword(&[","])?.is_none() {
101            break;
102          }
103        }  else {
104          Error!("unexpected", var, format!("Wrong format for multido variable {var:?}"));
105        }
106        gullet::skip_spaces()?;
107      }
108      DefMacro!(T_CS!("\\multido@stepvar"), None, Tokens::new(steps));
109      // Return the tokens to initialize the vars
110      Ok(inits)
111    });
112    read_result?
113  });
