11    "draft", "final", "referee",
12    "letter",
13  ].iter() {
14    DeclareOption!(*option, None);
15  }
16
17  // Anything else gets passed to article.

19    Digest!("\\PassOptionsToClass{\\CurrentOption}{article}")?;
20  });
21
22  // Font options — store choice in state
23  DeclareOption!("mathptmx", { AssignValue!("aipproc_font" => "mathptmx"); });
