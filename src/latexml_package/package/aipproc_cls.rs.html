66  // Perl aipproc.cls.ltxml L74-84: \author{name} RequiredKeyVals — the
67  // keyvals carry address / altaddress / email, and the Perl sub wraps
68  // each present key in `\lx@contact{key}{value}`. Prior Rust version
69  // used the bare stub "\\lx@author{#1}" that silently dropped all keys.
70  DefMacro!("\\author{} RequiredKeyVals", sub[(author, kv)] {
71    let mut out: Vec<Token> = Vec::new();
72    // \lx@author{author}
73    out.push(T_CS!("\\lx@author"));
74    out.push(T_BEGIN!());
75    out.extend(author.unlist_ref().iter().cloned());
76    out.push(T_END!());
77    for field in ["address", "altaddress", "email"] {
78      if let Some(val) = kv.get_value(field) {
79        // \lx@contact{field}{value}
80        out.push(T_CS!("\\lx@contact"));
81        out.push(T_BEGIN!());
82        out.extend(ExplodeText!(field));
83        out.push(T_END!());
84        out.push(T_BEGIN!());
85        out.extend(val.revert()?.unlist());
86        out.push(T_END!());
87      }
88    }
89    Ok(Tokens::new(out))
90  });
