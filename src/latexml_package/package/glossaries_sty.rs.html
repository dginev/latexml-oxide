1//! glossaries.sty — strict translation of Perl `glossaries.sty.ltxml` (126L).
2//!
3//! Perl raw-loads real TL glossaries.sty (~8700 lines) via
4//! `InputDefinitions(noltxml=>1)`, then layers surgical overrides:
5//!   - wrap `\@gls@link` output in `<ltx:glossaryref>`
6//!   - hook `\@newglossaryentryposthook` to emit `<ltx:glossarydefinition>`
7//!   - redefine `\printglossary` to emit `<ltx:glossary>`
8//!
9//! This Rust file mirrors the Perl 1:1.
10

19LoadDefinitions!({
20  // Pre-define \@gls@removedoption BEFORE glossaries.sty raw-loads.
21  // Real glossaries.sty L688 defines it via \newcommand and calls
22  // \PackageError on the obsolete acronym-style options (footnote /
23  // description / smallcaps / smaller / dua) — but this fires during
24  // package option processing, which happens DURING the raw-load. So
25  // a post-load override is too late.
26  //
27  // Pre-defining means \newcommand's check (\ifx\foo\relax-style)
28  // sees a non-relax definition and skips redefining. Use \def so
29  // our override is in place when option processing fires.
30  //
31  // The "error" is a PDF-rendering hint, not a structural failure.
32  // For XML output these options are moot. Downgrade to a warning.
33  // Witness 2110.10804 (\usepackage[smallcaps]{glossaries}).
34  // Perl emits 0 errors on the same input.
35  DefMacro!("\\@gls@removedoption{}",
36    "\\PackageWarning{glossaries}{obsolete package option #1 ignored}");
37
38  // Perl L18-19.
39  InputDefinitions!("glossaries", extension => Some(Cow::Borrowed("sty")), noltxml => true);
40  RequirePackage!("xspace");
41
42  // Perl L21: Silence pointless warnings.
43  def_macro_noop("\\glsnoidxstripaccents")?;
44
45  // Post-load: re-apply override in case raw-load's \newcommand
46  // overrode our pre-def. \newcommand errors if already defined, but
47  // \def silently overrides — but to be safe, reapply after load.
48  DefMacro!("\\@gls@removedoption{}",
49    "\\PackageWarning{glossaries}{obsolete package option #1 ignored}");
50
51  //======================================================================
52  // Perl L26-37: wrap `\@gls@link` in `<ltx:glossaryref>`.
53  Let!("\\lx@orig@glossaries@gls@link", "\\@gls@link");
54  DefMacro!("\\@gls@link[]{}{}",
55    "\\lx@glossaries@gls@link{\\csname glo@#2@type\\endcsname}{#2}{\\lx@orig@glossaries@gls@link[#1]{#2}{#3}}");
56  DefConstructor!("\\lx@glossaries@gls@link{}{}{}",
57    "<ltx:glossaryref inlist='#list' key='#2'>#3</ltx:glossaryref>",
58    enter_horizontal => true,
59    properties => sub[args] {
60      // Perl: $list = ToString($_[1]); $list = 'main' unless $_[1];
61      let list = args[0].as_ref().map(|t| t.to_string()).unwrap_or_default();
62      let list = if list.is_empty() { "main".to_string() } else { list };
63      Ok(stored_map!("list" => list))
64    });
65
66  //======================================================================
67  // Perl L40-42: skip over hyperref wrapping; we handle it.
68  DefMacro!("\\glsdohyperlink{}{}",   "#2");
69  DefMacro!("\\glsdonohyperlink{}{}", "#2");
70  RawTeX!("\\glsdisablehyper");
71
72  // Perl L45: This seems necessary, although it ought to be built in???
73  DefMacro!("\\glspostlinkhook", "\\xspace");
74
75  //======================================================================
76  // Perl L52-83: hook `\@newglossaryentryposthook` so each entry produces
77  // a structured `<ltx:glossarydefinition>` with one `<ltx:glossaryphrase>`
78  // per field. The keys mirror Perl exactly; the closing `}` is required
79  // because the hook body is interpreted as a single argument group.
80  DefMacro!("\\@newglossaryentryposthook",
81    "\\lx@glossaries@newentry{\\@glo@type}{\\glslabel}{\
82name=\\@glo@name,\
83description=\\@glo@desc,\
84symbol=\\@glo@symbol,\
85symbolplural=\\@glo@symbolplural,\
86text=\\@glo@text,\
87plural=\\@glo@plural,\
88first=\\@glo@first,\
89firstplural=\\@glo@firstplural,\
90sort=\\@glo@sort,\
91counter=\\@glo@counter,\
92see=\\@glo@see,\
93parent=\\@glo@parent,\
94prefix=\\@glo@prefix,\
95short=\\@glo@short,\
96shortplural=\\@glo@shortpl,\
97long=\\@glo@long,\
98longplural=\\@glo@longpl\
99}");
100
101  // Perl L85-97: DefConstructor that emits the structured definition.
102  // Iterate the keyvals in sorted-by-key order and insert one
103  // `<ltx:glossaryphrase>` per non-empty value (matches Perl's
104  // `if ToString($value)` guard).
105  DefConstructor!("\\lx@glossaries@newentry{}{} RequiredKeyVals",
106    sub[document, args, _props] {
107      let list = args[0].as_ref().map(|d| d.to_string()).unwrap_or_else(|| "main".to_string());
108      let key  = args[1].as_ref().map(|d| d.to_string()).unwrap_or_default();
109      document.open_element("ltx:glossarydefinition",
110        Some(string_map!("key" => key, "inlist" => list)), None)?;
111      if let Some(kv_digested) = args[2].as_ref()
112        && let DigestedData::KeyVals(ref kvs) = *kv_digested.data() {
113          // Sort by role (Perl: `sort keys %$hash`).
114          let mut pairs: Vec<(String, ArgWrap)> = kvs.get_pairs()
115            .map(|(k, v)| (k.clone(), v.clone())).collect();
116          pairs.sort_by(|a, b| a.0.cmp(&b.0));
117          for (role, val) in pairs {
118            let val_str = val.to_string();
119            if val_str.is_empty() { continue; }
120            // Insert <ltx:glossaryphrase key=key role=role>val</ltx:glossaryphrase>
121            document.open_element("ltx:glossaryphrase",
122              Some(string_map!("key" => key.clone(), "role" => role)), None)?;
123            document.absorb_string(&val_str, &NO_PROPERTIES)?;
124            document.close_element("ltx:glossaryphrase")?;
125          }
126        }
127      document.close_element("ltx:glossarydefinition")?;
128    }
129  );
130
131  //======================================================================
132  // Perl L101-104: redefine `\printglossary` to dispatch to our constructor.
133  DefMacro!("\\printglossary",
134    "\\global\\let\\warn@noprintglossary\\relax\
135\\@ifnextchar[{\\lx@printglossary}{\\lx@printglossary[type=main]}");
136  // Perl L105.
137  Let!("\\printnoidxglossary", "\\printglossary");
138
139  // Perl L107-117: emit `<ltx:glossary>` placeholder with computed id,
140  // list, and title. The XSLT pipeline later expands it into the actual
141  // rendered glossary entries.
142  DefConstructor!("\\lx@printglossary OptionalKeyVals",
143    "<ltx:glossary xml:id='#id' lists='#list'>\
144<ltx:title font='#titlefont' _force_font='true'>#title</ltx:title>\
145</ltx:glossary>",
146    properties => sub[args] {
147      // Perl L113-117 — compute type (default 'main'), title (digest
148      // \@glotype@<type>@title), and id (docid + ".glo." + cleaned type).
149      let typ = args[0].as_ref().and_then(|d| {
150        if let DigestedData::KeyVals(ref kvs) = *d.data() {
151          kvs.get_value("type").map(|v| v.to_string())
152        } else { None }
153      }).unwrap_or_else(|| "main".to_string());
154      let title_cs = s!("\\@glotype@{typ}@title");
155      let title = digest(T_CS!(&*title_cs))
156        .map(|d| d.to_string()).unwrap_or_default();
157      let docid = lookup_value("thedocument@ID")
158        .and_then(|v| match v { Stored::String(s) => Some(to_string(s)), _ => None })
159        .unwrap_or_default();
160      let cleaned = typ.chars().filter(|c| c.is_alphanumeric()).collect::<String>();
161      let id = if docid.is_empty() {
162        format!("glo.{cleaned}")
163      } else {
164        format!("{docid}.glo.{cleaned}")
165      };
166      Ok(stored_map!("list" => typ, "id" => id, "title" => title))
167    },
168    after_digest => sub[whatsit] {
169      // Perl L114: noteBackmatterElement(<whatsit>, 'ltx:glossary');
170      note_backmatter_element(whatsit, "ltx:glossary");
171    },
172    before_construct => sub[doc, whatsit] {
173      // Perl L117: adjustBackmatterElement($_[0], $_[1]);
174      adjust_backmatter_element(doc, whatsit)?;
175    }
176  );
177});
