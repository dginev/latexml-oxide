1use crate::prelude::*;
2
3LoadDefinitions!({
4  //**********************************************************************
5  // Define \name and \begin{name} to start an ignored section
6  // until \endname or \end{name}, respectively
7  let define_excluded: PrimitiveClosure = Rc::new(|mut args: Vec<ArgWrap>| {
8    let name = args.remove(0).owned_tokens().unwrap();
9    let begin_mark = s!("\\begin{{{name}}}");
10    let end_mark = s!("\\end{{{name}}}");
11    DefConstructor!(T_CS!(begin_mark), None, None,
12    after_digest => {
13      let mut nlines = 0;
14      read_raw_line();    // IGNORE 1st line (after the \begin{$name} !!!
15      // Perl comment.sty.ltxml L30 matches `/^\s*\Q$endmark\E\s*$/` —
16      // the end line may carry leading/trailing whitespace. Strict
17      // equality missed indented `  \end{comment}  `, stranding the
18      // excluded block consumption.
19      while let Some(line) = read_raw_line() {
20        if line.trim() == end_mark {
21          break;
22        }
23        nlines += 1;
24      }
25      note_progress(&s!("[Skipped {name} ({nlines} lines)]"));
26      Ok(Vec::new())
27    });
28    Ok(Vec::new())
29  });
30
31  // I don't understand Rust closures enough to figure out how to clone one, so instantiating it
32  // twice instead, via a macro
33  let define_included: PrimitiveBody = PrimitiveBody::Closure(Rc::new(|mut args: Vec<ArgWrap>| {
34    args.reverse(); // we'll be using .pop() from the front
35    let name = args
36      .pop()
37      .unwrap()
38      .owned_tokens()
39      .expect("expecting a Tokens argument")
40      .to_string();
41    let mut before_tokens = match args.pop() {
42      Some(arg) => arg.unlist(),
43      None => Vec::new(),
44    };
45    before_tokens.push(T_CS!("\\ignorespaces"));
46    let mut after_tokens = match args.pop() {
47      Some(arg) => arg.unlist(),
48      None => Vec::new(),
49    };
50    after_tokens.push(T_CS!("\\ignorespaces"));
51    // Note that we define the `magic' environment control sequences,
52    // but DO NOT do any of the normal environ things, like \begingroup \endgroup!
53    DefMacro!(T_CS!(s!("\\begin{{{name}}}")), None, {
54      read_raw_line(); // IGNORE 1st line (after the \begin{$name} !!!
55      before_tokens.clone()
56    });
57    DefMacro!(
58      T_CS!(s!("\\end{{{name}}}")),
59      None,
60      Tokens::new(after_tokens)
61    );
62    Ok(Vec::new())
63  }));
64
65  define_excluded(vec![ArgWrap::Tokens(Tokenize!("comment"))])?;
66
67  DefPrimitive!("\\includecomment{}", Some(define_included.clone()));
68  DefPrimitive!(
69    "\\excludecomment{}",
70    Some(PrimitiveBody::Closure(define_excluded))
71  );
72  DefPrimitive!("\\specialcomment{}{}{}", Some(define_included));
73  DefPrimitive!("\\processcomment{}{}{}{}", None);
74  DefMacro!("\\csarg{}{}", r"\expandafter#1\csname#2\endcsname");
75});
