6/// DEP-NEW (2026-05-19): data-drive helper for the 74 IOP-physics
7/// journal abbreviation macros that all expand to
8/// `\textit{<name>}` (some with trailing italic-correction `\/`).
9/// Replacing the per-call `DefMacro!` macro arms with this runtime
10/// helper collapses the repeated macro expansion to one inlined fn.
11/// Per [[wisdom_data_drive_min_call_sites]]: 74 ≫ 5 threshold.
12fn def_iop_journal(cs: &str, name: &str) -> Result<()> {
13  let (cs_tok, params) = parse_prototype(cs, true)?;
14  let body_toks = mouth::tokenize_internal(&format!("\\textit{{{name}}}"));
15  def_macro(cs_tok, params, ExpansionBody::Tokens(body_toks), None)?;
16  Ok(())
17}
