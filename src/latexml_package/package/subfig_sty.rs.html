11  // Perl L26-27: \refstepcounter@noreset passes noreset=1 to RefStepCounter,
12  // which steps the counter but skips the usual subcounter reset. Rust
13  // previously aliased it to plain \refstepcounter (which DOES reset
14  // subcounters), so a \subfloat inside an uncaptioned float temporarily
15  // stepping the parent counter would zero out the subcounter the next
16  // \subfloat was trying to increment.
17  DefPrimitive!("\\refstepcounter@noreset{}", sub[(cs)] {
18    let cs_expanded = Expand!(cs).to_string();
19    RefStepCounter!(&cs_expanded, true)?;
20  });
