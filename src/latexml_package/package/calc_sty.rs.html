5// calc.sty expression parser — ported from Perl calc.sty.ltxml
6// Grammar:
7//   <expression> -> <term> ((+|-) <term>)*
8//   <term>       -> <value> ((*|/) <value>)*  (where multiply/divide factor is float-aware)
9//   <value>      -> <literal> | \real{...} | \ratio{...}{...}
10//                 | \minof{...}{...} | \maxof{...}{...}
11//                 | \widthof{...} | ( <expression> )
12
13/// Internal representation for calc values: either an integer-based RegisterValue
14/// or a floating-point factor (from \real or \ratio).
15enum CalcValue {
16  Reg(RegisterValue),
17  Flt(f64),
18}
19
20impl CalcValue {
21  fn into_regval(self, expr_type: &str) -> RegisterValue {
22    match self {
23      CalcValue::Reg(rv) => rv,
24      CalcValue::Flt(f) => {
25        if expr_type == "Number" {
26          RegisterValue::Number(Number::new(f as i64))
27        } else {
28          RegisterValue::Glue(Glue::new(f as i64))
29        }
30      },
31    }
32  }
33}

35fn read_expression(expr_type: &str, tokens: Tokens) -> Result<RegisterValue> {
36  let reader_mouth = Mouth::new("", None)?;
37  gullet::reading_from_mouth(reader_mouth, move || {
38    gullet::unread(tokens);
39    let mut result = read_term(expr_type)?;
40    while let Some(op) = gullet::read_keyword(&["+", "-"])? {
41      let term2 = read_term(expr_type)?;
42      result = if op == "+" {
43        result.add(term2)
44      } else {
45        result.subtract(term2)
46      };
47    }
48    Ok(result)
49  })
50}

52fn read_term(expr_type: &str) -> Result<RegisterValue> {
53  gullet::skip_spaces()?;
54  let mut factor = read_value(expr_type)?.into_regval(expr_type);
55  while let Some(op) = gullet::read_keyword(&["*", "/"])? {
56    let factor2 = read_value("Number")?;
57    factor = match factor2 {
58      CalcValue::Flt(f) => apply_float_op(&factor, &op, f),
59      CalcValue::Reg(rv) => {
60        if op == "*" {
61          factor.multiply(rv)
62        } else {
63          factor.divide(rv)
64        }
65      },
66    };
67  }
68  Ok(factor)
69}

71/// Apply a float multiply/divide to a RegisterValue, preserving Glue components
72fn apply_float_op(val: &RegisterValue, op: &str, f: f64) -> RegisterValue {
73  match val {
74    RegisterValue::Glue(g) => {
75      if op == "*" {
76        RegisterValue::Glue(Glue {
77          skip:  (g.skip as f64 * f) as i64,
78          plus:  g.plus.map(|p| (p as f64 * f) as i64),
79          pfill: g.pfill,
80          minus: g.minus.map(|m| (m as f64 * f) as i64),
81          mfill: g.mfill,
82        })
83      } else {
84        let div = if f == 0.0 { 1.0 } else { f };
85        RegisterValue::Glue(Glue {
86          skip:  (g.skip as f64 / div) as i64,
87          plus:  g.plus.map(|p| (p as f64 / div) as i64),
88          pfill: g.pfill,
89          minus: g.minus.map(|m| (m as f64 / div) as i64),
90          mfill: g.mfill,
91        })
92      }
93    },
94    RegisterValue::Number(n) => {
95      let v = n.value_of() as f64;
96      if op == "*" {
97        RegisterValue::Number(Number::new((v * f) as i64))
98      } else {
99        let div = if f == 0.0 { 1.0 } else { f };
100        RegisterValue::Number(Number::new((v / div) as i64))
101      }
102    },
103    RegisterValue::Dimension(d) => {
104      let v = d.value_of() as f64;
105      if op == "*" {
106        RegisterValue::Dimension(Dimension::new((v * f) as i64))
107      } else {
108        let div = if f == 0.0 { 1.0 } else { f };
109        RegisterValue::Dimension(Dimension::new((v / div) as i64))
110      }
111    },
112    _ => val.clone(),
113  }
114}

233fn read_value(expr_type: &str) -> Result<CalcValue> {
234  gullet::skip_spaces()?;
235  let peek = gullet::read_x_token(None, false, None)?;
236  let peek = match peek {
237    Some(t) => t,
238    None => {
239      if expr_type == "Number" {
240        return Ok(CalcValue::Reg(RegisterValue::Number(Number::new(0))));
241      } else {
242        return Ok(CalcValue::Reg(RegisterValue::Glue(Glue::new(0))));
243      }
244    },
245  };
246  // \widthof{...} — Perl calc.sty.ltxml L139-143
247  if peek == T_CS!("\\widthof") {
248    let arg = gullet::read_arg(ExpansionLevel::Off)?;
249    let box_result = digest(arg)?;
250    if expr_type == "Number" {
251      Error!(
252        "unexpected",
253        "\\widthof",
254        format!("\\widthof not expected here (reading {expr_type})")
255      );
256    }
257    let width = box_result
258      .get_width(None)?
259      .unwrap_or_else(|| RegisterValue::Dimension(Dimension::new(0)));
260    return Ok(CalcValue::Reg(width));
261  }
262  // \heightof{...} — Perl calc.sty.ltxml L144-148
263  if peek == T_CS!("\\heightof") {
264    let arg = gullet::read_arg(ExpansionLevel::Off)?;
265    let box_result = digest(arg)?;
266    if expr_type == "Number" {
267      Error!(
268        "unexpected",
269        "\\heightof",
270        format!("\\heightof not expected here (reading {expr_type})")
271      );
272    }
273    let height = box_result
274      .get_height()
275      .unwrap_or_else(|| RegisterValue::Dimension(Dimension::new(0)));
276    return Ok(CalcValue::Reg(height));
277  }
278  // \depthof{...} — Perl calc.sty.ltxml L149-153
279  if peek == T_CS!("\\depthof") {
280    let arg = gullet::read_arg(ExpansionLevel::Off)?;
281    let box_result = digest(arg)?;
282    if expr_type == "Number" {
283      Error!(
284        "unexpected",
285        "\\depthof",
286        format!("\\depthof not expected here (reading {expr_type})")
287      );
288    }
289    let depth = box_result
290      .get_depth()
291      .unwrap_or_else(|| RegisterValue::Dimension(Dimension::new(0)));
292    return Ok(CalcValue::Reg(depth));
293  }
294  // \totalheightof{...} — Perl calc.sty.ltxml L154-158
295  if peek == T_CS!("\\totalheightof") {
296    let arg = gullet::read_arg(ExpansionLevel::Off)?;
297    let box_result = digest(arg)?;
298    if expr_type == "Number" {
299      Error!(
300        "unexpected",
301        "\\totalheightof",
302        format!("\\totalheightof not expected here (reading {expr_type})")
303      );
304    }
305    let height = box_result
306      .get_height()
307      .unwrap_or_else(|| RegisterValue::Dimension(Dimension::new(0)));
308    let depth = box_result
309      .get_depth()
310      .unwrap_or_else(|| RegisterValue::Dimension(Dimension::new(0)));
311    return Ok(CalcValue::Reg(height.add(depth)));
312  }
313  // \real{<decimal>} — returns a Float factor for multiplication
314  if peek == T_CS!("\\real") {
315    let arg = gullet::read_arg(ExpansionLevel::Off)?;
316    let reader_mouth = Mouth::new("", None)?;
317    let float = gullet::reading_from_mouth(reader_mouth, move || {
318      gullet::unread(arg);
319      let f = gullet::read_float()?;
320      // Round as fixpoint, matching Perl calc.sty.ltxml line 164
321      Ok(kround(f.value_f64() * UNITY as f64) as f64 / UNITY as f64)
322    })?;
323    return Ok(CalcValue::Flt(float));
324  }
325  // \ratio{<dimen expr>}{<dimen expr>} — returns a Float factor
326  if peek == T_CS!("\\ratio") {
327    let arg_x = gullet::read_arg(ExpansionLevel::Off)?;
328    let arg_y = gullet::read_arg(ExpansionLevel::Off)?;
329    let x = read_expression("Glue", arg_x)?;
330    let y = read_expression("Glue", arg_y)?;
331    let y_val = y.value_of() as f64;
332    let y_pt = if y_val == 0.0 { 1.0 } else { y_val };
333    return Ok(CalcValue::Flt(x.value_of() as f64 / y_pt));
334  }
335  // \minof{<expr>}{<expr>}
336  if peek == T_CS!("\\minof") {
337    let arg_x = gullet::read_arg(ExpansionLevel::Off)?;
338    let arg_y = gullet::read_arg(ExpansionLevel::Off)?;
339    let x = read_expression(expr_type, arg_x)?;
340    let y = read_expression(expr_type, arg_y)?;
341    return Ok(CalcValue::Reg(x.smaller(y)));
342  }
343  // \maxof{<expr>}{<expr>}
344  if peek == T_CS!("\\maxof") {
345    let arg_x = gullet::read_arg(ExpansionLevel::Off)?;
346    let arg_y = gullet::read_arg(ExpansionLevel::Off)?;
347    let x = read_expression(expr_type, arg_x)?;
348    let y = read_expression(expr_type, arg_y)?;
349    return Ok(CalcValue::Reg(x.larger(y)));
350  }
351  // Parenthesized subexpression: ( <expression> )
352  if peek == T_OTHER!("(") {
353    let inner = gullet::read_until(&Tokens!(T_OTHER!(")")));
354    return Ok(CalcValue::Reg(read_expression(expr_type, inner?)?));
355  }
356  // Else: literal value — put back token and read normally
357  gullet::unread_one(peek);
358  if expr_type == "Number" {
359    Ok(CalcValue::Reg(
360      RegisterValue::Number(gullet::read_number()?),
361    ))
362  } else {
363    Ok(CalcValue::Reg(RegisterValue::Glue(gullet::read_glue()?)))
364  }
365}

412      let (rtoken, params) = *dbox;
413      if let Some(defn) = rtoken.to_register() {

416        defn.set_value(old_value.add(delta), None, params);
417      }
418    }
