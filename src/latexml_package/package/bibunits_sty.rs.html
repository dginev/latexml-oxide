12  DeclareOption!("labelstoglobalaux", {});
13  DeclareOption!("sectionbib", {

16  DeclareOption!("subsectionbib", {
17    AssignMapping!("BACKMATTER_ELEMENT", "ltx:bibliography" => "ltx:subsection");

28  DefPrimitive!("\\lx@bibunits@setglobal", None,
29  after_digest => {
30    if !lookup_bool("CITE_UNIT_GLOBAL") {
31      let unit = Digest!("\\bu@unitname")?.to_string();

39      let unit = Digest!("\\bu@unitname")?.to_string();

50  // Perl: \bibliographyunit[\section] — intercepts sectional command to start bib units
51  DefPrimitive!("\\bibliographyunit [DefToken]", sub[args] {
52    let unit_arg = args.into_iter().next().unwrap();
53    if unit_arg.is_some() {
54      let unit_tok = unit_arg.expected_token();
55      let unit_cs = unit_tok.to_string();
56      let unit_cs_tok = T_CS!(unit_cs.clone());
57      Let!(T_CS!("\\old@bibunit"), unit_cs_tok.clone());
58      Let!(unit_cs_tok, T_CS!("\\@bibunit"));
59      // Map sectional unit to backmatter element
60      let bme = match unit_cs.as_str() {
61        "\\part" => Some("ltx:chapter"),
62        "\\chapter" => Some("ltx:section"),
63        "\\section" => Some("ltx:subsection"),
64        "\\subsection" => Some("ltx:subsubsection"),
65        "\\subsubsection" => Some("ltx:paragraph"),
66        "\\paragraph" => Some("ltx:subparagraph"),
67        _ => None,
68      };
69      if let Some(bme_val) = bme {
70        AssignMapping!("BACKMATTER_ELEMENT", "ltx:bibliography" => bme_val);
71      }
72    } else {
73      assign_value("CITE_UNIT", Stored::None, None);
74    }
75  });
76  DefMacro!("\\@bibunit", "\\lx@startbibunit\\old@bibunit");
77
78  // Perl: startBibunit() — increment counter and set CITE_UNIT
79  DefPrimitive!("\\lx@startbibunit", None,
80  after_digest => {
81    Digest!("\\global\\advance\\@bibunitauxcnt1")?;
82    let unit = Digest!("\\bu@unitname")?.to_string();
83    let cite_unit = if lookup_bool("CITE_UNIT_GLOBAL") {
84      s!("bibliography {unit}")
85    } else {
86      unit
87    };
88    assign_value("CITE_UNIT", pin(&cite_unit), None);
89  });

91  // Perl bibunits.sty.ltxml L72-76: {bibunit}[] env has afterDigestBegin
92  // that reads the optional arg (falling back to \bu@bibstyle), applies it
93  // via setBibstyle, and then startBibunit()s the counter/CITE_UNIT dance.
94  // Prior Rust stub was "#body" with no hooks, so `\begin{bibunit}[style]`
95  // didn't step counters nor switch the bib style — nested bibliographies
96  // all shared the global CITE_UNIT and original BIBSTYLE.
97  DefEnvironment!("{bibunit}[]", "#body",
98  after_digest_begin => sub[whatsit] {
99    let arg_style = whatsit.get_arg(1).map(|a| a.to_string()).unwrap_or_default();
100    let arg_style = arg_style.trim().to_string();
101    let style = if !arg_style.is_empty() {
102      arg_style
103    } else {
104      do_expand(T_CS!("\\bu@bibstyle")).map(|t| t.to_string()).unwrap_or_default()
105    };
106    if !style.is_empty() {
107      engine::latex_constructs::set_bibstyle(&style);
108    }
109    // startBibunit() equivalent: step counter + set CITE_UNIT.
110    Digest!("\\global\\advance\\@bibunitauxcnt1")?;
111    let unit = Digest!("\\bu@unitname")?.to_string();
112    let cite_unit = if lookup_bool("CITE_UNIT_GLOBAL") {
113      s!("bibliography {unit}")
114    } else {
115      unit
116    };
117    assign_value("CITE_UNIT", pin(&cite_unit), None);
118  });
119
120  DefMacro!(
121    "\\putbib[]",
122    "\\lx@bibliography[\\bu@unitname]{\\if.#1.\\bu@bibdata\\else#1\\fi}"
123  );
124
125  // Perl: make \bibliography reset the backmatter element
126  Let!("\\bu@orig@bibliography", "\\bibliography");
127  // Store original backmatter element for bibliography
128  if let Some(orig) = lookup_mapping("BACKMATTER_ELEMENT", "ltx:bibliography") {
129    assign_value("ORIG_BIBUNIT", orig, None);
130  }
131  DefMacro!("\\bibliography", "\\lx@reset@bibunit\\bu@orig@bibliography");
132  DefPrimitive!("\\lx@reset@bibunit", None,
133  after_digest => {
134    if let Some(orig) = lookup_value("ORIG_BIBUNIT") {
135      assign_mapping("BACKMATTER_ELEMENT", "ltx:bibliography", Some(orig));
136    }
137  });
