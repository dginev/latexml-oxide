1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: ijcai.sty.ltxml (PR #2767)
6  // borrow a few of the cite-related definitions from natbib
7  RequirePackage!("natbib");
8
9  // \author{} : used once, separates multiple authors by \and (\And for last), followed by \\ !!!
10  // Bizarrely, \affiliations and \emails should be WITHIN \author!
11  // \affiliations{} gives \\ separated set of affiliations, corresponding to author in order
12  // \emails{} gives a ", " separated set of emails, corresponding to author in order
13  // If the association isn't "obvious" (one-to-one, or only a single affil?),
14  // the author is expected to use $^{1,2}$ superscript markers!!
15  Let!("\\AND",      "\\and");
16  Let!("\\And",      "\\and");
17  Let!("\\leftcite", "\\cite");
18  DefMacro!("\\pubnote{}", "\\lx@add@pubnote[role=note]{#1}");
19
20  // These are used as separators WITHIN \author, so...
21  def_macro_noop("\\affiliations")?;
22  def_macro_noop("\\emails")?;
23
24  DefMacro!("\\author{}", "\\lx@ijcai@authorsplit#1\\affiliations\\done");
25  DefMacro!("\\lx@ijcai@authorsplit Until:\\affiliations Until:\\done",
26    "\\lx@add@authors{#1}\\ifx.#2.\\else\\lx@ijcai@affilsplit#2\\emails\\affiliations\\done\\fi");
27  DefMacro!("\\lx@ijcai@affilsplit  Until:\\emails Until:\\affiliations Until:\\done",
28    "\\ifx.#1.\\else\\expandafter\\lx@ijcai@affiliations\\expandafter{\\lx@strip@braces{#1}}\\fi\\ifx.#2.\\else\\expandafter\\lx@ijcai@emails\\expandafter{\\lx@strip@braces{#2}}\\fi");
29  DefMacro!("\\lx@ijcai@affiliations{}", "\\lx@add@affiliations[labelseq=author]{#1}");
30
31  // emails is expected WITHIN \author.
32  // Multiple affiliations separated by ,
33  // the n-th email is attached to n-th author
34  DefMacro!("\\lx@ijcai@emails{}",
35    "\\lx@clear@frontmatter{ltx:contact}[role=email]\\lx@splitting{\\lx@ijcai@email}{,}{#1}");
36  DefMacro!("\\lx@ijcai@email{}", "\\lx@add@email[labelseq=author]{#1}");
37});
