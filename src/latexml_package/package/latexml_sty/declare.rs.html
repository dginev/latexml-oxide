1//! The `\lxDeclare` / `\lxDefMath` declaration helpers — one Rust function
2//! per Perl sub in `latexml.sty.ltxml`, so the binding stays diffable against
3//! the reference source sub-by-sub:
4//!
5//! | Rust                         | Perl (latexml.sty.ltxml)                  |
6//! |------------------------------|-------------------------------------------|
7//! | [`next_declaration_id`]      | `next_declaration_id` (L544-550)          |
8//! | [`split_declare_tag`]        | `splitDeclareTag` (L437-449)              |
9//! | [`normalize_declare_keys`]   | `normalizeDeclareKeys` (L417-434)         |
10//! | [`get_declaration_scope`]    | `getDeclarationScope` (L552-559)          |
11//! | [`create_declaration_rewrite`] | `createDeclarationRewrite` (L561-580)   |
12//! | [`emit_declare_element`]     | the shared `<ltx:declare>` constructor    |
13//! |                              | body (`\lxDeclare` L474-485 /             |
14//! |                              | `\@lxDefMathDeclare` L387-396)            |
15//! | [`record_declaration_lines`] | Rust-only: the LATEXML_DECLARATIONS       |
16//! |                              | fast-path registry consumed by            |
17//! |                              | `apply_lx_declarations`                   |
18//!
19//! The pattern COMPILER lives with the rewrite engine
20//! (`latexml_core::rewrite::declare`), matching Perl's `domToXPath` in
21//! `Core/Rewrite.pm`.

27/// Perl `next_declaration_id()`: `StepCounter('@XMDECL')` then expand
28/// `\the@XMDECL@ID`. The `@XMDECL` counter is subordinate to `section`, so
29/// ids reset per-section: `S1.XMD1`, `S1.XMD2`, …, `S2.XMD1`, ….
30pub(super) fn next_declaration_id() -> Result<String> {
31  step_counter("@XMDECL", false)?;
32  Ok(
33    do_expand(T_CS!("\\the@XMDECL@ID"))
34      .ok()
35      .map(|t| t.to_string().trim().to_string())
36      .unwrap_or_default(),
37  )
38}

40/// Perl `splitDeclareTag`: the boxes before the first `:` box become the
41/// TERM (typically math), the rest the description; no `:` → neither.
42pub(super) fn split_declare_tag(
43  stuff: &Digested,
44) -> (Option<Vec<Digested>>, Option<Vec<Digested>>) {
45  let boxes = stuff.unlist();
46  match boxes
47    .iter()
48    .position(|b| b.get_string().map(|s| s.trim() == ":").unwrap_or(false))
49  {
50    Some(pos) => (Some(boxes[..pos].to_vec()), Some(boxes[pos + 1..].to_vec())),
51    None => (None, None),
52  }
53}

55/// Perl `normalizeDeclareKeys`: synthesize term/short/description for the
56/// `<ltx:declare>` element out of the DIGESTED tag/description values — a
57/// description like `$x$: a real variable` contains a real math box that
58/// must survive to the term tag (the term Math is then itself subject to
59/// the declaration rewrites). Stores the results as `*_boxes` whatsit
60/// properties for the constructor half ([`emit_declare_element`]).
61pub(super) fn normalize_declare_keys(
62  whatsit: &mut Whatsit,
63  tag_digested: Option<&Digested>,
64  description_digested: Option<&Digested>,
65) {
66  let stuff = description_digested.or(tag_digested);
67  let (term, mut desc) = stuff.map(split_declare_tag).unwrap_or((None, None));
68  let short: Option<Vec<Digested>> = if description_digested.is_some() {
69    tag_digested.map(|d| d.unlist()).or_else(|| desc.clone())
70  } else {
71    None
72  };
73  if desc.is_none() {
74    desc = description_digested.or(tag_digested).map(|d| d.unlist());
75  }
76  if let Some(t) = term {
77    whatsit.set_property("term_boxes", Stored::VecDigested(t));
78  }
79  if let Some(s) = short {
80    whatsit.set_property("short_boxes", Stored::VecDigested(s));
81  }
82  if let Some(d) = desc {
83    whatsit.set_property("desc_boxes", Stored::VecDigested(d));
84  }
85}

87/// Register a declaration in the LATEXML_DECLARATIONS state string — the
88/// math parser's font-aware fast-path registry (`apply_lx_declarations`).
89/// Rust-only companion to the rewrite path.
90///
91/// Line format: `body_text \t role \t name \t meaning \t decl_id \t
92/// match_font \t scope_prefix`. The trailing match_font makes
93/// apply_lx_declarations font-aware (a plain italic `$x$` must not annotate
94/// a bold `\mathbf{x}`), mirroring the font-aware rewrite path
95/// (declare_node_matches). Empty when the pattern carried no distinguishing
96/// font. The 7th field gates UNTAGGED `scope=section` declarations (no
97/// decl_id to carry the section prefix) so the fast path doesn't apply them
98/// document-globally (PR_READINESS cluster C).
99pub(super) fn record_declaration_lines(
100  body_text: &str,
101  role: &str,
102  name_val: &str,
103  meaning: &str,
104  decl_id: &str,
105  match_font: Option<&str>,
106  scope_opt: &str,
107) -> Result<()> {
108  let key = "LATEXML_DECLARATIONS";
109  let mut decls: Vec<String> = match lookup_value(key) {
110    Some(Stored::String(s)) => {
111      let s_str = with(s, |r| r.to_string());
112      if s_str.is_empty() {
113        Vec::new()
114      } else {
115        s_str.split('\n').map(String::from).collect()
116      }
117    },
118    _ => Vec::new(),
119  };
120  let match_font_field = match_font.unwrap_or("");
121  let scope_prefix = if scope_opt == "section" {
122    if !decl_id.is_empty() {
123      decl_id.split('.').next().unwrap_or("").to_string()
124    } else {
125      // afterDigest — where \thesection@ID is still correct.
126      do_expand(T_CS!("\\thesection@ID"))
127        .ok()
128        .map(|t| t.to_string().trim().to_string())
129        .unwrap_or_default()
130    }
131  } else {
132    String::new()
133  };
134  decls.push(format!(
135    "{}\t{}\t{}\t{}\t{}\t{}\t{}",
136    body_text, role, name_val, meaning, decl_id, match_font_field, scope_prefix
137  ));
138  // Mathcode decoding for single-char bodies
139  if body_text.chars().count() == 1 {
140    let ch = body_text.chars().next().unwrap();
141    if let Some(mathcode) = lookup_mathcode(&ch.to_string())
142      && mathcode > 0
143    {
144      let decoded_pos = (mathcode % 256) as u8;
145      let decoded_fam = (mathcode / 256) % 16;
146      let font_key = format!("textfont_{decoded_fam}");
147      if let Some(Stored::Token(ref ftok)) = lookup_value(&font_key) {
148        // Extract encoding before calling font::decode — decode may
149        // trigger preload_font_map → assign_value, and with_font_info
150        // holds a State borrow while its closure runs (see
151        // mathchar.rs fix for 0711.4787 RefCell panic pattern).
152        let mut encoding_opt: Option<String> = with_font_info(ftok, |fontinfo| {
153          if let Some(Stored::Font(info)) = fontinfo.unwrap_or(None) {
154            info.encoding.as_ref().map(|s| s.to_string())
155          } else {
156            None
157          }
158        });
159        // Fallback (mirror mathchar.rs L862-887): when `fontinfo_<cs>`
160        // didn't round-trip through the dump as a `Stored::Font`, but
161        // its `font_shared_key_<cs>` pointer DID, derive encoding from
162        // the font name via `decode_fontname`. Without this, dump-mode
163        // \lxDeclare doesn't add the alternate codepoint pattern (e.g.
164        // `*` → `∗`) and overrides on \ast etc. silently fail.
165        if encoding_opt.is_none() {
166          let shared_key = with_value(
167            &format!("font_shared_key_{}", ftok.with_str(ToString::to_string)),
168            |v| match v {
169              Some(Stored::String(s)) => with(*s, |str| Some(str.to_string())),
170              _ => None,
171            },
172          );
173          if let Some(sk) = shared_key
174            && let Some(name) = sk.strip_prefix("fontinfo_")
175          {
176            let props = font::decode_fontname(name, None, None);
177            if let Some(props) = props {
178              encoding_opt = props.encoding.as_ref().map(|s| s.to_string());
179            }
180          }
181        }
182        if let Some(encoding) = encoding_opt {
183          let decoded = font::decode(decoded_pos, Some(encoding), false);
184          if let Some(dc) = decoded {
185            let ds = dc.to_string();
186            if ds != body_text {
187              // Same 6-field shape (empty decl_id, trailing match_font)
188              // so apply_lx_declarations parses it uniformly.
189              decls.push(format!(
190                "{}\t{}\t{}\t{}\t\t{}",
191                ds,
192                role,
193                name_val,
194                meaning,
195                match_font.unwrap_or("")
196              ));
197            }
198          }
199        }
200      }
201    }
202  }
203  assign_value(
204    key,
205    Stored::String(pin(decls.join("\n"))),
206    Some(Scope::Global),
207  );
208  Ok(())
209}

211/// The shared `<ltx:declare>` element constructor — Perl `\lxDeclare`
212/// L474-485 / `\@lxDefMathDeclare` L387-396: `<tags><tag role="term">…</tag>
213/// <tag role="short">…</tag></tags>` then `<text>description</text>`, all
214/// from DIGESTED boxes (a `$x$: …` description term renders as real Math
215/// and is itself subject to the declaration rewrites).
216pub(super) fn emit_declare_element(
217  document: &mut Document,
218  decl_id: &str,
219  term_boxes: Option<Vec<Digested>>,
220  short_boxes: Option<Vec<Digested>>,
221  desc_boxes: Option<Vec<Digested>>,
222) -> Result<()> {
223  // Perl: floatToElement('ltx:declare') positions at a container that accepts <declare>
224  let saved = document.float_to_element("ltx:declare", false)?;
225  let mut attrs_map = HashMap::default();
226  attrs_map.insert("xml:id".to_string(), decl_id.to_string());
227  let _decl_node = document.open_element("ltx:declare", Some(attrs_map), None)?;
228  if term_boxes.is_some() || short_boxes.is_some() {
229    document.open_element("ltx:tags", None, None)?;
230    if let Some(term) = term_boxes {
231      let mut tag_attrs = HashMap::default();
232      tag_attrs.insert("role".to_string(), "term".to_string());
233      document.open_element("ltx:tag", Some(tag_attrs), None)?;
234      for b in &term {
235        document.absorb(b, None)?;
236      }
237      document.close_element("ltx:tag")?;
238    }
239    if let Some(short) = short_boxes {
240      let mut tag_attrs = HashMap::default();
241      tag_attrs.insert("role".to_string(), "short".to_string());
242      document.open_element("ltx:tag", Some(tag_attrs), None)?;
243      for b in &short {
244        document.absorb(b, None)?;
245      }
246      document.close_element("ltx:tag")?;
247    }
248    document.close_element("ltx:tags")?;
249  }
250  if let Some(desc) = desc_boxes {
251    let _text_node = document.open_element("ltx:text", None, None)?;
252    for b in &desc {
253      document.absorb(b, None)?;
254    }
255    document.close_element("ltx:text")?;
256  }
257  document.close_element("ltx:declare")?;
258  if let Some(ref save) = saved {
259    document.set_node(save);
260  }
261  Ok(())
262}

264/// Perl `getDeclarationScope`: resolve `scope=section` to the current
265/// section's `id:` scope. Uses the decl_id prefix (e.g. `S1` from
266/// `S1.XMD1`) since it was computed in afterDigest where `\thesection@ID`
267/// is correct — in afterConstruct it may be stale; falls back to walking
268/// the document node's ancestor section.
269pub(super) fn get_declaration_scope(
270  document: &Document,
271  scope_opt: &str,
272  decl_id: &str,
273) -> Option<Scope> {
274  if scope_opt != "section" {
275    return None;
276  }
277  let section_id = if !decl_id.is_empty() {
278    decl_id.split('.').next().unwrap_or("").to_string()
279  } else {
280    // Fallback: use the node's ancestor section id
281    let mut node = document.get_node().clone();
282    let mut sid = String::new();
283    loop {
284      if node.get_name() == "section" {
285        if let Some(id) = node
286          .get_property("xml:id")
287          .or_else(|| node.get_property("id"))
288        {
289          sid = id;
290        }
291        break;
292      }
293      match node.get_parent() {
294        Some(p) => node = p,
295        None => break,
296      }
297    }
298    sid
299  };
300  if !section_id.is_empty() {
301    Some(Scope::Named(pin(format!("id:{section_id}"))))
302  } else {
303    None
304  }
305}

307/// Perl `createDeclarationRewrite`: build the rewrite rule from whatever
308/// attributes exist — role/name/meaning AND decl_id alike (so a tag-only
309/// declaration still marks its matches) — compile the pattern, and UNSHIFT
310/// the rule in front of the others (later declarations preempt earlier ones
311/// via `_matched`). `replace` and `attributes` are mutually exclusive: a
312/// `replace=` declaration digests its replacement at rewrite time
313/// (Core/Rewrite.pm `compile_replacement`) instead of marking attributes.
314#[allow(clippy::too_many_arguments)]
315pub(super) fn create_declaration_rewrite(
316  scope: Option<Scope>,
317  role: String,
318  name_val: String,
319  meaning: String,
320  decl_id: String,
321  body_text: &str,
322  nowrap: bool,
323  replace_tokens: Option<Tokens>,
324) {
325  use latexml_core::rewrite::{Rewrite, RewriteOptions};
326  use rustc_hash::FxHashMap;
327  let mut attrs = FxHashMap::default();
328  if !role.is_empty() {
329    attrs.insert("role".to_string(), role);
330  }
331  if !name_val.is_empty() {
332    attrs.insert("name".to_string(), name_val);
333  }
334  if !meaning.is_empty() {
335    attrs.insert("meaning".to_string(), meaning);
336  }
337  if !decl_id.is_empty() {
338    attrs.insert("decl_id".to_string(), decl_id);
339  }
340  // Perl createDeclarationRewrite: ($nowrap ? (_nowrap => $nowrap) : ()) —
341  // read by set_attributes_wild; underscore-prefixed so it is never
342  // serialized onto the document.
343  if nowrap {
344    attrs.insert("_nowrap".to_string(), "1".to_string());
345  }
346  // Compile pattern: determine XPath, type, filters, wildcard paths.
347  // Font-awareness is applied Rust-side (declare_node_matches for the
348  // rewrite path, apply_lx_declarations for the post-rewrite fast path) —
349  // NOT baked into the XPath, since the serialized `@font` attribute isn't
350  // finalized until after math parsing (see compile_declare_pattern).
351  let pat = if body_text.contains('_') || body_text.contains('\\') || body_text.contains('\'') {
352    compile_declare_pattern(body_text)
353  } else {
354    // Simple single-token pattern: match XMTok by text; the Simple
355    // filter in declare_node_matches rejects non-matching fonts.
356    DeclarePattern {
357      xpath:          format!(
358        "descendant-or-self::*[local-name()='XMTok' and text()='{}']",
359        body_text.replace('\'', "&apos;")
360      ),
361      pattern_type:   DeclarePatternType::Simple,
362      base_text:      None,
363      sub_text:       None,
364      accent_name:    None,
365      has_wildcard:   false,
366      wildcard_paths: None,
367      font_class:     None,
368    }
369  };
370  if pat.xpath.is_empty() {
371    // Unrecognized pattern: make the compile failure VISIBLE — a silent
372    // skip is indistinguishable from a legitimate no-match, the exact
373    // precondition of the historical "wildcard XMDuals vanished" mode
374    // (PR_READINESS cluster C).
375    Warn!(
376      "unexpected",
377      "lxDeclare",
378      "\\lxDeclare pattern not recognized by the rewrite compiler; declaration will not match",
379      format!("pattern: '{body_text}'")
380    );
381    return;
382  }
383  // Pattern types determine select_count (see DeclarePattern::select_count):
384  // Subscript/prime patterns match base XMTok + POSTSUBSCRIPT/POSTSUPERSCRIPT
385  // sibling (select_count=2, pre-parsed DOM). Accents match the single XMApp.
386  let select_count = pat.select_count();
387  let rewrite = if let Some(replace_tks) = replace_tokens {
388    use std::rc::Rc;
389
390    use latexml_core::rewrite::RewriteReplaceClosure;
391    let closure: RewriteReplaceClosure = Rc::new(move |document, _nodes| {
392      // Perl Core/Rewrite.pm::compile_replacement (Tokens branch, as
393      // fixed by upstream b17cc621): digest the pattern in
394      // restricted_horizontal mode with a neutral font; for a math
395      // rule, unwrap the outer List (Rust's digest() always wraps its
396      // result in a List — the same shape Perl's changed autosimplify
397      // now produces), take its single body, then getBody and absorb.
398      begin_mode("restricted_horizontal")?;
399      neutralize_font();
400      let mut rbox = digest(replace_tks.clone())?;
401      end_mode("restricted_horizontal")?;
402      let unwrapped = if let DigestedData::List(l) = rbox.data() {
403        let items = l.borrow().unlist();
404        if items.len() == 1 {
405          Some(items[0].clone())
406        } else {
407          None
408        }
409      } else {
410        None
411      };
412      if let Some(u) = unwrapped {
413        rbox = u;
414      }
415      if let Some(body) = rbox.get_body()? {
416        rbox = body;
417      }
418      document.absorb(&rbox, None)?;
419      Ok(())
420    });
421    Rewrite::new("math", RewriteOptions {
422      xpath: Some(pat.xpath.clone()),
423      replace: Some(closure),
424      wildcard_paths: pat.wildcard_paths.clone(),
425      select_count,
426      scope,
427      // Replace rules need the SAME declare-side filtering as attribute
428      // rules — without it a `$x_\WildCard$` replace pattern deletes
429      // the matched x plus an ARBITRARY next sibling even with no
430      // subscript present (PR_READINESS cluster C).
431      declare_filter: Some(pat),
432      ..RewriteOptions::default()
433    })
434  } else {
435    Rewrite::new("math", RewriteOptions {
436      xpath: Some(pat.xpath.clone()),
437      attributes_map: Some(attrs),
438      wildcard_paths: pat.wildcard_paths.clone(),
439      select_count,
440      scope,
441      declare_filter: Some(pat),
442      ..RewriteOptions::default()
443    })
444  };
445  // Perl: "Put this rule IN FRONT of other rules!" — UnshiftValue.
446  unshift_value("DOCUMENT_REWRITE_RULES", vec![rewrite]);
447}
