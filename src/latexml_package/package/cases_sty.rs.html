3/// Perl: cases.sty.ltxml — numcases and subnumcases environments
4/// Perl: numcasesBindings($lhs) — creates 3-column alignment for numcases
5fn numcases_bindings(lhs: Tokens) -> Result<()> {
6  use latexml_core::alignment::cell::Cell;
7  let col1 = Cell {
8    before: Some(Tokens!(
9      T_CS!("\\hfil"),
10      T_MATH!(),
11      T_CS!("\\lx@hidden@bgroup"),
12      T_CS!("\\displaystyle")
13    )),
14    after: Some(Tokens!(T_CS!("\\lx@hidden@egroup"), T_MATH!())),
15    ..Cell::default()
16  };
17  let col2 = Cell {
18    before: Some(Tokens!(
19      T_MATH!(),
20      T_CS!("\\lx@hidden@bgroup"),
21      T_CS!("\\displaystyle")
22    )),
23    after: Some(Tokens!(
24      T_CS!("\\lx@hidden@egroup"),
25      T_MATH!(),
26      T_CS!("\\hfil")
27    )),
28    ..Cell::default()
29  };
30  let col3 = Cell {
31    before: Some(Tokens!(T_CS!("\\lx@hidden@bgroup"))),
32    after: Some(Tokens!(T_CS!("\\lx@hidden@egroup"), T_CS!("\\hfil"))),
33    ..Cell::default()
34  };
35  use latexml_core::alignment::template::TemplateConfig;
36  let mut template = Template::new(TemplateConfig::default());
37  template.add_column(col1);
38  template.add_column(col2);
39  template.add_column(col3);
40
41  let mut attrs = string_map! { "class" => "ltx_eqn_numcases" };
42  if let Ok(Some(colsep)) = lookup_register("\\arraycolsep", Vec::new()) {
43    let colsep_dim: Dimension = colsep.into();
44    let doubled = Dimension::new(colsep_dim.value_of() * 2);
45    attrs.insert(String::from("colsep"), doubled.to_attribute());
46  }
47
48  let alignment = Alignment::new(AlignmentConfig {
49    template:        Some(template),
50    open_container:  Rc::new(move |document, mut props| {
51      if let Ok(id_props) = ref_step_id("@equationgroup") {
52        if let Some(id) = id_props.get("id") {
53          props.insert(String::from("xml:id"), id.to_string());
54        }
55      }
56      // Perl: %attributes has class => 'ltx_eqn_numcases' which overrides
57      // the openContainer's default class. Use 'ltx_eqn_numcases' directly.
58      props
59        .entry(String::from("class"))
60        .or_insert_with(|| String::from("ltx_eqn_numcases"));
61      document
62        .open_element("ltx:equationgroup", Some(props), None)
63        .map(Option::Some)
64    }),
65    close_container: Rc::new(|document| document.close_element("ltx:equationgroup")),
66    open_row:        Rc::new(|document, mut props| {
67      if let Some(Stored::HashStored(eq_props)) = state::remove_value("EQUATIONROW_PROPS") {
68        if let Some(id) = eq_props.get("id") {
69          props.insert(String::from("xml:id"), Stored::from(id.to_string()));
70        }
71      }
72      let tags_digested = props.remove("tags");
73      let str_props: HashMap<String, String> =
74        props.into_iter().map(|(k, v)| (k, v.to_string())).collect();
75      document.open_element("ltx:equation", Some(str_props), None)?;
76      if let Some(Stored::Digested(d)) = tags_digested {
77        document.absorb(&d, None)?;
78      }
79      Ok(())
80    }),
81    close_row:       Rc::new(|document| document.close_element("ltx:equation")),
82    open_column:     Rc::new(|document, props| {
83      document
84        .open_element("ltx:_Capture_", Some(props), None)
85        .map(Option::Some)
86    }),
87    close_column:    Rc::new(|document| document.close_element("ltx:_Capture_")),
88    is_math:         true,
89    properties:      SymHashMap::default(),
90    xml_attributes:  attrs,
91  });
92  assign_alignment(alignment, None);
93  state::let_i(&T_MATH!(), &T_CS!("\\lx@dollar@in@mathmode"), None);
94  let mut lhs_expansion = lhs.unlist();
95  lhs_expansion.push(T_ALIGN!());
96  def_macro(
97    T_CS!("\\@numcases@LHS"),
98    None,
99    Tokens::new(lhs_expansion),
100    None,
101  )?;
102  Let!("\\\\", "\\@numcases@newline");
103  state::let_i(
104    &T_CS!("\\lx@alignment@row@before"),
105    &T_CS!("\\eqnarray@row@before"),
106    None,
107  );
108  state::let_i(
109    &T_CS!("\\lx@alignment@row@after"),
110    &T_CS!("\\eqnarray@row@after"),
111    None,
112  );
113  Ok(())
114}

116#[rustfmt::skip]
117LoadDefinitions!({
118  DefPrimitive!("\\@numcases@bindings{}", sub[(lhs)] {
119    numcases_bindings(lhs)?;
120  });
121
122  // Perl cases.sty.ltxml L41/L44/L51/L54 all carry `locked => 1`: the
123  // raw cases.sty is loaded alongside our binding for layout fidelity
124  // and tries to redefine \numcases / \endnumcases / \subnumcases /
125  // \endsubnumcases over our alignment trampolines. The lock keeps
126  // our versions in place.
127  DefMacro!("\\numcases{}",
128    "\\@numcases@bindings{#1}\\@@numcases\\@equationgroup@numbering{numbered=1,preset=1,deferretract=1,grouped=1,aligned=1}\\lx@begin@alignment\\@numcases@LHS",
129    locked => true);
130  DefMacro!("\\endnumcases", "\\lx@end@alignment\\end@numcases",
131    locked => true);
132
133  DefMacro!("\\subnumcases{}",
134    "\\@numcases@bindings{#1}\\lx@numcases@subnumbering@begin\\@@numcases\\@equationgroup@numbering{numbered=1,preset=1,deferretract=1,grouped=1,aligned=1}\\lx@begin@alignment\\@numcases@LHS",
135    locked => true);
136  DefMacro!("\\endsubnumcases", "\\lx@end@alignment\\end@numcases\\lx@numcases@subnumbering@end",
137    locked => true);
138
139  // Sub-numbering: step parent equation, save counter, reset, redefine \theequation
140  // Perl: cases.sty.ltxml L57-64
141  DefPrimitive!("\\lx@numcases@subnumbering@begin", sub[_args] {
142    use latexml_core::binding::counter::dialect::reset_counter;
143    use latexml_core::mouth;
144    // Step the equation counter and get properties (id, refnum)
145    let eqn_props = ref_step_counter("equation", false)?;
146    // Expand \theequation to get the parent equation number text (e.g. "3")
147    let eqnum_toks = gullet::do_expand(T_CS!("\\theequation"))?;
148    let eqnum_str = eqnum_toks.to_string();
149    // Save current equation counter value
150    let saved = state::lookup_register("\\c@equation", Vec::new())?.map_or(0, |rv| {
151      match rv {
152        RegisterValue::Number(n) => n.0,
153        _ => 0,
154      }
155    });
156    state::assign_value("SAVED_EQUATION_NUMBER", Stored::Number(Number::new(saved)), Some(Scope::Global));
157    // Reset equation counter to 0 for sub-lettering
158    reset_counter(&T_OTHER!("equation"))?;
159    // Redefine \theequation to parent_number + \alph{equation} (e.g. "3a", "3b")
160    let new_theequation = format!("{}\\alph{{equation}}", eqnum_str);
161    def_macro(T_CS!("\\theequation"), None, mouth::tokenize_internal(&new_theequation), None)?;
162    // Redefine \theequation@ID for xml:id generation (e.g. "S0.E3.\@equation@ID")
163    let id_str = eqn_props.iter().find_map(|(k, v)| {
164      if arena::with(*k, |ks| ks == "id") { Some(v.to_string()) } else { None }
165    }).unwrap_or_default();
166    if !id_str.is_empty() {
167      let new_id_macro = format!("{}.\\@equation@ID", id_str);
168      def_macro(T_CS!("\\theequation@ID"), None, mouth::tokenize_internal(&new_id_macro), None)?;
169    }
170  });
171  DefPrimitive!("\\lx@numcases@subnumbering@end", sub[_args] {
172    // Restore saved equation counter
173    if let Some(Stored::Number(n)) = state::lookup_value("SAVED_EQUATION_NUMBER") {
174      let _ = state::assign_register("\\c@equation", RegisterValue::Number(n), Some(Scope::Global), Vec::new());
175    }
176  });
177
178  DefMacro!("\\@numcases@newline[]",
179    "\\ifx.#1.\\lx@alignment@newline\\else\\lx@alignment@newline[#1]\\fi\\@numcases@LHS");
180  DefMacro!("\\@numcases@cr", "\\lx@alignment@cr\\@numcases@LHS");
181
182  DefConstructor!("\\@@numcases SkipSpaces DigestedBody", "#1",
183    before_digest => sub { bgroup(); },
184    after_construct => sub[document, _whatsit] {
185      let node = document.get_node();
186      if let Some(equationgroup) = node.get_last_child() {
187        rearrange_numcases(document, &equationgroup)?;
188      }
189    });
190
191  DefPrimitive!("\\end@numcases", sub[_args] { egroup()?; });
192});

194fn rearrange_numcases(document: &mut Document, equationgroup: &Node) -> Result<()> {
195  use crate::engine::base_xmath::equationgroup_join_cols;
196  let equations: Vec<Node> = document.findnodes("ltx:equation", Some(equationgroup));
197  for equation in equations {
198    let cells: Vec<Node> = document.findnodes("ltx:_Capture_", Some(&equation));
199    if cells.is_empty() {
200      continue;
201    }
202    let cell1cont: Vec<Node> = cells[0].get_child_elements();
203    if cells.len() == 1
204      && cell1cont.len() == 1
205      && cell1cont[0]
206        .get_attribute("class")
207        .unwrap_or_default()
208        .contains("ltx_intertext")
209    {
210      let mut cell1cont0 = cell1cont[0].clone();
211      cell1cont0.unlink_node();
212      let mut eq = equation.clone();
213      let _ = eq.add_prev_sibling(&mut cell1cont0);
214      eq.unlink_node();
215    } else if cells.len() == 1 && cell1cont.is_empty() {
216      let mut eq = equation.clone();
217      eq.unlink_node();
218    } else {
219      equationgroup_join_cols(document, 3, &mut equation.clone())?;
220    }
221  }
222  Ok(())
