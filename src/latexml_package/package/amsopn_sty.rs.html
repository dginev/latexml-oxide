1use crate::prelude::*;
2LoadDefinitions!({
3  RequirePackage!("amsgen");
4
5  // \DeclareMathOperator*{cs}{text}
6  //
7  // Use `.untex()` instead of `.to_string()` — the latter concatenates
8  // token texts with no separator, so `{\rm Aut}` (where the space after
9  // `\rm` was swallowed by control-word tokenization) becomes `{\rmAut}`,
10  // which tokenizes back as the single undefined CS `\rmAut`. `untex()`
11  // inserts a space at CS→letter boundaries (tokens.rs L392-405), so the
12  // round-trip through `def_math`'s internal `mouth::tokenize_internal`
13  // preserves the correct token structure.
14  //
15  // Perl avoids this entirely by passing Tokens directly to DefMathI via
16  // `Invocation(T_CS('\operatorname'), $star, $text)` — no stringify
17  // round-trip. Rust's `def_math` takes String, so we use the TeX-safe
18  // stringifier.
19  //
20  // Fixes sandbox papers 0806.2705 (`\rmTr`) and 0808.0535 (`\rmAut`/
21  // `\rmSpan`) whose `\DeclareMathOperator{\X}{{\rm X}}` patterns would
22  // otherwise produce undefined `\rmX` errors.
23  DefPrimitive!("\\DeclareMathOperator OptionalMatch:* {Token} {}", sub[(star, cs, text)] {
24    let text_str = text.untex();
25    let has_star = star.is_some();
26    // Perl L26-29: scriptpos => ($star ? \&doScriptpos : 'post') — starred form
27    // gets dynamic mid/post from current display style; bare form is always 'post'.
28    // revert_as => 'context' so source-export emits the user-facing CS name
29    // rather than the operatorname expansion. Both were previously dropped.
30    let opts = MathPrimitiveOptions {
31      role: Some(if has_star { "OPERATOR" } else { "OPFUNCTION" }.to_string()),
32      font: Some(fontmap!(family => "serif", series => "medium", shape => "upright").into()),
33      scriptpos: if has_star { None } else { Some("post".to_string()) },
34      dynamic_scriptpos: has_star,
35      revert_as: Some(Cow::Borrowed("context")),
36      ..Default::default()};
37    def_math(cs, None, text_str, opts)?;
38  });
39
40  // \operatorname*{text}
41  DefConstructor!("\\operatorname OptionalMatch:* {}",
42  "<ltx:XMWrap role='#role' scriptpos='#scriptpos'>#2</ltx:XMWrap>",
43  bounded => true, require_math => true,
44  font => { family => "serif", series => "medium", shape => "upright" },
45  properties => sub[args] {
46    let starred = args[0].is_some();
47    let role = if starred { "OPERATOR" } else { "OPFUNCTION" };
48    let scriptpos = if starred { "mid" } else { "post" };
49    Ok(stored_map!("role" => role, "scriptpos" => scriptpos))
50  });
51
52  DefConstructor!("\\operatornamewithlimits {}",
53    "<ltx:XMWrap role='OPERATOR' scriptpos='mid'>#1</ltx:XMWrap>",
54    bounded => true, require_math => true,
55    font => { family => "serif", series => "medium", shape => "upright" });
56
57  // Operator variants — Perl L33-38 ships scriptpos => \&doScriptpos so the
58  // operators sit mid (under/over) in display style and post (sub/super) in
59  // inline. Without it, Rust statically rendered everything as 'post', giving
60  // wrong placement in display-mode formulas.
61  DefMath!("\\injlim", "inj lim",
62    role => "LIMITOP", meaning => "injective-limit", dynamic_scriptpos => true);
63  DefMath!("\\projlim", "proj lim",
64    role => "LIMITOP", meaning => "projective-limit", dynamic_scriptpos => true);
65
66  // Perl: amsopn.sty.ltxml — var limit operators
67  DefMath!("\\varlimsup", "\\overline{\\operatorname{lim}}",
68    role => "LIMITOP", meaning => "limit-supremum");
69  DefMath!("\\varliminf", "\\underline{\\operatorname{lim}}",
70    role => "LIMITOP", meaning => "limit-infimum");
71  DefMath!("\\varinjlim", "\\underrightarrow{\\operatorname{lim}}",
72    role => "LIMITOP", meaning => "injective-limit");
73  DefMath!("\\varprojlim", "\\underleftarrow{\\operatorname{lim}}",
74    role => "LIMITOP", meaning => "projective-limit");
75
76  DefMacro!("\\nolimits@", "\\nolimits");
77  DefMacro!("\\nmlimits@", "\\displaylimits");
78  DefMacro!(
79    "\\qopname{}{}{}",
80    "\\mathop{#3}\\csname n#2limits@\\endcsname"
81  );
82});
