1use crate::prelude::*;
2LoadDefinitions!({
3  RequirePackage!("amsgen");
4
5  // \DeclareMathOperator*{cs}{text}
6  DefPrimitive!("\\DeclareMathOperator OptionalMatch:* {Token} {}", sub[(star, cs, text)] {
7    let text_str = text.to_string();
8    let has_star = star.is_some();
9    let opts = MathPrimitiveOptions {
10      role: Some(if has_star { "OPERATOR" } else { "OPFUNCTION" }.to_string()),
11      font: Some(fontmap!(family => "serif", series => "medium", shape => "upright").into()),
12      ..Default::default()};
13    def_math(cs, None, text_str, opts)?;
14  });
15
16  // \operatorname*{text}
17  DefConstructor!("\\operatorname OptionalMatch:* {}",
18    "<ltx:XMWrap role='#role' scriptpos='#scriptpos'>#2</ltx:XMWrap>",
19    bounded => true, require_math => true,
20    font => { family => "serif", series => "medium", shape => "upright" },
21    properties => sub[args] {
22      let starred = args[0].is_some();
23      let role = if starred { "OPERATOR" } else { "OPFUNCTION" };
24      let scriptpos = if starred { "mid" } else { "post" };
25      Ok(stored_map!("role" => role, "scriptpos" => scriptpos))
26    });
27
28  DefConstructor!("\\operatornamewithlimits {}",
29    "<ltx:XMWrap role='OPERATOR' scriptpos='mid'>#1</ltx:XMWrap>",
30    bounded => true, require_math => true,
31    font => { family => "serif", series => "medium", shape => "upright" });
32
33  // Operator variants
34  DefMath!("\\injlim", "inj lim",
35    role => "LIMITOP", meaning => "injective-limit");
36  DefMath!("\\projlim", "proj lim",
37    role => "LIMITOP", meaning => "projective-limit");
