5  // Perl: epstopdf.sty.ltxml
6  // Nothing to do here!
7  DefMacro!("\\epstopdfsetup{}", None);
8  // `\epstopdfDeclareGraphicsRule{ext}{type}{ext-out}{cmd}` (epstopdf-base.sty)
9  // registers an EPS→<other> conversion command for graphics inclusion.
10  // We delegate graphics format conversion to mutool/gs at the post-
11  // processing graphics phase, so author-supplied conversion rules
12  // have no effect on our HTML output. No-op stub mirrors Perl's
13  // "nothing to do" philosophy for the whole package.
14  // Witness: 2 papers in R06-R09 emitting `Error:undefined:
15  // \epstopdfDeclareGraphicsRule`.
16  DefMacro!("\\epstopdfDeclareGraphicsRule{}{}{}{}", None);
17  // `\epstopdfcall{cmd}` is the lower-level shell-out wrapper used by
18  // \epstopdfDeclareGraphicsRule. Same rationale: no-op.
19  DefMacro!("\\epstopdfcall{}", None);
20  // `\OutputFile` is `\edef`-set inside `\epstopdfDeclareGraphicsRule`
21  // / `\epstopdfcall` to the generated converted-file name. Since we
22  // stub those as no-ops, `\OutputFile` is never set. Subsequent
23  // references inside `\Gin@rule@` etc. then hit `Error:undefined`
24  // (~11 papers in R-stages). Initialize as empty so it never errors;
25  // its value would be the converted filename but we don't actually
26  // include via this path.
27  DefMacro!("\\OutputFile", None);
28});
