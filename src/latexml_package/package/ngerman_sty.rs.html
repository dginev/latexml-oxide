1use crate::prelude::*;

3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: ngerman.sty.ltxml
6  // This should be essentially the same, right?
7  // (considering we don't do hyphenation, etc)
8  RequirePackage!("babel", options => vec!["ngerman".to_string()]);
9
10  // Alias the `ngermanb` dialect's language number to `ngerman` (the real
11  // ngermanb.ldf does `\let\l@ngermanb\l@ngerman`). Without this,
12  // `\usepackage[…,ngermanb]{babel}` → `\selectlanguage{ngermanb}` errors
13  // "haven't defined the language 'ngermanb'". Parallels german_sty's
14  // `\l@germanb` alias (witness 1010.4065).
15  RawTeX!(r"\expandafter\ifx\csname l@ngermanb\endcsname\relax
16    \expandafter\ifx\csname l@ngerman\endcsname\relax
17      \expandafter\newlanguage\csname l@ngermanb\endcsname
18    \else
19      \expandafter\let\csname l@ngermanb\expandafter\endcsname\csname l@ngerman\endcsname
20    \fi
21  \fi");
22
23  // NGerman shares captions with german (reformed orthography, same strings).
24  RawTeX!(r"\providecommand\captionsngerman{%
25    \def\prefacename{Vorwort}\def\refname{Literatur}%
26    \def\abstractname{Zusammenfassung}\def\bibname{Literaturverzeichnis}%
27    \def\chaptername{Kapitel}\def\appendixname{Anhang}%
28    \def\contentsname{Inhaltsverzeichnis}%
29    \def\listfigurename{Abbildungsverzeichnis}%
30    \def\listtablename{Tabellenverzeichnis}%
31    \def\indexname{Index}\def\figurename{Abbildung}%
32    \def\tablename{Tabelle}\def\partname{Teil}%
33    \def\pagename{Seite}\def\seename{siehe}%
34    \def\alsoname{siehe auch}\def\proofname{Beweis}}");
35  RawTeX!(r"\providecommand\datengerman{}");
36  // ngermanb.ldf: `\def\dq{"}` — `\dq` yields a literal double-quote (see the
37  // matching note in german_sty.rs). [ngerman,english] babel loads this path
38  // (not german_sty.rs), so define it here too. Witness 1804.06196.
39  RawTeX!(r"\providecommand\dq{\textquotedbl}");
40});
