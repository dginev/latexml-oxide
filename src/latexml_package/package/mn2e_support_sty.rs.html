52  // Keywords — Perl mn2e_support.sty.ltxml L48-54:
53  //   DefEnvironment('{keywords}', '',
54  //     afterDigest => sub { push 'ltx:classification'→frontmatter });
55  //
56  // As an environment, `\endkeywords` is auto-defined, so raw
57  // `mn2e-breakabs.sty` redefinitions of `\endkeywords` (which reference
58  // undefined `\SFB@keywordstrue`) never fire.
59  //
60  // Was: emitted `<ltx:classification>` directly via constructor template,
61  // which inserted the element at the current insertion point. Driver:
62  // 2003.11440 `\begin{abstract}...\begin{keywords}{...}\end{keywords}\end{abstract}`
63  // — the `\begin{keywords}` was nested INSIDE the open abstract, where
64  // ltx:classification isn't a valid child, producing
65  // "ltx:classification isn't allowed in <ltx:abstract>".
66  //
67  // Mirror Perl directly:
68  //   DefEnvironment('{keywords}', '',
69  //     afterDigest => sub {
70  //       my $frontmatter = LookupValue('frontmatter');
71  //       push(@{ $$frontmatter{'ltx:classification'} },
72  //         ['ltx:classification', { scheme => 'keywords' }, @LaTeXML::LIST]);
73  //       return; });
74  // — empty constructor template, after-digest hook captures the digested
75  // body list and pushes a (tag, attrs, body) entry under the
76  // `ltx:classification` key of the global frontmatter hash. No mode
77  // bracket, no locked flag — matches Perl.
78  DefEnvironment!("{keywords}", "",
79    after_digest => {
80      let body = List::new(clone_box_list());
81      with_value_mut("frontmatter", |fm_opt| {
82        let frontmatter = match fm_opt {
83          Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
84          _ => return Ok::<(), latexml_core::Error>(()),
85        };
86        let entry = (
87          "ltx:classification".to_string(),
88          Some(string_map!("scheme" => "keywords")),
89          body.into(),
90        );
91        frontmatter.entry("ltx:classification".to_string())
92          .or_insert_with(Vec::new)
93          .push(entry);
94        Ok(())
95      })?;
96    });
