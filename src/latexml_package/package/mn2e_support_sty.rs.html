246  // inside `\begin{equation} … \end{equation}` where `$HJD$` is a
247  // text-escape to typeset "HJD" in roman. Without the override, Rust
248  // raised Error:expected:$ Missing $ closing display math at every
249  // occurrence. Matches the identical port in aa_support_sty.
250  use crate::engine::latex_constructs::{
251    after_equation, before_equation, prepare_equation_counter,
252  };
253  DefEnvironment!(
254    "{equation}",
255    "<ltx:equation xml:id='#id'>#tags<ltx:Math mode='display'><ltx:XMath>#body</ltx:XMath></ltx:Math></ltx:equation>",
256    mode => "display_math",
257    before_digest => {
258      prepare_equation_counter(stored_map!("numbered" => true, "preset" => true));
259      before_equation()?;
260      Let!(T_MATH!(), "\\lx@dollar@in@mathmode");
