6/// Perl subcaption.sty.ltxml L66/76/86/96: `properties => sub { (width => $_[2]) }`.
7/// Perl sets this on ALL four sub-float envs ({subfigure}, {subfigure*},
8/// {subtable}, {subtable*}); the Rust-only {subcaptionblock}/{subcaptionblock*}
9/// aliases inherit the same semantics. Records the sub-float's `{Dimension}`
10/// argument (args[1]) as the box's `width` property, so
11/// `arrange_panels_and_breaks` can compute the per-row layout from actual panel
12/// widths (without it, a panel reports its natural content width — the full
13/// float width — and every panel starts its own row; arXiv 2605.00347). Mirrors
14/// Perl storing the digested Dimension object.
15fn subcaption_width_props(args: &[Option<Digested>]) -> Result<SymHashMap<Stored>> {
16  let mut props: SymHashMap<Stored> = SymHashMap::default();
17  if let Some(w) = args
18    .get(1)
19    .and_then(|a| a.as_ref())
20    .and_then(|a| Dimension::spec_to_f64(&a.to_string()).ok())
21  {
22    props.insert("width", Stored::Dimension(Dimension::new_f64(w)));
23  }
24  Ok(props)
25}

65  // sub-prefixed) locally, then delegate to \caption.
66  DefMacro!("\\subcaption OptionalMatch:* []{}", sub[(_star, opt, caption)] {
67    let mut tokens = Vec::new();
68    if has_meaning(&T_CS!("\\@captype")) {
69      let ctype = do_expand(Tokens!(T_CS!("\\@captype")))?.to_string();
70      let ctype = ctype.trim().to_string();
71      if !ctype.is_empty() && !ctype.starts_with("sub") {
72        // Local redefinition via \def\@captype{sub<ctype>} tokens.
73        tokens.push(T_CS!("\\def"));
74        tokens.push(T_CS!("\\@captype"));
75        tokens.push(T_BEGIN!());
76        tokens.extend(Explode!(s!("sub{}", ctype)));
77        tokens.push(T_END!());
78      }
79    }
80    tokens.push(T_CS!("\\caption"));
81    if let Some(o) = opt {
82      tokens.push(T_OTHER!("["));
83      tokens.extend(o.unlist());
84      tokens.push(T_OTHER!("]"));
85    }
86    tokens.push(T_BEGIN!());
87    tokens.extend(caption.unlist());
88    tokens.push(T_END!());
89    Ok(Tokens::new(tokens))
90  });

93  // Subfigure environments
94  // Perl: beforeFloat('subfigure', preincrement => 'figure') / afterFloat
95  //
96  // EMULATE LaTeX's `\newenvironment` "already defined" guard. subcaption and
97  // the (unsupported) subfigure package are officially INCOMPATIBLE: both want
98  // to own `\subfigure`/`\subtable`. subfigure.sty binds them as
99  // `\subfigure[][]{}` / `\subtable[][]{}` MACROS (self-contained, one
100  // mandatory body); subcaption binds them as `{subfigure}[]{Dimension}` /
101  // `{subtable}[]{Dimension}` ENVIRONMENTS. In real LaTeX, subcaption declares
102  // these via `\newenvironment{subfigure}`, which REFUSES to redefine an
103  // already-defined `\subfigure` (it raises "Command \subfigure already defined"
104  // and keeps subfigure.sty's macro); our `DefEnvironment!` is unconditional
105  // (like `\def`, not `\newenvironment`), so it used to CLOBBER the macro.
106  // Then a `\subfigure[]{\includegraphics{...}}` (subfigure.sty's macro syntax)
107  // reparsed as `\begin{subfigure}` with `{\includegraphics{...}}` read as the
108  // `{Dimension}` (→ "Missing number", treated as zero) and the environment
109  // opened with no matching `\end{subfigure}` in the source, leaking an
110  // internal_vertical group that swallowed the rest of the document (figures,
111  // sections, bibliography). Vendored Perl clobbers the same way and *times out*
112  // on the witness — upstream candidate (KNOWN_PERL_ERRORS #48). So mirror the
113  // `\newenvironment` guard: only bind these two envs when the colliding command
114  // is not already defined, and Warn about the package conflict when it is.
115  // (`{subfigure*}`/`{subtable*}`/`{subcaptionblock}` names don't collide with
116  // subfigure.sty and stay unconditional.) Witness 2507.21938.
117  let subfigure_predefined = has_meaning(&T_CS!("\\subfigure"));

123       environment is not installed (mirrors \\newenvironment's guard); \
124       subfigure.sty's \\subfigure macro is kept");
