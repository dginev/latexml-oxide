3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: keyval.sty.ltxml
6  InputDefinitions!("keyval", noltxml => true, extension => Some(Cow::Borrowed("sty")));
7
8  // HOOK into define@key to make the latexml definitions as well
9  // \define@key{keyset}{key}[default]{code}
10  DefPrimitive!("\\define@key{}{}[]{}", sub[(keyset_tks, key_tks, default_opt, code)] {
11    let keyset = gullet::do_expand(keyset_tks)?.to_string();
12    let key = gullet::do_expand(key_tks)?.to_string();
13    let default = default_opt.map(|d: Tokens| d.to_string());
14
15    latexml_core::keyval::define(KeyvalConfig {
16      prefix: "KV",
17      keyset: &keyset,
18      key: &key,
19      vtype: "",
20      default: default.as_deref(),
21      code: Some(ExpansionBody::Tokens(code)),
22      ..KeyvalConfig::default()
23    })?;
24  });
25
26  // \setkeys{keyset}{keyvals}
27  DefMacro!("\\setkeys{}", sub[(keyset_tks)] {
28    let keyset = gullet::do_expand(keyset_tks)?.to_string();
29    let keysets: Vec<String> = keyset.split(',')
30      .map(|s| s.trim().to_string())
31      .filter(|s| !s.is_empty())
32      .collect();
33
34    let mut keyvals = KeyVals::new(KeyvalsConfig {
35      prefix: None,
36      keysets,
37      set_all: false,
38      set_internals: true,
39      skip: Vec::new(),
40      skip_missing: latexml_core::keyvals::SkipMissing::None,
41      hook_missing: None,
42    });
43    keyvals.read_from(T_END!(), false)?;
44    Ok(keyvals.set_keys_expansion())
45  });
46});
