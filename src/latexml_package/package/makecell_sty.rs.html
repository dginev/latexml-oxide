1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: makecell.sty.ltxml
6  // Load raw TeX first
7  InputDefinitions!("makecell", noltxml => true, extension => Some(Cow::Borrowed("sty")));
8
9  // Mark thead et.al as headers (row & column)
10  DefPrimitive!("\\lx@makecell@head", sub[_args] {
11    if let Some(alignment) = lookup_alignment() {
12      if let Some(data) = alignment.alignment_cell() {
13        if let Some(col) = data.borrow_mut().current_column() {
14          col.thead_in_column = true;
15          col.thead_in_row = true;
16        }
17      }
18    }
19    Ok(())
20  });
21
22  // Redefine \theadfont at BeginDocument to include heading marker
23  RawTeX!(r"\AtBeginDocument{\let\lx@orig@theadfont\theadfont\def\theadfont{\lx@orig@theadfont\lx@makecell@head}}");
24
25  // Since we use \thead, disable guessing
26  AssignValue!("GUESS_TABULAR_HEADERS" => false, Scope::Global);
27
28  // \rothead: override to prevent infinite recursion from nested \thead
29  // in raw TeX definition. Simplified: delegate to \thead with content.
30  DefMacro!("\\rothead[]{}",
31    "\\thead[#1]{#2}");
32  DefMacro!("\\rotcell[]{}",
33    "\\makecell[#1]{#2}");
34
35  // \diaghead (slopex,slopey) {width}{item A}{item B}
36  // Perl: arranges args in a table with appropriate alignment based on slope.
37  // Simplified: uses \shortstack for each head item (matching Perl's lx@diag@head)
38  DefMacro!("\\lx@diag@head{}{}",
39    "{\\theadfont\\shortstack[#1]{#2}}");
40
41  // Full \diaghead: overrides raw TeX with simplified Perl-style version
42  // OptionalPair parameter reads (x,y) slope if present
43  DefMacro!("\\diaghead OptionalPair {}{}{}",
44    "\\lx@diag@head{r}{#3}\\lx@diag@head{l}{#4}");
45});
