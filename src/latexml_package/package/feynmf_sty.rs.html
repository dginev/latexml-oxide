1//! feynmf.sty — Feynman diagrams with MetaFont
2//! Perl: feynmf.sty.ltxml — loads the raw sty with noltxml
3use crate::prelude::*;
4
5#[rustfmt::skip]
6LoadDefinitions!({
7  InputDefinitions!("feynmf", noltxml => true, extension => Some(Cow::Borrowed("sty")));
8
9  // feynmf {fmfgraph}/{fmfgraph*} environments: 2-arg-on-begin
10  // `(width,height)` followed by Feynman diagram body. Real package
11  // emits a Metafont diagram. For HTML rendering we drop the graphics
12  // body (no Metafont in our pipeline) but preserve the env so the
13  // surrounding equation/figure context still parses cleanly. Witness
14  // 2309.07343 (15 errors all from {fmfgraph*} undefined).
15  DefEnvironment!("{fmfgraph}{}{}",
16    "<ltx:note role='feynman-diagram'>(Feynman diagram, #1x#2)</ltx:note>",
17    mode => "internal_vertical");
18  DefEnvironment!("{fmfgraph*}{}{}",
19    "<ltx:note role='feynman-diagram'>(Feynman diagram, #1x#2)</ltx:note>",
20    mode => "internal_vertical");
21  // {fmffile}{name} - wraps a Feynman-diagram session. Render as no-op
22  // env (the diagrams inside are rendered by {fmfgraph}/{fmfgraph*}).
23  DefEnvironment!("{fmffile}{}", "#body", mode => "internal_vertical");
24  // \fmf, \fmfv, \fmfset, \fmflabel — diagram-content macros used
25  // inside {fmfgraph}. We don't render diagrams so absorb their args.
26  def_macro_noop("\\fmf{}{}")?;
27  def_macro_noop("\\fmfv{}{}")?;
28  def_macro_noop("\\fmfset{}{}")?;
29  def_macro_noop("\\fmflabel{}{}")?;
30});
