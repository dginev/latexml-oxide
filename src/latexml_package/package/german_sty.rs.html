1use crate::prelude::*;

3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: german.sty.ltxml
6  // This should be essentially the same, right?
7  // (considering we don't do hyphenation, etc)
8  RequirePackage!("babel", options => vec!["german".to_string()]);
9
10  // Alias the `germanb` dialect's language number to `german` (kernel/dump
11  // `\l@german`), exactly as the real germanb.ldf does via
12  // `\let\l@germanb\l@german`. Without this, `\usepackage[…,germanb]{babel}`
13  // selects `germanb` as the main language and babel's
14  // `\selectlanguage{germanb}` → `\bbl@iflanguage{germanb}` errors "You haven't
15  // defined the language 'germanb' yet" — because this binding REPLACES the raw
16  // germanb.ldf load (which is where `\l@germanb` would otherwise come from).
17  // Witness: arXiv:1010.4065 (`\usepackage[english,germanb]{babel}`).
18  RawTeX!(r"\expandafter\ifx\csname l@germanb\endcsname\relax
19    \expandafter\ifx\csname l@german\endcsname\relax
20      \expandafter\newlanguage\csname l@germanb\endcsname
21    \else
22      \expandafter\let\csname l@germanb\expandafter\endcsname\csname l@german\endcsname
23    \fi
24  \fi");
25
26  // German caption strings (from germanb.ldf). \providecommand so raw
27  // babel/germanb.ldf processing (if any) doesn't overwrite.
28  RawTeX!(r"\providecommand\captionsgerman{%
29    \def\prefacename{Vorwort}\def\refname{Literatur}%
30    \def\abstractname{Zusammenfassung}\def\bibname{Literaturverzeichnis}%
31    \def\chaptername{Kapitel}\def\appendixname{Anhang}%
32    \def\contentsname{Inhaltsverzeichnis}%
33    \def\listfigurename{Abbildungsverzeichnis}%
34    \def\listtablename{Tabellenverzeichnis}%
35    \def\indexname{Index}\def\figurename{Abbildung}%
36    \def\tablename{Tabelle}\def\partname{Teil}%
37    \def\pagename{Seite}\def\seename{siehe}%
38    \def\alsoname{siehe auch}\def\proofname{Beweis}}");
39  RawTeX!(r"\providecommand\dategerman{}");
40  RawTeX!(r"\providecommand\captionsngerman{\captionsgerman}");
41  RawTeX!(r"\providecommand\datengerman{\dategerman}");
42
43  // German " shorthand dispatch (from germanb.ldf). We replace germanb.ldf
44  // entirely via our binding dispatcher, so babel's \initiate@active@char
45  // + \declare@shorthand{german}{"a}{...} calls in germanb.ldf never fire.
46  // Simpler to implement the dispatch here as a native primitive that
47  // reads the next token and emits the umlaut/ß/guillemet directly. A
48  // future refactor could load germanb.ldf raw in parallel (the
49  // \initiate@active@char machinery does work in our engine now — verified
50  // 2026-04-17) and drop this custom primitive.
51  DefPrimitive!("\\lx@german@dq@dispatch", {
52    let tok = read_token()?;
53    let ch = tok.as_ref().map(|t| t.with_str(|s| s.to_string())).unwrap_or_default();
54    let expansion: &str = match ch.as_str() {
55      "a" => "\u{00E4}", "o" => "\u{00F6}", "u" => "\u{00FC}",
56      "e" => "\u{00EB}", "i" => "\u{00EF}",
57      "A" => "\u{00C4}", "O" => "\u{00D6}", "U" => "\u{00DC}",
58      "E" => "\u{00CB}", "I" => "\u{00CF}",
59      "s" | "z" => "\u{00DF}",
60      "S" => "SS", "Z" => "SZ",
61      "`" => "\u{201E}", "'" => "\u{201C}",
62      "<" => "\u{00AB}", ">" => "\u{00BB}",
63      "~" => "-", "=" => "-",
64      // consonants/unknowns: pass-through (below)
65      _ => "",
66    };
67    if !expansion.is_empty() {
68      unread(Tokenize!(expansion));
69    } else if !ch.is_empty()
70      && let Some(t) = tok { unread(Tokens!(t)); }
71  });
72  DefPrimitive!("\\mdqon", { assign_catcode('"', Catcode::ACTIVE, None); });
73  DefPrimitive!("\\mdqoff", { assign_catcode('"', Catcode::OTHER, None); });
74  // Faithful to babel germanb.ldf's `\initiate@active@char{"}`, which binds
75  // the active-`"` MEANING when german is LOADED — independent of catcode and
76  // of which language is the document main. Bind it here too, so that if ANY
77  // package later flips `"` to catcode-13 ACTIVE (witness 1006.0641:
78  // `\usepackage[german,english]{babel}` + a package + `fabfeynmp`, whose
79  // `{\catcode`\"=11 …}` group + babel's deferred activation leave `"` active),
80  // the active `"` always has the dispatch meaning rather than erroring
81  // "T_ACTIVE["] is not defined" on the first bare `"` in the body. The
82  // `\selectlanguage{german}` hook (babel_support_sty.rs) still (re)binds
83  // catcode+meaning together when german is actually selected.
84  if let Some(defn) = lookup_meaning(&T_CS!("\\lx@german@dq@dispatch")) {
85    assign_meaning(&T_ACTIVE!('"'), defn, Some(Scope::Global));
86  }
87  // germanb.ldf L173: `\def\dq{"}` — `\dq` yields a literal double-quote (the
88  // saved catcode-12 `"`, since `"` itself becomes the active shorthand). Map to
89  // `\textquotedbl`, LaTeXML's literal double-quote. (An earlier note recorded
90  // `\dq` as "actively undefined after this binding runs" — that was while the
91  // binding truncated past L57, before `\bbl@allowhyphens` below was added; the
92  // truncation is resolved, so this now sticks.)
93  RawTeX!(r"\providecommand\dq{\textquotedbl}");
94  // germanb.ldf helper stubs — no-op in Rust (no hyphenation / ligature phase).
95  RawTeX!(r"\providecommand\bbl@allowhyphens{}");
96  RawTeX!(r"\providecommand\bbl@ss{\ss}\providecommand\bbl@SS{SS}");
97  RawTeX!(r"\providecommand\bbl@sz{\ss}\providecommand\bbl@SZ{SZ}");
98});
