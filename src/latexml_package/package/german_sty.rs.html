1use crate::prelude::*;

3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: german.sty.ltxml
6  // This should be essentially the same, right?
7  // (considering we don't do hyphenation, etc)
8  RequirePackage!("babel", options => vec!["german".to_string()]);
9
10  // German caption strings (from germanb.ldf). \providecommand so raw
11  // babel/germanb.ldf processing (if any) doesn't overwrite.
12  RawTeX!(r"\providecommand\captionsgerman{%
13    \def\prefacename{Vorwort}\def\refname{Literatur}%
14    \def\abstractname{Zusammenfassung}\def\bibname{Literaturverzeichnis}%
15    \def\chaptername{Kapitel}\def\appendixname{Anhang}%
16    \def\contentsname{Inhaltsverzeichnis}%
17    \def\listfigurename{Abbildungsverzeichnis}%
18    \def\listtablename{Tabellenverzeichnis}%
19    \def\indexname{Index}\def\figurename{Abbildung}%
20    \def\tablename{Tabelle}\def\partname{Teil}%
21    \def\pagename{Seite}\def\seename{siehe}%
22    \def\alsoname{siehe auch}\def\proofname{Beweis}}");
23  RawTeX!(r"\providecommand\dategerman{}");
24  RawTeX!(r"\providecommand\captionsngerman{\captionsgerman}");
25  RawTeX!(r"\providecommand\datengerman{\dategerman}");
26
27  // German " shorthand dispatch (from germanb.ldf). We replace germanb.ldf
28  // entirely via our binding dispatcher, so babel's \initiate@active@char
29  // + \declare@shorthand{german}{"a}{...} calls in germanb.ldf never fire.
30  // Simpler to implement the dispatch here as a native primitive that
31  // reads the next token and emits the umlaut/ß/guillemet directly. A
32  // future refactor could load germanb.ldf raw in parallel (the
33  // \initiate@active@char machinery does work in our engine now — verified
34  // 2026-04-17) and drop this custom primitive.
35  DefPrimitive!("\\lx@german@dq@dispatch", {
36    let tok = gullet::read_token()?;
37    let ch = tok.as_ref().map(|t| t.with_str(|s| s.to_string())).unwrap_or_default();
38    let expansion: &str = match ch.as_str() {
39      "a" => "\u{00E4}", "o" => "\u{00F6}", "u" => "\u{00FC}",
40      "e" => "\u{00EB}", "i" => "\u{00EF}",
41      "A" => "\u{00C4}", "O" => "\u{00D6}", "U" => "\u{00DC}",
42      "E" => "\u{00CB}", "I" => "\u{00CF}",
43      "s" | "z" => "\u{00DF}",
44      "S" => "SS", "Z" => "SZ",
45      "`" => "\u{201E}", "'" => "\u{201C}",
46      "<" => "\u{00AB}", ">" => "\u{00BB}",
47      "~" => "-", "=" => "-",
48      // consonants/unknowns: pass-through (below)
49      _ => "",
50    };
51    if !expansion.is_empty() {
52      gullet::unread(Tokenize!(expansion));
53    } else if !ch.is_empty() {
54      if let Some(t) = tok { gullet::unread(Tokens!(t)); }
55    }
56  });
57  DefPrimitive!("\\mdqon", { state::assign_catcode('"', Catcode::ACTIVE, None); });
58  DefPrimitive!("\\mdqoff", { state::assign_catcode('"', Catcode::OTHER, None); });
59  // germanb.ldf helper stubs — no-op in Rust (no hyphenation / ligature phase).
60  RawTeX!(r"\providecommand\bbl@allowhyphens{}");
61  RawTeX!(r"\providecommand\bbl@ss{\ss}\providecommand\bbl@SS{SS}");
62  RawTeX!(r"\providecommand\bbl@sz{\ss}\providecommand\bbl@SZ{SZ}");
63});
