87  // Here's the interesting bit.
88  // Why do things the hard way, when we can pull lines out of the Mouth
89  // and match them as text ?
90  // Well, we have to dance a bit...
91  //
92  // NOTE: the part AFTER the \end{whatever}, should be lost (and message about it!)
93  DefMacro!("\\verbatim@", {
94    let env = state::lookup_string_from_sym(pin!("current_environment"));
95    // Note: This should allow a regexp, since there can be spaces between \end and { !!!
96    let mut lines = Vec::new();
97    // TODO: UGH!!! Isn't there a better way to approximate
98    // the Perl simplicity of writing an inline regex?
99    // the escaping is very easy to get wrong!
100    let env_re = Regex::new(&format!("^(.*)\\\\end\\s*\\{{{env}\\}}(.*)$")).unwrap();
101    while let Some(line) = gullet::read_raw_line() {
102      if let Some(caps) = env_re.captures(&line) {
103        let pre = caps.get(1).map_or("", |m| m.as_str()).to_string();
104        let post = caps.get(2).map_or("", |m| m.as_str()).to_string();
105        lines.push(pre);
106        if !post.is_empty() {
107          let message = s!("Characters dropped after '\\end{{{}}}'", env);
108          Info!("unexpected", "stuff", message);
109        }
110        break;
111      } else {
112        lines.push(line);
113      }
114    }
115    if lines.last() == Some(&String::new()) {
116      lines.pop();
117    }
118    let mut tokens = Vec::new();
119    for line in &lines {
120      tokens.push(T_CS!("\\verbatim@startline"));
121      tokens.extend(
122        Invocation!(T_CS!("\\verbatim@addtoline"), vec![Tokens::new(
123          ExplodeText!(line)
124        )])
125        .unlist(),
126      );
127      tokens.push(T_CS!("\\verbatim@processline"));
128    }
129    tokens.extend(Invocation!(T_CS!("\\end"), vec![T_OTHER!(env)]).unlist());
130    Ok(Tokens::new(tokens))
131  });

143            }
144          });

150            tokens.push(T_CS!("\\verbatim@processline"));
151          }
152          Ok(Tokens!(
153            T_CS!("\\begingroup"), T_CS!("\\@verbatim"),
154            T_CS!("\\frenchspacing"), T_CS!("\\@vobeyspaces"),
155            T_CS!("\\lx@verbatim@"), tokens, T_CS!("\\lx@end@verbatim@"), T_CS!("\\endgroup"))
156          )
157        },
158      )
159    } else {
160      let message = s!("\\verbatiminput found no file for {:?}, output may be incomplete", file);
161      Error!("binding", "missing_file", message);
162      Ok(Tokens!())
163    }
