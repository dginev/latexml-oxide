1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl ifplatform.sty.ltxml L36-39 reads `$^O` at binding-load time and
6  // branches on Win32/linux/darwin/cygwin. Rust equivalent uses
7  // `std::env::consts::OS` ("windows"/"linux"/"macos"/…). Previous Rust
8  // hardcoded Linux for every platform — on macOS builds `\ifmacosx` was
9  // false and `\platformname` came out as `Linux`, contrary to Perl.
10  let os = std::env::consts::OS;
11  let is_windows = os == "windows";
12  let is_linux   = os == "linux";
13  let is_macos   = os == "macos";
14  let is_cygwin  = false; // Rust has no dedicated cygwin target; matches
15                          // Perl's fall-through for uncommon OSes.
16
17  DefMacro!("\\windowsname",    "Windows");
18  DefMacro!("\\notwindowsname", "*NIX");
19  DefMacro!("\\linuxname",      "Linux");
20  DefMacro!("\\macosxname",     "Mac\\,OS\\,X");
21  DefMacro!("\\cygwinname",     "Cygwin");
22
23  // Perl L32: '\ifshellescape' hardcoded true — LaTeXML runs the macro
24  // expansion so shell-escape semantics don't apply.
25  DefConditional!("\\ifshellescape", { true });
26
27  // Perl L44-57: `\platformname` Let-aliased to whichever of
28  // \linuxname/\windowsname/\macosxname/\cygwinname matches $^O, falling
29  // back to `\unknownplatform` which carries the POSIX uname. Runtime
30  // string body → use the def_macro / let_alias function forms instead
31  // of the macro (the macro shape only takes literals).
32  let (platform_cs, unknown_name): (&str, String) = if is_windows {
33    ("\\windowsname", "[Unknown]".into())
34  } else if is_linux {
35    ("\\linuxname", "[Unknown]".into())
36  } else if is_macos {
37    ("\\macosxname", "[Unknown]".into())
38  } else if is_cygwin {
39    ("\\cygwinname", "[Unknown]".into())
40  } else {
41    ("\\unknownplatform", os.to_string())
42  };
43  def_macro(T_CS!("\\unknownplatform"), None,
44    Tokenize!(&unknown_name), None)?;
45  let_i(&T_CS!("\\platformname"), &T_CS!(platform_cs), None);
46
47  // Perl L63-66: one conditional per OS, returning the same-named sub.
48  DefConditional!("\\ifwindows", { is_windows });
49  DefConditional!("\\iflinux",   { is_linux });
50  DefConditional!("\\ifmacosx",  { is_macos });
51  DefConditional!("\\ifcygwin",  { is_cygwin });
52});
