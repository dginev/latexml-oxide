224  //
225  // Perl caption.sty.ltxml L110-115 routes through the `CAPTION_type` state
226  // value set by `\captionsetup{type=…}`: when the author has declared a
227  // float type, `\maybe@@generic@caption` expands to `\@captionof{type}`

231  DefMacro!("\\maybe@@generic@caption", sub[_args] {
232    if let Some(Stored::String(t)) = lookup_value("CAPTION_type") {
233      let ty = with(t, |s| s.to_string());
234      if !ty.is_empty() {
235        let mut out = vec![T_CS!("\\@captionof"), T_BEGIN!()];
236        out.extend(ExplodeText!(&ty));
237        out.push(T_END!());
238        return Ok(Tokens::new(out));
239      }
240    }
