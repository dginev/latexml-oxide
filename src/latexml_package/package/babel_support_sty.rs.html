75#[rustfmt::skip]
76LoadDefinitions!({
77  // Unicode quote characters (Perl L24-42)
78  //
79  // Perl: DefPrimitiveI('\ij', undef, "ij") etc. — DefPrimitive with literal
80  // string body, emits a Box. Rust uses DefMacro with string body which
81  // expands and re-tokenizes into character tokens. Both produce the same
82  // HTML output; for these simple language-shortcut/digraph CSes the DP
83  // audit flags 15 structural DefPrimitiveI↔DefMacro mismatches that are
84  // intentional — DefMacro is idiomatic Rust for plain-text CS aliases.
85  DefMacro!("\\ij", "ij");
86  DefMacro!("\\IJ", "IJ");
87
88  DefMacro!("\\flq", "\u{2039}");
89  DefMacro!("\\frq", "\u{203A}");
90  DefMacro!("\\flqq", "\u{00AB}");
91  DefMacro!("\\frqq", "\u{00BB}");
92
93  DefMacro!("\\glq", "\u{201A}");
94  DefMacro!("\\grq", "\u{2018}");
95  DefMacro!("\\glqq", "\u{201E}");
96  DefMacro!("\\grqq", "\u{201C}");
97
98  DefMacro!("\\SS", "SS");
99
100  DefMacro!("\\guilsinglleft", "\u{2039}");
101  DefMacro!("\\guilsinglright", "\u{203A}");
102  DefMacro!("\\guillemotleft", "\u{00AB}");
103  DefMacro!("\\guillemotright", "\u{00BB}");
104
105  // Shutup about hyphenation patterns (Perl L45)
106  DefMacro!("\\@nopatterns{}", "");
107
108  // Hook into \select@language, \foreign@language, \bbl@switch
109  // to set xml:lang attribute via MergeFont(language)
110  Let!("\\ltx@save@bbl@switch", "\\bbl@switch");
111  Let!("\\ltx@save@select@language", "\\select@language");
112  Let!("\\ltx@save@foreign@language", "\\foreign@language");
113
114  RawTeX!(r#"\def\select@language#1{\ltx@save@select@language{#1}\ltx@bbl@select@language{#1}}"#);
115  RawTeX!(r#"\def\foreign@language#1{\ltx@save@foreign@language{#1}\ltx@bbl@select@language{#1}}"#);
116  RawTeX!(r#"\def\bbl@switch#1{\ltx@save@bbl@switch{#1}\ltx@bbl@select@language{#1}}"#);
117
118  DefPrimitive!("\\ltx@bbl@select@language{}", sub[(language)] {
119    let lang = language.to_string();
120    let iso = babel_language_to_iso(&lang);
121    if let Some(code) = iso {
122      // Set cf@encoding to current encoding
123      def_macro(T_CS!("\\cf@encoding"), None,
124        gullet::do_expand(T_CS!("\\f@encoding"))?, None)?;
125      // Merge language into font → produces xml:lang attribute
126      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
127      // Perl: greek.ldf does \fontencoding{LGR}\selectfont in \extrasgreek
128      // and restores via \noextrasgreek. We replicate this here since our babel
129      // intercept doesn't load the real .ldf files.
130      if code == "el" {
131        load_font_map("LGR");
132        MergeFont!(encoding => "LGR");
133        // Greek accent shorthand: redefine active ~ to produce perispomeni
134        // (U+1FC0) for LGR ligature composition. In standard TeX, ~ produces
135        // tie/nobreakspace, but in Greek mode it's the circumflex accent
136        // combining character that triggers ligatures like ~a → ᾶ.
137        state::let_i(&T_CS!("\\ltx@save@greek@tilde"), &T_ACTIVE!('~'), None);
138        def_macro(T_ACTIVE!('~'), None, TokenizeInternal!("\u{1FC0}"), None)?;
139      } else {
140        // Restore non-Greek encoding: check if we're coming from LGR
141        let current_enc = lookup_font()
142          .and_then(|f| f.get_encoding().map(|e| e.to_string()))
143          .unwrap_or_else(|| "OT1".to_string());
144        if current_enc == "LGR" {
145          // Restore to OT1 (default Latin encoding) when leaving Greek
146          load_font_map("OT1");
147          MergeFont!(encoding => "OT1");
148          // Restore ~ to its pre-Greek meaning (tie/nobreakspace)
149          state::let_i(&T_ACTIVE!('~'), &T_CS!("\\ltx@save@greek@tilde"), None);
150        }
151      }
152      // French active punctuation: Perl's frenchb.ldf `\extrasfrench`
153      // hook activates `:`, `;`, `!`, `?` to emit a thin space before them;
154      // `\noextrasfrench` deactivates on language exit. We mirror that.
155      //
156      // Dispatch primitives are defined in babel_sty.rs unconditionally.
157      if code == "fr" {
158        // Entering French: flip catcodes to ACTIVE, attach dispatch meanings.
159        for &(ch, cs_name) in &[
160          (':', "\\lx@french@punct@colon"),
161          (';', "\\lx@french@punct@semi"),
162          ('!', "\\lx@french@punct@exclam"),
163          ('?', "\\lx@french@punct@question"),
164        ] {
165          if let Some(defn) = lookup_meaning(&T_CS!(cs_name)) {
166            state::assign_catcode(ch, Catcode::ACTIVE, Some(Scope::Global));
167            state::assign_meaning(&T_ACTIVE!(ch), defn, Some(Scope::Global));
168          }
169        }
170      }
171      // German: activate " as the shorthand dispatch for umlauts + opens
172      // the \mdqon / \mdqoff toggle. Babel's germanb.ldf normally does this
173      // via \initiate@active@char; we reproduce it with a direct meaning.
174      if code == "de" || code == "de-AT" {
175        if let Some(defn) = lookup_meaning(&T_CS!("\\lx@german@dq@dispatch")) {
176          state::assign_catcode('"', Catcode::ACTIVE, Some(Scope::Global));
177          state::assign_meaning(&T_ACTIVE!('"'), defn, Some(Scope::Global));
178        }
179      }
180      // Leaving French/German does not automatically deactivate the
181      // active-char meanings. The dispatch primitives (\lx@french@punct@*)
182      // check \languagename themselves and fall back to bare punctuation in
183      // non-French groups. DOCUMENT_LANGUAGE is only set once at babel-init
184      // by \lx@babel@activate@mainlang (setting per-\selectlanguage would
185      // clobber the root xml:lang when the body switches languages).
186    }
187  });
188
189  // Pretend we've got hyphenation patterns for ANY language (Perl L158-167)
190  DefMacro!("\\iflanguage{}", r#"\expandafter\ifx\csname l@#1\endcsname\relax
191  \expandafter\newlanguage\csname l@#1\endcsname\fi
192\expandafter\edef\expandafter\@@@@lang\expandafter{\csname l@#1\endcsname}
193\ifnum\csname l@#1\endcsname=\language
194  \expandafter\@firstoftwo
195\else
196  \expandafter\@secondoftwo
197\fi"#);
198});
