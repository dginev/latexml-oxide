75/// Flip the French active punctuation (`:`, `;`, `!`, `?`) to catcode ACTIVE
76/// and attach their thin-space dispatch meanings. Shared by the runtime
77/// `\selectlanguage` hook (body switches) and the `\begin{document}` deferral
78/// (preamble switches — see `\lx@bbl@begindoc@french@punct`). Real babel turns
79/// shorthands on only at `\begin{document}`; activating in the preamble
80/// corrupts packages loaded after `\selectlanguage{french}`.
81pub(crate) fn activate_french_active_punct() {
82  for &(ch, cs_name) in &[
83    (':', "\\lx@french@punct@colon"),
84    (';', "\\lx@french@punct@semi"),
85    ('!', "\\lx@french@punct@exclam"),
86    ('?', "\\lx@french@punct@question"),
87  ] {
88    if let Some(defn) = lookup_meaning(&T_CS!(cs_name)) {
89      state::assign_catcode(ch, Catcode::ACTIVE, Some(Scope::Global));
90      state::assign_meaning(&T_ACTIVE!(ch), defn, Some(Scope::Global));
91    }
92  }
93}

95#[rustfmt::skip]
96LoadDefinitions!({
97  // Many TL2025 babel language files (e.g. babel-italian italian.ldf,
98  // babel-spanish spanish.ldf) use etoolbox CSes like \ifdefstring inside
99  // hooks (e.g. `\bbl@beforestart`) and queue `\AtEndOfPackage{
100  // \RequirePackage{etoolbox}}` to satisfy them. Our `\AtEndOfPackage`
101  // machinery for raw-loaded `.ldf` chains doesn't reliably fire that
102  // queued load before babel hooks evaluate, so the test paper 0710.5177
103  // (`[english,italian]{article}`) fails with `\ifdefstring` undefined
104  // at `\bbl@beforestart`. Pre-loading etoolbox once for any babel-driven
105  // language tag closes that window. (Slight Perl divergence: Perl's
106  // `\AtEndOfPackage` chain succeeds organically.)
107  RequirePackage!("etoolbox");
108
109  // Pre-define `\bbl@languages` as an empty stub so any babel-XX.tex
110  // language file that references it at L21 (e.g. babel-german.tex,
111  // babel-italian.tex) finds something. babel.sty itself guards with
112  // `\ifx\bbl@languages\@undefined` (L266, L978) but the per-language
113  // .tex files do NOT — they assume `\bbl@languages` was set up by
114  // an earlier babel.sty load. With our `\usepackage[<lang>]{babel}`
115  // path, the language `.tex` is sometimes processed before babel.sty
116  // finishes its own setup; pre-defining here covers that load order.
117  // Mirrors the conditional definition in `nil.ldf.ltxml` (Perl L20-21
118  // / Rust nil_ldf.rs:6-8) but applied at the `babel_support` layer
119  // so it's always in scope before any lang.tex loads.
120  if !IsDefined!(&T_CS!("\\bbl@languages")) {
121    def_macro_noop("\\bbl@languages")?;
122  }
123
124  // Unicode quote characters (Perl L24-42)
125  //
126  // Perl: DefPrimitiveI('\ij', undef, "ij") etc. — DefPrimitive with literal
127  // string body, emits a Box. Rust uses DefMacro with string body which
128  // expands and re-tokenizes into character tokens. Both produce the same
129  // HTML output; for these simple language-shortcut/digraph CSes the DP
130  // audit flags 15 structural DefPrimitiveI↔DefMacro mismatches that are
131  // intentional — DefMacro is idiomatic Rust for plain-text CS aliases.
132  DefMacro!("\\ij", "ij");
133  DefMacro!("\\IJ", "IJ");
134
135  DefMacro!("\\flq", "\u{2039}");
136  DefMacro!("\\frq", "\u{203A}");
137  DefMacro!("\\flqq", "\u{00AB}");
138  DefMacro!("\\frqq", "\u{00BB}");
139
140  DefMacro!("\\glq", "\u{201A}");
141  DefMacro!("\\grq", "\u{2018}");
142  DefMacro!("\\glqq", "\u{201E}");
143  DefMacro!("\\grqq", "\u{201C}");
144
145  DefMacro!("\\SS", "SS");
146
147  DefMacro!("\\guilsinglleft", "\u{2039}");
148  DefMacro!("\\guilsinglright", "\u{203A}");
149  DefMacro!("\\guillemotleft", "\u{00AB}");
150  DefMacro!("\\guillemotright", "\u{00BB}");
151
152  // Shutup about hyphenation patterns (Perl L45)
153  def_macro_noop("\\@nopatterns{}")?;
154
155  // Hook into \select@language, \foreign@language, \bbl@switch
156  // to set xml:lang attribute via MergeFont(language)
157  Let!("\\ltx@save@bbl@switch", "\\bbl@switch");
158  Let!("\\ltx@save@select@language", "\\select@language");
159  Let!("\\ltx@save@foreign@language", "\\foreign@language");
160
161  RawTeX!(r#"\def\select@language#1{\ltx@save@select@language{#1}\ltx@bbl@select@language{#1}}"#);
162  RawTeX!(r#"\def\foreign@language#1{\ltx@save@foreign@language{#1}\ltx@bbl@select@language{#1}}"#);
163  RawTeX!(r#"\def\bbl@switch#1{\ltx@save@bbl@switch{#1}\ltx@bbl@select@language{#1}}"#);
164
165  DefPrimitive!("\\ltx@bbl@select@language{}", sub[(language)] {
166    let lang = language.to_string();
167    let iso = babel_language_to_iso(&lang);
168    if let Some(code) = iso {
169      // Set cf@encoding to current encoding
170      def_macro(T_CS!("\\cf@encoding"), None,
171        gullet::do_expand(T_CS!("\\f@encoding"))?, None)?;
172      // Merge language into font → produces xml:lang attribute
173      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
174      // Perl: greek.ldf does \fontencoding{LGR}\selectfont in \extrasgreek
175      // and restores via \noextrasgreek. We replicate this here since our babel
176      // intercept doesn't load the real .ldf files.
177      if code == "el" {
178        load_font_map("LGR");
179        MergeFont!(encoding => "LGR");
180        // Greek accent shorthand: redefine active ~ to produce perispomeni
181        // (U+1FC0) for LGR ligature composition. In standard TeX, ~ produces
182        // tie/nobreakspace, but in Greek mode it's the circumflex accent
183        // combining character that triggers ligatures like ~a → ᾶ.
184        state::let_i(&T_CS!("\\ltx@save@greek@tilde"), &T_ACTIVE!('~'), None);
185        def_macro(T_ACTIVE!('~'), None, TokenizeInternal!("\u{1FC0}"), None)?;
186      } else {
187        // Restore non-Greek encoding: check if we're coming from LGR
188        let current_enc = lookup_font()
189          .and_then(|f| f.get_encoding().map(|e| e.to_string()))
190          .unwrap_or_else(|| "OT1".to_string());
191        if current_enc == "LGR" {
192          // Restore to OT1 (default Latin encoding) when leaving Greek
193          load_font_map("OT1");
194          MergeFont!(encoding => "OT1");
195          // Restore ~ to its pre-Greek meaning (tie/nobreakspace)
196          state::let_i(&T_ACTIVE!('~'), &T_CS!("\\ltx@save@greek@tilde"), None);
197        }
198      }
199      // French active punctuation: Perl's frenchb.ldf `\extrasfrench`
200      // hook activates `:`, `;`, `!`, `?` to emit a thin space before them;
201      // `\noextrasfrench` deactivates on language exit. We mirror that.
202      //
203      // Dispatch primitives are defined in babel_sty.rs unconditionally.
204      // Real babel defers shorthand activation to `\begin{document}`:
205      // flipping `:;!?` to active while still in the PREAMBLE corrupts any
206      // package loaded after `\selectlanguage{french}` — e.g. adjustbox,
207      // whose graphicx `!` natural-size sentinel then tokenizes as an active
208      // char and the glue parser hits it: "Missing close parenthesis in Glue
209      // expr. Got T_ACTIVE[!]". Skip activation in the preamble; babel
210      // re-fires `\selectlanguage{\bbl@main@language}` at `\begin{document}`
211      // (via its AtBeginDocument hook), running this hook again with
212      // `inPreamble` cleared, which performs the activation then. A body-level
213      // `\selectlanguage` switch (inPreamble already false) still activates
214      // immediately. Witness 1712.07003 (`\selectlanguage{french}` in the
215      // preamble, then `\usepackage{adjustbox}`).
216      if code == "fr" && !lookup_bool("inPreamble") {
217        // Entering French in the document body: activate immediately. Preamble
218        // switches are deferred to \begin{document} by the begin-document hook
219        // below (so packages loaded after \selectlanguage see catcode-12 `!`).
220        activate_french_active_punct();
221      }
222      // German: activate " as the shorthand dispatch for umlauts + opens
223      // the \mdqon / \mdqoff toggle. Babel's germanb.ldf normally does this
224      // via \initiate@active@char; we reproduce it with a direct meaning.
225      if code == "de" || code == "de-AT" {
226        if let Some(defn) = lookup_meaning(&T_CS!("\\lx@german@dq@dispatch")) {
227          state::assign_catcode('"', Catcode::ACTIVE, Some(Scope::Global));
228          state::assign_meaning(&T_ACTIVE!('"'), defn, Some(Scope::Global));
229        }
230      }
231      // Leaving French/German does not automatically deactivate the
232      // active-char meanings. The dispatch primitives (\lx@french@punct@*)
233      // check \languagename themselves and fall back to bare punctuation in
234      // non-French groups. DOCUMENT_LANGUAGE is only set once at babel-init
235      // by \lx@babel@activate@mainlang (setting per-\selectlanguage would
236      // clobber the root xml:lang when the body switches languages).
237    }
238  });
239
240  // Deferred French-shorthand activation. The `\selectlanguage` hook above
241  // skips activation while `inPreamble`, so `\selectlanguage{french}` followed
242  // by `\usepackage{adjustbox}` (or any package whose macros embed a literal
243  // `!`, e.g. graphicx's natural-size sentinel) loads with `!` still catcode-12
244  // — matching real babel, which turns shorthands on at `\begin{document}`.
245  // Re-run the activation here if the resolved document language is French.
246  // Witness 1712.07003. (Body-level `\selectlanguage` switches happen after
247  // `inPreamble` is cleared and activate immediately in the hook above.)
248  DefPrimitive!("\\lx@bbl@begindoc@french@punct", {
249    if lookup_string("DOCUMENT_LANGUAGE") == "fr" {
250      activate_french_active_punct();
251    }
252  });
253  RawTeX!(r"\AtBeginDocument{\lx@bbl@begindoc@french@punct}");
254
255  // Pretend we've got hyphenation patterns for ANY language (Perl L158-167)
256  DefMacro!("\\iflanguage{}", r#"\expandafter\ifx\csname l@#1\endcsname\relax
257  \expandafter\newlanguage\csname l@#1\endcsname\fi
258\expandafter\edef\expandafter\@@@@lang\expandafter{\csname l@#1\endcsname}
259\ifnum\csname l@#1\endcsname=\language
260  \expandafter\@firstoftwo
261\else
262  \expandafter\@secondoftwo
263\fi"#);
264});
