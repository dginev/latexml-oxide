1//! babel.sty — multilingual support
2//!
3//! Perl: babel.sty.ltxml (30 lines) — `InputDefinitions('babel', noltxml=>1)`.
4//! Our Rust port carries a thin orchestration layer on top of the raw babel
5//! load. With the @currname leakage fix in commit 56b0c35d2, babel's own
6//! option pipeline (and therefore its entire language-loading / shorthand /
7//! captions story) now works end-to-end. Only two small workarounds remain
8//! here: pre-allocating `\l@polutonikogreek` for older TeX Live builds that
9//! don't include it in the kernel dump, and setting DOCUMENT_LANGUAGE +
10//! `\bbl@main@language` globally (babel's own raw-load path may resolve
11//! main to a language whose .ldf happened to run last — not always the
12//! user's intended last option).
13use crate::prelude::*;
14
15#[rustfmt::skip]
16LoadDefinitions!({
17  // \l@polutonikogreek: allocate if not present in kernel dump (newer TeXLive
18  // includes it, older may not).
19  RawTeX!(r"\expandafter\ifx\csname l@polutonikogreek\endcsname\relax\newlanguage\l@polutonikogreek\fi");
20
21  // `activeacute` was historically an option to babel-spanish.ldf that
22  // activated `'` as an active accent. Some 1995-2010 papers wrote
23  // `\usepackage[<lang>,activeacute]{babel}` treating it as a language;
24  // modern babel doesn't recognize it as a language and `\InputIfFileExists
25  // {activeacute.ldf}` silently fails (no on-disk file in TL). Babel then
26  // proceeds and `\selectlanguage{...}` errors with "haven't defined the
27  // language 'activeacute' yet". We pre-register `\l@activeacute` plus the
28  // empty `<lang>` hooks so `\selectlanguage` resolves silently — actual
29  // active-acute-on-quote semantics are not reproduced (most affected
30  // papers only set this option as a side effect of preamble copy-paste).
31  RawTeX!(r"%
32    \expandafter\ifx\csname l@activeacute\endcsname\relax
33      \newlanguage\l@activeacute
34    \fi
35    \providecommand\captionsactiveacute{}%
36    \providecommand\extrasactiveacute{}%
37    \providecommand\noextrasactiveacute{}%
38    \providecommand\dateactiveacute{}");
39
40  // \bbl@opt@safe = \@empty inhibits some risky redefinitions in babel.
41  // Mirror Perl LaTeXML/lib/LaTeXML/Package/babel.def.ltxml: `Let('\bbl@opt@safe', '\@empty')`.
42  // Without this, babel.sty's option processing enters an infinite loop on
43  // some redefinition paths (verified: triggers token_limit:Timeout 100M).
44  RawTeX!(r"\let\bbl@opt@safe\@empty");
45
46  // Pre-define `\bbl@main@language` as `english` so papers that load babel
47  // transitively (e.g. via aastex62 → revtex4 → bibstyles → babel chain
48  // without any explicit `\usepackage[<lang>]{babel}`) don't error with
49  // "Token \bbl@main@language is not defined" when `\selectlanguage` or
50  // `\lx@babel@activate@mainlang` later expands it. The real `\ldf@finish`
51  // overrides this when a language .ldf actually runs. Witness 2301.13322
52  // (aastex62 + lipsum + blindtext, no explicit babel options).
53  RawTeX!(r"\providecommand\bbl@main@language{english}");
54
55  InputDefinitions!("babel", noltxml => true, extension => Some(Cow::Borrowed("sty")));
56
57  // Sets DOCUMENT_LANGUAGE and force-sets \bbl@main@language from
58  // \opt@babel.sty so babel's `\AtBeginDocument{\selectlanguage{\bbl@main
59  // @language}}` picks up the user's intended main language (the last
60  // option), not whichever .ldf's \ldf@finish happened to run last.
61  // Everything else (captions activation, active-char shorthands, port
62  // dispatching) is handled end-to-end by babel's own chain.
63  DefPrimitive!("\\lx@babel@activate@mainlang", {
64    let main = gullet::do_expand(T_CS!("\\bbl@main@language"))
65      .map(|t| t.to_string()).unwrap_or_default();
66    let opt_babel = gullet::do_expand(Tokenize!(r"\csname opt@babel.sty\endcsname"))
67      .map(|t| t.to_string()).unwrap_or_default();
68    // Modern babel accepts `main=<lang>` to pin the document main language
69    // (driver: 2109.00402 \usepackage[main=english]{babel}). Two cases:
70    //   - `main=<lang>`  → use <lang> directly (highest priority)
71    //   - bare positional option (no `=`) → treat as a language candidate
72    //   - `<key>=<value>` for any other key (e.g. `shorthands=off`,
73    //     `provide=*`) → SKIP, it's an option not a language. Driver:
74    //     2001.00747 `\usepackage[english, shorthands=off]{babel}` was
75    //     selecting "off" as the language because we promoted any value-
76    //     half of any key=value option.
77    let main_kv = opt_babel.split(',')
78      .map(str::trim)
79      .find_map(|s| s.strip_prefix("main=").map(str::trim).map(str::to_string));
80    // Filter: drop key=value options AND babel-language MODIFIERS (sub-options
81    // recognised by a single .ldf rather than a language by themselves).
82    // Modifiers we know about:
83    //   `es-*` (babel-spanish: es-tabla, es-cuadro, es-noindentfirst, …)
84    // Driver: 2102.11084 `\usepackage[spanish, es-tabla]{babel}` selected
85    // "es-tabla" as the language → "haven't defined the language 'es-tabla'"
86    // GenericError.
87    let is_lang_candidate = |s: &str| -> bool {
88      !s.is_empty() && s != "nil" && !s.contains('=')
89        && !s.starts_with("es-")
90    };
91    let pkg_last = main_kv.clone().unwrap_or_else(|| {
92      opt_babel.split(',').map(str::trim).rfind(|s| is_lang_candidate(s)).unwrap_or_default().to_string()
93    });
94    // Pick the active main-lang. The non-trivial case is when the user's
95    // OPTION NAME differs from the .ldf's CANONICAL language name —
96    // e.g. `\usepackage[russianb]{babel}` loads russianb.ldf which calls
97    // `\ldf@finish{russian}`. babel's `\select@language` later needs
98    // `\l@<canonical>` (russian), so we must use the canonical name
99    // when an alias is in play. But when our Rust binding for a `.ldf`
100    // (e.g. french_ldf) bypasses raw load, no `\main@language` runs,
101    // and `\bbl@main@language` retains whatever the FIRST option's
102    // raw-loaded .ldf set — which is wrong (user wants the LAST option).
103    //
104    // Heuristic: trust `main` (the `\bbl@main@language` value) only
105    // when it maps to the same ISO code as `pkg_last`. That captures
106    // the canonical-alias case (russianb → russian, ukrainianb →
107    // ukrainian, brazilian → brazil) while still falling through to
108    // `pkg_last` for the no-Rust-ldf-binding case (where `main`
109    // points at the first option, not the user's intended last one).
110    //
111    // Witnesses: 2312.08012 (russianb → russian alias, needed canonical),
112    // tests/babel/elsart_keyword_brace_form (`[english,french]` →
113    // french_ldf binding doesn't update main, but user wants french).
114    let same_alias_class = !main.is_empty() && !pkg_last.is_empty()
115      && main != pkg_last
116      && crate::package::babel_support_sty::babel_language_to_iso(&main)
117         == crate::package::babel_support_sty::babel_language_to_iso(&pkg_last)
118      && crate::package::babel_support_sty::babel_language_to_iso(&main).is_some();
119    let lang = if let Some(m) = main_kv {
120      m
121    } else if same_alias_class {
122      // canonical aliases — use the canonical name from \ldf@finish
123      main
124    } else if !pkg_last.is_empty() {
125      pkg_last
126    } else if main != "nil" && !main.is_empty() {
127      main
128    } else {
129      return Ok(vec![]);
130    };
131    let iso = crate::package::babel_support_sty::babel_language_to_iso(&lang);
132    if let Some(code) = iso {
133      state::assign_value("DOCUMENT_LANGUAGE",
134        Stored::from(code.to_string()), Some(Scope::Global));
135      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
136    }
137    // Set \bbl@main@language so babel's AtBeginDocument
138    // \selectlanguage{\bbl@main@language} picks up the canonical name.
139    def_macro(T_CS!("\\bbl@main@language"), None,
140      Tokens!(Explode!(lang.clone())),
141      Some(ExpandableOptions { scope: Some(Scope::Global), ..ExpandableOptions::default() }))?;
142  });
143  // Run mainlang at load time so DOCUMENT_LANGUAGE is set before
144  // \begin{document} opens (and base_schema's after_open reads it).
145  RawTeX!(r"\lx@babel@activate@mainlang");
146
147});
