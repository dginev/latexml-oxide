1//! babel.sty — multilingual support
2//!
3//! Perl: babel.sty.ltxml (30 lines) — `InputDefinitions('babel', noltxml=>1)`.
4//! Our Rust port carries a thin orchestration layer on top of the raw babel
5//! load. With the @currname leakage fix in commit 56b0c35d2, babel's own
6//! option pipeline (and therefore its entire language-loading / shorthand /
7//! captions story) now works end-to-end. Only two small workarounds remain
8//! here: pre-allocating `\l@polutonikogreek` for older TeX Live builds that
9//! don't include it in the kernel dump, and setting DOCUMENT_LANGUAGE +
10//! `\bbl@main@language` globally (babel's own raw-load path may resolve
11//! main to a language whose .ldf happened to run last — not always the
12//! user's intended last option).
13use crate::prelude::*;
14
15#[rustfmt::skip]
16LoadDefinitions!({
17  // \l@polutonikogreek: allocate if not present in kernel dump (newer TeXLive
18  // includes it, older may not).
19  RawTeX!(r"\expandafter\ifx\csname l@polutonikogreek\endcsname\relax\newlanguage\l@polutonikogreek\fi");
20
21  // `activeacute` was historically an option to babel-spanish.ldf that
22  // activated `'` as an active accent. Some 1995-2010 papers wrote
23  // `\usepackage[<lang>,activeacute]{babel}` treating it as a language;
24  // modern babel doesn't recognize it as a language and `\InputIfFileExists
25  // {activeacute.ldf}` silently fails (no on-disk file in TL). Babel then
26  // proceeds and `\selectlanguage{...}` errors with "haven't defined the
27  // language 'activeacute' yet". We pre-register `\l@activeacute` plus the
28  // empty `<lang>` hooks so `\selectlanguage` resolves silently — actual
29  // active-acute-on-quote semantics are not reproduced (most affected
30  // papers only set this option as a side effect of preamble copy-paste).
31  RawTeX!(r"%
32    \expandafter\ifx\csname l@activeacute\endcsname\relax
33      \newlanguage\l@activeacute
34    \fi
35    \providecommand\captionsactiveacute{}%
36    \providecommand\extrasactiveacute{}%
37    \providecommand\noextrasactiveacute{}%
38    \providecommand\dateactiveacute{}");
39
40  // \bbl@opt@safe = \@empty inhibits some risky redefinitions in babel.
41  // Mirror Perl LaTeXML/lib/LaTeXML/Package/babel.def.ltxml: `Let('\bbl@opt@safe', '\@empty')`.
42  // Without this, babel.sty's option processing enters an infinite loop on
43  // some redefinition paths (verified: triggers token_limit:Timeout 100M).
44  RawTeX!(r"\let\bbl@opt@safe\@empty");
45
46  // Pre-define `\bbl@main@language` as `english` so papers that load babel
47  // transitively (e.g. via aastex62 → revtex4 → bibstyles → babel chain
48  // without any explicit `\usepackage[<lang>]{babel}`) don't error with
49  // "Token \bbl@main@language is not defined" when `\selectlanguage` or
50  // `\lx@babel@activate@mainlang` later expands it. The real `\ldf@finish`
51  // overrides this when a language .ldf actually runs. Witness 2301.13322
52  // (aastex62 + lipsum + blindtext, no explicit babel options).
53  RawTeX!(r"\providecommand\bbl@main@language{english}");
54
55  InputDefinitions!("babel", noltxml => true, extension => Some(Cow::Borrowed("sty")));
56
57  // Sets DOCUMENT_LANGUAGE and force-sets \bbl@main@language from
58  // \opt@babel.sty so babel's `\AtBeginDocument{\selectlanguage{\bbl@main
59  // @language}}` picks up the user's intended main language (the last
60  // option), not whichever .ldf's \ldf@finish happened to run last.
61  // Everything else (captions activation, active-char shorthands, port
62  // dispatching) is handled end-to-end by babel's own chain.
63  DefPrimitive!("\\lx@babel@activate@mainlang", {
64    let main = do_expand(T_CS!("\\bbl@main@language"))
65      .map(|t| t.to_string()).unwrap_or_default();
66    let opt_babel = do_expand(Tokenize!(r"\csname opt@babel.sty\endcsname"))
67      .map(|t| t.to_string()).unwrap_or_default();
68    // Modern babel accepts `main=<lang>` to pin the document main language
69    // (driver: 2109.00402 \usepackage[main=english]{babel}). Two cases:
70    //   - `main=<lang>`  → use <lang> directly (highest priority)
71    //   - bare positional option (no `=`) → treat as a language candidate
72    //   - `<key>=<value>` for any other key (e.g. `shorthands=off`,
73    //     `provide=*`) → SKIP, it's an option not a language. Driver:
74    //     2001.00747 `\usepackage[english, shorthands=off]{babel}` was
75    //     selecting "off" as the language because we promoted any value-
76    //     half of any key=value option.
77    let main_kv = opt_babel.split(',')
78      .map(str::trim)
79      .find_map(|s| s.strip_prefix("main=").map(str::trim).map(str::to_string));
80    // Filter: drop key=value options AND babel-language MODIFIERS (sub-options
81    // recognised by a single .ldf rather than a language by themselves).
82    // Modifiers we know about:
83    //   `es-*` (babel-spanish: es-tabla, es-cuadro, es-noindentfirst, …)
84    // Driver: 2102.11084 `\usepackage[spanish, es-tabla]{babel}` selected
85    // "es-tabla" as the language → "haven't defined the language 'es-tabla'"
86    // GenericError.
87    // ALSO drop babel's own bare KEYWORD options — `\DeclareOption{<kw>}` in
88    // babel.sty that are package switches, NOT languages (so babel consumes
89    // them via their `\ds@` handler and never treats them as a language to
90    // load/select). Without this, `\usepackage[english,strings]{babel}`
91    // selected the LAST bare option "strings" as the main language →
92    // `\selectlanguage{strings}` → "You haven't defined the language
93    // 'strings'". babel.sty L296/L336-379 (the no-`=` ones). Driver
94    // 2006.10240 (`[english,strings]`).
95    const BABEL_KEYWORD_OPTS: &[&str] = &[
96      "base", "showlanguages", "KeepShorthandsActive", "activeacute",
97      "activegrave", "debug", "noconfigs", "silent", "strings", "nocase",
98      "leqno", "fleqn",
99    ];
100    let is_lang_candidate = |s: &str| -> bool {
101      !s.is_empty() && s != "nil" && !s.contains('=')
102        && !s.starts_with("es-")
103        && !BABEL_KEYWORD_OPTS.contains(&s)
104    };
105    let pkg_last = main_kv.clone().unwrap_or_else(|| {
106      opt_babel.split(',').map(str::trim).rfind(|s| is_lang_candidate(s)).unwrap_or_default().to_string()
107    });
108    // Pick the active main-lang. The non-trivial case is when the user's
109    // OPTION NAME differs from the .ldf's CANONICAL language name —
110    // e.g. `\usepackage[russianb]{babel}` loads russianb.ldf which calls
111    // `\ldf@finish{russian}`. babel's `\select@language` later needs
112    // `\l@<canonical>` (russian), so we must use the canonical name
113    // when an alias is in play. But when our Rust binding for a `.ldf`
114    // (e.g. french_ldf) bypasses raw load, no `\main@language` runs,
115    // and `\bbl@main@language` retains whatever the FIRST option's
116    // raw-loaded .ldf set — which is wrong (user wants the LAST option).
117    //
118    // Heuristic: trust `main` (the `\bbl@main@language` value) only
119    // when it maps to the same ISO code as `pkg_last`. That captures
120    // the canonical-alias case (russianb → russian, ukrainianb →
121    // ukrainian, brazilian → brazil) while still falling through to
122    // `pkg_last` for the no-Rust-ldf-binding case (where `main`
123    // points at the first option, not the user's intended last one).
124    //
125    // Witnesses: 2312.08012 (russianb → russian alias, needed canonical),
126    // tests/babel/elsart_keyword_brace_form (`[english,french]` →
127    // french_ldf binding doesn't update main, but user wants french).
128    let same_alias_class = !main.is_empty() && !pkg_last.is_empty()
129      && main != pkg_last
130      && babel_support_sty::babel_language_to_iso(&main)
131         == babel_support_sty::babel_language_to_iso(&pkg_last)
132      && babel_support_sty::babel_language_to_iso(&main).is_some();
133    let lang = if let Some(m) = main_kv {
134      m
135    } else if same_alias_class {
136      // canonical aliases — use the canonical name from \ldf@finish
137      main
138    } else if !pkg_last.is_empty() {
139      pkg_last
140    } else if main != "nil" && !main.is_empty() {
141      main
142    } else {
143      return Ok(vec![]);
144    };
145    let iso = babel_support_sty::babel_language_to_iso(&lang);
146    if let Some(code) = iso {
147      assign_value("DOCUMENT_LANGUAGE",
148        Stored::from(code.to_string()), Some(Scope::Global));
149      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
150    }
151    // Set \bbl@main@language so babel's AtBeginDocument
152    // \selectlanguage{\bbl@main@language} picks up the canonical name.
153    def_macro(T_CS!("\\bbl@main@language"), None,
154      Tokens!(Explode!(lang.clone())),
155      Some(ExpandableOptions { scope: Some(Scope::Global), ..ExpandableOptions::default() }))?;
156  });
157  // Run mainlang at load time so DOCUMENT_LANGUAGE is set before
158  // \begin{document} opens (and base_schema's after_open reads it).
159  RawTeX!(r"\lx@babel@activate@mainlang");
160
161  // English-family caption/date/extras hook backfill. Modern babel's .ini
162  // path defines the per-variant `\captions<v>`/`\date<v>`/`\extras<v>`
163  // hooks only for the variant(s) whose .ini actually loaded (e.g.
164  // babel-british for `[british]`). A paper listing SEVERAL english
165  // variants — `\usepackage[british,USenglish]{babel}` — then invokes a
166  // variant whose .ini never ran (`\dateUSenglish`) or the `\captionsenglish`
167  // base, erroring at `\selectlanguage`/option dispatch. The classic .ldf
168  // loaders (`babel_lang_stubs::load_*`) also miss this because the .ini
169  // path bypasses them, and `english.sty`'s aliasing loop only runs for
170  // `\usepackage{english}`, not a direct `\usepackage[...]{babel}`. Backfill
171  // with `\@ifundefined` guards: never overrides a real definition; captions
172  // stay English (our HTML default); `\date<v>` aliases to `\dateenglish`
173  // (keeps `\today` faithful). Runs after babel's own option processing
174  // (InputDefinitions above), before the invocation in later preamble.
175  // Witness arXiv:1508.06150 (`\usepackage[british,USenglish]{babel}`); Perl rc=0.
176  // NB: no \makeatletter/\makeatother — RawTeX already digests with `@` as a
177  // letter; emitting \makeatother here would leave `@` catcode-12 globally and
178  // break babel's later `\l@<lang>`/`\bbl@…` parsing (manifested as a spurious
179  // "haven't defined the language" error).
180  RawTeX!(r"%
181    \@ifundefined{dateenglish}{\@namedef{dateenglish}{}}{}%
182    \@for\lx@bbl@engtmp:={english,USenglish,UKenglish,american,british,canadian,australian,newzealand}\do{%
183      \@ifundefined{captions\lx@bbl@engtmp}{\expandafter\let\csname captions\lx@bbl@engtmp\endcsname\@empty}{}%
184      \@ifundefined{extras\lx@bbl@engtmp}{\expandafter\let\csname extras\lx@bbl@engtmp\endcsname\@empty}{}%
185      \@ifundefined{noextras\lx@bbl@engtmp}{\expandafter\let\csname noextras\lx@bbl@engtmp\endcsname\@empty}{}%
186      \@ifundefined{date\lx@bbl@engtmp}{\expandafter\let\csname date\lx@bbl@engtmp\endcsname\dateenglish}{}%
187      \@ifundefined{l@\lx@bbl@engtmp}{\expandafter\let\csname l@\lx@bbl@engtmp\endcsname\l@english}{}}");
188  // ^ Also alias the hyphenation language register `\l@<variant>` to
189  // `\l@english` when undefined. The `.ini` path defines `\l@british` only for
190  // the variant whose `.ini` actually ran; a paper that loads several english
191  // variants then does `\selectlanguage{british}` → `\bbl@iflanguage{british}`
192  // tests `\ifx\csname l@british\endcsname\relax` and, finding it relax, errors
193  // "You haven't defined the language 'british' yet". english.ldf normally
194  // `\let`s every variant's `\l@` to `\l@english` (british uses English
195  // hyphenation); backfill the same. Witness 1508.06150.
196
197  // `\shorthandoff{<chars>}` / `\shorthandon{<chars>}`: babel shorthand
198  // toggles. Babel's raw implementation (babel.sty L1492-1496) iterates the
199  // argument switching each active-char shorthand, but ALSO fires
200  // `\PackageError{babel}{I can't switch '<c>' on or off--not a shorthand}`
201  // for any char that isn't a registered shorthand. Our language stubs
202  // (`install_lang_stub`) don't install the full shorthand machinery, so the
203  // raw impl errored on every call — which is why this was a no-op.
204  //
205  // But a pure no-op is WRONG: the *observable* effect of `\shorthandoff` is
206  // the CATCODE change (active → other), which genuinely affects tokenization,
207  // not just typesetting. French babel makes `;:!?` active; leaving them
208  // active when an author explicitly turned them off breaks downstream parsers
209  // that expect the literal chars — e.g. xy-pic `\xymatrix @!=8mm{…}` reads
210  // `@!=` expecting `!` at catcode 12; an active `!` derails the xymatrix so
211  // its `^`/`_` reach the math handler in text mode → 100+ `^`/`_` "can only
212  // appear in math mode" errors (witness arXiv:1804.10128, French + xy-pic).
213  //
214  // So apply the catcode change (the part our pipeline observes) and skip the
215  // error (the part the stub gap can't satisfy): `\shorthandoff` → OTHER(12),
216  // `\shorthandon` → ACTIVE(13). Local to the current group, mirroring TeX's
217  // local `\catcode`. Still correct for the defensive-call witness
218  // arXiv:1912.08056 (`\def\diag{\shorthandoff{;:!?}…}`) — catcode 12 is the
219  // intended effect, no error.
220  fn shorthand_set_catcode(chars: Tokens, cc: Catcode) {
221    for tok in chars.unlist() {
222      if tok.get_catcode() == Catcode::SPACE {
223        continue;
224      }
225      let single = tok.with_str(|s| {
226        let mut it = s.chars();
227        match (it.next(), it.next()) {
228          (Some(c), None) => Some(c),
229          _ => None,
230        }
231      });
232      if let Some(c) = single {
233        assign_catcode(c, cc, Some(Scope::Local));
234      }
235    }
236  }
237  DefPrimitive!("\\shorthandoff{}", sub[(chars)] {
238    shorthand_set_catcode(chars, Catcode::OTHER);
239  });
240  DefPrimitive!("\\shorthandon{}", sub[(chars)] {
241    shorthand_set_catcode(chars, Catcode::ACTIVE);
242  });
