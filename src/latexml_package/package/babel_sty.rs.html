1//! babel.sty — multilingual support
2//!
3//! Perl: babel.sty.ltxml (30 lines) — `InputDefinitions('babel', noltxml=>1)`.
4//! Our Rust port carries a thin orchestration layer on top of the raw babel
5//! load. With the @currname leakage fix in commit 56b0c35d2, babel's own
6//! option pipeline (and therefore its entire language-loading / shorthand /
7//! captions story) now works end-to-end. Only two small workarounds remain
8//! here: pre-allocating `\l@polutonikogreek` for older TeX Live builds that
9//! don't include it in the kernel dump, and setting DOCUMENT_LANGUAGE +
10//! `\bbl@main@language` globally (babel's own raw-load path may resolve
11//! main to a language whose .ldf happened to run last — not always the
12//! user's intended last option).
13use crate::prelude::*;
14
15#[rustfmt::skip]
16LoadDefinitions!({
17  // \l@polutonikogreek: allocate if not present in kernel dump (newer TeXLive
18  // includes it, older may not).
19  RawTeX!(r"\expandafter\ifx\csname l@polutonikogreek\endcsname\relax\newlanguage\l@polutonikogreek\fi");
20
21  // `activeacute` was historically an option to babel-spanish.ldf that
22  // activated `'` as an active accent. Some 1995-2010 papers wrote
23  // `\usepackage[<lang>,activeacute]{babel}` treating it as a language;
24  // modern babel doesn't recognize it as a language and `\InputIfFileExists
25  // {activeacute.ldf}` silently fails (no on-disk file in TL). Babel then
26  // proceeds and `\selectlanguage{...}` errors with "haven't defined the
27  // language 'activeacute' yet". We pre-register `\l@activeacute` plus the
28  // empty `<lang>` hooks so `\selectlanguage` resolves silently — actual
29  // active-acute-on-quote semantics are not reproduced (most affected
30  // papers only set this option as a side effect of preamble copy-paste).
31  RawTeX!(r"%
32    \expandafter\ifx\csname l@activeacute\endcsname\relax
33      \newlanguage\l@activeacute
34    \fi
35    \providecommand\captionsactiveacute{}%
36    \providecommand\extrasactiveacute{}%
37    \providecommand\noextrasactiveacute{}%
38    \providecommand\dateactiveacute{}");
39
40  // \bbl@opt@safe = \@empty inhibits some risky redefinitions in babel.
41  // Mirror Perl LaTeXML/lib/LaTeXML/Package/babel.def.ltxml: `Let('\bbl@opt@safe', '\@empty')`.
42  // Without this, babel.sty's option processing enters an infinite loop on
43  // some redefinition paths (verified: triggers token_limit:Timeout 100M).
44  RawTeX!(r"\let\bbl@opt@safe\@empty");
45
46  InputDefinitions!("babel", noltxml => true, extension => Some(Cow::Borrowed("sty")));
47
48  // Sets DOCUMENT_LANGUAGE and force-sets \bbl@main@language from
49  // \opt@babel.sty so babel's `\AtBeginDocument{\selectlanguage{\bbl@main
50  // @language}}` picks up the user's intended main language (the last
51  // option), not whichever .ldf's \ldf@finish happened to run last.
52  // Everything else (captions activation, active-char shorthands, port
53  // dispatching) is handled end-to-end by babel's own chain.
54  DefPrimitive!("\\lx@babel@activate@mainlang", {
55    let main = gullet::do_expand(T_CS!("\\bbl@main@language"))
56      .map(|t| t.to_string()).unwrap_or_default();
57    let opt_babel = gullet::do_expand(Tokenize!(r"\csname opt@babel.sty\endcsname"))
58      .map(|t| t.to_string()).unwrap_or_default();
59    // Modern babel accepts `main=<lang>` to pin the document main language
60    // (driver: 2109.00402 \usepackage[main=english]{babel}). Two cases:
61    //   - `main=<lang>`  → use <lang> directly (highest priority)
62    //   - bare positional option (no `=`) → treat as a language candidate
63    //   - `<key>=<value>` for any other key (e.g. `shorthands=off`,
64    //     `provide=*`) → SKIP, it's an option not a language. Driver:
65    //     2001.00747 `\usepackage[english, shorthands=off]{babel}` was
66    //     selecting "off" as the language because we promoted any value-
67    //     half of any key=value option.
68    let main_kv = opt_babel.split(',')
69      .map(str::trim)
70      .find_map(|s| s.strip_prefix("main=").map(str::trim).map(str::to_string));
71    // Filter: drop key=value options AND babel-language MODIFIERS (sub-options
72    // recognised by a single .ldf rather than a language by themselves).
73    // Modifiers we know about:
74    //   `es-*` (babel-spanish: es-tabla, es-cuadro, es-noindentfirst, …)
75    // Driver: 2102.11084 `\usepackage[spanish, es-tabla]{babel}` selected
76    // "es-tabla" as the language → "haven't defined the language 'es-tabla'"
77    // GenericError.
78    let is_lang_candidate = |s: &str| -> bool {
79      !s.is_empty() && s != "nil" && !s.contains('=')
80        && !s.starts_with("es-")
81    };
82    let pkg_last = main_kv.unwrap_or_else(|| {
83      opt_babel.split(',').map(str::trim).rfind(|s| is_lang_candidate(s)).unwrap_or_default().to_string()
84    });
85    let lang = if !pkg_last.is_empty() {
86      pkg_last
87    } else if main != "nil" && !main.is_empty() {
88      main
89    } else {
90      return Ok(vec![]);
91    };
92    let iso = crate::package::babel_support_sty::babel_language_to_iso(&lang);
93    if let Some(code) = iso {
94      state::assign_value("DOCUMENT_LANGUAGE",
95        Stored::from(code.to_string()), Some(Scope::Global));
96      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
97    }
98    // Force-set \bbl@main@language globally so babel's AtBeginDocument
99    // \selectlanguage{\bbl@main@language} picks up the user's intended
100    // main language (the LAST option), not whichever .ldf's \ldf@finish
101    // ran last. Babel's own chain then handles captions activation,
102    // active-char shorthands, and per-language port dispatching.
103    def_macro(T_CS!("\\bbl@main@language"), None,
104      Tokens!(Explode!(lang.clone())),
105      Some(ExpandableOptions { scope: Some(Scope::Global), ..ExpandableOptions::default() }))?;
106  });
107  // Run mainlang at load time so DOCUMENT_LANGUAGE is set before
108  // \begin{document} opens (and base_schema's after_open reads it).
109  RawTeX!(r"\lx@babel@activate@mainlang");
110
111});
