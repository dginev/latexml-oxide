148  // After babel loads: activate the main language's captions and shorthands.
149  // In Perl this happens via the precompiled kernel; we do it explicitly.
150  DefPrimitive!("\\lx@babel@activate@mainlang", {
151    let main = gullet::do_expand(T_CS!("\\bbl@main@language"))
152      .map(|t| t.to_string()).unwrap_or_default();
153    let loaded = gullet::do_expand(T_CS!("\\bbl@loaded"))
154      .map(|t| t.to_string()).unwrap_or_default();
155    // \opt@babel.sty has the actual \usepackage[...]{babel} options.
156    // Prefer these over \bbl@main@language which may reflect class options.
157    let opt_babel = gullet::do_expand(Tokenize!(r"\csname opt@babel.sty\endcsname"))
158      .map(|t| t.to_string()).unwrap_or_default();
159    let pkg_last = opt_babel.split(',').map(|s| s.trim().to_string())
160      .rfind(|s| !s.is_empty() && s != "nil").unwrap_or_default();
161    let loaded_last = loaded.split(',').map(|s| s.trim().to_string())
162      .rfind(|s| !s.is_empty() && s != "nil").unwrap_or_default();
163    let lang = if !pkg_last.is_empty() {
164      pkg_last
165    } else if !loaded_last.is_empty() {
166      loaded_last
167    } else if main != "nil" && !main.is_empty() {
168      main
169    } else {
170      String::new()
171    };
172    if !lang.is_empty() {
173      // Temporarily set @ to LETTER for CS name tokenization
174      // (at \begin{document} time, @ is OTHER which breaks \captions<lang>)
175      state::assign_catcode('@', Catcode::LETTER, None);
176      // Call \captions<lang> to set localized names
177      let cs = s!("\\captions{}", lang);
178      if lookup_definition(&T_CS!(cs.clone()))?.is_some() {
179        stomach::digest(Tokenize!(&cs))?;
180      }
181      // Set font language directly (don't go through stomach::digest which
182      // can interleave with pending babel hook processing).
183      let iso = match lang.as_str() {
184        "german" | "germanb" | "ngerman" | "ngermanb" => Some("de"),
185        "french" | "francais" | "frenchb" => Some("fr"),
186        "spanish" => Some("es"), "italian" => Some("it"),
187        "english" => Some("en"),
188        "american" | "USenglish" => Some("en-US"),
189        "british" | "UKenglish" => Some("en-GB"),
190        "portuguese" | "portuges" => Some("pt"),
191        "russian" | "russianb" => Some("ru"),
192        "greek" | "polutonikogreek" => Some("el"),
193        "dutch" => Some("nl"), "polish" => Some("pl"),
194        _ => None,
195      };
196      if let Some(code) = iso {
197        merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
198      }
199      // Restore @ to OTHER
200      state::assign_catcode('@', Catcode::OTHER, None);
201
202      // French active punctuation: make :;!? insert thin space BEFORE the char.
203      // Safe to activate here since we're at \begin{document} time (all packages loaded).
204      let is_french = lang == "french" || lang == "francais" || lang == "frenchb";
205      if is_french {
206        // Define dispatch primitives for French punctuation (if not already defined)
207        if lookup_definition(&T_CS!("\\lx@french@punct@colon"))?.is_none() {
208          // French punctuation spacing:
209          // Before ':' → regular non-breaking space (U+00A0, espace insécable)
210          // Before ';!?' → thin space (U+2006 SIX-PER-EM SPACE, espace fine)
211          DefPrimitive!("\\lx@french@punct@colon", {
212            enter_horizontal();
213            Tbox::new(arena::pin_static(" :"), None, None, Tokens!(), stored_map!())
214          });
215          DefPrimitive!("\\lx@french@punct@semi", {
216            enter_horizontal();
217            Tbox::new(arena::pin_static("\u{2006};"), None, None, Tokens!(), stored_map!())
218          });
219          DefPrimitive!("\\lx@french@punct@exclam", {
220            enter_horizontal();
221            Tbox::new(arena::pin_static("\u{2006}!"), None, None, Tokens!(), stored_map!())
222          });
223          DefPrimitive!("\\lx@french@punct@question", {
224            enter_horizontal();
225            Tbox::new(arena::pin_static("\u{2006}?"), None, None, Tokens!(), stored_map!())
226          });
227        }
228        for &(ch, cs_name) in &[
229          (':', "\\lx@french@punct@colon"),
230          (';', "\\lx@french@punct@semi"),
231          ('!', "\\lx@french@punct@exclam"),
232          ('?', "\\lx@french@punct@question"),
233        ] {
234          state::assign_catcode(ch, Catcode::ACTIVE, Some(Scope::Global));
235          let active_tok = T_ACTIVE!(ch);
236          let dispatch_cs = T_CS!(cs_name);
237          if let Some(defn) = lookup_meaning(&dispatch_cs) {
238            state::assign_meaning(&active_tok, defn, Some(Scope::Global));
239          }
240        }
241      }
242    }

249  // German " shorthand system (from germanb.ldf).
250  // babel's \initiate@active@char mechanism often fails during raw loading;
251  // implement the dispatch entirely in Rust as a Primitive that reads the
252  // next token and expands to the appropriate shorthand.
253  DefPrimitive!("\\lx@german@dq@dispatch", {
254    // Read the next token (the character after ")
255    let tok = gullet::read_token()?;
256    let ch = tok.as_ref().map(|t| t.with_str(|s| s.to_string())).unwrap_or_default();
257    // Map shorthand to expansion. Use Unicode directly for umlauts to avoid
258    // catcode issues with active " interfering with \" command.
259    let expansion: &str = match ch.as_str() {
260      "a" => "\u{00E4}", "o" => "\u{00F6}", "u" => "\u{00FC}",
261      "e" => "\u{00EB}", "i" => "\u{00EF}",
262      "A" => "\u{00C4}", "O" => "\u{00D6}", "U" => "\u{00DC}",
263      "E" => "\u{00CB}", "I" => "\u{00CF}",
264      "s" | "z" => "\u{00DF}", // ß
265      "S" => "SS", "Z" => "SZ",
266      "`" => "\u{201E}", // „ (German opening quote)
267      "'" => "\u{201C}", // " (German closing quote)
268      "<" => "\u{00AB}", // «
269      ">" => "\u{00BB}", // »
270      "~" => "-", "=" => "-",
271      // "" → empty (hskip) — handled below via empty check
272      // Consonant shorthands: discretionary hyphens, ignored in LaTeXML
273      _ => "",
274    };
275    if !expansion.is_empty() {
276      gullet::unread(Tokenize!(expansion));
277    } else if !ch.is_empty() {
278      // For consonants (c,f,l,m,n,p,r,t,...) and unknowns: just output the character
279      if let Some(t) = tok { gullet::unread(Tokens!(t)); }
280    }

308      .map(|t| t.to_string()).unwrap_or_default();
309    // In babel, the last explicitly loaded language is the main language.
310    // Prefer \opt@babel.sty (explicit package options) over \bbl@loaded
311    // (which may only contain class-option languages).
312    let pkg_last = opt_babel.split(',').map(|s| s.trim().to_string())
313      .rfind(|s| !s.is_empty() && s != "nil").unwrap_or_default();
314    let loaded_last = loaded.split(',').map(|s| s.trim().to_string())
315      .rfind(|s| !s.is_empty() && s != "nil").unwrap_or_default();
316    let lang_name = if !pkg_last.is_empty() {
317      pkg_last
318    } else if !loaded_last.is_empty() {
319      loaded_last
320    } else if main != "nil" && !main.is_empty() {
321      main
322    } else {
323      String::new()
324    };
325    // Map language name to ISO code
326    let iso = match lang_name.as_str() {
327      "german" | "germanb" | "ngerman" | "ngermanb" => Some("de"),
328      "french" | "francais" | "frenchb" => Some("fr"),
329      "spanish" => Some("es"), "italian" => Some("it"),
330      "english" => Some("en"),
331      "american" | "USenglish" => Some("en-US"),
332      "british" | "UKenglish" => Some("en-GB"),
333      "portuguese" | "portuges" => Some("pt"),
334      "russian" | "russianb" => Some("ru"),
335      "greek" | "polutonikogreek" => Some("el"),
336      "dutch" => Some("nl"), "polish" => Some("pl"),

342      merge_font(Font { language: Some(Cow::Owned(code.to_string())), ..Font::default() });
343    }
344    // Call \captions<lang> to set localized names
345    let captions_cs = s!("\\captions{}", lang_name);
346    if lookup_definition(&T_CS!(captions_cs.clone()))?.is_some() {
347      stomach::digest(Tokenize!(&captions_cs))?;
348    }
349    // French-specific: define French macros directly
350    let is_french = lang_name == "french" || lang_name == "francais"
351      || lang_name == "frenchb" || loaded.contains("french");
352    if is_french && lookup_definition(&T_CS!("\\up"))?.is_none() {
353      // Core French macros from french.ldf.ltxml / frenchb.ldf.ltxml
354      stomach::digest(Tokenize!(r"\def\up#1{\textsuperscript{#1}}"))?;
355      stomach::digest(Tokenize!(r"\def\fup#1{\textsuperscript{#1}}"))?;
356      stomach::digest(Tokenize!(r"\def\No{N\up{o}\xspace}\def\no{n\up{o}\xspace}"))?;
357      stomach::digest(Tokenize!(r"\def\Nos{N\up{os}\xspace}\def\nos{n\up{os}\xspace}"))?;

359      // Note: \ier/\iere/\ieme use \xspace (matching raw french.ldf).
360      // \primo/\secundo/... use \FBthickkern (matching french.ldf FrenchEnumerate).
361      // \No/\Nos/\no/\nos use \xspace.
362      stomach::digest(Tokenize!(r"\def\FBthickkern{\thinspace}"))?;
363      stomach::digest(Tokenize!(r"\def\ieme{\up{e}\xspace}\def\iemes{\up{es}\xspace}"))?;
364      stomach::digest(Tokenize!(r"\def\ier{\up{er}\xspace}\def\iers{\up{ers}\xspace}"))?;
365      stomach::digest(Tokenize!(r"\def\iere{\up{re}\xspace}\def\ieres{\up{res}\xspace}"))?;
366      stomach::digest(Tokenize!(r"\def\FrenchEnumerate#1{#1\up{o}\FBthickkern}"))?;
367      stomach::digest(Tokenize!(r"\def\FrenchPopularEnumerate#1{#1\up{o})\FBthickkern}"))?;
368      stomach::digest(Tokenize!(r"\def\primo{\FrenchEnumerate1}\def\secundo{\FrenchEnumerate2}"))?;
369      stomach::digest(Tokenize!(r"\def\tertio{\FrenchEnumerate3}\def\quarto{\FrenchEnumerate4}"))?;
370      stomach::digest(Tokenize!(r"\def\fprimo){\FrenchPopularEnumerate1}\def\fsecundo){\FrenchPopularEnumerate2}"))?;
371      stomach::digest(Tokenize!(r"\def\ftertio){\FrenchPopularEnumerate3}\def\fquarto){\FrenchPopularEnumerate4}"))?;
372      stomach::digest(Tokenize!(r"\def\og{\guillemotleft\nobreakspace}"))?;
373      stomach::digest(Tokenize!(r"\def\fg{\nobreakspace\guillemotright\xspace}"))?;
374      stomach::digest(Tokenize!(r"\def\degre{\textdegree}"))?;
375      stomach::digest(Tokenize!(r"\def\degres{\hbox to 0.3em{\degre}}"))?;
376      stomach::digest(Tokenize!(r"\let\tild\textasciitilde"))?;
377      stomach::digest(Tokenize!(r"\let\circonflexe\textasciicircum"))?;
378      stomach::digest(Tokenize!(r"\def\at{@}\def\boi{\textbackslash}"))?;

381      // \xspace implementation in xspace_sty.rs. Load it for French macros.

388    }
