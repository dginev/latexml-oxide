27  // Perl L35-45: override \include so each included chapter file gets
28  // its own CHAPTERBIB_UNIT / CITE_UNIT stamp (derived from the file
29  // basename). Without this override chapterbib's per-chapter
30  // bibliography never activated — every \cite resolved against a
31  // single global unit.
32  DefPrimitive!("\\include{}", sub[(path)] {
33    let path_str = Expand!(path).to_string();
34    let table = state::lookup_value("including@only");
35    let should_include = match &table {
36      None => true,
37      Some(Stored::HashString(map)) => map.contains_key(&path_str),
38      _ => true,
39    };
40    if should_include {
41      let name = std::path::Path::new(&path_str)
42        .file_stem()
43        .and_then(|s| s.to_str())
44        .unwrap_or(&path_str)
45        .to_string();
46      let cite_unit = if state::lookup_value("CITE_UNIT_GLOBAL").is_some() {
47        format!("bibliography {}", name)
48      } else {
49        name.clone()
50      };
51      AssignValue!("CHAPTERBIB_UNIT" => Stored::from(name), Some(Scope::Global));
52      AssignValue!("CITE_UNIT"       => Stored::from(cite_unit), Some(Scope::Global));
53      gullet::unread_one(T_CS!("\\lx@cb@reset"));
54      Input!(&path_str);
55    }
56  });

58  // Perl L47: expose the current chapterbib unit name as a token
59  // stream. The Perl uses a zero-arg `DefMacro(.., sub { Explode(…) })`
60  // closure; the Rust binding language doesn't yet have a typed
61  // zero-arg Expandable sub form, so emit a primitive that reads the
