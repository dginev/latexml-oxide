27  // Perl epsfig.sty.ltxml L35-52: \psfig RequiredKeyVals:epsGin emits
28  // an ltx:graphics element. properties closure extracts 'file' or
29  // 'figure' key as the graphic path, removes those keys from the kv
30  // set, injects clip=true if epsfclip was toggled on, and serializes
31  // remaining key=value pairs as the options string. Prior Rust was a
32  // \psfig{} primitive that discarded everything, so every \psfig call
33  // produced no graphic node at all.
34  DefConstructor!("\\psfig RequiredKeyVals:epsGin",
35    "<ltx:graphics graphic='#graphic' candidates='#candidates' options='#options'/>",
36    enter_horizontal => true,
37    properties => sub[args] {
38      use DigestedData::*;
39      let file = if let Some(kv_arg) = args[0].as_ref() {
40        if let KeyVals(kv) = kv_arg.data() {
41          let raw = kv.get_value("file")
42            .or_else(|| kv.get_value("figure"))
43            .map(|v| v.to_string())
44            .unwrap_or_default();
45          raw.trim().to_string()
46        } else { String::new() }
47      } else { String::new() };
48      let candidates = util::image::image_candidates(&file);
49      // Serialize remaining keyvals as "k=v,k=v" (skip file/figure,
50      // inject clip=true if epsfclip was toggled on).
51      let clip_on = lookup_value("epsfclip")
52        .map(|v| v.to_string() != "0" && !v.to_string().is_empty())
53        .unwrap_or(false);
54      let mut opts = Vec::<String>::new();
55      if let Some(kv_arg) = args[0].as_ref()
56        && let KeyVals(kv) = kv_arg.data() {
57          let mut saw_clip = false;
58          for (k, v) in kv.get_pairs() {
59            if k == "file" || k == "figure" { continue; }
60            if k == "clip" { saw_clip = true; }
61            let v_str = v.to_string();
62            if v_str.is_empty() { opts.push(k.clone()); }
63            else { opts.push(format!("{}={}", k, v_str)); }
64          }
65          if clip_on && !saw_clip { opts.push("clip=true".into()); }
66        }
67      let options = opts.join(",");
68      Ok(stored_map!("graphic" => file, "candidates" => candidates, "options" => options))
69    });
