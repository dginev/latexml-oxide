156  // \spnewtheorem*{env}[numberedlike]{caption}[within]{capfont}{bodyfont}
157  // Perl llncs.cls.ltxml L101-157 is `DefMacro('\spnewtheorem ...', sub
158  // { ... DefMacroI + NewCounter + MergeFont ... })` — a macro whose
159  // body imperatively installs new CSes/env bindings + counters.
160  // Rust uses DefPrimitive (stomach-level) because the installation
161  // needs to happen at digest-stable time so subsequent uses of the
162  // new `{env}` work. WISDOM #44: kind flip is intentional —
163  // `\spnewtheorem` is a preamble-declaration macro only, never
164  // captured by `\edef`/`\ifx`. capfont/bodyfont are TeX font
165  // commands (e.g. \bfseries, \itshape) — ignored in LaTeXML (both
166  // Perl and Rust do the same).
167  DefPrimitive!("\\spnewtheorem OptionalMatch:* {}[]{}[] {}{}", sub[(flag, thmset, otherthmset, typ, reset, _capfont, _bodyfont)] {
168    engine::latex_constructs::define_new_theorem(
169      flag.filter(|f| !f.is_empty()),
170      thmset,
171      otherthmset.filter(|t| !t.is_empty()),
172      if typ.is_empty() { None } else { Some(typ) },
173      reset.filter(|t| !t.is_empty()),
174    )?;
175  });
176  Let!("\\spdefaulttheorem", "\\spnewtheorem");
