1//! pgfmathcalc.code.tex — PGF math calculation macros
2//! Perl: pgfmathcalc.code.tex.ltxml (34 lines)

4//! Loads the raw TeX code and provides \pgfmathsetmacro in Rust.
5use crate::prelude::*;
6
7#[rustfmt::skip]
8LoadDefinitions!({
9  // Perl L20: Load pgf's TeX code for math calc first
10  InputDefinitions!("pgfmathcalc.code", extension => Some(Cow::Borrowed("tex")), noltxml => true);
11
12  // Perl L24-32: \pgfmathsetmacro — evaluates expression and defines macro.
13  // Perl's DefMacroI call explicitly passes `scope => 'local'`, so the
14  // new CS vanishes when the enclosing group closes — essential for
15  // tikz/pgf loops that reuse the same CS (e.g. \foreach \n …). The
16  // Rust port had dropped the scope, leaking redefinitions upward past
17  // the begingroup/endgroup wrapper. Restore Scope::Local to match Perl.
18  DefPrimitive!("\\pgfmathsetmacro{}{}", sub[(cs, expression)] {
19    let expr_str = do_expand(expression)?.to_string();
20    bgroup();
21    let result = crate::package::pgfmath_code_tex::pgfmathparse_eval(&expr_str);
22    egroup()?;
23    let result_tokens = mouth::tokenize_internal(&result);
24    def_macro(
25      cs.unlist().into_iter().next().unwrap_or(T_CS!("\\pgfmathresult")),
26      None,
27      result_tokens,
28      Some(ExpandableOptions { scope: Some(Scope::Local), ..Default::default() }))?;
29    Ok(Vec::new())
30  });
31});
