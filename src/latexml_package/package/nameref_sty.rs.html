1use crate::prelude::*;
2LoadDefinitions!({
3  // Perl: loads these packages
4  RequirePackage!("refcount");
5  RequirePackage!("gettitlestring");
6  RequirePackage!("ltxcmds");
7
8  // We want to display the actual "name" of the labeled structure (e.g. \section),
9  //   which is accessible via show="title"
10  //
11  // TODO: The star forms prevent nested double links.
12  // Perl L28 passes `enterHorizontal => 1` so \nameref in vertical mode
13  // transitions into horizontal like \ref does. Rust was missing it, so
14  // a bare `\nameref{…}` at the start of a cell or a potential vmode
15  // position could skip the mode flip.
16  DefConstructor!("\\nameref OptionalMatch:* Semiverbatim",
17  "<ltx:ref ?#1(class='ltx_refmacro_nameref ltx_nolink')(class='ltx_refmacro_nameref')\
18    show='title' labelref='#label' _force_font='true'/>",
19  enter_horizontal => true,
20  properties => sub[args] {
21    let label_arg = args[1].as_ref().map(ToString::to_string).unwrap_or_default();
22    Ok(stored_map!(
23      "label" => clean_label(&label_arg, None)))
24  });
25
26  DefMacro!("\\Nameref", "\\nameref"); //\def\Nameref#1{‘\nameref{#1}’ on page~\pageref{#1}}
27  DefMacro!("\\Sectionformat{}{}", "#1");
28  DefMacro!("\\Ref", "\\ref"); // can be improved if "varioref.sty" is loaded?
29  //The original nameref docs say: "Overload an AMS LaTEX command, which uses \newlabel. Sigh!"
30  DefMacro!("\\slabel", "\\label");
31  // We can improve if we had \vpageref
32  DefMacro!("\\vnameref", "\\nameref");
33});
