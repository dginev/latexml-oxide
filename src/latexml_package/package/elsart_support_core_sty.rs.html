25    let body_str = body.to_string();

27
28    // State machine to parse key=value pairs with brace nesting
29    let chars: Vec<char> = body_str.chars().collect();
30    let len = chars.len();
31    let mut i = 0;
32
33    while i < len {
34      // Skip whitespace and commas between key-value pairs
35      while i < len && (chars[i].is_whitespace() || chars[i] == ',') {
36        i += 1;
37      }
38      if i >= len { break; }
39

41      let key_start = i;
42      while i < len && chars[i] != '=' {

44      }

46      // Use char-vec slicing (NOT byte slicing on body_str) — affiliation
47      // text often contains UTF-8 multi-byte chars (accented names,
48      // diacritics) where the char-count `i` is not a byte boundary.
49      // Driver: 2407.00104 panic at `body_str[val_start..i]`.
50      let _key: String = chars[key_start..i].iter().collect::<String>().trim().to_lowercase();
51      i += 1; // skip '='
52
53      // Skip whitespace after '='
54      while i < len && chars[i].is_whitespace() {
55        i += 1;
56      }
57      if i >= len { break; }
58
59      // Read value — may be brace-delimited or plain
60      let value;

62        // Brace-delimited value: read until matching '}'
63        i += 1; // skip opening '{'
64        let val_start = i;
65        let mut depth = 1;
66        while i < len && depth > 0 {
67          if chars[i] == '{' { depth += 1; }
68          else if chars[i] == '}' { depth -= 1; }

70        }
71        value = chars[val_start..i].iter().collect::<String>().trim().to_string();

73      } else {

76        while i < len && chars[i] != ',' {
77          i += 1;
78        }

94          _ => {
95            // Unknown keys: include value as-is (matching elsarticle unknown handler)

98            }
99          }

268  // Perl elsart_support_core.sty.ltxml L138-141: `\keyword{...}` (1-arg
269  // form) reads a balanced group and wraps with `\@keyword <arg>
270  // \@keyword@cut` so XUntil terminates at the inserted token.
271  //
272  // History — DO NOT switch this back to `read_balanced(_, _, false)`:
273  // commit 09bc60c2 ("\keyword absorbs trailing unbalanced }") tried to
274  // mirror Perl's `$gullet->readBalanced` with no args (require_open=
275  // false → level starts at 1) to absorb the legacy unbalanced-`}`
276  // idiom from arXiv:1710.03688 / hep-ph0702114, where the abstract
277  // ends:
278  //     \keyword{Keyword1; ...} \vskip ...\noindent{...e3}}
279  // (a trailing literal `}` past the keyword arg). For that input
280  // alone, `require_open=false` correctly consumes `{...} }` and stops.
281  //
282  // BUT for the common balanced form — `\keyword{Higgs; Boson}` with
283  // no trailing extra `}` (the test fixture
284  // `tests/babel/elsart_keyword_brace_form.tex` and the vast majority
285  // of real-world usage) — `require_open=false` reads past the
286  // matching `}`, walks through `\end{abstract}` (raw `\end` is a CS
287  // and CC_END count is unchanged across that expansion), then
288  // `\end{frontmatter}`, … all the way to EOF. The reader's pushback
289  // grows unboundedly on the way; in CI we observed a single
290  // `memory allocation of 21743271936 bytes failed` (21 GB) → SIGABRT
291  // before the test could even report. The cost of one paper's
292  // 1-error reduction is the entire test suite OOM-aborting at
293  // `81_babel::elsart_keyword_brace_form_test`.
294  //
295  // RESOLVED 2026-05-30 via the design note's first suggestion: keep the
296  // strict balanced read of the keyword argument, then peek past spaces and
297  // gobble an OPTIONAL trailing `}` (catcode END). This reproduces Perl's
298  // lenient `readBalanced` *result* (it absorbs the legacy unbalanced `}`)
299  // for the trailing-`}` idiom — abstract bodies that end
300  //     \keyword{Kw1; Kw2; …}   <comments/spaces>   }   \end{abstract}
301  // (the stray `}` is the orphaned close of a commented-out brace group;
302  // witness 1601.01227, also 1710.03688 / hep-ph0702114) — WITHOUT the
303  // speculative read-to-EOF that OOM'd the balanced fixture: for the common
304  // `\keyword{Higgs; Boson}\end{abstract}` form the token after the strict
305  // read is `\end` (a CS, not `}`), so nothing is gobbled. The peek is
306  // bounded to one token, never crossing `\end`.
307  // Skip leading spaces and blank-line `\par`s, then gobble a single trailing
308  // `}` (catcode END) if one is there — Perl's `readBalanced` reads straight
309  // through the `\par` from a blank line before the stray brace (witness
310  // 1705.01354 separates `\keyword{…}` and the orphaned `}` by a blank line).
311  // The skipped whitespace/`\par` tokens are buffered and FULLY RESTORED when
312  // no trailing `}` follows, so the common `\keyword{…}\end{abstract}` form
313  // (and `\keyword{…}<blank>\end{abstract}`) is left byte-for-byte untouched.
314  // Bounded: stops at the first non-space, non-`\par` token; never crosses
315  // `\end`.
316  // Perl's `readBalanced` (L140) reads tokens up to the next UNMATCHED `}`
317  // (tracking brace depth), absorbing that stray `}` as its terminator —
318  // regardless of what non-brace material (spaces, `\par`, `\vskip <dimen>`,
319  // `\noindent`, …) sits between `\keyword{…}` and the orphaned brace. The
320  // earlier version only skipped spaces/`\par`s, so it stopped at the first
321  // real token (e.g. `\vskip`) and left the stray `}` to hit the abstract's
322  // mode-switch frame (`ltx:para`-free `}` → "close a group that switched to
323  // mode internal_vertical"; witness 1604.00855:
324  // `\keyword{…} \vskip 0.5\baselineskip}`).
325  //
326  // Walk forward tracking brace depth: an unmatched `}` (depth 0) is the
327  // stray terminator — DROP it, re-inject everything else (so the `\vskip`
328  // etc. still render, content-preserving). Bound the scan at `\end` (depth
329  // 0) — the env terminator — and a hard token cap, so the common
330  // `\keyword{Higgs; Boson}\end{abstract}` form (no stray `}`) reads one
331  // token (`\end`), restores it, and is left untouched (this is also what
332  // averts the read-to-EOF OOM a naive `readBalanced` hit). Witnesses:
333  // 1604.00855, 1601.01227, 1705.01354, 1710.03688, hep-ph0702114.
334  DefPrimitive!("\\lx@elsart@gobble@optbrace", {
335    let mut skipped: Vec<Token> = Vec::new();
336    let mut depth: i32 = 0;
337    let mut count: usize = 0;
338    while let Some(tok) = gullet::read_token()? {
339      count += 1;
340      if count > 4096 {
341        skipped.push(tok); // safety cap: restore everything, gobble nothing
342        break;
343      }
344      let cc = tok.get_catcode();
345      if cc == Catcode::BEGIN {
346        depth += 1;
347        skipped.push(tok);
348      } else if cc == Catcode::END {
349        if depth == 0 {
350          break; // unmatched stray `}` — drop it (absorbed, like readBalanced)
351        }
352        depth -= 1;
353        skipped.push(tok);
354      } else if depth == 0 && tok == T_CS!("\\end") {
355        skipped.push(tok); // env end reached before any stray `}` — restore all
356        break;
357      } else {
358        skipped.push(tok);
359      }
360    }
361    // Re-inject everything we buffered, in order. The only token ever removed
362    // is the stray `}` (which is never pushed onto `skipped`).
363    for tok in skipped.into_iter().rev() {
364      gullet::unread_one(tok);
365    }
366  });
