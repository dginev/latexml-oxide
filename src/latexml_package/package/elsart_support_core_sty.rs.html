15  // Not in Perl elsart_support_core, but needed for modern elsarticle papers.
16  // The TeX elsarticle.cls uses LaTeX3 \keys_set:nn to parse key-value pairs
17  // (organization, addressline, city, postcode, state, country) and concatenate
18  // the values with comma separators. We parse the key-value body in Rust

21  DefMacro!("\\affiliation[]{}", sub[(_opt_label, body)] {
22    // Parse key-value pairs from the body, respecting brace nesting.
23    // Keys: organization, addressline, city, postcode, state, country
24    // Values are typically brace-delimited: key={value with, commas}
25    let body_str = body.to_string();
26    let mut parts: Vec<String> = Vec::new();
27
28    // State machine to parse key=value pairs with brace nesting
29    let chars: Vec<char> = body_str.chars().collect();
30    let len = chars.len();
31    let mut i = 0;
32
33    while i < len {
34      // Skip whitespace and commas between key-value pairs
35      while i < len && (chars[i].is_whitespace() || chars[i] == ',') {
36        i += 1;
37      }
38      if i >= len { break; }
39
40      // Read key (until '=' or end)
41      let key_start = i;
42      while i < len && chars[i] != '=' {
43        i += 1;
44      }
45      if i >= len { break; }
46      // Use char-vec slicing (NOT byte slicing on body_str) — affiliation
47      // text often contains UTF-8 multi-byte chars (accented names,
48      // diacritics) where the char-count `i` is not a byte boundary.
49      // Driver: 2407.00104 panic at `body_str[val_start..i]`.
50      let _key: String = chars[key_start..i].iter().collect::<String>().trim().to_lowercase();
51      i += 1; // skip '='
52
53      // Skip whitespace after '='
54      while i < len && chars[i].is_whitespace() {
55        i += 1;
56      }
57      if i >= len { break; }
58
59      // Read value — may be brace-delimited or plain
60      let value;
61      if i < len && chars[i] == '{' {
62        // Brace-delimited value: read until matching '}'
63        i += 1; // skip opening '{'
64        let val_start = i;
65        let mut depth = 1;
66        while i < len && depth > 0 {
67          if chars[i] == '{' { depth += 1; }
68          else if chars[i] == '}' { depth -= 1; }
69          if depth > 0 { i += 1; }
70        }
71        value = chars[val_start..i].iter().collect::<String>().trim().to_string();
72        if i < len { i += 1; } // skip closing '}'
73      } else {
74        // Plain value: read until next comma or end
75        let val_start = i;
76        while i < len && chars[i] != ',' {
77          i += 1;
78        }
79        value = chars[val_start..i].iter().collect::<String>().trim().to_string();
80      }
81
82      // Only include known affiliation keys with non-empty values
83      let trimmed = value.trim();
84      if !trimmed.is_empty() {
85        match _key.as_str() {
86          "organization" | "organisation" | "o" | "or"
87          | "addressline" | "a" | "ad"
88          | "city" | "c" | "ci"
89          | "postcode" | "p" | "pc"
90          | "state" | "s" | "st"
91          | "country" | "cy" => {
92            parts.push(trimmed.to_string());
93          }
94          _ => {
95            // Unknown keys: include value as-is (matching elsarticle unknown handler)
96            if !trimmed.is_empty() {
97              parts.push(trimmed.to_string());
98            }
99          }
100        }
101      }
102    }
103
104    let affil_text = parts.join(", ");
105    let mut result = Vec::new();
106    result.push(T_CS!("\\@add@to@frontmatter"));
107    result.push(T_BEGIN!());
108    for ch in "ltx:creator".chars() {
109      result.push(Token { text: arena::pin_char(ch), code: Catcode::OTHER,
110      #[cfg(feature = "token-locators")] loc: 0
111    });
112    }
113    result.push(T_END!());
114    result.push(T_BEGIN!());
115    result.push(T_CS!("\\@@@affiliation"));
116    result.push(T_BEGIN!());
117    for ch in affil_text.chars() {
118      if ch == ' ' {
119        result.push(T_SPACE!());
120      } else if ch.is_ascii_alphabetic() {
121        result.push(Token { text: arena::pin_char(ch), code: Catcode::LETTER,
122      #[cfg(feature = "token-locators")] loc: 0
123    });
124      } else {
125        result.push(Token { text: arena::pin_char(ch), code: Catcode::OTHER,
126      #[cfg(feature = "token-locators")] loc: 0
127    });
128      }
129    }
130    result.push(T_END!());
131    result.push(T_END!());
132    result
133  });

262  // Perl elsart_support_core.sty.ltxml L138-141: `\keyword{...}` (1-arg
263  // form) reads a balanced group and wraps with `\@keyword <arg>
264  // \@keyword@cut` so XUntil terminates at the inserted token.
265  //
266  // History — DO NOT switch this back to `read_balanced(_, _, false)`:
267  // commit 09bc60c2 ("\keyword absorbs trailing unbalanced }") tried to
268  // mirror Perl's `$gullet->readBalanced` with no args (require_open=
269  // false → level starts at 1) to absorb the legacy unbalanced-`}`
270  // idiom from arXiv:1710.03688 / hep-ph0702114, where the abstract
271  // ends:
272  //     \keyword{Keyword1; ...} \vskip ...\noindent{...e3}}
273  // (a trailing literal `}` past the keyword arg). For that input
274  // alone, `require_open=false` correctly consumes `{...} }` and stops.
275  //
276  // BUT for the common balanced form — `\keyword{Higgs; Boson}` with
277  // no trailing extra `}` (the test fixture
278  // `tests/babel/elsart_keyword_brace_form.tex` and the vast majority
279  // of real-world usage) — `require_open=false` reads past the
280  // matching `}`, walks through `\end{abstract}` (raw `\end` is a CS
281  // and CC_END count is unchanged across that expansion), then
282  // `\end{frontmatter}`, … all the way to EOF. The reader's pushback
283  // grows unboundedly on the way; in CI we observed a single
284  // `memory allocation of 21743271936 bytes failed` (21 GB) → SIGABRT
285  // before the test could even report. The cost of one paper's
286  // 1-error reduction is the entire test suite OOM-aborting at
287  // `81_babel::elsart_keyword_brace_form_test`.
288  //
289  // The trailing-`}` arxiv 1710.03688 / hep-ph0702114 case is now a
290  // deferred parity gap. Any future fix must distinguish balanced vs
291  // unbalanced input WITHOUT speculatively reading to EOF — e.g. peek
292  // for the trailing `}` after a strict balanced read, or scope the
293  // lenient reader by an explicit token budget.
