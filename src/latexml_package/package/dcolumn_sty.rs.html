3/// Perl: absorbedString — digests tokens and extracts text content.
4/// Mirrors the Perl function in dcolumn.sty.ltxml that creates a temporary
5/// document to get the display form of math tokens (e.g. \cdot → ⋅).
6fn absorbed_string(todelim: &Tokens) -> String {
7  // Build \ensuremath{todelim} tokens
8  let mut toks = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
9  toks.extend_from_slice(todelim.unlist_ref());
10  toks.push(T_END!());
11  // Digest and extract text from resulting boxes
12  match stomach::digest(Tokens::new(toks)) {
13    Ok(digested) => collect_text(&digested),
14    Err(_) => todelim.to_string(),
15  }
16}
17
18/// Recursively extract leaf text content from a Digested tree.
19fn collect_text(digested: &Digested) -> String {
20  let mut result = String::new();
21  match digested.data() {
22    DigestedData::TBox(b) => {
23      let tbox = b.borrow();
24      arena::with(tbox.text, |text| result.push_str(text));
25    },
26    DigestedData::List(l) => {
27      let list = l.borrow();
28      for item in list.boxes.iter() {
29        result.push_str(&collect_text(item));
30      }
31    },
32    DigestedData::Whatsit(w) => {
33      let whatsit = w.borrow();
34      if let Some(Stored::Digested(body)) = whatsit.properties.get("body") {
35        result.push_str(&collect_text(body));
36      }
37    },
38    _ => {},
39  }
40  result
41}

48  // Perl: \lx@unactivate DefToken — resets mathcode of a character.
49  // Perl kind is DefMacro with an imperative sub body (no token return);
50  // Rust DefPrimitive runs the side effect at stomach time. WISDOM #44:
51  // the two kinds differ under expansion (`\edef` etc.); safe here because
52  // `\lx@unactivate` is only emitted inside `\DC@` expansions that execute
53  // at math-mode stomach time, never captured by `\edef`.
54  // WISDOM #44 verified 2026-04-23: zero `\edef`/`\ifx`/`\expandafter`
55  // uses of `\lx@unactivate` across LaTeXML/lib + ar5iv-bindings.
56  DefPrimitive!("\\lx@unactivate DefToken", sub[(delim_tok)] {
57    let delim_str = delim_tok.to_string();
58    if let Some(ch) = delim_str.chars().next() {
59      state::assign_mathcode(ch, 0u16, None);
60    }
61  });
62
63  // Perl: \DC@{}{}{} — activates the decimal delimiter in math mode
64  DefMacro!("\\DC@{}{}{}", sub[(delim, todelim, _ndec)] {
65    let delim_str = delim.to_string();
66    let todelim_str = todelim.to_string();
67    if delim_str != todelim_str {
68      if let Some(ch) = delim_str.chars().next() {
69        // Make the delimiter math-active (code 0x8000)
70        state::assign_mathcode(ch, 0x8000u16, None);
71      }
72      // Define the active character's expansion
73      let expansion_body = s!(
74        "\\lx@hidden@bgroup\\lx@unactivate{{{}}}\\lx@wrap[role=PERIOD]{{{}}}\\lx@hidden@egroup",
75        delim_str, todelim_str
76      );
77      let expansion = mouth::tokenize_internal(&expansion_body);
78      def_macro(T_CS!(delim_str), None, expansion, None)?;
79    }
80    // Save and deactivate $
81    Let!("\\DC@saved@dollar", "$");
82    state::let_i(&T_MATH!(), &T_CS!("\\relax"), None);
83    // Start inline math if not already in math
84    let in_math = state::lookup_bool_sym(pin!("IN_MATH"));
85    if in_math {
86      state::assign_value("DC_started_math", Stored::Bool(false), None);
87      Ok(Tokens::default())
88    } else {
89      state::assign_value("DC_started_math", Stored::Bool(true), None);
90      Ok(Tokens!(T_CS!("\\lx@begin@inline@math")))
91    }
92  });
93
94  // Perl: \DC@end — restores $ and ends inline math (only if we started it)
95  DefMacro!("\\DC@end", sub[_args] {
96    state::let_i(&T_MATH!(), &T_CS!("\\DC@saved@dollar"), None);
97    let started = state::lookup_value("DC_started_math")
98      .map(|v| matches!(v, Stored::Bool(true)))
99      .unwrap_or(false);
100    if started {
101      Ok(Tokens!(T_CS!("\\lx@end@inline@math")))
102    } else {
103      Ok(Tokens::default())
104    }
105  });
106
107  // Perl: DefColumnType('D{}{}{}', ...) — decimal alignment column
108  // Perl: align => 'char:' . absorbedString(Tokens(T_CS('\ensuremath'), T_BEGIN, $todelim, T_END))
109  DefColumnType!("D{}{}{}", sub[(delim, todelim, ndec)] {
110    // Perl: absorbedString — digest \ensuremath{todelim} to get display character
111    let alignment = absorbed_string(&todelim);
112    // Build before tokens: \DC@{delim}{todelim}{ndec}
113    let mut before = vec![T_CS!("\\DC@"), T_BEGIN!()];
114    before.extend(delim.unlist());
115    before.push(T_END!());
116    before.push(T_BEGIN!());
117    before.extend(todelim.unlist());
118    before.push(T_END!());
119    before.push(T_BEGIN!());
120    before.extend(ndec.unlist());
121    before.push(T_END!());
122    with_current_build_template(|template_opt| {
123      template_opt.unwrap().add_column(Cell {
124        before: Some(Tokens::new(before)),
125        after: Some(Tokens!(T_CS!("\\DC@end"))),
126        align: Some(Align::Char(alignment)),
127        ..Cell::default()
128      });
129    });
130  });
