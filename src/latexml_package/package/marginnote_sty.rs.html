33  // Perl marginnote.sty.ltxml L37-40: \marginnote[left]{right}[vshift]
34  // expands to \marginpar — with the left-margin text (#1) included as the
35  // optional [left] argument when present, and the vshift (#3) ignored.
36  // Prior Rust stub dropped #1 entirely; this port preserves the left branch.
37  DefMacro!("\\marginnote []{}[]", sub[(left, right, _vadjust)] {
38    let mut out: Vec<Token> = Vec::new();
39    out.push(T_CS!("\\marginpar"));
40    if let Some(l) = left {
41      // [\mn@parboxrestore\marginfont\raggedleftmarginnote <left>]
42      out.push(T_OTHER!("["));
43      out.push(T_CS!("\\mn@parboxrestore"));
44      out.push(T_CS!("\\marginfont"));
45      out.push(T_CS!("\\raggedleftmarginnote"));
46      out.push(T_SPACE!());
47      out.extend(l.unlist());
48      out.push(T_OTHER!("]"));
49    }
50    // {\mn@parboxrestore\marginfont\raggedrightmarginnote <right>}
51    out.push(T_BEGIN!());
52    out.push(T_CS!("\\mn@parboxrestore"));
53    out.push(T_CS!("\\marginfont"));
54    out.push(T_CS!("\\raggedrightmarginnote"));
55    out.push(T_SPACE!());
56    out.extend(right.unlist());
57    out.push(T_END!());
58    Ok(Tokens::new(out))
59  });

61  // Perl marginnote.sty.ltxml L42-46: \@mn@if@RTL dispatches at
62  // expansion time — if \if@RTL is defined (LookupValue) AND currently

64  DefMacro!("\\@mn@if@RTL", sub[_args] {
