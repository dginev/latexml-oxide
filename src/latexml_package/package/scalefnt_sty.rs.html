1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl scalefnt.sty.ltxml L20-22:
6  //   DefPrimitive('\scalefont{}', sub { MergeFont(scale => ToString($scale)); });
7  // Takes a float scale factor and multiplies the current font size. The
8  // Float parameter type delivers an f64 directly; fontmap! handles the
9  // Option wrap. Previous stub was a no-op DefMacro — user-visible size
10  // changes (e.g. `\scalefont{0.8}`) silently dropped.
11  // Perl: `\scalefont{}` — read ONE mandatory braced argument.
12  // Use `{Float}` so the binding strips the surrounding `{...}` and
13  // parses the contents as a float. Bare `Float` consumed only the
14  // unbraced numeric prefix and left the literal `{` on the stream,
15  // which caused mis-tracked brace depth → runaway gullet pushback
16  // (4 GB Vec growth) on subsequent body content. Witness:
17  // arXiv:1511.09288 — `\scalefont{0.9}{\hspace{0.2mm}…}` OOMed in
18  // a few ms during canvas stage_56 (second-500K).
19  DefPrimitive!("\\scalefont {Float}", sub[(scale)] {
20    merge_font(fontmap!(scale => scale.value_f64()));
21    Ok(Vec::new())
22  });
23});
