1//! arximspdf.cls / arxstspdf.cls — the arXiv IMS journal classes (Annals of
2//! Probability/Statistics/Applied Probability, `aop`/`aos`/`aap`/`aoas`).
3//!
4//! Neither Perl LaTeXML nor (previously) Rust bound these self-contained ~3000-
5//! line classes, so `\documentclass{arximspdf}` papers cascaded into dozens of
6//! undefined-macro errors (the structured `\b*` bibliography, `{barticle}`/…,
7//! plus `\eqntext`/`\dvtx`/`{longlist}`). This binding loads `article` as the
8//! base and defines the IMS-specific macros semantically so the papers convert —
9//! a strict improvement over Perl, which fails outright (both engines lack the
10//! class; same-host confirmed). Witnesses: 0910.0069 (aop632) + the 16-paper
11//! aop/aos sandbox cluster.
12//!
13//! Bibliography note: arximspdf uses `natbib` + `\bibitem`/`thebibliography`, so
14//! a bibitem produces `ltx:bibitem > ltx:bibblock` (Flow.model). The structured
15//! `ltx:bib-*` vocabulary is schema-valid only inside `ltx:bibentry` (the BibTeX
16//! path), NOT inside `bibblock` — so the `\b*` field macros are PASSTHROUGH text
17//! (readable + schema-valid), not mapped to `ltx:bib-*`.
18use crate::prelude::*;
19
20#[rustfmt::skip]
21LoadDefinitions!({
22  // ---- options -------------------------------------------------------------
23  // Journal selectors + numbering/layout options: all ignorable for a semantic
24  // conversion; pass anything unrecognized through to article.
25  for opt in [
26    "aop","aos","aap","aoas",
27    "10pt","11pt","12pt","draft","final","openright","openany",
28    "onecolumn","twocolumn","leqno","fleqn","twoside","oneside",
29    "number","nameyear","noMRlinks","MSNbibl","citesort","dvips","pdftex",
30    "autosecdot","noautosecdot","xxtheorem","nohyperref",
31    "normalfloat","secfloat","chapfloat","rotating",
32    "normaleqn","seceqn","chapeqn","normalthm","secthm","chapthm",
33  ] { DeclareOption!(opt, None); }
34  DeclareOption!(None, { Digest!("\\PassOptionsToClass{\\CurrentOption}{article}")?; });
35  ProcessOptions!();
36
37  load_class_with_options("article", Tokens!())?;
38
39  RequirePackage!("amssymb");
40  // NB: arximspdf does NOT load amsmath — it defines \tfrac/\dfrac/\operatorname/
41  // {aligned} itself. We mirror that: load only amsopn (for \operatorname /
42  // \DeclareMathOperator) and define the rest directly. Loading full amsmath would
43  // override plain-TeX \matrix with its env-form, breaking the command-form
44  // \matrix{...\cr...} these papers use (Perl is parity-broken the same way and
45  // also omits amsmath).
46  RequirePackage!("amsopn");
47  RequirePackage!("bm");        // \boldsymbol
48  RequirePackage!("natbib");    // \citep/\citet/\citeyear, thebibliography
49  RequirePackage!("graphicx");
50  RequirePackage!("hyperref");  // \href, \texorpdfstring (class loads it too)
51
52  // ---- math the class defines itself (arximspdf L1099-1100) ----------------
53  DefMacro!("\\tfrac{}{}", "{\\textstyle\\frac{#1}{#2}}");
54  DefMacro!("\\dfrac{}{}", "{\\displaystyle\\frac{#1}{#2}}");
55  DefMacro!("\\dvt",  "\\colon\\ ");
56  DefMacro!("\\dvtx", "\\colon\\;");
57  DefMacro!("\\divid{}{}", "\\frac{#1}{#2}");
58  DefMacro!("\\zs{}",  "#1");
59  DefMacro!("\\zss{}", "#1");
60  DefMacro!("\\eqntext{}", "#1");
61  Let!("\\bolds", "\\boldsymbol");
62
63  // ---- structured bibliography (passthrough) -------------------------------
64  // Entry environments: transparent (swallow the optional [type] arg, pass body).
65  for env in [
66    "barticle","bbook","bincollection","binproceedings","binbook","bproceedings",
67    "btechreport","bmanual","bmastersthesis","bphdthesis","bbooklet",
68    "bunpublished","bmisc","bchapter",
69  ] {
70    def_macro_noop(&format!("\\{env}[]"))?;
71    def_macro_noop(&format!("\\end{env}"))?;
72  }
73  // Field markup macros: emit their content as text.
74  for m in [
75    "bauthor","beditor","bsnm","bfnm","binits","bparticle","bsuffix",
76    "btitle","bjournal","bbooktitle","bseries","bvolume","byear","bpages",
77    "bedition","bpublisher","baddress","blocation","borganization","binstitution",
78    "bschool","btype","bnumber","bchapter","bhowpublished","bnote","banumber",
79    "bisbn","betal",
80  ] {
81    def_macro_identity(&format!("\\{m}{{}}"))?;
82  }
83  DefMacro!("\\AND", "and ");
84  def_macro_noop("\\bptok{}")?;
85  def_macro_noop("\\bptnote{}")?;
86  def_macro_noop("\\bid{}")?;          // keyval mr=/doi=… — drop (matches class display)
87  DefMacro!("\\bmrnumber{}", "\\MR{#1}");
88  DefMacro!("\\MR{}", "MR#1");         // MathReviews id, as text
89  def_macro_noop("\\endbibitem")?;
90
91  // ---- misc list / sectioning / theorems the class adds --------------------
92  DefEnvironment!("{longlist}[]", "<ltx:enumerate>#body</ltx:enumerate>",
93    mode => "internal_vertical");
94  DefMacro!("\\newproclaim", "\\newtheorem");
95  // arximspdf predefines a Theorem env and proof env(s) (L1405/1430-1435).
96  RawTeX!(r"\newtheorem{thm}{Theorem}");
97  RawTeX!(r"\newenvironment{pf}{\begin{proof}}{\end{proof}}");
98  RawTeX!(r"\newenvironment{pf*}[1]{\begin{proof}[#1]}{\end{proof}}");
99
100  // ---- frontmatter (standard LaTeXML frontmatter API; metadata preserved) --
101  // Standardize the IMS scaffolding onto the article frontmatter flow: the
102  // {frontmatter}/{aug}/{keyword} blocks are TRANSPARENT wrappers (no special
103  // mode), and the IMS metadata macros route to the same \lx@add@* sinks
104  // article/acmart use, so creators/title/keywords/pubnotes are emitted into the
105  // document frontmatter.
106  DefEnvironment!("{frontmatter}", "#body");
107  DefEnvironment!("{aug}", "#body");
108  // {keyword} collects MANY \kwd into one keywords list: \lx@add@keywords alone
109  // CLEARS on each call (keeps only the last), so use the collecting
110  // \lx@begin@keywords/\lx@end@keywords form and let each \kwd emit its text.
111  DefMacro!("\\keyword[]", "\\lx@begin@keywords");
112  DefMacro!("\\endkeyword", "\\lx@end@keywords");
113  // Author-name parts used inside \author{...}: pass their content through.
114  for m in ["fnms","snm","inits","degs","roles","suffix"] {
115    def_macro_identity(&format!("\\{m}{{}}"))?;
116  }
117  // \author[mark]{name} (IMS adds an optional thanks-mark) -> standard creator.
118  DefMacro!("\\author[]{}", "\\lx@add@creator[role=author]{#2}");
119  DefMacro!("\\affiliation{}", "\\lx@add@contact[role=affiliation]{#1}");
120  DefMacro!("\\address[]{}", "\\lx@add@contact[role=address]{#2}");
121  DefMacro!("\\email{}", "\\lx@add@contact[role=email,name={e-mail: }]{#1}");
122  DefMacro!("\\ead[]{}", "\\lx@add@contact[role=email,name={e-mail: }]{#2}");
123  DefMacro!("\\kwd[]{}", "#2, ");   // emit into the collecting {keyword} block
124  DefMacro!("\\journal{}", "\\lx@add@pubnote[role=journal]{#1}");
125  DefMacro!("\\volume{}", "\\lx@add@pubnote[role=volume]{#1}");
126  DefMacro!("\\issue{}", "\\lx@add@pubnote[role=number]{#1}");
127  DefMacro!("\\pubyear{}", "\\lx@add@date[role=publication]{#1}");
128  DefMacro!("\\doi{}", "\\lx@add@pubnote[role=doi]{#1}");
129  DefMacro!("\\received{}", "\\lx@add@date[role=received]{#1}");
130  DefMacro!("\\revised{}", "\\lx@add@date[role=revised]{#1}");
131  DefMacro!("\\accepted{}", "\\lx@add@date[role=accepted]{#1}");
132  // History date parts + misc frontmatter: passthrough / drop.
133  for m in ["sday","smonth","syear","sname","stitle","sdescription"] {
134    def_macro_identity(&format!("\\{m}{{}}"))?;
135  }
136  DefMacro!("\\dochead{}", "");
137  DefMacro!("\\dedicated{}", "");
138  DefMacro!("\\runtitle{}", "");
139  DefMacro!("\\runauthor{}", "");
140  for m in ["corref{}","docsubty{}","thankstext[]{}{}","thanksref[]{}",
141            "thanksmark[]{}","thankslabel[]{}","firstpage{}","lastpage{}",
142            "referstodoi{}","relateddoi[]{}{}","printead","printead*",
143            "printaddress{}","printaddressnum{}","printaddresses",
144            "pdftitle{}","pdfauthor{}","pdfsubject{}","pdfkeywords{}",
145            "sdatatype{}","sfilename{}","slink[]{}","thesuppdoi{}"] {
146    def_macro_noop(&format!("\\{m}"))?;
147  }
148  DefEnvironment!("{supplement}[]", "#body");
149
150  // ---- misc stubs ----------------------------------------------------------
151  DefMacro!("\\startlocaldefs", "\\makeatletter");
152  DefMacro!("\\endlocaldefs", "\\makeatother");
153  for cs in ["HPROOF","PROOF","CRC","psdraft","psfull"] { def_macro_noop(&format!("\\{cs}"))?; }
154  def_macro_noop("\\vtexed{}")?;
155});
