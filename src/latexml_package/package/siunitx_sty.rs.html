9/// Structured error emission for siunitx (parallels
10/// `latexml_post::diag::log_post_error!` and the engine `Error!` macro).
11///
12/// Why not `latexml_core::Error!`: the full macro early-returns
13/// `Err(LatexmlError)` on max-errors / runaway-loop, but the siunitx
14/// parsing helpers below return `Option<…>` / `Tokens` / `Vec<…>`,
15/// not `Result<_, LatexmlError>`, so the early-return would type-mismatch.
16/// The thin emit-only macro keeps the harness `Error:<class>:<object>`
17/// format contract while staying type-compatible.
18macro_rules! six_log_error {
19  ($category:expr, $object:expr, $msg:expr) => {
20    log::error!(target: &format!("{}:{}", $category, $object), "{}", $msg)
21  };
22  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {
23    log::error!(target: &format!("{}:{}", $category, $object), $fmt, $($arg)+)
24  };
25}

105/// `six_pin!("key")` — concatenates `"SIX_"` + `key` at compile time,
106/// then caches the interned `SymStr` per call site. Every subsequent
107/// lookup on the same thread is a `OnceCell` load + u32 compare — no
108/// `format!()` allocation, no hash probe into the arena.
109///
110/// siunitx reads its options on every `\num` / `\SI` / `\ang` expansion
111/// and `format!("SIX_{key}")` was one of the measurable chunks of the
112/// siunitx critical path. This macro replaces the `&str`-based `six_get`
113/// family for the (41 of 52) call sites that pass a string literal.
114macro_rules! six_pin {
115  ($key:literal) => {{
116    std::thread_local! {
117      static CACHED: std::cell::OnceCell<::latexml_core::common::arena::SymStr>
118        = const { std::cell::OnceCell::new() };
119    }
120    CACHED
121      .with(|c| *c.get_or_init(|| ::latexml_core::common::arena::pin_static(concat!("SIX_", $key))))
122  }};
123}

161/// Build raw keyvals content (without brackets) for passing to i_wrap
162fn make_kv_content(kv: &[(&str, Tokens)]) -> Option<Tokens> {
163  if kv.is_empty() {
164    return None;
165  }
166  let mut tks = Vec::new();
167  for (i, (key, value)) in kv.iter().enumerate() {
168    if i > 0 {
169      tks.push(T_OTHER!(","));
170    }
171    tks.extend(ExplodeText!(key));
172    tks.push(T_OTHER!("="));
173    tks.push(T_BEGIN!());
174    tks.extend_from_slice(value.unlist_ref());
175    tks.push(T_END!());
176  }
177  Some(Tokens::new(tks))
178}

180/// Perl: six_get_op — look up an option, wrap in \text{} and I_wrap
181fn six_get_op_sym(kv: &[(&str, Tokens)], key: SymStr) -> Tokens {
182  let text = six_get_tokens_sym(key);
183  if text.is_empty() {
184    i_wrap(make_kv_content(kv), Tokens::default())
185  } else {
186    let mut tks = vec![T_CS!("\\text"), T_BEGIN!()];
187    tks.extend(text.unlist());
188    tks.push(T_END!());
189    i_wrap(make_kv_content(kv), Tokens::new(tks))
190  }
191}

193/// Perl: six_setup — assign all keyvals to SIX_key state values
194fn six_setup(kv: &KeyVals) {
195  for (key, value) in kv.get_pairs() {
196    let key_str = key.clone();
197    match value {
198      ArgWrap::Tokens(t) => {
199        if t.is_empty() {
200          // Bare flag (e.g., "parse-numbers" without "= true") → set "true"
201          state::assign_value(&format!("SIX_{key_str}"), Stored::from("true"), None);
202        } else {
203          state::assign_value(&format!("SIX_{key_str}"), Stored::Tokens(t.clone()), None);
204        }
205      },
206      ArgWrap::KV(_) => {
207        state::assign_value(
208          &format!("SIX_{key_str}"),
209          Stored::from(value.to_string()),
210          None,
211        );
212      },
213      _ => {
214        // Empty value (boolean flag) or other
215        let s = value.to_string();
216        if s.is_empty() {
217          state::assign_value(&format!("SIX_{key_str}"), Stored::from("true"), None);
218        } else {
219          state::assign_value(&format!("SIX_{key_str}"), Stored::from(s), None);
220        }
221      },
222    }
223  }
224}

226/// Perl: six_begin_processing — bgroup + apply keyvals + redefine input-protect-tokens
227fn six_begin_processing(kv: Option<&KeyVals>) {
228  stomach::bgroup();
229  if let Some(kv) = kv {
230    six_setup(kv);
231  }
232  // Redefine input-protect-tokens: make each CS token into T_OTHER of its name
233  if let Some(Stored::Tokens(protect)) = six_get_sym(six_pin!("input-protect-tokens")) {
234    for token in protect.unlist() {
235      if token.get_catcode() == Catcode::ESCAPE {
236        let name = token.to_string();
237        let other_name = name.trim_start_matches('\\');
238        let expansion = Tokens::new(vec![T_OTHER!(other_name)]);
239        if let Ok(def) = Expandable::new(token, None, Some(ExpansionBody::Tokens(expansion)), None)
240        {
241          state::install_definition(def, None);
242        }
243      }
244    }
245  }
246}

255/// Parsed number structure (mirrors Perl hash)
256#[derive(Clone, Debug)]
257enum SixNumber {
258  Simple {
259    sign:     Option<Tokens>,
260    integer:  Option<Tokens>,
261    decimal:  Option<Tokens>,
262    fraction: Option<Tokens>,
263  },
264  Operator {
265    operator:   String,
266    arg1:       Option<Box<SixNumber>>,
267    arg2:       Option<Box<SixNumber>>,
268    sign:       Option<Tokens>,
269    symbol:     Option<Tokens>,
270    comparator: Option<Tokens>,
271  },

276    sign: Option<Tokens>,
277    integer: Option<Tokens>,

318/// Perl: six_match_keys — match and remove leading tokens matching sixkeys
319fn six_match_keys(tokens: &mut Vec<Token>, keys: &[SymStr]) -> Option<Tokens> {
320  let mut tomatch: Vec<Token> = vec![T_SPACE!()];
321  for key in keys {
322    if let Some(Stored::Tokens(toks)) = six_get_sym(*key) {
323      tomatch.extend(toks.unlist());
324    }
325  }
326
327  // Match by Token-text equality against `tomatch`. For a typical
328  // `tomatch` of a few dozen tokens and input of a few dozen tokens
329  // this is O(n*m) — still fast because `Token` eq is a `SymStr` u32
330  // compare, not a string compare.
331  let mut matched = Vec::new();
332  let mut consumed = 0;
333  for t in tokens.iter() {
334    if !tomatch.iter().any(|m| t == m) {
335      break;
336    }
337    consumed += 1;
338    if t.get_catcode() != Catcode::SPACE {
339      if t.get_catcode() == Catcode::ESCAPE {
340        matched.push(*t);
341        matched.push(T_SPACE!());
342      } else {
343        matched.push(*t);
344      }
345    }
346  }
347  // Single drain at the end — the prior `tokens.remove(0)` per match
348  // was O(n) each call (memmove of the tail on every consumed token).
349  if consumed > 0 {
350    tokens.drain(..consumed);
351  }
352  if matched.is_empty() {
353    None
354  } else {
355    Some(Tokens::new(matched))
356  }
357}

363fn six_match_simplenumber(tokens: &mut Vec<Token>) -> Option<SixNumber> {
364  let sign = six_match_sign(tokens);
365  let integer = six_match_keys(tokens, &[
366    six_pin!("input-digits"),
367    six_pin!("input-symbols"),
368  ]);
369  let (decimal, fraction) =
370    if six_match_keys(tokens, &[six_pin!("input-decimal-markers")]).is_some() {
371      (
372        Some(Tokens::default()),
373        six_match_keys(tokens, &[
374          six_pin!("input-digits"),
375          six_pin!("input-symbols"),
376        ]),
377      )
378    } else {
379      (None, None)
380    };
381
382  if sign.is_some() || integer.is_some() || decimal.is_some() || fraction.is_some() {
383    Some(SixNumber::simple(sign, integer, decimal, fraction))
384  } else {
385    None
386  }
387}

389fn six_match_uncertainnumber(tokens: &mut Vec<Token>) -> Option<SixNumber> {
390  let number = six_match_simplenumber(tokens)?;
391
392  if let Some(sign) = six_match_keys(tokens, &[six_pin!("input-uncertainty-signs")]) {
393    let int = six_match_keys(tokens, &[
394      six_pin!("input-digits"),
395      six_pin!("input-symbols"),
396    ]);
397    let (dec, frac) = if six_match_keys(tokens, &[six_pin!("input-decimal-markers")]).is_some() {
398      (
399        Some(Tokens::default()),
400        six_match_keys(tokens, &[
401          six_pin!("input-digits"),
402          six_pin!("input-symbols"),
403        ]),
404      )
405    } else {
406      (None, None)
407    };
408
409    if six_match_keys(tokens, &[
410      six_pin!("input-decimal-markers"),
411      six_pin!("input-complex-roots"),
412    ])
413    .is_some()
414    {
415      return Some(number);
416    }
417
418    let uncertainty = SixNumber::simple(Some(sign), int, dec, frac);
419    return Some(SixNumber::Operator {
420      operator:   "uncertain".to_string(),
421      arg1:       Some(Box::new(number)),
422      arg2:       Some(Box::new(uncertainty)),
423      sign:       None,
424      symbol:     None,
425      comparator: None,
426    });
427  }
428
429  if six_match_keys(tokens, &[six_pin!("input-open-uncertainty")]).is_some() {
430    let int = six_match_keys(tokens, &[
431      six_pin!("input-digits"),
432      six_pin!("input-symbols"),
433    ]);
434    six_match_keys(tokens, &[six_pin!("input-close-uncertainty")]);
435    let uncertainty = SixNumber::simple(None, int, None, None);
436    return Some(SixNumber::Operator {
437      operator:   "uncertain".to_string(),
438      arg1:       Some(Box::new(number)),
439      arg2:       Some(Box::new(uncertainty)),
440      sign:       None,
441      symbol:     None,
442      comparator: None,
443    });
444  }
445
446  Some(number)
447}

449fn six_match_complexnumber(tokens: &mut Vec<Token>) -> Option<SixNumber> {
450  let mut number = six_match_uncertainnumber(tokens)?;
451
452  if let Some(i) = six_match_keys(tokens, &[six_pin!("input-complex-roots")]) {
453    let sign = number.get_sign().cloned();
454    number.set_sign(None);
455    return Some(SixNumber::Operator {
456      operator: "complex".to_string(),
457      arg1: None,
458      arg2: Some(Box::new(number)),
459      sign,
460      symbol: Some(i),
461      comparator: None,
462    });
463  }
464
465  if let Some(sign) = six_match_sign(tokens) {
466    if let Some(i) = six_match_keys(tokens, &[six_pin!("input-complex-roots")]) {
467      if let Some(imag) = six_match_uncertainnumber(tokens) {
468        return Some(SixNumber::Operator {
469          operator:   "complex".to_string(),
470          arg1:       Some(Box::new(number)),
471          arg2:       Some(Box::new(imag)),
472          sign:       Some(sign),
473          symbol:     Some(i),
474          comparator: None,
475        });
476      }
477    } else if let Some(imag) = six_match_uncertainnumber(tokens) {
478      if let Some(i) = six_match_keys(tokens, &[six_pin!("input-complex-roots")]) {
479        return Some(SixNumber::Operator {
480          operator:   "complex".to_string(),
481          arg1:       Some(Box::new(number)),
482          arg2:       Some(Box::new(imag)),
483          sign:       Some(sign),
484          symbol:     Some(i),
485          comparator: None,
486        });
487      }
488      // Imaginary part matched but no `input-complex-roots` key — incomplete
489      // complex form. Perl siunitx.sty.ltxml:266 — Error('unexpected',
490      // 'sign', undef, "expected to find complex number")
491      six_log_error!(
492        "unexpected", "sign",
493        "expected to find complex number"
494      );
495    } else {
496      // Perl siunitx.sty.ltxml:266 — same Error site (no imaginary part
497      // followed the matched sign).
498      six_log_error!(
499        "unexpected", "sign",
500        "expected to find complex number"
501      );
502    }
503  }
504
505  Some(number)
506}

508fn six_match_scinumber(tokens: &mut Vec<Token>) -> Option<SixNumber> {
509  // Perl siunitx.sty.ltxml:270-279 — `$number` may be undef and the
510  // exponent-marker still matches. `\SI{e12}{G}` is a valid no-mantissa
511  // form that defaults to `10^12`. Without this branch our `?` short-
512  // circuited on a missing mantissa and the exponent-marker leaked into
513  // the leftover-tokens "Not matched in \num" error path. Witnesses:
514  // arXiv:2509.14043 / 2509.16675 — `\SIrange{e12}{e15}{G}` and
515  // `\SI{4e13}{G}` astro-physics magnetic-field strings.
516  let number = six_match_complexnumber(tokens);
517
518  if six_match_keys(tokens, &[six_pin!("input-exponent-markers")]).is_some() {
519    let sign = six_match_sign(tokens);
520    let exp = six_match_keys(tokens, &[
521      six_pin!("input-digits"),
522      six_pin!("input-symbols"),
523    ]);
524    let exp_number = SixNumber::simple(sign, exp, None, None);
525    return Some(SixNumber::Operator {
526      operator:   "exponent".to_string(),
527      arg1:       number.map(Box::new),
528      arg2:       Some(Box::new(exp_number)),
529      sign:       None,
530      symbol:     None,
531      comparator: None,
532    });
533  }
534
535  number
536}

538fn six_match_compoundnumber(tokens: &mut Vec<Token>) -> Option<SixNumber> {
539  if let Some(comp) = six_match_keys(tokens, &[six_pin!("input-comparators")]) {
540    return Some(SixNumber::Operator {
541      operator:   "comparator".to_string(),
542      arg1:       six_match_number(tokens).map(Box::new),
543      arg2:       None,
544      sign:       None,
545      symbol:     None,
546      comparator: Some(comp),
547    });
548  }
549
550  let mut number = six_match_scinumber(tokens)?;
551  loop {
552    if six_match_keys(tokens, &[six_pin!("input-product")]).is_some() {
553      let rhs = six_match_scinumber(tokens);
554      number = SixNumber::Operator {
555        operator:   "product".to_string(),
556        arg1:       Some(Box::new(number)),
557        arg2:       rhs.map(Box::new),
558        sign:       None,
559        symbol:     None,
560        comparator: None,
561      };
562    } else if six_match_keys(tokens, &[six_pin!("input-quotient")]).is_some() {
563      let rhs = six_match_scinumber(tokens);
564      number = SixNumber::Operator {
565        operator:   "quotient".to_string(),
566        arg1:       Some(Box::new(number)),
567        arg2:       rhs.map(Box::new),
568        sign:       None,
569        symbol:     None,
570        comparator: None,
571      };
572    } else {
573      return Some(number);
574    }
575  }
576}

586fn six_apply_mathligatures(tokens: Vec<Token>) -> Vec<Token> {
587  let mut result = Vec::with_capacity(tokens.len());
588  let mut iter = tokens.into_iter().peekable();
589  // Pre-intern the ligature trigger chars so the per-token dispatch
590  // below compares SymStr (u32 equality) instead of allocating
591  // `t.to_string()` to compare with "+" / ">" / "<" / etc.
592  let plus = pin!("+");
593  let minus = pin!("-");
594  let gt = pin!(">");
595  let lt = pin!("<");
596  let eq = pin!("=");
597  while let Some(t) = iter.next() {
598    if t.get_catcode() == Catcode::COMMENT {
599      continue;
600    }
601    if t.text == plus {
602      if iter.peek().is_some_and(|n| n.text == minus) {
603        iter.next();
604        result.push(T_CS!("\\pm"));
605      } else {
606        result.push(t);
607      }
608    } else if t.text == gt {
609      if let Some(ns) = iter.peek().map(|n| n.text) {
610        if ns == eq {
611          iter.next();
612          result.push(T_CS!("\\ge"));
613        } else if ns == gt {
614          iter.next();
615          result.push(T_CS!("\\gg"));
616        } else {
617          result.push(t);
618        }
619      } else {
620        result.push(t);
621      }
622    } else if t.text == lt {
623      if let Some(ns) = iter.peek().map(|n| n.text) {
624        if ns == eq {
625          iter.next();
626          result.push(T_CS!("\\le"));
627        } else if ns == lt {
628          iter.next();
629          result.push(T_CS!("\\ll"));
630        } else {
631          result.push(t);
632        }
633      } else {
634        result.push(t);
635      }
636    } else {
637      result.push(t);
638    }
639  }
640  result
641}

651fn six_postprocess_aux(mut number: SixNumber) -> SixNumber {
652  match &mut number {
653    SixNumber::Operator { arg1, arg2, .. } => {
654      *arg1 = arg1.take().map(|n| Box::new(six_postprocess_aux(*n)));
655      *arg2 = arg2.take().map(|n| Box::new(six_postprocess_aux(*n)));
656    },
657    SixNumber::Simple {
658      decimal,
659      fraction,
660      integer,
661      sign,
662      ..
663    } => {
664      if six_get_bool_sym(six_pin!("add-decimal-zero")) && decimal.is_some() && fraction.is_none() {
665        *fraction = Some(Tokens::new(vec![T_OTHER!("0")]));
666      }
667      if six_get_bool_sym(six_pin!("add-integer-zero")) && decimal.is_some() && integer.is_none() {
668        *integer = Some(Tokens::new(vec![T_OTHER!("0")]));
669      }
670      if sign.is_none() {
671        if let Some(Stored::Tokens(s)) = six_get_sym(six_pin!("explicit-sign")) {
672          if !s.is_empty() {
673            *sign = Some(s);
674          }
675        }
676      }
677    },
678  }
679  number
680}

691fn six_parse_number(expr: &Tokens) -> SixParseResult {
692  if six_get_bool_sym(six_pin!("parse-numbers")) {
693    let expanded = gullet::do_expand_partially(expr.clone()).unwrap_or_else(|_| expr.clone());
694    let mut tokens = six_apply_mathligatures(expanded.unlist());
695    let result = six_postprocess(six_match_number(&mut tokens));
696    if !tokens.is_empty() {
697      // Perl siunitx.sty.ltxml:410 — Error('unexpected', $$tokens[0],
698      //   $gullet, "Not matched in \\num: ...")
699      let leftover = Tokens::new(tokens.clone()).to_string();
700      let first_obj = tokens.first().map(|t| t.to_string()).unwrap_or_default();
701      six_log_error!(
702        "unexpected", first_obj,
703        "Not matched in \\num: {}", leftover
704      );
705      return SixParseResult::Raw(expr.clone());
706    }
707    match result {
708      Some(n) => SixParseResult::Parsed(n),
709      None => SixParseResult::Raw(expr.clone()),
710    }
711  } else {
712    SixParseResult::Raw(expr.clone())
713  }
714}
715
716fn six_parse_numbers(expr: &Tokens) -> Vec<SixParseResult> {
717  if six_get_bool_sym(six_pin!("parse-numbers")) {
718    let expanded = gullet::do_expand_partially(expr.clone()).unwrap_or_else(|_| expr.clone());
719    let mut tokens = six_apply_mathligatures(expanded.unlist());
720    let mut results = Vec::new();
721    loop {
722      let result = six_postprocess(six_match_number(&mut tokens));
723      match result {
724        Some(n) => results.push(SixParseResult::Parsed(n)),
725        None => break,
726      }
727      if tokens.first().is_some_and(|t| t.text == pin!(";")) {
728        tokens.remove(0);
729      } else {
730        break;
731      }
732    }
733    if !tokens.is_empty() {
734      // Perl siunitx.sty.ltxml:430 — Error('unexpected', $$tokens[0],
735      //   $gullet, "Not matched in \\num: ...")
736      let leftover = Tokens::new(tokens.clone()).to_string();
737      let first_obj = tokens.first().map(|t| t.to_string()).unwrap_or_default();
738      six_log_error!(
739        "unexpected", first_obj,
740        "Not matched in \\num: {}", leftover
741      );
742      return vec![SixParseResult::Raw(expr.clone())];
743    }
744    results
745  } else {
746    let mut results = Vec::new();
747    let mut current = Vec::new();
748    for t in expr.unlist_ref() {
749      if t.text == pin!(";") {
750        results.push(SixParseResult::Raw(Tokens::new(current)));
751        current = Vec::new();
752      } else {
753        current.push(*t);
754      }
755    }
756    if !current.is_empty() {
757      results.push(SixParseResult::Raw(Tokens::new(current)));
758    }
759    results
760  }
761}

767fn six_number_string(number: &SixNumber) -> String {
768  match number {
769    SixNumber::Operator {
770      operator,
771      arg1,
772      arg2,
773      sign,
774      comparator,
775      ..
776    } => match operator.as_str() {
777      "uncertain" => {
778        let a1 = arg1
779          .as_ref()
780          .map(|n| six_number_string(n))
781          .unwrap_or_default();
782        if let Some(a2) = arg2 {
783          format!("{}({})", a1, six_number_string(a2))
784        } else {
785          a1
786        }
787      },
788      "comparator" => {
789        let comp = comparator
790          .as_ref()
791          .map(|t| t.to_string())
792          .unwrap_or_default();
793        let a1 = arg1
794          .as_ref()
795          .map(|n| six_number_string(n))
796          .unwrap_or_default();
797        format!("{}{}", comp, a1)
798      },
799      "product" => {
800        let a1 = arg1
801          .as_ref()
802          .map(|n| six_number_string(n))
803          .unwrap_or_default();
804        let a2 = arg2
805          .as_ref()
806          .map(|n| six_number_string(n))
807          .unwrap_or_default();
808        format!("{}*{}", a1, a2)
809      },
810      "quotient" => {
811        let a1 = arg1
812          .as_ref()
813          .map(|n| six_number_string(n))
814          .unwrap_or_default();
815        let a2 = arg2
816          .as_ref()
817          .map(|n| six_number_string(n))
818          .unwrap_or_default();
819        format!("{}/{}", a1, a2)
820      },
821      "complex" => {
822        let a1 = arg1
823          .as_ref()
824          .map(|n| six_number_string(n))
825          .unwrap_or_default();
826        let a2 = arg2
827          .as_ref()
828          .map(|n| six_number_string(n))
829          .unwrap_or_default();
830        let s = sign
831          .as_ref()
832          .map(|t| t.to_string())
833          .unwrap_or_else(|| "+".to_string());
834        format!("{}{}{}\u{2148}", a1, s, a2)
835      },
836      "exponent" => {
837        let a1 = arg1
838          .as_ref()
839          .map(|n| six_number_string(n))
840          .unwrap_or_default();
841        let a2 = arg2
842          .as_ref()
843          .map(|n| six_number_string(n))
844          .unwrap_or_default();
845        format!("{}E{}", a1, a2)
846      },
847      _ => String::new(),
848    },
849    SixNumber::Simple { sign, integer, fraction, .. } => {
850      let s = sign.as_ref().map(|t| t.to_string()).unwrap_or_default();
851      let i = integer.as_ref().map(|t| t.to_string()).unwrap_or_default();
852      let f = fraction
853        .as_ref()
854        .map(|t| format!(".{}", t))
855        .unwrap_or_default();
856      format!("{}{}{}", s, i, f)
857    },
858  }
859}

861fn six_groupdigits(digits: &Tokens, direction: i32) -> Tokens {
862  let min: usize = six_get_choice_sym(six_pin!("group-minimum-digits"))
863    .parse()
864    .unwrap_or(5);
865  if min > digits.len() {
866    return digits.clone();
867  }
868  let digs: Vec<Token> = digits.unlist_ref().clone();
869  let sep = six_get_tokens_sym(six_pin!("group-separator"));
870  let g = 3usize;
871  let mut result = Vec::new();
872
873  if direction > 0 {
874    let mut remaining = digs;
875    while !remaining.is_empty() {
876      let mut chunk = Vec::new();
877      for _ in 0..g {
878        if remaining.is_empty() {
879          break;
880        }
881        chunk.insert(0, remaining.pop().unwrap());
882      }
883      if !result.is_empty() {
884        result.splice(0..0, sep.unlist_ref().iter().copied());
885      }
886      result.splice(0..0, chunk);
887    }
888  } else {
889    let mut remaining = digs;
890    while !remaining.is_empty() {
891      if !result.is_empty() {
892        result.extend_from_slice(sep.unlist_ref());
893      }
894      for _ in 0..g {
895        if remaining.is_empty() {
896          break;
897        }
898        result.push(remaining.remove(0));
899      }
900    }
901  }
902
903  Tokens::new(result)
904}

906fn six_format_simplenumber(number: &SixNumber) -> Tokens {
907  if let SixNumber::Simple {
908    sign,
909    integer,
910    fraction,
911    decimal,
912  } = number
913  {
914    let grouping = six_get_choice_sym(six_pin!("group-digits"));
915    let mut tokens = Vec::new();
916    let mut trailer = Vec::new();
917
918    if let Some(s) = sign {
919      let s_str = s.to_string();
920      if s_str.contains('-') {
921        if let Some(Stored::Tokens(c)) = six_get_sym(six_pin!("negative-color")) {
922          tokens.push(T_BEGIN!());
923          tokens.push(T_CS!("\\color"));
924          tokens.push(T_BEGIN!());
925          tokens.extend(c.unlist());
926          tokens.push(T_END!());
927          trailer.insert(0, T_END!());
928        }
929        if six_get_bool_sym(six_pin!("bracket-negative-numbers")) {
930          tokens.extend(six_get_tokens_sym(six_pin!("open-bracket")).unlist());
931          let cb = six_get_tokens_sym(six_pin!("close-bracket"));
932          trailer.splice(0..0, cb.unlist());
933        } else {
934          tokens.extend_from_slice(s.unlist_ref());
935        }
936      } else if s_str.contains('+') && !six_get_bool_sym(six_pin!("retain-explicit-plus")) {
937        // drop +
938      } else {
939        tokens.extend_from_slice(s.unlist_ref());
940      }
941    }
942
943    if let Some(int) = integer {
944      let i = if grouping == "true" || grouping == "integer" {
945        six_groupdigits(int, 1)
946      } else {
947        int.clone()
948      };
949      tokens.extend(i.unlist());
950    }
951
952    let has_decimal = decimal.is_some();
953    let has_fraction = fraction.is_some();
954    if six_get_bool_sym(six_pin!("copy-decimal-marker")) {
955      if has_decimal {
956        if let Some(d) = decimal {
957          if !d.is_empty() {
958            tokens.extend_from_slice(d.unlist_ref());
959          } else {
960            tokens.extend(six_get_tokens_sym(six_pin!("output-decimal-marker")).unlist());
961          }
962        }
963      }
964    } else if has_fraction || has_decimal {
965      tokens.extend(six_get_tokens_sym(six_pin!("output-decimal-marker")).unlist());
966    }
967
968    if let Some(frac) = fraction {
969      let f = if grouping == "true" || grouping == "decimal" {
970        six_groupdigits(frac, -1)
971      } else {
972        frac.clone()
973      };
974      tokens.extend(f.unlist());
975    }
976
977    tokens.extend(trailer);
978
979    let meaning = six_number_string(number);
980    let meaning_toks = Tokenize!(&meaning);
981    let symbol = i_symbol(
982      &[("role", Tokenize!("NUMBER")), ("meaning", meaning_toks)],
983      None,
984    );
985    let wrapped = i_wrap(None, Tokens::new(tokens));
986    i_dual(
987      &[("revert_as", Tokenize!("presentation"))],
988      symbol,
989      wrapped,
990      vec![],
991    )
992    .unwrap_or_default()
993  } else {
994    Tokens::default()
995  }
996}

998fn six_format_infix(
999  op: Tokens,
1000  left: Option<Tokens>,
1001  right: Option<Tokens>,
1002  args: Vec<Tokens>,
1003) -> Tokens {
1004  let n = args.len();
1005  if n < 1 {
1006    return Tokens::default();
1007  }
1008  if n == 1 {
1009    return args.into_iter().next().unwrap();
1010  }
1011
1012  // Perl: I_apply({}, map { I_arg($_) } 1..n+1)
1013  // First arg is the operator, rest are operands
1014  let content = i_apply(
1015    &[],
1016    Tokens::new(vec![i_arg("1")]),
1017    (2..=n + 1)
1018      .map(|i| Tokens::new(vec![i_arg(&i.to_string())]))
1019      .collect(),
1020  );
1021
1022  let mut pres = Vec::new();
1023  if let Some(l) = &left {
1024    pres.extend_from_slice(l.unlist_ref());
1025  }
1026  pres.push(i_arg("2"));
1027  for i in 3..=n + 1 {
1028    pres.push(i_arg("1"));
1029    pres.push(i_arg(&i.to_string()));
1030  }
1031  if let Some(r) = &right {
1032    pres.extend_from_slice(r.unlist_ref());
1033  }
1034
1035  let mut all_args = vec![op];
1036  all_args.extend(args);
1037
1038  i_dual(
1039    &[("revert_as", Tokenize!("presentation"))],
1040    content,
1041    Tokens::new(pres),
1042    all_args,
1043  )
1044  .unwrap_or_default()
1045}

1054fn six_format_number_inner(number: &SixNumber, bracket: i32) -> Tokens {
1055  let bracket = if bracket > 0 && six_get_bool_sym(six_pin!("bracket-numbers")) {
1056    bracket
1057  } else {
1058    0
1059  };
1060
1061  match number {
1062    SixNumber::Simple { .. } => six_format_simplenumber(number),
1063    SixNumber::Operator { operator, arg1, arg2, .. } => match operator.as_str() {
1064      "uncertain" => {
1065        let a1 = arg1.as_deref();
1066        let a2 = arg2.as_deref();
1067        if a2.is_none() || six_get_bool_sym(six_pin!("omit-uncertainty")) {
1068          return a1
1069            .map(|n| six_format_number_inner(n, 0))
1070            .unwrap_or_default();
1071        }
1072        let fa1 = a1
1073          .map(|n| six_format_number_inner(n, 0))
1074          .unwrap_or_default();
1075        let fa2 = a2
1076          .map(|n| six_format_number_inner(n, 0))
1077          .unwrap_or_default();
1078        if six_get_bool_sym(six_pin!("separate-uncertainty")) {
1079          six_format_infix(Tokens::new(vec![T_CS!("\\pm")]), None, None, vec![fa1, fa2])
1080        } else {
1081          let open = six_get_tokens_sym(six_pin!("output-open-uncertainty"));
1082          let close = six_get_tokens_sym(six_pin!("output-close-uncertainty"));
1083          let mut tks = fa1.unlist();
1084          tks.extend(open.unlist());
1085          tks.extend(fa2.unlist());
1086          tks.extend(close.unlist());
1087          Tokens::new(tks)
1088        }
1089      },
1090      "complex" => {
1091        if let SixNumber::Operator { arg1, arg2, sign, symbol, .. } = number {
1092          let real = arg1.as_deref().map(|n| six_format_number_inner(n, 0));
1093          let imag = arg2.as_deref().map(|n| six_format_number_inner(n, 0));
1094          let i_tok = if six_get_bool_sym(six_pin!("copy-complex-root")) {
1095            symbol
1096              .clone()
1097              .unwrap_or_else(|| six_get_tokens_sym(six_pin!("output-complex-root")))
1098          } else {
1099            six_get_tokens_sym(six_pin!("output-complex-root"))
1100          };
1101
1102          let mut result = Vec::new();
1103          if let Some(r) = &real {
1104            result.extend_from_slice(r.unlist_ref());
1105          }
1106          if let Some(s) = sign {
1107            result.extend_from_slice(s.unlist_ref());
1108          }
1109          if let Some(im) = &imag {
1110            result.extend_from_slice(im.unlist_ref());
1111          }
1112          result.extend(i_tok.unlist());
1113          Tokens::new(result)
1114        } else {
1115          Tokens::default()
1116        }
1117      },
1118      "exponent" => {
1119        let a1 = arg1.as_deref();
1120        let a2 = arg2.as_deref();
1121
1122        if let Some(a2) = a2 {
1123          let has_content = a2.get_integer().is_some_and(|i| !i.is_empty())
1124            || a2.get_fraction().is_some_and(|f| !f.is_empty());
1125          if !six_get_bool_sym(six_pin!("retain-zero-exponent")) && !has_content {
1126            return a1
1127              .map(|n| six_format_number_inner(n, 0))
1128              .unwrap_or_default();
1129          }
1130        }
1131
1132        let fa1 = a1
1133          .map(|n| six_format_number_inner(n, 1))
1134          .unwrap_or_default();
1135        let base = six_get_tokens_sym(six_pin!("exponent-base"));
1136        let fa2 = a2
1137          .map(|n| six_format_number_inner(n, 0))
1138          .unwrap_or_default();
1139        let power = i_superscript(&[("operator_meaning", Tokenize!("power"))], base, fa2);
1140
1141        let has_mantissa = a1.is_some_and(|n| {
1142          n.get_integer().is_some() || n.get_fraction().is_some() || n.is_operator()
1143        });
1144        let times = if has_mantissa {
1145          six_get_op_sym(
1146            &[
1147              ("role", Tokenize!("MULOP")),
1148              ("meaning", Tokenize!("times")),
1149            ],
1150            six_pin!("exponent-product"),
1151          )
1152        } else {
1153          Tokens::new(vec![T_CS!("\\lx@InvisibleTimes")])
1154        };
1155
1156        six_format_infix(
1157          times,
1158          if bracket > 1 {
1159            Some(six_get_tokens_sym(six_pin!("open-bracket")))
1160          } else {
1161            None
1162          },
1163          if bracket > 1 {
1164            Some(six_get_tokens_sym(six_pin!("close-bracket")))
1165          } else {
1166            None
1167          },
1168          vec![fa1, power],
1169        )
1170      },
1171      "comparator" => {
1172        if let SixNumber::Operator { comparator, .. } = number {
1173          let comp = comparator.clone().unwrap_or_default();
1174          let a1 = arg1
1175            .as_deref()
1176            .map(|n| six_format_number_inner(n, 0))
1177            .unwrap_or_default();
1178          let mut tks = comp.unlist();
1179          tks.extend(a1.unlist());
1180          i_wrap(None, Tokens::new(tks))
1181        } else {
1182          Tokens::default()
1183        }
1184      },
1185      "product" => {
1186        let fa1 = arg1
1187          .as_deref()
1188          .map(|n| six_format_number_inner(n, 1))
1189          .unwrap_or_default();
1190        let fa2 = arg2
1191          .as_deref()
1192          .map(|n| six_format_number_inner(n, 1))
1193          .unwrap_or_default();
1194        let times = six_get_op_sym(
1195          &[
1196            ("role", Tokenize!("MULOP")),
1197            ("meaning", Tokenize!("times")),
1198          ],
1199          six_pin!("output-product"),
1200        );
1201        six_format_infix(times, None, None, vec![fa1, fa2])
1202      },
1203      "quotient" => {
1204        let fa1 = arg1
1205          .as_deref()
1206          .map(|n| six_format_number_inner(n, 1))
1207          .unwrap_or_default();
1208        let fa2 = arg2
1209          .as_deref()
1210          .map(|n| six_format_number_inner(n, 2))
1211          .unwrap_or_default();
1212        if six_get_choice_sym(six_pin!("quotient-mode")) == "fraction" {
1213          let frac = six_get_tokens_sym(six_pin!("fraction-function"));
1214          let mut tks = frac.unlist();
1215          tks.push(T_BEGIN!());
1216          tks.extend(fa1.unlist());
1217          tks.push(T_END!());
1218          tks.push(T_BEGIN!());
1219          tks.extend(fa2.unlist());
1220          tks.push(T_END!());
1221          Tokens::new(tks)
1222        } else {
1223          let div = six_get_op_sym(
1224            &[
1225              ("role", Tokenize!("MULOP")),
1226              ("meaning", Tokenize!("divide")),
1227            ],
1228            six_pin!("output-quotient"),
1229          );
1230          six_format_infix(div, None, None, vec![fa1, fa2])
1231        }
1232      },
1233      other => {
1234        // Perl siunitx.sty.ltxml:642 — Error('unexpected', $op, undef,
1235        //   "Unrecognized operator $op in siunitx number")
1236        six_log_error!(
1237          "unexpected", other,
1238          "Unrecognized operator {} in siunitx number", other
1239        );
1240        Tokens::default()
1241      },
1242    },
1243  }
1244}

1383/// Parse unit definitions into structured units
1384fn six_parse_units(defns: Vec<SixUnitDefn>) -> Vec<SixUnit> {
1385  let mut units = Vec::new();
1386  let sticky_per = six_get_bool_sym(six_pin!("sticky-per"));
1387  let mut saved_per = false;
1388  // Consume defns as an iterator of Options — each slot holds the
1389  // SixUnitDefn until we move it into a SixUnit role slot. This skips
1390  // the per-role `.clone()` pass the prior `&[SixUnitDefn]` signature
1391  // forced (6 × N clones per unit block, each cloning 3 Strings + 3
1392  // optional allocations). Cost drops to zero allocations for the
1393  // role-binding phase.
1394  let mut slots: Vec<Option<SixUnitDefn>> = defns.into_iter().map(Some).collect();
1395  let mut idx = 0;
1396
1397  while idx < slots.len() {
1398    let mut unit = SixUnit::default();
1399
1400    for role in &[
1401      "per",
1402      "prepower",
1403      "prefix",
1404      "unit",
1405      "qualifier",
1406      "postpower",
1407    ] {
1408      while idx < slots.len() {
1409        let r_matches = slots[idx].as_ref().map(|d| d.unit_type.as_str()) == Some(*role);
1410        let r_is_style = slots[idx].as_ref().map(|d| d.unit_type.as_str()) == Some("style");
1411        if r_matches {
1412          let d = slots[idx].take().unwrap();
1413          match *role {
1414            "per" => unit.per = true,
1415            "prepower" => unit.prepower = Some(d),
1416            "prefix" => unit.prefix = Some(d),
1417            "unit" => unit.unit = Some(d),
1418            "qualifier" => unit.qualifier = Some(d),
1419            "postpower" => unit.postpower = Some(d),
1420            _ => {},
1421          }
1422          idx += 1;
1423          break;
1424        } else if r_is_style {
1425          let d = slots[idx].take().unwrap();
1426          if unit.prefix.is_none() && unit.unit.is_none() {
1427            if d.name == "cancel" {
1428              unit.cancel = true;
1429            }
1430            if d.name == "highlight" {
1431              unit.highlight_color = d.color;
1432            }
1433          }
1434          idx += 1;
1435        } else {
1436          break;
1437        }
1438      }
1439    }
1440
1441    if unit.unit.is_none()
1442      && unit.prefix.is_none()
1443      && !unit.per
1444      && unit.prepower.is_none()
1445      && unit.postpower.is_none()
1446    {
1447      if idx < slots.len() {
1448        idx += 1;
1449      }
1450      continue;
1451    }
1452
1453    if saved_per {
1454      unit.per = true;
1455    } else if sticky_per && unit.per {
1456      saved_per = true;
1457    }
1458
1459    units.push(unit);
1460  }
1461
1462  units
1463}

1465/// Format a single unit
1466fn six_format_1unit(unit: &SixUnit) -> Tokens {
1467  // Resolve presentations: replace unit/prefix CS references with their final text
1468  let pre_resolved = unit
1469    .prefix
1470    .as_ref()
1471    .map(|p| resolve_unit_presentation(&p.presentation));
1472  let u_resolved = unit
1473    .unit
1474    .as_ref()
1475    .map(|u| resolve_unit_presentation(&u.presentation));
1476  let mut p = unit
1477    .prepower
1478    .as_ref()
1479    .and_then(|pp| pp.power.clone())
1480    .or_else(|| unit.postpower.as_ref().and_then(|pp| pp.power.clone()));
1481
1482  if unit.per {
1483    p = Some(if let Some(pp) = p {
1484      let mut tks = vec![T_OTHER!("-")];
1485      tks.extend(pp.unlist());
1486      Tokens::new(tks)
1487    } else {
1488      Tokens::new(vec![T_OTHER!("-"), T_OTHER!("1")])
1489    });
1490  }
1491
1492  // Build unit name for meaning
1493  let unit_name = format!(
1494    "{}{}",
1495    unit.prefix.as_ref().map(|p| p.name.as_str()).unwrap_or(""),
1496    unit.unit.as_ref().map(|u| u.name.as_str()).unwrap_or(""),
1497  );
1498
1499  // Build presentation: \mathrm{resolved prefix + resolved unit}.
1500  //
1501  // Use `mouth::tokenize` (re-parses through the std catcode table)
1502  // rather than `ExplodeText!` (turns every char into an OTHER token).
1503  // For a prefix like `\micro` whose `resolve_unit_presentation` returns
1504  // the string "\SIUnitSymbolMicro", `ExplodeText!` would produce 17
1505  // literal OTHER tokens (\, S, I, U, n, i, t, S, y, m, b, o, l, M,
1506  // i, c, r, o), losing CS-ness and rendering as the literal text
1507  // `\SIUnitSymbolMicro` in the math output. With `mouth::tokenize`
1508  // we get a single CS token that downstream digestion can re-expand
1509  // to the µ glyph. Witness: 1410.8171 used to render `\SI{0,1}{\micro
1510  // \kelvin}` as the literal text `\SIUnitSymbolMicroK` in math.
1511  // Prefix and unit are tokenized separately (same as pre_resolved /
1512  // u_resolved are computed separately), so the prefix→unit boundary
1513  // is preserved at the Token level even though both strings happen
1514  // to be ASCII.
1515  let mut pres_inner = Vec::new();
1516  if let Some(pr) = &pre_resolved {
1517    pres_inner.extend(mouth::tokenize(pr).unlist());
1518  }
1519  if let Some(ut) = &u_resolved {
1520    pres_inner.extend(mouth::tokenize(ut).unlist());
1521  }
1522
1523  // \lx@unit{name}{\mathrm{presentation}}
1524  let mut result_tks = vec![T_CS!("\\lx@unit"), T_BEGIN!()];
1525  result_tks.extend(ExplodeText!(&unit_name));
1526  result_tks.push(T_END!());
1527  result_tks.push(T_BEGIN!());
1528  result_tks.push(T_CS!("\\mathrm"));
1529  result_tks.push(T_BEGIN!());
1530  result_tks.extend(pres_inner);
1531  result_tks.push(T_END!());
1532  result_tks.push(T_END!());
1533
1534  let mut result = Tokens::new(result_tks);
1535
1536  // Apply power
1537  if let Some(pp) = p {
1538    let mut tks = vec![T_CS!("\\lx@power"), T_BEGIN!()];
1539    tks.extend(result.unlist());
1540    tks.push(T_END!());
1541    tks.push(T_BEGIN!());
1542    tks.extend(pp.unlist());
1543    tks.push(T_END!());
1544    result = Tokens::new(tks);
1545  }
1546
1547  if unit.cancel {
1548    let mut tks = vec![T_CS!("\\cancel"), T_BEGIN!()];
1549    tks.extend(result.unlist());
1550    tks.push(T_END!());
1551    result = Tokens::new(tks);
1552  }
1553
1554  result
1555}

1650fn six_parse_literalunits(expr: &Tokens) -> Tokens {

1741/// Replace \lx@six@unitobject{name} tokens with \mathrm{presentation} tokens.
1742/// Must be called while siunitx_macros mapping is active (inside \SI{}{} processing).
1743fn six_resolve_unit_objects(tokens: &Tokens) -> Tokens {
1744  let mut result = Vec::new();
1745  let mut iter = tokens.unlist_ref().iter().copied().peekable();
1746  let mut had_substitution = false;
1747  // Pre-intern the two dispatch CS names so the per-token check is
1748  // u32 equality (not `t.to_string()` alloc + string compare).
1749  let unitobject_sym = pin!("\\lx@six@unitobject");
1750  let unitobject_arg_sym = pin!("\\lx@six@unitobject@arg");
1751
1752  while let Some(t) = iter.next() {
1753    if t.text == unitobject_sym {
1754      if let Some(name) = read_brace_group_str(&mut iter) {
1755        if let Some(Stored::String(encoded)) =
1756          state::lookup_mapping_sym(pin!("siunitx_macros"), &name)
1757        {
1758          // Decode directly from the arena-borrowed &str — was
1759          // cloning via `.to_string()` into a temporary String.
1760          if let Some(defn) = decode_unit_defn_from_encoded_sym(&name, encoded) {

1764              //   `return Tokens(T_CS('\mathrm'), T_BEGIN, $pres, T_END);`

1767              // prepower/postpower presentations like `^{3}` (from `\cubic`)
1768              // bare, which the math digester then sees as a second
1769              // superscript on the preceding atom (driver: 2304.12803).

1772              result.extend(pres.unlist());

1776            }

1778        }

1783    } else if t.text == unitobject_arg_sym {

1785        if let Some(arg) = read_brace_group_str(&mut iter) {
1786          if let Some(Stored::String(encoded)) =
1787            state::lookup_mapping_sym(pin!("siunitx_macros"), &name)
1788          {
1789            if let Some(defn) = decode_unit_defn_from_encoded_sym(&name, encoded) {
1790              let pres = defn.presentation;

1794                // `\tothe{2}` → `^{2}`). The presentation already supplies
1795                // its own grouping so we don't add a `\mathrm{...}` wrapper.

1799                result.push(T_END!());

1801                continue;
1802              }
1803            }
1804          }
1805        }
1806      }

1808      result.push(t);
1809    }
1810  }
1811
1812  if had_substitution {
1813    Tokens::new(result)
1814  } else {

1816  }

1828  let dot_sym = pin!(".");
1829
1830  while let Some(t) = iter.peek() {
1831    if t.text == unitobject_sym {
1832      iter.next();
1833      // Read {name} group
1834      if let Some(name) = read_brace_group_str(&mut iter) {
1835        // Look up in siunitx_macros mapping
1836        if let Some(Stored::String(encoded)) =
1837          state::lookup_mapping_sym(pin!("siunitx_macros"), &name)
1838        {
1839          if let Some(defn) = decode_unit_defn_from_encoded_sym(&name, encoded) {
1840            defns.push(defn);
1841          }
1842        }
1843      }
1844    } else if t.text == unitobject_arg_sym {
1845      iter.next();
1846      if let Some(name) = read_brace_group_str(&mut iter) {
1847        if let Some(arg) = read_brace_group_str(&mut iter) {
1848          if let Some(Stored::String(encoded)) =
1849            state::lookup_mapping_sym(pin!("siunitx_macros"), &name)
1850          {
1851            if let Some(mut defn) = decode_unit_defn_from_encoded_sym(&name, encoded) {
1852              // Apply arg to the appropriate field
1853              if defn.unit_type == "postpower" || defn.unit_type == "prepower" {
1854                defn.power = Some(Tokenize!(&arg));
1855              } else if defn.unit_type == "qualifier" {
1856                defn.presentation = Tokenize!(&arg);
1857              }
1858              defns.push(defn);
1859            }
1860          }
1861        }
1862      }
1863    } else if t.get_catcode() == Catcode::SPACE || t.text == dot_sym {
1864      iter.next(); // skip spaces and dots (unit product separators)
1865    } else if !defns.is_empty() {
1866      // Non-unit content after some units found — stop here, use what we have.
1867      // Perl handles mixed content (e.g., \pi\per\milli\meter) by parsing units
1868      // and passing non-unit content through. We stop at the first unrecognized token.
1869      break;
1870    } else {
1871      return None; // No units found yet — fall back to literal
1872    }
1873  }
1874
1875  Some(defns)
1876}

1878/// Read a brace-delimited group and return its string content
1879fn read_brace_group_str<I: Iterator<Item = Token>>(
1880  iter: &mut std::iter::Peekable<I>,
1881) -> Option<String> {
1882  if let Some(t) = iter.peek() {
1883    if t.get_catcode() == Catcode::BEGIN {
1884      iter.next();
1885      let mut result = String::new();
1886      let mut level = 1;
1887      for t in iter.by_ref() {
1888        if t.get_catcode() == Catcode::END {
1889          level -= 1;
1890          if level == 0 {
1891            break;
1892          }
1893        } else if t.get_catcode() == Catcode::BEGIN {
1894          level += 1;
1895        }
1896        // `.with_str` borrows the arena entry directly — no per-token
1897        // String alloc (the prior `&t.to_string()` was ~30 ns/token).
1898        t.with_str(|s| result.push_str(s));
1899      }
1900      return Some(result);
1901    }
1902  }
1903  None
1904}

1906// Cache of decoded SixUnitDefn "template" (everything except the
1907// caller-provided `name`) keyed by the encoded string's interned
1908// SymStr. Every \SI / \num / \si invocation re-decodes the same
1909// unit definitions (e.g. `\metre`, `\kilo`) for every usage — the
1910// raw decode allocates 3 Strings + 2 Token vectors each time.
1911// Caching by the encoded SymStr is automatically invalidation-safe:
1912// if a \DeclareSIUnit redefines `\metre`, the new encoded string
1913// interns to a different SymStr, so the old cache entry is simply
1914// never looked up again.
1915thread_local! {
1916  static UNIT_DEFN_CACHE: std::cell::RefCell<
1917    rustc_hash::FxHashMap<::latexml_core::common::arena::SymStr, SixUnitDefn>,
1918  > = std::cell::RefCell::new(rustc_hash::FxHashMap::default());
1919}

1921/// Decode a SixUnitDefn from "type|presentation|power|base" format.
1922/// Takes the already-interned `encoded_sym` so we can cache-lookup
1923/// without re-pinning. `name` is the caller's lookup key (not part of
1924/// the encoded value).
1925fn decode_unit_defn_from_encoded_sym(
1926  name: &str,
1927  encoded_sym: ::latexml_core::common::arena::SymStr,
1928) -> Option<SixUnitDefn> {
1929  // Fast path: cache hit by the encoded string's SymStr.
1930  if let Some(cached) = UNIT_DEFN_CACHE.with(|c| c.borrow().get(&encoded_sym).cloned()) {
1931    let mut defn = cached;
1932    defn.name = name.to_string();
1933    return Some(defn);
1934  }
1935
1936  let defn = ::latexml_core::common::arena::with(encoded_sym, |encoded| {
1937    let parts: Vec<&str> = encoded.splitn(4, '|').collect();
1938    if parts.is_empty() {
1939      return None;
1940    }
1941    Some(SixUnitDefn {
1942      name:         name.to_string(),
1943      unit_type:    parts[0].to_string(),
1944      presentation: Tokenize!(parts.get(1).unwrap_or(&"")),
1945      power:        parts.get(2).and_then(|p| {
1946        if p.is_empty() {
1947          None
1948        } else {
1949          Some(Tokenize!(p))
1950        }
1951      }),
1952      base:         parts.get(3).and_then(|b| b.parse().ok()),
1953      arg:          None,
1954      color:        None,
1955    })
1956  })?;
1957  UNIT_DEFN_CACHE.with(|c| c.borrow_mut().insert(encoded_sym, defn.clone()));
1958  Some(defn)
1959}

2001/// Resolve unit presentation: if the presentation contains CS tokens that are
2002/// known unit macros, replace them with their stored presentation text.
2003/// E.g., \DeclareSIUnit \metre { \meter } → resolve \meter → "m"
2004fn resolve_unit_presentation(pres: &Tokens) -> String {
2005  let mut result = String::new();
2006  for tok in pres.unlist_ref() {
2007    if tok.get_catcode() == Catcode::CS {
2008      let cs_name = tok.to_string();
2009      let unit_name = cs_name.trim_start_matches('\\');
2010      // Look up in siunitx_macros mapping
2011      if let Some(Stored::String(encoded)) =
2012        state::lookup_mapping_sym(pin!("siunitx_macros"), unit_name)
2013      {
2014        let encoded_str = arena::with(encoded, |s| s.to_string());
2015        let parts: Vec<&str> = encoded_str.splitn(4, '|').collect();
2016        if parts.len() >= 2 {
2017          let pres = parts[1];
2018          // If the presentation itself contains backslashes, try to resolve recursively
2019          if pres.contains('\\') {
2020            let sub = Tokenize!(pres);
2021            let sub_resolved = resolve_unit_presentation(&sub);
2022            result.push_str(&sub_resolved);
2023          } else {
2024            result.push_str(pres);
2025          }
2026          continue;
2027        }
2028      }
2029      // Not a unit macro — keep as-is
2030      result.push_str(&cs_name);
2031    } else if tok.get_catcode() != Catcode::SPACE {
2032      result.push_str(&tok.to_string());
2033    }
2034  }
2035  result
2036}
