1use crate::prelude::*;
2
3LoadDefinitions!({
4  // Redefine \documentclass to do nothing in subfiles
5  DefMacro!("\\documentclass OptionalSemiverbatim SkipSpaces Semiverbatim []", "");
6
7  // Fake \begin{document}/\end{document} for subfiles:
8  // After the main document starts, nested \begin{document} just opens/closes
9  // an internal_vertical mode group, keeping the rest of the main file intact.
10  DefPrimitive!("\\lx@subfiles@fake@begindocument", {
11    let nesting = lookup_int("subfiles_nesting") + 1;
12    AssignValue!("subfiles_nesting" => nesting, Some(Scope::Global));
13    // Make \end{document} close this fake \begin{document}
14    Let!("\\end{document}", "\\lx@subfiles@fake@enddocument", Scope::Global);
15    stomach::begin_mode("internal_vertical")?;
16  });
17
18  DefPrimitive!("\\lx@subfiles@fake@enddocument", {
19    let nesting = lookup_int("subfiles_nesting") - 1;
20    AssignValue!("subfiles_nesting" => nesting, Some(Scope::Global));
21    // Make sure the outermost \end{document} invokes our proper \end{document}
22    if nesting == 0 {
23      Let!("\\end{document}", "\\lx@subfiles@saved@enddocument", Scope::Global);
24    }
25    stomach::end_mode("internal_vertical")?;
26  });
27
28  // AtBeginDocument: after the main document starts, redirect nested
29  // \begin{document}/\end{document} to the fake versions
30  let _ = state::push_value("@at@begin@document",
31    Tokens!(
32      T_CS!("\\lx@subfiles@setup")
33    ));
34  DefPrimitive!("\\lx@subfiles@setup", {
35    AssignValue!("subfiles_nesting" => 0i64, Some(Scope::Global));
36    Let!("\\lx@subfiles@saved@enddocument", "\\end{document}", Scope::Global);
37    Let!("\\begin{document}", "\\lx@subfiles@fake@begindocument", Scope::Global);
38  });
39
40  // Define \subfile to be \input
41  Let!("\\subfile", "\\input");
42});
