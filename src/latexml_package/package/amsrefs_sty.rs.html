51  // properties closure patches old-style "12345 \# 67" → "12345:67" and
52  // emits both mr= and href= AMS lookup URL. Ported directly with
53  // a regex for the \# substitution.
54  DefConstructor!("\\MR{}",
55    "<ltx:ref href='#href' class='ltx_mathreviews'>MathReviews</ltx:ref>",
56    enter_horizontal => true,
57    properties => sub[args] {
58      let raw = args[0].as_ref().map(|a| a.to_string()).unwrap_or_default();
59      static RE: std::sync::OnceLock<regex::Regex> = std::sync::OnceLock::new();
60      let re = RE.get_or_init(|| regex::Regex::new(r"\s+\\#\s*").unwrap());
61      let mr = re.replace(&raw, ":").to_string();
62      let href = format!("http://www.ams.org/mathscinet-getitem?mr={}", mr);
63      Ok(stored_map!("mr" => mr, "href" => href))
64    });
