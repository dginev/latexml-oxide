1//! psfrag.sty — PostScript fragment overlays on EPS images
2//! Perl: psfrag.sty.ltxml — 166 lines
3//! Stores psfrag commands for later use when including EPS graphics.
4//! The actual overlay is done by LaTeX (we just preserve the fragments).
5use crate::prelude::*;
6
7#[rustfmt::skip]
8LoadDefinitions!({
9  // Perl L25-27 — initial state: psfrag_scan_all defaults to the
10  // 2.09_COMPATIBILITY flag (true when the document used \documentstyle),
11  // psfrag_scan starts off. No reader consumes these in Rust yet, but
12  // setting them at load time keeps state shape consistent with Perl
13  // so the future \includegraphics hook can LookupValue them directly.
14  AssignValue!("psfrag_scan_all" => state::lookup_bool("2.09_COMPATIBILITY"));
15  AssignValue!("psfrag_scan"     => 0i32);
16
17  // Options — Perl L28-32. Each declared option toggles psfrag_scan_all:
18  //   209mode, scanall → true
19  //   2emode           → false
20  DeclareOption!("209mode", { AssignValue!("psfrag_scan_all" => true); });
21  DeclareOption!("2emode",  { AssignValue!("psfrag_scan_all" => false); });
22  DeclareOption!("scanall", { AssignValue!("psfrag_scan_all" => true); });
23  ProcessOptions!();
24
25  // \psfrag — stores fragment for later overlay — Perl L46-55
26  // NOT a constructor since args should not be digested yet
27  DefPrimitive!("\\psfrag OptionalMatch:* Semiverbatim [][][][]{}", None);
28  // Perl L55: alias => '\psfrag' so the reversion `tex=` attribute
29  // serializes back to the author-facing `\psfrag` rather than the
30  // internal `\lx@delayed@psfrag` name.
31  DefConstructor!("\\lx@delayed@psfrag OptionalMatch:* Semiverbatim [][][][]{}", "",
32    alias => "\\psfrag");
33
34  // Scan control — Perl L57-64.
35  // Perl DefConstructor(...afterDigest {save_psfrag(cs); AssignValue(psfrag_scan=>0/1)});
36  // Rust implements the state toggle (psfrag_scan int). save_psfrag()
37  // would append to saved_psfragments, but the Rust \includegraphics
38  // hook doesn't consult that list yet (see the L24-28 TODO note), so
39  // skipping it is not observable. When that hook lands, extend these
40  // to also append the CS invocation to saved_psfragments.
41  //
42  // Intentional divergence (WISDOM #44 class: afterDigest-only no-op
43  // side effect): both \psfragscanon/off Perl DefConstructors emit
44  // nothing to the DOM — their entire effect is an afterDigest
45  // AssignValue. DefPrimitive is the idiomatic Rust shape for a
46  // pure-state-toggle CS with no XML emission; DefConstructor would
47  // only add an empty constructor string and be strictly equivalent.
48  // Audit counts 2 DefConstructor → DefPrimitive flips.
49  DefPrimitive!("\\psfragscanon", {
50    AssignValue!("psfrag_scan" => 1i32);
51  });
52  DefPrimitive!("\\psfragscanoff", {
53    AssignValue!("psfrag_scan" => 0i32);
54  });
55
56  // The Perl version hooks into \includegraphics and \epsfbox to check
57  // if the image is an EPS that needs psfrag processing, and if so,
58  // wraps it in a <ltx:picture> with the TeX overlay.
59  // This requires image type detection (psfrag_requirepicture) which
60  // we don't have. For now, includegraphics works normally without overlay.
61
62  // Rescan/debug macros — Perl psfrag.sty.ltxml.
63  // NOTE: do NOT define `\tex`. Real psfrag.sty does NOT define `\tex` as a
64  // LaTeX macro (its only `\tex` is `\string\\tex…` written verbatim into the
65  // .ps file for PostScript-side processing, psfrag.sty L201), and neither
66  // does Perl LaTeXML's psfrag binding. A prior Rust-only
67  // `DefMacro!("\\tex Semiverbatim", "#1")` intercepted papers that load
68  // psfrag but `\newcommand{\tex}{…}` for their own purposes — e.g.
69  // 1505.01267 does `\newcommand{\tex}{\textstyle}` and writes
70  // `\tex{…c_0…\mbox{as $\hat s\to\infty$}…}` inside display math. With the
71  // Rust `\tex` winning (newcommand ignored) and reading its arg Semiverbatim,
72  // `_` was neutralized to OTHER (literal `\_`) and the inline-math `\to`
73  // built an `ltx:XMApp` that leaked into the surrounding `ltx:text` →
74  // "ltx:XMApp isn't allowed in <ltx:text>". Leaving `\tex` undefined (Perl-
75  // faithful) lets the author's `\newcommand` win, so `\tex{…}`=`\textstyle{…}`
76  // digests as math.
77  def_macro_noop("\\psfragrescan")?;
78  def_macro_noop("\\psfragrescanoff")?;
79  def_macro_noop("\\psfragrescanon")?;
80  def_macro_noop("\\psfragdebugon")?;
81  def_macro_noop("\\psfragdebugoff")?;
82
83  // Perl psfrag.sty.ltxml L149: DefEnvironment('{psfrags}', '#body').
84  // Pure grouping, no content transform. Previously unported.
85  DefEnvironment!("{psfrags}", "#body");
86});
