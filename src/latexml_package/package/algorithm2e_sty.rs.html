23  // port only `DefEnvironment`'d `{algorithm}` and `\let`-aliased the others to
24  // `\algorithm`. That breaks when the `algorithm` (floats) package is ALSO
25  // loaded: it raw-defines a `{algorithm*}` (two-column float) environment, and
26  // a bare `\let\algorithm*\algorithm` leaves the env's name registration as
27  // `algorithm`, so `\begin{algorithm*}` opens the float-package's paragraph
28  // wrapper while algorithm2e's listing machinery runs inside it — the

30  // ("ltx:listingline isn't allowed in <ltx:text>"). A proper DefEnvironment
31  // for each name (matching Perl) registers `algorithm*` as its own listing

208  // Indentation prepending — Perl L197-208.
209  // Perl's `\lx@prepend@indentation@{}` does `$doc->floatToElement('ltx:tags')`
210  // FIRST, then prepends the indentation. That `floatToElement('ltx:tags')` is
211  // critical structurally: it repositions the cursor UP to the listingline,
212  // OUT of any open inline box — notably the `_noautoclose` `<ltx:text>` an
213  // `\hbox` opens when an algorithm2e listing is wrapped in `\colorbox{…}{…}`
214  // (→ `\hbox{…}`). With the cursor back at the listingline, the immediately
215  // following `\lx@algo@@endline` (`</ltx:listingline>`) closes cleanly.
216  //
217  // The previous Rust port emitted indentation at `\lx@algo@startline` instead
218  // and stubbed this as an EMPTY constructor "to avoid DOM manipulation" — but
219  // that dropped the reposition, so a listing inside an `\hbox`/`\colorbox`
220  // left the cursor inside the box's `_noautoclose` `<ltx:text>` and closing
221  // the listingline errored: "ltx:listingline … whose open descendents do not
222  // auto-close. Descendants are text". We keep Rust's startline-indentation
223  // approach (so we deliberately do NOT re-absorb `#1` here — that would double
224  // the indent), but restore Perl's cursor-repositioning float. Witnesses
225  // 1911.01815, 1903.04631 (algorithm2e inside `\colorbox`/`\hbox`).
226  DefMacro!("\\lx@prepend@indentation", "\\lx@prepend@indentation@{\\the\\lx@algo@indentation}");

233  //
234  // The earlier Rust port used the plain L195 template
235  // (`<ltx:tags><ltx:tag>#1</ltx:tag></ltx:tags>`), which emits the line-number
236  // tags at the CURRENT cursor. That errors "ltx:tags isn't allowed in
237  // <ltx:text>" whenever `\nl` fires while an inline `<ltx:text>` is open — e.g.
238  // a `\SetKwInput{KwInit}{\nl initialize}` line, where the KwInput label
239  // wrapper opens an `<ltx:text>` before `\nl`. Perl's active definition first
240  // `floatToElement('ltx:tags')` to climb OUT of that `<ltx:text>` up to the
241  // enclosing `<ltx:listingline>` (which can contain tags), emits the tags
242  // there, then leaves the cursor restored so following content flows on. This
243  // is the same float used by `\lx@prepend@indentation@` above. We restore the
244  // saved node (rather than Perl's manual childNode remove/re-append prepend)
245  // so the KwInput label's content keeps its wrapper. Witness 2104.02680
246  // (`\SetKwInput{KwInit}{\nl initialize}`).
247  DefConstructor!("\\algocf@printnl{}", sub[document, args] {
248    let num = args.first().and_then(|a| a.as_ref());
249    let savenode = document.float_to_element("ltx:tags", false)?;
250    document.open_element("ltx:tags", None, None)?;
251    match num {
252      Some(n) => { document.insert_element("ltx:tag", vec![n], None)?; },
253      None => { document.insert_element("ltx:tag", Vec::new(), None)?; },
254    }
255    document.close_element("ltx:tags")?;
256    if let Some(sn) = savenode { document.set_node(&sn); }
257  });
