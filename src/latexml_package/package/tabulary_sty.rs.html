10  // redefinition from replacing the tabulary→alignment trampoline
11  // (the raw tabulary.sty itself does this; the Perl lock prevents us
12  // from losing the binding when the raw sty is loaded alongside).
13  DefMacro!("\\tabulary{}[]{}",
14    "\\@tabular@bindings{#3}[vattach=#2,width=#1]\\@@tabulary{#1}[#2]{#3}\\lx@begin@alignment",
