1//! pdftexcmds.sty — pdfTeX utility commands
2//! Perl: pdftexcmds.sty.ltxml
3//! Everything is in pdfTeX.pool already; just require iftex.
4use crate::prelude::*;
5
6#[rustfmt::skip]
7LoadDefinitions!({
8  RequirePackage!("iftex");
9  // Stubs for the pdftex-primitive wrappers that bmpsize (TL L51-53)
10  // and other oberdiek packages probe via `\ifx\csname pdf@filedump
11  // \endcsname\relax`. The raw load of pdftexcmds.sty would create
12  // these conditionally; we bind instead so define defensively.
13  // Witness 2406.02536, 2406.03347 (bmpsize "pdfTeX 1.30 or newer").
14  def_macro_noop("\\pdf@filedump{}{}{}")?;
15  def_macro_noop("\\pdf@mdfivesum{}")?;
16  def_macro_noop("\\pdf@filemdfivesum{}")?;
17  DefMacro!("\\pdf@filesize{}", "0");
18  def_macro_noop("\\pdf@filemoddate{}")?;
19  DefMacro!("\\pdf@strcmp{}{}", "0");
20  // \pdf@shellescape returns the shell-escape level (0 = disabled).
21  // Used by packages probing whether \write18 is available. We don't
22  // execute shell commands; return 0.
23  DefMacro!("\\pdf@shellescape", "0");
24  // Other common pdftexcmds wrappers — return safe defaults.
25  def_macro_noop("\\pdf@unescapehex{}")?;
26  DefMacro!("\\pdf@escapestring{}", "#1");
27  DefMacro!("\\pdf@escapename{}", "#1");
28  DefMacro!("\\pdf@escapehex{}", "#1");
29  def_macro_noop("\\pdf@primitive{}")?;
30});
