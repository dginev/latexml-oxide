1//! `line` font encoding — the LaTeX picture-mode line fonts (`line10`,
2//! `linew10`), declared by `\font...=line10` and reached through
3//! `Font.pm`-table family `line`/`linew` → encoding `line`.
4//!
5//! **Why this map exists (root cause of a canvas OOM cluster):** LaTeX-2.09-era
6//! plain-TeX documents (math0102053, math0102089, math0212126, math0504436,
7//! math0506088, math0604321, …) inline picture mode's `\@sline`, whose drawing
8//! loop advances by the width of an `\hbox{\@linefnt\@getlinechar(x,y)}`:
9//!
10//! ```tex
11//! \@clnwd=\wd\@linechar
12//! \@whiledim \@clnwd <\@linelen \do {…\advance\@clnwd \wd\@linechar}
13//! ```
14//!
15//! Real TeX gets nonzero char widths (2.5–10 pt) from `line10.tfm`, so the loop
16//! terminates. Without a fontmap, `FontDecode` drops the char → empty box →
17//! width 0 pt → the `\ifdim` NEVER flips → unbounded box accumulation (~1.9 M
18//! boxes → 4.5 GB RSS). Perl LaTeXML ships no `line` fontmap
19//! (`Info:fontmap:line Couldn't find fontmap`) and OOMs identically — this map
20//! is a surpass-Perl reliability fix THROUGH the architecture's own mechanism
21//! (no control-flow divergence; Perl with the same map would behave the same).
22//! Modern latex.ltx even guards this exact hazard (`\ifdim\wd\@linechar=\z@
23//! \setbox\@linechar\hbox{.}\@badlinearg\fi`, latex.ltx ~L13777) — but these
24//! old documents inline the unguarded 2.09 macros, so the FONT's width is the
25//! only lever that reaches them.
26//!
27//! **Slot layout** (kernel `\@getlinechar`, latex.ltx L13815): segment slots
28//! are `8x-9+y` for rising `\line(x,y)` (y>0) and `8x-9+|y|+16` for falling
29//! (y<0), x,y ∈ 1..6 coprime; `\@getlarrow`/`\@getrarrow` put arrowheaded
30//! variants at 27 ('33, horizontal left head), 54 ('66, up head used by
31//! `\@upvector`), 63 ('77, down head used by `\@downvector`) and across the
32//! letter range (`16x-9±2y`, +64 for the falling/left set). The rising/falling
33//! bands overlap structurally, so a glyph-exact map is impossible — we choose a
34//! plausible diagonal per band. The semantically ESSENTIAL property is that
35//! every populated slot maps to a NONZERO-width glyph; the picture itself is an
36//! approximation either way (hand-rolled `\raise`/`\hskip` positioning).

84#[cfg(test)]
85mod tests {
86  use std::process::Command;
87
88  use super::LINE_SLOTS;
89  use crate::package::lcircle_fontmap::LCIRCLE_SLOTS;
90
91  /// The TFM-populated slots of a picture font, parsed from `tftopl` output
92  /// (lines like `(CHARACTER O 27` / `(CHARACTER C a`). Returns None when the
93  /// host TeX tree lacks the font or `tftopl` (test self-skips).
94  fn tfm_slots(font: &str) -> Option<Vec<usize>> {
95    let tfm = Command::new("kpsewhich")
96      .arg(format!("{font}.tfm"))
97      .output()
98      .ok()?;
99    if !tfm.status.success() {
100      return None;
101    }
102    let path = String::from_utf8_lossy(&tfm.stdout).trim().to_string();
103    let out = Command::new("tftopl").arg(&path).output().ok()?;
104    if !out.status.success() {
105      return None;
106    }
107    let text = String::from_utf8_lossy(&out.stdout).into_owned();
108    let mut slots = Vec::new();
109    for line in text.lines() {
110      if let Some(rest) = line.strip_prefix("(CHARACTER ") {
111        let mut it = rest.split_whitespace();
112        match (it.next(), it.next()) {
113          (Some("O"), Some(oct)) => {
114            if let Ok(n) = usize::from_str_radix(oct, 8) {
115              slots.push(n);
116            }
117          },
118          (Some("C"), Some(ch)) => {
119            if let Some(c) = ch.chars().next() {
120              slots.push(c as usize);
121            }
122          },
123          _ => {},
124        }
125      }
126    }
127    Some(slots)
128  }
129
130  /// Every TFM-populated slot MUST map to `Some` glyph: a `None` on a
131  /// populated slot gives that char a zero-width box and silently resurrects
132  /// the `\@whiledim` infinite-loop OOM (the canvas_3 cluster this map
133  /// fixed) for documents drawing that slope/arc. PR #249 review P3-16.
134  #[test]
135  fn every_tfm_populated_slot_is_mapped() {
136    for (font, map) in [("line10", &LINE_SLOTS), ("lcircle10", &LCIRCLE_SLOTS)] {
137      let Some(slots) = tfm_slots(font) else {
138        eprintln!("SKIP every_tfm_populated_slot_is_mapped: {font}.tfm / tftopl unavailable");
139        continue;
140      };
141      assert!(!slots.is_empty(), "{font}.tfm parsed to zero slots?");
142      for slot in slots {
143        if slot < map.len() {
144          assert!(
145            map[slot].is_some(),
146            "{font} slot {slot} (0o{slot:o}) is populated in the TFM but maps \
147             to None — zero-width box, \\@whiledim OOM risk"
148          );
149        }
150      }
151    }
152  }
