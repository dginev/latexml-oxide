5/// Runtime helper for the trivial `DefMath!` shape used 100+ times in
6/// txfonts (DEP-17, mirrors DEP-15 fontawesome approach). The macro
7/// arm expands at compile time into ~1.1 KiB of `.text` per invocation
8/// (parse_prototype + MathPrimitiveOptions builder); routing through
9/// this single helper drops `load_definitions` size at the cost of a
10/// runtime `parse_prototype` call per entry — paid once at engine
11/// bootstrap.
12fn def_math_sym(cs: &str, present: &str, role: Option<&str>, meaning: Option<&str>) -> Result<()> {
13  let (cs_tok, params) = parse_prototype(cs, true)?;
14  let mut opts = MathPrimitiveOptions::default();
15  if let Some(r) = role {
16    opts.role = Some(r.to_string());
17  }
18  if let Some(m) = meaning {
19    opts.meaning = Some(m.to_string());
20  }
21  def_math(cs_tok, params, present.to_string(), opts)?;
22  Ok(())
23}
24
25/// DEP-17 helper for the upright-Greek `DefMath!("\\xxxup", "char",
26/// font => { shape => "upright", forceshape => true })` shape — 29
27/// entries in txfonts (the lowercase + uppercase Greek `*up` family).
28fn def_math_upright_greek(cs: &str, present: &str) -> Result<()> {
29  let (cs_tok, params) = parse_prototype(cs, true)?;
30  let opts = MathPrimitiveOptions {
31    font: Some(FontDirective::from(
32      fontmap!(shape => "upright", forceshape => true),
33    )),
34    ..MathPrimitiveOptions::default()
35  };
36  def_math(cs_tok, params, present.to_string(), opts)?;
37  Ok(())
38}
