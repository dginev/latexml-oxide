6  //   LoadPool('LaTeX');
7  //   InputDefinitions('expl3', type => 'lua');
8  //   InputDefinitions('expl3', type => 'sty', noltxml => 1);

10  // The raw expl3.sty file has a TeX-level guard
11  //   \expandafter\ifx\csname tex_let:D\endcsname\relax
12  //     \expandafter\@firstofone\else\expandafter\@gobble\fi
13  //     {\input expl3-code.tex }%
14  // which detects the dump-loaded `\tex_let:D` PA-alias and skips
15  // re-loading expl3-code.tex. So this 3-line wrapper does the right
16  // thing: load lua portion, then load .sty (which short-circuits).
17  LoadPool!("LaTeX");
18  InputDefinitions!("expl3", extension => Some(Cow::Borrowed("lua")), notex => true);
19  let _ = input_definitions("expl3", NewDefault!(InputDefinitionOptions,
20    noltxml => true, extension => Some(Cow::Borrowed("sty"))));
21
22  // expl3 case-folding override.
23  //
24  // The kernel `\__kernel_codepoint_case:nn` walks per-codepoint case maps
25  // built from `c__codepoint_<case>_<cp>_tl` constants. Those are populated
26  // by reading UnicodeData.txt / CaseFolding.txt / SpecialCasing.txt during
27  // expl3-code.tex's group-end block at L33074-33180. Our raw expl3 load
28  // currently fails to open those files (the `ior_open` chain trips on a
29  // file_input dispatch issue tracked separately), leaving the codepoint
30  // tables empty — so `\str_lowercase:n {Hello}` returns "Hello" unchanged.
31  //
32  // Override the kernel function with a Rust impl using `char::to_lowercase`
33  // and `char::to_uppercase` from std. Returns a triple `{cp1}{cp2}{cp3}`
34  // matching expl3's expected return contract — first slot is the primary
35  // result codepoint, slots 2/3 hold combining chars for compound mappings
36  // (e.g. "ß" → "SS" upper has slot1=S, slot2=S; we model only single-cp
37  // mappings here, leaving slots 2/3 blank). For ASCII this is exact; for
38  // non-Latin scripts that map to multi-char sequences (Latin extended,
39  // Greek, etc.) Rust's std char::to_lowercase yields the right primary cp.
40  DefMacro!(T_CS!("\\__kernel_codepoint_case:nn"), "{}{}", sub[(case_type, cp_str)] {
41    let case = case_type.to_string().to_lowercase();
42    let cp_text = cp_str.to_string();
43    let cp_n: u32 = cp_text.trim().parse().unwrap_or(0);
44    let result_cp = if cp_n == 0 {
45      0u32
46    } else if let Some(c) = char::from_u32(cp_n) {
47      let folded: String = match case.as_str() {
48        "lowercase" | "casefold" => c.to_lowercase().collect(),
49        "uppercase" | "titlecase" => c.to_uppercase().collect(),
50        _ => c.to_string(),
51      };
52      folded.chars().next().map(|fc| fc as u32).unwrap_or(cp_n)
53    } else {
54      cp_n
55    };
56    Ok(Tokenize!(&format!("{{{}}}{{}}{{}}", result_cp)))
57  });
58});
