20  // Mirror expl3.sty's TeX-level guard so we know whether the .sty load
21  // about to run will actually `\input expl3-code.tex` (cascade-prone
22  // raw load) or short-circuit it (dump path). The guard inside
23  // expl3.sty is `\ifx\csname tex_let:D\endcsname\relax {\input ...}`,
24  // so an undefined `\tex_let:D` here ⇒ raw load will fire.

30  // Post-load fixup for `\__kernel_msg_info:nnxx`. xparse-2018-04-12.sty
31  // (line 101, 112, 218, 222) calls `\__kernel_msg_info:nnxx { xparse }
32  // { define-command }` etc. for every `\NewDocumentCommand`, but
33  // expl3-code.tex defines only the `:nnee` variant — `:nnxx` is a
34  // deprecated argument-spec letter (`x` = e-expanded, replaced by
35  // `e` in modern expl3) that xparse-2018-04-12 expects but expl3
36  // never auto-generates. Without this stub the CS is undefined →
37  // generate_error_stub installs an `<ltx:ERROR>` Constructor and
38  // EVERY `\NewDocumentCommand` invocation leaks the error element
39  // plus the unused message-body args into document text.
40  //
41  // We define `\__kernel_msg_info:nnxx` as a 4-arg no-op, matching
42  // Perl LaTeXML's effective end-state (`\msg_info:nnxx` is a
43  // log-only path; we have no log channel so a no-op is the closest
44  // equivalent).
45  //
46  // The historical "\cs_end: cascade" that this stub also masked was
47  // root-caused and fixed in latexml_core/src/binding/content.rs:
48  // \@pushfilename now runs BEFORE \@currname/\@currext are set,
49  // matching latex.ltx:15518-15519. With that fix the prior need to
50  // also stub `\g__file_record_seq` is gone.
51  //
52  // GATE: only install when the raw .sty actually re-loaded
53  // expl3-code.tex. On the dump path the guard short-circuits the
54  // re-load and the dump already provides the right state.
55  if raw_load_will_run {

57    state::assign_catcode('_', Catcode::LETTER, Some(Scope::Global));

63  // expl3 case-folding override.
64  //
65  // The kernel `\__kernel_codepoint_case:nn` walks per-codepoint case maps
66  // built from `c__codepoint_<case>_<cp>_tl` constants. Those are populated
67  // by reading UnicodeData.txt / CaseFolding.txt / SpecialCasing.txt during
68  // expl3-code.tex's group-end block at L33074-33180. Our raw expl3 load
69  // currently fails to open those files (the `ior_open` chain trips on a
70  // file_input dispatch issue tracked separately), leaving the codepoint
71  // tables empty — so `\str_lowercase:n {Hello}` returns "Hello" unchanged.
72  //
73  // Override the kernel function with a Rust impl using `char::to_lowercase`
74  // and `char::to_uppercase` from std. Returns a triple `{cp1}{cp2}{cp3}`
75  // matching expl3's expected return contract — first slot is the primary
76  // result codepoint, slots 2/3 hold combining chars for compound mappings
77  // (e.g. "ß" → "SS" upper has slot1=S, slot2=S; we model only single-cp
78  // mappings here, leaving slots 2/3 blank). For ASCII this is exact; for
79  // non-Latin scripts that map to multi-char sequences (Latin extended,
80  // Greek, etc.) Rust's std char::to_lowercase yields the right primary cp.
81  DefMacro!(T_CS!("\\__kernel_codepoint_case:nn"), "{}{}", sub[(case_type, cp_str)] {
82    let case = case_type.to_string().to_lowercase();
83    let cp_text = cp_str.to_string();
84    let cp_n: u32 = cp_text.trim().parse().unwrap_or(0);
85    let result_cp = if cp_n == 0 {
86      0u32
87    } else if let Some(c) = char::from_u32(cp_n) {
88      let folded: String = match case.as_str() {
89        "lowercase" | "casefold" => c.to_lowercase().collect(),
90        "uppercase" | "titlecase" => c.to_uppercase().collect(),
91        _ => c.to_string(),
92      };
93      folded.chars().next().map(|fc| fc as u32).unwrap_or(cp_n)
94    } else {
95      cp_n
96    };
97    Ok(Tokenize!(&format!("{{{}}}{{}}{{}}", result_cp)))
98  });
