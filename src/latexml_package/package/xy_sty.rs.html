231  DefConstructor!("\\lx@xy@svg Digested",
232    sub[document, args, props] {
233      // All values pre-computed at digest time via properties callback
234      let get_s = |k: &str| -> String {

236      };

249      document.open_element("svg:svg", Some(svg_attrs), None)?;
250

252      document.open_element("svg:g", Some(g_attrs), None)?;
253
254      if let Some(Some(content)) = args.first() {
255        document.absorb(content, None)?;
256      }
257      document.close_element("svg:g")?;
258      document.close_element("svg:svg")?;
259      document.close_element("ltx:picture")?;
260    },
261    // Perl: properties => sub { ... } — capture AND compute all values at digest time
262    after_digest => sub[whatsit] {
263      let range_str = lookup_string("saved_xy_range");
264      let dpi_val = lookup_int("DPI");
265      let dpi = if dpi_val > 0 { dpi_val as f64 } else { 100.0 };
266      let dims: Vec<f64> = range_str.split(',')
267        .filter_map(|s| s.trim().parse::<i64>().ok())
268        .map(|v| (v as f64 / 65536.0) * (dpi / 72.27))
269        .collect();
270      let (x0, y0, x1, y1) = (
271        dims.first().copied().unwrap_or(0.0),
272        dims.get(1).copied().unwrap_or(0.0),
273        dims.get(2).copied().unwrap_or(0.0),
274        dims.get(3).copied().unwrap_or(0.0),
275      );
276      let mut w = x1 - x0;
277      let mut h = y1 - y0;
278      let (x0f, y0f) = if w < 0.0 { w = 0.0; (0.0f64, y0) } else { (x0, y0) };
279      let y0f = if h < 0.0 { h = 0.0; 0.0 } else { y0f };
280      let x = -x0f;
281      let y = y1 - y0f;
282      let minx = x;
283      let miny = -y0f;
284      let pxwidth = if w > 1.0 { w } else { 1.0 };
285      let pxheight = if h > 1.0 { h } else { 1.0 };
286      whatsit.set_property("pxwidth", Stored::from(s!("{:.2}", pxwidth)));
287      whatsit.set_property("pxheight", Stored::from(s!("{:.2}", pxheight)));
288      whatsit.set_property("viewBox", Stored::from(
289        s!("{:.2} {:.2} {:.2} {:.2}", minx, miny, pxwidth, pxheight)));
290      whatsit.set_property("transform", Stored::from(
291        s!("matrix(1 0 0 -1 {} {})", x, y)));
292      if miny != 0.0 {
293        whatsit.set_property("style", Stored::from(
294          s!("vertical-align:{}px", -miny)));
295      }
296    }
297  );

353  // the real xy processing that sets `\xy@`≠`\xyinitial@` never runs,
354  // `\inxy@` always reports "not nested" → every internal `\xy` re-enters
