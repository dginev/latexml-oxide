111  // \lx@xy@capturerange — capture coordinate range (Perl L143-146)
112  DefPrimitive!("\\lx@xy@capturerange", {
113    let mut dims = String::new();
114    for reg in ["\\X@min", "\\Y@min", "\\X@max", "\\Y@max"] {
115      if let Ok(Some(val)) = state::lookup_register(reg, Vec::new()) {
116        let sp = match val {
117          RegisterValue::Dimension(d) => d.value_of(),
118          RegisterValue::Number(n) => n.value_of(),
119          _ => 0,
120        };
121        if !dims.is_empty() { dims.push(','); }
122        dims.push_str(&sp.to_string());
123      } else {
124        if !dims.is_empty() { dims.push(','); }
125        dims.push('0');
126      }
127    }
128    state::assign_value("saved_xy_range", Stored::String(arena::pin(&dims)), Some(Scope::Global));
129  });

135  // \lx@xy@svg — top-level xy picture SVG (Perl L95-141)
136  // Perl uses a `properties` callback (digestion time) to capture saved_xy_range,

138  // We replicate this with after_digest to snapshot the range at digest time.
139  DefConstructor!("\\lx@xy@svg Digested",
140    sub[document, args, props] {
141      // All values pre-computed at digest time via properties callback
142      let get_s = |k: &str| -> String {
143        match props.get(k) { Some(Stored::String(s)) => arena::to_string(*s), _ => String::new() }
144      };
145      let pxwidth = get_s("pxwidth");
146      let pxheight = get_s("pxheight");
147      let pic_attrs = string_map!("width" => pxwidth.clone(), "height" => pxheight.clone());
148      document.open_element("ltx:picture", Some(pic_attrs), None)?;
149
150      let mut svg_attrs = string_map!(
151        "version" => "1.1", "overflow" => "visible",
152        "width" => pxwidth.clone(), "height" => pxheight.clone(),
153        "viewBox" => get_s("viewBox")
154      );
155      let style = get_s("style");
156      if !style.is_empty() { svg_attrs.insert(String::from("style"), style); }
157      document.open_element("svg:svg", Some(svg_attrs), None)?;
158
159      let g_attrs = string_map!("transform" => get_s("transform"));
160      document.open_element("svg:g", Some(g_attrs), None)?;
161
162      if let Some(Some(content)) = args.first() {
163        document.absorb(content, None)?;
164      }
165      document.close_element("svg:g")?;
166      document.close_element("svg:svg")?;
167      document.close_element("ltx:picture")?;
168    },
169    // Perl: properties => sub { ... } — capture AND compute all values at digest time
170    after_digest => sub[whatsit] {
171      let range_str = state::lookup_string("saved_xy_range");
172      let dpi_val = state::lookup_int("DPI");
173      let dpi = if dpi_val > 0 { dpi_val as f64 } else { 100.0 };
174      let dims: Vec<f64> = range_str.split(',')
175        .filter_map(|s| s.trim().parse::<i64>().ok())
176        .map(|v| (v as f64 / 65536.0) * (dpi / 72.27))
177        .collect();
178      let (x0, y0, x1, y1) = (
179        dims.first().copied().unwrap_or(0.0),
180        dims.get(1).copied().unwrap_or(0.0),
181        dims.get(2).copied().unwrap_or(0.0),
182        dims.get(3).copied().unwrap_or(0.0),
183      );
184      let mut w = x1 - x0;
185      let mut h = y1 - y0;
186      let (x0f, y0f) = if w < 0.0 { w = 0.0; (0.0f64, y0) } else { (x0, y0) };
187      let y0f = if h < 0.0 { h = 0.0; 0.0 } else { y0f };
188      let x = -x0f;
189      let y = y1 - y0f;
190      let minx = x;
191      let miny = -y0f;
192      let pxwidth = if w > 1.0 { w } else { 1.0 };
193      let pxheight = if h > 1.0 { h } else { 1.0 };
194      whatsit.set_property("pxwidth", Stored::from(s!("{:.2}", pxwidth)));
195      whatsit.set_property("pxheight", Stored::from(s!("{:.2}", pxheight)));
196      whatsit.set_property("viewBox", Stored::from(
197        s!("{:.2} {:.2} {:.2} {:.2}", minx, miny, pxwidth, pxheight)));
198      whatsit.set_property("transform", Stored::from(
199        s!("matrix(1 0 0 -1 {} {})", x, y)));
200      if miny != 0.0 {
201        whatsit.set_property("style", Stored::from(
202          s!("vertical-align:{}px", -miny)));
203      }
204    }
205  );

295  // Step 7: Defensive stub for `\xymatrix` when bare `\usepackage{xy}`
296  // (no [matrix,arrow] options) leaves `xymatrix.tex` unloaded. Without
297  // a stub, papers using `\xymatrix@1{a & b \\ c & d}` (e.g.
298  // arXiv:1409.7007) cascade 100+ "Stray alignment '&'" errors as the
299  // unguarded body is parsed in math mode. The stub uses TeX's `#{`
300  // implicit-brace-stop pattern to read xy-pic's `@<modifier>` prefix
301  // tokens (`@1`, `@C=10pt`, `@R=2cm`, etc.) up to the brace, then
302  // swallows the brace-balanced body. The diagram is dropped — but
303  // the cascade is gone, matching latexml's general philosophy of
304  // "drop unsupported package content rather than fail conversion".
305  //
306  // Gated by `\@ifundefined{xymatrix}` so it only fires when xy.tex's
307  // option pipeline didn't auto-load matrix/arrow features.
308  RawTeX!(r"\@ifundefined{xymatrix}{%
309    \def\xymatrix#1#{\lx@xy@stub@xymatrix@body}%
310    \def\lx@xy@stub@xymatrix@body#1{}%
311  }{}");
