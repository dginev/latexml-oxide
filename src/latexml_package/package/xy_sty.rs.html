47  DefMacro!("\\xyoption{}", sub[(option)] {
48    let option_s = option.to_string();
49    let other_drivers = [
50      "16textures", "17oztex", "dvidrv", "dvips", "dvitops",
51      "oztex", "pdf", "textures", "dvi",
52    ];
53    let unsupported = ["barr", "movie", "necula", "smart"];
54    if other_drivers.contains(&option_s.as_str()) {
55      Info!("xy", "ignored", s!("Ignoring xy driver {} (using latexml)", option_s));
56      return Ok(Tokens!());
57    }
58    if unsupported.contains(&option_s.as_str()) {
59      Warn!("xy", "unsupported",
60        s!("The xy extension/feature {} may not be supported", option_s));
61    }
62    let cache_key = s!("loaded_xyoption_{}", option_s);
63    if state::lookup_bool(&cache_key) {
64      return Ok(Tokens!());
65    }
66    state::assign_value(&cache_key, true, Some(Scope::Global));
67    // Special case: "latexml" driver — load our Rust xylatexml overlay directly
68    // since the file xylatexml.tex doesn't exist on disk (it's compiled into the binary).
69    if option_s == "latexml" {
70      crate::package::xylatexml_tex::load_definitions()?;
71      return Ok(Tokens!());
72    }
73    // Special case: "all" — xyall.tex `\xyrequire`s every xy feature
74    // (curve, frame, cmtip, line, rotate, color, matrix, arrow, graph).
75    // Forwarding through `\lx@xy@xyoption@orig` does NOT actually
76    // \input the per-feature xy<name>.tex file in our pipeline
77    // (the `\xyinputorelse@` chain in xy.tex evaluates strangely
78    // without raw-TeX `\openin`), leaving things like `\xygraph`
79    // undefined. Directly load the per-feature files via
80    // `InputDefinitions` here. Witness 2311.05789
81    // (`\usepackage[all,tips]{xy}` + `\xygraph` → Error:undefined:\xygraph).
82    // Special case: known xy feature names — directly `\input` the
83    // feature file via input_definitions. `\lx@xy@xyoption@orig`'s
84    // `\xyinputorelse@` chain in xy.tex evaluates strangely in our
85    // pipeline (it relies on raw-TeX `\openin`/`\closein` to
86    // probe and \input, which the binding's input path doesn't fully
87    // honor), so features like `\xygraph` end up undefined despite
88    // `\usepackage[all,tips]{xy}`. Witness 2311.05789.
89    let feature_files: &[&str] = match option_s.as_str() {
90      "all" => &["curve","frame","cmtip","line","rotate","color","matrix","arrow","graph"],
91      "graph" | "matrix" | "arrow" | "curve" | "frame" | "cmtip"
92        | "line" | "rotate" | "color" | "tips" => {
93        let single = option_s.as_str();
94        let n = s!("xy{}", single);
95        let _ = input_definitions(&n, InputDefinitionOptions {
96          extension: Some(::std::borrow::Cow::Borrowed("tex")),
97          noltxml: true,
98          ..Default::default()
99        });
100        return Ok(Tokens!());
101      },
102      _ => &[][..],
103    };
104    if !feature_files.is_empty() {
105      for name in feature_files {
106        let n = s!("xy{}", name);
107        let _ = input_definitions(&n, InputDefinitionOptions {
108          extension: Some(::std::borrow::Cow::Borrowed("tex")),
109          noltxml: true,
110          ..Default::default()
111        });
112      }
113      return Ok(Tokens!());
114    }
115    Ok(Tokens!(T_CS!("\\lx@xy@xyoption@orig"), T_BEGIN!(), option, T_END!()))
116  });

162  // \lx@xy@capturerange — capture coordinate range (Perl L143-146)
163  DefPrimitive!("\\lx@xy@capturerange", {
164    let mut dims = String::new();
165    for reg in ["\\X@min", "\\Y@min", "\\X@max", "\\Y@max"] {
166      if let Ok(Some(val)) = state::lookup_register(reg, Vec::new()) {
167        let sp = match val {
168          RegisterValue::Dimension(d) => d.value_of(),
169          RegisterValue::Number(n) => n.value_of(),
170          _ => 0,
171        };
172        if !dims.is_empty() { dims.push(','); }
173        dims.push_str(&sp.to_string());
174      } else {
175        if !dims.is_empty() { dims.push(','); }
176        dims.push('0');
177      }
178    }
179    state::assign_value("saved_xy_range", Stored::String(arena::pin(&dims)), Some(Scope::Global));
180  });

186  // \lx@xy@svg — top-level xy picture SVG (Perl L95-141)
187  // Perl uses a `properties` callback (digestion time) to capture saved_xy_range,
188  // then the constructor body (construction time) reads from %props.
189  // We replicate this with after_digest to snapshot the range at digest time.
190  DefConstructor!("\\lx@xy@svg Digested",
191    sub[document, args, props] {
192      // All values pre-computed at digest time via properties callback
193      let get_s = |k: &str| -> String {
194        match props.get(k) { Some(Stored::String(s)) => arena::to_string(*s), _ => String::new() }
195      };
196      let pxwidth = get_s("pxwidth");
197      let pxheight = get_s("pxheight");
198      let pic_attrs = string_map!("width" => pxwidth, "height" => pxheight);
199      document.open_element("ltx:picture", Some(pic_attrs), None)?;
200
201      let mut svg_attrs = string_map!(
202        "version" => "1.1", "overflow" => "visible",
203        "width" => pxwidth, "height" => pxheight,
204        "viewBox" => get_s("viewBox")
205      );
206      let style = get_s("style");
207      if !style.is_empty() { svg_attrs.insert(String::from("style"), style); }
208      document.open_element("svg:svg", Some(svg_attrs), None)?;
209
210      let g_attrs = string_map!("transform" => get_s("transform"));
211      document.open_element("svg:g", Some(g_attrs), None)?;
212
213      if let Some(Some(content)) = args.first() {
214        document.absorb(content, None)?;
215      }
216      document.close_element("svg:g")?;
217      document.close_element("svg:svg")?;
218      document.close_element("ltx:picture")?;
219    },
220    // Perl: properties => sub { ... } — capture AND compute all values at digest time
221    after_digest => sub[whatsit] {
222      let range_str = state::lookup_string("saved_xy_range");
223      let dpi_val = state::lookup_int("DPI");
224      let dpi = if dpi_val > 0 { dpi_val as f64 } else { 100.0 };
225      let dims: Vec<f64> = range_str.split(',')
226        .filter_map(|s| s.trim().parse::<i64>().ok())
227        .map(|v| (v as f64 / 65536.0) * (dpi / 72.27))
228        .collect();
229      let (x0, y0, x1, y1) = (
230        dims.first().copied().unwrap_or(0.0),
231        dims.get(1).copied().unwrap_or(0.0),
232        dims.get(2).copied().unwrap_or(0.0),
233        dims.get(3).copied().unwrap_or(0.0),
234      );
235      let mut w = x1 - x0;
236      let mut h = y1 - y0;
237      let (x0f, y0f) = if w < 0.0 { w = 0.0; (0.0f64, y0) } else { (x0, y0) };
238      let y0f = if h < 0.0 { h = 0.0; 0.0 } else { y0f };
239      let x = -x0f;
240      let y = y1 - y0f;
241      let minx = x;
242      let miny = -y0f;
243      let pxwidth = if w > 1.0 { w } else { 1.0 };
244      let pxheight = if h > 1.0 { h } else { 1.0 };
245      whatsit.set_property("pxwidth", Stored::from(s!("{:.2}", pxwidth)));
246      whatsit.set_property("pxheight", Stored::from(s!("{:.2}", pxheight)));
247      whatsit.set_property("viewBox", Stored::from(
248        s!("{:.2} {:.2} {:.2} {:.2}", minx, miny, pxwidth, pxheight)));
249      whatsit.set_property("transform", Stored::from(
250        s!("matrix(1 0 0 -1 {} {})", x, y)));
251      if miny != 0.0 {
252        whatsit.set_property("style", Stored::from(
253          s!("vertical-align:{}px", -miny)));
254      }
255    }
256  );

346  // Step 7: Defensive stub for `\xymatrix` when bare `\usepackage{xy}`
347  // (no [matrix,arrow] options) leaves `xymatrix.tex` unloaded. Without
348  // a stub, papers using `\xymatrix@1{a & b \\ c & d}` (e.g.
349  // arXiv:1409.7007) cascade 100+ "Stray alignment '&'" errors as the
350  // unguarded body is parsed in math mode. The stub uses TeX's `#{`
351  // implicit-brace-stop pattern to read xy-pic's `@<modifier>` prefix
352  // tokens (`@1`, `@C=10pt`, `@R=2cm`, etc.) up to the brace, then
353  // swallows the brace-balanced body. The diagram is dropped — but
354  // the cascade is gone, matching latexml's general philosophy of
355  // "drop unsupported package content rather than fail conversion".
356  //
357  // Gated by `\@ifundefined{xymatrix}` so it only fires when xy.tex's
358  // option pipeline didn't auto-load matrix/arrow features.
359  RawTeX!(r"\@ifundefined{xymatrix}{%
360    \def\xymatrix#1#{\lx@xy@stub@xymatrix@body}%
361    \def\lx@xy@stub@xymatrix@body#1{}%
362  }{}");
