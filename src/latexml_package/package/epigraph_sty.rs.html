1use crate::prelude::*;
2
3
4LoadDefinitions!({
5  TeX!(
6    r#"""
7\newlength{\beforeepigraphskip}
8  \setlength{\beforeepigraphskip}{.5\baselineskip}
9\newlength{\afterepigraphskip}
10  \setlength{\afterepigraphskip}{.5\baselineskip}
11\newlength{\epigraphwidth}
12  \setlength{\epigraphwidth}{.4\textwidth}
13\newlength{\epigraphrule}
14  \setlength{\epigraphrule}{.4\p@}
15\newcommand{\epigraphsize}{\small}
16\newcommand{\epigraphflush}{flushright}
17\newcommand{\textflush}{flushleft}
18\newcommand{\sourceflush}{flushright}
19"""#
20  );
21
22  DefConstructor!("\\epigraph{}{}",
23    "<ltx:quote class='ltx_epigraph #epigraphflush' cssstyle='#qwidth #qalign'>\
24      <ltx:block class='ltx_epigraph_text' cssstyle='#talign'>#1</ltx:block>\
25      <ltx:block class='ltx_epigraph_source' cssstyle='#srule #salign'>#2</ltx:block>\
26    </ltx:quote>",
27    bounded => true,
28    before_digest => {
29      stomach::digest(Tokens!(T_CS!("\\epigraphsize")))?;
30    },
31    after_digest => sub[whatsit] {
32      let rule = LookupRegisterOrDefault!("\\epigraphrule");
33      let rule_pt = match rule {
34        RegisterValue::Dimension(d) => d.pt_value(None),
35        _ => 0.4,
36      };
37      let width = LookupRegisterOrDefault!("\\epigraphwidth");
38      let width_pt = match width {
39        RegisterValue::Dimension(d) => d.pt_value(None),
40        _ => 0.0,
41      };
42      let qa = gullet::do_expand(T_CS!("\\epigraphflush"))?.to_string();
43      let ta = gullet::do_expand(T_CS!("\\textflush"))?.to_string();
44      let sa = gullet::do_expand(T_CS!("\\sourceflush"))?.to_string();
45
46      let qalign = match qa.as_str() {
47        "center" => "margin-right:auto; margin-left:auto;",
48        "flushleft" => "margin-right:auto;",
49        "flushright" => "margin-left:auto;",
50        _ => "",
51      };
52      let talign = match ta.as_str() {
53        "center" => "text-align:center; ",
54        "flushleft" => "text-align:left; ",
55        "flushright" => "text-align:right; ",
56        _ => "",
57      };
58      let salign = match sa.as_str() {
59        "center" => "text-align:center; ",
60        "flushleft" => "text-align:left; ",
61        "flushright" => "text-align:right; ",
62        _ => "",
63      };
64
65      whatsit.set_property("srule", s!("border-top:solid {}pt;", rule_pt));
66      whatsit.set_property("qwidth", s!("width:{}pt;", width_pt));
67      whatsit.set_property("qalign", qalign);
68      whatsit.set_property("talign", talign);
69      whatsit.set_property("salign", salign);
70      // Note: do NOT set "epigraphflush" property — the template #epigraphflush
71      // should resolve to empty (Perl never sets this property)
72    }
73  );
74
75  DefEnvironment!("{epigraphs}", "#body",
76    before_digest => {
77      Let!("\\qitem", "\\epigraph");
78    }
79  );
80
81  DefMacro!("\\epigraphhead[]{}", "#1");
82  def_macro_noop("\\dropchapter{}")?;
83  def_macro_noop("\\undodrop")?;
84  def_macro_noop("\\cleartoevenpage")?;
85});
