19  for trigger in [
20    "\\citet", "\\citep", "\\citealt", "\\citealp", "\\citenum",
21    "\\citeauthor", "\\citefullauthor", "\\citeyear", "\\citeyearpar",
22    "\\citeauthoryear", "\\setcitestyle", "\\bibpunct",
23  ] {
24    let cs = T_CS!(trigger);
25    if !IsDefined!(&cs) {
26      let cs_clone = cs;
27      def_macro(cs, None,
28        latexml_core::definition::ExpansionBody::Closure(Rc::new(move |_args| {
29          require_package("natbib", RequireOptions::default())?;
30          Ok(Tokens::new(vec![cs_clone]))
31        })), None)?;
32    }
33  }
