8/// Push the digested body of a `{keyword}`/`{keywords}` env directly into
9/// the `frontmatter` map under `ltx:classification[scheme=keywords]`, the
10/// way Perl LaTeXML's after_digest_keywords does it
11/// (OmniBus.cls.ltxml `sub after_digest_keywords`). Avoids the
12/// raw_tex(`#body`) workaround that mistakenly tokenized `#` as PARAM
13/// and dumped the literal text "#body" into the output element.
14fn push_keyword_body_to_frontmatter(
15  whatsit: &mut latexml_core::whatsit::Whatsit,
16) -> latexml_core::Result<Vec<latexml_core::digested::Digested>> {
17  use latexml_core::BoxOps;
18  use latexml_core::common::store::Stored;
19  if let Some(body) = whatsit.get_body()? {
20    let mut attrs: rustc_hash::FxHashMap<String, String> =
21      rustc_hash::FxHashMap::default();
22    attrs.insert("scheme".to_string(), "keywords".to_string());
23    let entry = ("ltx:classification".to_string(), Some(attrs), body);
24    latexml_core::state::with_value_mut("frontmatter", |val_opt| {
25      if let Some(Stored::HashTagData(ref mut frnt)) = val_opt {
26        frnt
27          .entry("ltx:classification".to_string())
28          .or_insert_with(Vec::new)
29          .push(entry);
30      }
31    });
32  }
33  Ok(Vec::new())
34}

553  // Perl L286-297: math/package autoloads — when a trigger CS is used and
554  // not yet defined, require the specified package/class and re-trigger.
555  // The Perl `DefAutoload` macro registers this semantic; we implement it
556  // inline. Perl encodes the target via the `.sty.ltxml` / `.cls.ltxml`
557  // suffix; we mirror that by carrying an explicit "sty" / "cls" kind
558  // alongside the bare name.
559  //
560  // Why the kind matters: `\thechapter` autoloads `book.cls.ltxml` in Perl,
561  // not `book.sty`. Routing through `require_package("book")` instead
562  // finds the obsolete `book.sty` shim (TL's 2.09-compat file) which
563  // immediately fires `\LoadClass{book}` from inside the body — past the
564  // preamble — and errors with "\LoadClass can only appear in the preamble".
565  // Witness: arXiv:2602.10407 (`\documentclass{saunders}` + `\chapter` in
566  // an `\include`'d chapter; saunders.cls is unbound so OmniBus takes over,
567  // and the chapter trigger hits `\thechapter`). Perl handles it cleanly
568  // by autoloading the class binding instead of the obsolete sty.
569  for (trigger, name, ext) in [
570    // env triggers: `\begin{align}` etc. In Rust, we only dispatch on the
571    // bare CS name of the trigger — works for control sequences like
572    // `\multline`, `\numberwithin`, `\mathfrak`, `\mathbb`, `\deluxetable`,
573    // `\curraddr`, `\subjclass`, `\thechapter`. For envs, autoload key is
574    // `\begin{env}` which is a CS token.
575    ("\\begin{align}",                "amsmath",      "sty"),
576    ("\\begin{subequations}",         "amsmath",      "sty"),
577    ("\\begin{split}",                "amsmath",      "sty"),
578    ("\\multline",                    "amsmath",      "sty"),
579    ("\\csname multline*\\endcsname", "amsmath",      "sty"),
580    ("\\numberwithin",                "amsmath",      "sty"),
581    ("\\mathfrak",                    "amsfonts",     "sty"),
582    ("\\mathbb",                      "amsfonts",     "sty"),
583    ("\\begin{deluxetable}",          "deluxetable",  "sty"),
584    ("\\curraddr",                    "ams_support",  "sty"),
585    ("\\subjclass",                   "ams_support",  "sty"),
586    ("\\thechapter",                  "book",         "cls"),
587  ] {
588    let cs = T_CS!(trigger);
589    if !IsDefined!(&cs) {
590      let cs_clone = cs;
591      let name_str = name.to_string();
592      let trigger_str = trigger.to_string();
593      let is_cls = ext == "cls";
594      def_macro(cs, None,
595        latexml_core::definition::ExpansionBody::Closure(Rc::new(move |_args| {
596          // Mirrors Perl's DefAutoload → ClearAutoLoad in Package.pm:
597          // clear this autoload CS before loading, then re-emit the trigger as
598          // tokenized text. Re-tokenizing is important for `\begin{env}` triggers
599          // — amsmath defines `\split` (not `\begin{split}`), so the raw single-CS
600          // token would look undefined after clearing. Tokenizing expands into
601          // `\begin` + `{env}` which the standard `\begin{}` dispatcher resolves.
602          latexml_core::state::assign_meaning(

604            Some(Scope::Global));
605          if is_cls {
606            latexml_core::binding::content::load_class(
607              &name_str, Vec::new(), Tokens::default())?;
608          } else {
609            require_package(&name_str, RequireOptions::default())?;
610          }
611          Ok(mouth::tokenize_internal(&trigger_str))
612        })), None)?;
613    }
