8/// Push the digested body of a `{keyword}`/`{keywords}` env directly into
9/// the `frontmatter` map under `ltx:classification[scheme=keywords]`, the
10/// way Perl LaTeXML's after_digest_keywords does it
11/// (OmniBus.cls.ltxml `sub after_digest_keywords`). Avoids the
12/// raw_tex(`#body`) workaround that mistakenly tokenized `#` as PARAM
13/// and dumped the literal text "#body" into the output element.
14fn push_keyword_body_to_frontmatter(whatsit: &mut Whatsit) -> Result<Vec<Digested>> {
15  use latexml_core::{BoxOps, common::store::Stored};
16  if let Some(body) = whatsit.get_body()? {
17    let mut attrs: rustc_hash::FxHashMap<String, String> = rustc_hash::FxHashMap::default();
18    attrs.insert("scheme".to_string(), "keywords".to_string());
19    let entry = document::tag::TagData {
20      tag:     "ltx:classification".to_string(),
21      attr:    attrs,
22      content: vec![document::tag::TagContent::Box(body)],
23    };
24    with_value_mut("frontmatter", |val_opt| {
25      if let Some(Stored::HashTagData(frnt)) = val_opt {
26        frnt
27          .entry("ltx:classification".to_string())
28          .or_insert_with(Vec::new)
29          .push(entry);
30      }
31    });
32  }
33  Ok(Vec::new())
34}

571  // Perl L286-297: math/package autoloads — when a trigger CS is used and
572  // not yet defined, require the specified package/class and re-trigger.
573  // The Perl `DefAutoload` macro registers this semantic; we implement it
574  // inline. Perl encodes the target via the `.sty.ltxml` / `.cls.ltxml`
575  // suffix; we mirror that by carrying an explicit "sty" / "cls" kind
576  // alongside the bare name.
577  //
578  // Why the kind matters: `\thechapter` autoloads `book.cls.ltxml` in Perl,
579  // not `book.sty`. Routing through `require_package("book")` instead
580  // finds the obsolete `book.sty` shim (TL's 2.09-compat file) which
581  // immediately fires `\LoadClass{book}` from inside the body — past the
582  // preamble — and errors with "\LoadClass can only appear in the preamble".
583  // Witness: arXiv:2602.10407 (`\documentclass{saunders}` + `\chapter` in
584  // an `\include`'d chapter; saunders.cls is unbound so OmniBus takes over,
585  // and the chapter trigger hits `\thechapter`). Perl handles it cleanly
586  // by autoloading the class binding instead of the obsolete sty.
587  for (trigger, name, ext) in [
588    // env triggers: `\begin{align}` etc. In Rust, we only dispatch on the
589    // bare CS name of the trigger — works for control sequences like
590    // `\multline`, `\numberwithin`, `\mathfrak`, `\mathbb`, `\deluxetable`,
591    // `\curraddr`, `\subjclass`, `\thechapter`. For envs, autoload key is
592    // `\begin{env}` which is a CS token.
593    ("\\begin{align}",                "amsmath",      "sty"),
594    ("\\begin{subequations}",         "amsmath",      "sty"),
595    ("\\begin{split}",                "amsmath",      "sty"),
596    ("\\multline",                    "amsmath",      "sty"),
597    ("\\csname multline*\\endcsname", "amsmath",      "sty"),
598    ("\\numberwithin",                "amsmath",      "sty"),
599    ("\\mathfrak",                    "amsfonts",     "sty"),
600    ("\\mathbb",                      "amsfonts",     "sty"),
601    ("\\begin{deluxetable}",          "deluxetable",  "sty"),
602    ("\\curraddr",                    "ams_support",  "sty"),
603    ("\\subjclass",                   "ams_support",  "sty"),
604    ("\\thechapter",                  "book",         "cls"),
605  ] {
606    let cs = T_CS!(trigger);
607    if !IsDefined!(&cs) {
608      let cs_clone = cs;
609      let name_str = name.to_string();
610      let trigger_str = trigger.to_string();
611      let is_cls = ext == "cls";
612      def_macro(cs, None,
613        ExpansionBody::Closure(Rc::new(move |_args| {
614          // Mirrors Perl's DefAutoload → ClearAutoLoad in Package.pm:
615          // clear this autoload CS before loading, then re-emit the trigger as
616          // tokenized text. Re-tokenizing is important for `\begin{env}` triggers
617          // — amsmath defines `\split` (not `\begin{split}`), so the raw single-CS
618          // token would look undefined after clearing. Tokenizing expands into
619          // `\begin` + `{env}` which the standard `\begin{}` dispatcher resolves.
620          assign_meaning(
621            &cs_clone, Stored::None,
622            Some(Scope::Global));
623          if is_cls {
624            load_class(
625              &name_str, Vec::new(), Tokens::default())?;
626          } else {
627            require_package(&name_str, RequireOptions::default())?;
628          }
629          Ok(mouth::tokenize_internal(&trigger_str))
630        })), None)?;
631    }
632  }
