355  // Perl L286-297: math/package autoloads — when a trigger CS is used and
356  // not yet defined, require the specified package and re-trigger. The Perl
357  // `DefAutoload` macro registers this semantic; we implement it inline.
358  for (trigger, pkg) in [
359    // env triggers: `\begin{align}` etc. In Rust, we only dispatch on the
360    // bare CS name of the trigger — works for control sequences like
361    // `\multline`, `\numberwithin`, `\mathfrak`, `\mathbb`, `\deluxetable`,
362    // `\curraddr`, `\subjclass`, `\thechapter`. For envs, autoload key is
363    // `\begin{env}` which is a CS token.
364    ("\\begin{align}",         "amsmath"),
365    ("\\begin{subequations}",  "amsmath"),
366    ("\\begin{split}",         "amsmath"),
367    ("\\multline",             "amsmath"),
368    ("\\csname multline*\\endcsname", "amsmath"),
369    ("\\numberwithin",         "amsmath"),
370    ("\\mathfrak",             "amsfonts"),
371    ("\\mathbb",               "amsfonts"),
372    ("\\begin{deluxetable}",   "deluxetable"),
373    ("\\curraddr",             "ams_support"),
374    ("\\subjclass",            "ams_support"),
375    ("\\thechapter",           "book"),
376  ] {
377    let cs = T_CS!(trigger);
378    if !IsDefined!(&cs) {
379      let cs_clone = cs;
380      let pkg_str = pkg.to_string();
381      let trigger_str = trigger.to_string();
382      def_macro(cs, None,
383        latexml_core::definition::ExpansionBody::Closure(Rc::new(move |_args| {
384          // Mirrors Perl's DefAutoload → ClearAutoLoad in Package.pm:
385          // clear this autoload CS before loading, then re-emit the trigger as
386          // tokenized text. Re-tokenizing is important for `\begin{env}` triggers
387          // — amsmath defines `\split` (not `\begin{split}`), so the raw single-CS
388          // token would look undefined after clearing. Tokenizing expands into
389          // `\begin` + `{env}` which the standard `\begin{}` dispatcher resolves.
390          latexml_core::state::assign_meaning(
391            &cs_clone, latexml_core::common::store::Stored::None,
392            Some(Scope::Global));
393          require_package(&pkg_str, RequireOptions::default())?;
394          Ok(mouth::tokenize_internal(&trigger_str))
395        })), None)?;
396    }
