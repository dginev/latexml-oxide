8/// Push the digested body of a `{keyword}`/`{keywords}` env directly into
9/// the `frontmatter` map under `ltx:classification[scheme=keywords]`, the
10/// way Perl LaTeXML's after_digest_keywords does it
11/// (OmniBus.cls.ltxml `sub after_digest_keywords`). Avoids the
12/// raw_tex(`#body`) workaround that mistakenly tokenized `#` as PARAM
13/// and dumped the literal text "#body" into the output element.
14fn push_keyword_body_to_frontmatter(
15  whatsit: &mut latexml_core::whatsit::Whatsit,
16) -> latexml_core::Result<Vec<latexml_core::digested::Digested>> {
17  use latexml_core::BoxOps;
18  use latexml_core::common::store::Stored;
19  if let Some(body) = whatsit.get_body()? {
20    let mut attrs: rustc_hash::FxHashMap<String, String> =
21      rustc_hash::FxHashMap::default();
22    attrs.insert("scheme".to_string(), "keywords".to_string());
23    let entry = ("ltx:classification".to_string(), Some(attrs), body);
24    latexml_core::state::with_value_mut("frontmatter", |val_opt| {
25      if let Some(Stored::HashTagData(ref mut frnt)) = val_opt {
26        frnt
27          .entry("ltx:classification".to_string())
28          .or_insert_with(Vec::new)
29          .push(entry);
30      }
31    });
32  }
33  Ok(Vec::new())
34}

502  // Perl L286-297: math/package autoloads — when a trigger CS is used and
503  // not yet defined, require the specified package/class and re-trigger.
504  // The Perl `DefAutoload` macro registers this semantic; we implement it
505  // inline. Perl encodes the target via the `.sty.ltxml` / `.cls.ltxml`
506  // suffix; we mirror that by carrying an explicit "sty" / "cls" kind
507  // alongside the bare name.
508  //
509  // Why the kind matters: `\thechapter` autoloads `book.cls.ltxml` in Perl,
510  // not `book.sty`. Routing through `require_package("book")` instead
511  // finds the obsolete `book.sty` shim (TL's 2.09-compat file) which
512  // immediately fires `\LoadClass{book}` from inside the body — past the
513  // preamble — and errors with "\LoadClass can only appear in the preamble".
514  // Witness: arXiv:2602.10407 (`\documentclass{saunders}` + `\chapter` in
515  // an `\include`'d chapter; saunders.cls is unbound so OmniBus takes over,
516  // and the chapter trigger hits `\thechapter`). Perl handles it cleanly
517  // by autoloading the class binding instead of the obsolete sty.
518  for (trigger, name, ext) in [
519    // env triggers: `\begin{align}` etc. In Rust, we only dispatch on the
520    // bare CS name of the trigger — works for control sequences like
521    // `\multline`, `\numberwithin`, `\mathfrak`, `\mathbb`, `\deluxetable`,
522    // `\curraddr`, `\subjclass`, `\thechapter`. For envs, autoload key is
523    // `\begin{env}` which is a CS token.
524    ("\\begin{align}",                "amsmath",      "sty"),
525    ("\\begin{subequations}",         "amsmath",      "sty"),
526    ("\\begin{split}",                "amsmath",      "sty"),
527    ("\\multline",                    "amsmath",      "sty"),
528    ("\\csname multline*\\endcsname", "amsmath",      "sty"),
529    ("\\numberwithin",                "amsmath",      "sty"),
530    ("\\mathfrak",                    "amsfonts",     "sty"),
531    ("\\mathbb",                      "amsfonts",     "sty"),
532    ("\\begin{deluxetable}",          "deluxetable",  "sty"),
533    ("\\curraddr",                    "ams_support",  "sty"),
534    ("\\subjclass",                   "ams_support",  "sty"),
535    ("\\thechapter",                  "book",         "cls"),
536  ] {
537    let cs = T_CS!(trigger);
538    if !IsDefined!(&cs) {
539      let cs_clone = cs;
540      let name_str = name.to_string();
541      let trigger_str = trigger.to_string();
542      let is_cls = ext == "cls";
543      def_macro(cs, None,
544        latexml_core::definition::ExpansionBody::Closure(Rc::new(move |_args| {
545          // Mirrors Perl's DefAutoload → ClearAutoLoad in Package.pm:
546          // clear this autoload CS before loading, then re-emit the trigger as
547          // tokenized text. Re-tokenizing is important for `\begin{env}` triggers
548          // — amsmath defines `\split` (not `\begin{split}`), so the raw single-CS
549          // token would look undefined after clearing. Tokenizing expands into
550          // `\begin` + `{env}` which the standard `\begin{}` dispatcher resolves.
551          latexml_core::state::assign_meaning(
552            &cs_clone, latexml_core::common::store::Stored::None,
553            Some(Scope::Global));
554          if is_cls {
555            latexml_core::binding::content::load_class(
556              &name_str, Vec::new(), Tokens::default())?;
557          } else {
558            require_package(&name_str, RequireOptions::default())?;
559          }
560          Ok(mouth::tokenize_internal(&trigger_str))
561        })), None)?;
562    }
563  }
