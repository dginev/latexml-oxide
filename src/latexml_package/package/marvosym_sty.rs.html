3/// DEP-NEW (2026-05-19): data-drive helper for the 113 marvosym
4/// icon primitives that all expand to `DefPrimitive!("\\<cs>",
5/// "<unicode-char(s)>")` — the same simple-literal-body shape the
6/// `DefPrimitive!` macro's first arm expands. Inlining the body of
7/// that arm into a runtime fn collapses 113 macro expansions to
8/// one. Per [[wisdom_data_drive_min_call_sites]]: 113 ≫ 5
9/// threshold. Matches the fontawesome DEP-15 / jhep_cls /
10/// iopart_support_sty pattern.
11fn def_marvosym_icon(cs: &str, codepoint: &str) -> Result<()> {
12  let (cs_tok, params) = parse_prototype(cs, true)?;
13  let body_sym = pin(codepoint);
14  let cs_for_closure = cs_tok;
15  let closure: PrimitiveBody = PrimitiveBody::Closure(Rc::new(move |_args: Vec<ArgWrap>| {
16    Tbox::new(
17      body_sym,
18      None,
19      None,
20      Tokens!(cs_for_closure),
21      SymHashMap::default(),
22    )
23    .into_digested_result()
24  }));
25  def_primitive(cs_tok, params, Some(closure), PrimitiveOptions::default())?;
26  Ok(())
27}
