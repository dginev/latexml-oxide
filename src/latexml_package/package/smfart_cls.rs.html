1//! smfart.cls — Société Mathématique de France article class
2//! No Perl binding exists — Perl uses OmniBus (generic article fallback).
3//! smfart.cls is based on AMS classes (requires amsgen, amsfonts, amsmath).
4//! We load amsart as base, which provides the core AMS document structure.
5use crate::prelude::*;
6
7#[rustfmt::skip]
8LoadDefinitions!({
9  // Load amsart as the base class — provides AMS document structure,
10  // title/author/abstract environments, theorem-like environments, etc.
11  load_class("amsart", Vec::new(), Tokens!())?;
12
13  // smfart-specific commands that papers may use.
14  // These are simplified stubs matching the raw TeX definitions.
15  // \alttitle — alternate title (French/English toggle)
16  DefMacro!("\\alttitle{}", "");
17  // \altauthor — alternate author listing
18  DefMacro!("\\altauthor{}", "");
19  // \dedicatory — dedication line
20  DefMacro!("\\dedicatory{}", "\\par\\noindent\\itshape #1\\par");
21  // \keywords — keywords (often used via \subjclass in AMS)
22  DefMacro!("\\keywords{}", "");
23  // \altabstract environment — alternate language abstract
24  DefEnvironment!("{altabstract}", "#body");
25  // \resu environment — French resumé (abstract)
26  DefEnvironment!("{resu}", "#body");
27  // \smfbymark — SMF "by" connector for author lists
28  DefMacro!("\\smfbymark", " ");
29  // \altmaketitle — alternate maketitle (no-op, \maketitle handles it)
30  DefMacro!("\\altmaketitle", "");
31  // smfart.cls raw TeX saves/restores trivlist internals
32  Let!("\\smf@org@trivlist", "\\@trivlist");
33  Let!("\\smf@org@endtrivlist", "\\endtrivlist");
34  DefMacro!("\\smfbyline{}{}", "#1 \\textsc{#2}");
35  DefMacro!("\\@classname", "smfart");
36});
