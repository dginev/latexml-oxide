1//! frenchb.ldf — French (deprecated babel option name)
2//!
3//! Perl: `frenchb.ldf.ltxml` — calls `InputDefinitions('frenchb',
4//! type => 'ldf', noltxml => 1)` to raw-load TeX Live's `frenchb.ldf`.
5//!
6//! NOTE: This module's `load_definitions` is currently NOT registered
7//! in `latexml_package/src/lib.rs` for `("frenchb","ldf")` — that slot
8//! is wired to `french_ldf::load_definitions`, where the babel-level
9//! frenchb-language alias fix lives (TL2025 babel-french 3.7e turned
10//! `frenchb.ldf` into a deprecation shim that doesn't chain
11//! `french.ldf`). This module is retained for documentation parity
12//! with Perl's `frenchb.ldf.ltxml`.
13use crate::prelude::*;
14
15#[rustfmt::skip]
16LoadDefinitions!({
17  // frenchb doesn't normally require this package, but behaves better if it's loaded
18  RequirePackage!("textcomp");
19
20  // Raw-load TL's frenchb.ldf. Relies on `\CurrentOption` having been set
21  // by babel's option pipeline (\ProcessOptions → \ds@frenchb →
22  // \bbl@load@language{frenchb} → \input frenchb.ldf) with LETTER catcodes
23  // so the shim's `\ifx\CurrentOption\bbl@tempa` succeeds.
24  InputDefinitions!("frenchb", noltxml => true, extension => Some(Cow::Borrowed("ldf")));
25
26  // Patches over the raw frenchb/french.ldf for known portability gaps:
27  DefConstructor!("\\fup{}",        "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
28  DefConstructor!("\\FB@up@fake{}", "<ltx:sup>#1</ltx:sup>", enter_horizontal => true);
29
30  // Attempt to make it work with older & newer versions.
31  Let!("\\ltx@orig@nombre", "\\nombre");
32  DefMacro!("\\nombre{}",
33    "\\@ifpackageloaded{numprint}{\\numprint{#1}}{\\ltx@orig@nombre{#1}}");
34
35  // Perl: AtBeginDocument(sub { Let('\degre','\textdegree');
36  // DefMacro('\degres','\hbox to 0.3em{\degre}'); });
37  at_begin_document(TokenizeInternal!(
38    r"\let\degre\textdegree\def\degres{\hbox to 0.3em{\degre}}"
39  ))?;
40});
