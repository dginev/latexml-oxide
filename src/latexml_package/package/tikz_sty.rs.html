4LoadDefinitions!({
5  // Perl: tikz.sty.ltxml (57 lines)
6  // TODO: Full port requires InputDefinitions("tikz", noltxml => true) which loads
7  // the raw TeX tikz package. Also needs:
8  // 1. \use@@tikzlibrary{} — DefPrimitive that loads tikzlibrary*.code.tex files
9  // 2. \tikzcdset — redirect to pgfqkeys
10  // 3. pgf infrastructure (pgfsys-latexml.def)
11  //
12  // Perl source: LaTeXML/lib/LaTeXML/Package/tikz.sty.ltxml
13  // TikZ documents generate many warnings from unported pgf primitives.
14  // Increase MAX_ERRORS to allow processing to complete.
15  AssignValue!("MAX_ERRORS" => Stored::Int(1000));
16
17  DefMacro!("\\pgfmathresult", "0.0");
18  DefMacro!("\\tikz@align@temp", "\\pgfmathresult");
19  InputDefinitions!("tikz", noltxml => true, extension => Some(Cow::Borrowed("sty")));
20  // Perl L35-52: \use@@tikzlibrary — loads tikz/pgf library files
21  // Makes sure libraries are loaded as definitions (InputDefinitions) rather than
22  // content, with catcode management for | character.
23  DefPrimitive!("\\use@@tikzlibrary{}", sub[(libs)] {
24    let mut libs_str = do_expand(libs)?.to_string();
25    // tikzlibrarycd.code.tex checks for `decorations.pathmorphing`
26    // and `decorations.markings` @loaded flags at squiggly/markings
27    // arrow expansion time and emits a vendor PackageError if they're
28    // missing. arXiv's local pdflatex compiles these papers cleanly
29    // because the libraries are autoloaded by pdftex's TikZ; mirror
30    // by injecting both as deps whenever the cd library is requested.
31    // Papers commonly use `\usetikzlibrary{cd}` directly (without
32    // `\usepackage{tikz-cd}`), so the tikz_cd_sty binding never runs.
33    // Witness 2306.03232 (squiggly), 2308.06778 (markings).
34    if libs_str.split(',').any(|s| s.trim() == "cd") {
35      libs_str = s!("decorations.pathmorphing,decorations.markings,{libs_str}");
36    }
37    for lib in libs_str.split(',') {
38      let lib = lib.trim();
39      if lib.is_empty() {
40        continue;
41      }
42      let loaded_cs = T_CS!(s!("\\tikz@library@{lib}@loaded"));
43      if state::lookup_definition(&loaded_cs)?.is_some() {
44        continue; // already loaded
45      }
46      Info!("tikz", lib, &s!("TIKZ LIBRARY {lib}"));
47      Let!(loaded_cs.clone(), T_CS!("\\pgfutil@empty"));
48      // Save and set catcode for | to OTHER (Perl L42-50)
49      let bar_cc = state::lookup_catcode('|');
50      state::assign_catcode('|', Catcode::OTHER, None);
51      // Try tikzlibrary*.code.tex first, then pgflibrary*.code.tex
52      let tikz_name = s!("tikzlibrary{lib}.code");
53      let pgf_name = s!("pgflibrary{lib}.code");
54      let tikz_opts = NewDefault!(InputDefinitionOptions,
55        noerror => true, extension => Some(Cow::Borrowed("tex")));
56      let pgf_opts = NewDefault!(InputDefinitionOptions,
57        noerror => true, extension => Some(Cow::Borrowed("tex")));
58      if input_definitions(&tikz_name, tikz_opts).is_err()
59        && input_definitions(&pgf_name, pgf_opts).is_err() {
60          Warn!("missing_file", lib,
61            &s!("Can't find tikz library '{}' (neither pgf nor tikz). Anticipate undefined macros.", lib));
62        }
63      // Restore catcode for |
64      if let Some(cc) = bar_cc {
65        state::assign_catcode('|', cc, None);
66      }
67    }
68    Ok(Vec::new())
69  });
70
71  DefMacro!("\\tikzcdset", "\\pgfqkeys{/tikz/commutative diagrams}");
72});
