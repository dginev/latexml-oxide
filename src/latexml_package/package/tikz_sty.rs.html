4LoadDefinitions!({
5  // Perl: tikz.sty.ltxml (57 lines)
6  // TODO: Full port requires InputDefinitions("tikz", noltxml => true) which loads
7  // the raw TeX tikz package. Also needs:
8  // 1. \use@@tikzlibrary{} — DefPrimitive that loads tikzlibrary*.code.tex files
9  // 2. \tikzcdset — redirect to pgfqkeys
10  // 3. pgf infrastructure (pgfsys-latexml.def)
11  //
12  // Perl source: LaTeXML/lib/LaTeXML/Package/tikz.sty.ltxml
13  // TikZ documents generate many warnings from unported pgf primitives.
14  // Increase MAX_ERRORS to allow processing to complete.
15  AssignValue!("MAX_ERRORS" => Stored::Int(1000));
16
17  DefMacro!("\\pgfmathresult", "0.0");
18  DefMacro!("\\tikz@align@temp", "\\pgfmathresult");
19  InputDefinitions!("tikz", noltxml => true, extension => Some(Cow::Borrowed("sty")));
20  // Perl L35-52: \use@@tikzlibrary — loads tikz/pgf library files
21  // Makes sure libraries are loaded as definitions (InputDefinitions) rather than
22  // content, with catcode management for | character.
23  DefPrimitive!("\\use@@tikzlibrary{}", sub[(libs)] {
24    let libs_str = do_expand(libs)?.to_string();
25    for lib in libs_str.split(',') {
26      let lib = lib.trim();
27      if lib.is_empty() {
28        continue;
29      }
30      let loaded_cs = T_CS!(s!("\\\\tikz@library@{lib}@loaded"));
31      if state::lookup_definition(&loaded_cs)?.is_some() {
32        continue; // already loaded
33      }
34      Info!("tikz", lib, &s!("TIKZ LIBRARY {lib}"));
35      Let!(loaded_cs.clone(), T_CS!("\\pgfutil@empty"));
36      // Save and set catcode for | to OTHER (Perl L42-50)
37      let bar_cc = state::lookup_catcode('|');
38      state::assign_catcode('|', Catcode::OTHER, None);
39      // Try tikzlibrary*.code.tex first, then pgflibrary*.code.tex
40      let tikz_name = s!("tikzlibrary{lib}.code");
41      let pgf_name = s!("pgflibrary{lib}.code");
42      let tikz_opts = NewDefault!(InputDefinitionOptions,
43        noerror => true, extension => Some(Cow::Borrowed("tex")));
44      let pgf_opts = NewDefault!(InputDefinitionOptions,
45        noerror => true, extension => Some(Cow::Borrowed("tex")));
46      if input_definitions(&tikz_name, tikz_opts).is_err()
47        && input_definitions(&pgf_name, pgf_opts).is_err() {
48          Warn!("missing_file", lib,
49            &s!("Can't find tikz library '{}' (neither pgf nor tikz). Anticipate undefined macros.", lib));
50        }
51      // Restore catcode for |
52      if let Some(cc) = bar_cc {
53        state::assign_catcode('|', cc, None);
54      }
55    }
56    Ok(Vec::new())
57  });
58
59  DefMacro!("\\tikzcdset", "\\pgfqkeys{/tikz/commutative diagrams}");
60});
