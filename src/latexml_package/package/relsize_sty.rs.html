3// Perl relsize.sty.ltxml L23-24, L30-31: the same four chained s/…/…/
4// substitutions to collapse `++`/`--`/`+-`/`-+` prefixes before parsing.
5fn relsize_normalize_sign(input: &str) -> String {
6  let s = input.trim();
7  if let Some(rest) = s.strip_prefix("++") {
8    return rest.to_string();
9  }
10  if let Some(rest) = s.strip_prefix("--") {
11    return rest.to_string();
12  }
13  if let Some(rest) = s.strip_prefix("+-") {
14    return format!("-{}", rest);
15  }
16  if let Some(rest) = s.strip_prefix("-+") {
17    return format!("-{}", rest);
18  }
19  s.to_string()
20}

23LoadDefinitions!({
24  // Perl relsize.sty.ltxml L20-32: `\relsize{s}` multiplies the current
25  // font scale by 1.2^s; `\relscale{s}` multiplies by the literal s.
26  // Both normalize duplicated sign prefixes (`++`, `--`, `+-`, `-+`)
27  // before parsing, so users can chain `\relsize{+\s}` with a stored
28  // signed expansion. Previous stub was a no-op DefMacro — calls like
29  // `\larger` / `\smaller` (expanding to `\relsize{+1}` / `\relsize{-1}`)
30  // silently dropped the scale, leaving text at base size.
31  DefPrimitive!("\\relsize{}", sub[(size)] {
32    let s = relsize_normalize_sign(&size.to_string());
33    if let Ok(n) = s.trim().parse::<f64>() {
34      merge_font(fontmap!(scale => 1.2_f64.powf(n)));
35    }
36    Ok(Vec::new())
37  });
38  DefPrimitive!("\\relscale{}", sub[(size)] {
39    let s = relsize_normalize_sign(&size.to_string());
40    if let Ok(n) = s.trim().parse::<f64>() {
41      merge_font(fontmap!(scale => n));
42    }
43    Ok(Vec::new())
44  });
45
46  DefMacro!("\\textscale{}{}", "\\begingroup\\relscale{#1}#2\\endgroup");
47
48  DefMacro!("\\larger Optional:1",         "\\relsize{+#1}");
49  DefMacro!("\\smaller Optional:1",        "\\relsize{-#1}");
50  DefMacro!("\\textlarger Optional:1 {}",  "{\\relsize{+#1}#2}");
51  DefMacro!("\\textsmaller Optional:1 {}", "{\\relsize{-#1}#2}");
