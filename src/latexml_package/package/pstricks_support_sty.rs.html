144  // Angle units — Perl L653-654. Sets `\degrees` state to 360 (default)
145  // or to \degrees{angle}-provided value; \radians flips to 2π. The
146  // angle value is stored via AssignValue and consulted by pstricks's
147  // coord readers. Taken as-is from Perl.
148  // `[Float]` would map to a non-typed Tokens under the current prelude;
149  // Perl uses the raw string value as-is, so keep the argument as
150  // Optional tokens and parse at use time.
151  DefPrimitive!("\\degrees []", sub[(angle)] {
152    let v = angle
153      .as_ref()
154      .and_then(|t| t.to_string().trim().parse::<f64>().ok())
155      .unwrap_or(360.0);
156    AssignValue!("\\degrees" => Stored::from(v), None);
157  });
