1use crate::prelude::*;

3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: inst_support.sty.ltxml (PR #2767)
6  // This bit of code supports the \inst style institution markup
7  // used by several document classes and styles [aa, llncs, sv]
8  // HOWEVER, the care support is now built into Engine/Base_Utility
9  // so this shouldn't be needed.
10
11  // In inst style, the \author (used once for each author)
12  // gets some "labels" which are used to connect to \affiliation
13  // (also one per affiliation); the affiliation with the matching label
14  // is attached to the author.
15
16  // \author[marks]{author}
17  // Perl inst_support.sty.ltxml L33 defines `\author{}` — a SINGLE-arg macro
18  // that DROPS the optional `[marks]` its own comment documents, and whose
19  // `\lx@clear@creators` wipes prior authors on every call. A class using the
20  // inst convention of one `\author[label]{name}` PER author (ifacconf:
21  // `\author[First]{Eryn Vaid}` ×4) then reads the literal `[` as the name and
22  // keeps only the last — Perl and Rust both emit a single `[` personname
23  // (shared upstream bug; witness arXiv:2605.00004, whose pdflatex PDF lists all
24  // four authors). Surpass-Perl (OXIDIZED_DESIGN #53):
25  //   * accept the optional `[marks]` so `[` is never mistaken for the name,
26  //   * take the name from #2 and split it on \and/\And/comma as before,
27  //   * ACCUMULATE across calls — drop the per-call `\lx@clear@creators`, which
28  //     is a no-op on the first call anyway, so single-`\author` classes are
29  //     unaffected (and aa/llncs/sv define their own `\author` regardless).
30  // The `[marks]` (author↔affiliation label) are dropped, matching Perl's own
31  // handling; wiring them to the affiliation annotation is a separate follow-up.
32  DefMacro!("\\author[]{}",
33    "\\lx@splitting{\\lx@add@author}{\\and\\And,}{#2}");
34  DefMacro!("\\institute{}",
35    "\\lx@clear@frontmatter{ltx:contact}[role=affiliation]\\lx@splitting{\\lx@add@contact[role=affiliation,labelseq=affiliation]}{\\and\\And}{#1}");
36  DefMacro!("\\inst{}", "\\lx@request@frontmatter@annotation[affiliation]{#1}");
37
38  // \and variants — Perl L41-45
39  Let!("\\at", "\\and"); // Actually this is different than \and, but...
40  Let!("\\iand", "\\and");
41  Let!("\\nand", "\\and");
42  Let!("\\lastand", "\\and");
43  Let!("\\AND", "\\and");
44});
