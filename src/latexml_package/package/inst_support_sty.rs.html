5/// Perl inst_support.sty.ltxml L89-116 `relocateInstitute` — moves
6/// institutetext/email ltx:note bodies into the matching author's
7/// ltx:contact[@role='institutemark'/'emailmark']. Mirrors the engine-level
8/// `relocate_footnote` pattern (latex_constructs.rs L179).
9pub fn relocate_institute(document: &mut Document, instnode: &mut Node) -> Result<()> {
10  let role = instnode.get_attribute("role").unwrap_or_default();
11  if role == "institutetext" {
12    if let Some(mark) = instnode.get_attribute("mark") {
13      let matches = document.findnodes(
14        &format!(".//ltx:contact[@role='institutemark'][@_mark='{mark}']"),
15        None,
16      );
17      if !matches.is_empty() {
18        let children = instnode.get_child_nodes();
19        for mut author in matches {
20          document.append_clone(&mut author, children.clone())?;
21          document.set_attribute(&mut author, "role", "institute")?;
22          let _ = author.remove_attribute("_mark");
23        }
24        document.safe_unlink(instnode.clone());
25      }
26      // Perl L101-104 fallback (append a new ltx:contact to every author
27      // lacking an institutemark) uses append_tree to construct a fresh
28      // <ltx:contact role='institute'> subtree. Deferred — the matching-
29      // mark path covers the common case and the fallback triggers only
30      // for malformed inputs where \inst mark numbers don't line up with
31      // \institute entries.
32    }
33  } else if role == "email" {
34    if let Some(mark) = instnode.get_attribute("mark") {
35      let matches = document.findnodes(
36        &format!(".//ltx:contact[@role='emailmark'][@_mark='{mark}']"),
37        None,
38      );
39      if !matches.is_empty() {
40        let children = instnode.get_child_nodes();
41        for mut author in matches {
42          document.append_clone(&mut author, children.clone())?;
43          document.set_attribute(&mut author, "role", "email")?;
44          let _ = author.remove_attribute("_mark");
45        }
46        document.safe_unlink(instnode.clone());
47      }
48    }
49  }
50  Ok(())
51}
