9  // Perl: proof.sty.ltxml
10  // The premises can be separated by "&" (which is NOT being used for alignment!)
11  DefMath!("\\lx@proof@logical@and", "\u{2003}", role => "ADDOP", meaning => "and");
12
13  // Perl proof.sty.ltxml L20-28: split the argument on T_ALIGN (`&`),
14  // which in proof.sty separates premises (not for column alignment).
15  // Single clause → pass through; multiple → wrap with I_apply around
16  // \lx@proof@logical@and so the semantic structure is an n-ary AND.
17  DefMacro!("\\lx@proof@split@and {}", sub[(tokens)] {
18    let clauses = split_tokens(tokens, vec![T_ALIGN!().into()]);
19    if clauses.is_empty() {
20      Ok(Tokens!())
21    } else if clauses.len() == 1 {
22      Ok(clauses.into_iter().next().unwrap())
23    } else {
24      Ok(i_apply(&[], Tokens!(T_CS!("\\lx@proof@logical@and")), clauses))
25    }
26  });
27
28  // \lx@proof@stack{}{}{}{}:
29  // An extremely contrived stack of premises and conclusion.
30  // Args are $top, $middle (if any), $bottom, $sidelabel (if any).
31  // Perl L44-49 after_digest compares top/bot widths; if top is wider,
32  // stretches bot (the bars whatsit) to match. Without this, the bars
33  // over the conclusion are only as wide as the conclusion itself.
34  DefConstructor!("\\lx@proof@stack{}{}{}{}",
35    "<ltx:XMArray vattach='bottom'><ltx:XMRow><ltx:XMCell>#1</ltx:XMCell>?#4(<ltx:XMCell rowspan='3'>#4</ltx:XMCell>)()</ltx:XMRow>?#2(<ltx:XMRow><ltx:XMCell>#2</ltx:XMCell></ltx:XMRow>)()<ltx:XMRow><ltx:XMCell>#3</ltx:XMCell></ltx:XMRow></ltx:XMArray>",
36    after_digest => sub[whatsit] {
37      let top_w = whatsit.get_arg(1).and_then(|a| a.clone().get_width(None).ok().flatten());
38      let bot_w = whatsit.get_arg(3).and_then(|a| a.clone().get_width(None).ok().flatten());
39      if let (Some(RegisterValue::Dimension(tw)),
40              Some(RegisterValue::Dimension(bw))) = (top_w, bot_w)
41        && tw.value_of() > bw.value_of()
42          && let Some(bot) = whatsit.get_arg_mut(3) {
43            bot.set_property("stretchto", Stored::from(
44              RegisterValue::Dimension(tw)
45            ));
46          }
47    });
48
49  // Perl L52-57 emits `width='#stretchto'` on the XMWrap so the bars
50  // stretch to the width stashed by \lx@proof@stack's after_digest.
51  DefConstructor!("\\lx@proof@bars OptionalMatch:= {}",
52    "<ltx:XMApp><ltx:XMTok role='OVERACCENT'>\u{203E}</ltx:XMTok>?#1(<ltx:XMApp><ltx:XMTok role='OVERACCENT'>\u{203E}</ltx:XMTok>)()<ltx:XMWrap width='#stretchto'>#2</ltx:XMWrap>?#1(</ltx:XMApp>)()</ltx:XMApp>");
53
54  // Perl proof.sty.ltxml L60-71: \infer (* or =) [label] {lower}{uppers}
55  // builds an XMDual — content branch records
56  //   I_apply(I_symbol(meaning => "infer"|"multistep-infer"), lower, uppers-split)
57  // presentation branch wraps \lx@proof@stack around the lower + bars + label.
58  // Args shared between branches (in order): lower, split-and(uppers), label.
59  DefMacro!("\\infer OptionalMatch:* OptionalMatch:= [] {}{}",
60    sub[(multistep, double, label, lower, uppers)] {
61      let meaning = if multistep.is_some() { "multistep-infer" } else { "infer" };
62      let content = i_apply(
63        &[],
64        i_symbol(&[("meaning", Tokenize!(meaning))], None),
65        vec![Tokens!(i_arg("1")), Tokens!(i_arg("2"))],
66      );
67      let presentation = if multistep.is_some() {
68        // Invocation(\lx@proof@stack, I_arg(2), \vdots, I_arg(1), I_arg(3))
69        Invocation!(T_CS!("\\lx@proof@stack"), vec![
70          Tokens!(i_arg("2")),
71          Tokens!(T_CS!("\\vdots")),
72          Tokens!(i_arg("1")),
73          Tokens!(i_arg("3")),
74        ])
75      } else {
76        // Invocation(\lx@proof@bars, double, I_arg(1))
77        let bars = Invocation!(T_CS!("\\lx@proof@bars"), vec![
78          double.unwrap_or_default(),
79          Tokens!(i_arg("1")),
80        ]);
81        // Invocation(\lx@proof@stack, I_arg(2), undef, bars, I_arg(3))
82        Invocation!(T_CS!("\\lx@proof@stack"), vec![
83          Tokens!(i_arg("2")),
84          Tokens!(),
85          bars,
86          Tokens!(i_arg("3")),
87        ])
88      };
89      let uppers_split =
90        Invocation!(T_CS!("\\lx@proof@split@and"), vec![uppers]);
91      let cmd = i_dual(&[], content, presentation, vec![
92        lower,
93        uppers_split,
94        label.unwrap_or_default(),
95      ])?;
96      let mut out: Vec<Token> = Vec::with_capacity(3 + cmd.len());
97      out.push(T_CS!("\\ensuremath"));
98      out.push(T_BEGIN!());
99      out.extend(cmd.unlist());
100      out.push(T_END!());
101      Ok(Tokens::new(out))
102    }
103  );
104
105  // Perl proof.sty.ltxml L74-81: \deduce [label] {lower}{uppers}
106  // Similar to \infer but no bars; label (if any) replaces the bars row.
107  DefMacro!("\\deduce [] {}{}", sub[(label, lower, uppers)] {
108    let content = i_apply(
109      &[],
110      i_symbol(&[("meaning", Tokenize!("deduce"))], None),
111      vec![Tokens!(i_arg("1")), Tokens!(i_arg("2"))],
112    );
113    // Invocation(\lx@proof@stack, I_arg(2), label, I_arg(1))
114    // Perl passes only 3 args to \lx@proof@stack{}{}{}{}; 4th (sidelabel)
115    // defaults to empty.
116    let presentation = Invocation!(T_CS!("\\lx@proof@stack"), vec![
117      Tokens!(i_arg("2")),
118      label.unwrap_or_default(),
119      Tokens!(i_arg("1")),
120      Tokens!(),
121    ]);
122    let uppers_split =
123      Invocation!(T_CS!("\\lx@proof@split@and"), vec![uppers]);
124    let cmd = i_dual(&[], content, presentation, vec![
125      lower,
126      uppers_split,
127    ])?;
128    let mut out: Vec<Token> = Vec::with_capacity(3 + cmd.len());
129    out.push(T_CS!("\\ensuremath"));
130    out.push(T_BEGIN!());
131    out.extend(cmd.unlist());
132    out.push(T_END!());
133    Ok(Tokens::new(out))
134  });
