7  // Override numprint's `n` and `N` column type rewrites. The raw package
8  // defines these using \nprt@rewrite@ with \@ifnextchar and \nprt@digittoks
9  // which produce unrecognized tokens in our alignment template parser.
10  // Simplify to plain right-aligned columns (loses decimal alignment but
11  // prevents 54+ stray alignment errors).
12  RawTeX!(r#"\makeatletter
13\renewcommand{\NC@rewrite@n}[1]{\NC@find r}%
14\renewcommand{\NC@rewrite@N}[1]{\NC@find r}%
15\makeatother"#);
16
17  Let!("\\ltx@orig@numprint", "\\numprint");
18  DefMacro!("\\numprint[]{}",
19    "\\ifx.#1.\\ltx@numprint@{#2}\\else\\ltx@numprint@@{#1}{#2}\\fi");
20  DefMacro!("\\ltx@numprint@{}",
21    "\\ifmmode\\ltx@math@numprint@{#1}\\else\\ltx@text@numprint@{#1}\\fi");
22  DefMacro!("\\ltx@numprint@@{}{}",
23    "\\ifmmode\\ltx@math@numprint@@{#1}{#2}\\else\\ltx@text@numprint@@{#1}{#2}\\fi");
24  DefMacro!("\\ltx@text@numprint@{}",    "\\ltx@text@number{\\ltx@orig@numprint{#1}}");
25  DefMacro!("\\ltx@text@numprint@@{}{}", "\\ltx@text@number{\\ltx@orig@numprint[#1]{#2}}");
26  // In text mode, \numprint wraps output in ltx:text class="ltx_number".
27  // But numprint's internal formatting can produce math elements (XMTok)
28  // that cause "not allowed in ltx:text" schema errors. Use pass-through
29  // to avoid schema violations. The ltx_number class is cosmetic only.
30  DefMacro!("\\ltx@text@number{}", "#1");
31  DefMacro!("\\ltx@math@numprint@{}",
32    "\\ltx@math@@numprint@{#1}{\\ltx@orig@numprint{#1}}");
33  DefMacro!("\\ltx@math@numprint@@{}{}",
34    "\\ltx@math@@numprint@@{#1}{#2}{\\ltx@mark@units{#1}}{\\ltx@orig@numprint[#1]{#2}}");
35
36  // Math constructors (Perl L59-76)
37  DefConstructor!("\\ltx@math@@numprint@ {} {}",
38    "<ltx:XMDual>\
39       <ltx:XMTok meaning='#value' role='NUMBER'>#value</ltx:XMTok>\
40       <ltx:XMWrap>#2</ltx:XMWrap>\
41     </ltx:XMDual>",
42    properties => sub[args] {
43      let value = args.first().and_then(|a| a.as_ref())
44        .map(|a| a.to_string()).unwrap_or_default();
45      Ok(stored_map!("value" => value))
46    });
47  DefConstructor!("\\ltx@math@@numprint@@ {} {} {} {}",
48    "<ltx:XMDual>\
49       <ltx:XMApp>\
50         <ltx:XMTok meaning='times' role='MULOP'>\u{2062}</ltx:XMTok>\
51         <ltx:XMTok meaning='#value' role='NUMBER'>#value</ltx:XMTok>\
52         <ltx:XMWrap>#3</ltx:XMWrap>\
53       </ltx:XMApp>\
54       <ltx:XMWrap>#4</ltx:XMWrap>\
55     </ltx:XMDual>",
56    properties => sub[args] {
57      let value = args.get(1).and_then(|a| a.as_ref())
58        .map(|a| a.to_string()).unwrap_or_default();
59      Ok(stored_map!("value" => value))
60    });
61
62  // Unit marking (Perl L99-111): simplified — just absorb content
63  DefConstructor!("\\ltx@mark@units{}", "#1", reversion => "#1");
64
65  // Sign symbols (Perl L79-84)
66  DefMacro!(T_CS!("\\nprt@sign@+"),  None, "\\ifmmode+\\else+\\fi");
67  DefMacro!(T_CS!("\\nprt@sign@-"),  None, "\\ifmmode-\\else-\\fi");
68  DefMacro!(T_CS!("\\nprt@sign@+-"), None, "\\ifmmode\\pm\\else\\pm\\fi");

70  // Product sign (Perl L87-94)

72  RawTeX!(r"\expandafter\def\csname ltx@text@prod\string\times\endcsname{×}");
73  RawTeX!(r"\expandafter\def\csname ltx@text@prod\string\cdot\endcsname{⋅}");
74
75  DefMacro!("\\npunitcommand{}", "\\ensuremath{\\mathrm{#1}}");
