7  // Override numprint's `n` and `N` column type rewrites. The raw package
8  // defines these using \nprt@rewrite@ with \@ifnextchar and \nprt@digittoks
9  // which produce unrecognized tokens in our alignment template parser.
10  // Simplify to plain right-aligned columns (loses decimal alignment but
11  // prevents 54+ stray alignment errors).
12  RawTeX!(r#"\makeatletter
13\renewcommand{\NC@rewrite@n}[1]{\NC@find r}%
14\renewcommand{\NC@rewrite@N}[1]{\NC@find r}%
15\makeatother"#);

17  Let!("\\ltx@orig@numprint", "\\numprint");
18  DefMacro!("\\numprint[]{}",
19    "\\ifx.#1.\\ltx@numprint@{#2}\\else\\ltx@numprint@@{#1}{#2}\\fi");
20  DefMacro!("\\ltx@numprint@{}",
21    "\\ifmmode\\ltx@math@numprint@{#1}\\else\\ltx@text@numprint@{#1}\\fi");
22  DefMacro!("\\ltx@numprint@@{}{}",
23    "\\ifmmode\\ltx@math@numprint@@{#1}{#2}\\else\\ltx@text@numprint@@{#1}{#2}\\fi");
24  DefMacro!("\\ltx@text@numprint@{}",    "\\ltx@text@number{\\ltx@orig@numprint{#1}}");
25  DefMacro!("\\ltx@text@numprint@@{}{}", "\\ltx@text@number{\\ltx@orig@numprint[#1]{#2}}");
26  // In text mode, \numprint wraps output in ltx:text class="ltx_number".
27  // Port of Perl: DefConstructor('\ltx@text@number{}',
28  //   "<ltx:text class='ltx_number' _noautoclose='1'>#1</ltx:text>",
29  //   enterHorizontal => 1);
30  // Use \ifmmode guard: skip wrapping in math mode (ltx:text invalid inside XMath).
31  DefMacro!("\\ltx@text@number{}",
32    "\\ifmmode#1\\else\\ltx@text@number@wrap{#1}\\fi");
33  DefConstructor!("\\ltx@text@number@wrap{}",
34    "<ltx:text class='ltx_number' _noautoclose='1'>#1</ltx:text>");
35  DefMacro!("\\ltx@math@numprint@{}",
36    "\\ltx@math@@numprint@{#1}{\\ltx@orig@numprint{#1}}");
37  DefMacro!("\\ltx@math@numprint@@{}{}",
38    "\\ltx@math@@numprint@@{#1}{#2}{\\ltx@mark@units{#1}}{\\ltx@orig@numprint[#1]{#2}}");

40  // Math constructors (Perl L59-76)
41  DefConstructor!("\\ltx@math@@numprint@ {} {}",
42    "<ltx:XMDual>\
43       <ltx:XMTok meaning='#value' role='NUMBER'>#value</ltx:XMTok>\
44       <ltx:XMWrap>#2</ltx:XMWrap>\
45     </ltx:XMDual>",
46    properties => sub[args] {
47      let value = args.first().and_then(|a| a.as_ref())
48        .map(|a| a.to_string()).unwrap_or_default();
49      Ok(stored_map!("value" => value))
50    });
51  DefConstructor!("\\ltx@math@@numprint@@ {} {} {} {}",
52    "<ltx:XMDual>\
53       <ltx:XMApp>\
54         <ltx:XMTok meaning='times' role='MULOP'>\u{2062}</ltx:XMTok>\
55         <ltx:XMTok meaning='#value' role='NUMBER'>#value</ltx:XMTok>\
56         <ltx:XMWrap>#3</ltx:XMWrap>\
57       </ltx:XMApp>\
58       <ltx:XMWrap>#4</ltx:XMWrap>\
59     </ltx:XMDual>",
60    properties => sub[args] {
61      let value = args.get(1).and_then(|a| a.as_ref())
62        .map(|a| a.to_string()).unwrap_or_default();
63      Ok(stored_map!("value" => value))
64    });

74  DefMacro!(T_CS!("\\nprt@sign@-"),  None, "\\ifmmode-\\else\\ltx@text@minus\\fi");
