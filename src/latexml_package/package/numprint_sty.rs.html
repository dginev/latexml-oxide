3/// Perl numprint.sty.ltxml L127-145 — body of the `n`/`N` `DefColumnType`
4/// closures. Reconstructs the alignment-template `before`/`after` token
5/// streams plus `align => char:.` and pushes the column to the active
6/// BUILD_TEMPLATE.
7///
8/// `math_wrap = true`  for `n` (raw TeX `\nprt@rewrite@@{$}{$}...` — `#5`/`#6` of
9///   `\nprt@end` are math shifts, so the rebuilt `\numprint{…}` digests in
10///   math mode).
11/// `math_wrap = false` for `N` (raw TeX `\nprt@rewrite@@{}{}...` — empty `#5`/`#6`,
12///   text-mode digestion).
13///
14/// `\nprt@end` (numprint.sty L433) takes 6 mandatory args:
15///   #1 = man_before digits, #2 = man_after digits,
16///   #3 = exp_before digits, #4 = exp_after digits,
17///   #5, #6 = math-shift wrappers (or empty).
18fn add_numprint_column(args: Vec<ArgWrap>, math_wrap: bool) -> Result<()> {
19  let nd_exp_before = args
20    .first()
21    .cloned()
22    .and_then(ArgWrap::owned_tokens)
23    .unwrap_or_default();
24  let nd_exp_after = args
25    .get(1)
26    .cloned()
27    .and_then(ArgWrap::owned_tokens)
28    .unwrap_or_default();
29  let nd_man_before = args
30    .get(2)
31    .cloned()
32    .and_then(ArgWrap::owned_tokens)
33    .unwrap_or_default();
34  let nd_man_after = args
35    .get(3)
36    .cloned()
37    .and_then(ArgWrap::owned_tokens)
38    .unwrap_or_default();
39
40  let before = Tokens!(T_CS!("\\nprt@begin"), T_CS!("\\ignorespaces"));
41
42  // Perl: after => Invocation(T_CS('\nprt@end'),
43  //   $nd_man_before, $nd_man_after,
44  //   $nd_exp_before, $nd_exp_after,
45  //   T_MATH/Tokens(), T_MATH/Tokens())
46  let mut after_toks: Vec<Token> = Vec::with_capacity(32);
47  after_toks.push(T_CS!("\\nprt@end"));
48  for arg_tokens in [&nd_man_before, &nd_man_after, &nd_exp_before, &nd_exp_after] {
49    after_toks.push(T_BEGIN!());
50    after_toks.extend_from_slice(arg_tokens.unlist_ref());
51    after_toks.push(T_END!());
52  }
53  after_toks.push(T_BEGIN!());
54  if math_wrap {
55    after_toks.push(T_MATH!());
56  }
57  after_toks.push(T_END!());
58  after_toks.push(T_BEGIN!());
59  if math_wrap {
60    after_toks.push(T_MATH!());
61  }
62  after_toks.push(T_END!());
63
64  with_current_build_template(|template_opt| {
65    if let Some(t) = template_opt {
66      t.add_column(Cell {
67        before: Some(before),
68        after: Some(Tokens::new(after_toks)),
69        // Perl: align => 'char:' . ToString(Digest(T_CS('\nprt@decimal')))
70        // \nprt@decimal default expansion is `{.}` (set by \npdecimalsign).
71        align: Some(Align::Char(".".to_string())),
72        ..Cell::default()
73      });
74    }
75  });
76  Ok(())
77}

198  // Track which children belong to the absorbed unit by snapshotting the
199  // last child BEFORE absorb; everything inserted after that point came
200  // from the unit's tokens.
201  DefConstructor!("\\ltx@mark@units{}", sub[document, args, _props] {
202    let parent = document.get_node().clone();
203    let pre_last = parent.get_last_child();
204    if let Some(unit) = args.first().and_then(|a| a.as_ref()) {
205      document.absorb(unit, None)?;
206    }
207    // Iterate newly-added children: start at pre_last.next_sibling()
208    // (or the parent's first_child if pre_last was None).
209    let mut cursor = match pre_last {
210      Some(n) => n.get_next_sibling(),
211      None => parent.get_first_child(),
212    };
213    while let Some(mut node) = cursor {
214      cursor = node.get_next_sibling();
215      // XML_ELEMENT_NODE only — text/comment nodes skip.
216      if node.get_type() != Some(libxml::tree::NodeType::ElementNode) {
217        continue;
218      }
219      // Tag XMTok-like elements with role missing/None or in the
220      // {ID, UNKNOWN, FLOATSUPERSCRIPT} set.
221      let role = node.get_attribute("role").unwrap_or_default();
222      let tag = role.is_empty() || role == "ID" || role == "UNKNOWN" || role == "FLOATSUPERSCRIPT";
223      if tag {
224        document.add_class(&mut node, "ltx_unit")?;
225      }
226    }
227  },
