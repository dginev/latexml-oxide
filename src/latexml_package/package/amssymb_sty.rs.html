3/// Runtime helper for the trivial `DefMath!` shape used 90+ times in
4/// amssymb (DEP-17c). Mirrors DEP-17/DEP-17b: route the same shape
5/// through one fn so each call site compiles to a short call instead
6/// of ~1 KiB of inlined macro expansion. Engine bootstrap pays
7/// `parse_prototype` once per entry.
8fn def_math_sym(cs: &str, present: &str, role: Option<&str>, meaning: Option<&str>) -> Result<()> {
9  let (cs_tok, params) = parse_prototype(cs, true)?;
10  let mut opts = MathPrimitiveOptions::default();
11  if let Some(r) = role { opts.role = Some(r.to_string()); }
12  if let Some(m) = meaning { opts.meaning = Some(m.to_string()); }
13  def_math(cs_tok, params, present.to_string(), opts)?;
14  Ok(())
15}
