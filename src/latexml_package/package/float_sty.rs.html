18  // Creates a new float environment with counter, title format, etc.
19  DefPrimitive!("\\newfloat{}{}{}[]", sub[(ftype, _placement, auxext, within)] {
20    let ftype = ftype.to_string();
21    let auxext = auxext.to_string();
22    let within = within.map(|t| t.to_string()).unwrap_or_default();
23
24    // Set \lx@name@ to type name if not already defined (float.sty only, not newfloat)
25    let name_cs_str = s!("\\lx@name@{ftype}");
26    let name_tok = T_CS!(name_cs_str);
27    if !has_meaning(&name_tok) {
28      def_macro(name_tok, None, Tokens::new(ExplodeText!(ftype)), None)?;
29    }
30
31    // Get current float style for format@title
32    let style = stomach::digest(T_CS!("\\float@style"))
33      .map(|d| d.to_string())
34      .unwrap_or_else(|_| "plain".to_string());
35    let isplain = style.starts_with("plain");
36
37    // \format@title@type{} (float.sty only, not newfloat)
38    let format_cs = s!("\\format@title@{ftype}");
39    let format_body = if isplain {
40      s!("\\lx@tag[][: ]{{\\lx@fnum@@{{{ftype}}}}} #1")
41    } else {
42      s!("\\lx@tag[][ ]{{\\lx@fnum@@{{{ftype}}}}} #1")
43    };
44    let format_cs_tok = T_CS!(format_cs);
45    let format_paramlist = parse_parameters("{}", &format_cs_tok, true)?;
46    def_macro(format_cs_tok, format_paramlist,
47      mouth::tokenize_internal(&format_body), None)?;

53/// Shared helper: creates a float environment with counter, formatting macros,
54/// and DefEnvironmentI. Used by both float.sty \newfloat and newfloat.sty
55/// \DeclareFloatingEnvironment.
56pub fn define_float_environment(ftype: &str, auxext: &str, within: &str) -> Result<()> {
57  // Get current float style. `\float@style` is only defined by float.sty;
58  // newfloat.sty is independent of float.sty and Perl's newfloat.sty.ltxml
59  // never reads `\float@style` (it hardcodes the default layout). To keep
60  // the shared helper usable from both call sites without a spurious
61  // `undefined:\float@style` error when newfloat is loaded alone, probe
62  // the definition first and fall back to "plain" silently when absent.
63  let style = if lookup_definition(&T_CS!("\\float@style"))?.is_some() {
64    stomach::digest(T_CS!("\\float@style"))
65      .map(|d| d.to_string())
66      .unwrap_or_else(|_| "plain".to_string())
67  } else {
68    "plain".to_string()
69  };
70
71  // NewCounter($type, $within)
72  new_counter(ftype, within, None)?;
73
74  // DefMacroI('\the'.$type, ...) if $within
75  if !within.is_empty() {
76    let the_cs = s!("\\the{ftype}");
77    let the_body = s!("\\the{within}.\\arabic{{{ftype}}}");
78    def_macro(
79      T_CS!(the_cs),
80      None,
81      mouth::tokenize_internal(&the_body),
82      None,
83    )?;
84  }
85
86  // DefMacroI('\fnum@font@'.$type, ...)
87  let isplain = style.starts_with("plain");
88  let fnum_cs = s!("\\fnum@font@{ftype}");
89  let fnum_body = if isplain { "\\rmfamily" } else { "\\bfseries" };
90  def_macro(
91    T_CS!(fnum_cs),
92    None,
93    mouth::tokenize_internal(fnum_body),
94    None,
95  )?;
96
97  // DefMacroI('\ext@'.$type, ..., $auxext)
98  let ext_cs = s!("\\ext@{ftype}");
99  def_macro(T_CS!(ext_cs), None, Tokens::new(ExplodeText!(auxext)), None)?;
100
101  // Create the float environment and starred variant
102  let class = s!("ltx_float_{ftype}");
103  create_float_env(ftype, &class, &style)?;
104  let starred_name = s!("{ftype}*");
105  create_float_env(&starred_name, &class, &style)?;
106
107  Ok(())
108}

110fn create_float_env(name: &str, class: &str, style: &str) -> Result<()> {
111  use crate::engine::latex_constructs::{after_float, before_float_ex};
112
113  let class_val = class.to_string();
114  // Extract the base type for before_float (remove trailing *)
115  let base_type = name.trim_end_matches('*').to_string();
116  // Perl float.sty.ltxml L70: starred variant calls beforeFloat with
117  // `double => 1` so \hsize gets \textwidth (spans both columns) rather
118  // than \columnwidth (single column). The detect-by-name-suffix mirrors
119  // the DefEnvironmentI("$type*", ...) branch in Perl's \newfloat.
120  let is_double = name.ends_with('*');
121  let style_str = style.to_string();
122
123  let replacement: ReplacementClosure = Rc::new({
124    let class_val = class_val.clone();
125    move |document: &mut Document,
126          args: &Vec<Option<Digested>>,
127          props: &arena::SymHashMap<Stored>| {
128      let mut av: HashMap<String, String> = HashMap::default();
129      if let Some(stored) = props.get("id") {
130        av.insert("xml:id".into(), stored.to_string());
131      }
132      if let Some(stored) = props.get("inlist") {
133        let inlist_str = stored.to_string();
134        if !inlist_str.is_empty() {
135          av.insert("inlist".into(), inlist_str);
136        }
137      }
138      // ?#1(placement='#1') — placement from optional arg
139      if let Some(Some(ref arg1)) = args.first() {
140        let placement = arg1.to_string();
141        if !placement.is_empty() {
142          av.insert("placement".into(), placement);
143        }
144      }
145      av.insert("class".into(), class_val.clone());
146      document.open_element("ltx:float", Some(av), None)?;
147      // #tags
148      if let Some(stored) = props.get("tags") {
149        let digested_opt: Option<Digested> = stored.into();
150        if let Some(ref digested) = digested_opt {
151          document.absorb(digested, None)?;
152        }
153      }
154      // #body
155      if let Some(stored) = props.get("body") {
156        let digested_opt: Option<Digested> = stored.into();
157        if let Some(ref digested) = digested_opt {
158          document.absorb(digested, None)?;
159        }
160      }
161      document.close_element("ltx:float")?;
162      Ok(())
163    }
164  });
165
166  let env_cs = T_CS!(s!("\\begin{{{name}}}"));
167  let paramlist = parse_parameters("[]", &env_cs, true)?;
168
169  let mut options = ConstructorOptions {
170    mode: Some("internal_vertical".into()),
171    ..Default::default()
172  };
173
174  // before_digest: beforeFloat($type [, double => 1])
175  let bt = base_type.clone();
176  let before_closure: BeforeDigestClosure = Rc::new(move || {
177    before_float_ex(&bt, None, is_double);
178    Ok(Vec::new())
179  });
180  options.before_digest.push(before_closure);
181
182  // after_digest: afterFloat($whatsit)
183  let after_closure: DigestionClosure = Rc::new(move |whatsit: &mut Whatsit| {
184    after_float(whatsit);
185    Ok(Vec::new())
186  });
187  options.after_digest.push(after_closure);
188
189  // after_construct: addFloatFrames
190  let style_for_construct = style_str.clone();
191  let after_construct_closure: ConstructionClosure =
192    Rc::new(move |document: &mut Document, _whatsit: &Whatsit| {
193      add_float_frames(document, &style_for_construct)?;
194      Ok(())
195    });
196  options.after_construct.push(after_construct_closure);
197
198  options.properties = Rc::new(|_| Ok(stored_map!("layout" => "vertical")));
199
200  def_environment(name.to_string(), paramlist, Some(replacement), options);
201  Ok(())
202}

204/// Perl: addFloatFrames (float.sty.ltxml L76-85)
205pub fn add_float_frames(document: &mut Document, style: &str) -> Result<()> {
206  let caption_qname = arena::pin_static("ltx:caption");
207  let toccaption_qname = arena::pin_static("ltx:toccaption");
208  let node = document.get_node();
209  if let Some(float_node) = node.get_last_child() {
210    match style {
211      "ruled" => {
212        let mut float_mut = float_node.clone();
213        document.set_attribute(&mut float_mut, "framed", "top")?;
214        // inner frame: topbottom on first non-caption child
215        for child in float_node.get_child_elements() {
216          let qname = latexml_core::document::get_node_qname(&child);
217          if qname != caption_qname && qname != toccaption_qname {
218            let mut child_mut = child;
219            document.set_attribute(&mut child_mut, "framed", "topbottom")?;
220            break;
221          }
222        }
223      },
224      "boxed" => {
225        // inner frame: rectangle on first non-caption child
226        for child in float_node.get_child_elements() {
227          let qname = latexml_core::document::get_node_qname(&child);
228          if qname != caption_qname && qname != toccaption_qname {
229            let mut child_mut = child;
230            document.set_attribute(&mut child_mut, "framed", "rectangle")?;
231            break;
232          }
233        }
234      },
235      _ => {}, // plain, plaintop — no framing
236    }
237  }
238  Ok(())
