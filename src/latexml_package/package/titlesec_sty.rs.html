3// Perl titlesec.sty.ltxml L35-40: titlesec "shape" option → CSS class.
4// Rust inlines the same four-entry map at each lookup site so both
5// primitives reference a single source of truth.
6fn titlesec_shape_class(shape: &str) -> Option<&'static str> {
7  match shape {
8    "runin" => Some("ltx_runin"),
9    "frame" => Some("ltx_framed ltx_framed_rectangle"),
10    "rightmargin" => Some("ltx_align_right"),
11    "leftmargin" => Some("ltx_align_left"),
12    _ => None,
13  }
14}

18  // Perl: titlesec.sty.ltxml — stubbed since no styling was implemented,
19  // but the star + non-star forms actually DO dynamic-macro work in Perl.
20  // This cycle brings those to parity.

22  def_macro_noop("\\titlelabel{}")?;
23  // \titleformat: star and normal forms
24  DefMacro!("\\titleformat", "\\@ifstar{\\lx@titleformat@star}{\\lx@titleformat}");

26  // Perl L30-34: \titleformat*{\cmd}{format} redefines
27  // `\format@title@<cmd>` to `<format> <space> #1` with 1 parameter.
28  // Users writing \titleformat*{\section}{\bfseries} get a working
29  // override instead of a silent drop. Strip leading backslash from
30  // the command name per Perl L32.
31  // Perl kind is DefMacro with sub body that installs a macro via
32  // DefMacroI. Rust DefPrimitive does the install at stomach time.
33  // WISDOM #44: NOT universally equivalent — safe here because
34  // `\lx@titleformat@star` is only invoked via `\titleformat*{\cmd}{format}`
35  // at preamble/document time, never captured by `\edef`.
36  // WISDOM #44 verified 2026-04-23: zero `\edef`/`\ifx`/`\expandafter`
37  // uses of `\lx@titleformat@star` across LaTeXML/lib + ar5iv-bindings.
38  DefPrimitive!("\\lx@titleformat@star {}{}", sub[(cmd, format)] {
39    let cs_str = cmd.to_string();
40    let sec = cs_str.strip_prefix('\\').unwrap_or(&cs_str);
41    let target = s!("\\format@title@{sec}");
42    let mut body: Vec<Token> = format.unlist();
43    body.push(T_SPACE!());
44    body.push(T_PARAM!());
45    body.push(T_OTHER!("1"));
46    def_macro(T_CS!(&target), convert_latex_args(1, None)?,
47      Tokens::new(body), None)?;
48  });
49
50  // Perl L42-57: \titleformat{cmd}[shape]{format}{label}{sep}{before}[after]
51  // Ignores before/after (Perl too). If shape maps to a CSS class, inject
52  // `\@ADDCLASS{<class>}` after the format tokens. Then defines two
53  // macros:
54  //   \format@title@font@<sec>        := <format> [\@ADDCLASS <class>]
55  //   \format@title@<sec>   ( 1 arg ) := \format@title@font@<sec> <label>
56  //                                      \hspace{<sep>} #1
57  DefPrimitive!("\\lx@titleformat {} [] {}{}{}{}[]",
58    sub[(cmd, shape, format, label, sep, _before, _after)] {
59    let cs_str = cmd.to_string();
60    let sec = cs_str.strip_prefix('\\').unwrap_or(&cs_str);
61    let shape_str = shape.as_ref().map(|s| s.to_string()).unwrap_or_default();
62    let class = titlesec_shape_class(&shape_str);
63
64    // \format@title@font@<sec>
65    let font_target = s!("\\format@title@font@{sec}");
66    let mut font_body: Vec<Token> = format.unlist();
67    if let Some(cls) = class {
68      font_body.push(T_CS!("\\@ADDCLASS"));
69      font_body.push(T_OTHER!(cls));
70    }
71    def_macro(T_CS!(&font_target), None, Tokens::new(font_body), None)?;
72
73    // \format@title@<sec>   (1 arg body)
74    let body_target = s!("\\format@title@{sec}");
75    let mut body: Vec<Token> = Vec::new();
76    body.push(T_CS!(&font_target));
77    body.extend(label.unlist());
78    body.push(T_CS!("\\hspace"));
79    body.push(T_BEGIN!());
80    body.extend(sep.unlist());
81    body.push(T_END!());
82    body.push(T_PARAM!());
83    body.push(T_OTHER!("1"));
84    def_macro(T_CS!(&body_target), convert_latex_args(1, None)?,
85      Tokens::new(body), None)?;
86  });
