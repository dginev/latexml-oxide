132/// Perl: replace_star($tokens, $replacement) — enumitem.sty.ltxml L114-119
133fn replace_star(tokens: &Tokens, replacement: &Token) -> Tokens {
134  let src = tokens.unlist_ref();
135  let mut out = Vec::with_capacity(src.len());
136  for t in src {
137    if t.with_str(|s| s == "*") && t.get_catcode() == Catcode::OTHER {
138      out.push(*replacement);
139    } else {
140      out.push(*t);
141    }
142  }
143  Tokens::new(out)

166/// Perl: store_enumitem_defaults($name, $kv) — enumitem.sty.ltxml L228-237
167fn store_enumitem_defaults(name: &str, kv: &KeyVals) {
168  // Load existing keys directly inside the state/arena closure pair —
169  // the intermediate keys_str String is avoided; we split the interned
170  // &str and collect owned keys straight into the Vec.
171  let mut keys: Vec<String> = with_value(&s!("{name}@keys"), |v| match v {
172    Some(Stored::String(s)) => with(*s, |ks| {
173      ks.split(',')
174        .filter(|k| !k.is_empty())
175        .map(String::from)
176        .collect()
177    }),
178    _ => Vec::new(),
179  });
180
181  for (key, val) in kv.get_pairs() {
182    let val_key = s!("{name}@{key}");
183    match val {
184      ArgWrap::Tokens(t) => {
185        assign_value(&val_key, Stored::Tokens(t.clone()), Some(Scope::Global));
186      },
187      ArgWrap::None => {
188        assign_value(&val_key, Stored::None, Some(Scope::Global));
189      },
190      _ => {
191        assign_value(
192          &val_key,
193          Stored::String(pin(val.to_string())),
194          Some(Scope::Global),
195        );
196      },
197    }
198    if !keys.contains(key) {
199      keys.push(key.clone());
200    }
201  }
202  assign_value(
203    &s!("{name}@keys"),
204    Stored::String(pin(keys.join(","))),
205    Some(Scope::Global),
206  );
207}

209/// Perl: merged_enumitem_keyvals($name, $level, $argkv) — enumitem.sty.ltxml L239-249
210fn merged_enumitem_keyvals(
211  name: &str,
212  level: i64,
213  argkv: Option<&KeyVals>,
214) -> rustc_hash::FxHashMap<String, ArgWrap> {
215  let mut hash = rustc_hash::FxHashMap::default();
216
217  let default_names = [
218    "enumitem_defaults".to_string(),
219    s!("enumitem_{name}_defaults"),
220    s!("enumitem_{name}{level}_defaults"),
221  ];
222
223  for def_name in &default_names {
224    // with_value pulls the keys-string out of the Stored::String arm
225    // without cloning the envelope; the per-key inner lookup still
226    // needs to produce an owned ArgWrap, so we pay the clone there.
227    // Collect keys as owned Vec<String> inside state+arena closures
228    // so the split happens on the interned &str and the owned
229    // intermediary is smaller (Vec<String> of just the keys, not
230    // the whole comma-separated string plus allocs).
231    let keys: Vec<String> = with_value(&s!("{def_name}@keys"), |v| match v {
232      Some(Stored::String(s)) => with(*s, |ks| {
233        ks.split(',')
234          .filter(|k| !k.is_empty())
235          .map(String::from)
236          .collect()
237      }),
238      _ => Vec::new(),
239    });
240    if keys.is_empty() {
241      continue;
242    }
243    for key in &keys {
244      if !key.is_empty() {
245        let val_key = s!("{def_name}@{key}");
246        if let Some(val) = lookup_value(&val_key) {
247          let aw = match val {
248            Stored::Tokens(t) => ArgWrap::Tokens(t),
249            Stored::Number(n) => ArgWrap::Number(n),
250            Stored::None => ArgWrap::None,
251            Stored::String(s) => ArgWrap::Tokens(with(s, mouth::tokenize_internal)),
252            _ => ArgWrap::None,
253          };
254          hash.insert(key.clone(), aw);
255        }
256      }
257    }
258  }
259
260  // Merge argument keyvals last (highest priority)
261  if let Some(kv) = argkv {
262    for (key, val) in kv.get_pairs() {
263      hash.insert(key.clone(), val.clone());
264    }
265  }
266
267  hash
268}

359      } else {

363    ),
364    locked: true,

366    before_digest_end,
367    after_digest_body: vec![after_digest_body],
368    ..Default::default()
369  };
370  def_environment(listname.to_string(), paramlist, Some(replacement), options);
371  Ok(())
372}
