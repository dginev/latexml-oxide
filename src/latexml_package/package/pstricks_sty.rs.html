1//! pstricks.sty — PSTricks graphics package
2//! PSTricks requires DVI backend; we raw-load real pstricks.sty to set
3//! up internal state (\ifpst@useCalc, \ifpst@psfonts, …) then override
4//! user-facing drawing commands as HTML-friendly no-ops.
5//! Perl: pstricks.sty.ltxml (44L) + pstricks_support.sty.ltxml (1057L)
6use crate::prelude::*;
7
8#[rustfmt::skip]
9LoadDefinitions!({
10  RequirePackage!("xcolor");
11  // Perl pstricks.sty.ltxml L40-42:
12  //   InputDefinitions('pstricks', type => 'sty', noltxml => 1);
13  // The raw-load executes the 7 \newifs (\ifpst@useCalc,
14  // \ifpst@psfonts, \ifpstGSfonts, \if@check@engine, \ifpst@xetex,
15  // \ifpst@autopdf, \ifpst@distiller) plus the option processing
16  // machinery. Downstream pstricks.tex (line 1228 \ifpst@useCalc)
17  // depends on these being defined. Without the raw-load the hand-
18  // stub left \ifpst@useCalc/\ifpst@psfonts undefined; witness:
19  // 1907.03162 (`\usepackage{pstricks, pst-plot, pst-eps, pst-grad}`
20  // → 2 Error:undefined diagnostics, vs Perl 0 errors).
21  InputDefinitions!("pstricks", extension => Some(Cow::Borrowed("sty")), noltxml => true);
22  // Perl pstricks.sty.ltxml L44: `RequirePackage('pstricks_support')`.
23  // pstricks_support defines color-CS shorthands (`\blue`, `\red`, …)
24  // that PSTricks-using papers (e.g. arxiv 1107.3732) reference inside
25  // `\tikzpicture{\node{\blue{…}}}`. Without it those CSes are undefined.
26  RequirePackage!("pstricks_support");
27
28  // Core PSTricks parameter setting
29  DefMacro!("\\psset{}", "");
30  DefMacro!("\\newpsobject{}{}{}", "");
31  DefMacro!("\\newpsstyle{}{}", "");
32
33  // PSCoordList-emulator. Perl's pstricks_support.sty.ltxml uses parameter
34  // type `PSCoordList` (variable-arity `(x,y)(x,y)...`) to absorb the paren
35  // tuples that follow most pstricks drawing commands. Without it those
36  // tuples leak as raw text into the document — opening an `<ltx:p>` that
37  // doesn't auto-close before subsequent block content (witness:
38  // hep-ph0102192 minipage-in-figure failure). Recursive `\@ifnextchar`
39  // idiom: peek for `(`; consume one tuple; recurse.
40  RawTeX!("\\def\\lx@psgobble@parens{\\@ifnextchar({\\lx@psgobble@one}{}}");
41  RawTeX!("\\def\\lx@psgobble@one(#1){\\lx@psgobble@parens}");
42
43  // Drawing commands — all no-ops for HTML, but MUST consume trailing
44  // PSCoordList via `\lx@psgobble@parens` so coords don't leak as text.
45  DefMacro!("\\psline OptionalMatch:* []{}", "\\lx@psgobble@parens");
46  DefMacro!("\\psframe OptionalMatch:* []{}", "\\lx@psgobble@parens");
47  DefMacro!("\\pscircle OptionalMatch:* []{}{}", "\\lx@psgobble@parens");
48  DefMacro!("\\psarc OptionalMatch:* []{}{}{}{}", "\\lx@psgobble@parens");
49  DefMacro!("\\psbezier OptionalMatch:* []{}", "\\lx@psgobble@parens");
50  DefMacro!("\\pscurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
51  DefMacro!("\\psecurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
52  DefMacro!("\\psccurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
53  DefMacro!("\\parabola OptionalMatch:* []{}{}", "\\lx@psgobble@parens");
54  DefMacro!("\\pspolygon OptionalMatch:* []{}", "\\lx@psgobble@parens");
55  DefMacro!("\\psdots OptionalMatch:* []{}", "\\lx@psgobble@parens");
56  DefMacro!("\\psdot OptionalMatch:* []{}", "\\lx@psgobble@parens");
57  DefMacro!("\\qline{}{}", "");
58
59  // \Rput[refpoint](x,y){body} — placement at coords (real pstricks
60  // defines this in pstricks.tex / pst-code-put.tex, raw-loaded by
61  // Perl's `InputDefinitions('pstricks', noltxml=>1)`. Our Rust binding
62  // doesn't raw-load pstricks.sty, so define a HTML-shrug stub here:
63  // emit just the body, dropping placement. Witness: 0905.1885
64  // (`\Rput[t](0,0){$H_1$}`).
65  DefMacro!("\\Rput OptionalMatch:* [] Pair {}", "#4");
66  DefMacro!("\\rput OptionalMatch:* [] Pair {}", "#4");
67  DefMacro!("\\uput OptionalMatch:* {} [] Pair {}", "#5");
68  DefMacro!("\\qdisk{}{}", "");
69
70  // Text placement — drop both coords AND the text body. Perl's
71  // `DefPSConstructor` would wrap the labelled text inside a
72  // `<ltx:picture>`, so the picture auto-closes cleanly when block
73  // content (e.g. `\begin{minipage}` inside a figure) follows. Rust's
74  // pstricks port doesn't yet generate `<ltx:picture>`; emitting the
75  // text into the surrounding paragraph traps later block content
76  // inside an `<ltx:p>` (witness: hep-ph0102192 minipage-in-figure
77  // schema errors). Dropping the text body is a fidelity regression —
78  // visible labels like "cocktail"/"thermal" placed via `\rput` are
79  // lost — but it eliminates the cascading schema errors. TODO: port
80  // `DefPSConstructor` framework so pstricks output lives in
81  // `<ltx:picture>` and labels survive.
82  RawTeX!("\\def\\lx@rput@parens(#1)#2{}");
83  RawTeX!("\\def\\lx@rput@bracket[#1]{\\lx@rput@parens}");
84  RawTeX!("\\def\\rput{\\@ifstar\\lx@rput@i\\lx@rput@i}");
85  RawTeX!("\\def\\lx@rput@i{\\@ifnextchar[\\lx@rput@bracket{\\lx@rput@parens}}");
86  RawTeX!("\\def\\lx@uput@parens#1(#2)#3{}"); // {dist}(coord){text} → drop
87  RawTeX!("\\def\\lx@uput@bracket[#1]{\\lx@uput@parens}");
88  RawTeX!("\\def\\uput{\\@ifstar\\lx@uput@i\\lx@uput@i}");
89  RawTeX!("\\def\\lx@uput@i{\\@ifnextchar[\\lx@uput@bracket{\\lx@uput@parens}}");
90  RawTeX!("\\def\\lx@cput@parens(#1)#2{}");
91  RawTeX!("\\def\\lx@cput@bracket[#1]{\\lx@cput@parens}");
92  RawTeX!("\\def\\cput{\\@ifstar\\lx@cput@i\\lx@cput@i}");
93  RawTeX!("\\def\\lx@cput@i{\\@ifnextchar[\\lx@cput@bracket{\\lx@cput@parens}}");
94
95  // Box commands
96  DefMacro!("\\psframebox OptionalMatch:* []{}", "#2");
97  DefMacro!("\\psshadowbox OptionalMatch:* []{}", "#2");
98  DefMacro!("\\pscirclebox OptionalMatch:* []{}", "#2");
99  DefMacro!("\\psovalbox OptionalMatch:* []{}", "#2");
100  DefMacro!("\\psdblframebox OptionalMatch:* []{}", "#2");
101
102  // Environment
103  DefEnvironment!("{pspicture} OptionalMatch:* []{}", "#body");
104  DefEnvironment!("{pspicture*} OptionalMatch:* []{}", "#body");
105
106  // Grid
107  DefMacro!("\\psgrid OptionalMatch:* []{}", "");
108
109  // Misc
110  DefMacro!("\\pscustom OptionalMatch:* []{}", "");
111  DefMacro!("\\psclip{}", "");
112  DefMacro!("\\endpsclip", "");
113  DefMacro!("\\SpecialCoor", "");
114  DefMacro!("\\NormalCoor", "");
115  DefMacro!("\\degrees[]", "");
116  DefMacro!("\\radians", "");
117
118  // \multips(rotation)(translation){n}{stuff} — pstricks "multiple put"
119  // for drawing N copies of an object along a translated step. Rust port
120  // doesn't raw-load pstricks.tex so this CS would otherwise be undefined.
121  // Use RawTeX with a `\def` that consumes the paren-delimited args plus
122  // the two brace args; the body is a no-op since pstricks output is
123  // already suppressed in pspicture stubs. Same pattern as
124  // `iopart_support_sty.rs:185`'s `\def\pt(#1){...}`.
125  // Witness: math0104011 (was 17 errors → 0 with this stub).
126  RawTeX!("\\def\\multips(#1)(#2)#3#4{}");
127});
