1//! pstricks.sty — PSTricks graphics package (stubs)
2//! PSTricks requires DVI backend; we just stub commands to prevent errors.
3//! Perl: pstricks.sty.ltxml (44L) + pstricks_support.sty.ltxml (1057L)
4use crate::prelude::*;
5
6#[rustfmt::skip]
7LoadDefinitions!({
8  RequirePackage!("xcolor");
9  // Perl pstricks.sty.ltxml L44: `RequirePackage('pstricks_support')`.
10  // pstricks_support defines color-CS shorthands (`\blue`, `\red`, …)
11  // that PSTricks-using papers (e.g. arxiv 1107.3732) reference inside
12  // `\tikzpicture{\node{\blue{…}}}`. Without it those CSes are undefined.
13  RequirePackage!("pstricks_support");
14
15  // Core PSTricks parameter setting
16  DefMacro!("\\psset{}", "");
17  DefMacro!("\\newpsobject{}{}{}", "");
18  DefMacro!("\\newpsstyle{}{}", "");
19
20  // PSCoordList-emulator. Perl's pstricks_support.sty.ltxml uses parameter
21  // type `PSCoordList` (variable-arity `(x,y)(x,y)...`) to absorb the paren
22  // tuples that follow most pstricks drawing commands. Without it those
23  // tuples leak as raw text into the document — opening an `<ltx:p>` that
24  // doesn't auto-close before subsequent block content (witness:
25  // hep-ph0102192 minipage-in-figure failure). Recursive `\@ifnextchar`
26  // idiom: peek for `(`; consume one tuple; recurse.
27  RawTeX!("\\def\\lx@psgobble@parens{\\@ifnextchar({\\lx@psgobble@one}{}}");
28  RawTeX!("\\def\\lx@psgobble@one(#1){\\lx@psgobble@parens}");
29
30  // Drawing commands — all no-ops for HTML, but MUST consume trailing
31  // PSCoordList via `\lx@psgobble@parens` so coords don't leak as text.
32  DefMacro!("\\psline OptionalMatch:* []{}", "\\lx@psgobble@parens");
33  DefMacro!("\\psframe OptionalMatch:* []{}", "\\lx@psgobble@parens");
34  DefMacro!("\\pscircle OptionalMatch:* []{}{}", "\\lx@psgobble@parens");
35  DefMacro!("\\psarc OptionalMatch:* []{}{}{}{}", "\\lx@psgobble@parens");
36  DefMacro!("\\psbezier OptionalMatch:* []{}", "\\lx@psgobble@parens");
37  DefMacro!("\\pscurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
38  DefMacro!("\\psecurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
39  DefMacro!("\\psccurve OptionalMatch:* []{}", "\\lx@psgobble@parens");
40  DefMacro!("\\parabola OptionalMatch:* []{}{}", "\\lx@psgobble@parens");
41  DefMacro!("\\pspolygon OptionalMatch:* []{}", "\\lx@psgobble@parens");
42  DefMacro!("\\psdots OptionalMatch:* []{}", "\\lx@psgobble@parens");
43  DefMacro!("\\psdot OptionalMatch:* []{}", "\\lx@psgobble@parens");
44  DefMacro!("\\qline{}{}", "");
45  DefMacro!("\\qdisk{}{}", "");
46
47  // Text placement — drop both coords AND the text body. Perl's
48  // `DefPSConstructor` would wrap the labelled text inside a
49  // `<ltx:picture>`, so the picture auto-closes cleanly when block
50  // content (e.g. `\begin{minipage}` inside a figure) follows. Rust's
51  // pstricks port doesn't yet generate `<ltx:picture>`; emitting the
52  // text into the surrounding paragraph traps later block content
53  // inside an `<ltx:p>` (witness: hep-ph0102192 minipage-in-figure
54  // schema errors). Dropping the text body is a fidelity regression —
55  // visible labels like "cocktail"/"thermal" placed via `\rput` are
56  // lost — but it eliminates the cascading schema errors. TODO: port
57  // `DefPSConstructor` framework so pstricks output lives in
58  // `<ltx:picture>` and labels survive.
59  RawTeX!("\\def\\lx@rput@parens(#1)#2{}");
60  RawTeX!("\\def\\lx@rput@bracket[#1]{\\lx@rput@parens}");
61  RawTeX!("\\def\\rput{\\@ifstar\\lx@rput@i\\lx@rput@i}");
62  RawTeX!("\\def\\lx@rput@i{\\@ifnextchar[\\lx@rput@bracket{\\lx@rput@parens}}");
63  RawTeX!("\\def\\lx@uput@parens#1(#2)#3{}"); // {dist}(coord){text} → drop
64  RawTeX!("\\def\\lx@uput@bracket[#1]{\\lx@uput@parens}");
65  RawTeX!("\\def\\uput{\\@ifstar\\lx@uput@i\\lx@uput@i}");
66  RawTeX!("\\def\\lx@uput@i{\\@ifnextchar[\\lx@uput@bracket{\\lx@uput@parens}}");
67  RawTeX!("\\def\\lx@cput@parens(#1)#2{}");
68  RawTeX!("\\def\\lx@cput@bracket[#1]{\\lx@cput@parens}");
69  RawTeX!("\\def\\cput{\\@ifstar\\lx@cput@i\\lx@cput@i}");
70  RawTeX!("\\def\\lx@cput@i{\\@ifnextchar[\\lx@cput@bracket{\\lx@cput@parens}}");
71
72  // Box commands
73  DefMacro!("\\psframebox OptionalMatch:* []{}", "#2");
74  DefMacro!("\\psshadowbox OptionalMatch:* []{}", "#2");
75  DefMacro!("\\pscirclebox OptionalMatch:* []{}", "#2");
76  DefMacro!("\\psovalbox OptionalMatch:* []{}", "#2");
77  DefMacro!("\\psdblframebox OptionalMatch:* []{}", "#2");
78
79  // Environment
80  DefEnvironment!("{pspicture} OptionalMatch:* []{}", "#body");
81  DefEnvironment!("{pspicture*} OptionalMatch:* []{}", "#body");
82
83  // Grid
84  DefMacro!("\\psgrid OptionalMatch:* []{}", "");
85
86  // Misc
87  DefMacro!("\\pscustom OptionalMatch:* []{}", "");
88  DefMacro!("\\psclip{}", "");
89  DefMacro!("\\endpsclip", "");
90  DefMacro!("\\SpecialCoor", "");
91  DefMacro!("\\NormalCoor", "");
92  DefMacro!("\\degrees[]", "");
93  DefMacro!("\\radians", "");
94
95  // \multips(rotation)(translation){n}{stuff} — pstricks "multiple put"
96  // for drawing N copies of an object along a translated step. Rust port
97  // doesn't raw-load pstricks.tex so this CS would otherwise be undefined.
98  // Use RawTeX with a `\def` that consumes the paren-delimited args plus
99  // the two brace args; the body is a no-op since pstricks output is
100  // already suppressed in pspicture stubs. Same pattern as
101  // `iopart_support_sty.rs:185`'s `\def\pt(#1){...}`.
102  // Witness: math0104011 (was 17 errors → 0 with this stub).
103  RawTeX!("\\def\\multips(#1)(#2)#3#4{}");
104});
