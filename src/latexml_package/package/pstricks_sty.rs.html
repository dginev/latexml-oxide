3//! up internal state (\ifpst@useCalc, \ifpst@psfonts, …) then override
4//! user-facing drawing commands as HTML-friendly no-ops.

12  //   InputDefinitions('pstricks', type => 'sty', noltxml => 1);
13  // The raw-load executes the 7 \newifs (\ifpst@useCalc,
14  // \ifpst@psfonts, \ifpstGSfonts, \if@check@engine, \ifpst@xetex,

31  // Perl pstricks_support.sty.ltxml L849-861: `\newpsobject{name}{oldname}{keyval}`
32  // dynamically defines `\<name>` to forward to `\<oldname>` with the saved
33  // `<keyval>` baked into the optional argument. The paper's drawing object is
34  // then drawn by the resolved `\<oldname>` (typically `\psline`, `\psdots`).
35  // Perl stores `oldname` and `keyval` in two LookupValue keys so the
36  // generated forwarder can read them at call time, and additionally merges
37  // a user-supplied `[opt]` into the saved `keyval`.
38  //
39  // Witness: physics/9710028 uses
40  //   \newpsobject{PST@Border}{psline}{linewidth=.0015,linestyle=solid}
41  // then later calls `\PST@Border(...)`. With the prior no-op stub
42  // `\PST@Border` stayed undefined and Rust errored; Perl recovered.
43  DefPrimitive!("\\newpsobject{}{}{}", sub[(newname, oldname, keyval)] {
44    let newcs    = s!("\\{}", newname.to_string());
45    let oldcs    = s!("\\{}", oldname.to_string());
46    let keystr   = keyval.to_string();
47    let new_tok  = T_CS!(newcs);
48    let params   = parse_parameters("OptionalMatch:* []", &new_tok, true)?;
49    // Generated forwarder closure: read OptionalMatch:* and []; emit
50    //   \<old>(*)([combined-key])
51    // combined-key = saved-key + ',' + user-key (Perl L855).
52    let oldcs_owned = oldcs;
53    let key_owned   = keystr;
54    let body_closure: ExpansionBody = ExpansionBody::Closure(Rc::new(move |args| {
55      let star = args.first().map(|a| !a.is_none()).unwrap_or(false);
56      let usr  = args.get(1)
57        .and_then(|a| match a.as_tokens() { Ok(Some(t)) => Some(t.to_string()), _ => None })
58        .unwrap_or_default();
59      let combined = match (key_owned.is_empty(), usr.is_empty()) {
60        (false, false) => s!("{},{}", key_owned, usr),
61        (false, true)  => key_owned.clone(),
62        (true,  false) => usr,
63        (true,  true)  => String::new(),
64      };
65      let mut out = vec![T_CS!(oldcs_owned.clone())];
66      if star { out.push(T_OTHER!("*")); }
67      if !combined.is_empty() {
68        out.push(T_OTHER!("["));
69        out.extend(Explode!(combined));
70        out.push(T_OTHER!("]"));
71      }
72      // Perl L856 — emit the suffix only; the paren-coords tuple is
73      // consumed by the resolved \psline (or sibling) macro's own
74      // `\lx@psgobble@parens` chain.
75      Ok(Tokens::new(out))
76    }));
77    def_macro(new_tok, params, Some(body_closure), None)?;
78  });

82  // PSCoordList-emulator. Perl's pstricks_support.sty.ltxml uses parameter
83  // type `PSCoordList` (variable-arity `(x,y)(x,y)...`) to absorb the paren
84  // tuples that follow most pstricks drawing commands. Without it those
85  // tuples leak as raw text into the document — opening an `<ltx:p>` that
86  // doesn't auto-close before subsequent block content (witness:
87  // hep-ph0102192 minipage-in-figure failure). Recursive `\@ifnextchar`
88  // idiom: peek for `(`; consume one tuple; recurse.
89  RawTeX!("\\def\\lx@psgobble@parens{\\@ifnextchar({\\lx@psgobble@one}{}}");
90  RawTeX!("\\def\\lx@psgobble@one(#1){\\lx@psgobble@parens}");

119  // Text placement — drop both coords AND the text body. Perl's
120  // `DefPSConstructor` would wrap the labelled text inside a
121  // `<ltx:picture>`, so the picture auto-closes cleanly when block
122  // content (e.g. `\begin{minipage}` inside a figure) follows. Rust's
123  // pstricks port doesn't yet generate `<ltx:picture>`; emitting the
124  // text into the surrounding paragraph traps later block content
125  // inside an `<ltx:p>` (witness: hep-ph0102192 minipage-in-figure
126  // schema errors). Dropping the text body is a fidelity regression —
127  // visible labels like "cocktail"/"thermal" placed via `\rput` are
128  // lost — but it eliminates the cascading schema errors. TODO: port
129  // `DefPSConstructor` framework so pstricks output lives in
130  // `<ltx:picture>` and labels survive.
131  RawTeX!("\\def\\lx@rput@parens(#1)#2{}");
132  RawTeX!("\\def\\lx@rput@bracket[#1]{\\lx@rput@parens}");
133  RawTeX!("\\def\\rput{\\@ifstar\\lx@rput@i\\lx@rput@i}");
134  RawTeX!("\\def\\lx@rput@i{\\@ifnextchar[\\lx@rput@bracket{\\lx@rput@parens}}");

136  RawTeX!("\\def\\lx@uput@bracket[#1]{\\lx@uput@parens}");
137  RawTeX!("\\def\\uput{\\@ifstar\\lx@uput@i\\lx@uput@i}");

139  RawTeX!("\\def\\lx@cput@parens(#1)#2{}");
