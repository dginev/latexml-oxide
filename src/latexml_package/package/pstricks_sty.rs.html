31  // Perl pstricks_support.sty.ltxml L849-861: `\newpsobject{name}{oldname}{keyval}`
32  // dynamically defines `\<name>` to forward to `\<oldname>` with the saved
33  // `<keyval>` baked into the optional argument. The paper's drawing object is
34  // then drawn by the resolved `\<oldname>` (typically `\psline`, `\psdots`).
35  // Perl stores `oldname` and `keyval` in two LookupValue keys so the
36  // generated forwarder can read them at call time, and additionally merges
37  // a user-supplied `[opt]` into the saved `keyval`.
38  //
39  // Witness: physics/9710028 uses
40  //   \newpsobject{PST@Border}{psline}{linewidth=.0015,linestyle=solid}
41  // then later calls `\PST@Border(...)`. With the prior no-op stub
42  // `\PST@Border` stayed undefined and Rust errored; Perl recovered.
43  DefPrimitive!("\\newpsobject{}{}{}", sub[(newname, oldname, keyval)] {
44    let newcs    = s!("\\{}", newname.to_string());
45    let oldcs    = s!("\\{}", oldname.to_string());
46    let keystr   = keyval.to_string();
47    let new_tok  = T_CS!(newcs);
48    let params   = parse_parameters("OptionalMatch:* []", &new_tok, true)?;
49    // Generated forwarder closure: read OptionalMatch:* and []; emit
50    //   \<old>(*)([combined-key])
51    // combined-key = saved-key + ',' + user-key (Perl L855).
52    let oldcs_owned = oldcs;
53    let key_owned   = keystr;
54    let body_closure: ExpansionBody = ExpansionBody::Closure(Rc::new(move |args| {
55      let star = args.first().map(|a| !a.is_none()).unwrap_or(false);
56      let usr  = args.get(1)
57        .and_then(|a| match a.as_tokens() { Ok(Some(t)) => Some(t.to_string()), _ => None })
58        .unwrap_or_default();
59      let combined = match (key_owned.is_empty(), usr.is_empty()) {
60        (false, false) => s!("{},{}", key_owned, usr),
61        (false, true)  => key_owned.clone(),
62        (true,  false) => usr,
63        (true,  true)  => String::new(),
64      };
65      let mut out = vec![T_CS!(oldcs_owned.clone())];
66      if star { out.push(T_OTHER!("*")); }
67      if !combined.is_empty() {
68        out.push(T_OTHER!("["));
69        out.extend(Explode!(combined));
70        out.push(T_OTHER!("]"));
71      }
72      // Perl L856 — emit the suffix only; the paren-coords tuple is
73      // consumed by the resolved \psline (or sibling) macro's own
74      // `\lx@psgobble@parens` chain.
75      Ok(Tokens::new(out))
76    }));
77    def_macro(new_tok, params, Some(body_closure), None)?;
78  });

82  // PSCoordList-emulator. Perl's pstricks_support.sty.ltxml uses parameter
83  // type `PSCoordList` (variable-arity `(x,y)(x,y)...`) to absorb the paren
84  // tuples that follow most pstricks drawing commands. Without it those
85  // tuples leak as raw text into the document — opening an `<ltx:p>` that
86  // doesn't auto-close before subsequent block content (witness:
87  // hep-ph0102192 minipage-in-figure failure). Recursive `\@ifnextchar`
88  // idiom: peek for `(`; consume one tuple; recurse.
89  RawTeX!("\\def\\lx@psgobble@parens{\\@ifnextchar({\\lx@psgobble@one}{}}");
90  RawTeX!("\\def\\lx@psgobble@one(#1){\\lx@psgobble@parens}");
91  // `\lx@psgobble@shape`: consume an OPTIONAL leading `{<arrows>}` brace group
92  // (e.g. `\psline{->}…`), THEN the trailing `(x,y)(x,y)…` coordinate tuples.
93  // The pstricks open-curve commands take an OPTIONAL arrow spec before the
94  // coordinates; the previous signatures declared it as a MANDATORY `{}` arg,
95  // so when arrows were absent (`\pscurve[opts](x,y)…`) the `{}` swallowed the
96  // first `(` and `\lx@psgobble@parens` then saw a digit, stopped, and left the
97  // remaining coordinates as stray picture text. Following an open `\put{…}`
98  // `<ltx:text>`, that stray text trapped every later block in an un-closeable
99  // `<ltx:text>` (witness 1112.2096). Peeking for the leading brace makes the
100  // arrow spec optional without over-gobbling any trailing document braces.
101  RawTeX!("\\def\\lx@psgobble@shape{\\@ifnextchar\\bgroup{\\lx@psgobble@arrows}{\\lx@psgobble@parens}}");
102  RawTeX!("\\def\\lx@psgobble@arrows#1{\\lx@psgobble@parens}");

104  // Drawing commands — all no-ops for HTML, but MUST consume their
105  // optional `{arrows}` + trailing `(x,y)…` coordinate tuples so nothing
106  // leaks as text. Open curves use `\lx@psgobble@shape` (optional arrows);
107  // closed shapes have no arrows so `\lx@psgobble@parens` (coords only)
108  // suffices and is equally arrow-tolerant via the peek.
109  DefMacro!("\\psline OptionalMatch:* []", "\\lx@psgobble@shape");
110  DefMacro!("\\psframe OptionalMatch:* []", "\\lx@psgobble@shape");
111  // \pscircle[par](x,y){radius}: one coord pair + braced radius (real pstricks
112  // `\def\pscircle(#1){#2}` style). Pair reads `(x,y)`, `{}` the radius.
113  DefMacro!("\\pscircle OptionalMatch:* [] Pair {}", "\\lx@psgobble@parens");
114  DefMacro!("\\psarc OptionalMatch:* []{}{}{}{}", "\\lx@psgobble@parens");
115  DefMacro!("\\psbezier OptionalMatch:* []", "\\lx@psgobble@shape");
116  DefMacro!("\\pscurve OptionalMatch:* []", "\\lx@psgobble@shape");

123  // \qline(x1,y1)(x2,y2) — two coordinate pairs, real pstricks

130  // \Rput[refpoint](x,y){body} — placement at coords (real pstricks

139  // \qdisk(x0,y0){radius} — coordinate pair + braced radius, real pstricks
140  // `\def\qdisk(#1)#2{…}` (pstricks.tex L3411). The old `{}{}` signature read
141  // two brace/single-token args instead of `(coord){radius}`, so
142  // `\qdisk(3,2.5){2.5pt}` consumed only `\qdisk` + `(` + `3` and left the

146  // un-closeable `<ltx:text>` → "ltx:* isn't allowed in <ltx:text>" cascade.
147  // Witness: arXiv:1112.2096 (`\put(2.5,1.4){$a$}` then `\qdisk(3,2.5){2.5pt}`
148  // inside `\mbox{\scalebox{\begin{pspicture}…}}`).

151  // Text placement — drop both coords AND the text body. Perl's
152  // `DefPSConstructor` would wrap the labelled text inside a
153  // `<ltx:picture>`, so the picture auto-closes cleanly when block
154  // content (e.g. `\begin{minipage}` inside a figure) follows. Rust's
155  // pstricks port doesn't yet generate `<ltx:picture>`; emitting the
156  // text into the surrounding paragraph traps later block content
157  // inside an `<ltx:p>` (witness: hep-ph0102192 minipage-in-figure
158  // schema errors). Dropping the text body is a fidelity regression —
159  // visible labels like "cocktail"/"thermal" placed via `\rput` are
160  // lost — but it eliminates the cascading schema errors. TODO: port
161  // `DefPSConstructor` framework so pstricks output lives in
162  // `<ltx:picture>` and labels survive.
163  // Runaway-safe placement gobbler shared by \rput/\cput. Consumes (and
164  // drops) optional [refpoint], optional {angle}, optional (coords), and the
165  // mandatory {body}. The PREVIOUS def used a *delimited* `(#1)` parameter
166  // (`\def\lx@rput@parens(#1)#2{}`): for the braced-angle / no-coords form
167  // `\rput{angle}{body}` there is no `(`, so TeX scanned FORWARD eating
168  // tokens — including `\end{pspicture}` — until the next `(` anywhere
169  // later. That swallowed the env end, so pspicture's `end_mode` never fired
170  // and its mode-switch frame leaked, tripping `\endgroup Attempt to close a
171  // group that switched to mode restricted_horizontal` (witness 1505.07999 +
172  // the ~17-paper `\endgroup` mode-leak cluster). Perl avoids this with
173  // `OptionalBracketed`+`ZeroPSCoord` (coords optional); we PEEK for `(`
174  // instead of requiring it. (Body still dropped — see the <ltx:p>-cascade
175  // note above; faithful `<ltx:g>`-with-body is the separate TODO.)
176  RawTeX!("\\def\\lx@put@cb(#1)#2{}");      // (coords){body} -> drop
177  RawTeX!("\\def\\lx@put@bb#1{}");          // {body} -> drop (no coords)
178  RawTeX!("\\def\\lx@put@b#1{\\@ifnextchar(\\lx@put@cb\\lx@put@bb}"); // {angle}; then (coords)? body
179  RawTeX!("\\def\\lx@put@s{\\@ifnextchar(\\lx@put@cb\\lx@put@b}");    // ( -> coords; else {angle}|{body}
180  RawTeX!("\\def\\lx@put@opt[#1]{\\lx@put@s}");                       // [refpoint] -> continue
181  RawTeX!("\\def\\lx@put@start{\\@ifnextchar[\\lx@put@opt\\lx@put@s}");
182  RawTeX!("\\def\\rput{\\@ifstar\\lx@put@start\\lx@put@start}");
183  RawTeX!("\\def\\lx@uput@parens#1(#2)#3{}"); // {dist}(coord){text} → drop
184  RawTeX!("\\def\\lx@uput@bracket[#1]{\\lx@uput@parens}");
185  RawTeX!("\\def\\uput{\\@ifstar\\lx@uput@i\\lx@uput@i}");
186  RawTeX!("\\def\\lx@uput@i{\\@ifnextchar[\\lx@uput@bracket{\\lx@uput@parens}}");
187  // \cput shares the runaway-safe gobbler (same delimited-`(` hazard).
188  RawTeX!("\\def\\cput{\\@ifstar\\lx@put@start\\lx@put@start}");
