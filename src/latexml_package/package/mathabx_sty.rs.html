3/// Runtime helper for the trivial `DefMath!` shape used 170+ times in
4/// mathabx (DEP-17b, mirrors DEP-15/DEP-17 approach). The macro arm
5/// expands at compile time into ~1.1 KiB of `.text` per invocation;
6/// routing through this helper drops `load_definitions` size at the
7/// cost of a runtime `parse_prototype` call per entry — paid once at
8/// engine bootstrap.
9fn def_math_sym(cs: &str, present: &str, role: Option<&str>, meaning: Option<&str>) -> Result<()> {
10  let (cs_tok, params) = parse_prototype(cs, true)?;
11  let mut opts = MathPrimitiveOptions::default();
12  if let Some(r) = role {
13    opts.role = Some(r.to_string());
14  }
15  if let Some(m) = meaning {
16    opts.meaning = Some(m.to_string());
17  }
18  def_math(cs_tok, params, present.to_string(), opts)?;
19  Ok(())
20}
