1//! subfloat.sty — Subfigure/subtable container environments
2//! Perl: subfloat.sty.ltxml — 100 lines
3use crate::engine::latex_constructs::{after_float, before_float};
4use crate::prelude::*;
5
6#[rustfmt::skip]
7LoadDefinitions!({
8  // Convenience macros — Perl L25-28
9  DefMacro!("\\subfiguresbegin", "\\begin{subfigures}");
10  DefMacro!("\\subfiguresend",   "\\end{subfigures}");
11  DefMacro!("\\subtablesbegin",  "\\begin{subtables}");
12  DefMacro!("\\subtablesend",    "\\end{subtables}");
13
14  // Counters — Perl L30-32
15  NewCounter!("subfloatfigure", "figure", idprefix => "sf", idwithin => "figure");
16  NewCounter!("subfloattable",  "table",  idprefix => "st", idwithin => "table");
17  // Perl sets \themainfigure/\themaintable in {subfigures}/{subtables} beforeDigest;
18  // we provide global Let aliases so they resolve outside that scope too.
19  Let!("\\themainfigure", "\\thefigure");
20  Let!("\\themaintable",  "\\thetable");
21  DefMacro!("\\thesubfloatfigure", "\\themainfigure\\alph{subfloatfigure}");
22  DefMacro!("\\thesubfloattable",  "\\themaintable\\alph{subfloattable}");
23  DefMacro!("\\subfloatfigurename", "Figure");
24  DefMacro!("\\subfloattablename",  "Table");
25  Let!("\\ext@subfloatfigure", "\\ext@figure");
26  Let!("\\ext@subfloattable",  "\\ext@table");
27  DefMacro!("\\fnum@subfigure", "(\\thesubfigure)");
28  DefMacro!("\\fnum@subtable",  "(\\thesubtable)");
29
30  // Container environments — Perl L40-67 / L70-97. The previous Rust port
31  // omitted the per-env `before_digest`/`after_digest` hooks entirely:
32  // Perl wires `beforeFloat('figure')` (steps the figure counter,
33  // pushes float scope) plus `afterFloat($whatsit)` (pops scope, sets
34  // refnum), so a `\begin{subfigures}` properly registers as a float and
35  // its caption can resolve. Also `inlist='lof'` was hardcoded — Perl
36  // uses `inlist='#inlist'` driven by RefStepCounter.
37  //
38  // BLOCKER: Perl ALSO redefines {figure}/{figure*} (and table/table*)
39  // INSIDE the before_digest closure to switch them to the
40  // subfloat<figure|table> counter and drop the inlist attribute. That
41  // requires a runtime def_environment from inside a before_digest
42  // closure (enumitem/float pattern). Deferred — the simple case
43  // (single-level subfigure with manual \subfigure marks) still works
44  // with the basic before/after_float hooks added here.
45  DefEnvironment!("{subfigures}",
46    "<ltx:figure xml:id='#id' inlist='#inlist'>#tags#body</ltx:figure>",
47    before_digest => {
48      Let!("\\themainfigure", "\\thefigure");
49      before_float("figure", None);
50    },
51    after_digest => sub[whatsit] { after_float(whatsit); },
52    mode => "internal_vertical"
53  );
54  DefEnvironment!("{subtables}",
55    "<ltx:table xml:id='#id' inlist='#inlist'>#tags#body</ltx:table>",
56    before_digest => {
57      Let!("\\themaintable", "\\thetable");
58      before_float("table", None);
59    },
60    after_digest => sub[whatsit] { after_float(whatsit); },
61    mode => "internal_vertical"
62  );
63});
