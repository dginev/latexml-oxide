1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: \xspace reads the next token. If it's NOT in a set of
6  // "no-space" tokens, insert a space. The no-space set includes:
7  // T_BEGIN, T_END, punctuation (.,:;!?/')-~), and certain CS tokens.
8  // This makes \No\xspace produce "No " before words but "No." before periods.
9  // Perl kind is DefMacro with a gullet-reading sub body; Rust DefPrimitive
10  // reads + unreads at stomach time. WISDOM #44: this is NOT a universal
11  // equivalence — `\xspace` inside `\edef` would behave differently (Perl
12  // expands at \edef time; Rust freezes the CS). Safe in practice because
13  // `\xspace` call sites are post-expansion text-mode position.
14  DefPrimitive!("\\xspace", {
15    let next = gullet::read_token()?;
16    let is_no_space = next.as_ref().map(|t| {
17      let cc = t.get_catcode();
18      // T_BEGIN ({) and T_END (}) suppress space (Perl: @XSPACES includes T_BEGIN, T_END)
19      if matches!(cc, Catcode::BEGIN | Catcode::END) {
20        return true;
21      }
22      // CS tokens: \/, \ , \xspace, \space, \@sptoken, \@xobeysp
23      if cc == Catcode::CS {
24        return t.with_str(|s| matches!(s,
25          "\\/"|"\\ "|"\\xspace"|"\\space"|"\\@sptoken"|"\\@xobeysp"
26        ));
27      }
28      // Punctuation and special characters
29      t.with_str(|s| matches!(s,
30        "." | "," | ":" | ";" | "!" | "?" | "/" | "'" | ")" | "-" | "~"
31        | "\u{00A0}" // non-breaking space
32      ))
33    }).unwrap_or(false);
34    if let Some(tok) = next {
35      gullet::unread(Tokens!(tok));
36    }
37    if !is_no_space {
38      // Insert a space token
39      gullet::unread(Tokens!(T_SPACE!()));
40    }
41  });
42
43  // xspace.sty L57-60 `\xspaceaddexceptions{chars}` adds chars to the
44  // dynamic exception list. Our `\xspace` primitive above uses a
45  // hardcoded suppress-list; the dynamic list mechanism isn't modeled.
46  // Stub as no-op so raw-loaded sibling packages that call this don't
47  // emit undefined-CS errors. Author's added exceptions silently
48  // won't take effect (cosmetic — potential extra space, never
49  // missing-space). Companion `\xspaceremoveexception` too.
50  def_macro_noop("\\xspaceaddexceptions{}")?;
51  def_macro_noop("\\xspaceremoveexception{}")?;
52});
