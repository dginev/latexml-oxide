24    };

32    // provide `\color`. Papers that only load `colordvi` (e.g. 0705.1190)
33    // should still work.
34    //
35    // We achieve the same by routing `\text<name>` through the

42    // disappear at group close. Use \global\def to match Perl's global
43    // scope semantics — the bare \def fallback is local by default.
