1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: aa.cls.ltxml — Astronomy & Astrophysics
6  // Ignorable options
7  for option in ["10pt", "11pt", "12pt", "twoside", "onecolumn", "twocolumn",
8    "draft", "final", "referee", "longauth", "rnote", "oldversion",
9    "runningheads", "envcountreset", "envcountsect",
10    "structabstract", "traditabstract", "letter"].iter()
11  {
12    DeclareOption!(*option, None);
13  }
14  DeclareOption!(None, {
15    Digest!("\\PassOptionsToClass{\\CurrentOption}{article}")?;
16  });
17  ProcessOptions!();
18  LoadClass!("article");
19  RequirePackage!("aa_support");
20  // Raw aa.cls unconditionally loads natbib (not just for bibnumber/bibauthoryear options).
21  // Many aa papers use \citep/\citet without explicit \usepackage{natbib}.
22  RequirePackage!("natbib");
23  // Override \pmatrix to use plain TeX version (not amsmath)
24  DefMacro!("\\pmatrix{}",
25    "\\lx@gen@plain@matrix{name=pmatrix,datameaning=matrix,left=\\lx@left(,right=\\lx@right)}{#1}");
26  // Override \cases to use plain TeX version
27  DefMacro!("\\cases{}",
28    "\\lx@gen@plain@cases{meaning=cases,left=\\lx@left\\{,conditionmode=text,style=\\textstyle}{#1}");
29  // Perl aa.cls.ltxml L57 resets the amsmath loaded-flag so documents
30  // that need amsmath's {cases} can `\usepackage{amsmath}` again after
31  // aa_support pulled amsmath in implicitly. Example:
32  // arXiv:astro-ph/0203101. Without this reset the re-RequirePackage
33  // is a no-op and {cases} stays as the plain-TeX version above.
34  AssignValue!("amsmath.sty_loaded" => Stored::None, Some(Scope::Global));
35
36  // aa.cls L1651-1664: \tablebib{...} / \tablefoot{...} emit a labeled
37  // table-notes block below the tabular, inside the table float. Perl
38  // (aa_support.sty.ltxml L229) renders `\tablefoot` as a plain `\footnote`
39  // → `ltx:note` (Meta.class, valid in a table float, and — crucially —
40  // able to hold the *multi-paragraph* note bodies A&A papers write, where a
41  // blank line inside `\tablefoot{...}` becomes a `\par`).
42  //
43  // A bare `\par\textbf{Notes.} #1` (the old form, witnesses 2406.05044 /
44  // 2406.14661) works only for single-paragraph notes: a `\par` *inside*
45  // `#1` builds an `ltx:para`, which `table_model` forbids
46  // (`ltx:para isn't allowed in ltx:table`; witness 1701.02312, whose
47  // `\tablefoot` has a blank-line `\par`). Route through `\footnote` like
48  // Perl — but keep the visible "Notes."/"References." label (real aa.cls
49  // shows it) by prefixing it *inside* the note, where block content is legal.
50  DefMacro!("\\tablebib{}", "\\footnote{\\textbf{References.} #1}");
51  DefMacro!("\\tablefoot{}", "\\footnote{\\textbf{Notes.} #1}");
52  DefMacro!("\\tablebibname", "References");
53  DefMacro!("\\tablefootname", "Notes");
54  // A&A authors use \orcid for ORCID identifier. Preserve as ltx:note.
55  DefMacro!("\\orcid{}",
56    "\\@add@frontmatter{ltx:note}[role=orcid]{#1}");
57
58  // Real aa.cls ends with `\RequirePackage[modulo,mathlines,switch,running,
59  // columnwise,pagewise]{linenoaa}` (the A&A-bundled lineno variant), so
60  // documents call \nolinenumbers / \linenumbers freely (witness
61  // 2605.00223 line 123). Our lineno binding stubs the full macro set —
62  // require it (options irrelevant to the XML stubs). Perl's aa.cls.ltxml
63  // lacks this (\nolinenumbers undefined there); candidate to upstream.
64  RequirePackage!("lineno");
65});
