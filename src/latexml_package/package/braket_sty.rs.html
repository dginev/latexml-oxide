83/// Split a token list on top-level `|` OTHER tokens. Nested braces are
84/// respected: a `|` inside `{…}` stays within the current part. Mirrors
85/// Perl's `splitBraketArg` for the single-bar case (we do not yet
86/// distinguish `\|` double-bar from `|`, matching the simpler branch of
87/// the Perl helper).
88fn split_braket_parts(arg: Tokens) -> Vec<Tokens> {
89  let vbar = T_OTHER!("|");
90  let mut result: Vec<Tokens> = Vec::new();
91  let mut current: Vec<Token> = Vec::new();
92  let mut depth: i32 = 0;
93  for t in arg.unlist() {
94    if t.get_catcode() == Catcode::BEGIN { depth += 1; }
95    else if t.get_catcode() == Catcode::END { depth -= 1; }
96    if depth == 0 && t == vbar {
97      result.push(Tokens::new(std::mem::take(&mut current)));
98    } else {
99      current.push(t);
100    }
101  }
102  result.push(Tokens::new(current));
103  result
104}

106/// Pick the `\lx@braket@…` / `\lx@Braket@…` dispatch target plus the
107/// actual argument count to forward, based on how many `|`-separated
108/// parts the user supplied.
109fn pick_braket_cs(prefix: &str, n_parts: usize) -> (String, usize) {
110  match n_parts {
111    2 => (format!("{prefix}V"), 2),
112    n if n >= 3 => (format!("{prefix}VV"), 3),
113    _ => (prefix.to_string(), 1),
114  }
115}

117/// Build `\cs{arg1}{arg2}…` as a raw Token stream — mirrors Perl's
118/// `Invocation(T_CS($cs), @args)`. Each arg is wrapped in an explicit
119/// `{…}` group so the downstream macro reads it as a single argument
120/// without re-tokenizing a Display-formatted form (which would fuse
121/// `\mbf r` into `\mbfr`).
122fn build_invocation(cs: impl AsRef<str>, args: &[Tokens]) -> Tokens {
123  let total_tokens: usize = args.iter().map(|a| a.len() + 2).sum();
124  let mut out: Vec<Token> = Vec::with_capacity(1 + total_tokens);
125  out.push(T_CS!(cs.as_ref()));
126  for arg in args {
127    out.push(T_BEGIN!());
128    out.extend(arg.clone().unlist());
129    out.push(T_END!());
130  }
131  Tokens::new(out)
132}
