1//! PiCTeX definition stubs.
2//!
3//! Raw PiCTeX expands plotted points and line rules into very large TeX
4//! boxes. In HTML this becomes thousands of positioned inline spans, which is
5//! slow and usually less useful than keeping the figure/caption structure.
6//! These stubs consume common PiCTeX drawing commands without rendering them.
7use crate::prelude::*;
8
9#[rustfmt::skip]
10LoadDefinitions!({
11  RawTeX!("\\expandafter\\ifx\\csname plotsymbolspacing\\endcsname\\relax\\newdimen\\plotsymbolspacing\\fi");
12
13  DefMacro!("\\beginpicture", "");
14  DefMacro!("\\endpicture", "");
15  DefMacro!("\\setsolid", "");
16  DefMacro!("\\setlinear", "");
17  DefMacro!("\\setquadratic", "");
18  DefMacro!("\\stoprotation", "");
19
20  RawTeX!("\\def\\setcoordinatesystem units <#1,#2>{}");
21  RawTeX!("\\def\\setplotsymbol (#1){}");
22  RawTeX!("\\def\\setdashes <#1>{}");
23  RawTeX!("\\def\\startrotation by #1 #2{}");
24  RawTeX!("\\def\\plot#1/{}");
25  RawTeX!("\\def\\putrule from #1 #2 to #3 #4{}");
26  RawTeX!("\\def\\circulararc #1 degrees from #2 #3 center at #4 #5{}");
