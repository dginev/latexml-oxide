18/// Format a RAW pgfmath result, as set by pgf's internal `@`-suffixed
19/// functions (e.g. `\pgfmathmod@`, the trig used by
20/// `\pgfmathanglebetweenpoints`). Perl LaTeXML / real pgf:
21///   `sub pgfmathresult { sprintf("%.5f", $value); $value =~ s/0+$//; ... }`
22/// — i.e. it STRIPS trailing zeros AND the now-bare decimal point, so an
23/// integer prints as `10`, not `10.0`. (The PUBLIC `\pgfmathparse` path keeps
24/// the `.0` — that is `format_parse_result`, deliberately separate.)
25///
26/// Faithfulness matters here beyond cosmetics: pgf's
27/// `\pgfmathpointintersectionoflineandarc` runs an UNBOUNDED bisection
28/// (`pgfmathcalc.code.tex`) whose only exit is the exact comparison
29/// `\ifdim\x pt=\q pt`. When the `@`-functions append a spurious `.0`, the
30/// probe angle's string never matches the target the way TeX's raw fixed-point
31/// output does, the bisection over-narrows to `\p == \s` (no break there) and
32/// spins forever — each iteration opening a `{` group that accumulates as an
33/// empty box → OOM (witness 2201.09268, a `rounded rectangle`/callout node
34/// boundary).
35fn pgfmath_result_str(value: f64) -> String {
36  if value.is_nan() || value.is_infinite() {
37    return "0".to_string();
38  }
39  let clamped = value.clamp(-MAX_PGF_NUMBER, MAX_PGF_NUMBER);
40  if clamped == (clamped as i64) as f64 {
41    // Integer result — raw, NO `.0` (matches Perl `@`-function output).
42    return format!("{}", clamped as i64);
43  }
44  let mut s = format!("{:.5}", clamped);
45  // Strip trailing zeros after the decimal point (the fractional case always
46  // keeps at least one non-zero fractional digit, so this never bares the dot).
47  while s.ends_with('0') {
48    s.pop();
49  }
50  s
51}

100    51090942171709440000.0,

102  let n_orig = n;
103  let n = n.unsigned_abs() as usize;
104  if n >= FACTS.len() {
105    // Perl pgfmath.code.tex.ltxml L68:
106    //   Error("pgfmath", "overflow", undef,
107    //     "Arithmetic overflow: $arg! is too large.");
108    //   return Pgfmath::FALSE;   # i.e. 0
109    // Emit the diagnostic and return 0 exactly like Perl. We must NOT run the

113    // B2). 0 is also benign downstream and any n ≥ 22! already overflows pgf's
114    // MAX_PGF_NUMBER range, so the prior `clamp`-to-max output was meaningless.
115    let _ = (|| -> Result<()> {
116      Error!(
117        "pgfmath",
118        "overflow",
119        format!("Arithmetic overflow: {n_orig}! is too large.")
120      );
121      Ok(())
122    })();
123    0.0
124  } else {
125    FACTS[n]
126  }

129/// Parse a pgfNumber argument: expand macros, then convert to f64
130/// Perl: pgfNumber parameter type reads and evaluates the number
131fn parse_pgf_number(arg: &Tokens) -> f64 {
132  // First try direct string (avoids expansion overhead for literal numbers)
133  let s = arg.to_string();
134  let s = strip_leading_double_negation(s.trim());
135  if s == "." {
136    return 0.0;
137  }
138  if let Ok(v) = s.parse::<f64>() {
139    return v;
140  }
141  // If direct parse fails (e.g. unexpanded macro), expand and retry
142  match do_expand(arg.clone()) {
143    Ok(expanded) => {
144      let s = expanded.to_string();
145      let s = strip_leading_double_negation(s.trim());
146      if s == "." {
147        return 0.0;
148      }
149      s.parse::<f64>().unwrap_or(0.0)
150    },
151    _ => 0.0,
152  }
153}

155// Perl #2711 uses `s/^\-\-//g`, which strips every leading `--` pair
156// (e.g. `----5` → `5`). `strip_prefix` only removes one pair.
157fn strip_leading_double_negation(mut s: &str) -> &str {
158  while let Some(rest) = s.strip_prefix("--") {
159    s = rest;
160  }
161  s
162}

167/// Try to parse input as a simple number and format per Perl rules.
168/// Returns None if input is not a simple number.
169fn try_simple_number(input: &str) -> Option<String> {
170  let bytes = input.as_bytes();
171  let mut pos = 0;
172
173  // Optional sign
174  let sign: Option<char> = if pos < bytes.len() && (bytes[pos] == b'+' || bytes[pos] == b'-') {
175    let c = bytes[pos] as char;
176    pos += 1;
177    Some(c)
178  } else {
179    None
180  };
181
182  // Required: at least one digit
183  let int_start = pos;
184  while pos < bytes.len() && bytes[pos].is_ascii_digit() {
185    pos += 1;
186  }
187  if pos == int_start {
188    return None;
189  }
190  let integer = &input[int_start..pos];
191
192  // Optional decimal part: \.[\d.]*
193  let decimals: Option<&str> = if pos < bytes.len() && bytes[pos] == b'.' {
194    let dec_start = pos;
195    pos += 1;
196    while pos < bytes.len() && (bytes[pos].is_ascii_digit() || bytes[pos] == b'.') {
197      pos += 1;
198    }
199    Some(&input[dec_start..pos])
200  } else {
201    None
202  };
203
204  // Must be at end of string
205  if pos != bytes.len() {
206    return None;
207  }
208
209  // Format result (Perl L332-344)
210  let mut result = input.to_string();
211  if let Some(s) = sign {
212    if s == '+' {
213      result = match decimals {
214        Some(d) => format!("{}{}", integer, d),
215        None => integer.to_string(),
216      };
217    } else {
218      // sign == '-'
219      if !result.contains(|c: char| ('1'..='9').contains(&c)) {
220        result = "0.0".to_string();
221      } else if decimals.is_none() {
222        result = format!("{}.0", result);
223      }
224      // Overflow check for negative
225      if let Ok(numval) = result.parse::<f64>()
226        && numval < -MAX_PGF_NUMBER
227      {
228        result = format!("{}", -MAX_PGF_NUMBER);
229      }
230    }
231  }
232  Some(result)
233}

531/// Apply a user-defined pgf math function by calling TeX
532/// Perl: sub pgfmath_apply (L447-459)
533fn pgfmath_apply_user(name: &str, args: &[f64]) -> Option<f64> {
534  let cs_name = format!("\\pgfmath{}@", name);
535  let cs_tok = Token {
536    text: pin(&cs_name),
537    code: Catcode::CS,
538    #[cfg(feature = "token-locators")]
539    loc: 0,
540  };
541  // Check if the function is defined
542  lookup_definition(&cs_tok).ok()?.as_ref()?;
543  // Build invocation tokens: \pgfmath{name}@{arg1}{arg2}...
544  let mut tokens: Vec<Token> = vec![cs_tok];
545  for arg in args {
546    let s = pgfmath_result_str(*arg);
547    tokens.push(T_BEGIN!());
548    tokens.extend(Explode!(s));
549    tokens.push(T_END!());
550  }
551  // Reset the global `\ifpgfmathunitsdeclared` flag BEFORE digesting the body
552  // so that, afterwards, the flag reflects ONLY whether THIS function body
553  // parsed a unit'd value. A user function body may run a nested
554  // `\pgfmathparse` (e.g. pgfplots' `pgfplotsbarwidthgeneric` body parses the
555  // `<bar width>pt` dimension); the outer parser must inherit that via
556  // `absorb_units_flag` so it doesn't clobber the flag back to false on exit.
557  // Without the reset a prior parse's stale flag would leak in for bodies that
558  // set `\pgfmathresult` directly (e.g. `\def\pgfmathresult{42}`). Mirrors
559  // Perl, where `\ifpgfmathunitsdeclared` is a persistent global threaded
560  // through nested evaluations. Fixes `bar shift={\pgfplotbarwidth}` under
561  // `symbolic x coords` (2110.14597).
562  let _ = digest(Tokens::from(vec![T_CS!("\\pgfmathunitsdeclaredfalse")]));
563  // Digest the invocation (sets \pgfmathresult)
564  let _ = digest(Tokens::from(tokens));
565  // Read back \pgfmathresult via expansion
566  let result_tokens = do_expand(Tokens::from(vec![T_CS!("\\pgfmathresult")])).ok()?;
567  let s = result_tokens.to_string();
568  s.trim().parse::<f64>().ok()
569}

633// ==================== Recursive Descent Parser ====================
634// Faithful translation of the Parse::RecDescent grammar from
635// pgfmath.code.tex.ltxml L653-732
636//
637// Grammar rules:
638//   formula     : expr ? expr : expr | expr CMP expr | expr
639//   expr        : term (ADDOP term)*
640//   term        : factor (MULOP factor)*
641//   factor      : simplefactor ^ simplefactor | simplefactor postfix*
642//   simplefactor: ( formula ) | PREFIX simplefactor | FUNCTION(...) |
643//                 FUNCTION0 | NUMBER UNIT | NUMBER | REGISTER
644
645/// Read the global `\ifpgfmathunitsdeclared` boolean — true iff the most
646/// recent (possibly nested) evaluation parsed a unit'd value. The winnow
647/// grammar inherits this after a user function/constant via
648/// `pgfmath_grammar::absorb_units_flag`; see `pgfmath_apply_user`, which resets
649/// the flag before digesting each user body. Mirrors Perl, where
650/// `\ifpgfmathunitsdeclared` is a persistent global threaded through nested
651/// evaluations. Lives at file scope (not in the grammar module) because
652/// `if_condition` / `T_CS!` come from the crate prelude here.

690  /// The grammar-wide skip: `[\s\{\}]*`.
691  fn sb(i: &mut In) {
692    let n = i
693      .input
694      .bytes()
695      .take_while(|b| matches!(b, b' ' | b'\t' | b'\n' | b'\r' | b'{' | b'}'))
696      .count();
697    i.input = &i.input[n..];
698  }
699  fn first(i: &In) -> Option<u8> { i.input.as_bytes().first().copied() }
700  fn second(i: &In) -> Option<u8> { i.input.as_bytes().get(1).copied() }
701  fn bump(i: &mut In, n: usize) { i.input = &i.input[n..]; }
702  fn eat(i: &mut In, b: u8) -> bool {
703    sb(i);
704    if first(i) == Some(b) {
705      bump(i, 1);
706      true
707    } else {
708      false
709    }
710  }

712  /// formula := expr ('?' expr ':' expr)? | expr CMPOP expr
713  pub(super) fn formula(i: &mut In) -> ModalResult<f64> {
714    let left = expr(i)?;
715    sb(i);
716    if eat(i, b'?') {
717      let then_val = expr(i)?;
718      sb(i);
719      eat(i, b':');
720      let else_val = expr(i)?;
721      return Ok(if left != 0.0 { then_val } else { else_val });
722    }
723    let cp = i.checkpoint();
724    if let Some(op) = cmp_op(i) {
725      if let Ok(right) = expr(i) {
726        i.state.int_result = true;
727        return Ok(pgfmath_cmp_op(&op, left, right));
728      }
729      i.reset(&cp);
730    }
731    Ok(left)
732  }

734  fn cmp_op(i: &mut In) -> Option<String> {
735    sb(i);
736    let b = i.input.as_bytes();
737    if b.len() >= 2
738      && let two @ (b"==" | b"!=" | b">=" | b"<=" | b"&&" | b"||") = &[b[0], b[1]]
739    {
740      let op = std::str::from_utf8(two.as_slice()).unwrap().to_string();
741      bump(i, 2);
742      return Some(op);
743    }
744    match b.first() {
745      Some(c @ (b'>' | b'<')) => {
746        let op = (*c as char).to_string();
747        bump(i, 1);
748        Some(op)
749      },
750      _ => None,
751    }
752  }

754  /// expr := term (('+'|'-') term)*
755  fn expr(i: &mut In) -> ModalResult<f64> {
756    let mut result = term(i)?;
757    loop {
758      sb(i);
759      match first(i) {
760        Some(b'+') => {
761          bump(i, 1);
762          result += term(i)?;
763          // Arithmetic yields a real: `2+(1<2)` → "3.0", NOT the integer a bare
764          // comparison produces (pgfmath ground truth). A comparison anywhere in
765          // an operand set int_result; combining it arithmetically clears that.
766          i.state.int_result = false;
767        },
768        Some(b'-') => {
769          bump(i, 1);
770          result -= term(i)?;
771          i.state.int_result = false;
772        },
773        _ => break,
774      }
775    }
776    Ok(result)
777  }

779  /// term := factor (('*'|'/') factor)*
780  fn term(i: &mut In) -> ModalResult<f64> {
781    let mut result = factor(i)?;
782    loop {
783      sb(i);
784      match first(i) {
785        Some(b'*') => {
786          bump(i, 1);
787          result *= factor(i)?;
788          // See expr(): arithmetic demotes a comparison's integer to a real
789          // (`2*(1<2)` → "2.0", not "2").
790          i.state.int_result = false;
791        },
792        Some(b'/') => {
793          bump(i, 1);
794          result /= pgfmath_divisor(factor(i)?);
795          i.state.int_result = false;
796        },
797        _ => break,
798      }
799    }
800    Ok(result)
801  }

803  /// factor := simplefactor '^' simplefactor | simplefactor ('!'|'r')*
804  fn factor(i: &mut In) -> ModalResult<f64> {
805    let base = simplefactor(i)?;
806    sb(i);
807    if eat(i, b'^') {
808      let exp = simplefactor(i)?;
809      i.state.int_result = false; // exponentiation yields a real
810      return Ok(base.powf(exp));
811    }
812    let mut result = base;
813    loop {
814      sb(i);
815      match first(i) {
816        Some(b'!') if second(i) != Some(b'=') => {
817          bump(i, 1);
818          result = pgfmath_factorial(result as i64);
819        },
820        Some(b'r') if !second(i).map(|c| c.is_ascii_alphabetic()).unwrap_or(false) => {
821          bump(i, 1);
822          result = result.to_degrees();
823        },
824        _ => break,
825      }
826    }
827    Ok(result)
828  }

830  /// simplefactor — the atom: parens, prefix ops, numbers (with unit/register
831  /// suffix), registers, identifiers (functions/constants), lone dot.
832  fn simplefactor(i: &mut In) -> ModalResult<f64> {
833    sb(i);
834    match first(i) {
835      None => return fail(),
836      Some(b'(') => {
837        bump(i, 1);
838        let val = formula(i)?;
839        sb(i);
840        eat(i, b')'); // forgive unclosed at end (Perl)
841        return Ok(val);
842      },
843      Some(b'-') => {
844        bump(i, 1);
845        return Ok(-simplefactor(i)?);
846      },
847      Some(b'+') => {
848        bump(i, 1);
849        return simplefactor(i);
850      },
851      Some(b'!') => {
852        if second(i) == Some(b'=') {
853          return fail();
854        }
855        bump(i, 1);
856        let val = simplefactor(i)?;
857        return Ok(if val == 0.0 { 1.0 } else { 0.0 });
858      },
859      _ => {},
860    }
861
862    if let Some(num) = try_number(i) {
863      sb(i);
864      if let Some(unit) = try_unit(i) {
865        i.state.units_declared = true;
866        return Ok(pgfmath_convert(num, &unit));
867      }
868      if first(i) == Some(b'\\') {
869        let cp = i.checkpoint();
870        if let Some(reg) = try_cs_register(i) {
871          return Ok(num * reg);
872        }
873        i.reset(&cp);
874      }
875      return Ok(num);
876    }
877
878    if first(i) == Some(b'\\')
879      && let Some(reg) = try_cs_register(i)
880    {
881      sb(i);
882      if first(i) == Some(b'\\') {
883        let cp = i.checkpoint();
884        if let Some(reg2) = try_cs_register(i) {
885          return Ok(reg * reg2);
886        }
887        i.reset(&cp);
888      }
889      return Ok(reg);
890    }
891
892    if first(i).map(|c| c.is_ascii_alphabetic()).unwrap_or(false) {
893      let name = read_identifier(i);
894      if is_builtin_function(&name) {
895        return function_call(i, &name);
896      }
897      if is_builtin_constant(&name) {
898        return Ok(pgfmath_apply_fn(&name, &[]));
899      }
900      if is_user_function(&name) {
901        // A user function whose TeX body may declare units — inherit them
902        // afterwards (see `absorb_units_flag`). Builtin functions above must
903        // NOT absorb: they compute purely in Rust and never touch the global
904        // `\ifpgfmathunitsdeclared`, so reading it could only pick up a STALE
905        // true from an earlier evaluation and wrongly mark e.g. `sin(30)` as
906        // unit'd. (Real `\pgfmathparse` resets the global at parse-start, which
907        // is why #249's recdescent — which absorbed for builtins too — never
908        // tripped; the golden corpus runs evals back-to-back without that
909        // reset and exposes the staleness.)
910        let v = function_call(i, &name)?;
911        absorb_units_flag(i);
912        return Ok(v);
913      }
914      if is_user_constant(&name) {
915        // A 0-arg user pseudo-constant whose body may parse a unit'd value
916        // (e.g. `pgfplotsbarwidthgeneric`). Inherit its units flag — see
917        // `absorb_units_flag`. Builtin constants above are pure and need no
918        // absorb.
919        let v = pgfmath_apply_fn(&name, &[]);
920        absorb_units_flag(i);
921        return Ok(v);
922      }
923      return fail();
924    }
925
926    if first(i) == Some(b'.') {
927      bump(i, 1);
928      return Ok(0.0);
929    }
930    fail()
931  }

933  /// After evaluating a USER-declared pgfmath function/constant (NOT a
934  /// builtin), inherit the global `\ifpgfmathunitsdeclared` flag that its body
935  /// may have set (its body runs with the flag reset by `pgfmath_apply_user`,
936  /// then a nested `\pgfmathparse` of a unit'd value sets it). Mirrors Perl,
937  /// where the flag is a persistent global threaded through nested evaluations.
938  /// Monotonic — only ever sets the flag, matching `try_cs_register`'s
939  /// unconditional set. Call ONLY from the user-function / user-constant
940  /// dispatch arms: builtins never touch the global, so absorbing after one
941  /// would leak a stale true into a dimensionless result. Fixes
942  /// `bar shift={\pgfplotbarwidth}` under `symbolic x coords` (2110.14597):
943  /// `\pgfplotbarwidth` → `pgfplotsbarwidthgeneric` (a 0-arg pseudo-constant)
944  /// whose body parses the bar-width dimension.
945  fn absorb_units_flag(i: &mut In) {
946    if super::pgfmath_units_declared_global() {
947      i.state.units_declared = true;
948    }
949  }

951  /// FUNCTION '(' formula (',' formula)* ')' | FUNCTION simplefactor
952  fn function_call(i: &mut In, name: &str) -> ModalResult<f64> {
953    sb(i);
954    if first(i) == Some(b'(') {
955      bump(i, 1);
956      let mut args = vec![formula(i)?];
957      loop {
958        sb(i);
959        if eat(i, b',') {
960          match formula(i) {
961            Ok(arg) => {
962              args.push(arg);
963            },
964            _ => {
965              break;
966            },
967          }
968        } else {
969          break;
970        }
971      }
972      sb(i);
973      eat(i, b')');
974      return Ok(pgfmath_apply_fn(name, &args));
975    }
976    let arg = simplefactor(i)?;
977    Ok(pgfmath_apply_fn(name, &[arg]))
978  }

980  /// `(\d+\.?[\d.]*|\d*\.?\d+)([eE][+-]?\d+)?` with hex/binary forms and
981  /// the embedded-dot tolerance ("1.2.3"), exactly as the recdescent had it.
982  fn try_number(i: &mut In) -> Option<f64> {
983    sb(i);
984    let b = i.input.as_bytes();
985    if b.is_empty() {
986      return None;
987    }
988    if b.len() >= 2 && b[0] == b'0' && b[1] == b'x' {
989      let n = b[2..].iter().take_while(|c| c.is_ascii_hexdigit()).count();
990      if n == 0 {
991        return None;
992      }
993      let v = i64::from_str_radix(&i.input[2..2 + n], 16).unwrap_or(0) as f64;
994      bump(i, 2 + n);
995      return Some(v);
996    }
997    if b.len() >= 2 && b[0] == b'0' && b[1] == b'b' {
998      let n = b[2..]
999        .iter()
1000        .take_while(|c| matches!(c, b'0' | b'1'))
1001        .count();
1002      if n == 0 {
1003        return None;
1004      }
1005      let v = i64::from_str_radix(&i.input[2..2 + n], 2).unwrap_or(0) as f64;
1006      bump(i, 2 + n);
1007      return Some(v);
1008    }
1009    let mut pos = 0;
1010    let mut has_digit = false;
1011    while pos < b.len() && b[pos].is_ascii_digit() {
1012      pos += 1;
1013      has_digit = true;
1014    }
1015    if pos < b.len() && b[pos] == b'.' {
1016      if has_digit || (pos + 1 < b.len() && b[pos + 1].is_ascii_digit()) {
1017        pos += 1;
1018        while pos < b.len() && (b[pos].is_ascii_digit() || b[pos] == b'.') {
1019          pos += 1;
1020        }
1021        has_digit = true;
1022      } else if has_digit {
1023        pos += 1;
1024      }
1025    }
1026    if !has_digit || pos == 0 {
1027      return None;
1028    }
1029    if pos < b.len() && (b[pos] == b'e' || b[pos] == b'E') {
1030      let saved = pos;
1031      pos += 1;
1032      if pos < b.len() && (b[pos] == b'+' || b[pos] == b'-') {
1033        pos += 1;
1034      }
1035      if pos < b.len() && b[pos].is_ascii_digit() {
1036        while pos < b.len() && b[pos].is_ascii_digit() {
1037          pos += 1;
1038        }
1039      } else {
1040        pos = saved;
1041      }
1042    }
1043    let v = i.input[..pos].parse::<f64>().unwrap_or(0.0);
1044    bump(i, pos);
1045    Some(v)
1046  }

1048  fn try_unit(i: &mut In) -> Option<String> {
1049    sb(i);
1050    for &unit in PGF_UNITS {
1051      if let Some(rest) = i.input.strip_prefix(unit) {
1052        // not a unit if a letter follows ("ptx" is not "pt")
1053        if rest
1054          .as_bytes()
1055          .first()
1056          .map(|c| c.is_ascii_alphabetic())
1057          .unwrap_or(false)
1058        {
1059          continue;
1060        }
1061        bump(i, unit.len());
1062        return Some(unit.to_string());
1063      }
1064    }
1065    None
1066  }
1067
1068  /// `\\[a-zA-Z@]+` — a TeX register; reading one declares units (see the
1069  /// pgfplots ybar witnesses on the original implementation).
1070  fn try_cs_register(i: &mut In) -> Option<f64> {
1071    sb(i);
1072    let b = i.input.as_bytes();
1073    if b.first() != Some(&b'\\') {
1074      return None;
1075    }
1076    let n = b[1..]
1077      .iter()
1078      .take_while(|c| c.is_ascii_alphabetic() || **c == b'@')
1079      .count();
1080    if n == 0 {
1081      return None;
1082    }
1083    let cs = &i.input[..1 + n];
1084    let value = pgfmath_register_lookup(cs);
1085    bump(i, 1 + n);
1086    i.state.units_declared = true;
1087    Some(value)
1088  }
1089
1090  fn read_identifier(i: &mut In) -> String {
1091    let n = i
1092      .input
1093      .bytes()
1094      .take_while(|c| c.is_ascii_alphanumeric() || *c == b'_')
1095      .count();
1096    let name = i.input[..n].to_string();
1097    bump(i, n);
1098    name
1099  }

1102/// Apply a comparison operator
1103fn pgfmath_cmp_op(op: &str, left: f64, right: f64) -> f64 {
1104  let result = match op {
1105    "==" => left == right,
1106    "!=" => left != right,
1107    ">" => left > right,
1108    "<" => left < right,
1109    ">=" => left >= right,
1110    "<=" => left <= right,
1111    "&&" => left != 0.0 && right != 0.0,
1112    "||" => left != 0.0 || right != 0.0,
1113    _ => false,
1114  };
1115  if result { 1.0 } else { 0.0 }
1116}

1120/// Format the result of pgfmathparse for output
1121/// Perl: L383-393 of pgfmath.code.tex.ltxml
1122fn format_parse_result(result: f64, input: &str) -> String {
1123  // A non-finite result — NaN from `sqrt(-1)`/`ln(-1)`/`asin(5)`, or ±inf from
1124  // an overflow — must NOT serialize as the literal "NaN"/"inf": that string
1125  // poisons every downstream `\pgfmathresult` dimension read. Mirror the
1126  // sibling `pgfmath_result_str` guard (L21) and degrade to 0; the public path
1127  // keeps the `.0`. NB `clamp` below does NOT rescue NaN (`NaN.clamp(..) ==
1128  // NaN`), so this must come first. Review M4.
1129  if result.is_nan() || result.is_infinite() {
1130    return "0.0".to_string();
1131  }
1132  // Overflow check
1133  let result = result.clamp(-MAX_PGF_NUMBER, MAX_PGF_NUMBER);
1134
1135  // Integer result
1136  if result == (result as i64) as f64 {
1137    let i = result as i64;
1138    // Perl: $result .= '.0' unless $input =~ /^int\(/;
1139    if input.starts_with("int(") {
1140      return format!("{}", i);
1141    }
1142    return format!("{}.0", i);
1143  }
1144
1145  // Decimal result: sprintf("%.5f"), strip trailing zeros
1146  let mut s = format!("{:.5}", result);
1147  if s.contains('.') {
1148    while s.ends_with('0') && !s.ends_with(".0") {
1149      s.pop();
1150    }
1151  }
1152  s
1153}

1155/// Expand the argument to a pgfmath parse with the seven calc-package
1156/// compatibility CSes (`\real`, `\minof`, `\maxof`, `\ratio`, `\widthof`,
1157/// `\heightof`, `\depthof`) transiently `\let` to the `\pgfmath@calc@*`
1158/// internals, then restored to their prior meanings.
1159///
1160/// Perl `pgfmath.code.tex.ltxml` L320-327 performs these `Let`s at the *start
1161/// of `sub pgfmathparse`* — i.e. only while the pgfmath argument is being
1162/// expanded; being local (non-`global`) `Let`s they revert with the enclosing
1163/// tikz/pgf group. The earlier Rust port hoisted them to package-load time
1164/// (because the native parser "can't re-bind per call"), which globally
1165/// clobbered `\real` (and the six siblings) for the *whole document*. That
1166/// broke the very common use of `\real` as the blackboard-ℝ symbol: e.g.
1167/// `\int_\real p_m`, where the 1-argument `\pgfmath@calc@real` ate the
1168/// following `p` as its argument and yielded a spurious "Double subscript".
1169/// Witness: 1608.06741 (`\newcommand\real{\mathbb{R}}`, ignored because
1170/// mathtools→calc already defined `\real`, then pgfmath via todonotes→tikz
1171/// clobbered it). We restore Perl's exact scope with a tight
1172/// save → let → expand → restore around just this one expansion, so `\real`
1173/// used as an ordinary math macro elsewhere is never touched.
1174fn expand_pgfmath_arg<T: Into<Tokens>>(tokens: T) -> Tokens {
1175  const PAIRS: [(&str, &str); 7] = [
1176    ("\\real", "\\pgfmath@calc@real"),
1177    ("\\minof", "\\pgfmath@calc@minof"),
1178    ("\\maxof", "\\pgfmath@calc@maxof"),
1179    ("\\ratio", "\\pgfmath@calc@ratio"),
1180    ("\\widthof", "\\pgfmath@calc@widthof"),
1181    ("\\heightof", "\\pgfmath@calc@heightof"),
1182    ("\\depthof", "\\pgfmath@calc@depthof"),
1183  ];
1184  let saved: Vec<(Token, Stored)> = PAIRS
1185    .iter()
1186    .map(|(cs, _)| {
1187      let t = T_CS!(*cs);
1188      let m = lookup_meaning(&t).unwrap_or(Stored::None);
1189      (t, m)
1190    })
1191    .collect();
1192  for (cs, helper) in PAIRS.iter() {
1193    let_i(&T_CS!(*cs), &T_CS!(*helper), None);
1194  }
1195  let expanded = do_expand(tokens).unwrap_or_default();
1196  for (t, m) in saved {
1197    assign_meaning(&t, m, None);
1198  }
1199  expanded
1200}

1202/// Main pgfmathparse evaluation function
1203/// Perl: sub pgfmathparse (L316-394)
1204/// Returns (result_string, units_declared)
1205pub(crate) fn pgfmathparse_eval_with_units(raw_input: &str) -> (String, bool) {
1206  // Normalize whitespace
1207  let input: String = raw_input.split_whitespace().collect::<Vec<_>>().join(" ");
1208  let input = input.trim();
1209
1210  // 0. String-valued ternary / ifthenelse — Perl pgfmath uses untyped scalars so the `?:` operator
1211  //    and `ifthenelse(...)` can return string values like `"black"` or `"pgreen!50"` for color
1212  //    names. Our parser is f64-only, so detect the pattern <test> ? "a" : "b"     OR
1213  //    ifthenelse(<test>, "a", "b") early, evaluate the test as a number, and return the chosen
1214  //    string literal. Drivers: 2601.14798 (color jitter heatmap), Stage-20 v6 ~1579 errors from
1215  //    \pgfmathsetmacro{\clr}{...}.
1216  if let Some(result) = try_string_ternary(input) {
1217    return (result, false);
1218  }
1219
1220  // 0b. Bare string literal (arXiv-fork e0dd3e33/aeb24f2d: string results
1221  //     pass through untouched, including the empty string) — PGF strings
1222  //     like `\pgfmathparse{"south east"}` feed anchors/colors verbatim.
1223  {
1224    let trimmed = input.trim();
1225    if trimmed.len() >= 2 && trimmed.starts_with('"') && trimmed.ends_with('"') {
1226      let inner = &trimmed[1..trimmed.len() - 1];
1227      if !inner.contains('"') {
1228        return (inner.to_string(), false);
1229      }
1230    }
1231  }
1232
1233  // 1. Simple number check (Perl L332-345)

1245    state: pgfmath_grammar::Flags::default(),
1246  };

1249    if stream.state.int_result && result == (result as i64) as f64 {

1252    return (

1265/// Detect string-valued pgfmath ternaries `<test> ? "a" : "b"` or
1266/// `ifthenelse(<test>, "a", "b")` and return the chosen literal.
1267/// Returns None if the pattern doesn't match or contains no string args.
1268fn try_string_ternary(input: &str) -> Option<String> {
1269  // Strip ifthenelse(...) wrapper if present
1270  let inner = if let Some(rest) = input.strip_prefix("ifthenelse(") {
1271    let body = rest.strip_suffix(')')?;
1272    // Re-shape as ternary: split on top-level commas (depth 0 of nested parens/quotes)
1273    let parts = split_top_commas(body)?;
1274    if parts.len() != 3 {
1275      return None;
1276    }
1277    format!(
1278      "{} ? {} : {}",
1279      parts[0].trim(),
1280      parts[1].trim(),
1281      parts[2].trim()
1282    )
1283  } else {
1284    input.to_string()
1285  };
1286  // Find top-level `?` and `:` (skipping inside quotes and parens)
1287  let (cond, then_part, else_part) = split_ternary(&inner)?;
1288  // Branches must include at least one quoted string for this path to make sense
1289  let then_is_str = then_part.trim().starts_with('"');
1290  let else_is_str = else_part.trim().starts_with('"');
1291  if !then_is_str && !else_is_str {
1292    return None;
1293  }
1294  // Evaluate the test as a number via the regular f64 grammar
1295  let mut stream = winnow::Stateful {
1296    input: cond.trim(),
1297    state: pgfmath_grammar::Flags::default(),
1298  };
1299  let test_val = pgfmath_grammar::formula(&mut stream).ok()?;
1300  let chosen = if test_val != 0.0 {
1301    then_part
1302  } else {
1303    else_part
1304  };
1305  // Strip outer quotes if present; otherwise return as-is
1306  let trimmed = chosen.trim();
1307  let unq = if trimmed.starts_with('"') && trimmed.ends_with('"') && trimmed.len() >= 2 {
1308    &trimmed[1..trimmed.len() - 1]
1309  } else {
1310    trimmed
1311  };
1312  Some(unq.to_string())
1313}

1315/// Split a string on commas at the top level (depth 0), skipping
1316/// nested parens/brackets and quoted substrings.
1317fn split_top_commas(s: &str) -> Option<Vec<&str>> {
1318  let bytes = s.as_bytes();
1319  let mut parts = Vec::new();
1320  let mut start = 0usize;
1321  let mut depth: i32 = 0;
1322  let mut in_quote = false;
1323  for (i, &b) in bytes.iter().enumerate() {
1324    match b {
1325      b'"' => in_quote = !in_quote,
1326      b'(' | b'[' | b'{' if !in_quote => depth += 1,
1327      b')' | b']' | b'}' if !in_quote => depth -= 1,
1328      b',' if !in_quote && depth == 0 => {
1329        parts.push(&s[start..i]);
1330        start = i + 1;
1331      },
1332      _ => {},
1333    }
1334  }
1335  parts.push(&s[start..]);
1336  Some(parts)
1337}

1339/// Split `cond ? then : else` at the top-level `?` and `:`.
1340/// Skips inside quotes and parens. Returns (cond, then, else).
1341fn split_ternary(s: &str) -> Option<(&str, &str, &str)> {
1342  let bytes = s.as_bytes();
1343  let mut depth: i32 = 0;
1344  let mut in_quote = false;
1345  let mut q_pos: Option<usize> = None;
1346  for (i, &b) in bytes.iter().enumerate() {
1347    match b {
1348      b'"' => in_quote = !in_quote,
1349      b'(' | b'[' | b'{' if !in_quote => depth += 1,
1350      b')' | b']' | b'}' if !in_quote => depth -= 1,
1351      b'?' if !in_quote && depth == 0 => {
1352        q_pos = Some(i);
1353        break;
1354      },
1355      _ => {},
1356    }
1357  }
1358  let qp = q_pos?;
1359  // Now find the matching `:` after qp at depth 0
1360  let mut depth: i32 = 0;
1361  let mut in_quote = false;
1362  for (j, &b) in bytes.iter().enumerate().skip(qp + 1) {
1363    match b {
1364      b'"' => in_quote = !in_quote,
1365      b'(' | b'[' | b'{' if !in_quote => depth += 1,
1366      b')' | b']' | b'}' if !in_quote => depth -= 1,
1367      b':' if !in_quote && depth == 0 => {
1368        return Some((&s[..qp], &s[qp + 1..j], &s[j + 1..]));
1369      },
1370      _ => {},
1371    }
1372  }
1373  None
1374}

1809  /// A comparison yields an INTEGER string, but arithmetic that consumes a
1810  /// comparison yields a REAL — `int_result` must be scoped to the actual
1811  /// result, not left globally set by a nested comparison. Ground truth from
1812  /// real pgfmath (`\pgfmathparse` + `\pgfmathresult`, TeX Live): `2==2`→1,
1813  /// `(1<2)`→1 (parens preserve), but `2*(1<2)`→2.0 and `2+(1<2)`→3.0.
1814  /// Regression guard for the arXiv-fork c52d3cf6 comparison port, which set a
1815  /// stream-global flag so `2*(1<2)` wrongly truncated to "2".
1816  #[test]
1817  fn comparison_int_result_is_scoped_not_global() {
1818    latexml_core::state::set_state(latexml_core::state::State::new(
1819      latexml_core::state::StateOptions::default(),
1820    ));
1821    // Bare / parenthesised comparison → integer.
1822    assert_eq!(super::pgfmathparse_eval("2==2"), "1");
1823    assert_eq!(super::pgfmathparse_eval("1<2"), "1");
1824    assert_eq!(super::pgfmathparse_eval("(1<2)"), "1");
1825    // Arithmetic over a comparison → real (the leak this guards against).
1826    assert_eq!(super::pgfmathparse_eval("2*(1<2)"), "2.0");
1827    assert_eq!(super::pgfmathparse_eval("2+(1<2)"), "3.0");
1828    // A comparison that evaluates false, then multiplied, is still a real.
1829    assert_eq!(super::pgfmathparse_eval("5*(3<2)"), "0.0");
1830    // Plain arithmetic is unaffected (no comparison anywhere).
1831    assert_eq!(super::pgfmathparse_eval("1+1"), "2.0");
1832  }
