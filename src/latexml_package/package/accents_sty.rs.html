17  // Perl: \lx@if@isaccent{thing}{true}{false}
18  // Checks if thing is a single token whose definition has exactly 1 parameter.
19  DefMacro!("\\lx@if@isaccent{}{}{}", sub[(thing, true_branch, false_branch)] {
20    let toks = thing.unlist();
21    // Must be a single token, with a definition that has exactly 1 parameter
22    let is_accent = toks.len() == 1 && {
23      match lookup_definition(&toks[0])? { Some(defn) => {
24        defn.get_num_args() == 1
25      } _ => {
26        false
27      }}
28    };
29    if is_accent {
30      Ok(true_branch)
31    } else {
32      Ok(false_branch)
33    }
34  });
35
36  // Perl: \underaccent{acc}{base} — checks if acc is an accent command
37  DefMacro!(
38    "\\underaccent{}{}",
39    "\\lx@if@isaccent{#1}{\\lx@converttounder{#1}{#2}{#1{#2}}}{\\lx@underaccentset{#1}{#2}}"
40  );

42  DefConstructor!("\\lx@underaccentset ScriptStyle {}",
43    "<ltx:XMApp><ltx:XMWrap role='UNDERACCENT'>#1</ltx:XMWrap><ltx:XMArg>#2</ltx:XMArg></ltx:XMApp>",
44    sizer => "#1",
45    alias => "\\underaccent");
46
47  // Perl: \lx@converttounder Undigested Undigested {}
48  // Absorbs #3 (which is #1{#2}, e.g. \hat{x}), then walks the XML to find
49  // XMTok[@role='OVERACCENT'] and changes to UNDERACCENT.
50  DefConstructor!("\\lx@converttounder Undigested Undigested {}", sub[document, args, _props] {
51    // #3 is already digested and absorbed — absorb it into the document
52    let thing = args[2].as_ref().unwrap();
53    document.absorb(thing, None)?;
54    // Find the OVERACCENT token in the last child and change to UNDERACCENT
55    let current = document.get_node().clone();
56    if let Some(last_child) = current.get_last_child() {
57      let acc_nodes = document.findnodes("ltx:XMTok[@role='OVERACCENT']", Some(&last_child));
58      if let Some(mut acc) = acc_nodes.into_iter().next() {
59        document.set_attribute(&mut acc, "role", "UNDERACCENT")?;
60      }
61    }
62  },
63  reversion => sub[_whatsit, args] {
64    let mut rev = vec![T_CS!("\\underaccent")];
65    rev.push(T_BEGIN!());
66    if let Some(arg0) = &args[0] { rev.extend(arg0.revert()?.unlist()); }
67    rev.push(T_END!());
68    rev.push(T_BEGIN!());
69    if let Some(arg1) = &args[1] { rev.extend(arg1.revert()?.unlist()); }
70    rev.push(T_END!());
71    Ok(Tokens::new(rev))
72  });
