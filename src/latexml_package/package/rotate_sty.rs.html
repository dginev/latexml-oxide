1use crate::prelude::*;
2
3#[rustfmt::skip]
4LoadDefinitions!({
5  // Perl: rotate.sty.ltxml
6  RequirePackage!("graphicx");
7
8  // \rotate[l]{stuff}
9  // l=left, r=right, u=upsidedown, f=flip
10  DefConstructor!("\\rotate Optional:l Digested",
11    "<ltx:inline-block angle='#angle' width='#width' height='#height' depth='#depth'\
12     innerwidth='#innerwidth' innerheight='#innerheight' innerdepth='#innerdepth'\
13     xscale='#xscale' yscale='#yscale'\
14     xtranslate='#xtranslate' ytranslate='#ytranslate'>\
15     #2\
16     </ltx:inline-block>",
17    after_digest => sub[whatsit] {
18      let rot_str = whatsit.get_arg(1)
19        .map(|a| a.to_string())
20        .unwrap_or_default();
21      let rot = if rot_str.is_empty() { "l".to_string() } else { rot_str };
22      if rot == "l" || rot == "r" {
23        let angle = if rot == "l" { -90.0 } else { 90.0 };
24        if let Some(body) = whatsit.get_arg(2) {
25          if let Ok(props) = crate::package::graphics_sty::rotated_properties(body.clone(), angle, false) {
26            for (k, v) in props {
27              whatsit.set_property(k, v);
28            }
29          }
30        }
31      } else {
32        // u=upsidedown, f=flip
33        let xscale: f64 = if rot == "u" { 1.0 } else { -1.0 };
34        let yscale: f64 = if rot == "u" { -1.0 } else { 1.0 };
35        if let Some(body) = whatsit.get_arg(2) {
36          let mut body = body.clone();
37          if let Ok((w_dim, h_dim, d_dim, _, _, _)) = body.get_size(None) {
38            let w = w_dim.value_of() as f64;
39            let h = h_dim.value_of() as f64;
40            let d = d_dim.value_of() as f64;
41            if w != 0.0 || h != 0.0 || d != 0.0 {
42              let dim_attr = |v: f64| Stored::from(latexml_core::common::dimension::attribute_format(v as i64, None));
43              let ytranslate = (h + d) * (yscale - 1.0) / 2.0;
44              whatsit.set_property("width", dim_attr(w));
45              whatsit.set_property("height", dim_attr(h * yscale));
46              whatsit.set_property("depth", dim_attr(d * yscale));
47              whatsit.set_property("xscale", Stored::from(s!("{xscale}")));
48              whatsit.set_property("yscale", Stored::from(s!("{yscale}")));
49              whatsit.set_property("ytranslate", dim_attr(ytranslate));
50            }
51          }
52        }
53      }
54    },
55    mode => "internal_vertical"
56  );
57});
