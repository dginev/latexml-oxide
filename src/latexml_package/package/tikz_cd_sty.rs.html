4LoadDefinitions!({
5  // Perl: tikz-cd.sty.ltxml
6  InputDefinitions!("tikz-cd", noltxml => true, extension => Some(Cow::Borrowed("sty")));
7
8  // tikz-cd's `rightsquigarrow` / `leadsto` arrow styles look for the
9  // flag `\tikz@library@decorations.pathmorphing@loaded` (set by
10  // `\usetikzlibrary{decorations.pathmorphing}`) and emit a
11  // `\PackageError` when it's missing. arXiv's compile cluster
12  // appears to load the library by default; papers that rely on
13  // squiggly arrows in tikzcd diagrams thus compile cleanly there
14  // but fire 1 error per squiggly use on our pipeline. Pull in the
15  // library after raw-load so the flag is set before any \begin{tikzcd}
16  // expansion. Witness: arXiv:2508.13059 — 4 errors -> 0.
17  TeX!(r"\usetikzlibrary{decorations.pathmorphing}");
18  // Also pull `decorations.markings` — used by tikz-cd's arrow-marking
19  // styles; same vendor pattern as pathmorphing. Witness 2308.06778.
20  TeX!(r"\usetikzlibrary{decorations.markings}");
21  // Defensive: also set both `@loaded` flags GLOBALLY (pgf uses
22  // `\global\let \csname ... \endcsname \pgfutil@empty` — without
23  // \global the flag stays local to the binding's load group and
24  // disappears after group exit, leaving the user's squiggly arrow
25  // hitting the missing-flag PackageError). Witness 2306.03232.
26  TeX!(r"\global\expandafter\let\csname tikz@library@decorations.pathmorphing@loaded\endcsname\@empty");
27  TeX!(r"\global\expandafter\let\csname tikz@library@decorations.markings@loaded\endcsname\@empty");
28});
