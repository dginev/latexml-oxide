3/// Perl attachfile.sty.ltxml L39-40: `%attachfileicon` name → codepoint
4/// table. Keys are case-insensitive in Perl (`lc(ToString(...))`) so
5/// this table is stored lowercase and callers should downcase their
6/// lookup key.
7fn attachfile_icon(name: &str) -> &'static str {
8  match name {
9    "pushpin" => "\u{1F4CC}",
10    "paperclip" => "\u{1F4CE}",
11    "tag" => "\u{1F3F7}",
12    "graph" => "\u{1F4CA}",
13    _ => "\u{1F4CC}", // Perl default: pushpin fallback
14  }
15}

17/// Perl attachfile.sty.ltxml L42-49 `attachfileAttributes`. Extracts the
18/// `icon`, `color`, and `file` values from the keyvals (+ optional file
19/// arg) and packs them into a stored-map for the constructor XML to
20/// reference. `file` is None for the noattach / notextattach variants.
21fn attachfile_properties(kv: Option<&Digested>, file_arg: Option<&Digested>) -> SymHashMap<Stored> {
22  let icon_key = kv
23    .and_then(|d| {
24      if let DigestedData::KeyVals(kvs) = d.data() {
25        kvs.get_value("icon")
26      } else {
27        None
28      }
29    })
30    .map(|v| v.to_string().to_ascii_lowercase())
31    .unwrap_or_else(|| "pushpin".to_string());
32  let color = kv
33    .and_then(|d| {
34      if let DigestedData::KeyVals(kvs) = d.data() {
35        kvs.get_value("color")
36      } else {
37        None
38      }
39    })
40    .map(|v| v.to_string())
41    .unwrap_or_default();
42  let file = file_arg.map(|f| f.to_string()).unwrap_or_default();
43  stored_map!(
44    "icon"  => attachfile_icon(&icon_key).to_string(),
45    "color" => color,
46    "file"  => file
47  )
48}
