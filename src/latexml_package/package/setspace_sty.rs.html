2//! Perl: setspace.sty.ltxml
3use crate::prelude::*;

5LoadDefinitions!({
6  def_macro_noop("\\singlespacing")?;
7  def_macro_noop("\\onehalfspacing")?;
8  def_macro_noop("\\doublespacing")?;
9  def_macro_noop("\\setstretch{}")?;
10  def_macro_noop("\\SetSinglespace{}")?;
11  def_macro_noop("\\setdisplayskipstretch{}")?;
12  def_macro_noop("\\restore@spacing")?;
13
14  // {spacing}{}: the body-wrapping variant. Keep BOUND_MODE vertical so `$$`
15  // inside it still enters display math. Witness 2305.08368:
16  // `\begin{spacing}{1.25}` wrapping the whole body made `$$x_1=...$$` fall
17  // through the `$` handler's vertical-only check (tex_math.rs:447) → 199
18  // `Error:unexpected:_` cascades. Perl-faithful binding is `#body` only, but
19  // the default Package.pm mode is `restricted_horizontal`; in Rust we have
20  // to make `internal_vertical` explicit so paragraphs and display math
21  // survive the wrap.
22  DefEnvironment!("{spacing}{}", "#body", mode => "internal_vertical");
23  DefEnvironment!("{singlespace}", "#body");
24  DefEnvironment!("{onehalfspace}", "#body");
25  DefEnvironment!("{doublespace}", "#body");
26  // Standalone-switch overrides: some papers (witness 2310.08233 IEEEtran)
27  // use `\singlespace` as a SWITCH inside an arg-grabbing context such as
28  // `\title{\singlespace ...}`. DefEnvironment binds `\singlespace` to the
29  // env-begin CS, which pushes a (restricted_)horizontal mode-switch group.
30  // That group cannot be cleanly popped at arg-close, leaking into the
31  // following `\@add@frontmatter@now`. Override the env-begin CS (and its
32  // sibling `\endsinglespace`) with plain noops so standalone use is inert.
33  // For `\begin{singlespace}...\end{singlespace}` blocks: \begin/\end track
34  // env scope themselves; the begin/end CSes are just hooks. Without the
35  // hooks the body still renders (it is just literal #body), so the env
36  // form remains correct.
37  def_macro_noop("\\singlespace")?;
38  def_macro_noop("\\endsinglespace")?;
39  def_macro_noop("\\onehalfspace")?;
40  def_macro_noop("\\endonehalfspace")?;
41  def_macro_noop("\\doublespace")?;
42  def_macro_noop("\\enddoublespace")?;
43});
