4LoadDefinitions!({
5  // Perl: xypic.tex.ltxml — the `\input xypic` entry point.
6  //
7  //   InputDefinitions('xy', type => 'tex');
8  //   RawTeX('\xyoption{v2}');
9  //
10  // CRITICAL (Perl-faithful split): this loads the xy *tex* binding
11  // overlay (= our `xy_sty.rs`, registered for `("xy","tex")`) but does
12  // NOT `RequirePackage('xy')`, so the xy *package* (`xy.sty`) is left
13  // UN-marked. A subsequent `\usepackage[all]{xy}` therefore still runs
14  // its option processing (loading the curve/arrow/... feature files),
15  // rather than being short-circuited by the "already loaded" option-clash
16  // early-stop in `input_definitions`.
17  //
18  // Why this matters: a curved arrow `\ar@/^1pc/[u]` digests through
19  // `\crvi` (xycurve.tex L69), which only defines when the `curve` feature
20  // loads via `\xyoption{all}` (or `{curve}`). When `\input xypic` mis-marks
21  // `xy.sty` loaded (the previous `RequirePackage('xy')` behaviour), the
22  // document's own `\usepackage[all]{xy}` got dropped, curve never loaded,
23  // and `\crvi` stayed undefined. Witness 2011.01105 (`\input xypic` +
24  // `\usepackage[all]{xy}` + `\ar@/^1pc/`, Perl-clean); also 2311.05789
25  // (`\xygraph` from `[all,tips]`).
26  //
27  // `\usepackage{xypic}` (the *.sty* entry) keeps Perl `xypic.sty.ltxml`'s
28  // `RequirePackage('xy', options => ['v2'])` semantics in `xypic_sty.rs`.
29  InputDefinitions!("xy", extension => Some(Cow::Borrowed("tex")));
30  Digest!("\\xyoption{v2}")?;
31});
