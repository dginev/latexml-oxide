3/// Metadata for a compiled \lxDeclare pattern.
4/// Contains the XPath, pattern type for Rust-side filtering, and wildcard info.
5pub struct DeclarePattern {
6  pub xpath:          String,
7  /// "simple", "subscript", "prime", "accent"
8  pub pattern_type:   &'static str,
9  /// Base token text for subscript/prime/accent base matching (e.g. "x")
10  pub base_text:      Option<String>,
11  /// For literal subscripts: the subscript content text (e.g. "1")
12  pub sub_text:       Option<String>,
13  /// For accent patterns: the accent name (e.g. "hat")
14  pub accent_name:    Option<String>,
15  #[allow(dead_code)]
16  pub has_wildcard:   bool,
17  pub wildcard_paths: Option<Vec<Vec<usize>>>,
18}

20/// Generate an XPath text predicate for a base token specification.
21fn base_text_predicate(base: &str) -> String {
22  if base.starts_with('\\') {
23    let cmd = base.trim_start_matches('\\');
24    if let Some(inner) = cmd
25      .strip_prefix("mathcal{")
26      .and_then(|s| s.strip_suffix('}'))
27    {
28      format!("@font='caligraphic' and text()='{inner}'")
29    } else {
30      match cmd {
31        "varepsilon" => "@meaning='varepsilon'".to_string(),
32        _ => format!("@meaning='{cmd}'"),
33      }
34    }
35  } else {
36    format!("text()='{}'", base.replace('\'', "&apos;"))
37  }
38}

51fn compile_declare_pattern(body_text: &str) -> DeclarePattern {
52  // === Subscript patterns ===
53  // IMPORTANT: Rewrites run BEFORE math parsing. The pre-parsed DOM has:
54  //   <XMTok>x</XMTok> <XMApp role="POSTSUBSCRIPT"><XMTok>n</XMTok></XMApp>
55  // NOT the post-parsed: <XMApp><XMTok role="SUBSCRIPTOP"/><XMTok>x</XMTok><XMTok>n</XMTok></XMApp>
56  // Match the BASE XMTok, with select_count=2 to include the POSTSUBSCRIPT sibling.
57  // Rust-side filtering verifies the sibling structure.
58
59  // Wildcard: x_\WildCard, \varepsilon_\WildCard, \mathcal{T}_\WildCard
60  if let Some(base) = body_text.strip_suffix("_\\WildCard") {
61    let base = base.trim().to_string();
62    let base_pred = base_text_predicate(&base);
63    return DeclarePattern {
64      // Match the base XMTok; Rust-side filter checks POSTSUBSCRIPT sibling
65      xpath:          format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
66      pattern_type:   "subscript",
67      base_text:      Some(base),
68      sub_text:       None,
69      accent_name:    None,
70      has_wildcard:   true,
71      // Wildcard = child 1 of sibling 2 (the content of POSTSUBSCRIPT XMApp)
72      wildcard_paths: Some(vec![vec![2, 1]]),
73    };
74  }
75  // Braced wildcard subscripts: x_{\WildCard}, x_{\WildCard,\WildCard}
76  if body_text.contains("_{\\WildCard") {
77    if let Some(idx) = body_text.find("_{") {
78      let base = body_text[..idx].trim().to_string();
79      let base_pred = base_text_predicate(&base);
80      let brace_content = &body_text[idx + 2..body_text.len().saturating_sub(1)];
81      let nwilds = brace_content.matches("\\WildCard").count();
82      let wpaths = if nwilds <= 1 {
83        vec![vec![2, 1]] // child 1 of sibling 2 (POSTSUBSCRIPT content)
84      } else {
85        (1..=nwilds).map(|i| vec![2, 1, i]).collect()
86      };
87      return DeclarePattern {
88        xpath:          format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
89        pattern_type:   "subscript",
90        base_text:      Some(base),
91        sub_text:       None,
92        accent_name:    None,
93        has_wildcard:   true,
94        wildcard_paths: Some(wpaths),
95      };
96    }
97  }
98  // Literal subscript: x_1, x_{1}, x_{2n-1}
99  // Pre-parsed: XMTok[x] + XMApp[POSTSUBSCRIPT, XMTok[1]]
100  if let Some((base, sub)) = parse_subscript_literal(body_text) {
101    let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
102    return DeclarePattern {
103      xpath:          format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
104      pattern_type:   "literal_subscript",
105      base_text:      Some(base),
106      sub_text:       Some(sub),
107      accent_name:    None,
108      has_wildcard:   false,
109      wildcard_paths: None,
110    };
111  }
112
113  // === Accent patterns ===
114  // Wildcard accent: \hat{\WildCard}, \widehat{\WildCard}
115  for accent in &[
116    "hat", "widehat", "tilde", "bar", "vec", "dot", "ddot", "check", "breve",
117  ] {
118    let pattern = format!("\\{accent}{{\\WildCard}}");
119    if body_text == pattern {
120      return DeclarePattern {
121        // Broad: match any XMApp. Rust filters by accent name in first child.
122        xpath:          "descendant-or-self::*[local-name()='XMApp']".to_string(),
123        pattern_type:   "accent",
124        base_text:      None,
125        sub_text:       None,
126        accent_name:    Some(accent.to_string()),
127        has_wildcard:   true,
128        // Wildcard = child 2 (base content) of the accent XMApp
129        wildcard_paths: Some(vec![vec![1, 2]]),
130      };
131    }
132  }
133  // Literal accent: \hat{x}, \widehat{x}
134  for accent in &[
135    "hat", "widehat", "tilde", "bar", "vec", "dot", "ddot", "check", "breve",
136  ] {
137    if let Some(rest) = body_text.strip_prefix(&format!("\\{accent}{{")) {
138      if let Some(inner) = rest.strip_suffix('}') {
139        if !inner.contains("WildCard") {
140          return DeclarePattern {
141            xpath:          "descendant-or-self::*[local-name()='XMApp']".to_string(),
142            pattern_type:   "accent",
143            base_text:      Some(inner.to_string()),
144            sub_text:       None,
145            accent_name:    Some(accent.to_string()),
146            has_wildcard:   false,
147            wildcard_paths: None,
148          };
149        }
150      }
151    }
152  }
153
154  // === Prime pattern ===
155  // x^{\prime} → after parsing: XMApp[SUPERSCRIPTOP, XMTok(x), XMTok(prime)]
156  // Match the XMApp with SUPERSCRIPTOP and base text.
157  if let Some(base) = body_text.strip_suffix("^{\\prime}") {
158    let base = base.trim().to_string();
159    if !base.is_empty() && !base.contains('\\') {
160      let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
161      return DeclarePattern {
162        // Pre-parsed: XMTok[x] + XMApp[POSTSUPERSCRIPT, XMTok[prime]]
163        xpath:          format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
164        pattern_type:   "prime",
165        base_text:      Some(base),
166        sub_text:       None,
167        accent_name:    None,
168        has_wildcard:   false,
169        wildcard_paths: None,
170      };
171    }
172  }
173  // Also handle raw prime: x'
174  if body_text.ends_with('\'') && body_text.len() > 1 {
175    let base = body_text[..body_text.len() - 1].trim().to_string();
176    if !base.is_empty() && !base.contains('\\') {
177      let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
178      return DeclarePattern {
179        // Pre-parsed: XMTok[x] + XMApp[POSTSUPERSCRIPT, XMTok[prime]]
180        xpath:          format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
181        pattern_type:   "prime",
182        base_text:      Some(base),
183        sub_text:       None,
184        accent_name:    None,
185        has_wildcard:   false,
186        wildcard_paths: None,
187      };
188    }
189  }
190
191  // === Fallback: simple token pattern ===
192  // For single characters/words without special structure, match as XMTok by text.
193  // This handles DefMathRewrite match strings like 'a', 'f', 'x', etc.
194  if !body_text.is_empty() && !body_text.contains('\\') {
195    return DeclarePattern {
196      xpath:          format!(
197        "descendant-or-self::*[local-name()='XMTok' and text()='{}']",
198        body_text.replace('\'', "&apos;")
199      ),
200      pattern_type:   "simple",
201      base_text:      None,
202      sub_text:       None,
203      accent_name:    None,
204      has_wildcard:   false,
205      wildcard_paths: None,
206    };
207  }
208
209  // Truly unrecognized pattern (e.g. complex TeX commands without matching rules)
210  DeclarePattern {
211    xpath:          String::new(),
212    pattern_type:   "unknown",
213    base_text:      None,
214    sub_text:       None,
215    accent_name:    None,
216    has_wildcard:   false,
217    wildcard_paths: None,
218  }
219}

221/// Parse a literal (non-wildcard) subscript pattern like "x_1" or "x_{2n-1}".
222/// Returns (base, subscript_content) if recognized.
223fn parse_subscript_literal(body_text: &str) -> Option<(String, String)> {
224  if body_text.contains("WildCard") {
225    return None;
226  }
227  // Check for _ subscript
228  let idx = body_text.find('_')?;
229  let base = body_text[..idx].trim().to_string();
230  if base.is_empty() {
231    return None;
232  }
233  let sub = body_text[idx + 1..].trim();
234  // Strip braces: {1} → 1, {2n-1} → 2n-1
235  let sub = sub
236    .strip_prefix('{')
237    .and_then(|s| s.strip_suffix('}'))
238    .unwrap_or(sub);
239  Some((base, sub.to_string()))
240}
