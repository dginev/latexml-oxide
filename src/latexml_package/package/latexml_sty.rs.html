3/// Metadata for a compiled \lxDeclare pattern.
4/// Contains the XPath, pattern type for Rust-side filtering, and wildcard info.
5pub struct DeclarePattern {
6  pub xpath:          String,
7  /// "simple", "subscript", "prime", "accent"
8  pub pattern_type:   &'static str,
9  /// Base token text for subscript/prime/accent base matching (e.g. "x")
10  pub base_text:      Option<String>,
11  /// For literal subscripts: the subscript content text (e.g. "1")
12  pub sub_text:       Option<String>,
13  /// For accent patterns: the accent name (e.g. "hat")
14  pub accent_name:    Option<String>,
15  #[allow(dead_code)]
16  pub has_wildcard:   bool,
17  pub wildcard_paths: Option<Vec<Vec<usize>>>,
18}

20/// Generate an XPath text predicate for a base token specification.
21fn base_text_predicate(base: &str) -> String {
22  if base.starts_with('\\') {
23    let cmd = base.trim_start_matches('\\');
24    if let Some(inner) = cmd.strip_prefix("mathcal{").and_then(|s| s.strip_suffix('}')) {
25      format!("@font='caligraphic' and text()='{inner}'")
26    } else {
27      match cmd {
28        "varepsilon" => "@meaning='varepsilon'".to_string(),
29        _ => format!("@meaning='{cmd}'"),
30      }
31    }
32  } else {
33    format!("text()='{}'", base.replace('\'', "&apos;"))
34  }
35}

38/// Handles both wildcard and non-wildcard patterns.

40/// Perl: compile_match1 digests tokens to DOM, then domToXPath.

42/// with Rust-side filtering criteria (avoids XPath nested predicate bug).
43/// Public entry point for the .latexml file loader.
44pub fn compile_declare_pattern_pub(body_text: &str) -> DeclarePattern {
45  compile_declare_pattern(body_text)

48fn compile_declare_pattern(body_text: &str) -> DeclarePattern {
49  // === Subscript patterns ===
50  // IMPORTANT: Rewrites run BEFORE math parsing. The pre-parsed DOM has:
51  //   <XMTok>x</XMTok> <XMApp role="POSTSUBSCRIPT"><XMTok>n</XMTok></XMApp>
52  // NOT the post-parsed: <XMApp><XMTok role="SUBSCRIPTOP"/><XMTok>x</XMTok><XMTok>n</XMTok></XMApp>
53  // Match the BASE XMTok, with select_count=2 to include the POSTSUBSCRIPT sibling.
54  // Rust-side filtering verifies the sibling structure.
55
56  // Wildcard: x_\WildCard, \varepsilon_\WildCard, \mathcal{T}_\WildCard
57  if let Some(base) = body_text.strip_suffix("_\\WildCard") {
58    let base = base.trim().to_string();
59    let base_pred = base_text_predicate(&base);
60    return DeclarePattern {
61      // Match the base XMTok; Rust-side filter checks POSTSUBSCRIPT sibling
62      xpath: format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
63      pattern_type: "subscript",
64      base_text: Some(base),
65      sub_text: None,
66      accent_name: None,
67      has_wildcard: true,
68      // Wildcard = child 1 of sibling 2 (the content of POSTSUBSCRIPT XMApp)
69      wildcard_paths: Some(vec![vec![2, 1]]),
70    };
71  }
72  // Braced wildcard subscripts: x_{\WildCard}, x_{\WildCard,\WildCard}
73  if body_text.contains("_{\\WildCard") {
74    if let Some(idx) = body_text.find("_{") {
75      let base = body_text[..idx].trim().to_string();
76      let base_pred = base_text_predicate(&base);
77      let brace_content = &body_text[idx + 2..body_text.len().saturating_sub(1)];
78      let nwilds = brace_content.matches("\\WildCard").count();
79      let wpaths = if nwilds <= 1 {
80        vec![vec![2, 1]]  // child 1 of sibling 2 (POSTSUBSCRIPT content)
81      } else {
82        (1..=nwilds).map(|i| vec![2, 1, i]).collect()
83      };
84      return DeclarePattern {
85        xpath: format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
86        pattern_type: "subscript",
87        base_text: Some(base),
88        sub_text: None,
89        accent_name: None,
90        has_wildcard: true,
91        wildcard_paths: Some(wpaths),
92      };
93    }
94  }
95  // Literal subscript: x_1, x_{1}, x_{2n-1}
96  // Pre-parsed: XMTok[x] + XMApp[POSTSUBSCRIPT, XMTok[1]]
97  if let Some((base, sub)) = parse_subscript_literal(body_text) {
98    let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
99    return DeclarePattern {
100      xpath: format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
101      pattern_type: "literal_subscript",
102      base_text: Some(base),
103      sub_text: Some(sub),
104      accent_name: None,
105      has_wildcard: false,
106      wildcard_paths: None,
107    };
108  }
109
110  // === Accent patterns ===
111  // Wildcard accent: \hat{\WildCard}, \widehat{\WildCard}
112  for accent in &["hat", "widehat", "tilde", "bar", "vec", "dot", "ddot", "check", "breve"] {
113    let pattern = format!("\\{accent}{{\\WildCard}}");
114    if body_text == pattern {
115      return DeclarePattern {
116        // Broad: match any XMApp. Rust filters by accent name in first child.
117        xpath: "descendant-or-self::*[local-name()='XMApp']".to_string(),
118        pattern_type: "accent",
119        base_text: None,
120        sub_text: None,
121        accent_name: Some(accent.to_string()),
122        has_wildcard: true,
123        // Wildcard = child 2 (base content) of the accent XMApp
124        wildcard_paths: Some(vec![vec![1, 2]]),
125      };
126    }
127  }
128  // Literal accent: \hat{x}, \widehat{x}
129  for accent in &["hat", "widehat", "tilde", "bar", "vec", "dot", "ddot", "check", "breve"] {
130    if let Some(rest) = body_text.strip_prefix(&format!("\\{accent}{{")) {
131      if let Some(inner) = rest.strip_suffix('}') {
132        if !inner.contains("WildCard") {
133          return DeclarePattern {
134            xpath: "descendant-or-self::*[local-name()='XMApp']".to_string(),
135            pattern_type: "accent",
136            base_text: Some(inner.to_string()),
137            sub_text: None,
138            accent_name: Some(accent.to_string()),
139            has_wildcard: false,
140            wildcard_paths: None,
141          };
142        }
143      }
144    }
145  }
146
147  // === Prime pattern ===
148  // x^{\prime} → after parsing: XMApp[SUPERSCRIPTOP, XMTok(x), XMTok(prime)]
149  // Match the XMApp with SUPERSCRIPTOP and base text.
150  if let Some(base) = body_text.strip_suffix("^{\\prime}") {
151    let base = base.trim().to_string();
152    if !base.is_empty() && !base.contains('\\') {
153      let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
154      return DeclarePattern {
155        // Pre-parsed: XMTok[x] + XMApp[POSTSUPERSCRIPT, XMTok[prime]]
156        xpath: format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
157        pattern_type: "prime",
158        base_text: Some(base),
159        sub_text: None,
160        accent_name: None,
161        has_wildcard: false,
162        wildcard_paths: None,
163      };
164    }
165  }
166  // Also handle raw prime: x'
167  if body_text.ends_with('\'') && body_text.len() > 1 {
168    let base = body_text[..body_text.len() - 1].trim().to_string();
169    if !base.is_empty() && !base.contains('\\') {
170      let base_pred = format!("text()='{}'", base.replace('\'', "&apos;"));
171      return DeclarePattern {
172        // Pre-parsed: XMTok[x] + XMApp[POSTSUPERSCRIPT, XMTok[prime]]
173        xpath: format!("descendant-or-self::*[local-name()='XMTok' and {base_pred}]"),
174        pattern_type: "prime",
175        base_text: Some(base),
176        sub_text: None,
177        accent_name: None,
178        has_wildcard: false,
179        wildcard_paths: None,
180      };
181    }
182  }
183
184  // === Fallback: simple token pattern ===
185  // For single characters/words without special structure, match as XMTok by text.
186  // This handles DefMathRewrite match strings like 'a', 'f', 'x', etc.
187  if !body_text.is_empty() && !body_text.contains('\\') {
188    return DeclarePattern {
189      xpath: format!(
190        "descendant-or-self::*[local-name()='XMTok' and text()='{}']",
191        body_text.replace('\'', "&apos;")),
192      pattern_type: "simple",
193      base_text: None,
194      sub_text: None,
195      accent_name: None,
196      has_wildcard: false,
197      wildcard_paths: None,
198    };
199  }
200
201  // Truly unrecognized pattern (e.g. complex TeX commands without matching rules)
202  DeclarePattern {
203    xpath: String::new(),
204    pattern_type: "unknown",
205    base_text: None,
206    sub_text: None,
207    accent_name: None,
208    has_wildcard: false,
209    wildcard_paths: None,
210  }
211}

213/// Parse a literal (non-wildcard) subscript pattern like "x_1" or "x_{2n-1}".
214/// Returns (base, subscript_content) if recognized.
215fn parse_subscript_literal(body_text: &str) -> Option<(String, String)> {
216  if body_text.contains("WildCard") { return None; }
217  // Check for _ subscript
218  let idx = body_text.find('_')?;
219  let base = body_text[..idx].trim().to_string();
220  if base.is_empty() { return None; }
221  let sub = body_text[idx + 1..].trim();
222  // Strip braces: {1} → 1, {2n-1} → 2n-1
223  let sub = sub.strip_prefix('{').and_then(|s| s.strip_suffix('}')).unwrap_or(sub);
224  Some((base, sub.to_string()))
225}
