1//! ctable.sty binding: deps + conditional raw-load.
2//!
3//! Perl ships no ctable binding and raw-loads the real ctable.sty (it's in TL),
4//! so `\ctable[…]{…}{…}{…}` works. We do the same — BUT guarded on tikz being
5//! absent, because of a known clash:
6//!
7//! ctable.sty (TL) does a guarded
8//! `\@ifpackageloaded{tikz}{…define \transparent…}{\RequirePackage{transparent}}`
9//! at load time, then `\@ifpackageloaded{tikz}{\@ifpackageloaded{transparent}
10//! {\PackageError "You must load ctable after tikz"}{}}{}` inside
11//! `\AtBeginDocument`. When tikz IS loaded, our raw-load path ends up with both
12//! tikz and transparent in scope and fires that error. So for tikz papers we
13//! keep the deps-only behavior (no `\ctable`, table dropped — same net effect
14//! as Perl when its TEXINPUTS misses ctable). Witnesses: arXiv:1912.08312,
15//! 1912.08818, 2001.00802, 2001.05616, 2001.09838, 2001.09978 (tikz+ctable).
16//!
17//! For the COMMON non-tikz case the earlier pure-deps stub was WRONG: it left
18//! `\ctable` undefined, so papers that actually use `\ctable` errored where
19//! Perl is clean. Witness arXiv:2011.04706 (`\usepackage{ctable}` +
20//! `\ctable[caption=…]{lcccccr}{…}{…}`, no tikz): 3 err → 0. We now raw-load
21//! ctable.sty there, defining `\ctable` exactly as Perl does.
22
23use crate::prelude::*;
24
25LoadDefinitions!({
26  // Pull in ctable's real dependencies (`\RequirePackage{ifpdf,
27  // etoolbox,xcolor,xkeyval,array,tabularx,booktabs,rotating}` —
28  // ctable.sty L28). Papers that rely on ctable for its transitive
29  // dependencies (the most common being booktabs for `\toprule`/
30  // `\midrule`/`\bottomrule`) need this — without it our previous
31  // pure-no-op stub silently dropped them. Witness 2002.05708 (loaded
32  // ctable, used \toprule/\midrule/\bottomrule from booktabs without
33  // a direct \usepackage{booktabs}).
34  RequirePackage!("ifpdf");
35  RequirePackage!("etoolbox");
36  RequirePackage!("xcolor");
37  RequirePackage!("xkeyval");
38  RequirePackage!("array");
39  RequirePackage!("tabularx");
40  RequirePackage!("booktabs");
41  RequirePackage!("rotating");
42  // Raw-load the real ctable.sty so `\ctable[…]{…}{…}{…}` is actually defined
43  // (Perl ships no ctable binding and raw-loads it → clean). The documented
44  // "You must load ctable after tikz" AtBeginDocument error only fires when
45  // tikz is ALSO loaded, so guard the raw-load on tikz being absent. tikz
46  // papers keep the deps-only behavior (matching the prior stub).
47  // Witness arXiv:2011.04706 (`\usepackage{ctable}` + `\ctable[caption=…]{…}`,
48  // no tikz: Perl 0 err; the old deps-only stub left `\ctable` undefined → 3 err).
49  if !lookup_bool("tikz.sty_loaded") {
50    InputDefinitions!("ctable", noltxml => true, extension => Some(Cow::Borrowed("sty")));
51  }
52});
