54  // Perl latex_constructs L4316-4353: {filecontents} and {filecontents*} environments
55  // Read raw lines until \end{filecontents}, cache content for later \input.
56  fn cache_filecontents(end_marker: &str, header_star: bool) -> Result<()> {
57    gullet::skip_spaces()?;
58    let filename_toks = gullet::read_arg(ExpansionLevel::Off)?;
59    let filename = filename_toks.to_string();
60    let mut lines: Vec<String> = vec![
61      format!("%% LaTeX2e file `{filename}'"),
62      if header_star {
63        "%% generated by the `filecontents*' environment".to_string()
64      } else {
65        "%% generated by the `filecontents' environment".to_string()
66      },
67    ];
68    if !header_star { lines.push("%%".to_string()); }
69    // Discard remainder of \begin{filecontents} line
70    gullet::read_raw_line();
71    // Read raw lines until end marker
72    loop {
73      match gullet::read_raw_line() {
74        Some(line) if !line.contains(end_marker) => lines.push(line),
75        _ => break,
76      }
77    }
78    let n = lines.len();
79    let content = lines.join("\n");
80    Info!("note", "filecontents", s!("Cached filecontents for {filename} ({n} lines)"));
81    state::assign_value(&s!("{filename}_contents"), Stored::from(content), Some(Scope::Global));
82    Ok(())
83  }
