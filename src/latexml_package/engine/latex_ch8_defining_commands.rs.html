13  DefPrimitive!("\\newcommand OptionalMatch:* DefToken [Number][]{}",
14  sub[(_star,cs_token,nargs,opt,body)] {
15    let nargs = nargs.value_of() as usize;
16    if !IsDefinable!(&cs_token) {
17      if !has_value(&s!("{}:locked", cs_token.to_string())) { // not locked, inform.
18        let message = s!("Ignoring redefinition (\\newcommand) of {}", cs_token.stringify());
19        Info!("ignore", cs_token, message);
20      }
21      return Ok(vec![]);
22    }
23    let macro_args = convert_latex_args(nargs, opt)?;
24    DefMacro!(cs_token, macro_args, body);
25  });
26
27  DefPrimitive!("\\renewcommand OptionalMatch:* DefToken [Number][]{}",
28  sub[(_star, cs, nargs_num, opt, body)] {
29    let nargs = nargs_num.value_of() as usize;
30    let opt = if let Some(ref opt_content) = opt {
31      if opt_content.is_empty() { None } else { opt }
32    } else { opt };
33    let macro_args = convert_latex_args(nargs, opt)?;
34    DefMacro!(cs, macro_args, body);
35  });

57      let cs_args = convert_latex_args(nargs, opt_checked)?;
58      DefMacro!(cs, cs_args, body);

90  // The following commands define encoding-specific expansions
91  // or glyphs.  The control-sequence is defined to use the expansion for
92  // the current encoding, if any, or the default expansion (for encoding "?").
93  // We don't want to redefine control-sequence if it already has a definition:
94  // It may be that we've already defined it to expand into the above conditional.
95  // But more importantly, we don't want to override a hand-written definition (if any).

97  DefPrimitive!("\\DeclareTextCommand DefToken {}[Number][]{}",
98  sub[(cs, encoding, nargs, opts, expansion)] {
99    let cs_str = cs.to_string();
100    let nargs = nargs.value_of() as usize;
101
102    let encoding = Expand!(encoding);
103    if !IsDefined!(&cs) {    // If not already defined...
104      DefMacro!(cs, None, Some(s!(r#"""
105      \expandafter\ifx\csname\cf@encoding\string{}\endcsname\relax\csname?\string{}\endcsname\else
106      \csname\cf@encoding\string{}\endcsname\fi
107      """#, cs_str, cs_str, cs_str).into()));
108    }
109     let ecs = T_CS!(s!("\\{}{}", encoding, cs_str));
110     let ecs_args = convert_latex_args(nargs, opts)?;
111     DefMacro!(ecs, ecs_args, expansion);
112  });

142  DefPrimitive!("\\DeclareTextSymbol DefToken {}{Number}", sub[(cs, encoding, code)] {
143    let code_value = code.value_of() as u8;
144    let cs_str = cs.to_string();
145    let ecs = T_CS!(s!("\\{encoding}{cs_str}"));
146    let encoding_str = Expand!(encoding).to_string();
147    if IsDefinable!(&cs) {    // If not already defined...
148      DefMacro!(cs, None, Some(s!(r"\expandafter\ifx\csname\cf@encoding\string{cs_str}\endcsname\relax
149      \csname?\string{cs_str}\endcsname\else\csname\cf@encoding\string{cs_str}\endcsname\fi").into()));
150    }
151    let replacement_value = font::decode(code_value, Some(encoding_str), false).unwrap();
152    let replacement = PrimitiveBody::from(replacement_value);
153    def_primitive(ecs, None, Some(replacement), PrimitiveOptions::default())?;
154  });
