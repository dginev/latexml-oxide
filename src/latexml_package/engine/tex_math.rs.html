25  // Decide whether we're going into or out of math, inline or display.
26  Tag!("ltx:XMText", auto_open => true, auto_close => true);
27  // This really should be T_MATH
28  // and it should (or not) check for a second $ only if not in restricted horizontal mode!
29  // (and then all the \lx@dollar@in@(text|math|normal)mode defns would not be needed.
30  DefPrimitive!(T_CS!("\\lx@dollar@in@normalmode"), None, {
31    let mut op = "\\lx@begin@inline@math";
32    {
33      let mode = state::lookup_string("MODE");
34      Debug!("T_MATH primitive current mode: {:?}", mode);
35      if mode == "display_math" {
36        if gullet::if_next(T_MATH!())? {
37          gullet::read_token()?;
38          op = "\\lx@end@display@math";
39        } else {
40          // Avoid a Fatal, but we're likely in trouble.
41          // Should we switch to text mode? (LaTeX normally wouldn't)
42          // Did we miss something and would should have already been in text mode? Possibly...
43          Error!(
44            "expected",
45            "$",
46            "Missing $ closing display math.\nIgnoring; expect to be in wrong math/text mode."
47          );
48          op = "";
49        }
50      } else if mode == "inline_math" {
51        op = "\\lx@end@inline@math";
52      } else if gullet::if_next(T_MATH!())? {
53        gullet::read_token()?;
54        op = "\\lx@begin@display@math";
55      }
56    }
57    if !op.is_empty() {
58      Ok(stomach::invoke_token(&T_CS!(op))?)
59    } else {
60      Ok(Vec::new())
61    }
62  });

96    let level = stomach::get_boxing_level();
97    if lookup_int("MATH_ALIGN_$_BEGUN") == (level as i64) {
98      // If we're begun making _something_ with $.
99      let l = if lookup_bool("IN_MATH") {
100        // But we're somehow in math?

105      assign_value("MATH_ALIGN_$_BEGUN", 0, None); // Reset this AFTER finishing the something

109      if lookup_bool("IN_MATH") {
110        // If we're "still" in math

112      } else {
113        stomach::invoke_token(&T_CS!("\\lx@begin@inline@math"))

183          set_dual_branch("content");

186          if ctex != tex {

188          }
189          Some(tex)
190        } else {
191          None
192        }
193      } else {
194        None
195      };
196      if let Some(tex_string) = tex_opt {
