1use crate::prelude::*;
2//======================================================================
3// C.7.8 Changing Style
4//======================================================================
5// For Math style changes, we record the current font, which is then merged
6// into the Whatsit's created for letters, etc.  The merging depends on
7// the type of letter, greek, symbol, etc.
8// Apparently, with the normal TeX setup, these fonts don't really merge,
9// rather they override all of family, series and shape.
10LoadDefinitions!({
11  DefConstructor!("\\mathrm{}", "#1", bounded => true, require_math => true,
12    locked => true,
13    font => {family => "serif", series => "medium", shape => "upright"});
14  DefConstructor!("\\mathit{}", "#1", bounded => true, require_math => true,
15    locked => true,
16    font => {shape => "italic", family => "serif", series => "medium"});
17  DefConstructor!("\\mathbf{}", "#1", bounded => true, require_math => true,
18    locked => true,
19    font => {series => "bold", family => "serif", shape => "upright"});
20  DefConstructor!("\\mathsf{}", "#1", bounded => true, require_math => true,
21    locked => true,
22    font => {family => "sansserif", series => "medium", shape => "upright"});
23  DefConstructor!("\\mathtt{}", "#1", bounded => true, require_math => true,
24    locked => true,
25    font => {family => "typewriter", series => "medium", shape => "upright"});
26  DefConstructor!("\\mathcal{}", "#1", bounded => true, require_math => true,
27    locked => true,
28    font => {family => "caligraphic", series => "medium", shape => "upright"});
29  DefConstructor!("\\mathscr{}", "#1", bounded => true, require_math => true,
30    locked => true,
31    font => {family => "script", series => "medium", shape => "upright"});
32  DefConstructor!("\\mathnormal{}", "#1", bounded => true, require_math => true,
33    locked => true,
34    font => {family => "math", shape => "italic", series => "medium"});
35
36  DefMacro!("\\fontsubfuzz", ".4pt");
37  DefMacro!("\\oldstylenums", "");
38
39  DefPrimitive!("\\operator@font", None,
40    font => {family => "serif", series => "medium", shape => "upright"});
41});
