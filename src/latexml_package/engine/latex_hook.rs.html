3/// Perl: DefAutoload — define a macro that auto-loads a package on first use.
4/// When the command is first invoked, it loads the specified package (via RequirePackage),
5/// then re-emits the original CS so it gets re-executed with the proper definition.
6fn def_autoload(cs_name: &str, package: &str) -> Result<()> {
7  use latexml_core::definition::ExpansionBody;
8  let cs_tok = T_CS!(cs_name);
9  // Don't overwrite if already defined
10  if IsDefined!(&cs_tok) {
11    return Ok(());
12  }
13  let pkg_name = package.to_string();
14  let cs_for_closure = cs_tok;
15  def_macro(
16    cs_tok,
17    None,
18    ExpansionBody::Closure(Rc::new(move |_args| {
19      require_package(&pkg_name, RequireOptions::default())?;
20      Ok(Tokens::new(vec![cs_for_closure]))
21    })),
22    None,
23  )?;
24  Ok(())
25}
