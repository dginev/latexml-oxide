12  //======================================================================
13  // Font declaration
14  //----------------------------------------------------------------------
15  // \font             iq loads information about a font into TeX's memory.
16  // \fontname         c  returns the system file name for a font.
17  // \fontdimen        iq holds font parameters.
18  // \nullfont         iq is a predefined font with no characters.
19
20  // # 2nd arg is <font> = <fontdef token> | \font | <family member>
21  // #  <family member> = <font range><4bit number>
22  // #  <font range> = \textfont | \scriptfont | \scriptscriptfont
23  // Perl: FontDef param type — for \textfont/\scriptfont/\scriptscriptfont, reads family
24  // number and looks up the stored font CS token.  For \font, returns current_FontDef.
25  DefParameterType!(FontToken, sub[_inner, _extra] {
26    let token = gullet::read_token()?.unwrap();
27    if let Some(font_type) = token.with_str(|ts| {
28      if ts.starts_with("\\textfont") && ts == "\\textfont" { Some("textfont") }
29      else if ts.starts_with("\\scriptscriptfont") && ts == "\\scriptscriptfont" { Some("scriptscriptfont") }
30      else if ts.starts_with("\\scriptfont") && ts == "\\scriptfont" { Some("scriptfont") }
31      else { None }
32    }) {
33      // Perl: $token = LookupValue($type . 'font_' . $fam->valueOf)
34      let fam = gullet::read_number()?.value_of();
35      let key = s!("{font_type}_{fam}");
36      match state::lookup_value(&key) {
37        Some(Stored::Token(t)) => t,
38        _ => token,
39      }
40    } else if token.with_str(|ts| ts == "\\font") {
41      // Perl: $token = LookupValue('current_FontDef') || T_CS('\lx@default@font')
42      match state::lookup_value("current_FontDef") {
43        Some(Stored::Token(t)) => t,
44        _ => T_CS!("\\lx@default@font"),
45      }
46    } else {
47      token
48    }
49  });

51  DefPrimitive!("\\font Token SkipMatch:= SkipSpaces TeXFileName",
52  sub[(cs, name_arg)] {
53    let name = name_arg.to_string();
54    // Read optional "at <dimen>" or "scaled <number>"
55    let at_pt = gullet::read_keyword(&["at"])?
56      .map(|_| gullet::read_dimension().unwrap().pt_value(None));
57    let scaled = gullet::read_keyword(&["scaled"])?
58      .map(|_| gullet::read_number().unwrap().value_of() as f64 / 1000.0);
59    let props_opt = if let Some(mut props) = font::decode_fontname(&name, at_pt, scaled) {
60      props.name = Some(Cow::Owned(name.clone()));
61      Some(props)
62    } else { // Failed?
63      let message = s!("Unrecognized font name {:?} Font switch macro {:?}
64      will have no effect", name, cs.stringify());
65      Info!("unexpected", name, message);
66      None
67    };
68    gullet::skip_spaces()?;
69    let cs_str = cs.to_string();
70    if let Some(ref props) = props_opt {
71      AssignValue!(&s!("fontinfo_{cs_str}"), props.clone());
72    }
73    // Perl: $key = 'fontinfo_' . $name; $key .= " at " . ToString($at) if $at;
74    // Shared font key: fonts with same name+size share hyphenchar/skewchar
75    let at_str_opt = if let Some(at_val) = at_pt {
76      Some(s!("{at_val:.1}pt"))
77    } else if let Some(_sc) = scaled {
78      props_opt.as_ref().and_then(|p| p.size).map(|sz| s!("{sz:.1}pt"))
79    } else {
80      None
81    };
82    let shared_key = if let Some(ref at_str) = at_str_opt {
83      s!("fontinfo_{name} at {at_str}")
84    } else {
85      s!("fontinfo_{name}")
86    };
87    // Store CS → shared key mapping
88    state::assign_value(
89      &s!("font_shared_key_{cs_str}"),
90      Stored::String(arena::pin(&shared_key)),
91      None,
92    );
93    // Store explicit "at" value for \meaning (Perl: $$fontinfo{at} = ToString($at))
94    if let Some(ref at_str) = at_str_opt {
95      state::assign_value(
96        &s!("fontinfo_at_{cs_str}"),
97        Stored::String(arena::pin(at_str)),
98        None,
99      );
100    }
101    // Perl: only initialize hyphenchar/skewchar if this font key hasn't been seen before
102    // (shared fontinfo means second \font with same name+size reuses existing values)
103    let hc_key = s!("hyphenchar_{shared_key}");
104    if !state::has_value(&hc_key) {
105      let default_hyphen = lookup_int("\\defaulthyphenchar");
106      state::assign_value(
107        &hc_key,
108        Stored::Number(Number::new(default_hyphen as i64)),
109        Some(Scope::Global),
110      );
111      let default_skew = lookup_int("\\defaultskewchar");
112      state::assign_value(
113        &s!("skewchar_{shared_key}"),
114        Stored::Number(Number::new(default_skew as i64)),
115        Some(Scope::Global),
116      );
117    }
118    // Perl: installDefinition(FontDef->new($cs, $key))
119    // When the font switch CS is invoked, set current_FontDef so \fontname\font works
120    let cs_for_fontdef = cs;
121    DefPrimitive!(cs, None, None, font => props_opt,
122      before_digest => sub {
123        AssignValue!("current_FontDef", cs_for_fontdef, None);
124      }
125    );
126  });

128  // Perl: DefMacro('\fontname FontDef', sub { Explode($fontinfo && $$fontinfo{name}
129  //       || "fontname not available") })
130  // Perl's FontDef param type: for \font, looks up current_FontDef, falls back to \lx@default@font
131  DefMacro!("\\fontname FontToken", sub[args] {
132    let token = args.into_iter().next().unwrap().expected_token();
133    let cs_str = token.to_string();
134    // Determine which font CS to look up
135    let lookup_cs = if cs_str == "\\font" {
136      // Current font — look up current_FontDef, fallback to \lx@default@font
137      match state::lookup_value("current_FontDef") {
138        Some(Stored::Token(t)) => t.to_string(),
139        _ => s!("\\lx@default@font"),
140      }
141    } else {
142      cs_str
143    };
144    let key = s!("fontinfo_{}", lookup_cs);
145    let name = state::lookup_value(&key).and_then(|stored| {
146      if let Stored::Font(f) = stored {
147        f.name.as_ref().map(|n| n.to_string())
148      } else {
149        None
150      }
151    });
152    Tokens::new(Explode!(name.unwrap_or_else(|| s!("fontname not available"))))
153  });
