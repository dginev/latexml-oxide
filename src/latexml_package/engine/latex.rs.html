24
25  // Perl: LaTeX.pool.ltxml — LoadPool('TeX'); LoadFormat('latex');
26  LoadPool!("TeX");
27
28  // Load order — always the same, whether the dump is present or not:
29  //   bootstrap → _base → dump → _constructs
30  //
31  // `_base` always runs BEFORE the dump so its closure-backed defs
32  // (which can't be serialized) are installed. The dump then adds
33  // serializable state (~25k entries of post-raw-load kernel state)
34  // via add-only policy — CSes already defined by `_base` are not
35  // overwritten. `_constructs` runs last and defines closure-backed
36  // CSes that must be available regardless of dump state.
37  //
38  // This differs from Perl's strictly-mutually-exclusive `LoadFormat`
39  // (bootstrap+dump+constructs XOR bootstrap+base+constructs) because
40  // in our Rust port `_base` runs in ~3-5 ms of compiled code; there
41  // is no meaningful speed win from skipping it, and keeping it loaded
42  // means closure-backed defs don't vanish.
43  InnerPool!(latex_bootstrap);
44
45  // SYNC_STATUS D0 (d.1): stage a snapshot right after bootstrap.
46  // ini_tex::dump_format reads it so its diff captures "bootstrap →
47  // fully-initialized kernel", matching Perl's DumpFile semantics.
48  // Has no effect at normal runtime (just stored in a thread-local).
49  latexml_core::state::stage_snapshot("bootstrap");
50
51  InnerPool!(latex_base);
52
53  // Dump load — add-only: CSes already defined by _base skipped.
54  // Honours `LATEXML_NODUMP=1` (Perl-parity) to disable the dump path.
55  if let Err(e) = crate::engine::latex_dump::load_definitions() {
56    log::warn!("latex_dump: {}", e);
57  }
58
59  InnerPool!(latex_constructs);
60});
