1use crate::prelude::*;
2//**********************************************************************
3// C.12 Line and Page Breaking
4//**********************************************************************
5LoadDefinitions!({
6  //======================================================================
7  // C.12.1 Line Breaking
8  //======================================================================
9  DefPrimitive!("\\linebreak[]");
10  DefPrimitive!("\\nolinebreak[]");
11  DefPrimitive!("\\-"); // We don't do hyphenation.
12  // \hyphenation in TeX.pool
13  DefPrimitive!("\\sloppy");
14  DefPrimitive!("\\fussy");
15  // sloppypar can be used as an environment, or by itself.
16  DefMacro!("\\sloppypar", "\\par\\sloppy");
17  DefMacro!("\\endsloppypar", "\\par");
18  DefMacro!("\\nobreakdashes", "-");
19  DefMacro!("\\showhyphens{}", "#1");
20  //======================================================================
21  // C.12.2 Page Breaking
22  //======================================================================
23  DefMacro!("\\pagebreak[Default:4]", sub[(arg_opt)] {
24      let arg : u32 = if let Some(arg_t) = arg_opt {
25        arg_t.to_string().parse::<u32>().unwrap_or(0)
26      } else { 0 };
27      if arg <= 2 {
28        Ok(Tokens!()) }
29      else {
30        Ok(Invocation!(T_CS!("\\vadjust"), vec![T_CS!("\\clearpage")]))
31      }
32  });
33  DefPrimitive!("\\nopagebreak[]");
34  DefPrimitive!("\\columnbreak"); // latex? or multicol?
35  DefPrimitive!("\\enlargethispage OptionalMatch:* {}");
36
37  DefMacro!("\\clearpage", "\\lx@newpage");
38  DefMacro!("\\cleardoublepage", "\\lx@newpage");
39  DefPrimitive!("\\samepage");
40});
