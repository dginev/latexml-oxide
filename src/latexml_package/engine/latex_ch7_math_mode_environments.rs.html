20fn before_equation() -> Result<()> {
21  let mut has_preset = false;
22  let mut is_numbered = false;
23  maybe_peek_label()?;
24  let ctr = with_value_mut("EQUATION_NUMBERING", |val_opt| {
25    if let Some(Stored::HashStored(ref mut numbering)) = val_opt {
26      numbering.insert("in_equation", true.into());
27      is_numbered = matches!(numbering.get("numbered"), Some(&Stored::Bool(true)));
28      has_preset = numbering.contains_key("preset");
29      match numbering.get("counter") {
30        Some(Stored::String(v)) => arena::to_string(*v),
31        Some(other) => panic!("eq counter should be stored as string, was instead: {other:?}"),
32        _ => String::from("equation"),
33      }
34    } else {
35      String::from("equation")
36    }
37  });
38  if has_preset {
39    let mut tags = if is_numbered {
40      ref_step_counter(&ctr, false)?
41    } else {
42      ref_step_id(&ctr)?
43    };
44    tags.insert("preset", true.into());
45    state::assign_value("EQUATIONROW_TAGS", tags, Some(Scope::Global));
46  } else {
47    state::assign_value(
48      "EQUATIONROW_TAGS",
49      Stored::HashStored(SymHashMap::default()),
50      Some(Scope::Global),
51    );
52  }
53  state::let_i(
54    &T_CS!("\\lx@end@display@math"),
55    &T_CS!("\\lx@eDM@in@equation"),
56    None,
57  );
58  state::let_i(
59    &T_CS!("\\lx@begin@display@math"),
60    &T_CS!("\\lx@bDM@in@equation"),
61    None,
62  );
63  Ok(())
64}

66fn after_equation(whatsit: Option<&mut Whatsit>) -> Result<()> {
67  // Phase 1: Gather all needed data from state (immutable borrows only)
68  enum EqAction { Retract, Postset, TagsUpdate, None }
69  let mut action = EqAction::None;
70  let mut is_aligned = false;
71  let mut is_numbered_for_postset = false;
72  let mut ctr = String::from("equation");
73
74  with_value("EQUATION_NUMBERING", |eq_num_opt| {
75    if let Some(Stored::HashStored(ref numbering)) = eq_num_opt {
76      is_aligned = matches!(numbering.get("aligned"), Some(&Stored::Bool(true)));
77      is_numbered_for_postset =
78        matches!(numbering.get("numbered"), Some(&Stored::Bool(true)));
79      with_value("EQUATIONROW_TAGS", |tags_opt| {
80        if let Some(Stored::HashStored(ref tags)) = tags_opt {
81          ctr = tags
82            .get("counter")
83            .map_or_else(|| numbering.get("counter"), Some)
84            .map(ToString::to_string)
85            .unwrap_or_else(|| String::from("equation"));
86
87          if !matches!(tags.get("noretract"), Some(&Stored::Bool(true)))
88            && (matches!(tags.get("retract"), Some(&Stored::Bool(true)))
89              || (matches!(numbering.get("retract"), Some(&Stored::Bool(true)))
90                && matches!(numbering.get("preset"), Some(&Stored::Bool(true)))
91                && matches!(tags.get("preset"), Some(&Stored::Bool(true)))))
92          {
93            action = EqAction::Retract;
94          } else if matches!(numbering.get("postset"), Some(&Stored::Bool(true)))
95            && !matches!(tags.get("reset"), Some(&Stored::Bool(true)))
96          {
97            action = EqAction::Postset;
98          } else if !matches!(tags.get("reset"), Some(&Stored::Bool(true)))
99            && matches!(numbering.get("numbered"), Some(&Stored::Bool(true)))
100          {
101            action = EqAction::TagsUpdate;
102          }
103        }
104      });
105    }
106  });
107
108  // Phase 2: Act on gathered data (borrows released, safe to mutate state)
109  match action {
110    EqAction::Retract => {
111      retract_equation();
112    },
113    EqAction::Postset => {
114      let new_tags = if is_numbered_for_postset {
115        ref_step_counter(&ctr, false)?
116      } else {
117        ref_step_id(&ctr)?
118      };
119      state::assign_value(
120        "EQUATIONROW_TAGS",
121        Stored::HashStored(new_tags),
122        Some(Scope::Global),
123      );
124    },
125    EqAction::TagsUpdate => {
126      let invoked_tags = build_invocation(
127        T_CS!("\\lx@make@tags"),
128        vec![Some(Tokens::new(Explode!(ctr)))],
129      )?;
130      let stored_tags_update =
131        Stored::Digested(stomach::digest(invoked_tags)?);
132      with_value_mut("EQUATIONROW_TAGS", |tags_opt| {
133        if let Some(Stored::HashStored(ref mut tags)) = tags_opt {
134          tags.insert("tags", stored_tags_update);
135        }
136      });
137    },
138    EqAction::None => {},
139  }
140
141  // Phase 3: Reset in_equation flag
142  with_value_mut("EQUATION_NUMBERING", |eq_num_opt| {
143    if let Some(Stored::HashStored(ref mut numbering)) = eq_num_opt {
144      numbering.insert("in_equation", Stored::Bool(false));
145    }
146  });
147
148  // Phase 4: Install tags in $whatsit or current Row, as appropriate.
149  #[allow(clippy::manual_unwrap_or_default)]
150  let props = match state::remove_value("EQUATIONROW_TAGS") {
151    Some(Stored::HashStored(hs)) => hs,
152    _ => SymHashMap::default(),
153  };
154  if is_aligned {
155    // Perl: propagate id/tags to current alignment row properties.
156    // In Perl, these get stored as $$row{id}, $$row{tags} and later passed to
157    // the openRow hook. We store them in EQUATIONROW_PROPS for the open_row hook.
158    state::assign_value(
159      "EQUATIONROW_PROPS",
160      Stored::HashStored(props),
161      Some(Scope::Global),
162    );
163  } else if let Some(w) = whatsit {
164    w.set_properties(props);
165  }
166  Ok(())
167}

169/// Perl: latex_constructs.pool.ltxml lines 2025-2035
170fn retract_equation() {
171  // Phase 1: Gather data (immutable borrows)
172  let (ctr, is_preset, is_numbered) =
173    with_value("EQUATION_NUMBERING", |eq_num_opt| {
174      let numbering = match eq_num_opt {
175        Some(Stored::HashStored(n)) => n,
176        _ => return (String::from("equation"), false, false),
177      };
178      let is_numbered =
179        matches!(numbering.get("numbered"), Some(&Stored::Bool(true)));
180      with_value("EQUATIONROW_TAGS", |tags_opt| {
181        let tags = match tags_opt {
182          Some(Stored::HashStored(t)) => t,
183          _ => return (String::from("equation"), false, is_numbered),
184        };
185        let ctr = tags
186          .get("counter")
187          .map_or_else(|| numbering.get("counter"), Some)
188          .map(ToString::to_string)
189          .unwrap_or_else(|| String::from("equation"));
190        let is_preset =
191          matches!(tags.get("preset"), Some(&Stored::Bool(true)));
192        (ctr, is_preset, is_numbered)
193      })
194    });
195
196  // Phase 2: Mutate state (borrows released)
197  if is_preset {
198    // counter (or ID counter) was stepped, so decrement it.
199    let counter_name =
200      if is_numbered { ctr.clone() } else { s!("UN{}", ctr) };
201    let _ = add_to_counter(&counter_name, Number::new(-1));
202  }
203  if let Ok(mut new_tags) = ref_step_id(&ctr) {
204    new_tags.insert("reset", true.into());
205    state::assign_value(
206      "EQUATIONROW_TAGS",
207      Stored::HashStored(new_tags),
208      Some(Scope::Global),
209    );
210  }
211}
