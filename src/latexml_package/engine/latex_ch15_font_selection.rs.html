101  DefPrimitive!("\\selectfont", {
102    let family = Expand!(T_CS!("\\f@family")).to_string();
103    let series = Expand!(T_CS!("\\f@series")).to_string();
104    let shape = Expand!(T_CS!("\\f@shape")).to_string();
105    if let Some(sh) = font::lookup_font_family(&family) {
106      MergeFont!(sh.clone());
107    } else {
108      let message = s!("Unrecognized font family {:?}.", family);
109      Info!("unexpected", family, message);
110    }
111    if let Some(sh) = font::lookup_font_series(&series) {
112      MergeFont!(sh.clone());
113    } else {
114      let message = s!("Unrecognized font series {:?}.", series);
115      Info!("unexpected", series, message);
116    }
117    if let Some(sh) = font::lookup_font_shape(&shape) {
118      MergeFont!(sh.clone());
119    } else {
120      let message = s!("Unrecognized font shape {:?}.", shape);
121      Info!("unexpected", shape, message);
122    }
123    Ok(Vec::new())
124  });
