6pub fn reenter_text_mode(vertical_mode: bool) {
7  let mode_key = if vertical_mode {
8    "VTEXT_MODE_BINDINGS"
9  } else {
10    "HTEXT_MODE_BINDINGS"
11  };
12  let text_key = "TEXT_MODE_BINDINGS";
13  let mode_bindings = checkout_value(mode_key);
14  let text_bindings = checkout_value(text_key);
15  let mut bindings: VecDeque<&Stored> = match mode_bindings {
16    Some(Stored::VecDequeStored(ref vdq)) => vdq.iter().collect::<VecDeque<&Stored>>(),
17    _ => VecDeque::new(),
18  };
19  if let Some(Stored::VecDequeStored(ref vdq)) = text_bindings {
20    bindings.extend(vdq.iter().collect::<Vec<_>>());
21  }
22  for binding in bindings {
23    if let Stored::Tokens(tks) = binding {
24      let vec = tks.unlist_ref();
25      state::let_i(&vec[0], &vec[1], None);
26    }
27  }
28  if let Some(value) = mode_bindings {
29    checkin_value(mode_key, value);
30  }
31  if let Some(value) = text_bindings {
32    checkin_value(text_key, value);
33  }
34}

69pub fn parse_def_parameters(cs: &Token, params_in: Tokens) -> Result<Option<Parameters>> {
70  let mut tokens: VecDeque<Token> = VecDeque::from(params_in.pack_parameters()?.unlist());
71  // Now, recognize parameters and delimiters.
72  let mut params = Vec::new();
73  let mut n = 0;
74  while let Some(mut t) = tokens.pop_front() {
75    let cc = t.get_catcode();
76    if cc == Catcode::PARAM || cc == Catcode::ARG {
77      if cc == Catcode::PARAM {
78        if tokens.is_empty() {
79          // Special case: lone # NOT following a numbered parameter
80          // Note that we require a { to appear next, but do NOT read it!
81          params.push(Parameter::new(
82            Cow::Borrowed("RequireBrace"),
83            Cow::Borrowed("RequireBrace"),
84            None,
85          )?);
86          break;
87        } else {
88          n += 1;
89          if let Some(t_next) = tokens.pop_front() {
90            t = t_next;
91          } else {
92            todo!(); // hm, this is a bit of a pain to port without making t into an
93            // Option<Token>...
94          }
95        }
96      } else {
97        // CC_ARG case, keep looking at this token
98        n += 1;
99      }
100      if n > 0 {
101        let t_num = t.with_str(|ts| ts.parse::<i8>()).unwrap_or(-1);
102        if t_num != n {
103          fatal!(
104            ParamSpec,
105            Expected,
106            s!(
107              "Parameters for {:?} not in order. Got {:?}, expected {:?}. in {:?}",
108              cs,
109              t,
110              n,
111              params
112            )
113          );
114        }
115      }
116      // Check for delimiting text following the parameter #n
117      let mut delim = Vec::new();
118      let mut pc = Catcode::MARKER; // throwaway initial val
119      while !tokens.is_empty() {
120        let inner_cc = tokens.front().unwrap().get_catcode();
121        if inner_cc == Catcode::PARAM || inner_cc == Catcode::ARG {
122          break;
123        }
124        let d = tokens.pop_front().unwrap();
125        if !(pc == Catcode::SPACE && inner_cc == Catcode::SPACE) {
126          // BUT collapse whitespace!
127          delim.push(d);
128        }
129        pc = inner_cc;
130      }
131      // Found text that marks the end of the parameter
132      if !delim.is_empty() {
133        let extra = Tokens::new(delim);
134        params.push(
135          Parameter {
136            name: arena::pin_static("Until"),
137            spec: arena::pin(format!("Until:{extra}")),
138            extra: vec![extra],
139            ..Parameter::default()
140          }
141          .init()?,
142        );
143      } else if tokens.len() == 1 && tokens.front().unwrap().get_catcode() == Catcode::PARAM {
144        // Special case: trailing sole # => delimited by next opening brace.
145        tokens.pop_front();
146        params.push(Parameter::new("UntilBrace", "UntilBrace", None)?);
147      } else {
148        // Nothing? Just a plain parameter.
149        params.push(Parameter::new("Plain", "{}", None)?);
150      }
151    } else {
152      // Initial delimiting text is required.
153      let mut lit: Vec<Token> = vec![t];
154      while !tokens.is_empty() {
155        let lit_cc = tokens.front().unwrap().get_catcode();
156        if lit_cc == Catcode::PARAM || lit_cc == Catcode::ARG {
157          break;
158        }
159        lit.push(tokens.pop_front().unwrap());
160      }
161      let expected = Tokens::new(lit);
162      params.push(
163        Parameter {
164          name: arena::pin_static("Match"),
165          spec: arena::pin(s!("Match:{expected}")),
166          extra: vec![expected],
167          novalue: true,
168          ..Parameter::default()
169        }
170        .init()?,
171      );
172    }
173  }
174  // return (@params ? LaTeXML::Core::Parameters->new(@params) : undef);
175  if params.is_empty() {
176    Ok(None)
177  } else {
178    Ok(Some(Parameters::new(params)))
179  }
180}

182pub fn do_def(globally: bool, cs: Token, params: Tokens, body: Tokens) -> Result<()> {
183  let paramlist = parse_def_parameters(&cs, params)?;
184  let scope = if globally { Some(Scope::Global) } else { None };
185  state::install_definition(
186    Expandable::new(
187      cs,
188      paramlist,
189      Some(ExpansionBody::Tokens(body)),
190      Some(ExpandableOptions {
191        nopack_parameters: true,
192        ..ExpandableOptions::default()
193      }),
194    )?,
195    scope,
196  );
197  after_assignment();
198  Ok(())
199}

222/// Properties for a decoded math character, mirroring Perl's decodeMathChar return
223#[derive(Debug, Clone, Default)]
224pub struct MathCharProps {
225  pub role: Option<String>,
226  pub glyph: Option<char>,
227  pub meaning: Option<String>,
228  pub name: Option<String>,
229  pub stretchy: Option<String>,
230  pub need_scriptpos: bool,
231  pub need_mathstyle: bool,
232  pub scriptpos: Option<String>,
233  pub mathstyle: Option<String>,
234}
235
236impl MathCharProps {
237  /// Convert need_scriptpos/need_mathstyle flags to actual values based on display mode
238  pub fn resolve_style_props(&mut self) {
239    let in_display = state::lookup_string("IN_MATH_DISPLAY") == "true"
240      || lookup_font()
241        .map(|f| f.get_mathstyle().map(|s| s.as_ref()) == Some("display"))
242        .unwrap_or(false);
243    if self.need_scriptpos {
244      self.scriptpos = Some(if in_display { "mid" } else { "post" }.to_string());
245    }
246    if self.need_mathstyle {
247      self.mathstyle = Some(if in_display { "display" } else { "text" }.to_string());
248    }
249  }
250
251  /// Insert all properties into a HashMap for Tbox construction
252  pub fn into_props_map(self) -> HashMap<&'static str, Stored> {
253    let mut props = HashMap::default();
254    if let Some(role) = self.role {
255      props.insert("role", Stored::String(arena::pin(role)));
256    }
257    if let Some(meaning) = self.meaning {
258      props.insert("meaning", Stored::String(arena::pin(meaning)));
259    }
260    if let Some(name) = self.name {
261      props.insert("name", Stored::String(arena::pin(name)));
262    }
263    if let Some(stretchy) = self.stretchy {
264      props.insert("stretchy", Stored::String(arena::pin(stretchy)));
265    }
266    if let Some(scriptpos) = self.scriptpos {
267      props.insert("scriptpos", Stored::String(arena::pin(scriptpos)));
268    }
269    if let Some(mathstyle) = self.mathstyle {
270      props.insert("mathstyle", Stored::String(arena::pin(mathstyle)));
271    }
272    props
273  }
274}

493// Is this "fontinfo" stuff sufficient to maintain a math font "family" ??
494// What we're really after is a connection to a font encoding mapping.
495pub fn decode_math_char(mut n: u16) -> Result<MathCharProps> {
496  let class: u16 = n / (16 * 256);
497  n %= 16 * 256;
498  let fam: u16 = n / 256;
499  n %= 256;
500  let font = lookup_value(&s!("textfont_{fam}")).unwrap_or_else(|| {
501    lookup_value(&s!("scriptfont_{fam}"))
502      .unwrap_or_else(|| lookup_value(&s!("scriptscriptfont_{fam}")).unwrap_or(Stored::Bool(false)))
503  });
504  let c = n as u8 as char;
505  let class_role = MATH_CLASS_ROLE[class as usize];
506
507  // Decode the glyph from the font encoding
508  let glyph = with_font_info(&T_CS!(font.to_string()), |fontinfo| {
509    let cinfo = if let Some(Stored::Font(ref info)) = fontinfo? {
510      if let Some(ref data) = info.encoding {
511        font::decode(n as u8, Some(data.to_string()), false)
512      } else {
513        Some(c)
514      }
515    } else {
516      None
517    };
518    Ok::<Option<char>, latexml_core::Error>(cinfo)
519  })?;
520
521  // Look up unicode math properties for the decoded glyph
522  let glyph_char = glyph.unwrap_or(c);
523  let mut props = unicode_math_properties(glyph_char).unwrap_or_default();
524  props.glyph = glyph;
525
526  // Apply class-based role if no role from unicode properties, or class gives a specific role
527  if !class_role.is_empty() {
528    // Class-based role takes precedence when it's specific (not empty)
529    // But unicode_math_properties may have a more specific role (e.g. SUMOP vs BIGOP)
530    if props.role.is_none() {
531      props.role = Some(class_role.to_string());
532    } else if let Some(ref unicode_role) = props.role {
533      // If class says BIGOP but unicode says SUMOP, keep SUMOP (more specific)
534      // If class says something else, and unicode has a role, keep unicode's role
535      if unicode_role == class_role || class_role == "BIGOP" {
536        // keep the unicode role
537      } else {
538        props.role = Some(class_role.to_string());
539      }
540    }
541  } else if props.role.is_none() {
542    // No class role and no unicode role — try math_token_attributes
543    with_value(&s!("math_token_attributes_{}", c), |charinfo| {
544      if let Some(Stored::HashString(ref info)) = charinfo {
545        let inner_role = &info["role"];
546        if !inner_role.is_empty() {
547          props.role = Some(inner_role.to_string());
548        }
549      }
550    });
551  }
552
553  // Resolve need_scriptpos/need_mathstyle to actual values
554  props.resolve_style_props();
555
556  Ok(props)
557}

636/// This attempts to be a generalize vbox construction;
637///
638/// The idea is to receeive block-like material, possibly wrapped in appropriate
639/// container which gets attributes.
640///
641/// The contents are constructed in an ltx:_CaptureBlock_ element,
642/// designed to accept all reasonable block material from several levels,
643/// and then determine which container element is most apprpriate for both the conent & context
644/// from block, logical-block or sectional-block, or the inline- variants.
645pub fn insert_block(
646  document: &mut Document,
647  contents: &Digested,
648  block_attr: HashMap<String, String>,
649) -> Result<Vec<Node>> {
650  // Create something like:
651  // "<ltx:inline-block vattach='$vattach' height='#height'>#2</ltx:inline-block>"
652  let context_opt = document.get_element(); // Where we originally start inserting.
653  if context_opt.is_none() {
654    // edge case: if we start the doc with a block, the context is empty
655    document.absorb(contents, None)?;
656    return Ok(Vec::new());
657  }
658  let mut context = context_opt.unwrap();
659  let mut context_tag = document::get_node_qname(&context);
660  // svg is slightly tricky
661  let (is_svg, is_xmath, is_xmtext) = arena::with(context_tag, |tag| {
662    (
663      tag.starts_with("svg:"),
664      tag.starts_with("ltx:XM"),
665      tag == "ltx:XMText",
666    )
667  });
668  let ignorable_attr = is_svg || block_attr.is_empty(); // if we do not REQUIRE the attributes
669  if is_xmath && !is_xmtext {
670    // but math always needs this
671    context = document.open_element("ltx:XMText", None, None)?;
672    context_tag = document::get_node_qname(&context);
673  }
674  let is_inline = is_svg || document::can_contain(&context, "#PCDATA");
675  let mut container_attr = block_attr.clone();
676  container_attr.insert("_vertical_mode_".to_string(), "true".to_string());
677  let container = document.open_element("ltx:_CaptureBlock_", Some(container_attr), None)?;
678  document.absorb(contents, None)?;
679
680  let mut nodes = content_nodes(&container);
681  let node_tags = nodes
682    .iter()
683    .map(document::get_node_qname)
684    .collect::<Vec<_>>();
685  let nnodes = nodes.len();
686  document.close_to_node(&container, true)?;
687  document.close_node(&container)?;
688  document.close_to_node(&context, true)?;
689
690  if nnodes < 1 {
691    // Insertion came up empty?
692    document.remove_node(container); // then remove the new block entirely
693    return Ok(nodes);
694  } else if ignorable_attr
695    && node_tags
696      .iter()
697      .all(|tag| document::can_contain_qsym(context_tag, *tag))
698  {
699    // No attributes, contents allowed in context?
700    document.unwrap_nodes(container)?; // No container needed, at all.
701    return Ok(nodes);
702  } else if nnodes == 1 {
703    if document::can_contain_qsym(context_tag, node_tags[0])
704      && (ignorable_attr
705        || block_attr
706          .keys()
707          .all(|key| document::sym_can_have_attribute(node_tags[0], arena::pin(key))))
708    {
709      // IF: Single node, allowed in context & accepts attributes
710      // THEN: Add attributes and unwrap the single node
711      for (k, v) in block_attr.iter() {
712        document.set_attribute(&mut nodes[0], k, v)?;
713      }
714      document.unwrap_nodes(container)?;
715      return Ok(nodes);
716    } else if let Some(newcontainer) = document::sym_can_contain_somehow(context_tag, node_tags[0])
717    {
718      if ignorable_attr
719        || block_attr.keys().all(|key| {
720          newcontainer
721            .map(|nc| document::sym_can_have_attribute(nc, arena::pin(key)))
722            .unwrap_or(false)
723        })
724      {
725        if let Some(nc) = newcontainer {
726          // rename the capture to that container
727          document.rename_node_qsym(container, nc, true)?;
728          return Ok(nodes);
729        }
730      }
731    }
732  }
733  // This jagged conditional is a "code smell", due to the difficulty of refactoring
734  // the in-conditional-assignments from Perl.
735
736  // Otherwise, rename the capture
737  // MAY need foreignObject wrapper
738  if is_svg
739    && node_tags
740      .iter()
741      .any(|tag| arena::with(*tag, |tag_str| tag_str.starts_with("ltx:")))
742  {
743    context = document
744      .wrap_nodes("svg:foreignObject", vec![container.clone()])?
745      .expect("foreign object wrap should always succeed in SVG");
746    context_tag = document::get_node_qname(&context);
747  }
748  let candidates = if is_inline {
749    [
750      "ltx:inline-block",
751      "ltx:inline-logical-block",
752      "ltx:inline-sectional-block",
753    ]
754    .map(arena::pin_static)
755    .to_vec()
756  } else {
757    [
758      "ltx:block",
759      "ltx:logical-block",
760      "ltx:sectional-block",
761      "ltx:figure",
762    ]
763    .map(arena::pin_static)
764    .to_vec()
765  };
766  let filtered_candidates = candidates
767    .into_iter()
768    .filter(|candidate| {
769      node_tags
770        .iter()
771        .all(|tag| document::sym_can_contain_somehow(*candidate, *tag).is_some())
772    })
773    .collect::<Vec<_>>();
774  // and are allowed in the context
775  let allowed_candidates = filtered_candidates
776    .iter()
777    .filter(|candidate| document::can_contain_qsym(context_tag, **candidate))
778    .copied()
779    .collect::<Vec<_>>();
780  if let Some(final_tag) = allowed_candidates
781    .first()
782    .map_or(filtered_candidates.first(), Some)
783  {
784    // Rename the capture to the correct container
785    // TODO: There is an arena code smell here. The `Model` interface needs to become lock-free
786    // where Symbol tickets and &str are equally intuitive to use without runtime panics from
787    // arena mutability exceptions.
788    document.rename_node(container, &arena::to_string(*final_tag), true)?;
789  } else {
790    // we didn't know what to do?
791    let message = arena::with(context_tag, |ctxt_str| {
792      s!(
793        "Did not find a block-like candidate in {} (with attributes ({})",
794        ctxt_str,
795        block_attr
796          .iter()
797          .map(|(k, v)| s!("{k}={v}"))
798          .collect::<Vec<_>>()
799          .join(";")
800      )
801    });
802    Warn!("malformed", "_CaptureBlock_", message);
803    document.rename_node(container, "ltx:block", true)?;
804  }
805  Ok(nodes)
806}

808pub fn cleanup_math(document: &mut Document, mathnode: Node) -> Result<()> {
809  // Cleanup ltx:Math elements; particularly if they aren't "really" math.
810  // But record the oddity with class=ltx_markedasmath
811
812  // If the Math ONLY contains XMath/XMText, it apparently isn't math at all!?!
813  if document
814    .findnodes("ltx:XMath/ltx:*[local-name() != 'XMText']", Some(&mathnode))
815    .is_empty()
816  {
817    // So unwrap down to the contents of the XMText's.
818    let xmtexts = mathnode.get_child_nodes().into_iter().flat_map(|child| {
819      child
820        .get_child_nodes()
821        .into_iter()
822        .flat_map(|grandhcild| grandhcild.get_child_nodes())
823    });
824    let mut texts = vec![];
825    for mut text in xmtexts {
826      text = if text.get_type() == Some(NodeType::ElementNode) {
827        // Make sure we've got an element
828        text
829      } else {
830        document.wrap_nodes("ltx:text", vec![text])?.unwrap()
831      };
832      // Now record that it originally was marked as math
833      document.add_class(&mut text, "ltx_markedasmath")?;
834      texts.push(text)
835    }
836    document.replace_node(mathnode.clone(), texts)?; // and replace the whole Math with the pieces
837  } else {
838    // Cleanup any remaining XMTexts
839    cleanup_xmtext_outer(document, &mathnode)?;
840  }
841  Ok(())
842}

963// A random collection of utility functions.
964// [maybe need to do some reorganization?]
965// Since this is used for textual tokens, typically to split author lists,
966// we don't split within braces or math
967pub fn split_tokens(tokens: Tokens, delims: Vec<Token>) -> Vec<Tokens> {
968  let mut items = Vec::new();
969  let mut toks = Vec::new();
970  if !tokens.is_empty() {
971    let tokens = tokens.unlist();
972    let mut tokens_iter = tokens.into_iter();
973    while let Some(t) = tokens_iter.next() {
974      if delims.iter().any(|d| d == &t) {
975        items.push(Tokens::new(std::mem::take(&mut toks)));
976      } else if t == T_BEGIN!() {
977        toks.push(t);
978        let mut level = 1;
979        for t in tokens_iter.by_ref() {
980          match t.get_catcode() {
981            Catcode::BEGIN => level += 1,
982            Catcode::END => level -= 1,
983            _ => {},
984          }
985          toks.push(t);
986          if level < 1 {
987            // done if balanced.
988            break;
989          }
990        }
991      } else if t == T_MATH!() {
992        toks.push(t);
993        for t in tokens_iter.by_ref() {
994          let is_math = t.get_catcode() == Catcode::MATH;
995          toks.push(t);
996          if is_math {
997            break;
998          }
999        }
1000      } else {
1001        toks.push(t);
1002      }
1003    }
1004    // last author is in toks, add to items
1005    items.push(Tokens::new(toks));
1006  }
1007  items
1008}

1022/// Converts tokens to a string in the fashion of \message and others
1023///
1024/// doubles #, converts to string; optionally adds spaces after control sequences
1025/// in the spirit of the B Book, "show_token_list" routine, in 292.
1026/// [This could be a $tokens->unpackParameters, but for the curious space treatment]
1027pub fn writable_tokens(tokens: &Tokens) -> String {
1028  let mut wv = Vec::new();
1029  for t in tokens.unlist_ref().iter() {
1030    match t.code {
1031      Catcode::CS => {
1032        wv.push(*t);
1033        // Perl: add space after CS unless it's a single non-alpha char CS (like \{, \\, \#)
1034        // i.e. skip space only for "\X" where X is exactly one non-[a-zA-Z] character
1035        let is_single_nonalpha_cs = arena::with(t.text, |s| {
1036          s.starts_with('\\') && {
1037            let rest = &s[1..];
1038            rest.chars().count() == 1 && !rest.chars().next().unwrap_or(' ').is_ascii_alphabetic()
1039          }
1040        });
1041        if !is_single_nonalpha_cs {
1042          wv.push(T_SPACE!());
1043        }
1044      },
1045      Catcode::SPACE => {
1046        wv.push(T_SPACE!());
1047      },
1048      Catcode::PARAM => {
1049        wv.push(*t);
1050        wv.push(*t);
1051      },
1052      Catcode::ARG => {
1053        // B Book, 294. Reduce to param+integer
1054        wv.push(T_PARAM!());
1055        wv.push(t.as_other());
1056      },
1057      _ => {
1058        wv.push(*t);
1059      },
1060    }
1061  }
1062  Tokens::new(wv).untex()
1063}

1065/// Support for Key / Value arguments.
1066// The very basic form is
1067//   RequiredKeyVals: $keyset
1068//   OptionalKeyVals: $keyset
1069// to parse Key-Value pairs from a given keyset (see the 'keyval' package
1070// documentation for more information). These types of KeyVal
1071// parameters will return a LaTeXML::Core::KeyVals object, which can then be
1072// used to access the values of the individual items.
1073// The difference between the two forms is that RequiredKeyVals expects a set of
1074// key-value pairs wrapped in T_BEGIN T_END, where as OptionalKeyVals optionally
1075// expects a set of KeyValue pairs wrapped in T_OTHER('[') T_OTHER(']')
1076//
1077// Several extension of the keyval package exist, the most common one we support
1078// is the xkeyval package. This introduces further variations on the keyval
1079// arguments parsing, in particular it allows to read keys from more than one
1080// keyset at once. These can be specified by giving comma-seperated values in
1081// the keyset argument. By default, a key will only be set in the **first**
1082// keyset it occurs in. By using
1083//   RequiredKeyVals+: $keysets
1084//   OptionalKeyVals+: $keysets
1085// the key will be set in all keysets instead.
1086//
1087// All keys to be parsed with these arguments should be declared using
1088// DefKeyVal in LaTeXML::Package. By default, an error is thrown if an unknown
1089// key is encountered. To surpress this behaviour, and instead store all
1090// undefined keys, use
1091//   RequiredKeyVals*: $keysets
1092//   OptionalKeyVals*: $keysets
1093// instead. The '*' and '+' modifiers can be combined by using:
1094//   RequiredKeyVals*+: $keysets
1095//   OptionalKeyVals*+: $keysets
1096//
1097// Furthermore, the xkeyval package supports giving prefixes to keys,
1098//   RequiredKeyVals[*][+]: $prefix|$keysets
1099//   OptionalKeyVals[*][+]: $prefix|$keysets
1100//
1101// Finally, it is possible to specify specific keys to skip when digesting the
1102// object. This can be achieved using comma-seperated key values in
1103//   RequiredKeyVals[*][+]: $prefix|$keysets|$skip
1104//   OptionalKeyVals[*][+]: $prefix|$keysets|$skip
1105
1106// function to handle all the
1107#[derive(Default)]
1108pub struct KVSpec {
1109  pub star:    bool,
1110  pub plus:    bool,
1111  pub prefix:  Option<String>,
1112  pub keysets: Vec<String>,
1113  pub skip:    Vec<String>,
1114}
1115pub fn keyvals_aux(until: Option<Token>, spec: KVSpec) -> Result<KeyVals> {
1116  let KVSpec {
1117    mut star,
1118    plus,
1119    mut prefix,
1120    mut keysets,
1121    skip,
1122  } = spec;
1123  // support both "keysets" and "prefix|keysets"
1124  if keysets.is_empty() {
1125    if let Some(pfx) = prefix.take() {
1126      keysets = vec![pfx];
1127    }
1128
1129    // to emulate old behaviour, throw no errors
1130    // when we have a single keyset and no prefix (or no keyset at all)
1131    if keysets.is_empty() {
1132      star = true;
1133    }
1134  }
1135
1136  // create a new set of Key-Value arguments
1137  let mut keyvals = KeyVals::new(KeyvalsConfig {
1138    prefix,
1139    keysets,
1140    set_all: plus,
1141    set_internals: true,
1142    skip,
1143    skip_missing: if star {
1144      keyvals::SkipMissing::All
1145    } else {
1146      keyvals::SkipMissing::None
1147    },
1148    hook_missing: None,
1149  });
1150  // and read it from the gullet
1151  if let Some(until_token) = until {
1152    keyvals.read_from(until_token, false)?;
1153  }
1154  // we still want to make use of the hash
1155  Ok(keyvals)
1156}
