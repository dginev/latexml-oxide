134  //
135  // The predigest closure is new for latexml-oxide, as it was a single closure in Perl
136  // The key problem is that in latexml-oxide the parameter type interfaces are well-typed,
137  // so it is not possible to remain elegant while at the same time
138  // have access to the stomach AND digest.
139  // Hence, the `reader` is exclusively responsible for using the gullet to obtain tokens,
140  // while early/immediate digestion via the stomach can be achieved
141  // by using the separate `predigest` interface
142  // Importantly, predigest forces the parameter to be usable
143  // only for stomach-capable bindings,
144  // namely DefConstructor, DefPrimitive or DefEnvironment
145  predigest => sub[key] {

147      let mut keyvals = KeyVals::new(

149      let dim = gullet::read_dimension()?;
150      keyvals.set_value(&key.owned_tokens().unwrap().to_string(), dim.into(), false)?;
151      keyvals.into()
152    } else {
153      Ok(None)
154    }
155  },

191  DefConstructor!("\\hbox BoxSpecification HBoxContents", sub[document, args, props] {
192    // "<ltx:text width='#width' _noautoclose='1'>#2</ltx:text>",
193    unpack_opt_ref!(args => _spec_opt, contents_opt);
194    let contents = contents_opt.as_ref().unwrap();
195    let current_opt = document.get_element();
196
197    // What is the CORRECT (& general) way to ask whether we're in "vertical mode"??
198    //  my $vmode = $tag eq 'ltx:inline-block'; # ie, explicitly \vbox !?!?!?!
199    let is_svg  = if let Some(ref current) = current_opt {
200      document::with_node_qname(current, |qname| qname.starts_with("svg:"))
201    } else { false };
202    let vmode = if let Some(ref current) = current_opt {
203      current.has_attribute("_vertical_mode_")
204    } else { false };
205    let newtag = if is_svg { "svg:g" }
206      else if vmode { "ltx:p" } else { "ltx:text" };
207    let width : String = if let Some(Stored::Dimension(ref w)) = props.get("width") {
208      w.to_attribute()
209    } else {
210      String::new()
211    };
212    let node = document.open_element(newtag,
213      Some(string_map!("_noautoclose" => "true", "width" => width)), None)?;
214    // Note on the clone: Remember that contents is a Digested,
215    // i.e. we are cloning an Rc<> wrapper, which is relatively cheap.
216    // see the documentation on `Digested` on why we don't have a neater way of dealing with this.
217    document.absorb(contents, None)?;
218    if !is_svg {
219      while !document.get_element().unwrap().has_attribute("_beginscope") &&
220        document.maybe_close_element("svg:g")?.is_some() {}
221      document.maybe_close_element("svg:svg")?;
222      document.maybe_close_node(&node)?;
223    } else {
224      document.maybe_close_element("svg:g")?;
225    }

315    // If there is any afterAssignment tokens, move them over so BoxContents parameter will use them
316    if let Some(after_token) = state::remove_value("afterAssignment") {
317      state::assign_value("BeforeNextBox", after_token, None);
318    }
319    // Save global flag, since we're digesting to get the box content, which resets the flag!
320    // Should afterDigest be responsible for resetting flags?
321    let scope = if get_prefix("global") {
322      Some(Scope::Global)
323    } else {
324      None
325    };
326    clear_prefixes(); // before invoke, below; we've saved the only relevant one (global)
327    let mut rest = if let Some(xtoken) = gullet::read_x_token(None, false, None)? {
328        stomach::invoke_token(&xtoken)?
329    } else { Vec::new() };
330    let stuff = if !rest.is_empty() {
331      Stored::Digested(rest.remove(0))
332    } else {
333      Stored::None
334    };
335    state::assign_value(&format!("box{}", number.value_of()), stuff, scope);
336    rest

571// Risky: I think this needs to be digested as a body to work like TeX (?)
572// but parameter think's it's just parsing from gullet...
573pub fn read_box_contents(everybox_opt: Option<Tokens>) -> Result<Tokens> {
574  while let Some(t) = gullet::read_token()? {
575    if t.get_catcode() == Catcode::BEGIN {
576      break;
577    } // Skip till { or \bgroup
578  }
579  // Now, insert some extra tokens, if any, possibly from \afterassignment
580  match state::remove_value("BeforeNextBox") {
581    Some(Stored::Tokens(tokens)) => gullet::unread(tokens),
582    Some(Stored::Token(token)) => gullet::unread_one(token),
583    None | Some(Stored::None) => {},
584    Some(other) => panic!("afterAssignment should be a token, got: {}", other),
585  };
586  // AND, insert any extra tokens passed in, due to everyhbox or everyvbox
587  if let Some(everybox) = everybox_opt {
588    gullet::unread(everybox);
589  }
590  Ok(Tokens!())
591}
