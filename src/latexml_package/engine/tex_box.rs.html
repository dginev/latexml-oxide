569    let h_pt: Option<f64> = height.as_ref().and_then(|h| h.strip_suffix("pt").and_then(|s| s.parse().ok()));

571    // Perl: rwidth => $width (raw, could be undef); cwidth => $width || Dimension('0.4pt')
572    if let Some(w) = &width { whatsit.set_property("rwidth", w.clone()); }
573    if let Some(h) = &height { whatsit.set_property("rheight", h.clone()); }
574    if let Some(d) = &depth { whatsit.set_property("rdepth", d.clone()); }
575
576    if let Some(_alignment) = lookup_alignment() {
577      // Perl: set isVerticalRule only if dimensions suggest a real rule
578      let dominated_by_height = match (h_pt, w_pt) {
579        (None, None) => true,
580        (Some(h), None) => h > 20.0,
581        (Some(h), Some(w)) => h > 3.0 * w,
582        _ => false,
583      };
584      if dominated_by_height {
585        whatsit.set_property("isVerticalRule", true);
586      }
587    } else if w_pt == Some(0.0) {
588      whatsit.set_property("invisible", true);
589    }
590    // TODO: color handling
591    Ok(Vec::new())
592  });
593
594  DefConstructor!("\\hrule RuleSpecification",
595    "?#isHorizontalRule()(<ltx:rule ?#rheight(height='#rheight') ?#rdepth(depth='#rdepth')\
596       ?#rwidth(width='#rwidth') ?#color(color='#color')/>)",
597  before_construct => sub[document, whatsit] {
598    // Perl: maybeCloseElement('ltx:p') if rwidth is '100%'
599    if whatsit.get_property_string("rwidth") == "100%" {
600      document.maybe_close_element("ltx:p")?;
601    }
602  },
603  after_digest => sub [whatsit] {
604    use latexml_core::digested::DigestedData;
605    let (width, height, depth) = if let Some(d) = whatsit.get_arg(1) {
606      if let DigestedData::KeyVals(kv) = d.data() {
607        (kv.get_value("width").map(ToString::to_string),
608         kv.get_value("height").map(ToString::to_string),
609         kv.get_value("depth").map(ToString::to_string))
610      } else { (None, None, None) }
611    } else { (None, None, None) };
612
613    // Perl: $stomach->leaveHorizontal;
614    leave_horizontal()?;
615    let w_pt: Option<f64> = width.as_ref().and_then(|w| w.strip_suffix("pt").and_then(|s| s.parse().ok()));
616    let h_pt: Option<f64> = height.as_ref().and_then(|h| h.strip_suffix("pt").and_then(|s| s.parse().ok()));
617
618    whatsit.set_property("rwidth", width.unwrap_or_else(|| "100%".to_string()));
619    whatsit.set_property("rheight", height.unwrap_or_else(|| "1px".to_string()));
620    if let Some(d) = &depth { whatsit.set_property("rdepth", d.clone()); }
621
622    if let Some(_alignment) = lookup_alignment() {
623      // Perl: set isHorizontalRule only if dimensions suggest a real rule
624      let dominated_by_width = match (h_pt, w_pt) {
625        (None, None) => true,
626        (None, Some(w)) => w > 20.0,
627        (Some(h), Some(w)) => w > 3.0 * h,
628        _ => false,
629      };
630      if dominated_by_width {
631        // TODO: $alignment->addLine('t');
632        whatsit.set_property("isHorizontalRule", true);
633      }
634    }
635    // Outside alignment: isHorizontalRule is NOT set, so template outputs <ltx:rule>
636    // TODO: color handling
637    Ok(Vec::new())
638  });
639
640  // Various leaders, ignored for now...
641  DefPrimitive!("\\leaders", None);
642  DefPrimitive!("\\cleaders", None);
643  DefPrimitive!("\\xleaders", None);
644
645  // Overlay one glyph on another (used by \accent fallback)
646  // Perl: enterHorizontal => 1
647  DefConstructor!("\\lx@overlay{}{}",
648    "<ltx:text class='ltx_overlay' _noautoclose='1'>\
649       <ltx:text class='ltx_overlay_base' _noautoclose='1'>#1</ltx:text>\
650       <ltx:text class='ltx_overlay_over' _noautoclose='1'>#2</ltx:text>\
651     </ltx:text>",
652    enter_horizontal => true
653  );
654});

656// Risky: I think this needs to be digested as a body to work like TeX (?)
657// but parameter think's it's just parsing from gullet...
658pub fn read_box_contents(everybox_opt: Option<Tokens>) -> Result<Tokens> {
659  while let Some(t) = gullet::read_token()? {
660    if t.get_catcode() == Catcode::BEGIN {
661      break;
662    } // Skip till { or \bgroup
663  }
664  // Now, insert some extra tokens, if any, possibly from \afterassignment
665  match state::remove_value("BeforeNextBox") {
666    Some(Stored::Tokens(tokens)) => gullet::unread(tokens),
667    Some(Stored::Token(token)) => gullet::unread_one(token),
668    None | Some(Stored::None) => {},
669    Some(other) => panic!("afterAssignment should be a token, got: {}", other),
670  };
671  // AND, insert any extra tokens passed in, due to everyhbox or everyvbox
672  if let Some(everybox) = everybox_opt {
673    gullet::unread(everybox);
674  }
675  Ok(Tokens!())
676}
