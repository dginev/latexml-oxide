7/// Perl: hackVBoxAttachment($box, $valign)
8/// Sets vattach on the box, with special handling for \halign alignment objects.
9///
10/// For \halign inside \vbox/\vtop, vattach must be set on the alignment's XML
11/// attributes (so it becomes an attribute on `<tabular>`), NOT on the box property
12/// (which would end up on `<para>` via insertBlock).
13///
14/// Perl's List() simplification returns single-item vertical lists unwrapped,
15/// so $box->getProperty('alignment') finds the alignment directly. In Rust,
16/// Lists are not simplified, so we must walk into children to find it.
17fn hack_vbox_attachment(whatsit: &mut Whatsit, valign: &'static str) {
18  if let Some(content_box) = whatsit.get_arg_mut(2) {
19    if !set_halign_vattach(content_box, valign) {
20      // No \halign alignment found — set vattach as property on the box
21      content_box.set_property("vattach", valign);
22    }
23  }
24}

26/// Walk into a Digested tree to find a \halign Whatsit with an 'alignment' property.
27/// If found, set vattach on the alignment's XML attributes and return true.
28/// Returns false if no \halign alignment was found.
29fn set_halign_vattach(digested: &Digested, valign: &str) -> bool {
30  match digested.data() {
31    DigestedData::Whatsit(ref w) => {
32      let w_ref = w.borrow();
33      if w_ref.get_property("alignment").is_some() {
34        // Check if this whatsit's definition is \halign
35        let def = w_ref.get_definition();
36        let is_halign = *def.get_cs_name() == *"\\halign";
37        if is_halign {
38          // Get the alignment property value and set vattach on it
39          if let Some(Cow::Borrowed(Stored::Digested(ref alignment_dig))) =
40            w_ref.get_property("alignment")
41          {
42            if let DigestedData::Alignment(ref alignment_cell) = *alignment_dig.data() {
43              alignment_cell
44                .borrow_mut()
45                .get_xml_attributes_mut()
46                .insert(String::from("vattach"), String::from(valign));
47              return true;
48            }
49          }
50        }
51        // Has alignment but not \halign (e.g. tabular) — don't set vattach
52        return false;
53      }
54      false
55    },
56    DigestedData::List(ref list_cell) => {
57      // Walk children to find a \halign
58      let children = list_cell.borrow().unlist();
59      for child in children.iter() {
60        if set_halign_vattach(child, valign) {
61          return true;
62        }
63      }
64      false
65    },
66    _ => false,
67  }
68}

576    let box_key = s!("box{}", number.value_of());
577    if let Some(Stored::Digested(stuff)) = lookup_value(&box_key) {
578      let mode = stuff.get_property_string("mode");
579      if mode.ends_with("horizontal") {
580        Ok(stuff.unlist())
581      } else {
582        Ok(vec![stuff])
583      }
584    } else {
585      Ok(Vec::new())
586    }
587  });
588
589  // \unvbox<8bit>, \unvcopy<8bit>
590  // Perl: $stomach->leaveHorizontal (TeX_Box.pool.ltxml lines 683, 693)
591  DefPrimitive!("\\unvbox Number", sub[(number)] {
592    leave_horizontal()?;
593    let box_key = s!("box{}", number.value_of());
594    if let Some(Stored::Digested(stuff)) = state::remove_value(&box_key) {
595      // Only unlist if box is vertical (mode ends with "vertical")
596      let mode = stuff.get_property_string("mode");
597      if mode.ends_with("vertical") {
598        Ok(stuff.unlist())
599      } else {
600        Ok(vec![stuff])
601      }
602    } else {
603      Ok(Vec::new())
604    }
605  });
606
607  // Perl: $stomach->leaveHorizontal (TeX_Box.pool.ltxml line 693)
608  DefPrimitive!("\\unvcopy Number", sub[(number)] {
609    leave_horizontal()?;
610    let box_key = s!("box{}", number.value_of());
611    if let Some(Stored::Digested(stuff)) = lookup_value(&box_key) {
612      let mode = stuff.get_property_string("mode");
613      if mode.ends_with("vertical") {
614        Ok(stuff.unlist())
615      } else {
616        Ok(vec![stuff])
617      }
618    } else {
619      Ok(Vec::new())
620    }
621  });

623  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
624  // Various box related parameters
625  // ----------------------------------------------------------------------
626  // \prevdepth      iq is the depth of the last box added to the current vertical list.
627  // \boxmaxdepth    pd is the maximum possible depth of a vertical box.
628  // \badness        iq is 0-10,000 and represents the badness of the glue settings
629  //                    in the last constructed box.
630  // \hbadness       pi is the badness above which bad hboxes are reported.
631  // \vbadness       pi is the badness above which bad vboxes are reported.
632  // \hfuzz          pd is the overrun allowed before overfull hboxes are reported.
633  // \vfuzz          pd is the overrun allowed before overfull vboxes are reported.
634  // \overfullrule   pd is the width of the rule appended to an overfull box.
635  // ======================================================================
636  DefRegister!("\\prevdepth", Dimension::new(0));
637  DefRegister!("\\boxmaxdepth", Dimension!("16383.99999pt"));
638  DefRegister!("\\hfuzz", Dimension!("0.1pt"));
639  DefRegister!("\\vfuzz", Dimension!("0.1pt"));
640  DefRegister!("\\overfullrule", Dimension!("5pt"));
641  DefRegister!("\\badness", Number::new(0), readonly => true);  // Perl: readonly
642  DefRegister!("\\hbadness", Number!(1000));   // Perl: NOT readonly (writable threshold)
643  DefRegister!("\\vbadness", Number!(1000));
644
645  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
646  // Rules and Leaders
647  // ----------------------------------------------------------------------
648  // \hrule          c  makes a rule box in vertical mode.
649  // \vrule          c  makes a rule box in horizontal mode.
650  // \cleaders       c  insert centered leaders.
651  // \leaders        c  fill space using specified glue with a box or rule.
652  // \xleaders       c  insert expanded leaders.
653  // ======================================================================
654  DefParameterType!(RuleSpecification, sub[_inner, _extra] {
655    let mut keyvals = KeyVals::new(
656      KeyvalsConfig{ skip_missing: keyvals::SkipMissing::All, .. KeyvalsConfig::default()});
657    while let Some(key) = gullet::read_keyword(&["width", "height", "depth"])? {
658      keyvals.set_value(&key, ArgWrap::Dimension(gullet::read_dimension()?), false)?;
659    }
660    keyvals
661  },
662  optional => true,
663  predigest => sub[arg] {
664    match arg {
665      ArgWrap::KV(kv) => Ok(Some(Digested::from(*kv))),
666      _ => Ok(arg.undigested()),
667    }
668  });
669
670  // \hrule, \vrule are awkward in trying to deal with 3 cases
671  //  * as rules within an alignment/table
672  //  * as separating lines within text
673  //  * as graphical lines within svg
674  // and each has different requirements for size
675  DefConstructor!("\\vrule RuleSpecification",
676    "?#invisible()(?#isVerticalRule()(<ltx:rule ?#rheight(height='#rheight') ?#rdepth(depth='#rdepth')\
677       ?#rwidth(width='#rwidth') ?#color(color='#color')/>))",
678  after_digest => sub [whatsit] {
679    // Extract dimensions from keyvals arg (Perl: TeX_Box.pool.ltxml L752-760)
680    use latexml_core::digested::DigestedData;
681    use latexml_core::definition::argument::ArgWrap;
682    use latexml_core::common::dimension::Dimension;
683    let arg1 = whatsit.get_arg(1);
684    let (width, height, depth) = if let Some(d) = &arg1 {
685      if let DigestedData::KeyVals(kv) = d.data() {
686        let w = kv.get_value("width").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
687        let h = kv.get_value("height").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
688        let d = kv.get_value("depth").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
689        (w, h, d)
690      } else { (None, None, None) }
691    } else { (None, None, None) };
692
693    // Perl: $stomach->enterHorizontal
694    enter_horizontal();
695
696    // Perl: rwidth => $width, cwidth => $width || Dimension('0.4pt'), etc.
697    // Use to_attribute() for 1-decimal-place formatting matching Perl's Dimension->toAttribute
698    use latexml_core::common::numeric_ops::NumericOps;
699    if let Some(w) = width { whatsit.set_property("rwidth", w.to_attribute()); }
700    if let Some(h) = height { whatsit.set_property("rheight", h.to_attribute()); }
701    if let Some(d) = depth { whatsit.set_property("rdepth", d.to_attribute()); }
702    // Set computed sizes (Perl: cwidth/cheight/cdepth) as cached_width/cached_height/cached_depth
703    // These are used by get_size() to determine box dimensions for alignment cell sizing.
704    // Perl: cwidth => $width || Dimension('0.4pt')
705    let default_width: Dimension = "0.4pt".parse().unwrap_or_default();
706    let cwidth = width.unwrap_or(default_width);
707    let cheight = height.unwrap_or_default();
708    let cdepth = depth.unwrap_or_default();
709    whatsit.set_property("cached_width", cwidth);
710    whatsit.set_property("cached_height", cheight);
711    whatsit.set_property("cached_depth", cdepth);
712
713    let w_pt = width.map(|d| d.value_of() as f64 / 65536.0);
714    let h_pt = height.map(|d| d.value_of() as f64 / 65536.0);
715
716    if let Some(_alignment) = lookup_alignment() {
717      // Perl: set isVerticalRule only if dimensions suggest a real rule
718      let dominated_by_height = match (h_pt, w_pt) {
719        (None, None) => true,
720        (Some(h), None) => h > 20.0,
721        (Some(h), Some(w)) => h > 3.0 * w,
722        _ => false,
723      };
724      if dominated_by_height {
725        whatsit.set_property("isVerticalRule", true);
726      }
727    } else if w_pt == Some(0.0) {
728      whatsit.set_property("invisible", true);
729    }
730    // Set color from current font (Perl: only if NOT black)
731    if let Some(font) = lookup_font() {
732      if let Some(color) = font.color {
733        if color != latexml_core::common::color::BLACK {
734          whatsit.set_property("color", color.to_attribute());
735        }
736      }
737    }
738    Ok(Vec::new())
739  });
740
741  DefConstructor!("\\hrule RuleSpecification",
742    "?#isHorizontalRule()(<ltx:rule ?#rheight(height='#rheight') ?#rdepth(depth='#rdepth')\
743       ?#rwidth(width='#rwidth') ?#color(color='#color')/>)",
744  before_construct => sub[document, whatsit] {
745    // Perl: maybeCloseElement('ltx:p') if rwidth is '100%'
746    if whatsit.get_property_string("rwidth") == "100%" {
747      document.maybe_close_element("ltx:p")?;
748    }
749  },
750  after_digest => sub [whatsit] {
751    use latexml_core::digested::DigestedData;
752    use latexml_core::definition::argument::ArgWrap;
753    use latexml_core::common::numeric_ops::NumericOps;
754    let (width, height, depth) = if let Some(d) = whatsit.get_arg(1) {
755      if let DigestedData::KeyVals(kv) = d.data() {
756        let w = kv.get_value("width").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
757        let h = kv.get_value("height").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
758        let d = kv.get_value("depth").and_then(|a| if let ArgWrap::Dimension(d) = a { Some(*d) } else { None });
759        (w, h, d)
760      } else { (None, None, None) }
761    } else { (None, None, None) };
762
763    // Perl: $stomach->leaveHorizontal;
764    leave_horizontal()?;
765    let w_pt = width.map(|d| d.value_of() as f64 / 65536.0);
766    let h_pt = height.map(|d| d.value_of() as f64 / 65536.0);
767
768    // Perl: rwidth => $width || '100%', rheight => $height || '1px'
769    whatsit.set_property("rwidth", width.map(|w| w.to_attribute()).unwrap_or_else(|| "100%".to_string()));
770    whatsit.set_property("rheight", height.map(|h| h.to_attribute()).unwrap_or_else(|| "1px".to_string()));
771    if let Some(d) = depth { whatsit.set_property("rdepth", d.to_attribute()); }
772    // Set computed sizes for alignment cell sizing
773    let cheight = height.unwrap_or_else(|| "1px".parse::<latexml_core::common::dimension::Dimension>().unwrap_or_default());
774    let cdepth = depth.unwrap_or_default();
775    whatsit.set_property("cached_height", cheight);
776    whatsit.set_property("cached_depth", cdepth);
777    // hrule defaults to full width — don't cache a specific width
778
779    if let Some(_alignment) = lookup_alignment() {
780      // Perl: set isHorizontalRule only if dimensions suggest a real rule
781      let dominated_by_width = match (h_pt, w_pt) {
782        (None, None) => true,
783        (None, Some(w)) => w > 20.0,
784        (Some(h), Some(w)) => w > 3.0 * h,
785        _ => false,
786      };
787      if dominated_by_width {
788        _alignment.alignment_cell().unwrap().borrow_mut()
789          .add_line("t", Vec::new());
790        whatsit.set_property("isHorizontalRule", true);
791      }
792    }
793    // Outside alignment: isHorizontalRule is NOT set, so template outputs <ltx:rule>
794    // Set color from current font (Perl: only if NOT black)
795    if let Some(font) = lookup_font() {
796      if let Some(color) = font.color {
797        if color != latexml_core::common::color::BLACK {
798          whatsit.set_property("color", color.to_attribute());
799        }
800      }
801    }
802    Ok(Vec::new())
803  });
804
805  // Various leaders — fill space with box or rule
806  // Perl: DefConstructor('\leaders Digested Digested', sub { ... },
807  //   bounded => 1, beforeDigest => sub { $STATE->assignValue(Alignment => undef); });
808  DefConstructor!("\\leaders Digested Digested", sub [document, args] {
809    let filler = &args[0];
810    let _glue = &args[1];
811
812    // Get the context element's box to check for explicit width
813    let context = document.get_element();
814    let cbox = context.as_ref().and_then(|n| document.get_node_box(n));
815    let req_width: Option<Dimension> = cbox.as_ref().and_then(|b| {
816      b.get_property("width").and_then(|s| {
817        let dim_opt: Option<Dimension> = (&*s).into();
818        dim_opt
819      })
820    });
821
822    let mut noautoclose_attrs = HashMap::default();
823    noautoclose_attrs.insert(String::from("_noautoclose"), String::from("1"));
824    let mut container = document.open_element("ltx:text", Some(noautoclose_attrs), None)?;
825
826    if let Some(ref filler_d) = filler {
827      document.absorb(filler_d, None)?;
828    }
829
830    // Check if we should extend a rule to fill the requested width
831    let mut unwrap = false;
832    if let (Some(_), Some(ref rw)) = (&cbox, &req_width) {
833      // Find the last child of the container — should be the absorbed rule
834      if let Some(mut fnode) = container.get_last_child() {
835        let qname = fnode.get_name();
836        if qname == "rule" {
837          // Extend the rule to fill the width
838          document.set_attribute(&mut fnode, "width", &rw.to_attribute())?;
839          document.add_class(&mut fnode, "ltx_filled_leader")?;
840          unwrap = true;
841        }
842      }
843    }
844    if !unwrap {
845      document.add_class(&mut container, "ltx_leader")?;
846    }
847
848    document.close_element("ltx:text")?;
849    if unwrap {
850      document.unwrap_nodes(container)?;
851    }
852  },
853    bounded => true,
854    before_digest => sub {
855      // Hide alignment so that \hrule inside \leaders doesn't add border="t"
856      // Perl: $STATE->assignValue(Alignment => undef);
857      state::assign_value("Alignment", Stored::None, None);
858    }
859  );
860
861  state::let_i(&T_CS!("\\cleaders"), &T_CS!("\\leaders"), None);
862  state::let_i(&T_CS!("\\xleaders"), &T_CS!("\\leaders"), None);
863
864  // Overlay one glyph on another (used by \accent fallback)
865  // Perl: enterHorizontal => 1
866  DefConstructor!("\\lx@overlay{}{}",
867    "<ltx:text class='ltx_overlay' _noautoclose='1'>\
868       <ltx:text class='ltx_overlay_base' _noautoclose='1'>#1</ltx:text>\
869       <ltx:text class='ltx_overlay_over' _noautoclose='1'>#2</ltx:text>\
870     </ltx:text>",
871    enter_horizontal => true
872  );
873});

875// Risky: I think this needs to be digested as a body to work like TeX (?)
876// but parameter think's it's just parsing from gullet...
877pub fn read_box_contents(everybox_opt: Option<Tokens>) -> Result<Tokens> {
878  while let Some(t) = gullet::read_token()? {
879    if t.get_catcode() == Catcode::BEGIN {
880      break;
881    } // Skip till { or \bgroup
882  }
883  // Now, insert some extra tokens, if any, possibly from \afterassignment
884  match state::remove_value("BeforeNextBox") {
885    Some(Stored::Tokens(tokens)) => gullet::unread(tokens),
886    Some(Stored::Token(token)) => gullet::unread_one(token),
887    None | Some(Stored::None) => {},
888    Some(other) => panic!("afterAssignment should be a token, got: {}", other),
889  };
890  // AND, insert any extra tokens passed in, due to everyhbox or everyvbox
891  if let Some(everybox) = everybox_opt {
892    gullet::unread(everybox);
893  }
894  Ok(Tokens!())
895}
