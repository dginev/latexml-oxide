68fn stored_string_list(keys: &[&str]) -> Stored {
69  let deque: VecDeque<Stored> = keys.iter().map(|k| Stored::from(k.to_string())).collect();
70  Stored::VecDequeStored(deque)

102pub fn save_theorem_style(name: &str, saved: Vec<(String, Stored)>) {
103  let key = s!("THEOREM_{name}_PARAMETERS");
104  let deque: VecDeque<Stored> = saved
105    .into_iter()
106    .flat_map(|(k, v)| vec![Stored::from(k), v])
107    .collect();
108  state::assign_value(&key, Stored::VecDequeStored(deque), Some(Scope::Global));
109}

111pub fn use_theorem_style(name: &str) {
112  let savable_keys = get_savable_keys();
113  let params_key = s!("THEOREM_{name}_PARAMETERS");
114  if let Some(Stored::VecDequeStored(params)) = state::lookup_value(&params_key) {
115    let params_vec: Vec<Stored> = params.into_iter().collect();
116    let mut i = 0;
117    while i + 1 < params_vec.len() {
118      let key = params_vec[i].to_string();
119      let val = params_vec[i + 1].clone();
120      if savable_keys.iter().any(|k| k == &key) {
121        if key.starts_with('\\') {
122          let tokens = match val {
123            Stored::Tokens(t) => t,
124            Stored::Bool(_) => {
125              // bool stored for a register key — skip
126              i += 2;
127              continue;
128            },
129            _ => mouth::tokenize(&val.to_string()),
130          };
131          let _ = state::assign_register(
132            &key,
133            RegisterValue::Tokens(tokens),
134            None,
135            vec![],
136          );
137        } else {
138          state::assign_value(&key, val, None);
139        }
140      }
141      i += 2;
142    }
143  }
144}

146fn define_new_theorem(
147  flag: Option<Tokens>,
148  thmset: Tokens,
149  otherthmset: Option<Tokens>,
150  typ: Option<Tokens>,
151  within: Option<Tokens>,
152) -> Result<()> {
153  let thmset_str = thmset.to_string();
154  let classname = clean_class_name(&thmset_str);
155  let listname = {
156    let mut ln = s!("theorem:{thmset_str}");
157    ln.retain(|c| !c.is_whitespace());
158    ln = ln.replace('\'', "prime");
159    ln = ln.replace('?', "question");
160    ln = ln.replace('#', "hash");
161    ln
162  };
163  let otherthmset_str = otherthmset
164    .as_ref()
165    .map(|t| t.to_string())
166    .filter(|s| !s.is_empty());
167  let has_type = typ.as_ref().is_some_and(|t| !t.is_empty());
168  let is_starred = flag.is_some();
169
170  let within_str = if let Some(ref w) = within {
171    let ws = digest_literal(w.clone())?.to_string();
172    if ws.is_empty() { None } else { Some(ws) }
173  } else {
174    None
175  };
176
177  let counter = otherthmset_str.clone().unwrap_or_else(|| thmset_str.clone());
178  let counter = counter.replace(' ', ".");
179
180  // If counter != thmset, record mapping
181  if counter != thmset_str {
182    AssignMapping!("counter_for_type", &thmset_str => &counter);
183    DefMacro!(
184      T_CS!(s!("\\the{thmset_str}")),
185      None,
186      Some(ExpansionBody::Tokens(Tokens::new(vec![T_CS!(s!("\\the{counter}"))]))),
187      scope => Some(Scope::Global)
188    );
189  }
190
191  let numbering = {
192    let reg = LookupRegister!("\\thm@numbering");
193    if let RegisterValue::Tokens(t) = reg { t.to_string() } else { "\\arabic".into() }
194  };
195
196  let is_starred = is_starred || numbering.is_empty();
197
198  if otherthmset_str.is_none() {
199    let idprefix = s!("Thm{}", classname.replace('*', "."));
200    let c_counter = s!("\\c@{counter}");
201    if !is_defined(&c_counter) {
202      let within_ref = within_str.as_deref().unwrap_or("");
203      NewCounter!(&counter, within_ref, idprefix => &idprefix);
204    }
205    // Define \the<counter>
206    if !numbering.is_empty() {
207      let the_counter_body = if let Some(ref w) = within_str {
208        s!("\\csname the{w}\\endcsname\\@thmcountersep{numbering}{{{counter}}}")
209      } else {
210        s!("{numbering}{{{counter}}}")
211      };
212      DefMacro!(
213        T_CS!(s!("\\the{counter}")),
214        None,
215        Some(ExpansionBody::Tokens(mouth::tokenize_internal(&the_counter_body))),
216        scope => Some(Scope::Global)
217      );
218    }
219  }
220
221  // Save current theorem style params for this theorem name
222  let savable_keys = get_savable_keys();
223  let mut saved_params: Vec<(String, Stored)> = Vec::new();
224  for key in &savable_keys {
225    if key.starts_with('\\') {
226      let reg = LookupRegisterOrDefault!(key);
227      let tokens = match reg {
228        RegisterValue::Tokens(t) => t,
229        _ => Tokens!(),
230      };
231      saved_params.push((key.clone(), Stored::Tokens(tokens)));
232    } else {
233      let val = state::lookup_value(key).unwrap_or(Stored::None);
234      saved_params.push((key.clone(), val));
235    }
236  }
237  save_theorem_style(&thmset_str, saved_params);
238
239  // Define \lx@name@<thmset>
240  let thmname_cs = s!("\\lx@name@{thmset_str}");
241  if has_type {
242    let type_tokens = typ.clone().unwrap();
243    DefMacro!(
244      T_CS!(&thmname_cs),
245      None,
246      Some(ExpansionBody::Tokens(type_tokens)),
247      scope => Some(Scope::Global)
248    );
249  } else {
250    DefMacro!(
251      T_CS!(&thmname_cs),
252      None,
253      Some(ExpansionBody::Tokens(Tokens!())),
254      scope => Some(Scope::Global)
255    );
256  }
257
258  // Read swap value
259  let swap = state::lookup_value("thm@swap")
260    .map(|v| match v {
261      Stored::Int(n) => n != 0,
262      Stored::Bool(b) => b,
263      _ => false,
264    })
265    .unwrap_or(false);
266
267  // Define \fnum@<thmset>
268  let fnum_cs = s!("\\fnum@{thmset_str}");
269  let fnum_tokens = if is_starred || counter.is_empty() {
270    Tokens::new(vec![T_CS!(&thmname_cs)])
271  } else if swap {
272    let mut toks = vec![T_CS!(s!("\\the{counter}"))];
273    if has_type {
274      toks.push(T_SPACE!());
275    }
276    toks.push(T_CS!(&thmname_cs));
277    Tokens::new(toks)
278  } else {
279    let mut toks = vec![T_CS!(&thmname_cs)];
280    if has_type {
281      toks.push(T_SPACE!());
282    }
283    toks.push(T_CS!(s!("\\the{counter}")));
284    Tokens::new(toks)
285  };
286  DefMacro!(
287    T_CS!(&fnum_cs),
288    None,
289    Some(ExpansionBody::Tokens(fnum_tokens)),
290    scope => Some(Scope::Global)
291  );
292
293  // Define \format@title@<thmset>
294  let format_title_cs = s!("\\format@title@{thmset_str}");
295  let headformatter = LookupRegisterOrDefault!("\\thm@headformatter");
296  let headformatter_tokens = match headformatter {
297    RegisterValue::Tokens(t) => t,
298    _ => Tokens!(),
299  };
300
301  let format_cs_token = T_CS!(&format_title_cs);
302  if !headformatter_tokens.is_empty() {
303    // amsthm-style head formatter
304    let mut fmt_toks = vec![
305      T_CS!("\\the"), T_CS!("\\thm@headfont"),
306    ];
307    fmt_toks.extend(headformatter_tokens.unlist());
308    fmt_toks.push(T_BEGIN!());
309    if has_type {
310      fmt_toks.extend(typ.clone().unwrap().unlist());
311    }
312    fmt_toks.push(T_END!());
313    fmt_toks.push(T_CS!(s!("\\the{counter}")));
314    fmt_toks.push(T_BEGIN!());
315    fmt_toks.push(T_PARAM!());
316    fmt_toks.push(T_OTHER!("1"));
317    fmt_toks.push(T_END!());
318    fmt_toks.push(T_CS!("\\the"));
319    fmt_toks.push(T_CS!("\\thm@headpunct"));
320
321    let params = parse_parameters("{}", &format_cs_token, true)?;
322    DefMacro!(
323      format_cs_token,
324      params,
325      Some(ExpansionBody::Tokens(Tokens::new(fmt_toks))),
326      scope => Some(Scope::Global)
327    );
328  } else {
329    // Standard format
330    let note_part = if has_type {
331      "\\ifx.#1.\\else\\space\\the\\thm@notefont(#1)\\fi"
332    } else {
333      "#1"
334    };
335    let fmt_str = s!(
336      "{{\\the\\thm@headfont\\lx@tag{{\\csname fnum@{thmset_str}\\endcsname}}{{{note_part}}}\\the\\thm@headpunct}}"
337    );
338    let params = parse_parameters("{}", &format_cs_token, true)?;
339    DefMacro!(
340      format_cs_token,
341      params,
342      Some(ExpansionBody::Tokens(mouth::tokenize_internal(&fmt_str))),
343      scope => Some(Scope::Global)
344    );
345  }
346
347  // Define the environment
348  let thmset_for_env = thmset_str.clone();
349
350  // Hand-written replacement closure (compile_replacement! only works with literals)
351  let inlist_val = s!("thm {listname}");
352  let class_val = s!("ltx_theorem_{classname}");
353  let compiled_replacement: Option<ReplacementClosure> = Some(Rc::new(
354    move |document: &mut Document,
355          _args: &Vec<Option<Digested>>,
356          props: &SymHashMap<Stored>| {
357      let mut av_props: HashMap<String, String> = HashMap::default();
358      if let Some(stored) = props.get("id") {
359        av_props.insert("xml:id".into(), stored.to_string());
360      }
361      av_props.insert("inlist".into(), inlist_val.clone());
362      av_props.insert("class".into(), class_val.clone());
363      let this_font_opt = match props.get("font") {
364        Some(Stored::Font(f)) => Some(Cow::Borrowed(&**f)),
365        Some(Stored::FontDirective(FontDirective::Asset(fa))) => Some(Cow::Borrowed(&**fa)),
366        Some(Stored::FontDirective(FontDirective::Closure(code))) =>
367          Some(Cow::Owned(code(None)?)),
368        _ => None
369      };
370      if let Some(this_font) = this_font_opt {
371        document.open_element("ltx:theorem", Some(av_props), Some(&this_font))?;
372      } else {
373        document.open_element("ltx:theorem", Some(av_props), None)?;
374      }
375      // #tags
376      if let Some(stored_digested) = props.get("tags") {
377        let digested_opt: Option<Digested> = stored_digested.into();
378        if let Some(ref digested) = digested_opt {
379          document.absorb(digested, None)?;
380        }
381      }
382      // <ltx:title font='#titlefont' _force_font='true'>#title</ltx:title>
383      let mut title_av: HashMap<String, String> = HashMap::default();
384      if let Some(stored) = props.get("titlefont") {
385        title_av.insert("font".into(), stored.to_string());
386      }
387      title_av.insert("_force_font".into(), "true".into());
388      let title_font_opt = match props.get("titlefont") {
389        Some(Stored::Font(f)) => Some(Cow::Borrowed(&**f)),
390        Some(Stored::FontDirective(FontDirective::Asset(fa))) => Some(Cow::Borrowed(&**fa)),
391        Some(Stored::FontDirective(FontDirective::Closure(code))) =>
392          Some(Cow::Owned(code(None)?)),
393        _ => None
394      };
395      if let Some(title_font) = title_font_opt {
396        document.open_element("ltx:title", Some(title_av), Some(&title_font))?;
397      } else {
398        document.open_element("ltx:title", Some(title_av), None)?;
399      }
400      if let Some(stored_digested) = props.get("title") {
401        let digested_opt: Option<Digested> = stored_digested.into();
402        if let Some(ref digested) = digested_opt {
403          document.absorb(digested, None)?;
404        }
405      }
406      document.close_element("ltx:title")?;
407      // #body
408      if let Some(stored_digested) = props.get("body") {
409        let digested_opt: Option<Digested> = stored_digested.into();
410        if let Some(ref digested) = digested_opt {
411          document.absorb(digested, None)?;
412        }
413      }
414      Ok(())
415    }));
416
417  let thmset_for_before = thmset_str.clone();
418  let counter_for_props = counter.clone();
419  let thmset_for_props = thmset_str.clone();
420  let is_starred_for_props = is_starred;
421  let has_type_for_props = has_type;
422
423  let mut options = ConstructorOptions { 
424    mode: Some("internal_vertical".into()), 
425    scope: Some(Scope::Global), 
426    ..Default::default() };
427
428  // before_digest
429  let before_digest_closure: BeforeDigestClosure = Rc::new(move || {
430    use_theorem_style(&thmset_for_before);
431    let digested = stomach::digest(mouth::tokenize_internal("\\normalfont\\the\\thm@prework"))?;
432    Ok(vec![digested])
433  });
434  options.before_digest.push(before_digest_closure);
435
436  // after_digest_begin
437  let after_digest_begin_closure: DigestionClosure = Rc::new(move |whatsit| {
438    let name_opt = whatsit.get_arg(1);
439    let name_str = name_opt
440      .map(|n| n.revert().map(|t| t.to_string()).unwrap_or_default())
441      .unwrap_or_default();
442    let digest_str = s!(
443      "\\the\\thm@bodyfont\\the\\thm@styling\\def\\lx@thistheorem{{{name_str}}}"
444    );
445    let digested = stomach::digest(mouth::tokenize_internal(&digest_str))?;
446    Ok(vec![digested])
447  });
448  options.after_digest_begin.push(after_digest_begin_closure);
449
450  // before_digest_end
451  let before_digest_end_closure: BeforeDigestClosure = Rc::new(move || {
452    let digested = stomach::digest(mouth::tokenize_internal("\\thm@doendmark\\the\\thm@postwork"))?;
453    Ok(vec![digested])
454  });
455  options.before_digest_end.push(before_digest_end_closure);
456
457  // after_construct
458  let after_construct_closure: ConstructionClosure =
459    Rc::new(move |document: &mut Document, _whatsit: &Whatsit| {
460      document.maybe_close_element("ltx:theorem")?;
461      Ok(())
462    });
463  options.after_construct.push(after_construct_closure);
464
465  // properties
466  let thmset_for_tags = thmset_for_props.clone();
467  let counter_for_tags = counter_for_props.clone();
468  let props_closure: PropertiesClosure = Rc::new(
469    #[allow(clippy::ptr_arg)]
470    move |args: &Vec<Option<Digested>>| {
471      let mut props = SymHashMap::default();
472
473      if !counter_for_tags.is_empty() {
474        if is_starred_for_props {
475          let ctr_props = ref_step_id(&counter_for_tags)?;
476          for (k, v) in ctr_props.iter() {
477            props.insert_sym(*k, v.clone());
478          }
479          // For starred theorems with a type, create tags without the counter number
480          if has_type_for_props {
481            let tag_tokens = Tokens::new(vec![
482              T_BEGIN!(),
483              T_CS!("\\let"),
484              T_CS!(s!("\\the{}", counter_for_tags)),
485              T_CS!("\\@empty"),
486              T_CS!("\\lx@make@tags"),
487              T_BEGIN!(),
488            ]);
489            let mut full_toks = tag_tokens.unlist();
490            full_toks.extend(mouth::tokenize(&thmset_for_tags).unlist());
491            full_toks.push(T_END!());
492            full_toks.push(T_END!());
493            let tags = stomach::digest(Tokens::new(full_toks))?;
494            props.insert("tags", tags.into());
495          }
496        } else {
497          let ctr_props = ref_step_counter(&thmset_for_tags, false)?;
498          for (k, v) in ctr_props.iter() {
499            props.insert_sym(*k, v.clone());
500          }
501        }
502      }
503
504      // Compute title
505      let format_title_cs = s!("\\format@title@{}", thmset_for_tags);
506      let mut title_tokens = vec![
507        T_BEGIN!(),
508        T_CS!("\\the"),
509        T_CS!("\\thm@headstyling"),
510        T_CS!(&format_title_cs),
511        T_BEGIN!(),
512      ];
513      if let Some(Some(ref arg)) = args.first() {
514        title_tokens.extend(arg.revert()?.unlist());
515      }
516      title_tokens.push(T_END!());
517      title_tokens.push(T_END!());
518
519      let title = digest_text(Tokens::new(title_tokens))?;
520      let titlefont = title.get_font()?.map(|f| f.into_owned());
521      props.insert("title", title.into());
522      if let Some(f) = titlefont {
523        props.insert("titlefont", Stored::Font(Rc::new(f)));
524      }
525
526      Ok(props)
527    },
528  );
529  options.properties = props_closure;
530
531  // Use the OptionalUndigested parameter
532  let env_cs = T_CS!(s!("\\begin{{{thmset_for_env}}}"));
533  let paramlist = parse_parameters("OptionalUndigested", &env_cs, true)?;
534  def_environment(thmset_for_env, paramlist, compiled_replacement, options);
535
536  Ok(())
537}
