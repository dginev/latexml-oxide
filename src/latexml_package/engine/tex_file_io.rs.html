3//! Core TeX Implementation for LaTeXML
4use crate::prelude::*;
5static PSFILE_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r"\bpsfile=(.+?)(?:\s|\})").unwrap());

7LoadDefinitions!({
8  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
9  // File I/O Family of primitive control sequences
10  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
11  //======================================================================
12  // Low-level input
13  //----------------------------------------------------------------------
14  // \openin           c  opens an auxiliary file for reading.
15  // \closein          c  closes an auxiliary file opened for reading.
16  // \read             c  reads one or more lines from an auxiliary file.
17  // \endinput         c  stops input from a file at the end of the current line.
18  // \inputlineno      iq holds the line number of the line last read in the current input file.
19
20  // TeX I/O primitives
21  DefPrimitive!("\\openin Number SkipMatch:= SkipSpaces TeXFileName",
22  sub[(port, filename)] {
23    let port = port.to_string();
24    let filename = filename.to_string();
25    // possibly should close $port if it's already been opened?
26    // Rely on FindFile to enforce any access restrictions
27    if let Some(path) = find_file(&filename, Some(
28      FindFileOptions {forbid_ltxml: true, ..FindFileOptions::default()})) {
29      let content_str = LookupString!(&s!("{}_contents",path));
30      let content = if content_str.is_empty() {
31        None
32      } else {
33        Some(content_str)
34      };
35      let mouth = Mouth::create(&path, MouthOptions {
36        content,
37        .. MouthOptions::default()
38      })?;
39      AssignValue!(&s!("input_file:{}", port), mouth, Some(Scope::Global));
40    }
41  });
42  DefPrimitive!("\\closein Number", sub[(port)] {
43    let file_key = s!("input_file:{}", port);
44    let mut finished = false;
45    //   close the mouth (if any) and clear the variable
46    with_value(&file_key, |mouth_opt|
47      if let Some(Stored::Mouth(ref mouth)) = mouth_opt {
48        mouth.borrow_mut().finish();
49        finished = true;
50      });
51    if finished {
52      AssignValue!(&file_key, false, Some(Scope::Global));
53    }
54  });
55
56  DefPrimitive!("\\read Number SkipKeyword:to SkipSpaces Token", sub[(port, token)] {
57    let mouth_opt =
58      if let Some(Stored::Mouth(mouth_stored)) = lookup_value(&format!("input_file:{port}")) {
59        Some(mouth_stored)
60      } else { None };
61    if let Some(mouth_obj) = mouth_opt {
62      bgroup();
63      AssignValue!("PRESERVE_NEWLINES", 2); // Special EOL/EOF treatment for \read
64      AssignValue!("INCLUDE_COMMENTS", false);
65      let mut tokens = Vec::new();
66      let mut level = 0;
67      let mut mouth = mouth_obj.borrow_mut();
68      while let Some(t) = mouth.read_token() {
69        let cc = t.get_catcode();
70        if cc != Catcode::MARKER {
71          tokens.push(t);
72        }
73        match cc {
74          Catcode::BEGIN => {level += 1},
75          Catcode::END => {level -= 1},
76          _ => {}
77        };
78        if level == 0 && mouth.is_eol() {
79          break;
80        }
81      }
82      egroup()?;
83      DefMacro!(token, None, Tokens::new(tokens), nopack_parameters => true);
84    }
85  });
86  // Note that TeX doesn't actually close the mouth;
87  // it just flushes it so that it will close the next time it's read!
88  DefMacro!(T_CS!("\\endinput"), None, {
89    gullet::flush_mouth();
90  });
91  DefRegister!("\\inputlineno",Number!(0), readonly => true, getter=> {
92    Number::new(gullet::get_locator().from_line as i64)
93  });
94
95  //======================================================================
96  // Low-level output
97  //----------------------------------------------------------------------
98  // \openout          c  opens an auxiliary file for writing.
99  // \closeout         c  closes an auxiliary file opened for writing.
100  // \write            c  writes material to an auxiliary file.
101  // \immediate        c  performs the following output command without waiting for \shipout.
102
103  // For output files, we'll write the data to a cached internal copy
104  // rather than to the actual file system.
105  DefPrimitive!("\\openout Number SkipMatch:= SkipSpaces TeXFileName",
106    sub[(port, filename)] {
107    let port = port.to_string();
108    let filename = filename.to_string();
109    let contents_key = &s!("{}_contents",filename);
110    AssignValue!(&s!("output_file:{}",port)  => filename,  Some(Scope::Global));
111    AssignValue!(contents_key => "",  Some(Scope::Global));
112  });
113
114  DefPrimitive!("\\closeout Number", sub[(port)] {
115    AssignValue!(&s!("output_file:{}",port), false, Some(Scope::Global));
116  });
117  DefPrimitive!("\\write Number {}", sub[(port_n, tokens)] {
118    let port = port_n.value_of();
119    let handle = with_value(&s!("output_file:{}", port), |val_opt|
120    if let Some(filename) = val_opt {
121       s!("{}_contents",filename)
122    } else { String::new() });
123    if !handle.is_empty() {
124      let mut contents : String = LookupString!(&handle);
125      contents.push_str(&Expand!(tokens).untex());
126      contents.push('\n');
127      AssignValue!(&handle => contents, Some(Scope::Global));
128    } else if port < 0 {
129      NoteLog!(Expand!(tokens).untex());
130    } else {
131      Note!(Expand!(tokens).untex());
132    }
133  });
134
135  // Since we don't paginate, we're effectively always "shipping out",
136  // so all operations are \immediate
137  DefPrimitive!("\\immediate", None);
138
139  //======================================================================
140  // High-level input
141  //----------------------------------------------------------------------
142  // \input            c  inserts a file at the current position in the source file.
143  DefMacro!("\\input TeXFileName", sub[(name)] {
144    let mut tks = name.unlist();
145    // If given a LaTeX-style argument, strip braces
146    if tks.len() > 1 && tks.first().unwrap().get_catcode() == Catcode::BEGIN
147      && tks.last().unwrap().get_catcode() == Catcode::END {
148      tks.remove(0);
149      tks.pop();
150      // and load LaTeX.pool if not already
151      if !lookup_bool("LaTeX.pool_loaded") {
152        LoadPool!("LaTeX");
153      }
154    }
155    let reloadable_opts = InputOptions { reloadable: true, ..InputOptions::default() };
156    input(&Tokens::new(tks).to_string(), reloadable_opts)?;
157  });
158  //======================================================================
159  // Special output
160  //----------------------------------------------------------------------
161  // \special          c  sends material to the dvi file for special processing.
162  DefPrimitive!("\\special {}", sub[(arg)] {
163    let special_str = arg.to_string();
164    // recognize one special graphics inclusion case
165    if let Some(cap) = PSFILE_REGEX.captures(&special_str) {
166      let graphic = cap.get(1).unwrap().as_str();
167      RequirePackage!("graphicx", searchpaths_only => true);
168      let mut kv = Vec::new();
169      for prop in ["voffset","hoffset","hscale","vscale","hsize","vsize","angle"] {
170        let prop_regex = Regex::new(&s!("\\b{prop}=(.+?)(?:\\s|\\}})")).unwrap();
171        if let Some(cap) = prop_regex.captures(&special_str) {
172          let prop_val = cap.get(1).unwrap().as_str();
173          if !kv.is_empty() {
174            kv.push(T_OTHER!(","));
175          }
176          kv.push(T_OTHER!(prop));
177          kv.push(T_OTHER!("="));
178          kv.push(T_OTHER!(prop_val));
179        }
180      }
181      if !kv.is_empty() {
182        let mut wrapped = vec![T_OTHER!("[")];
183        wrapped.extend(kv);
184        wrapped.push(T_OTHER!("]"));
185        kv = wrapped;
186      }
187
188      gullet::unread_vec(vec![T_BEGIN!(), T_OTHER!(graphic), T_END!()]);
189      gullet::unread_vec(kv);
190      gullet::unread_one(T_CS!("\\ltx@special@graphics"));
191    } else {
192      Info!("ignored", "special", s!("Unrecognized TeX Special: {arg}"));
193    }
194  });
