13  // Perl: PushValue('@at@begin@document', $_[1]->unlist)
14  // Note: in modern LaTeX with expl3, \AtBeginDocument is redefined to use
15  // the L3 hook system (\AddToHook{begindocument}{...}). Our definition here
16  // serves as a fallback when expl3 isn't loaded. When expl3 IS loaded, it
17  // overrides this with its own version that routes through \hook_gput_code:nnn.
18  // Perl 93f875a6: support optional [label] from modern LaTeX hooks system
19  DefMacro!("\\AtBeginDocument[]{}", sub[(_label, rules)] {
20    state::push_value("@at@begin@document", rules)
21  });
22  DefMacro!("\\AtEndDocument[]{}", sub[(_label, rules)] {
23    state::push_value("@at@end@document", rules)
24  });

26  // Like  "<ltx:document xml:id='#id'>#body</ltx:document>",
27  // But more complicated due to id, at begin/end document and so forth.
28  // AND, lower-level so that we can cope with common errors at document end.
29  DefConstructor!(T_CS!("\\begin{document}"), None, sub[document, _args, props] {
30    let id = prop_str!(props,"id");
31    // Already (auto) created?
32    if let Some(mut docel) = document.findnode("/ltx:document", None) {
33      if id != *EMPTY_SYM {
34        let id_s = arena::with(id, |s| s.to_string());
35        document.set_attribute(&mut docel, "xml:id", &id_s)?;
36      }
37    } else {
38      let props = arena::with(id, |id_str| string_map!("xml:id" => id_str));
39      document.open_element("ltx:document", Some(props), None)?;
40    }
41  },
42  after_digest => sub[whatsit] {
43    // Perl: beginMode('internal_vertical', 1) — noframe=1

51    Let!("\\@nodocument", "\\relax", Scope::Global);
52    // Clear \everypar at document start (Perl parity)
53    state::assign_value("\\everypar", Tokens!(), Some(Scope::Global));
54    let mut boxes = Vec::new();
55    if let Some(ops) = state::lookup_tokens("@document@preamble@atend") {
56      boxes.push(stomach::digest(ops)?);
57    }
58    if let Some(ops) = state::lookup_tokens("@at@begin@document") {
59      boxes.push(stomach::digest(ops)?);
60    }
61    // Fire the L3 hook system for begindocument.
62    // Modern LaTeX (with expl3) uses \hook_use:n{begindocument} instead of
63    // \@begindocumenthook. This fires hooks registered via \AtBeginDocument
64    // when expl3 has redefined it to use \AddToHook{begindocument}{...}.
65    // Includes babel's \lx@babel@activate@mainlang.
66    if lookup_definition(&T_CS!("\\hook_use:n"))?.is_some() {
67      boxes.push(stomach::digest(Tokenize!(r"\hook_use:n{begindocument}"))?);
68    }
69    // Fire babel language activation AFTER all hooks (including babel's own
70    // \selectlanguage call). This runs even if babel's hook code has errors.
71    // Use T_CS! directly since @ is OTHER catcode at \begin{document} time.
72    if lookup_definition(&T_CS!("\\lx@babel@activate@mainlang"))?.is_some() {
73      boxes.push(stomach::digest(Tokens!(T_CS!("\\lx@babel@activate@mainlang")))?);
74    }
75    state::assign_value("inPreamble", false, None); // atbegin is still (sorta) preamble
76    if let Some(ops) = state::lookup_tokens("@document@preamble@afterend") {
77      boxes.push(stomach::digest(ops)?);
78    }
79    whatsit.set_font(lookup_font().unwrap()); // Start w/ whatever font was last selected.
80    leave_horizontal_internal();
81    boxes
82  });
