137/// Perl: tabbingBindings() — sets up alignment with repeated template and rebinds control chars
138fn tabbing_bindings() -> Result<()> {
139  // Template: repeated column with before=\lx@text@intercol, after=\hfil\lx@text@intercol
140  let col = Cell {
141    before: Some(Tokens::new(vec![T_CS!("\\lx@text@intercol")])),
142    after: Some(Tokens::new(vec![T_CS!("\\hfil"), T_CS!("\\lx@text@intercol")])),
143    empty: true,
144    ..Cell::default()
145  };
146  let template = Template::new(TemplateConfig {
147    repeated: vec![col],
148    ..TemplateConfig::default()
149  });
150
151  let mut xml_attrs = HashMap::default();
152  xml_attrs.insert(String::from("class"), String::from("ltx_tabbing"));
153
154  let alignment = Alignment::new(AlignmentConfig {
155    template: Some(template),
156    open_container: Rc::new(|document, props| {
157      document
158        .open_element("ltx:tabular", Some(props), None)
159        .map(Option::Some)
160    }),
161    close_container: Rc::new(|document| document.close_element("ltx:tabular")),
162    open_row: Rc::new(|document, props| {
163      let str_props: HashMap<String, String> = props.into_iter()
164        .map(|(k, v)| (k, v.to_string()))
165        .collect();
166      document
167        .open_element("ltx:tr", Some(str_props), None)
168        .and(Ok(()))
169    }),
170    close_row: Rc::new(|document| document.close_element("ltx:tr")),
171    open_column: Rc::new(|document, props| {
172      document
173        .open_element("ltx:td", Some(props), None)
174        .map(Option::Some)
175    }),
176    close_column: Rc::new(|document| document.close_element("ltx:td")),
177    is_math: false,
178    properties: SymHashMap::default(),
179    xml_attributes: xml_attrs,
180  });
181  assign_alignment(alignment, None);
182
183  // Rebind control characters within tabbing
184  // Perl: Let("\\=", '\@tabbing@tabset') etc.
185  state::let_i(&T_CS!("\\="), &T_CS!("\\@tabbing@tabset"), None);
186  state::let_i(&T_CS!("\\>"), &T_CS!("\\@tabbing@nexttab"), None);
187  state::let_i(&T_CS!("\\\\"), &T_CS!("\\@tabbing@newline"), None);
188  state::let_i(&T_CS!("\\kill"), &T_CS!("\\@tabbing@kill"), None);
189  state::let_i(&T_CS!("\\+"), &T_CS!("\\@tabbing@increment"), None);
190  state::let_i(&T_CS!("\\-"), &T_CS!("\\@tabbing@decrement"), None);
191  state::let_i(&T_CS!("\\<"), &T_CS!("\\@tabbing@untab"), None);
192  // Save accent definitions before rebinding \' and \`
193  state::let_i(&T_CS!("\\@tabbing@'"), &T_CS!("\\'"), None);
194  state::let_i(&T_CS!("\\@tabbing@`"), &T_CS!("\\`"), None);
195  state::let_i(&T_CS!("\\a"), &T_CS!("\\@tabbing@accent"), None);
196  // Rebind \' and \` to tabbing-specific (flush right / hfil)
197  state::let_i(&T_CS!("\\'"), &T_CS!("\\@tabbing@flushright"), None);
198  state::let_i(&T_CS!("\\`"), &T_CS!("\\@tabbing@hfil"), None);
199  state::let_i(&T_CS!("\\pushtabs"), &T_CS!("\\@tabbing@pushtabs"), None);
200  state::let_i(&T_CS!("\\poptabs"), &T_CS!("\\@tabbing@poptabs"), None);
201
202  Ok(())
