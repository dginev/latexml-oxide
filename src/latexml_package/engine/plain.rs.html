89/// Set role and augment delimiter properties on the last child element.
90/// Mirrors Perl's augmentDelimiterProperties (TeX_Math.pool.ltxml).
91/// Perl: augmentDelimiterProperties($doc, $whatsit, $role, $stretchy)
92/// Look up delimiter character in DELIM_CHAR_MAP and set name/meaning/role.
93/// When role is empty, don't change the role (Perl: $role=undef).
94fn augment_delimiter_properties(document: &mut Document, role: &str) -> Result<()> {
95  let current = document.get_node().clone();
96  let delim_opt = current
97    .get_child_nodes()
98    .into_iter()
99    .filter(|n| n.get_type() == Some(NodeType::ElementNode))
100    .last();
101  if let Some(mut delim) = delim_opt {
102    let char_content = delim.get_content();
103    let first_char = char_content.chars().next();
104    // Look up delimiter properties by char content
105    if let Some(entry) = first_char.and_then(|c| DELIM_CHAR_MAP.get(&c)) {
106      // Role: only change if explicitly requested (non-empty role)
107      if !role.is_empty() {
108        let new_role = if role == "OPEN" { entry.left_role } else { entry.right_role };
109        let current_role = delim.get_attribute("role");
110        match current_role.as_deref() {
111          None | Some("OPEN") | Some("MIDDLE") | Some("CLOSE") | Some("VERTBAR") => {
112            document.set_attribute(&mut delim, "role", new_role)?;
113          },
114          _ => {},
115        }
116      }
117      // Set name if entry has one
118      if let Some(name) = entry.name {
119        document.set_attribute(&mut delim, "name", name)?;
120      }
121      // Handle meaning
122      if entry.remove_meaning {
123        let _ = delim.remove_attribute("meaning");
124      }
125      // Handle char replacement (e.g. < → ⟨)
126      if let Some(replacement) = entry.replace_char {
127        if let Some(mut first_child) = delim.get_first_child() {
128          let _ = first_child.set_content(&replacement.to_string());
129        }
130      }
131    } else if !role.is_empty() {
132      // No map entry — just set role if explicitly requested
133      let current_role = delim.get_attribute("role");
134      match current_role.as_deref() {
135        None | Some("OPEN") | Some("MIDDLE") | Some("CLOSE") | Some("VERTBAR") => {
136          document.set_attribute(&mut delim, "role", role)?;
137        },
138        _ => {},
139      }
140    }
141  }
142  Ok(())
143}

2397  if document.is_openable("ltx:p") {

2400      "ltx:p",

2406    let line_content = line.iter().filter_map(|c| c.as_ref()).collect();
2407    document.insert_element(
2408      "ltx:text",
2409      line_content,

2411      "class" => s!("ltx_align_{alignment}"))),
2412    )?;
2413    document.insert_element("ltx:break", Vec::new(), None)?;
2414  } else if let Some(Some(line_content)) = line.first() {
2415    document.absorb(line_content, None)?;
2416  }
