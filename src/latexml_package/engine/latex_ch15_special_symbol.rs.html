3/// Perl: %unicode_enclosed_alphanumerics table
4/// Maps single chars (0-9, a-z, A-Z) and numbers 10-20 to their circled Unicode equivalents.
5fn unicode_enclosed_alphanumeric(text: &str) -> Option<String> {
6  let ch = match text {
7    "0" => '\u{24EA}', "1" => '\u{2460}', "2" => '\u{2461}', "3" => '\u{2462}',
8    "4" => '\u{2463}', "5" => '\u{2464}', "6" => '\u{2465}', "7" => '\u{2466}',
9    "8" => '\u{2467}', "9" => '\u{2468}', "10" => '\u{2469}', "11" => '\u{246A}',
10    "12" => '\u{246B}', "13" => '\u{246C}', "14" => '\u{246D}', "15" => '\u{246E}',
11    "16" => '\u{246F}', "17" => '\u{2470}', "18" => '\u{2471}', "19" => '\u{2472}',
12    "20" => '\u{2473}',
13    "a" => '\u{24D0}', "b" => '\u{24D1}', "c" => '\u{24D2}', "d" => '\u{24D3}',
14    "e" => '\u{24D4}', "f" => '\u{24D5}', "g" => '\u{24D6}', "h" => '\u{24D7}',
15    "i" => '\u{24D8}', "j" => '\u{24D9}', "k" => '\u{24DA}', "l" => '\u{24DB}',
16    "m" => '\u{24DC}', "n" => '\u{24DD}', "o" => '\u{24DE}', "p" => '\u{24DF}',
17    "q" => '\u{24E0}', "r" => '\u{24E1}', "s" => '\u{24E2}', "t" => '\u{24E3}',
18    "u" => '\u{24E4}', "v" => '\u{24E5}', "w" => '\u{24E6}', "x" => '\u{24E7}',
19    "y" => '\u{24E8}', "z" => '\u{24E9}',
20    "A" => '\u{24B6}', "B" => '\u{24B7}', "C" => '\u{24B8}', "D" => '\u{24B9}',
21    "E" => '\u{24BA}', "F" => '\u{24BB}', "G" => '\u{24BC}', "H" => '\u{24BD}',
22    "I" => '\u{24BE}', "J" => '\u{24BF}', "K" => '\u{24C0}', "L" => '\u{24C1}',
23    "M" => '\u{24C2}', "N" => '\u{24C3}', "O" => '\u{24C4}', "P" => '\u{24C5}',
24    "Q" => '\u{24C6}', "R" => '\u{24C7}', "S" => '\u{24C8}', "T" => '\u{24C9}',
25    "U" => '\u{24CA}', "V" => '\u{24CB}', "W" => '\u{24CC}', "X" => '\u{24CD}',
26    "Y" => '\u{24CE}', "Z" => '\u{24CF}',
27    _ => return None,
28  };
29  Some(ch.to_string())
30}
