329  // We've got the same problem as LaTeX: Lather, Rinse, Repeat.
330  // It would be nice to know the bib info at digestion time
331  //  * whether author lists will collapse
332  //  * whether there are "a","b".. extensions on the year.
333  // We could process the bibliography first, (IF it is a separate *.bib!)
334  // but won't know which entries are included (and so can't resolve the a/b/c..)
335  // until we've finished looking at (all of) the source(s) that will refer to them!
336  //
337  // We can do this in 2 passes, however
338  //  (1) convert (latexml) both the source document(s) and the bibliography
339  //  (2) extract the required bibitems and integrate (latexmlpost) it into the documents.
340  // [Note that for mult-document sites, step (2) becomes 2 stages: scan and integrate]
341  //
342  // Here's the general layout.
343  //   <ltx:cite> contains everything that the citations produce,
344  //     including parens, pre-note, punctunation that precede the <ltx:bibcite>
345  //     and punctuation, post-note, parens, that follow it.
346  //   <ltx:bibcite show="string" bibrefs="keys" sep="" yysep="">phrases</ltx:bibcite>
347  //     encodes the actual citation.
348  //
349  //     bibrefs : lists the bibliographic keys that will be used
350  //     show    : gives the pattern for formatting using data from the bibliography
351  //       It can contain:
352  //         authors or fullauthors
353  //         year
354  //         number
355  //         phrase1,phrase2,... selects one of the phrases from the content of the <ltx:bibref>
356  //     This format is used as follows:
357  //       If author and year is present, and a subset of the citations share the same authors,
358  //         then the format is used, but the year is repeated for each citation in the subset,
359  //         as a link to the bib entry.
360  //       Otherwise, the format is applied to each entry.
361  //
362  // The design is intended to support natbib, as well as plain LaTeX.

433fn adjust_backmatter_element(document: &mut Document, whatsit: &Whatsit) -> Result<()> {
434  let asif_opt =
435    if let Some(Stored::String(asif_sym)) = whatsit.get_property("backmatterelement").as_deref() {
436      Some(arena::to_string(*asif_sym))
437    } else {
438      None
439    };
440  // Note: We allocate a string here, since
441  // it looks like arena::with can deadlock with find_insertion_point
442  // we may need a find_insertion_point_sym to avoid that...
443  if let Some(asif) = asif_opt {
444    let point = document.find_insertion_point(&asif, None)?;
445    document.set_node(&point);
446  }
447  Ok(())
448}

450// Since SOME people seem to write bibliographies w/o \bibitem,
451// just blank lines between apparent entries,
452// Making \par do a \bibitem{} works, but screws up valid
453// bibliographies with blank lines!
454// So, let's do some redirection!
455fn setup_pseudo_bibitem() -> Result<()> {
456  Let!("\\save@bibitem", "\\bibitem");
457  Let!("\\save@par", "\\par");
458  Let!("\\save@backbackslash", "\\\\");
459  Let!("\\bibitem", "\\restoring@bibitem");
460  Let!("\\par", "\\par@in@bibliography");
461  Let!("\\\\", "\\par@in@bibliography");
462  Let!("\\vskip", "\\vskip@in@bibliography");
463  // Moreover some people use \item instead of \bibitem
464  Let!("\\item", "\\item@in@bibliography");
465  // And protect from redefinitions.
466  Let!("\\newblock", "\\lx@bibnewblock");
467  // Risky, but when bibliography immediatesly starts with text (no implied \par)
468  if let Some(token) = gullet::read_non_space()? {
469    gullet::unread_one(token);
470    if !token.is_executable() {
471      gullet::unread_one(T_CS!("\\par"));
472    }
473  }
474  Ok(())
475}

478fn begin_bibliography(whatsit: &mut Whatsit) -> Result<()> {
479  begin_bibliography_clean(whatsit)?;
480  // Fix for missing \bibitems!
481  setup_pseudo_bibitem()

484fn begin_bibliography_clean(whatsit: &mut Whatsit) -> Result<()> {
485  // Check if \bibsection is defined and try to decipher it.
486  // Expecting something like \section*{sometext}
487
488  // TODO: Continue updating here...
489  // let bs_opt       = lookup_definition(&T_CS!("\\bibsection"))?;
490
491  note_backmatter_element(whatsit, "ltx:bibliography");
492  // Try to compute a reasonable, but unique ID;
493  // relative to the document's ID, if any.
494  // But also, if there are multiple bibliographies,
495  let bibnumber = 1 + lookup_int("n_bibliographies");
496  assign_value("n_bibliographies", bibnumber, Some(Scope::Global));
497  let mut docid: String = Expand!(T_CS!("\\thedocument@ID")).to_string();
498  if !docid.is_empty() {
499    docid += ".";
500  }
501  let bibid = s!("{}bib{}", docid, radix::radix_alpha(bibnumber - 1));
502  DefMacro!(T_CS!("\\thebibliography@ID"), None, T_OTHER!(&bibid), scope => Some(Scope::Global));
503  whatsit.set_property("id", bibid);
504  let title_opt = match DigestIf!("\\refname")? {
505    Some(v) => Some(v),
506    None => DigestIf!("\\bibname")?,
507  };
508  if let Some(title) = title_opt {
509    whatsit.set_property("titlefont", title.get_font()?.unwrap());
510    whatsit.set_property("title", title);
511  }
512  {
513    whatsit.set_property("bibstyle", lookup_value("BIBSTYLE"));
514    whatsit.set_property("citestyle", lookup_value("CITE_STYLE"));
515  }
516  // And prepare for the likely nonsense that appears within bibliographies
517  ResetCounter!("enumiv");
518  Ok(())
