342  // We've got the same problem as LaTeX: Lather, Rinse, Repeat.
343  // It would be nice to know the bib info at digestion time
344  //  * whether author lists will collapse
345  //  * whether there are "a","b".. extensions on the year.
346  // We could process the bibliography first, (IF it is a separate *.bib!)
347  // but won't know which entries are included (and so can't resolve the a/b/c..)
348  // until we've finished looking at (all of) the source(s) that will refer to them!
349  //
350  // We can do this in 2 passes, however
351  //  (1) convert (latexml) both the source document(s) and the bibliography
352  //  (2) extract the required bibitems and integrate (latexmlpost) it into the documents.
353  // [Note that for mult-document sites, step (2) becomes 2 stages: scan and integrate]
354  //
355  // Here's the general layout.
356  //   <ltx:cite> contains everything that the citations produce,
357  //     including parens, pre-note, punctunation that precede the <ltx:bibcite>
358  //     and punctuation, post-note, parens, that follow it.
359  //   <ltx:bibcite show="string" bibrefs="keys" sep="" yysep="">phrases</ltx:bibcite>
360  //     encodes the actual citation.
361  //
362  //     bibrefs : lists the bibliographic keys that will be used
363  //     show    : gives the pattern for formatting using data from the bibliography
364  //       It can contain:
365  //         authors or fullauthors
366  //         year
367  //         number
368  //         phrase1,phrase2,... selects one of the phrases from the content of the <ltx:bibref>
369  //     This format is used as follows:
370  //       If author and year is present, and a subset of the citations share the same authors,
371  //         then the format is used, but the year is repeated for each citation in the subset,
372  //         as a link to the bib entry.
373  //       Otherwise, the format is applied to each entry.
374  //
375  // The design is intended to support natbib, as well as plain LaTeX.

449pub(crate) fn adjust_backmatter_element(document: &mut Document, whatsit: &Whatsit) -> Result<()> {
450  let asif_opt =
451    if let Some(Stored::String(asif_sym)) = whatsit.get_property("backmatterelement").as_deref() {
452      Some(arena::to_string(*asif_sym))
453    } else {
454      None
455    };
456  // Note: We allocate a string here, since
457  // it looks like arena::with can deadlock with find_insertion_point
458  // we may need a find_insertion_point_sym to avoid that...
459  if let Some(asif) = asif_opt {
460    let point = document.find_insertion_point(&asif, None)?;
461    document.set_node(&point);
462  }
463  Ok(())
464}

466// Since SOME people seem to write bibliographies w/o \bibitem,
467// just blank lines between apparent entries,
468// Making \par do a \bibitem{} works, but screws up valid
469// bibliographies with blank lines!
470// So, let's do some redirection!
471fn setup_pseudo_bibitem() -> Result<()> {
472  Let!("\\save@bibitem", "\\bibitem");
473  Let!("\\save@par", "\\par");
474  Let!("\\save@backbackslash", "\\\\");
475  Let!("\\bibitem", "\\restoring@bibitem");
476  Let!("\\par", "\\par@in@bibliography");
477  Let!("\\\\", "\\par@in@bibliography");
478  Let!("\\vskip", "\\vskip@in@bibliography");
479  // Moreover some people use \item instead of \bibitem
480  Let!("\\item", "\\item@in@bibliography");
481  // And protect from redefinitions.
482  Let!("\\newblock", "\\lx@bibnewblock");
483  // Risky, but when bibliography immediatesly starts with text (no implied \par)
484  if let Some(token) = gullet::read_non_space()? {
485    gullet::unread_one(token);
486    if !token.is_executable() {
487      gullet::unread_one(T_CS!("\\par"));
488    }
489  }
490  Ok(())
491}

500fn begin_bibliography_clean(whatsit: &mut Whatsit) -> Result<()> {
501  // Check if \bibsection is defined and try to decipher it.
502  // Expecting something like \section*{sometext}
503  // Perl: beginBibliography_clean in latex_constructs.pool.ltxml
504  let mut bibtitle: Option<Tokens> = None;
505  if let Some(bs) = lookup_definition(&T_CS!("\\bibsection"))? {
506    if bs.is_expandable() {
507      if let Some(ExpansionBody::Tokens(expansion_toks)) = bs.get_expansion() {
508        let mut tokens = expansion_toks.clone().unlist();
509        if !tokens.is_empty() {
510          let bibunitmap: &[(&str, &str)] = &[
511            ("\\part", "ltx:part"), ("\\chapter", "ltx:chapter"),
512            ("\\section", "ltx:section"), ("\\subsection", "ltx:subsection"),
513            ("\\subsubsection", "ltx:subsubsection"), ("\\paragraph", "ltx:paragraph"),
514            ("\\subparagraph", "ltx:subparagraph"),
515          ];
516          let first_cs = tokens.remove(0).to_string();
517          if let Some((_, unit)) = bibunitmap.iter().find(|(cs, _)| *cs == first_cs) {
518            state::assign_mapping("BACKMATTER_ELEMENT", "ltx:bibliography", Some(arena::pin(unit)));
519            // Strip * if present
520            if !tokens.is_empty() && tokens[0].to_string() == "*" {
521              tokens.remove(0);
522            }
523            if !tokens.is_empty() {
524              bibtitle = Some(Tokens::new(tokens));
525            }
526          }
527        }
528      }
529    }
530  }
531
532  note_backmatter_element(whatsit, "ltx:bibliography");
533  // Try to compute a reasonable, but unique ID;
534  // relative to the document's ID, if any.
535  // But also, if there are multiple bibliographies,
536  let bibnumber = 1 + lookup_int("n_bibliographies");
537  assign_value("n_bibliographies", bibnumber, Some(Scope::Global));
538  let mut docid: String = Expand!(T_CS!("\\thedocument@ID")).to_string();
539  if !docid.is_empty() {
540    docid += ".";
541  }
542  let bibid = s!("{}bib{}", docid, radix::radix_alpha(bibnumber - 1));
543  DefMacro!(T_CS!("\\thebibliography@ID"), None, T_OTHER!(&bibid), scope => Some(Scope::Global));
544  whatsit.set_property("id", bibid);
545  let title_opt = if let Some(bt) = bibtitle {
546    Some(Digest!(bt)?)
547  } else {
548    match DigestIf!("\\refname")? {
549      Some(v) => Some(v),
550      None => DigestIf!("\\bibname")?,
551    }
552  };
553  if let Some(title) = title_opt {
554    whatsit.set_property("titlefont", title.get_font()?.unwrap());
555    whatsit.set_property("title", title);
556  }
557  if let Some(bs) = lookup_value("BIBSTYLE") {
558    whatsit.set_property("bibstyle", bs);
559  }
560  if let Some(cs) = lookup_value("CITE_STYLE") {
561    whatsit.set_property("citestyle", cs);
562  }
563  // And prepare for the likely nonsense that appears within bibliographies
564  ResetCounter!("enumiv");
565  Ok(())
566}

568// Perl: $BIBSTYLES hash — maps bib style names to (citestyle, sort) pairs
569fn lookup_bibstyle_params(style: &str) -> Option<(&'static str, &'static str)> {
570  match style {
571    "plain"    => Some(("numbers", "true")),
572    "unsrt"    => Some(("numbers", "false")),
573    "alpha"    => Some(("AY",      "true")),
574    "abbrv"    => Some(("numbers", "true")),
575    "plainnat" => Some(("numbers", "true")),
576    "unsrtnat" => Some(("numbers", "false")),
577    "alphanat" => Some(("AY",      "true")),
578    "abbrvnat" => Some(("numbers", "true")),
579    _ => None,
580  }
581}

583// Perl: setBibstyle($style) — set BIBSTYLE, CITE_STYLE, CITE_SORT
584fn set_bibstyle(style: &str) {
585  assign_value("BIBSTYLE", arena::pin(style), None);
586  if let Some((cs, so)) = lookup_bibstyle_params(style) {
587    assign_value("CITE_STYLE", arena::pin(cs), None);
588    assign_value("CITE_SORT", arena::pin(so), None);
589  }
