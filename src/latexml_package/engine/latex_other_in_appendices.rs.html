8/// Port of Perl's `latexChangeCase` function.
9/// Applies Unicode case conversion (not TeX uccode/lccode tables) to tokens.
10/// Converts CC_SPACE to T_SPACE (matching latex3 behavior).
11/// Handles \protect + excluded CS tokens (text_case_exclude mapping).
12fn lx_change_case_tokens(req_case: &str, tokens: &Tokens) -> Result<Vec<Token>> {
13  // Match Perl's readingFromMouth($tokens, ...) behavior:
14  // create an empty mouth, then unread the tokens so they're read from the pushback.
15  // This avoids the endline-char trailing space that would appear if we converted
16  // tokens to a string and created a Mouth from that string.
17  let mouth = Mouth::new("", None)?;
18  gullet::open_mouth(mouth, false);
19  gullet::unread(tokens.clone());
20  let result = lx_read_and_change_case(req_case)?;
21  gullet::close_mouth(true)?;
22  Ok(result)
23}
24
25fn lx_read_and_change_case(req_case: &str) -> Result<Vec<Token>> {
26  let mut result = vec![];
27  let mut in_math = false;
28  let mut is_upper = req_case == "upper" || req_case == "sentence" || req_case == "title";
29  loop {
30    let tok = match gullet::read_x_token(Some(false), false, None)? {
31      None => break,
32      Some(t) => t,
33    };
34    let cc = tok.get_catcode();
35    if cc == Catcode::MATH {
36      in_math = !in_math;
37      result.push(tok);
38    } else if in_math {
39      result.push(tok);
40    } else if cc == Catcode::LETTER || cc == Catcode::OTHER {
41      // Compute new case outside the with_str borrow to avoid arena RefCell conflict
42      let new_str: String = tok.with_str(|s| {
43        if is_upper {
44          s.chars().flat_map(|c| c.to_uppercase()).collect()
45        } else {
46          s.chars().flat_map(|c| c.to_lowercase()).collect()
47        }
48      });
49      let changed = tok.with_str(|s| s != new_str.as_str());
50      let new_tok = if changed { Token::new(new_str, cc) } else { tok };
51      result.push(new_tok);
52      if req_case == "sentence" || req_case == "title" {
53        is_upper = false;
54      }
55    } else if cc == Catcode::SPACE {
56      result.push(T_SPACE!()); // HACK: match Perl/latex3 latexChangeCase
57      if req_case == "title" {
58        is_upper = true;
59      }
60    } else if cc == Catcode::CS && tok.with_str(|s| s == "\\protect") {
61      // Handle \protect + next token: check text_case_exclude
62      if let Some(next_tok) = gullet::read_token()? {
63        // Trim trailing space to handle both "\NoCaseChange " (DeclareRobustCommand) and
64        // "\@ensuremath" (manual definition) patterns
65        let next_key = next_tok.with_str(|s| s.trim_end().to_string());
66        if lookup_mapping("text_case_exclude", &next_key).is_some() {
67          // Protected: read optional and required arg, preserve unchanged
68          let opt = gullet::read_optional(None)?;
69          let arg = gullet::read_arg(ExpansionLevel::Off)?;
70          result.push(tok);       // \protect
71          result.push(next_tok);  // \cs_space
72          if let Some(opt_tokens) = opt {
73            // Optional arg gets case-changed too (per Perl)
74            let converted = lx_change_case_tokens(req_case, &opt_tokens)?;
75            result.push(T_OTHER!("["));
76            result.extend(converted);
77            result.push(T_OTHER!("]"));
78          }
79          result.push(T_BEGIN!());
80          result.extend(arg.unlist()); // required arg unchanged
81          result.push(T_END!());
82        } else if let Some(changed) =
83          lookup_mapping(if is_upper { "text_uppercase" } else { "text_lowercase" }, &next_key)
84        {
85          // Map this CS to its case-changed form
86          if let Stored::Token(changed_tok) = changed {
87            result.push(changed_tok);
88          } else {
89            result.push(tok);
90            result.push(next_tok);
91          }
92          if req_case == "sentence" || req_case == "title" {
93            is_upper = false;
94          }
95        } else {
96          result.push(tok);
97          result.push(next_tok);
98        }
99      }
100    } else {
101      result.push(tok);
102    }
103  }
104  Ok(result)
105}
