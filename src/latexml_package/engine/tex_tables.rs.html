413pub fn alignment_bindings(
414  template: Template,
415  mode: String,
416  properties: SymHashMap<Stored>,
417  xml_attributes: HashMap<String, String>,
418) {
419  let mode = if mode.is_empty() {
420    state::lookup_string("MODE")
421  } else {
422    mode
423  };
424  let is_math = mode.ends_with("math");
425  let (container, rowtype, coltype) = if is_math {
426    ("ltx:XMArray", "ltx:XMRow", "ltx:XMCell")
427  } else {
428    ("ltx:tabular", "ltx:tr", "ltx:td")
429  };
430  let alignment = Alignment::new(AlignmentConfig {
431    template: Some(template),
432    open_container: Rc::new(|document, props| {
433      document
434        .open_element(container, Some(props), None)
435        .map(Option::Some)
436    }),
437    close_container: Rc::new(|document| document.close_element(container)),
438    open_row: Rc::new(|document, props| {
439      document
440        .open_element(rowtype, Some(props), None)
441        .and(Ok(()))
442    }),
443    close_row: Rc::new(|document| document.close_element(rowtype)),
444    open_column: Rc::new(|document, props| {
445      document
446        .open_element(coltype, Some(props), None)
447        .map(Option::Some)
448    }),
449    close_column: Rc::new(|document| document.close_element(coltype)),
450    is_math,
451    properties,
452    xml_attributes,
453  });
454  assign_alignment(alignment, None);
455  // Debug("Halign $alignment: New " . $template->show) if $LaTeXML::DEBUG{halign};
456  state::let_i(
457    &T_MATH!(),
458    &if is_math {
459      T_CS!("\\lx@dollar@in@mathmode")
460    } else {
461      T_CS!("\\lx@dollar@in@textmode")
462    },
463    None,
464  );
465}

467pub fn digest_alignment_body(whatsit: &mut Whatsit) -> Result<()> {
468  // Now read & digest the body.
469  // Note that the body MUST end with a \cr, and that we've made Special Arrangments
470  // with \alignment@cr to recognize the end of the \halign
471  local_align_group_count(0);
472  let alignment_stored = if let Some(alignment) = lookup_alignment() {
473    alignment
474  } else {
475    Error!(
476      "missing",
477      "alignment",
478      "There is no open alignment structure here"
479    );
480    return Ok(());
481  };
482  local_reading_alignment(&alignment_stored);
483  whatsit.set_property("alignment", Stored::Digested(alignment_stored.clone()));
484
485  // Debug!("Halign {}: BODY Processing...",alignment) if $LaTeXML::DEBUG{halign};
486  let mut lastwascr = false;
487  let mut reversion: Vec<Token> = Vec::new();
488  let mut creversion: Vec<Token> = Vec::new();
489  let alignment_cell = alignment_stored.alignment_cell().unwrap();
490  loop {
491    let (cell_opt, next, vtype, hidden) = digest_alignment_column(alignment_cell, lastwascr)?;
492    //     Debug("Halign $alignment: BODY got CELL"
493    //         . "[" . $alignment->currentRowNumber . "," . $alignment->currentColumnNumber . "]"
494    //         . ToString($cell) . " ended at " . Stringify($next)) if $LaTeXML::DEBUG{halign};
495
496    if let Some(cell) = cell_opt {
497      reversion.extend(
498        trim_column_template(alignment_cell.borrow_mut(), p_revert(cell.clone())?).unlist(),
499      );
500      creversion.extend(
501        trim_column_template(alignment_cell.borrow_mut(), c_revert(cell.clone())?).unlist(),
502      );
503      extract_alignment_column(alignment_cell.borrow_mut(), cell)?;
504    } else {
505      // Debug("Halign $alignment: BODY DONE!") if $LaTeXML::DEBUG{halign};
506      break;
507    }
508    lastwascr = false;
509    if (vtype.is_none() || vtype.as_ref().unwrap().is_empty())
510      && (next.is_none() || next == Some(T_END!()) || next == Some(T_CS!("\\lx@close@alignment")))
511    {
512      // End of alignment
513      alignment_cell.borrow_mut().end_row()?;
514      break;
515    } else if vtype.as_deref() == Some("align") {
516      alignment_cell.borrow_mut().end_column()?;
517      if !hidden {
518        reversion.push(next.unwrap()); // and record the &
519        creversion.push(next.unwrap()); // and record the &
520      }
521    } else if vtype.as_deref() == Some("insert") {
522      alignment_cell.borrow_mut().end_column()?;
523    } else if vtype.as_deref() == Some("cr") || vtype.as_deref() == Some("crcr") {
524      alignment_cell.borrow_mut().end_row()?;
525      if !hidden {
526        reversion.push(next.unwrap());
527        creversion.push(next.unwrap());
528      } else if vtype.as_deref() == Some("cr") {
529        let arg_toks = gullet::read_arg(ExpansionLevel::Off)?;
530        let arg = stomach::digest(arg_toks)?;
531        reversion.extend(p_revert(arg.clone())?.unlist());
532        creversion.extend(c_revert(arg)?.unlist());
533      } else if vtype.as_deref() == Some("crcr") {
534      }
535      lastwascr = true;
536    } else if let Some(next_tok) = next {
537      // Note, in case next is \crcr
538      Error!("unexpected", next_tok, s!("Column ended with {next_tok}"));
539    }
540  }
541  alignment_cell.borrow_mut().end_row()?;
542  alignment_cell
543    .borrow_mut()
544    .set_reversion(Tokens!(reversion));
545  alignment_cell
546    .borrow_mut()
547    .set_content_reversion(Tokens!(creversion));
548  //   Debug("Halign $alignment: BODY DONE!\n"
549  //       . "=> " . join(',', map { Stringify($_); } @reversion)) if $LaTeXML::DEBUG{halign};
550  expire_align_group_count();
551  expire_reading_alignment();
552  Ok(())
553}

555// Read & digest an alignment column's data,
556// accommodating the current template and any special cs's
557// Returns the column's digested boxes, the ending token, and it's alignment type.
558type DigestedColumn = Result<(Option<Digested>, Option<Token>, Option<String>, bool)>;
559pub fn digest_alignment_column(alignment: &RefCell<Alignment>, lastwascr: bool) -> DigestedColumn {
560  new_local_box_list();
561  let ismath = lookup_bool("IN_MATH");
562  // Scan for leading \omit, skipping over (& saving) \hline.
563  //   Debug("Halign $alignment: COLUMN starting scan "
564  //       . "(" . ($ismath ? " math" : " text") . ")") if $LaTeXML::DEBUG{halign};
565  let mut last_token: Option<Token> = None;
566  let mut spanning = false;
567  loop {
568    // Outer loop; collects 1 column (possibly multiple spans) return from within!
569    // Scan till we get something NOT \omit, \noalign
570    while let Some(xtoken) = gullet::read_x_token(Some(false), false, None)? {
571      last_token = Some(xtoken);
572      let token = last_token.as_ref().unwrap();
573      // Skip leading space. Skip \par or blank line(?). Or \crcr following a \cr
574      if *token == T_SPACE!()
575        || *token == T_CS!("\\par")
576        || (lastwascr && (*token == T_CS!("\\crcr") || *token == T_CS!("\\lx@hidden@crcr")))
577      {
578      } else if *token == T_CS!("\\omit") {
579        // \omit removes template for this column.
580        //         Debug("Halign $alignment: OMIT at " . Stringify($token)) if
581        // $LaTeXML::DEBUG{halign};
582        if !alignment.borrow().is_in_row() {
583          alignment.borrow_mut().start_row(false)?;
584        }
585        alignment.borrow_mut().omit_next_column();
586      } else if *token == T_CS!("\\noalign") {
587        // \puts something in vertical list
588        // Debug("Halign $alignment: noalign at " . Stringify($token)) if $LaTeXML::DEBUG{halign};
589        if alignment.borrow().is_in_row() {
590          alignment.borrow_mut().end_row()?;
591        }
592        alignment.borrow_mut().start_column(true)?;
593        alignment.borrow_mut().last_column();
594        let next_arg = gullet::read_arg(ExpansionLevel::Off)?;
595        let r = stomach::digest(next_arg)?;
596        alignment.borrow_mut().end_row()?;
597        expire_local_box_list();
598        return Ok((Some(r), Some(T_CS!("\\cr")), some!("cr"), false)); // Pretend this is a whole
599      // row???
600      } else if *token == T_CS!("\\lx@hidden@noalign") {
601        // \puts something in vertical list
602        //         Debug("Halign $alignment: COLUMN invisible noalign") if $LaTeXML::DEBUG{halign};
603        let invoked = stomach::invoke_token(token)?;
604        extend_box_list(invoked);
605      } else {
606        break;
607      }
608    }
609    //     Debug("Halign $alignment: COLUMN end scan at " . Stringify($token)) if
610    // $LaTeXML::DEBUG{halign};
611    if last_token.is_none()
612      || last_token == Some(T_END!())
613      || last_token == Some(T_CS!("\\lx@close@alignment"))
614    {
615      expire_local_box_list();
616      return Ok((None, last_token, None, false));
617    }
618    // Next column, unless spanning (then combine columns)
619    if spanning {
620      spanning = false;
621      alignment.borrow_mut().next_column()?;
622    } else {
623      alignment.borrow_mut().start_column(false)?;
624    }
625    // Push before template,  Marker and put the token back
626    let to_unread = Tokens!(
627      alignment.borrow_mut().get_column_before(),
628      T_MARKER!("before-column"),
629      last_token.unwrap()
630    );
631    // eprintln!("Halign: COLUMN preload at {}", to_unread.stringify());
632    gullet::unread(to_unread);
633    while let Some(token) = gullet::read_x_token(Some(false), false, None)? {
634      if let Some((_atoken, vtype, hidden)) = gullet::is_column_end(&token) {
635        if vtype == "span" {
636          // next column, but continue accumulating
637          // Debug("Halign $alignment: COLUMN span") if $LaTeXML::DEBUG{halign};
638          spanning = true;
639          break;
640        } else {
641          // Debug("Halign $alignment: COLUMN ended with " . Stringify($token) . "\n"
642          //     . "  => " . ToString(List(@LaTeXML::LIST))) if $LaTeXML::DEBUG{halign};
643          let current_list = expire_local_box_list();
644          let mut out_list = List::new(current_list);
645          out_list.mode = Some(if ismath { TexMode::Math } else { TexMode::Text });
646          return Ok((
647            Some(out_list.into()),
648            Some(token),
649            Some(String::from(vtype)),
650            hidden,
651          ));
652        }
653      // DG: Note this block is commented out as clippy warned it has the exact identical logic
654      // as the "all other cases" else that follows it
655      //
656      // } else if token == T_CS!("\\lx@hidden@noalign") { //  \puts something in vertical list
657      //   // Debug("Halign $alignment: COLUMN invisible noalign") if $LaTeXML::DEBUG{halign};
658      //   let invoked = stomach.invoke_token(&token)?;
659      //   stomach.box_list.extend(invoked.into_iter());
660      } else {
661        // Else, we're getting some actual content for the column
662        // eprintln!("Halign: COLUMN invoking {}", token.stringify());// if $LaTeXML::DEBUG{halign};
663        let invoked = stomach::invoke_token(&token)?;
664        extend_box_list(invoked);
665        // eprintln!("Halign: COLUMN {} ==> {}",token.stringify(),
666        // List::new(stomach.box_list.clone()).stringify()); //       if
667        // $LaTeXML::DEBUG{halign};
668      }
669    }
670  }
671}

673// This attempts to trim off the column template parts from contents of the full column,
674// leaving only the author supplied part for a sensible reversion.
675// It's not nearly clever enough, given that macros can be in the template,
676// but works surprisingly well so far.
677// A better alternative might be based on sneaking some Marker tokens/boxes through
678// but they would likely interfere with the macros tehmselves.
679pub fn trim_column_template(mut alignment: RefMut<Alignment>, tokens: Tokens) -> Tokens {
680  if let Some(row) = alignment.current_row() {
681    if row.is_pseudo() {
682      return tokens;
683    }
684  }
685  let mut pre = alignment.get_column_before().unlist();
686  let mut post = alignment.get_column_after().unlist();
687  //   Debug("Halign $alignment: COLUMN Compare:\n"
688  //       . "  Column: " . ToString(Tokens(@tokens)) . "\n"
689  //       . "  Before: " . ToString(Tokens(@pre)) . "\n"
690  //       . "  After : " . ToString(Tokens(@post)) . "\n") if $LaTeXML::DEBUG{halign};
691  let mut tks_vec = tokens.unlist();
692  while !pre.is_empty() && !tks_vec.is_empty() {
693    let t = pre.remove(0);
694    if let Some(tks_first) = tks_vec.first() {
695      if t == *tks_first {
696        tks_vec.remove(0);
697      }
698    }
699  }
700  while !post.is_empty() && !tks_vec.is_empty() {
701    let t = post.pop().unwrap();
702    if let Some(tks_last) = tks_vec.last() {
703      if t == *tks_last {
704        tks_vec.pop();
705      }
706    }
707  }
708  //   Debug("  Trimmed: " . ToString(Tokens(@tokens))) if $LaTeXML::DEBUG{halign};
709  Tokens::new(tks_vec)
710}
711// Given the boxes for an alignment cell,
712// extract & remove the various fills and rules from the ends to annotate the cell structure
713pub fn extract_alignment_column(
714  mut alignment: RefMut<Alignment>,
715  in_box: Digested,
716) -> Result<Digested> {
717  let mut boxes = VecDeque::new();
718  boxes.extend(in_box.unlist());
719  let is_math = lookup_bool("IN_MATH");
720  //Note: $n0,$n1 is a VERY round-about way of tracking the column spanning!
721  let n0 = lookup_int("alignmentStartColumn") as usize + 1;
722  let n1 = alignment.current_column_number();
723  let colspec = alignment.get_column(n0).unwrap();
724  let mut align = colspec.align.unwrap_or(Align::Left);
725  let mut border = String::new();
726  // Peel off any boxes from both sides until we get the "meat" of the column.
727  // from this we can establish borders, alignment and emptiness.
728  // But we, of course, immediately put them back...
729  let mut saveleft = VecDeque::new();
730  let mut saveright = VecDeque::new();
731  while let Some(front_box) = boxes.pop_front() {
732    match front_box.data() {
733      DigestedData::List(_) => {
734        for fbox in front_box.unlist().into_iter().rev() {
735          boxes.push_front(fbox);
736        }
737      },
738      _ if front_box.get_property("isFill").is_some() => {
739        align = Align::Right;
740        break;
741      },
742      _ if front_box.get_property("isVerticalRule").is_some() => {
743        border.push('l');
744      },
745      item
746        if front_box.get_property("isHorizontalRule").is_some()
747          || front_box.get_property("alignmentSkippable").is_some()
748          || front_box.get_property("isSpace").is_some()
749          || matches!(item, DigestedData::Comment(_))
750          || front_box.is_empty()? =>
751      {
752        saveleft.push_front(front_box)
753      },
754      _ => {
755        // put the box back, and terminate left side loop.
756        boxes.push_front(front_box);
757        break;
758      },
759    }
760  }
761  while let Some(last_box) = boxes.pop_back() {
762    match last_box.data() {
763      DigestedData::List(_) => {
764        for lbox in last_box.unlist().into_iter() {
765          boxes.push_back(lbox);
766        }
767      },
768      _ if last_box.get_property("isFill").is_some() => {
769        if align == Align::Right {
770          align = Align::Center
771        };
772        break;
773      },
774      _ if last_box.get_property("isVerticalRule").is_some() => {
775        border.push('r');
776      },
777      item
778        if last_box.get_property("isHorizontalRule").is_some()
779          || last_box.get_property("alignmentSkippable").is_some()
780          || last_box.get_property("isSpace").is_some()
781          || matches!(item, DigestedData::Comment(_))
782          || last_box.is_empty()? =>
783      {
784        saveright.push_front(last_box);
785      },
786      _ => {
787        // put the box back, and terminate right side loop.
788        boxes.push_back(last_box);
789        break;
790      },
791    }
792  }
793  if align != Align::Justify {
794    colspec.width = None;
795  }
796  // Replacing boxes with the fil padding & vertical rules stripped off
797  let mut final_boxes = Vec::from(saveleft);
798  final_boxes.extend(boxes);
799  final_boxes.extend(saveright);
800  let mut boxes_list = List::new(final_boxes);
801  boxes_list.mode = Some(if is_math {
802    TexMode::Math
803  } else {
804    TexMode::Text
805  });
806  let digested_out = Digested::from(boxes_list);
807  // record relevant info in the Alignment.
808  colspec.align = Some(align);
809  border = s!("{}{}", colspec.border, border);
810  colspec.border = border;
811  colspec.boxes = Some(digested_out.clone());
812  colspec.colspan = Some(n1 - n0 + 1);
813  //   if ($$alignment{in_tabular_head} || $$alignment{in_tabular_foot}) {
814  //     $$colspec{thead}{column} = 1; }
815  //   for (my $i = $n0 + 1 ; $i <= $n1 ; $i++) {
816  //     my $c = $alignment->getColumn($i);
817  //     $$c{skipped} = 1 if $c; }
818  //   Debug("Halign $alignment: INSTALL column " . join(',', map { $_ . "=" .
819  // ToString($$colspec{$_}); } sort keys %$colspec)) if $LaTeXML::DEBUG{halign};
820  Ok(digested_out)
821}

824// Support for \\[dim] .... TO BE WORKED OUT!
825// NOTE that this does NOT skip spaces before * or []!!!!!
826//  As if: \lx@alignment@newline OptionalMatch:* [Dimension]
827// Read arguments for \\, namely * and/or [Dimension]
828// BUT optionally do it while skipping spaces (latex style) or not (ams style)
829fn read_newline_args(skipspaces: bool) -> Result<(bool, Option<Tokens>)> {
830  if lookup_alignment().is_some() {
831    local_align_group_count(1000000);
832    if skipspaces {
833      gullet::skip_spaces()?;
834    }
835    let (mut star, mut optional) = (false, None);
836    let mut next_opt = gullet::read_token()?;
837    if next_opt == Some(T_OTHER!("*")) {
838      star = true;
839      if skipspaces {
840        gullet::skip_spaces()?;
841      }
842      next_opt = gullet::read_token()?;
843    }
844    if next_opt == Some(T_OTHER!("[")) {
845      optional = Some(gullet::read_until(&Tokens!(T_OTHER!("]")))?);
846      next_opt = None;
847    }
848    if let Some(next) = next_opt {
849      gullet::unread_one(next);
850    }
851    expire_align_group_count();
852    Ok((star, optional))
853  } else {
854    Err("read_newline_args should only be called with a proper 'Alignment' active in state".into())
855  }
