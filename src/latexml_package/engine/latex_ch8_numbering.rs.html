20  // explicitly during digestion (via a whatsit), we can't use the usual LaTeX counter mechanism.

25  DefPrimitive!("\\newcounter{}[]", sub[(cs, default_opt)] {
26    let default = if let Some(tks) = default_opt {
27      if !tks.is_empty() {
28        Expand!(tks)
29      } else {
30        Tokens!()
31      }
32    } else {
33      Tokens!()
34    };
35    let cs_expanded = &Expand!(cs).to_string();
36    NewCounter!(cs_expanded, &default.to_string());
37  });
38  DefPrimitive!("\\setcounter{}{Number}", sub[(cs, default)] {
39    let cs_expanded = &Expand!(cs).to_string();
40    SetCounter!(cs_expanded, default);
41  });
42  DefPrimitive!("\\addtocounter{}{Number}", sub[(cs,default)] {
43    let cs_expanded = &Expand!(cs).to_string();
44    AddToCounter!(cs_expanded, default);
45  });
46  DefPrimitive!("\\stepcounter{}",    sub[(cs)] {
47    let cs_expanded = &Expand!(cs).to_string();
48    StepCounter!(cs_expanded, false)?;
49  });
50  DefPrimitive!("\\refstepcounter{}", sub[(cs)] {
51    let cs_expanded = &Expand!(cs).to_string();
52    RefStepCounter!(cs_expanded, false)?;
53  });
