28  DefPrimitive!("\\@check@length DefToken", sub[(cs)] {
29    match lookup_definition(&cs)? {
30      None => {
31        let message = s!("'{}' is not a length; defining it now", cs.stringify());
32        Warn!("undefined", cs, message);
33        DefRegister!(cs, None, Dimension::new(0));
34      },
35      Some(defn) => if !defn.is_register() {
36        let message = s!("'{}' length was expected, got {:?} instead of register.",
37          cs.to_string(), defn.register_type());
38        Error!("misdefined", cs, message);
39      }
40    };
41  });
42
43  DefPrimitive!("\\newlength DefToken", sub[(cs)] {
44    DefRegister!(cs, None, Glue::new(0), allocate => "\\skip");
45    Ok(vec![])
46  });

48  DefPrimitive!("\\setlength {Variable}{Dimension}", sub[(variable,length)] {
49    if let ArgWrap::RegisterDefinition(dbox) = variable {
50      let (rtoken, params) = *dbox;
51      let defn = rtoken.to_register()
52        .expect("if a Variable parameter provides a token, it must have a Register definition.");
53      defn.set_value(length.into(), None, params);
54    }
55    Ok(Vec::new())
56  });
57  DefPrimitive!("\\addtolength {Variable}{Dimension}", sub[(variable,length)] {
58    if let ArgWrap::RegisterDefinition(dbox) = variable {
59      let (rtoken, params) = *dbox;
60      let defn = rtoken.to_register()
61        .expect("if a Variable parameter provides a token, it must have a Register definition.");
62      // TODO: can we avoid cloning the params?
63      let oldlength = defn.value_of(params.clone()).unwrap_or_default();
64      defn.set_value(oldlength.add(length), None, params);
65    }
66    Ok(Vec::new())
67  });

91        stored_map!("width" => length, "isSpace" => true));
92    }
