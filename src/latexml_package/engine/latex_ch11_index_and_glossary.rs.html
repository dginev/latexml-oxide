3/// Perl: addIndexPhraseKey — sets the `key` attribute on index/glossary phrase
4/// nodes from their text content, applying CleanIndexKey normalization.
5fn add_index_phrase_key(node: &mut Node) -> Result<()> {
6  if node.get_attribute("key").is_none() {
7    let text = node.get_content();
8    let key = clean_index_key(&text);
9    if !key.is_empty() {
10      node.set_attribute("key", &key)?;
11    }
12  }
13  Ok(())
14}
15
16/// Perl: CleanIndexKey — trim whitespace, remove trailing punctuation.
17fn clean_index_key(key: &str) -> String {
18  let key = key.trim();
19  // Remove trailing punctuation
20  key.trim_end_matches(['.', ',', ';']).to_string()
21}
