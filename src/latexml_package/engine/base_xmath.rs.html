281  // These are used within XMDual
282  // The XMDual represents both a content & presentation representation of some
283  // possibly exotic structure ("Transfix notation"),
284  // or just a somewhat complex presentation that corresponds (often) to a simpler
285  // applicative content structure.
286  // Invoking such a mathematical object to "arguments" requires that both the
287  // content & presentation branches contain those arguments.
288  // There will be an XMArg, with an ID, containing the actual markup, and an XMRef that referrs to
289  // it. The XMArg will usually be in the presentation branch (so that it inherits appropriate
290  // style), unless the arg is "hidden" (ie. semantic, but not displayed).
291  // This means that we don't know which one appears first! (See Package's dualize_arglist)
292  //
293  // To get a "proper id", we'll use a temporary label-like attribute (_xmkey)
294  // and establish an id and idref later.

321  // Connect up the XMRef/XMArg pairs (actually can be multiple XMRef's)
322  // We want to set the idref of the XMRef's to point to the id of the XMArg (or other XM element),
323  // but usually the XMRef is created first, and we want to let the referred to element
324  // get it's id computed by whatever means it prefers.
325  // so we have to work both ways (use state::to record associations, to avoid expensive xpath)
326  // Set id's on any non-XMRef nodes that have an _xmkey
327  // This gets a more natural ordering
328  Tag!("ltx:*", after_open_late => sub[document,node] {
329    if node.has_attribute("_xmkey") {
330      let qname = document::get_node_qname(node);
331      if (qname != arena::pin_static("ltx:XMRef")) &&
332        arena::with(qname, |qstr| qstr.starts_with("ltx:XM")) && !node.has_attribute("xml:id") {
333        document.generate_id(node, "")?;
334      }
335    }
336  });

338  Tag!("ltx:XMDual", after_close_late => sub[document,node] {
339    let mut ids  = HashMap::default();
340    let mut refs = Vec::new();
341    // Collect all children with _xmkey attribute
342    for mut n in document.findnodes("descendant::*[@_xmkey]", Some(node)) {
343      if document::with_node_qname(&n, |qname| qname == "ltx:XMRef")
344          && !n.has_attribute("idref") {
345        refs.push(n);    // we'll fill these in next
346      } else { // generate & record ids for all referenced noces
347        let key = n.get_attribute("_xmkey").unwrap();
348        if let Entry::Vacant(e) = ids.entry(key) {
349          document.generate_id(&mut n, "")?; // Generate id if none already.
350          e.insert(n.get_attribute_ns("id",XML_NS).unwrap_or_default());
351        }
352      }
353    }
354    for mut r in refs {                        // Now fill in the references
355      let r_xmkey = r.get_attribute("_xmkey").unwrap();
356      document.set_attribute(&mut r, "idref", ids.get(&r_xmkey).unwrap())?;
357      r.remove_attribute("_xmkey")?;
358    }
359  });
