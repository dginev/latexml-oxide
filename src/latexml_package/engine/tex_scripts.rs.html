5//======================================================================
6// Scripts are a bit of a strange beast, with respect to when the arguments
7// are processed, and what kind of object should be created.
8//
9// While scripts look like they take a normal TeX argument, they really
10// take the next BOX (AFTER expansion & digestion)!  Thus, while
11//   a^\frac{b}{c} and a^\mathcal{B}
12// DO work in TeX, other things like
13//   a^\sqrt{3} or a^\acute{b}
14// DO NOT! (Hint: consider the expansions)
15// Note that with
16//  \def\xyz{xyz}
17//   a^\xyz   =>  a^{x}yz
18// So, we try to mimic, but note that our boxes don't correspond 100% to TeX's
19//
20// Normally, sub/super scripts should be turned into a sort of postfix operator:
21// The parser will attach the script to the appropriate preceding object.
22// However, there are a few special cases involving empty boxes {}.
23// If the argument is an empty box $x^{}$, the whole script should just disappear.
24// If the PRECEDING box is {} (in ${}^{p}$, a sort of `floating' script should be created.
25// This may combine, in the parser, with the following object to generate
26// a prescript.
27
28// DG: Note: TeX.pool's isEmpty seems best reorganized as Digested::is_empty
29// implemented by each concrete data structure. That should now be the case.
30
31// Remember a "safe" way to test a script Whatsit.
32// Returns [ (FLOATING|POST) , (SUBSCRIPT|SUPERSCRIPT) ] or nothing
33pub fn is_script(object: &Digested) -> Option<(String, Catcode)> {
34  let box_opt = match object.data() {
35    DigestedData::List(obj) => obj.borrow().boxes.last().map(|v| Cow::Owned(v.clone())),
36    _ => Some(Cow::Borrowed(object)),
37  };
38  if let Some(boxobj) = box_opt {
39    if let DigestedData::Whatsit(ref obj) = boxobj.data() {
40      // careful w/alias in getCSName!
41      obj.borrow().get_definition().get_cs().with_cs_name(|name| {
42        SCRIPT_NAME_RE.captures(name).map(|cap| {
43          (
44            cap.get(1).map_or("", |m| m.as_str()).to_owned(),
45            if cap.get(2).map_or("", |m| m.as_str()) == "SUBSCRIPT" {
46              Catcode::SUB
47            } else {
48              Catcode::SUPER
49            },
50          )
51        })
52      })
53    } else {
54      None
55    }
56  } else {
57    None
58  }
59}

61// TODO: Something is really off with the Rust version of the data model for
62// Digested here, we keep having to *clone* incorrectly. The Perl expectation
63// was for something Rust deems highly illegal/unsafe: multiple owners of a
64// mutable reference to a Digested object. This needs to be disentangled
65// in the new codebase, so as to avoid both 1) cloning and
66// 2) mutably referencing the same Digested object from multiple unrelated pieces of code.

68fn script_handler(cc: Catcode) -> Result<Vec<Digested>> {
69  //
70  //   gullet::skip_spaces();
71  let font = lookup_font().unwrap();
72  if font.get_mathstyle().is_some() {
73    let mut putback = VecDeque::new();
74    let mut nscripts = 0;
75
76    let mut cs = if cc == Catcode::SUPER {
77      "\\lx@floating@superscript"
78    } else {
79      "\\lx@floating@subscript"
80    };
81    let mut prevscript = None;
82    let mut prevspace = false;
83    let mut base = None;
84    // Check preceding boxes to determine possible attachment (floating vs post),
85    // Note that this analysis has to be done now (or sometime like it) before grouping lists go
86    // away; and whether there are conflicting preceding scripts, which is an error
87    // Parsing is too late!
88    while let Some(prev) = { pop_box_list() } {
89      if prev.get_property_bool("isSpace") {
90        prevspace = true; // a space avoids double-scripts
91        putback.push_front(prev); // put back? assuming it will add rpadding to previous???
92        continue;
93      } else if prev.is_empty()? {
94        // If empty, the script floats, can't conflict, but don't put back
95        break;
96      } else if let Some(prevop) = is_script(&prev) {
97        if prevop.1 == cc {
98          // Whoops, duplicated; better use FLOATING
99          putback.push_front(prev);
100          let lcode = if prevop.1 == Catcode::SUPER {
101            "superscript"
102          } else {
103            "subscript"
104          };
105          if !prevspace {
106            Error!("unexpected", s!("double-{lcode}"), s!("Double {lcode}"));
107          }
108          cs = if cc == Catcode::SUPER {
109            "\\lx@floating@superscript"
110          } else {
111            "\\lx@floating@subscript"
112          };
113          break;
114        } else {
115          // Else, is OK (so far) assume POST (it will stack previous script)
116          prevscript = Some(prev.clone()); // we'll overlap the width of the previous.
117          putback.push_front(prev);
118          cs = if cc == Catcode::SUPER {
119            "\\lx@post@superscript"
120          } else {
121            "\\lx@post@subscript"
122          };
123        }
124        // if we hit a FLOATING script, terminate, as the floating empty group avoids double scripts
125        if prevop.0 == "FLOATING" {
126          break;
127        }
128        nscripts += 1;
129        if nscripts > 1 {
130          break;
131        }
132      } else {
133        //  We found something "normal", so assume we'll attach to it, and we're done.
134        base = Some(prev.clone());
135        putback.push_front(prev);
136        cs = if cc == Catcode::SUPER {
137          "\\lx@post@superscript"
138        } else {
139          "\\lx@post@subscript"
140        };
141        break;
142      }
143    }
144    extend_box_list(putback);
145    MergeFont!(scripted => true);
146    // Now, get following boxes (may have to process several tokens!)
147    let mut stuff = Vec::new();
148    while let Some(tok) = gullet::read_x_token(Some(false), false, None)? {
149      stuff = stomach::invoke_token(&tok)?;
150      if !stuff.is_empty() {
151        break;
152      }
153    }
154    if stuff.is_empty() {
155      Error!("expected", "{", "Missing sub/superscript argument"); //$gullet->showUnexpected);
156      stuff.push(Digested::default());
157    }
158    let script = stuff.remove(0); // ONLY the first box is the script!
159
160    if !script.is_empty()? {
161      let mut properties = {
162        stored_map!(
163          "isMath" => true,
164          "base"        => if let Some(b) = base { Stored::Digested(b) }
165            else { Stored::None },                      // for sizing/positioning
166          "scriptlevel" => get_script_level(),
167          "level"       => get_boxing_level()
168        )
169      };
170      if let Some(pvs) = prevscript {
171        properties.insert("prevscript", pvs.into());
172      }
173      if let Some(font) = script.get_font()? {
174        properties.insert("font", font.into());
175      }
176      let mut with_script = vec![Digested::from(Whatsit {
177        definition: lookup_definition(&T_CS!(cs))?.unwrap(),
178        args: vec![Some(script)],
179        properties,
180        // TODO:
181        // locator: stomach.get_gullet().get_locator(),
182        ..Whatsit::default()
183      })];
184      with_script.extend(stuff);
185      stuff = with_script;
186    }
187    assign_font(font, Some(Scope::Local)); // revert
188    Ok(stuff)
189  } else {
190    let c = if cc == Catcode::SUPER { '^' } else { '_' };
191    Error!(
192      "Unexpected",
193      c,
194      format!("Script {c} can only appear in math mode")
195    );
196    let placeholder = if cc == Catcode::SUPER {
197      T_SUPER!()
198    } else {
199      T_SUB!()
200    };
201    Ok(vec![Digested::from(Tbox::new(
202      arena::pin_char(c),
203      None,
204      None,
205      Tokens!(placeholder),
206      SymHashMap::default(),
207    ))])
208  }
209}

211// The `argument' to a sub or superscript will typically be processed as a box,
212// and either has braces, or is something that results in a single box.
213// When we revert these, we DON'T want to wrap extra braces around, because they'll accumulate;
214// at the least they're ugly; in some applications they affect "round trip" processing.
215// OTOH, direct use of \lx@post@superscript, etal, MAY need to have extra braces around them.
216// So, when reverting, we're going to a bit of extra trouble to make sure we have ONE set
217// of braces, but no extras!!  [Worry about lists of lists...]
218pub fn revert_script(script: &Digested) -> Result<Vec<Token>> {
219  let tokens = script.revert()?;
220  let mut ts = tokens.unlist();
221  // let mut level = 0;
222  if ts.len() > 1
223    && ts.first().unwrap().code == Catcode::BEGIN
224    && ts.last().unwrap().code == Catcode::END
225  {
226    Ok(ts)
227  } else {
228    let mut wrapped = vec![T_BEGIN!()];
229    wrapped.append(&mut ts);
230    wrapped.push(T_END!());
231    Ok(wrapped)
232  }
233}

235// Compute the 'advance' of this script.
236// can we do this before parsing? we can do the advance or something.... Hmmmm.
237// * Need to know scriptpos (mid or post) to determine position.
238// * need to know sub/super
239fn script_sizer(
240  script: &Digested,
241  base_opt: Option<&Stored>,
242  prev_opt: Option<&Stored>,
243  op: &str,
244  pos: &str,
245) -> Result<(Dimension, Dimension, Dimension)> {
246  // NOTE: Currently, the mathstyle is NOT reflected in the font of the script!!!!
247  // Or is it now ?????
248  // [unless it's different from the 'expected' style!!!]
249  let script_size = script.clone().get_size(None)?;
250  let (mut ws, mut hs, mut ds) = (
251    script_size.0.value_of() as f64,
252    script_size.1.value_of() as f64,
253    script_size.2.value_of() as f64,
254  );
255  ws *= 0.8;
256  hs *= 0.8;
257  ds *= 0.8; // HACK!@!!
258  let (wb, hb, db) = if let Some(Stored::Digested(ref base)) = base_opt {
259    let base_size = base.clone().get_size(None)?;
260    (
261      base_size.0.value_of() as f64,
262      base_size.1.value_of() as f64,
263      base_size.2.value_of() as f64,
264    )
265  } else {
266    let nominal_size = lookup_font().unwrap().get_nominal_size();
267    (
268      nominal_size.0.value_of() as f64,
269      nominal_size.1.value_of() as f64,
270      nominal_size.2.value_of() as f64,
271    )
272  };
273  let w;
274  let (mut h, mut d) = (0.0, 0.0);
275  // Fishing for the scriptpos on the base (if any)
276  let inferred_pos = if pos.is_empty() {
277    if let Some(Stored::Digested(ref base)) = base_opt {
278      let base_pos = base
279        .get_property("scriptpos")
280        .map(|s| s.to_string())
281        .unwrap_or_default();
282      if base_pos.is_empty() {
283        Cow::Borrowed("post")
284      } else {
285        Cow::Owned(base_pos)
286      }
287    } else {
288      Cow::Borrowed("post")
289    }
290  } else {
291    Cow::Borrowed("post")
292  };
293  if inferred_pos == "mid" {
294    w = (ws - wb).max(0.0); // as if max width of base & script
295    if op == "SUPERSCRIPT" {
296      h = hb + ds + hs;
297    } else {
298      d = db + hs + ds;
299    }
300  } else {
301    // as if max of width & prev script's width
302    let wp = if let Some(Stored::Digested(ref prev)) = prev_opt {
303      prev.get_width(None)?.unwrap_or_default().value_of() as f64
304    } else {
305      0.0
306    };
307    w = (ws - wp).max(0.0);
308    if op == "SUPERSCRIPT" {
309      h = hb + hs / 2.0;
310    } else {
311      d = hs / 2.0 + ds;
312    }
313  }
314  Ok((
315    Dimension::new_f64(w),
316    Dimension::new_f64(h),
317    Dimension::new_f64(d),
318  ))
319}
