26  DefMacro!("\\lx@ifundefined{}{}{}", sub[(name, if_token, else_token)] {
27    let cs = T_CS!(s!("\\{}", Expand!(name).to_string()));
28    if IsDefined!(&cs) {

37    let mut boxes = Vec::new();
38    while let Some(token) = gullet::read_x_token(None, false, None)? {
39      boxes = stomach::invoke_token(&token)?;
40      if boxes.is_empty() {
41        break;
42      }
43      while !boxes.is_empty() {
44        if match boxes[0].get_property("isSpace") {
45          Some(Cow::Borrowed(Stored::Bool(space_bool))) => *space_bool,
46          Some(Cow::Owned(Stored::Bool(ref space_bool))) => *space_bool,
47          _ => false,
48        } {
49          boxes = boxes[1..].to_vec();
50        } else {
51          break;
52        }
53      }
54      if !boxes.is_empty() {
55        break;
56      }
57    }
58    Ok(boxes)

127    sub[(keys_opt,tag_tks,attrs_opt,tokens)] {
128    // Digest this as if we're already in the document body!
129    let inpreamble = lookup_bool("inPreamble");
130    assign_value("inPreamble", false, None);
131    // Be careful since the contents may also want to add frontmatter
132    // (which should be inside or after this one!)
133    // So, we append this entry before digesting
134    let tag = tag_tks.to_string();
135    if let Some(keys) = keys_opt {
136      let known_key = if let Some(Stored::HashTagData(ref mut frnt)) = lookup_value("frontmatter") {
137        frnt.contains_key(&tag)
138      } else { false };

147      }

152      }

156      if let DigestedData::KeyVals(digested) = attr_kvs.be_digested()?.data() {
157        Some(digested.get_hash_digested())

159        None
160      }
161    } else {
162      None
163    };

165    wrapped_tokens.extend(tokens.unlist());
166    wrapped_tokens.push(T_END!());
167    let digested_tokens = stomach::digest(Tokens::new(wrapped_tokens))?;
168    let entry = (tag.clone(), attrs_digested, digested_tokens);

171        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
172        _ => fatal!(TexPool, Expected, "Global TeX Frontmatter hash was not available, should never happen"),
173      };

175      f_entry.push(entry);

235  // Tags & Titles
236  // The reference numbers, titles, captions etc, for various objects have
237  // different styling conventions, and the styling various depending on context.
238  // We'll use ltx:tags as a container for the various forms of ltx:tag with different @role's.
239  // The role=refnum form is simply formatted by \the<counter> and used by \ref;
240  // An ltx:tag w/o @role are for the numbers, often formatted differently, which
241  // appear alongside the object; Such a tag also may be embedded within the title or caption.
242  // Cross-references automatically generated by LaTeXML benefit from a bit more context:
243  // these are the role=typerefnum forms.
244  // Additional forms are needed for bibliographies, hyperref's autoref, etc.
245  // An additional complication is that while the "type" determines the formatting
246  // of the various forms, some types (eg. theorems) share the same counter.
247  // LaTeX defines this handling on an adhoc basis; defines \fnum@table, \fnum@figure for some types
248  // but \labelenumi, etc for others.
249
250  // This section synthesizes a more uniform support for reference numbers,
251  // references to reference numbers, title formatting etc.
252  // It allows you to customize each of the forms for each type encountered.

255  // This collects up the various declared ltx:tag's into an ltx:tags
256  DefMacro!("\\lx@make@tags {}", sub[(ttype)] {
257    let mut tags = Vec::new();
258    if let Some(Stored::HashStored(formatters)) =
259      lookup_value("type_tag_formatter") {
260      let keys_sym : Vec<_> = formatters.keys().copied().collect();
261      let mut sorted_keys : Vec<String> = arena::with_many(&keys_sym, |keys| {
262        keys.into_iter().map(str::to_owned).collect()
263      });
264      sorted_keys.sort();
265      for role in sorted_keys.iter() {
266        let formatter_opt = match formatters.get(role) {
267          Some(Stored::Token(t)) => Some(*t),
268          Some(Stored::String(sym)) => Some(Token { text: *sym, code: Catcode::CS }),
269          _ => None
270        };
271        if let Some(formatter_token) = formatter_opt {
272          tags.push(Invocation!(T_CS!("\\lx@tag@intags"),
273            vec![
274              Tokens!(T_OTHER!(role)),
275              build_invocation(formatter_token, vec![Some(ttype.clone())])?
276            ])
277          );
278        }
279      }
280    }
281
282    let mut lx_tags = vec![T_CS!("\\lx@tags"), T_BEGIN!()];
283    for invoked_tag in tags {
284      lx_tags.append(&mut invoked_tag.unlist());
285    }
286    lx_tags.push(T_END!());
287    Ok(Tokens::new(lx_tags))
288  });

465/// Insert FrontMatter into document, if not already added
466pub fn insert_frontmatter(document: &mut Document) -> Result<()> {
467  if lookup_bool("frontmatter_done") {
468    return Ok(());
469  }
470  let frontmatter_elements_set: HashSet<String> =
471    FRONTMATTER_ELEMENTS.iter().map(ToString::to_string).collect();
472
473  // Get frontmatter hash
474  let frontmatter_ref = match state::lookup_value("frontmatter") {
475    Some(Stored::HashTagData(frnt)) => frnt,
476    _ => return Ok(()),
477  };
478  let set_keys: Vec<String> = frontmatter_ref.keys().cloned().collect();
479
480  // If doc ONLY has abstract as frontmatter, defer until abstract's document location
481  if set_keys.len() == 1
482    && set_keys[0] == "ltx:abstract"
483    && !lookup_bool("frontmatter_deferred")
484  {
485    state::assign_value("frontmatter_deferred", true, Some(Scope::Global));
486    return Ok(());
487  }
488
489  state::assign_value("frontmatter_done", true, Some(Scope::Global));
490
491  // Remove frontmatter and replace with empty
492  let mut frontmatter = match state::remove_value("frontmatter") {
493    Some(Stored::HashTagData(frnt)) => frnt,
494    _ => return Ok(()),
495  };
496  state::assign_value(
497    "frontmatter",
498    Stored::HashTagData(HashMap::default()),
499    Some(Scope::Global),
500  );
501
502  // Order: first go through frontmatter_elements order, then any custom keys
503  let custom_keys: Vec<String> = frontmatter
504    .keys()
505    .filter(|key| !frontmatter_elements_set.contains(key.as_str()))
506    .map(ToString::to_string)
507    .collect();
508  let mut all_keys: Vec<String> =
509    FRONTMATTER_ELEMENTS.iter().map(ToString::to_string).collect();
510  all_keys.extend(custom_keys);
511
512  for key in &all_keys {
513    if let Some(list) = frontmatter.remove(key) {
514      // Dubious, but assures that frontmatter appears in text mode...
515      document.set_box_to_absorb(
516        Tbox::new(
517          *EMPTY_SYM,
518          lookup_font(),
519          None,
520          Tokens!(T_SPACE!()),
521          SymHashMap::default(),
522        )
523        .into(),
524      );
525      for (tag, attr, stuff) in list {
526        // Add a dedicated class for frontmatter notes
527        let attr = if tag == "ltx:note" {
528          let mut a = attr.unwrap_or_default();
529          let existing = a.get("class").cloned().unwrap_or_default();
530          let new_class = if existing.is_empty() {
531            "ltx_note_frontmatter".to_string()
532          } else {
533            s!("{existing} ltx_note_frontmatter")
534          };
535          a.insert("class".to_string(), new_class);
536          Some(a)
537        } else {
538          attr
539        };
540        document.open_element(&tag, attr, None)?;
541        document.absorb(&stuff, None)?;
542        let completed_node = document.close_element(&tag)?;
543        // Prune empty frontmatter elements (except ltx:rdf)
544        if tag != "ltx:rdf" {
545          if let Some(ref node) = completed_node {
546            if node.get_child_nodes().is_empty() {
547              document.remove_node(node.clone());
548            }
549          }
550        }
551      }
552      document.expire_box_to_absorb();
553    }
554  }
555  Ok(())
