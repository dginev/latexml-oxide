28    if IsDefined!(&cs) {

30    } else {

36  DefPrimitive!("\\lx@ignorehardspaces", {
37    let mut boxes = Vec::new();
38    while let Some(token) = gullet::read_x_token(None, false, None)? {
39      boxes = stomach::invoke_token(&token)?;
40      if boxes.is_empty() {
41        break;
42      }
43      while !boxes.is_empty() {
44        if match boxes[0].get_property("isSpace") {
45          Some(Cow::Borrowed(Stored::Bool(space_bool))) => *space_bool,
46          Some(Cow::Owned(Stored::Bool(ref space_bool))) => *space_bool,
47          _ => false,
48        } {
49          boxes = boxes[1..].to_vec();
50        } else {
51          break;
52        }
53      }
54      if !boxes.is_empty() {
55        break;
56      }
57    }
58    Ok(boxes)
59  });

120  // Add a new frontmatter item that will be enclosed in <$tag %attr>...</$tag>
121  // The content is the result of digesting $tokens.
122  // \\@add@frontmatter[keys]{tag}[attributes]{content}
123  // Perl: DEFERS processing by pushing \@add@frontmatter@now invocation
124  // into @at@begin@maketitle, which is digested at \maketitle time.
125  // This is critical for correct ordering: \author creates entry BEFORE
126  // \address/\email append to it.
127  DefPrimitive!("\\@add@frontmatter OptionalKeyVals {} OptionalKeyVals {}",
128    sub[(keys_opt,tag_tks,attrs_opt,tokens)] {
129    // Build invocation: \@add@frontmatter@now[keys]{tag}[attrs]{tokens}
130    let mut inv_tokens: Vec<Token> = vec![T_CS!("\\@add@frontmatter@now")];
131    // Pass keys if present
132    if let Some(ref keys) = keys_opt {
133      inv_tokens.push(T_OTHER!("["));
134      inv_tokens.extend(keys.revert()?.unlist());
135      inv_tokens.push(T_OTHER!("]"));
136    }
137    inv_tokens.push(T_BEGIN!());
138    inv_tokens.extend(tag_tks.unlist());
139    inv_tokens.push(T_END!());
140    // Pass attrs if present
141    if let Some(ref attrs) = attrs_opt {
142      inv_tokens.push(T_OTHER!("["));
143      inv_tokens.extend(attrs.revert()?.unlist());
144      inv_tokens.push(T_OTHER!("]"));
145    }
146    inv_tokens.push(T_BEGIN!());
147    inv_tokens.extend(tokens.unlist());
148    inv_tokens.push(T_END!());
149    let _ = state::push_value("@at@begin@maketitle", Stored::Tokens(Tokens::new(inv_tokens)));
150  });

152  // \@add@frontmatter@now — actually processes the frontmatter entry
153  // keys can have: replace (to replace the current entry, if any),
154  //                ifnew (only add if no previous entry)
155  DefPrimitive!("\\@add@frontmatter@now OptionalKeyVals {} OptionalKeyVals {}",
156    sub[(keys_opt,tag_tks,attrs_opt,tokens)] {
157    // Digest this as if we're already in the document body!
158    let inpreamble = lookup_bool("inPreamble");
159    assign_value("inPreamble", false, None);
160    // Be careful since the contents may also want to add frontmatter
161    // (which should be inside or after this one!)
162    // So, we append this entry before digesting (Perl comment from Base_Utility.pool.ltxml)
163    let tag = tag_tks.to_string();
164    if let Some(keys) = keys_opt {
165      let known_key = if let Some(Stored::HashTagData(ref mut frnt)) = lookup_value("frontmatter") {
166        frnt.contains_key(&tag)
167      } else { false };
168      if known_key && keys.has_key("replace") {
169        state::with_value_mut("frontmatter", |val_opt| {
170          if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt {
171            frnt.insert(tag.clone(), Vec::new());
172          }
173        });
174      }
175      if known_key && keys.has_key("ifnew") {
176        return Ok(Vec::new());
177      }
178    }
179
180    let attrs_digested = if let Some(attr_kvs) = attrs_opt {
181      if let DigestedData::KeyVals(digested) = attr_kvs.be_digested()?.data() {
182        Some(digested.get_hash_digested())
183      } else {
184        None
185      }
186    } else {
187      None
188    };
189    // Perl: Create entry FIRST, then digest content (so nested @add@to@ appends to this entry)
190    let placeholder = Digested::from(List::new(Vec::new()));
191    let entry = (tag.clone(), attrs_digested, placeholder);
192    state::with_value_mut("frontmatter", |val_opt| {
193      let frontmatter = match val_opt {
194        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
195        _ => fatal!(TexPool, Expected, "Global TeX Frontmatter hash was not available, should never happen"),
196      };
197      let f_entry = frontmatter.entry(tag.clone()).or_insert_with(Vec::new);
198      f_entry.push(entry);
199      Ok(())
200    })?;
201    // NOW digest the content — nested @add@to@frontmatter calls will see this entry as "last"
202    let mut wrapped_tokens = vec![T_BEGIN!()];
203    wrapped_tokens.extend(tokens.unlist());
204    wrapped_tokens.push(T_END!());
205    let digested_tokens = stomach::digest(Tokens::new(wrapped_tokens))?;
206    // Fill in the placeholder
207    state::with_value_mut("frontmatter", |val_opt| {
208      let frontmatter = match val_opt {
209        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
210        _ => return Ok::<(), latexml_core::Error>(()),
211      };
212      if let Some(list) = frontmatter.get_mut(&tag) {
213        if let Some(last) = list.last_mut() {
214          last.2 = digested_tokens;
215        }
216      }
217      Ok(())
218    })?;
219    AssignValue!("inPreamble", inpreamble);
220  },
221  before_digest => { stomach::bgroup(); },
222  after_digest => { let _ = stomach::egroup(); });

224  // Append a piece of data to an existing frontmatter item that is contained in <$tag>
225  // If $label is given, look for an item which has label=>$label,

228  // \@add@to@frontmatter{tag}[label]{content}
229  // Perl: defers processing by pushing \@add@to@frontmatter@now invocation
230  // into @at@begin@maketitle, which is digested at \maketitle time.
231  DefPrimitive!("\\@add@to@frontmatter {} [] {}", sub[(tag, label, tokens)] {
232    // Build invocation: \@add@to@frontmatter@now{tag}[label]{tokens}
233    let mut inv_tokens = vec![T_CS!("\\@add@to@frontmatter@now"), T_BEGIN!()];
234    inv_tokens.extend(tag.unwrap_or_default().unlist());
235    inv_tokens.push(T_END!());
236    if let Some(ref lbl) = label {
237      inv_tokens.push(T_OTHER!("["));
238      inv_tokens.extend(lbl.clone().unlist());
239      inv_tokens.push(T_OTHER!("]"));
240    }
241    inv_tokens.push(T_BEGIN!());
242    inv_tokens.extend(tokens.unwrap_or_default().unlist());
243    inv_tokens.push(T_END!());
244    let _ = state::push_value("@at@begin@maketitle", Stored::Tokens(Tokens::new(inv_tokens)));
245  });

247  // \@add@to@frontmatter@now{tag}[label]{content}
248  // Actually processes content into the frontmatter hash.
249  DefPrimitive!("\\@add@to@frontmatter@now {} [] {}",
250    sub[(tag_tks, label_opt, tokens)] {
251    let tag = tag_tks.unwrap().to_string();
252    let label = label_opt.as_ref().map(|l| l.to_string());
253    let inpreamble = lookup_bool("inPreamble");
254    assign_value("inPreamble", false, None);
255
256    // Digest the content tokens
257    let mut wrapped = vec![T_BEGIN!()];
258    wrapped.extend(tokens.unwrap_or_default().unlist());
259    wrapped.push(T_END!());
260    let datum = stomach::digest(Tokens::new(wrapped))?;
261
262    assign_value("inPreamble", inpreamble, None);
263
264    state::with_value_mut("frontmatter", |val_opt| {
265      let frontmatter = match val_opt {
266        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
267        _ => fatal!(TexPool, Expected, "Frontmatter hash missing"),
268      };
269      if let Some(ref lbl) = label {
270        // Look for existing item with matching label
271        if let Some(list) = frontmatter.get_mut(&tag) {
272          for item in list.iter_mut() {
273            let (_, ref iattr, _) = item;
274            if let Some(ref attrs) = iattr {
275              if attrs.get("label").map(|s| s.as_str()) == Some(lbl.as_str()) {
276                // Append datum to existing item
277                let (_, _, ref mut existing_content) = item;
278                let items = vec![existing_content.clone(), datum.clone()];
279                *existing_content = Digested::from(List::new(items));
280                return Ok(());
281              }
282            }
283          }
284        }
285      } else if let Some(list) = frontmatter.get_mut(&tag) {
286        // No label: append datum to last item in tag
287        if let Some(last) = list.last_mut() {
288          let (_, _, ref mut existing_content) = last;
289          let items = vec![existing_content.clone(), datum.clone()];
290          *existing_content = Digested::from(List::new(items));
291          return Ok(());
292        }
293      }
294      // New entry
295      let attrs = label.map(|l| {
296        let mut m = HashMap::default();
297        m.insert("label".to_string(), l);
298        m
299      });
300      let entry = (tag.clone(), attrs, datum);
301      frontmatter.entry(tag).or_insert_with(Vec::new).push(entry);
302      Ok(())
303    })?;
304  },
305  before_digest => { stomach::bgroup(); },
306  after_digest => { let _ = stomach::egroup(); });

365  // Tags & Titles
366  // The reference numbers, titles, captions etc, for various objects have
367  // different styling conventions, and the styling various depending on context.
368  // We'll use ltx:tags as a container for the various forms of ltx:tag with different @role's.
369  // The role=refnum form is simply formatted by \the<counter> and used by \ref;
370  // An ltx:tag w/o @role are for the numbers, often formatted differently, which
371  // appear alongside the object; Such a tag also may be embedded within the title or caption.
372  // Cross-references automatically generated by LaTeXML benefit from a bit more context:
373  // these are the role=typerefnum forms.
374  // Additional forms are needed for bibliographies, hyperref's autoref, etc.
375  // An additional complication is that while the "type" determines the formatting
376  // of the various forms, some types (eg. theorems) share the same counter.
377  // LaTeX defines this handling on an adhoc basis; defines \fnum@table, \fnum@figure for some types
378  // but \labelenumi, etc for others.
379
380  // This section synthesizes a more uniform support for reference numbers,
381  // references to reference numbers, title formatting etc.
382  // It allows you to customize each of the forms for each type encountered.
383  // The design reflects LaTeX needs, more than TeX, but support starts here!
384
385  // This collects up the various declared ltx:tag's into an ltx:tags
386  DefMacro!("\\lx@make@tags {}", sub[(ttype)] {
387    let mut tags = Vec::new();
388    if let Some(Stored::HashStored(formatters)) =
389      lookup_value("type_tag_formatter") {
390      let keys_sym : Vec<_> = formatters.keys().copied().collect();
391      let mut sorted_keys : Vec<String> = arena::with_many(&keys_sym, |keys| {
392        keys.into_iter().map(str::to_owned).collect()
393      });
394      sorted_keys.sort();
395      for role in sorted_keys.iter() {
396        let formatter_opt = match formatters.get(role) {
397          Some(Stored::Token(t)) => Some(*t),
398          Some(Stored::String(sym)) => Some(Token { text: *sym, code: Catcode::CS }),
399          _ => None
400        };
401        if let Some(formatter_token) = formatter_opt {
402          tags.push(Invocation!(T_CS!("\\lx@tag@intags"),
403            vec![
404              Tokens!(T_OTHER!(role)),
405              build_invocation(formatter_token, vec![Some(ttype.clone())])?
406            ])
407          );
408        }
409      }
410    }
411
412    let mut lx_tags = vec![T_CS!("\\lx@tags"), T_BEGIN!()];
413    for invoked_tag in tags {
414      lx_tags.append(&mut invoked_tag.unlist());
415    }
416    lx_tags.push(T_END!());
417    Ok(Tokens::new(lx_tags))
418  });

605/// Check if a token is "definable" — undefined or equivalent to `\relax`.
606///
607/// Port of Perl `isDefinable($token)` (Base_Utility.pool.ltxml L33-40).
608/// Returns true if the token can be (re)defined: either undefined, equivalent
609/// to `\relax`, or in 2.09 compatibility mode. Excludes `\relax` itself
610/// and tokens starting with `\end`.
611#[allow(dead_code)]
612pub fn is_definable(token: &Token) -> bool {
613  let meaning = state::lookup_meaning(token);
614  let relax_meaning = state::lookup_meaning(&T_CS!("\\relax"));
615  let has_no_meaning = meaning.is_none()
616    || meaning == relax_meaning
617    || state::lookup_bool("2.09_COMPATIBILITY");
618  if !has_no_meaning {
619    return false;
620  }
621  let mut name_ok = true;
622  token.with_cs_name(|name| {
623    if name == "relax" || name.starts_with("end") {
624      name_ok = false;
625    }
626    Ok::<(), latexml_core::common::error::Error>(())
627  }).ok();
628  name_ok
629}

631/// Split a token list by delimiter tokens, respecting brace nesting and math mode.
632///
633/// Port of Perl `SplitTokens($tokens, @delims)` (Base_Utility.pool.ltxml L106-132).
634/// Used for textual tokens — typically to split author lists. Does NOT split
635/// within braces `{...}` or math `$...$`.
636#[allow(dead_code)]
637pub fn split_tokens(tokens: &Tokens, delims: &[Token]) -> Vec<Tokens> {
638  let mut items: Vec<Vec<Token>> = Vec::new();
639  let mut current: Vec<Token> = Vec::new();
640  let token_list = tokens.clone().unlist();
641  let mut i = 0;
642  while i < token_list.len() {
643    let t = &token_list[i];
644    // Check if this token is a delimiter
645    if delims.iter().any(|d| t == d) {
646      items.push(current);
647      current = Vec::new();
648      i += 1;
649      continue;
650    }
651    // Brace group: collect everything inside, preserving nesting
652    if t.get_catcode() == Catcode::BEGIN {
653      current.push(*t);
654      let mut level = 1i32;
655      i += 1;
656      while i < token_list.len() && level > 0 {
657        let inner = &token_list[i];
658        if inner.get_catcode() == Catcode::BEGIN {
659          level += 1;
660        } else if inner.get_catcode() == Catcode::END {
661          level -= 1;
662        }
663        current.push(*inner);
664        i += 1;
665      }
666      continue;
667    }
668    // Math mode: collect until matching $
669    if t.get_catcode() == Catcode::MATH {
670      current.push(*t);
671      i += 1;
672      while i < token_list.len() {
673        let inner = &token_list[i];
674        current.push(*inner);
675        i += 1;
676        if inner.get_catcode() == Catcode::MATH {
677          break;
678        }
679      }
680      continue;
681    }
682    // Regular token
683    current.push(*t);
684    i += 1;
685  }
686  items.push(current);
687  items.into_iter().map(Tokens::new).collect()
688}

690/// Join token groups with a conjunction token between them.
691///
692/// Port of Perl `JoinTokens($conjunction, @things)` (Base_Utility.pool.ltxml L142-148).
693#[allow(dead_code)]
694pub fn join_tokens(conjunction: &Tokens, things: Vec<Tokens>) -> Tokens {
695  if things.is_empty() {
696    return Tokens::new(vec![]);
697  }
698  let mut result: Vec<Token> = Vec::new();
699  let mut first = true;
700  for thing in things {
701    if !first {
702      result.extend(conjunction.clone().unlist());
703    }
704    result.extend(thing.unlist());
705    first = false;
706  }
707  Tokens::new(result)
708}

710/// Insert FrontMatter into document, if not already added
711pub fn insert_frontmatter(document: &mut Document) -> Result<()> {
712  if lookup_bool("frontmatter_done") {
713    return Ok(());
714  }
715  let frontmatter_elements_set: HashSet<String> =
716    FRONTMATTER_ELEMENTS.iter().map(ToString::to_string).collect();
717
718  // Get frontmatter hash
719  let frontmatter_ref = match state::lookup_value("frontmatter") {
720    Some(Stored::HashTagData(frnt)) => frnt,
721    _ => return Ok(()),
722  };
723  let set_keys: Vec<String> = frontmatter_ref.keys().cloned().collect();
724
725  // If doc ONLY has abstract as frontmatter, defer until abstract's document location
726  if set_keys.len() == 1
727    && set_keys[0] == "ltx:abstract"
728    && !lookup_bool("frontmatter_deferred")
729  {
730    state::assign_value("frontmatter_deferred", true, Some(Scope::Global));
731    return Ok(());
732  }
733
734  state::assign_value("frontmatter_done", true, Some(Scope::Global));
735
736  // Remove frontmatter and replace with empty
737  let mut frontmatter = match state::remove_value("frontmatter") {
738    Some(Stored::HashTagData(frnt)) => frnt,
739    _ => return Ok(()),
740  };
741  state::assign_value(
742    "frontmatter",
743    Stored::HashTagData(HashMap::default()),
744    Some(Scope::Global),
745  );
746
747  // Order: first go through frontmatter_elements order, then any custom keys
748  let custom_keys: Vec<String> = frontmatter
749    .keys()
750    .filter(|key| !frontmatter_elements_set.contains(key.as_str()))
751    .map(ToString::to_string)
752    .collect();
753  let mut all_keys: Vec<String> =
754    FRONTMATTER_ELEMENTS.iter().map(ToString::to_string).collect();
755  all_keys.extend(custom_keys);
756
757  for key in &all_keys {
758    if let Some(list) = frontmatter.remove(key) {
759      // Dubious, but assures that frontmatter appears in text mode...
760      document.set_box_to_absorb(
761        Tbox::new(
762          *EMPTY_SYM,
763          lookup_font(),
764          None,
765          Tokens!(T_SPACE!()),
766          SymHashMap::default(),
767        )
768        .into(),
769      );
770      for (tag, attr, stuff) in list {
771        // Add a dedicated class for frontmatter notes
772        let attr = if tag == "ltx:note" {
773          let mut a = attr.unwrap_or_default();
774          let existing = a.get("class").cloned().unwrap_or_default();
775          let new_class = if existing.is_empty() {
776            "ltx_note_frontmatter".to_string()
777          } else {
778            s!("{existing} ltx_note_frontmatter")
779          };
780          a.insert("class".to_string(), new_class);
781          Some(a)
782        } else {
783          attr
784        };
785        document.open_element(&tag, attr, None)?;
786        document.absorb(&stuff, None)?;
787        let completed_node = document.close_element(&tag)?;
788        // Prune empty frontmatter elements (except ltx:rdf)
789        if tag != "ltx:rdf" {
790          if let Some(ref node) = completed_node {
791            if node.get_child_nodes().is_empty() {
792              document.remove_node(node.clone());
793            }
794          }
795        }
796      }
797      document.expire_box_to_absorb();
798    }
799  }
800  Ok(())
801}
