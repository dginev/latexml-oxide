27    let cs = T_CS!(s!("\\{}", Expand!(name).to_string()));

37    let mut boxes = Vec::new();
38    while let Some(token) = gullet::read_x_token(None, false, None)? {
39      boxes = stomach::invoke_token(&token)?;
40      if boxes.is_empty() {
41        break;
42      }
43      while !boxes.is_empty() {
44        if match boxes[0].get_property("isSpace") {
45          Some(Cow::Borrowed(Stored::Bool(space_bool))) => *space_bool,
46          Some(Cow::Owned(Stored::Bool(ref space_bool))) => *space_bool,
47          _ => false,
48        } {
49          boxes = boxes[1..].to_vec();
50        } else {
51          break;
52        }
53      }
54      if !boxes.is_empty() {
55        break;
56      }
57    }
58    Ok(boxes)

85  // Add a new frontmatter item that will be enclosed in <$tag %attr>...</$tag>
86  // The content is the result of digesting $tokens.
87  // \\@add@frontmatter[keys]{tag}[attributes]{content}
88  // keys can have
89  //   replace (to replace the current entry, if any)
90  //   ifnew   (only add if no previous entry)//

92  DefPrimitive!("\\@add@frontmatter OptionalKeyVals {} OptionalKeyVals {}",
93    sub[(keys_opt,tag_tks,attrs_opt,tokens)] {
94    // Digest this as if we're already in the document body!
95    let inpreamble = lookup_bool("inPreamble");
96    assign_value("inPreamble", false, None);
97    // Be careful since the contents may also want to add frontmatter
98    // (which should be inside or after this one!)
99    // So, we append this entry before digesting
100    let tag = tag_tks.to_string();
101    if let Some(keys) = keys_opt {
102      let known_key = if let Some(Stored::HashTagData(ref mut frnt)) = lookup_value("frontmatter") {
103        frnt.contains_key(&tag)
104      } else { false };
105      // if replace and previous entries
106      // remove previous entries
107      if known_key && keys.has_key("replace") {
108        state::with_value_mut("frontmatter", |val_opt| {
109          if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt {
110            frnt.insert(tag.clone(), Vec::new());
111          }
112        });
113      }
114      // if ifnew and previous entries
115      // skip this one.
116      if known_key && keys.has_key("ifnew") {
117        return Ok(Vec::new());
118      }
119    }
120
121    let attrs_digested = if let Some(attr_kvs) = attrs_opt {
122      if let DigestedData::KeyVals(digested) = attr_kvs.be_digested()?.data() {
123        Some(digested.get_hash_digested())
124      } else {
125        None
126      }
127    } else {
128      None
129    };
130    let mut wrapped_tokens = vec![T_BEGIN!()];
131    wrapped_tokens.extend(tokens.unlist());
132    wrapped_tokens.push(T_END!());
133    let digested_tokens = stomach::digest(Tokens::new(wrapped_tokens))?;
134    let entry = (tag.clone(), attrs_digested, digested_tokens);
135    state::with_value_mut("frontmatter", |val_opt| {
136      let frontmatter = match val_opt {
137        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
138        _ => fatal!(TexPool, Expected, "Global TeX Frontmatter hash was not available, should never happen"),
139      };
140      let f_entry = frontmatter.entry(tag).or_insert_with(Vec::new);
141      f_entry.push(entry);
142      Ok(())
143    })?;
144
145    AssignValue!("inPreamble", inpreamble);
146  });

242  // Tags & Titles
243  // The reference numbers, titles, captions etc, for various objects have
244  // different styling conventions, and the styling various depending on context.
245  // We'll use ltx:tags as a container for the various forms of ltx:tag with different @role's.
246  // The role=refnum form is simply formatted by \the<counter> and used by \ref;
247  // An ltx:tag w/o @role are for the numbers, often formatted differently, which
248  // appear alongside the object; Such a tag also may be embedded within the title or caption.
249  // Cross-references automatically generated by LaTeXML benefit from a bit more context:
250  // these are the role=typerefnum forms.
251  // Additional forms are needed for bibliographies, hyperref's autoref, etc.
252  // An additional complication is that while the "type" determines the formatting
253  // of the various forms, some types (eg. theorems) share the same counter.
254  // LaTeX defines this handling on an adhoc basis; defines \fnum@table, \fnum@figure for some types
255  // but \labelenumi, etc for others.
256
257  // This section synthesizes a more uniform support for reference numbers,
258  // references to reference numbers, title formatting etc.
259  // It allows you to customize each of the forms for each type encountered.
260  // The design reflects LaTeX needs, more than TeX, but support starts here!

262  // This collects up the various declared ltx:tag's into an ltx:tags
263  DefMacro!("\\lx@make@tags {}", sub[(ttype)] {
264    let mut tags = Vec::new();
265    if let Some(Stored::HashStored(formatters)) =
266      lookup_value("type_tag_formatter") {
267      let keys_sym : Vec<_> = formatters.keys().copied().collect();
268      let mut sorted_keys : Vec<String> = arena::with_many(&keys_sym, |keys| {
269        keys.into_iter().map(str::to_owned).collect()
270      });
271      sorted_keys.sort();
272      for role in sorted_keys.iter() {
273        let formatter_opt = match formatters.get(role) {
274          Some(Stored::Token(t)) => Some(*t),
275          Some(Stored::String(sym)) => Some(Token { text: *sym, code: Catcode::CS }),
276          _ => None
277        };
278        if let Some(formatter_token) = formatter_opt {
279          tags.push(Invocation!(T_CS!("\\lx@tag@intags"),
280            vec![
281              Tokens!(T_OTHER!(role)),
282              build_invocation(formatter_token, vec![Some(ttype.clone())])?
283            ])
284          );
285        }
286      }
287    }
288
289    let mut lx_tags = vec![T_CS!("\\lx@tags"), T_BEGIN!()];
290    for invoked_tag in tags {
291      lx_tags.append(&mut invoked_tag.unlist());
292    }
293    lx_tags.push(T_END!());
294    Ok(Tokens::new(lx_tags))
295  });
