4/// Perl: beforeFloat (latex_constructs.pool.ltxml L3430-3438)

12/// Extended version with `double` flag for `*` variants (span both columns).
13pub fn before_float_ex(float_type: &str, preincrement: Option<&str>, double: bool) {

24  let dim_val = state::lookup_dimension(dim_name).unwrap_or_default();
25  state::assign_register("\\hsize", dim_val.into(), None, Vec::new()).ok();
26  // Perl: if (my $main = $options{preincrement}) {
27  //   if (($type ne (LookupValue('LAST_FLOATTYPE') || ''))
28  //     && !IfCondition('\iflx@donecaption')) {
29  //     AssignValue('PREINCREMENTED_' . $main => { RefStepCounter($main) }, 'global'); } }
30  if let Some(main_counter) = preincrement {
31    let last_type = state::lookup_value("LAST_FLOATTYPE")
32      .map(|s| s.to_string())

35      .unwrap_or(None)

40        state::assign_value(&prekey, props, Some(Scope::Global));

43  }

59/// Simplified version of Perl's arrange_panels_and_breaks().
60/// When a figure/table/float has 2+ child figure/table/float elements (panels),
61/// add the ltx_figure_panel class to each panel.
62fn arrange_panels(document: &mut Document, node: &mut libxml::tree::Node) -> Result<()> {
63  // Perl: arrange_panels_and_breaks (latex_constructs L3286-3406)
64  // Simplified: we mark panel children with ltx_figure_panel class
65  // but skip the full break-insertion / width-based row-splitting logic.
66  //
67  // panel_break_names (Perl L3302-3307): elements that are NOT panels.
68  // Includes: ltx:break, Caption class (caption, toccaption),
69  // SectionalFrontMatter class (title, toctitle, subtitle, creator, contact, date,
70  // tags, classification, acknowledgements), Meta class (resource, navigation, etc.)
71  let is_panel_break = |qname: arena::SymStr| -> bool {
72    arena::with(qname, |name| {
73      matches!(
74        name,
75        "ltx:break"
76          | "ltx:caption"
77          | "ltx:toccaption"
78          | "ltx:title"
79          | "ltx:toctitle"
80          | "ltx:subtitle"
81          | "ltx:creator"
82          | "ltx:contact"
83          | "ltx:date"
84          | "ltx:tags"
85          | "ltx:classification"
86          | "ltx:acknowledgements"
87          | "ltx:resource"
88          | "ltx:navigation"
89      )
90    })
91  };
92  let note_qname = arena::pin_static("ltx:note");
93  let caption_qname = arena::pin_static("ltx:caption");
94  let mut panels: Vec<libxml::tree::Node> = Vec::new();
95  let mut notes: Vec<libxml::tree::Node> = Vec::new();
96  let mut caption: Option<libxml::tree::Node> = None;
97  for child in node.get_child_elements() {
98    let qname = latexml_core::document::get_node_qname(&child);
99    if qname == note_qname {
100      notes.push(child);
101    } else if is_panel_break(qname) {
102      if qname == caption_qname {
103        caption = Some(child);
104      }
105    } else {
106      // Perl L3342-3390: non-break children are potential panels
107      // (Perl also checks child_width > 0 at L3390, but we skip width checks)
108      panels.push(child);
109    }
110  }
111  // Perl BuildPanelsAndID L3317-3324: move top-level ltx:note to nearest caption
112  if let Some(mut cap) = caption {
113    for mut note in notes {
114      note.unlink_node();
115      cap.add_child(&mut note).ok();
116    }
117  }
118  // Perl L3403-3405: only add class if >1 panel (complex figure)
119  if panels.len() >= 2 {
120    // Perl: standalone panels get breaks between them.
121    // Perl has width-based row-splitting logic, but without box width tracking,
122    // we use a simpler heuristic: insert break after each "standalone" panel
123    // (p, listing, equation, equationgroup, itemize, enumerate, quote, theorem,
124    // proof, description, verbatim, math) when there are multiple panels.
125    let is_standalone = |p: &libxml::tree::Node| -> bool {
126      let qname = latexml_core::document::get_node_qname(p);
127      arena::with(qname, |name| {
128        matches!(name,
129          "ltx:p" | "ltx:listing" | "ltx:math" | "ltx:itemize" | "ltx:enumerate"
130          | "ltx:quote" | "ltx:theorem" | "ltx:proof" | "ltx:description"
131          | "ltx:equation" | "ltx:equationgroup" | "ltx:verbatim")
132      })
133    };
134    for panel in &mut panels {
135      document.add_class(panel, "ltx_figure_panel")?;
136    }
137    // Insert breaks between panels.
138    // Perl inserts break before a standalone panel (if there are prior panels in the row),
139    // and after standalone panels at the start. We simplify: insert break between consecutive
140    // panels where either the current or next panel is standalone.
141    for i in 0..panels.len().saturating_sub(1) {
142      if is_standalone(&panels[i]) || is_standalone(&panels[i + 1]) {
143        let ns = panels[i].get_namespace();
144        let mut break_node =
145          libxml::tree::Node::new("break", ns, document.get_document()).unwrap();
146        let _ = break_node.set_attribute("class", "ltx_break");
147        panels[i].add_next_sibling(&mut break_node)?;
148      }
149    }
150  }
151  Ok(())
152}

154/// Perl: collapseFloat (latex_constructs.pool.ltxml L3493-3520)
155/// If a figure/table/float contains exactly one inner float child,
156/// and they don't BOTH have captions, collapse the inner into the outer.
157fn collapse_float(document: &mut Document, float: &mut libxml::tree::Node) -> Result<()> {
158  let caption_qname = arena::pin_static("ltx:caption");
159  let figure_qname = arena::pin_static("ltx:figure");
160  let table_qname = arena::pin_static("ltx:table");
161  let float_qname = arena::pin_static("ltx:float");
162  // Find inner float/figure/table children
163  let mut inners: Vec<libxml::tree::Node> = Vec::new();
164  for child in float.get_child_elements() {
165    let qname = latexml_core::document::get_node_qname(&child);
166    if qname == figure_qname || qname == table_qname || qname == float_qname {
167      inners.push(child);
168    }
169  }
170  if inners.len() != 1 {
171    return Ok(());
172  }
173  let mut inner = inners.into_iter().next().unwrap();
174  // Check captions: collapse only if they don't BOTH have captions
175  let outer_has_caption = float.get_child_elements().iter()
176    .any(|c| latexml_core::document::get_node_qname(c) == caption_qname);
177  let inner_has_caption = inner.get_child_elements().iter()
178    .any(|c| latexml_core::document::get_node_qname(c) == caption_qname);
179  if outer_has_caption && inner_has_caption {
180    return Ok(());
181  }
182  // Copy inner's attributes to outer (except xml:id)
183  let attrs = inner.get_attributes();
184  for (name, value) in &attrs {
185    // get_attributes() may return the key as "id" (local name) or "xml:id" (prefixed)
186    if name != "xml:id" && name != "id" {
187      document.set_attribute(float, name, value)?;
188    }
189  }
190  // If inner has caption, promote inner's xml:id to outer
191  if inner_has_caption {
192    let inner_id = inner.get_attribute("xml:id")
193      .or_else(|| inner.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace"));
194    if let Some(id) = inner_id {
195      // Unrecord the outer's old ID and remove the attribute before setting the new one
196      if let Some(old_id) = float.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace") {
197        document.unrecord_id(&old_id);
198      }
199      float.remove_attribute("xml:id").ok();
200      document.unrecord_id(&id);
201      document.set_attribute(float, "xml:id", &id)?;
202    }
203  }
204  // Replace inner element with its children (unwrap inner)
205  let children: Vec<libxml::tree::Node> = inner.get_child_nodes();
206  for mut child in children {
207    child.unlink_node();
208    float.add_child(&mut child).ok();
209  }
210  inner.unlink_node();
211  Ok(())
212}
