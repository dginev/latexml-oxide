4/// Perl: beforeFloat (latex_constructs.pool.ltxml L3430-3438)
5/// Sets \@captype, adjusts \hsize for single/double column floats.
6/// `preincrement`: if Some("figure"), pre-increments the parent float counter
7///   on first subfloat entry (before main caption), storing result for later use.
8pub fn before_float(float_type: &str, preincrement: Option<&str>) {
9  def_macro(
10    T_CS!("\\@captype"), None,
11    Tokens::new(ExplodeText!(float_type)),
12    None,
13  ).ok();
14  // Perl: if (my $main = $options{preincrement}) {
15  //   if (($type ne (LookupValue('LAST_FLOATTYPE') || ''))
16  //     && !IfCondition('\iflx@donecaption')) {
17  //     AssignValue('PREINCREMENTED_' . $main => { RefStepCounter($main) }, 'global'); } }
18  if let Some(main_counter) = preincrement {
19    let last_type = state::lookup_value("LAST_FLOATTYPE")
20      .map(|s| s.to_string())
21      .unwrap_or_default();
22    let done_caption = if_condition(&T_CS!("\\iflx@donecaption"))
23      .unwrap_or(None)
24      .unwrap_or(false);
25    if float_type != last_type && !done_caption {
26      if let Ok(props) = ref_step_counter(main_counter, false) {
27        let prekey = s!("PREINCREMENTED_{main_counter}");
28        state::assign_value(&prekey, props, Some(Scope::Global));
29      }
30    }
31  }
32}

34/// Perl: afterFloat (latex_constructs.pool.ltxml L3440-3448)
35/// Rescues caption counters into the whatsit properties.
36pub fn after_float(whatsit: &mut Whatsit) {
37  let captype = stomach::digest(T_CS!("\\@captype"))
38    .map(|d| d.to_string())
39    .unwrap_or_default();
40  // Perl: AssignValue('PREINCREMENTED_' . $type => undef, 'global');
41  let prekey = s!("PREINCREMENTED_{captype}");
42  state::remove_value(&prekey);
43  rescue_caption_counters(&captype, whatsit);
44  state::assign_value("LAST_FLOATTYPE", Stored::String(arena::pin(&captype)), Some(Scope::Global));
45}

47/// Simplified version of Perl's arrange_panels_and_breaks().
48/// When a figure/table/float has 2+ child figure/table/float elements (panels),
49/// add the ltx_figure_panel class to each panel.
50fn arrange_panels(document: &mut Document, node: &mut libxml::tree::Node) -> Result<()> {
51  let figure_qname = arena::pin_static("ltx:figure");
52  let table_qname = arena::pin_static("ltx:table");
53  let float_qname = arena::pin_static("ltx:float");
54  let mut panels: Vec<libxml::tree::Node> = Vec::new();
55  for child in node.get_child_elements() {
56    let qname = latexml_core::document::get_node_qname(&child);
57    if qname == figure_qname || qname == table_qname || qname == float_qname {
58      panels.push(child);
59    }
60  }
61  if panels.len() >= 2 {
62    for mut panel in panels {
63      document.add_class(&mut panel, "ltx_figure_panel")?;
64    }
65  }
66  Ok(())
67}

69/// Perl: collapseFloat (latex_constructs.pool.ltxml L3493-3520)
70/// If a figure/table/float contains exactly one inner float child,
71/// and they don't BOTH have captions, collapse the inner into the outer.
72fn collapse_float(document: &mut Document, float: &mut libxml::tree::Node) -> Result<()> {
73  let caption_qname = arena::pin_static("ltx:caption");
74  let figure_qname = arena::pin_static("ltx:figure");
75  let table_qname = arena::pin_static("ltx:table");
76  let float_qname = arena::pin_static("ltx:float");
77  // Find inner float/figure/table children
78  let mut inners: Vec<libxml::tree::Node> = Vec::new();
79  for child in float.get_child_elements() {
80    let qname = latexml_core::document::get_node_qname(&child);
81    if qname == figure_qname || qname == table_qname || qname == float_qname {
82      inners.push(child);
83    }
84  }
85  if inners.len() != 1 {
86    return Ok(());
87  }
88  let mut inner = inners.into_iter().next().unwrap();
89  // Check captions: collapse only if they don't BOTH have captions
90  let outer_has_caption = float.get_child_elements().iter()
91    .any(|c| latexml_core::document::get_node_qname(c) == caption_qname);
92  let inner_has_caption = inner.get_child_elements().iter()
93    .any(|c| latexml_core::document::get_node_qname(c) == caption_qname);
94  if outer_has_caption && inner_has_caption {
95    return Ok(());
96  }
97  // Copy inner's attributes to outer (except xml:id)
98  let attrs = inner.get_attributes();
99  for (name, value) in &attrs {
100    // get_attributes() may return the key as "id" (local name) or "xml:id" (prefixed)
101    if name != "xml:id" && name != "id" {
102      document.set_attribute(float, name, value)?;
103    }
104  }
105  // If inner has caption, promote inner's xml:id to outer
106  if inner_has_caption {
107    let inner_id = inner.get_attribute("xml:id")
108      .or_else(|| inner.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace"));
109    if let Some(id) = inner_id {
110      // Unrecord the outer's old ID and remove the attribute before setting the new one
111      if let Some(old_id) = float.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace") {
112        document.unrecord_id(&old_id);
113      }
114      float.remove_attribute("xml:id").ok();
115      document.unrecord_id(&id);
116      document.set_attribute(float, "xml:id", &id)?;
117    }
118  }
119  // Replace inner element with its children (unwrap inner)
120  let children: Vec<libxml::tree::Node> = inner.get_child_nodes();
121  for mut child in children {
122    child.unlink_node();
123    float.add_child(&mut child).ok();
124  }
125  inner.unlink_node();
126  Ok(())
127}
