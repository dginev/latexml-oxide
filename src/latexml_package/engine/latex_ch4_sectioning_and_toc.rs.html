57  DefMacro!(
58    "\\@startsection{}{}{}{}{}{} OptionalMatch:*",
59    sub[(type_tokens, level_arg, _ignore3, _ignore4, _ignore5, _ignore6, flag)] {
60      // Aside: Guard mode
61      // Never start sections in math mode -- this is a good recovery point for broken documents
62      if lookup_bool("IN_MATH") {
63        let mode = state::lookup_string("MODE");
64        if mode.contains("math") { // double-check we're really in math
65          end_mode(&mode)?;
66        } else { // otherwise, just unset the flag?
67          state::assign_value("IN_MATH", false, Some(Scope::Global));
68        }
69      }
70      // Main logic
71      let level = level_arg.to_string();
72      let level_int = if level.is_empty() { 0 } else { level.parse::<i64>().expect(&level) };
73      let mut tokens: Vec<Token>;
74      if flag.is_some() { // No number, not in TOC
75        tokens = vec![
76          T_CS!("\\par"), T_CS!("\\@startsection@hook"), T_CS!("\\@@unnumbered@section"),
77        T_BEGIN!()];
78        tokens.extend(type_tokens.unlist());
79        tokens.extend(vec![T_END!(), T_BEGIN!(), T_END!()]);
80      } else if level_int > CounterValue!("secnumdepth").value_of() ||
81        lookup_bool("no_number_sections") {
82        // No number, but in TOC
83        tokens = vec![
84          T_CS!("\\par"), T_CS!("\\@startsection@hook"), T_CS!("\\@@unnumbered@section"),
85        T_BEGIN!()];
86        tokens.extend(type_tokens.unlist());
87        tokens.extend(vec![T_END!(), T_BEGIN!(), T_OTHER!("toc"), T_END!()]);
88      } else { // Number and in TOC
89        tokens = vec![T_CS!("\\par"), T_CS!("\\@startsection@hook"), T_CS!("\\@@numbered@section"),
90        T_BEGIN!()];
91        tokens.extend(type_tokens.unlist());
92        tokens.extend(vec![T_END!(), T_BEGIN!(), T_OTHER!("toc"), T_END!()]);
93      };
94      Ok(Tokens::new(tokens))
95    },
96    locked => true
97  );

99  DefConstructor!(
100    "\\@@numbered@section{} Undigested OptionalUndigested Undigested",
101    sub[document, args, props] {
102      // args:=(stype,inlist,toctitle,title)
103      let stype = args[0].as_ref().unwrap().to_string();
104      let inlist = args[1].as_ref().unwrap().to_string();
105      // TODO: This bizarre argument API interaction needs to be simplified down to Perl's
106      // intuitive level of:       let (x,y,z, ...) = @args;
107      // If backmatter, find insertion point as if inserting the backmatter element type
108      if let Some(asif) = props.get("backmatterelement") {
109        let asif_str = asif.to_string();
110        let point = document.find_insertion_point(&asif_str, None)?;
111        document.set_node(&point);
112      }
113      let clean_id = prop_string!(props,"id"); // TODO: CleanID($id);
114      let tagname = s!("ltx:{stype}");
115      document.open_element(&tagname,
116        Some(string_map!("xml:id" => clean_id, "inlist" => inlist)),
117        None,
118          )?;
119      // TODO: Another instance where the immutability of props causes endless cloning
120      //       which is slow and wasteful.
121      // The big problem is that for props to be mutable, the entire parent whatsit needs to
122      // be mutable, and Rust hits a mutability conflict between the parent, and the
123      // "args" and "props" children ... will come back here after performance becomes
124      // an issue again
125      //
126      // Part 2: I have now, with great attention and profiling, solidified the position that
127      //       Whatsits are immutable during the absorbtion phase -- and hence
128      // the args and props passed in here will remain immutable in latexml_oxide.
129      // Hence, for this absorb call to run correctly, it must either:
130      // 1) Accept a cloned value as currently, paying with performance
131      // 2) Accept immutable references to digested objects,
132      // which may lead to far-reaching borrowing constraints
133      //   e.g. unlist()-ing a digested List will have to produce box references,
134      //  rather than provide the owned boxes directly.
135      //   would have to experiment with this - as it is of course much lighter on performance
136      //
137
138      // Update 2022: The notes are generally still accurate,
139      // but cloning a Digested object is now cheap enough,
140      // as each enum variant is guarded by an Rc reference counter. Rc<Tbox>, Rc<List>, etc.
141      if let Some(Stored::Digested(tags)) = props.get("tags") {
142        document.absorb(tags, None)?;
143      }
144      let title = prop_digested!(props, "title");
145      document.insert_element("ltx:title", title, None)?;
146
147      let toctitle = prop_digested!(props, "toctitle");
148      if !toctitle.is_empty() {
149        document.insert_element("ltx:toctitle", toctitle, None)?;
150      }
151    },
152    properties => sub[args] {
153      let stype = args[0].as_ref().unwrap();
154      // let inlist = args[1].as_ref().unwrap();
155      let toctitle_arg = args[2].as_ref();
156      let title = args[3].as_ref().unwrap();
157
158      maybe_peek_label()?;
159      let stype_str = stype.to_string();
160      let mut props = ref_step_counter(&stype_str, false)?;
161      // For appendix, look up the backmatter element mapping
162      if stype_str == "appendix" {
163        if let Some(bme) = state::lookup_mapping("BACKMATTER_ELEMENT", &s!("ltx:{stype_str}")) {
164          props.insert("backmatterelement", bme);
165        }
166      }
167      let toctitle = match toctitle_arg {
168        Some(v) => if !v.to_string().is_empty() {
169          args[2].as_ref().unwrap()
170        } else {
171          title
172        },
173        None => title
174      };
175      let stype_tokens = stype.revert()?;
176      let title_tokens = title.revert()?;
177      let invoked_title =
178        Invocation!(T_CS!("\\lx@format@title@@"), vec![stype_tokens, title_tokens]);
179      let xtitle    = stomach::digest(invoked_title)?;
180      let invoked_toctitle = Invocation!(T_CS!("\\lx@format@toctitle@@"),
181          vec![stype.revert()?, toctitle.revert()?]);
182      let xtoctitle = stomach::digest(invoked_toctitle)?;
183
184      if xtoctitle.to_string() != xtitle.to_string() {
185        props.insert("toctitle", xtoctitle.into());
186      }
187      props.insert("title", xtitle.into());
188
189      Ok(props)
190    }
191  );

194  DefConstructor!("\\@@unnumbered@section{} Undigested OptionalUndigested Undigested",
195  sub[document, args, props] {
196      let stype = args[0].as_ref().unwrap();
197      let inlist = args[1].as_ref().unwrap();
198      // If backmatter, find insertion point as if inserting the backmatter element type
199      if let Some(asif) = props.get("backmatterelement") {
200        let asif_str = asif.to_string();
201        let point = document.find_insertion_point(&asif_str, None)?;
202        document.set_node(&point);
203      }
204      let id = props.get("id").unwrap().to_string();
205      document.open_element(&s!("ltx:{stype}"),
206        Some(string_map!(
207          "xml:id" => clean_id(&id),
208          "inlist"  => inlist.to_string()
209        )), None)?;
210      let title = prop_digested!(props, "title");
211      document.insert_element("ltx:title", title, None)?;
212
213      let toctitle = prop_digested!(props, "toctitle");
214      if !toctitle.is_empty() {
215        document.insert_element("ltx:toctitle", toctitle, None)?;
216      }
217    },
218    properties => sub[args] {
219      use DigestedData::*;
220      let stype = args[0].as_ref().unwrap();
221      // let inlist = args[1].as_ref().unwrap();
222      let toctitle_arg = args[2].as_ref();
223      let title = args[3].as_ref().unwrap();
224      maybe_peek_label()?;
225      let stype_str = stype.to_string();
226      let mut props = RefStepID!(&stype_str)?;
227      // For appendix, look up the backmatter element mapping
228      if stype_str == "appendix" {
229        if let Some(bme) = state::lookup_mapping("BACKMATTER_ELEMENT", &s!("ltx:{stype_str}")) {
230          props.insert("backmatterelement", bme);
231        }
232      }
233      let title_digested = if let Postponed(tokens) = title.data() {
234        // TODO: is .clone() on the tokens before they are unlisted a sign that
235        // the DigestedData::Postponed variant isn't ideal?
236        // should we be draining it? Or is there a better conceptual organization?
237        stomach::digest(
238          Tokens!(T_CS!("\\lx@hidden@bgroup"), tokens.clone().unlist(), T_CS!("\\lx@hidden@egroup")))?
239      } else {
240        title.clone()
241      };
242      props.insert("title", title_digested.into());
243
244      if let Some(toctitle) = toctitle_arg {
245        if let Postponed(toctokens) = toctitle.data() {
246          if !toctokens.is_empty() {
247            let toctitle_digested = stomach::digest(
248              Tokens!(T_CS!("\\lx@hidden@bgroup"),
249                toctokens.clone().unlist(), T_CS!("\\lx@hidden@egroup")))?;
250            props.insert("toctitle", toctitle_digested.into());
251          }
252        }
253      }
254      Ok(props)
255    }
256  );

338  DefConstructor!("\\tableofcontents",
339    "<ltx:TOC lists='toc' scope='global' select='#select'><ltx:title>#name</ltx:title></ltx:TOC>",
340    properties => {
341      let mut td = CounterValue!("tocdepth").value_of() as usize + 1;
342      let s  = ["ltx:part", "ltx:chapter", "ltx:section", "ltx:subsection", "ltx:subsubsection",
343          "ltx:paragraph", "ltx:subparagraph"];
344      let max_level = s.len()-1;
345      td = std::cmp::min(td,max_level);
346      let mut s_depth : Vec<&'static str> = s.into_iter().take(td+1).collect();
347      if !s_depth.is_empty() {
348        s_depth.push("ltx:appendix");
349        s_depth.push("ltx:index");
350        s_depth.push("ltx:bibliography");
351      }
352
353      Ok(stored_map!("select" => s_depth.join(" | "),
354        "name" => digest(T_CS!("\\contentsname"))?))
355    }
356  );

371  DefConstructor!("\\addcontentsline{}{}{}", sub[document,args] {
372      if let [inlist,_vtype,_title @ ..] = args.as_slice() {
373        // Note that the node can be inlist $inlist.
374        // Could conceivably want to add $title as toctitle???
375        if let Some(savenode) = document.float_to_label() {
376          // DG: The Document+Node mutability API is strange
377          //     w.r.t the original Perl ergonomics.
378          // if we use `.get_node_mut()` we can no longer `doc.set_attribute(node)`,
379          // as it induces TWO simultaneous mutable pointers into document.
380          // cloning Node is now cheap enough (as the Node data lives in C's libxml)
381          // but it's not yet an idiomatic Rust interface. Something to ponder...
382          let mut node  = document.get_node().clone();
383          let inlist_str = inlist.as_ref().map(|v|v.to_string()).unwrap_or_default();
384          let inlist_v = if let Some(lists) = node.get_attribute("inlist") {
385            if !lists.is_empty() {
386              s!("{lists} {inlist_str}")
387            } else { inlist_str }
388          } else {
389            inlist_str
390          };
391          document.set_attribute(&mut node, "inlist", &inlist_v)?;
392          document.set_node(&savenode);
393        }
394      }
395    }
396  );
