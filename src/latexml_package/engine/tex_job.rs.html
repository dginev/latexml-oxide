3//! Core TeX Implementation for LaTeXML

5use crate::prelude::*;
6use chrono::prelude::*;
7
8LoadDefinitions!({
9  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
10  // Job Family of primitive control sequences
11  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
12
13  //======================================================================
14  // The current Job
15  //----------------------------------------------------------------------
16  // \jobname          c  is the underlying file name for a job.
17  // \time             pi holds the current time in minutes after midnight (0-1439).
18  // \day              pi holds the current day of the month (1-31).
19  // \month            pi holds the current month of the year (1-12).
20  // \year             pi holds the current year (e.g., 2000).
21  // \mag              pi holds the magnification ratio times 1000.
22  DefMacro!(T_CS!("\\jobname"), None, Tokens!()); // Set to the filename by initialization
23  DefRegister!("\\time", Number!(0));
24  DefRegister!("\\day", Number!(0));
25  DefRegister!("\\month", Number!(0));
26  DefRegister!("\\year", Number!(0));
27  DefRegister!("\\mag", Number!(1000));
28
29  // TODO: This may mess up Daemon state? Reinit when setting jobname?
30  // Respect SOURCE_DATE_EPOCH env var for reproducible builds (like Perl)
31  let dt: DateTime<Local> = if let Ok(epoch_str) = std::env::var("SOURCE_DATE_EPOCH") {
32    if let Ok(epoch) = epoch_str.trim().parse::<i64>() {
33      DateTime::from_timestamp(epoch, 0)
34        .map(|utc| utc.with_timezone(&Local))
35        .unwrap_or_else(Local::now)
36    } else { Local::now() }
37  } else { Local::now() };
38  AssignValue!("\\day", Number!(dt.day()), Scope::Global);
39  AssignValue!("\\month", Number!(dt.month()), Scope::Global);
40  AssignValue!("\\year", Number!(dt.year()), Scope::Global);
41  AssignValue!(
42    "\\time",
43    Number!(60 * dt.hour() + dt.minute()),
44    Scope::Global
45  );
46
47  //======================================================================
48  // Random Job related things
49  //----------------------------------------------------------------------
50  // \end              c  terminates the current job.
51  // \everyjob         pt holds tokens which are inserted at the start of every job.
52  // \deadcycles       iq is the number of times \output was called since the last \shipout.
53  // \maxdeadcycles    pi is the maximum allowed value of \deadcycles before an error is generated.
54  // Perl: $stomach->leaveHorizontal; $stomach->getGullet->flush;
55  DefPrimitive!("\\lx@end@document", {
56    // When called during package/definition loading (e.g., expl3's error handler
57    // calls \tex_end:D via \msg_fatal), ignore it. Package errors should not
58    // terminate the entire document processing.
59    if !state::lookup_bool_sym(pin!("INTERPRETING_DEFINITIONS")) {
60      leave_horizontal()?;
61      gullet::flush();
62    }
63    // else: silently ignore during definition loading
64  });
65  Let!("\\end", "\\lx@end@document");
66  // Save the TeX primitive \end under \@@end so that expl3's primitive rename
67  // (\__kernel_primitive:NN \end \tex_end:D) gets the real primitive, not
68  // LaTeX's \end{environment} handler which consumes {} arguments.
69  // In Perl, @@end is the saved TeX \end (latex.ltx saves it).
70  Let!("\\@@end", "\\lx@end@document");
71
72  DefRegister!("\\everyjob", Tokens!());
73  DefRegister!("\\deadcycles", Number!(0));
74  DefRegister!("\\maxdeadcycles", Number!(0));
75
76  //======================================================================
77  // Dumping
78  //----------------------------------------------------------------------
79  // \dump             c  outputs a format file in INITEX; otherwise it is equivalent to \end.
80
81  DefMacro!("\\dump", {
82    Warn!("unexpected", "dump", "Do not know how to \\dump yet, sorry");
83  });
84
85  // TODO: load_dump
86  // TODO: load_latex
87
88  //======================================================================
89  // LaTeX 2.09 compatibility
90  //----------------------------------------------------------------------
91  // Perl: TeX.pool.ltxml L60-65
92  // \documentstyle[opts]{class} — LaTeX 2.09 command.
93  // Perl loads LaTeX.pool then re-queues \documentstyle token.
94  // Since our \documentclass already loads LaTeX.pool on first encounter,
95  // we can simply redirect \documentstyle → \documentclass.
96  // Perl: TeX.pool.ltxml L60-65
97  // Reads [options]{class}, loads LaTeX (or AmSTeX) pool, then re-emits
98  // \documentclass [options]{class} for the now-loaded LaTeX pool to handle.
99  DefMacro!("\\documentstyle[]{}", sub[(options_opt, class_tks)] {
100    let class = class_tks.to_string();
101    let pool = if class == "amsppt" { "AmSTeX" } else { "LaTeX" };
102    input_definitions(pool, InputDefinitionOptions {
103      extension: Some(Cow::Borrowed("pool")),
104      ..InputDefinitionOptions::default()
105    })?;
106
107    state::assign_value("2.09_COMPATIBILITY", true, Some(Scope::Global));
108
109    // In LaTeX 2.09, options are both class options AND packages to load.
110    // First load the class, then try to load each option as a package.
111    let mut result = Tokens!(T_CS!("\\documentclass"));
112    if let Some(ref opts) = options_opt {
113      let opts: &Tokens = opts;
114      result = Tokens!(result, T_OTHER!("["), opts.clone(), T_OTHER!("]"));
115    }
116    result = Tokens!(result, T_BEGIN!(), class_tks, T_END!());
117
118    // After class loads, try each option as a package (Perl \compat@loadpackages)
119    if let Some(opts) = options_opt {
120      let opts_str = opts.to_string();
121      for opt in opts_str.split(',') {
122        let opt = opt.trim();
123        if opt.is_empty() { continue; }
124        // Skip numeric options (10pt, 11pt, 12pt) and known class options
125        if opt.ends_with("pt") || opt == "landscape" || opt == "twocolumn"
126          || opt == "onecolumn" || opt == "draft" || opt == "final"
127          || opt == "preprint" || opt == "tighten" || opt == "manuscript" {
128          continue;
129        }
130        result = Tokens!(result,
131          T_CS!("\\IfFileExists"),
132          T_BEGIN!(), Explode!(format!("{opt}.sty")), T_END!(),
133          T_BEGIN!(), T_CS!("\\RequirePackage"), T_BEGIN!(), Explode!(opt), T_END!(), T_END!(),
134          T_BEGIN!(), T_END!()
135        );
136      }
137    }
138    Ok(result)
139  });
140});
