3// C.10.2 The array and tabular Environments
4//======================================================================
5// Tabular are a bit tricky in that we have to arrange for tr and td to
6// be openned and closed at the right times; the only real markup is
7// the & and \\. Also \multicolumn has to be cooperative.
8// Along with this, we have to track which column specification applies
9// to the current column.
10// To simulate LaTeX's tabular borders & hlines, we simply add border
11// attributes to all cells.  For HTML, CSS will be necessary to display them.
12// [We'll ignore HTML's frame, rules and colgroup mechanisms.]
13

27  DefPrimitive!("\\@tabular@bindings AlignmentTemplate OptionalKeyVals:tabular",
28    sub[(template, attributes_opt)] {
29    let attrs_stored = attributes_opt.map(KeyVals::as_flat_hash).unwrap_or_default();
30    let mut attrs = HashMap::default();
31    for (k,v) in attrs_stored {
32      attrs.insert(k, v.to_string());
33    }
34    if let Some(va) = attrs.get("vattach") {
35      attrs.insert(String::from("vattach"), translate_attachment(va).to_string());
36    }
37
38    tabular_bindings(template, SymHashMap::default(), attrs)?;
39  });

95  DefMacro!("\\cline{}", r"\noalign{\@cline{#1}}");
96  DefConstructor!("\\@cline{}", "",
97    after_digest => sub[whatsit] {
98      let cols = whatsit.get_arg(1).map(ToString::to_string).unwrap_or_default();
99      let mut cols_vec = Vec::new();
100      let cols_chars = cols.chars();
101      let mut from : Option<usize> = None;
102      let mut num = String::new();
103      for c_next in cols_chars {
104        match c_next {
105          ',' => if !num.is_empty() {
106            let this_num = num.parse::<usize>().unwrap();
107            if let Some(from_num) = from {
108              for num_in_range in from_num..=this_num {
109                cols_vec.push(num_in_range);
110              }
111            } else {
112              cols_vec.push(this_num);
113            }
114            from = None;
115            num = String::new();
116          },
117          '-' => {
118            from = Some(num.parse::<usize>().unwrap());
119            num = String::new();
120          }
121          c if c.is_ascii_digit() => num.push(c_next),
122          _ => break
123        }
124      }
125      if !num.is_empty() {
126        let this_num = num.parse::<usize>().unwrap();
127        if let Some(from_num) = from {
128          for num_in_range in from_num..=this_num {
129            cols_vec.push(num_in_range);
130          }
131        } else {
132          cols_vec.push(this_num);
133        }
134      }
135      if let Some(alignment_stored) = lookup_alignment() {
136        alignment_stored.alignment_cell().unwrap().borrow_mut()
137          .add_line("t", cols_vec);
138      }
139    },
140    sizer      => 0, alias => "\\cline",
141    // properties => { "isHorizontalRule" => true }
142  );
