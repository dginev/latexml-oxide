1//! TeX Logic
2//!
3//! Core TeX Implementation for LaTeXML
4
5use crate::prelude::*;
6LoadDefinitions!({
7  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
8  // Logic Family of primitive control sequences
9  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
10  //======================================================================
11  // Basic logic
12  //----------------------------------------------------------------------
13  // \iftrue           c  is a conditional which is always true.
14  // \iffalse          c  is a conditional which is always false.
15  // \else             c  begins the false part of a conditional.
16  // \fi               c  is the concluding command of a conditional.
17  // \or               c  separates cases in an \ifcase conditional.
18  DefConditional!("\\iftrue", { true });
19  DefConditional!("\\iffalse", { false });
20  DefConditional!("\\else"); // BUILT-IN to Definition
21  DefConditional!("\\or"); // BUILT-IN to Definition
22  DefConditional!("\\fi"); // BUILT-IN to Definition
23
24  //======================================================================
25  // Token testing
26  //----------------------------------------------------------------------
27  // \if               c  tests if two tokens have the same character codes (i.e., values 0-256).
28  // \ifx              c  tests if two tokens are the same.
29  // \ifcat            c  tests if two tokens have the same category codes (i.e., values 0-16).
30  DefParameterType!(ExpandedIfToken, sub[_inner, _extra] {
31    let token_opt = gullet::read_x_token(Some(false), true, None)?;
32    match token_opt {
33      Some(t) => t,
34      None => {
35        Error!("expected", "ExpandedIfToken",
36          "conditional expected a token argument, came back empty. Falling back to \\@empty");
37        T_CS!("\\@empty")
38      }}
39  });
40
41  DefConditional!("\\if ExpandedIfToken ExpandedIfToken", sub[(left,right)] {
42    left.get_charcode() == right.get_charcode()
43  });
44  DefConditional!("\\ifx Token Token", sub[(left,right)] { x_equals(&left, &right) });
45  DefConditional!("\\ifcat ExpandedIfToken ExpandedIfToken", sub[(left,right)] {
46    left.get_catcode() == right.get_catcode()
47  });
48
49  //======================================================================
50  // Number testing
51  //----------------------------------------------------------------------
52  // \ifnum            c  compares two integers.
53  // \ifodd            c  tests for an odd integer.
54  // \ifcase           c  begins a multi-case conditional.
55  DefConditional!("\\ifnum Number Token Number", sub[(u,rel,v)] {
56    compare(u.value_of(), rel, v.value_of())
57  });
58  DefConditional!("\\ifodd Number", sub[(u)] {
59    u.value_of() % 2 == 1
60  });
61  DefConditional!("\\ifcase Number");
62
63  //======================================================================
64  // Dimension testing
65  //----------------------------------------------------------------------
66  // \ifdim            c  compares two dimensions.
67  DefConditional!("\\ifdim Dimension Token Dimension", sub[(u,rel,v)] {
68    compare(u.value_of(), rel, v.value_of())
69  });
70
71  //======================================================================
72  // Box testing
73  //----------------------------------------------------------------------
74  // \ifhbox           c  is true if a box register contains an \hbox.
75  // \ifvbox           c  is true if a box register contains a \vbox.
76  // \ifvoid           c  is true if a box register is void.
77  //
78  DefConditional!("\\ifhbox Number", sub[(arg)] { classify_box(arg)? == "hbox" });
79  DefConditional!("\\ifvbox Number", sub[(arg)] { classify_box(arg)? == "vbox" });
80  DefConditional!("\\ifvoid Number", sub[(arg)] { classify_box(arg)?.is_empty() });
81
82  //======================================================================
83  // Mode testing
84  //----------------------------------------------------------------------
85  // \ifhmode          c  is true if TeX is in horizontal or restricted horizontal mode.
86  // \ifinner          c  is true if TeX is in internal vertical, restricted horizontal, or
87  // nondisplay math mode. \ifmmode          c  is true if TeX is in math or display math mode.
88  // \ifvmode          c  is true if TeX is in vertical or internal vertical mode.
89
90  // NOTE: We don't KNOW if we're in vertical, horizontal or inner mode!!!!!!!
91  DefConditional!("\\ifvmode", { false });
92  DefConditional!("\\ifhmode", { false });
93  DefConditional!("\\ifinner", { false });
94  DefConditional!("\\ifmmode", { lookup_bool("IN_MATH") });
95
96  //======================================================================
97  // I/O testing
98  //----------------------------------------------------------------------
99  // \ifeof c tests for the end of a file .
100  DefConditional!("\\ifeof Number", sub[(port)] {
101    with_value(&s!("input_file:{}", port), |val_opt|
102      if let Some(Stored::Mouth(mouth)) = val_opt {
103        mouth.borrow().at_eof()
104      } else {
105        true
106      })
107  });
108});
109
110fn compare(u: i64, rel: Token, v: i64) -> bool {
111  // NOTE: One would expect this to be best written as an advanced match state::ent
112  // however, due to the shallow comparison of Cow<str> the Cow::Borrowed("<") and
113  // Cow::Owned("<") variants will NOT be equal via a destructuring match.
114  // However, since we've defined our own PartialEq trait over Token, an equality comparison
115  // will produce the right behavior
116  if rel == T_OTHER!("<") || rel == T_CS!("\\@@<") {
117    u < v
118  } else if rel == T_OTHER!("=") {
119    u == v
120  } else if rel == T_OTHER!(">") || rel == T_CS!("\\@@>") {
121    u > v
122  } else {
123    let message = s!(
124      "Expected a relational token for comparision. Got {:?} (cc {:?})",
125      rel,
126      rel.get_catcode()
127    );
128    let err = || {
129      Error!("expected", "<relationaltoken>", message);
130      Ok(())
131    };
132    err().ok();
133    false
134  }
135}
