515/// Runtime lookup: route `filename` (e.g. `"aastex.cls"`, `"pgfmath.code.tex"`)
516/// through its compiled `load_definitions` fn, or return `None` when the
517/// filename has no registered binding (caller falls back to raw
518/// `.sty` / `.cls` / similar). Splits on the *first* `.` so multi-dot
519/// filenames like `pgfmath.code.tex` are matched as `("pgfmath", "code.tex")`
520/// rather than `("pgfmath.code", "tex")` — the latter silently dropped
521/// pgf / pgfmath / pgfmathcalc bindings, breaking tikz tests.
522pub fn dispatch(filename: &str) -> Option<Result<()>> {
523  let (base, ext) = filename.split_once('.')?;
524  BINDINGS
525    .iter()
526    .find(|(name, extension, _)| *name == base && *extension == ext)
527    .map(|(_, _, loader)| loader())
528}

530/// Names of all registered class bindings (without the `.cls` suffix),
531/// in registration order. Consumed by Perl-parity `LoadClass` prefix-match
532/// fallback — when `mn2ebis.cls` has no direct binding, the caller sorts
533/// these longest-first and picks the first whose name is a prefix of
534/// `mn2ebis` (→ `mn2e`). Lazily materialized once from `BINDINGS` and
535/// cached via `OnceLock`, so callers can take `&'static [&'static str]`
536/// without allocating on every query.
537pub fn class_binding_names() -> &'static [&'static str] {
538  use std::sync::OnceLock;
539  static NAMES: OnceLock<Vec<&'static str>> = OnceLock::new();
540  NAMES
541    .get_or_init(|| {
542      BINDINGS
543        .iter()
544        .filter(|(_, ext, _)| *ext == "cls")
545        .map(|(name, _, _)| *name)
546        .collect()
547    })
548    .as_slice()
549}
