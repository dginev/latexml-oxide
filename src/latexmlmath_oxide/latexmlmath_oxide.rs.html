15const USAGE: &str = "\
16latexmlmath_oxide — convert a single TeX formula (a LaTeXML reimplementation in Rust)
17
18Usage: latexmlmath_oxide [OPTIONS] '<formula>'
19
20Emits the math element itself — no document wrapper. With no conversion option,
21the default is Presentation MathML on stdout (Perl `latexmlmath` parity).
22
23Options:
24      --pmml, --presentationmathml  Presentation MathML (the default)
25      --cmml, --contentmathml       Content MathML
26      --xmath, --XMath              LaTeXML's internal <XMath> parse tree
27  -q, --quiet                       Only warnings and errors; no progress/lexeme notes
28  -h, --help                        Print this help
29
30Passing both --pmml and --cmml emits parallel markup: Content MathML annotated
31into the Presentation primary, in one <m:semantics>.
32
33Examples:
34  latexmlmath_oxide '1+1=2'
35  latexmlmath_oxide 'a^2+b^2=c^2'
36  latexmlmath_oxide --cmml '\\frac{a}{b}+c'
37
38Convert whole documents with `latexml_oxide` instead.";

43/// Perl `finalize` (`Core/Document.pm` L452, `$node->removeAttribute($n) if $n =~ /^_/`)
44/// drops every `_`-prefixed helper attribute — the ones holding fonts, source boxes, id
45/// bookkeeping — before anything is output. `--xmath` reads the digested document
46/// directly, upstream of our equivalent sweep (`latexml_core::document` L599), so run it
47/// here or `_font="4568…"` leaks into the output where Perl shows none.
48fn strip_internal_attrs(node: &mut libxml::tree::Node) {
49  let internal: Vec<String> = node
50    .get_attributes()
51    .keys()
52    .filter(|n| n.starts_with('_'))
53    .cloned()
54    .collect();
55  for name in internal {
56    let _ = node.remove_attribute(&name);
57  }
58  for mut child in node.get_child_elements() {
59    strip_internal_attrs(&mut child);
60  }
61}

63/// Print `node` alone, with `href` as its DEFAULT namespace — `<math xmlns="…">`, the
64/// way Perl `latexmlmath` does it, not `<m:math>`.
65///
66/// Perl's `outputXML` reparents the node into a fresh document, then
67/// `setNamespaceDeclPrefix($oldprefix, undef)` rewrites the one namespace *declaration*
68/// so every element referencing it drops the prefix at once. Serializing in place
69/// instead emits `<m:math>` with NO `xmlns:m` — the declaration lived on the
70/// `<document>` root we discard — which is not well-formed standalone (`xmllint`:
71/// "Namespace prefix m on math is not defined"). libxml-rs exposes no prefix mutator,
72/// so the equivalent is: declare the default namespace on the extracted root and
73/// re-point every element already in that namespace at it.
74///
75/// Indented like Perl's `toString(1)` via a serialize → re-parse → serialize round trip,
76/// NOT by reparenting into a fresh document. Reparenting is what Perl does and would be
77/// the direct route, but `Document::set_root_element` is `xmlDocSetRootElement`, which
78/// does NOT call `xmlSetTreeDoc`: the subtree keeps pointing at the old document, and
79/// dropping either one frees the other's nodes — `free(): invalid pointer`, SIGABRT on
80/// every run (measured). XML::LibXML's `setDocumentElement` adopts the node; libxml-rs
81/// exposes no equivalent. The round trip is safe because the namespace fix above makes
82/// the serialized node well-formed standalone; `no_blanks` drops the whitespace text
83/// nodes carrying the node's ORIGINAL nesting depth, so it re-indents from column 0
84/// instead of inheriting `--xmath`'s ten-space lead-in.
85fn print_node_defaulted(doc: &libxml::tree::Document, node: &mut libxml::tree::Node, href: &str) {
86  use libxml::tree::Namespace;
87  if let Ok(default_ns) = Namespace::new("", href, node) {
88    fn repoint(node: &mut libxml::tree::Node, ns: &Namespace, href: &str) {
89      if node.get_namespace().is_some_and(|n| n.get_href() == href) {
90        let _ = node.set_namespace(ns);
91      }
92      for mut child in node.get_child_elements() {
93        repoint(&mut child, ns, href);
94      }
95    }
96    repoint(node, &default_ns, href);
97  }
98  let raw = doc.node_to_string(node);
99  let pretty = libxml::parser::Parser::default()
100    .parse_string_with_options(&raw, libxml::parser::ParserOptions {
101      no_blanks: true,
102      ..libxml::parser::ParserOptions::default()
103    })
104    .map(|reparsed| {
105      reparsed.to_string_with_options(SaveOptions {
106        format: true,
107        no_declaration: true, // Perl serializes the NODE — no <?xml …?> preamble
108        ..SaveOptions::default()
109      })
110    })
111    // Never lose the output to a formatting nicety: fall back to the flat form.
112    .unwrap_or(raw);
113  print!("{pretty}");
114}

116fn main() -> Result<()> {
117  // 256 MB stack — see cortex_worker.rs for rationale (#17).
118  std::thread::Builder::new()
119    .stack_size(256 * 1024 * 1024)
120    .spawn(|| real_main().map_err(|e| e.to_string()))
121    .expect("spawn worker thread")
122    .join()
123    .expect("worker thread panicked")
124    .map_err(|s| s.into())
125}

127fn real_main() -> Result<()> {
128  let mut argv: Vec<String> = env::args().skip(1).collect();
129
130  // Parse flags
131  let mut pmml_flag = argv
132    .iter()
133    .any(|a| a == "--pmml" || a == "--presentationmathml");
134  let cmml_flag = argv.iter().any(|a| a == "--cmml" || a == "--contentmathml");
135  let xmath_flag = argv.iter().any(|a| a == "--xmath" || a == "--XMath");
136  let quiet_flag = argv.iter().any(|a| a == "--quiet" || a == "-q");
137  argv.retain(|a| {
138    ![
139      "--pmml",
140      "--presentationmathml",
141      "--cmml",
142      "--contentmathml",
143      "--xmath",
144      "--XMath",
145      "--quiet",
146      "-q",
147    ]
148    .contains(&a.as_str())
149  });
150  // Perl L169: `$pmml = '-' unless (defined $mathimage || ... || $xmath || $unimath)` —
151  // with no conversion option, presentation MathML on stdout is the default.
152  if !pmml_flag && !cmml_flag && !xmath_flag {
153    pmml_flag = true;
154  }
155
156  let log_level = if quiet_flag {
157    log::LevelFilter::Warn
158  } else {
159    log::LevelFilter::Info
160  };
161  latexml_core::util::logger::init(log_level).ok();
162
163  // `--help` before anything else: without it the flag falls through as the formula
164  // and gets typeset, landing the user in TeX's interactive error prompt.
165  if argv.iter().any(|a| a == "--help" || a == "-h") {
166    println!("{USAGE}");
167    process::exit(0);
168  }
169
170  let source = match argv.first() {
171    Some(s) => s.clone(),
172    None => {
173      eprintln!("{USAGE}");
174      process::exit(1);
175    },
176  };
177
178  let mut core_engine = new_test_engine();
179  let (lexemes, mut lex_nodes, xmath_opt, mut doc) =
180    lex_single_tex_formula(&source, &mut core_engine);
181  if lexemes.is_empty() {
182    Error!("latexmlmath", "lex", "No lexemes produced from input");
183    process::exit(1);
184  }
185  if !quiet_flag {
186    eprintln!("lexemes: {lexemes:?}");
187  }
188
189  state::set_nomathparse_flag(false);
190  let mut parser = MathParser::default();
191  match parser.parse_lexemes(lexemes, &lex_nodes, &mut doc) {
192    Ok(Some(parse_tree)) => {
193      let mut xmath = xmath_opt.unwrap();
194      for mut node in xmath.get_child_nodes() {
195        node.unlink();
196      }
197      let xml_tree = parse_tree.into_xmath(&mut xmath, &mut lex_nodes, &mut doc)?;
198      xmath
199        .get_parent()
200        .unwrap()
201        .set_attribute("text", &text_form(&xml_tree, &doc))
202        .unwrap();
203
204      if pmml_flag || cmml_flag {
205        // Post-process with MathML
206        let xml_str = doc.get_document().to_string_with_options(SaveOptions {
207          format: true,
208          ..SaveOptions::default()
209        });
210        use latexml_post::{
211          document::{PostDocument, PostDocumentOptions},
212          processor::Processor,
213        };
214        let post_doc = PostDocument::new_from_string(&xml_str, PostDocumentOptions::default())
215          .expect("parse XML for MathML post-processing");
216        let mut post = latexml_post::Post::new();
217        let mut processors: Vec<Box<dyn Processor>> = Vec::new();
218        // Parallel P+C: the Content processor is a parallel *secondary* of the
219        // Presentation primary, merged into one <m:semantics>/<m:annotation-xml>
220        // by combine_parallel — not an independent pass (which left content as
221        // an orphan <apply> sibling of <m:semantics>). Mirrors latexml_oxide::post.
222        if pmml_flag {
223          let mut presentation = latexml_post::mathml::MathML::new_presentation();
224          if cmml_flag {
225            presentation = presentation.with_secondaries(vec![Box::new(
226              latexml_post::mathml::MathML::new_content().secondary(),
227            )]);
228          }
229          processors.push(Box::new(presentation));
230        } else if cmml_flag {
231          processors.push(Box::new(latexml_post::mathml::MathML::new_content()));
232        }
233        match post.process_chain(vec![post_doc], &mut processors) {
234          Ok(results) => {
235            // Perl emits the math element alone (`findnode('//m:math')`), not the
236            // document around it. With parallel P+C the content markup is annotated
237            // INTO the presentation <m:math>, so one node carries both.
238            match results[0].findnodes("//m:math").into_iter().next() {
239              Some(mut math) => {
240                print_node_defaulted(results[0].get_document(), &mut math, MATHML_NS)
241              },
242              None => {
243                Error!(
244                  "latexmlmath",
245                  "output",
246                  "no <m:math> produced for the formula"
247                );
248                process::exit(1);
249              },
250            }
251          },
252          Err(e) => {
253            eprintln!("MathML post-processing failed: {}", e);
254            process::exit(1);
255          },
256        }
257      }
258
259      // Perl emits XMath after pmml/cmml (bin/latexmlmath L232-241), and the order is
260      // load-bearing here: `print_node_defaulted` REPARENTS the node, so running this
261      // first would tear <XMath> out of `doc` before the MathML path serialises it.
262      // Read straight from the digested document — the MathML path above serialises +
263      // re-parses, which re-materialises indentation whitespace at the node's original
264      // depth and drags it into the extracted subtree.
265      if xmath_flag {
266        strip_internal_attrs(&mut xmath);
267        print_node_defaulted(doc.get_document(), &mut xmath, LATEXML_NS);
268      }
269    },
270    _ => {
271      Warn!("math", "parse", "Grammar did not recognize expression.");
272      process::exit(1);
273    },
274  }
275  Ok(())
