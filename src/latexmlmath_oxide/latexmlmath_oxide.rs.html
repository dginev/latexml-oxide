15fn main() -> Result<()> {
16  // 256 MB stack — see cortex_worker.rs for rationale (#17).
17  std::thread::Builder::new()
18    .stack_size(256 * 1024 * 1024)
19    .spawn(|| real_main().map_err(|e| e.to_string()))
20    .expect("spawn worker thread")
21    .join()
22    .expect("worker thread panicked")
23    .map_err(|s| s.into())
24}

26fn real_main() -> Result<()> {
27  let mut argv: Vec<String> = env::args().skip(1).collect();
28
29  // Parse flags
30  let pmml_flag = argv
31    .iter()
32    .any(|a| a == "--pmml" || a == "--presentationmathml");
33  let cmml_flag = argv.iter().any(|a| a == "--cmml" || a == "--contentmathml");
34  let quiet_flag = argv.iter().any(|a| a == "--quiet" || a == "-q");
35  argv.retain(|a| {
36    ![
37      "--pmml",
38      "--presentationmathml",
39      "--cmml",
40      "--contentmathml",
41      "--quiet",
42      "-q",
43    ]
44    .contains(&a.as_str())
45  });
46
47  let log_level = if quiet_flag {
48    log::LevelFilter::Warn
49  } else {
50    log::LevelFilter::Info
51  };
52  latexml_core::util::logger::init(log_level).ok();
53
54  let source = match argv.first() {
55    Some(s) => s.clone(),
56    None => {
57      eprintln!("Usage: latexmlmath_oxide [--pmml] [--cmml] [--quiet] '<formula>'");
58      process::exit(1);
59    },
60  };
61
62  let mut core_engine = new_test_engine();
63  let (lexemes, mut lex_nodes, xmath_opt, mut doc) =
64    lex_single_tex_formula(&source, &mut core_engine);
65  if lexemes.is_empty() {
66    Error!("latexmlmath", "lex", "No lexemes produced from input");
67    process::exit(1);
68  }
69  if !quiet_flag {
70    eprintln!("lexemes: {lexemes:?}");
71  }
72
73  state::set_nomathparse_flag(false);
74  let mut parser = MathParser::default();
75  match parser.parse_lexemes(lexemes, &lex_nodes, &mut doc) {
76    Ok(Some(parse_tree)) => {
77      let mut xmath = xmath_opt.unwrap();
78      for mut node in xmath.get_child_nodes() {
79        node.unlink();
80      }
81      let xml_tree = parse_tree.into_xmath(&mut xmath, &mut lex_nodes, &mut doc)?;
82      xmath
83        .get_parent()
84        .unwrap()
85        .set_attribute("text", &text_form(&xml_tree, &doc))
86        .unwrap();
87
88      if pmml_flag || cmml_flag {
89        // Post-process with MathML
90        let xml_str = doc.get_document().to_string_with_options(SaveOptions {
91          format: true,
92          ..SaveOptions::default()
93        });
94        use latexml_post::{
95          document::{PostDocument, PostDocumentOptions},
96          processor::Processor,
97        };
98        let post_doc = PostDocument::new_from_string(&xml_str, PostDocumentOptions::default())
99          .expect("parse XML for MathML post-processing");
100        let mut post = latexml_post::Post::new();
101        let mut processors: Vec<Box<dyn Processor>> = Vec::new();
102        // Parallel P+C: the Content processor is a parallel *secondary* of the
103        // Presentation primary, merged into one <m:semantics>/<m:annotation-xml>
104        // by combine_parallel — not an independent pass (which left content as
105        // an orphan <apply> sibling of <m:semantics>). Mirrors latexml_oxide::post.
106        if pmml_flag {
107          let mut presentation = latexml_post::mathml::MathML::new_presentation();
108          if cmml_flag {
109            presentation = presentation.with_secondaries(vec![Box::new(
110              latexml_post::mathml::MathML::new_content().secondary(),
111            )]);
112          }
113          processors.push(Box::new(presentation));
114        } else if cmml_flag {
115          processors.push(Box::new(latexml_post::mathml::MathML::new_content()));
116        }
117        match post.process_chain(vec![post_doc], &mut processors) {
118          Ok(results) => println!("{}", results[0].to_xml_string()),
119          Err(e) => {
120            eprintln!("MathML post-processing failed: {}", e);
121            process::exit(1);
122          },
123        }
124      } else {
125        // Raw XML output
126        println!(
127          "{}",
128          doc.get_document().to_string_with_options(SaveOptions {
129            format: true,
130            ..SaveOptions::default()
131          })
132        );
133      }
134    },
135    _ => {
136      Warn!("math", "parse", "Grammar did not recognize expression.");
137      process::exit(1);
138    },
139  }
140  Ok(())
141}
