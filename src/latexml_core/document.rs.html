131  pub fn new() -> Self {
132    crate::ensure_libxml_init(); // Thread-safe libxml2 initialization
133    // `NODE_RC_MAX_GUARD` is libxml's diagnostic threshold for mutable
134    // access to Rc-shared nodes; the real aliasing guarantee is the
135    // `weak_count == 0` check in `Node::mut_node`. For legitimate large
136    // documents (e.g. arxiv 0805.2376 with deep dcpic commutative-diagram
137    // sharing), natural ref counts exceed the crate's default of 2 by
138    // several orders of magnitude. Raise to 8192 to accommodate.
139    //
140    // libxml-rs implements this as a `pub static mut NODE_RC_MAX_GUARD:
141    // usize` with `unsafe { NODE_RC_MAX_GUARD = value; }` — concurrent
142    // writes from N test threads each constructing their own Document
143    // are a classic data race on a `static mut`. Gate the write through
144    // a `std::sync::Once` so it happens exactly once per process; reads
145    // from `Node::mut_node` then see a stable value.
146    static GUARD_INIT: std::sync::Once = std::sync::Once::new();
147    GUARD_INIT.call_once(|| set_node_rc_guard(8192));
148    let doc_scaffold = XmlDoc::new().unwrap();
149    let root = match doc_scaffold.get_root_element() {
150      Some(root) => root,
151      None => doc_scaffold.as_node(), // when empty, set the document node as a node.
152    };
153    Document {
154      document:                    doc_scaffold,
155      node:                        root,
156      node_boxes:                  HashMap::default(),
157      node_fonts:                  HashMap::default(),

201  /// Get an XPath context that knows about our namespace mappings.
202  pub fn get_xpath(&mut self) -> &mut XPath {
203    if let Some(ref mut ctxt) = self.context {
204      ctxt
205    } else {
206      let mut context = XPath::new(&self.document, HashMap::default());
207      model::with_code_namespaces(|code_ns| {
208        for (prefix, ns) in code_ns {
209          // TODO: Is this too slow? We may need to store an active context in the state as an
210          // alternative
211          arena::with2(*prefix, *ns, |p_str, ns_str| {
212            context
213              .register_namespace(p_str, ns_str)
214              .expect("register_namespace has no reason to fail during get_xpath?");
215          });
216        }
217      });
218      self.context = Some(context);
219      self.context.as_mut().unwrap()
220    }

254  // This should be called before returning the final XML::LibXML::Document to the
255  // outside world.  It resolves the fonts for each node relative to it's
256  // ancestors. It removes the `helper' attributes that store fonts, source
257  // box, etc.
258  pub fn finalize(&mut self) -> Result<()> {
259    // Belt-and-suspenders idstore rebuild before prune_xmduals.
260    // Originally guarded a SIGSEGV on arxiv 1605.08055 where
261    // `mark_xmnode_visibility` dereferenced dangling lookup_id
262    // entries while recursing through XMRef nodes. Cycle 72 audited
263    // the 5 hazard call sites the earlier comment listed (math-parser
264    // `replace_tree` at parser.rs:456/690, `unbind_node` loops at
265    // parser.rs:639/856 and rewrite.rs:522); all 5 now have proper
266    // unrecord_node_ids / remove_node-cascade coverage. The rebuild
267    // call is retained as a safety net pending empirical
268    // 10k-sandbox verification on 1605.08055 — see SYNC_STATUS.md
269    // D3b [~] entry. A fresh DOM walk drops any surviving dangling
270    // entries; duplicates in DOM get modify_id via record_node_ids.
271    self.rebuild_idstore_from_dom()?;
272    // Sweep dangling XMRefs that were specifically created by
273    // amsmath::rearrange_ams_split (tagged with `_split_ref="1"`).
274    // The math parser later absorbs some XMArray cells (inserted
275    // MULOPs, etc.) and the parallel XMWrap refs end up pointing
276    // at vanished targets, cascading through Warn:expected:node
277    // (here) and Error:expected:id (post-process) for ~1500 wp3
278    // canvas papers. Restricting the sweep to `_split_ref` avoids
279    // breaking declare_test's renamed-id case (XMRefs pointing to
280    // `S1.Ex1.m1.1`-style ids that resolve through Perl-faithful
281    // idstore staleness; those don't carry the marker).
282    self.prune_dangling_split_xmrefs()?;
283    self.prune_xmduals()?;
284    if let Some(mut root) = self.document.get_root_element() {
285      self.set_local_font(Rc::new(Font::text_default()));
286      self.finalize_rec(&mut root)?;
287      self.set_rdfa_prefixes();
288      self.apply_document_namespace_declarations(&mut root);
289      self.expire_local_font();
290    }
291    Ok(())
292  }

294  /// Apply registered document namespace declarations to the root element.
295  /// Perl's RegisterDocumentNamespace stores prefix→URI mappings in the model.
296  /// These must appear as xmlns:prefix="URI" on the root element during serialization.
297  /// Only emit namespaces that are actually used in the document (Perl behavior).
298  fn apply_document_namespace_declarations(&self, root: &mut Node) {
299    let prefixes = model::get_document_namespace_prefixes();
300    // Collect which prefixes are actually used in the document
301    let nsnodes = root.get_namespace_declarations();
302    let existing_prefixes: Vec<String> = nsnodes.iter().map(|ns| ns.get_prefix()).collect();
303    for (prefix, ns_uri) in prefixes {
304      // Skip internal/default namespaces
305      if prefix.is_empty() || prefix == "ltx" || prefix == "xml" {
306        continue;
307      }
308      // Skip namespaces containing "DEFAULT#" (internal model entries)
309      if ns_uri.contains("DEFAULT#") {
310        continue;
311      }
312      // Only add if this prefix appears as a namespace declaration on some child node
313      // (meaning it's actually used in the document)
314      if existing_prefixes.contains(&prefix) {
315        continue; // already declared on root
316      }
317      // Check if any descendant element uses this namespace prefix
318      // by looking for namespace declarations on descendant elements
319      let has_usage = self.has_namespace_usage(root, &prefix);
320      if has_usage {
321        let attr_name = format!("xmlns:{prefix}");
322        root.set_attribute(&attr_name, &ns_uri).ok();
323      }
324    }
325  }

327  /// Check if any descendant of node uses the given namespace prefix.
328  fn has_namespace_usage(&self, node: &Node, prefix: &str) -> bool {
329    // Check attributes of this node for prefix: usage. The
330    // allocation-free check `starts_with(prefix) + byte-at-prefix.len()`
331    // replaces `format!("{prefix}:")` which would heap-allocate per node
332    // visited during the recursive descent.
333    let plen = prefix.len();
334    for (key, _) in node.get_attributes() {
335      if key.len() > plen && key.starts_with(prefix) && key.as_bytes()[plen] == b':' {
336        return true;
337      }
338    }
339    // Check children recursively
340    for child in node.get_child_nodes() {
341      if child.get_type() == Some(NodeType::ElementNode) {
342        // Check if element itself is in this namespace
343        for ns in child.get_namespace_declarations() {
344          if ns.get_prefix() == prefix {
345            return true;
346          }
347        }
348        if self.has_namespace_usage(&child, prefix) {
349          return true;
350        }
351      }
352    }
353    false
354  }

356  /// Remove xml:ids from XMTok elements that aren't referenced by any idref.
357  /// The Rust math parser generates xml:ids on XMTok nodes for internal XMRef linkage
358  /// during parsing. After finalization (which includes prune_xmduals), some ids
359  /// are no longer referenced. Perl's parser doesn't generate these ids.
360  pub fn cleanup_unreferenced_xmtok_ids(&mut self) {
361    use rustc_hash::FxHashSet as HashSet;
362    let mut referenced_ids: HashSet<String> = HashSet::default();
363    for node in self.findnodes("descendant-or-self::*[@idref]", None) {
364      if let Some(idref) = node.get_attribute("idref") {
365        referenced_ids.insert(idref);
366      }
367    }
368    let xml_ns = "http://www.w3.org/XML/1998/namespace";
369    let toks = self.findnodes("descendant-or-self::ltx:XMTok[@xml:id]", None);
370    for mut tok in toks {
371      if let Some(id) = tok.get_attribute_ns("id", xml_ns) {
372        if !referenced_ids.contains(&id) {
373          self.unrecord_id(&id);
374          // Remove both the prefixed attribute and the ns attribute
375          let _ = tok.remove_attribute("xml:id");
376          let _ = tok.remove_attribute_ns("id", xml_ns);
377        }
378      }
379    }
380  }

382  /// Iterative implementation of finalize_rec to avoid stack overflow.
383  /// Uses an explicit heap-allocated work stack instead of call-stack recursion.
384  /// Resolves fonts for each node relative to its ancestors, and removes
385  /// helper attributes (_font, _standalone_font, etc).
386  fn finalize_rec(&mut self, node: &mut Node) -> Result<()> {
387    // Work items for the iterative traversal.
388    // Enter: process font declarations and children (text children handled inline,
389    //        element children deferred to the stack).
390    // PostElement: after finalizing an element child, check for font wrapper collapse.
391    // PostWork: after all children processed, remove bookkeeping attrs and expire font.
392    #[allow(clippy::enum_variant_names)]
393    enum Work {
394      Enter(Node),
395      PostElement {
396        child:         Node,
397        parent_qname:  SymStr,
398        was_forcefont: bool,
399      },
400      PostWork {
401        node:              Node,
402        /// Bookkeeping attribute names captured during Enter, before tree modifications
403        bookkeeping_attrs: Vec<String>,
404      },
405    }
406
407    let mut stack: Vec<Work> = Vec::new();
408    // Track nodes that have no visible attributes after bookkeeping removal.
409    // Used in PostElement to decide collapse without calling get_attributes()
410    // (which can crash on corrupted libxml2 attribute lists after replace_node).
411    let mut empty_attr_nodes: HashSet<usize> = HashSet::default();
412    // Defer collapse operations to avoid libxml2 memory corruption.
413    // Collapsing nodes during tree traversal corrupts attribute linked lists
414    // of ancestor nodes. By collecting collapses and running them after
415    // the entire traversal completes, we avoid accessing corrupted state.
416    let mut deferred_collapses: Vec<(Node, SymStr)> = Vec::new();
417    stack.push(Work::Enter(node.clone()));
418
419    while let Some(work) = stack.pop() {
420      match work {
421        Work::Enter(mut current) => {
422          let qname = get_node_qname(&current);
423          let local_font = self.get_local_font().unwrap();
424          // _standalone_font is typically for metadata that gets extracted out of context
425          let mut declared_font = if current.has_attribute("_standalone_font") {
426            Cow::Borrowed(&*FONT_TEXT_DEFAULT)
427          } else {
428            Cow::Borrowed(&*local_font)
429          };
430
431          // TODO: _pre_comment / _comment insertion requires create_comment support in libxml
432          // wrapper Perl: parent.insertBefore(XML::LibXML::Comment.new(comment), node)
433          // Perl: parent.insertAfter(XML::LibXML::Comment.new(comment), node)
434
435          // Use boxed HashMap to reduce work item size — Font is ~500 bytes per entry
436          let mut pending_declaration: Box<HashMap<String, (String, Font)>> = Box::default();
437
438          if self.has_node_font(&current) {
439            let desired_font = self.get_node_font(&current);
440            *pending_declaration = desired_font.relative_to(&declared_font);
441            if (!current.get_child_nodes().is_empty() || current.has_attribute("_force_font"))
442              && !pending_declaration.is_empty()
443            {
444              let mut keys_to_remove: Vec<SymStr> = Vec::new();
445              let mut attrs_to_set: Vec<(SymStr, SymStr)> = Vec::new();
446              for (key, (value, properties)) in pending_declaration.iter() {
447                if model::can_have_attribute(qname, arena::pin(key)) {
448                  let key_sym = arena::pin(key);
449                  attrs_to_set.push((key_sym, arena::pin(value)));
450                  // Merge to set the font currently in effect
451                  declared_font = Cow::Owned(declared_font.merge_ref(properties));
452                  keys_to_remove.push(key_sym);
453                }
454              }
455
456              for (key, mut value) in attrs_to_set {
457                if key == pin!("class") {
458                  // Merge and sort class values alphabetically, matching Perl's behavior
459                  if let Some(ovalue) = current.get_attribute("class") {
460                    let new_s = arena::with(value, |s| s.to_string());
461                    let mut classes: Vec<&str> = new_s
462                      .split_whitespace()
463                      .chain(ovalue.split_whitespace())
464                      .collect();
465                    classes.sort_unstable();
466                    classes.dedup();
467                    value = arena::pin(classes.join(" "));
468                  }
469                }
470                // Resolve to owned Strings before calling set_attribute,
471                // to avoid holding an arena borrow while set_attribute may need arena::pin
472                // for schema-based attribute filtering (canHaveAttribute check).
473                let key_s = arena::with(key, |s| s.to_string());
474                let value_s = arena::with(value, |s| s.to_string());
475                self.set_attribute(&mut current, &key_s, &value_s)?;
476              }
477              for key in keys_to_remove {
478                arena::with(key, |key_str| pending_declaration.remove(key_str));
479              }
480            }
481          }
482          // Optionally add ids to all nodes (AFTER all parsing, rearrangement, etc)
483          if qname != pin!("ltx:document")
484            && state::lookup_bool("GENERATE_IDS")
485            && !current.has_attribute("xml:id")
486            && !current.has_attribute("id") // SVG elements with plain id don't need xml:id
487            && arena::with(qname, |qname_str| can_have_attribute(qname_str, "xml:id"))
488          {
489            self.generate_id(&mut current, "")?;
490          }
491          self.set_local_font(Rc::new(declared_font.into_owned()));
492
493          // Capture bookkeeping attribute names now, before tree modifications.
494          // This avoids calling get_attributes() later when the attribute list
495          // might be corrupted by replace_node operations.
496          let bookkeeping_attrs: Vec<String> = current
497            .get_attributes()
498            .into_keys()
499            .filter(|name| name.starts_with('_'))
500            .collect();
501
502          // Process children using the snapshot from get_child_nodes().
503          // Element children are deferred to the stack; text children are handled inline.
504          // PostWork is pushed first (runs after all children), then children in reverse.
505          stack.push(Work::PostWork {
506            node: current.clone(),
507            bookkeeping_attrs,
508          });
509
510          let children = current.get_child_nodes();
511          // Collect work items forward, then push in reverse for left-to-right processing
512          let mut child_work: Vec<Work> = Vec::new();
513          for child in &children {
514            let child_type = child.get_type();
515            if child_type == Some(NodeType::ElementNode) {
516              let was_forcefont = child.has_attribute("_force_font");
517              // Enter first, PostElement after (reversed when pushed to stack)
518              child_work.push(Work::Enter(child.clone()));
519              child_work.push(Work::PostElement {
520                child: child.clone(),
521                parent_qname: qname,
522                was_forcefont,
523              });
524            } else if child_type == Some(NodeType::TextNode) {
525              // Text node: wrap with font element if needed (handled inline)
526              let mut text_keys_to_remove = Vec::new();
527              for key in pending_declaration.keys() {
528                if !can_have_attribute(FONT_ELEMENT_NAME, key) {
529                  text_keys_to_remove.push(key.clone());
530                }
531              }
532              for key in text_keys_to_remove {
533                pending_declaration.remove(&key);
534              }
535              if can_contain(&current, FONT_ELEMENT_NAME) && !pending_declaration.is_empty() {
536                if let Some(mut text) = self.wrap_nodes(FONT_ELEMENT_NAME, vec![child.clone()])? {
537                  for (key, (value, _properties)) in pending_declaration.iter() {
538                    self.set_attribute(&mut text, key, value)?;
539                  }
540                  // Text wrapper finalization is shallow (only text content), push to stack
541                  child_work.push(Work::Enter(text));
542                }
543              }
544            }
545          }
546          // Push in reverse so left-to-right processing order is maintained
547          for work_item in child_work.into_iter().rev() {
548            stack.push(work_item);
549          }
550        },
551        Work::PostElement {
552          child,
553          parent_qname,
554          was_forcefont,
555        } => {
556          // After finalizing a child element, check if it should be collapsed.
557          // Use empty_attr_nodes set instead of child.get_attributes().is_empty()
558          // to avoid traversing the attribute linked list, which can be corrupted
559          // by prior replace_node operations in libxml2.
560          // Defer the actual collapse to after the traversal completes, since
561          // replace_node can corrupt ancestor attribute lists in libxml2.
562          if (get_node_qname(&child) == pin!("ltx:text"))
563            && !was_forcefont
564            && empty_attr_nodes.contains(&child.to_hashable())
565          {
566            let grandchildren = child.get_child_nodes();
567            if grandchildren
568              .iter()
569              .all(|gchild| can_contain_qsym(parent_qname, get_node_qname(gchild)))
570            {
571              deferred_collapses.push((child, parent_qname));
572            }
573          }
574        },
575        Work::PostWork {
576          mut node,
577          bookkeeping_attrs: _captured_at_enter,
578        } => {
579          // Mirrors Perl `Document.pm:452`: at finalize time, ANY attribute
580          // whose name starts with `_` is internal bookkeeping and gets
581          // stripped. Re-derive the set at PostWork time rather than relying
582          // on the captured-at-Enter snapshot, because descendant
583          // `generate_id` calls can write `_ID_counter_<prefix>_` attributes
584          // ONTO the current node (their nearest-id-bearing ancestor) during
585          // child traversal — those late-added attrs were missing from the
586          // Enter snapshot and would otherwise leak into the output XML,
587          // causing duplicate xml:id collisions in the post-processing
588          // libxml2 validator (1312.5864 cluster: 70× `S8.T5.m2241 already
589          // defined`, where the Math element carried `_ID_counter__="1"`
590          // populated post-Enter).
591          let attrs_now = node.get_attributes();
592          let total_attrs = attrs_now.len();
593          let bookkeeping_attrs: Vec<&String> = attrs_now
594            .keys()
595            .filter(|name| name.starts_with('_'))
596            .collect();
597          let bookkeeping_count = bookkeeping_attrs.len();
598          for name in bookkeeping_attrs {
599            let _ = node.remove_attribute(name);
600          }
601          // If all attributes were bookkeeping, the node now has empty attrs.
602          if total_attrs <= bookkeeping_count {
603            empty_attr_nodes.insert(node.to_hashable());
604          }
605          self.expire_local_font();
606        },
607      }
608    }
609    // Execute deferred font wrapper collapses now that the entire tree has been
610    // finalized. Process from deepest (last) to shallowest (first) to avoid
611    // corrupting ancestor attribute lists during the replacement operations.
612    // This deferred approach prevents libxml2 memory corruption that occurs
613    // when replace_node runs during tree traversal.
614    for (child, _parent_qname) in deferred_collapses.into_iter().rev() {
615      let grandchildren = child.get_child_nodes();
616      if grandchildren.is_empty() {
617        // Empty font wrapper — just remove it
618        self.remove_node(child);
619      } else {
620        Debug!(
621          "will replace {} grandchildren nodes in finalize_rec (deferred)",
622          grandchildren.len()
623        );
624        self.replace_node(child, grandchildren)?;
625      }
626    }
627    Ok(())
628  }

630  /// Document construction at the Current Insertion Point.
631  ///
632  /// absorb the given $box into the DOM (called from constructors).
633  /// This will return a list of whatever nodes were created.
634  /// Note that this may include nodes that are children of other nodes in the list
635  /// or nodes that are no longer in the document.
636  /// Also, note that when a text nodes is appended to, the complete text node is in the list,
637  /// not just the portion that was added.
638  /// [Note that recording the nodes being constructed isn't all that costly,
639  /// but filtering them for parent/child relations IS, particularly since it usually isn't needed]
640  ///
641  /// A box that is a TBox, or List, or Whatsit, is responsible for carrying out
642  /// its own insertion, but it should ultimately call methods of Document
643  /// that will record the nodes that were created.
644  /// $box can also be a plain string (Digested::Postponed)
645  /// which will be inserted according to whatever
646  /// font, mode, etc, are in %props.
647  pub fn absorb(&mut self, object: &Digested, props_opt: Option<SymHashMap<Stored>>) -> Result<()> {
648    use DigestedData::*;
649    let props = props_opt.unwrap_or_default();
650    let mut boxes = vec![Cow::Borrowed(object)];
651    while let Some(front_box) = boxes.pop() {
652      match front_box.data() {
653        List(ref list) => {
654          // Simply unwind Lists to avoid unneccessary recursion; This occurs quite frequently!
655          for tbox in list.borrow().unlist().into_iter().rev() {
656            boxes.push(Cow::Owned(tbox));
657          }
658        },
659        // A Proper Box or Whatsit? Absorb it.
660        TBox(ref digested) => {
661          self.set_box_to_absorb(Some((*front_box).clone()));
662          self.init_constructed_nodes();
663          digested.borrow().be_absorbed(self)?;
664          // record these for OUTER caller, but return only the most recent set
665          self.close_constructed_nodes();
666          self.expire_box_to_absorb();
667        },
668        Whatsit(ref digested) => {
669          self.set_box_to_absorb(Some((*front_box).clone()));
670          self.init_constructed_nodes();
671          digested.borrow().be_absorbed(self)?;
672          self.close_constructed_nodes();
673          self.expire_box_to_absorb();
674        },
675        Alignment(ref alignment) => {
676          self.set_box_to_absorb(Some((*front_box).clone()));
677          self.init_constructed_nodes();
678          alignment.borrow_mut().be_absorbed_mut(self)?;
679          self.close_constructed_nodes();
680          self.expire_box_to_absorb();
681        },
682        Comment(ref comment) => {
683          comment.be_absorbed(self)?;
684        },
685        Postponed(ref tokens) => {
686          let text_font_opt = if let Some(Stored::Font(ref prop_font)) = props.get("font") {
687            Some(Rc::clone(prop_font))
688          } else {
689            match self.box_to_absorb {
690              Some(ref thisbox) => thisbox
691                .get_font()?
692                .map(|thisfont| Rc::new(thisfont.into_owned())),
693              None => None,
694            }
695          };
696          let text_font = text_font_opt.unwrap_or_default();
697          if let Some(new_text) = self.open_text(&tokens.to_string(), &text_font)? {
698            self.record_constructed_node(&new_text);
699          }
700        },
701        KeyVals(ref kv) => {
702          // When KeyVals appear in body absorption (e.g. #1 for RequiredKeyVals),
703          // convert them to text representation matching Perl's stringify behavior.
704          let text = kv.to_string();
705          if !text.is_empty() {
706            let text_font = match self.box_to_absorb {
707              Some(ref thisbox) => thisbox
708                .get_font()?
709                .map(|f| Rc::new(f.into_owned()))
710                .unwrap_or_default(),
711              None => Rc::default(),
712            };
713            if let Some(new_text) = self.open_text(&text, &text_font)? {
714              self.record_constructed_node(&new_text);
715            }
716          }
717        },
718        RegisterValue(_) => {
719          // RegisterValue should not normally appear in the absorption pipeline.
720        },
721      }
722    }
723    Ok(())
724  }

726  fn init_constructed_nodes(&mut self) {
727    // Pop a buffer from the free-list (retaining its previously-grown capacity);
728    // fall back to a fresh empty Vec if the pool is dry. Swap it in as the new
729    // inner frame; the outgoing outer frame goes onto the save stack.
730    let fresh = self.reusable_node_buffers.pop().unwrap_or_default();
731    let prev = std::mem::replace(&mut self.constructed_nodes, fresh);
732    self.localized_constructed_nodes.push(prev);
733  }
734
735  /// Close the current constructed-nodes frame, restoring the outer frame and
736  /// re-recording the inner frame's nodes into it. The drained inner buffer
737  /// is returned to `reusable_node_buffers` (empty but with capacity intact)
738  /// so the next `init_constructed_nodes` can reuse it without allocating.
739  fn close_constructed_nodes(&mut self) {
740    let outer = self.localized_constructed_nodes.pop().unwrap_or_default();
741    let mut inner = std::mem::replace(&mut self.constructed_nodes, outer);
742    for n in inner.drain(..) {
743      self.record_constructed_node(&n);
744    }
745    // `inner` is now empty; its capacity is preserved. Return it to the pool.
746    self.reusable_node_buffers.push(inner);
747  }

752  pub fn absorb_string(
753    &mut self,
754    object: &str,
755    props: &SymHashMap<Stored>,
756  ) -> Result<Option<Node>> {
757    // Else, plain string in text mode.
758    let ismath: bool = match props.get("isMath") {
759      Some(v) => v.into(),
760      None => false,
761    };
762    if !ismath {
763      // Perf: avoid cloning Rc<Font> into owned Font in the common case.
764      // We pull out the Rc (shared reference is fine since open_text only
765      // borrows the Font, not self) and go through Cow<Font>.
766      let font_opt: Option<Rc<Font>> = match props.get("font") {
767        Some(Stored::Font(fnt)) => Some(Rc::clone(fnt)),
768        Some(Stored::FontDirective(FontDirective::Asset(fnt))) => Some(Rc::clone(fnt)),
769        _ => None,
770      };
771      if let Some(fnt) = font_opt {
772        return self.open_text(object, &fnt);
773      }
774      if let Some(Stored::FontDirective(FontDirective::Closure(code))) = props.get("font") {
775        let fnt = code(None)?;
776        return self.open_text(object, &fnt);
777      }
778      // Fallback to box_to_absorb font.
779      let fnt = self
780        .box_to_absorb
781        .as_ref()
782        .unwrap()
783        .get_font()?
784        .unwrap()
785        .into_owned();
786      self.open_text(object, &fnt)
787    } else if get_node_qname(&self.node) == pin!("ltx:XMTok") {
788      // Or plain string in math mode.
789      // Note text nodes can ONLY appear in <XMTok> or <text>!!!
790      // Have we already opened an XMTok? Then insert into it.
791      Ok(Some(self.open_math_text_internal(object)?))
792    // Else create the XMTok now.
793    } else {
794      // Odd case: constructors that work in math & text can insert raw strings in Math mode.
795      let font_math_opt = match props.get("font") {
796        Some(Stored::Font(fnt)) => Some(Cow::Borrowed(&**fnt)),
797        Some(Stored::FontDirective(FontDirective::Asset(fnt))) => Some(Cow::Borrowed(&**fnt)),
798        Some(Stored::FontDirective(FontDirective::Closure(code))) => Some(Cow::Owned(code(None)?)),
799        _ => None,
800      };
801      if let Some(font_math) = font_math_opt {
802        Ok(Some(self.insert_math_token(
803          object,
804          HashMap::default(),
805          Some(&font_math),
806        )?))
807      } else {
808        Ok(Some(self.insert_math_token(
809          object,
810          HashMap::default(),
811          None,
812        )?))
813      }
814    }
815  }

878  /// Does NOT move the current insertion point to the PI,
879  /// but may move up past a text node.
880  // Rust note: attrib would have been best as Vec<(String,String)> but
881  // currently quote!() doesn't work out of the box on them
882  pub fn insert_pi(
883    &mut self,
884    op: &str,
885    attributes_opt: Option<HashMap<String, String>>,
886  ) -> Result<()> {
887    let mut attr_data = Vec::new();
888    if let Some(attributes) = attributes_opt {
889      let mut keys = vec!["class", "package", "options"];
890      let other_keys = attributes
891        .keys()
892        .filter(|k| k.as_str() != "class" && k.as_str() != "package" && k.as_str() != "options")
893        .map(String::as_str)
894        .collect::<Vec<_>>();
895      keys.extend(other_keys);
896      for key in keys {
897        if let Some(value) = attributes.get(key) {
898          attr_data.push(s!("{}=\"{}\"", key, value));
899        }
900      }
901    }
902    // self.close_text_internal();  // Close any open text node
903    let mut pi_node = self
904      .document
905      .create_processing_instruction(op, &attr_data.join(" "))
906      .unwrap();
907    if self.node.get_type() == Some(NodeType::DocumentNode) {
908      self.pending.push(pi_node);
909    } else {
910      // Perl: insertPI always places PIs before the root element.
911      // Find the document root and insert before it.
912      let doc_node = self.document.clone();
913      if let Some(mut root) = doc_node.get_root_element() {
914        root.add_prev_sibling(&mut pi_node)?;
915      } else {
916        self.node.add_prev_sibling(&mut pi_node)?;
917      }
918    }
919    Ok(())
920  }

948  /// Stamp a freshly-opened element with its source range as a
949  /// `data-sourcepos` attribute, for the `--source-map` feature (issues
950  /// #47/#92). The range comes from the construct currently being absorbed
951  /// (`box_to_absorb`); the integer file `tag` is resolved through the
952  /// document-level `sources` table (`state::source_tag`) so no path is
953  /// inlined. See `docs/SOURCE_PROVENANCE.md` §0/§2.
954  ///
955  /// Math is kept **opaque** per the MVP scope: the `ltx:Math` wrapper is
956  /// stamped, but its `ltx:XM*` MathML internals are skipped (the Marpa
957  /// math parser has no locator awareness — §7 A.3). Only invoked when the
958  /// source-map switch is on (the caller gates it).
959  fn stamp_source_locator(&mut self, node: &Node, qname: &str) {
960    // Math internals: stamp only the leaf token elements (`ltx:XMTok` — the
961    // operators / identifiers / numbers) when token-locators gives them a real
962    // located box locator (the math char's source origin). This is the per-token
963    // in-equation provenance step (§7 A.3). The structural XM* (XMApp/XMDual/
964    // XMArray) are rebuilt by the Marpa parser — created directly, not via
965    // `open_element` — so they never reach here; the remaining digestion-built
966    // wrappers (XMArg/XMHint/XMText/XMRef/XMWrap) stay opaque. The `data:sourcepos`
967    // rides the XMTok element through the parser's restructuring (attribute on a
968    // reparented node) and through the XMath→MathML XSLT.
969    //
970    // Gated at compile time: feature-OFF keeps math fully opaque (the MVP scope
971    // and the golden's math-opacity assertion); only the token-locators build
972    // exposes the located XMTok leaves.
973    #[cfg(not(feature = "token-locators"))]
974    if qname.starts_with("ltx:XM") {
975      return;
976    }
977    #[cfg(feature = "token-locators")]
978    if qname.starts_with("ltx:XM") && qname != "ltx:XMTok" {
979      return;
980    }
981    // Read the pre-captured Copy locator — never re-borrow the box here.
982    let Some(loc) = self.current_box_locator else {
983      return;
984    };
985    // Skip locators with no real source position (default/synthetic).
986    if loc.from_line == 0 {
987      return;
988    }
989    // User-source only (§7.B): emit a navigable locator only into an editable
990    // user document — `.tex`/`.ltx`, plus the bibliography sources `.bbl` (the
991    // BibTeX-generated, but author-editable, list of `\bibitem`s) and `.bib`
992    // (BibTeX database entries). All four are files the editor may legitimately
993    // scroll into. This skips both synthetic default locators (whose source is
994    // `…/locator.rs`, from `Locator::default()`'s `file!()`) and foreign
995    // package/class files (`.sty`/`.cls`/…) — the editor must never scroll into
996    // those. Foreign/unstamped elements inherit their nearest user-source
997    // ancestor's range client-side (DOM walk-up). (MVP heuristic; a tracked
998    // user-input set would be more precise.)
999    let src = loc.get_source();
1000    let is_user_source = arena::with(src, |s| {
1001      let s = s.to_ascii_lowercase();
1002      s.ends_with(".tex") || s.ends_with(".ltx") || s.ends_with(".bbl") || s.ends_with(".bib")
1003    });
1004    if !is_user_source {
1005      return;
1006    }
1007    let tag = state::source_tag(src);
1008    // Emit in LaTeXML's `data:` namespace (`http://dlmf.nist.gov/LaTeXML/data`,
1009    // registered in `base_schema.rs:19`). The post XSLT's `copy_foreign_attributes`
1010    // path converts a `data:`-prefixed *foreign-namespaced* attribute to the HTML
1011    // `data-sourcepos` attribute (`LaTeXML-common.xsl`: `data:` prefix → `data-…`
1012    // when `USE_DATA_ATTRIBUTES` = true, i.e. HTML5). Faithful to Perl LaTeXML's
1013    // foreign-attribute convention; no XSLT change needed. The general namespaced-
1014    // attribute binding lives in `set_attribute` (shared with `aria:` etc.).
1015    let mut n = node.clone();
1016    let _ = self.set_attribute(&mut n, "data:sourcepos", &loc.to_sourcepos(tag));
1017  }

1019  /// Note: This closes the deepest open node of a given type.
1020  /// This can cause problems with auto-opened nodes, esp. ones for fontswitches!
1021  /// Since this is an "explicit request", we're currently skipping over those nodes,
1022  /// ie. we're automatically closing them, even if they're the same type as we're asking to
1023  /// close!!! This is kinda risky! Maybe we should try to request closing of specific nodes.
1024  pub fn close_element(&mut self, qname: &str) -> Result<Option<Node>> {
1025    Debug!(
1026      "document",
1027      "close_element",
1028      s!(
1029        "Close element {:?} at {:?}",
1030        qname,
1031        self.document.node_to_string(&self.node)
1032      )
1033    );
1034    let qsym = arena::pin(qname);
1035    self.close_text_internal()?;
1036    let mut node = self.node.clone();
1037    let mut cant_close = Vec::new();
1038    while node.get_type() != Some(NodeType::DocumentNode) {
1039      let t = get_node_qname(&node);
1040      // autoclose until node of same name BUT also close nodes opened' for font
1041      // switches!
1042      if t == qsym && !(t == pin!("ltx:text") && node.has_attribute("_fontswitch")) {
1043        break;
1044      }
1045      if !can_auto_close(&node) {
1046        cant_close.push(node.clone());
1047      }
1048      match node.get_parent() {
1049        Some(parent) => {
1050          node = parent;
1051        },
1052        None => break, // detached node — treat as not found
1053      }
1054    }
1055
1056    if node.get_type() == Some(NodeType::DocumentNode) {
1057      // Didn't find $qname at all!!
1058      let qname_msg: String = match qname {
1059        "#PCDATA" => qname.to_owned(),
1060        _ => s!("</{qname}>"),
1061      };
1062      let message = s!(
1063        "Attempt to close {}, which isn't open. Currently in {}",
1064        qname_msg,
1065        self.get_insertion_context(None)?
1066      );
1067      Error!("malformed", qname, message);
1068      Ok(None)
1069    } else {
1070      // Found node.
1071      if !cant_close.is_empty() {
1072        // Intervening non-auto-closeable nodes!!
1073        let message = s!(
1074          "Closing tag {:?} whose open descendents do not auto-close. Descendants are {:?}",
1075          qname,
1076          cant_close
1077            .into_iter()
1078            .map(|n| n.get_name())
1079            .collect::<Vec<String>>()
1080            .join(",")
1081        );
1082        Error!("malformed", qname, message);
1083      }
1084      // So, now close up to the desired node.
1085      self.close_node_internal(&node)?;
1086      Ok(Some(node))
1087    }
1088  }

1110  /// returning the last Some(node) that would be closed if it is possible,
1111  /// otherwise None
1112  pub fn is_closeable<T: IntoVDQS>(&self, tags: T) -> Option<Node> {
1113    let mut tags: VecDeque<SymStr> = tags.into_vdqs();
1114    let mut node_opt = if self.node.get_type() == Some(NodeType::TextNode) {
1115      self.node.get_parent()
1116    } else {
1117      Some(self.node.clone())
1118    };
1119    while let Some(qname) = tags.pop_front() {
1120      'inner: loop {
1121        let node: &Node = match node_opt {
1122          None => break,
1123          Some(ref n) => n,
1124        };
1125        let node_type = node.get_type();
1126        if node_type == Some(NodeType::DocumentNode) || node_type.is_none() {
1127          return None;
1128        }
1129        let this_qname = get_node_qname(node);
1130        if this_qname == qname {
1131          break 'inner;
1132        }
1133        if !can_auto_close(node) {
1134          Debug!(
1135            "It was impossible to autoclose node: {:?}",
1136            self.document.node_to_string(node)
1137          );
1138          return None;
1139        }
1140        node_opt = node.get_parent();
1141      }
1142      if !tags.is_empty() {
1143        if let Some(node) = node_opt {
1144          node_opt = node.get_parent();
1145        }
1146      }
1147    }
1148    node_opt
1149  }

1161  /// Closes all nodes until $node becomes the current point.
1162  pub fn close_to_node(&mut self, node: &Node, ifopen: bool) -> Result<()> {
1163    let mut cant_close = Vec::new();
1164    let mut lastopen: Option<Node> = None;
1165    let mut n = self.node.clone();
1166    let mut n_type = n.get_type();
1167    // go up the tree from current node, till we find `node`
1168    while n_type != Some(NodeType::DocumentNode) && &n != node {
1169      if !can_auto_close(&n) {
1170        cant_close.push(n.clone());
1171      }
1172      lastopen = Some(n.clone());
1173      if let Some(p) = n.get_parent() {
1174        n = p;
1175        n_type = n.get_type();
1176      } else {
1177        break;
1178      }
1179    }
1180    if n_type == Some(NodeType::DocumentNode) {
1181      // Didn't find $node at all!!
1182      // Perl: suppress error when $ifopen is true
1183      if !ifopen {
1184        let message = s!("Attempt to close {:?}, which isn't open", node.get_name());
1185        arena::with(get_node_qname(node), |qname_str| {
1186          {
1187            Error!("malformed", qname_str, message)
1188          };
1189          Ok(())
1190        })?;
1191      }
1192    } else {
1193      // Found node.
1194      if !cant_close.is_empty() {
1195        // But found has intervening non-auto-closeable nodes!!
1196        let qname = get_node_qname(node);
1197        let message = s!(
1198          "Closing {:?} whose open descendents do not auto-close. Descendants are: {:?}",
1199          qname,
1200          cant_close
1201            .into_iter()
1202            .map(|n| n.get_name())
1203            .collect::<Vec<String>>()
1204            .join(",")
1205        );
1206        arena::with(qname, |qname_str| {
1207          {
1208            Error!("malformed", qname_str, message)
1209          };
1210          Ok(())
1211        })?;
1212      }
1213      if let Some(lastopen_node) = lastopen {
1214        self.close_node_internal(&lastopen_node)?;
1215      }
1216    }
1217    Ok(())
1218  }

1229  pub fn close_node_with_strictness(&mut self, strict: bool, node: &Node) -> Result<()> {
1230    // Perl: my ($t, @cant_close) = (); ... while ((($t = $n->getType) != XML_DOCUMENT_NODE) ...
1231    let mut cant_close: Vec<Node> = Vec::new();
1232    let mut n = self.node.clone();
1233    let mut t = n.get_type(); // track walker node type, not target
1234    while t.is_some() && t != Some(NodeType::DocumentNode) && &n != node {
1235      if !can_auto_close(&n) {
1236        cant_close.push(n.clone());
1237      }
1238      match n.get_parent() {
1239        Some(parent) => {
1240          n = parent;
1241          t = n.get_type();
1242        },
1243        None => {
1244          t = None; // detached node — stop walking
1245        },
1246      }
1247    }
1248
1249    if t == Some(NodeType::DocumentNode) || t.is_none() {
1250      // Didn't find $qname at all!!
1251      if strict {
1252        let qname = get_node_qname(node);
1253        arena::with(qname, |qname_str| {
1254          let message = s!(
1255            "Attempt to close {}, which isn't open. Currently in {:?}",
1256            qname_str,
1257            self.get_insertion_context(None)?
1258          );
1259          {
1260            Error!("malformed", qname_str, message)
1261          };
1262          Ok(())
1263        })?;
1264      }
1265    } else {
1266      // Found node.
1267      // Intervening non-auto-closeable nodes!!
1268      if !cant_close.is_empty() {
1269        model::with_node_qname(node, |qname| {
1270          let message = s!(
1271            "Closing {} whose open descendents do not auto-close. Descendents are {}",
1272            qname,
1273            cant_close
1274              .iter()
1275              .map(Node::get_name)
1276              .collect::<Vec<String>>()
1277              .join(", ")
1278          );
1279          if strict {
1280            Error!("malformed", qname, message);
1281          } else {
1282            Info!("malformed", qname, message);
1283          }
1284          Ok(())
1285        })?;
1286      }
1287      self.close_node_internal(node)?;
1288    }
1289    Ok(())
1290  }

1352  pub fn serialize_to_string(&self) -> String {
1353    // This line is to use libxml2's built-in serializer w/indentation heuristic.
1354    // Apparently, libxml2 is giving us "binary" or byte strings which we'd prefer
1355    // to have as text. return decode('UTF-8',
1356    // $self->getDocument->toString($format)); } This uses our own serializer
1357    // emulating libxml2's heuristic indentation.
1358    // This uses our own serializer with the correct schema-based indentation rules:
1359    // noindent_children=true when the element can contain #PCDATA per the schema.
1360    let result = self.serialize_aux(&self.document.as_node(), 0, false, false);
1361    // Trim trailing newline (the root element adds \n after </document>
1362    // but Perl doesn't include it)
1363    result.trim_end_matches('\n').to_string() + "\n"
1364  }

1366  /// We ought to try for something close to C14N (<http://www.w3.org/TR/xml-c14n>),
1367  /// but keep XML declaration, comments and don't convert empty elements.
1368  pub fn serialize_aux(
1369    &self,
1370    node: &Node,
1371    depth: usize,
1372    noindent: bool,
1373    heuristic: bool,
1374  ) -> String {
1375    let indent = "  ".repeat(depth);
1376    let mut serialized = String::new();
1377
1378    match node.get_type() {
1379      Some(NodeType::DocumentNode) => {
1380        serialized.push_str("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
1381        if let Some(child) = node.get_first_child() {
1382          let child_serialized = self.serialize_aux(&child, depth, noindent, heuristic);
1383          serialized.push_str(&child_serialized);
1384          let mut current_child = child;
1385          while let Some(sibling) = current_child.get_next_sibling() {
1386            let sibling_serialized = self.serialize_aux(&sibling, depth, noindent, heuristic);
1387            serialized.push_str(&sibling_serialized);
1388            current_child = sibling;
1389          }
1390        }
1391      },
1392      Some(NodeType::ElementNode) => {
1393        // Get the qualified name (prefix:localname) for namespace-prefixed elements
1394        let local_name = node.get_name();
1395        let tag = if let Some(ns) = node.get_namespace() {
1396          let prefix = ns.get_prefix();
1397          if prefix.is_empty() {
1398            local_name
1399          } else {
1400            s!("{}:{}", prefix, local_name)
1401          }
1402        } else {
1403          local_name
1404        };
1405        let children = node.get_child_nodes();
1406        let mut open_tag = s!("<{tag}");
1407
1408        let nsnodes = node.get_namespace_declarations();
1409        for ns in nsnodes {
1410          let prefix = ns.get_prefix();
1411          let prefix_declaration = if prefix.is_empty() {
1412            s!("xmlns")
1413          } else {
1414            s!("xmlns:{}", prefix)
1415          };
1416          let href = ns.get_href();
1417          write!(open_tag, " {prefix_declaration}=\"{href}\"").ok();
1418        }
1419
1420        let anodes = node.get_attributes();
1421        let mut anodes_keys: Vec<&String> = anodes.keys().collect();
1422        // Sort: xmlns:* declarations first (matching Perl's output order), then alphabetically
1423        anodes_keys.sort_by(|a, b| {
1424          let a_is_xmlns = a.starts_with("xmlns:");
1425          let b_is_xmlns = b.starts_with("xmlns:");
1426          match (a_is_xmlns, b_is_xmlns) {
1427            (true, false) => std::cmp::Ordering::Less,
1428            (false, true) => std::cmp::Ordering::Greater,
1429            _ => a.cmp(b),
1430          }
1431        });
1432        for key in anodes_keys {
1433          // Plain "id" is duplicated by libxml2 when xml:id is set.
1434          // Skip it for non-SVG elements (which use xml:id).
1435          // SVG elements use plain "id" (not xml:id).
1436          if key == "id" && !tag.starts_with("svg:") {
1437            continue;
1438          }
1439          // For SVG elements that have plain "id", skip the xml:id duplicate
1440          if key == "xml:id" && tag.starts_with("svg:") && node.get_attribute("id").is_some() {
1441            continue;
1442          }
1443          let key_sym = model::get_node_document_qname(&node.get_attribute_node(key).unwrap());
1444          let val_serialized = serialize_attr(&node.get_property(key).unwrap_or_default());
1445          arena::with(key_sym, |key_str| {
1446            write!(open_tag, " {key_str}=\"{val_serialized}\"")
1447          })
1448          .ok();
1449        }
1450        // HACK for xml:id for now, assuming last element.
1451        // SVG elements use plain "id" (not xml:id) — skip xml:id conversion for them.
1452        if anodes.contains_key("id") && !tag.starts_with("svg:") {
1453          let val_serialized = serialize_attr(&node.get_property("id").unwrap_or_default());
1454          write!(open_tag, " xml:id=\"{val_serialized}\"").ok();
1455        }
1456
1457        let noindent_children: bool = if heuristic {
1458          // libxml2's heuristic: inline (noindent) if ANY direct child is a text node.
1459          // Crucially, this does NOT propagate the parent's noindent — each element
1460          // independently checks its own children for text nodes.
1461          children
1462            .iter()
1463            .any(|e| e.get_type() == Some(NodeType::TextNode))
1464        } else {
1465          // This is the "Correct" way to determine whether to add indentation
1466          let node_qname = get_node_qname(node);
1467          model::can_contain_sym(node_qname, pin!("#PCDATA"))
1468        };
1469
1470        if !noindent {
1471          serialized.push_str(&indent)
1472        }
1473        serialized.push_str(&open_tag);
1474        // Perl serializes elements with children (including empty text nodes) as
1475        // <tag>...</tag>, and truly childless elements as <tag/>. Match this behavior.
1476        if !children.is_empty() {
1477          // with contents.
1478          serialized.push('>');
1479          if !noindent_children {
1480            serialized.push('\n');
1481          }
1482          for child in children {
1483            serialized.push_str(&self.serialize_aux(
1484              &child,
1485              depth + 1,
1486              noindent_children,
1487              heuristic,
1488            ));
1489          }
1490          if !noindent_children {
1491            serialized.push_str(&indent)
1492          }
1493          write!(serialized, "</{tag}>").ok();
1494        } else {
1495          // empty element.
1496          serialized.push_str("/>");
1497        }
1498        if !noindent {
1499          serialized.push('\n');
1500        }
1501      },
1502      Some(NodeType::TextNode) => {
1503        return serialize_string(&node.get_content());
1504      },
1505      Some(NodeType::PiNode) => {
1506        // should code this by hand, as well...
1507        if !noindent {
1508          serialized.push_str(&indent);
1509        }
1510        serialized.push_str(&self.document.node_to_string(node));
1511        if !noindent {
1512          serialized.push('\n');
1513        }
1514      },
1515      Some(NodeType::CommentNode) => {
1516        write!(
1517          serialized,
1518          "<!-- {}-->",
1519          serialize_string(&node.get_content())
1520        )
1521        .ok();
1522      },
1523      _ => {},
1524    }
1525    serialized
1526  }

1528  pub fn set_node(&mut self, node: &Node) {
1529    // Perl Document.pm:setNode L74-87: if the candidate is a
1530    // DOCUMENT_FRAG_NODE, validate that it has exactly one child and
1531    // descend to that child. The original Rust port had this check
1532    // commented-out with a wrong-node-type marker (`DocumentNode`
1533    // instead of `DocumentFragNode`); revived with the correct enum.
1534    let mut chosen = node.clone();
1535    if chosen.get_type() == Some(NodeType::DocumentFragNode) {
1536      let children = chosen.get_child_nodes();
1537      // Wrap the Error!/note_status side-effects in an IIFE to swallow the
1538      // `Result<()>` the macro returns (it can early-return Err if the
1539      // MAX_ERRORS cap is hit). `set_node` returns `()` and is called from
1540      // 32 call-sites; threading `Result` through all of them is out of
1541      // scope for an audit-fix. The hot path stays the same; the rare
1542      // hit-the-cap case loses one escape attempt but the next Error!
1543      // anywhere else will trigger the cap regardless.
1544      let _ = (|| -> Result<()> {
1545        if children.len() > 1 {
1546          Error!(
1547            "unexpected",
1548            "multiple-nodes",
1549            "Cannot set insertion point to a DOCUMENT_FRAG_NODE"
1550          );
1551        } else if children.is_empty() {
1552          Error!(
1553            "unexpected",
1554            "empty-nodes",
1555            "Cannot set insertion point to an empty DOCUMENT_FRAG_NODE"
1556          );
1557        }
1558        Ok(())
1559      })();
1560      if let Some(first) = children.into_iter().next() {
1561        chosen = first;
1562      }
1563    }
1564    self.node = chosen;
1565  }

1568  /// Scan for RDFa attributes in the document and set the `prefix` attribute
1569  /// on the root element based on which RDFa prefixes are actually used.
1570  fn set_rdfa_prefixes(&mut self) {
1571    // Collect the RDFa prefix mapping from state
1572    let prefix_map: HashMap<String, String> = state::with_mapping_keys("RDFa_prefixes", |keys| {
1573      let mut map = HashMap::default();
1574      for key_sym in keys {
1575        let key_str = arena::to_string(key_sym);
1576        if let Some(stored) = state::lookup_mapping("RDFa_prefixes", &key_str) {
1577          map.insert(key_str, stored.to_string());
1578        }
1579      }
1580      map
1581    });
1582    if prefix_map.is_empty() {
1583      return;
1584    }
1585
1586    let non_rdf_prefixes: HashSet<&str> = ["http", "https", "ftp"].iter().copied().collect();
1587    let rdf_term_attrs = [
1588      "about", "resource", "property", "typeof", "rel", "rev", "datatype",
1589    ];
1590
1591    // Build XPath to find elements with any RDFa term attribute
1592    let xpath = format!(
1593      "descendant::*[{}]",
1594      rdf_term_attrs
1595        .iter()
1596        .map(|a| format!("@{a}"))
1597        .collect::<Vec<_>>()
1598        .join(" or ")
1599    );
1600
1601    let mut used_prefixes: BTreeSet<String> = BTreeSet::new();
1602
1603    let nodes = self.findnodes(&xpath, None);
1604    for node in &nodes {
1605      for attr_name in &rdf_term_attrs {
1606        if let Some(value) = node.get_attribute(attr_name) {
1607          for term in value.split_whitespace() {
1608            if let Some(colon_pos) = term.find(':') {
1609              let prefix = &term[..colon_pos];
1610              if !non_rdf_prefixes.contains(prefix) && prefix_map.contains_key(prefix) {
1611                used_prefixes.insert(prefix.to_string());
1612              }
1613            }
1614          }
1615        }
1616      }
1617    }
1618
1619    if !used_prefixes.is_empty() {
1620      if let Some(mut root) = self.document.get_root_element() {
1621        let prefix_str = used_prefixes
1622          .iter()
1623          .map(|p| format!("{}: {}", p, prefix_map[p]))
1624          .collect::<Vec<_>>()
1625          .join(" ");
1626        let _ = root.set_attribute("prefix", &prefix_str);
1627      }
1628    }
1629  }

1631  pub fn insert_math_token(
1632    &mut self,
1633    text: &str,
1634    mut attributes: HashMap<String, String>,
1635    font_opt: Option<&Font>,
1636  ) -> Result<Node> {
1637    // Perf: avoid allocating the "role" String key unless the entry is missing.
1638    // HashMap::entry() takes an owned K, which forces allocation even on the
1639    // common path where "role" is already present.
1640    if !attributes.contains_key("role") {
1641      attributes.insert(String::from("role"), String::from("UNKNOWN"));
1642    }
1643    // Remove internal-only properties that should not become XML attributes.
1644    // In Perl, these are filtered by canHaveAttribute (model validation),
1645    // but we filter them explicitly here.
1646    attributes.remove("mode");
1647    attributes.remove("isMath");
1648    attributes.remove("cached_width");
1649    attributes.remove("cached_height");
1650    attributes.remove("cached_depth");
1651    // attributes.remove("stretchy");
1652
1653    let is_space = attributes.contains_key("isSpace");
1654    let qname = if is_space {
1655      MATH_HINT_NAME
1656    } else {
1657      MATH_TOKEN_NAME
1658    };
1659    let cur_qname = get_node_qname(&self.node);
1660    let text = if is_space && !text.is_empty() && text.chars().all(|c| c.is_whitespace()) {
1661      "" // Make empty hint, of only spaces
1662    } else {
1663      text
1664    };
1665    if qname == MATH_TOKEN_NAME && cur_qname == pin!("ltx:XMTok") {
1666      // Already INSIDE a token!
1667      if !text.is_empty() {
1668        self.open_math_text_internal(text)?;
1669      }
1670    } else {
1671      let mut node = self.open_element(qname, Some(attributes), None)?;
1672      // let tbox  = $attributes{_box} || $LaTeXML::BOX;
1673      let font = match font_opt {
1674        Some(f) => f.clone(),
1675        None => match self.box_to_absorb {
1676          Some(ref tbox) => match tbox.get_font()? {
1677            Some(f) => f.into_owned(),
1678            None => Font::math_default(), // should never happen?
1679          },
1680          None => Font::math_default(), // should never happen?
1681        },
1682      };
1683      self.set_node_font(&mut node, &font)?;
1684      if let Some(ref digested) = self.box_to_absorb {
1685        // TODO: The Rc<Digested> node boxes still have some way to go until they are fully
1686        // ergonomic...
1687        self.set_node_box(&node, digested.clone());
1688      }
1689      if !text.is_empty() {
1690        self.open_math_text_internal(text)?;
1691      }
1692      self.close_node_internal(&node)?; // Should be safe.
1693    }
1694    Ok(self.node.clone())

1717  /// Insert a new comment, or append to previous comment.
1718  /// Does NOT move the current insertion point to the Comment,
1719  /// but may move up past a text node.
1720  /// Perl: Document.pm lines 678-698
1721  pub fn insert_comment(&mut self, text: &str) -> Result<Node> {
1722    let trimmed = text.trim_end();
1723    let clean = DASHES_RE.replace_all(trimmed, "__");
1724    // Perl does NOT close the text node here — it uses getElement() to find
1725    // the nearest element, then inserts the comment relative to element children.
1726    // This preserves self.node as the current text node, so subsequent text
1727    // appends correctly and ligatures fire on the full text run.
1728
1729    let comment_text = s!(" {} ", clean);
1730
1731    if self.node.get_type() == Some(NodeType::DocumentNode) {
1732      Self::add_comment(
1733        &self.document,
1734        &self.node,
1735        &comment_text,
1736        Placement_::AppendChild,
1737      );
1738    } else if let Some(node) = self.get_element() {
1739      // Get the nearest element node (Perl: getElement)
1740      let prev = node.get_last_child();
1741      let prevtype = prev.as_ref().and_then(|n| n.get_type());
1742
1743      if prevtype == Some(NodeType::CommentNode) {
1744        // Merge with previous comment
1745        if let Some(mut prev_comment) = prev {
1746          let existing = prev_comment.get_content();
1747          let merged = s!("{}\n     {} ", existing, clean);
1748          prev_comment.set_content(&merged).ok();
1749        }
1750      } else if prevtype == Some(NodeType::TextNode) {
1751        let prev_node = prev.unwrap();
1752        // If the node before the text is already a comment, just append new comment
1753        // Otherwise, insert before the text to avoid splitting text runs
1754        let before_text = prev_node.get_prev_sibling();
1755        let before_is_comment =
1756          before_text.as_ref().and_then(|n| n.get_type()) == Some(NodeType::CommentNode);
1757
1758        if before_is_comment {
1759          Self::add_comment(
1760            &self.document,
1761            &node,
1762            &comment_text,
1763            Placement_::AppendChild,
1764          );
1765        } else {
1766          Self::add_comment(
1767            &self.document,
1768            &prev_node,
1769            &comment_text,
1770            Placement_::PrevSibling,
1771          );
1772        }
1773      } else {
1774        Self::add_comment(
1775          &self.document,
1776          &node,
1777          &comment_text,
1778          Placement_::AppendChild,
1779        );
1780      }
1781    }
1782    Ok(self.node.clone())
1783  }

1801  pub fn open_text(&mut self, text: &str, font: &Font) -> Result<Option<Node>> {
1802    let node_type = self.node.get_type();
1803    {
1804      // Ignore initial whitespace
1805      if (text.is_empty() || ONLY_SPACE_RE.is_match(text))
1806        && (node_type == Some(NodeType::DocumentNode)
1807          || (node_type == Some(NodeType::ElementNode) && !can_contain(&self.node, "#PCDATA")))
1808      {
1809        return Ok(None);
1810      }
1811    }
1812    if matches!(&font.family.as_deref(), Some("nullfont")) {
1813      return Ok(None);
1814    };
1815    Debug!(
1816      "document",
1817      "open_text",
1818      s!(
1819        "Insert text {:?} at {:?}",
1820        text,
1821        self.document.node_to_string(&self.node)
1822      )
1823    );
1824
1825    // Get the desired font attributes, particularly the desired element
1826    // (usually ltx:text, but let Font override, eg for \emph)
1827    let declared_font = self.get_node_font(&self.node);
1828    let pending_declaration = font.relative_to(declared_font);
1829    let elementname = match pending_declaration.get("element") {
1830      Some((k, _v)) => k,
1831      None => FONT_ELEMENT_NAME,
1832    };
1833    let element_sym = arena::pin(elementname);
1834    // If not at document begin. And not appending text in same font.
1835    if node_type != Some(NodeType::DocumentNode)
1836      && !(node_type == Some(NodeType::TextNode)
1837        && (font.distance(self.get_node_font(&self.node.get_parent().unwrap())) == 0))
1838    {
1839      // then we'll need to do some open/close to get fonts matched.
1840      let node = self.close_text_internal()?; // Close text node, if any.
1841      let mut bestdiff = 99;
1842      let rc_node = Rc::new(node);
1843      let mut closeto: Rc<Node> = Rc::clone(&rc_node);
1844      let mut n: Rc<Node> = Rc::clone(&rc_node);
1845      while n.get_type() != Some(NodeType::DocumentNode) {
1846        let node_font = self.get_node_font(&n);
1847        let d = font.distance(node_font);
1848        if d < bestdiff {
1849          bestdiff = d;
1850          closeto = n.clone();
1851          if d == 0 {
1852            break;
1853          }
1854        }
1855        // Stop if not a font element, or if marked _noautoclose, or if
1856        // this is an explicit (non-fontswitch) text wrapper. A constructor-
1857        // opened `<ltx:text class='...'>` (e.g. `\uline{...}`) MUST NOT be
1858        // closed-out-of by a font-distance heuristic just because the parent
1859        // happens to score better — that produces an empty wrapper and
1860        // siblings the inner content (driver: 2402.16319 `\uline{\textbf{2}}`
1861        // inside `\sc` tabular). Only auto-opened fontswitch wrappers are
1862        // safe to walk past.
1863        if get_node_qname(&n) != element_sym
1864          || n.has_attribute("_noautoclose")
1865          || !n.has_attribute("_fontswitch")
1866        {
1867          break;
1868        }
1869        match n.get_parent() {
1870          Some(p) => n = Rc::new(p),
1871          None => break,
1872        }
1873      }
1874
1875      // Move to best starting point for this text.
1876      if *closeto != *rc_node {
1877        self.close_to_node(&closeto, false)?;
1878      }
1879      if bestdiff > 0 {
1880        // Open if needed.
1881        self.open_element(
1882          elementname,
1883          Some(string_map!("_fontswitch" => "true", "_autoopened" => "true")),
1884          Some(font),
1885        )?;
1886      }
1887    }
1888
1889    // Finally, insert the darned text.
1890    let outnode = self.open_text_internal(text)?;
1891    self.record_constructed_node(&outnode);
1892    Ok(Some(outnode))
1893  }

1951        },

1958  /// Avoid redundant nesting of font switching elements:
1959  /// If we're closing a node that can take font switches and it contains
1960  /// a single FONT_ELEMENT_NAME node; pull it up.
1961  fn auto_collapse_children(&mut self, node: &mut Node) -> Result<()> {
1962    let qname = get_node_qname(node);
1963    if qname != pin!("ltx:_Capture_") {
1964      let mut c = node.get_child_nodes();
1965      // with single child, AND, $node can have all the attributes that the child has (but at least
1966      // "font") BUT, it isn"t being forced somehow
1967      if c.len() == 1
1968        && (get_node_qname(&c[0]) == pin!("ltx:text"))
1969        && model::can_have_attribute(qname, pin!("font"))
1970        && c[0]
1971          .get_attributes()
1972          .keys()
1973          .filter(|x| !x.starts_with('_'))
1974          .all(|v| {
1975            model::can_have_attribute(qname, arena::pin(v))
1976              && !(NON_MERGEABLE_ATTRIBUTES.contains(v.as_str()))
1977          })
1978        && !c[0].has_attribute("_force_font")
1979      {
1980        let c_first = c.pop().unwrap();
1981        let c_first_font = self.get_node_font(&c_first).clone();
1982        self.set_node_font(node, &c_first_font)?;
1983        for mut gc in c_first.get_child_nodes().into_iter() {
1984          gc.unlink();
1985          node.add_child(&mut gc)?;
1986          self.record_node_ids(node)?;
1987        }
1988        // Merge the attributes from the child onto $node
1989        self.merge_attributes(&c_first, node, None)?;
1990        self.remove_node(c_first);
1991      }
1992    }
1993    Ok(())
1994  }

1996  pub fn merge_attributes(
1997    &mut self,
1998    from: &Node,
1999    to: &mut Node,
2000    force: Option<&HashSet<&'static str>>,
2001  ) -> Result<()> {
2002    for (key, val) in from.get_attributes().iter() {
2003      // Skip internal attributes
2004      if key.starts_with('_') {
2005        continue;
2006      }
2007      // Normalize key: get_attributes() returns "id" for xml:id attributes.
2008      // Check both "xml:id" and bare "id" with XML namespace for the special case.
2009      let is_xml_id = key.as_str() == "xml:id"
2010        || (key.as_str() == "id"
2011          && from
2012            .get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace")
2013            .is_some());
2014      let effective_key = if is_xml_id { "xml:id" } else { key.as_str() };
2015      let is_forced = force.is_some_and(|f| f.contains(effective_key));
2016      // Special case attributes
2017      if is_xml_id {
2018        // Use the replacement id. record_id_with_node returns a
2019        // deduplicated id when a DIFFERENT node already claims the
2020        // same one; must use the return value, not the original `val`.
2021        let to_has_id = to.has_attribute("xml:id")
2022          || to
2023            .get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace")
2024            .is_some();
2025        if !to_has_id || is_forced {
2026          self.unrecord_id(val);
2027          let deduped = self.record_id_with_node(val, to);
2028          to.set_attribute("xml:id", &deduped)?;
2029        }
2030      } else if MERGE_ATTRIBUTE_SPACEJOIN.contains(key.as_str()) {
2031        self.add_ss_values(to, key, val)?;
2032      } else if MERGE_ATTRIBUTE_SEMICOLONJOIN.contains(key.as_str()) {
2033        if let Some(existing) = to.get_attribute(key) {
2034          let merged = format!("{existing}; {val}");
2035          to.set_attribute(key, &merged)?;
2036        } else {
2037          to.set_attribute(key, val)?;
2038        }
2039      } else if MERGE_ATTRIBUTE_SUMLENGTH.contains(key.as_str()) {
2040        if let Some(val2) = to.get_attribute(key) {
2041          // Parse and sum pt values
2042          let v1 = val.trim_end_matches("pt").parse::<f64>().unwrap_or(0.0);
2043          let v2 = val2.trim_end_matches("pt").parse::<f64>().unwrap_or(0.0);
2044          to.set_attribute(key, &format!("{}pt", v1 + v2))?;
2045        } else {
2046          to.set_attribute(key, val)?;
2047        }
2048      } else if !to.has_attribute(key) || is_forced {
2049        // Else if attribute not present on $to, or if we specifically override it, just copy
2050        to.set_attribute(key, val)?;
2051      }
2052    }
2053    Ok(())
2054  }

2360  /// Return a string indicating the path to the current insertion point in the document.
2361  /// if $levels is defined, show only that many levels
2362  pub fn get_insertion_context(&self, levels_opt: Option<usize>) -> Result<String> {
2363    let mut levels = match levels_opt {
2364      None => {
2365        // Default depth is based on verbosity
2366        if state::current_verbosity() <= 1 {
2367          Some(5)
2368        } else {
2369          None
2370        }
2371      },
2372      Some(t) => Some(t),
2373    };
2374    let mut node = self.node.clone();
2375    let node_type = node.get_type();
2376    if node_type != Some(NodeType::TextNode)
2377      && node_type != Some(NodeType::ElementNode)
2378      && node_type != Some(NodeType::DocumentNode)
2379    {
2380      let message = s!(
2381        "Insertion point is not an element, document or text: {:?}",
2382        self.document.node_to_string(&node)
2383      );
2384      Error!("internal", "context", message);
2385      return Ok(String::new());
2386    }
2387    // Build a context path like "<ltx:document><ltx:section><ltx:p>" by walking
2388    // ancestors and prepending each element's qname. Mirrors Perl's
2389    // `Stringify($node)` chain with a depth cap from `levels_opt`.
2390    //
2391    // Cap each qname at 80 chars. Pathological inputs (e.g. xy-pic
2392    // emitting unparsed `\fontdimen 17 \cmr10 at NNsp` strings that
2393    // become element names through a sequence of recovery errors)
2394    // can produce multi-MB qnames; walking 5+ ancestors and
2395    // `format!`-ing each yields a 3.25 GB allocation request that
2396    // OOM-kills the worker. Truncating preserves the diagnostic
2397    // signal ("first 80 chars + …") without the unbounded growth.
2398    // Witness papers: math0203082, math0402448 (R35.B, sandbox).
2399    const QNAME_CAP: usize = 80;
2400    let truncate_qname = |qname: &str| -> String {
2401      if qname.len() <= QNAME_CAP {
2402        qname.to_string()
2403      } else {
2404        let mut s: String = qname.chars().take(QNAME_CAP).collect();
2405        s.push('…');
2406        s
2407      }
2408    };
2409    let qn_for = |n: &Node| -> String {
2410      match n.get_type() {
2411        Some(NodeType::ElementNode) => {
2412          with_node_qname(n, |qname| format!("<{}>", truncate_qname(qname)))
2413        },
2414        Some(NodeType::TextNode) => "#text".to_string(),
2415        Some(NodeType::DocumentNode) => "#document".to_string(),
2416        _ => "?".to_string(),
2417      }
2418    };
2419    let mut path = qn_for(&node);
2420    while let Some(parent_node) = node.get_parent() {
2421      node = parent_node;
2422      if let Some(levels_val) = levels {
2423        levels = Some(levels_val - 1);
2424        if levels_val <= 1 {
2425          path = format!("...{path}");
2426          break;
2427        }
2428      }
2429      path = format!("{}{}", qn_for(&node), path);
2430    }
2431    Ok(path)
2432  }

2528        // Rust from 33 → 19, exact parity with Perl=19.

3304      // Unlink presentation from dual before replacing, since presentation is a child of dual
3305      pres.unlink();
3306      self.replace_node(dual, vec![pres])?;
3307      return Ok(());
3308    }
3309
3310    // Case 2: Compact mirror XMApp nodes
3311    if c_name != "ltx:XMApp" || p_name != "ltx:XMApp" {
3312      return Ok(());
3313    }
3314    let content_args = xml::element_nodes(&content);
3315    let pres_args = xml::element_nodes(&presentation);
3316    if content_args.len() != pres_args.len() {
3317      return Ok(());
3318    }
3319    let n_args = content_args.len();
3320
3321    // Walk the corresponding children, double-check they are referenced in the same order
3322    enum NewArg {
3323      Single(Node),
3324      Pair(Node, Node), // (content_arg, pres_arg) — to be merged
3325    }
3326    let mut new_args: Vec<NewArg> = Vec::with_capacity(n_args);
3327    for (c_arg, p_arg) in content_args.into_iter().zip(pres_args) {
3328      if let Some(c_idref) = c_arg.get_attribute("idref") {
3329        if c_idref == p_arg.get_attribute_ns("id", XML_NS).unwrap_or_default() {
3330          new_args.push(NewArg::Single(p_arg));
3331          continue;
3332        }
3333      }
3334      if let Some(p_idref) = p_arg.get_attribute("idref") {
3335        if p_idref == c_arg.get_attribute_ns("id", XML_NS).unwrap_or_default() {
3336          new_args.push(NewArg::Single(c_arg));
3337          continue;
3338        }
3339      }
3340      // We can handle content-side XMToks to any XM* presentation subtree
3341      let c_arg_name = with_node_qname(&c_arg, |n| n.to_string());
3342      if c_arg_name != "ltx:XMTok" {
3343        return Ok(()); // Can't compact this structure
3344      }
3345      new_args.push(NewArg::Pair(c_arg, p_arg));
3346    }
3347
3348    // If we made it here, this dual has two mirrored applications — compact it.
3349    let mut parent = match dual.get_parent() {
3350      Some(p) => p,
3351      None => return Ok(()),
3352    };
3353    let mut compact_apply = self.open_element_at(&mut parent, "ltx:XMApp", None, None)?;
3354    for n_arg in new_args {
3355      let mut node = match n_arg {
3356        NewArg::Single(n) => n,
3357        NewArg::Pair(c_arg, mut p_arg) => {
3358          self.merge_attributes(&c_arg, &mut p_arg, Some(&CONTENT_TRANSFER))?;
3359          p_arg
3360        },
3361      };
3362      node.unlink();
3363      compact_apply.add_child(&mut node)?;
3364    }
3365    // Migrate dual attributes to the new XMApp
3366    self.merge_attributes(&dual, &mut compact_apply, Some(&DUAL_TRANSFER))?;

3368    // Perl uses replaceChild which is a direct swap; replace_tree calls append_tree
3369    // which re-creates elements and fires afterOpen/afterClose hooks a second time.
3370    compact_apply.unlink();
3371    let mut dual_mut = dual;
3372    dual_mut.add_prev_sibling(&mut compact_apply).ok();
3373    self.remove_node(dual_mut);
3374    Ok(())
3375  }

3721  fn open_element_internal(
3722    &mut self,
3723    point: &mut Node,
3724    ns_opt: Option<String>,
3725    tag: &str,
3726  ) -> Result<Node> {
3727    // TODO:
3728    //
3729    // I am seriously irritated by the XML namespace and the confusion of the "default#"
3730    // tricks and libxml2's custom decisions about namespace interactions
3731    //
3732    // I have "hacked together" a working flow for now, but I expect to
3733    // encounter bugs related to the shortcuts taken here. I would welcome a
3734    // redesign       that simplifies the namespace logic dramatically.
3735    let new_ns = match ns_opt {
3736      Some(ns_uri) => {
3737        match point.lookup_namespace_prefix(&ns_uri) {
3738          // namespace not already declared?
3739          None => {
3740            if let Some(prefix) = model::get_document_namespace_prefix(&ns_uri, false, false) {
3741              if prefix != pin!("") {
3742                let mut root = self.document.get_root_element().unwrap();
3743                match arena::with(prefix, |prefix_str| {
3744                  Namespace::new(prefix_str, &ns_uri, &mut root)
3745                }) {
3746                  Ok(ns) => Some(ns),
3747                  Err(_) => {
3748                    // The namespace already exists on root (declared by an
3749                    // earlier element of the same namespace — e.g. a prior
3750                    // tikz/SVG picture) but `lookup_namespace_prefix` did not
3751                    // find it from this deeply-nested insertion point. Recover
3752                    // by reusing the root declaration (or creating it on the
3753                    // insertion point), exactly as the already-declared branch
3754                    // below — do NOT drop the namespace. Witness 1802.00756:
3755                    // a `tikzpicture` inside a nested `gather*`/`minipage`/
3756                    // `figure*` emitted 14× "failed to create namespace: svg"
3757                    // and the `<svg:svg>`/`<svg:g>` lost their namespace.
3758                    arena::with(prefix, |prefix_str| {
3759                      let found = root
3760                        .get_namespace_declarations()
3761                        .into_iter()
3762                        .find(|ns| ns.get_prefix() == prefix_str);
3763                      if found.is_none() {
3764                        Namespace::new(prefix_str, &ns_uri, point).ok()
3765                      } else {
3766                        found
3767                      }
3768                    })
3769                  },
3770                }
3771              } else {
3772                // default namespace?
3773                None
3774              }
3775            } else {
3776              // default namespace?
3777              None
3778            }
3779          },
3780          Some(prefix) => {
3781            if !prefix.is_empty() {
3782              let mut root = self.document.get_root_element().unwrap();
3783              match Namespace::new(&prefix, &ns_uri, &mut root) {
3784                Ok(ns) => Some(ns),
3785                Err(_) => {
3786                  // Namespace already exists on root — find and reuse it.
3787                  // We search declarations then fall back to creating on the
3788                  // insertion point (which inherits from root).
3789                  let found = root
3790                    .get_namespace_declarations()
3791                    .into_iter()
3792                    .find(|ns| ns.get_prefix() == prefix);
3793                  if found.is_none() {
3794                    // Try creating on the insertion point instead
3795                    Namespace::new(&prefix, &ns_uri, point).ok()
3796                  } else {
3797                    found
3798                  }
3799                },
3800              }
3801            } else {
3802              // default namespace?
3803              None
3804            }
3805          },
3806        }
3807      },
3808      None => None,
3809    };
3810
3811    let no_ns = new_ns.is_none();
3812    let mut newnode = match Node::new(tag, new_ns.clone(), &self.document) {
3813      Ok(n) => n,
3814      Err(_) => {
3815        // libxml2 rejected the tag (e.g. NUL byte, malformed name).
3816        // Bail out of element creation rather than aborting; caller
3817        // can recover. Driver: 2304.07380 panic at Node::new unwrap.
3818        let message = s!("failed to create element {:?}", tag);
3819        Error!("document", "open_element_internal", message);
3820        return Err(message.into());
3821      },
3822    };
3823    point.add_child(&mut newnode)?;
3824    if no_ns {

3826      // try to find the root's default namespace first. This prevents inheriting
3827      // the parent's namespace when inside a different namespace context (e.g.,
3828      // SVG elements getting svg: prefix on LaTeXML elements like Math/XMath).
3829      let root_ns = self
3830        .document
3831        .get_root_element()
3832        .and_then(|r| r.get_namespace());
3833      let parent_ns = point.get_namespace();
3834      if let Some(ref rns) = root_ns {
3835        // Use root's namespace for default-namespace elements
3836        let _ = newnode.set_namespace(rns);
3837      } else if let Some(ns) = parent_ns {
3838        // Fallback: inherit from parent (original behavior)
3839        newnode.set_namespace(&ns)?;
3840      }
3841    } else if let Some(ref ns) = new_ns {
3842      // For explicitly namespaced elements (e.g., svg:svg), ensure the namespace
3843      // is set after add_child — Node::new may not properly bind the namespace
3844      // when the Namespace was retrieved from get_namespace_declarations().
3845      let _ = newnode.set_namespace(ns);
3846    }

3850  }

3885  // Appending clones of nodes
3886
3887  // Inserting clones of nodes into the document.
3888  // Nodes that exist in some other part of the document (or some other document)
3889  // will need to be cloned so that they can be part of the new document;
3890  // otherwise, they would be removed from thier previous document.
3891  // Also, we want to have a clean namespace node structure
3892  // (otherwise, libxml2 has a tendency to introduce annoying "default" namespace prefix
3893  // declarations) And, finally, we need to modify any id's present in the old nodes,
3894  // since otherwise they may be duplicated.
3895
3896  // # Should have variants here for prepend, insert before, insert after.... ???
3897  pub fn append_clone(&mut self, node: &mut Node, new_children: Vec<Node>) -> Result<()> {
3898    // Expand any document fragments
3899    let new_children = new_children
3900      .into_iter()
3901      .flat_map(|child| {
3902        if child.get_type() == Some(NodeType::DocumentFragNode) {
3903          child.get_child_nodes()
3904        } else {
3905          vec![child]
3906        }
3907      })
3908      .collect::<Vec<Node>>();
3909    // Now find all xml:id's in the new_children and record replacement id's for them
3910    let mut id_map = HashMap::default();
3911    // Find all id's defined in the copy and change the id.
3912    // Note: XPath ".//@xml:id" can fail to find namespace-qualified attributes.
3913    // Use DOM walking as fallback to ensure all ids are found.
3914    for child in new_children.iter() {
3915      let mut xpath_ids: Vec<String> = self.findvalues(".//@xml:id", Some(child));
3916      if xpath_ids.is_empty() {
3917        // Fallback: walk DOM to find xml:id attributes
3918        Self::collect_xml_ids_from(child, &mut xpath_ids);
3919      }
3920      for id in xpath_ids {
3921        id_map.insert(id.clone(), self.modify_id(id));
3922      }
3923    }
3924    // Now do the cloning (actually copying) and insertion.
3925    self.append_clone_aux(node, new_children, &mut id_map)
3926  }

3928  /// Walk DOM to collect xml:id attribute values (fallback when XPath fails).
3929  fn collect_xml_ids_from(node: &Node, ids: &mut Vec<String>) {

3931      ids.push(id);

3937        Self::collect_xml_ids_from(&child, ids);
3938      }
3939    }

3942  fn append_clone_aux(
3943    &mut self,
3944    node: &mut Node,
3945    new_children: Vec<Node>,
3946    id_map: &mut HashMap<String, String>,
3947  ) -> Result<()> {
3948    for child in new_children.into_iter() {
3949      match child.get_type() {
3950        Some(NodeType::ElementNode) => {
3951          let mut new = self.open_element_internal(
3952            node,
3953            child.get_namespace().map(|ns| ns.get_href()),
3954            &child.get_name(),
3955          )?;
3956          for (key, val) in child.get_attributes() {
3957            match key.as_str() {
3958              "xml:id" | "id" => {
3959                // Use the replacement id. The pre-walk
3960                // (findvalues/collect_xml_ids_from) normally populates
3961                // id_map with every `xml:id` it can see, but the
3962                // namespace/prefix of the attribute key returned by
3963                // libxml2's `get_attributes()` can differ from what the
3964                // XPath / DOM walk picked up (e.g. when the incoming
3965                // node's tree came through a cloneNode that stripped
3966                // the xml namespace). Fall back to minting a fresh
3967                // replacement id on-the-fly rather than panicking —
3968                // 1410.8508 hit this when the pre-walk found zero ids
3969                // but attributes on a child node did carry xml:id.
3970                let fresh;
3971                let mapped_id = match id_map.get(&val) {
3972                  Some(id) => id,
3973                  None => {
3974                    fresh = self.modify_id(val.clone());
3975                    id_map.insert(val.clone(), fresh.clone());
3976                    id_map.get(&val).unwrap()
3977                  },
3978                };
3979                let newid = self.record_id_with_node(mapped_id, &new);
3980                // Write the literal "xml:id" key (not the bare "id" local-
3981                // name returned by libxml's get_attributes). Otherwise the
3982                // cloned node only gets a plain `id` attribute, and the
3983                // subsequent `after_open` chain's `has_attribute_ns("id",
3984                // XML_NS)` check returns false, causing `generate_id` to
3985                // mint a fresh `.<parent>.N` xml:id that doesn't match the
3986                // sibling XMRef idrefs (which were rewritten from id_map).
3987                // Witness: arXiv:2509.07628 — MathFork mainfork emitted
3988                // 154 XMRefs with `.mf` idrefs while the cloned target
3989                // nodes received parent-scoped `.m2.N` xml:ids, leaving
3990                // every XMRef dangling and triggering 4
3991                // `Error:expected:id` per equation during post-processing
3992                // visibility marking. Same shape applies anywhere a
3993                // cloned subtree carries xml:id attributes — MathFork,
3994                // tabular-cell clone, _Capture_ flush.
3995                new.set_attribute("xml:id", &newid)?;
3996                // Update id_map so subsequent idref lookups use the ACTUAL recorded id.
3997                // record_id_with_node may change the id (e.g., if there are conflicts),
3998                // so the mapped_id and newid may differ.
3999                if *mapped_id != newid {
4000                  id_map.insert(val.clone(), newid);
4001                }
4002              },
4003              "idref" => {
4004                // Refer to the replacement id if it was replaced
4005                let id = id_map.get(&val).unwrap_or(&val);
4006                new.set_attribute(&key, id)?;
4007              },
4008              other_key =>
4009              // TODO: Are namespaced attributes successfully handled here? Check.
4010              {
4011                new.set_attribute(other_key, &val)?
4012              },
4013            };
4014          }
4015          self.after_open(&mut new)?;
4016          self.append_clone_aux(&mut new, child.get_child_nodes(), id_map)?;
4017          self.after_close(&mut new)?;
4018        },
4019        Some(NodeType::TextNode) => node.append_text(&child.get_content())?,
4020        Some(NodeType::CommentNode) => {
4021          // Skip XML comments during cloning (Perl also skips them in most contexts)
4022        },
4023        other => {
4024          log::warn!("append_clone_aux: skipping unsupported {other:?} node type");
4025        },
4026      };
4027    }
4028    Ok(())
4029  }

4069  /// Replace `node` by `nodes` (presumably descendants of some kind?)
4070  // DG: Don't return the replaced `node`, as it is groudns for memory management trouble
4071  //     with the low-level libxml layer. I've encountered segfaults here.
4072  pub fn replace_node(&mut self, mut node: Node, with: Vec<Node>) -> Result<()> {
4073    if let Some(_parent) = node.get_parent() {
4074      // libxml2's xmlAddNextSibling merges consecutive text nodes: when both
4075      // the reference sibling and the new node are TextNode, it appends the
4076      // new node's content to the reference node and frees the new node. The
4077      // Rust wrapper doesn't surface the merged result, so naively advancing
4078      // `c0_opt` to `with_node` would capture a pointer to freed memory,
4079      // producing silent data loss for the third+ insertion and eventually
4080      // a libxml2 SIGSEGV when the dangling pointer is re-traversed (e.g.
4081      // during a later XPath evaluation). Detect the text-text case and
4082      // coalesce in-place instead.
4083      let mut c0_opt: Option<Node> = None;
4084      for mut with_node in with.into_iter() {
4085        with_node.unlink();
4086        let is_text = with_node.get_type() == Some(NodeType::TextNode);
4087        if let Some(mut c0) = c0_opt {
4088          let c0_is_text = c0.get_type() == Some(NodeType::TextNode);
4089          if is_text && c0_is_text {
4090            let existing = c0.get_content();
4091            let added = with_node.get_content();
4092            c0.set_content(&format!("{existing}{added}"))?;
4093            // with_node is still a standalone (unlinked) text node; drop it.
4094            c0_opt = Some(c0);
4095            continue;
4096          }
4097          c0.add_next_sibling(&mut with_node)?;
4098        } else {
4099          // first node, swap in
4100          node.add_next_sibling(&mut with_node)?;
4101        }
4102        c0_opt = Some(with_node);
4103      }
4104      self.remove_node(node);
4105    }
4106    Ok(())
4107  }

4127  ) -> Result<Node> {

4129      .get_parent()
4130      .expect("rename should never be called on an orphan or root node.");

4132    // Move to the position AFTER node

4136    for (key, value) in node.get_attributes() {

4166    // THEN call afterOpen... ?

4382      if let Some(labels) = node.get_attribute("labels") {
4383        // A labelled node MUST carry an xml:id so `\ref` can resolve to it.
4384        // Normally the `Tag('ltx:*', afterClose:late)` GenerateID hook
4385        // (latex_constructs.rs) stamps one, but it does not reach every node
4386        // — notably the <ltx:document> root, which receives a label when a
4387        // bare `\label{…}` appears with no enclosing id'd sectioning (e.g.
4388        // `\input{abs}` then `\label{sec:intro}` before any \section; witness
4389        // 1703.09326). Perl handles this by giving the root an xml:id — its
4390        // output is `<document … labels="LABEL:sec:intro" xml:id="id1">` —
4391        // NOT by erroring. Match Perl: generate an id here when one is
4392        // missing, exactly as Perl's GenerateID does (an id-less root yields
4393        // "id1" from `generate_id`'s empty-prefix→"id", no-ancestor path).
4394        let id = match node.get_attribute("id") {
4395          Some(id) => Some(id),
4396          None => {
4397            self.generate_id(&mut node, "")?;
4398            node.get_attribute_ns("id", XML_NS)
4399          },
4400        };
4401        if let Some(id) = id {
4402          for label in labels.split_whitespace() {
4403            self
4404              .rewrite_labels
4405              .insert(label.to_string(), id.clone());
4406          }
4407        }
4408        // If generate_id still couldn't assign one (a node the model forbids
4409        // an xml:id on), the label is simply unresolvable — drop it silently
4410        // (Perl does not error here either).
4411      }
4412    }
4413    Ok(())
4414  }

4421  /// This function computes an xml:id for a node, if it hasn't already got one.
4422  /// It is suitable for use in Tag afterOpen as
4423  ///  `Tag('ltx:para',afterOpen=>sub { GenerateID(@_,'p'); });`
4424  /// It generates an id of the form `<parentid>.<prefix><number>`
4425  /// The parent node (the one with `ID=<parentid>`) also maintains a counter
4426  /// stored in an attribute `_ID_counter_<prefix>` recording the last used
4427  /// `number` for `prefix` amongst its descendents.
4428  pub fn generate_id(&mut self, node: &mut Node, mut prefix: &str) -> Result<()> {
4429    // If node doesn't already have an id, and can
4430    // but isn't a _Capture_ node (which ultimately should disappear)
4431    let qname = get_node_qname(node);
4432    if !node.has_attribute_ns("id", XML_NS)
4433      && model::can_have_attribute(qname, pin!("xml:id"))
4434      && (qname != pin!("ltx:_Capture_"))
4435    {
4436      let mut ancestor = self
4437        .findnode("ancestor::*[@xml:id][1]", Some(node))
4438        .unwrap_or_else(|| self.get_document().get_root_element().unwrap());
4439      //// Old versions don't like ancestor.getAttribute('xml:id');
4440      let ancestor_id = ancestor.get_attribute_ns("id", XML_NS);
4441      // If we've got no ancestor_id, then we've got no ancestor (no document yet!),
4442      // or ancestor IS the root element (but without an id);
4443      // If we also have no prefix, we'll end up with an illegal id (just digits)!!!
4444      // We'll use "id" for an id prefix; this will work whether or not we have an ancestor.
4445      if prefix.is_empty() && ancestor_id.is_none() {
4446        prefix = "id";
4447      }
4448
4449      // Perl `Package.pm:939` (`'_ID_counter_' . ($prefix ? $prefix . '_' : '')`)
4450      // — empty prefix uses `_ID_counter_` with a single trailing underscore,
4451      // not `_ID_counter__`. Matters for interop with code that reads the
4452      // attribute by exact name (e.g. `Base_XMath.pool.ltxml:940` reads
4453      // `_ID_counter_` for the empty-prefix counter).
4454      let ctrkey = if prefix.is_empty() {
4455        s!("_ID_counter_")
4456      } else {
4457        s!("_ID_counter_") + prefix + "_"
4458      };
4459      let a_ctr = ancestor.get_attribute(&ctrkey).unwrap_or_else(|| s!("0"));
4460
4461      let ctr_int = 1 + a_ctr.parse::<u32>().unwrap_or(0);
4462      let ctr = ctr_int.to_string();
4463
4464      let id = match ancestor_id {
4465        Some(aid) => aid + ".",
4466        None => String::new(),
4467      } + prefix
4468        + &ctr;
4469
4470      ancestor.set_attribute(&ctrkey, &ctr)?;
4471      self.set_attribute(node, "xml:id", &id)?;
4472    }
4473    Ok(())
4474  }

4483  pub fn replace_tree(&mut self, new: Node, old: Node) -> Result<Option<Node>> {
4484    if let Some(mut parent) = old.get_parent() {
4485      let mut following = VecDeque::new(); // Collect the matching and following nodes
4486      while let Some(mut sib) = parent.get_last_child() {
4487        if sib == old {
4488          break;
4489        }
4490        // parent.remove_child(sib); // We're putting these back, in a moment!
4491        sib.unlink();
4492        following.push_front(sib);
4493      }
4494      // NOTE: remove_node calls old.unlink() which sets unlinked=true and detaches
4495      // the node from the DOM. The document's node cache holds another Rc reference,
4496      // so _Node::drop (and xmlFreeNode) doesn't run until the Document itself drops.
4497      // By that time, any children reused by `new` have been moved at the C level,
4498      // so xmlFreeNode on old only frees the shell.
4499      self.remove_node(old);
4500      self.append_tree(&mut parent, vec![new])?;
4501      let inserted = parent.get_last_child();
4502      for mut child in following {
4503        parent.add_child(&mut child)?; // No need for clone
4504      }
4505      Ok(inserted)
4506    } else {
4507      Ok(None)
4508    }
4509  }

4511  pub fn append_tree(&mut self, node: &mut Node, data: Vec<Node>) -> Result<()> {
4512    for child in data {
4513      match child.get_type() {
4514        Some(NodeType::ElementNode) => {
4515          let mut attributes: HashMap<String, String> =
4516            child.get_attributes().into_iter().collect();
4517          // Perl appendTree: REMOVE xml:id from source node before re-creation.
4518          // This prevents duplicate ID registration. The ID will be re-registered
4519          // by open_element_at when the new node is created with the same ID.
4520          if let Some(xmlid) = child.get_attribute_ns("id", XML_NS) {
4521            attributes
4522              .entry("xml:id".to_string())
4523              .or_insert_with(|| xmlid.clone());
4524            // Unrecord before re-creation (Perl: $child->removeAttribute('xml:id') + unRecordID)
4525            self.unrecord_id(&xmlid);
4526          }
4527
4528          let tag_sym = get_node_qname(&child);
4529          let tag = arena::to_string(tag_sym);
4530          let mut new = self.open_element_at(node, &tag, Some(attributes), None)?;
4531          self.append_tree(&mut new, child.get_child_nodes())?;
4532          self.close_element_at(&mut new)?;
4533        },
4534        Some(NodeType::DocumentFragNode) => {
4535          self.append_tree(node, child.get_child_nodes())?;
4536        },
4537        Some(NodeType::TextNode) => {
4538          node.append_text(&child.get_content())?;
4539        },
4540        other => {
4541          log::debug!("append_tree: unhandled libxml NodeType {other:?}");
4542        },
4543      }
4544    }
4545    Ok(())
4546  }

4568fn trim_node_left_whitespace(node: &Node) -> Result<()> {
4569  if let Some(mut first_child) = node.get_first_child() {
4570    match first_child.get_type() {
4571      Some(NodeType::TextNode) => {
4572        let content = first_child.get_content();
4573        // Perl: s/^ +// — only trim ASCII spaces, preserve unicode spaces (nbsp, em-space, etc.)
4574        let trimmed_content = content.trim_start_matches(' ');
4575        if !content.is_empty() && (trimmed_content != content) {
4576          first_child.set_content(trimmed_content)?;
4577        }
4578      },
4579      Some(NodeType::ElementNode) => trim_node_left_whitespace(&first_child)?,
4580      _ => {},
4581    };
4582  }
4583  Ok(())
4584}

4586fn trim_node_right_whitespace(node: &Node) -> Result<()> {
4587  // Skip trailing empty <text> font wrapper elements to find the real last content.
4588  // These are artifacts of font change tracking during alignment absorption.
4589  let mut candidate = node.get_last_child();
4590  while let Some(ref child) = candidate {
4591    if child.get_type() == Some(NodeType::ElementNode)
4592      && child.get_name() == "text"
4593      && child.get_child_nodes().is_empty()
4594      && child.has_attribute("_noautoclose")
4595    {
4596      candidate = child.get_prev_sibling();
4597    } else {
4598      break;
4599    }
4600  }
4601  if let Some(mut last_child) = candidate {
4602    match last_child.get_type() {
4603      Some(NodeType::TextNode) => {
4604        let content = last_child.get_content();
4605        // Perl: s/\s+$// — but we can't trim all Unicode whitespace because some
4606        // tests have significant thin spaces (U+2009) from DimensionToSpaces.
4607        // Trim: ASCII whitespace, nbsp (U+00A0), em-space (U+2003), en-space (U+2002).
4608        let trimmed_content = content.trim_end_matches(|c: char| {
4609          c.is_ascii_whitespace() || c == '\u{00A0}' || c == '\u{2003}' || c == '\u{2002}'
4610        });
4611        if !content.is_empty() && (trimmed_content != content) {
4612          if trimmed_content.is_empty() {
4613            // Remove the entirely-whitespace text node
4614            last_child.unlink();
4615          } else {
4616            last_child.set_content(trimmed_content)?;
4617          }
4618        }
4619      },
4620      Some(NodeType::ElementNode) => trim_node_right_whitespace(&last_child)?,
4621      _ => {},
4622    };
4623  }
4624  Ok(())
4625}
