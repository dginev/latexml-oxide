29/// An `Rc`-guarded abstraction for any object encountered at the "digested" phase of processing
30// Each variant is wrapped in an `Rc`, for cheap(er) cloning when passing around
31// these objects to various auxiliary state (e.g. bookkeeping current box),
32// but also for repeatedly passing them as owned into binding closures
33// while also storing them in their owner Box.
34//
35// This model is incredibly hard to achieve with lifetimes, so
36// we employ reference counting instead (close to their original Perl design).
37// A strict OO-hierarchy of object ownership (with no auxiliary state metadata)
38// would allow a Rust-like redesign. But it could be too hard to achieve in practice.
39#[derive(Clone)]
40pub struct Digested(Rc<DigestedData>);
41/// These are all kinds of data which we consider officially supported
42/// as outputs from the digestion phase of TeX, i.e. from invoking a token.
43#[allow(clippy::large_enum_variant)]
44// TODO: Investigate if the outer Rc<> wrap of Digested is enough to avoid performance penalties
45// from having       the concrete structs in DigestedData vary a lot in size.
46pub enum DigestedData {
47  /// A TeX Box
48  TBox(RefCell<Tbox>),
49  /// A TeX Whatsit (with interior mutability, for setters invoked while stored in state)
50  Whatsit(RefCell<Whatsit>),
51  /// A TeX Alignment (with interior mutability, for setters invoked while stored in state)
52  Alignment(Box<RefCell<Alignment>>),
53  /// A list of Digested data
54  List(RefCell<List>),
55  /// Raw Tokens that were postponed to the digestion phase uninvoked/undigested
56  Postponed(Tokens),
57  /// A LaTeX-like digested key-value map
58  KeyVals(KeyVals),
59  /// A TeX-like `RegisterValue` (e.g. a Dimension or Glue)
60  RegisterValue(RegisterValue),
61  /// A TeX comment
62  Comment(Comment),
63}

85impl PartialEq for Digested {
86  fn eq(&self, other: &Digested) -> bool {
87    use DigestedData::*;
88    match *self.0 {
89      TBox(ref tb) => {
90        if let TBox(ref tb2) = *other.0 {
91          tb == tb2
92        } else {
93          false
94        }
95      },
96      Whatsit(ref tb) => {
97        if let Whatsit(ref tb2) = *other.0 {
98          *tb.borrow() == *tb2.borrow()
99        } else {
100          false
101        }
102      },
103      Alignment(ref tb) => {
104        if let Alignment(ref tb2) = *other.0 {
105          *tb.borrow() == *tb2.borrow()
106        } else {
107          false
108        }
109      },
110      List(ref tb) => {
111        if let List(ref tb2) = *other.0 {
112          tb == tb2
113        } else {
114          false
115        }
116      },
117      Postponed(ref tb) => {
118        if let Postponed(ref tb2) = *other.0 {
119          tb == tb2
120        } else {
121          false
122        }
123      },
124      KeyVals(ref tb) => {
125        if let KeyVals(ref tb2) = *other.0 {
126          tb == tb2
127        } else {
128          false
129        }
130      },
131      RegisterValue(ref tb) => {
132        if let RegisterValue(ref tb2) = *other.0 {
133          tb == tb2
134        } else {
135          false
136        }

262      List(ref l) => l.borrow().get_locator(),
263      Comment(ref c) => c.get_locator(),

271  fn revert(&self) -> Result<Tokens> {
272    use DigestedData::*;
273    match *self.0 {
274      TBox(ref b) => b.borrow().revert(),
275      List(ref l) => l.borrow().revert(),
276      Whatsit(ref w) => w.borrow().revert(),
277      // Re-entrant guard: a broken alignment (e.g. a `\matrix`/`\pmatrix` left
278      // mid-mutation by a failed mode-group close) can re-enter its own RefCell
279      // during reversion → "already mutably borrowed" panic. Perl has no
280      // borrow-checker, so its `$alignment->revert` is plain re-entrant data
281      // access. `try_borrow` + an empty reversion on the re-entrant cycle,
282      // mirroring the base_xmath fix (75c452843d) and `compute_size`/`with_properties`.
283      Alignment(ref w) => match w.try_borrow() {
284        Ok(al) => al.revert(),
285        Err(_) => {
286          Error!(
287            "unexpected",
288            "self_referential_alignment",
289            "Reverting a re-entrant alignment to empty tokens (source text is lost)"
290          );
291          Ok(Tokens::default())
292        },
293      },
294      Postponed(ref t) => Ok(t.clone()),
295      KeyVals(ref kvs) => kvs.revert(),
296      Comment(ref c) => c.revert(),
297      RegisterValue(ref rv) => rv.revert(),
298    }
299  }

316      TBox(_) | Whatsit(_) | Alignment(_) | KeyVals(_) | Comment(_) | Postponed(_)
317      | RegisterValue(_) => {

324  fn be_absorbed(&self, document: &mut Document) -> Result<Vec<Node>> {
325    use DigestedData::*;
326    match &*self.0 {
327      TBox(b) => b.borrow().be_absorbed(document),
328      List(l) => l.borrow().be_absorbed(document),
329      Comment(c) => c.be_absorbed(document),
330      Whatsit(w) => w.borrow().be_absorbed(document),
331      // Re-entrant guard: a broken alignment mid-`be_absorbed_mut` (which holds
332      // an exclusive borrow) can be re-absorbed by recursive document
333      // construction → "already borrowed" panic. Absorb nothing on the
334      // re-entrant cycle (the alignment is mid-mutation; producing no nodes is
335      // the safe degradation, like `Postponed`). Mirrors `with_properties` /
336      // `compute_size` / the base_xmath fix (75c452843d).
337      Alignment(w) => match w.try_borrow_mut() {
338        Ok(mut al) => al.be_absorbed_mut(document),
339        Err(_) => {
340          Error!(
341            "unexpected",
342            "self_referential_alignment",
343            "Skipping absorption of a re-entrant alignment (its cells are lost)"
344          );
345          Ok(Vec::new())
346        },
347      },
348      KeyVals(kvs) => kvs.be_absorbed(document),
349      Postponed(_) => Ok(Vec::new()), // Postponed items absorbed silently
350      RegisterValue(_rv) => Ok(Vec::new()), // Register values not absorbable
351    }
352  }

354  fn with_properties<R, FnR>(&self, caller: FnR) -> R
355  where FnR: FnOnce(&HashMap<Stored>) -> R {
356    use DigestedData::*;
357    // Defensive `try_borrow`: when a Digested wrapper is mid-`be_absorbed_mut`
358    // (which holds an exclusive `borrow_mut`) and document construction
359    // recursively asks the SAME node for its properties, an infallible
360    // `.borrow()` panics with "RefCell already mutably borrowed". Fall
361    // back to NO_PROPERTIES instead — property access during the
362    // mid-absorption window is read-only and a missing-properties result
363    // is benign (matches Perl: properties default to empty in this state).
364    // Witness: 1205.0376 (article + plain-TeX `\AND`/`\at` redefs +
365    // align environment) — previously FATAL_101 panic at digested.rs:329,
366    // now succeeds.
367    match &*self.0 {
368      TBox(b) => match b.try_borrow() {
369        Ok(b) => caller(b.get_properties()),
370        Err(_) => caller(&NO_PROPERTIES),
371      },
372      List(l) => match l.try_borrow() {
373        Ok(l) => caller(l.get_properties()),
374        Err(_) => caller(&NO_PROPERTIES),
375      },
376      Comment(c) => caller(c.get_properties()),
377      Whatsit(w) => match w.try_borrow() {
378        Ok(w) => caller(w.get_properties()),
379        Err(_) => caller(&NO_PROPERTIES),
380      },
381      Alignment(w) => match w.try_borrow() {
382        Ok(w) => caller(w.get_properties()),
383        Err(_) => caller(&NO_PROPERTIES),
384      },
385      KeyVals(_) | Postponed(_) | RegisterValue(_) => caller(&NO_PROPERTIES),
386    }
387  }

469  /// Note the difference between calling `compute_size` on a Digested object, and calling it on a
470  /// concrete box type. When called on `Digested` it will opt for caching the computed sizes,
471  /// but when called on the concrete types it will always compute sizes fresh.
472  fn compute_size(&self, options: HashMap<Stored>) -> Result<(Dimension, Dimension, Dimension)> {
473    use DigestedData::*;
474    // A self-referential box (its size computation recurses into the SAME
475    // RefCell — e.g. a box that transitively contains itself) would re-enter
476    // `borrow_mut` and panic ("already borrowed"). The other traversals on
477    // `Digested` (fingerprint, serialization) already guard with `try_borrow`;
478    // do the same here and treat a re-entrant cycle as zero-size to break it.
479    // Witness: astro-ph0310145 (panicked at digested.rs Alignment arm).
480    let zero = (Dimension::new(0), Dimension::new(0), Dimension::new(0));
481    match *self.0 {
482      TBox(ref b) => match b.try_borrow_mut() {
483        Ok(mut x) => x.compute_size_and_cache(options),
484        Err(_) => Ok(zero),
485      },
486      List(ref l) => match l.try_borrow_mut() {
487        Ok(mut x) => x.compute_size_and_cache(options),
488        Err(_) => Ok(zero),
489      },
490      KeyVals(ref kvs) => kvs.compute_size(options),
491      Whatsit(ref w) => match w.try_borrow_mut() {
492        Ok(mut x) => x.compute_size_and_cache(options),
493        Err(_) => Ok(zero),
494      },
495      Alignment(ref w) => match w.try_borrow_mut() {
496        Ok(mut x) => x.compute_size_and_cache(options),
497        Err(_) => Ok(zero),
498      },
499      Postponed(_) | RegisterValue(_) | Comment(_) => Ok(zero),
500    }
501  }

504/// Hard cap on the number of sub-boxes visited by [`Digested::cycle_fingerprint`].
505/// Bounds the fingerprint cost to O(1) per box while still sampling enough
506/// content (depth-first) to tell apart same-shaped boxes.
507const FP_BUDGET: u32 = 48;
508
509/// Per-box traversal cap for [`Digested::estimate_bytes`]. Larger than
510/// [`FP_BUDGET`] (the estimate is computed only on a small *sample* of the box
511/// list, so a deeper walk is affordable and improves accuracy for nested
512/// boxes).
513const EB_BUDGET: u32 = 256;
514
515impl Digested {
516  /// immutably borrow the inner Digested data
517  pub fn data(&self) -> &DigestedData { &self.0 }
518
519  /// A content-aware but COST-BOUNDED fingerprint for the stomach cycle guard
520  /// ([`crate::cycle_guard`]).
521  ///
522  /// Design tension: it must (a) distinguish boxes by *content* so two
523  /// different boxes that merely share a shape (e.g. two `List`s of equal
524  /// length but different children) don't collide into a false cycle, yet
525  /// (b) be cheap on the digestion path. We reconcile both with a hard
526  /// **node budget**: at most [`FP_BUDGET`] sub-boxes are ever visited (a
527  /// depth-first sample of the content), so cost is O(1) per box regardless
528  /// of how large or deeply nested the structure is, while the sample is rich
529  /// enough that real content differences change the hash. (It is also only
530  /// ever invoked once `box_list` has already blown past the stomach guard's
531  /// activation size, so ordinary conversions never pay for it at all.)
532  /// NOT a stable cross-process hash — for in-run loop detection only.
533  pub fn cycle_fingerprint(&self) -> u64 {
534    use std::hash::Hasher;
535    let mut h = rustc_hash::FxHasher::default();
536    let mut budget: u32 = FP_BUDGET;
537    self.fingerprint_into(&mut h, &mut budget);
538    h.finish()
539  }
540
541  // NOTE: `fingerprint_into` and `estimate_bytes_into` are PAIRED budgeted
542  // traversals over `DigestedData` (one hashes, one sizes — different
543  // per-variant work, same walk shape). Both matches are deliberately
544  // EXHAUSTIVE (no `_` catch-all): adding a `DigestedData` variant breaks
545  // both at compile time, so the two cannot silently drift on coverage —
546  // only keep that property when editing either (PR #249 review P3-11).
547  fn fingerprint_into<H: std::hash::Hasher>(&self, h: &mut H, budget: &mut u32) {
548    use std::hash::Hash;
549    if *budget == 0 {
550      return;
551    }
552    *budget -= 1;
553    match self.data() {
554      DigestedData::TBox(b) => {
555        0u8.hash(h);
556        if let Ok(tb) = b.try_borrow() {
557          tb.text.hash(h);
558        }
559      },
560      DigestedData::Whatsit(w) => {
561        1u8.hash(h);
562        if let Ok(wb) = w.try_borrow() {
563          // The creating definition's identity distinguishes whatsits of
564          // different kinds (Rc data-pointer, stable within a run); the args
565          // distinguish their content.
566          (Rc::as_ptr(&wb.definition) as *const () as usize).hash(h);
567          wb.args.len().hash(h);
568          for arg in &wb.args {
569            if *budget == 0 {
570              break;
571            }
572            match arg {
573              Some(d) => d.fingerprint_into(h, budget),
574              None => {
575                *budget -= 1;
576                0xFEu8.hash(h);
577              },
578            }
579          }
580        }
581      },
582      DigestedData::Alignment(_) => 2u8.hash(h),
583      DigestedData::List(l) => {
584        3u8.hash(h);
585        if let Ok(lb) = l.try_borrow() {
586          lb.boxes.len().hash(h);
587          for child in &lb.boxes {
588            if *budget == 0 {
589              break;
590            }
591            child.fingerprint_into(h, budget);
592          }
593        }
594      },
595      DigestedData::Postponed(t) => {
596        4u8.hash(h);
597        t.len().hash(h);
598      },
599      DigestedData::KeyVals(_) => 5u8.hash(h),
600      DigestedData::RegisterValue(r) => {
601        6u8.hash(h);
602        std::mem::discriminant(r).hash(h);
603      },
604      DigestedData::Comment(c) => {
605        7u8.hash(h);
606        c.0.hash(h);
607      },
608    }
609  }
610  /// A COST-BOUNDED estimate of the heap bytes this digested box (and its
611  /// nested content) occupies. Used by the stomach's portable
612  /// memory-budget guard ([`crate::stomach`]) to detect a runaway *by the
613  /// resource that actually matters — bytes — rather than box COUNT*, since
614  /// per-box weight varies several-fold (a bare text box vs a deeply nested
615  /// `\hbox{\raise…\hbox{…}}`). Traversal is capped at [`EB_BUDGET`] sub-boxes
616  /// (depth-first) so the estimate stays O(1) per box; deeply nested boxes
617  /// beyond the budget are under-counted, which is safe (the guard only needs
618  /// a monotone lower bound to catch unbounded growth).
619  pub fn estimate_bytes(&self) -> usize {
620    let mut budget: u32 = EB_BUDGET;
621    self.estimate_bytes_into(&mut budget)
622  }
623
624  // Paired with `fingerprint_into` above — keep both matches exhaustive
625  // (see the note there).
626  fn estimate_bytes_into(&self, budget: &mut u32) -> usize {
627    if *budget == 0 {
628      return 0;
629    }
630    *budget -= 1;
631    // Per-node fixed overhead: the `Rc<DigestedData>` control block + the
632    // enum discriminant + the inner `RefCell`/`Box`. Deliberately coarse.
633    const NODE: usize = 64;
634    // Note: text and the `Rc<Font>` are shared (interned / ref-counted), so they
635    // add no marginal per-box bytes. What DOES accumulate per box and dominates
636    // RSS is each box's OWNED data: the `properties` HashMap, the `tokens`
637    // source-TeX vector (`Tbox`), the args/children vectors, and — crucially —
638    // the nested children themselves.
639    //
640    // `map_bytes`: a `SymHashMap` (hashbrown) allocates a control-byte table +
641    // key/value slots at ~7/8 load; ~96 B per live entry plus the base table
642    // covers control bytes, the `Stored` value enum, and growth slack.
643    fn map_bytes(n: usize) -> usize { if n == 0 { 0 } else { 64 + n * 96 } }
644    match self.data() {
645      DigestedData::TBox(b) => {
646        let mut bytes = NODE + 48;
647        if let Ok(tb) = b.try_borrow() {
648          bytes += map_bytes(tb.properties.len());
649          bytes += tb.tokens.len() * 16; // owned source-TeX tokens
650        }
651        bytes
652      },
653      DigestedData::Whatsit(w) => {
654        let mut bytes = NODE + 64;
655        if let Ok(wb) = w.try_borrow() {
656          bytes += map_bytes(wb.properties.len());
657          bytes += wb.args.len() * 16;
658          for arg in &wb.args {
659            if *budget == 0 {
660              break;
661            }
662            if let Some(d) = arg {
663              bytes += d.estimate_bytes_into(budget);
664            }
665          }
666        }
667        bytes
668      },
669      DigestedData::List(l) => {
670        let mut bytes = NODE + 48;
671        if let Ok(lb) = l.try_borrow() {
672          bytes += map_bytes(lb.properties.len());
673          bytes += lb.boxes.len() * 8;
674          for child in &lb.boxes {
675            if *budget == 0 {
676              break;
677            }
678            bytes += child.estimate_bytes_into(budget);
679          }
680        }
681        bytes
682      },
683      DigestedData::Postponed(_) => NODE + 32,
684      DigestedData::Comment(_) => NODE + 16,
685      DigestedData::Alignment(_) | DigestedData::KeyVals(_) | DigestedData::RegisterValue(_) => {
686        NODE
687      },
688    }
689  }

751  /// Check if all items are "empty" or only spaces or otherwise skippable in a table cell.
752  /// Perl: isSkippable (Alignment.pm L484-508)
753  pub fn is_skippable(&self) -> bool {
754    use DigestedData::*;
755    match *self.0 {
756      Comment(_) => true,
757      TBox(ref b) => {
758        let b = b.borrow();
759        if b.get_property_bool("alignmentPreserve") {
760          // Perl PR #2767: explicitly preserved content is never skippable
761          false
762        } else if b.get_property_bool("isEmpty")
763          || b.get_property_bool("isSpace")
764          || b.get_property_bool("alignmentSkippable")
765        {
766          true
767        } else {
768          // Perl: getString, check if only whitespace
769          b.get_string()
770            .ok()
771            .map(|s| s.trim().is_empty())
772            .unwrap_or(false)
773        }
774      },
775      List(ref l) => {
776        let l = l.borrow();
777        // Perl PR #2767: explicitly preserved content is never skippable
778        !l.get_property_bool("alignmentPreserve") && l.boxes.iter().all(|d| d.is_skippable())
779      },
780      Whatsit(ref w) => {
781        let w = w.borrow();
782        if w.get_property_bool("alignmentPreserve") {
783          // Perl PR #2767: explicitly preserved content is never skippable
784          false
785        } else if w.get_property_bool("isEmpty")
786          || w.get_property_bool("isSpace")
787          || w.get_property_bool("alignmentSkippable")
788        {
789          true
790        } else {
791          match w.get_body() {
792            Ok(Some(body)) => body.is_skippable(),
793            _ => {
794              match w.get_property("content_box") {
795                Some(ref prop) => {
796                  // Perl: $thing->getProperty('content_box') — for \hbox etc.
797                  match &**prop {
798                    Stored::Digested(cb) => cb.is_skippable(),
799                    _ => false,
800                  }
801                },
802                _ => false,
803              }
804            },
805          }
806        }
807      },
808      Postponed(ref tks) => {
809        // Perl checks token catcodes: letters, others, active, CS are NOT skippable
810        tks.unlist_ref().iter().all(|t| {
811          let cc = t.get_catcode();
812          !matches!(
813            cc,
814            crate::token::Catcode::LETTER
815              | crate::token::Catcode::OTHER
816              | crate::token::Catcode::ACTIVE
817              | crate::token::Catcode::CS
818          )
819        })
820      },
821      _ => false,
822    }
823  }

953  /// Iteratively dismantle a deeply-nested `Digested` so its `Drop` does NOT
954  /// recurse once per nesting level. A deep singly-owned `Rc<DigestedData>`
955  /// chain otherwise overflows the small per-test-thread stack under CI's
956  /// `--test-threads=2` (a SIGABRT that passes locally on the larger default
957  /// stack). Descends into the FIRST child of each `List`, dropping the
958  /// shallow remainder (leaf boxes + the now-childless outer) as it goes, so
959  /// teardown is O(1) in stack depth. (The analogous PRODUCTION concern — a
960  /// boxing-depth-cap Fatal unwinding a deep structure — is tracked
961  /// separately; the real box-list runaway is a WIDE `Vec`, already
962  /// iteratively dropped.)
963  fn drain_nest(mut cur: Digested) {
964    loop {
965      let child = if let DigestedData::List(l) = &*cur.0 {
966        let mut boxes = std::mem::take(&mut l.borrow_mut().boxes);
967        (!boxes.is_empty()).then(|| boxes.swap_remove(0))
968      } else {
969        None
970      };
971      match child {
972        Some(c) => cur = c,
973        None => break,
974      }
975    }
976  }
