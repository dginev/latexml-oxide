6pub const UNITY: i64 = 65536;
7pub const UNITY_F64: f64 = 65536.0;
8pub const EPSILON: f64 = 0.000_000_119_209_29;
9pub const ROUNDING_HALF: f64 = 0.49999994;
10pub const SCALES: &[i32] = &[1, 10, 100, 1000, 10000, 100_000];
11
12/// Round $number to $prec decimals (0...6) attempting to do so portably.
13pub fn round_to(number: f64, prec_opt: Option<u8>) -> f64 {
14  let mut prec = prec_opt.unwrap_or(2);
15  if prec > 5 {
16    prec = 5;
17  }
18  let scale = SCALES[prec as usize];
19  // scale to integer, w/some slop in case arbitrarily close to an integer...
20  let n = number * scale as f64 * (1.0 + 100.0 * EPSILON);
21  let adjusted: f64 = if n < -EPSILON {
22    n - 0.5
23  } else if n > EPSILON {
24    n + 0.5
25  } else {
26    0.0
27  };
28  adjusted.trunc() / scale as f64
29}

31/// An attempt at rounding floats to integers (like scaled points),
32/// in a (hopefully) Knuthian manner (like round_decimals \S102 in Tex The Program)
33// DG: Note that we have to go to the largest `i64` type to contain the truncation
34// of large SP values multiplied up by UNITY
35pub fn kround(number: f64) -> i64 {
36  let rounded = if number < 0.0 {
37    number - ROUNDING_HALF
38  } else {
39    number + ROUNDING_HALF
40  };
41  rounded.trunc() as i64
42}
43
44/// Convert `float` to a fixed-point number
45///
46/// If `unit` is given, it is number of units PER SCALED-POINT! (hence, extra division)
47/// AND, note that the float is rounded and THEN truncated after multiplying by units!
48/// to mimic TeX's behavior.
49pub fn fixpoint(float: f64, unit_opt: Option<f64>) -> i64 {
50  let fix = kround(float * UNITY_F64);
51  if let Some(unit) = unit_opt {
52    (fix as f64 * unit / UNITY_F64).trunc() as i64
53  } else {
54    fix
55  }
56}

58pub trait NumericOps {
59  fn new(num: i64) -> Self
60  where Self: Sized;
61  fn new_f64(num: f64) -> Self
62  where Self: Sized;
63  fn unit(&self) -> Option<&'static str> { None }
64  fn value_of(self) -> i64;
65  fn value_f64(self) -> f64
66  where Self: Sized {
67    self.value_of() as f64
68  }
69  fn pt_value(self, prec: Option<u8>) -> f64
70  where Self: Sized {
71    round_to(self.value_of() as f64 / UNITY_F64, prec)
72  }
73  fn px_value(self, prec: Option<u8>) -> f64
74  where Self: Sized {
75    let dpi = crate::state::lookup_int("DPI");
76    let dpi = if dpi > 0 { dpi as f64 } else { 100.0 };
77    round_to((self.value_f64() / UNITY_F64) * (dpi / 72.27), prec)
78  }
79
80  fn absolute(self) -> Self
81  where Self: Sized {
82    Self::new(self.value_of().abs())
83  }
84
85  fn sign(self) -> i8
86  where Self: Sized {
87    use std::cmp::Ordering::*;
88    match self.value_of().cmp(&0) {
89      Less => -1,
90      Equal => 0,
91      Greater => 1,
92    }
93  }
94
95  fn negate(self) -> Self
96  where Self: Sized {
97    Self::new(-self.value_of())
98  }
99  fn add<T: NumericOps>(self, other: T) -> Self
100  where Self: Sized {
101    Self::new(self.value_of() + other.value_of())
102  }
103  fn subtract<T: NumericOps>(self, other: T) -> Self
104  where Self: Sized {
105    Self::new(self.value_of() - other.value_of())
106  }
107  // Perl: int($self->valueOf * $other->valueOf) — uses float arithmetic to
108  // handle Float multipliers correctly, then truncates.
109  fn multiply<T: NumericOps>(self, other: T) -> Self
110  where Self: Sized {
111    Self::new((self.value_of() as f64 * other.value_f64()) as i64)
112  }
113  /// Truncating division
114  fn divide<T: NumericOps>(self, other: T) -> Self
115  where Self: Sized {
116    let mut other_value: f64 = other.value_of() as f64;
117    if other_value == 0.0 {
118      other_value = EPSILON; // avoid dividing by zero
119    }
120    Self::new((self.value_of() as f64 / other_value).trunc() as i64)
121  }
122
123  /// Rounding division
124  fn divideround<T: NumericOps>(self, other: T) -> Self
125  where Self: Sized {
126    let mut other_value: f64 = other.value_of() as f64;
127    if other_value == 0.0 {
128      other_value = EPSILON; // avoid dividing by zero
129    }
130    Self::new((0.5 + self.value_of() as f64 / other_value).trunc() as i64)
131  }
132
133  fn smaller<T: NumericOps>(self, other: T) -> Self
134  where Self: Sized {
135    Self::new(self.value_of().min(other.value_of()))
136  }
137
138  fn larger<T: NumericOps>(self, other: T) -> Self
139  where Self: Sized {
140    Self::new(self.value_of().max(other.value_of()))
141  }
142
143  fn to_token(self) -> Token
144  where Self: Sized {
145    T_OTHER!(self.value_of().to_string())
146  }
147  // dancing around meta-programming in the Glue case... is there a better way?
148  fn into_glue_type(self) -> Glue
149  where Self: Sized {
150    Glue::new(0) // default: zero glue
151  }
152  fn register_type(&self) -> RegisterType;
153  fn to_attribute(&self) -> String
154  where Self: Display {
155    self.to_string()
156  }
157}

163  #[test]
164  fn round_to_default_precision_is_two() {
165    assert_eq!(round_to(1.2345, None), 1.23);
166    assert_eq!(round_to(1.2355, None), 1.24);
167    assert_eq!(round_to(0.0, None), 0.0);
168  }

170  #[test]
171  fn round_to_respects_precision() {
172    assert_eq!(round_to(1.23456, Some(3)), 1.235);
173    assert_eq!(round_to(1.23456, Some(0)), 1.0);
174    assert_eq!(round_to(1.5, Some(0)), 2.0);
175  }
176
177  #[test]
178  fn round_to_caps_precision_at_five() {
179    // The doc-comment says 0..=5 is the intended range; precisions
180    // above 5 are clamped to 5.
181    let a = round_to(1.12345, Some(5));
182    let b = round_to(1.12345, Some(10));
183    assert_eq!(a, b, "precision > 5 clamps to 5 (got {a} vs {b})");
184  }
185
186  #[test]
187  fn round_to_negative_numbers() {
188    assert_eq!(round_to(-1.235, None), -1.24);
189    assert_eq!(round_to(-0.005, None), -0.01);
190  }
191
192  #[test]
193  fn kround_basic() {
194    assert_eq!(kround(0.0), 0);
195    assert_eq!(kround(0.49), 0);
196    // 0.5 + ROUNDING_HALF (0.49999994) = 0.99999994 → trunc = 0
197    // Knuthian rounding below is actually a bit different from banker's.
198    assert_eq!(kround(1.49), 1);
199    assert_eq!(kround(1.5), 1);
200    assert_eq!(kround(-0.49), 0);
201    assert_eq!(kround(-1.49), -1);
202  }
203
204  #[test]
205  fn fixpoint_without_unit() {
206    // fixpoint(x, None) returns kround(x * 65536).
207    assert_eq!(fixpoint(1.0, None), UNITY);
208    assert_eq!(fixpoint(0.0, None), 0);
209    assert_eq!(fixpoint(0.5, None), UNITY / 2);
210  }

212  #[test]
213  fn fixpoint_with_unit_scales() {
214    // unit=1.0 means 1 unit per scaled-point, so:
215    //   fix(1.0, Some(1.0)) = kround(65536.0) * 1.0 / 65536.0 = 1 (truncated)
216    let out = fixpoint(1.0, Some(1.0));
217    // Result depends on the unit semantics ("units PER SCALED-POINT").

220  }
