14/// Properties for a decoded math character, mirroring Perl's decodeMathChar return
15#[derive(Debug, Clone, Default)]
16pub struct MathCharProps {
17  pub role: Option<String>,
18  pub glyph: Option<char>,
19  pub meaning: Option<String>,
20  pub name: Option<String>,
21  pub stretchy: Option<String>,
22  pub need_scriptpos: bool,
23  pub need_mathstyle: bool,
24  pub scriptpos: Option<String>,
25  pub mathstyle: Option<String>,
26  pub reversion: Option<crate::tokens::Tokens>,
27  pub font: Option<crate::common::font::Font>,
28}
29
30impl MathCharProps {
31  /// Convert need_scriptpos/need_mathstyle flags to actual values based on display mode
32  pub fn resolve_style_props(&mut self) {
33    let in_display = state::lookup_bool("IN_MATH_DISPLAY")
34      || state::lookup_font()
35        .map(|f| f.get_mathstyle().map(|s| s.as_ref()) == Some("display"))
36        .unwrap_or(false);
37    if self.need_scriptpos {
38      self.scriptpos = Some(if in_display { "mid" } else { "post" }.to_string());
39    }
40    if self.need_mathstyle {
41      self.mathstyle = Some(if in_display { "display" } else { "text" }.to_string());
42    }
43  }
44
45  /// Insert all properties into a HashMap for Tbox construction
46  pub fn into_props_map(self) -> HashMap<&'static str, Stored> {
47    let mut props = HashMap::default();
48    if let Some(role) = self.role {
49      props.insert("role", Stored::String(arena::pin(role)));
50    }
51    if let Some(meaning) = self.meaning {
52      props.insert("meaning", Stored::String(arena::pin(meaning)));
53    }
54    if let Some(name) = self.name {
55      props.insert("name", Stored::String(arena::pin(name)));
56    }
57    if let Some(stretchy) = self.stretchy {
58      props.insert("stretchy", Stored::String(arena::pin(stretchy)));
59    }
60    if let Some(scriptpos) = self.scriptpos {
61      props.insert("scriptpos", Stored::String(arena::pin(scriptpos)));
62    }
63    if let Some(mathstyle) = self.mathstyle {
64      props.insert("mathstyle", Stored::String(arena::pin(mathstyle)));
65    }
66    props
67  }
68}

289// Is this "fontinfo" stuff sufficient to maintain a math font "family" ??
290// What we're really after is a connection to a font encoding mapping.
291pub fn decode_math_char(mut n: u16, reversion: Option<crate::tokens::Tokens>) -> Result<MathCharProps> {
292  let class: u16 = n / (16 * 256);
293  n %= 16 * 256;
294  let mut fam: u16 = n / 256;
295  
296  let curfam_val: i32 = match state::lookup_register("\\fam", Vec::new()) {
297    Ok(Some(crate::definition::register::RegisterValue::Number(curfam))) => curfam.0 as i32,
298    _ => -1,
299  };
300
301  if class == 7
302    && (0..=15).contains(&curfam_val) {
303      fam = curfam_val as u16;
304    }
305  n %= 256;
306
307  let curfont = state::lookup_font().unwrap();
308  let initfont = match state::lookup_value("initial_math_font") {
309    Some(Stored::Font(f)) => f.clone(),
310    _ => curfont.clone(),
311  };
312
313  let mut use_current_font = false;
314  let mut maybe_rev = curfam_val >= 0 && fam != 1;
315  let mut fontdef_tok: Option<Token> = None;
316  
317  if class == 7 && curfam_val < 0 && curfont != initfont {
318    use_current_font = true;
319    maybe_rev = true;
320    fontdef_tok = Some(crate::T_CS!("\\font"));
321  }
322
323  let mut downsize = 0;
324  if fontdef_tok.is_none() {
325    let style = curfont.get_mathstyle().map(|s| s.to_string()).unwrap_or_default();
326    let style_str = if style == "script" || style == "scriptscript" || style == "text" { style.as_str() } else { "text" };
327    if style_str == "text" {
328      if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("textfont_{fam}")) { fontdef_tok = Some(t); }
329    } else if style_str == "script" {
330      if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("scriptfont_{fam}")) { fontdef_tok = Some(t); }
331      else if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("textfont_{fam}")) { fontdef_tok = Some(t); downsize = 1; }
332    } else if style_str == "scriptscript" {
333      if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("scriptscriptfont_{fam}")) { fontdef_tok = Some(t); }
334      else if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("scriptfont_{fam}")) { fontdef_tok = Some(t); downsize = 1; }
335      else if let Some(Stored::Token(t)) = state::lookup_value(&crate::s!("textfont_{fam}")) { fontdef_tok = Some(t); downsize = 2; }
336    }
337  }
338  
339  let c = n as u8 as char;
340  // Guard against invalid class values from corrupted mathchar codes
341  // (e.g., during expl3 loading when \__int_eval_end: errors corrupt state)
342  if (class as usize) >= MATH_CLASS_ROLE.len() {
343    return Ok(MathCharProps::default());
344  }
345  let class_role = MATH_CLASS_ROLE[class as usize];
346
347  let mut f = (*curfont).clone();
348  if let Some(ftok) = &fontdef_tok {
349    if use_current_font {
350      // f is already curfont
351    } else {
352      // Merge textfont info (family, encoding, etc.) but preserve the current
353      // font's size. The textfont defines design-time properties, while the
354      // current size comes from context (e.g. \big's font => { size => 12 }).
355      let preserved_size = f.size;
356      state::with_font_info(ftok, |fontinfo| {
357        if let Some(Stored::Font(ref info)) = fontinfo.unwrap_or(None) {
358          f = f.merge_ref(info);
359        } else {
360          // Perl: fallback to \lx@default@font if not found
361          if let Some(Stored::Token(d_tok)) = state::lookup_value("\\lx@default@font") {
362             state::with_font_info(&d_tok, |d_info| {
363               if let Some(Stored::Font(ref d_f)) = d_info.unwrap_or(None) {
364                 f = f.merge_ref(d_f);
365               }
366             });
367          }
368        }
369      });
370      f.size = preserved_size;
371    }
372  }
373
374  if downsize > 0 {
375    f.scripted = Some(true);
376  }
377  if downsize > 1 {
378    f.scripted = Some(true);
379  }
380
381  let d = f.relative_to(&curfont);
382
383  let glyph = if use_current_font {
384    if let Some(ref data) = curfont.encoding {
385      crate::common::font::decode(n as u8, Some(data.to_string()), false)
386    } else {
387      Some(c)
388    }
389  } else if let Some(ftok) = &fontdef_tok {
390    // Extract the encoding BEFORE calling font::decode. font::decode may
391    // call preload_font_map which mutates state, and with_font_info holds
392    // a State borrow while its closure runs — the reentrant mutation
393    // panics with "RefCell already borrowed" (sandbox paper 0711.4787).
394    let encoding_opt: Option<String> = state::with_font_info(ftok, |fontinfo| {
395      if let Some(Stored::Font(ref info)) = fontinfo? {
396        Ok::<Option<String>, crate::common::error::Error>(
397          info.encoding.as_ref().map(|s| s.to_string())
398        )
399      } else {
400        Ok(None)
401      }
402    })?;
403    if let Some(data) = encoding_opt {
404      crate::common::font::decode(n as u8, Some(data), false)
405    } else {
406      Some(c)
407    }
408  } else {
409    Some(c)
410  };
411
412  let glyph_char = glyph.unwrap_or(c);
413  let charinfo = unicode_math_properties(glyph_char);
414  let mut props = charinfo.clone().unwrap_or_default();
415  props.glyph = glyph;
416
417  let mut role = charinfo.as_ref().and_then(|info| info.role.clone());
418  if role.is_none() && !class_role.is_empty() {
419    role = Some(class_role.to_string());
420  }
421  if role.is_some() && props.role.is_none() {
422    props.role = role;
423  }
424
425  props.resolve_style_props();
426
427  let mut final_reversion = reversion;
428  if let Some(rev) = final_reversion.clone() {
429    let mut wrap = maybe_rev && !d.is_empty();
430    if state::lookup_value("LaTeX.pool.ltxml_loaded").is_some() {
431      wrap = false;
432    }
433    if wrap {
434      if let Some(ftok) = fontdef_tok {
435        let mut new_rev = vec![crate::T_BEGIN!(), ftok];
436        new_rev.extend(rev.unlist());
437        new_rev.push(crate::T_END!());
438        final_reversion = Some(crate::tokens::Tokens::new(new_rev));
439      }
440    }
441    props.reversion = final_reversion;
442  }
443
444  props.font = Some(f);
445
446  Ok(props)
447}

449/// Stomach-level hook for decoding math characters.
450/// Called from stomach::invoke_token_simple when IN_MATH and mathcode is set.
451/// Perl: decodeMathChar($mathcode, $meaning) in Stomach::invokeToken_simple
452pub fn decode_math_char_for_stomach(
453  mathcode: u16,
454  meaning: Token,
455) -> Result<Option<Digested>> {
456  let props = decode_math_char(mathcode, Some(crate::Tokens!(meaning)))?;
457
458  let glyph = match props.glyph {
459    Some(g) => g,
460    None => return Ok(None),
461  };
462
463  let mut properties = SymHashMap::default();
464  properties.insert("mode", Stored::String(*arena::MATH_SYM));
465  if let Some(ref role) = props.role {
466    properties.insert("role", Stored::String(arena::pin(role)));
467  }
468  if let Some(ref m) = props.meaning {
469    properties.insert("meaning", Stored::String(arena::pin(m)));
470  }
471  if let Some(ref name) = props.name {
472    properties.insert("name", Stored::String(arena::pin(name)));
473  }
474  if let Some(ref stretchy) = props.stretchy {
475    properties.insert("stretchy", Stored::String(arena::pin(stretchy)));
476  }
477  if let Some(ref scriptpos) = props.scriptpos {
478    properties.insert("scriptpos", Stored::String(arena::pin(scriptpos)));
479  }
480  if let Some(ref mathstyle) = props.mathstyle {
481    properties.insert("mathstyle", Stored::String(arena::pin(mathstyle)));
482  }
483
484  let glyph_sym = arena::pin(glyph.to_string());
485
486  let font = props.font.map(|f| {
487    Rc::new(arena::with(glyph_sym, |s| f.specialize(s)))
488  });
489  Ok(Some(Digested::from(Tbox::new(
490    glyph_sym,
491    font,
492    None,
493    props.reversion.unwrap_or(crate::Tokens!(meaning)),
494    properties,
495  ))))
496}
