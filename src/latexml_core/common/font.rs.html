200pub fn decode_fontname(name: &str, at_opt: Option<f64>, scaled_opt: Option<f64>) -> Option<Font> {
201  if let Some(cap) = FONT_RE.captures(name) {
202    let mut props = Font::default();
203    let fam = cap.get(1).map_or("", |m| m.as_str());
204    let ser = cap.get(2).map_or("", |m| m.as_str());
205    let shp = cap.get(3).map_or("", |m| m.as_str());
206    let size_str = cap.get(4).map_or("", |m| m.as_str());
207    // TODO: Maybe a straight merge isn't the way to go here -- and we should do a property
208    // whitelist as in Perl? but the merge is convenient, especially if we can state "only the
209    // values different from the default Font" which may already work courtesy of "None"? Need
210    // to think it through...
211    if let Some(ffam) = lookup_font_family(fam) {
212      props = props.merge(ffam.clone());
213    }
214    if let Some(fser) = lookup_font_series(ser) {
215      props = props.merge(fser.clone());
216    }
217    if let Some(fsh) = lookup_font_shape(shp) {
218      props = props.merge(fsh.clone());
219    }
220    let mut size = if let Some(at) = at_opt {
221      at
222    } else {
223      let size_f64 = size_str.parse::<f64>().unwrap_or(1.0);
224      if size_f64 == 0.0 { 1.0 } else { size_f64 } // Yes, also if 0, "" (from regexp)
225    };
226    if let Some(scaled) = scaled_opt {
227      size *= scaled;
228    }
229    props.size = Some(size);
230    // Experimental Hack !?!?!?
231    if props.encoding.is_none() {
232      props.encoding = Some(Cow::Borrowed("OT1"));
233    }
234    // TODO: What is this field for?
235    // if let Some(at) = at_opt {
236    //   props.at = Some(s!("{at}pt"));
237    // }
238    Some(props)
239  } else {
240    None
241  }
242}

244/// A data structure containing Font information, but also related textual properties (such as
245/// color)
246///
247/// This struct is a little interesting, as we want to pass overrides that partially modify (via a
248/// merge) the current font, in each definitional binding. To accommodate that with this struct,
249/// every single field needs to be an Option, in order to unambiguously tell the "intend" of
250/// override (Some) vs no intent (None).
251#[derive(Clone, PartialEq, Default)]
252pub struct Font {
253  pub family:        Option<Cow<'static, str>>,
254  pub series:        Option<Cow<'static, str>>,
255  pub shape:         Option<Cow<'static, str>>,
256  pub size:          Option<f64>,
257  pub color:         Option<Cow<'static, str>>,
258  pub bg:            Option<Cow<'static, str>>,
259  pub opacity:       Option<Cow<'static, str>>,
260  pub encoding:      Option<Cow<'static, str>>,
261  pub language:      Option<Cow<'static, str>>,
262  pub mathstyle:     Option<Cow<'static, str>>,
263  pub mathstylestep: Option<i32>,
264  pub name:          Option<Cow<'static, str>>,
265  pub emph:          Option<bool>,
266  pub scripted:      Option<bool>,
267  pub fraction:      Option<bool>,
268  // Note: forcefamily, forceseries, forceshape (& forcebold for compatibility)
269  // are only useful for fonts in math; See the specialize method below.
270  pub forceseries:   Option<bool>,
271  pub forcefamily:   Option<bool>,
272  pub forceshape:    Option<bool>,
273  pub forcebold:     Option<bool>,
274  pub scale:         Option<f64>,
275  pub flags:         Option<u8>,
276}

278impl Hash for Font {
279  // We need to implement hash since we have to tell Rust how to hash `f64` values
280  // for now I have decided to go for a precision of 4 digits after the decimal point,
281  // so multiplying by 1000
282  fn hash<H: Hasher>(&self, hasher: &mut H) {
283    self.family.hash(hasher);
284    self.series.hash(hasher);
285    self.shape.hash(hasher);
286    self.size.map(|size| (size * 1000.0) as i64).hash(hasher);
287    self.color.hash(hasher);
288    self.bg.hash(hasher);
289    self.opacity.hash(hasher);
290    self.encoding.hash(hasher);
291    self.language.hash(hasher);
292    self.mathstyle.hash(hasher);
293    self.mathstylestep.hash(hasher);
294    self.name.hash(hasher);
295    self.emph.hash(hasher);
296    self.scripted.hash(hasher);
297    self.forceseries.hash(hasher);
298    self.forcefamily.hash(hasher);
299    self.forceshape.hash(hasher);
300    self.scale.map(|scale| (scale * 1000.0) as i64).hash(hasher);
301    self.flags.hash(hasher);
302  }
303}

482  // NOTE: In math, NORMALLY, setting any one of
483  //    family, series or shape
484  // will, usually, automatically reset the others to thier defaults!
485  // You must arrange this in the calls....
486  pub fn merge(&self, other: Font) -> Self {
487    let mut newfont = self.clone();
488    // first set direct overrides.
489    if let Some(value) = other.family {
490      newfont.family = Some(value);
491    }
492    if let Some(value) = other.series {
493      newfont.series = Some(value);
494    }
495    if let Some(value) = other.shape {
496      newfont.shape = Some(value);
497    }
498    if other.emph == Some(true) {
499      newfont.shape = if newfont.shape.unwrap_or(Cow::Borrowed("")) == "italic" {
500        Some(Cow::Borrowed("upright"))
501      } else {
502        Some(Cow::Borrowed("italic"))
503      };
504    }
505    if let Some(value) = other.size {
506      newfont.size = Some(value);
507    }
508    if let Some(value) = other.color {
509      newfont.color = Some(value);
510    }
511    if let Some(value) = other.bg {
512      newfont.bg = Some(value);
513    }
514    if let Some(value) = other.opacity {
515      newfont.opacity = Some(value);
516    }
517    if let Some(value) = other.encoding {
518      newfont.encoding = Some(value);
519    }
520    if let Some(value) = other.language {
521      newfont.language = Some(value);
522    }
523    let mut has_mathstyle = false;
524    if let Some(value) = other.mathstyle {
525      has_mathstyle = true;
526      newfont.mathstyle = Some(value);
527    }
528    if let Some(value) = other.forceseries {
529      newfont.forceseries = Some(value);
530    }
531    if let Some(value) = other.forcefamily {
532      newfont.forcefamily = Some(value);
533    }
534    if let Some(value) = other.forceshape {
535      newfont.forceshape = Some(value);
536    }
537
538    // Now perform any dynamic adjustment directives
539    if let Some(scale) = other.scale {
540      if let Some(size) = newfont.size {
541        newfont.size = Some(scale * size);
542      }
543    }
544    // Set the mathstyle, and also the size from the mathstyle
545    // But we may need to scale that size against the existing or requested size.
546    let style_scale = if let Some(size) = newfont.size {
547      let key = self.mathstyle.as_ref().unwrap_or(&Cow::Borrowed("display"));
548      // the explicit &str typecast is currently needed for rust to
549      // figure out how to use the Cow<str> in the HashMap lookup.
550      let str_key: &str = key;
551      size / *STYLE_SIZE.get(str_key).unwrap() as f64
552    } else {
553      1.0
554    };
555    // Explicitly requested size, use it; else
556    if other.size.is_none() {
557      if has_mathstyle {
558        // otherwise set the size from mathstyle
559        let str_mathstyle: &str = newfont.mathstyle.as_ref().unwrap();
560        newfont.size = Some(style_scale * *STYLE_SIZE.get(str_mathstyle).unwrap() as f64);
561      } else if Some(true) == other.scripted {
562        // Or adjust both the mathstyle & size for scripts
563        let str_stylekey: &str = self.mathstyle.as_ref().unwrap_or(&Cow::Borrowed("display"));
564        newfont.mathstyle = SCRIPT_STYLE_MAP
565          .get(str_stylekey)
566          .map(|c| Cow::Borrowed(*c));
567        let str_mathstylekey: &str = newfont
568          .mathstyle
569          .as_ref()
570          .unwrap_or(&Cow::Borrowed("display"));
571        newfont.size = Some(style_scale * *STYLE_SIZE.get(str_mathstylekey).unwrap() as f64);
572      }
573    }
574
575    // TODO:
576    // elsif ($options{fraction}) {     # Or adjust both for fractions
577    //   $mathstyle = $fracstylemap{ $mathstyle            || 'display' };
578    //   $size      = $style_scale * $stylesize{ $mathstyle || 'display' }; }
579
580    // if ($options{emph}) {
581    //   $shape = ($shape eq 'italic' ? 'upright' : 'italic');
582    //   $flags |= $FLAG_EMPH; }
583    // $flags &= ~$FLAG_EMPH if $mathstyle;    # Disable emph in math
584    //   newfont
585    // }
586    newfont
587  }

589  /// Instanciate the font for a particular class of symbols.
590  /// NOTE: This works in `normal' latex, but probably needs some tunability.
591  /// Depending on the fonts being used, the allowable combinations may be different.
592  /// Getting the font right is important, since the author probably
593  /// thinks of the identity of the symbols according to what they SEE in the printed
594  /// document.  Even though the markup might seem to indicate something else...
595  ///
596  /// Use Unicode properties to determine font merging.
597  pub fn specialize(&self, text: &str) -> Self {
598    let mut new = self.clone();
599    if text.is_empty() {
600      return new; // ?
601    }
602    let deffamily = if self.forcefamily.unwrap_or(false) {
603      self.family.clone().unwrap_or_else(|| DEFFAMILY.into())
604    } else {
605      DEFFAMILY.into()
606    };
607    let defseries = if self.forceseries.unwrap_or(false) {
608      self.series.clone().unwrap_or_else(|| DEFSERIES.into())
609    } else {
610      DEFSERIES.into()
611    };
612    let defshape = if self.forceshape.unwrap_or(false) {
613      self.shape.clone().unwrap_or_else(|| DEFSERIES.into())
614    } else {
615      DEFSHAPE.into()
616    };
617    if LATIN_LETTER_RE.is_match(text) {
618      // Latin Letter
619      if new.shape.is_none() && new.family.is_none() {
620        new.shape = Some("italic".into());
621      }
622    } else if GREEK_LETTER_RE.is_match(text) {
623      // Single Greek character?
624      if UPPER_LETTER_RE.is_match(text) {
625        // Uppercase
626        if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
627          new.family = Some(deffamily);
628          if new.shape.is_some() && (new.shape != Some(DEFSHAPE.into())) {
629            new.shape = Some(defshape); // if ANY shape, must be default
630          }
631        }
632      } else {
633        // Lowercase
634        match new.family {
635          None => {
636            new.family = Some(deffamily);
637          },
638          Some(nf) if nf != DEFFAMILY => {
639            new.family = Some(deffamily);
640          },
641          _ => {},
642        };
643        // if new.shape.is_none() {// always ?
644        new.shape = Some("italic".into());
645        // }
646        if new.series.is_some() && (new.series != Some(DEFSERIES.into())) {
647          new.series = Some(defseries);
648        }
649      }
650    } else if DIGIT_LETTER_RE.is_match(text) {
651      // Digit
652      if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
653        new.family = Some(deffamily);
654        new.shape = Some(defshape); // defaults, always.
655      }
656    } else {
657      // Other Symbol
658      new.family = Some(deffamily);
659      new.shape = Some(defshape); // defaults, always.
660      if new.series.is_some() && (new.series.as_ref().unwrap() != DEFSERIES) {
661        new.series = Some(defseries);
662      } // defaults, always.
663    }
664    new
665  }

711  /// This method compares 2 fonts, returning the differences between them.
712  /// Noting that the font-related attributes in the schema distill the
713  /// font properties into fewer attributes (font,fontsize,color,background,opacity),
714  /// the return value encodes both the attribute changes that would be needed to effect
715  /// the font change, along with the font properties that differed
716  /// Namely, the result is a hash keyed on the attribute name and whose value is a FontDiff
717  ///    value      => "new_attribute_value"
718  ///    properties => { %fontproperties }
719  /// or (String, Font)
720  pub fn relative_to(&self, other: &Font) -> HashMap<String, (String, Font)> {
721    let family = match self.family {
722      Some(ref fam) => {
723        if fam == "math" {
724          Some(Cow::Borrowed("serif"))
725        } else {
726          Some(fam.clone())
727        }
728      },
729      None => None,
730    };
731    let other_family = match other.family {
732      Some(ref fam) => {
733        if fam == "math" {
734          Some(Cow::Borrowed("serif"))
735        } else {
736          Some(fam.clone())
737        }
738      },
739      None => None,
740    };
741    let mut diffs = vec![];
742    let mut font_properties = Font::default();
743    if is_diff(family.as_ref(), other_family.as_ref()) {
744      diffs.push(family.clone().unwrap());
745      font_properties.family = family;
746    }
747    if is_diff(self.series.as_ref(), other.series.as_ref()) {
748      let series = self.series.clone().unwrap();
749      diffs.push(series);
750      font_properties.series.clone_from(&self.series);
751    }
752    if is_diff(self.shape.as_ref(), other.shape.as_ref()) {
753      let shape = self.shape.clone().unwrap();
754      diffs.push(shape);
755      font_properties.shape.clone_from(&self.shape);
756    }
757    let mut result = HashMap::default();
758
759    if !diffs.is_empty() {
760      let font_value = diffs.join(" ");
761      result.insert(s!("font"), (font_value, font_properties));
762    }
763
764    if is_diff_f64(self.size.as_ref().copied(), other.size.as_ref().copied()) {
765      result.insert(
766        "fontsize".to_string(),
767        (
768          relative_font_size(self.size.unwrap(), other.size.unwrap()),
769          Font {
770            size: self.size,
771            ..Font::default()
772          },
773        ),
774      );
775    }
776    // ////      ? (fontsize => { value => $siz, properties => { size => $siz } })
777    //   ? (fontsize => { value => relativeFontSize($siz, $osiz), properties => { size => $siz } })
778    //   : ()),
779
780    // (is_diff($col, $ocol)
781    //   ? (color => { value => $col, properties => { color => $col } })
782    //   : ()),
783    // (is_diff($bkg, $obkg)
784    //   ? (backgroundcolor => { value => $bkg, properties => { background => $bkg } })
785    //   : ()),
786    // (is_diff($opa, $oopa)
787    //   ? (opacity => { value => $opa, properties => { opacity => $opa } })
788    //   : ()),
789    // (is_diff($lang, $olang)
790    //   ? ('xml:lang' => { value => $lang, properties => { language => $lang } })
791    //   : ()),
792
793    //// Contemplate this: We do NOT want mathstyle showing up (automatically) in the attributes
794    //// So, we presumably want to ignore differences in mathstyle
795    //// They shouldn't (by themselves) affect the display?
796    ////    (is_diff($mstyle, $omstyle)
797    ////      ? ('mathstyle' => { value => $mstyle, properties => { mathstyle => $mstyle } })
798    ////      : ()),
799    result

863  pub fn compute_string_size(
864    &self,
865    text: &str,
866    _options: SymHashMap<Stored>,
867  ) -> (Dimension, Dimension, Dimension) {
868    if text.is_empty()
869      || self
870        .get_family()
871        .map(|fam| fam == "nullfont")
872        .unwrap_or(false)
873    {
874      return (
875        Dimension::default(),
876        Dimension::default(),
877        Dimension::default(),
878      );
879    }
880    let size = self.get_size().unwrap_or(DEFSIZE);
881    let _ismath = self.get_family().map(|fam| fam == "math").unwrap_or(false);
882    let (mut w, mut h, mut d) = (0, 0, 0);
883    for char in text.chars() {
884      let metric = self.get_metric(Some(char));
885      let entry_opt = metric.sizes.get(char.to_string().as_str());
886      // let entry_opt  = metric.sizes.get(char);
887      let (cw, ch, cd, _ci) = if let Some(entry) = entry_opt {
888        *entry
889      } else {
890        (0.75 * UNITY_F64, 0.7 * UNITY_F64, 0.2 * UNITY_F64, 0.0)
891      };
892      w += (cw * size).trunc() as i64;
893      // if (my $kern = $chars[0] && $$metric{kerns}{ $char . $chars[0] }) {
894      //   $w += int($size * $kern); }
895      // if ($ismath && $ci) {
896      //   $w += int($size * $ci); }
897      h = max(h, (ch * size).trunc() as i64);
898      d = max(d, (cd * size).trunc() as i64);
899    }
900    // The 1 is so that any actual glyph appears to be non-empty.
901    // This is presumably only necessary to deal with the flawed emptiness heiristics in Alignment?
902    if w == 0 {
903      w = 1;
904    }
905    (Dimension::new(w), Dimension::new(h), Dimension::new(d))
906  }

1070/// Decode a codepoint using the fontmap for a given font and/or fontencoding.
1071///
1072/// If `encoding` not provided, then lookup according to the current font's
1073/// encoding; the font family may also be used to choose the fontmap (think tt fonts!).
1074/// When `implicit` is false, we are "explicitly" asking for a decoding, such as
1075/// with \char, \mathchar, \symbol, DeclareTextSymbol and such cases.
1076/// In such cases, only codepoints specifically within the map are covered; the rest are undef.
1077/// If `implicit` is true, we'll decode token content that has made it to the stomach:
1078/// We're going to assume that SOME sort of handling of input encoding is taking place,
1079/// so that if anything above 128 comes in, it must already be Unicode!.
1080/// The lower half plane still needs to go through decoding, though, to deal
1081/// with TeX's rearrangement of ASCII...
1082pub fn decode(code: u8, encoding_opt: Option<String>, implicit: bool) -> Option<char> {
1083  let mut font = None;
1084  let encoding = match encoding_opt {
1085    Some(enc) => Cow::Owned(enc),
1086    None => {
1087      font = lookup_font();
1088      if let Some(ref font) = font {
1089        match font.get_encoding() {
1090          None => Cow::Borrowed(""),
1091          Some(encoding) => encoding.clone(),
1092        }
1093      } else {
1094        Cow::Borrowed("")
1095      }
1096    },
1097  };
1098
1099  let mut map: Option<Fontmap> = None;
1100  if !encoding.is_empty() {
1101    preload_font_map(&encoding).expect("preloading a font map should succeed?");
1102    if let Some(encmap) = load_font_map(&encoding) {
1103      // OK got some map.
1104      map = Some(encmap);
1105      if let Some(ref font) = font {
1106        if let Some(family) = (*font).get_family() {
1107          with_value(&s!("{encoding}_{family}_fontmap"), |fmap_opt| {
1108            if let Some(fmap) = fmap_opt {
1109              map = fmap.into(); // Use the family specific map, if any.
1110            }
1111          });
1112        }
1113      }
1114    }
1115  }
1116
1117  if implicit {
1118    if let Some(map) = map {
1119      if code < 128 {
1120        match map.get(code as usize) {
1121          None => None,
1122          Some(c) => *c,
1123        }
1124      } else {
1125        Some(code.into())
1126      }
1127    } else {
1128      Some(code.into())
1129    }
1130  } else if let Some(map) = map {
1131    match map.get(code as usize) {
1132      None => None,
1133      Some(c) => *c,
1134    }
1135  } else {
1136    None
1137  }
1138}

1140pub fn decode_string(string: SymStr, encoding_opt: Option<&str>, implicit: bool) -> SymStr {
1141  let empty_sym = *EMPTY_SYM;
1142  if string == empty_sym {
1143    return empty_sym;
1144  }
1145  let mut font = None;
1146  let encoding = match encoding_opt {
1147    None => {
1148      font = lookup_font();
1149      if let Some(ref font) = font {
1150        font.get_encoding().unwrap_or(&Cow::Borrowed(""))
1151      } else {
1152        ""
1153      }
1154    },
1155    Some(encoding) => encoding,
1156  };
1157
1158  let mut map: Option<Fontmap> = None;
1159  if !encoding.is_empty() {
1160    preload_font_map(encoding).expect("preload_font_map should succeed?");
1161    if let Some(encmap) = load_font_map(encoding) {
1162      // OK got some map.
1163      map = Some(encmap);
1164      if let Some(ref font) = font {
1165        if let Some(family) = (*font).get_family() {
1166          with_value(&s!("{}_{}_fontmap", encoding, family), |fmap_opt| {
1167            if let Some(fmap) = fmap_opt {
1168              map = fmap.into(); // Use the family specific map, if any.
1169            }
1170          });
1171        }
1172      }
1173    }
1174  }
1175
1176  let mut result_string: String = String::new();
1177  arena::with(string, |str| {
1178    for c in str.chars() {
1179      if implicit {
1180        if let Some(ref map_ref) = map {
1181          let code = c as u16; // u16, so that Unicode chars get cast correctly
1182          if code < 128 {
1183            if let Some(Some(mapc_val)) = map_ref.get(code as usize) {
1184              result_string.push(*mapc_val);
1185            }
1186          } else {
1187            result_string.push(c);
1188          }
1189        } else {
1190          result_string.push(c)
1191        }
1192      } else if let Some(ref map_ref) = map {
1193        let code = c as u8;
1194        if let Some(Some(mapc_val)) = map_ref.get(code as usize) {
1195          result_string.push(*mapc_val);
1196        }
1197      }
1198    }
1199  });
1200  arena::pin(result_string)
1201}
