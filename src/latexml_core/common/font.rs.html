308/// Find a Font Metric for a given fontname, fallback to 10pt or cmr as needed.
309/// Perl: getMetricForName
310pub fn get_metric_for_name(name: &str) -> &'static MetricData {
311  let base = if let Some(idx) = name.find(|c: char| c.is_ascii_digit()) {
312    &name[..idx]
313  } else {
314    name
315  };
316  // Try exact name first (e.g. "cmr10")
317  if let Some(m) = STDMETRICS.get(name) {
318    return m;
319  }
320  // Try base without size (e.g. "cmr" from "cmr10")
321  if let Some(m) = STDMETRICS.get(base) {
322    return m;
323  }
324  // Try base + "10". Stack-buffer concat avoids the per-call `format!`
325  // heap alloc — `get_metric_for_name` is reached through the
326  // per-character `get_metric` loop, so this is a real allocation
327  // site. Font basenames are short (≤ ~20 ASCII bytes); 32 bytes is
328  // ample padding.
329  let base_bytes = base.as_bytes();
330  if base_bytes.len() + 2 <= 32 {
331    let mut buf = [0u8; 32];
332    buf[..base_bytes.len()].copy_from_slice(base_bytes);
333    buf[base_bytes.len()] = b'1';
334    buf[base_bytes.len() + 1] = b'0';
335    if let Ok(s) = std::str::from_utf8(&buf[..base_bytes.len() + 2]) {
336      if let Some(m) = STDMETRICS.get(s) {
337        return m;
338      }
339    }
340  }
341  // Ultimate fallback to "cmr"
342  STDMETRICS
343    .get("cmr")
344    .expect("STDMETRICS must contain 'cmr'")
345}

347pub fn decode_fontname(name: &str, at_opt: Option<f64>, scaled_opt: Option<f64>) -> Option<Font> {
348  if let Some(cap) = FONT_RE.captures(name) {
349    // Perl: my %props = (series => 'medium', shape => 'upright', encoding => 'OT1');
350    let mut props = Font {
351      series: Some(Cow::Borrowed(DEFSERIES)),
352      shape: Some(Cow::Borrowed(DEFSHAPE)),
353      encoding: Some(Cow::Borrowed("OT1")),
354      ..Font::default()
355    };
356    let fam = cap.get(1).map_or("", |m| m.as_str());
357    let ser = cap.get(2).map_or("", |m| m.as_str());
358    let shp = cap.get(3).map_or("", |m| m.as_str());
359    let size_str = cap.get(4).map_or("", |m| m.as_str());
360    if let Some(ffam) = lookup_font_family(fam) {
361      props = props.merge_ref(ffam);
362    }
363    if let Some(fser) = lookup_font_series(ser) {
364      props = props.merge_ref(fser);
365    }
366    if let Some(fsh) = lookup_font_shape(shp) {
367      props = props.merge_ref(fsh);
368    }
369    let mut size = if let Some(at) = at_opt {
370      at
371    } else {
372      let size_f64 = size_str.parse::<f64>().unwrap_or(1.0);
373      if size_f64 == 0.0 { 1.0 } else { size_f64 } // Yes, also if 0, "" (from regexp)
374    };
375    if let Some(scaled) = scaled_opt {
376      size *= scaled;
377    }
378    props.size = Some(size);
379    // Experimental Hack !?!?!?
380    if props.encoding.is_none() {
381      props.encoding = Some(Cow::Borrowed("OT1"));
382    }
383    // TODO: What is this field for?
384    // if let Some(at) = at_opt {
385    //   props.at = Some(s!("{at}pt"));
386    // }
387    Some(props)
388  } else {
389    None
390  }
391}

605  pub fn to_hashable(&self) -> u64 {
606    // MUST be deterministic: equal Fonts must hash equal, stably across calls
607    // AND across process runs. `set_node_font`/`get_node_font` use this as the
608    // `_font` key and `node_fonts` map key, so a randomized seed
609    // (`RandomState::new()`, used here previously) gave the same Font a
610    // different id on every call — breaking font dedup and making the document
611    // build run-to-run non-deterministic (intermittent locked-frame/mode
612    // FATALs, e.g. 1510.04473). `FxHasher` has a fixed seed.
613    let mut hasher = rustc_hash::FxHasher::default();
614    Hash::hash(self, &mut hasher);
615    hasher.finish()
616  }

695      && check(&self.language, &other.language)
696      && check(&self.mathstyle, &other.mathstyle)

759  /// Compute math bearing (inter-atom spacing) between two boxes.
760  /// Perl: math_bearing
761  pub fn math_bearing(&self, thisbox: &Digested, prevbox: &Digested) -> f64 {
762    let r0 = prevbox
763      .get_property("role")
764      .and_then(|s| {
765        if let Stored::String(sym) = s.into_owned() {
766          Some(arena::with(sym, |s| s.to_string()))
767        } else {
768          None
769        }
770      })
771      .unwrap_or_else(|| "ID".to_string());
772    let r1 = thisbox
773      .get_property("role")
774      .and_then(|s| {
775        if let Stored::String(sym) = s.into_owned() {
776          Some(arena::with(sym, |s| s.to_string()))
777        } else {
778          None
779        }
780      })
781      .unwrap_or_else(|| "ID".to_string());
782    let t0 = *MATH_ATOM_TYPE.get(r0.as_str()).unwrap_or(&0);
783    let t1 = *MATH_ATOM_TYPE.get(r1.as_str()).unwrap_or(&0);
784    let bearing = MATH_BEARINGS[t0][t1];
785    let style = self
786      .get_mathstyle()
787      .map(|s| s.to_string())
788      .unwrap_or_else(|| "text".to_string());
789    if bearing == 0 || (bearing < 0 && style != "display" && style != "text") {
790      return 0.0;
791    }
792    // Look up the bearing register: 1=thinmuskip, 2=medmuskip, 3=thickmuskip
793    let reg_cs = match bearing.unsigned_abs() {
794      1 => T_CS!("\\thinmuskip"),
795      2 => T_CS!("\\medmuskip"),
796      3 => T_CS!("\\thickmuskip"),
797      _ => return 0.0,
798    };
799    if let Ok(Some(def)) = lookup_definition(&reg_cs) {
800      if let Some(val) = def.value_of(Vec::new()) {
801        // Perl: $STATE->lookupDefinition(...)->valueOf->spValue
802        // MuGlue->spValue = fixpoint($skip/UNITY, font->getMUWidth)
803        //                 = kround((skip/UNITY) * MUWidth)
804        // The raw skip is in mu*UNITY units; convert to sp via MUWidth.
805        let skip = val.value_of();
806        let mu_width = self.get_mu_width() as f64;
807        return (skip as f64 / UNITY_F64 * mu_width).trunc();
808      }
809    }
810    0.0
811  }

840  /// Like `merge` but borrows `other` to avoid requiring callers to own or
841  /// clone a Font just to pass into merge. Clones only the retained fields
842  /// from `other` (cheap — Option<Cow<'static,str>> clones are free for
843  /// Borrowed variants, and most Font fields are None in typical uses).
844  pub fn merge_ref(&self, other: &Font) -> Self {
845    // Handle forcebold for compatibility (Perl lines 873-874)
846    let mut series = other.series.clone();
847    let mut force_series = other.forceseries;
848    if other.forcebold == Some(true) {
849      series = Some(Cow::Borrowed("bold"));
850      force_series = Some(true);
851    }
852
853    // Build flags from force options
854    let mut flags: u8 = 0;
855    if other.forcefamily == Some(true) {
856      flags |= FLAG_FORCE_FAMILY;
857    }
858    if force_series == Some(true) {
859      flags |= FLAG_FORCE_SERIES;
860    }
861    if other.forceshape == Some(true) {
862      flags |= FLAG_FORCE_SHAPE;
863    }
864
865    let oflags = self.flags.unwrap_or(0);
866    // Perl: fallback to self if not overridden, or if force-flags on self prevent override
867    let family = if other.family.is_none() || (oflags & FLAG_FORCE_FAMILY != 0) {
868      self.family.clone()
869    } else {
870      other.family.clone()
871    };
872    let series = if series.is_none() || (oflags & FLAG_FORCE_SERIES != 0) {
873      self.series.clone()
874    } else {
875      series
876    };
877    let mut shape = if other.shape.is_none() || (oflags & FLAG_FORCE_SHAPE != 0) {
878      self.shape.clone()
879    } else {
880      other.shape.clone()
881    };
882    let mut size = other.size.or(self.size);
883    let color = other.color.or(self.color);
884    // Perl: $bg = $$self[5] if (!exists $options{background});
885    // Only override bg if `other` actually specifies one
886    let bg = if other.bg.is_some() {
887      other.bg
888    } else {
889      self.bg
890    };
891    let opacity = other.opacity.clone().or_else(|| self.opacity.clone());
892    let encoding = other.encoding.clone().or_else(|| self.encoding.clone());
893    let language = other.language.clone().or_else(|| self.language.clone());
894    let mut mathstyle = other.mathstyle.clone().or_else(|| self.mathstyle.clone());
895    flags |= self.flags.unwrap_or(0);
896
897    // Dynamic adjustment directives
898    if let Some(scale) = other.scale {
899      if let Some(ref mut sz) = size {
900        *sz *= scale;
901      }
902    }
903
904    // Scale factor for mathstyle-based sizing
905    let style_scale = if let Some(sz) = self.size {
906      let key: &str = self.mathstyle.as_deref().unwrap_or("display");
907      sz / *STYLE_SIZE.get(key).unwrap_or(&10) as f64
908    } else {
909      1.0
910    };
911
912    if other.size.is_some() {
913      // Explicitly requested size, use it
914    } else if other.mathstyle.is_some() {
915      // Set the size from mathstyle
916      let ms: &str = mathstyle.as_deref().unwrap_or("display");
917      size = Some(style_scale * *STYLE_SIZE.get(ms).unwrap_or(&10) as f64);
918    } else if other.scripted == Some(true) {
919      // Adjust both the mathstyle & size for scripts
920      let ms: &str = mathstyle.as_deref().unwrap_or("display");
921      mathstyle = SCRIPT_STYLE_MAP.get(ms).map(|c| Cow::Borrowed(*c));
922      let new_ms: &str = mathstyle.as_deref().unwrap_or("display");
923      size = Some(style_scale * *STYLE_SIZE.get(new_ms).unwrap_or(&10) as f64);
924    } else if other.fraction == Some(true) {
925      // Adjust both for fractions
926      let ms: &str = mathstyle.as_deref().unwrap_or("display");
927      mathstyle = FRAC_STYLE_MAP.get(ms).map(|c| Cow::Borrowed(*c));
928      let new_ms: &str = mathstyle.as_deref().unwrap_or("display");
929      size = Some(style_scale * *STYLE_SIZE.get(new_ms).unwrap_or(&10) as f64);
930    }
931
932    // Emphasis handling (Perl lines 909-912)
933    if other.emph == Some(true) {
934      shape = if shape.as_deref() == Some("italic") {
935        Some(Cow::Borrowed("upright"))
936      } else {
937        Some(Cow::Borrowed("italic"))
938      };
939      flags |= FLAG_EMPH;
940    }
941    // Disable emph in math (Perl: $flags &= ~$FLAG_EMPH if $mathstyle)
942    if mathstyle.is_some() {
943      flags &= !FLAG_EMPH;
944    }
945
946    let newfont = Font {
947      family,
948      series,
949      shape,
950      size,
951      color,
952      bg,
953      opacity,
954      encoding,
955      language,
956      mathstyle,
957      flags: Some(flags),
958      // Carry over fields that aren't part of Perl's merge:
959      mathstylestep: other.mathstylestep.or(self.mathstylestep),
960      name: other.name.clone().or_else(|| self.name.clone()),
961      emph: None,
962      scripted: None,
963      fraction: None,
964      forceseries: if flags & FLAG_FORCE_SERIES != 0 {
965        Some(true)
966      } else {
967        None
968      },
969      forcefamily: if flags & FLAG_FORCE_FAMILY != 0 {
970        Some(true)
971      } else {
972        None
973      },
974      forceshape: if flags & FLAG_FORCE_SHAPE != 0 {
975        Some(true)
976      } else {
977        None
978      },
979      forcebold: None,
980      scale: None,
981    };
982    // Note: Perl's merge() has an optional `specialize` option that is passed
983    // explicitly (e.g. merge(specialize => $text)). It's NOT keyed on the font name.
984    // Specialize is called at TBox creation time (tbox.rs) with the actual text content.
985    // Do NOT call specialize here with the font name — it corrupts font properties
986    // (e.g. resetting series "bold" to "medium" for font names like "cmb10").
987    newfont

999    let mut new = self.clone();
1000    if text.is_empty() {

1002    }
1003    let deffamily = if self.forcefamily.unwrap_or(false) {
1004      self.family.clone().unwrap_or_else(|| DEFFAMILY.into())
1005    } else {
1006      DEFFAMILY.into()
1007    };
1008    let defseries = if self.forceseries.unwrap_or(false) {
1009      self.series.clone().unwrap_or_else(|| DEFSERIES.into())
1010    } else {
1011      DEFSERIES.into()
1012    };
1013    let defshape = if self.forceshape.unwrap_or(false) {
1014      self.shape.clone().unwrap_or_else(|| DEFSHAPE.into())
1015    } else {

1017    };

1019      // Latin Letter

1021        new.shape = Some("italic".into());
1022      }
1023    } else if GREEK_LETTER_RE.is_match(text) {
1024      // Single Greek character?
1025      if UPPER_LETTER_RE.is_match(text) {
1026        // Uppercase
1027        if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
1028          new.family = Some(deffamily);
1029          if new.shape.is_some() && (new.shape != Some(DEFSHAPE.into())) {
1030            new.shape = Some(defshape); // if ANY shape, must be default
1031          }
1032        }
1033      } else {
1034        // Lowercase
1035        if new.family.is_none() || (new.family.as_deref() != Some(DEFFAMILY)) {
1036          new.family = Some(deffamily);
1037        }

1039        if new.shape.is_none() || !self.forceshape.unwrap_or(false) {
1040          new.shape = Some("italic".into());
1041        }
1042        if new.series.is_some() && (new.series != Some(DEFSERIES.into())) {
1043          new.series = Some(defseries);
1044        }
1045      }
1046    } else if DIGIT_LETTER_RE.is_match(text) {
1047      // Digit
1048      if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
1049        new.family = Some(deffamily);
1050        new.shape = Some(defshape); // defaults, always.
1051      }
1052    } else {

1054      new.family = Some(deffamily);
1055      new.shape = Some(defshape); // defaults, always.
1056      if new.series.is_some() && (new.series.as_ref().unwrap() != DEFSERIES) {
1057        new.series = Some(defseries);
1058      } // defaults, always.
1059    }

1063  pub fn distance(&self, other: &Font) -> i8 {
1064    let mut distance: i8 = 0;
1065    // Normalize "math" → "serif" for comparison (Perl lines 436-437)
1066    let fam = self
1067      .family
1068      .as_deref()
1069      .map(|f| if f == "math" { "serif" } else { f });
1070    let ofam = other
1071      .family
1072      .as_deref()
1073      .map(|f| if f == "math" { "serif" } else { f });
1074    if is_diff_opt_str(fam, ofam) {
1075      distance += 1;
1076    }
1077    if is_diff_opt_str(self.series.as_deref(), other.series.as_deref()) {
1078      distance += 1;
1079    }
1080    if is_diff_opt_str(self.shape.as_deref(), other.shape.as_deref()) {
1081      distance += 1;
1082    }
1083    if is_diff_f64(self.size, other.size) {
1084      distance += 1;
1085    }
1086    // Color: use reference-style comparison (different Color variant = different).
1087    // Perl's isDiff uses object reference equality: Cmyk(0,0,0,1) ≠ Rgb(0,0,0)
1088    // even though both are visually black.
1089    if is_diff_font_color_ref(self.color.as_ref(), other.color.as_ref()) {
1090      distance += 1;
1091    }
1092    if is_diff_color(self.bg.as_ref(), other.bg.as_ref()) {
1093      distance += 1;
1094    }
1095    if is_diff_opt_str(self.opacity.as_deref(), other.opacity.as_deref()) {
1096      distance += 1;
1097    }
1098    // Perl does NOT count encoding differences
1099    // Perl does NOT count mathstyle differences
1100    if is_diff_opt_str(self.language.as_deref(), other.language.as_deref()) {
1101      distance += 1;
1102    }
1103    // Perl: ($flags & $FLAG_EMPH) ^ ($oflags & $FLAG_EMPH) ? 1 : 0
1104    let flags = self.flags.unwrap_or(0);
1105    let oflags = other.flags.unwrap_or(0);
1106    if (flags & FLAG_EMPH) ^ (oflags & FLAG_EMPH) != 0 {
1107      distance += 1;
1108    }
1109    distance
1110  }

1138  /// font properties into fewer attributes (font,fontsize,color,background,opacity),
1139  /// the return value encodes both the attribute changes that would be needed to effect
1140  /// the font change, along with the font properties that differed
1141  /// Namely, the result is a hash keyed on the attribute name and whose value is a FontDiff
1142  ///    value      => "new_attribute_value"
1143  ///    properties => { %fontproperties }
1144  /// or (String, Font)
1145  pub fn relative_to(&self, other: &Font) -> HashMap<String, (String, Font)> {
1146    let family = match self.family {
1147      Some(ref fam) => {
1148        if fam == "math" {
1149          Some(Cow::Borrowed("serif"))
1150        } else {
1151          Some(fam.clone())
1152        }
1153      },
1154      None => None,
1155    };
1156    let other_family = match other.family {
1157      Some(ref fam) => {
1158        if fam == "math" {
1159          Some(Cow::Borrowed("serif"))
1160        } else {
1161          Some(fam.clone())
1162        }
1163      },
1164      None => None,
1165    };
1166    let mut diffs = vec![];
1167    let mut font_properties = Font::default();
1168    if is_diff(family.as_ref(), other_family.as_ref()) {
1169      diffs.push(family.clone().unwrap());
1170      font_properties.family = family;
1171    }
1172    if is_diff(self.series.as_ref(), other.series.as_ref()) {
1173      let series = self.series.clone().unwrap();
1174      diffs.push(series);
1175      font_properties.series.clone_from(&self.series);
1176    }
1177    if is_diff(self.shape.as_ref(), other.shape.as_ref()) {
1178      let shape = self.shape.clone().unwrap();
1179      diffs.push(shape);
1180      font_properties.shape.clone_from(&self.shape);
1181    }
1182    let mut result = HashMap::default();
1183
1184    if !diffs.is_empty() {
1185      let font_value = diffs.join(" ");
1186      result.insert(s!("font"), (font_value, font_properties));
1187    }
1188
1189    if is_diff_f64(self.size.as_ref().copied(), other.size.as_ref().copied()) {
1190      result.insert(
1191        "fontsize".to_string(),
1192        (
1193          relative_font_size(self.size.unwrap(), other.size.unwrap()),
1194          Font {
1195            size: self.size,
1196            ..Font::default()
1197          },
1198        ),
1199      );
1200    }
1201    // Emit color when Color variants differ (reference-style comparison).
1202    // Perl's `ne` treats Cmyk(0,0,0,1) ≠ Rgb(0,0,0) even though both are black.
1203    if is_diff_font_color_ref(self.color.as_ref(), other.color.as_ref()) {
1204      let effective_color = self.color.unwrap_or(DEFCOLOR);
1205      result.insert(
1206        "color".to_string(),
1207        (effective_color.to_attribute(), Font {
1208          color: Some(effective_color),
1209          ..Font::default()
1210        }),
1211      );
1212    }
1213    if is_diff_color(self.bg.as_ref(), other.bg.as_ref()) {
1214      result.insert(
1215        "backgroundcolor".to_string(),
1216        (self.bg.as_ref().unwrap().to_attribute(), Font {
1217          bg: self.bg,
1218          ..Font::default()
1219        }),
1220      );
1221    }
1222    if is_diff(self.opacity.as_ref(), other.opacity.as_ref()) {
1223      result.insert(
1224        "opacity".to_string(),
1225        (self.opacity.as_ref().unwrap().to_string(), Font {
1226          opacity: self.opacity.clone(),
1227          ..Font::default()
1228        }),
1229      );
1230    }
1231    if is_diff(self.encoding.as_ref(), other.encoding.as_ref()) {
1232      result.insert(
1233        "encoding".to_string(),
1234        (self.encoding.as_ref().unwrap().to_string(), Font {
1235          encoding: self.encoding.clone(),
1236          ..Font::default()
1237        }),
1238      );
1239    }
1240    if is_diff(self.language.as_ref(), other.language.as_ref()) {
1241      result.insert(
1242        "xml:lang".to_string(),
1243        (self.language.as_ref().unwrap().to_string(), Font {
1244          language: self.language.clone(),
1245          ..Font::default()
1246        }),
1247      );
1248    }
1249    // Emph: (!$mstyle && $flags && ($flags & $FLAG_EMPH) && (!$oflags || !($oflags & $FLAG_EMPH))
1250    let flags = self.flags.unwrap_or(0);
1251    let oflags = other.flags.unwrap_or(0);
1252    if self.mathstyle.is_none()
1253      && flags != 0
1254      && (flags & FLAG_EMPH) != 0
1255      && (oflags == 0 || (oflags & FLAG_EMPH) == 0)
1256    {
1257      result.insert(
1258        "element".to_string(),
1259        ("ltx:emph".to_string(), Font::default()),
1260      );
1261    }
1262    // We do NOT want mathstyle showing up automatically in the attributes
1263    result
1264  }

1290  /// Find a Font Metric corresponding to this font's family_series_shape_size
1291  /// that contains the given `char`, if given.
1292  /// Try to find a fallback metric if `char` is not in the current Font.
1293  /// Perl: getMetric
1294  pub fn get_metric(&self, c_opt: Option<char>) -> &MetricData {
1295    let family = self.family.as_deref().unwrap_or("serif");
1296    let series = self.series.as_deref().unwrap_or("medium");
1297    let shape = self.shape.as_deref().unwrap_or("upright");
1298    let size = self.size.unwrap_or_else(defsize) as i64;
1299    // Stack buffer for char→&str lookup key, reused across paths. Avoids
1300    // one String allocation per character per get_metric call (which is
1301    // called per-character inside compute_string_size).
1302    let mut ch_buf = [0u8; 4];
1303    let ch_key = c_opt.map(|c| c.encode_utf8(&mut ch_buf) as &str);
1304    if let Some(name) = lookup_metric_name(family, series, shape) {
1305      let fullname = format!("{name}{size}");
1306      if let Some(metric) = STDMETRICS.get(fullname.as_str()) {
1307        if ch_key.is_none_or(|k| metric.sizes.contains_key(k)) {
1308          return metric;
1309        }
1310      }
1311      // Try base name fallback
1312      let metric = get_metric_for_name(name);
1313      if ch_key.is_none_or(|k| metric.sizes.contains_key(k)) {
1314        return metric;
1315      }
1316    }
1317    // Look for a fallback metric if char given
1318    if let Some(k) = ch_key {
1319      for name in METRIC_FALLBACKS {
1320        let fullname = format!("{name}{size}");
1321        let metric = STDMETRICS
1322          .get(fullname.as_str())
1323          .unwrap_or_else(|| get_metric_for_name(name));
1324        if metric.sizes.contains_key(k) {
1325          return metric;
1326        }
1327      }
1328    }
1329    get_metric_for_name("cmr")
1330  }

1348  pub fn compute_string_size(
1349    &self,
1350    text: &str,
1351    _options: SymHashMap<Stored>,
1352  ) -> (Dimension, Dimension, Dimension) {
1353    if text.is_empty()
1354      || self
1355        .get_family()
1356        .map(|fam| fam == "nullfont")
1357        .unwrap_or(false)
1358    {
1359      return (
1360        Dimension::default(),
1361        Dimension::default(),
1362        Dimension::default(),
1363      );
1364    }
1365    let size = self.get_size().unwrap_or_else(defsize);
1366    let ismath = self.get_family().map(|fam| fam == "math").unwrap_or(false);
1367    let (mut w, mut h, mut d) = (0, 0, 0);
1368    // Iterate via Peekable — no intermediate Vec<char> allocation,
1369    // and we get O(1) lookahead for kerning between consecutive chars.
1370    let mut chars_iter = text.chars().peekable();
1371    // Stack buffers for char→&str and char+char→&str lookups, avoiding
1372    // String::to_string() + String::format!() heap allocations inside
1373    // the per-character hot loop. encode_utf8 writes directly to the
1374    // buffer and returns a borrowed &str slice — no allocation.
1375    let mut ch_buf = [0u8; 4];
1376    let mut kern_buf = [0u8; 8];
1377    while let Some(ch) = chars_iter.next() {
1378      let metric = self.get_metric(Some(ch));
1379      let ch_key = ch.encode_utf8(&mut ch_buf);
1380      let entry_opt = metric.sizes.get(ch_key);
1381      let (cw, ch_sz, cd, ci) = if let Some(entry) = entry_opt {
1382        *entry
1383      } else {
1384        (0.75 * UNITY_F64, 0.7 * UNITY_F64, 0.2 * UNITY_F64, 0.0)
1385      };
1386      w += (cw * size).trunc() as i64;
1387      // Kerning: check kern between this char and next.
1388      if let Some(&next_ch) = chars_iter.peek() {
1389        let first_len = ch.encode_utf8(&mut kern_buf).len();
1390        let second_len = next_ch.encode_utf8(&mut kern_buf[first_len..]).len();
1391        let kern_key = std::str::from_utf8(&kern_buf[..first_len + second_len]).unwrap();
1392        if let Some(kern) = metric.kerns.get(kern_key) {
1393          w += (size * kern).trunc() as i64;
1394        }
1395      }
1396      // Italic correction in math
1397      if ismath && ci != 0.0 {
1398        w += (size * ci).trunc() as i64;
1399      }
1400      h = max(h, (ch_sz * size).trunc() as i64);
1401      d = max(d, (cd * size).trunc() as i64);
1402    }
1403    // The 1 is so that any actual glyph appears to be non-empty.
1404    // This is presumably only necessary to deal with the flawed emptiness heiristics in Alignment?
1405    if w == 0 {
1406      w = 1;
1407    }
1408    (Dimension::new(w), Dimension::new(h), Dimension::new(d))
1409  }

1498            .map(|v| v.to_string())
1499            .unwrap_or_default();
1500          if sub_mode == "horizontal" {
1501            let sub_boxes: Vec<Digested> = bx.unlist();

1504              .get_property("width")

1506                Stored::Dimension(d) => Some(d.value_of()),

1842/// with TeX's rearrangement of ASCII...

1844fn push_fontmap_char(result: &mut String, c: char, _code: u8) {

1854  }
