344/// Find a Font Metric for a given fontname, fallback to 10pt or cmr as needed.
345/// Perl: getMetricForName
346pub fn get_metric_for_name(name: &str) -> &'static MetricData {
347  let base = if let Some(idx) = name.find(|c: char| c.is_ascii_digit()) {
348    &name[..idx]
349  } else {
350    name
351  };
352  // Try exact name first (e.g. "cmr10")
353  if let Some(m) = STDMETRICS.get(name) {
354    return m;
355  }
356  // Try base without size (e.g. "cmr" from "cmr10")
357  if let Some(m) = STDMETRICS.get(base) {
358    return m;
359  }
360  // Try base + "10". Stack-buffer concat avoids the per-call `format!`
361  // heap alloc — `get_metric_for_name` is reached through the
362  // per-character `get_metric` loop, so this is a real allocation
363  // site. Font basenames are short (≤ ~20 ASCII bytes); 32 bytes is
364  // ample padding.
365  let base_bytes = base.as_bytes();
366  if base_bytes.len() + 2 <= 32 {
367    let mut buf = [0u8; 32];
368    buf[..base_bytes.len()].copy_from_slice(base_bytes);
369    buf[base_bytes.len()] = b'1';
370    buf[base_bytes.len() + 1] = b'0';
371    if let Ok(s) = std::str::from_utf8(&buf[..base_bytes.len() + 2])
372      && let Some(m) = STDMETRICS.get(s)
373    {
374      return m;
375    }
376  }
377  // Ultimate fallback to "cmr"
378  STDMETRICS
379    .get("cmr")
380    .expect("STDMETRICS must contain 'cmr'")
381}

383pub fn decode_fontname(name: &str, at_opt: Option<f64>, scaled_opt: Option<f64>) -> Option<Font> {
384  if let Some(cap) = FONT_RE.captures(name) {
385    // Perl: my %props = (series => 'medium', shape => 'upright', encoding => 'OT1');
386    let mut props = Font {
387      series: Some(Cow::Borrowed(DEFSERIES)),
388      shape: Some(Cow::Borrowed(DEFSHAPE)),
389      encoding: Some(Cow::Borrowed("OT1")),
390      ..Font::default()
391    };
392    let fam = cap.get(1).map_or("", |m| m.as_str());
393    let ser = cap.get(2).map_or("", |m| m.as_str());
394    let shp = cap.get(3).map_or("", |m| m.as_str());
395    let size_str = cap.get(4).map_or("", |m| m.as_str());
396    if let Some(ffam) = lookup_font_family(fam) {
397      props = props.merge_ref(ffam);
398    }
399    if let Some(fser) = lookup_font_series(ser) {
400      props = props.merge_ref(fser);
401    }
402    if let Some(fsh) = lookup_font_shape(shp) {
403      props = props.merge_ref(fsh);
404    }
405    let mut size = if let Some(at) = at_opt {
406      at
407    } else {
408      let size_f64 = size_str.parse::<f64>().unwrap_or(1.0);
409      if size_f64 == 0.0 { 1.0 } else { size_f64 } // Yes, also if 0, "" (from regexp)
410    };
411    if let Some(scaled) = scaled_opt {

795  /// Compute math bearing (inter-atom spacing) between two boxes.
796  /// Perl: math_bearing
797  pub fn math_bearing(&self, thisbox: &Digested, prevbox: &Digested) -> f64 {
798    let r0 = prevbox
799      .get_property("role")
800      .and_then(|s| match s.into_owned() {
801        Stored::String(sym) => Some(arena::with(sym, |s| s.to_string())),
802        _ => None,
803      })
804      .unwrap_or_else(|| "ID".to_string());
805    let r1 = thisbox
806      .get_property("role")
807      .and_then(|s| match s.into_owned() {
808        Stored::String(sym) => Some(arena::with(sym, |s| s.to_string())),
809        _ => None,
810      })
811      .unwrap_or_else(|| "ID".to_string());
812    let t0 = *MATH_ATOM_TYPE.get(r0.as_str()).unwrap_or(&0);
813    let t1 = *MATH_ATOM_TYPE.get(r1.as_str()).unwrap_or(&0);
814    let bearing = MATH_BEARINGS[t0][t1];
815    let style = self
816      .get_mathstyle()
817      .map(|s| s.to_string())
818      .unwrap_or_else(|| "text".to_string());
819    if bearing == 0 || (bearing < 0 && style != "display" && style != "text") {
820      return 0.0;
821    }
822    // Look up the bearing register: 1=thinmuskip, 2=medmuskip, 3=thickmuskip
823    let reg_cs = match bearing.unsigned_abs() {
824      1 => T_CS!("\\thinmuskip"),
825      2 => T_CS!("\\medmuskip"),
826      3 => T_CS!("\\thickmuskip"),
827      _ => return 0.0,
828    };
829    if let Ok(Some(def)) = lookup_definition(&reg_cs)
830      && let Some(val) = def.value_of(Vec::new())
831    {
832      // Perl: $STATE->lookupDefinition(...)->valueOf->spValue
833      // MuGlue->spValue = fixpoint($skip/UNITY, font->getMUWidth)
834      //                 = kround((skip/UNITY) * MUWidth)
835      // The raw skip is in mu*UNITY units; convert to sp via MUWidth.
836      let skip = val.value_of();
837      let mu_width = self.get_mu_width() as f64;
838      return (skip as f64 / UNITY_F64 * mu_width).trunc();
839    }
840    0.0
841  }

870  /// Like `merge` but borrows `other` to avoid requiring callers to own or
871  /// clone a Font just to pass into merge. Clones only the retained fields
872  /// from `other` (cheap — Option<Cow<'static,str>> clones are free for
873  /// Borrowed variants, and most Font fields are None in typical uses).
874  pub fn merge_ref(&self, other: &Font) -> Self {
875    // Handle forcebold for compatibility (Perl lines 873-874)
876    let mut series = other.series.clone();
877    let mut force_series = other.forceseries;
878    if other.forcebold == Some(true) {
879      series = Some(Cow::Borrowed("bold"));
880      force_series = Some(true);
881    }
882
883    // Build flags from force options
884    let mut flags: u8 = 0;
885    if other.forcefamily == Some(true) {
886      flags |= FLAG_FORCE_FAMILY;
887    }
888    if force_series == Some(true) {
889      flags |= FLAG_FORCE_SERIES;
890    }
891    if other.forceshape == Some(true) {
892      flags |= FLAG_FORCE_SHAPE;
893    }
894
895    let oflags = self.flags.unwrap_or(0);
896    // Perl: fallback to self if not overridden, or if force-flags on self prevent override
897    let family = if other.family.is_none() || (oflags & FLAG_FORCE_FAMILY != 0) {
898      self.family.clone()
899    } else {
900      other.family.clone()
901    };
902    let series = if series.is_none() || (oflags & FLAG_FORCE_SERIES != 0) {
903      self.series.clone()
904    } else {
905      series
906    };
907    let mut shape = if other.shape.is_none() || (oflags & FLAG_FORCE_SHAPE != 0) {
908      self.shape.clone()
909    } else {
910      other.shape.clone()
911    };
912    let mut size = other.size.or(self.size);
913    let color = other.color.or(self.color);
914    // Perl: $bg = $$self[5] if (!exists $options{background});
915    // Only override bg if `other` actually specifies one
916    let bg = if other.bg.is_some() {
917      other.bg
918    } else {
919      self.bg
920    };
921    let opacity = other.opacity.clone().or_else(|| self.opacity.clone());
922    let encoding = other.encoding.clone().or_else(|| self.encoding.clone());
923    let language = other.language.clone().or_else(|| self.language.clone());
924    let mut mathstyle = other.mathstyle.clone().or_else(|| self.mathstyle.clone());
925    flags |= self.flags.unwrap_or(0);
926
927    // Dynamic adjustment directives
928    if let Some(scale) = other.scale
929      && let Some(ref mut sz) = size
930    {
931      *sz *= scale;
932    }
933
934    // Scale factor for mathstyle-based sizing
935    let style_scale = if let Some(sz) = self.size {
936      let key: &str = self.mathstyle.as_deref().unwrap_or("display");
937      sz / *STYLE_SIZE.get(key).unwrap_or(&10) as f64
938    } else {
939      1.0
940    };
941
942    if other.size.is_some() {
943      // Explicitly requested size, use it
944    } else if other.mathstyle.is_some() {
945      // Set the size from mathstyle
946      let ms: &str = mathstyle.as_deref().unwrap_or("display");
947      size = Some(style_scale * *STYLE_SIZE.get(ms).unwrap_or(&10) as f64);
948    } else if other.scripted == Some(true) {
949      // Adjust both the mathstyle & size for scripts
950      let ms: &str = mathstyle.as_deref().unwrap_or("display");
951      mathstyle = SCRIPT_STYLE_MAP.get(ms).map(|c| Cow::Borrowed(*c));
952      let new_ms: &str = mathstyle.as_deref().unwrap_or("display");
953      size = Some(style_scale * *STYLE_SIZE.get(new_ms).unwrap_or(&10) as f64);
954    } else if other.fraction == Some(true) {
955      // Adjust both for fractions
956      let ms: &str = mathstyle.as_deref().unwrap_or("display");
957      mathstyle = FRAC_STYLE_MAP.get(ms).map(|c| Cow::Borrowed(*c));
958      let new_ms: &str = mathstyle.as_deref().unwrap_or("display");
959      size = Some(style_scale * *STYLE_SIZE.get(new_ms).unwrap_or(&10) as f64);
960    }
961
962    // Emphasis handling (Perl lines 909-912)
963    if other.emph == Some(true) {
964      shape = if shape.as_deref() == Some("italic") {
965        Some(Cow::Borrowed("upright"))
966      } else {
967        Some(Cow::Borrowed("italic"))
968      };
969      flags |= FLAG_EMPH;
970    }

1020  /// Instanciate the font for a particular class of symbols.
1021  /// NOTE: This works in `normal' latex, but probably needs some tunability.
1022  /// Depending on the fonts being used, the allowable combinations may be different.
1023  /// Getting the font right is important, since the author probably
1024  /// thinks of the identity of the symbols according to what they SEE in the printed
1025  /// document.  Even though the markup might seem to indicate something else...
1026  ///
1027  /// Use Unicode properties to determine font merging.
1028  pub fn specialize(&self, text: &str) -> Self {
1029    let mut new = self.clone();
1030    if text.is_empty() {
1031      return new; // ?
1032    }
1033    let deffamily = if self.forcefamily.unwrap_or(false) {
1034      self.family.clone().unwrap_or_else(|| DEFFAMILY.into())
1035    } else {
1036      DEFFAMILY.into()
1037    };
1038    let defseries = if self.forceseries.unwrap_or(false) {
1039      self.series.clone().unwrap_or_else(|| DEFSERIES.into())
1040    } else {
1041      DEFSERIES.into()
1042    };
1043    let defshape = if self.forceshape.unwrap_or(false) {
1044      self.shape.clone().unwrap_or_else(|| DEFSHAPE.into())
1045    } else {
1046      DEFSHAPE.into()
1047    };
1048    if LATIN_LETTER_RE.is_match(text) {
1049      // Latin Letter
1050      if new.shape.is_none() && new.family.is_none() {
1051        new.shape = Some("italic".into());
1052      }
1053    } else if GREEK_LETTER_RE.is_match(text) {
1054      // Single Greek character?
1055      if UPPER_LETTER_RE.is_match(text) {
1056        // Uppercase
1057        if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
1058          new.family = Some(deffamily);
1059          if new.shape.is_some() && (new.shape != Some(DEFSHAPE.into())) {
1060            new.shape = Some(defshape); // if ANY shape, must be default
1061          }
1062        }
1063      } else {
1064        // Lowercase
1065        if new.family.is_none() || (new.family.as_deref() != Some(DEFFAMILY)) {
1066          new.family = Some(deffamily);
1067        }
1068        // Perl: $shape = 'italic' if !$shape || !($flags & $FLAG_FORCE_SHAPE);
1069        if new.shape.is_none() || !self.forceshape.unwrap_or(false) {
1070          new.shape = Some("italic".into());
1071        }
1072        if new.series.is_some() && (new.series != Some(DEFSERIES.into())) {
1073          new.series = Some(defseries);
1074        }
1075      }
1076    } else if DIGIT_LETTER_RE.is_match(text) {
1077      // Digit
1078      if new.family.is_none() || (new.family.as_ref().unwrap() == "math") {
1079        new.family = Some(deffamily);
1080        new.shape = Some(defshape); // defaults, always.
1081      }
1082    } else {
1083      // Other Symbol
1084      new.family = Some(deffamily);
1085      new.shape = Some(defshape); // defaults, always.
1086      if new.series.is_some() && (new.series.as_ref().unwrap() != DEFSERIES) {
1087        new.series = Some(defseries);
1088      } // defaults, always.
1089    }
1090    new
1091  }

1093  pub fn distance(&self, other: &Font) -> i8 {
1094    let mut distance: i8 = 0;
1095    // Normalize "math" → "serif" for comparison (Perl lines 436-437)
1096    let fam = self
1097      .family
1098      .as_deref()
1099      .map(|f| if f == "math" { "serif" } else { f });
1100    let ofam = other
1101      .family
1102      .as_deref()
1103      .map(|f| if f == "math" { "serif" } else { f });
1104    if is_diff_opt_str(fam, ofam) {
1105      distance += 1;
1106    }
1107    if is_diff_opt_str(self.series.as_deref(), other.series.as_deref()) {
1108      distance += 1;
1109    }
1110    if is_diff_opt_str(self.shape.as_deref(), other.shape.as_deref()) {
1111      distance += 1;
1112    }
1113    if is_diff_f64(self.size, other.size) {
1114      distance += 1;
1115    }
1116    // Color: use reference-style comparison (different Color variant = different).
1117    // Perl's isDiff uses object reference equality: Cmyk(0,0,0,1) ≠ Rgb(0,0,0)
1118    // even though both are visually black.
1119    if is_diff_font_color_ref(self.color.as_ref(), other.color.as_ref()) {
1120      distance += 1;
1121    }
1122    if is_diff_color(self.bg.as_ref(), other.bg.as_ref()) {
1123      distance += 1;
1124    }
1125    if is_diff_opt_str(self.opacity.as_deref(), other.opacity.as_deref()) {
1126      distance += 1;
1127    }
1128    // Perl does NOT count encoding differences
1129    // Perl does NOT count mathstyle differences
1130    if is_diff_opt_str(self.language.as_deref(), other.language.as_deref()) {
1131      distance += 1;
1132    }
1133    // Perl: ($flags & $FLAG_EMPH) ^ ($oflags & $FLAG_EMPH) ? 1 : 0

1135    let oflags = other.flags.unwrap_or(0);
1136    if (flags & FLAG_EMPH) ^ (oflags & FLAG_EMPH) != 0 {
1137      distance += 1;
1138    }
1139    distance
1140  }

1167  /// Noting that the font-related attributes in the schema distill the
1168  /// font properties into fewer attributes (font,fontsize,color,background,opacity),
1169  /// the return value encodes both the attribute changes that would be needed to effect
1170  /// the font change, along with the font properties that differed
1171  /// Namely, the result is a hash keyed on the attribute name and whose value is a FontDiff
1172  ///    value      => "new_attribute_value"
1173  ///    properties => { %fontproperties }
1174  /// or (String, Font)
1175  pub fn relative_to(&self, other: &Font) -> HashMap<String, (String, Font)> {
1176    let family = match self.family {
1177      Some(ref fam) => {
1178        if fam == "math" {
1179          Some(Cow::Borrowed("serif"))
1180        } else {
1181          Some(fam.clone())
1182        }
1183      },
1184      None => None,
1185    };
1186    let other_family = match other.family {
1187      Some(ref fam) => {
1188        if fam == "math" {
1189          Some(Cow::Borrowed("serif"))
1190        } else {
1191          Some(fam.clone())
1192        }
1193      },
1194      None => None,
1195    };
1196    let mut diffs = vec![];
1197    let mut font_properties = Font::default();
1198    if is_diff(family.as_ref(), other_family.as_ref()) {
1199      diffs.push(family.clone().unwrap());
1200      font_properties.family = family;
1201    }
1202    if is_diff(self.series.as_ref(), other.series.as_ref()) {
1203      let series = self.series.clone().unwrap();
1204      diffs.push(series);
1205      font_properties.series.clone_from(&self.series);
1206    }
1207    if is_diff(self.shape.as_ref(), other.shape.as_ref()) {
1208      let shape = self.shape.clone().unwrap();
1209      diffs.push(shape);
1210      font_properties.shape.clone_from(&self.shape);
1211    }
1212    let mut result = HashMap::default();
1213
1214    if !diffs.is_empty() {
1215      let font_value = diffs.join(" ");
1216      result.insert(s!("font"), (font_value, font_properties));
1217    }
1218
1219    if is_diff_f64(self.size.as_ref().copied(), other.size.as_ref().copied()) {
1220      result.insert(
1221        "fontsize".to_string(),
1222        (
1223          relative_font_size(self.size.unwrap(), other.size.unwrap()),
1224          Font {
1225            size: self.size,
1226            ..Font::default()
1227          },
1228        ),
1229      );
1230    }
1231    // Emit color when Color variants differ (reference-style comparison).
1232    // Perl's `ne` treats Cmyk(0,0,0,1) ≠ Rgb(0,0,0) even though both are black.
1233    if is_diff_font_color_ref(self.color.as_ref(), other.color.as_ref()) {
1234      let effective_color = self.color.unwrap_or(DEFCOLOR);
1235      result.insert(
1236        "color".to_string(),
1237        (effective_color.to_attribute(), Font {
1238          color: Some(effective_color),
1239          ..Font::default()
1240        }),
1241      );
1242    }
1243    if is_diff_color(self.bg.as_ref(), other.bg.as_ref()) {
1244      result.insert(
1245        "backgroundcolor".to_string(),
1246        (self.bg.as_ref().unwrap().to_attribute(), Font {
1247          bg: self.bg,
1248          ..Font::default()
1249        }),
1250      );
1251    }
1252    if is_diff(self.opacity.as_ref(), other.opacity.as_ref()) {
1253      result.insert(
1254        "opacity".to_string(),
1255        (self.opacity.as_ref().unwrap().to_string(), Font {
1256          opacity: self.opacity.clone(),
1257          ..Font::default()
1258        }),
1259      );
1260    }
1261    if is_diff(self.encoding.as_ref(), other.encoding.as_ref()) {
1262      result.insert(
1263        "encoding".to_string(),
1264        (self.encoding.as_ref().unwrap().to_string(), Font {
1265          encoding: self.encoding.clone(),
1266          ..Font::default()
1267        }),
1268      );
1269    }
1270    if is_diff(self.language.as_ref(), other.language.as_ref()) {
1271      result.insert(
1272        "xml:lang".to_string(),
1273        (self.language.as_ref().unwrap().to_string(), Font {
1274          language: self.language.clone(),
1275          ..Font::default()
1276        }),
1277      );
1278    }
1279    // Emph: (!$mstyle && $flags && ($flags & $FLAG_EMPH) && (!$oflags || !($oflags & $FLAG_EMPH))
1280    let flags = self.flags.unwrap_or(0);
1281    let oflags = other.flags.unwrap_or(0);
1282    if self.mathstyle.is_none()
1283      && flags != 0
1284      && (flags & FLAG_EMPH) != 0
1285      && (oflags == 0 || (oflags & FLAG_EMPH) == 0)
1286    {
1287      result.insert(
1288        "element".to_string(),
1289        ("ltx:emph".to_string(), Font::default()),
1290      );
1291    }
1292    // We do NOT want mathstyle showing up automatically in the attributes
1293    result
1294  }

1320  /// Find a Font Metric corresponding to this font's family_series_shape_size
1321  /// that contains the given `char`, if given.
1322  /// Try to find a fallback metric if `char` is not in the current Font.
1323  /// Perl: getMetric
1324  pub fn get_metric(&self, c_opt: Option<char>) -> &MetricData {
1325    let family = self.family.as_deref().unwrap_or("serif");
1326    let series = self.series.as_deref().unwrap_or("medium");
1327    let shape = self.shape.as_deref().unwrap_or("upright");
1328    let size = self.size.unwrap_or_else(defsize) as i64;
1329    // Stack buffer for char→&str lookup key, reused across paths. Avoids
1330    // one String allocation per character per get_metric call (which is
1331    // called per-character inside compute_string_size).
1332    let mut ch_buf = [0u8; 4];
1333    let ch_key = c_opt.map(|c| c.encode_utf8(&mut ch_buf) as &str);
1334    if let Some(name) = lookup_metric_name(family, series, shape) {
1335      let fullname = format!("{name}{size}");
1336      if let Some(metric) = STDMETRICS.get(fullname.as_str())
1337        && ch_key.is_none_or(|k| metric.sizes.contains_key(k))
1338      {
1339        return metric;
1340      }
1341      // Try base name fallback
1342      let metric = get_metric_for_name(name);
1343      if ch_key.is_none_or(|k| metric.sizes.contains_key(k)) {
1344        return metric;
1345      }
1346    }
1347    // Look for a fallback metric if char given
1348    if let Some(k) = ch_key {
1349      for name in METRIC_FALLBACKS {
1350        let fullname = format!("{name}{size}");
1351        let metric = STDMETRICS
1352          .get(fullname.as_str())
1353          .unwrap_or_else(|| get_metric_for_name(name));
1354        if metric.sizes.contains_key(k) {
1355          return metric;
1356        }
1357      }
1358    }
1359    get_metric_for_name("cmr")
1360  }

1378  pub fn compute_string_size(
1379    &self,
1380    text: &str,
1381    _options: SymHashMap<Stored>,
1382  ) -> (Dimension, Dimension, Dimension) {
1383    if text.is_empty()
1384      || self
1385        .get_family()
1386        .map(|fam| fam == "nullfont")
1387        .unwrap_or(false)
1388    {
1389      return (
1390        Dimension::default(),
1391        Dimension::default(),
1392        Dimension::default(),
1393      );
1394    }
1395    let size = self.get_size().unwrap_or_else(defsize);
1396    let ismath = self.get_family().map(|fam| fam == "math").unwrap_or(false);
1397    let (mut w, mut h, mut d) = (0, 0, 0);
1398    // Iterate via Peekable — no intermediate Vec<char> allocation,
1399    // and we get O(1) lookahead for kerning between consecutive chars.
1400    let mut chars_iter = text.chars().peekable();
1401    // Stack buffers for char→&str and char+char→&str lookups, avoiding
1402    // String::to_string() + String::format!() heap allocations inside
1403    // the per-character hot loop. encode_utf8 writes directly to the
1404    // buffer and returns a borrowed &str slice — no allocation.
1405    let mut ch_buf = [0u8; 4];
1406    let mut kern_buf = [0u8; 8];
1407    while let Some(ch) = chars_iter.next() {
1408      let metric = self.get_metric(Some(ch));
1409      let ch_key = ch.encode_utf8(&mut ch_buf);
1410      let entry_opt = metric.sizes.get(ch_key);
1411      let (cw, ch_sz, cd, ci) = if let Some(entry) = entry_opt {
1412        *entry
1413      } else {
1414        (0.75 * UNITY_F64, 0.7 * UNITY_F64, 0.2 * UNITY_F64, 0.0)
1415      };
1416      w += (cw * size).trunc() as i64;
1417      // Kerning: check kern between this char and next.
1418      if let Some(&next_ch) = chars_iter.peek() {
1419        let first_len = ch.encode_utf8(&mut kern_buf).len();
1420        let second_len = next_ch.encode_utf8(&mut kern_buf[first_len..]).len();
1421        let kern_key = std::str::from_utf8(&kern_buf[..first_len + second_len]).unwrap();
1422        if let Some(kern) = metric.kerns.get(kern_key) {
1423          w += (size * kern).trunc() as i64;
1424        }
1425      }
1426      // Italic correction in math
1427      if ismath && ci != 0.0 {
1428        w += (size * ci).trunc() as i64;
1429      }
1430      h = max(h, (ch_sz * size).trunc() as i64);
1431      d = max(d, (cd * size).trunc() as i64);
1432    }
1433    // The 1 is so that any actual glyph appears to be non-empty.
1434    // This is presumably only necessary to deal with the flawed emptiness heiristics in Alignment?
1435    if w == 0 {
1436      w = 1;
1437    }
1438    (Dimension::new(w), Dimension::new(h), Dimension::new(d))
1439  }

1506    // horizontal List without a width property is restricted_horizontal (a single
1507    // line, no wrapping), matching Perl.

1837  /// `wd` is the max, inter-line spacing uses each line's `baseline` (`bs < 0` =
1838  /// no adjustment), and `ht`/`dp` are split per `vattach` (`mathaxis` = size/4).
1839  fn compute_boxes_size_stack(vattach: &str, mathaxis: i64, lines: &[[i64; 4]]) -> (i64, i64, i64) {
1840    let nlines = lines.len();
1841    if nlines == 0 {
1842      return (0, 0, 0);
1843    }
1844    if nlines == 1 {
1845      let [_bs, w, h, d] = lines[0];
1846      return (w, h, d);
1847    }
1848    // Perl: $lineskip = lookupDefinition('\lineskip')->valueOf->valueOf
1849    let lineskip = lookup_definition(&T_CS!("\\lineskip"))
1850      .ok()
1851      .flatten()
1852      .and_then(|def| def.value_of(Vec::new()))
1853      .map(|v| v.value_of())
1854      .unwrap_or(0);
1855    let mut wd: i64 = 0;
1856    let mut prevdepth: i64 = -99999;
1857    let mut th: i64 = 0;
1858    for line in lines {
1859      let [bs, w, h, d] = *line;
1860      wd = max(w, wd);
1861      th += h + d;
1862      if prevdepth >= 0 && bs >= 0 {
1863        if prevdepth + h < bs {
1864          th += bs - prevdepth - h;
1865        } else {
1866          th += lineskip;
1867        }
1868      }
1869      // TeX vpack \prevdepth discipline (divergence from Perl #2798 — see
1870      // the vertical branch above): boxes set prevdepth to their depth;
1871      // glue (bs == -1) is TRANSPARENT (prevdepth unchanged, so the next
1872      // box still receives \baselineskip accounting across the skip);
1873      // rules (bs == -2) disable it (TeX's \prevdepth = -1000pt sentinel).
1874      prevdepth = if bs >= 0 {
1875        d
1876      } else if bs == -1 {
1877        prevdepth
1878      } else {
1879        -99999
1880      };
1881    }
1882    let (ht, dp) = match vattach {
1883      "middle" => (th / 2 + mathaxis, th / 2 - mathaxis),
1884      "bottom" => {
1885        let d = lines[nlines - 1][3];
1886        (th - d, d)
1887      },
1888      // else (baseline / top): align to baseline of top row.
1889      _ => {
1890        let h = lines[0][2];
1891        (h, th - h)
1892      },
1893    };
1894    (wd, ht, dp)
1895  }

2042  if let Some(c) = decode(code, encoding_opt, implicit) {
2043    // T1 position 223: "SS" (capital sharp S as two chars)
2044    if c == '\u{1E9E}' {
2045      return Some(pin!("SS"));
2046    }
2047    // For standalone combining characters, prepend NBSP as base character
2048    // (Perl fontmaps encode these as UTF(0xA0)."\x{combining}")
2049    if is_combining_mark(c) {
2050      return Some(arena::pin(format!("\u{00A0}{c}")));
2051    }
2052    Some(arena::pin_char(c))
2053  } else {
2054    None
2055  }
2056}

2084    Some(enc) => Cow::Owned(enc),
2085    None => {
2086      font = lookup_font();
2087      if let Some(ref font) = font {
2088        match font.get_encoding() {
2089          None => Cow::Borrowed(""),
2090          Some(encoding) => encoding.clone(),
2091        }
2092      } else {
2093        Cow::Borrowed("")
2094      }
2095    },
2096  };
2097
2098  let mut map: Option<Fontmap> = None;
2099  if !encoding.is_empty() {

2102      // OK got some map.
2103      map = Some(encmap);
2104      if let Some(ref font) = font
2105        && let Some(family) = (*font).get_family()
2106      {
2107        with_value(&s!("{encoding}_{family}_fontmap"), |fmap_opt| {
2108          if let Some(fmap) = fmap_opt {
2109            map = fmap.into(); // Use the family specific map, if any.
2110          }
2111        });
2112      }
2113    }
2114  }
2115
2116  if implicit {
2117    if let Some(map) = map {
2118      if code < 128 {
2119        match map.get(code as usize) {
2120          None => None,
2121          Some(c) => *c,
2122        }
2123      } else {
2124        Some(code.into())
2125      }
2126    } else {
2127      Some(code.into())
2128    }
2129  } else if let Some(map) = map {
2130    match map.get(code as usize) {
2131      None => None,
2132      Some(c) => *c,
2133    }
2134  } else {
2135    None
2136  }

2139pub fn decode_string(string: SymStr, encoding_opt: Option<&str>, implicit: bool) -> SymStr {
2140  let empty_sym = pin!("");
2141  if string == empty_sym {
2142    return empty_sym;
2143  }
2144  let mut font = None;
2145  let encoding = match encoding_opt {
2146    None => {
2147      font = lookup_font();
2148      if let Some(ref font) = font {
2149        font.get_encoding().unwrap_or(&Cow::Borrowed(""))
2150      } else {
2151        ""
2152      }
2153    },
2154    Some(encoding) => encoding,
2155  };
2156
2157  let mut map: Option<Fontmap> = None;
2158  if !encoding.is_empty() {
2159    let _ = preload_font_map(encoding); // infallible in practice; swallow Result
2160    if let Some(encmap) = load_font_map(encoding) {
2161      // OK got some map.
2162      map = Some(encmap);
2163      if let Some(ref font) = font
2164        && let Some(family) = (*font).get_family()
2165      {
2166        with_value(&s!("{}_{}_fontmap", encoding, family), |fmap_opt| {
2167          if let Some(fmap) = fmap_opt {
2168            map = fmap.into(); // Use the family specific map, if any.
2169          }
2170        });
2171      }
2172    }
2173  }
2174
2175  // Load multi-char overrides if available
2176  let multichar_map: Option<HashMap<String, String>> = if !encoding.is_empty() {
2177    let mapname = format!("{encoding}_fontmap_multichar");
2178    with_value(&mapname, |val_opt| {
2179      if let Some(Stored::HashString(m)) = val_opt {
2180        Some(m.clone())
2181      } else {
2182        None
2183      }
2184    })
2185  } else {
2186    None
2187  };
2188
2189  let mut result_string: String = String::new();
2190  arena::with(string, |str| {
2191    for c in str.chars() {
2192      if implicit {
2193        if let Some(ref map_ref) = map {
2194          let code = c as u16; // u16, so that Unicode chars get cast correctly
2195          if code < 128 {
2196            // Check multi-char override first
2197            if let Some(ref mc) = multichar_map
2198              && let Some(mc_str) = mc.get(&(code as u8).to_string())
2199            {
2200              result_string.push_str(mc_str);
2201              continue;
2202            }
2203            if let Some(Some(mapc_val)) = map_ref.get(code as usize) {
2204              push_fontmap_char(&mut result_string, *mapc_val, code as u8);
2205            }
2206          } else {
2207            result_string.push(c);
2208          }
2209        } else {
2210          result_string.push(c)
2211        }
2212      } else if let Some(ref map_ref) = map {
2213        let code = c as u8;
2214        // Check multi-char override first
2215        if let Some(ref mc) = multichar_map
2216          && let Some(mc_str) = mc.get(&code.to_string())
2217        {
2218          result_string.push_str(mc_str);
2219          continue;
2220        }
2221        if let Some(Some(mapc_val)) = map_ref.get(code as usize) {
2222          push_fontmap_char(&mut result_string, *mapc_val, code);
2223        }
2224      }
2225    }
2226  });
2227  arena::pin(result_string)
2228}
