4/// Color in a specific color model, matching Perl's LaTeXML::Common::Color hierarchy.
5///
6/// Core models: rgb, cmy, cmyk, hsb, gray.
7/// PartialEq compares by model + components, matching Perl's Object::ne
8/// behavior via toString (e.g., cmyk(0,0,0,1) ≠ rgb(0,0,0)).
9#[derive(Clone, Copy, Debug, PartialEq)]
10pub enum Color {
11  // Eq is manually implemented below since f64 doesn't derive Eq,
12  // but our floats are always valid (no NaN).
13  Rgb(f64, f64, f64),
14  Cmy(f64, f64, f64),
15  Cmyk(f64, f64, f64, f64),
16  Hsb(f64, f64, f64),
17  Gray(f64),
18}
19
20/// Perl: use constant Black => bless ['rgb', 0, 0, 0], '...::rgb';
21pub const BLACK: Color = Color::Rgb(0.0, 0.0, 0.0);
22/// Perl: use constant White => bless ['rgb', 1, 1, 1], '...::rgb';
23pub const WHITE: Color = Color::Rgb(1.0, 1.0, 1.0);
24
25// f64 doesn't implement Eq, but our color components are always valid floats (no NaN).
26impl Eq for Color {}

28impl Hash for Color {
29  fn hash<H: Hasher>(&self, hasher: &mut H) {
30    std::mem::discriminant(self).hash(hasher);
31    for c in self.components() {
32      ((c * 100000.0) as i64).hash(hasher);
33    }
34  }
35}

37/// Perl: toString → "model(c1,c2,...)"
38impl fmt::Display for Color {
39  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
40    let model = self.model();
41    let comps: Vec<String> = self.components().iter().map(|c| format_component(*c)).collect();
42    write!(f, "{model}({})", comps.join(","))
43  }
44}

46impl Color {
47  /// Return the color model name. Perl: $self->model
48  pub fn model(&self) -> &'static str {
49    match self {
50      Color::Rgb(..) => "rgb",
51      Color::Cmy(..) => "cmy",
52      Color::Cmyk(..) => "cmyk",
53      Color::Hsb(..) => "hsb",
54      Color::Gray(..) => "gray",
55    }
56  }
57
58  /// Return the component values. Perl: $self->components
59  pub fn components(&self) -> Vec<f64> {
60    match self {
61      Color::Rgb(r, g, b) => vec![*r, *g, *b],
62      Color::Cmy(c, m, y) => vec![*c, *m, *y],
63      Color::Cmyk(c, m, y, k) => vec![*c, *m, *y, *k],
64      Color::Hsb(h, s, b) => vec![*h, *s, *b],
65      Color::Gray(g) => vec![*g],
66    }
67  }

69  /// Convert to RGB model. Perl: $self->rgb
70  pub fn to_rgb(&self) -> Color {
71    match self {
72      Color::Rgb(..) => *self,
73      Color::Cmy(c, m, y) => Color::Rgb(1.0 - c, 1.0 - m, 1.0 - y),
74      Color::Cmyk(..) => self.to_cmy().to_rgb(),
75      Color::Hsb(h, s, b) => {
76        let i = (6.0 * h) as i32;
77        let f = 6.0 * h - i as f64;
78        let u = b * (1.0 - s * (1.0 - f));
79        let v = b * (1.0 - s * f);
80        let w = b * (1.0 - s);
81        match i {
82          0 => Color::Rgb(*b, u, w),
83          1 => Color::Rgb(v, *b, w),
84          2 => Color::Rgb(w, *b, u),
85          3 => Color::Rgb(w, v, *b),
86          4 => Color::Rgb(u, w, *b),
87          5 => Color::Rgb(*b, w, v),
88          6 => Color::Rgb(*b, w, w),
89          _ => Color::Rgb(*b, w, w), // fallback
90        }
91      },
92      Color::Gray(g) => Color::Rgb(*g, *g, *g),
93    }
94  }
95
96  /// Convert to CMY model. Perl: $self->cmy
97  pub fn to_cmy(&self) -> Color {
98    match self {
99      Color::Cmy(..) => *self,
100      Color::Rgb(r, g, b) => Color::Cmy(1.0 - r, 1.0 - g, 1.0 - b),
101      Color::Cmyk(c, m, y, k) => {
102        Color::Cmy((c + k).min(1.0), (m + k).min(1.0), (y + k).min(1.0))
103      },
104      Color::Hsb(..) => self.to_rgb().to_cmy(),
105      Color::Gray(g) => Color::Cmy(1.0 - g, 1.0 - g, 1.0 - g),
106    }
107  }
108
109  /// Convert to CMYK model. Perl: $self->cmyk
110  pub fn to_cmyk(&self) -> Color {
111    match self {
112      Color::Cmyk(..) => *self,
113      Color::Cmy(c, m, y) => {
114        // Perl: undercolor-removal with beta parameters all = 1
115        let k = c.min(*m).min(*y);
116        Color::Cmyk(
117          (c - k).clamp(0.0, 1.0),
118          (m - k).clamp(0.0, 1.0),
119          (y - k).clamp(0.0, 1.0),
120          k,
121        )
122      },
123      Color::Rgb(..) => self.to_cmy().to_cmyk(),
124      Color::Hsb(..) => self.to_rgb().to_cmyk(),
125      Color::Gray(g) => Color::Cmyk(0.0, 0.0, 0.0, 1.0 - g),
126    }
127  }
128
129  /// Convert to HSB model. Perl: $self->hsb
130  pub fn to_hsb(&self) -> Color {
131    match self {
132      Color::Hsb(..) => *self,
133      Color::Rgb(r, g, b) => {
134        // Perl: rgb.pm Phi function + hsb dispatch
135        let i =
136          4 * (if *r >= *g { 1 } else { 0 })
137            + 2 * (if *g >= *b { 1 } else { 0 })
138            + (if *b >= *r { 1 } else { 0 });
139        match i {
140          1 => phi(*b, *g, *r, 3.0, 1.0),
141          2 => phi(*g, *r, *b, 1.0, 1.0),
142          3 => phi(*g, *b, *r, 3.0, -1.0),
143          4 => phi(*r, *b, *g, 5.0, 1.0),
144          5 => phi(*b, *r, *g, 5.0, -1.0),
145          6 => phi(*r, *g, *b, 1.0, -1.0),
146          7 => Color::Hsb(0.0, 0.0, *b),
147          _ => Color::Hsb(0.0, 0.0, 0.0),
148        }
149      },
150      Color::Cmy(..) => self.to_rgb().to_hsb(),
151      Color::Cmyk(..) => self.to_cmy().to_hsb(),
152      Color::Gray(g) => Color::Hsb(0.0, 0.0, *g),
153    }
154  }
155
156  /// Convert to gray model. Perl: $self->gray
157  pub fn to_gray(&self) -> Color {
158    match self {
159      Color::Gray(..) => *self,
160      Color::Rgb(r, g, b) => Color::Gray(0.3 * r + 0.59 * g + 0.11 * b),
161      Color::Cmy(c, m, y) => Color::Gray(1.0 - (0.3 * c + 0.59 * m + 0.11 * y)),
162      Color::Cmyk(c, m, y, k) => {
163        Color::Gray(1.0 - (0.3 * c + 0.59 * m + 0.11 * y + k).min(1.0))
164      },
165      Color::Hsb(..) => self.to_rgb().to_gray(),
166    }
167  }

169  /// Convert to another model by name. Perl: $self->convert($tomodel)
170  pub fn convert(&self, to_model: &str) -> Color {
171    match to_model {
172      "rgb" => self.to_rgb(),
173      "cmy" => self.to_cmy(),
174      "cmyk" => self.to_cmyk(),
175      "hsb" => self.to_hsb(),
176      "gray" => self.to_gray(),
177      _ => *self,
178    }
179  }

181  /// Convert to hex attribute string. Perl: $self->toAttribute() = $self->rgb->toHex()
182  pub fn to_attribute(&self) -> String {
183    let rgb = self.to_rgb();
184    if let Color::Rgb(r, g, b) = rgb {
185      format!(
186        "#{:02X}{:02X}{:02X}",
187        component_to_u8(r),
188        component_to_u8(g),
189        component_to_u8(b)
190      )
191    } else {
192      unreachable!()
193    }
194  }

196  /// Complement. Perl: $self->complement
197  pub fn complement(&self) -> Color {
198    match self {
199      Color::Rgb(r, g, b) => Color::Rgb(1.0 - r, 1.0 - g, 1.0 - b),
200      Color::Cmy(c, m, y) => Color::Cmy(1.0 - c, 1.0 - m, 1.0 - y),
201      Color::Cmyk(..) => self.to_cmy().complement().to_cmyk(),
202      Color::Hsb(h, s, b) => {
203        let hp = if *h < 0.5 { h + 0.5 } else { h - 0.5 };
204        let bp = 1.0 - b * (1.0 - s);
205        let sp = if bp == 0.0 { 0.0 } else { b * s / bp };
206        Color::Hsb(hp, sp, bp)
207      },
208      Color::Gray(g) => Color::Gray(1.0 - g),
209    }
210  }

212  /// Mix self*fraction + other*(1-fraction). Perl: $self->mix($other, $fraction)
213  pub fn mix(&self, other: &Color, fraction: f64) -> Color {
214    let (base, other) = self.align_models(other);
215    // Hsb: mix in rgb space then convert back
216    if matches!(&base, Color::Hsb(..)) {
217      return base.to_rgb().mix(&other, fraction).to_hsb();
218    }
219    let a = base.components();
220    let b = other.components();
221    let mixed: Vec<f64> = a.iter().zip(b.iter()).map(|(ai, bi)| fraction * ai + (1.0 - fraction) * bi).collect();
222    from_model_components(base.model(), &mixed)
223  }
224
225  /// Add component-wise. Perl: $self->add($other)
226  pub fn add(&self, other: &Color) -> Color {
227    let (base, other) = self.align_models(other);
228    let a = base.components();
229    let b = other.components();
230    let added: Vec<f64> = a.iter().zip(b.iter()).map(|(ai, bi)| ai + bi).collect();
231    from_model_components(base.model(), &added)
232  }

234  /// Scale all components. Perl: $self->scale($m)
235  pub fn scale(&self, m: f64) -> Color {
236    let scaled: Vec<f64> = self.components().iter().map(|c| m * c).collect();
237    from_model_components(self.model(), &scaled)
238  }
239
240  /// Multiply by component vector. Perl: $self->multiply(@m)
241  pub fn multiply(&self, factors: &[f64]) -> Color {
242    let comps = self.components();
243    let result: Vec<f64> = comps.iter().zip(factors.iter()).map(|(c, f)| c * f).collect();
244    from_model_components(self.model(), &result)
245  }

247  /// Align two colors to the same model for operations.
248  /// Perl: if base is gray, convert to other's model; else convert other to base's model.
249  fn align_models(&self, other: &Color) -> (Color, Color) {
250    if self.model() == other.model() {
251      return (*self, *other);
252    }
253    if self.model() == "gray" {
254      (self.convert(other.model()), *other)
255    } else {
256      (*self, other.convert(self.model()))
257    }
258  }

260  /// Format the RGB components as a comma-separated string for reversion.
261  /// Perl: join(',', $color->rgb->components)
262  pub fn rgb_components_string(&self) -> String {
263    let rgb = self.to_rgb();
264    let comps = rgb.components();
265    comps.iter().map(|c| format_component(*c)).collect::<Vec<_>>().join(",")
266  }

268  /// Encode for state storage: "model c1 c2 ..."
269  pub fn to_stored(&self) -> String {
270    let model = self.model();
271    let comps: Vec<String> = self.components().iter().map(|c| format_component(*c)).collect();
272    format!("{model} {}", comps.join(" "))
273  }
274
275  /// Decode from state storage format "model c1 c2 ..."
276  pub fn from_stored(s: &str) -> Option<Color> {
277    let parts: Vec<&str> = s.split_whitespace().collect();
278    if parts.is_empty() {
279      return None;
280    }
281    let model = parts[0];
282    let comps: Vec<f64> = parts[1..].iter().filter_map(|p| p.parse::<f64>().ok()).collect();
283    Some(from_model_components(model, &comps))
284  }

287/// Convert color component float (0.0-1.0) to u8 (0-255).
288/// Matches Perl's `roundto($n * 255, 0)` which adds a small epsilon factor.
289fn component_to_u8(v: f64) -> u8 {
290  let scaled = v.clamp(0.0, 1.0) * 255.0 * (1.0 + 100.0 * f64::EPSILON);
291  scaled.round() as u8
292}

294/// Format a float component like Perl: integers without decimal point.
295pub fn format_component(v: f64) -> String {
296  if (v - v.round()).abs() < 1e-10 {
297    format!("{}", v.round() as i64)
298  } else {
299    format!("{v}")
300  }
301}

303/// Perl rgb.pm: Phi function for RGB→HSB conversion
304fn phi(x: f64, y: f64, z: f64, u: f64, v: f64) -> Color {
305  Color::Hsb(
306    (u * (x - z) + v * (x - y)) / (6.0 * (x - z)),
307    (x - z) / x,
308    x,
309  )
310}

312/// Parse color components from a spec string (comma or space separated)
313fn parse_components(spec: &str) -> Vec<f64> {
314  if spec.contains(',') {
315    spec.split(',').filter_map(|s| s.trim().parse::<f64>().ok()).collect()
316  } else {
317    spec.split_whitespace().filter_map(|s| s.parse::<f64>().ok()).collect()
318  }
319}

321/// Create a Color from model name and component values
322pub fn from_model_components(model: &str, comps: &[f64]) -> Color {
323  match model {
324    "rgb" if comps.len() >= 3 => Color::Rgb(comps[0], comps[1], comps[2]),
325    "cmy" if comps.len() >= 3 => Color::Cmy(comps[0], comps[1], comps[2]),
326    "cmyk" if comps.len() >= 4 => Color::Cmyk(comps[0], comps[1], comps[2], comps[3]),
327    "hsb" if comps.len() >= 3 => Color::Hsb(comps[0], comps[1], comps[2]),
328    "gray" if !comps.is_empty() => Color::Gray(comps[0]),
329    _ => BLACK,
330  }
331}

333/// Parse a color from model name + spec string.
334/// Perl: Color($model, components)->toCore
335pub fn color_from_model_spec(model: &str, spec: &str) -> Color {
336  let spec = spec.trim().trim_matches(|c| c == '{' || c == '}').trim();
337  let c = parse_components(spec);
338  from_model_components(model, &c)
