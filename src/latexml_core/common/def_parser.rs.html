16static CSNAME_MACRO_RE: Lazy<Regex> =
17  Lazy::new(|| Regex::new(r"^\\csname\s+(.*)\\endcsname").unwrap());
18// Includes `_` (expl3 private/word-internal) and an optional `:<letters>`
19// suffix (expl3 parameter-type sigil) so prototype strings like
20// "\\draw_path_arc:nnn{}{}{}" parse with the entire `\draw_path_arc:nnn`
21// as the control-sequence name. Under normal LaTeX catcodes `_` is SUB
22// and `:` is OTHER, so these names only round-trip through the
23// tokenizer under expl3 catcode regime — but compile-time prototype
24// strings bypass the tokenizer, so this is purely a string-parsing
25// concern. Witness: l3draw_sty stubs would previously fail with
26// "Unrecognized parameter type with name '_begin', spec '_begin:'".
27static CS_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\\[a-zA-Z@_]+(?::[a-zA-Z]*)?)").unwrap());
28static SINGLE_CHAR_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\\.)").unwrap());
29static ACTIVE_CHAR_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(.)").unwrap());

31/// If calling at compile-time, pass `None` for state, to avoid initialization.
32pub fn parse_prototype(proto: &str, init_flag: bool) -> Result<(Token, Option<Parameters>)> {
33  let cs;
34  let normalized_proto = if let Some(captures) = CSNAME_MACRO_RE.captures(proto) {
35    let csname_content = captures.get(1).map_or("", |m| m.as_str());
36    // At compile time, reject \csname patterns with braces — these produce CS names
37    // like \begin{env} that can cause infinite loops in the proc macro expansion.
38    // Starred environments (\csname eqnarray*\endcsname) are fine.
39    if !init_flag && (csname_content.contains('{') || csname_content.contains('}')) {
40      panic!(
41        "\\csname...\\endcsname with braces in definition prototype is not supported at compile time: \"{}\". \
42         Use RawTeX!() or runtime DefMacroI() for CS names containing {{}} characters.",
43        proto
44      );
45    }
46    cs = T_CS!(s!("\\{}", csname_content));
47    // also replace in proto
48    CSNAME_MACRO_RE.replace(proto, "")
49  } else if let Some(captures) = CS_RE.captures(proto) {
50    // Match a cs
51    let csname = captures.get(1).map_or("", |m| m.as_str()).to_string();
52    cs = T_CS!(csname);
53    // also replace in proto
54    CS_RE.replace(proto, "")
55  } else if let Some(captures) = SINGLE_CHAR_RE.captures(proto) {
56    // Match a single char cs, env name,...
57    cs = T_CS!(captures.get(1).map_or("", |m| m.as_str()));
58    // also replace in proto
59    SINGLE_CHAR_RE.replace(proto, "")
60  } else if let Some(captures) = ACTIVE_CHAR_RE.captures(proto) {
61    // Match an active char
62    cs = mouth::tokenize_internal(captures.get(1).map_or("", |m| m.as_str()))
63      .unlist()
64      .remove(0);
65    // also replace in proto
66    ACTIVE_CHAR_RE.replace(proto, "")
67  } else {
68    let message = s!(
69      "Definition prototype doesn't have proper control sequence: \"{}\"",
70      proto
71    );
72    fatal!(Prototype, Misdefined, message);
73  };
74  let final_proto = normalized_proto.trim();
75  let paramlist = parse_parameters(final_proto, &cs, init_flag)?;
76  Ok((cs, paramlist))
77}

79/// If calling at compile-time, pass `None` for state, to avoid initialization.
80pub fn parse_parameters(
81  outer_prototype: &str,
82  cs: &Token,
83  init_flag: bool,
84) -> Result<Option<Parameters>> {
85  // The prototype grammar, as a winnow parse (#171 family; the golden corpus
86  // in `golden_tests` pins byte-equivalence with the prior regex munch loop):
87  //   item := "{" inner "}" \s*          (Plain, recursive inner)
88  //         | "[" inner "]" \s*          (Optional; inner may be Default:…)
89  //         | word (":" extra)? \s*      (named type, extra |-split)   [nonempty]
90  //         | any-single-char            (literal Token; spaces too — each
91  //                                       space is its own Token, faithfully)
92  use winnow::{
93    combinator::{delimited, opt, preceded},
94    prelude::*,
95    token::{any, take_while},
96  };
97
98  fn ws0(input: &mut &str) -> ModalResult<()> {
99    take_while(0.., |c: char| c.is_whitespace())
100      .void()
101      .parse_next(input)
102  }
103  /// `{inner}` / `[inner]` group: returns the inner slice; trailing \s* eaten.
104  fn group<'a>(open: char, close: char) -> impl FnMut(&mut &'a str) -> ModalResult<&'a str> {
105    move |input: &mut &'a str| {
106      let inner = delimited(open, take_while(0.., move |c| c != close), close).parse_next(input)?;
107      ws0(input)?;
108      Ok(inner)
109    }
110  }
111  /// `word(:extra)?` with word ∈ \w+ or a bare nonempty `:extra`; \s* eaten.
112  fn paramspec<'a>(input: &mut &'a str) -> ModalResult<(&'a str, Option<&'a str>)> {
113    let word = take_while(0.., |c: char| c.is_alphanumeric() || c == '_').parse_next(input)?;
114    let extra = opt(preceded(
115      ':',
116      take_while(0.., |c: char| !c.is_whitespace() && c != '{' && c != '['),
117    ))
118    .parse_next(input)?;
119    if word.is_empty() && extra.is_none() {
120      return Err(winnow::error::ErrMode::Backtrack(
121        winnow::error::ContextError::new(),
122      ));
123    }
124    ws0(input)?;
125    Ok((word, extra))
126  }
127
128  let mut parameters = Vec::with_capacity(4);
129  let mut rest: &str = outer_prototype;
130  let input = &mut rest;
131  while !input.is_empty() {
132    // Probe-and-commit on a copy per branch: a failing branch must not consume
133    // (`delimited` advances past its opening token before failing otherwise —
134    // e.g. a lone "{" from the non-nesting inner of "{{}}").
135    let mut probe;
136    let res = {
137      probe = *input;
138      group('{', '}')
139        .parse_next(&mut probe)
140        .inspect(|_| *input = probe)
141    };
142    let mut p: Parameter = match res {
143      Ok(inner_spec) => {
144        // Plain (possibly typed-inner) braced group, spec keeps its braces.
145        let inner: Option<Parameters> = if inner_spec.is_empty() {
146          None
147        } else {
148          parse_parameters(inner_spec, cs, init_flag)?
149        };
150        Parameter {
151          name: pin!("Plain"),
152          spec: arena::pin(format!("{{{inner_spec}}}")),
153          inner: inner.map(|ps| ps.into()).unwrap_or_default(),
154          ..Parameter::default()
155        }
156      },
157      _ => {
158        let res = {
159          probe = *input;
160          group('[', ']')
161            .parse_next(&mut probe)
162            .inspect(|_| *input = probe)
163        };
164        match res {
165          Ok(inner_spec) => {
166            let spec = arena::pin(format!("[{inner_spec}]"));
167            if let Some(default_str) = inner_spec.strip_prefix("Default:") {
168              let extra = if default_str.is_empty() {
169                vec![]
170              } else {
171                vec![mouth::tokenize_internal(default_str)]
172              };
173              Parameter {
174                name: pin!("Optional"),
175                spec,
176                extra,
177                ..Parameter::default()
178              }
179            } else if !inner_spec.is_empty() {
180              Parameter {
181                name: pin!("Optional"),
182                spec,
183                inner: parse_parameters(inner_spec, cs, init_flag)?
184                  .map(|ps| ps.into())
185                  .unwrap_or_default(),
186                ..Parameter::default()
187              }
188            } else {
189              Parameter {
190                name: pin!("Optional"),
191                spec,
192                ..Parameter::default()
193              }
194            }
195          },
196          _ => {
197            let res = {
198              probe = *input;
199              paramspec.parse_next(&mut probe).inspect(|_| *input = probe)
200            };
201            match res {
202              Ok((word, extra_opt)) => {
203                let spec_str = match extra_opt {
204                  Some(extra) => format!("{word}:{extra}"),
205                  None => word.to_string(),
206                };
207                let extra: Vec<Tokens> = match extra_opt {
208                  None | Some("") => Vec::new(),
209                  Some(extra_str) => extra_str
210                    .split('|')
211                    .map(|t| Tokens::new(mouth::tokenize_internal(t).unlist()))
212                    .collect(),
213                };
214                Parameter {
215                  name: arena::pin(word),
216                  spec: arena::pin(&spec_str),
217                  extra,
218                  ..Parameter::default()
219                }
220              },
221              _ => {
222                // Literal single char (incl. each whitespace char) as a Token parameter.
223                let ch = any.parse_next(input).map_err(
224                  |_: winnow::error::ErrMode<winnow::error::ContextError>| {
225                    Error::from(s!(
226                      "parse_parameters: unreadable prototype tail for {:?}",
227                      cs
228                    ))
229                  },
230                )?;
231                let ch_token = CharToken!(ch, Catcode::OTHER);
232                Parameter {
233                  name: pin!("Token"),
234                  spec: pin!("Token"),
235                  extra: vec![Tokens::new(vec![ch_token])],
236                  ..Parameter::default()
237                }
238              },
239            }
240          },
241        }
242      },
243    };
244    if init_flag {
245      p = p.init()?;
246    }
247    parameters.push(p);
248  }
249  if parameters.is_empty() {
250    Ok(None)
251  } else {
252    Ok(Some(Parameters::new(parameters)))
253  }
254}

256#[cfg(test)]
257mod golden_tests {
258  /// Render a parse result in a stable, comparison-friendly form.
259  fn describe(proto: &str) -> String {
260    match super::parse_parameters(proto, &crate::T_CS!("\\x"), false) {
261      Ok(None) => "None".to_string(),
262      Ok(Some(ps)) => ps
263        .get_parameters()
264        .iter()
265        .map(|p| {
266          format!(
267            "{}:{}/x{}/i{}",
268            crate::common::arena::with(p.name, |s| s.to_string()),
269            crate::common::arena::with(p.spec, |s| s.to_string()),
270            p.extra.len(),
271            p.inner
272              .as_ref()
273              .map(|ps| ps.get_parameters().len())
274              .unwrap_or(0)
275          )
276        })
277        .collect::<Vec<_>>()
278        .join(" | "),
279      Err(e) => format!("ERR:{e}"),
280    }
281  }
