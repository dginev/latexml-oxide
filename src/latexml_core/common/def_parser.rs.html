13static CSNAME_MACRO_RE: Lazy<Regex> =
14  Lazy::new(|| Regex::new(r"^\\csname\s+(.*)\\endcsname").unwrap());
15static CS_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\\[a-zA-Z@]+)").unwrap());
16static SINGLE_CHAR_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\\.)").unwrap());
17static ACTIVE_CHAR_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(.)").unwrap());
18static DEFAULT_CHECK_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^Default:(.*)$").unwrap());
19static NESTED_CHECK_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\{([^\}]*)\})\s*").unwrap());
20static OPTIONAL_CHECK_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\[([^\]]*)\])\s*").unwrap());
21static PARAMSPECT_CHECK_RE: Lazy<Regex> =
22  Lazy::new(|| Regex::new(r"^((\w*)(:([^\s\{\[]*))?)\s*").unwrap());

24/// If calling at compile-time, pass `None` for state, to avoid initialization.
25pub fn parse_prototype(proto: &str, init_flag: bool) -> Result<(Token, Option<Parameters>)> {
26  let cs;
27  let normalized_proto = if let Some(captures) = CSNAME_MACRO_RE.captures(proto) {
28    cs = T_CS!(s!("\\{}", captures.get(1).map_or("", |m| m.as_str())));
29    // also replace in proto
30    CSNAME_MACRO_RE.replace(proto, "")
31  } else if let Some(captures) = CS_RE.captures(proto) {
32    // Match a cs
33    let csname = captures.get(1).map_or("", |m| m.as_str()).to_string();
34    cs = T_CS!(csname);
35    // also replace in proto
36    CS_RE.replace(proto, "")
37  } else if let Some(captures) = SINGLE_CHAR_RE.captures(proto) {
38    // Match a single char cs, env name,...
39    cs = T_CS!(captures.get(1).map_or("", |m| m.as_str()));
40    // also replace in proto
41    SINGLE_CHAR_RE.replace(proto, "")
42  } else if let Some(captures) = ACTIVE_CHAR_RE.captures(proto) {
43    // Match an active char
44    cs = mouth::tokenize_internal(captures.get(1).map_or("", |m| m.as_str()))
45      .unlist()
46      .remove(0);
47    // also replace in proto
48    ACTIVE_CHAR_RE.replace(proto, "")
49  } else {
50    let message = s!(
51      "Definition prototype doesn't have proper control sequence: \"{}\"",
52      proto
53    );
54    fatal!(Prototype, Misdefined, message);
55  };
56  let final_proto = normalized_proto.trim();
57  let paramlist = parse_parameters(final_proto, &cs, init_flag)?;
58  Ok((cs, paramlist))
59}

61/// If calling at compile-time, pass `None` for state, to avoid initialization.
62pub fn parse_parameters(
63  outer_prototype: &str,
64  cs: &Token,
65  init_flag: bool,
66) -> Result<Option<Parameters>> {
67  let mut prototype = Cow::Borrowed(outer_prototype);
68  let mut parameters = Vec::new();
69  while !prototype.is_empty() {
70    let next_proto: Cow<str>;
71    // Handle possibly nested cases, such as {Number}
72    if NESTED_CHECK_RE.is_match(&prototype) {
73      let captures = NESTED_CHECK_RE.captures(&prototype).unwrap();
74      next_proto = NESTED_CHECK_RE.replace(&prototype, "");
75      let spec = captures.get(1).map_or("", |m| m.as_str());
76      let inner_spec = captures.get(2).map_or("", |m| m.as_str());
77      let inner: Option<Parameters> = if inner_spec.is_empty() {
78        None
79      } else {
80        parse_parameters(inner_spec, cs, init_flag)?
81      };
82      let mut p = Parameter {
83        name: arena::pin_static("Plain"),
84        spec: if spec.is_empty() {
85          *EMPTY_SYM
86        } else {
87          arena::pin(spec)
88        },
89        inner: inner.map(|ps| ps.into()).unwrap_or_default(),
90        ..Parameter::default()
91      };
92      if init_flag {
93        p = p.init()?;
94      }
95      parameters.push(p);
96    } else if let Some(captures) = OPTIONAL_CHECK_RE.captures(&prototype) {
97      // Ditto for Optional
98      let spec = captures.get(1).map_or("", |m| m.as_str());
99      let inner_spec = captures.get(2).map_or("", |m| m.as_str());
100      next_proto = OPTIONAL_CHECK_RE.replace(&prototype, "");
101      if let Some(_default_captures) = DEFAULT_CHECK_RE.captures(inner_spec) {
102        // TODO: Add the defaults !
103        let mut p = Parameter {
104          name: arena::pin_static("Optional"),
105          spec: if spec.is_empty() {
106            *EMPTY_SYM
107          } else {
108            arena::pin(spec)
109          },
110          // extra: vec![TokenizeInternal!(default_captures.get(0).map_or("", |m| m.as_str()))],
111          ..Parameter::default()
112        };
113        if init_flag {
114          p = p.init()?;
115        }
116        parameters.push(p);
117      } else if !inner_spec.is_empty() {
118        let mut p = Parameter {
119          name: arena::pin_static("Optional"),
120          spec: if spec.is_empty() {
121            *EMPTY_SYM
122          } else {
123            arena::pin(spec)
124          },
125          inner: parse_parameters(inner_spec, cs, init_flag)?
126            .map(|ps| ps.into())
127            .unwrap_or_default(),
128          ..Parameter::default()
129        };
130        if init_flag {
131          p = p.init()?;
132        }
133        parameters.push(p);
134      } else {
135        let mut p = Parameter {
136          name: arena::pin_static("Optional"),
137          spec: arena::pin(spec),
138          ..Parameter::default()
139        };
140        if init_flag {
141          p = p.init()?;
142        }
143        parameters.push(p);
144      }
145    } else if let Some(captures) = PARAMSPECT_CHECK_RE.captures(&prototype) {
146      let spec = arena::pin(captures.get(1).map_or("", |m| m.as_str()));
147      let name = arena::pin(captures.get(2).map_or("", |m| m.as_str()));
148      let extra_str = captures.get(4).map_or("", |m| m.as_str()).to_string();
149      next_proto = PARAMSPECT_CHECK_RE.replace(&prototype, "");
150      let extra: Vec<Tokens> = if extra_str.is_empty() {
151        Vec::new()
152      } else {
153        extra_str
154          .split('|')
155          .map(|t| Tokens::new(mouth::tokenize_internal(t).unlist()))
156          .collect()
157      };
158      let mut p = Parameter {
159        name,
160        spec,
161        extra,
162        ..Parameter::default()
163      };
164      if init_flag {
165        p = p.init()?;
166      }
167      parameters.push(p);
168    } else {
169      fatal!(
170        Parameter,
171        Misdefined,
172        s!(
173          "Unrecognized parameter specification at \"prototype\" {:?}",
174          cs
175        )
176      );
177    }
178    prototype = Cow::Owned(next_proto.to_string());
179  }
180  if parameters.is_empty() {
181    Ok(None)
182  } else {
183    Ok(Some(Parameters::new(parameters)))
184  }
185}
