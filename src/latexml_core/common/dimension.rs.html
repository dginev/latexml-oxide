45impl Dimension {
46  pub fn spec_to_f64(spec: &str) -> Result<f64> {
47    if spec.is_empty() {
48      Ok(0.0)
49    } else if let Some(cap) = SPEC_RE.captures(spec) {
50      // Dimensions given.
51      let num_str = cap.get(1).map_or(String::new(), |m| m.as_str().to_string());
52      let num: f64 = num_str.parse::<f64>().expect(&num_str);
53      let unit = cap.get(2).map_or(String::new(), |m| m.as_str().to_string());
54      let converted_unit = convert_unit(&unit);
55      Ok(fixpoint(num, Some(converted_unit)) as f64)
56    } else {
57      // When scaled points passed in (typically the result of Perl calculations on other
58      // Dimensions), you might think truncation (int) is more TeX-like.
59      // Recall that TeX arithmatic truncates, whereas reading by Gullet tends to round.
60      // The Perl arithmetic is replacing an unknown combination of those truncates & rounds.
61      // As it turns out, using int() here results in non-terminating loops in pgf/tikz.
62      // So, we use round (Knuth style)
63      // Note that divide and such explicitly use int(), however!
64      Ok(kround(spec.parse::<f64>().expect(spec)) as f64)
65    }
66  }
67}

78// This is Knuth's print_scaled (See TeX the Program, \S 103)
79// It (should) round-trip with kround.
80pub fn fixedformat(mut s: i64, unit_opt: Option<&str>) -> String {
81  let mut string = String::new();
82  if s < 0 {
83    string.push('-');
84    s = -s;
85  }
86  string.push_str(&(s / UNITY).to_string());
87  string.push('.');
88  s = 10 * (s % UNITY) + 5;
89  let mut delta = 10;
90  loop {
91    if delta > UNITY {
92      s += 0x8000 - 50000;
93    }
94    string.push_str(&(s / UNITY).to_string());
95    s = 10 * (s % UNITY);
96    delta *= 10;
97    if s <= delta {
98      break;
99    }
100  }
101  if let Some(unit) = unit_opt {
102    string.push_str(unit);
103  }
104  string
105}
