86  ///**********************************************************************
87  /// Namespaces
88  ///**********************************************************************
89  /// There are TWO namespace mappings!!!
90  /// One for coding, one for the DocType.
91  ///
92  /// Coding: this namespace mapping associates prefixes to namespace URIs for
93  ///   use in the latexml code, constructors and such.
94  ///   This must be a one to one mapping and there are no default namespaces.
95  /// Document: this namespace mapping associates prefixes to namespace URIs
96  ///   as used in the generated document, and will be the
97  ///   set of prefixes used in the generated output.
98  ///   This mapping may also use a prefix of "#default" which is for
99  ///   the unprefixed form of elements (not used for attributes!)
100  pub fn register_namespace(&mut self, codeprefix: &str, namespace_opt: Option<&str>) {
101    self.register_namespace_sym(arena::pin(codeprefix), namespace_opt.map(arena::pin))
102  }
103  pub fn register_namespace_sym(&mut self, codeprefix: SymStr, namespace_opt: Option<SymStr>) {
104    // double-check empty strings are None
105    let namespace_opt_checked = namespace_opt.filter(|val| *val != *EMPTY_SYM);
106    match namespace_opt_checked {
107      Some(namespace) => {
108        self
109          .code_namespace_prefixes
110          .insert_sym(namespace, codeprefix);
111        self.code_namespaces.insert_sym(codeprefix, namespace);
112      },
113      None => {
114        if let Some(prev) = self.code_namespaces.get_sym(codeprefix) {
115          self.code_namespace_prefixes.remove_sym(*prev);
116        };
117        self.code_namespaces.remove_sym(codeprefix);
118      },
119    };
120  }

122  pub fn register_document_namespace(&mut self, docprefix: &str, namespace_opt: Option<&str>) {
123    let default_sym = *H_DEFAULT_SYM;
124    let docprefix_sym = if docprefix.is_empty() {
125      default_sym
126    } else {
127      arena::pin(docprefix)
128    };
129
130    match namespace_opt {
131      Some(namespace) => {
132        // Since the default namespace url can still ALSO have a prefix associated,
133        // we prepend "DEFAULT#url" when using as a hash key in the prefixes table.
134        let ns_sym = arena::pin(namespace);
135        let regnamespace = if docprefix_sym == default_sym {
136          arena::pin(s!("DEFAULT#{namespace}"))
137        } else {
138          ns_sym
139        };
140        self
141          .document_namespace_prefixes
142          .insert_sym(regnamespace, docprefix_sym);
143        self.document_namespaces.insert_sym(docprefix_sym, ns_sym);
144      },
145      None => {
146        if let Some(prev) = self.document_namespaces.get_sym(docprefix_sym) {
147          self.document_namespace_prefixes.remove_sym(*prev);
148        };
149        self.document_namespaces.remove_sym(docprefix_sym);
150      },
151    };
152  }

157  /// TODO: This is another component that would fit perfectly as a compiler plugin.
158  /// For now, simply reimplementing the runtime loading of
159  /// LaTeXML.model as-is from Model.pm
160  pub fn load_compiled_schema(&mut self, path: &str) {
161    note_begin(&s!("Loading compiled schema {}\n", path));
162    let compiled_fh = File::open(path).unwrap();
163    let compiled_reader = BufReader::new(&compiled_fh);
164    for line_item in compiled_reader.lines().map_while(std::result::Result::ok) {
165      let line: String = line_item;
166      if let Some(caps) = TAG_MODEL_LINE_RE.captures(&line) {
167        let tag = caps.get(1).map_or("", |m| m.as_str());
168        let attr = caps.get(2).map_or("", |m| m.as_str());
169        let children = caps.get(3).map_or("", |m| m.as_str());
170        self.add_tag_attribute(tag, attr.split(',').collect());
171        self.add_tag_content(tag, children.split(',').collect());
172      } else if let Some(caps) = CLASS_MODEL_LINE_RE.captures(&line) {
173        let classname = caps.get(1).map_or("", |m| m.as_str());
174        let elements = caps.get(2).map_or("", |m| m.as_str());
175        let mut class_set = HashSet::default();
176        for set_element in elements.split(',').collect::<Vec<&str>>() {
177          class_set.insert(arena::pin(set_element));
178        }
179        self.set_schema_class(classname, class_set);
180      } else if let Some(caps) = NAMESPACE_MODEL_LINE_RE.captures(&line) {
181        let prefix = caps.get(1).map_or("", |m| m.as_str());
182        let namespace = caps.get(2).map_or("", |m| m.as_str());
183        self.register_document_namespace(prefix, Some(namespace));
184      } else {
185        panic!("Fatal:internal:{path} Compiled model '{path}' is malformatted at \"{line}\"");
186      }
187    }
188
189    note_end(&s!("Loading compiled schema {path}\n"));
190  }

239  /// Clone the tagprop's (allowed content & attributes) of @other to $tag
240  fn synthesize_element(&mut self, tag: &str, others: &[&str]) {
241    let mut to_add_in_model = Vec::new();
242    let mut to_add_in_attrs = Vec::new();
243    for other in others {
244      if let Some(content) = self.schema_class.get(other) {
245        for child in content {
246          to_add_in_model.push(*child);
247        }
248      } else if let Some(entry) = self.tagprop.get(other) {
249        for child in &entry.model {
250          to_add_in_model.push(*child);
251        }
252        for attr in &entry.attributes {
253          to_add_in_attrs.push(*attr);
254        }
255      }
256    }
257    let capture = self.tagprop.entry(tag).or_default();
258    for child in to_add_in_model {
259      capture.model.insert(child);
260    }
261    for attr in to_add_in_attrs {
262      capture.attributes.insert(attr);
263    }
264  }

283pub fn load_schema(search_paths: &[&str]) -> Result<()> {
284  // Only load once
285  let mut model = model_mut!();
286  if model.schema.is_some() {
287    return Ok(());
288  }
289  let mut name = String::new();
290  if model.schema_data.is_none() {
291    // TODO: Return this code path to normal once we properly load schemas
292    Warn!("expected", "<model>", "TODO");
293    // Warn('expected', '<model>', undef, "No Schema Model has been declared; assuming LaTeXML");
294    // // article ??? or what ? undef gives problems!
295
296    model.register_document_namespace("ltx", Some(LTX_NAMESPACE));
297    model.set_relaxng_schema("LaTeXML");
298    model.register_namespace("ltx", Some(LTX_NAMESPACE));
299    model.register_namespace("svg", Some("http://www.w3.org/2000/svg"));
300    model.register_namespace("xlink", Some("http://www.w3.org/1999/xlink")); // Needed for SVG
301    model.register_namespace("m", Some("http://www.w3.org/1998/Math/MathML"));
302    model.register_namespace("xhtml", Some("http://www.w3.org/1999/xhtml"));
303    model.permissive = true;
304  } // Actually, they could have declared all sorts of Tags....
305  let mut schema_type = *EMPTY_SYM;
306  match model.schema_data {
307    None => {},
308    Some(ref data) => {
309      schema_type = data[0];
310      if schema_type == *DTD_SYM {
311        // NOTE: This is a hack, as DTD should be deprecated, just making xii test work for now
312        // ($roottag, $publicid, $systemid) = @data;
313        name = arena::with(*data.last().unwrap(), |data_str| {
314          data_str.replace(".dtd", "")
315        });
316        model.schema = Some(Relaxng {
317          name: "DTD".to_string(), // HACK, phase out DTD support!
318          ..Relaxng::default()
319        });
320        // $systemid);
321      } else if schema_type == *RELAXNG_SYM {
322        name = arena::to_string(data[1]);
323        model.schema = Some(Relaxng {
324          name: name.clone(),
325          ..Relaxng::default()
326        });
327      } else {
328        let message = arena::with(schema_type, |schema_type_str| {
329          s!("Can't load a schema of type {schema_type_str:?}")
330        });
331        Error!("unknown", "schematype", message)
332      }
333    },
334  };
335
336  if !model.no_compiled {
337    let paths: Option<Vec<String>> = if search_paths.is_empty() {
338      None
339    } else {
340      Some(search_paths.iter().map(ToString::to_string).collect())
341    };
342    let pathname_opt = pathname::find(&name, pathname::PathnameFindOptions {
343      paths,
344      extensions: Some(vec![s!("model")]),
345      installation_subdir: Some(arena::with(schema_type, |str| s!("resources/{str}"))),
346    });
347
348    match pathname_opt {
349      Some(compiled_path) => model.load_compiled_schema(&compiled_path),
350      None => model.schema.as_mut().unwrap().load_schema(),
351    };
352  }
353  model.load_internal_extensions();
354  if model.debug_mode {
355    model.describe_model()
356  }
357
358  Ok(())
359}

361pub fn get_document_namespace_prefix(
362  namespace: &str,
363  forattribute: bool,
364  probe: bool,
365) -> Option<SymStr> {
366  // Get the prefix associated with the namespace url, noting that for elements, it might by
367  // "#default", but for attributes would never be.
368  // log!("Searching for {:?} in {:?}", namespace, self.document_namespace_prefixes);
369  let mut docprefix = if !forattribute {
370    model!()
371      .document_namespace_prefixes
372      .get(&s!("DEFAULT#{namespace}"))
373      .copied()
374  } else {
375    None
376  };
377  let ns_sym = arena::pin(namespace);
378  if docprefix.is_none() {
379    docprefix = model!()
380      .document_namespace_prefixes
381      .get_sym(ns_sym)
382      .copied();
383  }
384
385  if docprefix.is_none() && !probe {
386    {
387      model_mut!().namespace_errors += 1;
388    }
389    let ns_err = s!("namespace{}", &model!().namespace_errors.to_string());
390    docprefix = Some(arena::pin(&ns_err));
391    {
392      model_mut!().register_document_namespace(&ns_err, Some(namespace));
393    }
394    let message2 = if let Some(dp) = docprefix {
395      arena::with(dp, |dp_str| s!("Using '{dp_str}' instead"))
396    } else {
397      String::from("No prefix to fall back on.")
398    };
399    Warn!(
400      "malformed",
401      namespace,
402      "No prefix has been registered for namespace (in document)",
403      message2
404    );
405  }
406  let default_sym = *H_DEFAULT_SYM;
407  docprefix.filter(|p| p != &default_sym)
408}

410pub fn get_document_namespace(docprefix: &str, probe: bool) -> Option<String> {
411  let h_default_sym = *H_DEFAULT_SYM;
412  let docprefix_sym = if docprefix.is_empty() {
413    h_default_sym
414  } else {
415    arena::pin(docprefix)
416  };
417  let ns_str = match model!().document_namespaces.get_sym(docprefix_sym) {
418    None => String::new(),
419    Some(sym) => arena::with(*sym, |s| {
420      if s.starts_with("DEFAULT#") {
421        s.replacen("DEFAULT#", "", 1)
422      } else {
423        s.to_string()
424      }
425    }),
426  };
427
428  if docprefix_sym != h_default_sym && ns_str.is_empty() && !probe {
429    {
430      model_mut!().namespace_errors += 1;
431    }
432    let ns_error = s!(
433      "http://example.com/namespace{}",
434      &model!().namespace_errors.to_string()
435    );
436    {
437      model_mut!().register_document_namespace(docprefix, Some(&ns_error));
438    }
439    let msg1 = arena::with(docprefix_sym, |dp_str| {
440      s!("No namespace has been registered for prefix '{dp_str}' (in document)")
441    });
442    let msg2 = s!("Using '{ns_str}' instead");
443    let err = || {
444      Error!("malformed", docprefix, msg1, msg2);
445      Ok(())
446    };
447    err().ok();
448  }
449  if ns_str.is_empty() {
450    None
451  } else {
452    Some(ns_str)
453  }
454}

456/// Get the (code) prefix associated with $namespace,
457/// creating a dummy prefix and signalling an error if none has been registered.
458///
459/// In the following:
460/// $forattribute is 1 if the namespace is for an attribute (in which case, there must be a
461/// non-empty prefix) $probe, if non 0, just test for namespace, without creating an entry
462/// if missing.
463pub fn get_namespace_prefix(namespace: &str, _forattribute: bool, probe: bool) -> Option<SymStr> {
464  let mut codeprefix: Option<SymStr> = None;
465  let ns_sym = arena::pin(namespace);
466  let mut model = model_mut!();
467  if !namespace.is_empty() {
468    codeprefix = model.code_namespace_prefixes.get_sym(ns_sym).copied();
469
470    if codeprefix.is_some() && !probe {
471      {
472        let docprefix = model.document_namespace_prefixes.get_sym(ns_sym);
473        // if there's a doc prefix and it's NOT already used in code namespace mapping
474        if docprefix.is_some() && !model.code_namespaces.contains_key_sym(docprefix.unwrap()) {
475          codeprefix = docprefix.copied();
476        }
477      }
478    } else {
479      // Else synthesize one
480      model.namespace_errors += 1;
481      let auto_prefix = arena::pin(s!("namespace{}", &model.namespace_errors.to_string()));
482      codeprefix = Some(auto_prefix);
483    }
484    model.register_namespace_sym(codeprefix.unwrap(), Some(arena::pin(namespace)));
485    // Warn!('malformed', $namespace, undef,
486    //   "No prefix has been registered for namespace '$namespace' (in code)",
487    //   "Using '$codeprefix' instead"); }
488  }
489
490  codeprefix
491}

493pub fn get_namespace(codeprefix: &str, probe: bool) -> Result<Option<SymStr>> {
494  let mut model = model_mut!();
495  let mut ns: Option<SymStr> = model.code_namespaces.get(codeprefix).copied();
496  if ns.is_none() && !probe {
497    model.namespace_errors += 1;
498    let example_namespace = s!(
499      "http://example.com/namespace{}",
500      &model.namespace_errors.to_string()
501    );
502    ns = Some(arena::pin(&example_namespace));
503    model.register_namespace(codeprefix, Some(&example_namespace));
504    Error!(
505      "malformed",
506      codeprefix,
507      s!("No namespace has been registered for prefix '{codeprefix}' (in code)"),
508      s!("Using '{example_namespace}' instead")
509    );
510  }
511  Ok(ns)
512}

567/// Same as get_node_qname, but using the Document namespace prefixes
568pub fn get_node_document_qname(node: &Node) -> SymStr {
569  use libxml::tree::NodeType::*;
570  let node_type = node.get_type();
571  if node_type.is_none() {
572    return arena::pin_static("#BrokenNode");
573  }
574
575  match node_type.unwrap() {
576    TextNode => arena::pin_static("#PCDATA"),
577    DocumentNode => arena::pin_static("#Document"),
578    CommentNode => arena::pin_static("#Comment"),
579    PiNode => arena::pin_static("#ProcessingInstruction"),
580    DTDNode => arena::pin_static("#DTD"),
581
582    // TODO
583    // elsif ($type == XML_NAMESPACE_DECL) {
584    //   my $ns = $node->declaredURI;
585    //   my $prefix = $ns && $self->getDocumentNamespacePrefix($ns, 0, 1);
586    //   return ($prefix ? 'xmlns:' . $prefix : 'xmlns'); }
587    NamespaceDecl => arena::pin_static("xmlns"),
588
589    ElementNode | AttributeNode => {
590      let empty_sym = *EMPTY_SYM;
591      let mut prefix = empty_sym;
592      if let Some(ns) = node.get_namespace() {
593        let href = ns.get_href();
594        if !href.is_empty() {
595          prefix = get_document_namespace_prefix(&href, false, true).unwrap_or(empty_sym);
596        }
597      }
598      if prefix == empty_sym {
599        arena::pin(node.get_name())
600      } else {
601        arena::pin(arena::with(prefix, |prefix_str| {
602          s!("{}:{}", prefix_str, node.get_name())
603        }))
604      }
605    },
606    // Need others?
607    t => {
608      panic!("Fatal:misdefined:<caller> should not ask for qualified name for node of type {t:?}")
609    },
610  }
611}

613/// Given a Qualified name, possibly prefixed with a namespace prefix,
614/// as defined by the code namespace mapping,
615/// return the NamespaceURI and localname.
616pub fn decode_qname(codetag: &str) -> Result<(Option<String>, String)> {
617  match PREFIXED_LOCALNAME_RE.captures(codetag) {
618    Some(captures) => {
619      let prefix = captures.get(1).map_or("", |m| m.as_str());
620      let localname = captures.get(2).map_or("", |m| m.as_str());
621
622      if prefix == "xml" {
623        Ok((None, codetag.to_string()))
624      } else {
625        Ok((
626          get_namespace(prefix, false)?.map(arena::to_string),
627          localname.to_string(),
628        ))
629      }
630    },
631    None => Ok((None, codetag.to_string())),
632  }
633}

649/// A check for allowed direct element containment, using ticket-based `SymStr` names.
650///
651/// TODO: This is a major code smell, experimental prototyping to see how to interoperate
652/// strings with the inerned arena.
653/// `can_contain` and `can_contain_sym` should be implemented once, and one should be an
654/// interning-only helper.
655pub fn can_contain_sym(tag: SymStr, child: SymStr) -> bool {
656  // Handle obvious cases explicitly.
657  if tag == *H_PCDATA_SYM || tag == *H_COMMENT_SYM || tag == *EMPTY_SYM {
658    return false;
659  } else if tag == *WILD_CARD_SYM {
660    return true;
661  };
662  if arena::with(tag, |tag_str| tag_str.ends_with("_Capture_"))
663    || arena::with(child, |child_str| {
664      child_str.ends_with("_Capture_") || child_str.ends_with("_CaptureBlock_")
665    })
666  {
667    // with or without namespace prefix
668    return true;
669  }
670
671  if child == *WILD_CARD_SYM || child == *H_COMMENT_SYM || child == *H_PI_SYM || child == *H_DTD_SYM
672  {
673    return true;
674  }
675
676  let mut model = model_mut!();
677  if model.permissive && tag == *H_DOC_SYM && child != *H_PCDATA_SYM {
678    return true; // No DTD? Punt!
679  }
680
681  // Else query tag properties.
682  let model_entry = &mut model.tagprop.entry_sym(tag).or_default().model;
683  model_entry.contains(&ANY_SYM) || model_entry.contains(&child)
684}

686/// Can an element with (qualified name) `tag` contain a `child` element?
687pub fn can_contain(tag: &str, child: &str) -> bool {
688  // Handle obvious cases explicitly.
689  match tag {
690    "#PCDATA" | "#Comment" | "" => return false,
691    "_WildCard_" => return true,
692    _ => {},
693  };
694  if tag.ends_with("_Capture_") || child.ends_with("_Capture_") || tag.ends_with("_CaptureBlock_") {
695    // with or without namespace prefix
696    return true;
697  }
698
699  match child {
700    "_WildCard_" | "#Comment" | "#ProcessingInstruction" | "#DTD" => return true,
701    _ => {},
702  };
703  let model = model!();
704  if model.permissive && tag == "#Document" && child != "#PCDATA" {
705    return true; // No DTD? Punt!
706  }
707
708  // Else query tag properties.
709  let model = &model.tagprop.get(tag).unwrap_or(&*DEFAULT_TAG_FRAME).model;
710  model.contains(&ANY_SYM) || model.contains(&arena::pin(child))
711}

713pub fn can_have_attribute(tag: SymStr, attrib: SymStr) -> bool {
714  // Handle obvious cases explicitly.
715  if let Some(early_choice) = arena::with(tag, |tag_str| match tag_str {
716    "#PCDATA" | "#Comment" | "#Document" | "#ProcessingInstruction" | "#DTD" => Some(false),
717    "_WildCard_" => Some(true),
718    other if other.ends_with("_Capture_") => Some(true),
719    _ => None,
720  }) {
721    return early_choice;
722  };
723  let model = model!();
724  if model.permissive {
725    return true;
726  }
727
728  // Else query tag properties.
729  let attributes = &model
730    .tagprop
731    .get_sym(tag)
732    .unwrap_or(&*DEFAULT_TAG_FRAME)
733    .attributes;
734  attributes.contains(&attrib)
735}

785pub(crate) fn compute_indirect_model_aux(
786  tag: SymStr,
787  start_opt: Option<SymStr>,
788  desirability: usize,
789  openable: &mut HashSet<SymStr>,
790  desc: &mut SymHashMap<SymHashMap<usize>>,
791) {
792  let start = match start_opt {
793    Some(s) => s,
794    None => *EMPTY_SYM,
795  };
796
797  // A bit tricky here, we need to release the model_mut!() borrow immediately, which is why we
798  // move ownership of the tag strings into the tag_contents vector.
799  // That leads to a bunch of .clone()s later one, but stays close to the original algorithm
800  let tag_contents: Vec<SymStr> = get_tag_contents(tag);
801
802  for kid in tag_contents {
803    if desc.entry_sym(kid).or_default().contains_key_sym(&start) {
804      continue;
805    } // Already solved
806
807    if start != *EMPTY_SYM {
808      desc
809        .entry_sym(kid)
810        .or_default()
811        .insert_sym(start, desirability);
812    }
813
814    if kid != *H_PCDATA_SYM && openable.contains(&kid) {
815      let inner = if start != *EMPTY_SYM { start } else { kid };
816
817      compute_indirect_model_aux(kid, Some(inner), desirability, openable, desc);
818    }
819  }
820}
