25static PREFIXED_LOCALNAME_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^([^:]+):(.+)$").unwrap());
26static TAG_MODEL_LINE_RE: Lazy<Regex> =
27  Lazy::new(|| Regex::new(r"^([^\{]+)\{(.*?)\}\((.*?)\)$").unwrap());

30// `classname:=(elt1,elt2,...)` so the elements split cleanly.

33static NAMESPACE_MODEL_LINE_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^([^=]+)=(.*?)$").unwrap());

91  /// There are TWO namespace mappings!!!
92  /// One for coding, one for the document output.
93  ///
94  /// Coding: this namespace mapping associates prefixes to namespace URIs for
95  ///   use in the latexml code, constructors and such.
96  ///   This must be a one to one mapping and there are no default namespaces.
97  /// Document: this namespace mapping associates prefixes to namespace URIs
98  ///   as used in the generated document, and will be the
99  ///   set of prefixes used in the generated output.
100  ///   This mapping may also use a prefix of "#default" which is for
101  ///   the unprefixed form of elements (not used for attributes!)
102  pub fn register_namespace(&mut self, codeprefix: &str, namespace_opt: Option<&str>) {
103    self.register_namespace_sym(arena::pin(codeprefix), namespace_opt.map(arena::pin))
104  }
105  pub fn register_namespace_sym(&mut self, codeprefix: SymStr, namespace_opt: Option<SymStr>) {
106    // double-check empty strings are None
107    let namespace_opt_checked = namespace_opt.filter(|val| *val != pin!(""));
108    match namespace_opt_checked {
109      Some(namespace) => {
110        self
111          .code_namespace_prefixes
112          .insert_sym(namespace, codeprefix);
113        self.code_namespaces.insert_sym(codeprefix, namespace);
114      },
115      None => {
116        if let Some(prev) = self.code_namespaces.get_sym(codeprefix) {
117          self.code_namespace_prefixes.remove_sym(*prev);
118        };
119        self.code_namespaces.remove_sym(codeprefix);
120      },
121    };
122  }

124  pub fn register_document_namespace(&mut self, docprefix: &str, namespace_opt: Option<&str>) {
125    let default_sym = pin!("#default");
126    let docprefix_sym = if docprefix.is_empty() {
127      default_sym
128    } else {
129      arena::pin(docprefix)
130    };
131
132    match namespace_opt {
133      Some(namespace) => {
134        // Since the default namespace url can still ALSO have a prefix associated,
135        // we prepend "DEFAULT#url" when using as a hash key in the prefixes table.
136        let ns_sym = arena::pin(namespace);
137        let regnamespace = if docprefix_sym == default_sym {
138          arena::pin(s!("DEFAULT#{namespace}"))
139        } else {
140          ns_sym
141        };
142        self
143          .document_namespace_prefixes
144          .insert_sym(regnamespace, docprefix_sym);
145        self.document_namespaces.insert_sym(docprefix_sym, ns_sym);
146      },
147      None => {
148        if let Some(prev) = self.document_namespaces.get_sym(docprefix_sym) {
149          self.document_namespace_prefixes.remove_sym(*prev);
150        };
151        self.document_namespaces.remove_sym(docprefix_sym);
152      },
153    };
154  }

162  pub fn load_compiled_schema(&mut self, path: &str) {
163    let compiled_fh = File::open(path).unwrap();
164    let compiled_reader = BufReader::new(&compiled_fh);
165    let content: String = compiled_reader
166      .lines()
167      .map_while(std::result::Result::ok)
168      .collect::<Vec<_>>()
169      .join("\n");
170    self.load_compiled_schema_str(&content, path);
171  }

173  /// Same as [`Self::load_compiled_schema`] but consumes an already-
174  /// loaded `.model` body. `source` is used purely for diagnostic
175  /// messages (the `note_begin`/`note_end` envelope and the
176  /// malformed-line panic). Avoids the disk read when the model is
177  /// served from the binary's own embedded RelaxNG table.
178  pub fn load_compiled_schema_str(&mut self, content: &str, source: &str) {
179    note_begin(&s!("Loading compiled schema {}\n", source));
180    for line in content.lines() {
181      if let Some(caps) = TAG_MODEL_LINE_RE.captures(line) {
182        let tag = caps.get(1).map_or("", |m| m.as_str());
183        let attr = caps.get(2).map_or("", |m| m.as_str());
184        let children = caps.get(3).map_or("", |m| m.as_str());
185        self.add_tag_attribute(tag, attr.split(',').collect());
186        self.add_tag_content(tag, children.split(',').collect());
187      } else if let Some(caps) = CLASS_MODEL_LINE_RE.captures(line) {
188        let classname = caps.get(1).map_or("", |m| m.as_str());
189        let elements = caps.get(2).map_or("", |m| m.as_str());
190        let mut class_set = HashSet::default();
191        for set_element in elements.split(',') {
192          class_set.insert(arena::pin(set_element));
193        }
194        self.set_schema_class(classname, class_set);
195      } else if let Some(caps) = NAMESPACE_MODEL_LINE_RE.captures(line) {
196        let prefix = caps.get(1).map_or("", |m| m.as_str());
197        let namespace = caps.get(2).map_or("", |m| m.as_str());
198        self.register_document_namespace(prefix, Some(namespace));
199      } else {
200        panic!("Fatal:internal:{source} Compiled model '{source}' is malformatted at \"{line}\"");
201      }
202    }
203    note_end(&s!("Loading compiled schema {source}\n"));
204  }

236  /// Serialise the loaded schema into the `.model` plain-text format
237  /// emitted by Perl `LaTeXML::Common::Model::compileSchema`
238  /// (Model.pm L121-136). Three kinds of lines, all newline-separated:
239  ///
240  /// * `prefix=namespace` for every entry in `document_namespaces` (sorted by prefix).
241  /// * `classname:=(elt1,elt2,...)` for every entry in `schema_class` (sorted by classname; each
242  ///   element list sorted).
243  /// * `tag{attr1,attr2}(child1,child2)` for every entry in `tagprop` (sorted by tag; attrs and
244  ///   children sorted; tags whose name starts with `!` are skipped — they are content-model-only
245  ///   negations).
246  ///
247  /// Output is identical to the Perl tool so a downstream
248  /// `tools/compileschema.sh` can diff Rust vs. Perl-generated
249  /// `LaTeXML.model` files byte-for-byte (modulo schema content).
250  pub fn dump_compiled_schema(&self) -> String {
251    fn sym_to_string(sym: SymStr) -> String { arena::with(sym, |s| s.to_string()) }
252    fn syms_sorted(set: impl IntoIterator<Item = SymStr>) -> Vec<String> {
253      let mut v: Vec<String> = set.into_iter().map(sym_to_string).collect();
254      v.sort();
255      v
256    }
257    let mut out = String::new();
258    let prefixes = syms_sorted(self.document_namespaces.keys().copied());
259    for prefix in &prefixes {
260      let ns_opt = self
261        .document_namespaces
262        .get_sym(arena::pin(prefix.as_str()));
263      let ns = match ns_opt {
264        Some(v) => sym_to_string(*v),
265        None => continue,
266      };
267      out.push_str(prefix);
268      out.push('=');
269      out.push_str(&ns);
270      out.push('\n');
271    }
272    let classnames = syms_sorted(self.schema_class.keys().copied());
273    for classname in &classnames {
274      let elements = match self.schema_class.get_sym(arena::pin(classname.as_str())) {
275        Some(set) => set,
276        None => continue,
277      };
278      let elt_names = syms_sorted(elements.iter().copied());
279      out.push_str(classname);
280      out.push_str(":=(");
281      out.push_str(&elt_names.join(","));
282      out.push_str(")\n");
283    }
284    let tags = syms_sorted(self.tagprop.keys().copied());
285    for tag in &tags {
286      if tag.starts_with('!') {
287        continue;
288      }
289      let frame = match self.tagprop.get_sym(arena::pin(tag.as_str())) {
290        Some(f) => f,
291        None => continue,
292      };
293      let attrs = syms_sorted(frame.attributes.iter().copied());
294      let children = syms_sorted(frame.model.iter().copied());
295      out.push_str(tag);
296      out.push('{');
297      out.push_str(&attrs.join(","));
298      out.push_str("}(");
299      out.push_str(&children.join(","));
300      out.push_str(")\n");
301    }
302    out
303  }

329  /// Clone the tagprop's (allowed content & attributes) of @other to $tag
330  fn synthesize_element(&mut self, tag: &str, others: &[&str]) {
331    let mut to_add_in_model = Vec::new();
332    let mut to_add_in_attrs = Vec::new();
333    for other in others {
334      if let Some(content) = self.schema_class.get(other) {
335        for child in content {
336          to_add_in_model.push(*child);
337        }
338      } else if let Some(entry) = self.tagprop.get(other) {
339        for child in &entry.model {
340          to_add_in_model.push(*child);
341        }
342        for attr in &entry.attributes {
343          to_add_in_attrs.push(*attr);
344        }
345      }
346    }
347    let capture = self.tagprop.entry(tag).or_default();
348    for child in to_add_in_model {
349      capture.model.insert(child);
350    }
351    for attr in to_add_in_attrs {
352      capture.attributes.insert(attr);
353    }
354  }

364pub fn load_schema(search_paths: &[&str]) -> Result<()> {
365  // Only load once
366  let mut model = model_mut!();
367  if model.schema.is_some() {
368    return Ok(());
369  }
370  let mut name = String::new();
371  if model.schema_data.is_none() {
372    // TODO: Return this code path to normal once we properly load schemas
373    Warn!("expected", "<model>", "TODO");
374    // Warn('expected', '<model>', undef, "No Schema Model has been declared; assuming LaTeXML");
375    // // article ??? or what ? undef gives problems!
376
377    model.register_document_namespace("ltx", Some(LTX_NAMESPACE));
378    model.set_relaxng_schema("LaTeXML");
379    model.register_namespace("ltx", Some(LTX_NAMESPACE));
380    model.register_namespace("svg", Some("http://www.w3.org/2000/svg"));
381    model.register_namespace("xlink", Some("http://www.w3.org/1999/xlink")); // Needed for SVG
382    model.register_namespace("m", Some("http://www.w3.org/1998/Math/MathML"));
383    model.register_namespace("xhtml", Some("http://www.w3.org/1999/xhtml"));
384    model.permissive = true;
385  } // Actually, they could have declared all sorts of Tags....
386  // Only RelaxNG schemas are supported (DTD support removed from Rust port)
387  if let Some(ref data) = model.schema_data {
388    if data[0] == pin!("RelaxNG") {
389      name = arena::to_string(data[1]);
390      model.schema = Some(Relaxng {
391        name: name.clone(),
392        ..Relaxng::default()
393      });
394    } else {
395      let message = arena::with(data[0], |schema_type_str| {
396        s!("Can't load a schema of type {schema_type_str:?}")
397      });
398      Error!("unknown", "schematype", message)
399    }
400  }
401
402  if !model.no_compiled && model.schema.is_some() {
403    let paths: Option<Vec<String>> = if search_paths.is_empty() {
404      None
405    } else {
406      Some(search_paths.iter().map(ToString::to_string).collect())
407    };
408    let pathname_opt = pathname::find(&name, pathname::PathnameFindOptions {
409      paths,
410      extensions: Some(vec![s!("model")]),
411      installation_subdir: Some(s!("resources/RelaxNG")),
412    });
413
414    match pathname_opt {
415      Some(compiled_path) => model.load_compiled_schema(&compiled_path),
416      None => {
417        // No `.model` on the searchpath. Try the embedded RelaxNG
418        // table for a same-named compiled schema first (the
419        // distribution path — binary running outside the source
420        // tree); fall through to raw `.rng` parsing if that misses
421        // too.
422        let embed_key = format!("{}.model", name);
423        if let Some(content) = crate::common::relaxng::embedded::lookup(&embed_key) {
424          model.load_compiled_schema_str(content, &format!("<embedded>/{embed_key}"));
425        } else {
426          let paths: Vec<&std::path::Path> =
427            search_paths.iter().map(std::path::Path::new).collect();
428          let schema = model.schema.as_mut().unwrap();
429          let schema_name = schema.name.clone();
430          if let Err(err) = schema.load_schema(&schema_name, &paths) {
431            let msg = format!("load_schema failed for {}: {}", schema_name, err);
432            Warn!("expected", "RelaxNG", msg);
433          }
434        }
435      },
436    };
437  }
438  model.load_internal_extensions();
439  if model.debug_mode {
440    model.describe_model()
441  }
442
443  Ok(())
444}

446pub fn get_document_namespace_prefix(
447  namespace: &str,
448  forattribute: bool,
449  probe: bool,
450) -> Option<SymStr> {
451  // Get the prefix associated with the namespace url, noting that for elements, it might by
452  // "#default", but for attributes would never be.
453  // log!("Searching for {:?} in {:?}", namespace, self.document_namespace_prefixes);
454  let mut docprefix = if !forattribute {
455    model!()
456      .document_namespace_prefixes
457      .get(&s!("DEFAULT#{namespace}"))
458      .copied()
459  } else {
460    None
461  };
462  let ns_sym = arena::pin(namespace);
463  if docprefix.is_none() {
464    docprefix = model!()
465      .document_namespace_prefixes
466      .get_sym(ns_sym)
467      .copied();
468  }
469
470  if docprefix.is_none() && !probe {
471    {
472      model_mut!().namespace_errors += 1;
473    }
474    let ns_err = s!("namespace{}", &model!().namespace_errors.to_string());
475    docprefix = Some(arena::pin(&ns_err));
476    {
477      model_mut!().register_document_namespace(&ns_err, Some(namespace));
478    }
479    let message2 = if let Some(dp) = docprefix {
480      arena::with(dp, |dp_str| s!("Using '{dp_str}' instead"))
481    } else {
482      String::from("No prefix to fall back on.")
483    };
484    Warn!(
485      "malformed",
486      namespace,
487      "No prefix has been registered for namespace (in document)",
488      message2
489    );
490  }
491  let default_sym = pin!("#default");
492  docprefix.filter(|p| p != &default_sym)
493}

495pub fn get_document_namespace(docprefix: &str, probe: bool) -> Option<String> {
496  let h_default_sym = pin!("#default");
497  let docprefix_sym = if docprefix.is_empty() {
498    h_default_sym
499  } else {
500    arena::pin(docprefix)
501  };
502  let ns_str = match model!().document_namespaces.get_sym(docprefix_sym) {
503    None => String::new(),
504    Some(sym) => arena::with(*sym, |s| {
505      if s.starts_with("DEFAULT#") {
506        s.replacen("DEFAULT#", "", 1)
507      } else {
508        s.to_string()
509      }
510    }),
511  };
512
513  if docprefix_sym != h_default_sym && ns_str.is_empty() && !probe {
514    {
515      model_mut!().namespace_errors += 1;
516    }
517    let ns_error = s!(
518      "http://example.com/namespace{}",
519      &model!().namespace_errors.to_string()
520    );
521    {
522      model_mut!().register_document_namespace(docprefix, Some(&ns_error));
523    }
524    let msg1 = arena::with(docprefix_sym, |dp_str| {
525      s!("No namespace has been registered for prefix '{dp_str}' (in document)")
526    });
527    let msg2 = s!("Using '{ns_str}' instead");
528    let err = || {
529      Error!("malformed", docprefix, msg1, msg2);
530      Ok(())
531    };
532    err().ok();
533  }
534  if ns_str.is_empty() {
535    None
536  } else {
537    Some(ns_str)
538  }
539}

541/// Get the (code) prefix associated with $namespace,
542/// creating a dummy prefix and signalling an error if none has been registered.
543///
544/// In the following:
545/// $forattribute is 1 if the namespace is for an attribute (in which case, there must be a
546/// non-empty prefix) $probe, if non 0, just test for namespace, without creating an entry
547/// if missing.
548pub fn get_namespace_prefix(namespace: &str, _forattribute: bool, probe: bool) -> Option<SymStr> {
549  let mut codeprefix: Option<SymStr> = None;
550  let ns_sym = arena::pin(namespace);
551  let mut model = model_mut!();
552  if !namespace.is_empty() {
553    codeprefix = model.code_namespace_prefixes.get_sym(ns_sym).copied();
554
555    if codeprefix.is_some() && !probe {
556      {
557        let docprefix = model.document_namespace_prefixes.get_sym(ns_sym);
558        // if there's a doc prefix and it's NOT already used in code namespace mapping
559        if docprefix.is_some() && !model.code_namespaces.contains_key_sym(docprefix.unwrap()) {
560          codeprefix = docprefix.copied();
561        }
562      }
563    } else {
564      // Else synthesize one
565      model.namespace_errors += 1;
566      let auto_prefix = arena::pin(s!("namespace{}", &model.namespace_errors.to_string()));
567      codeprefix = Some(auto_prefix);
568    }
569    model.register_namespace_sym(codeprefix.unwrap(), Some(arena::pin(namespace)));
570    // Warn!('malformed', $namespace, undef,
571    //   "No prefix has been registered for namespace '$namespace' (in code)",
572    //   "Using '$codeprefix' instead"); }
573  }
574
575  codeprefix
576}

578pub fn get_namespace(codeprefix: &str, probe: bool) -> Result<Option<SymStr>> {
579  let mut model = model_mut!();
580  let mut ns: Option<SymStr> = model.code_namespaces.get(codeprefix).copied();
581  if ns.is_none() && !probe {
582    model.namespace_errors += 1;
583    let example_namespace = s!(
584      "http://example.com/namespace{}",
585      &model.namespace_errors.to_string()
586    );
587    ns = Some(arena::pin(&example_namespace));
588    model.register_namespace(codeprefix, Some(&example_namespace));
589    Error!(
590      "malformed",
591      codeprefix,
592      s!("No namespace has been registered for prefix '{codeprefix}' (in code)"),
593      s!("Using '{example_namespace}' instead")
594    );
595  }
596  Ok(ns)
597}

599/// Get the node's qualified name in standard form
600/// Ie. using the registered (code) prefix for that namespace.
601/// NOTE: Reconsider how _Capture_ & _WildCard_ should be integrated!?!
602pub fn get_node_qname(node: &Node) -> SymStr {
603  use libxml::tree::NodeType::*;
604  let node_type = node.get_type();
605  if node_type.is_none() {
606    return arena::pin_static("#BrokenNode");
607  }
608  match node_type.unwrap() {
609    TextNode => arena::pin_static("#PCDATA"),
610    DocumentNode => arena::pin_static("#Document"),
611    CommentNode => arena::pin_static("#Comment"),
612    PiNode => arena::pin_static("#ProcessingInstruction"),
613    DTDNode => arena::pin_static("#DTD"),
614    NamespaceDecl => {
615      // match node.declared_uri() {
616      //   Some(ns) => match self.get_namespace_prefix(ns, false, true) {
617      //     Some(prefix) => s!("xmlns:")+prefix,
618      //     None => s!("xmlns")
619      //   },
620      //   None => s!("xmlns")
621      // }
622      arena::pin_static("xmlns")
623    },
624    ElementNode | AttributeNode => {
625      let name_str = node.get_name();
626      // Use the actual namespace prefix from the node when available.
627      // For elements in the default (ltx) namespace, the prefix is empty
628      // so we prepend "ltx:". For SVG/MathML/etc with explicit prefix,
629      // use that prefix directly.
630      if let Some(ns) = node.get_namespace() {
631        let prefix = ns.get_prefix();
632        if prefix.is_empty() {
633          // Default namespace — use ltx: prefix
634          arena::pin(s!("ltx:{}", name_str))
635        } else {
636          // Explicit prefix (e.g., "svg", "m") — use it
637          arena::pin(s!("{}:{}", prefix, name_str))
638        }
639      } else {
640        // No namespace — special cases for non-namespaced elements
641        match name_str.as_str() {
642          "song" | "verse" => arena::pin(name_str),
643          regular => arena::pin(s!("ltx:{}", regular)),
644        }
645      }
646    },
647    // Need others?
648    t => {
649      panic!("Fatal:misdefined:<caller> should not ask for qualified name for node of type {t:?}")
650    },
651  }
652}

654pub fn with_node_qname<R, FnR>(node: &Node, caller: FnR) -> R
655where FnR: FnOnce(&str) -> R {
656  let qsym = get_node_qname(node);
657  arena::with(qsym, |qname_str| caller(qname_str))
658}

661pub fn get_node_document_qname(node: &Node) -> SymStr {
662  use libxml::tree::NodeType::*;
663  let node_type = node.get_type();
664  if node_type.is_none() {
665    return arena::pin_static("#BrokenNode");
666  }
667
668  match node_type.unwrap() {
669    TextNode => arena::pin_static("#PCDATA"),
670    DocumentNode => arena::pin_static("#Document"),
671    CommentNode => arena::pin_static("#Comment"),
672    PiNode => arena::pin_static("#ProcessingInstruction"),
673    DTDNode => arena::pin_static("#DTD"),
674
675    // TODO
676    // elsif ($type == XML_NAMESPACE_DECL) {
677    //   my $ns = $node->declaredURI;
678    //   my $prefix = $ns && $self->getDocumentNamespacePrefix($ns, 0, 1);
679    //   return ($prefix ? 'xmlns:' . $prefix : 'xmlns'); }
680    NamespaceDecl => arena::pin_static("xmlns"),
681
682    ElementNode | AttributeNode => {
683      let empty_sym = pin!("");
684      let mut prefix = empty_sym;

692        arena::pin(node.get_name())
693      } else {
694        arena::pin(arena::with(prefix, |prefix_str| {
695          s!("{}:{}", prefix_str, node.get_name())
696        }))
697      }
698    },
699    // Need others?
700    t => {
701      panic!("Fatal:misdefined:<caller> should not ask for qualified name for node of type {t:?}")
702    },
703  }

706/// Given a Qualified name, possibly prefixed with a namespace prefix,
707/// as defined by the code namespace mapping,
708/// return the NamespaceURI and localname.
709pub fn decode_qname(codetag: &str) -> Result<(Option<String>, String)> {
710  match PREFIXED_LOCALNAME_RE.captures(codetag) {
711    Some(captures) => {
712      let prefix = captures.get(1).map_or("", |m| m.as_str());
713      let localname = captures.get(2).map_or("", |m| m.as_str());
714
715      if prefix == "xml" {
716        Ok((None, codetag.to_string()))
717      } else {
718        Ok((
719          get_namespace(prefix, false)?.map(arena::to_string),
720          localname.to_string(),
721        ))
722      }
723    },
724    None => Ok((None, codetag.to_string())),
725  }
726}

742/// A check for allowed direct element containment, using ticket-based `SymStr` names.
743///
744/// TODO: This is a major code smell, experimental prototyping to see how to interoperate
745/// strings with the inerned arena.
746/// `can_contain` and `can_contain_sym` should be implemented once, and one should be an
747/// interning-only helper.
748pub fn can_contain_sym(tag: SymStr, child: SymStr) -> bool {
749  // Handle obvious cases explicitly.
750  if tag == pin!("#PCDATA") || tag == pin!("#Comment") || tag == pin!("") {
751    return false;
752  } else if tag == pin!("_WildCard_") {
753    return true;
754  };
755  if arena::with(tag, |tag_str| tag_str.ends_with("_Capture_"))
756    || arena::with(child, |child_str| {
757      child_str.ends_with("_Capture_") || child_str.ends_with("_CaptureBlock_")
758    })
759  {
760    // with or without namespace prefix
761    return true;
762  }
763
764  if child == pin!("_WildCard_")
765    || child == pin!("#Comment")
766    || child == pin!("#ProcessingInstruction")
767    || child == pin!("#DTD")
768  {
769    return true;
770  }
771
772  let mut model = model_mut!();
773  if model.permissive && tag == pin!("#Document") && child != pin!("#PCDATA") {
774    return true; // No schema? Punt!
775  }
776
777  // Else query tag properties.
778  let model_entry = &mut model.tagprop.entry_sym(tag).or_default().model;
779  model_entry.contains(&pin!("ANY")) || model_entry.contains(&child)
780}

782/// Can an element with (qualified name) `tag` contain a `child` element?
783pub fn can_contain(tag: &str, child: &str) -> bool {
784  // Handle obvious cases explicitly.
785  match tag {
786    "#PCDATA" | "#Comment" | "" => return false,
787    "_WildCard_" => return true,
788    _ => {},
789  };
790  if tag.ends_with("_Capture_") || child.ends_with("_Capture_") || tag.ends_with("_CaptureBlock_") {
791    // with or without namespace prefix
792    return true;
793  }
794
795  match child {
796    "_WildCard_" | "#Comment" | "#ProcessingInstruction" | "#DTD" => return true,
797    _ => {},
798  };
799  let model = model!();
800  if model.permissive && tag == "#Document" && child != "#PCDATA" {
801    return true; // No schema? Punt!
802  }
803
804  // Else query tag properties.
805  let model = &model.tagprop.get(tag).unwrap_or(&*DEFAULT_TAG_FRAME).model;
806  model.contains(&pin!("ANY")) || model.contains(&arena::pin(child))
807}

810  // Handle obvious cases explicitly.
811  if let Some(early_choice) = arena::with(tag, |tag_str| match tag_str {
812    "#PCDATA" | "#Comment" | "#Document" | "#ProcessingInstruction" | "#DTD" => Some(false),

826    .tagprop
827    .get_sym(tag)
828    .unwrap_or(&*DEFAULT_TAG_FRAME)
829    .attributes;
830  attributes.contains(&attrib)
831}

881pub(crate) fn compute_indirect_model_aux(
882  tag: SymStr,
883  start_opt: Option<SymStr>,
884  desirability: usize,
885  openability: &mut SymHashMap<u32>,
886  desc: &mut SymHashMap<SymHashMap<usize>>,
887) {
888  let start = match start_opt {
889    Some(s) => s,
890    None => pin!(""),
891  };
892
893  // A bit tricky here, we need to release the model_mut!() borrow immediately, which is why we
894  // move ownership of the tag strings into the tag_contents vector.
895  // That leads to a bunch of .clone()s later one, but stays close to the original algorithm
896  let tag_contents: Vec<SymStr> = get_tag_contents(tag);
897
898  for kid in tag_contents {
899    // Memoise on (kid, start) to bound recursion in cyclic schemas, but
900    // retain the *maximum* desirability observed across paths — the
901    // outer loop in compute_indirect_model picks the highest-scoring
902    // starting tag, so the score stored here must reflect the best path,
903    // not the first one the hashmap iteration happened to surface.
904    //
905    // The prior "first visit wins" behavior (WISDOM #49) caused paralists
906    // test-harness runs to assign `desc[#PCDATA][ltx:text] = 50` when
907    // `contents(text)` iterated `ltx:picture` before `#PCDATA`: the
908    // sub-recursion `text → picture → #PCDATA` inserted 50 first and the
909    // direct `text → #PCDATA` path was skipped, forcing the auto-open
910    // path to pick `<ltx:picture>` instead of `<ltx:text>`.
911    let prior = desc.entry_sym(kid).or_default().get_sym(start).copied();
912    if let Some(prior_d) = prior {
913      if prior_d >= desirability {
914        continue;
915      }
916    }
917
918    if start != pin!("") {
919      desc
920        .entry_sym(kid)
921        .or_default()
922        .insert_sym(start, desirability);
923    }
924
925    if kid != pin!("#PCDATA") {
926      if let Some(priority) = openability.get_sym(kid).copied() {
927        let inner = if start != pin!("") { start } else { kid };
928        // Perl Document.pm L220: `$desirability * $x`. We keep integer
929        // arithmetic (priorities scaled by 100), so this is a scaled multiply.
930        let next_desirability = desirability * (priority as usize) / 100;
931        compute_indirect_model_aux(kid, Some(inner), next_desirability, openability, desc);
932      }
933    }
934  }
935}
