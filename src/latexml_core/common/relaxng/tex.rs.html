1//! Schema-doc TeX emission. Port of `RelaxNG.pm` lines 545–815.
2//!
3//! Walks the simplified AST (in [`Relaxng::modules`] and the lookup
4//! tables populated by [`super::simplify`]) and produces a single
5//! `schema.tex` string consumable by `latexmlman.sty`'s `\schemamodule`
6//! / `\patterndef` / `\elementdef` / `\attrdef` / `\moduleref` /
7//! `\patternref` / `\elementref` macros.
8//!
9//! Behaviour intentionally byte-equivalent (modulo whitespace settled
10//! at the unit-test level) with the Perl `documentModules` for the
11//! same simplified state. The schema-doc-style omissions
12//! (`SKIP_SVG`/`SKIP_ARIA`/`SKIP_XHTML`) live on [`Options`] with the
13//! same defaults as upstream.
14
15// The `to_tex*` methods take `&mut self` because `EmitState` is a
16// state-machine accumulator threaded through the entire walk — depth,
17// per-module counters, and the schema-mappings cache all mutate as
18// emission proceeds. Clippy's default `wrong_self_convention` rule
19// expects `to_*` to consume `self`; that's the wrong shape here.
20#![allow(clippy::wrong_self_convention)]

47/// Emission state — mutated as `document_modules` walks the AST.
48struct EmitState<'a> {
49  rng:               &'a Relaxng,
50  opts:              Options,
51  /// Mirrors Perl's `$$self{defined_patterns}{$name}`.
52  /// `1`  = at least one `\patterndef{name}` already emitted,
53  /// `-1` = at least one `\patternadd{name}` emitted but no `\patterndef` yet.
54  /// Final pass upgrades `\patternadd` → `\patterndefadd` for any -1.
55  defined_patterns: HashMap<String, i8>,
56}

58/// Top-level emission. Returns a single `schema.tex` string.
59pub fn document_modules(rng: &Relaxng, opts: Options) -> String {
60  let mut emit = EmitState { rng, opts, defined_patterns: HashMap::default() };
61  let mut docs = String::new();
62  // Each Module renders as one page (`--splitat=section`), regardless
63  // of def count. Page-size is mitigated client-side by CSS lazy
64  // paint and a JS search/filter input — splitting the module across
65  // multiple pages would break Ctrl-F across the whole module, which
66  // is the primary navigation affordance.
67  for module in &rng.modules {
68    let (op, name, content) = match module {
69      Pattern::Module { name, body } => ("module", name.clone(), body),
70      _ => continue,
71    };
72    let _ = op;
73    if emit.opts.skip_svg && is_svg_module(&name) {
74      continue;
75    }
76    let mod_name = strip_urn_prefix(&name);
77
78    // Modules typically wrap their content in a single
79    // `Pattern::Grammar`; descend into it so the iteration sees each
80    // individual def directly.
81    let to_emit: Vec<&Pattern> = content
82      .iter()
83      .flat_map(|item| match item {
84        Pattern::Grammar { body, .. } => body.iter().collect::<Vec<_>>(),
85        other => vec![other],
86      })
87      .collect();
88
89    let mut preamble = String::new();
90    // Outer-Grammar "Includes:" preamble line (collected once,
91    // emitted on the first synthetic group's page if partitioning).
92    for item in content {
93      if let Pattern::Grammar { body, .. } = item {
94        let mods: Vec<String> = body
95          .iter()
96          .filter_map(|d| match d {
97            Pattern::Module { name, .. } => Some(name.clone()),
98            _ => None,
99          })
100          .collect();
101        if !mods.is_empty() {
102          let refs: Vec<String> = mods
103            .iter()
104            .map(|m| format!("\\moduleref{{{}}}", clean_tex(m)))
105            .collect();
106          if !preamble.is_empty() {
107            preamble.push('\n');
108          }
109          preamble.push_str(&format!(
110            "\\par\\noindent\\textit{{Includes:}} {}.",
111            refs.join(", ")
112          ));
113        }
114      }
115    }
116
117    // Render each def, preserving source order in `defs`.
118    let mut defs: Vec<String> = Vec::new();
119    for item in &to_emit {
120      // Module entries already accounted for in the Includes line.
121      if matches!(item, Pattern::Module { .. }) {
122        continue;
123      }
124      let rendered = emit.to_tex(item);
125      if rendered.is_empty() {
126        continue;
127      }
128      match item {
129        Pattern::Doc(_) | Pattern::Start { .. } => {
130          if !preamble.is_empty() {
131            preamble.push('\n');
132          }
133          preamble.push_str(&rendered);
134        },
135        _ => defs.push(rendered),
136      }
137    }
138
139    docs.push_str(&format!("\n\\begin{{schemamodule}}{{{}}}", mod_name));
140    if !preamble.is_empty() {
141      docs.push('\n');
142      docs.push_str(&preamble);
143    }
144    let body = defs.join("\n");
145    if !body.is_empty() {
146      docs.push_str(&format!(
147        "\n\\begin{{description}}\n{}\n\\end{{description}}",
148        body
149      ));
150    }
151    docs.push_str("\n\\end{schemamodule}");
152  }
153  // Final pass: any pattern emitted only as `\patternadd` becomes
154  // `\patterndefadd`. Mirrors Perl `$docs =~ s/\\patternadd\{$name\}/\\patterndefadd{$name}/s`
155  // — single substitution per name.
156  let mut keys: Vec<String> = emit
157    .defined_patterns
158    .iter()
159    .filter(|(_, v)| **v < 0)
160    .map(|(k, _)| k.clone())
161    .collect();
162  keys.sort();
163  for name in keys {
164    let from = format!("\\patternadd{{{}}}", name);
165    let to = format!("\\patterndefadd{{{}}}", name);
166    if let Some(idx) = docs.find(&from) {
167      docs.replace_range(idx..idx + from.len(), &to);
168    }
169  }
170  docs
171}

175/// `cleanTeX`: escape `#`, escape `_`, wrap `<...>` in `\texttt{...}`,
176/// strip URN prefix, recognise `#PCDATA`.
177pub fn clean_tex(s: &str) -> String {
178  if s == "#PCDATA" {
179    return String::from(r"\typename{text}");
180  }
181  let mut out = strip_urn_prefix(s);
182  // Order matters: escape # before \texttt{...} expansion (no #s in
183  // the wrapper), and _ at the end so the others' inserted text
184  // doesn't collide.
185  out = out.replace('#', "\\#");
186  out = wrap_angle_text(&out);
187  out = out.replace('_', "\\_");
188  out
189}
190
191/// `cleanTeXName`: clean_tex + strip any leading prefix listed in
192/// `display_strip_prefixes`. The strip list is auto-populated from
193/// the schema's `default namespace` URI (mapped to its prefix) — see
194/// `Relaxng::auto_strip_primary_namespace`. So a LaTeXML schema
195/// (primary `http://dlmf.nist.gov/LaTeXML` → `ltx`) renders
196/// `\elementref{para}` rather than `\elementref{ltx:para}`; an
197/// XHTML-flavoured schema's `xhtml:div` reads as `div`; a MathML
198/// schema's `m:math` reads as `math`.
199fn clean_tex_name(s: &str, strip_prefixes: &[String]) -> String {
200  let cleaned = clean_tex(s);
201  for prefix in strip_prefixes {
202    let with_colon = format!("{}:", prefix);
203    if let Some(rest) = cleaned.strip_prefix(&with_colon) {
204      return rest.to_string();
205    }
206  }
207  cleaned
208}
209
210fn strip_urn_prefix(s: &str) -> String {
211  s.strip_prefix("urn:x-LaTeXML:RelaxNG:").unwrap_or(s).to_string()
212}

214/// Replace each `<TEXT>` substring with `\texttt{TEXT}`.
215fn wrap_angle_text(s: &str) -> String {
216  let mut out = String::with_capacity(s.len());
217  let bytes = s.as_bytes();
218  let mut i = 0;
219  while i < bytes.len() {
220    if bytes[i] == b'<' {
221      if let Some(end) = s[i + 1..].find('>') {
222        let inner = &s[i + 1..i + 1 + end];
223        out.push_str("\\texttt{");
224        out.push_str(inner);
225        out.push('}');
226        i += 1 + end + 1;
227        continue;
228      }
229    }
230    out.push(bytes[i] as char);
231    i += 1;
232  }
233  out
234}

238impl EmitState<'_> {
239  fn to_tex(&mut self, p: &Pattern) -> String {
240    match p {
241      // Trailing blank line so adjacent `<a:documentation>` annotations
242      // (trang emits one per `## comment` block separated by a blank
243      // line) survive into TeX as *paragraph-separated* prose, not a
244      // single run-on. LaTeXML reads a blank line as `\par`.
245      Pattern::Doc(s) => format!("{}\n\n", clean_tex(s)),
246      Pattern::Ref { qname } => self.to_tex_ref(qname),
247      Pattern::Def { combiner, name, body } => {
248        let combiner_label = match combiner {
249          DefCombiner::Group => "",
250          DefCombiner::Choice => "choice",
251          DefCombiner::Interleave => "interleave",
252        };
253        self.to_tex_def(combiner_label, name, body)
254      },
255      Pattern::Element { name, body } => self.to_tex_element(name, body),
256      Pattern::Attribute { name, body } => self.to_tex_attribute(name, body),
257      Pattern::Combination { op, body } => self.to_tex_combination(*op, body),
258      Pattern::Data(t) => format!("\\typename{{{}}}", clean_tex(t)),
259      Pattern::Value(v) => format!("\\attrval{{{}}}", clean_tex(v)),
260      Pattern::Start { body } => {
261        // Module-level <start>: emit as a paragraph, not a description-
262        // list item. The Perl original emitted
263        // `\item[\textit{Start}]\textbf{==}\ root`, but that depended
264        // on living inside the moduledescription environment that the
265        // old section-split layout opened. With per-def subsection
266        // splitting there's no enclosing list at module scope, so
267        // module-preamble notes flow as prose.
268        let (docs, spec) = self.extract_docs(body);
269        let content = spec
270          .iter()
271          .map(|p| self.to_tex(p))
272          .collect::<Vec<_>>()
273          .join(" ");
274        let mut s = format!("\\par\\noindent\\textit{{Start symbol:}} {}", content);
275        if !docs.is_empty() {
276          s.push_str(&format!(" \\par{}", docs));
277        }
278        s
279      },
280      Pattern::Grammar { body, .. } => {
281        // The grammar's leading <include>'s become an "Includes" line
282        // of `\moduleref{…}`s, then the rest of the body (defs, doc,
283        // etc.) flows through normally. Module preamble is paragraph
284        // text — see Pattern::Start above for rationale.
285        let mut mods: Vec<String> = Vec::new();
286        let mut rest: Vec<&Pattern> = Vec::new();
287        for d in body {
288          match d {
289            Pattern::Module { name, .. } => mods.push(name.clone()),
290            other => rest.push(other),
291          }
292        }
293        let mut out = String::new();
294        if !mods.is_empty() {
295          let refs: Vec<String> = mods
296            .iter()
297            .map(|m| format!("\\moduleref{{{}}}", clean_tex(m)))
298            .collect();
299          out.push_str(&format!(
300            "\\par\\noindent\\textit{{Includes:}} {}.\n",
301            refs.join(", ")
302          ));
303        }
304        for r in rest {
305          out.push_str(&self.to_tex(r));
306          out.push('\n');
307        }
308        out
309      },
310      Pattern::Module { name, .. } => {
311        // Standalone Module reference (rare — most are absorbed into
312        // the parent Grammar's "Includes" line). Emit as a brief
313        // paragraph note rather than a list item.
314        if self.opts.skip_svg && is_svg_module(name) {
315          format!(
316            "\\par\\noindent\\textit{{Module}} \\texttt{{{}}} \\textit{{included}}.",
317            clean_tex(name)
318          )
319        } else {
320          format!(
321            "\\par\\noindent\\textit{{Module}} \\moduleref{{{}}} \\textit{{included}}.",
322            clean_tex(name)
323          )
324        }
325      },
326      Pattern::ParentRef { qname } => self.to_tex_ref(qname),
327      Pattern::ElementRef { qname } => format!("\\elementref{{{}}}", clean_tex_name(qname, &self.rng.display_strip_prefixes)),
328      Pattern::Override { module, .. } => self.to_tex(module),
329      Pattern::Text => clean_tex("#PCDATA"),
330    }
331  }

333  fn to_tex_ref(&self, name: &str) -> String {
334    if let Some(el) = self.rng.elementdefs.get(name) {
335      let cleaned = clean_tex_name(el, &self.rng.display_strip_prefixes);
336      if self.opts.skip_xhtml && cleaned == "xhtml:*" {
337        return String::from("\\texttt{xhtml:*}");
338      }
339      return format!("\\elementref{{{}}}", cleaned);
340    }
341    if name.ends_with("_attributes") || name.ends_with("_model") {
342      if let Some(def) = self.rng.defs.get(name) {
343        // Read-only recursion is fine here; we don't mutate state on
344        // the ref-expansion path (Perl doesn't either).
345        let cloned = def.clone();
346        let mut tmp = EmitState {
347          rng:              self.rng,
348          opts:             self.opts,
349          defined_patterns: HashMap::default(),
350        };
351        return tmp.to_tex(&cloned);
352      }
353    }
354    let stripped = strip_first_qualifier(name);
355    if self.opts.skip_svg && stripped == "svg" {
356      return String::from("\\texttt{svg:svg}");
357    }
358    format!("\\patternref{{{}}}", clean_tex(&stripped))
359  }

361  fn to_tex_def(&mut self, combiner: &str, qname: &str, data: &[Pattern]) -> String {
362    if self.opts.skip_aria && qname.contains("aria") {
363      return String::new();
364    }
365    if qname.ends_with("_attributes") || qname.ends_with("_model") {
366      return String::new();
367    }
368    let stripped = strip_first_qualifier(qname);
369    if self.opts.skip_svg && stripped.starts_with("svg") {
370      return String::new();
371    }
372    let cleaned_name = clean_tex(&stripped);
373    let (docs, spec) = self.extract_docs(data);
374
375    // Compact rendering for `X = element a {B} | element b {B} | …`
376    // shapes, where every alternative is a same-bodied element. The
377    // generic path emits one `\elementdef` card per branch wrapped in
378    // `(... ~\textbar~ ...)`; that produces orphan `(`, `|`, `)` text
379    // around the cards once LaTeXML promotes each `\item` into a
380    // sibling list. The compact form lists the names monospaced and
381    // shows the shared body once.
382    if combiner.is_empty() {
383      if let Some((op, elements)) = self.detect_element_choice(&spec) {
384        let (pattern_body, element_defs) =
385          self.render_element_choice(qname, &cleaned_name, op, &elements);
386        if matches!(self.defined_patterns.get(&cleaned_name), Some(v) if *v > 0) {
387          return String::new();
388        }
389        self.defined_patterns.insert(cleaned_name.clone(), 1);
390        let mut out = format!("\\patterndef{{{}}}{{{}}}{{{}}}\n", cleaned_name, docs, pattern_body);
391        for ed in element_defs {
392          out.push_str(&ed);
393        }
394        return out;
395      }
396    }
397
398    let (attr, content) = self.to_tex_body(&spec);
399
400    if !combiner.is_empty() {
401      let mut body = attr;
402      if !content.is_empty() {
403        let sep = if combiner == "choice" { "\\textbar=" } else { "\\&=" };
404        body.push_str(&format!("\\item[{}] {}", sep, content));
405      }
406      self
407        .defined_patterns
408        .entry(cleaned_name.clone())
409        .or_insert(-1);
410      return format!("\\patternadd{{{}}}{{{}}}{{{}}}\n", cleaned_name, docs, body);
411    }
412
413    // Bare def
414    let mut attr = attr;
415    let mut content = content;
416    if attr.is_empty() && cleaned_name.contains("\\_attributes") {
417      attr = String::from("\\item[\\textit{Attributes:}] \\textit{empty}");
418    }
419    if content.is_empty() && cleaned_name.contains("\\_model") {
420      content = String::from("\\textit{empty}");
421    }
422    let mut body = attr;
423    if !content.is_empty() {
424      body.push_str(&format!("\\item[\\textit{{Content}}:] {}", content));
425    }
426    // Expansion line (when defs[qname] is content-shaped and differs).
427    if !cleaned_name.contains("\\_attributes") {
428      if let Some(stored) = self.rng.defs.get(qname) {
429        if self.is_content(stored) && !self.is_attributes(stored) {
430          let (xattr, xcontent) = self.to_tex_body(std::slice::from_ref(stored));
431          // Suppress when the stored form differs from `content` only by
432          // the outer `(...)` wrap that `to_tex_combination(Group)` adds:
433          // for patterns like `anyElement` the def-args path renders the
434          // body unwrapped while the stored-Combination path re-wraps it,
435          // and emitting both yields a near-duplicate Expansion block.
436          let unwrapped = strip_outer_parens(&xcontent);
437          if xattr.is_empty() && !xcontent.is_empty() && xcontent != content && unwrapped != content {
438            body.push_str(&format!("\\item[\\textit{{Expansion}}:] {}", xcontent));
439          }
440        }
441      }
442    }
443    if let Some(uses) = self.symbol_uses(qname) {
444      body.push_str(&format!("\\item[\\textit{{Used by}}:] {}", uses));
445    }
446    if matches!(self.defined_patterns.get(&cleaned_name), Some(v) if *v > 0) {
447      return String::new();
448    }
449    self.defined_patterns.insert(cleaned_name.clone(), 1);
450    // Extract any nested non-wildcard `Pattern::Element` descendants
451    // into sibling `\elementdef` cards so the inline `\elementref`
452    // links in the patterndef body resolve to a card on the same
453    // page. Skip when no descendants — most patterns have none.
454    let extras = collect_element_descendants(&spec);
455    let mut out = format!("\\patterndef{{{}}}{{{}}}{{{}}}\n", cleaned_name, docs, body);
456    for (el_name, el_body) in &extras {
457      let cleaned_el = clean_tex_name(el_name, &self.rng.display_strip_prefixes);
458      let (el_attr, el_content) = self.to_tex_body(el_body);
459      let mut eb = el_attr;
460      if !el_content.is_empty() {
461        eb.push_str(&format!("\\item[\\textit{{Content}}:] {}", el_content));
462      }
463      eb.push_str(&format!(
464        "\\item[\\textit{{Used by}}:] \\patternref{{{}}}",
465        cleaned_name
466      ));
467      out.push_str(&format!("\\elementdef{{{}}}{{}}{{{}}}\n", cleaned_el, eb));
468    }
469    out
470  }

472  fn to_tex_element(&mut self, qname: &str, data: &[Pattern]) -> String {
473    let local = qname.strip_prefix("ltx:").unwrap_or(qname);
474    if self.opts.skip_xhtml && local == "xhtml:*" {
475      return String::new();
476    }
477    // Wildcard element names (`*`, `*:*`, `prefix:*`) come from `<anyName/>`
478    // / `<nsName/>` in the source schema — they describe "an element of
479    // any name", not a real definable element. Render them inline as a
480    // content-model expression so they sit gracefully inside the parent
481    // pattern's body. Emitting `\elementdef{*}{...}` here would inject a
482    // nested definition card (with its own Content/Attribute rows) into
483    // the parent card, which is the rendering bug visible on patterns
484    // like `anyElement`.
485    if is_wildcard_name(qname) {
486      return self.render_inline_element(qname, data);
487    }
488    let cleaned = clean_tex_name(qname, &self.rng.display_strip_prefixes);
489    let (docs, spec) = self.extract_docs(data);
490    let (attr, content) = self.to_tex_body(&spec);
491    let content = if content.is_empty() {
492      String::from("\\typename{empty}")
493    } else {
494      content
495    };
496    let mut body = attr;
497    body.push_str(&format!("\\item[\\textit{{Content}}:] {}", content));
498    if let Some(ename) = self.rng.element_reverse_defs.get(qname) {
499      if let Some(uses) = self.symbol_uses(ename) {
500        body.push_str(&format!("\\item[\\textit{{Used by}}:] {}", uses));
501      }
502    }
503    format!("\\elementdef{{{}}}{{{}}}{{{}}}\n", cleaned, docs, body)
504  }

506  fn to_tex_attribute(&mut self, name: &str, data: &[Pattern]) -> String {
507    let cleaned = clean_tex_name(name, &self.rng.display_strip_prefixes);
508    if let Some(rest) = cleaned.strip_prefix('!') {
509      return format!("\\item[\\textit{{Excluding attribute }}]\\texttt{{{}}}", rest);
510    }
511    // Same wildcard-handling rationale as `to_tex_element`: render inline
512    // so the parent pattern's body doesn't pick up a nested `\attrdef`
513    // item card for a name like `*` or `*:*`.
514    if is_wildcard_name(&cleaned) {
515      return self.render_inline_attribute(&cleaned, data);
516    }
517    let (docs, spec) = self.extract_docs(data);
518    let content = if spec.is_empty() {
519      String::from("\\typename{text}")
520    } else {
521      spec
522        .iter()
523        .map(|p| self.to_tex(p))
524        .collect::<Vec<_>>()
525        .join(" ")
526    };
527    format!("\\attrdef{{{}}}{{{}}}{{{}}}", cleaned, docs, content)
528  }

530  /// Inline rendering of an `<element>` pattern when it sits inside
531  /// another pattern's body — produce text that won't trip LaTeXML's
532  /// `\item` promotion.
533  ///
534  /// For real-name elements (e.g. `xhtml:div`) returns just
535  /// `\elementref{xhtml:div}` — a link to the sibling `\elementdef`
536  /// card that `to_tex_def`'s extraction emits. The body itself
537  /// belongs on that card, not inlined here.
538  ///
539  /// For wildcard names (`*`, `*:*`, `prefix:*`), there is no card to
540  /// link to, so render the wildcard inline as
541  /// `\textit{element}~\texttt{NAME}~\{BODY\}` so the body's content
542  /// model is at least visible somewhere.
543  fn render_inline_element(&mut self, qname: &str, data: &[Pattern]) -> String {
544    if !is_wildcard_name(qname) {
545      return format!("\\elementref{{{}}}", clean_tex_name(qname, &self.rng.display_strip_prefixes));
546    }
547    let cleaned = clean_tex_name(qname, &self.rng.display_strip_prefixes);
548    let (_docs, spec) = self.extract_docs(data);
549    let parts: Vec<String> = spec.iter().map(|p| self.to_tex(p)).collect();
550    let parts: Vec<String> = parts.into_iter().filter(|s| !s.is_empty()).collect();
551    if parts.is_empty() {
552      format!("\\textit{{element}}~\\texttt{{{}}}", cleaned)
553    } else {
554      format!(
555        "\\textit{{element}}~\\texttt{{{}}}~\\{{{}\\}}",
556        cleaned,
557        parts.join(", ")
558      )
559    }
560  }

562  /// Detect a pattern body that's "purely a list of element
563  /// definitions" — either a bare singleton (`pattern X = element Y {…}`
564  /// with a leading `## doc` that blocks the simplify shortcut) or a
565  /// Choice/Group/Interleave of N non-wildcard elements
566  /// (`X = element a {…} | element b {…} | …`). Both shapes
567  /// mishandle in the generic path: nested `\elementdef` macros
568  /// render as cards inside the patterndef body, jumping out of
569  /// their `\item` and leaving empty `<dd>`s or orphan `(... | ...)`
570  /// punctuation. `to_tex_def` swaps them for an alphabetised
571  /// `\elementref` Choice expression in the patterndef body plus
572  /// sibling `\elementdef` cards (one per unique name).
573  ///
574  /// Returns `(combiner, [(element_name, element_body)])`.
575  fn detect_element_choice<'a>(
576    &self,
577    spec: &'a [Pattern],
578  ) -> Option<ElementChoice<'a>> {
579    if spec.len() != 1 {
580      return None;
581    }
582    // Bare singleton: `spec = [Element]`. Treat as a 1-element Choice
583    // (op doesn't matter — the renderer never inserts a separator for
584    // a single name).
585    if let Pattern::Element { name, body } = &spec[0] {
586      if is_wildcard_name(name) {
587        return None;
588      }
589      return Some((CombineOp::Choice, vec![(name.clone(), body.as_slice())]));
590    }
591    let Pattern::Combination { op, body } = &spec[0] else {
592      return None;
593    };
594    if !matches!(op, CombineOp::Choice | CombineOp::Group | CombineOp::Interleave) {
595      return None;
596    }
597    if body.is_empty() {
598      return None;
599    }
600    let mut elements: Vec<(String, &'a [Pattern])> = Vec::with_capacity(body.len());
601    for child in body {
602      let Pattern::Element { name, body: el_body } = child else {
603        return None;
604      };
605      if is_wildcard_name(name) {
606        return None;
607      }
608      elements.push((name.clone(), el_body.as_slice()));
609    }
610    Some((*op, elements))
611  }

613  /// Render `detect_element_choice` output. Returns:
614  ///
615  /// * the patterndef body — a single `Content` line whose value is
616  ///   an alphabetised expression `(name1 | name2 | …)` of
617  ///   `\elementref` links to the sibling cards. The post-pass's
618  ///   `render_content_models` then pretty-prints it with operator-
619  ///   leading layout. Plus the regular Used-by line.
620  /// * a list of `\elementdef{name}{}{…}` strings — one per unique
621  ///   element name (deduped, source-order kept). Each carries its
622  ///   own body (so distinct-bodied variants of the same name keep
623  ///   their first-seen body), plus a Used-by line citing the parent
624  ///   pattern.
625  fn render_element_choice(
626    &mut self,
627    qname: &str,
628    parent_clean: &str,
629    op: CombineOp,
630    elements: &[(String, &[Pattern])],
631  ) -> (String, Vec<String>) {
632    // Pattern body: one Choice/Group/Interleave expression of
633    // alphabetised \elementref links. Use the op-appropriate join
634    // string so the post-pass tokenizer sees the same operator
635    // tokens it does for any other content-model expression.
636    let mut sorted_names: Vec<String> = elements.iter().map(|(n, _)| n.clone()).collect();
637    sorted_names.sort();
638    sorted_names.dedup();
639    let names_tex: Vec<String> = sorted_names
640      .iter()
641      .map(|n| format!("\\elementref{{{}}}", clean_tex_name(n, &self.rng.display_strip_prefixes)))
642      .collect();
643    let sep = match op {
644      CombineOp::Choice => " ~\\textbar~ ",
645      CombineOp::Interleave => " ~\\&~ ",
646      CombineOp::Group => ", ",
647      _ => " ~\\textbar~ ",
648    };
649    let content = if names_tex.len() == 1 {
650      names_tex[0].clone()
651    } else {
652      format!("({})", names_tex.join(sep))
653    };
654    let mut body = format!("\\item[\\textit{{Content}}:] {}", content);
655    if let Some(uses) = self.symbol_uses(qname) {
656      body.push_str(&format!("\\item[\\textit{{Used by}}:] {}", uses));
657    }
658
659    // Sibling \elementdef cards. Dedupe by name (first occurrence
660    // wins): two `element a {B1} | element a {B2}` branches collapse
661    // to a single `xhtml:a` card carrying B1, since the post-pass
662    // can only assign `id="schema.xhtml..a"` to one anchor anyway.
663    let mut seen: rustc_hash::FxHashSet<String> = rustc_hash::FxHashSet::default();
664    let mut element_defs: Vec<String> = Vec::new();
665    for (name, el_body) in elements {
666      if !seen.insert(name.clone()) {
667        continue;
668      }
669      let cleaned = clean_tex_name(name, &self.rng.display_strip_prefixes);
670      let (el_attr, el_content) = self.to_tex_body(el_body);
671      let mut eb = el_attr;
672      if !el_content.is_empty() {
673        eb.push_str(&format!("\\item[\\textit{{Content}}:] {}", el_content));
674      }
675      eb.push_str(&format!(
676        "\\item[\\textit{{Used by}}:] \\patternref{{{}}}",
677        parent_clean
678      ));
679      element_defs.push(format!("\\elementdef{{{}}}{{}}{{{}}}\n", cleaned, eb));
680    }
681    (body, element_defs)
682  }

684  /// Inline rendering of an `<attribute><anyName/>...</attribute>` (or
685  /// `<nsName/>`) pattern: `\textit{attribute}~\texttt{NAME}=CONTENT`.
686  fn render_inline_attribute(&mut self, cleaned: &str, data: &[Pattern]) -> String {
687    let (_docs, spec) = self.extract_docs(data);

695        .join(" ")
696    };

700  fn to_tex_combination(&mut self, op: CombineOp, data: &[Pattern]) -> String {
701    // Collapse adjacent wildcard pairs (`*` followed by `*:*`) — they
702    // come from a single `<anyName/>` and would otherwise render twice.
703    let dedup_owned: Vec<Pattern>;
704    let data: &[Pattern] = if has_wildcard_pair(data) {
705      dedup_owned = dedupe_wildcard_pairs(data);
706      &dedup_owned
707    } else {
708      data
709    };
710    // Render Element / Attribute children inline (text-shape) instead
711    // of as `\elementdef` / `\attrdef` cards. The card macros expand
712    // to `\item[…]` which LaTeXML promotes out of the surrounding
713    // paragraph, leaving the Combination's `(`, `~\textbar~`, `)`
714    // tokens as orphan text fragments around an unrelated sibling
715    // list. Inline rendering keeps the whole Combination on a single
716    // text line. (For wildcard names, `to_tex_element`/`to_tex_attribute`
717    // already routes to inline.)
718    let inner: Vec<String> = data
719      .iter()
720      .map(|p| match p {
721        Pattern::Element { name, body } if !is_wildcard_name(name) => {
722          self.render_inline_element(name, body)
723        },
724        Pattern::Attribute { name, body } if !is_wildcard_name(name) => {
725          let cleaned = clean_tex_name(name, &self.rng.display_strip_prefixes);
726          self.render_inline_attribute(&cleaned, body)
727        },
728        _ => self.to_tex(p),
729      })
730      .collect();
731    match op {
732      CombineOp::Group => {
733        if inner.len() == 1 {
734          inner.into_iter().next().unwrap()
735        } else {
736          format!("({})", inner.join(", "))
737        }
738      },
739      CombineOp::Interleave => format!("({})", inner.join(" ~\\&~ ")),
740      CombineOp::Choice => format!("({})", inner.join(" ~\\textbar~ ")),
741      CombineOp::Optional => {
742        // Single attribute body: emit without the textsuperscript wrapper.
743        if inner.len() == 1 && matches!(data[0], Pattern::Attribute { .. }) {
744          inner.into_iter().next().unwrap()
745        } else {
746          format!("{}\\textsuperscript{{?}}", inner.first().cloned().unwrap_or_default())
747        }
748      },
749      CombineOp::ZeroOrMore | CombineOp::OneOrMore => {
750        // Note: Perl emits ^{*} for both zeroOrMore and oneOrMore — preserved.
751        format!(
752          "{}\\textsuperscript{{*}}",
753          inner.first().cloned().unwrap_or_default()
754        )
755      },
756      CombineOp::List => format!("({})", inner.join(", ")),
757    }
758  }

763  /// Mirrors `toTeXExtractDocs`.

777  /// Partition `data` into `(attrs_string, content_string)`, with the
778  /// same heuristics as Perl `toTeXBody`. Recursive expansion of
779  /// `*_attributes` / `*_model` refs, and pattern refs whose name ends
780  /// with `attributes` flow into the attribute list as-is.
781  fn to_tex_body(&mut self, data: &[Pattern]) -> (String, String) {
782    let mut attributes: Vec<String> = Vec::new();
783    let mut content: Vec<String> = Vec::new();
784    let mut attr_patterns: Vec<String> = Vec::new();
785    // Perl uses shift+unshift to inline-expand `*_attributes`/`*_model`
786    // refs and attribute-shaped Combinations as their members are
787    // encountered. A front-poppable deque mirrors that traversal.
788    // Dedupe `*`/`*:*` wildcard pairs at this layer too: the def-args
789    // path (`<define>` with a single wildcard `<element>` body) feeds
790    // them straight in without a Combination wrapper.
791    let dedup_owned: Vec<Pattern>;
792    let data: &[Pattern] = if has_wildcard_pair(data) {
793      dedup_owned = dedupe_wildcard_pairs(data);
794      &dedup_owned
795    } else {
796      data
797    };
798    // Group "trivial-body" attributes by their datatype so a long run
799    // of identical `attribute foo {text}` rows collapses into a single
800    // `Text attributes: a, b, c` line. Wildcards (`*`, `*:*`) skip the
801    // grouping path — they have no enumerable name. Attributes carrying
802    // a `Doc` annotation also skip it (we'd lose the doc otherwise).
803    // Key = the type label ("text" / "string" / …); BTreeMap gives a
804    // stable alphabetical render order across types.
805    let mut grouped: BTreeMap<String, Vec<String>> = BTreeMap::new();
806    let mut deque: std::collections::VecDeque<Pattern> = data.iter().cloned().collect();
807    while let Some(item) = deque.pop_front() {
808      match &item {
809        Pattern::Attribute { name, body } => {
810          if let Some(t) = simple_attr_type(body) {
811            if !is_wildcard_name(name) {
812              grouped.entry(t).or_default().push(clean_tex_name(name, &self.rng.display_strip_prefixes));
813              continue;
814            }
815          }
816          attributes.push(self.to_tex(&item));
817        },
818        Pattern::Combination { body, .. } if self.is_attributes(&item) => {
819          for c in body.iter().cloned().rev() {
820            deque.push_front(c);
821          }
822        },
823        Pattern::Ref { qname } if qname.ends_with("_attributes") || qname.ends_with("_model") => {
824          if let Some(def) = self.rng.defs.get(qname).cloned() {
825            deque.push_front(def);
826          }
827        },
828        Pattern::Ref { qname } if qname_ends_with_attributes(qname) => {
829          attr_patterns.push(self.to_tex(&item));
830        },
831        // Direct element children — render inline (just the link)
832        // instead of `\elementdef{…}` so a pattern body like
833        // `text, element a {…}, text` doesn't end up with the card
834        // promoting itself out of the surrounding paragraph.
835        // `to_tex_def` extracts these into sibling cards.
836        Pattern::Element { name, body } if !is_wildcard_name(name) => {
837          content.push(self.render_inline_element(name, body));
838        },
839        _ => content.push(self.to_tex(&item)),
840      }
841    }
842
843    let mut attr_str = String::new();
844    if !attr_patterns.is_empty() {
845      attr_str.push_str("\\item[\\textit{Attributes}:] ");
846      attr_str.push_str(&attr_patterns.join(", "));
847    }
848    // Grouped lines render before the per-attribute cards: the bulk
849    // overview reads first, then any non-trivial typed attributes.
850    // Each name is wrapped in `\texttt{...}` so it lands under
851    // `.ltx_font_typewriter` (var(--font-code) — SF Mono / Fira Mono
852    // / etc.) in the rendered HTML; commas stay in body type so the
853    // names visually separate.
854    for (type_name, mut names) in grouped {
855      names.sort();
856      let monospaced: Vec<String> =
857        names.iter().map(|n| format!("\\texttt{{{}}}", n)).collect();
858      attr_str.push_str(&format!(
859        "\\item[\\textit{{{}}}:] {}",
860        attr_group_label(&type_name),
861        monospaced.join(", "),
862      ));
863    }
864    for a in attributes {
865      attr_str.push_str(&a);
866    }
867    let content_str = content.join(", ");
868    (attr_str, content_str)
869  }

873    match item {
874      Pattern::Attribute { .. } => true,
875      Pattern::Ref { qname } => self

884          | CombineOp::Choice
885          | CombineOp::Group
886          | CombineOp::ZeroOrMore

912          | CombineOp::Choice
913          | CombineOp::Group
914          | CombineOp::ZeroOrMore

923  /// Format the "Used by:" link list for `qname`. Returns `None` when
924  /// the symbol has no recorded uses.
925  fn symbol_uses(&self, qname: &str) -> Option<String> {
926    let uses = self.rng.uses_name.get(qname)?;
927    let mut sorted: Vec<&String> = uses.iter().collect();
928    sorted.sort();
929    let mut transformed: Vec<String> = Vec::new();
930    for u in sorted {
931      if self.opts.skip_svg && u.contains("SVG.") {
932        continue;
933      }
934      // pattern:[^:]*:NAME_(attributes|model) → element:NAME
935      if let Some(rest) = u.strip_prefix("pattern:") {
936        if let Some(idx) = rest.find(':') {
937          let after = &rest[idx + 1..];
938          if let Some(name) = after
939            .strip_suffix("_attributes")
940            .or_else(|| after.strip_suffix("_model"))
941          {
942            transformed.push(format!("element:{}", name));
943            continue;
944          }
945        }
946      }
947      transformed.push(u.clone());
948    }
949    let mut parts: Vec<String> = Vec::new();
950    for t in &transformed {
951      if let Some(rest) = t.strip_prefix("pattern:") {
952        if let Some(idx) = rest.find(':') {
953          let name = &rest[idx + 1..];
954          parts.push(format!("\\patternref{{{}}}", clean_tex(name)));
955        }
956      }
957    }
958    for t in &transformed {
959      if let Some(name) = t.strip_prefix("element:") {
960        parts.push(format!("\\elementref{{{}}}", clean_tex_name(name, &self.rng.display_strip_prefixes)));
961      }
962    }
963    if parts.is_empty() {
964      None
965    } else {
966      Some(parts.join(", "))
967    }
968  }
969}

993/// Walk `spec` (recursively into `Combination`s but never into
994/// `Element` bodies) and collect every distinct non-wildcard
995/// `Pattern::Element { name, body }`. First occurrence per name wins
996/// — so `element a {B1} | element a {B2}` extracts as a single
997/// `xhtml:a` carrying `B1`, since the post-pass can only assign
998/// `id="schema.xhtml..a"` to one anchor anyway.
999///
1000/// The collected list drives `to_tex_def`'s per-element sibling
1001/// `\elementdef` extraction: pattern bodies render Elements inline as
1002/// `\elementref{name}` links, and the corresponding card sits as a
1003/// sibling of the patterndef so the link resolves on the same page.
1004fn collect_element_descendants(
1005  spec: &[Pattern],
1006) -> Vec<(String, &[Pattern])> {
1007  let mut out = Vec::new();
1008  let mut seen: rustc_hash::FxHashSet<String> = rustc_hash::FxHashSet::default();
1009  for p in spec {
1010    walk_for_elements(p, &mut out, &mut seen);
1011  }
1012  out
1013}
1014
1015fn walk_for_elements<'a>(
1016  p: &'a Pattern,
1017  out: &mut Vec<(String, &'a [Pattern])>,
1018  seen: &mut rustc_hash::FxHashSet<String>,
1019) {
1020  match p {
1021    Pattern::Element { name, body } => {
1022      if !is_wildcard_name(name) && seen.insert(name.clone()) {
1023        out.push((name.clone(), body.as_slice()));
1024      }
1025      // Don't recurse into Element body — that's the element's own
1026      // content model, not separate elements that should become
1027      // siblings of the parent pattern.
1028    },
1029    Pattern::Combination { body, .. } => {
1030      for c in body {
1031        walk_for_elements(c, out, seen);
1032      }
1033    },
1034    _ => {},
1035  }
1036}

1079/// True if `name` is an `<anyName/>` / `<nsName/>` wildcard:
1080/// `*` (no namespace), `*:*` (any namespace, any local) or `prefix:*`
1081/// (any local within a namespace). These names come from the scanner's
1082/// expansion of `<anyName/>` / `<nsName/>` in `scan_name_class` and
1083/// don't denote real definable element / attribute names.
1084fn is_wildcard_name(name: &str) -> bool {
1085  name == "*" || name == "*:*" || name.ends_with(":*")
1086}

1088/// True if `data` contains an adjacent `*` then `*:*` Element pair
1089/// (or the same shape for Attribute). Used as a cheap pre-check so the
1090/// dedupe path only allocates when there's actually something to fold.
1091fn has_wildcard_pair(data: &[Pattern]) -> bool {
1092  data.windows(2).any(|w| is_wildcard_pair(&w[0], &w[1]))
1093}

1095fn is_wildcard_pair(a: &Pattern, b: &Pattern) -> bool {
1096  match (a, b) {
1097    (Pattern::Element { name: n1, .. }, Pattern::Element { name: n2, .. })
1098    | (Pattern::Attribute { name: n1, .. }, Pattern::Attribute { name: n2, .. }) => {
1099      n1 == "*" && n2 == "*:*"
1100    },
1101    _ => false,
1102  }
1103}

1105/// Walk `data` and collapse each adjacent `*` / `*:*` Element- or
1106/// Attribute-pair (produced by the scanner from a single `<anyName/>`)
1107/// into the single `*:*` form. The pair always shares its body since
1108/// `scan_pattern_element` / `scan_pattern_attribute` build both members
1109/// from the same `body_proto.clone()`, so dropping the `*` half loses
1110/// no information.
1111fn dedupe_wildcard_pairs(data: &[Pattern]) -> Vec<Pattern> {
1112  let mut out = Vec::with_capacity(data.len());
1113  let mut i = 0;
1114  while i < data.len() {
1115    if i + 1 < data.len() && is_wildcard_pair(&data[i], &data[i + 1]) {
1116      // Keep the `*:*` member (data[i+1]) — it's the broader form and
1117      // reads more clearly in the rendered content model.
1118      out.push(data[i + 1].clone());
1119      i += 2;
1120    } else {
1121      out.push(data[i].clone());
1122      i += 1;
1123    }
1124  }
1125  out
1126}

1128/// If `s` is wrapped in matching outer parentheses (no other unbalanced
1129/// content at top level), return the unwrapped slice; otherwise `s`.
1130/// Used by the Expansion suppression in `to_tex_def` to detect when the
1131/// stored-Combination form differs from the raw def-args form by only
1132/// an outer `(...)` wrap.
1133fn strip_outer_parens(s: &str) -> &str {
1134  let bytes = s.as_bytes();
1135  if bytes.first() != Some(&b'(') || bytes.last() != Some(&b')') {
1136    return s;
1137  }
1138  // Confirm the outer `(` matches the outer `)` (no `(A)(B)` slip).
1139  let mut depth = 0i32;
1140  for (i, b) in bytes.iter().enumerate() {
1141    match b {
1142      b'(' => depth += 1,
1143      b')' => {
1144        depth -= 1;
1145        if depth == 0 && i + 1 != bytes.len() {
1146          return s;
1147        }
1148      },
1149      _ => {},
1150    }
1151  }
1152  if depth != 0 {
1153    return s;
1154  }
1155  &s[1..s.len() - 1]
1156}

1158fn qname_ends_with_attributes(qname: &str) -> bool {
1159  // Matches the Perl regex `[^a-zA-Z]attributes$`.
1160  let rest = qname.strip_suffix("attributes").unwrap_or("");
1161  if qname == "attributes" {
1162    return false;
1163  }
1164  match rest.chars().last() {
1165    Some(c) => !c.is_ascii_alphabetic() && qname.ends_with("attributes"),
1166    None => false,
1167  }
1168}
