1//! An arena for interning strings
2//!  (global, mutable, single thread only)
3//!
4//! Note: works under the assumption of a single threaded, short-lived process,
5//! where memory starvation will not be an issue. It would need a hard reset if the same process
6//! does multiple conversions, and a different implementation if one needs a thread-local arena.
7//!
8//! ## Borrow safety
9//!
10//! All arena access goes through `with_arena_mut()`, which acquires a mutable
11//! borrow from `RefCell` on first call and caches the raw pointer in a
12//! thread-local `Cell`. Re-entrant calls (e.g., `pin()` inside a `with()`
13//! closure) reuse the cached pointer without touching `RefCell`, eliminating
14//! borrow conflicts entirely. A RAII guard clears the pointer on scope exit
15//! (including panics), so the invariant holds through unwinds.
16

29type Interner = StringInterner<BufferBackend, BuildHasherDefault<FxHasher>>;

31#[thread_local]
32static ARENA: Lazy<RefCell<Interner>> =
33  Lazy::new(|| {
34    RefCell::new(StringInterner::with_capacity_and_hasher(
35      32_768,
36      BuildHasherDefault::<FxHasher>::default(),
37    ))
38  });

40/// Cached raw pointer to the arena's inner interner, valid only while the
41/// outermost `with_arena_mut` holds its `RefMut` guard. Null when idle.
42#[thread_local]
43static ACTIVE: Cell<*mut Interner> = Cell::new(std::ptr::null_mut());
44
45/// RAII guard that clears `ACTIVE` on drop (including during unwinds).
46/// Declared AFTER the `RefMut` guard so it drops FIRST (Rust drops in
47/// reverse declaration order), ensuring no window where ACTIVE is stale.
48struct ArenaCleanup;
49impl Drop for ArenaCleanup {
50  fn drop(&mut self) {
51    ACTIVE.set(std::ptr::null_mut());
52  }
53}
54
55/// Execute `f` with mutable access to the interner.
56/// First (outermost) call acquires `borrow_mut()` from `RefCell` and caches
57/// the raw pointer. Re-entrant calls reuse the cached pointer — no RefCell
58/// interaction, so no borrow conflicts.
59///
60/// # Safety
61/// Sound because: (1) `#[thread_local]` guarantees single-thread access,
62/// (2) `ArenaCleanup` clears the pointer before the `RefMut` drops,
63/// (3) re-entrant access is strictly nested (same stack, same thread).
64#[inline]
65fn with_arena_mut<R>(f: impl FnOnce(&mut Interner) -> R) -> R {
66  let ptr = ACTIVE.get();
67  if !ptr.is_null() {
68    // Re-entrant call — reuse existing mutable borrow.
69    // SAFETY: ptr was set by the outermost call on this thread, which still
70    // holds the RefMut guard. We are nested on the same stack.
71    f(unsafe { &mut *ptr })
72  } else {
73    // Outermost call — acquire mutable borrow from RefCell.
74    let mut guard = ARENA.borrow_mut();
75    let ptr = &mut *guard as *mut Interner;
76    ACTIVE.set(ptr);
77    let _cleanup = ArenaCleanup; // drops BEFORE guard (reverse order)
78    f(&mut guard)
79  }
80}

140/// Assign a static str into the arena, returning a unique symbol.
141pub fn pin_static(text: &'static str) -> SymStr {
142  with_arena_mut(|arena| arena.get_or_intern_static(text))
143}

145/// Assign a string into the arena, returning a unique symbol.
146pub fn pin<S: AsRef<str>>(text: S) -> SymStr {
147  with_arena_mut(|arena| arena.get_or_intern(text))

150pub fn pin_char(c: char) -> SymStr {
151  let mut tmp = [0u8; 4];
152  let s = c.encode_utf8(&mut tmp);
153  pin(s)
154}

158/// Resolve a symbol and call the closure with a `&str` reference.
159/// The closure may safely call `pin()` or any other arena function —
160/// re-entrant access reuses the cached borrow.
161pub fn with<R, FnR>(sym: SymStr, caller: FnR) -> R
162where FnR: FnOnce(&str) -> R {
163  with_arena_mut(|arena| {
164    let s = arena
165      .resolve(sym)
166      .expect("arena::with: symbol not found in arena");
167    caller(s)
168  })
169}

200pub fn with_many<R, FnR>(syms: &[SymStr], caller: FnR) -> R
201where FnR: FnOnce(Vec<&str>) -> R {
202  with_arena_mut(|arena| {
203    let many = syms
204      .iter()
205      .map(|sym| {
206        arena
207          .resolve(*sym)
208          .expect("arena::with_many: symbol not found in arena")
209      })
210      .collect();
211    caller(many)
212  })
213}

215pub fn to_string(sym: SymStr) -> String {
216  with_arena_mut(|arena| {
217    arena
218      .resolve(sym)
219      .expect("arena::to_string: symbol not found in arena")
220      .to_owned()
221  })
222}
