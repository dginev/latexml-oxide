53/// Per-thread tracker for the most recently emitted error's
54/// `category:object` signature, plus the count of how many
55/// consecutive errors share the same signature. Used to detect
56/// runaway loops where a single pathological control-sequence (like
57/// plain-TeX `\tabalign` invoked in math mode → unbounded `\halign`
58/// cell loop) keeps emitting the same error indefinitely. See
59/// `wisdom_tabalign_math_runaway.md` for the canonical witness.
60#[thread_local]
61static LAST_ERROR_KEY: RefCell<Option<String>> = RefCell::new(None);
62#[thread_local]
63static CONSECUTIVE_ERROR_COUNT: std::cell::Cell<usize> = std::cell::Cell::new(0);
64
65/// Threshold for "same error fired this many times in a row → bail."
66/// Set well above any legitimate same-error pattern (a paper with
67/// 500+ identical errors would already be near-useless output) but
68/// well below the 10000 MAX_ERRORS cap so runaway papers don't
69/// accumulate huge noise logs. Empirically, the pathological
70/// `\tabalign`-in-math-mode runaway hits >9000 consecutive same
71/// errors; this catches that at 500 instead. The threshold was
72/// tightened from an initial 2000 after verifying no test in the
73/// 1112-test suite exceeds it.
74pub const MAX_CONSECUTIVE_ERRORS: usize = 500;
75
76/// Record an error signature; returns the new consecutive count.
77/// Call from the Error! macro after note_status. Resets count to 1
78/// on a different signature, increments on a match.
79pub fn note_consecutive_error(key: &str) -> usize {
80  let mut last = LAST_ERROR_KEY.borrow_mut();
81  if last.as_deref() == Some(key) {
82    let c = CONSECUTIVE_ERROR_COUNT.get() + 1;
83    CONSECUTIVE_ERROR_COUNT.set(c);
84    c
85  } else {
86    *last = Some(key.to_string());
87    CONSECUTIVE_ERROR_COUNT.set(1);
88    1
89  }
90}

92/// Reset the consecutive-error tracker (called from initialize_report).
93fn reset_consecutive_error_tracker() {
94  *LAST_ERROR_KEY.borrow_mut() = None;
95  CONSECUTIVE_ERROR_COUNT.set(0);
96}

120pub fn note_status(status: LogStatus, what: Option<&str>) {
121  let mut report = REPORT.borrow_mut();
122  use LogStatus::*;
123  match status {
124    Debug => report.debug += 1,
125    Info => report.info += 1,
126    Warning => report.warning += 1,
127    Error => report.error += 1,
128    Fatal => {
129      // Diagnostic for "phantom fatals" (a fatal counted in the final summary
130      // with no `Fatal:` line in the log — an `Err` raised via `Fatal!` that
131      // some caller swallowed without `log_fatal`): dump a backtrace at the
132      // moment the tally first flips. Witness math0402448.
133      if !report.fatal && debug_fatal_enabled() {
134        eprintln!("[debug-fatal] LogStatus::Fatal first noted here:");
135        eprintln!("{}", std::backtrace::Backtrace::force_capture());
136      }
137      report.fatal = true;
138    },
139    Undefined => {
140      let entry = report
141        .undefined
142        .entry(what.unwrap_or_default())
143        .or_insert(0);
144      *entry += 1;
145    },
146    Missing => {
147      let entry = report.missing.entry(what.unwrap_or_default()).or_insert(0);
148      *entry += 1;
149    },
150  }
151}

153pub fn get_status(status: LogStatus) -> usize {
154  let report = REPORT.borrow();
155  use LogStatus::*;
156  match status {
157    Debug => report.debug,
158    Info => report.info,
159    Warning => report.warning,
160    Error => report.error,
161    Fatal => {
162      if report.fatal {
163        1
164      } else {
165        0
166      }
167    },
168    Undefined => report.undefined.0.values().sum(),
169    Missing => report.missing.0.values().sum(),
170  }
171}

184  let mut report = REPORT.borrow_mut();

190thread_local! {
191  /// Latch for the most recent RESOURCE-class fatal (`ErrorTarget::Timeout`
192  /// with a unit category: token/pushback/if limits, cycle-guard recursion,
193  /// memory budget, conversion deadline). Some layers flatten `Error` into a
194  /// plain string on the way up (the marpa semantics boundary turns it into
195  /// `marpa::error::Error`), destroying the structured identity — which made
196  /// resource fatals indistinguishable from semantic parse rejections and
197  /// produced "phantom fatals" (counted in the summary, never logged, parse
198  /// grinding on). The `Fatal!` macro records here at raise time; consumers
199  /// `take` it to re-classify a flattened error. PR #249 review P1-4.
200  static LAST_RESOURCE_FATAL: RefCell<Option<Error>> = const { RefCell::new(None) };
201}
202
203/// Record a fatal into the resource-fatal latch — only Timeout-target fatals
204/// with payload-free categories are kept (the latch exists for resource
205/// fatals; payload-carrying `ErrorCategory` variants are not cloneable and
206/// are never resource-class).
207pub fn record_last_fatal(e: &Error) {
208  use ErrorCategory as C;
209  if !matches!(e.target, ErrorTarget::Timeout) {
210    return;
211  }
212  let category = match &e.category {
213    C::TokenLimit => C::TokenLimit,
214    C::PushbackLimit => C::PushbackLimit,
215    C::Recursion => C::Recursion,
216    C::IfLimit => C::IfLimit,
217    C::MemoryBudget => C::MemoryBudget,
218    C::Convert => C::Convert,
219    _ => return,
220  };
221  LAST_RESOURCE_FATAL.with(|c| {
222    *c.borrow_mut() = Some(Error {
223      target: ErrorTarget::Timeout,
224      category,
225      message: e.message.clone(),
226    });
227  });
228}

230/// Take (and clear) the latched resource fatal, if any. Returns the
231/// structured `Error` so a boundary that received only a flattened string can
232/// propagate the real thing.
233pub fn take_last_resource_fatal() -> Option<Error> {
234  LAST_RESOURCE_FATAL.with(|c| c.borrow_mut().take())
235}

237/// Build a status message matching Perl's `getStatusMessage()`.
238/// Format: "N warnings; M errors; K fatal error; L undefined macros[\foo, \bar]; P missing
239/// files[x.sty]" Returns "No obvious problems" when no issues detected.
240pub fn get_status_message() -> String {
241  let report = REPORT.borrow();
242  let mut parts = Vec::new();
243  if report.warning > 0 {
244    parts.push(format!(
245      "{} warning{}",
246      report.warning,
247      if report.warning > 1 { "s" } else { "" }
248    ));
249  }
250  if report.error > 0 {
251    parts.push(format!(
252      "{} error{}",
253      report.error,
254      if report.error > 1 { "s" } else { "" }
255    ));
256  }
257  if report.fatal {
258    parts.push("1 fatal error".to_string());
259  }
260  let undef_keys: Vec<String> = report
261    .undefined
262    .keys()
263    .map(|k| crate::common::arena::to_string(*k))
264    .collect();
265  if !undef_keys.is_empty() {
266    parts.push(format!(
267      "{} undefined macro{}[{}]",
268      undef_keys.len(),
269      if undef_keys.len() > 1 { "s" } else { "" },
270      undef_keys.join(", ")
271    ));
272  }
273  let miss_keys: Vec<String> = report
274    .missing
275    .keys()
276    .map(|k| crate::common::arena::to_string(*k))
277    .collect();
278  if !miss_keys.is_empty() {
279    parts.push(format!(
280      "{} missing file{}[{}]",
281      miss_keys.len(),
282      if miss_keys.len() > 1 { "s" } else { "" },
283      miss_keys.join(", ")
284    ));
285  }
286  if parts.is_empty() {
287    "No obvious problems".to_string()
288  } else {
289    parts.join("; ")
290  }
291}

293/// Compute the status code from the report state (Perl getStatusCode).
294/// 3 = fatal, 2 = errors, 1 = warnings, 0 = clean.
295pub fn get_status_code() -> usize {
296  let report = REPORT.borrow();
297  if report.fatal {
298    3
299  } else if report.error > 0 {
300    2
301  } else if report.warning > 0 {
302    1
303  } else {
304    0
305  }
306}

352/// Feature-gated debug logging — Perl's `Debug(...) if $LaTeXML::DEBUG{feature}`.
353/// Usage: `DebugFeature!("frontmatter", "FRONT Add {}", entry)`.
354/// Logs with the feature name as the `log` target (so output matches the
355/// previous `log::debug!(target: "frontmatter", ...)` form) and counts a
356/// Debug in the status report, like `Debug!`. NB deliberately does NOT
357/// forward to `Debug!` — its 3-expr `(category, object, message)` arm
358/// would mis-capture a format string with two arguments.
359#[macro_export]
360macro_rules! DebugFeature {
361  ($feature:literal, $($arg:tt)*) => {{
362    if $crate::common::error::debug_enabled($feature) {
363      $crate::common::error::note_status(
364        $crate::common::error::LogStatus::Debug, None);
365      use log::debug;
366      debug!(target: $feature, $($arg)*);
367    }
368  }};
369}

443#[macro_export]
444macro_rules! Error {
445  ($category:expr_2021, $object:expr_2021, $message:expr_2021) => {{
446    $crate::Error!($category,$object,$message,"")
447  }};
448 ($category:expr_2021, $object:expr_2021, $message:expr_2021, $($details:expr_2021),*) => {{
449    $crate::common::error::note_status(
450      $crate::common::error::LogStatus::Error, None);
451    if !$crate::common::error::is_log_output_suppressed() {
452      use log::error;
453      error!(target: &format!("{}:{}", $category, $object), "{}",
454        $crate::generate_message!($message, $($details),*));
455    }
456    // Borrow-safe read: an Error! can be raised from inside a `state_mut()`
457    // scope (e.g. push_value's BUG branch, a constructor's after_digest),
458    // where a plain `lookup_int` would panic "RefCell already mutably
459    // borrowed" and abort the conversion (tikz-cd 2001.08973).
460    let max_from_state = $crate::state::try_lookup_int("MAX_ERRORS");
461    // Match Perl LaTeXML default of 100 errors before Fatal('too_many_errors').
462    // Past 100 errors a paper has already failed comprehension; continuing
463    // produces noise without information. Override via state for tests
464    // or specific bindings (e.g. tikz_sty raises to 1000, dump-build raises
465    // to 1_000_000).
466    let maxerrors = match max_from_state {
467      // STATE contended: we cannot read the (possibly raised) cap, so skip the
468      // too-many-errors check for *this* error rather than risk a spurious
469      // Fatal from a stale default. The next uncontended error re-applies it.
470      None => usize::MAX,
471      Some(v) if v > 0 => v as usize,
472      Some(_) => 100,
473    };
474    if $crate::common::error::get_status($crate::common::error::LogStatus::Error) > maxerrors {
475      Fatal!(TooManyErrors, MaxLimit(maxerrors), format!("Too many errors (> {maxerrors})!"));
476    }
477    // Runaway-loop early-bail: if the same error signature has fired
478    // MAX_CONSECUTIVE_ERRORS times in a row, we're stuck in a loop
479    // (the canonical witness is plain-TeX `\tabalign` invoked in math
480    // mode → unbounded `\halign` cell loop emitting `\hbox` end-mode
481    // mismatches). Bail before MAX_ERRORS so logs stay short and
482    // post-processing sees a clear cause. The threshold is well above
483    // any legitimate same-error pattern (real papers max out at a few
484    // hundred unique errors).
485    let __consec_key = format!("{}:{}", $category, $object);
486    let __consec = $crate::common::error::note_consecutive_error(&__consec_key);
487    if __consec > $crate::common::error::MAX_CONSECUTIVE_ERRORS {
488      Fatal!(
489        TooManyErrors,
490        MaxLimit($crate::common::error::MAX_CONSECUTIVE_ERRORS),
491        format!(
492          "Runaway: same error '{}' fired {} times in a row (cap = {})",
493          __consec_key, __consec, $crate::common::error::MAX_CONSECUTIVE_ERRORS
494        )
495      );
496    }
497  }}
498}

627// SAFETY: `Error` contains a `Locator` which embeds a Rc<RefCell<Mouth>> — !Send/!Sync
628// by default. The invariant is the same as for `Stored`: errors propagate within a
629// single thread's conversion pipeline; they never cross thread boundaries at runtime.
630// These impls exist to satisfy `Box<dyn std::error::Error + Send + Sync>` bounds on
631// error return types, which transitively require Send/Sync on all error variants.
632unsafe impl Send for Error {}
633unsafe impl Sync for Error {}

726impl Error {
727  pub fn log_fatal(&self) {
728    let target_str = s!("Fatal:{:?}:{:?} ", self.target, self.category);
729    use log::error;
730    error!(target: &target_str, "{}", self.message);
731    // Mark the global report as fatal so cortex_worker's exit code is
732    // 3 (conversion failure) instead of 0 (success). Without this,
733    // `Fatal:Timeout:MemoryBudget` etc. printed but the runtime
734    // status_code stayed at 0 — canvas would classify the worker as
735    // OK with an empty HTML output. R35.A.
736    note_status(LogStatus::Fatal, None);
737  }

741      category: ErrorCategory::ToDo,
742      message:  String::from(
743        "This section of the code is not yet implemented / ported over from Perl.",
744      ),
745    }
746  }

908  #[test]
909  fn fatal_macro_latches_resource_fatals() {
910    // The `Fatal!` macro must record Timeout-target fatals in the
911    // resource-fatal latch so boundaries that flatten errors to strings
912    // (marpa semantics) can recover the structured identity (P1-4).
913    initialize_report();
914    fn raise() -> Result<()> {
915      Fatal!(
916        Timeout,
917        TokenLimit,
918        "Token limit of 5 exceeded, infinite loop?"
919      );
920    }
921    let err = raise().unwrap_err();
922    assert!(matches!(err.target, ErrorTarget::Timeout));
923    let latched = take_last_resource_fatal().expect("latch must hold the fatal");
924    assert!(matches!(latched.target, ErrorTarget::Timeout));
925    assert!(matches!(latched.category, ErrorCategory::TokenLimit));
926    assert_eq!(latched.message, "Token limit of 5 exceeded, infinite loop?");
927    // take() clears the latch.
928    assert!(take_last_resource_fatal().is_none());
929    // Non-Timeout fatals are NOT latched (the latch serves resource fatals).
930    fn raise_other() -> Result<()> {
931      Fatal!(Internal, EoF, "fell off the end");
932    }
933    let _ = raise_other().unwrap_err();
934    assert!(take_last_resource_fatal().is_none());
935    // initialize_report clears a stale latch.
936    let _ = raise().unwrap_err();
937    initialize_report();
938    assert!(take_last_resource_fatal().is_none());
939  }
