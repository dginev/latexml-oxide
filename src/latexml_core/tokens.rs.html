31/// Tokens are a thin wrapper over a vector of Token objects
32///
33/// They are usually read from a `Mouth` and treated as an immutable interface.
34/// For access to the inner Token contents, use one of the `unlist` methods.
35#[derive(Debug, Clone, Default)]
36pub struct Tokens(Vec<Token>);

38impl PartialEq for Tokens {
39  fn eq(&self, other: &Tokens) -> bool {

44/// convenience macro for assembling a Tokens object from different pieces (`Token`, `Vec<Token>`,
45/// `Tokens`)
46#[macro_export]
47macro_rules! Tokens(
48  () => ( $crate::tokens::NO_TOKENS );
49  ($( $tokens:expr ),+) => ({
50    let mut collected : Vec<$crate::token::Token> = Vec::new();
51    $(
52      let t_vec : Vec<$crate::token::Token> = $tokens.into();
53      collected.extend(t_vec);
54    )*
55    $crate::tokens::Tokens::new(collected)
56  }));

228  /// to_keyvals casts back to a parsed KeyVals (usually via a KeyVals parameter type)
229  /// which had to be re-converted to a Tokens for reentering the expansion flow
230  pub fn to_keyvals(&self) -> Result<KeyVals> {
231    let mut toks_iter = self.unlist_ref().iter();
232    let mut kvs = KeyVals::default();
233    while let Some(key) = toks_iter.next() {
234      key.with_str(|key_str| {
235        if let Some(value) = toks_iter.next() {
236          kvs.add_value(key_str, ArgWrap::Token(*value), false, false)
237        } else {
238          kvs.add_value(key_str, ArgWrap::Tokens(Tokens!()), false, false)
239        }
240      })?;
241    }
242    Ok(kvs)
243  }

245  /// Methods for overloaded ops.
246  pub fn equals(&self, other: Tokens) -> bool {
247    let self_tokens: Vec<&Token> = self
248      .0
249      .iter()
250      .filter(|t| t.code != Catcode::COMMENT && t.code != Catcode::MARKER)
251      .collect();
252    let other_tokens: Vec<&Token> = other
253      .0
254      .iter()
255      .filter(|t| t.code != Catcode::COMMENT && t.code != Catcode::MARKER)
256      .collect();
257    if self_tokens.len() != other_tokens.len() {
258      false
259    } else {
260      self_tokens
261        .into_iter()
262        .zip(other_tokens)
263        .all(|(t_self, t_other)| *t_self == *t_other)
264    }
265  }

297  /// Checks if any BEGIN/END code groups are correctly nested and closed
298  pub fn is_balanced(&self) -> bool {
299    let mut level = 0;
300    for t in &self.0 {
301      level += match t.get_catcode() {
302        Catcode::BEGIN => 1,
303        Catcode::END => -1,
304        _ => 0,
305      };
306      if level < 0 {
307        // a negative level encountered at any point is ill-formed,
308        // return early
309        return false;
310      }
311    }
312    level == 0
313  }

315  // NOTE: Assumes each arg either undef or also Tokens
316  // Using inline accessors on those assumptions
317  /// substitutes the parameters (ARG catcode) in a Tokens list for concrete arguments
318  pub fn substitute_parameters(&self, args: &[Option<Cow<Tokens>>]) -> Self {
319    let mut result = Vec::new();
320    for token in self.0.iter() {
321      if token.get_catcode() != Catcode::ARG {
322        // Non-match; copy it
323        result.push(*token);
324      } else {
325        let idx = token.with_str(|ts| ts.parse::<usize>().unwrap_or(0));
326        if idx > 0 && idx <= args.len() {
327          if let Some(ref arg) = args[idx - 1] {
328            result.extend(arg.clone().into_owned().unlist());
329          }
330        }
331      }
332    }
333    Tokens::new(result)
334  }

336  /// Consumes a Tokens to a string containing TeX that created it (or could have).
337  /// Note that this is not necessarily the original TeX code; expansions or other substitutions may
338  /// have taken place.
339  ///
340  /// **Design decision:** The Perl `UnTeX` inserts `%\n` line-breaks (TeX comment + newline) when
341  /// a token string would exceed 78 characters. The Rust port deliberately omits this feature.
342  /// Line-break insertion is purely cosmetic and makes test expectations fragile — the `%\n`
343  /// appears verbatim in `tex=` attributes of `ltx:Math` elements, causing test XML files to
344  /// contain `%&#10;` escape sequences that depend on exact token lengths. We instead always
345  /// produce compact, single-line output. Test `.xml` files should not contain `%&#10;`.
346  pub fn untex(self) -> String {
347    let mut tokens: VecDeque<Token> = self.revert().into_iter().collect();
348    let mut tex_string = String::new();
349    let mut length = 0;
350    let mut level = 0;
351    let mut prevs = String::new();
352    let mut prevcc = Catcode::COMMENT;
353    while let Some(token) = tokens.pop_front() {
354      let cc = token.get_catcode();
355      if cc == Catcode::COMMENT {
356        continue;
357      }
358      let mut token_string = token.to_string();
359      // Note: \n only-used to fail alphanumeric test
360      let first_char = token_string.chars().next().unwrap_or('\n');
361      if cc == Catcode::LETTER {
362        // keep "words" together, just for aesthetics
363        while !tokens.is_empty() && tokens[0].get_catcode() == Catcode::LETTER {
364          tokens
365            .pop_front()
366            .unwrap()
367            .with_str(|front_str| token_string.push_str(front_str));
368        }
369      }
370
371      let l = token_string.len();
372      if cc == Catcode::BEGIN {
373        level += 1;
374      }
375      //  Seems a reasonable & safe time to line break, for readability, etc.
376      if cc == Catcode::SPACE && token_string == "\n" {
377        // preserve newlines already present
378        if length > 0 {
379          tex_string.push_str(&token_string);
380          length = 0;
381        }
382      // If this token is a letter (or otherwise starts with a letter or digit): space or linebreak
383      } else {
384        let last_prevs = prevs.chars().last().unwrap_or('_');
385        // Perl: $STATE->lookupCatcode($1) == CC_LETTER
386        // Must use actual catcode lookup, not just is_alphabetic(), because
387        // characters like @ may have catcode LETTER in some contexts.
388        let prev_is_letter = crate::state::lookup_catcode(last_prevs)
389          .map(|cc| cc == Catcode::LETTER)
390          .unwrap_or_else(|| last_prevs.is_alphabetic());
391
392        if (cc == Catcode::LETTER || (cc == Catcode::OTHER && first_char.is_alphanumeric()))
393          && prevcc == Catcode::CS
394          && prev_is_letter
395        {
396          // Insert a (virtual) space before a letter if previous token was a CS w/letters
397          // This is required for letters, but just aesthetic for digits (to me?)
398          let space = ' ';
399          tex_string.push(space);
400          tex_string.push_str(&token_string);
401          length += 1 + l;
402        } else {
403          tex_string.push_str(&token_string);
404          length += l;
405        }
406        if cc == Catcode::END {
407          level -= 1;
408        }
409        prevs = token_string;
410        prevcc = cc;
411      }
412    }
413    // Patch up nesting for valid TeX !!!
414    match level {
415      1..=i32::MAX => {
416        for _ in 0..level {
417          tex_string.push('}');
418        }
419      },
420      i32::MIN..=-1 => {
421        for _ in 0..(-level) {
422          tex_string = String::from("{") + &tex_string;
423        }
424      },
425      0 => {},
426    }
427    tex_string
428  }

430  /// Packs repeated CC_PARAM tokens into CC_ARG tokens for use as a macro body (and other token
431  /// lists) Also unwraps \noexpand tokens, since that is also needed for macro bodies
432  /// (but not strictly part of packing parameters)
433  pub fn pack_parameters(self) -> Result<Self> {
434    let mut rescanned = Vec::new();
435    let mut toks = self.unlist().into_iter().collect::<VecDeque<_>>();
436    while let Some(t) = toks.pop_front() {
437      if t.get_catcode() == Catcode::PARAM && !toks.is_empty() {
438        let next_t = toks.pop_front();
439        let next_cc = next_t.as_ref().map(|t| t.get_catcode());
440        if next_cc == Some(Catcode::OTHER) {
441          // only group clear match token cases
442          rescanned.push(Token {
443            text: next_t.unwrap().get_sym(),
444            code: Catcode::ARG,
445          });
446        } else if next_cc == Some(Catcode::PARAM) {
447          rescanned.push(t);
448        } else {
449          // any other case, preserve as-is, let the higher level call resolve any errors
450          // e.g. \detokenize{#,} is legal, while \textbf{#,} is not
451          // Note: this also fires for alignment templates (\halign{#\hfil&...}) which is valid TeX.
452          // Perl has the same warning (Tokens.pm packParameters line 139). Non-fatal.
453          Error!(
454            "misdefined",
455            "expansion",
456            "Parameter has a malformed arg, should be #1-#9 or ##. In expansion {}",
457            Tokens::new(toks.clone().into_iter().collect()).to_string()
458          );
459        }
460      } else {
461        rescanned.push(t);
462      }
463    }
464    Ok(Tokens::new(rescanned))
465  }

467  /// Trims outer braces (if they balance each other).
468  /// Strips exactly 1 layer of matching outer braces by default.
469  /// Should this also trim whitespace? or only if there are braces?
470  pub fn strip_braces(self) -> Self { self.strip_braces_n(1) }
471
472  /// Trims `layers` outer brace pairs (if they balance each other).
473  /// Also trims whitespace *outer to* the removed braces.
474  /// Follows the Perl Tokens.pm algorithm: first collects all balanced
475  /// brace pairs, then strips from outside-in, only removing pairs that
476  /// span the full remaining width.
477  pub fn strip_braces_n(self, mut layers: usize) -> Self {
478    let tokens = self.0;
479    let n = tokens.len();
480    if n <= 1 {
481      return Tokens::new(tokens);
482    }
483
484    let mut i0: usize = 0;
485    let mut i1: usize = n;
486
487    // skip past spaces at ends
488    while i0 < i1 && tokens[i0].get_catcode() == Catcode::SPACE {
489      i0 += 1;
490    }
491    while i1 > i0 && tokens[i1 - 1].get_catcode() == Catcode::SPACE {
492      i1 -= 1;
493    }
494
495    // Collect balanced pairs (innermost first due to stack order)
496    let mut opens: Vec<usize> = Vec::new();
497    let mut pairs: Vec<(usize, usize)> = Vec::new();
498    for i in i0..i1 {
499      match tokens[i].get_catcode() {
500        Catcode::BEGIN => opens.push(i),
501        Catcode::END => {
502          if let Some(j) = opens.pop() {
503            pairs.push((j, i));
504          } else {
505            return Tokens::new(tokens); // Unbalanced: Too many }
506          }
507        },
508        _ => {},
509      }
510    }
511    if !opens.is_empty() {
512      return Tokens::new(tokens); // Unbalanced: Too many {
513    }
514
515    // Strip layers from outside-in.
516    // pairs is ordered innermost-first, so pop() gives outermost pair first.
517    while layers > 0 {
518      layers -= 1;
519      if let Some((j0, j1)) = pairs.pop() {
520        if j0 == i0 && j1 == i1 - 1 {
521          i0 += 1;
522          i1 -= 1;
523        }
524      }
525    }
526
527    // Empty after stripping
528    if i0 >= i1 {
529      return Tokens::new(Vec::new());
530    }
531
532    if i0 > 0 || i1 < n {
533      Tokens::new(tokens[i0..i1].to_vec())
534    } else {
535      Tokens::new(tokens)
536    }
537  }
538}

549impl ToTokens for Catcode {
550  fn to_tokens(&self, stream: &mut TokenStream) {
551    use crate::token::Catcode::*;
552    let kind = match *self {
553      ESCAPE => "ESCAPE",
554      BEGIN => "BEGIN",
555      END => "END",
556      MATH => "MATH",
557      ALIGN => "ALIGN",
558      EOL => "EOL",
559      PARAM => "PARAM",
560      SUPER => "SUPER",
561      SUB => "SUB",
562      SPACE => "SPACE",
563      // Non-primitive
564      IGNORE => "IGNORE",
565      LETTER => "LETTER",
566      OTHER => "OTHER",
567      ACTIVE => "ACTIVE",
568      COMMENT => "COMMENT",
569      INVALID => "INVALID",
570      CS => "CS",
571      MARKER => "MARKER",
572      ARG => "ARG",
573    };
574    stream.append(Ident::new("Catcode", Span::call_site()));
575    stream.append(Punct::new(':', Spacing::Joint));
576    stream.append(Punct::new(':', Spacing::Alone));
577    stream.append(Ident::new(kind, Span::call_site()));
578  }
