93  fn be_digested(mut self) -> Result<Digested> {
94    if self.was_digested {
95      Info!(
96        "ignore",
97        "keyvals",
98        "Skipping digestion of \\setkeys as requested (did you digest a KeyVals twice?) "
99      );
100    } else {
101      crate::stomach::digest(self.set_keys_expansion())?;
102    }
103
104    // iterate over the tuples, digesting the values
105    for tuple in self.tuples.iter_mut() {
106      let KVData {
107        key,
108        value,
109        primary_keyset,
110        digested_value,
111        ..
112      } = tuple;
113      if digested_value.is_none() {
114        // avoid accidental repeats?
115        let keytype_opt = keyval_get(&keyval_qname(&self.prefix, primary_keyset, key), "type");
116        let v = if let Some(Stored::Parameter(keytype)) = keytype_opt {
117          match value.take() {
118            Some(v) => keytype.digest(v, None)?,
119            _ => None,
120          }
121        } else {
122          match value.take() {
123            Some(v) => Some(v.be_digested()?),
124            _ => None,
125          }
126        };
127        tuple.digested_value = v;
128      }
129    }
130    // TODO: DG: KeyVals digestion feels very iffy while porting it over to Rust.
131    // had to add an explicit "rebuild" to cache the digested values in the new
132    // "cached_hash_digested" It feels like the entire object should be reorganized to leverage
133    // a little more of the well-typed capabilities we have here.
134    self.rebuild(None);
135    self.was_digested = true;
136    Ok(self.into())
137  }

191  /// The KeyVals constructor
192  ///======================================================================
193  /// This defines the KeyVals data object that can appear in the datastream
194  /// along with tokens, boxes, etc.
195  /// Thus it has to be digestible, however we may not want to digest it more
196  /// than once.
197  ///**********************************************************************
198  pub fn new(options: KeyvalsConfig) -> Self {
199    // parse all the arguments
200    let KeyvalsConfig {
201      prefix,
202      mut keysets,
203      set_all,
204      set_internals,
205      skip,
206      skip_missing,
207      hook_missing,
208    } = options;
209    let prefix = prefix.unwrap_or_else(|| String::from("KV"));
210    // Perl KeyVals.pm #2777 (fdc8bf91, 2026-03-27):
211    // filter empty strings from the keyset list. Split("," , ",pstricks")
212    // (e.g. \pst@famlist accumulates as ",pstricks") yields ["", "pstricks"];
213    // the empty keyset caused keyval_qname("psset","","ArrowInside") to
214    // collide with raw \def\psset@@ArrowInside (a delimited-argument helper)
215    // and emit spurious "Missing argument" errors. Hardening here matches
216    // the Perl fix regardless of how keysets was constructed at the call
217    // site.
218    keysets.retain(|k| !k.is_empty());
219    if keysets.is_empty() {
220      keysets = vec![String::from("_anonymous_")];
221    }
222    KeyVals {
223      prefix,
224      keysets,
225      skip,
226      set_all,
227      set_internals,
228      skip_missing,
229      hook_missing,
230      ..KeyVals::default()
231    }
232  }

238  /// Return a list of the keysets in which this key is defined
239  fn resolve_keyval_for(&self, key: &str) -> Vec<String> {
240    let prefix = &self.prefix;
241    let allkeysets = &self.keysets;
242    let keysets: Vec<_> = self
243      .keysets
244      .iter()
245      .filter(|kset| has_keyval(prefix, kset, key))
246      .collect();
247    // throw an error (not really), unless we record the missing macros
248    // Since we're not as obsessive about declaring ALL keys, we'll soften the blow
249    if keysets.is_empty() {
250      if self.skip_missing == SkipMissing::None {
251        // Rate-limit: only emit Info the first time this (prefix, key,
252        // keysets) tuple fires. A large `tabular` with 700 rows can
253        // otherwise produce 700 identical "Encountered unknown KeyVals
254        // key 'vattach'" messages (arxiv 1709.05096), each allocating
255        // a formatted String + going through the log backend. Perl's
256        // Info() has an equivalent deduper in Error.pm via
257        // maxWarnings limits; our rate-limit is per (prefix,key,keysets)
258        // and unbounded in count, so the first occurrence is always
259        // visible but repeats are silently dropped.
260        type SeenSet = rustc_hash::FxHashSet<(String, String, String)>;
261        thread_local! {
262          static SEEN_MISSING: std::cell::RefCell<SeenSet> =
263            std::cell::RefCell::new(SeenSet::default());
264        }
265        let all_joined = allkeysets.join(",");
266        let is_new = SEEN_MISSING.with(|cell| {
267          cell
268            .borrow_mut()
269            .insert((prefix.clone(), key.to_string(), all_joined.clone()))
270        });
271        if is_new {
272          // Intentional divergence from Perl (KeyVals.pm L97 uses Info).
273          // An unknown KeyVal key in `\setkeys` (non-starred) is the
274          // package binding admitting it doesn't recognise an option
275          // the user actually requested — the key's effect (formatting,
276          // rendering options) is silently dropped. For siunitx
277          // specifically this cascades into broken `\SI{}` expansion,
278          // which leaves bare control sequences in math and produces
279          // duplicated xml:id (witness: 1410.8171). Promoted to Warn
280          // so each unique missing key surfaces as a status_code=1
281          // (`[warn]` in the canvas), and a binding gap can't ship
282          // green.
283          //
284          // EXCEPT for the 'Frontmatter' keyset (PR #2767): its design
285          // passes undeclared keys by construction (the engine's own
286          // \lx@add@date uses `name={...}`; class bindings pass through
287          // arbitrary attributes). Perl Infos there; mirror that level
288          // so frontmatter-bearing papers don't all turn status_code=1.
289          if all_joined == "Frontmatter" {
290            Info!(
291              "undefined",
292              "Encountered unknown KeyVals key",
293              s!("'{key}' with prefix '{prefix}' not defined in '{all_joined}'")
294            );
295          } else {
296            Warn!(
297              "undefined",
298              "Encountered unknown KeyVals key",
299              s!(
300                "'{key}' with prefix '{prefix}' not defined in '{all_joined}', were you perhaps using \\setkeys instead of \\setkeys*?"
301              )
302            );
303          }
304        }
305      }
306      return Vec::new();
307    }
308    // return either the first or all of the KeyVal objects
309    // TODO: SymStr would avoid the allocation.
310    if self.set_all {
311      keysets.into_iter().cloned().collect()
312    } else {
313      vec![keysets[0].clone()]
314    }
315  }

333  fn read_keyword_from(&self, close: Token) -> Result<(Tokens, Option<Token>)> {
334    // set of tokens we will expand
335    let mut tokens = Vec::new();
336    let delim = &[close, T_OTHER!(","), T_OTHER!("=")];
337    // skip leading spaces
338    gullet::skip_spaces()?;
339
340    let mut last_token = None;
341    while let Some(token) = gullet::read_x_token(None, false, None)? {
342      // skip to the next iteration if we have a paragraph
343      if token == T_CS!("\\par") {
344        continue;
345      }
346      // if we have one of out delimiters, we end
347      if delim.contains(&token) {
348        last_token = Some(token);
349        break;
350      }
351      tokens.push(token);
352    }
353    // return the tokens and the last token
354    Ok((Tokens::new(tokens), last_token))
355  }

362  /// return the value of a given key. If multiple values are given, return the last one.
363  pub fn get_value(&self, key: &str) -> Option<&ArgWrap> {
364    // Since we (by default) accumulate lists of values when repeated,
365    // we need to provide the "common" thing: return the last value given.
366    match self.cached_hash.get(key) {
367      None => None,
368      Some(value) => value.last(),
369    }

439  pub fn set_keys_expansion(&self) -> Tokens {
440    let skip_keys = &self.skip;
441    let set_internals = self.set_internals;
442    let prefix = &self.prefix;
443
444    // Handle skipMissing store token (xkeyval feature)
445    let rmmacro = match &self.skip_missing {
446      SkipMissing::Store(token) => Some(*token),
447      _ => None,
448    };
449    let hook_missing = self.hook_missing;
450
451    // Read existing tokens from rmmacro (if defined and has meaning)
452    let mut rmtokens: Vec<Token> = Vec::new();
453    if let Some(rm) = rmmacro
454      && state::has_meaning(&rm)
455      && let Ok(expanded) = gullet::do_expand(Tokens!(rm))
456    {
457      rmtokens = expanded.unlist();
458    }
459
460    let mut tokens: Vec<Token> = Vec::new();
461
462    // Define xkeyval internals if needed
463    if set_internals {
464      let keysets_joined = self.keysets.join(",");
465      let skip_joined = self.skip.join(",");
466      tokens.push(T_CS!("\\def"));
467      tokens.push(T_CS!("\\XKV@fams"));
468      tokens.push(T_BEGIN!());
469      tokens.extend(Explode!(keysets_joined));
470      tokens.push(T_END!());
471      tokens.push(T_CS!("\\def"));
472      tokens.push(T_CS!("\\XKV@na"));
473      tokens.push(T_BEGIN!());
474      tokens.extend(Explode!(skip_joined));
475      tokens.push(T_END!());
476    }
477
478    // Iterate over key-value pairs
479    for tuple in &self.tuples {
480      let KVData {
481        key,
482        value,
483        use_default,
484        primary_keyset,
485        keysets,
486        ..
487      } = tuple;
488
489      // Skip keys in the skip list
490      if skip_keys.iter().any(|s| s == key) {
491        continue;
492      }
493
494      // If no keysets resolved for this key
495      if keysets.is_empty() {
496        // Store in rmmacro if defined
497        if rmmacro.is_some()
498          && let Ok(rev) = self.revert_keyval(
499            key,
500            primary_keyset,
501            value.as_ref(),
502            *use_default,
503            rmtokens.is_empty(),
504          )
505        {
506          rmtokens.extend(rev);
507        }
508        // Call hookMissing if defined
509        if let Some(hm) = hook_missing
510          && let Ok(rev) =
511            self.revert_keyval(key, primary_keyset, value.as_ref(), *use_default, true)
512        {
513          tokens.push(hm);
514          tokens.push(T_BEGIN!());
515          tokens.extend(rev);
516          tokens.push(T_END!());
517        }
518        continue;
519      }
520
521      // Iterate over all valid keysets
522      for keyset in keysets {
523        let qname = keyval_qname(prefix, keyset, key);
524        if !has_keyval(prefix, keyset, key) {
525          Info!(
526            "undefined",
527            "Encountered unknown KeyVals key",
528            s!("'{key}' with prefix '{prefix}' not defined in '{keyset}'")
529          );
530        } else if matches!(keyval_get(&qname, "disabled"), Some(Stored::Bool(true))) {
531          Warn!("undefined", "keyval", s!("`{key}' has been disabled. "));
532        } else {
533          // Define xkeyval internals per-key if needed
534          if set_internals {
535            tokens.push(T_CS!("\\def"));
536            tokens.push(T_CS!("\\XKV@prefix"));
537            tokens.push(T_BEGIN!());
538            tokens.extend(Explode!(s!("{prefix}@")));
539            tokens.push(T_END!());
540            tokens.push(T_CS!("\\def"));
541            tokens.push(T_CS!("\\XKV@tfam"));
542            tokens.push(T_BEGIN!());
543            tokens.extend(Explode!(keyset));
544            tokens.push(T_END!());
545            tokens.push(T_CS!("\\def"));
546            tokens.push(T_CS!("\\XKV@header"));
547            tokens.push(T_BEGIN!());
548            tokens.extend(Explode!(s!("{prefix}@{keyset}@")));
549            tokens.push(T_END!());
550            tokens.push(T_CS!("\\def"));
551            tokens.push(T_CS!("\\XKV@tkey"));
552            tokens.push(T_BEGIN!());
553            tokens.extend(Explode!(key));
554            tokens.push(T_END!());
555          }
556
557          // Perl: if ($useDefault) { push(@tokens, T_CS('\\' . $qname . '@default')); }
558          //       else { push(@tokens, T_CS('\\' . $qname), T_BEGIN, Revert($value), T_END); }
559          // Note: Perl unconditionally emits \qname@default for bare keys. In Rust, we guard
560          // with has_meaning to avoid undefined-CS errors when @default was never registered
561          // (e.g., xkeyval DeclareOptionX keys without default values).
562          if *use_default && state::has_meaning(&T_CS!(s!("\\{qname}@default"))) {
563            // Call the @default macro (bare key with registered default)
564            tokens.push(T_CS!(s!("\\{qname}@default")));
565          } else {
566            // Call the macro with the value (or empty if bare key without default)
567            tokens.push(T_CS!(s!("\\{qname}")));
568            tokens.push(T_BEGIN!());
569            if let Some(v) = value
570              && let Ok(reverted) = v.revert()
571            {
572              tokens.extend(reverted.unlist());
573            }
574            tokens.push(T_END!());
575          }
576
577          // Reset xkeyval internals per-key
578          if set_internals {
579            tokens.push(T_CS!("\\def"));
580            tokens.push(T_CS!("\\XKV@prefix"));
581            tokens.push(T_BEGIN!());
582            tokens.push(T_END!());
583            tokens.push(T_CS!("\\def"));
584            tokens.push(T_CS!("\\XKV@tfam"));
585            tokens.push(T_BEGIN!());
586            tokens.push(T_END!());
587            tokens.push(T_CS!("\\def"));
588            tokens.push(T_CS!("\\XKV@header"));
589            tokens.push(T_BEGIN!());
590            tokens.push(T_END!());
591            tokens.push(T_CS!("\\def"));
592            tokens.push(T_CS!("\\XKV@tkey"));
593            tokens.push(T_BEGIN!());
594            tokens.push(T_END!());
595          }
596        }
597      }
598    }
599
600    // Assign rmmacro with collected missing keys
601    if let Some(rm) = rmmacro {
602      tokens.push(T_CS!("\\def"));
603      tokens.push(rm);
604      tokens.push(T_BEGIN!());
605      tokens.extend(rmtokens);
606      tokens.push(T_END!());
607    }
608
609    // Reset all internals if applicable
610    if set_internals {
611      tokens.push(T_CS!("\\def"));
612      tokens.push(T_CS!("\\XKV@fams"));
613      tokens.push(T_BEGIN!());
614      tokens.push(T_END!());
615      tokens.push(T_CS!("\\def"));
616      tokens.push(T_CS!("\\XKV@na"));
617      tokens.push(T_BEGIN!());
618      tokens.push(T_END!());
619    }
620
621    Tokens::new(tokens)
622  }

624  pub fn revert(&self) -> Result<Tokens> {
625    let mut tokens = Vec::new();
626    // iterate over the key-value pairs
627    for tuple in &self.tuples {
628      let KVData {
629        key,
630        value,
631        use_default,
632        keysets: _,
633        primary_keyset,
634        digested_value: _,
635      } = tuple;
636      if !primary_keyset.is_empty() {
637        let reverted = self.revert_keyval(
638          key,
639          primary_keyset,
640          value.as_ref(),
641          *use_default,
642          tokens.is_empty(),
643        )?;
644        tokens.extend(reverted);
645      }
646    }
647    // and return the list of tokens
648    Ok(Tokens::new(tokens))
649  }

651  fn revert_keyval(
652    &self,
653    key: &str,
654    keyset: &str,
655    value_opt: Option<&ArgWrap>,
656    use_default: bool,
657    is_first: bool,
658  ) -> Result<Vec<Token>> {
659    // get the key-value definition
660    let keytype_stored = keyval_get(&keyval_qname(&self.prefix, keyset, key), "type");
661    // define the tokens
662    let mut tokens = Vec::new();
663    // write comma and key, unless in the first iteration
664    if !is_first {
665      tokens.push(T_OTHER!(","));
666    }
667    tokens.extend(Explode!(key));
668    // write the default (if applicable)
669    if !use_default && let Some(value) = value_opt {
670      tokens.push(T_OTHER!("="));
671      let mut reverted_tokens = Vec::new();
672      if let Some(Stored::Parameter(keytype)) = keytype_stored {
673        // TODO: The types here are a little curious. The stored value must be cast back into
674        // Tokens if Parameter's revert works on Tokens. Or should that revert call work on
675        // ArgWrap?
676        if let Some(reverted) = keytype.revert(Some(value.revert()?))? {
677          reverted_tokens.extend(reverted.unlist());
678        }
679      } else {
680        reverted_tokens.extend(value.revert()?.unlist());
681      }
682      tokens.extend(self.rebrace(Tokens::new(reverted_tokens)).unlist());
683    }
684    Ok(tokens)
685  }
686
687  /// When reverting a KeyVals value, we may need to wrap in {}
688  /// eg. if a "," appears outside of any bracing
689  /// Other cases?
690  fn rebrace(&self, tokens: Tokens) -> Tokens {
691    let mut level: i32 = 0;
692    let mut needs_brace = tokens.is_empty();
693    for t in tokens.unlist_ref() {
694      let cc = t.get_catcode();
695      if cc == Catcode::BEGIN {
696        level += 1;
697      }
698      if cc == Catcode::END {
699        level -= 1;
700        // Note that '{ }} {' is still unbalanced
701        // even though the left and right braces match in count.
702        if level < 0 {
703          break;
704        }
705      } else if level <= 0 && cc == Catcode::OTHER && t.with_str(|s| s == ",") {
706        // Outer comma?
707        needs_brace = true;
708        break;
709      }
710    }
711    if needs_brace {
712      let mut wrapped = vec![T_BEGIN!()];
713      wrapped.extend(tokens.unlist());
714      wrapped.push(T_END!());
715      Tokens::new(wrapped)
716    } else {
717      tokens
718    }
719  }

725  pub fn add_value(
726    &mut self,
727    key: &str,
728    value_arg: ArgWrap,
729    use_default: bool,
730    no_rebuild: bool,
731  ) -> Result<()> {
732    // figure out the keyset(s) for the key to be added
733    let keysets = self.resolve_keyval_for(key);
734    let primary_keyset = self.get_primary_keyval(keysets.as_slice()).to_owned();
735
736    // and add the new tuple to the set of tuples
737    let value = if use_default {
738      match keyval_get(&keyval_qname(&self.prefix, &primary_keyset, key), "default") {
739        None => Some(ArgWrap::Tokens(Tokens!())), // bare key with no default: empty value
740        Some(v) => {
741          let arg: Result<ArgWrap> = v.into();
742          Some(arg?)
743        },
744      }
745    } else {
746      Some(value_arg)
747    };
748    self.tuples.push(KVData {
749      key: key.to_string(),
750      value,
751      use_default,
752      keysets,
753      primary_keyset,
754      digested_value: None,
755    });
756    // we now need to rebuild, unless we were asked not to
757    // TODO: Maybe only update the last element?
758    if !no_rebuild {
759      self.rebuild(None);
760    }
761    Ok(())
762  }
763
764  pub fn set_value(&mut self, key: &str, value: ArgWrap, use_default: bool) -> Result<()> {
765    // delete the existing values by skipping key
766    self.rebuild(Some(key));
767    // Perl: if (ref $value eq 'ARRAY') { foreach ... addValue(..., 1) } rebuild()
768    //       elsif (defined($value)) { addValue($key, $value, $useDefault) }
769    //       else { just delete (already done by rebuild above) }
770    match &value {
771      ArgWrap::None => {
772        // undef — just delete (already done by rebuild above)
773        Ok(())
774      },
775      _ => {
776        // single value — set normally
777        self.add_value(key, value, use_default, false)
778      },
779    }
780  }

782  fn rebuild(&mut self, skip_opt: Option<&str>) {
783    // the new data structures to create
784    let mut newtuples: Vec<KVData> = Vec::new();
785    let mut pairs = Vec::new();
786    let mut hash: HashMap<String, Vec<ArgWrap>> = HashMap::default();
787    let mut hash_digested: HashMap<String, Vec<Digested>> = HashMap::default();
788
789    for tuple in self.tuples.drain(..) {
790      // take all the elements we need from the stack
791      let KVData {
792        key,
793        value,
794        use_default,
795        primary_keyset,
796        keysets,
797        digested_value,
798      } = tuple;
799      // if we want to skip some values, we need to store new tuples
800      let key_str = key.as_str();
801      if let Some(skip) = skip_opt
802        && skip == key_str
803      {
804        continue;
805      }
806      if let Some(v) = value.as_ref() {
807        // push key / value into the pair
808        pairs.push((key.clone(), v.clone()));
809
810        // we always use Vec<ArgWrap> storage, just push the new value in
811        let entry = hash.entry(key.clone()).or_default();
812        entry.push(v.clone());
813      } else if let Some(ref dv) = digested_value {
814        // After digestion, value is taken but digested_value is set.
815        // Populate cached_pairs from the digested value (matching Perl's rebuild behavior).
816        let fallback = ArgWrap::Tokens(dv.revert().unwrap_or_default());
817        pairs.push((key.clone(), fallback.clone()));
818        let entry = hash.entry(key.clone()).or_default();
819        entry.push(fallback);
820      }
821      // if we have a digested value, push that in the Vec<Digested> hash storage
822      if let Some(ref dvalue) = digested_value {
823        let entry = hash_digested.entry(key.clone()).or_default();
824        entry.push(dvalue.clone());
825      }
826
827      // Record.
828      newtuples.push(KVData {
829        key,
830        value,
831        use_default,
832        primary_keyset,
833        keysets,
834        digested_value,
835      });
836    }
837    // store all of the values
838    self.cached_pairs = pairs;
839    self.cached_hash = hash;
840    self.cached_hash_digested = hash_digested;
841    self.tuples = newtuples;
842  }

848  // A KeyVal argument MUST be delimited by either braces or brackets (if optional)
849  // This method reads the keyval pairs INCLUDING the delimiters, (rather than
850  // parsing after the fact), since some values may have special catcode needs.
851
852  pub fn read_from(&mut self, until: Token, silence_missing: bool) -> Result<()> {
853    // if we want to force skip_missing keys, we set it up here
854    let skip_missing = self.skip_missing.clone();
855    let hook_missing = self.hook_missing;
856    // if we want to silence all missing errors, store them in a hook
857    if silence_missing {
858      self.skip_missing = SkipMissing::All;
859      self.hook_missing = None;
860    }
861
862    // read the opening token and figure out where we are
863    let startloc = gullet::get_locator();
864    // set and read tokens
865    let _open = gullet::read_token()?;
866
867    let punct_tks = Tokens!(T_OTHER!(","));
868    let until_tks = Tokens!(until);
869    // iterate over all the key-value pairs to read
870    loop {
871      // gobble leading spaces
872      gullet::skip_spaces()?;
873      if gullet::if_next(T_BEGIN!())? {
874        // Protect against redundant {} wrapping
875        gullet::read_token()?;
876        gullet::unread(gullet::read_balanced(ExpansionLevel::Off, false, false)?.strip_braces());
877        gullet::skip_spaces()?;
878      }
879      // Read a single keyword, get a delimiter and a set of keyword tokens
880      let (ktoks, mut delim_opt) = self.read_keyword_from(until)?;
881
882      // if there was no delimiter at the end, we throw an error
883      if delim_opt.is_none() {
884        let message = s!(
885          "Fell off end expecting {} while reading KeyVal key",
886          until.stringify()
887        );
888        let message2 = s!("key started at {}", startloc.to_string());
889        Error!("expected", until, message, message2);
890      }
891
892      // turn the key tokens into a string and trim whitespace
893      let key_str = ktoks.to_string();
894      let key = key_str.trim();
895
896      // if we have a non-empty key
897      if !key.is_empty() {
898        let mut value = ArgWrap::None;
899        // if we have an '=', we explcity assign a value
900        let is_explicit = delim_opt == Some(T_OTHER!("="));
901        if is_explicit {
902          // setup the key-codes to properly read
903          let resolved_kv = self.resolve_keyval_for(key);
904          let keyset = self.get_primary_keyval(&resolved_kv);
905          let keytype_opt = keyval_get(&keyval_qname(&self.prefix, keyset, key), "type");
906          if let Some(Stored::Parameter(ref keytype)) = keytype_opt {
907            keytype.setup_catcodes();
908          }
909          // read until comma
910          let mut toks = Vec::new();
911          loop {
912            // TODO: The types are a bit unnatural here - we need the plural Tokens for read_match,
913            //       but we expect the singular Token as a delimiter result, since we are matching
914            // on a char separator
915            delim_opt = gullet::read_match(&[&punct_tks, &until_tks])?.map(|tks| tks.into());
916            if delim_opt.is_some() {
917              break; // only until we hit a delim.
918            }
919            if let Some(tok) = gullet::read_token()? {
920              // Copy next token to args
921              toks.push(tok);
922              if tok.get_catcode() == Catcode::BEGIN {
923                let balanced_arg = gullet::read_balanced(ExpansionLevel::Off, false, false)?;
924                if !balanced_arg.is_empty() {
925                  toks.extend(balanced_arg.unlist());
926                }
927                toks.push(T_END!());
928              }
929            } else {
930              break;
931            }
932          }
933          // reparse (and expand) the tokens representing the value
934          if !toks.is_empty() {
935            let stripped_toks = Tokens::new(toks).strip_braces_n(2);
936            if !stripped_toks.is_empty() {
937              if let Some(Stored::Parameter(ref keytype)) = keytype_opt {
938                value = keytype.reparse(stripped_toks)?;
939              } else {
940                value = ArgWrap::Tokens(stripped_toks);
941              }
942            }
943          }
944          // An explicit `=` always assigns a value, even when it is empty
945          // (`key=` or `key={}`): that is an EXPLICIT empty override, distinct
946          // from a missing key. Keep it as empty Tokens rather than the
947          // `ArgWrap::None` the value was initialised to — `None` is reserved
948          // for a missing key and its Display is the literal string "None",
949          // which leaks into consumers that stringify the value. Concretely, a
950          // starred matrix with no alignment bracket emits `alignment=` (empty);
951          // without this the keyval value was "None", so `\lx@gen@matrix@bindings`
952          // saw `alignment="None"` instead of defaulting to "c", producing a
953          // malformed column alignment that made a `\dots` cell swallow the next
954          // `&` → "Stray alignment". Witness 1910.00678.
955          if value.is_none() {
956            value = ArgWrap::Tokens(Tokens!());
957          }
958          // and cleanup
959          if let Some(Stored::Parameter(ref keydef)) = keytype_opt {
960            keydef.revert_catcodes()?;
961          }
962        }
963        // and store our value please
964        if !silence_missing || self.can_resolve_keyval_for(key) {
965          self.add_value(key, value, !is_explicit, false)?;
966        }
967      }
968
969      // we finish if we have the last element
970      if delim_opt.as_ref() == Some(&until) {
971        break;
972      }
973    }
974
975    // rebuild and return nothing
976    self.rebuild(None);
977
978    // restore all settings if we silenced the missing keys
979    if silence_missing {
980      self.skip_missing = skip_missing;
981      self.hook_missing = hook_missing;
982    }
983    Ok(())
984  }

986  /// TODO: This is an improvised method for switching KeyVals into Tokens, but losing all collected
987  /// metadata.
988  /// The long-term solution ought to be via a type system extension, where the
989  /// arguments to our before-digest closures are a vector of a new type
990  /// ReadValue ::= [Token, KeyVals, RegisterValue]       potentially?
991  /// On the other hand, we can also put the
992  /// extra effort of *postponing* the build of KV metadata until digestion,
993  /// this way not losing any time reserializing metadata
994  pub fn into_tokens(self) -> Result<Tokens> {
995    let mut tks: Vec<Token> = Vec::new();
996    for (k, v) in self.cached_pairs.into_iter() {
997      tks.push(T_OTHER!(k));
998      match v {
999        ArgWrap::Tokens(vtks) => {
1000          let expanded = gullet::do_expand(vtks)?;
1001          let mut exp_str = expanded.to_string();
1002          if exp_str == "{}" {
1003            exp_str = String::new();
1004          }
1005          tks.push(T_OTHER!(exp_str));
1006        },
1007        ArgWrap::Token(vtk) => tks.push(vtk),
1008        other => {
1009          log::warn!("Unexpected ArgWrap variant in KeyVals revert: {:?}", other);
1010        },
1011      }
1012    }
1013    Ok(Tokens::new(tks))
1014  }
1015}
