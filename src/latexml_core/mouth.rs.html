97impl Default for Mouth {
98  fn default() -> Self {
99    // Historically the source was `"Anonymous String {gid}"` with a
100    // per-instance gid, which Locator::source then pinned into the arena.
101    // The gid served no functional purpose and made every anonymous mouth
102    // unique at the SymStr layer — fine for a handful of mouths, but
103    // catastrophic when a runaway error-recovery path creates millions
104    // (arxiv 1210.4211 under parallel load: 50M anonymous mouths saturated
105    // the u32 interner offset). Collapsing onto a shared static label makes
106    // the per-mouth cost arena-free, and the pin-count sentinel remains as
107    // a symptom detector for the *actual* bug (something is still creating
108    // 50M anonymous mouths — that's a runaway loop to track down, now with
109    // the arena side-effect removed).
110    Mouth {
111      notes:                  false,
112      note_message:           None,
113      fordefinitions:         false,
114      at_letter:              false,
115      at_eof:                 false,
116      skipping_spaces:        false,
117      lineno:                 0,
118      colno:                  0,
119      chars:                  VecDeque::new(),
120      nchars:                 0,
121      source:                 String::from("Anonymous String"),
122      shortsource:            s!("String"),
123      // handle : None,
124      foodtype:               FoodType::File,
125      saved_at_cc:            None,
126      saved_include_comments: None,
127      buffer:                 VecDeque::new(),
128      raw_buffer:             VecDeque::new(),
129      reader:                 None,
130    }
131  }

139  fn get_locator(&self) -> Locator {
140    let (to_line, to_column) = (self.lineno, self.colno);
141    let max_col = if self.nchars > 0 {
142      self.nchars - 1
143    } else {
144      self.nchars
145    }; // There is always a trailing EOL char, if any
146    let (from_line, from_column) = if to_column > 0 && to_column >= max_col {
147      (to_line, 0)
148    } else {

152    // internal colno counter is 0-indexed (character array index), so we add 1
153    // when producing the Locator for error-message display.
154    Locator::new(
155      &self.source,
156      from_line as u32,
157      (from_column + 1) as u32,
158      to_line as u32,
159      (to_column + 1) as u32,
160    )
161  }

175  pub fn create(source: &str, mut options: MouthOptions) -> Result<Self> {
176    if let Some(content) = options.content.take() {
177      // we've cached the content of this source
178      let (_dir, name, ext) = pathname::split(source);
179      options.source = Some(source.to_string());
180      options.shortsource = Some(s!("{}.{}", name, ext));
181      Mouth::new(&content, Some(options))
182    } else if source.starts_with("literal:") {
183      let source = source.replacen("literal:", "", 1);
184      // we've supplied literal data
185      options.source = None; // the source does not have a corresponding file name
186      options.foodtype = FoodType::opt_from_str("literal");
187      Mouth::new(&source, Some(options))
188    } else if source.is_empty() {
189      Mouth::new("", Some(options))
190    } else {
191      let (_dir, name, ext) = pathname::split(source);
192      options.foodtype = FoodType::opt_from_str(&pathname::protocol(source));
193      options.source = Some(source.to_string());
194      if options.shortsource.is_none() {
195        options.shortsource = Some(if ext.is_empty() {
196          name
197        } else {
198          s!("{}.{}", name, ext)
199        });
200      }
201      Mouth::new(source, Some(options))
202    }
203  }

241  fn open_file(&mut self, pathname: &str) -> Result<()> {
242    if self.foodtype == FoodType::File {
243      // Perl: check readable, then check binary (non-empty), then open
244      let metadata = std::fs::metadata(pathname);
245      match &metadata {
246        Err(e) if e.kind() == io::ErrorKind::NotFound => {
247          fatal!(Mouth, MissingFile, s!("Can't find file {}", pathname));
248        },
249        Err(e) if e.kind() == io::ErrorKind::PermissionDenied => {
250          Error!(
251            "I/O",
252            "unreadable",
253            s!("File {} is not readable. Ignoring.", pathname),
254            "",
255            "",
256            self.get_location()
257          );
258          return Ok(());
259        },
260        Err(e) => {
261          return Err(io::Error::new(e.kind(), e.to_string()).into());
262        },
263        Ok(meta) => {
264          // Check for binary file (non-empty and appears binary)
265          // Perl's -B heuristic: check first block for high proportion of non-text bytes
266          if meta.len() > 0 {
267            if let Ok(mut f) = File::open(pathname) {
268              let mut buf = [0u8; 512];
269              if let Ok(n) = f.read(&mut buf) {
270                if n > 0 {
271                  let non_text = buf[..n]
272                    .iter()
273                    .filter(|&&b| {
274                      b == 0 || (b < 0x20 && b != b'\n' && b != b'\r' && b != b'\t' && b != 0x1b)
275                    })
276                    .count();
277                  if non_text * 3 > n {
278                    // High ratio of non-text bytes — likely binary
279                    Error!(
280                      "invalid",
281                      "binary",
282                      s!("Input file {} appears to be binary. Ignoring.", pathname),
283                      "",
284                      "",
285                      self.get_location()
286                    );
287                    return Ok(());
288                  }
289                }
290              }
291            }
292          }
293        },
294      }
295      let f = match File::open(pathname) {
296        Ok(f) => f,
297        Err(e) => {
298          Error!(
299            "I/O",
300            "open",
301            s!("Can't open {} for reading: {}", pathname, e),
302            "",
303            "",
304            self.get_location()
305          );
306          return Err(e.into());
307        },
308      };
309      let reader = BufReader::new(f);
310      self.reader = Some(reader);
311      self.buffer = VecDeque::new();
312      self.raw_buffer = VecDeque::new();
313    }
314    Ok(())
315  }

325  fn initialize(&mut self) {
326    self.note_message = if self.notes {
327      let source = if !self.source.is_empty() {
328        &self.source
329      } else {
330        "Anonymous String"
331      };
332      let kind = if self.fordefinitions {
333        "definitions"
334      } else {
335        "content"
336      };
337      let at_note = if self.fordefinitions && !self.at_letter {
338        " w/@ other"
339      } else {
340        ""
341      };
342      Some(s!("Processing {}{} {}", kind, at_note, source))
343    } else {
344      None
345    };
346    // Perl Mouth.pm L97: ProgressSpinup($$self{note_message}) — emit
347    // `(Processing definitions <source>...` when this mouth begins reading.
348    // The matching ProgressSpindown (Mouth.pm L121) is in `finish()` below.
349    if let Some(ref msg) = self.note_message {
350      note_begin(msg);
351    }
352    // Perl: at_letter saves/restores @ catcode independently of fordefinitions.
353    // Use Scope::Global to ensure it persists across scope frame pops during file loading.
354    if self.at_letter {
355      self.saved_at_cc = lookup_catcode('@');
356      assign_catcode('@', Catcode::LETTER, Some(Scope::Global));
357    }
358    // Perl: fordefinitions saves/restores INCLUDE_COMMENTS
359    if self.fordefinitions {
360      self.saved_include_comments = match lookup_value("INCLUDE_COMMENTS") {
361        Some(Stored::Bool(x)) => Some(x),
362        _ => None,
363      };
364      assign_value("INCLUDE_COMMENTS", false, Some(Scope::Local));
365    }
366  }

380  /// Fully finish this mouth: stop reading AND restore catcodes/state.
381  /// Called by close_mouth when the mouth is popped from the stack.
382  pub fn finish(&mut self) {
383    self.stop_reading();
384    // Perl: at_letter restores @ catcode (independent of fordefinitions).
385    // Use Scope::Global to ensure it takes effect regardless of scope frame state.
386    if self.at_letter {
387      let cc = self.saved_at_cc.take().unwrap_or(Catcode::OTHER);
388      assign_catcode('@', cc, Some(Scope::Global));
389    }
390    // Perl: fordefinitions restores INCLUDE_COMMENTS
391    if let Some(sic) = self.saved_include_comments.take() {
392      assign_value("INCLUDE_COMMENTS", sic, Some(Scope::Local))
393    }
394    if self.notes {
395      if let Some(ref msg) = self.note_message {
396        note_end(msg);
397      }
398    }
399  }

402  /// This is (hopefully) a platform independent way of splitting a string
403  /// into "lines" ending with CRLF, CR or LF (DOS, Mac or Unix).
404  /// Note that TeX considers newlines to be \r, ie CR, ie ^^M
405  fn split_lines(lines: &str) -> VecDeque<String> {
406    let mut lines: VecDeque<String> = LINEBREAK_REGEX.split(lines).map(str::to_owned).collect();
407    if let Some(last_line) = lines.back() {
408      if last_line.is_empty() {
409        lines.pop_back();
410      }
411    }
412    lines
413  }

415  /// Split raw bytes into lines without decoding, splitting on \r\n, \r, or \n.
416  fn split_raw_lines(bytes: &[u8]) -> VecDeque<Vec<u8>> {
417    let mut lines = VecDeque::new();
418    let mut start = 0;
419    let mut i = 0;
420    while i < bytes.len() {
421      if bytes[i] == b'\r' {
422        lines.push_back(bytes[start..i].to_vec());
423        if i + 1 < bytes.len() && bytes[i + 1] == b'\n' {
424          i += 1; // skip \n after \r
425        }
426        start = i + 1;
427      } else if bytes[i] == b'\n' {
428        lines.push_back(bytes[start..i].to_vec());
429        start = i + 1;
430      }
431      i += 1;
432    }
433    // Add remaining bytes (last line without trailing newline)
434    if start < bytes.len() {
435      lines.push_back(bytes[start..].to_vec());
436    }
437    lines
438  }

440  /// Decode a raw byte line using the current encoding setting.
441  /// Matches Perl's per-line decode behavior.
442  fn decode_bytes(raw_line: &[u8], location: String) -> String {
443    if let Some(ref encoding_sym) = get_input_encoding() {
444      // Probe the encoding without allocating — this fires per input
445      // line, so even a small heap alloc per call adds up on large
446      // documents. Only resolve the symbol to an owned String when we
447      // actually need it for the misdefined-encoding Info! message.
448      let is_latin1 = crate::common::arena::with(*encoding_sym, |s| {
449        s.eq_ignore_ascii_case("iso-8859-1")
450          || s.eq_ignore_ascii_case("latin1")
451          || s.eq_ignore_ascii_case("latin-1")
452      });
453      let file_str = if is_latin1 {
454        raw_line.iter().map(|&b| b as char).collect::<String>()
455      } else {
456        // Fallback: try UTF-8 with lossy conversion
457        String::from_utf8_lossy(raw_line).into_owned()
458      };
459      let replaced = file_str.replace('\u{FFFD}', " ");
460      if replaced.len() != file_str.len() {
461        let encoding_name = crate::common::arena::to_string(*encoding_sym);
462        Info!(
463          "misdefined",
464          &encoding_name,
465          s!("input isn't valid under encoding {}", &encoding_name),
466          "",
467          "",
468          location
469        );
470      }
471      replaced
472    } else {
473      // No encoding set — interpret as UTF-8, with fallback.
474      // For non-UTF-8 bytes, we keep them as raw Latin-1 chars.
475      // This matches Perl which passes raw bytes through when
476      // PERL_INPUT_ENCODING is undef (disabled by inputenc).
477      match str::from_utf8(raw_line) {
478        Ok(s) => s.to_string(),
479        Err(_) => {
480          // When no encoding is set but bytes aren't valid UTF-8,
481          // treat as raw bytes (Latin-1 passthrough).
482          // This happens after inputenc disables PERL_INPUT_ENCODING
483          // and the remaining file lines contain high bytes.
484          raw_line.iter().map(|&b| b as char).collect::<String>()
485        },
486      }
487    }
488  }

490  /// Original LaTeXML:
491  /// This is (hopefully) a correct way to split a line into "chars",
492  /// or what is probably more desired is "Grapheme clusters" (even "extended")
493  /// These are unicode characters that include any following combining chars, accents & such.
494  /// I am thinking that when we deal with unicode this may be the most correct way?
495  /// If it's not the way XeTeX does it, perhaps, it must be that ALL combining chars
496  /// have to be converted to the proper accent control sequences!
497  fn get_next_line(&mut self) -> Option<String> {
498    if self.buffer.is_empty() && !self.raw_buffer.is_empty() {
499      // Decode the next raw byte line lazily using the current encoding.
500      // This matches Perl's approach: each line is decoded with the encoding
501      // that is active at the time the line is read, allowing inputenc to
502      // change encoding mid-file.
503      if let Some(raw_line) = self.raw_buffer.pop_front() {
504        let decoded = Mouth::decode_bytes(&raw_line, self.get_location());
505        self.buffer.push_back(decoded);
506      }
507    }
508    if self.buffer.is_empty() {
509      if let Some(ref mut reader) = self.reader {
510        // file mouth case — read all bytes, split into raw lines, decode lazily
511        let mut file_bytes = Vec::new();
512        let _num_bytes = match reader.read_to_end(&mut file_bytes) {
513          Ok(count) => count,
514          Err(e) => {
515            let message = s!("BufReader::read_to_end returned an error: {:?}", e);
516            Warn!("mouth", "io", message, "", "", self.get_location());
517            0
518          },
519        };
520        // remove the now exhausted reader
521        self.reader.take();
522        // Split raw bytes into lines without decoding (preserving raw bytes).
523        // Each line is decoded lazily via decode_bytes() using the CURRENT encoding.
524        self.raw_buffer = Mouth::split_raw_lines(&file_bytes);
525        // Decode the first line now
526        if let Some(raw_line) = self.raw_buffer.pop_front() {
527          let decoded = Mouth::decode_bytes(&raw_line, self.get_location());
528          self.buffer.push_back(decoded);
529        }
530      }
531    }
532    self.buffer.pop_front()
533  }

535  /// Get the next character & it's catcode from the current line of input, even ignored chars,
536  /// handling TeX's "^^" encoding.
537  /// Note that this is the only place where catcode lookup is done (well almost),
538  /// and that it is somewhat `inlined'.
539  fn get_next_char(&mut self) -> Option<(char, Catcode)> {
540    if self.colno >= self.nchars {
541      return None;
542    };
543    let ch_opt = self.chars.get(self.colno);
544    self.colno += 1;
545    if let Some(ch) = ch_opt {
546      let mut ch = *ch;
547      let mut cc = lookup_catcode(ch).unwrap_or(Catcode::OTHER);
548      // Possible convert ^^x
549      // Perl: (cc == CC_SUPER) && (colno + 1 < nchars) && (ch == chars[colno])
550      if cc == Catcode::SUPER
551        && self.colno + 1 < self.nchars
552        && Some(&ch) == self.chars.get(self.colno)
553      {
554        let c1_opt = self.chars.get(self.colno + 1);
555        let c2_opt = self.chars.get(self.colno + 2);
556        let mut two_hex = false;
557        // ^^ followed by TWO LOWERCASE Hex digits???
558        if let Some(c1) = c1_opt {
559          if let Some(c2) = c2_opt {
560            // Perf: avoid per-char String alloc + regex match by using
561            // direct ASCII class check. LOWERHEX_REGEX = ^[0-9a-f]$, i.e.
562            // lowercase hex digits only.
563            let is_lowerhex = |c: char| -> bool { matches!(c, '0'..='9' | 'a'..='f') };
564            if (self.colno + 2 < self.nchars) && is_lowerhex(*c1) && is_lowerhex(*c2) {
565              // TODO: Maybe Result type warranted here?
566              let hex = u8::from_str_radix(&s!("{}{}", c1, c2), 16).unwrap();
567              ch = hex as char;
568              self.splice(self.colno - 1..self.colno + 3, &[ch]);
569              self.nchars -= 3;
570              two_hex = true;
571            }
572          }
573        }
574        if !two_hex {
575          // OR ^^ followed by a SINGLE Control char type code???
576          let c = self.chars[self.colno + 1];
577          let cn = c as i16;
578
579          ch = (cn + if cn >= 64 { -64 } else { 64 }) as u8 as char;
580          self.splice(self.colno - 1..self.colno + 2, &[ch]);
581          self.nchars -= 2;
582        }
583        cc = lookup_catcode(ch).unwrap_or(Catcode::OTHER);
584      }
585      Some((ch, cc))
586    } else {
587      None
588    }
589  }

591  /// Checks if there is more input to process.
592  ///
593  /// Note: we need mutability, as we may refill the internal BufReader
594  /// when performing the check.
595  pub fn has_more_input(&mut self) -> bool {
596    if !self.is_eol() || !self.buffer.is_empty() || !self.raw_buffer.is_empty() {
597      return true;
598    }
599    // Peek the underlying reader if present. A fill_buf I/O error is treated
600    // as end-of-input (return false) rather than panicking — the caller will
601    // naturally stop requesting tokens and the Mouth will be closed out.
602    match self.reader.as_mut() {
603      Some(r) => r.fill_buf().map(|buf| !buf.is_empty()).unwrap_or(false),
604      None => false,
605    }
606  }

608  /// Read the next token, or undef if exhausted.
609  /// Note that this also returns COMMENT tokens containing source comments,
610  /// and also locator comments (file, line# info).
611  /// LaTeXML::Core::Gullet intercepts them and passes them on at appropriate times.
612  pub fn read_token(&mut self) -> Option<Token> {
613    loop {
614      // Iterate till we find a token, or run out. (use return)
615      // ===== Get next line, if we need to.
616      if self.colno >= self.nchars {
617        self.lineno += 1;
618        self.colno = 0;
619        let line_opt = self.get_next_line();
620        // For \read, we have to return something for EOL, and handle implicit final newline
621        let read_mode = lookup_int("PRESERVE_NEWLINES") > 1;
622        let eolch = if let Some(defn) = lookup_definition(&T_CS!("\\endlinechar")).unwrap() {
623          if defn.is_register() {
624            if let Some(eol) = defn.value_of(Vec::new()) {
625              let eol = eol.value_of() as i16;
626              if eol > 0 && eol <= 255 {
627                let mch = (eol as u8) as char;
628                Some(mch)
629              } else {
630                None
631              }
632            } else {
633              None
634            }
635          } else {
636            None
637          }
638        } else {
639          Some('\r')
640        };
641        if line_opt.is_none() {
642          // Exhausted the input.
643          let eolcc = if let Some(ch) = eolch {
644            lookup_catcode(ch).unwrap_or(Catcode::OTHER)
645          } else {
646            Catcode::OTHER
647          };
648          let eoftoken = if let Some(eolch_content) = eolch {
649            if read_mode && !self.at_eof && !self.source.is_empty() {
650              if eolcc == Catcode::EOL {
651                Some(T_CS!("\\par"))
652              } else {
653                Some(CharToken!(eolch_content, eolcc))
654              }
655            } else {
656              None
657            }
658          } else {
659            None
660          };
661          self.at_eof = true;
662          self.chars = VecDeque::new();
663          self.nchars = 0;
664          return eoftoken;
665        }
666        // Remove trailing spaces from external sources
667        let mut line = line_opt.unwrap();
668        if !self.source.is_empty() && line.ends_with(' ') {
669          line = TRAILING_SPACE_CHARS.replace(&line, "").to_string();
670        }
671        // Then append the appropriate \endlinechar, or "\r";
672        if let Some(ch) = eolch {
673          line.push(ch);
674        }
675
676        self.chars = line.chars().collect::<VecDeque<char>>();
677        self.nchars = self.chars.len();
678        // In state N, skip leading spaces & ignored, possibly decoding (trailing space removed
679        // above)
680        while let Some((_ch, cc)) = self.get_next_char() {
681          match cc {
682            Catcode::SPACE | Catcode::IGNORE => {},
683            Catcode::EOL => {
684              // Eolch already? empty line!
685              self.colno = self.nchars; // ignore rest of line.
686              return Some(T_CS!("\\par"));
687            },
688            _ => break,
689          }
690        }
691        if self.nchars == 0 || self.colno > self.nchars {
692          // Past end of line?
693          // If upcoming line is empty, and there is no recognizable EOL, fake one
694          if read_mode && eolch != Some('\r') {
695            return Some(T_MARKER!("EOL"));
696          }
697        } else {
698          // Back up over peeked char
699          self.colno -= 1;
700        }
701        // Sneak a comment out, every so often.
702        if self.lineno.is_multiple_of(READLINE_PROGRESS_QUANTUM) && lookup_bool("INCLUDE_COMMENTS")
703        {
704          // Perl T_COMMENT prepends '%' (Token.pm L81)
705          return Some(T_COMMENT!(s!(
706            "%**** {} Line {} ****",
707            &self.shortsource,
708            &self.lineno.to_string()
709          )));
710        }
711      }
712      // In state::S, skip spaces
713      if self.skipping_spaces {
714        let mut cc = None;
715        // This is very awkward as a loop,
716        //  but I had to port the Perl logic without going crazy...
717        // tokenizer/verb.tex depends on it.
718        while let Some((_, ncc)) = self.get_next_char() {
719          cc = Some(ncc);
720          if ncc != Catcode::SPACE {
721            break;
722          }
723        }
724        if self.colno <= self.nchars && cc.is_some() && cc != Some(Catcode::SPACE) {
725          self.colno -= 1;
726        }
727        if cc == Some(Catcode::EOL) {
728          // If we've got an EOL
729          self.get_next_char();
730          if self.colno < self.nchars {
731            self.colno -= 1;
732          }
733        }
734        self.skipping_spaces = false;
735      }
736      // ==== Extract next token from line.
737      if let Some((ch, cc)) = self.get_next_char() {
738        if let Some(token) = Mouth::dispatch_char(self, ch, cc) {
739          return Some(token);
740        } // Else, repeat till we get something or run out.
741      }
742    }
743  }

746  /// Read all tokens until a token equal to $until (if given), or until exhausted.
747  /// Returns an empty Tokens list, if there is no input
748  pub fn read_tokens(&mut self) -> Tokens {
749    let mut tokens = Vec::new();
750    while let Some(token) = self.read_token() {
751      tokens.push(token);
752    }
753    while let Some(Token { code: Catcode::SPACE, .. }) = tokens.last() {
754      // Remove trailing space
755      tokens.pop();
756    }
757    Tokens::new(tokens)
758  }

760  //**********************************************************************
761  // Read a raw lines; there are so many variants of how it should end,
762  // that the Mouth API is left as simple as possible.
763  // Alas: $noread true means NOT to read a new line, but only return
764  // the remainder of the current line, if any. This is useful when combining
765  // with previously peeked tokens from the Gullet.
766  pub fn read_raw_line(&mut self, noread: bool) -> Option<String> {
767    let mut line = String::new();
768    if self.colno < self.nchars {
769      line = self.chars.iter().skip(self.colno).collect();
770      // Strip the final carriage return, if it has been added back (Perl: s/\r$//s)
771      if line.ends_with('\r') {
772        line.pop();
773      }
774      self.colno = self.nchars;
775    } else if !noread {
776      match self.get_next_line() {
777        None => {
778          // We've exhausted this mouth
779          self.at_eof = true;
780          self.chars = VecDeque::new();
781          self.nchars = 0;
782          self.colno = 0;
783          return None;
784        },
785        Some(next_line) => {
786          // Strip trailing spaces (Perl: s/ *$//s)
787          line = next_line.trim_end_matches(' ').to_string();
788          self.lineno += 1;
789          self.chars = line.chars().collect();
790          self.nchars = self.chars.len();
791          self.colno = self.nchars;
792        },
793      }
794    }
795    Some(line)
796  }

798  fn dispatch_char(&mut self, ch: char, cc: Catcode) -> Option<Token> {
799    // Possibly want to think about caching (common) letters, etc to keep from
800    // creating tokens like crazy... or making them more compact... or ???
801    use crate::token::Catcode::*;
802    match cc {
803      ESCAPE => self.handle_escape(), // T_ESCAPE
804      BEGIN => {
805        if ch == '{' {
806          Some(T_BEGIN!())
807        } else {
808          Some(CharToken!(ch, BEGIN))
809        }
810      },
811      END => {
812        if ch == '}' {
813          Some(T_END!())
814        } else {
815          Some(CharToken!(ch, END))
816        }
817      },
818      MATH => {
819        if ch == '$' {
820          Some(T_MATH!())
821        } else {
822          Some(CharToken!(ch, MATH))
823        }
824      },
825      ALIGN => {
826        if ch == '&' {
827          Some(T_ALIGN!())
828        } else {
829          Some(CharToken!(ch, ALIGN))
830        }
831      },
832      EOL => Some(self.handle_end_of_line()),
833      PARAM => {
834        if ch == '#' {
835          Some(T_PARAM!())
836        } else {
837          Some(CharToken!(ch, PARAM))
838        }
839      }, // T_PARAM
840      SUPER => {
841        if ch == '^' {
842          Some(T_SUPER!())
843        } else {
844          Some(CharToken!(ch, SUPER))
845        }
846      }, // T_SUPER
847      SUB => {
848        if ch == '_' {
849          Some(T_SUB!())
850        } else {
851          Some(CharToken!(ch, SUB))
852        }
853      }, // T_SUB
854      SPACE => self.handle_space(),
855      LETTER => Some(CharToken!(ch, Catcode::LETTER)),
856      OTHER => Some(CharToken!(ch, Catcode::OTHER)),
857      ACTIVE => Some(T_ACTIVE!(ch)),
858      COMMENT => self.handle_comment(),
859      INVALID => Some(CharToken!(ch, Catcode::OTHER)), // T_INVALID (we could get unicode!)
860      _ => None,                                       // IGNORE, others
861    }
862  }

864  fn handle_end_of_line(&mut self) -> Token {
865    self.colno = self.nchars; // Ignore any remaining characters after EOL
866    if lookup_int("PRESERVE_NEWLINES") != 0 {
867      Token!("\n", Catcode::SPACE)
868    } else {
869      T_SPACE!()
870    }
871  }
872
873  fn handle_space(&mut self) -> Option<Token> {
874    // Skip any following spaces!
875    while let Some((_ch, cc)) = self.get_next_char() {
876      if (cc != Catcode::SPACE) && (cc != Catcode::EOL) {
877        // backup at nonspace/eol
878        if self.colno <= self.nchars {
879          self.colno -= 1;
880        }
881        break;
882      }
883    }
884    Some(T_SPACE!())
885  }
886
887  fn handle_comment(&mut self) -> Option<Token> {
888    let n = self.colno;
889    self.colno = self.nchars;
890    let mut comment = String::new();
891    for c in self.chars.iter().skip(n).take(self.nchars - n) {
892      comment.push(*c);
893    }
894    let trimmed_comment = comment.trim();
895    if !trimmed_comment.is_empty() && lookup_bool("INCLUDE_COMMENTS") {
896      // Perl T_COMMENT prepends '%' to the comment text (Token.pm L81)
897      Some(T_COMMENT!(s!("%{}", trimmed_comment)))
898    } else if lookup_int("PRESERVE_NEWLINES") > 1 {
899      Some(T_MARKER!("EOL")) // Required EOL during \read
900    } else {
901      None
902    }
903  }

906  // See The TeXBook, Chapter 8, The Characters You Type, pp.46--47.
907  //**********************************************************************
908
909  /// Read control sequence
910  fn handle_escape(&mut self) -> Option<Token> {
911    // NOTE: We're using control sequences WITH the \ prepended!!!
912    if let Some((ch, mut cc)) = self.get_next_char() {
913      // Knuth, p.46 says that Newlines are converted to spaces,
914      // Bit I believe that he does NOT mean within control sequences
915      let mut cs = s!("\\{}", ch);
916      if cc == Catcode::LETTER {
917        // For letter, read more letters for csname.
918        while let Some((nch, ncc)) = self.get_next_char() {
919          cc = ncc;
920          if ncc == Catcode::LETTER {
921            cs.push(nch);
922          } else {
923            break;
924          }
925        }
926        // We WILL skip spaces, but not till next token is read (in case catcode changes!!!!)
927        self.skipping_spaces = true;
928        if cc != Catcode::LETTER {
929          self.colno -= 1;
930        }
931      }
932      Some(T_CS!(cs))
933    } else {
934      None
935    }
936  }

939  fn splice<R>(&mut self, range: R, with: &[char])
940  where R: RangeBounds<usize> {
941    let mut v: Vec<char> = self.chars.drain(..).collect();
942    v.splice(range, with.iter().cloned());
943    self.chars = v.into_iter().collect();

946  /// Checks if Mouth read is at the end of a line.
947  ///
948  /// Careful:
949  /// used BOTH for flushing input for `\endinput`
950  /// and for detecting line end for `\read`
951  pub fn is_eol(&mut self) -> bool {
952    let savecolno = self.colno;
953    // We have to peek past any ignored tokens & also spaces, if skipping
954    let mut cc = None;
955    while let Some((_, ncc)) = self.get_next_char() {
956      if ncc != Catcode::IGNORE && (!self.skipping_spaces || ncc != Catcode::SPACE) {
957        cc = Some(ncc);
958        break;
959      }
960    }
961    if self.colno <= self.nchars && cc.is_some() {
962      // Back-up if too far.
963      self.colno -= 1;
964    }
965    // If skipping spaces (really, reading for input (\endinput) ?), jump to end of EOL or comments
966    if self.skipping_spaces && (cc == Some(Catcode::EOL) || cc == Some(Catcode::COMMENT)) {
967      // If we've got an EOL | COMMENT
968      self.colno = self.nchars
969    }
970    let eol = self.colno >= self.nchars;
971    self.colno = savecolno;
972    eol
973  }

983pub fn tokenize(text: &str) -> Tokens {
984  // special case! empty input is empty Tokens
985  if text.is_empty() {
986    return NO_TOKENS;
987  }
988  state::use_std_state();
989  let result = Mouth::new(text, None).unwrap().read_tokens();
990  state::use_main_state();
991  result
992}
