1//! Generate Rust source code from a kernel dump file.
2//!
3//! Reads the text dump produced by `dump_writer.rs` and emits a `.rs` file
4//! containing typed static data tables. The compiler checks every entry at
5//! build time — no runtime parsing, no text format errors possible.
6//!
7//! Usage:
8//!   1. Generate dump: `latexml_oxide --init=latex.ltx --dest=/tmp/dump.tmp`
9//!   2. Generate Rust: `dump_codegen::generate_rs("/tmp/dump.tmp", "latex_dump.rs")`
10//!   3. Place in `latexml_package/src/engine/latex_dump.rs`
11//!   4. Load at runtime via `latex_dump::load_definitions()`
12
13use std::io::Write;
14use std::path::Path;

16/// Value entries to skip (runtime-specific or cause regressions).
17const SKIP_VALUES: &[&str] = &[
18  "INTERPRETING_DEFINITIONS",
19  "if_count",
20  "absorb_count",
21  "_loaded",
22  "INCLUDE_COMMENTS",
23  "INCLUDE_STYLES",
24  "\\everyjob",
25  "\\toks",
26  "input_file:",
27  "output_file:",
28  "texsys",
29];
30
31fn should_skip_value(key: &str) -> bool {
32  SKIP_VALUES
33    .iter()
34    .any(|skip| key == *skip || key.starts_with(skip) || key.ends_with(skip) || key.contains(skip))
35}

37/// Escape a string for use as a Rust string literal.
38fn rust_escape(s: &str) -> String {
39  if s.contains("\"#")
40    || s.contains('\r')
41    || s
42      .chars()
43      .any(|c| c.is_ascii_control() && c != '\n' && c != '\t')
44  {
45    let mut out = String::with_capacity(s.len() + 8);
46    out.push('"');
47    for ch in s.chars() {
48      match ch {
49        '\\' => out.push_str("\\\\"),
50        '"' => out.push_str("\\\""),
51        '\n' => out.push_str("\\n"),
52        '\r' => out.push_str("\\r"),
53        '\t' => out.push_str("\\t"),
54        c if c.is_ascii_control() => out.push_str(&format!("\\x{:02x}", c as u8)),
55        c => out.push(c),
56      }
57    }
58    out.push('"');
59    out
60  } else {
61    format!("r#\"{}\"#", s)
62  }
63}

65fn url_decode(s: &str) -> String {
66  let mut result = String::with_capacity(s.len());
67  let mut chars = s.chars();
68  while let Some(ch) = chars.next() {
69    if ch == '%' {
70      let hex: String = chars.by_ref().take(2).collect();
71      if let Ok(byte) = u8::from_str_radix(&hex, 16) {
72        result.push(byte as char);
73      }
74    } else {
75      result.push(ch);
76    }
77  }
78  result
79}

152  ints:          Vec<IntEntry>,
153  strings:       Vec<StrEntry>,
154  dims:          Vec<DimEntry>,
155  glues:         Vec<GlueEntry>,
156  mudims:        Vec<DimEntry>,
157  muglues:       Vec<GlueEntry>,
158  charcodes:     Vec<CharcodeEntry>,
159  catcode_vals:  Vec<CatcodeEntry>,
160  single_tokens: Vec<TokenEntry>,
161  token_lists:   Vec<TokenListEntry>,
162  nones:         Vec<String>,
163  vecdeques:     Vec<String>,
164
165  // Code tables
166  catcodes:  Vec<CatcodeEntry>,
167  lccodes:   Vec<CharcodeEntry>,
168  uccodes:   Vec<CharcodeEntry>,
169  sfcodes:   Vec<CharcodeEntry>,
170  delcodes:  Vec<CharcodeEntry>,
171  mathcodes: Vec<CharcodeEntry>,

174  let_defs:    Vec<TokenEntry>,

210/// Parse dump content into categorized entries.
211fn parse_dump(content: &str) -> DumpData {
212  let mut data = DumpData::default();
213
214  for line in content.lines() {
215    let line = line.trim();
216    if line.is_empty() || line.starts_with('#') {
217      continue;
218    }
219    let parts: Vec<&str> = line.splitn(3, '\t').collect();
220    if parts.len() < 2 {
221      continue;
222    }
223    let table = parts[0];
224    let key = url_decode(parts[1]);
225    let rest = if parts.len() > 2 { parts[2] } else { "" };
226
227    match table {
228      "V" => parse_value(&mut data, &key, rest),
229      "M" => parse_meaning(&mut data, &key, rest),
230      "C" => {
231        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
232        if val_parts.len() >= 2 {
233          if let Ok(v) = val_parts[1].parse::<u8>() {
234            data.catcodes.push(CatcodeEntry { key, val: v });
235          }
236        }
237      },
238      "LC" => {
239        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
240        if val_parts.len() >= 2 {
241          if let Ok(v) = val_parts[1].parse::<u16>() {
242            data.lccodes.push(CharcodeEntry { key, val: v });
243          }
244        }
245      },
246      "UC" => {
247        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
248        if val_parts.len() >= 2 {
249          if let Ok(v) = val_parts[1].parse::<u16>() {
250            data.uccodes.push(CharcodeEntry { key, val: v });
251          }
252        }
253      },
254      "SC" => {
255        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
256        if val_parts.len() >= 2 {
257          if let Ok(v) = val_parts[1].parse::<u16>() {
258            data.sfcodes.push(CharcodeEntry { key, val: v });
259          }
260        }
261      },
262      "DC" => {
263        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
264        if val_parts.len() >= 2 {
265          if let Ok(v) = val_parts[1].parse::<u16>() {
266            data.delcodes.push(CharcodeEntry { key, val: v });
267          }
268        }
269      },
270      "MC" => {
271        let val_parts: Vec<&str> = rest.splitn(2, '\t').collect();
272        if val_parts.len() >= 2 {
273          if let Ok(v) = val_parts[1].parse::<u16>() {
274            data.mathcodes.push(CharcodeEntry { key, val: v });
275          }
276        }
277      },
278      _ => {}, // Skip unknown tables
279    }
280  }
281
282  data
283}

285fn parse_value(data: &mut DumpData, key: &str, rest: &str) {
286  // Skip \ver@ and other runtime-specific entries
287  if key.starts_with("\\ver@") || should_skip_value(key) {
288    return;
289  }
290  let parts: Vec<&str> = rest.splitn(2, '\t').collect();
291  if parts.is_empty() {
292    return;
293  }
294  let val_data = parts.get(1).copied().unwrap_or("");
295
296  match parts[0] {
297    "N" => data.nones.push(key.to_string()),
298    "B" => data.bools.push(BoolEntry {
299      key: key.to_string(),
300      val: val_data == "1",
301    }),
302    "I" => {
303      if let Ok(v) = val_data.parse::<i64>() {
304        data.ints.push(IntEntry { key: key.to_string(), val: v });
305      }
306    },
307    "S" => data.strings.push(StrEntry {
308      key: key.to_string(),
309      val: url_decode(val_data),
310    }),
311    "CH" => {
312      if let Ok(v) = val_data.parse::<u16>() {
313        data
314          .charcodes
315          .push(CharcodeEntry { key: key.to_string(), val: v });
316      }
317    },
318    "CC" => {
319      if let Ok(v) = val_data.parse::<u8>() {
320        data
321          .catcode_vals
322          .push(CatcodeEntry { key: key.to_string(), val: v });
323      }
324    },
325    "D" => {
326      if let Ok(v) = val_data.parse::<i64>() {
327        data.dims.push(DimEntry { key: key.to_string(), val: v });
328      }
329    },
330    "G" => {
331      if let Some(g) = parse_glue_data(key, val_data) {
332        data.glues.push(g);
333      }
334    },
335    "MD" => {
336      if let Ok(v) = val_data.parse::<i64>() {
337        data.mudims.push(DimEntry { key: key.to_string(), val: v });
338      }
339    },
340    "MG" => {
341      if let Some(g) = parse_glue_data(key, val_data) {
342        data.muglues.push(g);
343      }
344    },
345    "T" => {
346      if let Some((cc, text)) = parse_tok(val_data) {
347        data
348          .single_tokens
349          .push(TokenEntry { key: key.to_string(), cc, text });
350      }
351    },
352    "TK" => {
353      if val_data.is_empty() {
354        data.token_lists.push(TokenListEntry {
355          key:       key.to_string(),
356          tok_start: data.value_tokens.len() as u32,
357          tok_count: 0,
358        });
359      } else {
360        let start = data.value_tokens.len() as u32;
361        let mut count = 0u16;
362        for tok_s in val_data.split(',') {
363          if let Some((cc, text)) = parse_tok(tok_s) {
364            data.value_tokens.push(TokData { cc, text });
365            count += 1;
366          }
367        }
368        data.token_lists.push(TokenListEntry {
369          key:       key.to_string(),
370          tok_start: start,
371          tok_count: count,
372        });
373      }
374    },
375    "VD" => data.vecdeques.push(key.to_string()),
376    _ => {},
377  }
378}

380fn parse_meaning(data: &mut DumpData, key: &str, rest: &str) {
381  let parts: Vec<&str> = rest.splitn(2, '\t').collect();
382  if parts.is_empty() {
383    return;
384  }
385  match parts[0] {
386    "N" => {}, // Skip None meanings
387    "E" => {
388      let eparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(4, '\t').collect();
389      if eparts.len() < 4 {
390        return;
391      }
392      let nargs: u8 = eparts[1].parse().unwrap_or(0);
393      let flags_str = eparts[2];
394      let tok_data = eparts[3];
395
396      let mut flags: u8 = 0;
397      if flags_str.contains('L') {
398        flags |= 1;
399      }
400      if flags_str.contains('P') {
401        flags |= 2;
402      }
403
404      let start = data.expand_tokens.len() as u32;
405      let mut count = 0u16;
406      if !tok_data.is_empty() {
407        for tok_s in tok_data.split(',') {
408          if let Some((cc, text)) = parse_tok(tok_s) {
409            data.expand_tokens.push(TokData { cc, text });
410            count += 1;
411          }
412        }
413      }
414      data.expandables.push(ExpandEntry {
415        cs: key.to_string(),
416        nargs,
417        flags,
418        tok_start: start,
419        tok_count: count,
420      });
421    },
422    "T" => {
423      let tok_s = parts.get(1).unwrap_or(&"");
424      if let Some((cc, text)) = parse_tok(tok_s) {
425        data
426          .let_defs
427          .push(TokenEntry { key: key.to_string(), cc, text });
428      }
429    },
430    "PA" | "MPA" => {
431      // Primitive alias: \let <key> = <target>.
432      // Emit as a call to state::let_i at load time; target must already
433      // be defined (guaranteed by the (d.2) early/late ordering).
434      let target = url_decode(parts.get(1).unwrap_or(&""));
435      if target.is_empty() || target == key {
436        return; // self-alias or malformed: skip
437      }
438      data.let_aliases.push(LetAliasEntry {
439        key: key.to_string(),
440        target,
441        is_math: parts[0] == "MPA",
442      });
443    },
444    _ => {},
445  }
446}

448fn parse_tok(s: &str) -> Option<(u8, String)> {
449  let (cc_str, text) = s.split_once(':')?;
450  let cc: u8 = cc_str.parse().ok()?;
451  Some((cc, url_decode(text)))
452}
453
454fn parse_glue_data(key: &str, s: &str) -> Option<GlueEntry> {
455  let mut skip = 0i64;
456  let mut plus = None;
457  let mut pfill = 0u8;
458  let mut minus = None;
459  let mut mfill = 0u8;
460
461  for (i, part) in s.split(',').enumerate() {
462    if i == 0 {
463      skip = part.parse().ok()?;
464    } else if let Some(rest) = part.strip_prefix("pf") {
465      pfill = rest.parse().unwrap_or(0);
466    } else if let Some(rest) = part.strip_prefix('p') {
467      plus = Some(rest.parse().ok()?);
468    } else if let Some(rest) = part.strip_prefix("mf") {
469      mfill = rest.parse().unwrap_or(0);
470    } else if let Some(rest) = part.strip_prefix('m') {
471      minus = Some(rest.parse().ok()?);
472    }
473  }
474
475  Some(GlueEntry {
476    key: key.to_string(),
477    skip,
478    plus,
479    pfill,
480    minus,
481    mfill,
482  })
483}

487/// Generate a Rust source file with typed static data tables from a text dump.
488/// Every entry is compiler-checked. Zero runtime parsing needed.
489pub fn generate_rs(dump_path: &Path, output_path: &Path) -> Result<usize, String> {
490  let content =
491    std::fs::read_to_string(dump_path).map_err(|e| format!("Failed to read dump: {}", e))?;
492
493  let data = parse_dump(&content);
494  let total = data.total_entries();
495
496  let mut out =
497    std::fs::File::create(output_path).map_err(|e| format!("Failed to create output: {}", e))?;
498
499  // Header
500  writeln!(
501    out,
502    "//! Auto-generated kernel dump — static data tables.\n\
503     //! DO NOT EDIT — regenerate with: `cargo run --release --bin latexml_oxide -- --init=latex.ltx`\n\
504     //!\n\
505     //! {total} entries, compiler-checked at build time. Zero runtime parsing.\n\
506     #![allow(unused)]\n\
507     #![allow(clippy::type_complexity)]\n"
508  )
509  .map_err(we)?;
510
511  // Imports
512  writeln!(
513    out,
514    "use std::collections::VecDeque;\n\
515     \n\
516     use latexml_core::common::arena;\n\
517     use latexml_core::common::def_parser::parse_parameters;\n\
518     use latexml_core::common::dimension::Dimension;\n\
519     use latexml_core::common::glue::{{FillCode, Glue}};\n\
520     use latexml_core::common::mudimension::MuDimension;\n\
521     use latexml_core::common::muglue::MuGlue;\n\
522     use latexml_core::common::store::Stored;\n\
523     use latexml_core::definition::expandable::{{Expandable, ExpandableOptions}};\n\
524     use latexml_core::state;\n\
525     use latexml_core::state::Scope;\n\
526     use latexml_core::token::{{Catcode, Token}};\n\
527     use latexml_core::tokens::Tokens;\n"
528  )
529  .map_err(we)?;
530
531  // Local struct for expandable definitions
532  writeln!(
533    out,
534    "struct ExpandDef {{\n\
535     {s}cs: &'static str,\n\
536     {s}nargs: u8,\n\
537     {s}flags: u8, // bit 0 = long, bit 1 = protected\n\
538     {s}tok_start: u32,\n\
539     {s}tok_count: u16,\n\
540     }}\n",
541    s = "  "
542  )
543  .map_err(we)?;
544
545  // Emit static arrays
546  emit_bool_array(&mut out, "BOOL_VALUES", &data.bools)?;
547  emit_int_array(&mut out, "INT_VALUES", &data.ints)?;
548  emit_str_array(&mut out, "STRING_VALUES", &data.strings)?;
549  emit_dim_array(&mut out, "DIMENSION_VALUES", &data.dims)?;
550  emit_glue_array(&mut out, "GLUE_VALUES", &data.glues)?;
551  emit_dim_array(&mut out, "MUDIM_VALUES", &data.mudims)?;
552  emit_glue_array(&mut out, "MUGLUE_VALUES", &data.muglues)?;
553  emit_charcode_array(&mut out, "CHARCODE_VALUES", &data.charcodes)?;
554  emit_catcode_array(&mut out, "CATCODE_VALUE_ENTRIES", &data.catcode_vals)?;
555  emit_token_array(&mut out, "SINGLE_TOKEN_VALUES", &data.single_tokens)?;
556  emit_none_array(&mut out, "NONE_VALUES", &data.nones)?;
557  emit_none_array(&mut out, "VECDEQUE_VALUES", &data.vecdeques)?;
558
559  // Token list values (V/TK) — entries reference VALUE_TOKENS pool
560  emit_token_list_array(&mut out, "TOKEN_LIST_VALUES", &data.token_lists)?;
561  emit_tok_pool(&mut out, "VALUE_TOKENS", &data.value_tokens)?;
562
563  // Code tables
564  emit_catcode_array(&mut out, "CATCODE_TABLE", &data.catcodes)?;
565  emit_charcode_array(&mut out, "LCCODE_TABLE", &data.lccodes)?;
566  emit_charcode_array(&mut out, "UCCODE_TABLE", &data.uccodes)?;
567  emit_charcode_array(&mut out, "SFCODE_TABLE", &data.sfcodes)?;
568  emit_charcode_array(&mut out, "DELCODE_TABLE", &data.delcodes)?;
569  emit_charcode_array(&mut out, "MATHCODE_TABLE", &data.mathcodes)?;
570
571  // Meaning table
572  emit_token_array(&mut out, "LET_DEFINITIONS", &data.let_defs)?;
573  emit_expand_array(&mut out, "EXPANDABLE_DEFS", &data.expandables)?;
574  emit_tok_pool(&mut out, "EXPAND_TOKENS", &data.expand_tokens)?;
575  emit_let_alias_array(&mut out, "LET_ALIAS_DEFS", &data.let_aliases)?;
576
577  // load_definitions function
578  emit_load_fn(&mut out, &data)?;
579
580  Info!(
581    "dump_codegen", "generated",
582    s!("Generated {} entries to {} (static data tables)",
583      total, output_path.display())
584  );
585
586  Ok(total)
587}

591fn emit_bool_array(
592  out: &mut std::fs::File,
593  name: &str,
594  entries: &[BoolEntry],
595) -> Result<(), String> {
596  writeln!(out, "static {name}: &[(&str, bool)] = &[").map_err(we)?;
597  for e in entries {
598    writeln!(out, "  ({}, {}),", rust_escape(&e.key), e.val).map_err(we)?;
599  }
600  writeln!(out, "];\n").map_err(we)
601}

627fn emit_glue_array(
628  out: &mut std::fs::File,
629  name: &str,
630  entries: &[GlueEntry],
631) -> Result<(), String> {
632  // (key, skip, plus, pfill, minus, mfill)
633  writeln!(
634    out,
635    "static {name}: &[(&str, i64, Option<i64>, u8, Option<i64>, u8)] = &["
636  )
637  .map_err(we)?;
638  for e in entries {
639    let plus_s = match e.plus {
640      Some(v) => format!("Some({v})"),
641      None => "None".to_string(),
642    };
643    let minus_s = match e.minus {
644      Some(v) => format!("Some({v})"),
645      None => "None".to_string(),
646    };
647    writeln!(
648      out,
649      "  ({}, {}, {}, {}, {}, {}),",
650      rust_escape(&e.key),
651      e.skip,
652      plus_s,
653      e.pfill,
654      minus_s,
655      e.mfill
656    )
657    .map_err(we)?;
658  }
659  writeln!(out, "];\n").map_err(we)
660}

784/// Emit the load_definitions() function that iterates over all static arrays.
785fn emit_load_fn(out: &mut std::fs::File, data: &DumpData) -> Result<(), String> {
786  let s = "  ";
787  writeln!(
788    out,
789    "/// Load the precompiled kernel definitions into the global state.\n\
790     /// Perl equivalent: LoadFormat → LoadPool(format_dump)\n\
791     pub fn load_definitions() -> latexml_core::common::error::Result<()> {{"
792  )
793  .map_err(we)?;
794
795  // Bool values
796  if !data.bools.is_empty() {
797    writeln!(
798      out,
799      "{s}for &(key, val) in BOOL_VALUES {{\n\
800       {s}{s}state::assign_value(key, Stored::Bool(val), Some(Scope::Global));\n\
801       {s}}}"
802    )
803    .map_err(we)?;
804  }
805
806  // Int values
807  if !data.ints.is_empty() {
808    writeln!(
809      out,
810      "{s}for &(key, val) in INT_VALUES {{\n\
811       {s}{s}state::assign_value(key, Stored::Int(val), Some(Scope::Global));\n\
812       {s}}}"
813    )
814    .map_err(we)?;
815  }
816
817  // String values
818  if !data.strings.is_empty() {
819    writeln!(
820      out,
821      "{s}for &(key, val) in STRING_VALUES {{\n\
822       {s}{s}state::assign_value(key, Stored::from(val), Some(Scope::Global));\n\
823       {s}}}"
824    )
825    .map_err(we)?;
826  }
827
828  // Dimension values
829  if !data.dims.is_empty() {
830    writeln!(
831      out,
832      "{s}for &(key, val) in DIMENSION_VALUES {{\n\
833       {s}{s}state::assign_value(key, Stored::Dimension(Dimension(val)), Some(Scope::Global));\n\
834       {s}}}"
835    )
836    .map_err(we)?;
837  }
838
839  // Glue values
840  if !data.glues.is_empty() {
841    writeln!(
842      out,
843      "{s}for &(key, skip, plus, pfill, minus, mfill) in GLUE_VALUES {{\n\
844       {s}{s}state::assign_value(key, Stored::Glue(Glue {{\n\
845       {s}{s}{s}skip, plus, pfill: FillCode::new(pfill as usize),\n\
846       {s}{s}{s}minus, mfill: FillCode::new(mfill as usize),\n\
847       {s}{s}}}), Some(Scope::Global));\n\
848       {s}}}"
849    )
850    .map_err(we)?;
851  }
852
853  // MuDimension values
854  if !data.mudims.is_empty() {
855    writeln!(
856      out,
857      "{s}for &(key, val) in MUDIM_VALUES {{\n\
858       {s}{s}state::assign_value(key, Stored::MuDimension(MuDimension(val)), Some(Scope::Global));\n\
859       {s}}}"
860    )
861    .map_err(we)?;
862  }
863
864  // MuGlue values
865  if !data.muglues.is_empty() {
866    writeln!(
867      out,
868      "{s}for &(key, skip, plus, pfill, minus, mfill) in MUGLUE_VALUES {{\n\
869       {s}{s}state::assign_value(key, Stored::MuGlue(MuGlue {{\n\
870       {s}{s}{s}skip, plus, pfill: FillCode::new(pfill as usize),\n\
871       {s}{s}{s}minus, mfill: FillCode::new(mfill as usize),\n\
872       {s}{s}}}), Some(Scope::Global));\n\
873       {s}}}"
874    )
875    .map_err(we)?;
876  }
877
878  // Charcode values
879  if !data.charcodes.is_empty() {
880    writeln!(
881      out,
882      "{s}for &(key, val) in CHARCODE_VALUES {{\n\
883       {s}{s}state::assign_value(key, Stored::Charcode(val), Some(Scope::Global));\n\
884       {s}}}"
885    )
886    .map_err(we)?;
887  }
888
889  // Catcode values (in value table, not catcode table)
890  if !data.catcode_vals.is_empty() {
891    writeln!(
892      out,
893      "{s}for &(key, val) in CATCODE_VALUE_ENTRIES {{\n\
894       {s}{s}state::assign_value(key, Stored::Catcode(Catcode::from(val)), Some(Scope::Global));\n\
895       {s}}}"
896    )
897    .map_err(we)?;
898  }
899
900  // Single token values
901  if !data.single_tokens.is_empty() {
902    writeln!(
903      out,
904      "{s}for &(key, cc, text) in SINGLE_TOKEN_VALUES {{\n\
905       {s}{s}let tok = Token {{ text: arena::pin(text), code: Catcode::from(cc) }};\n\
906       {s}{s}state::assign_value(key, Stored::Token(tok), Some(Scope::Global));\n\
907       {s}}}"
908    )
909    .map_err(we)?;
910  }
911
912  // Token list values
913  if !data.token_lists.is_empty() {
914    writeln!(
915      out,
916      "{s}for &(key, tok_start, tok_count) in TOKEN_LIST_VALUES {{\n\
917       {s}{s}let toks: Vec<Token> = VALUE_TOKENS[tok_start as usize..][..tok_count as usize]\n\
918       {s}{s}{s}.iter()\n\
919       {s}{s}{s}.map(|&(cc, text)| Token {{ text: arena::pin(text), code: Catcode::from(cc) }})\n\
920       {s}{s}{s}.collect();\n\
921       {s}{s}state::assign_value(key, Stored::Tokens(Tokens::from(toks)), Some(Scope::Global));\n\
922       {s}}}"
923    )
924    .map_err(we)?;
925  }
926
927  // None values
928  if !data.nones.is_empty() {
929    writeln!(
930      out,
931      "{s}for &key in NONE_VALUES {{\n\
932       {s}{s}state::assign_value(key, Stored::None, Some(Scope::Global));\n\
933       {s}}}"
934    )
935    .map_err(we)?;
936  }
937
938  // VecDeque values
939  if !data.vecdeques.is_empty() {
940    writeln!(
941      out,
942      "{s}for &key in VECDEQUE_VALUES {{\n\
943       {s}{s}state::assign_value(key, Stored::VecDequeStored(VecDeque::new()), Some(Scope::Global));\n\
944       {s}}}"
945    )
946    .map_err(we)?;
947  }
948
949  // Catcode table
950  if !data.catcodes.is_empty() {
951    writeln!(
952      out,
953      "{s}for &(key, val) in CATCODE_TABLE {{\n\
954       {s}{s}if let Some(ch) = key.chars().next() {{\n\
955       {s}{s}{s}state::assign_catcode(ch, Catcode::from(val), Some(Scope::Global));\n\
956       {s}{s}}}\n\
957       {s}}}"
958    )
959    .map_err(we)?;
960  }
961
962  // LC/UC/SF/DC/MC code tables
963  for (array_name, fn_name) in [
964    ("LCCODE_TABLE", "assign_lccode"),
965    ("UCCODE_TABLE", "assign_uccode"),
966    ("SFCODE_TABLE", "assign_sfcode"),
967    ("DELCODE_TABLE", "assign_delcode"),
968    ("MATHCODE_TABLE", "assign_mathcode"),
969  ] {
970    let entries_exist = match array_name {
971      "LCCODE_TABLE" => !data.lccodes.is_empty(),
972      "UCCODE_TABLE" => !data.uccodes.is_empty(),
973      "SFCODE_TABLE" => !data.sfcodes.is_empty(),
974      "DELCODE_TABLE" => !data.delcodes.is_empty(),
975      "MATHCODE_TABLE" => !data.mathcodes.is_empty(),
976      _ => false,
977    };
978    if entries_exist {
979      writeln!(
980        out,
981        "{s}for &(key, val) in {array_name} {{\n\
982         {s}{s}if let Some(ch) = key.chars().next() {{\n\
983         {s}{s}{s}state::{fn_name}(ch, val, Some(Scope::Global));\n\
984         {s}{s}}}\n\
985         {s}}}"
986      )
987      .map_err(we)?;
988    }
989  }
990
991  // Let definitions (M/T)
992  if !data.let_defs.is_empty() {
993    writeln!(
994      out,
995      "{s}for &(cs_name, cc, text) in LET_DEFINITIONS {{\n\
996       {s}{s}let cs = Token {{ text: arena::pin(cs_name), code: Catcode::CS }};\n\
997       {s}{s}if !state::has_meaning(&cs) {{\n\
998       {s}{s}{s}let tok = Token {{ text: arena::pin(text), code: Catcode::from(cc) }};\n\
999       {s}{s}{s}state::assign_meaning(&cs, tok, Some(Scope::Global));\n\
1000       {s}{s}}}\n\
1001       {s}}}"
1002    )
1003    .map_err(we)?;
1004  }
1005
1006  // Expandable definitions (M/E) — the bulk (77% of entries)
1007  if !data.expandables.is_empty() {
1008    writeln!(
1009      out,
1010      "{s}for def in EXPANDABLE_DEFS {{\n\
1011       {s}{s}let cs = Token {{ text: arena::pin(def.cs), code: Catcode::CS }};\n\
1012       {s}{s}if state::has_meaning(&cs) {{ continue; }}\n\
1013       {s}{s}let toks: Vec<Token> = EXPAND_TOKENS[def.tok_start as usize..][..def.tok_count as usize]\n\
1014       {s}{s}{s}.iter()\n\
1015       {s}{s}{s}.map(|&(cc, text)| Token {{ text: arena::pin(text), code: Catcode::from(cc) }})\n\
1016       {s}{s}{s}.collect();\n\
1017       {s}{s}let params = if def.nargs > 0 {{\n\
1018       {s}{s}{s}let proto = \"{{}}\".repeat(def.nargs as usize);\n\
1019       {s}{s}{s}// init_flag=true: engine is up when this table is consumed\n\
1020       {s}{s}{s}// at runtime, so Parameter::init() can resolve readers\n\
1021       {s}{s}{s}// via PARAMETER_TYPES; without init the Plain reader\n\
1022       {s}{s}{s}// falls back to mock_reader and invocation fails.\n\
1023       {s}{s}{s}parse_parameters(&proto, &cs, true).ok().flatten()\n\
1024       {s}{s}}} else {{ None }};\n\
1025       {s}{s}let opts = Some(ExpandableOptions {{\n\
1026       {s}{s}{s}long: def.flags & 1 != 0,\n\
1027       {s}{s}{s}protected: def.flags & 2 != 0,\n\
1028       {s}{s}{s}nopack_parameters: true,\n\
1029       {s}{s}{s}..ExpandableOptions::default()\n\
1030       {s}{s}}});\n\
1031       {s}{s}if let Ok(exp) = Expandable::new(cs, params, Some(Tokens::from(toks).into()), opts) {{\n\
1032       {s}{s}{s}state::install_definition(exp, Some(Scope::Global));\n\
1033       {s}{s}}}\n\
1034       {s}}}"
1035    )
1036    .map_err(we)?;
1037  }
1038
1039  // Primitive-alias entries (M/PA, M/MPA). The write pass ordered these
1040  // so that `target` is either a bootstrap primitive (always present) or
1041  // an entry defined earlier in this pass. Skip self-aliases and any
1042  // alias whose key already has a meaning (add-only parity).
1043  if !data.let_aliases.is_empty() {
1044    writeln!(
1045      out,
1046      "{s}for &(key_s, target_s) in LET_ALIAS_DEFS {{\n\
1047       {s}{s}if key_s == target_s {{ continue; }}\n\
1048       {s}{s}let key_tok = Token {{ text: arena::pin(key_s), code: Catcode::CS }};\n\
1049       {s}{s}if state::has_meaning(&key_tok) {{ continue; }}\n\
1050       {s}{s}let target_tok = Token {{ text: arena::pin(target_s), code: Catcode::CS }};\n\
1051       {s}{s}if !state::has_meaning(&target_tok) {{ continue; }}\n\
1052       {s}{s}state::let_i(&key_tok, &target_tok, Some(Scope::Global));\n\
1053       {s}}}"
1054    )
1055    .map_err(we)?;
1056  }
1057
1058  writeln!(out, "\n{s}Ok(())\n}}").map_err(we)?;
1059
1060  Ok(())
1061}
