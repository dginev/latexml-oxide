121  /// engages. Defaults to [`CYCLE_GUARD_ACTIVATE`] (20M). Graphics packages
122  /// whose healthy expansion legitimately runs to 100M+ tokens (pgf/tikz/xy)
123  /// raise it via [`raise_cycle_guard_activate`] so their streams stay out of
124  /// the per-token fingerprint regime; the 400M `token_limit` remains the hard

126  /// set explicitly in the constructor and the per-conversion reset.

128  /// Windowed cycle detector over the expansion (read-token) stream — catches
129  /// small-period infinite expansion loops (`\def\x{a\x}` etc.) far earlier
130  /// and more cheaply than `token_limit`/`pushback_limit`. Gated on a high
131  /// `progress` so normal documents never touch it. See [`crate::cycle_guard`].

133  /// Reading-context serial, mixed into every cycle-guard fingerprint so that
134  /// windows never match ACROSS `reading_from_mouth` contexts: a cycle is only
135  /// a cycle within one expansion context. Each `reading_from_mouth` entry
136  /// allocates a fresh serial (from `ctx_next`) and restores the outer one on
137  /// exit (`ctx_stack`), so (a) consecutive IDENTICAL short expansions — the
138  /// math0402448 xymatrix per-cell `get_xmarg_id` stream — get distinct
139  /// serials and can never concatenate into a pseudo-periodic window (the
140  /// false positive an earlier blanket `reset()` suppressed), while (b) an
141  /// OUTER loop's tokens keep their serial across inner expansions, so a
142  /// runaway whose body calls `do_expand` each iteration (~164 call sites)
143  /// remains detectable — the blind spot the blanket reset had (PR #249
144  /// review P2-7).
145  pub ctx_serial:           u64,
146  ctx_next:                 u64,
147  ctx_stack:                Vec<u64>,

267/// Get the current location of input getting read
268pub fn get_locator() -> Locator {
269  let gullet = gullet!();
270  let mut runtime_opt = gullet.runtime.as_ref();
271  let mut mouthstack_iter = gullet.mouthstack.iter();
272  while runtime_opt.is_some() && runtime_opt.as_ref().unwrap().mouth.get_source().is_empty() {
273    runtime_opt = mouthstack_iter.next();
274  }
275  // The free fn stays `-> Locator` ("where the parser is now" — always a real
276  // position during digestion; the workhorse for errors + box creation). A
277  // Mouth's `get_locator` is `Option` (per the `Object` trait) but is always
278  // `Some`, so unwrap to the default only in the no-mouth backup.
279  if let Some(runtime) = runtime_opt {
280    // First exit condition: we found a mouth with a source, and asked it for a locator
281    runtime.mouth.get_locator().unwrap_or_default()
282  } else if let Some(runtime) = gullet.mouthstack.front() {
283    // Backup strategy: return the first locator in the mouthstack:
284    runtime.mouth.get_locator().unwrap_or_default()
285  } else {
286    // Final backup -- the default locator
287    Locator::default()
288  }
289}

292/// mouthstack walk, but reads the mouth's `get_locator_from_start` (`from` = the
293/// last token's captured start) instead of the heuristic `from`. Used for the
294/// construct-START snapshot at constructor digest under `--source-map`.
295pub fn get_locator_from_start() -> Locator {
296  let gullet = gullet!();
297  let mut runtime_opt = gullet.runtime.as_ref();
298  let mut mouthstack_iter = gullet.mouthstack.iter();
299  while runtime_opt.is_some() && runtime_opt.as_ref().unwrap().mouth.get_source().is_empty() {

301  }
302  if let Some(runtime) = runtime_opt {

304  } else if let Some(runtime) = gullet.mouthstack.front() {
305    runtime.mouth.get_locator_from_start()
306  } else {
307    Locator::default()
308  }
309}

332/// Variant of `unread`, but drains the contents of `tokens` without taking ownership.
333pub fn unread_mut(tokens: &mut Tokens) {
334  if let Some(ref mut runtime) = gullet_mut!().runtime {
335    // Iterate in reverse and push to the stack top — the first element
336    // of `tokens` ends up on top (= next to read). Same semantics as
337    // the old VecDeque push_front pattern.
338    for token in tokens.unlist_mut().drain(..).rev() {
339      runtime.pushback.push(token);
340    }

343/// Unreads a single `Token` to the start of the token stream.
344/// Perl: unread() always adjusts $ALIGN_STATE when unreading { or } tokens.
345pub fn unread_one(token: Token) {
346  match token.get_catcode() {
347    Catcode::BEGIN => decrement_align_group_count(), // Retract scanned brace
348    Catcode::END => increment_align_group_count(),
349    _ => {},
350  }
351  if let Some(ref mut runtime) = gullet_mut!().runtime {
352    runtime.pushback.push(token);
353  };
354}
355/// Unreads a `Vec<Token>` to the start of the token stream
356/// Perl: also adjusts ALIGN_STATE by retracting scanned braces (Gullet.pm lines 343-358)
357pub fn unread_vec(tokens: Vec<Token>) {
358  let mut level: i64 = 0;
359  if let Some(ref mut runtime) = gullet_mut!().runtime {
360    // Reserve once, push each token in reverse-iteration order so the
361    // first element of `tokens` ends up at the stack top. Same
362    // semantics as the old VecDeque push_front loop, but without
363    // per-element head-pointer arithmetic.
364    runtime.pushback.reserve(tokens.len());
365    for token in tokens.into_iter().rev() {
366      match token.get_catcode() {
367        Catcode::BEGIN => level -= 1, // Retract scanned braces
368        Catcode::END => level += 1,
369        _ => {},
370      }
371      runtime.pushback.push(token);
372    }
373  }
374  if level != 0 {
375    set_align_group_count(align_group_count() + level as i32);
376  }
377}

398pub fn close_mouth(forced: bool) -> Result<()> {
399  let mut shift_from_mouthstack = false;
400  let mut error_has_more_input = false;
401  if let Some(ref mut runtime) = runtime!()
402    && !forced
403    && (!runtime.pushback.is_empty() || runtime.mouth.has_more_input())
404  {
405    error_has_more_input = true
406  }
407  if error_has_more_input {
408    let next = match read_token()? {
409      Some(t) => t.stringify(),
410      None => String::from("Empty"),
411    };
412    let message = s!("Closing mouth with input remaining '{}'", next);
413    Error!("unexpected", next, message);
414  }
415  let mut gullet = gullet_mut!();
416  if let Some(ref mut runtime) = gullet.runtime {
417    runtime.mouth.finish();
418    shift_from_mouthstack = true;
419  }
420  if shift_from_mouthstack {
421    gullet.runtime = gullet.mouthstack.pop_front();
422  }
423  Ok(())
424}
425/// This flushes a mouth so that it will be automatically closed, next time it's read
426/// Corresponds to TeX's \endinput
427pub fn flush_mouth() {
428  if let Some(ref mut runtime) = runtime!() {
429    // Collect remaining mouth tokens in mouth order (t1, t2, t3, …),
430    // then splice them into the stack's BOTTOM in reverse order so
431    // that after the stack's existing top is popped, the mouth tokens
432    // come out in the original mouth order (t1 first, then t2, …).
433    let mut trailer: Vec<Token> = Vec::new();
434    while !runtime.mouth.is_eol() {
435      if let Some(token) = runtime.mouth.read_token() {
436        trailer.push(token);
437      }
438    }
439    if !trailer.is_empty() {
440      trailer.reverse();
441      runtime.pushback.splice(0..0, trailer);
442    }
443    // Stop reading (clear buffers, close file) but do NOT restore catcodes.
444    // Catcodes are restored by close_mouth → finish() when the mouth is
445    // properly popped from the stack.
446    runtime.mouth.stop_reading();
447  }
448}

497// internal low-level reader that extracts a token from a mouth,
498// but always keeps comment tokens pending.
499fn read_internal_token() -> Option<Token> {
500  let mut next_token = None;
501  let Gullet {
502    ref mut runtime,
503    ref mut pending_comments,
504    ..
505  } = *gullet_mut!();
506  // Defensive: gullet runtime can be None during early shutdown /
507  // recovery from a fatal error. Treat as "no more tokens" instead
508  // of panicking. Driver: 2404.06289 (natbib \NAT@@wrout cascade
509  // landed here after the conversion was already in error-recovery
510  // mode).
511  let rt = runtime.as_mut()?;
512  let pushback = &mut rt.pushback;
513  // Check in pushback first....
514  while let Some(pushback_token) = pushback.pop() {
515    match pushback_token.get_catcode() {
516      Catcode::COMMENT => pending_comments.push_back(pushback_token),
517      Catcode::MARKER => handle_marker(pushback_token),
518      _ => {
519        next_token = Some(pushback_token);
520        break;
521      },
522    };
523  }
524  // Not in pushback, read from the current Mouth
525  if next_token.is_none() {
526    while let Some(token) = runtime.as_mut().unwrap().mouth.read_token() {
527      match token.get_catcode() {
528        Catcode::COMMENT => pending_comments.push_back(token),
529        Catcode::MARKER => handle_marker(token),
530        _ => {
531          next_token = Some(token);
532          break;
533        },
534      };
535    }
536  }
537  next_token
538}

540/// Per-token-read resource checkpoint — runtime probe, progress/token-limit
541/// accounting, pushback-limit probe, and cycle-guard activation — in ONE
542/// mutable borrow. Returns `Ok(None)` when the gullet has no runtime (caller
543/// returns end-of-input), else `Ok(Some(cycle_guard_active))` for the caller to
544/// pass to [`cycle_guard_checkpoint`] (single-borrow fast path, PR #249 P2-9).
545///
546/// Shared by all FOUR reader loops (`read_token`, `read_x_token`,
547/// `read_balanced`, `read_next_conditional`) — siblings over
548/// `read_internal_token`, NOT a delegation chain, so each must run its own or
549/// full-expansion paths (`\edef`, csname, conditional skipping) bypass every
550/// gullet guard and a runaway grinds to the watchdog (gap found on math0402448).
551#[inline]
552fn read_resource_checkpoint() -> Result<Option<bool>> {
553  let mut g = gullet_mut!();
554  if g.runtime.is_none() {
555    return Ok(None);
556  }
557  // Progress counts UNCONDITIONALLY: the cycle guard's activation gate
558  // (`progress > CYCLE_GUARD_ACTIVATE`) feeds off it, so nesting the
559  // increment inside the token-limit branch made `LATEXML_TOKEN_LIMIT=0`
560  // (and `set_token_limit(None)` during format init) silently disable the
561  // cycle guard as well — exactly when an operator disables the limit to
562  // let a big document through is when the loop guard matters most.
563  // (PR #249 review P2-5.) Only the limit COMPARISON stays conditional.
564  g.progress += 1;
565  if let Some(limit) = g.token_limit
566    && g.progress > limit
567  {
568    let msg = s!("Token limit of {} exceeded, infinite loop?", limit);
569    drop(g);
570    // The cycle guard already dumps its window when it trips, but a runaway
571    // that reaches THIS limit is by definition one the cycle guard did NOT
572    // recognise — an aperiodic grind (a counter that keeps advancing, a list
573    // that keeps growing). Those are exactly the ones with no other clue in
574    // the log, so dump the same recent-token ring here. Without it a
575    // TokenLimit fatal says only "infinite loop?" and every investigation
576    // starts from zero (witness 2606.21610, a 42 s silent grind).
577    if *DEBUG_FATAL {
578      eprintln!("[debug-fatal] token limit tripped: {msg}");
579      DEBUG_RECENT_TOKENS.with(|ring| {
580        let ring = ring.borrow();
581        let recent: Vec<&str> = ring.iter().map(String::as_str).collect();
582        eprintln!(
583          "[debug-fatal] last {} read tokens: {}",
584          ring.len(),
585          recent.join(" ")
586        );
587      });
588    }
589    Fatal!(Timeout, TokenLimit, msg);
590  }
591  if let Some(limit) = g.pushback_limit {
592    let pb_len = g.runtime.as_ref().map(|r| r.pushback.len()).unwrap_or(0);
593    if pb_len > limit {
594      // Diagnostic: the looping token window is right here in the
595      // pushback — dump its head so the cycle is identifiable from logs.
596      if *DEBUG_FATAL && let Some(rt) = g.runtime.as_ref() {
597        let head: Vec<String> = rt
598          .pushback
599          .iter()
600          .take(48)
601          .map(|t| format!("{t:?}"))
602          .collect();
603        eprintln!("[debug-fatal] pushback head: {}", head.join(" "));
604      }
605      let msg = s!("Pushback limit of {} exceeded, infinite loop?", limit);
606      drop(g);
607      Fatal!(Timeout, PushbackLimit, msg);
608    }
609  }
610  Ok(Some(g.progress > g.cycle_guard_activate))
611}

613/// Windowed cycle-guard checkpoint over the token-read stream. Only engaged
614/// once `progress` is already pathologically high — the caller passes the
615/// activation state computed by [`read_resource_checkpoint`] in the SAME
616/// borrow, so below the gate this costs one branch and zero borrows (and no
617/// fingerprint hash). A small-period infinite expansion loop then trips a
618/// clean Fatal in O(window) extra tokens instead of grinding to the token
619/// limit (and the gigabytes of RSS that implies). Shared by all four reader
620/// loops (see [`read_resource_checkpoint`] on why each loop needs its own
621/// checkpoints — they do NOT delegate to one another).
622#[inline]
623fn cycle_guard_checkpoint(active: bool, nextt: &Token) -> Result<()> {
624  // Record BEFORE the activation gate: the ring is also what the token-limit
625  // fatal dumps, and that fatal fires precisely for runaways the cycle guard
626  // never recognised — which includes ones that trip a lowered
627  // `LATEXML_TOKEN_LIMIT` before the guard ever activates. Gating the ring on
628  // `active` left it empty in exactly that case. Debug-only, so free in
629  // production.
630  if *DEBUG_FATAL {
631    DEBUG_RECENT_TOKENS.with(|ring| {
632      let mut ring = ring.borrow_mut();
633      if ring.len() >= 512 {
634        ring.pop_front();
635      }
636      ring.push_back(format!("{nextt:?}"));
637    });
638  }
639  if !active {
640    return Ok(());
641  }
642  let mut g = gullet_mut!();
643  {
644    // Mix the reading-context serial into the fingerprint (see the
645    // `Gullet::ctx_serial` field doc): tokens read in different
646    // `reading_from_mouth` contexts can then never form a matching window,
647    // scoping cycle detection to ONE expansion context without destroying
648    // the outer context's history. The multiplier spreads the serial across
649    // the hash bits (splitmix64's odd constant).
650    let fp = nextt.cycle_fingerprint() ^ g.ctx_serial.wrapping_mul(0x9E37_79B9_7F4A_7C15);
651    if let Some(period) = g.cycle_guard.push(fp) {
652      drop(g);
653      let msg = s!(
654        "Infinite expansion loop: a window of {} token(s) repeated {}+ times",
655        period,
656        crate::cycle_guard::REPEAT
657      );
658      if *DEBUG_FATAL {
659        eprintln!("[debug-fatal] gullet cycle guard tripping on token {nextt:?}: {msg}");
660        DEBUG_RECENT_TOKENS.with(|ring| {
661          let ring = ring.borrow();
662          let recent: Vec<&str> = ring.iter().map(String::as_str).collect();
663          eprintln!("[debug-fatal] last 512 read tokens: {}", recent.join(" "));
664        });
665      }
666      Fatal!(Timeout, Recursion, msg);
667    }
668  }
669  Ok(())
670}

672/// Read a token that the calling macro/primitive REQUIRES, holding the
673/// "argument expected but input ended" diagnostic in one place.
674///
675/// `read_token()` returning `None` (input exhausted) is a normal, expected
676/// control-flow signal in most contexts (end of file/group/optional-arg scan) —
677/// so the primitive deliberately keeps the `Option`. But for a caller that
678/// genuinely requires a token here, `None` is TeX's *"File ended while scanning
679/// use of \cs"* error state (real `pdftex` raises an `! Emergency stop`). This
680/// helper emits that parity `Error!` once, centrally, instead of every call site
681/// `.unwrap()`-panicking on the `None`. It STILL returns the `Option` (it does
682/// NOT fabricate a token), so the type system keeps each caller honest about how
683/// it degrades — close its group, substitute a default, etc.
684pub fn read_token_required(what: &str) -> Result<Option<Token>> {
685  let tok = read_token()?;
686  if tok.is_none() {
687    Error!(
688      "expected",
689      what,
690      format!("input ended while scanning use of {what}")
691    );
692  }
693  Ok(tok)
694}

696pub fn read_token() -> Result<Option<Token>> {
697  let mut next_token: Option<Token>;
698  loop {
699    // Single combined borrow: runtime probe + limits + activation state.
700    let guard_active = match read_resource_checkpoint()? {
701      None => return Ok(None),
702      Some(active) => active,
703    };
704    // internal low-level reader that extracts a token from a mouth,
705    // but always keeps comment tokens pending.
706    next_token = read_internal_token();
707    // ProgressStep() if ($$self{progress}++ % $TOKEN_PROGRESS_QUANTUM) == 0;
708
709    // Strict-Perl translation of Gullet.pm `readToken`:
710    //   alignment column-end check → \dont_expand check → break
711    //   ALIGN_STATE tracking happens AFTER the loop, on the FINAL
712    //   token to be returned (Perl L320-324).
713    if let Some(ref nextt) = next_token {
714      // Uniform placement across all reader loops: fingerprint the token AS
715      // READ (before alignment/\dont_expand special-casing), so the cycle
716      // guard sees the same stream regardless of which loop reads it.
717      cycle_guard_checkpoint(guard_active, nextt)?;
718      if (align_group_count() == 0)
719        && has_reading_alignment()
720        && let Some((atoken, atype, ahidden)) = is_column_end(nextt)
721      {
722        let reading_alignment = get_reading_alignment().unwrap();
723        if let DigestedData::Alignment(data) = reading_alignment.data() {
724          handle_template(data.borrow_mut(), atoken, atype, ahidden)?;
725        } else {
726          return Err("reading_alignment should always contain DigestedData::Alignment".into());
727        }
728        continue; // Perl: handleTemplate then continue while(1) loop
729      }
730      if nextt.code == Catcode::CS && nextt.text == pin!("\\dont_expand") {
731        // `\noexpand <tok>`: collapse to the per-token `\special_relax` family,
732        // encoding <tok>'s identity in the name (faithful to TeX's
733        // `no_expand_flag`, which keeps `cur_cs`). End-of-input ⇒ bare
734        // `\special_relax` (nothing to shadow).
735        next_token = Some(match read_token()? {
736          Some(tok) => crate::token::noexpand_family(&tok),
737          None => T_CS!("\\special_relax"),
738        });
739      }
740      break;
741    } else {
742      break;
743    }
744  }
745  // Perl Gullet.pm L320-324: ALIGN_STATE tracking happens AFTER the loop,
746  // applied only to the FINAL returned token. Previously this was inside
747  // the loop BEFORE the alignment check, which prevented column-end
748  // template handling on `{` tokens (count became 1 before the check).
749  if let Some(ref nextt) = next_token {
750    match nextt.get_catcode() {
751      Catcode::BEGIN => increment_align_group_count(),
752      Catcode::END => decrement_align_group_count(),
753      _ => {},
754    }
755  }
756  Ok(next_token)
757}

790// Cluster F: bound gullet expansion-recursion depth (= `read_x_token`
791// re-entrancy) so a runaway — an xint number-arg chain, a self-referential
792// `\csname`/`\number`/`\romannumeral` — raises a fast `Fatal:Timeout:Recursion`
793// rather than grinding to the watchdog / RSS fuse. Legit docs nest ≲20; the cap
794// is 12_000. Env override `LATEXML_EXPAND_DEPTH_LIMIT` (0 disables).
795#[thread_local]
796static EXPAND_DEPTH: Cell<usize> = Cell::new(0);
797#[thread_local]
798static EXPAND_DEPTH_LIMIT: Cell<usize> = Cell::new(12_000);
799
800/// Reset the counter + re-read the env limit each conversion (the thread-local
801/// engine is reused; a caught unwind could otherwise leave the counter high).
802fn reset_expand_depth() {
803  EXPAND_DEPTH.set(0);
804  EXPAND_DEPTH_LIMIT.set(
805    std::env::var("LATEXML_EXPAND_DEPTH_LIMIT")
806      .ok()
807      .and_then(|v| v.trim().parse().ok())
808      .unwrap_or(12_000),
809  );
810}

812/// RAII depth counter for `read_x_token`: `enter` increments (Fatals past the
813/// limit), drop decrements — so every return path stays balanced.
814struct ExpandDepthGuard;
815impl ExpandDepthGuard {
816  #[inline]
817  fn enter() -> Result<ExpandDepthGuard> {
818    let d = EXPAND_DEPTH.get() + 1;
819    EXPAND_DEPTH.set(d);
820    let limit = EXPAND_DEPTH_LIMIT.get();
821    if limit != 0 && d > limit {
822      EXPAND_DEPTH.set(d - 1); // Drop won't run — decrement here.
823      Fatal!(
824        Timeout,
825        Recursion,
826        format!("Excessive expansion recursion (depth {d} > {limit}); infinite macro loop?")
827      );
828    }
829    Ok(ExpandDepthGuard)
830  }
831}
832impl Drop for ExpandDepthGuard {
833  #[inline]
834  fn drop(&mut self) { EXPAND_DEPTH.set(EXPAND_DEPTH.get().saturating_sub(1)); }
835}

837/// Read the next non-expandable token, expanding until one appears. Hot path —
838/// `read_token` is folded in. `toplevel` (default true): on mouth exhaustion,
839/// step to the containing mouth. `fully_expand` (default = toplevel): expand
840/// even protected defns ("for execution"). Unlike `read_balanced`, does NOT
841/// defer `\the` & friends; `\noexpand`'d tokens act like `\relax`. For `\if`/
842/// `\ifx` arguments pass `for_conditional=true` (handles `\noexpand` and CS
843/// `\let` to tokens specially).
844pub fn read_x_token(
845  toplevel_opt: Option<bool>,
846  for_conditional: bool,
847  fully_expand_opt: Option<bool>,
848) -> Result<Option<Token>> {
849  // toplevel should be true by default
850  let toplevel = toplevel_opt.unwrap_or(true);
851  let fully_expand = fully_expand_opt.unwrap_or(toplevel);
852  let _depth_guard = ExpandDepthGuard::enter()?; // Cluster F expansion-depth cap
853  loop {
854    // Resource + cycle checkpoints: this loop reads via the low-level
855    // `read_internal_token` (NOT `read_token`), so it must run the same
856    // guards itself — otherwise every full-expansion read path (`\edef`,
857    // `read_balanced`, csname construction) bypasses the token/pushback
858    // limits and the expansion cycle guard entirely, and a `\def\x{a\x}`
859    // runaway grinds to the multi-GB watchdog instead of a clean Fatal.
860    // (One combined borrow; the returned activation state feeds the cycle
861    // checkpoint below so the fast path pays a single RefCell borrow.)
862    let guard_active = read_resource_checkpoint()?.unwrap_or(false);
863    // internal low-level reader that extracts a token from a mouth,
864    // but always keeps comment tokens pending.
865    let next_token = read_internal_token();
866    //ProgressStep() if ($$self{progress}++ % $TOKEN_PROGRESS_QUANTUM) == 0;
867    if next_token.is_none() {
868      {
869        let gullet = gullet!();
870        let current_is_autoclose = gullet
871          .runtime
872          .as_ref()
873          .map(|r| r.autoclose)
874          .unwrap_or(false);
875        // Drain a *transparent autoclose injection* (\scantokens, raw_tex) and
876        // resume the enclosing mouth even for a BOUNDED reader (toplevel==false,
877        // e.g. the InputDefinitions file loop) — these are part of the current
878        // logical stream. Faithful to tex.web `get_next` §362-365 (exhausting any
879        // input level resumes the enclosing one; \scantokens is a pseudo-file
880        // level). DIVERGES from Perl `Gullet.pm` readXToken, which gates on
881        // `autoclose = toplevel` and so returns at the first exhausted mouth —
882        // truncating a `.sty` whenever `\scantokens` runs mid-load (witness
883        // 1906.03240: real babel.sty `\selectlanguage` dropped every later def →
884        // undefined-CS cascade; Perl's hand-written babel.ltxml dodges it).
885        // A non-autoclose boundary is left to its owner (return None).
886        if !current_is_autoclose || gullet.mouthstack.is_empty() {
887          return Ok(None);
888        }
889      }
890      close_mouth(false)?; // Drain the autoclose injection; resume the parent.
891      continue;
892    }
893    // we got a token
894    let token = next_token.unwrap();
895    cycle_guard_checkpoint(guard_active, &token)?;
896    if token.get_catcode() == Catcode::CS && token.text == pin!("\\dont_expand") {
897      let unexpanded = match read_token()? {
898        Some(t) => t,
899        None => return Ok(Some(T_CS!("\\special_relax"))), // \dont_expand at end-of-input
900      };
901      if for_conditional && unexpanded.code == Catcode::ACTIVE {
902        return Ok(Some(unexpanded));
903      } else {
904        // `\noexpand <tok>`: per-token `\special_relax` family encoding <tok>'s
905        // identity in the name — faithful to TeX's `no_expand_flag`, which keeps
906        // `cur_cs` while giving relax meaning for this one access. (Perl
907        // readXToken returns a bare `\special_relax`, dropping the identity;
908        // recovering it is a deliberate, SURPASS-PERL fidelity fix so a
909        // `\noexpand`'d delimiter — e.g. xint's `\XINTfstop`, witness
910        // 1804.01117 — survives a number/macro scan for the surrounding parser.)
911        return Ok(Some(crate::token::noexpand_family(&unexpanded)));
912      }
913    }
914    // Wow!!!!! See TeX the Program \S 309
915    // SHOULD count nesting of { }!!! when SCANNED (not digested)
916    let check_alignment_data = {
917      if has_reading_alignment() && align_group_count() == 0 {
918        if let Some((_atoken, atype, ahidden)) = is_column_end(&token) {
919          let reading_alignment = get_reading_alignment().unwrap();
920          Some((reading_alignment, atype, ahidden))
921        } else {
922          None
923        }
924      } else {
925        None
926      }
927    };
928    if let Some((reading_alignment, atype, ahidden)) = check_alignment_data {
929      if let DigestedData::Alignment(data) = reading_alignment.data() {
930        handle_template(data.borrow_mut(), token, atype, ahidden)?;
931      } else {
932        panic!("malformed alignmed was stored?");
933      }
934      // And *then* continue the main loop checks
935    } else if token.get_catcode().is_active_or_cs() {
936      // Read the meaning via closure so we can branch on the borrowed
937      // Stored without cloning (Stored::clone was ~1% of total on
938      // siunitx-heavy profiles; this site fires on every CS/ACTIVE
939      // expansion — the hottest lookup_meaning caller).
940      enum Outcome {
941        LetTo(Token),
942        Undefined,
943        NonExpandable,
944        Invoke(std::rc::Rc<dyn Definition>),
945      }
946      let outcome = with_meaning(&token, |defn_opt| match defn_opt {
947        Some(Stored::Token(t)) => Outcome::LetTo(*t),
948        Some(Stored::None) | None => Outcome::Undefined,
949        Some(other) => match other.to_definition() {
950          Some(defn) => {
951            if !defn.is_expandable() || (defn.is_protected() && !fully_expand) {
952              Outcome::NonExpandable
953            } else {
954              Outcome::Invoke(defn)
955            }
956          },
957          None => Outcome::Undefined,
958        },
959      });
960      match outcome {
961        Outcome::LetTo(let_token) => {
962          return Ok(Some(if for_conditional { let_token } else { token }));
963        },
964        Outcome::Undefined => {
965          if token.get_catcode() == Catcode::CS {
966            return Ok(Some(generate_error_stub(&token)?));
967          } else {
968            return Ok(Some(token));
969          }
970        },
971        Outcome::NonExpandable => {
972          return Ok(Some(token));
973        },
974        Outcome::Invoke(defn) => {
975          local_current_token(token);
976          // Grow the native stack ahead of deep expansion recursion (xint
977          // `\XINT_…` number-arg chains nest tens of thousands deep) so
978          // finite-deep recursion completes instead of overflowing the conversion
979          // thread's 256 MB stack → SIGABRT (Perl degrades via `$MAXSTACK`). This
980          // only grows the stack; the depth CAP is `ExpandDepthGuard` at the top
981          // of `read_x_token`. Same idiom as the recursive walks in `document.rs`
982          // / the math parser; params in `crate::stack_guard`.
983          #[cfg_attr(not(feature = "token-locators"), allow(unused_mut))]
984          let mut invoked = crate::stack_guard::maybe_grow(|| defn.invoke(false))?;
985          // token-locators: fill-only origin inheritance. A macro that
986          // expands into synthesized tokens with no origin — e.g.
987          // `\today → ExplodeText!(Today!())` yielding "May 25, 2026" —
988          // would leave its output unlocatable. Attribute such tokens to
989          // the invocation site so the rendered text is source-mapped.
990          // The inherited handle is flagged `inherited` (one push per
991          // expansion, shared by every result token), so `child_span`'s
992          // genuine-origin-first scan never lets a macro's structural body
993          // literals widen its arguments' content-exact span. We also never
994          // overwrite a token that already carries an origin. See
995          // SOURCE_PROVENANCE.md §3.1.3.
996          #[cfg(feature = "token-locators")]
997          {
998            let inv_loc = token.loc;
999            if inv_loc != 0 {
1000              let mut inherited = 0u32;
1001              for t in invoked.unlist_mut() {
1002                if t.loc == 0 {
1003                  if inherited == 0 {
1004                    inherited = crate::token::push_inherited_origin(inv_loc);
1005                  }
1006                  t.loc = inherited;
1007                }
1008              }
1009            }
1010          }
1011          if *TRACE_GROUP_END {
1012            // Print per-event {macro, delta} so post-processing can sum
1013            // by-macro to find which expandable CS contributes net +/- 1
1014            // imbalance across the run. Format: TRACE_GE delta CS
1015            let (mut begs, mut ends) = (0, 0);
1016            for t in invoked.unlist_ref() {
1017              if *t == T_CS!("\\group_begin:") || *t == T_CS!("\\begingroup") {
1018                begs += 1;
1019              } else if *t == T_CS!("\\group_end:") || *t == T_CS!("\\endgroup") {
1020                ends += 1;
1021              }
1022            }
1023            if begs > 0 || ends > 0 {
1024              eprintln!(
1025                "TRACE_GE delta={} begs={} ends={} cs={}",
1026                begs - ends,
1027                begs,
1028                ends,
1029                token
1030              );
1031            }
1032          }
1033          unread(invoked);
1034          expire_current_token();
1035          continue;
1036        },
1037      }
1038    } else {
1039      // Perl Gullet.pm L421-422: track { and } at scan level for ALIGN_STATE
1040      match token.get_catcode() {
1041        Catcode::BEGIN => increment_align_group_count(),
1042        Catcode::END => decrement_align_group_count(),
1043        _ => {},
1044      }
1045      return Ok(Some(token));
1046    }
1047  }
1048}

1093      Some(t) => {

1128/// Approximates TeX's scan_toks (but doesn't parse \def parameter lists)
1129/// and only optionally requires the openning "{".
1130///
1131/// It may return comments in the token lists.
1132/// The `is_macrodef` flag affects whether # parameters are "packed" for macro bodies.
1133/// If `require_open` is true, the opening T_BEGIN has not yet been read, and is required.
1134///
1135/// If `toplevel` is true, it will automatically close empty mouths as it reads,
1136/// and will also fully expand macros (unless overridden by `expansion_level` being explicitly Off).
1137pub fn read_balanced(
1138  expansion_level: ExpansionLevel,
1139  is_macrodef: bool,
1140  require_open: bool,
1141) -> Result<Tokens> {
1142  use ExpansionLevel::*;
1143  if !require_open {
1144    decrement_align_group_count();
1145  }
1146  local_align_group_count(1000000);
1147  // let startloc = if lookup_verbosity() > 0 { Some(get_locator()) } else { None };
1148  // Do we need to expand to get the { ???
1149  if require_open {
1150    let token_opt = if expansion_level != Off {
1151      read_x_token(Some(false), false, None)?
1152    } else {
1153      read_token()?
1154    };
1155    let is_open = match token_opt {
1156      None => false,
1157      Some(token) => {
1158        token.get_catcode() == Catcode::BEGIN
1159          || with_meaning(
1160            &token,
1161            |m| matches!(m, Some(Stored::Token(t)) if *t == T_BEGIN!()),
1162          )
1163      },
1164    };
1165    if !is_open {
1166      // Push the token back so subsequent reads (especially alignment `&`,
1167      // newline `\\`, or `\end{tabular}`) recover gracefully when an
1168      // upstream macro had a required `{}` arg with no `{...}` available
1169      // (e.g. mn2e/multirow with a missing 3rd arg — 0903.4199 cascade).
1170      // Without this, the consumed `&` was lost from alignment context,
1171      // turning a 1-error "missing arg" into a 10001-cap `&`-cascade.
1172      if let Some(t) = token_opt {
1173        unread_one(t);
1174      }
1175      Error!("expected", "{", s!("Expected opening '{{'"));
1176      return Ok(Tokens!());
1177    }
1178  }
1179  // Pre-size the token accumulator: most balanced reads are short
1180  // macro arguments (~4–16 tokens). This skips the Vec's early
1181  // doublings that the callgrind profile attributes to
1182  // `raw_vec::finish_grow` (1% of total instructions in read_balanced
1183  // alone).
1184  let mut tokens: Vec<Token> = Vec::with_capacity(16);
1185  let mut level = 1;
1186  loop {
1187    // Resource checkpoint: this loop reads RAW (pushback.pop / mouth
1188    // read_token below), not via read_token/read_x_token — without its own
1189    // checkpoint, `\edef`-body expansion loops (`\def\x{a\x}\edef\y{\x}`)
1190    // bypass the token/pushback limits AND the expansion cycle guard, and
1191    // grind to the multi-GB process watchdog instead of a clean early Fatal.
1192    let guard_active = read_resource_checkpoint()?.unwrap_or(false);
1193    // we'll keep comments in the result
1194    let mut next_token = None;
1195    if !gullet!().pending_comments.is_empty() {
1196      tokens.extend(gullet_mut!().pending_comments.drain(..));
1197    }
1198    // Examine pushback first
1199    while let Some(pushback_token) = runtime_mut!().unwrap().pushback.pop() {
1200      match pushback_token.get_catcode() {
1201        Catcode::COMMENT => tokens.push(pushback_token),
1202        Catcode::MARKER => handle_marker(pushback_token),
1203        _ => {
1204          next_token = Some(pushback_token);
1205          break;
1206        },
1207      };
1208    }
1209    // Not in pushback, read from the current Mouth
1210    if next_token.is_none() {
1211      while let Some(token) = runtime_mut!().unwrap().mouth.read_token() {
1212        match token.get_catcode() {
1213          Catcode::COMMENT => tokens.push(token),
1214          Catcode::MARKER => handle_marker(token),
1215          _ => {
1216            next_token = Some(token);
1217            break;
1218          },
1219        };
1220      }
1221    }
1222    // ProgressStep() if ($$self{progress}++ % $TOKEN_PROGRESS_QUANTUM) == 0;
1223    if let Some(ref t) = next_token {
1224      // Cycle-guard checkpoint on the raw read stream (see the resource
1225      // checkpoint at the loop top for why read_balanced needs its own).
1226      cycle_guard_checkpoint(guard_active, t)?;
1227    }
1228    match next_token {
1229      // Mouth exhausted mid-balanced-read: mirror read_x_token / tex.web get_next
1230      // §362-365 — a transparent autoclose injection (\scantokens, raw_tex) is
1231      // part of the current logical stream, so drain it and resume the enclosing
1232      // mouth rather than reporting an unbalanced read. xint's
1233      // `\edef\X{\scantokens{...}}` opens an autoclose mouth mid-edef whose
1234      // matching `}` lives in the PARENT file; not crossing breaks the read at the
1235      // boundary and leaks `\xintexprSafeCatcodes`' `\begingroup`, corrupting
1236      // everything after. SURPASS-PERL: Perl readBalanced (Gullet.pm:466) `last`s
1237      // here and also fails this xint input.
1238      //
1239      // Gate on the mouth KIND, not just the autoclose bit: `\input` file
1240      // mouths are ALSO opened autoclose, and a truncated/unbalanced included
1241      // file must ERROR like TeX ("File ended while scanning use of …") and
1242      // Perl — not silently absorb the parent document into the argument
1243      // (PR_READINESS should-fix 9). Only string/literal injections
1244      // (\scantokens, RawTeX) are transparent to a balanced read.
1245      None => {
1246        let cross = {
1247          let gullet = gullet!();
1248          gullet
1249            .runtime
1250            .as_ref()
1251            .map(|r| r.autoclose && r.mouth.foodtype() != crate::mouth::FoodType::File)
1252            .unwrap_or(false)
1253            && !gullet.mouthstack.is_empty()
1254        };
1255        if cross {
1256          close_mouth(false)?;
1257          continue;
1258        }
1259        break;
1260      },
1261      Some(token) => match token.get_catcode() {
1262        Catcode::CS if token.text == pin!("\\dont_expand") => {
1263          if let Some(next_t) = read_token()? {
1264            tokens.push(next_t); // Pass on NEXT token, unchanged.
1265          }
1266        },
1267        Catcode::END => {
1268          // Perl Gullet.pm L476: track ALIGN_STATE for } inside readBalanced
1269          decrement_align_group_count();
1270          level -= 1;
1271          if level <= 0 {
1272            break;
1273          }
1274          tokens.push(token);
1275        },
1276        Catcode::BEGIN => {
1277          // Perl Gullet.pm L482: track ALIGN_STATE for { inside readBalanced
1278          increment_align_group_count();
1279          level += 1;
1280          tokens.push(token);
1281        },
1282        cc => {
1283          // Wow!!!!! See TeX the Program \S 309
1284          // Not sure if this code still applies within scan_toks???
1285          // SHOULD count nesting of { }!!! when SCANNED (not digested)
1286          if has_reading_alignment()
1287            && align_group_count() == 0
1288            && let Some((_atoken, atype, ahidden)) = is_column_end(&token)
1289          {
1290            match get_reading_alignment().unwrap().data() {
1291              DigestedData::Alignment(data) => {
1292                handle_template(data.borrow_mut(), token, atype, ahidden)?;
1293              },
1294              _ => {
1295                panic!("malformed alignmed was stored?");
1296              },
1297            }
1298            continue;
1299          }
1300          // Note: use general-purpose lookup, since we may reexamine $defn below
1301          if expansion_level != Off && cc.is_active_or_cs() {
1302            // Borrow the stored meaning via with_meaning so the Stored
1303            // enum is not cloned per token. We extract (a) whether a
1304            // meaning exists at all (for the undefined-CS diagnostic
1305            // below) and (b) the Rc<dyn Definition> if it's a proper
1306            // definition — both are cheap (bool + Rc-clone).
1307            let (has_meaning, defn_opt) =
1308              with_meaning(&token, |m| (m.is_some(), m.and_then(|s| s.to_definition())));
1309            if let Some(defn) = defn_opt {
1310              if defn.is_expandable() && (!defn.is_protected() || expansion_level == Full) {
1311                local_current_token(token);
1312                let expansion = defn.invoke(false)?;
1313                if expansion.is_empty() {
1314                  expire_current_token();
1315                  continue;
1316                }
1317                // If a special \the type command, push the expansion directly into the result
1318                // Well, almost directly: handle any MARKER tokens now, and possibly un-pack T_PARAM
1319                //
1320                // Perl `Gullet.pm:505` checks `$$defn{cs}[0]` — but in Perl the Lt-aliases
1321                // (e.g. `Lt('\\exp_not:n','\\unexpanded')`) share the SAME Definition, so its
1322                // cs field IS `\unexpanded`. In Rust the dump-writer emits `\exp_not:n` as a
1323                // separate Expandable with alias=`\unexpanded`; check the alias too so the
1324                // DEFERRED_COMMANDS gate fires for `\exp_not:n {…}` inside `\edef` bodies.
1325                // Without this, expl3's `\seq_gpush:Nn` (which uses `\exp_not:n` to wrap
1326                // `\__seq_item:n {…}`) loses its item — the item gets re-expanded into the
1327                // expandable-error trap, leaving the seq stack empty and triggering
1328                // `extra-pop-label`/`\q_no_value`-recursion cascades during `\@pushfilename`.
1329                let cs_matches = DEFERRED_COMMANDS.contains(&defn.get_cs().text);
1330                let alias_matches = defn
1331                  .get_alias()
1332                  .map(|a| DEFERRED_COMMANDS.contains(&arena::pin(a)))
1333                  .unwrap_or(false);
1334                if expansion_level != Full && (cs_matches || alias_matches) {
1335                  for t in expansion.unlist() {
1336                    match t.get_catcode() {
1337                      Catcode::MARKER => handle_marker(t),
1338                      Catcode::PARAM if is_macrodef => {
1339                        // "unpack" to cover the packParameters at end!
1340                        tokens.push(t);
1341                        tokens.push(t);
1342                      },
1343                      _ => tokens.push(t),
1344                    }
1345                  }
1346                } else {
1347                  // otherwise, prepend to pushback to be expanded further.
1348                  unread(expansion);
1349                }
1350                expire_current_token();
1351                continue;
1352              }
1353            } else if cc == Catcode::CS && !has_meaning {
1354              // cs SHOULD have defn by now; report early!
1355              generate_error_stub(&token)?;
1356            }
1357          }
1358          // Return the token — EXCEPT a `\special_relax` (noexpand'd) family token
1359          // collected into an expanded token list reverts to its plain shadowed
1360          // identity: TeX's no_expand_flag is transient (tex.web §1149-1153), so
1361          // `\edef`/`\xdef` store the PLAIN token, not a relax marker (etex ground
1362          // truth: `\def\s{\noexpand\s}\edef\r{\romannumeral0\s}` → `\meaning\r` =
1363          // "macro:->\s", xint's f-stop idiom). Otherwise the family token persists
1364          // into the `\edef` body and a later number scan hits "Missing number".
1365          // Gated on CS/active so the hot per-token push pays only a catcode check.
1366          if cc.is_active_or_cs() {
1367            tokens.push(token.noexpand_shadowed().unwrap_or(token));
1368          } else {
1369            tokens.push(token);
1370          }
1371        },
1372      },
1373    }
1374  }
1375  if level > 0 {
1376    // Reached for a genuinely unbalanced read: a balancing end in a LITERAL
1377    // (string-injection) mouth IS recognized via the autoclose crossing above;
1378    // a FILE boundary deliberately is not (TeX/Perl parity — "file ended
1379    // while scanning"), so a truncated \input lands here with the loud Error.
1380    // TODO: add the startloc details
1381    // my $loc_message = $startloc ? ("Started at " . ToString($startloc)) : ("Ended at " .
1382    // ToString($self->getLocator));
1383    Error!(
1384      "expected",
1385      "}",
1386      "Gullet->readBalanced ran out of input in an unbalanced state"
1387    );
1388  }
1389  expire_align_group_count();
1390  if tokens.is_empty() {
1391    Ok(Tokens!())
1392  } else {
1393    Ok(if is_macrodef {
1394      Tokens::new(tokens).pack_parameters()?
1395    } else {
1396      Tokens::new(tokens)
1397    })
1398  }
1399}

1401/// Match the input against a set of keywords; Similar to readMatch, but the keywords are strings,
1402/// and Case and catcodes are ignored; additionally, leading spaces are skipped.
1403/// AND, macros are expanded.
1404///
1405/// Perf: zero-allocation char-wise comparison against each keyword.
1406/// The previous version allocated two Strings per char-match (via `to_uppercase()`
1407/// and `char::to_string()`), which was expensive in hot parameter parsing loops.
1408pub fn read_keyword(keywords: &[&str]) -> Result<Option<String>> {
1409  skip_spaces()?;
1410  for keyword in keywords.iter() {
1411    // Pre-size to the keyword length — `matched` holds one token per
1412    // matched char, and we unread them on no-match. Keyword-match
1413    // runs on every parameter/keyword read; small win per call.
1414    let mut matched = Vec::with_capacity(keyword.len());
1415    let mut ok = true;
1416    for expected in keyword.chars() {
1417      let Some(tok) = read_x_token(Some(false), false, None)? else {
1418        ok = false;
1419        break;
1420      };
1421      // Compare char-by-char against the token's text, case-insensitively.
1422      let eq = tok.with_str(|s| {
1423        let mut it = s.chars();
1424        match it.next() {
1425          Some(c) if it.next().is_none() => {
1426            // single-char token: case-insensitive compare
1427            c.to_uppercase().eq(expected.to_uppercase())
1428          },
1429          _ => false,
1430        }
1431      });
1432      matched.push(tok);
1433      if !eq {
1434        ok = false;
1435        break;
1436      }
1437    }
1438    if ok {
1439      return Ok(Some(keyword.to_string()));
1440    } else {
1441      unread(matched.into());
1442    }
1443  }
1444  Ok(None)
1445}

1447/// Return a (balanced) sequence tokens until a match against one of the Tokens in @delims.
1448///
1449/// Note that Braces on input hides the contents from matching,
1450/// so this assumes there wont be braces in $delim!
1451/// But, see readUntilBrace for that case.
1452pub fn read_until(delim: &Tokens) -> Result<Tokens> {
1453  // Pre-size like `read_balanced`: the accumulator is grown one token at a time
1454  // in the loops below, so an unsized `Vec::new()` pays the 0→1→2→4→8 doubling
1455  // reallocations on every call. 16 covers the common short delimited read in a
1456  // single allocation (a top `grow_one` site in the allocation profile).
1457  let mut tokens: Vec<Token> = Vec::with_capacity(16);
1458  let mut nbraces = 0;
1459  let want = delim.unlist_ref();
1460  let ntomatch = want.len();
1461  let mut has_matched;
1462
1463  if ntomatch == 1 {
1464    let want = &want[0];
1465    loop {
1466      let token = match read_token()? {
1467        Some(t) => t,
1468        None => {
1469          // Ran out!
1470          unread(Tokens::new(tokens));
1471          return Ok(Tokens!()); // Not more correct, but maybe less confusing?
1472        },
1473      };
1474      // Perl: check direct match OR \special_relax smuggling (Gullet.pm line 662)
1475      if token == *want || special_relax_matches(&token, want) {
1476        break;
1477      }
1478      match token.get_catcode() {
1479        Catcode::MARKER => {
1480          // would have been handled by readToken, but we're bypassing
1481          handle_marker(token);
1482        },
1483        Catcode::BEGIN => {
1484          // And if it's a BEGIN, copy till balanced END
1485          nbraces += 1;
1486          tokens.push(token);
1487          let balanced_arg = read_balanced(ExpansionLevel::Off, false, false)?;
1488          if !balanced_arg.is_empty() {
1489            tokens.extend(balanced_arg.unlist());
1490          }
1491          tokens.push(T_END!());
1492        },
1493        _ => {
1494          tokens.push(token);
1495        },
1496      }
1497    }
1498  } else {
1499    let mut ring = VecDeque::new();
1500    loop {
1501      // prefill the required number of tokens
1502      while ring.len() < ntomatch {
1503        let token = match read_token()? {
1504          Some(t) => t,
1505          None => {
1506            // Ran out!
1507            unread(Tokens::new(tokens));
1508            return Ok(Tokens!()); // Not more correct, but maybe less confusing?
1509          },
1510        };
1511        // Perl: $$token[1] == CC_BEGIN — direct catcode check
1512        if token.get_catcode() == Catcode::BEGIN {
1513          // read balanced, and refill ring.
1514          nbraces += 1;
1515          for r_token in ring {
1516            tokens.push(r_token);
1517          }
1518          tokens.push(token);
1519          let balanced_arg = read_balanced(ExpansionLevel::Off, false, false)?;
1520          if !balanced_arg.is_empty() {
1521            tokens.append(&mut balanced_arg.unlist());
1522          }
1523          tokens.push(T_END!()); // Copy directly to result
1524          ring = VecDeque::new(); // and retry
1525        } else {
1526          ring.push_back(token);
1527        }
1528      }
1529      has_matched = &ring == want; // Test match
1530      if has_matched {
1531        break;
1532      } // Matched all!
1533      if let Some(ring_token) = ring.pop_front() {
1534        tokens.push(ring_token);
1535      }
1536    }
1537  }
1538  // Notice that IFF the arg looks like {balanced}, the outer braces are stripped
1539  // so that delimited arguments behave more similarly to simple, undelimited arguments.
1540  // Perl: ($nbraces == 1) && ($tokens[0][1] == CC_BEGIN) && ($tokens[-1][1] == CC_END)
1541  if nbraces == 1
1542    && tokens.first().unwrap().get_catcode() == Catcode::BEGIN
1543    && tokens.last().unwrap().get_catcode() == Catcode::END
1544  {
1545    tokens.remove(0);
1546    tokens.pop();
1547  }
1548  Ok(Tokens::new(tokens))
1549}

1576/// Quiet version of read_cs_name — used by \ifcsname.
1577/// In TeX, \ifcsname silently skips non-expandable CS tokens and returns the constructed name
1578/// without emitting errors (unlike \csname which DOES emit errors).

1581fn read_cs_name_inner(quiet: bool) -> Result<Token> {
1582  // TeX does NOT store the csname with the leading `\`, BUT stores active chars with a flag
1583  // However, so long as the Mouth's CS and \string properly respect \escapechar, all's well!
1584
1585  // Safety bound: a real CS name fits in well under 256 chars. We've seen
1586  // pathological cases (lipsum.sty with malformed \cs_set_nopar:Npe expansion,
1587  // or expl3 raw-load before \endcsname is bound) where `\csname` reads
1588  // thousands of tokens accumulating into `cs`, eventually OOMing the
1589  // `read_x_token` pushback Vec. Cap at 4096 bytes — beyond that there's no
1590  // legitimate CS name, just a runaway. Emit one clear error and break.
1591  const MAX_CS_NAME_BYTES: usize = 4096;
1592  let mut cs = String::from("\\");
1593  // keep newlines from having \n inside!
1594  while let Some(token) = read_x_token(Some(true), false, None)? {
1595    if token.defined_as(&TOKEN_ENDCSNAME) {
1596      break;
1597    }
1598    if cs.len() > MAX_CS_NAME_BYTES {
1599      Error!(
1600        "runaway",
1601        "csname",
1602        format!(
1603          "CS-name read exceeded {MAX_CS_NAME_BYTES} bytes; aborting at partial cs: {:?}",
1604          // Truncate by CHARS, not bytes — a byte slice at 200 can split a
1605          // multi-byte UTF-8 char (e.g. 'ä') and panic. Witness 2601.03403.
1606          cs.chars().take(200).collect::<String>()
1607        )
1608      );
1609      break;
1610    }
1611    match token.get_catcode() {
1612      Catcode::CS => {
1613        // Soft-substitute the underlying char for a character-equivalent CS
1614        // token in the \csname stream — a documented divergence from Knuth TeX
1615        // (tex.web L7745-7758 hard-errors "Missing \endcsname inserted" for any
1616        // CS that isn't \endcsname). Our expansion pipeline surfaces PA-aliased
1617        // `Stored::Token` CSes (expl3 `\exp_stop_f:` = frozen space, `\lx@NBSP`
1618        // from CLUSTER-NBSP) into the csname stream where real TeX wouldn't reach
1619        // this state; erroring like Knuth would break real expl3/mhchem/glossaries
1620        // loads, so we substitute the char the author meant. Witnesses: `\lx@NBSP`
1621        // (CLUSTER-NBSP, 18 papers; `~`→U+00A0 in `\csname r@LABEL\endcsname`),
1622        // `\exp_stop_f:` (mhchem raw-load). The `\lx@NBSP` carve-out below stays
1623        // for clarity; the general `Stored::Token` case handles the rest uniformly.
1624        let cs_str = token.with_str(|s| s.to_string());
1625        // Well-known `\text…` primitives that map to a single char in real
1626        // pdflatex's csname-stream interpretation. The `DefPrimitive!(name,
1627        // "char")` body is a closure that wraps a Tbox — not statically
1628        // inspectable from here — so we maintain an explicit table for the
1629        // canonical set. Witnesses (stage-1..3 of 100k warning corpus):
1630        //   \\textquoteright surfacing in `\twemoji flag: Côte d` cluster
1631        //   (≥4 papers across 2603.08303, 2604.13899, 2604.17338, 2604.20621).
1632        let soft_char: Option<char> = match cs_str.as_str() {
1633          "\\lx@NBSP" | "\\lx@nobreakspace" | "\\nobreakspace" => Some('\u{00A0}'),
1634          "\\textquoteright" => Some('\u{2019}'),
1635          "\\textquoteleft" => Some('\u{2018}'),
1636          "\\textquotedblright" => Some('\u{201D}'),
1637          "\\textquotedblleft" => Some('\u{201C}'),
1638          "\\textquotedbl" => Some('"'),
1639          "\\textemdash" => Some('\u{2014}'),
1640          "\\textendash" => Some('\u{2013}'),

1649          "\\textasciicircum" => Some('^'),
1650          "\\textasciitilde" => Some('~'),
1651          "\\textunderscore" => Some('_'),
1652          "\\textasteriskcentered" => Some('*'),
1653          // NFSS encoding-specific glyph CS names (`\<encoding>\<glyph>`)
1654          // built by \DeclareTextSymbol for the i/j dotless letters. These
1655          // surface when a paper composes `\'\i` style accented chars
1656          // that travel through `\lx@applyaccent` and a downstream
1657          // encoding-specific dispatcher. Substitute the dotless glyph
1658          // (U+0131 / U+0237) so the constructed csname carries the
1659          // character the author meant.
1660          // Witnesses: arXiv:2603.22193, 2603.23433, 2604.20621 (twemoji
1661          // São Tomé & Príncipe / St. Barthélemy / Côte d'Ivoire cluster).
1662          "\\T1\\i" | "\\OT1\\i" | "\\LY1\\i" => Some('\u{0131}'),
1663          "\\T1\\j" | "\\OT1\\j" | "\\LY1\\j" => Some('\u{0237}'),
1664          _ => None,
1665        };
1666        if let Some(c) = soft_char {
1667          cs.push(c);
1668        } else if cs_str == "\\lx@applyaccent" {
1669          // Accent macros (`\'`, `\"`, `\^`, …) expand to
1670          // `\lx@applyaccent <accent> <combining> <standalone> {<letter>}`
1671          // (tex_character.rs::accent_def). pdflatex's `\csname` skips the
1672          // accented char (the accent runs in the gullet); ours is a stomach
1673          // `DefPrimitive`, so a literal `\lx@applyaccent` surfaces in the csname
1674          // stream and aborts the read (witnesses: twemoji.sty, arXiv:2603.22193 /
1675          // 2603.23433). Faithful fix: peek the 4 args, append the standalone char
1676          // (arg 3, T_OTHER!) to the name, discard the rest — mirrors the
1677          // implicit-character substitution above.
1678          let _accent = read_x_token(Some(true), false, None)?;
1679          let _combiner = read_x_token(Some(true), false, None)?;
1680          let standalone = read_x_token(Some(true), false, None)?;
1681          // The 4th arg is a brace group `{<letter>}` — consume the
1682          // T_BEGIN, then read tokens until matching T_END.
1683          if let Some(t) = read_x_token(Some(true), false, None)?
1684            && t.get_catcode() == Catcode::BEGIN
1685          {
1686            let mut depth: i32 = 1;
1687            while depth > 0 {
1688              match read_x_token(Some(true), false, None)? {
1689                Some(t2) => match t2.get_catcode() {
1690                  Catcode::BEGIN => depth += 1,
1691                  Catcode::END => depth -= 1,
1692                  _ => {},
1693                },
1694                None => break,
1695              }
1696            }
1697          }
1698          if let Some(c) = standalone {
1699            c.with_str(|s| cs.push_str(s));
1700          }
1701        } else {
1702          match lookup_meaning(&token) {
1703            Some(Stored::Token(letted)) => {
1704              // CS is \let-equivalent to a single token. If that token is
1705              // a character (LETTER/OTHER/SPACE), append its string repr
1706              // to the constructed csname — mirrors real TeX's behaviour
1707              // of substituting the let-target into the csname stream.
1708              // Non-character lets (Catcode::CS, MATH, etc.) fall through
1709              // to the error branches below.
1710              let target_cc = letted.get_catcode();
1711              if matches!(target_cc, Catcode::LETTER | Catcode::OTHER | Catcode::SPACE) {
1712                if target_cc == Catcode::SPACE {
1713                  cs.push(' ');
1714                } else {
1715                  letted.with_str(|s| cs.push_str(s));
1716                }
1717              } else if !quiet {
1718                let message = s!(
1719                  "The control sequence {:?} should not appear between \\csname and \\endcsname (partial cs so far: {:?})",
1720                  token,
1721                  cs
1722                );
1723                Error!("unexpected", token, message);
1724              }
1725            },
1726            _ => {
1727              if !quiet {
1728                if lookup_definition(&token)?.is_some() {
1729                  let message = s!(
1730                    "The control sequence {:?} should not appear between \\csname and \\endcsname (partial cs so far: {:?})",
1731                    token,
1732                    cs
1733                  );
1734                  Error!("unexpected", token, message);
1735                } else {
1736                  let message = s!("The token {:?} is not defined", token);
1737                  Error!("undefined", token, message);
1738                }
1739              }
1740            },
1741          }
1742        }
1743        // In quiet mode (ifcsname), just skip the CS token
1744      },
1745      Catcode::SPACE => cs.push(' '), // Keep newlines from having \n!
1746      _ => {
1747        token.with_str(|s| cs.push_str(s));
1748      },
1749    };
1750  }
1751  Ok(T_CS!(cs))
1752}

1754/// reads and discards tokens, until it encounters a conditional, if any.
1755/// Perl: skipConditionalBody inner loop (Conditional.pm L127-133) reads tokens directly
1756/// from pushback/mouth (NOT through readToken) and manually tracks
1757/// `$LaTeXML::ALIGN_STATE` for `{` and `}`. Critically, this bypasses the
1758/// "alignment-template trigger" check that fires `handleTemplate` on `&`/`\cr`
1759/// when align_group_count==0 — that check belongs to digestion, not to
1760/// `\else`-skip. Rust's `read_token` includes the trigger; calling it from
1761/// here would let pmatrix's `&` get treated as the OUTER alignment's
1762/// column-end during `\ifx.#1.\else…\fi` skip when `#1` contains
1763/// `\begin{pmatrix}…&…\end{pmatrix}`. (REG-2 / math-ph0501074: the 9-line
1764/// `\nonumber+\lefteqn+\pmatrix` repro.) Use `read_internal_token` instead
1765/// and track BEGIN/END manually, matching Perl byte-for-byte.
1766pub fn read_next_conditional() -> Result<Option<(Token, ConditionalType)>> {
1767  loop {
1768    // The FOURTH reader loop over `read_internal_token` (the `\else`/`\fi`
1769    // skipper — it must not delegate to read_token, whose alignment trigger
1770    // misfires during conditional-skip; see the fn doc). It still needs the
1771    // resource/cycle checkpoints: a skip over a pathologically large stream
1772    // must count progress and remain loop-detectable like every other read
1773    // path (PR #249 review P2-9).
1774    let guard_active = match read_resource_checkpoint()? {
1775      None => return Ok(None),
1776      Some(active) => active,
1777    };
1778    let next_low = read_internal_token();
1779    match next_low {
1780      Some(token) => {
1781        cycle_guard_checkpoint(guard_active, &token)?;
1782        let cc = token.get_catcode();
1783        // Perl L128-130: manual ALIGN_STATE tracking for `{` / `}` (avoids
1784        // the trigger check that read_token applies).
1785        match cc {
1786          Catcode::BEGIN => increment_align_group_count(),
1787          Catcode::END => decrement_align_group_count(),
1788          _ => {},
1789        }
1790        if cc.is_active_or_cs()
1791          && let Some(cond_type) = lookup_conditional(&token)
1792        {
1793          return Ok(Some((token, cond_type)));
1794        }
1795      },
1796      None => {
1797        // Current mouth exhausted. Try closing if autoclosable and there are
1798        // more mouths on the stack (TeX continues reading across input boundaries).
1799        let (autoclose, stack_len) = {
1800          let gullet = gullet!();
1801          let ac = gullet
1802            .runtime
1803            .as_ref()
1804            .map(|r| r.autoclose)
1805            .unwrap_or(false);
1806          let sl = gullet.mouthstack.len();
1807          (ac, sl)
1808        };
1809        if autoclose && stack_len > 0 {
1810          close_mouth(false)?;
1811          continue;
1812        }
1813        return Ok(None);
1814      },
1815    }
1816  }
1817}

1824/// Read and return a "normal" TeX argument
1825///
1826/// The next Token or Tokens (if surrounded by braces).
1827/// `expansion_level` controls expansion as if the argument were read
1828///  and then expanded in isolation:
1829///
1830/// In the case of a single unbraced expandable token,
1831///  it will **not** read any macro arguments from the following input!
1832pub fn read_arg(expansion_level: ExpansionLevel) -> Result<Tokens> {
1833  match read_non_space()? {
1834    None => Ok(Tokens!()),
1835    Some(token) => {
1836      // Perl: $$token[1] == CC_BEGIN — checks actual catcode, NOT defined_as.
1837      // \bgroup (catcode CS) does NOT match here; only literal { does.
1838      if token.get_catcode() == Catcode::BEGIN {
1839        read_balanced(expansion_level, false, false)
1840      } else if matches!(expansion_level, ExpansionLevel::Off) {
1841        // A `\noexpand`'d token captured as an (undelimited) macro argument
1842        // reverts to its plain shadowed identity — faithful to TeX, where the
1843        // `no_expand_flag` is transient and never stored in the captured arg
1844        // (`#1` is `cur_cs`, the plain token). This lets `\ifx\X#1` match (xint's
1845        // `\def\XINTfstop{\noexpand\XINTfstop}` f-stop, witness 1804.01117).
1846        // `\let`/`\ifx` of the LIVE token read via `read_token` (the `Token`
1847        // param), NOT here, so their relax-meaning capture is unaffected.
1848        Ok(Tokens!(token.noexpand_shadowed().unwrap_or(token)))
1849      } else {
1850        // Perl Gullet.pm `readArg`:
1851        //   return $self->readingFromMouth(Tokens(T_BEGIN, $token, T_END), sub {
1852        //       readBalanced($self, $expanded, 0, 1); });
1853        // Use an isolated mouth so leftover tokens (e.g. an extra `}` when
1854        // `$token` itself happens to be T_END) cannot leak back into the
1855        // caller's stream. `unread_vec` here would pollute the parent mouth.
1856        let synth = Tokens::new(vec![T_BEGIN!(), token, T_END!()]);
1857        reading_from_mouth(Mouth::default(), move || -> Result<Tokens> {
1858          unread(synth);
1859          read_balanced(expansion_level, false, true)
1860        })
1861      }
1862    },
1863  }
1864}

1885/// Shared core of [`read_optional`] / [`read_optional_angled`]: if the next
1886/// non-space token is the `open` delimiter, read up to (and consuming) `close`;
1887/// otherwise unread the token and yield `default`.
1888///
1889/// The peek/unread contract is the whole point, and is why an optional
1890/// delimited argument must NOT be spelled `OptionalMatch:x OptionalUntil:y`:
1891/// `Until` never checks for the OPENING delimiter, so when the argument is
1892/// absent it scans to the next `y` ANYWHERE downstream (`\citeA{Smith} and
1893/// $a > b$` swallows the cite and the math, yielding the key `b`).
1894fn read_optional_delimited(
1895  open: Token,
1896  close: Token,
1897  default: Option<Tokens>,
1898) -> Result<Option<Tokens>> {
1899  match read_non_space()? {
1900    None => Ok(None),
1901    // `Token`'s `PartialEq` compares catcode + text (and deliberately NOT the
1902    // token-locators origin handle), so this is the catcode-and-symbol match
1903    // the delimiters need.
1904    Some(t) => {
1905      if t == open {
1906        Ok(Some(read_until(&Tokens!(close))?))
1907      } else {
1908        unread_one(t);
1909        Ok(default)
1910      }
1911    },
1912  }
1913}

1997///   Number    <- Dimension, Glue     (extract raw i64)

2006    Some(token) => {

2009        Some(defn) => {

2014            }

2017              Ok(defn.value_of(args))

2026            } else {
2027              unread_one(token); // Unread

2029            }
2030          } else {

2034        },

2044/// Attempt to coerce a register value from `source_type` to `target_type`.
2045fn coerce_register(
2046  target_type: RegisterType,
2047  source_type: RegisterType,
2048  defn: &Register,
2049) -> Result<Option<RegisterValue>> {
2050  use crate::common::numeric_ops::NumericOps;
2051  // Perl fix 50f0061d: include self-coercions (Number→Number, etc.)
2052  // so \number \fam works when \fam is already a Number register
2053  let can_coerce = matches!(
2054    (target_type, source_type),
2055    (RegisterType::Number, RegisterType::Number)
2056      | (RegisterType::Number, RegisterType::Dimension)
2057      | (RegisterType::Number, RegisterType::Glue)
2058      | (RegisterType::Dimension, RegisterType::Dimension)
2059      | (RegisterType::Dimension, RegisterType::Glue)
2060      | (RegisterType::MuDimension, RegisterType::MuDimension)
2061      | (RegisterType::MuDimension, RegisterType::MuGlue)
2062      | (RegisterType::Glue, RegisterType::Glue)
2063      | (RegisterType::MuGlue, RegisterType::MuGlue)
2064  );
2065  if !can_coerce {
2066    return Ok(None);
2067  }
2068  let args = defn.read_arguments()?;
2069  if let Some(val) = defn.value_of(args) {
2070    let raw = match val {
2071      RegisterValue::Dimension(d) => d.value_of(),
2072      RegisterValue::Glue(g) => g.value_of(),
2073      RegisterValue::MuGlue(mg) => mg.value_of(),
2074      RegisterValue::Number(n) => n.value_of(),
2075      RegisterValue::MuDimension(md) => md.value_of(),
2076      _ => return Ok(None),
2077    };
2078    let coerced = match target_type {
2079      RegisterType::Number => RegisterValue::Number(Number::new(raw)),
2080      RegisterType::Dimension => RegisterValue::Dimension(Dimension::new(raw)),
2081      RegisterType::MuDimension => RegisterValue::MuDimension(MuDimension::new(raw)),
2082      _ => return Ok(None),
2083    };
2084    Ok(Some(coerced))
2085  } else {
2086    Ok(None)
2087  }
2088}

2090/// Match the input against one of the Token or Tokens in @choices; return the matching one or
2091/// undef.
2092pub fn read_match(choices: &[&Tokens]) -> Result<Option<Tokens>> {
2093  for choice in choices {
2094    let mut to_match: Vec<&Token> = choice.unlist_ref().iter().rev().collect();
2095    // `matched` accumulates tokens read so far, bounded by `choice.len()`.
2096    // Pre-size to avoid reallocations on multi-token match attempts.
2097    let mut matched = Vec::with_capacity(choice.unlist_ref().len());
2098    while !to_match.is_empty() {
2099      match read_token()? {
2100        None => break,
2101        Some(token) => {
2102          let cc = token.get_catcode();
2103          // Perl: also check smuggled \special_relax token (Gullet.pm line 612)
2104          let was_last_match = if let Some(&&want) = to_match.last() {
2105            token == want || special_relax_matches(&token, &want)
2106          } else {
2107            false
2108          };
2109          matched.push(token);
2110          if was_last_match {
2111            to_match.pop();
2112          } else {
2113            break;
2114          }
2115
2116          if cc == Catcode::SPACE {
2117            // If this was space, SKIP any following!!!
2118            while let Some(space_token) = read_token()? {
2119              if space_token.get_catcode() != Catcode::SPACE {
2120                // Unread non-space and end — use unread_one for proper agc adjustment
2121                unread_one(space_token);
2122                break;
2123              } else {
2124                matched.push(space_token);
2125              }
2126            }
2127          }
2128        },
2129      }
2130    }
2131    if to_match.is_empty() {
2132      return Ok(Some((*choice).clone())); // All matched!!!
2133    } else {
2134      // Put 'em back and try next — use unread_vec for proper agc adjustment
2135      unread_vec(matched);
2136    }
2137  }
2138  Ok(None)
2139}

2141//======================================================================
2142// Integer, Number
2143//======================================================================
2144// ```
2145// <number> = <optional signs><unsigned number>
2146// <unsigned number> = <normal integer> | <coerced integer>
2147// <coerced integer> = <internal dimen> | <internal glue>
2148// ```
2149pub fn read_number() -> Result<Number> {
2150  let is_negative = read_optional_signs()?;
2151  let s = if is_negative { -1 } else { 1 };
2152  if let Some(n) = read_normal_integer()? {
2153    if is_negative { Ok(n.negate()) } else { Ok(n) }
2154  } else if let Some(n) = read_internal_dimension()? {
2155    Ok(Number::new(s * n.value_of()))
2156  } else if let Some(n) = read_internal_glue()? {
2157    Ok(Number::new(s * n.value_of()))
2158  } else {
2159    let next = read_token()?;
2160    // Perl Gullet.pm:904-905: the primary message is just "Missing number,
2161    // treated as zero"; the processing context and the unexpected-token
2162    // (showUnexpected) are SEPARATE Error details rendered on their own
2163    // lines. Render the tokens with ToString/Stringify, not Rust-Debug
2164    // (which leaks `Some("\\relax")` into user-facing diagnostics).
2165    let current = get_current_token()
2166      .map(|t| t.to_string())
2167      .unwrap_or_default();
2168    let unexpected = match next {
2169      Some(t) => s!("Next token is {}", t.stringify()),
2170      None => s!("Input is empty"),
2171    };
2172    Warn!(
2173      "expected",
2174      "<number>",
2175      "Missing number, treated as zero",
2176      s!("while processing {current}"),
2177      unexpected
2178    );
2179    if let Some(next) = next {
2180      unread_one(next);
2181    }

2183  }

2186/// ```bnf
2187/// <normal integer> = <internal integer> | <integer constant>
2188///   | '<octal constant><one optional space> | "<hexadecimal constant><one optional space>
2189///   | `<character token><one optional space>
2190/// ```
2191pub fn read_normal_integer() -> Result<Option<Number>> {
2192  match read_x_token(None, false, None)? {
2193    None => Ok(None),
2194    Some(token) => {
2195      let cc = token.get_catcode();
2196      let mut text = token.to_string();
2197      if cc == Catcode::OTHER && text.chars().all(|c| c.is_ascii_digit()) {
2198        // Read decimal literal. Overflow is rare but possible on weird
2199        // input (digit runs wider than i64::MAX); Perl's TeX silently
2200        // truncates such values, so we fall back to i64::MAX / MIN on
2201        // parse failure rather than panicking with .expect().
2202        text.push_str(&read_digits(&DIGIT_RE, true)?);
2203        let n = text.parse::<i64>().unwrap_or_else(|_| {
2204          if text.starts_with('-') {
2205            i64::MIN
2206          } else {
2207            i64::MAX
2208          }
2209        });
2210        Ok(Some(Number::new(n)))
2211      } else if token == T_OTHER!("'") {
2212        // Read Octal literal. Perl: `Number(oct(readDigits(...)))`, and
2213        // Perl's `oct("")` is 0 — so a `'` with no octal digit following
2214        // yields 0 (TeX's "Missing number, treated as zero"), NOT a fatal
2215        // error. Mirror that, and clamp overflow to i64::MAX like the
2216        // decimal arm rather than propagating a ParseIntError.
2217        let digits = read_digits(&OCT_RE, true)?;
2218        let decimal = if digits.is_empty() {
2219          0
2220        } else {
2221          i64::from_str_radix(&digits, 8).unwrap_or(i64::MAX)
2222        };
2223        Ok(Some(Number::new(decimal)))
2224      } else if token == T_OTHER!("\"") {
2225        //  Read Hex literal. Perl: `Number(hex(readDigits(...)))`, and
2226        // Perl's `hex("")` is 0 — so a `"` with no hex digit following
2227        // yields 0, NOT a fatal error. (Witness 2008.10843: mdwmath.sty
2228        // raw-load reads a bare `"` with no hex digit → previously a
2229        // `Fatal:Document:Generic(ParseIntError)` aborting the run.)
2230        let digits = read_digits(&HEX_RE, true)?;
2231        let decimal = if digits.is_empty() {
2232          0
2233        } else {
2234          i64::from_str_radix(&digits, 16).unwrap_or(i64::MAX)
2235        };
2236        Ok(Some(Number::new(decimal)))
2237      } else if token == T_OTHER!("`") {
2238        //  Read Charcode: `<character token><one optional space>
2239        let mut s = match read_token()? {
2240          None => String::new(),
2241          Some(next) => next.to_string(),
2242        };
2243        if s.starts_with('\\') {
2244          s.remove(0);
2245        }
2246        let s_char = s.chars().next().unwrap_or('\0');
2247        // Perl: skip1Space($self, 1); — expanded space-skip after charcode
2248        skip_one_space(true)?;
2249        Ok(Some(Number::new(s_char as i64))) //  Only a character token!!! NOT expanded!!!!
2250      } else {
2251        unread_one(token); // Unread
2252        read_internal_integer()
2253      }
2254    },
2255  }
2256}

2382    Some((u.value_of(), UNITY))
2383  } else if let Some(u) = read_internal_glue()? {

2385  } else {
2386    read_keyword(&["true"])?; // But ignore, we're not bothering with mag...
2387    if let Some(u) = read_keyword(&["pt", "pc", "in", "bp", "cm", "mm", "dd", "cc", "sp", "px"])? {
2388      skip_one_space(true)?;
2389      Some(convert_unit_ratio(&u))
2390    } else {
2391      None
2392    }
2393  };

2621/// Check if a token is a space token (catcode SPACE) or an "implicit space"

2631      token,
2632      |m| matches!(m, Some(Stored::Token(t)) if t.get_catcode() == Catcode::SPACE),

2674fn read_digits(range_regex: &Regex, skip: bool) -> Result<String> {
2675  let mut result = String::new();
2676  while let Some(token) = read_x_token(None, false, None)? {
2677    let digit_opt = token.with_str(|s| {
2678      if s.len() == 1 && range_regex.is_match(s) {
2679        s.chars().next()
2680      } else {
2681        None

2683    });

2685      result.push(digit);
2686    } else {
2687      if !(skip && is_space_or_implicit_space(&token)) {
2688        unread_one(token);
2689      }
2690      break;
2691    }
2692  }

2756pub fn is_column_end(token: &Token) -> Option<(Token, &'static str, bool)> {
2757  match token.get_catcode() {

2759    Catcode::CS | Catcode::ACTIVE => {
2760      // Embedded version of Equals, knowing both are tokens
2761      let defn = lookup_meaning(token).unwrap_or_else(|| Stored::Token(*token));
2762      // Perl Gullet.pm L273: if meaning is a Token with CC_ALIGN, treat as alignment tab
2763      if let Stored::Token(t) = &defn
2764        && t.get_catcode() == Catcode::ALIGN
2765      {
2766        return Some((*token, "align", false));
2767      }
2768      for end in *COLUMN_ENDS {

2776    },
2777    _ => None,

2795/// Do something, while reading tokens from a specific Mouth.
2796///
2797/// This reads ONLY from that mouth (or any mouth openned by code in that source),
2798/// and the mouth should end up empty afterwards, and only be closed here.
2799pub fn reading_from_mouth<R, FnR>(mouth: Mouth, reader: FnR) -> Result<R>
2800where FnR: FnOnce() -> Result<R> {
2801  let context_mouth_source = arena::pin(mouth.get_source());
2802  // A cycle is only a cycle WITHIN one expansion context. Rather than
2803  // resetting the guard history here (an earlier fix that also BLINDED the
2804  // guard to outer loops whose body calls `do_expand` each iteration), give
2805  // this reading context a fresh SERIAL that `cycle_guard_checkpoint` mixes
2806  // into every fingerprint: windows can never match across contexts, so
2807  // consecutive identical short expansions — the math0402448 xymatrix
2808  // per-cell `get_xmarg_id` stream that false-positived as a "loop" — stay
2809  // inert, while the OUTER context's serial (restored on exit) keeps outer
2810  // periodicity intact and detectable. PR #249 review P2-7.
2811  {
2812    let mut g = gullet_mut!();
2813    let outer = g.ctx_serial;
2814    g.ctx_stack.push(outer);
2815    g.ctx_next += 1;
2816    g.ctx_serial = g.ctx_next;
2817  }
2818  open_mouth(mouth, false); // only allow mouth to be explicitly closed here.
2819  let reader_result = reader();
2820  // Reading in this context is over (whether Ok or Err): restore the outer
2821  // context's serial before any cleanup/return path below.
2822  {
2823    let mut g = gullet_mut!();
2824    let restored = g.ctx_stack.pop().unwrap_or(0);
2825    g.ctx_serial = restored;
2826  }
2827  // If the reader returned an error (e.g., Fatal from token limit),
2828  // we STILL need to clean up the mouth to preserve the caller's state.
2829  let results: R = match reader_result {
2830    Ok(v) => v,
2831    Err(e) => {
2832      // Force-close our mouth and any autoclosable mouths above it
2833      loop {
2834        let current = gullet!()
2835          .runtime
2836          .as_ref()
2837          .map(|r| arena::pin(r.mouth.get_source()));
2838        if current == Some(context_mouth_source) {
2839          close_mouth(true).ok();
2840          break;
2841        } else if gullet!().mouthstack.is_empty() {
2842          break; // Our mouth was already consumed
2843        } else {
2844          close_mouth(true).ok(); // Close stale mouth above ours
2845        }
2846      }
2847      // Reset progress counter so subsequent processing isn't immediately killed
2848      gullet_mut!().progress = 0;
2849      return Err(e);
2850    },
2851  };
2852  // `mouth` must still be open, with (at worst) empty autoclosable mouths in front of it.
2853  // Rate-limit the "mouth closed" error — when the gullet gets into a state
2854  // where the cleanup loop keeps finding stale mouths above the target, the
2855  // same error can fire on EVERY caller of reading_from_mouth. Arxiv 0906.1883
2856  // (birkmult + local .cls) can trigger 10K+ such firings, one per stack frame.
2857  // Fatal out after 50 repeat firings so the process surfaces a clear "we lost
2858  // the mouth stack" signal instead of filling the log with identical messages.
2859  thread_local! {
2860    static MOUTH_CLOSED_ERRORS: Cell<usize> = const { Cell::new(0) };
2861  }
2862  fn record_mouth_closed_error() { MOUTH_CLOSED_ERRORS.with(|c| c.set(c.get().saturating_add(1))); }
2863  fn should_emit_mouth_closed() -> bool { MOUTH_CLOSED_ERRORS.with(|c| c.get() < 10) }
2864  fn mouth_closed_budget_exhausted() -> bool { MOUTH_CLOSED_ERRORS.with(|c| c.get() >= 50) }
2865  loop {
2866    let mouth_source = gullet!()
2867      .runtime
2868      .as_ref()
2869      .map(|r| arena::pin(r.mouth.get_source()));
2870    if mouth_source == Some(context_mouth_source) {
2871      close_mouth(true)?;
2872      break;
2873    } else if gullet!().mouthstack.is_empty() {
2874      if should_emit_mouth_closed() {
2875        // `arena::to_string` clones the resolved &str into an owned String
2876        // BEFORE we hand it to Error! — a following `arena::pin` triggered
2877        // deep inside generate_message!/get_location() can grow the
2878        // interner's buffer and invalidate a borrowed &str (observed as
2879        // garbled, buffer-adjacent symbol content in 0906.1883 errors).
2880        let src = arena::to_string(context_mouth_source);
2881        Error!(
2882          "unexpected",
2883          "<closed>",
2884          "Mouth is unexpectedly already closed",
2885          s!("Reading from {src}, but it has already been closed.")
2886        );
2887      }
2888      record_mouth_closed_error();
2889      if mouth_closed_budget_exhausted() {
2890        Fatal!(
2891          Stomach,
2892          Recursion,
2893          "Too many unexpectedly-closed mouth errors (>50); gullet mouth-stack state is inconsistent"
2894        );
2895      }
2896      break;
2897    } else {
2898      let is_autoclosable = gullet!()
2899        .runtime
2900        .as_ref()
2901        .map(|r| r.autoclose)
2902        .unwrap_or(false);
2903      if is_autoclosable {
2904        // Auto-closable mouth (e.g. from \scantokens, raw_tex) — safe to close
2905        close_mouth(true)?;
2906      } else {
2907        // Non-autoclosable mouth that isn't our target — this means our target
2908        // mouth was already consumed. Don't close this mouth (it belongs to an
2909        // outer reading_from_mouth call). Just error and stop.
2910        if should_emit_mouth_closed() {
2911          let src = arena::to_string(context_mouth_source);
2912          Error!(
2913            "unexpected",
2914            "<closed>",
2915            "Mouth is unexpectedly already closed",
2916            s!(
2917              "Reading from {src}, but it has already been closed (found different non-closable mouth on top)."
2918            )
2919          );
2920        }
2921        record_mouth_closed_error();
2922        if mouth_closed_budget_exhausted() {
2923          Fatal!(
2924            Stomach,
2925            Recursion,
2926            "Too many unexpectedly-closed mouth errors (>50); gullet mouth-stack state is inconsistent"
2927          );
2928        }
2929        break;
2930      }
2931    }
2932  }
2933  Ok(results)
2934}

2944/// Obscure, but the only way I can think of to End!! (see \bye or \end{document})
2945/// Flush all sources (close all pending mouth's)
2946pub fn flush() {
2947  let mut g = gullet_mut!();
2948  if let Some(ref mut runtime) = g.runtime {
2949    runtime.mouth.finish();
2950  }
2951  while !g.mouthstack.is_empty() {
2952    if let Some(mut entry) = g.mouthstack.pop_front() {
2953      entry.mouth.finish();
2954    }
2955  }
2956  g.runtime = Some(MouthRuntime {
2957    mouth:     Mouth::default(),
2958    pushback:  Vec::with_capacity(128),
2959    autoclose: true,
2960  });
2961  g.mouthstack = VecDeque::new();
2962}
