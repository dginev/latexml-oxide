18#[derive(Debug, Clone, Default, PartialEq, Eq)]
19pub struct Comment(pub String);
20
21impl fmt::Display for Comment {
22  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result { write!(f, "") }
23}
24impl Object for Comment {
25  fn revert(&self) -> Result<Tokens> { Ok(NO_TOKENS) }
26}
27impl BoxOps for Comment {
28  fn get_properties(&self) -> &HashMap<Stored> { &NO_PROPERTIES }
29  fn with_properties<R, FnR>(&self, caller: FnR) -> R
30  where FnR: FnOnce(&HashMap<Stored>) -> R {
31    caller(&NO_PROPERTIES)
32  }
33  fn get_property(&self, key: &str) -> Option<Cow<'_, Stored>> {
34    // Perl: Comment->getProperty('isEmpty') returns 1
35    if key == "isEmpty" {
36      Some(Cow::Owned(Stored::Bool(true)))
37    } else {
38      None
39    }
40  }
41  fn set_property<T: Into<Stored>>(&mut self, _key: &str, _value: T) {} // no-op
42  fn get_string(&self) -> Result<Cow<'_, str>> { Ok(Cow::Borrowed(&self.0)) }
43  fn be_absorbed(&self, document: &mut Document) -> Result<Vec<Node>> {
44    document.insert_comment(&self.0)?;
45    Ok(Vec::new())
46  }
47  fn get_font(&self) -> Result<Option<Cow<'_, Font>>> { Ok(None) }
48  fn get_width(&self, _options: Option<HashMap<Stored>>) -> Result<Option<RegisterValue>> {
49    Ok(Some(RegisterValue::Dimension(Dimension::new(0))))
50  }
51
52  fn compute_size(&self, _options: HashMap<Stored>) -> Result<(Dimension, Dimension, Dimension)> {
53    Ok((
54      Dimension::default(),
55      Dimension::default(),
56      Dimension::default(),
57    ))
58  }

84  fn comment_display_is_always_empty_string() {
85    // Display of a Comment is always "" (comments produce no visible
86    // output).
87    let c = Comment("visible".to_string());
88    assert_eq!(format!("{c}"), "");
89    let c2 = Comment::default();
90    assert_eq!(format!("{c2}"), "");
91  }

100  #[test]
101  fn comment_is_empty_property() {
102    // Perl: Comment->getProperty('isEmpty') returns 1.
103    let c = Comment("anything".to_string());
104    match c.get_property("isEmpty") {
105      Some(Cow::Owned(Stored::Bool(true))) => {},
106      other => panic!("expected Some(Bool(true)), got {other:?}"),
107    }
108    // Other keys return None.
109    assert!(c.get_property("random_key").is_none());
110  }
