1//! Shared XML-replacement template AST + parser + runtime interpreter (#171).
2//!
3//! A constructor's `"<ltx:…>"` replacement is a tiny templating language (see the
4//! grammar comment in `latexml_codegen::constructable`). Historically it had **two**
5//! independent implementations: the compile-time proc-macro
6//! (`latexml_codegen/src/constructable.rs`, a regex-strip state machine fused to
7//! codegen) and the runtime byte-scanner (`latexml_contrib::script_bindings`).
8//! Two implementations of one language is drift waiting to happen.
9//!
10//! This module is the single source of truth: one [`winnow`] parser produces one
11//! [`ReplacementOp`] AST, consumed by **both** front-ends —
12//! * the compile-time codegen walks `&[ReplacementOp]` and emits `quote!`,
13//! * the runtime interpreter [`apply_ops`] walks the same AST against a live
14//!   `Document`.
15//!
16//! The semantics mirror the Perl `LaTeXML::Core::Definition::Constructor::Compiler`
17//! (`Compiler.pm`) faithfully — the existing `constructable.rs` is the ground
18//! truth this reproduces, including its quirks (see [`unquote`]). The dialect:
19//!
20//! ```text
21//!  #1..#9            n-th digested argument                 (Value::Arg)
22//!  #name             named whatsit property                 (Value::Prop)
23//!  &func(args,…)     function call (whitelisted at runtime)  (Value::Func)
24//!  <q a='v' …>       open element + attributes              (OpenElement)
25//!  <q … />           empty element (open + close)
26//!  </q>              close element                          (CloseElement)
27//!  <?q a='v' …?>     processing instruction                 (ProcessingInstruction)
28//!  ?test(if)(else)   conditional                            (Conditional)
29//!  ^ / ^^  prefix    float the next element/attribute       (FloatKind)
30//!  key='v'           set attribute on current node          (SetAttribute)
31//!  literal text      absorb as a string                     (Text)
32//! ```

54/// A `^` / `^^` float prefix attaching to the next open-element. `^` floats to
55/// where the element is allowed; `^^` additionally closes intervening open
56/// elements if possible (Perl Compiler.pm float_type 1 vs 2). Counts ≥2 collapse
57/// to `Double` (counts ≥3 never occur in practice).
58#[derive(Debug, Clone, Copy, PartialEq, Eq)]
59pub enum FloatKind {
60  Single,
61  Double,
62}
63
64/// A substitutable value: `#n`, `#name`, `&func(…)`, or literal text. Literal only
65/// arises inside function arguments / attribute strings — never at content
66/// position (a bare literal there is [`ReplacementOp::Text`]).
67#[derive(Debug, Clone, PartialEq, Eq)]
68pub enum Value {
69  /// `#n` (1..=9) — the n-th digested argument (1-based).
70  Arg(usize),
71  /// `#name` — a named whatsit property.
72  Prop(String),
73  /// `&func(args…)` — a function call (resolved through a whitelist at runtime,
74  /// a Rust call at compile time).
75  Func { name: String, args: Vec<FuncArg> },
76  /// Literal text, already `unquote`d.
77  Literal(String),
78}

80/// A `&func(…)` argument — either a bare value or a quoted interpolated string.
81#[derive(Debug, Clone, PartialEq, Eq)]
82pub enum FuncArg {
83  Value(Value),
84  Str(AttrValue),
85}

87/// An interpolated attribute value: `'role-#1'` → `[Literal("role-"), Value(Arg 1)]`.
88#[derive(Debug, Clone, PartialEq, Eq)]
89pub struct AttrValue {
90  pub parts: Vec<AttrPart>,
91}

93/// One piece of an [`AttrValue`].
94#[derive(Debug, Clone, PartialEq, Eq)]
95pub enum AttrPart {
96  Literal(String),
97  Value(Value),
98  /// `?test(ifval)(elseval)` inside a quoted attribute string — branches are
99  /// single values (Perl `translate_string`).
100  Conditional { test: Value, then_val: Value, else_val: Value },
101}

103/// An attribute-list entry inside a `<tag …>` or `<?pi …?>` — a key/value pair or
104/// a conditional set of pairs (Perl `translate_avpairs`).
105#[derive(Debug, Clone, PartialEq, Eq)]
106pub enum AttrPair {
107  KeyValue { key: String, value: AttrValue },
108  Conditional { test: Value, then_attrs: Vec<AttrPair>, else_attrs: Vec<AttrPair> },
109}

111/// One operation in a compiled replacement template.
112#[derive(Debug, Clone, PartialEq, Eq)]
113pub enum ReplacementOp {
114  /// `<q a='v' …>` (or `<q … />` when `self_closing`). `float` set when a `^`/`^^`
115  /// prefix attaches here.
116  OpenElement {
117    qname: String,
118    attrs: Vec<AttrPair>,
119    float: Option<FloatKind>,
120    self_closing: bool,
121  },
122  /// `</q>`.
123  CloseElement { qname: String },
124  /// `<?q a='v' …?>`.
125  ProcessingInstruction { qname: String, attrs: Vec<AttrPair> },
126  /// `#n` / `#name` / `&func(…)` at content position — absorb the value.
127  AbsorbValue { value: Value },
128  /// `key='v'` at content position — set the attribute on the current node.
129  /// `float` set when a `^` prefix attaches here.
130  SetAttribute { key: String, value: AttrValue, float: bool },
131  /// Literal text to absorb.
132  Text { text: String },
133  /// `?test(if)(else)` — branches are op-lists.
134  Conditional { test: Value, then_ops: Vec<ReplacementOp>, else_ops: Vec<ReplacementOp> },
135}

139/// Parse an XML-replacement template into a [`ReplacementOp`] op-list. The entry
140/// point for both consumers (codegen at compile time, [`apply_ops`] at runtime).
141pub fn parse_replacement(template: &str) -> Result<Vec<ReplacementOp>> {
142  ops_with_float
143    .parse(template)
144    .map_err(|e| Error::from(format!("replacement template parse error: {e}")))
145}
146
147/// A leading `^`/`^^` float prefix, then a sequence of ops, with the float
148/// attached to the first open-element / top-level attribute that follows. Used as
149/// the top-level parser and for conditional op-branches (each re-parsed
150/// independently, mirroring Perl's recursive `compile_replacement_tokens`).
151fn ops_with_float(input: &mut &str) -> ModalResult<Vec<ReplacementOp>> {
152  let float = opt(float_prefix).parse_next(input)?;
153  let mut ops: Vec<ReplacementOp> = repeat(0.., op).parse_next(input)?;
154  if let Some(fk) = float {
155    attach_float(&mut ops, fk);
156  }
157  Ok(ops)
158}
159
160/// Attach a leading float to the first open-element / top-level attribute. Scans
161/// only the top-level op list (a float does not reach into conditional branches).
162fn attach_float(ops: &mut [ReplacementOp], fk: FloatKind) {
163  for o in ops.iter_mut() {
164    match o {
165      ReplacementOp::OpenElement { float, .. } => {
166        *float = Some(fk);
167        return;
168      },
169      ReplacementOp::SetAttribute { float, .. } => {
170        *float = true;
171        return;
172      },
173      _ => {},
174    }
175  }
176}

178/// `^+ \s*` — one or more carets then whitespace (Perl `FLOAT_RE`).
179fn float_prefix(input: &mut &str) -> ModalResult<FloatKind> {
180  let carets = take_while(1.., '^').parse_next(input)?;
181  ws(input)?;
182  Ok(if carets.chars().count() >= 2 { FloatKind::Double } else { FloatKind::Single })
183}

185/// One top-level operation. Branch order mirrors `compile_replacement_tokens`'s
186/// `while`-loop priority. `pi`/`open` consume leading whitespace (Perl regexes
187/// allow `^\s*<`); `close`/`text` do not (asymmetry preserved faithfully).
188fn op(input: &mut &str) -> ModalResult<ReplacementOp> {
189  alt((
190    conditional_op,
191    pi_op,
192    open_tag_op,
193    close_tag_op,
194    absorb_value_op,
195    attribute_op,
196    text_op,
197  ))
198  .parse_next(input)
199}

201fn conditional_op(input: &mut &str) -> ModalResult<ReplacementOp> {
202  peek((literal("?"), one_of(['#', '&']))).parse_next(input)?;
203  let (test, if_s, else_s) = parse_conditional_raw(input)?;
204  let then_ops = reparse_ops(&if_s)?;
205  let else_ops = reparse_ops(&else_s)?;
206  Ok(ReplacementOp::Conditional { test, then_ops, else_ops })
207}

209fn pi_op(input: &mut &str) -> ModalResult<ReplacementOp> {
210  ws(input)?;
211  literal("<?").parse_next(input)?;
212  let name = cut_err_(qname, input)?;
213  let attrs = avpairs(input)?;
214  ws(input)?;
215  cut_err_lit("?>", input)?;
216  Ok(ReplacementOp::ProcessingInstruction { qname: name, attrs })
217}
218
219fn open_tag_op(input: &mut &str) -> ModalResult<ReplacementOp> {
220  ws(input)?;
221  literal("<").parse_next(input)?;
222  // Backtrackable: `</…` or `<?…` must fall through to other branches.
223  let name = qname.parse_next(input)?;
224  let attrs = avpairs(input)?;
225  let self_closing = opt(literal("/")).parse_next(input)?.is_some();
226  cut_err_lit(">", input)?;
227  Ok(ReplacementOp::OpenElement { qname: name, attrs, float: None, self_closing })
228}
229
230fn close_tag_op(input: &mut &str) -> ModalResult<ReplacementOp> {
231  // `</qname\s*>` — no leading whitespace (Perl `LEAD_CLOSE_TAG_RE`).
232  literal("</").parse_next(input)?;
233  let name = cut_err_(qname, input)?;
234  ws(input)?;
235  cut_err_lit(">", input)?;
236  Ok(ReplacementOp::CloseElement { qname: name })
237}

239fn absorb_value_op(input: &mut &str) -> ModalResult<ReplacementOp> {
240  // Gate on `#`/`&` (Perl `LEAD_VALUE_RE`); content_value has no literal fallback.
241  peek(one_of(['#', '&'])).parse_next(input)?;
242  let value = content_value(input)?;
243  Ok(ReplacementOp::AbsorbValue { value })
244}
245
246fn attribute_op(input: &mut &str) -> ModalResult<ReplacementOp> {
247  // `qname\s*=\s*'…'` (Perl `QNAME_KEY_RE`), no leading whitespace.
248  let key = qname.parse_next(input)?;
249  if opt((ws_p, literal("="), ws_p)).parse_next(input)?.is_none() {
250    // qname without `=` is not an attribute — let alt try `text`.
251    return Err(ErrMode::Backtrack(ContextError::new()));
252  }
253  let value = attr_string(input)?;
254  Ok(ReplacementOp::SetAttribute { key, value, float: false })
255}

257fn text_op(input: &mut &str) -> ModalResult<ReplacementOp> {
258  // Random text stops at `<` and the specials (Perl `LEAD_RANDOM_TEXT_RE`).
259  let raw = take_literal_run(input, "", true)?;
260  Ok(ReplacementOp::Text { text: unquote(&raw) })
261}

265/// `?test(a)(b)` → `(test, a_raw, b_raw)`. The caller (op / avpair / string
266/// context) re-parses the raw branch strings appropriately. Mirrors Perl
267/// `parse_conditional`: a missing first paren ⇒ both branches empty; a missing
268/// second paren ⇒ else branch empty.
269fn parse_conditional_raw(input: &mut &str) -> ModalResult<(Value, String, String)> {
270  literal("?").parse_next(input)?;
271  let test = match content_value(input) {
272    Ok(v) => v,
273    Err(ErrMode::Backtrack(e)) => return Err(ErrMode::Cut(e)),
274    Err(e) => return Err(e),
275  };
276  match opt(bracketed).parse_next(input)? {
277    None => Ok((test, String::new(), String::new())),
278    Some(a) => {
279      let b = opt(bracketed).parse_next(input)?.unwrap_or_default();
280      Ok((test, a, b))
281    },
282  }
283}

285/// Extract a balanced `(…)` group, returning the inner text (Perl
286/// `extract_bracketed`). Leading whitespace is skipped; a non-whitespace char
287/// before any `(` means "no group here" (backtrack, input unchanged). Nested
288/// parens balance; quotes are *not* special (faithful to the original).
289fn bracketed(input: &mut &str) -> ModalResult<String> {
290  let s: &str = input;
291  let mut level: i32 = 0;
292  let mut has_open = false;
293  let mut extracted = String::new();
294  let mut consumed = 0usize;
295  let mut closed = false;
296  for (i, c) in s.char_indices() {
297    match c {
298      ')' => {
299        level -= 1;
300        if level < 1 {
301          consumed = i + c.len_utf8();
302          closed = true;
303          break;
304        }
305        extracted.push(c);
306      },
307      '(' => {
308        has_open = true;
309        level += 1;
310        if level > 1 {
311          extracted.push(c);
312        }
313      },
314      other => {
315        if level > 0 {
316          extracted.push(other);
317        } else if !other.is_whitespace() {
318          break; // non-ws before any '(' — not a group
319        }
320      },
321    }
322  }
323  if has_open && closed {
324    *input = &s[consumed..];
325    Ok(extracted)
326  } else {
327    Err(ErrMode::Backtrack(ContextError::new()))
328  }
329}

331/// A value with no literal fallback: `&func(…)`, `#n`, `#name` (Perl
332/// `translate_value` gated by `LEAD_VALUE_RE`).
333fn content_value(input: &mut &str) -> ModalResult<Value> {
334  alt((func_value, arg_value, prop_value)).parse_next(input)
335}

337/// A value *with* literal fallback (used for `&func` arguments): tries the value
338/// forms, else a literal run excluding `exclude`.
339fn full_value(input: &mut &str, exclude: &str) -> ModalResult<Value> {
340  if let Some(v) = opt(content_value).parse_next(input)? {
341    return Ok(v);
342  }
343  let raw = take_literal_run(input, exclude, false)?;
344  Ok(Value::Literal(unquote(&raw)))
345}

347fn arg_value(input: &mut &str) -> ModalResult<Value> {
348  literal("#").parse_next(input)?;
349  let digits = take_while(1.., |c: char| c.is_ascii_digit()).parse_next(input)?;
350  let n: usize = digits.parse().map_err(|_| ErrMode::Backtrack(ContextError::new()))?;
351  if !(1..=9).contains(&n) {
352    return Err(ErrMode::Cut(ContextError::new()));
353  }
354  Ok(Value::Arg(n))
355}

357fn prop_value(input: &mut &str) -> ModalResult<Value> {
358  literal("#").parse_next(input)?;
359  let name = take_while(1.., |c: char| c.is_alphanumeric() || c == '_' || c == '-')
360    .map(str::to_string)
361    .parse_next(input)?;
362  Ok(Value::Prop(name))
363}

365fn func_value(input: &mut &str) -> ModalResult<Value> {
366  // `&([\w:]*)\(` (Perl `FN_RE`).
367  literal("&").parse_next(input)?;
368  let name = take_while(0.., |c: char| c.is_alphanumeric() || c == '_' || c == ':')
369    .map(str::to_string)
370    .parse_next(input)?;
371  literal("(").parse_next(input)?; // no '(' ⇒ backtrack (a bare `&amp;` is text)
372  let mut args = Vec::new();
373  loop {
374    if probe(|i| (ws_p, literal(")")).void().parse_next(i), input) {
375      break;
376    }
377    ws(input)?;
378    let arg = if probe(|i| one_of(['\'', '"']).parse_next(i), input) {
379      FuncArg::Str(attr_string(input)?)
380    } else {
381      FuncArg::Value(full_value(input, ",)")?)
382    };
383    args.push(arg);
384    if opt((ws_p, literal(","), ws_p)).parse_next(input)?.is_none() {
385      break;
386    }
387  }
388  ws(input)?;
389  cut_err_lit(")", input)?;
390  Ok(Value::Func { name, args })
391}

393/// A quoted, interpolated attribute string (Perl `translate_string`). If no
394/// opening quote is present, it consumes a single char and yields an empty value
395/// — faithfully reproducing the original's quirk.
396fn attr_string(input: &mut &str) -> ModalResult<AttrValue> {
397  ws(input)?;
398  let quote = match opt(one_of(['\'', '"'])).parse_next(input)? {
399    Some(q) => q,
400    None => {
401      if let Some(c) = input.chars().next() {
402        *input = &input[c.len_utf8()..];
403      }
404      return Ok(AttrValue { parts: Vec::new() });
405    },
406  };
407  let mut parts = Vec::new();
408  loop {
409    if input.is_empty() {
410      break;
411    }
412    if input.starts_with(quote) {
413      *input = &input[quote.len_utf8()..];
414      break;
415    }
416    if probe(|i| (literal("?"), one_of(['#', '&'])).void().parse_next(i), input) {
417      let (test, if_s, else_s) = parse_conditional_raw(input)?;
418      let then_val = parse_single_value(&if_s)?;
419      let else_val = parse_single_value(&else_s)?;
420      parts.push(AttrPart::Conditional { test, then_val, else_val });
421      continue;
422    }
423    if probe(|i| one_of(['#', '&']).parse_next(i), input) {
424      parts.push(AttrPart::Value(content_value(input)?));
425      continue;
426    }
427    let raw = take_literal_run(input, "'\"", false)?;
428    parts.push(AttrPart::Literal(unquote(&raw)));
429  }
430  Ok(AttrValue { parts })
431}

433/// A set of attribute pairs (Perl `translate_avpairs`): conditionals and
434/// `key='v'` pairs, leading whitespace trimmed each iteration.
435fn avpairs(input: &mut &str) -> ModalResult<Vec<AttrPair>> {
436  let mut pairs = Vec::new();
437  loop {
438    ws(input)?;
439    if let Some(c) = opt(avpair_conditional).parse_next(input)? {
440      pairs.push(c);
441      continue;
442    }
443    if let Some(kv) = opt(avpair_keyval).parse_next(input)? {
444      pairs.push(kv);
445      continue;
446    }
447    break;
448  }
449  Ok(pairs)
450}
451
452fn avpair_conditional(input: &mut &str) -> ModalResult<AttrPair> {
453  peek((literal("?"), one_of(['#', '&']))).parse_next(input)?;
454  let (test, if_s, else_s) = parse_conditional_raw(input)?;
455  Ok(AttrPair::Conditional {
456    test,
457    then_attrs: reparse_avpairs(&if_s),
458    else_attrs: reparse_avpairs(&else_s),
459  })
460}
461
462fn avpair_keyval(input: &mut &str) -> ModalResult<AttrPair> {
463  let key = qname.parse_next(input)?;
464  if opt((ws_p, literal("="), ws_p)).parse_next(input)?.is_none() {
465    return Err(ErrMode::Backtrack(ContextError::new()));
466  }
467  let value = attr_string(input)?;
468  Ok(AttrPair::KeyValue { key, value })
469}

471/// A single value parsed from a conditional branch in string context (Perl
472/// `translate_value` on the branch). Empty branch ⇒ empty literal; trailing text
473/// is ignored (lenient, matching the original).
474fn parse_single_value(s: &str) -> ModalResult<Value> {
475  if s.is_empty() {
476    return Ok(Value::Literal(String::new()));
477  }
478  let mut inp: &str = s;
479  full_value(&mut inp, "")
480}

482/// Re-parse a conditional op-branch as a full op-list (strict: must consume all,
483/// mirroring `compile_replacement_tokens` looping to empty).
484fn reparse_ops(s: &str) -> ModalResult<Vec<ReplacementOp>> {
485  ops_with_float.parse(s).map_err(|_| ErrMode::Cut(ContextError::new()))
486}

488/// Re-parse a conditional avpair-branch as an attribute list (lenient: trailing
489/// text ignored, like `translate_avpairs`).
490fn reparse_avpairs(s: &str) -> Vec<AttrPair> {
491  let mut inp: &str = s;
492  avpairs(&mut inp).unwrap_or_default()
493}

495/// `qname` (XML Name, Perl `QNAME_RE`). Approximated as alnum/`_`/`:`/`.`/`-`
496/// with an alpha/`_`/`:` start — covers every `ltx:`-namespaced name in practice.
497fn qname(input: &mut &str) -> ModalResult<String> {
498  (one_of(is_qname_start), take_while(0.., is_qname_continue))
499    .take()
500    .map(str::to_string)
501    .parse_next(input)
502}
503
504fn is_qname_start(c: char) -> bool { c.is_alphabetic() || c == '_' || c == ':' }
505fn is_qname_continue(c: char) -> bool {
506  c.is_alphanumeric() || matches!(c, '_' | ':' | '.' | '-')
507}

509/// Consume optional whitespace, discarding it.
510fn ws(input: &mut &str) -> ModalResult<()> {
511  let _ = take_while(0.., |c: char| c.is_whitespace()).parse_next(input)?;
512  Ok(())
513}

515/// Whitespace as a tuple-usable parser (returns `()`).
516fn ws_p(input: &mut &str) -> ModalResult<()> { ws(input) }

518/// Non-consuming lookahead test: run `p` on a copy and report whether it
519/// succeeds, leaving `input` untouched. (Pins the winnow error type, which a bare
520/// `peek(...).is_ok()` cannot infer.)
521fn probe<O>(mut p: impl FnMut(&mut &str) -> ModalResult<O>, input: &str) -> bool {
522  let mut s: &str = input;
523  p(&mut s).is_ok()
524}

526/// A literal that hard-fails (`cut_err`) instead of backtracking.
527fn cut_err_lit(lit: &'static str, input: &mut &str) -> ModalResult<()> {
528  match literal(lit).parse_next(input) {
529    Ok(_) => Ok(()),
530    Err(ErrMode::Backtrack(e)) => Err(ErrMode::Cut(e)),
531    Err(e) => Err(e),
532  }
533}

535/// Run a parser, converting a `Backtrack` into a `Cut` (committed position).
536fn cut_err_<O>(
537  mut p: impl FnMut(&mut &str) -> ModalResult<O>,
538  input: &mut &str,
539) -> ModalResult<O> {
540  match p(input) {
541    Ok(o) => Ok(o),
542    Err(ErrMode::Backtrack(e)) => Err(ErrMode::Cut(e)),
543    Err(e) => Err(e),
544  }
545}

547/// Maximal run of literal text. A "unit" is `&amp;`, a TeX control sequence
548/// `\letters`, an escape `\X`, or one ordinary char. Stops at `#`/`?`, a bare
549/// `&`, a lone `\`, the `exclude` chars, and (when `exclude_lt`) `<`. Returns the
550/// raw run (NOT yet `unquote`d). Faithful to Perl's `QUOTED_SPECIALS` classes.
551fn take_literal_run(input: &mut &str, exclude: &str, exclude_lt: bool) -> ModalResult<String> {
552  let s: &str = input;
553  let mut pos = 0usize;
554  while pos < s.len() {
555    let rest = &s[pos..];
556    let c = rest.chars().next().unwrap();
557    if c == '&' {
558      if rest.starts_with("&amp;") {
559        pos += 5;
560        continue;
561      }
562      break; // bare '&' is special
563    }
564    if c == '\\' {
565      let after = &rest[1..];
566      let letters: usize = after
567        .chars()
568        .take_while(|ch| ch.is_ascii_alphabetic() || *ch == '@')
569        .map(char::len_utf8)
570        .sum();
571      if letters > 0 {
572        pos += 1 + letters; // \textbf etc.
573        continue;
574      }
575      if let Some(nc) = after.chars().next() {
576        pos += 1 + nc.len_utf8(); // \X escape
577        continue;
578      }
579      break; // lone trailing backslash
580    }
581    if c == '#' || c == '?' {
582      break;
583    }
584    if exclude_lt && c == '<' {
585      break;
586    }
587    if exclude.contains(c) {
588      break;
589    }
590    pos += c.len_utf8();
591  }
592  if pos == 0 {
593    return Err(ErrMode::Backtrack(ContextError::new()));
594  }
595  let raw = s[..pos].to_string();
596  *input = &s[pos..];
597  Ok(raw)
598}

600/// Reverse the template's escape conventions (Perl `unquote`). Reproduces the
601/// original's exact behavior, including the quirk that `\X` (X ∈ `#?(&,<>\%`) is
602/// **removed entirely** (the original `ESCAPED_OP` regex has no capture group, so
603/// the replacement is the empty string), then `##`→`#` and `&amp;`→`&`.
604pub fn unquote(text: &str) -> String {
605  const ESCAPED: &[char] = &['#', '?', '(', '&', ',', '<', '>', '\\', '%'];
606  let mut out = String::with_capacity(text.len());
607  let mut i = 0usize;
608  while i < text.len() {
609    let rest = &text[i..];
610    let c = rest.chars().next().unwrap();
611    if c == '\\' {
612      if let Some(nc) = rest[1..].chars().next() {
613        if ESCAPED.contains(&nc) {
614          i += 1 + nc.len_utf8(); // drop the whole `\X`
615          continue;
616        }
617      }
618    }
619    out.push(c);
620    i += c.len_utf8();
621  }
622  out.replace("##", "#").replace("&amp;", "&")
623}

625/// Double every backslash so the text can be embedded as a Rust string literal
626/// (Perl `slashify`). Used by the compile-time codegen consumer.
627pub fn slashify(text: &str) -> String { text.replace('\\', "\\\\") }

631/// Execute a parsed replacement against a live `Document` — the runtime consumer
632/// of the AST. Mirrors the Document operations the compile-time codegen emits.
633pub fn apply_ops(
634  ops: &[ReplacementOp],
635  document: &mut Document,
636  args: &[Option<Digested>],
637  props: &SymHashMap<Stored>,
638) -> Result<()> {
639  let mut savenode: Option<Node> = None;
640  exec_ops(ops, document, args, props, &mut savenode)?;
641  if let Some(sn) = savenode {
642    document.set_node(&sn);
643  }
644  Ok(())
645}
646
647fn exec_ops(
648  ops: &[ReplacementOp],
649  document: &mut Document,
650  args: &[Option<Digested>],
651  props: &SymHashMap<Stored>,
652  savenode: &mut Option<Node>,
653) -> Result<()> {
654  for op in ops {
655    match op {
656      ReplacementOp::OpenElement { qname, attrs, float, self_closing } => {
657        if let Some(fk) = float {
658          *savenode = document.float_to_element(qname, matches!(fk, FloatKind::Double))?;
659        }
660        let av = eval_avpairs(attrs, args, props)?;
661        if av.is_empty() {
662          document.open_element(qname, None, None)?;
663        } else {
664          let mut map: HashMap<String, String> = HashMap::default();
665          for (k, v) in av {
666            map.insert(k, v);
667          }
668          let this_font_opt: Option<Cow<Font>> = match props.get("font") {
669            Some(Stored::Font(f)) => Some(Cow::Borrowed(&**f)),
670            Some(Stored::FontDirective(FontDirective::Asset(fa))) => Some(Cow::Borrowed(&**fa)),
671            Some(Stored::FontDirective(FontDirective::Closure(code))) => {
672              Some(Cow::Owned(code(None)?))
673            },
674            _ => None,
675          };
676          if let Some(this_font) = this_font_opt {
677            document.open_element(qname, Some(map), Some(&this_font))?;
678          } else {
679            document.open_element(qname, Some(map), None)?;
680          }
681        }
682        if *self_closing {
683          document.close_element(qname)?;
684        }
685      },
686      ReplacementOp::CloseElement { qname } => {
687        document.close_element(qname)?;
688      },
689      ReplacementOp::ProcessingInstruction { qname, attrs } => {
690        let av = eval_avpairs(attrs, args, props)?;
691        if av.is_empty() {
692          document.insert_pi(qname, None)?;
693        } else {
694          let mut map: HashMap<String, String> = HashMap::default();
695          for (k, v) in av {
696            map.insert(k, v);
697          }
698          document.insert_pi(qname, Some(map))?;
699        }
700      },
701      ReplacementOp::AbsorbValue { value } => {
702        absorb_value(value, document, args, props)?;
703      },
704      ReplacementOp::SetAttribute { key, value, float } => {
705        let val_str = eval_attr_value(value, args, props)?;
706        if *float {
707          *savenode = document.float_to_attribute(key);
708          let mut node = document.get_node().clone();
709          document.set_attribute(&mut node, key, &val_str)?;
710          if let Some(ref sn) = savenode {
711            document.set_node(sn);
712          }
713        } else {
714          let mut node = document.get_node().clone();
715          document.set_attribute(&mut node, key, &val_str)?;
716        }
717      },
718      ReplacementOp::Text { text } => {
719        document.absorb_string(text, props)?;
720      },
721      ReplacementOp::Conditional { test, then_ops, else_ops } => {
722        if eval_bool(test, args, props)? {
723          exec_ops(then_ops, document, args, props, savenode)?;
724        } else {
725          exec_ops(else_ops, document, args, props, savenode)?;
726        }
727      },
728    }
729  }
730  Ok(())
731}

733/// Evaluate attribute pairs to `(key, value)` strings (font key dropped, as the
734/// codegen drops it — the open font comes from `props["font"]` instead).
735fn eval_avpairs(
736  attrs: &[AttrPair],
737  args: &[Option<Digested>],
738  props: &SymHashMap<Stored>,
739) -> Result<Vec<(String, String)>> {
740  let mut out = Vec::new();
741  for a in attrs {
742    match a {
743      AttrPair::KeyValue { key, value } => {
744        if key == "font" {
745          continue;
746        }
747        out.push((key.clone(), eval_attr_value(value, args, props)?));
748      },
749      AttrPair::Conditional { test, then_attrs, else_attrs } => {
750        let branch = if eval_bool(test, args, props)? { then_attrs } else { else_attrs };
751        out.extend(eval_avpairs(branch, args, props)?);
752      },
753    }
754  }
755  Ok(out)
756}
757
758fn eval_attr_value(
759  v: &AttrValue,
760  args: &[Option<Digested>],
761  props: &SymHashMap<Stored>,
762) -> Result<String> {
763  let mut s = String::new();
764  for part in &v.parts {
765    match part {
766      AttrPart::Literal(lit) => s.push_str(lit),
767      AttrPart::Value(val) => s.push_str(&value_to_attribute(val, args, props)?),
768      AttrPart::Conditional { test, then_val, else_val } => {
769        let chosen = if eval_bool(test, args, props)? { then_val } else { else_val };
770        s.push_str(&value_to_attribute(chosen, args, props)?);
771      },
772    }
773  }
774  Ok(s)
775}

777/// Render a value as an attribute string: `to_attribute()` of the resolved
778/// argument/property, or empty when absent (codegen's
779/// `match … { Some(v) => v.to_attribute(), None => String::new() }`).
780fn value_to_attribute(
781  v: &Value,
782  args: &[Option<Digested>],
783  props: &SymHashMap<Stored>,
784) -> Result<String> {
785  Ok(match v {
786    Value::Arg(n) => match args.get(n - 1) {
787      Some(Some(d)) => d.to_attribute(),
788      _ => String::new(),
789    },
790    Value::Prop(name) => match props.get(name) {
791      Some(stored) => stored.to_attribute(),
792      None => String::new(),
793    },
794    Value::Func { name, args: fargs } => call_func(name, fargs, args, props)?,
795    Value::Literal(lit) => lit.clone(),
796  })
797}
798
799/// Absorb a value at content position (codegen's `Into<Option<Digested>>` +
800/// `document.absorb`).
801fn absorb_value(
802  v: &Value,
803  document: &mut Document,
804  args: &[Option<Digested>],
805  props: &SymHashMap<Stored>,
806) -> Result<()> {
807  match v {
808    Value::Arg(n) => {
809      if let Some(Some(d)) = args.get(n - 1) {
810        document.absorb(d, None)?;
811      }
812    },
813    Value::Prop(name) => {
814      if let Some(stored) = props.get(name) {
815        let dig: Option<Digested> = stored.into();
816        if let Some(ref d) = dig {
817          document.absorb(d, None)?;
818        }
819      }
820    },
821    Value::Func { name, args: fargs } => {
822      let s = call_func(name, fargs, args, props)?;
823      if !s.is_empty() {
824        document.absorb_string(&s, props)?;
825      }
826    },
827    Value::Literal(_) => {}, // a literal never reaches content position
828  }
829  Ok(())
830}

832/// The truth test of a conditional (codegen's
833/// `!v.to_string().is_empty() && v.to_string() != "false"`).
834fn eval_bool(
835  v: &Value,
836  args: &[Option<Digested>],
837  props: &SymHashMap<Stored>,
838) -> Result<bool> {
839  Ok(match v {
840    Value::Arg(n) => match args.get(n - 1) {
841      Some(Some(d)) => is_truthy(&d.to_string()),
842      _ => false,
843    },
844    Value::Prop(name) => match props.get(name) {
845      Some(stored) => is_truthy(&stored.to_string()),
846      None => false,
847    },
848    Value::Func { name, args: fargs } => is_truthy(&call_func(name, fargs, args, props)?),
849    Value::Literal(s) => is_truthy(s),
850  })
851}
852
853fn is_truthy(s: &str) -> bool { !s.is_empty() && s != "false" }
854
855/// Resolve a `&func(…)` at runtime through a curated whitelist (the compile-time
856/// path resolves these to Rust calls; the runtime path keeps untrusted scripts
857/// safe by only allowing vetted helpers). Arguments are rendered to strings.
858/// No active binding uses `&func` (they appear only in commented Perl-only
859/// templates), so the whitelist is intentionally minimal and grows on demand.
860fn call_func(
861  name: &str,
862  fargs: &[FuncArg],
863  args: &[Option<Digested>],
864  props: &SymHashMap<Stored>,
865) -> Result<String> {
866  let mut argv: Vec<String> = Vec::with_capacity(fargs.len());
867  for fa in fargs {
868    argv.push(match fa {
869      FuncArg::Value(v) => value_to_attribute(v, args, props)?,
870      FuncArg::Str(s) => eval_attr_value(s, args, props)?,
871    });
872  }
873  match name {
874    "ToString" => Ok(argv.join("")),
875    _ => Err(Error::from(format!(
876      "runtime template: function &{name}(…) is not in the whitelist"
877    ))),
878  }
879}
