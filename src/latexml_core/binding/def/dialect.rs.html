53/// regex for the prefix of a conditional command sequence
54pub static CONDITIONAL_CS_RE: Lazy<Regex> =
55  Lazy::new(|| Regex::new(r"^\\(?:if(.*)|unless)$").unwrap());
56/// regex for the prefix of a protocol (such as literal:)
57pub static LEADING_PROTOCOL_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^\w+:").unwrap());
58/// regex for a trailing slash (trivial, but aids replacement of said slash)
59pub static TRAILING_SLASH_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"/$").unwrap());
60/// regex for one-or-more spaces
61pub static SPACES_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"\s+").unwrap());
62/// regex for ${}^{label}$
63pub static DIRTY_ID_IDIOM_RE: Lazy<Regex> =
64  Lazy::new(|| Regex::new(r"\$\{\}\^\{(?P<label>[^\}]*)\}\$").unwrap());
65/// regex for characters not expected in a usual id attribute
66pub static NON_ID_CHARSET_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"[^\w_\-.]+").unwrap());
67/// regex for a strange noisy TeX `\\~{}`
68pub static TILDE_NOISE_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"\\~\{\}").unwrap());
69/// regex for a TeX argument specifier or any command sequence
70pub static HAS_ARG_OR_CS: Lazy<Regex> = Lazy::new(|| Regex::new(r"#\d|\\.").unwrap());

80/// Token variant of `is_defined`. Defined in the LaTeX-y sense of also not being let to \relax.
81pub fn is_defined_token(cs: &Token) -> bool {
82  match lookup_meaning(cs) {
83    Some(store) => match store {
84      Stored::Token(_) => true,
85      Stored::Expandable(ref m) => m.get_cs_name() != "\\relax",
86      Stored::Primitive(ref m) => m.get_cs_name() != "\\relax",
87      Stored::Constructor(ref m) => m.get_cs_name() != "\\relax",
88      Stored::Register(ref m) => m.get_cs_name() != "\\relax",
89      Stored::Conditional(ref m) => m.get_cs_name() != "\\relax",
90      Stored::MathPrimitive(ref m) => m.get_cs_name() != "\\relax",
91      _ => true, // other stored values are considered defined
92    },
93    _ => false,
94  }
95}

97/// Check if the `token` is not yet defined, or let to `\relax`
98pub fn is_definable(token: &Token) -> bool {
99  // Non-CS / non-ACTIVE tokens (letters, digits, punctuation, etc.) have
100  // a trivial "self" meaning from `lookup_meaning` (= `Stored::Token(self)`).
101  // That's not a real `\def`/`\let` definition — kernel `\@ifdefinable`
102  // would treat them as not-yet-defined. Without this check, calls like
103  // `\@ifdefinable{Z@L@\foo}{…}` (zref-base.sty:118) get our DefToken
104  // reader's first-token-from-brace-group (the letter `Z`) and fail with
105  // "Command \Z already defined" because letter Z's lookup_meaning is
106  // non-None. Witness: arXiv:2504.18121 / 2504.17729 / 2504.17871 et al
107  // (Task #23 zref-base \Z collision cluster).
108  if !token.get_catcode().is_active_or_cs() {
109    return true;
110  }
111  let meaning = lookup_meaning(token);
112  token.with_str(|name| name != "\\relax" && !name.starts_with("\\end"))
113    && (meaning.is_none()
114      || (meaning == lookup_meaning(&TOKEN_RELAX))
115      || lookup_bool("2.09_COMPATIBILITY"))
116}

121/// Define a conditional control sequence.
122///
123/// Its processing takes place in the Gullet.
124/// The test is applied to the arguments (if any),
125/// which determines which branch is executed.
126/// If the test is undefined, the conditional is a "user defined" one;
127/// Two additional primitives are defined \footrue and \foofalse;
128/// the test is then determined by the most recently called of those.
129///
130/// If you supply a skipper instead of a test, it is also applied to the arguments
131/// and should skip to the right place in the following \or, \else, \fi.
132pub fn def_conditional(
133  cs: Token,
134  paramlist: Option<Parameters>,
135  test: Option<ConditionalClosure>,
136  options: ConditionalOptions,
137) -> Result<()> {
138  let locked_key_opt = if let Some(true) = options.locked {
139    Some(arena::with(cs.get_sym(), |cs_name| s!("{cs_name}:locked")))
140  } else {
141    None
142  };
143  if cs.with_str(|cs_name| matches!(cs_name, "\\fi" | "\\else" | "\\or" | "\\unless")) {
144    install_definition(
145      Conditional {
146        cs,
147        paramlist,
148        test,
149        conditional_type: cs.with_str(|cs_name| ConditionalType::from(cs_name)),
150        skipper: options.skipper,
151      },
152      options.scope,
153    )
154  } else {
155    // Perl Package.pm L1210-1219 — match `\<2chars><rest>` and warn (not
156    // error) when prefix is not `if`, but still proceed with user-defined
157    // conditional creation. Bug-for-bug compatible: e.g. `\newif\pgf@lib@svg@relative`
158    // (PGF library naming) creates `\f@lib@svg@relativetrue` etc. with the
159    // `pg` prefix consumed — package code that uses `\if\pgf@lib@svg@relative`
160    // still works because the Let to `\iffalse` runs unconditionally.
161    let (name_opt, warn_misnamed) = cs.with_str(|custom| {
162      // First try strict `\if<name>` / `\unless` — preferred path.
163      if let Some(captures) = CONDITIONAL_CS_RE.captures(custom) {
164        let name = captures.get(1).map_or("", |m| m.as_str()).to_string();
165        return (Some(name), false);
166      }
167      // Perl-loose fallback: `\<2chars><rest>` — capture rest as `name`,
168      // emit a `misdefined` Warn since prefix isn't `if`.
169      let bytes = custom.as_bytes();
170      if bytes.len() > 3 && bytes[0] == b'\\' {
171        let rest = std::str::from_utf8(&bytes[3..]).ok().map(str::to_string);
172        (rest, true)
173      } else {
174        (None, false)
175      }
176    });
177    if warn_misnamed {
178      let message = s!(
179        "The conditional {} is being defined but doesn't start with \\if",
180        cs
181      );
182      Warn!("misdefined", cs, message);
183    }
184    if let Some(name) = name_opt {
185      if !name.is_empty() && name != "case" && test.is_none() {
186        // user-defined conditional, like with \newif
187        // Note: setting up these macros is compile-time expensive, maybe there is some way to
188        // avoid...
189        // Note: the double clones are technically correct Rust if annoying to write and read.
190        //       first, we want to capture a cloned value of cs, to be able to keep using cs here.
191        // second, each invocation of the conditional macro needs to create new tokens to
192        // return,       hence a clone is required on each call.
193        def_macro(
194          T_CS!(s!("\\{}true", name)),
195          None,
196          Tokens!(T_CS!("\\let"), cs, T_CS!("\\iftrue")),
197          None,
198        )?;
199        def_macro(
200          T_CS!(s!("\\{}false", name)),
201          None,
202          Tokens!(T_CS!("\\let"), cs, T_CS!("\\iffalse")),
203          None,
204        )?;
205        let_i(&cs, &T_CS!("\\iffalse"), None);
206      } else {
207        //  For \ifcase, the parameter list better be a single Number !!
208        install_definition(
209          Conditional {
210            cs,
211            paramlist,
212            test,
213            conditional_type: ConditionalType::If,
214            skipper: options.skipper,
215          },
216          options.scope,
217        );
218      }
219    } else {
220      let message = s!(
221        "The conditional {} is being defined but doesn't start with \\if",
222        cs
223      );
224      Error!("misdefined", cs, message);
225    }
226  }
227
228  if let Some(locked_key) = locked_key_opt {
229    assign_value(&locked_key, true, Some(Scope::Global));
230  }
231  Ok(())
232}

234/// Defines the macro expansion for a command sequence.
235///
236/// A macro control sequence that reads parameters
237/// as specified by `paramlist` and is expanded during macro expansion time in the `Gullet`.
238/// See `ExpansionBody` for the possible kinds of `expansion` material.
239pub fn def_macro<T: Into<Option<ExpansionBody>>>(
240  cs: Token,
241  paramlist: Option<Parameters>,
242  expansion: T,
243  options_opt: Option<ExpandableOptions>,
244) -> Result<()> {
245  let expansion_opt: Option<ExpansionBody> = expansion.into();
246  // TODO: The None case could be refactored to feel much cleaner.
247  // For now it's equivalent to Tokens!()
248  let mut options = options_opt.unwrap_or_default();
249  let scope = options.scope.take();
250  if options.mathactive && cs.with_str(|s| s.len()) == 1 {
251    assign_mathcode(
252      cs.with_str(|cstr| cstr.chars().next().unwrap()),
253      0x8000u16,
254      scope,
255    );
256  }
257  let locked_key_opt = if options.locked {
258    Some(format!("{cs}:locked"))
259  } else {
260    None
261  };
262  let defcs = if options.robust {
263    def_robust_cs(cs, options.locked, options.scope)?
264  } else {
265    cs
266  };
267  install_definition(
268    Expandable::new(defcs, paramlist, expansion_opt, Some(options))?,
269    scope,
270  );
271  if let Some(locked_key) = locked_key_opt {
272    assign_value(&locked_key, true, Some(Scope::Global));
273  }
274  Ok(())
275}

292/// Defines a register with an initial value.
293///
294/// (a Number, Dimension, Glue, MuGlue or Tokens --- I haven't handled Box's yet).
295/// Usually, the `prototype` is just the control sequence,
296/// but registers are also handled by prototypes like `\count{Number}`. `DefRegister` arranges
297/// that the register value can be accessed when a numeric, dimension, ... value is being read,
298/// and also defines the control sequence for assignment.
299pub fn def_register<T: Into<RegisterValue>>(
300  cs: Token,
301  parameters: Option<Parameters>,
302  value: T,
303  options: Option<RegisterOptions>,
304) -> Result<()> {
305  let mut options: RegisterOptions = options.unwrap_or_default();
306  let value: RegisterValue = value.into();
307  let has_address_option = options.address.is_some();
308  let mut address = match options.address.take() {
309    Some(v) => v,
310    None => match options.allocate {
311      Some(v) => allocate_register(&v)?.unwrap_or_default(),
312      None => String::new(),
313    },
314  };
315  // by adding this check here, we no longer need to use Register::new in the Rust version
316  if address.is_empty() {
317    address = cs.to_string();
318  }
319  // Assign, but do not RE-assign
320  if !has_address_option || !has_value(&address) {
321    assign_value(&address, value.clone(), Some(Scope::Global));
322  }
323
324  let register_type: RegisterType = (&value).into();
325  install_definition(
326    Register {
327      cs,
328      address,
329      parameters,
330      register_type,
331      readonly: options.readonly,
332      getter: options.getter,
333      setter: options.setter,
334      default: Some(value),
335      value: None,
336      locator: gullet::get_locator(),
337      ..Register::default()
338    },
339    Some(Scope::Global),
340  );
341  Ok(())
342}

344/// Defines a primitive control sequence
345///
346/// A primitive is processed during
347/// digestion (in the  `Stomach`), after macro expansion but before Construction time.
348/// Primitive control sequences generate Boxes or Lists, generally
349/// containing basic Unicode content, rather than structured XML.
350/// Primitive control sequences are also executed for side effect during digestion,
351/// effecting changes to the `State`.
352pub fn def_primitive(
353  cs: Token,
354  paramlist: Option<Parameters>,
355  compiled_replacement: Option<PrimitiveBody>,
356  options: PrimitiveOptions,
357) -> Result<()> {
358  let options_locked = options.locked;
359  let scope = options.scope;
360  let mut before_digest_env: Vec<BeforeDigestClosure> = Vec::new();
361  let cs_name = cs.with_cs_name(ToString::to_string);
362
363  // Perl: mode => 'text' becomes restricted_horizontal + enterHorizontal
364  let mut needs_enter_horizontal = options.enter_horizontal;
365  let mode = if options.mode.as_deref() == Some("text") {
366    needs_enter_horizontal = true;
367    Some("restricted_horizontal".to_string())
368  } else {
369    options.mode
370  };
371
372  if options.require_math {
373    let cs_name_cloned = cs_name.clone();
374    let require_math_closure = before_digest_simple!({ requireMath!(cs_name_cloned) });
375    before_digest_env.push(require_math_closure);
376  }
377
378  if options.forbid_math {
379    let cs_name_cloned = cs_name.clone();
380    let forbid_math_closure = before_digest_simple!({ forbidMath!(cs_name_cloned) });
381    before_digest_env.push(forbid_math_closure);
382  }
383  if needs_enter_horizontal {
384    before_digest_env.push(before_digest_simple!({
385      enter_horizontal();
386    }));
387  }
388  if options.leave_horizontal {
389    before_digest_env.push(before_digest_simple!({
390      leave_horizontal()?;
391    }));
392  }
393  if let Some(ref mode) = mode {
394    let mode_clone = mode.clone();
395    let begin_mode_closure = before_digest_simple!({
396      begin_mode(&mode_clone)?;
397    });
398    before_digest_env.push(begin_mode_closure);
399  } else if options.bounded {
400    let bgroup_closure = before_digest_simple!({
401      bgroup();
402    });
403    before_digest_env.push(bgroup_closure);
404  }
405  match options.font {
406    Some(FontDirective::Asset(chosen_font)) => {
407      // Perf: capture Rc<Font> directly; closure borrows through it.
408      // Previously: `(*chosen_font).clone()` cloned the Font per invocation.
409      let merge_font_closure = before_digest_simple!({
410        merge_font_ref(&chosen_font);
411      });
412      before_digest_env.push(merge_font_closure);
413    },
414    Some(FontDirective::Closure(font_closure)) => {
415      let execute_font_closure = before_digest_simple!({
416        merge_font(font_closure(None)?);
417      });
418      before_digest_env.push(execute_font_closure);
419    },
420    None => {},
421  }
422  before_digest_env.extend(options.before_digest);
423
424  let mut after_digest_env: Vec<DigestionClosure> = options.after_digest;
425  if let Some(ref mode_str) = mode {
426    let mode_clone = mode_str.clone();
427    let end_mode_closure: DigestionClosure = after_digest_simple!(_whatsit, {
428      end_mode(&mode_clone)?;
429    });
430    after_digest_env.push(end_mode_closure);
431  } else if options.bounded {
432    let egroup_closure: DigestionClosure = after_digest_simple!(_whatsit, {
433      egroup()?;
434    });
435    after_digest_env.push(egroup_closure);
436  }
437  //  Not sure robust entirely makes sense for Primitives, other than LaTeXML vs LaTeX mismatch
438  let defcs = if options.robust {
439    def_robust_cs(cs, options.locked, scope)?
440  } else {
441    cs
442  };
443
444  install_definition(
445    Primitive {
446      cs: defcs,
447      paramlist,
448      replacement: compiled_replacement,
449      before_digest: before_digest_env,
450      after_digest: after_digest_env,
451      alias: options.alias,
452      nargs: options.nargs,
453      is_prefix: options.is_prefix,
454      reversion: options.reversion,
455      font_id: options.font_id,
456    },
457    scope,
458  );
459  if options_locked {
460    assign_value(&s!("{}:locked", cs_name), true, Some(Scope::Global));
461  }
462  Ok(())
463}

465/// Advanced math replacements require a XMDual representation
466pub fn def_math_dual(
467  cs: Token,
468  paramlist: Option<Parameters>,
469  presentation: String,
470  options: MathPrimitiveOptions,
471) -> Result<()> {
472  let (cont_cs_str, pres_cs_str) =
473    cs.with_str(|csname| (s!("{csname}@content"), s!("{csname}@presentation")));
474  let cont_cs = T_CS!(cont_cs_str);
475  let pres_cs = T_CS!(pres_cs_str);
476  let defcs = if options.robust {
477    def_robust_cs(cs, options.locked, options.scope)?
478  } else {
479    cs
480  };
481  let presentation_toks = mouth::tokenize_internal(&presentation);
482
483  // Make the original CS expand into a DUAL invoking a presentation macro and content constructor
484  let captured_role = options.role.clone();
485  let captured_revert_as = options.revert_as.clone();
486  let captured_cont_cs = cont_cs;
487  let captured_pres_cs = pres_cs;
488  let captured_pres = presentation.clone();
489  install_definition(
490    Expandable::new(
491      defcs,
492      paramlist.clone(),
493      Some(ExpansionBody::Closure(Rc::new(move |args| {
494        let args_opt_tks = args
495          .into_iter()
496          .map(|arg| arg.into())
497          .collect::<Vec<Option<Tokens>>>();
498        let (cargs, pargs) = dualize_arglist(&captured_pres, args_opt_tks)?;
499
500        let mut dtks = vec![T_CS!("\\lx@dual")];
501        // optional keyval arg
502        if captured_role.is_some() || captured_revert_as.is_some() {
503          dtks.push(T_OTHER!("["));
504          if let Some(ref role) = captured_role {
505            dtks.extend(vec![T_OTHER!("role"), T_OTHER!("="), T_OTHER!(role)]);
506            if let Some(ref _revert_as) = captured_revert_as {
507              dtks.push(T_OTHER!(","));
508            }
509          }
510          if let Some(ref revert_as) = captured_revert_as {
511            dtks.extend(vec![
512              T_OTHER!("revert_as"),
513              T_OTHER!("="),
514              T_OTHER!(revert_as),
515            ]);
516          }
517          dtks.push(T_OTHER!("]"));
518        }
519        // end optional keyval arg
520        // Perl: Invocation($content_cs, @content_args) wraps each arg in braces.
521        // If no args (no params), just emit the CS without braces.
522        dtks.push(T_BEGIN!());
523        dtks.push(captured_cont_cs);
524        for carg in cargs.into_iter().flatten() {
525          dtks.push(T_BEGIN!());
526          dtks.extend(carg.unlist());
527          dtks.push(T_END!());
528        }
529        dtks.push(T_END!());
530        dtks.push(T_BEGIN!());
531        dtks.push(captured_pres_cs);
532        for parg in pargs.into_iter().flatten() {
533          dtks.push(T_BEGIN!());
534          dtks.extend(parg.unlist());
535          dtks.push(T_END!());
536        }
537        dtks.push(T_END!());
538
539        Ok(Tokens::new(dtks))
540      }))),
541      Some(ExpandableOptions {
542        protected: options.protected,
543        ..ExpandableOptions::default()
544      }),
545    )?,
546    options.scope,
547  );
548
549  // Make the presentation macro.
550  install_definition(
551    Expandable::new(
552      pres_cs,
553      paramlist.clone(),
554      Some(ExpansionBody::Tokens(presentation_toks)),
555      Some(ExpandableOptions {
556        protected: options.protected,
557        ..ExpandableOptions::default()
558      }),
559    )?,
560    options.scope,
561  );
562
563  // content: Make the content constructor
564  // content: build the replacement closure
565  let nargs = paramlist
566    .as_ref()
567    .map(|pl| pl.get_parameters().len())
568    .unwrap_or(0);
569  let content_closure: ReplacementClosure = if nargs == 0 {
570    Rc::new(|document, _args, props| {
571      let mut attrs = HashMap::default();
572      for key in ["role", "scriptpos", "stretchy"] {
573        if let Some(v) = props.get(key) {
574          attrs.insert(key.to_owned(), v.to_string());
575        }
576      }
577      for key in MATH_CONSTRUCTOR_ATTRIBUTES {
578        if let Some(v) = props.get(key) {
579          attrs.insert(key.to_string(), v.to_string());
580        }
581      }
582      document.insert_element("ltx:XMTok", Vec::new(), Some(attrs))?;
583      Ok(())
584    })
585  } else {
586    Rc::new(|document, args, props| {
587      let mut app_attrs = HashMap::default();
588      for key in ["role", "scriptpos"] {
589        if let Some(v) = props.get(key) {
590          app_attrs.insert(key.to_owned(), v.to_string());
591        }
592      }
593      document.open_element("ltx:XMApp", Some(app_attrs), None)?;
594      let mut op_attrs = HashMap::default();
595      if let Some(v) = props.get("operator_stretchy") {
596        op_attrs.insert("stretchy".to_owned(), v.to_string());
597      }
598      if let Some(v) = props.get("operator_role") {
599        op_attrs.insert("role".to_owned(), v.to_string());
600      }
601      if let Some(v) = props.get("operator_scriptpos") {
602        op_attrs.insert("scriptpos".to_owned(), v.to_string());
603      }
604      for key in MATH_CONSTRUCTOR_ATTRIBUTES {
605        if let Some(v) = props.get(key) {
606          op_attrs.insert(key.to_string(), v.to_string());
607        }
608      }
609      // operator
610      document.insert_element("ltx:XMTok", Vec::new(), Some(op_attrs))?;
611      // arguments
612      // TODO: options.reorder?
613      for arg in args.iter().flatten() {
614        document.absorb(arg, None)?;
615      }
616      document.close_element("ltx:XMApp")?;
617      Ok(())
618    })
619  };
620  // content: install the constructor
621  let mut content_constructor = Constructor {
622    cs: cont_cs,

626  };
627  let scope = options.scope;
628  transfer_common_constructor_options(&cs, &presentation, options, &mut content_constructor);
629  install_definition(content_constructor, scope);
630  Ok(())
631}

720/// `Constructor` definition
721pub fn def_math_constructor(
722  cs: Token,
723  paramlist: Option<Parameters>,
724  presentation: String,
725  mut options: MathPrimitiveOptions,
726) -> Result<()> {
727  // TODO: do we need to do anything about digesting the presentation?
728  let nargs = paramlist
729    .as_ref()
730    .map(|pl| pl.get_parameters().len())
731    .unwrap_or(0);
732  // let csname_alias = if options.alias.is_none() && options.robust {
733  //   Some(String::from(cs.get_cs_name()))
734  // } else {
735  //   None
736  // };
737  let defcs = if options.robust {
738    def_robust_cs(cs, options.locked, options.scope)?
739  } else {
740    cs
741  };

743    if options.revert_as.is_none()
744      || options.revert_as == Some(Cow::Borrowed("content"))
745      || options.revert_as == Some(Cow::Borrowed("context"))
746    {
747      // TODO :&& (($LaTeXML::DUAL_BRANCH || 'content') eq 'content'))
748      options.reversion = Some(Reversion::Tokens(Tokens!(cs)));
749    } else {
750      // TODO: This differs from the Perl, where `presentation` comes in as Tokens
751      //       we have it come in as a `String`,
752      //       so need to tokenize when reusing it as a reversion.
753      options.reversion = Some(Reversion::Tokens(Tokens::new(Explode!(presentation))));
754    }
755  }
756  let presentation_for_sizer = presentation.clone();
757  let presentation_for_replacement = presentation.clone();
758  let is_mathstyle = options.mathstyle.is_some();
759  let mathstyle_for_font = options.mathstyle.clone();
760  let presentation_for_font = presentation.clone();
761  options.font = Some(FontDirective::Closure(if is_mathstyle {
762    Rc::new(move |_whatsit| {
763      Ok(
764        lookup_font()
765          .unwrap()
766          .merge(Font {
767            mathstyle: mathstyle_for_font
768              .as_ref()
769              .map(|ms| Cow::Owned(ms.to_owned())),
770            ..Font::default()
771          })
772          .specialize(&presentation_for_font),
773      )
774    })
775  } else {
776    Rc::new(move |_whatsit| Ok(lookup_font().unwrap().specialize(&presentation_for_font)))
777  }));
778  let compiled_replacement: Option<ReplacementClosure> = Some(if nargs == 0 {
779    // Perl defmath_cons (Package.pm L1841-1844):
780    //   $nargs == 0
781    //     && $presentation !~ /(?:\(|\)|\\)/
782    //   ? "?#isMath(<ltx:XMTok …$end_tok)($qpresentation)"
783    //   : "<ltx:XMTok …$end_tok"
784    //
785    // The `?#isMath(…)(…)` form is Perl's construction-time conditional —
786    // in math context emit the XMTok, otherwise emit the bare presentation
787    // character. The check is gated on presentation NOT containing `(`,
788    // `)`, or `\`, which would collide with template-specials.
789    //
790    // Ported here as a runtime DOM-ancestry walk (same predicate as
791    // `tbox.rs::be_absorbed`), and the presentation-content guard mirrors
792    // Perl's regex. `\rightarrowfill` (`\x{2192}`, no specials) gets the
793    // text fallback; symbols like `\(` / `\)` / anything with a backslash
794    // keep the unconditional XMTok — matching Perl.
795    let presentation_is_trivial = !presentation_for_replacement
796      .chars()
797      .any(|c| c == '(' || c == ')' || c == '\\');
798    Rc::new(
799      move |document: &mut Document, _, props: &SymHashMap<Stored>| {
800        let font_opt = match props.get("font") {
801          Some(Stored::Font(f)) => Some(Cow::Borrowed(&**f)),
802          Some(Stored::FontDirective(FontDirective::Closure(code))) => {
803            Some(Cow::Owned(code(None)?))
804          },
805          Some(Stored::FontDirective(FontDirective::Asset(font))) => Some(Cow::Borrowed(&**font)),
806          _ => None,
807        };
808        // Perl's `?#isMath(…)` conditional compiles to `ToString($prop{'isMath'})`
809        // (Constructor/Compiler.pm parse_conditional L164-173 + L197: `#prop` →
810        // `$prop{'prop'}`). That property is set on the Whatsit at digestion
811        // time from the stomach's math-mode flag (Constructor.pm L108). So the
812        // check here is on the *Whatsit's* isMath, not document-ancestry — an
813        // `\hbox{\rightarrowfill}` inside `\mathop{…}` digests the XMTok with
814        // isMath=false (hbox switched stomach to text mode), even though the
815        // document insertion point is nested under <ltx:Math>.
816        let is_math = matches!(props.get("isMath"), Some(Stored::Bool(true)));
817        if !is_math && presentation_is_trivial {
818          // Perl `?#isMath(…)(plain)` text branch — just emit the char.
819          document.absorb_string(&presentation_for_replacement, props)?;
820          return Ok(());
821        }
822        let mut attrs = HashMap::default();
823        for key in ["role", "scriptpos", "stretchy"] {
824          if let Some(v) = props.get(key) {
825            attrs.insert(key.to_owned(), v.to_string());
826          }
827        }
828        for key in MATH_CONSTRUCTOR_ATTRIBUTES {
829          if let Some(v) = props.get(key) {
830            attrs.insert(key.to_string(), v.to_string());
831          }
832        }
833        if let Some(font) = font_opt {
834          document.open_element("ltx:XMTok", Some(attrs), Some(&font))?;
835        } else {
836          document.open_element("ltx:XMTok", Some(attrs), None)?;
837        }
838        document.absorb_string(&presentation_for_replacement, props)?;
839        document.close_element("ltx:XMTok")?;
840        Ok(())
841      },
842    )
843  } else {
844    // Perl defmath_cons (Package.pm L1847-1851): when `$nargs` > 0 the
845    // template is always `<ltx:XMApp>…<ltx:XMTok …/></ltx:XMApp>` — there
846    // is NO text-mode fallback for this arm. The `requireMath` beforeDigest
847    // (Perl L1689, same as Rust L1606) has already warned the user; math
848    // context is assumed. Keep parity — emit the XMApp/XMTok structure as
849    // before.
850    Rc::new(
851      move |document: &mut Document, args: &Vec<Option<Digested>>, props: &SymHashMap<Stored>| {
852        let mut attrs = HashMap::default();
853        for key in ["role", "scriptpos", "stretchy"] {
854          if let Some(v) = props.get(key) {
855            attrs.insert(key.to_owned(), v.to_string());
856          }
857        }
858        let font_opt = match props.get("font") {
859          Some(Stored::Font(f)) => Some(Cow::Borrowed(&**f)),
860          Some(Stored::FontDirective(FontDirective::Closure(code))) => {
861            Some(Cow::Owned(code(None)?))
862          },
863          Some(Stored::FontDirective(FontDirective::Asset(font))) => Some(Cow::Borrowed(&**font)),
864          _ => None,
865        };
866        if let Some(ref font) = font_opt {
867          document.open_element("ltx:XMApp", Some(attrs), Some(font))?;
868        } else {
869          document.open_element("ltx:XMApp", Some(attrs), None)?;
870        }
871        // operator
872        let mut op_attrs = HashMap::default();
873        if let Some(role) = props.get("operator_role") {
874          op_attrs.insert(String::from("role"), role.to_string());
875        }
876        if let Some(stretchy) = props.get("operator_stretchy") {
877          op_attrs.insert(String::from("stretchy"), stretchy.to_string());
878        }
879        if let Some(scriptpos) = props.get("operator_scriptpos") {
880          op_attrs.insert(String::from("scriptpos"), scriptpos.to_string());
881        }
882        for key in MATH_CONSTRUCTOR_ATTRIBUTES {
883          if let Some(v) = props.get(key) {
884            op_attrs.insert(key.to_string(), v.to_string());
885          }
886        }
887        if let Some(font) = font_opt {
888          document.open_element("ltx:XMTok", Some(op_attrs), Some(&font))?;
889        } else {
890          document.open_element("ltx:XMTok", Some(op_attrs), None)?;
891        }
892        document.absorb_string(&presentation_for_replacement, props)?;
893        document.close_element("ltx:XMTok")?;
894        // arguments
895        for arg in args {
896          document.open_element("ltx:XMArg", None, None)?;
897          if let Some(arg_v) = arg {
898            document.absorb(arg_v, None)?;
899          }
900          document.close_element("ltx:XMArg")?;
901        }
902
903        document.close_element("ltx:XMApp")?;
904        Ok(())
905      },
906    )
907  });
908  let sizer: Option<SizingClosure> = Some(Rc::new(move |_| {
909    Ok(Font::math_default().compute_string_size(&presentation_for_sizer, SymHashMap::default()))
910  }));
911
912  // let mut prop_options = options.clone();
913  let mut constructor = Constructor {
914    cs: defcs,
915    paramlist,
916    replacement: compiled_replacement,
917    nargs: Some(nargs),
918    sizer,
919    // capture_body: options.capture_body,
920    // outer
921    // long
922    ..Constructor::default()
923  };
924  let scope = options.scope;
925  transfer_common_constructor_options(&cs, &presentation, options, &mut constructor);
926  install_definition(constructor, scope);
927  Ok(())
928}

965/// absorption by the `Document`, the `Whatsit` will generate the XML fragment according

969  paramlist: Option<Parameters>,
970  compiled_replacement: Option<ReplacementClosure>,
971  options: ConstructorOptions,
972) {
973  // TODO: This won't work, as we can only invoke method calls on paramlist in runtime
974  //*latexml_codegen::constructable::NARGS = $paramlist.get_num_args();

976  let cs = if options.robust {
977    def_robust_cs(cs, options.locked, scope).expect("def_robust_cs for constructor failed")
978  } else {
979    cs
980  };

982  let locked_key_opt = if options.locked {
983    Some(s!("{cs_name}:locked"))
984  } else {
985    None
986  };
987
988  let mut before_digest_closures: Vec<BeforeDigestClosure> = Vec::new();
989
990  // Perl: mode => 'text' becomes restricted_horizontal + enterHorizontal
991  let mut needs_enter_horizontal = options.enter_horizontal;
992  let mode = if options.mode.as_deref() == Some("text") {
993    needs_enter_horizontal = true;
994    Some("restricted_horizontal".to_string())
995  } else {
996    options.mode
997  };
998
999  if options.require_math {
1000    let cs_name_cloned = cs_name.clone();
1001    let require_math_closure = before_digest_simple!({ requireMath!(cs_name_cloned) });

1011      enter_horizontal();
1012    }));
1013  }
1014  if options.leave_horizontal {
1015    before_digest_closures.push(before_digest_simple!({
1016      leave_horizontal()?;
1017    }));

1023    });

1028    });

1057    });

1061      egroup()?;
1062    });
1063    after_digest_closures.push(egroup_closure);

1091/// Defines an Environment that generates a specific XML fragment.
1092///
1093/// `compiled_replacement` is of the same form as for DefConstructor, but will generally include
1094/// reference to the `#body` property.
1095/// Upon encountering a `\begin{env}`:  the mode is switched, if needed, else a new group is opened;
1096/// then the environment name is noted; the beforeDigest hook is run.
1097/// Then the Whatsit representing the begin command (but ultimately the whole environment) is
1098/// created and the `after_digest_begin` hook is run.
1099/// Next, the body will be digested and collected until the balancing `\end{env}`.
1100/// Then, any `after_digest` hook is run, the environment is ended, finally the mode is ended or the
1101/// group is closed.  The body and `\end{env}` whatsit are added to the `\begin{env}`'s whatsit as
1102/// body and trailer, respectively.
1103pub fn def_environment(
1104  name: String,
1105  paramlist: Option<Parameters>,
1106  compiled_replacement: Option<ReplacementClosure>,
1107  options: ConstructorOptions,
1108) {
1109  // This is for the common case where the environment is opened by \begin{env}
1110  let begin_name = s!("\\begin{{{name}}}");
1111  let end_name = s!("\\end{{{name}}}");
1112  let mut before_digest_env: Vec<BeforeDigestClosure> = Vec::new();
1113
1114  // Perl Package.pm line 1885: $mode = 'restricted_horizontal' if !$mode || ($mode eq 'text');
1115  // Environments ALWAYS have a mode — defaults to restricted_horizontal.
1116  // This means \end{env} always calls endMode(), never egroup().
1117  let mode = match options.mode.as_deref() {
1118    None | Some("text") => Some("restricted_horizontal".to_string()),
1119    _ => options.mode,
1120  };
1121
1122  if options.require_math {
1123    let require_name = begin_name.clone();
1124    let require_math_closure = before_digest_simple!({ requireMath!(require_name) });
1125    before_digest_env.push(require_math_closure);
1126  }
1127  if options.forbid_math {
1128    let forbid_name = begin_name.clone();
1129    let forbid_math_closure = before_digest_simple!({ forbidMath!(forbid_name) });
1130    before_digest_env.push(forbid_math_closure);
1131  }
1132  let bgroup_closure = before_digest_simple!({
1133    bgroup();
1134  });
1135  before_digest_env.push(bgroup_closure);
1136  let atbegin_key = s!("@environment@{name}@atbegin");
1137  let atbegin_hook_closure = before_digest_simple!({
1138    if let Some(b) = lookup_tokens(&atbegin_key) {
1139      vec![digest(b.unlist())?]
1140    } else {
1141      Vec::new()
1142    }
1143  });
1144
1145  before_digest_env.push(atbegin_hook_closure);
1146  if options.enter_horizontal {
1147    before_digest_env.push(before_digest_simple!({
1148      enter_horizontal();
1149    }));
1150  }
1151  if options.leave_horizontal {
1152    before_digest_env.push(before_digest_simple!({
1153      leave_horizontal()?;
1154    }));
1155  }
1156  // Perl Package.pm line 1908: beginMode($mode, 1) — noframe=1 since bgroup already pushed
1157  if let Some(ref mode) = mode {
1158    let bmode = mode.clone();
1159    let mode_closure = before_digest_simple!({
1160      begin_mode_opt(&bmode, true)?;
1161    });
1162    before_digest_env.push(mode_closure);
1163  }
1164
1165  let env_name = name.clone();
1166  let current_environment_closure = before_digest_simple!({
1167    assign_value_sym(crate::pin!("current_environment"), env_name.clone(), None);
1168    let body = T_LETTER!(env_name.clone());
1169    def_macro(
1170      T_CS!("\\@currenvir"),
1171      None,
1172      Some(ExpansionBody::Tokens(Tokens!(body))),
1173      None,
1174    )?;
1175  });
1176  before_digest_env.push(current_environment_closure);
1177
1178  match options.font {
1179    Some(FontDirective::Asset(chosen_font)) => {
1180      // Perf: capture Rc<Font> directly; closure borrows through it.
1181      // Previously: `(*chosen_font).clone()` cloned the Font per invocation.

1184      });

1186    },

1204  before_digest_env.extend(options.before_digest);

1207  // before they are moved into the \begin{name} constructor below.
1208  let bare_after_digest_begin = options.after_digest_begin.clone();
1209  let bare_after_digest_body = options.after_digest_body.clone();
1210  let bare_before_construct = options.before_construct.clone();
1211  let bare_after_construct = options.after_construct.clone();
1212  let bare_sizer = options.sizer.clone();
1213  let bare_reversion = options.reversion.clone();
1214  let bare_alias = options.alias.clone();
1215

1217    push_frame();
1218    Ok(())
1219  });
1220  let mut before_construct_with_frame: Vec<ConstructionClosure> = vec![push_frame_closure];
1221  before_construct_with_frame.extend(options.before_construct);
1222
1223  let mut after_construct_with_frame: Vec<ConstructionClosure> = options.after_construct;
1224
1225  let pop_frame_closure = Rc::new(|_document: &mut Document, _whatsit: &Whatsit| {
1226    pop_frame()?;
1227    Ok(())
1228  });
1229  after_construct_with_frame.push(pop_frame_closure);
1230
1231  // Perl Package.pm L1891-1895: "in pure LaTeX would usually have expanded to \env
1232  // and would have skipped spaces before parsing args, if any."
1233  // Prepend SkipSpaces parameter when the environment has arguments.
1234  let paramlist_skips = match paramlist {
1235    Some(ref pl) if pl.get_num_args() > 0 => {
1236      let skip_spaces_param = Parameter {
1237        novalue: true,
1238        name: arena::pin_static("SkipSpaces"),
1239        spec: arena::pin_static("SkipSpaces"),
1240        reader: Rc::new(|_inner, _extra| {
1241          gullet::skip_spaces()?;
1242          Ok(ArgWrap::None)
1243        }),
1244        ..Parameter::default()
1245      };
1246      let mut params = vec![skip_spaces_param];
1247      params.extend(pl.get_parameters().into_iter().cloned());
1248      Some(Parameters::new(params))
1249    },
1250    _ => paramlist.clone(),
1251  };

1254    cs:                T_CS!(begin_name),
1255    paramlist:         paramlist_skips,
1256    replacement:       compiled_replacement.clone(),
1257    nargs:             options.nargs,
1258    before_digest:     before_digest_env,
1259    after_digest:      options.after_digest_begin,
1260    after_digest_body: options.after_digest_body,
1261    before_construct:  before_construct_with_frame,
1262    // Curiously, it's the \begin whose afterConstruct gets called.

1265    properties:        options.properties.clone(),

1268    // ), $options{scope});

1271    alias:             options.alias,

1277  let end_name_clone = end_name.clone();

1280    if env.is_empty() || name_clone != env {
1281      let message1 = s!("Can't close environment {}", name_clone);
1282      let message2 = s!(
1283        "Current are {} ",
1284        with_stacked_values_sym(crate::pin!("current_environment"), |vals| vals
1285          .iter()
1286          .map(|x| s!("{:?}", x))
1287          .collect::<Vec<String>>()
1288          .join(", "))
1289      );
1290      Error!("unexpected", end_name_clone, message1, message2);
1291    }
1292    Ok(Vec::new())
1293  });
1294  after_digest_env.push(unexpected_end_closure);

1297    Some(ref emode) => {
1298      let emode = emode.clone();
1299      let emode_closure = Rc::new(move |_whatsit: &mut Whatsit| {
1300        // Perl Package.pm L1944-1945:
1301        //   # Switch mode (w/stack frame pop), OR egroup
1302        //   ($mode ? (sub { $_[0]->endMode($mode) }) : sub { $_[0]->egroup; }),
1303        // endMode(mode) with no second arg defaults to noframe=0 — it DOES
1304        // pop a frame. This pairs with L1902's `bgroup` + L1908's
1305        // `beginMode($mode, 1)` (noframe=1, no push) on the begin side:
1306        // bgroup pushes exactly one frame, beginMode writes MODE/BOUND_MODE
1307        // Local into that frame, endMode pops the frame and reverts.

1309        Ok(Vec::new())
1310      });
1311      after_digest_env.push(emode_closure);
1312    },
1313    None => {
1314      let egroup_closure = Rc::new(|_whatsit: &mut Whatsit| {
1315        egroup()?;
1316        Ok(Vec::new())
1317      });
1318      after_digest_env.push(egroup_closure);
1319    },
1320  };
1321
1322  let (mut before_digest_for_endenv, before_digest_end_clone) = {
1323    let cloned = options.before_digest_end.clone();
1324    (options.before_digest_end, cloned)
1325  };
1326  let atend_key = s!("@environment@{name}@atend");
1327  let atend_hook_closure = before_digest_simple!({
1328    if let Some(e) = lookup_tokens(&atend_key) {
1329      vec![digest(e.unlist())?]
1330    } else {

1332    }
1333  });

1335
1336  let end_envname_constructor = Rc::new(Constructor {
1337    cs: T_CS!(end_name),
1338    replacement: None,
1339    paramlist: None,
1340    before_digest: before_digest_for_endenv,
1341    after_digest: after_digest_env,
1342    ..Constructor::default() // TODO ? fill in missing ones
1343  });

1347  // Perl Package.pm lines 1949-1969: \FOO gets the same hook pipeline as \begin{FOO}
1348  let mut before_digest_bare: Vec<BeforeDigestClosure> = Vec::new();
1349  before_digest_bare.push(before_digest_simple!({
1350    bgroup();
1351  }));
1352  if options.enter_horizontal {
1353    before_digest_bare.push(before_digest_simple!({
1354      enter_horizontal();
1355    }));
1356  }
1357  if options.leave_horizontal {
1358    before_digest_bare.push(before_digest_simple!({
1359      leave_horizontal()?;
1360    }));
1361  }
1362  if let Some(ref bmode) = mode {
1363    let bmode = bmode.clone();
1364    before_digest_bare.push(before_digest_simple!({
1365      begin_mode_opt(&bmode, true)?;
1366    }));
1367  }
1368  // Perl Package.pm L1949-1969: bare `\name` runs the same user beforeDigest
1369  // hooks as `\begin{name}` (e.g. beforeFloat for `{figure}`). Order matters:
1370  // bgroup + mode have already been pushed; the user hooks come last, mirroring

1372  before_digest_bare.extend(bare_before_digest);
1373  let push_frame_bare = Rc::new(|_document: &mut Document, _whatsit: &Whatsit| {
1374    push_frame();
1375    Ok(())
1376  });
1377  let pop_frame_bare = Rc::new(|_document: &mut Document, _whatsit: &Whatsit| {
1378    pop_frame()?;
1379    Ok(())
1380  });
1381  // Perl: \name gets the same afterDigest, afterDigestBody, beforeConstruct, afterConstruct,
1382  // sizer, reversion, alias as \begin{name}
1383  let mut before_construct_bare: Vec<ConstructionClosure> = vec![push_frame_bare];
1384  before_construct_bare.extend(bare_before_construct);
1385  let mut after_construct_bare: Vec<ConstructionClosure> = bare_after_construct;
1386  after_construct_bare.push(pop_frame_bare);
1387  let name_constructor = Rc::new(Constructor {
1388    cs: T_CS!(s!("\\{}", &name)),
1389    paramlist,
1390    replacement: compiled_replacement,
1391    nargs: options.nargs,
1392    capture_body: true,
1393    properties: options.properties.clone(),
1394    before_digest: before_digest_bare,
1395    after_digest: bare_after_digest_begin,
1396    after_digest_body: bare_after_digest_body,
1397    before_construct: before_construct_bare,
1398    after_construct: after_construct_bare,
1399    sizer: infer_sizer(bare_sizer.as_ref(), bare_reversion.as_ref()),
1400    reversion: bare_reversion,
1401    alias: bare_alias,
1402  });
1403  install_definition(name_constructor, options.scope);
1404  let end_name = s!("\\end{}", &name);
1405  let mut after_digest_end = options.after_digest;
1406  // Perl Package.pm lines 1970-1975: \endFOO calls endMode if mode was specified
1407  match mode {
1408    Some(ref emode) => {
1409      let emode = emode.clone();
1410      after_digest_end.push(Rc::new(move |_whatsit: &mut Whatsit| {
1411        end_mode(&emode)?;
1412        Ok(Vec::new())
1413      }));
1414    },
1415    None => {
1416      // No mode specified — no egroup needed for this simplified closer.
1417      // (The \end{FOO} constructor already handles egroup.)
1418    },
1419  };
1420
1421  // Perl Package.pm lines 1970-1975: \endFOO has beforeDigestEnd + afterDigest + endMode
1422  let end_name_constructor = Constructor {
1423    cs: T_CS!(end_name),
1424    paramlist: None,
1425    replacement: None,
1426    before_digest: before_digest_end_clone,
1427    after_digest: after_digest_end,
1428    ..Constructor::default()
1429  };
1430  install_definition(Rc::new(end_name_constructor), options.scope);
1431
1432  if options.locked {
1433    assign_value(
1434      &s!("\\begin{{{}}}:locked", &name),
1435      true,
1436      Some(Scope::Global),
1437    );
1438    assign_value(&s!("\\end{{{}}}:locked", &name), true, Some(Scope::Global));
1439    assign_value(&s!("\\{}:locked", &name), true, Some(Scope::Global));
1440    assign_value(&s!("\\end{}:locked", &name), true, Some(Scope::Global));
1441  }
1442}

1472/// Flesh out two dual (mathematical) forms of a given list of arguments.
1473///
1474/// Given a list of Tokens (to be expanded into mathematical objects)
1475/// return two lists
1476///   (1) The Tokens' wrapped in an XMAarg, with an ID added
1477///   (2) a corresponding list of Tokens creating XMRef's to those IDs
1478///
1479/// Ah, but there are complications!!!
1480/// On the one hand, arguments may be hidden, never appearing on the presentation side
1481/// (all will be passed to the content side); This argues for putting the XMArg's on the content
1482/// side. OTOH, they ought to be on the presentation side, so that they can be expanded & digested
1483/// in the proper context they will be presented, and pick up all the styling (font size,
1484/// displaystyle..) I don't know how to work around the latter, so we'll put args on the
1485/// presentation side, UNLESS they are hidden, in which case they'll be on the content side.
1486/// So, how do we know if they're hidden? We'll scan the presentation for #\d, that's how!
1487pub fn dualize_arglist(
1488  presentation: &str,
1489  args: Vec<Option<Tokens>>,
1490) -> Result<(ArgsUnpacked, ArgsUnpacked)> {
1491  let mut used = HashMap::default();
1492  for cap in ARG_HOLE.captures_iter(presentation) {
1493    // Get the args that were actually used!
1494    let argi = cap.get(1).unwrap().as_str();
1495    let entry = used.entry(argi.parse::<usize>().expect(argi)).or_insert(0);
1496    *entry += 1;
1497  }
1498  let mut cargs = Vec::new();
1499  let mut pargs = Vec::new();
1500  for (index, arg_opt) in args.into_iter().enumerate() {
1501    match arg_opt {
1502      None => {
1503        pargs.push(None);
1504        cargs.push(None);
1505      },
1506      Some(arg) if arg.unlist_ref().is_empty() => {
1507        pargs.push(Some(arg.clone()));
1508        cargs.push(Some(arg));
1509      },
1510      Some(arg_toks) => {
1511        if used.get(&(1 + index)).unwrap_or(&0) > &0 {
1512          // used in presentation?
1513          let id = get_xmarg_id()?;
1514          pargs.push(Some(Tokens!(
1515            T_CS!("\\lx@xmarg"),
1516            T_BEGIN!(),
1517            id.clone().unlist(),
1518            T_END!(),
1519            T_BEGIN!(),
1520            arg_toks.unlist(),
1521            T_END!()
1522          ))); // put XMArg in presentation
1523          cargs.push(Some(Tokens!(
1524            T_CS!("\\lx@xmref"),
1525            T_BEGIN!(),
1526            id.unlist(),
1527            T_END!()
1528          )));
1529        } else {
1530          // Hidden arg, put XMArg in content.
1531          let id = get_xmarg_id()?;
1532          cargs.push(Some(Tokens!(
1533            T_CS!("\\lx@xmarg"),
1534            T_BEGIN!(),
1535            id.clone().unlist(),
1536            T_END!(),
1537            T_BEGIN!(),
1538            arg_toks.unlist(),
1539            T_END!()
1540          )));
1541          pargs.push(Some(Tokens!(
1542            T_CS!("\\lx@xmref"),
1543            T_BEGIN!(),
1544            id.unlist(),
1545            T_END!()
1546          )));
1547        }
1548      },
1549    }
1550  }
1551  Ok((cargs, pargs))
1552}

1554/// Define a Mathematical symbol or function.
1555///
1556/// There are two sets of cases:
1557///  (1) If the presentation appears to be TeX code, we create an XMDual,
1558/// since the presentation may end up with structure, etc.
1559///  (2) But if the presentation is a simple string, or unicode,
1560/// it is just the content of the symbol; even if the function takes arguments.
1561// ALSO
1562//  arrange that the operator token gets cs="$cs"
1563// ALSO
1564//  Possibly some trick with SUMOP/INTOP affecting limits ?
1565//  Well, not exactly, but....
1566// HMM.... Still fishy.
1567// When to make a dual ?
1568// If the $presentation seems to be TeX (ie. it involves #1... but not ONLY!)
1569pub fn def_math(
1570  cs: Token,
1571  paramlist: Option<Parameters>,
1572  presentation: String,
1573  mut options: MathPrimitiveOptions,
1574) -> Result<()> {
1575  // Can't defer parsing parameters since we need to know number of args!
1576  // $paramlist = parseParameters($paramlist, $cs) if defined $paramlist && !ref $paramlist;
1577
1578  let nargs = match paramlist {
1579    Some(ref plist) => plist.get_num_args(),
1580    None => 0,
1581  };
1582  let csname = cs.with_str(ToString::to_string);
1583  let name_opt = {
1584    let name = match options.name {
1585      Some(ref name) => Cow::Owned(name.to_owned()),
1586      None => {
1587        let mut inferred_name = match options.alias {
1588          Some(ref alias) => Cow::Owned(alias.to_owned()),
1589          None => Cow::Borrowed(&csname),
1590        };
1591        if inferred_name.starts_with('\\') {
1592          inferred_name = Cow::Owned(inferred_name.replacen('\\', "", 1))
1593        }
1594        inferred_name
1595      },
1596    };
1597    let meaning_check = options
1598      .meaning
1599      .as_ref()
1600      .map_or_else(|| Cow::Owned(String::new()), Cow::Borrowed);
1601    if (*name == presentation) || (name.is_empty()) || *name == *meaning_check {
1602      None
1603    } else {
1604      Some(name.into_owned())
1605    }
1606  };
1607  options.name = name_opt;
1608  if nargs == 0 && options.role.is_none() {
1609    options.role = Some(String::from("UNKNOWN"))
1610  }
1611  if nargs > 0 && options.operator_role.is_none() {
1612    options.operator_role = Some(String::from("UNKNOWN"))
1613  }
1614  if options.hide_content_reversion {
1615    options.revert_as = Some(Cow::Borrowed("context"));
1616  }
1617
1618  let locked = options.locked;
1619  // Store some data for introspection
1620  // defmath_introspective(cs, paramlist, presentation, options);
1621
1622  // If single character, handle with a rewrite rule
1623  if csname.len() == 1 {
1624    let mut math_attr_hash: HashMap<String, String> = HashMap::default();
1625    transfer_opt_default!(name, options, math_attr_hash);
1626    transfer_opt_default!(meaning, options, math_attr_hash);
1627    transfer_opt_default!(omcd, options, math_attr_hash);
1628    transfer_opt_default!(decl_id, options, math_attr_hash);
1629    transfer_opt_default!(role, options, math_attr_hash);
1630    transfer_opt_default!(replace, options, math_attr_hash);
1631    transfer_opt_default!(mathstyle, options, math_attr_hash);
1632    transfer_opt_default!(stretchy, options, math_attr_hash);
1633    assign_value(
1634      &s!("math_token_attributes_{}", csname),
1635      math_attr_hash,
1636      Some(Scope::Global),
1637    );
1638  }
1639  // If the macro involves arguments,
1640  // we will create an XMDual to separate simple content application
1641  // from the (likely) convoluted presentation.
1642  else if HAS_ARG_OR_CS.is_match(&presentation) {
1643    // TODO: Are the code variants still applicable in Rust?
1644    //((ref presentation eq "CODE")
1645    // || ((ref presentation) && grep { $_->equals(T_PARAM) } presentation->unlist)
1646    // || ((ref presentation) && (grep { $_->isExecutable } presentation->unlist)))
1647    def_math_dual(cs, paramlist, presentation, options)?;
1648  }
1649  // EXPERIMENT: Introduce an intermediate case for simple symbols
1650  // Define a primitive that will create a Box with the appropriate set of XMTok attributes.
1651  else if nargs == 0 && !options.has_complex_option() {
1652    def_math_primitive(cs, paramlist, presentation, options);
1653  } else {
1654    def_math_constructor(cs, paramlist, presentation, options)?;
1655  }
1656  if locked {
1657    assign_value(&format!("{csname}:locked"), true, Some(Scope::Global));
1658  }
1659  Ok(())
1660}

1662/// Transfers the common MathPrimitive options to a (ideally freshly instantiated) Constructor.
1663fn transfer_common_constructor_options(
1664  cs: &Token,
1665  presentation: &str,
1666  options: MathPrimitiveOptions,
1667  cons: &mut Constructor,
1668) {
1669  let cs_str = cs.with_str(ToString::to_string);
1670  let mut properties = options.to_hash_stored();
1671  cons.alias = Some(options.alias.unwrap_or_else(|| cs_str.clone()));
1672  if let Some(sizer) = infer_sizer(options.sizer.as_ref(), options.reversion.as_ref()) {
1673    cons.sizer = Some(sizer);
1674  }
1675  if let Some(reversion) = options.reversion {
1676    cons.reversion = Some(reversion);
1677  }
1678  //
1679  // before_digest
1680  //
1681  let mut before_digest_closures: Vec<BeforeDigestClosure> = vec![before_digest_simple!({
1682    requireMath!(cs_str);
1683  })];
1684  if !options.nogroup {
1685    before_digest_closures.push(before_digest_simple!({
1686      bgroup();
1687    }));
1688  }
1689  if let Some(font) = options.font {
1690    before_digest_closures.push(before_digest_simple!({
1691      if let FontDirective::Asset(ref chosen_font) = font {
1692        merge_font((**chosen_font).clone());
1693      }
1694    }));
1695  }
1696  before_digest_closures.extend(options.before_digest);
1697  cons.before_digest = before_digest_closures;
1698  //
1699  // after_digest
1700  //
1701  let mut after_digest_closures = options.after_digest;
1702  // Perl: mathstyle => \&doVariablesizeOp — compute mathstyle at digest time
1703  if options.dynamic_mathstyle {
1704    after_digest_closures.push(after_digest_simple!(_args, {
1705      let state_font = lookup_font().unwrap();
1706      let is_display = state_font
1707        .get_mathstyle()
1708        .is_some_and(|s| s.as_ref() == "display");
1709      let mathstyle = if is_display { "display" } else { "text" };
1710      _args.set_property("mathstyle", Stored::from(mathstyle.to_string()));
1711    }));
1712  }
1713  if !options.nogroup {
1714    after_digest_closures.push(after_digest_simple!(_args, {
1715      egroup()?;
1716    }));
1717  }
1718  cons.after_digest = after_digest_closures;
1719  cons.before_construct = options.before_construct;
1720  cons.after_construct = options.after_construct;
1721  let presentation_for_font = presentation.to_owned();
1722  properties.insert(
1723    "font",
1724    Stored::FontDirective(FontDirective::Closure(
1725      if let Some(mathstyle) = options.mathstyle {
1726        Rc::new(move |_whatsit| {
1727          Ok(
1728            lookup_font()
1729              .unwrap()
1730              .merge(Font {
1731                mathstyle: Some(Cow::Owned(mathstyle.clone())),
1732                ..Font::default()
1733              })
1734              .specialize(&presentation_for_font),
1735          )
1736        })
1737      } else {
1738        Rc::new(move |_whatsit| Ok(lookup_font().unwrap().specialize(&presentation_for_font)))
1739      },
1740    )),
1741  );
1742
1743  cons.properties = Rc::new(move |_args| Ok(properties.clone()));
1744}

1746//======================================================================
1747// Allocated registers.
1748// We ASSUME the same set of \count positions used by TeX & LaTeX
1749// for recording the next available position in \count,\dimen,\skip,\muskip.
1750
1751pub fn allocate_register(rtype: &str) -> Result<Option<String>> {
1752  let addr = match rtype {
1753    "\\count" => "\\count10",
1754    "\\dimen" => "\\count11",
1755    "\\skip" => "\\count12",
1756    "\\muskip" => "\\count13",
1757    "\\box" => "\\count14",
1758    "\\toks" => "\\count15",
1759    _ => "",
1760  };
1761  if !addr.is_empty() {
1762    // addr is a Register but MUST be stored as \count<#>
1763    if let Some(n) = lookup_number(addr) {
1764      // Perl Package.pm L617-622: the allocation counter picks the NEXT
1765      // unbound slot. If `\<type>N+1` is already an explicit DefRegister
1766      // (e.g. a system-allocated `\count10`, `\toks0`), advance past it
1767      // so the new register doesn't collide. Matches Perl's
1768      //   while ($STATE->isValueBound($loc)) { $next++; $loc = $type . $next; }
1769      let mut next = n.value_of() + 1;
1770      let mut loc = format!("{rtype}{next}");
1771      while crate::state::is_value_bound(&loc, None) {
1772        next += 1;
1773        loc = format!("{rtype}{next}");
1774      }
1775      assign_value(addr, Number::new(next), Some(Scope::Global));
1776      Ok(Some(loc))
1777    } else {
1778      Ok(None)
1779    }
1780  } else {
1781    Error!(
1782      "misdefined",
1783      rtype,
1784      format!("Type {rtype} is not an allocated register type")
1785    );
1786    Ok(None)
1787  }
