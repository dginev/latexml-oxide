34/// Maximum nesting depth for package/class loading to prevent infinite recursion.
35/// Perl LaTeXML has no explicit limit but rarely exceeds 20 levels in practice.
36const MAX_INPUT_DEPTH: usize = 500;

38thread_local! {
39  /// Current nesting depth of input_definitions calls.
40  static INPUT_DEPTH: std::cell::Cell<usize> = const { std::cell::Cell::new(0) };

43/// a configuration for loading LaTeX definition files (such as .sty, .cls, and their bindings)
44pub struct InputDefinitionOptions {
45  /// an optional extension (such as "sty")
46  pub extension:     Option<Cow<'static, str>>,
47  /// package options to pass into the loaded library
48  pub options:       Vec<String>,
49  /// Tokens to process after the definition is loaded
50  pub after:         Tokens,
51  /// flag to forbid raw TeX sources
52  pub notex:         bool,
53  /// flag to forbid errors ?
54  pub noerror:       bool,
55  /// flag to forbid binding dispatch
56  pub noltxml:       bool,
57  /// collection of (package) options to process when loading the dependency
58  pub withoptions:   Option<Vec<String>>,
59  /// flag to handle options, or ignore them
60  pub handleoptions: bool,
61  /// flag to process in .cls mode (default: false)
62  pub as_class:      bool,
63  /// flag to indicate reading the file raw in Gullet
64  pub raw:           bool,
65  /// flag to allow reloading a previously loaded definitions file
66  pub reloadable:    bool,
67  /// flag: set @ catcode to LETTER during loading (default true).
68  /// Set to false for packages like xy.tex that need @ to stay as OTHER.
69  pub at_letter:     bool,
70}
71impl Default for InputDefinitionOptions {
72  fn default() -> Self {
73    InputDefinitionOptions {
74      extension:     None,
75      options:       Vec::new(),
76      after:         Tokens!(),
77      notex:         false,
78      noerror:       false,
79      noltxml:       false,
80      raw:           false,
81      reloadable:    false,
82      withoptions:   None,
83      handleoptions: false,
84      as_class:      false,
85      at_letter:     true,
86    }
87  }

91pub fn input_definitions(raw_file: &str, mut options: InputDefinitionOptions) -> Result<()> {
92  let name = raw_file.trim();
93
94  // Guard: prevent infinite recursion from circular or runaway package loading.
95  // When a binding is missing, raw TeX loading can trigger macro loops.
96  let depth = INPUT_DEPTH.with(|d| {
97    let current = d.get();
98    d.set(current + 1);
99    current + 1
100  });
101  if depth > MAX_INPUT_DEPTH {
102    INPUT_DEPTH.with(|d| d.set(d.get() - 1));
103    Fatal!(
104      Stomach, Recursion,
105      s!("Package loading depth exceeded {} (loading '{}').\
106        This usually means a missing binding causes infinite recursion.", MAX_INPUT_DEPTH, name)
107    );
108  }
109
110  // Ensure depth cleanup on all exit paths via a guard
111  struct InputDepthGuard;
112  impl Drop for InputDepthGuard {
113    fn drop(&mut self) {
114      INPUT_DEPTH.with(|d| d.set(d.get() - 1));
115    }
116  }
117  let _guard = InputDepthGuard;
118
119  // Note: we always need a gullet to expand, and we sometimes need a stomach to load_definitions...
120  // so let's make stomach a mandatory option.
121  let prevname = if options.handleoptions && lookup_definition(&T_CS!("\\@currname"))?.is_some() {
122    gullet::do_expand(T_CS!("\\@currname"))?.to_string()
123  } else {
124    String::new()
125  };
126  let prevext = if options.handleoptions && lookup_definition(&T_CS!("\\@currext"))?.is_some() {
127    gullet::do_expand(T_CS!("\\@currext"))?.to_string()
128  } else {
129    String::new()
130  };
131  // This file will be treated somewhat as if it were a class
132  // IF as_class is true
133  // OR if it is loaded by such a class, and has withoptions true!!! (yikes)
134  if options.handleoptions && options.withoptions.is_some() {
135    with_vecdeque("@masquerading@as@class", |vdq_opt| {
136      if let Some(vdq) = vdq_opt {
137        if vdq.iter().any(|x| {
138          if let Stored::String(ref v) = x {
139            arena::with(*v, |str| str == prevname)
140          } else {
141            false
142          }
143        }) {
144          options.as_class = true;
145        }
146      }
147    });
148  }
149  if options.noltxml {
150    options.raw = true; // so it will be read as raw by Gullet.
151  }
152  let as_type = if options.as_class {
153    Cow::Borrowed("cls")
154  } else {
155    options
156      .extension
157      .as_ref()
158      .cloned()
159      .unwrap_or(Cow::Borrowed(""))
160  };
161
162  // If loading a class, store class options (Perl Package.pm lines 2561-2564)
163  if as_type == "cls" && !options.options.is_empty() {
164    for opt in &options.options {
165      push_value("class_options", arena::pin(opt))?;
166    }
167    let class_opts_str = options.options.join(",");
168    def_macro(
169      T_CS!("\\@classoptionslist"),
170      None,
171      Tokens!(Explode!(class_opts_str)),
172      None,
173    )?;
174  }
175
176  // Compute the exact name based on the type
177  let filename = match &options.extension {
178    None => name.to_string(),
179    Some(ext) => s!("{}.{}", name, ext),
180  };
181  // Store the document class filename for xkeyval's isInClassFile check
182  if as_type == "cls" && options.handleoptions {
183    assign_value("document_class_filename", filename.clone(), Some(Scope::Global));
184  }
185  let current_options = options.options.join(",");
186  if !current_options.is_empty() {
187    if let Some(Stored::String(prevoptions)) = lookup_value(&s!("{filename}_loaded_with_options")) {
188      if arena::with(prevoptions, |prev_str| current_options != prev_str) {
189        let message = s!(
190          "Option clash for file {} with options {:?}, previously loaded with {:?}",
191          filename,
192          current_options,
193          prevoptions
194        );
195        Info!("unexpected", "options", message);
196      }
197    }
198  }
199
200  // TODO: This needs reorganization, bindings are not found as "files" in rust,
201  // we need to have a registry (we don't yet)
202
203  // Perl: early-stop if already loaded (checks request_loaded, name_loaded, etc.)
204  // This prevents double-loading and breaks circular loading chains.
205  // IMPORTANT: check BEFORE printing "Loading..." message to avoid spurious output.
206  let loaded_key = s!("{filename}_loaded");
207  if !options.reloadable && lookup_bool(&loaded_key) {
208    return Ok(());
209  }
210  // Also check without extension (Perl checks name_loaded too)
211  let name_loaded_key = s!("{name}_loaded");
212  if !options.reloadable && name != filename && lookup_bool(&name_loaded_key) {
213    return Ok(());
214  }
215
216  // Mark as loaded, then process the definitions
217  note_begin(&s!("Loading {:?} definitions", filename));
218  def_macro(T_CS!("\\@currname"), None, Tokens!(Explode!(name)), None)?;
219  def_macro(T_CS!("\\@currext"), None, Tokens!(Explode!(as_type)), None)?;
220
221  // TODO: Is this inaccurate with latexml? It only sets the macros if the file is found, we set
222  // them *always*, as a matter of course TODO: This *IS* inaccurate with the Package.pm
223  // InputDefinitions, revisit at the right time and make sure it matches line-by-line (including
224  // the subordinated methods)
225  if options.handleoptions {
226    before_input_handle_options(&mut options, &prevname, &prevext, name, &as_type)?;
227    def_macro(
228      T_CS!(s!("\\{}.{}-h@@k", name, as_type)),
229      None,
230      options.after,
231      None,
232    )?;
233  }
234
235  if !current_options.is_empty() {
236    assign_value(
237      &s!("{}_loaded_with_options", filename),
238      current_options,
239      Some(Scope::Global),
240    );
241  }
242
243  // Track loaded files in \@filelist BEFORE loading (Perl: Package.pm calls
244  // \@addtofilelist before reading the file, so \@filelist is available inside)
245  if options.handleoptions && lookup_definition(&T_CS!("\\@addtofilelist"))?.is_some() {
246    digest(Tokens!(
247      T_CS!("\\@addtofilelist"), T_BEGIN!(), Explode!(filename), T_END!()
248    ))?;
249  }
250
251  // Catch Fatal errors during binding loading (e.g., token limit exceeded during
252  // expl3 kernel loading). Convert to non-fatal so document processing continues.
253  let is_binding = if options.noltxml {
254    false
255  } else {
256    match load_external_binding(&filename).and_then(|ext| {
257      if ext { Ok(true) } else { load_binding(&filename) }
258    }) {
259      Ok(v) => v,
260      Err(e) => {
261        Error!("unexpected", &filename, s!("Error loading binding for '{}': {}", filename, e));
262        // Mark as loaded even on error to prevent re-loading via raw path
263        assign_value(&s!("{filename}_loaded"), true, Some(Scope::Global));
264        false
265      }
266    }
267  };
268  let mut is_found_raw = false;
269  if is_binding {
270    // We found and loaded a binding successfully, mark it as such.
271    let loaded_flag = format!("{filename}_loaded");
272    assign_value(&loaded_flag, true, Some(Scope::Global));
273    // Perl: Let(T_CS('\ver@'.$trequest), T_CS('\fmtversion'), 'global');
274    // Set \ver@name.ext = \fmtversion so \RequirePackage with date checks works.
275    // Without this, LaTeX's package loading guard thinks the package isn't loaded.
276    // Perl: Let(T_CS('\ver@'.$trequest), T_CS('\fmtversion'), 'global');
277    // Set \ver@name.ext to \fmtversion so LaTeX's \RequirePackage guard works.
278    // Without this, \RequirePackage date checks fail and packages get re-loaded.
279    if options.handleoptions {
280      let ver_cs = T_CS!(s!("\\ver@{}", filename));
281      if lookup_definition(&ver_cs).ok().flatten().is_none() {
282        let fmtversion_cs = T_CS!("\\fmtversion");
283        let_i(&ver_cs, &fmtversion_cs, Some(Scope::Global));
284      }
285    }
286  } else {
287    // We're inverting the control flow, because it is near-instant to check whether we have an
288    // available binding dispatcher, in both contributed and core binding names
289    // Now that we have ensured there is no compiled target of this name, we can start the file
290    // system search dance, call to kpsewhich, etc.
291    //
292    // Perl Package.pm FindFile search order (L2109-2139):
293    //   1. .ltxml binding (handled above by load_binding/load_external_binding)
294    //   2. Raw TeX in search paths, BUT only if INTERPRETING_DEFINITIONS is true
295    //      (i.e. we're inside recursive loading from another raw TeX file)
296    //   3. FindFile_fallback — strip version suffixes, find generic .ltxml binding
297    //      (e.g. icml2024.sty → icml.sty.ltxml)
298    //   4. Raw TeX in search paths (without INTERPRETING_DEFINITIONS gate)
299    //   5. kpsewhich
300    //
301    // This ordering ensures versioned-package fallback bindings take priority
302    // over raw .sty files that may contain layout checks (like ICML's \ifdim
303    // page-margin checks) that produce spurious warnings.
304    let interpreting = lookup_bool("INTERPRETING_DEFINITIONS");
305
306    // Step 2: If we're already interpreting raw TeX definitions, look for the file directly
307    let found_raw = if interpreting && !options.notex {
308      find_file(
309        &filename,
310        Some(FindFileOptions {
311          forbid_ltxml:      options.noltxml,
312          notex:             false,
313          ext_type:          options.extension.as_ref().cloned(),
314          search_paths_only: false,
315        }),
316      )
317    } else {
318      None
319    };
320
321    // Step 3: Try fallback (strip version suffixes) before raw TeX
322    // Perl Package.pm L2118-2121: FindFile_fallback
323    // e.g. natbib-arxiv_v2.sty → natbib.sty, icml2024.sty → icml.sty
324    let found_raw = if found_raw.is_some() {
325      found_raw
326    } else if !options.noltxml {
327      if let Some(fallback) = find_file_fallback(name, &as_type) {
328        Info!("fallback", name,
329          s!("Interpreted as versioned package, falling back to {fallback}"));
330        // Load the fallback binding — use reloadable since we already marked original as "loaded"
331        let fallback_name = fallback.trim_end_matches(&format!(".{as_type}")).to_string();
332        let fb_result = input_definitions(&fallback_name, InputDefinitionOptions {
333          extension: Some(Cow::Borrowed(if as_type == "sty" { "sty" } else { "cls" })),
334          options: Vec::new(),
335          after: Tokens::default(),
336          handleoptions: options.handleoptions,
337          noerror: true,
338          reloadable: true,
339          ..InputDefinitionOptions::default()
340        });
341        if fb_result.is_ok() {
342          assign_value(&s!("{filename}_loaded"), true, Some(Scope::Global));
343        }
344        None // fallback handled the loading; no raw file to load
345      } else {
346        None
347      }
348    } else {
349      None
350    };
351
352    // Step 4: Raw TeX in search paths (without INTERPRETING_DEFINITIONS gate)
353    // Perl Package.pm L2122-2125
354    let found_raw = if found_raw.is_some() {
355      found_raw
356    } else if !options.notex && !interpreting && !lookup_bool(&s!("{filename}_loaded")) {
357      find_file(
358        &filename,
359        Some(FindFileOptions {
360          forbid_ltxml:      options.noltxml,
361          notex:             false,
362          ext_type:          options.extension.as_ref().cloned(),
363          search_paths_only: false,
364        }),
365      )
366    } else {
367      None
368    };
369
370    if let Some(file) = found_raw {
371      is_found_raw = true;
372      load_tex_definitions(&filename, &file, options.reloadable, options.at_letter)?;
373    } else if !lookup_bool(&s!("{filename}_loaded")) {
374      if options.noerror {
375        // With noerror: don't mark as loaded and return Err so callers can
376        // try fallback names (e.g. tikzlibrary → pgflibrary). Matches Perl's
377        // InputDefinitions which returns undef on not-found even with noerror=>1.
378        note_end(&s!("Loading {:?} definitions", filename));
379        return Err(s!("File not found: {}", filename).into());
380      }
381      // Mark as loaded even on failure — prevents retrying a missing file
382      // in a loop (e.g. when raw TeX repeatedly calls \RequirePackage).
383      assign_value(&s!("{filename}_loaded"), true, Some(Scope::Global));
384      Warn!("missing_file", name,
385        s!("Can't find binding or file for '{filename}'. \
386          No dispatcher entry and no raw file found on disk."));
387    }
388  }
389
390  if options.handleoptions {
391    if is_binding || is_found_raw {
392      digest(T_CS!(s!("\\{name}.{as_type}-h@@k")))?;
393    }
394    // Always restore @currname/@currext and pop filename stack,
395    // even when no binding was found, to keep the stack balanced.
396    // Note: @popfilename uses \gdef to restore @currname/@currext from the stack,
397    // so it takes precedence. We also set them with def_macro as a fallback
398    // (matches Perl Package.pm lines 2635-2637).
399    if !prevname.is_empty() {
400      def_macro(
401        T_CS!("\\@currname"),
402        None,
403        Tokens!(Explode!(prevname)),
404        None,
405      )?;
406    }
407    if !prevext.is_empty() {
408      def_macro(T_CS!("\\@currext"), None, Tokens!(Explode!(prevext)), None)?;
409    }
410    digest(T_CS!("\\@popfilename"))?;
411    // Verify @currname was correctly restored, and force-fix if not
412    let restored_name = if lookup_definition(&T_CS!("\\@currname"))?.is_some() {
413      do_expand(T_CS!("\\@currname"))?.to_string()
414    } else {
415      String::new()
416    };
417    if !prevname.is_empty() && restored_name != prevname {
418      // @popfilename may have popped a stale entry; force correct value
419      def_macro(
420        T_CS!("\\@currname"),
421        None,
422        Tokens!(Explode!(prevname)),
423        Some(ExpandableOptions { scope: Some(Scope::Global), ..ExpandableOptions::default() }),
424      )?;
425    }
426    if !prevext.is_empty() {
427      let restored_ext = if lookup_definition(&T_CS!("\\@currext"))?.is_some() {
428        do_expand(T_CS!("\\@currext"))?.to_string()
429      } else {
430        String::new()
431      };
432      if restored_ext != prevext {
433        def_macro(
434          T_CS!("\\@currext"),
435          None,
436          Tokens!(Explode!(prevext)),
437          Some(ExpandableOptions { scope: Some(Scope::Global), ..ExpandableOptions::default() }),
438        )?;
439      }
440    }
441    reset_options()?;
442  }
443  note_end(&s!("Loading {:?} definitions", filename));
444  Ok(())
445}

452fn _load_binding(internal: bool, request: &str) -> Result<bool> {
453  // avoid double-loads, but be binding-specific
454  let loaded_key = s!("{request}_binding_loaded");
455  if lookup_bool(&loaded_key) {
456    return Ok(true);
457  }
458  // TODO? || lookup_bool(&s!("{trequest}_loaded"))
459  //|| lookup_bool(&s!("{name}_loaded")) || lookup_bool(&s!("{ltxname}_loaded"));
460
461  let taken_dispatcher = if internal {
462    get_bindings_dispatch()
463  } else {
464    get_extra_bindings_dispatch()
465  };
466  match taken_dispatcher {
467    Some(ref dispatcher) => {
468      let result_opt = dispatcher(request);
469      match result_opt {
470        Some(result) => {
471          // Here and only here we are certain we have binding support.
472          // Preemptively mark as loaded to avoid recursion.
473
474          // TODO: is this still true?
475          // Note (only!) that the binding version of this was loaded; still could load raw tex!
476          assign_value(&loaded_key, true, Some(Scope::Global));
477          // if a binding load succeeded, mark the generic request as loaded.
478          assign_value(&s!("{request}_loaded"), true, Some(Scope::Global));
479          match result {
480            Ok(()) => Ok(true),
481            Err(e) => Err(e),
482          }
483        },
484        None => Ok(false),
485      }
486    },
487    None => Ok(false),
488  }
489}

491// Factor out handling and passing loading options from input_content,
492// to simplify main routine
493fn before_input_handle_options(
494  options: &mut InputDefinitionOptions,
495  prevname: &str,
496  prevext: &str,
497  name: &str,
498  as_type: &str,
499) -> Result<()> {
500  // Note: this is trying to emulate the LaTeX 2 (latex.ltx) use of \@pushfilename. For expl3, see
501  // expl3.sty.ltxml
502  digest(T_CS!("\\@pushfilename"))?;
503
504  // For \RequirePackageWithOptions, pass the options from the outer class/style to the inner one.
505  if let Some(with_options_to_pass) = options.withoptions.take() {
506    if !prevname.is_empty() && has_value(&s!("opt@{}.{}", prevname, prevext)) {
507      // Only pass those class options that are declared by the package!
508      let mut topass = Vec::new();
509      with_vecdeque("@declaredoptions", |vdq_opt| {
510        if let Some(declared_options) = vdq_opt {
511          for op in with_options_to_pass.into_iter() {
512            if declared_options.iter().any(|x| {
513              if let Stored::String(val) = x {
514                arena::with(*val, |str| str == op)
515              } else {
516                false
517              }
518            }) {
519              topass.push(op)
520            }
521          }
522        }
523      });
524      if !topass.is_empty() {
525        pass_options(name, as_type, topass)?;
526      }
527    }
528  }
529  def_macro(T_CS!("\\@currname"), None, Tokens!(Explode!(name)), None)?;
530  def_macro(T_CS!("\\@currext"), None, Tokens!(Explode!(as_type)), None)?;
531  // reset options (Note reset & pass were in opposite order in LoadClass ????)
532  reset_options()?;
533  pass_options(name, as_type, options.options.clone())?;
534
535  // Note which packages are pretending to be classes.
536  if options.as_class {
537    push_value("@masquerading@as@class", arena::pin(name))?;
538  }
539  let current_opt_val = with_vecdeque(&s!("opt@{}.{}", name, as_type), |vdq_opt| match vdq_opt {
540    Some(vdq) => {
541      let mut pieces = String::new();
542      for x in vdq.iter() {
543        if let Stored::String(val) = x {
544          arena::with(*val, |str| pieces.push_str(str));
545        }
546        pieces.push(',');
547      }
548      pieces.pop();
549      pieces
550    },
551    None => String::new(),
552  });
553  def_macro(
554    T_CS!(s!("\\opt@{}.{}", name, as_type)),
555    None,
556    Tokens!(Explode!(current_opt_val)),
557    None,
558  )?;
559  Ok(())
560}

590  }

593/// This is essentially the `\input` equivalent
594///
595/// we are most likely expecting to get actual content,
596/// (possibly with definitions included, as well)
597/// but might actually be getting pure definitions,
598/// (like a proper style file)
599/// in which case we may really want to load a binding.
600/// Note that generic style files (non-latex) often have a .tex extension.
601pub fn input(request: &str, options: InputOptions) -> Result<()> {
602  // unwrap if in quotes \input{"file name"}

1193    let paths: Vec<String> = get_search_paths();

1199    // has an entry for this file, consider it "found". This is how Perl's

1201    // We check the `{file}_binding_available` flag, which binding packages

1205    }

1209        paths: Some(paths),
1210        ..PathnameFindOptions::default()

1213      }
1214    }
1215    // Otherwise, pass on to kpsewhich
1216    // Depending on flags, maybe search for ltxml in texmf or for plain tex in ours!

1218    // return if grep { pathname::is_nasty($_) } @$paths;    // SECURITY! No nasty paths in cmdline
1219    //       // Do we need to sanitize these environment variables?
1220    // my $kpsewhich = which($ENV{LATEXML_KPSEWHICH} || 'kpsewhich');
1221    // local $ENV{TEXINPUTS} = join($Config::Config{'path_sep'},
1222    //   @$paths, $ENV{TEXINPUTS} || $Config::Config{'path_sep'});
1223    // my @candidates = (((!$options{noltxml} && !$nopaths) ? ("$file.ltxml") : ()),
1224    //   (!$options{notex} ? ($file) : ()));
1225    // if (my $result = pathname::kpsewhich(@candidates)) {
1226    //   return (-f $result ? $result : undef); }
1227    // if ($urlbase && ($path = url_find($file, urlbase => $urlbase))) {
1228    //   return $path; }
1229    // When notex is set, don't search for the raw TeX file via kpsewhich.
1230    // This prevents non-TeX files (e.g. .lua) from being loaded as raw TeX.
1231    if options.notex {
1232      None
1233    } else {
1234      pathname::kpsewhich(&[file])
1235    }
1236  }

1280/// Define a named color (Perl: DefColor).
1281/// Stores as color_{name} and also defines \\color@{name} macro.
1282pub fn def_color(name: &str, color: &crate::common::color::Color, scope: Option<Scope>) -> Result<()> {
1283  use crate::common::color;
1284  // Check ifglobalcolors
1285  let effective_scope = if if_condition(&T_CS!("\\ifglobalcolors"))? == Some(true) {
1286    Some(Scope::Global)
1287  } else {
1288    scope
1289  };
1290  // Store in state as "model c1 c2 ..."
1291  let stored = color.to_stored();
1292  assign_value(
1293    &s!("color_{name}"),
1294    Stored::String(arena::pin(stored)),
1295    effective_scope,
1296  );
1297  // Define \\color@{name} macro for reversion
1298  // Perl: DefMacroI('\\color@'.$name, undef,
1299  //   '\relax\relax{model spec}{model}{spec_commas}')
1300  let model = color.model();
1301  let comps = color.components();
1302  let spec_parts: Vec<String> = comps.iter().map(|c| color::format_component(*c)).collect();
1303  let spec_space = spec_parts.join(" ");
1304  let spec_comma = spec_parts.join(",");
1305  let model_spec = s!("\\relax\\relax{{{model} {spec_space}}}{{{model}}}{{{spec_comma}}}");
1306  def_macro(
1307    T_CS!(s!("\\\\color@{name}")),
1308    None,
1309    crate::mouth::tokenize_internal(&model_spec),
1310    Some(crate::definition::expandable::ExpandableOptions {
1311      scope: effective_scope,
1312      ..Default::default()
1313    }),
1314  )?;
1315  Ok(())
1316}

1364/// Test a conditional `\ifXXX` and return its boolean result (Perl: IfCondition).
1365/// Looks up the conditional's test closure and invokes it.
1366pub fn if_condition(if_token: &Token) -> Result<Option<bool>> {
1367  use crate::definition::conditional::ConditionalType;
1368  if let Some(defn) = lookup_definition(if_token)? {
1369    if defn.get_conditional_type() == Some(ConditionalType::If) {
1370      if let Some(test) = defn.get_test() {
1371        // Read arguments for the conditional test
1372        let args = match defn.get_parameters() {
1373          Some(params) => params.read_arguments(Some(defn.as_ref()))?,
1374          None => Vec::new(),
1375        };
1376        return Ok(Some(test(args)?));
1377      }
1378    }
1379  }
1380  if x_equals(if_token, &T_CS!("\\iftrue")) {
1381    return Ok(Some(true));
1382  }
1383  if x_equals(if_token, &T_CS!("\\iffalse")) {
1384    return Ok(Some(false));
1385  }
1386  Error!(
1387    "expected",
1388    "conditional",
1389    s!("Expected a conditional, got '{}'", if_token.stringify())
1390  );
1391  Ok(None)
1392}

1414/// Creates a single `Tokens` representing a TeX invocation of the lead `token` over a list of
1415/// arguments.
1416///
1417/// Note: currently this is near the `Mouth` representation of data, and deals purely with tokens.
1418/// A more generic version of this method may be able to support `ArgWrap` for the argument list.
1419/// Indeed, the return type may also be lifted to a new generic ArgWrap::Invocation, if there was
1420/// benefit.
1421pub fn build_invocation<T: Into<Token>>(token: T, args: Vec<Option<Tokens>>) -> Result<Tokens> {
1422  let token: Token = token.into();
1423  // Note: token may have been \let to another defn!
1424  if let Some(defn) = lookup_definition(&token)? {
1425    let mut invoked_tokens = vec![token];
1426    if let Some(params) = defn.get_parameters() {
1427      invoked_tokens.extend(params.revert_arguments(args)?);
1428    }
1429    Ok(Tokens::new(invoked_tokens))
1430  } else {
1431    let message = s!("Can't invoke {:?}; it is undefined", token.stringify());
1432    token.with_cs_name(|csname| {
1433      Error!("undefined", csname, message);
1434      Ok(())
1435    })?;
1436    let mut invoked_tokens = vec![token];
1437    // DefConstructor!(token, convert_latex_args(args.len(), 0),
1438    // sub { LaTeXML::Core::Stomach::makeError($_[0], 'undefined', token); });
1439    let wrapped_args: Vec<Token> = args
1440      .into_iter()
1441      .flat_map(|arg_opt| {
1442        let mut wrapped = vec![T_BEGIN!()];
1443        if let Some(arg) = arg_opt {
1444          wrapped.extend(arg.unlist());
1445        }
1446        wrapped.push(T_END!());
1447        wrapped
1448      })
1449      .collect();
1450    invoked_tokens.extend(wrapped_args);
1451    Ok(Tokens::new(invoked_tokens))
1452  }
1453}

1455/// Convert a LaTeX-style argument spec to our Package form.
1456/// Ie. given $nargs and $optional, being the two optional arguments to
1457/// something like \newcommand, convert it to the form we use
1458pub fn convert_latex_args(
1459  mut nargs: usize,
1460  optional: Option<Tokens>,
1461) -> Result<Option<Parameters>> {
1462  let mut params = Vec::new();
1463  if let Some(tks) = optional {
1464    params.push(
1465      Parameter {
1466        name: arena::pin_static("Optional"),
1467        spec: arena::pin(s!("[Default:{}]", tks.clone().untex())),
1468        extra: vec![tks],
1469        ..Parameter::default()
1470      }
1471      .init()?,
1472    );
1473    nargs -= 1;
1474  }
1475
1476  for _ in 1..=nargs {
1477    params.push(
1478      Parameter {
1479        name: arena::pin_static("Plain"),
1480        spec: arena::pin_static("{}"),

1482      }
1483      .init()?,
1484    );
1485  }
1486  if params.is_empty() {
1487    Ok(None)
1488  } else {
1489    Ok(Some(Parameters::new(params)))
1490  }

1493/// Decode a codepoint using the fontmap for a given font and/or encoding (Perl: FontDecode).
1494/// Returns the decoded glyph (if any) and the possibly-adjusted font.
1495pub fn font_decode(
1496  code: i32,
1497  encoding_opt: Option<&str>,
1498  font_opt: Option<Rc<Font>>,
1499) -> (Option<char>, Option<Rc<Font>>) {
1500  if code < 0 {
1501    return (None, font_opt);
1502  }
1503  let font = font_opt.unwrap_or_else(|| lookup_font().unwrap());
1504  let encoding = match encoding_opt {
1505    Some(enc) => enc.to_string(),
1506    None => font.get_encoding().map_or("OT1".to_string(), |c| c.to_string()),
1507  };
1508  let map = load_font_map(&encoding);
1509  // Check for family-specific map
1510  let (effective_map, _effective_enc) = if let Some(family) = font.get_family() {
1511    let fam_key = s!("{}_{}_fontmap", encoding, family);
1512    if let Some(fmap) = lookup_value(&fam_key) {
1513      let fmap: Option<Fontmap> = fmap.into();
1514      if let Some(fm) = fmap {
1515        (Some(fm), s!("{}_{}", encoding, family))
1516      } else {
1517        (map, encoding.clone())
1518      }
1519    } else {
1520      (map, encoding.clone())
1521    }
1522  } else {
1523    (map, encoding.clone())
1524  };
1525  let glyph = effective_map
1526    .as_ref()
1527    .and_then(|m| m.get(code as usize).copied().flatten());
1528  // In-math alphanumeric mathstyle handling is done in decodeMathChar instead
1529  (glyph, Some(font))
1530}

1532/// Decode a string using the fontmap for a given encoding (Perl: FontDecodeString).
1533/// If `implicit` is true, codepoints missing from the map decode to themselves.
1534pub fn font_decode_string(
1535  string: &str,
1536  encoding_opt: Option<&str>,
1537  implicit: bool,
1538) -> String {
1539  let font = lookup_font().unwrap();
1540  let encoding = match encoding_opt {
1541    Some(enc) => enc.to_string(),
1542    None => font.get_encoding().map_or("OT1".to_string(), |c| c.to_string()),
1543  };
1544  let map = load_font_map(&encoding);
1545  // Check for family-specific map
1546  let effective_map = if let Some(family) = font.get_family() {
1547    let fam_key = s!("{}_{}_fontmap", encoding, family);
1548    if let Some(fmap) = lookup_value(&fam_key) {
1549      let fmap: Option<Fontmap> = fmap.into();
1550      fmap.or(map)
1551    } else {
1552      map
1553    }
1554  } else {
1555    map
1556  };
1557  let input_enc = lookup_string("INPUT_ENCODING");
1558  let map_max: usize = if input_enc == "utf8" { 128 } else { 256 };
1559  // Also limit for short font maps
1560  let map_max = if let Some(ref m) = effective_map {
1561    if m.len() < map_max { m.len() } else { map_max }
1562  } else {
1563    map_max
1564  };
1565
1566  let mut result = String::new();
1567  for ch in string.chars() {
1568    let code = ch as usize;
1569    if implicit {
1570      if let Some(ref m) = effective_map {
1571        if code < map_max {
1572          if let Some(Some(glyph)) = m.get(code) {
1573            result.push(*glyph);
1574          }
1575        } else {
1576          result.push(ch);
1577        }
1578      } else {
1579        result.push(ch);
1580      }
1581    } else if let Some(ref m) = effective_map {
1582      if let Some(Some(glyph)) = m.get(code) {
1583        result.push(*glyph);
1584      }
1585    }
1586  }
1587  result
1588}

1598pub fn preload_font_map(encoding: &str) -> Result<()> {
1599  // This check is done as a "preload" step for mutability reasons.
1600  let key = s!("{encoding}_fontmap");

1603  }
1604  let fail_key = s!("{encoding}_fontmap_failed_to_load");
1605  let failed_flag = lookup_bool(&fail_key);
1606  if !failed_flag {
1607    assign_value(&fail_key, true, None); // Stop recursion?

1609      extension: Some(Cow::Borrowed("fontmap")),

1612    });
1613    if has_value(&s!("{encoding}_fontmap")) {
1614      // Got map?
1615      assign_value(&fail_key, false, None);

1617      assign_value(&fail_key, true, Some(Scope::Global));
