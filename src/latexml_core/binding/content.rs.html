34/// Maximum nesting depth for package/class loading to prevent infinite recursion.
35/// Perl LaTeXML has no explicit limit but rarely exceeds 20 levels in practice.
36const MAX_INPUT_DEPTH: usize = 500;
37
38thread_local! {
39  /// Current nesting depth of input_definitions calls.
40  static INPUT_DEPTH: std::cell::Cell<usize> = const { std::cell::Cell::new(0) };
41}

43/// a configuration for loading LaTeX definition files (such as .sty, .cls, and their bindings)
44pub struct InputDefinitionOptions {
45  /// an optional extension (such as "sty")
46  pub extension:        Option<Cow<'static, str>>,
47  /// package options to pass into the loaded library
48  pub options:          Vec<String>,
49  /// Tokens to process after the definition is loaded
50  pub after:            Tokens,
51  /// flag to forbid raw TeX sources
52  pub notex:            bool,
53  /// flag to forbid errors ?
54  pub noerror:          bool,
55  /// flag to forbid binding dispatch
56  pub noltxml:          bool,
57  /// collection of (package) options to process when loading the dependency
58  pub withoptions:      Option<Vec<String>>,
59  /// flag to handle options, or ignore them
60  pub handleoptions:    bool,
61  /// flag to process in .cls mode (default: false)
62  pub as_class:         bool,
63  /// flag to indicate reading the file raw in Gullet
64  pub raw:              bool,
65  /// flag to allow reloading a previously loaded definitions file
66  pub reloadable:       bool,
67  /// flag: set @ catcode to LETTER during loading (default true).
68  /// Set to false for packages like xy.tex that need @ to stay as OTHER.
69  pub at_letter:        bool,
70  /// When set, raw-file lookup is restricted to the user-supplied
71  /// SEARCHPATHS (SOURCEDIRECTORY + graphicspaths), skipping the
72  /// kpsewhich fallback into system texmf. Matches Perl Package.pm's
73  /// `searchpaths_only => 1` — enabled by the `localrawstyles` option
74  /// to latexml.sty. Perl ref: Package.pm L2135, L2674.
75  pub searchpaths_only: bool,
76}
77impl Default for InputDefinitionOptions {
78  fn default() -> Self {
79    InputDefinitionOptions {
80      extension:        None,
81      options:          Vec::new(),
82      after:            Tokens!(),
83      notex:            false,
84      noerror:          false,
85      noltxml:          false,
86      raw:              false,
87      reloadable:       false,
88      withoptions:      None,
89      handleoptions:    false,
90      as_class:         false,
91      at_letter:        true,
92      searchpaths_only: false,
93    }
94  }
95}

98pub fn input_definitions(raw_file: &str, mut options: InputDefinitionOptions) -> Result<()> {
99  let name = raw_file.trim();
100
101  // Guard: prevent infinite recursion from circular or runaway package loading.
102  // When a binding is missing, raw TeX loading can trigger macro loops.
103  let depth = INPUT_DEPTH.with(|d| {
104    let current = d.get();
105    d.set(current + 1);
106    current + 1
107  });
108  if depth > MAX_INPUT_DEPTH {
109    INPUT_DEPTH.with(|d| d.set(d.get() - 1));
110    Fatal!(
111      Stomach,
112      Recursion,
113      s!(
114        "Package loading depth exceeded {} (loading '{}').\
115        This usually means a missing binding causes infinite recursion.",
116        MAX_INPUT_DEPTH,
117        name
118      )
119    );
120  }
121
122  // Ensure depth cleanup on all exit paths via a guard
123  struct InputDepthGuard;
124  impl Drop for InputDepthGuard {
125    fn drop(&mut self) { INPUT_DEPTH.with(|d| d.set(d.get() - 1)); }
126  }
127  let _guard = InputDepthGuard;
128
129  // Note: we always need a gullet to expand, and we sometimes need a stomach to load_definitions...
130  // so let's make stomach a mandatory option.
131  //
132  // Snapshot \@currname/\@currext only when handleoptions=true. Perl
133  // Package.pm:2549-2550 does the same — both prevname and prevext are
134  // gated on options{handleoptions}. The handleoptions=false branch
135  // does NOT mutate \@currname/\@currext (mirrors plain LaTeX `\input`
136  // semantics; Perl L2580 likewise only mutates them inside its
137  // handleoptions=true branch).
138  let prevname = if options.handleoptions && lookup_definition(&T_CS!("\\@currname"))?.is_some() {
139    gullet::do_expand(T_CS!("\\@currname"))?.to_string()
140  } else {
141    String::new()
142  };
143  let prevext = if options.handleoptions && lookup_definition(&T_CS!("\\@currext"))?.is_some() {
144    gullet::do_expand(T_CS!("\\@currext"))?.to_string()
145  } else {
146    String::new()
147  };
148  // This file will be treated somewhat as if it were a class
149  // IF as_class is true
150  // OR if it is loaded by such a class, and has withoptions true!!! (yikes)
151  if options.handleoptions && options.withoptions.is_some() {
152    with_vecdeque("@masquerading@as@class", |vdq_opt| {
153      if let Some(vdq) = vdq_opt {
154        if vdq.iter().any(|x| {
155          if let Stored::String(ref v) = x {
156            arena::with(*v, |str| str == prevname)
157          } else {
158            false
159          }
160        }) {
161          options.as_class = true;
162        }
163      }
164    });
165  }
166  if options.noltxml {
167    options.raw = true; // so it will be read as raw by Gullet.
168  }
169  let as_type = if options.as_class {
170    Cow::Borrowed("cls")
171  } else {
172    options
173      .extension
174      .as_ref()
175      .cloned()
176      .unwrap_or(Cow::Borrowed(""))
177  };
178
179  // If loading a class, store class options (Perl Package.pm lines 2561-2564).
180  // Perl L2561: `if ($astype eq 'cls' and $options{options})` — only
181  // (re)define `\@classoptionslist` when THIS cls load actually carries
182  // options. A nested `\LoadClass` with empty options (e.g. amsart →
183  // ams_core; our `*_cls.rs` bindings pass `Tokens!()` rather than forwarding
184  // the outer options as Perl's `withoptions=>1` does) must NOT clobber the
185  // document class's option list — babel iterates `\@classoptionslist`
186  // (`\bbl@foreach`, babel.sty L4270) to pick up a GLOBAL language option such
187  // as `\documentclass[french]{amsart}` and only then declares/loads that
188  // language. Clobbering it to empty silently dropped global babel languages
189  // for every bound class. Witness 1911.07001 (`[oneside,french,titlepage]
190  // {amsart}` + bare `\usepackage{babel}` → french.ldf must load).
191  //
192  // DIVERGENCE retained from Perl: still define `\@classoptionslist` as EMPTY
193  // for an option-less *document* class so babel's
194  // `\csname\ds@\@classoptionslist\endcsname` doesn't run away (the kernel
195  // default `\let\@classoptionslist\relax`; witness 2504.00009). We gate that
196  // on "no class options recorded yet", so it fires for the first/outermost
197  // class load but never clobbers an already-populated list on a nested load.
198  if as_type == "cls" {
199    for opt in &options.options {
200      push_value("class_options", arena::pin(opt))?;
201    }
202    let class_opts_str = options.options.join(",");
203    let have_recorded_opts = lookup_vecdeque("class_options")
204      .map(|v| !v.is_empty())
205      .unwrap_or(false);
206    if !class_opts_str.is_empty() || !have_recorded_opts {
207      def_macro(
208        T_CS!("\\@classoptionslist"),
209        None,
210        Tokens!(Explode!(class_opts_str)),
211        None,
212      )?;
213    }
214  }
215
216  // Compute the exact name based on the type
217  let filename = match &options.extension {
218    None => name.to_string(),
219    Some(ext) => s!("{}.{}", name, ext),
220  };
221  // Store the document class filename for xkeyval's isInClassFile check
222  if as_type == "cls" && options.handleoptions {
223    assign_value(
224      "document_class_filename",
225      filename.clone(),
226      Some(Scope::Global),
227    );
228  }
229  let current_options = options.options.join(",");
230  if !current_options.is_empty() {
231    if let Some(Stored::String(prevoptions)) = lookup_value(&s!("{filename}_loaded_with_options")) {
232      if arena::with(prevoptions, |prev_str| current_options != prev_str) {
233        let message = s!(
234          "Option clash for file {} with options {:?}, previously loaded with {:?}",
235          filename,
236          current_options,
237          prevoptions
238        );
239        Info!("unexpected", "options", message);
240      }
241    }
242  }
243
244  // TODO: This needs reorganization, bindings are not found as "files" in rust,
245  // we need to have a registry (we don't yet)
246
247  // Perl: early-stop if already loaded (checks request_loaded, name_loaded, etc.)
248  // This prevents double-loading and breaks circular loading chains.
249  // IMPORTANT: check BEFORE printing "Loading..." message to avoid spurious output.
250  //
251  // Per OXIDIZED_DESIGN #23: gate on the flag matching the load path
252  // we'll actually take. CRITICAL invariant: a binding `<file>.rs` is
253  // allowed to call `InputDefinitions(noltxml=>1)` for its same-named
254  // raw .sty/.cls/.def AFTER its own `_loaded` flag was set — the raw
255  // load gates on `_raw_loaded`, not `_loaded`. Examples: babel_sty
256  // → raw babel.sty; cite_sty → raw cite.sty.
257  let opt_noltxml = options.noltxml;
258  let opt_notex = options.notex;
259  // Rust-only `_load_attempted` flag: set in the miss-handler below to
260  // prevent retry loops while keeping `_loaded` reserved for genuine
261  // binding success. Without this split the `_loaded`-on-miss hack
262  // shadowed `require_package`'s `!_loaded && !_raw_loaded`
263  // post-call check, disabling `maybe_require_dependencies` for any
264  // package that had no binding (e.g. paper-local `jinstpub.sty`).
265  let already_handled = |fkey: &str| -> bool {
266    if opt_noltxml {
267      lookup_bool(&s!("{fkey}_raw_loaded"))
268    } else if opt_notex {
269      lookup_bool(&s!("{fkey}_loaded")) || lookup_bool(&s!("{fkey}_load_attempted"))
270    } else {
271      lookup_bool(&s!("{fkey}_loaded"))
272        || lookup_bool(&s!("{fkey}_raw_loaded"))
273        || lookup_bool(&s!("{fkey}_load_attempted"))
274    }
275  };
276  if !options.reloadable && already_handled(&filename) {
277    return Ok(());
278  }
279  // Also check without extension (Perl checks name_loaded too)
280  if !options.reloadable && name != filename && already_handled(name) {
281    return Ok(());
282  }
283
284  // Mark as loaded, then process the definitions.
285  //
286  // Suppress the "Loading X definitions" banner when we're inside a
287  // nested input_definitions call for the SAME file — that's the
288  // pattern where an .ltxml binding (e.g. babel_sty.rs) immediately
289  // calls `input_definitions(name, noltxml=true)` to raw-load the
290  // texlive .sty. Both frames used to print, producing the confusing
291  // duplicate `(Loading "babel.sty" definitions... (Loading
292  // "babel.sty" definitions...` trace. Now only the outermost frame
293  // announces — tracked per-filename via a state-value marker.
294  let banner_key = s!("__loading_banner__{filename}");
295  let this_frame_announces = crate::state::with_value(&banner_key, |v| v.is_none());
296  if this_frame_announces {
297    note_begin(&s!("Loading {:?} definitions", filename));
298    crate::state::assign_value(&banner_key, true, Some(crate::state::Scope::Global));
299  }
300
301  // Snapshot options.after / options.options BEFORE handleoptions consumes
302  // them so the fallback-binding recursive call (Step 3 below) can forward
303  // both to the fallback. Without this snapshot, mn1 → mn.cls.ltxml fallback
304  // ran with empty options/after and the user's `[epsfig]` was lost (see
305  // astro-ph0002213 root cause).
306  let original_after = options.after.clone();
307  let original_options = options.options.clone();
308  // Snapshot the GRANDPARENT's expl3 state BEFORE `\@pushfilename`'s
309  // `\ExplSyntaxOff` flips `_` to SUB. The post-load cleanup hook in
310  // load_tex_definitions uses this to know whether the calling context
311  // was in expl3 mode (so it can skip the `\ExplSyntaxOff` cleanup that
312  // would otherwise stick post-`\@popfilename`). Witness cluster:
313  // arXiv:2509.05997 / .07893 / .02344, 2510.13206/.13942/.17317
314  // (xsavebox + sys_load_backend + l3backend-dvips.def chain — minimal
315  // repro: \usepackage{xsavebox}).
316  let grandparent_in_expl3 = lookup_catcode('_') == Some(Catcode::LETTER);
317  // Strict-LaTeX-kernel order (latex.ltx `\@onefilewithoptions`, L15518-L15519):
318  //   \@pushfilename                        % capture OLD \@currname / \@currext
319  //   \xdef\@currname{ <new name> }         % then update to NEW
320  //
321  // We previously set \@currname/\@currext to the new file's name BEFORE
322  // calling `before_input_handle_options` (which performs the push). That
323  // captured the NEW name in the pushed triple, so `\@currnamestack` never
324  // held the empty `{}{}{<catcode>}` initial-state triple that
325  // expl3-code.tex's `\__file_tmp:w` recursion uses as its termination
326  // sentinel. Result: under raw expl3-code.tex load (LATEXML_NODUMP=1),
327  // the recursion ate past `\group_end:` into subsequent
328  // `\seq_new:N` / `\cs_new:Npn …` lines, producing the cs_end:
329  // cascade documented in .investigation/cs_end_bisect_round22/.
330  //
331  // Now: push first (uses current/OLD \@currname), then update inside
332  // before_input_handle_options (line 756-757). For the
333  // `handleoptions == false` path no push happens, so set the names
334  // directly here.
335  if options.handleoptions {
336    before_input_handle_options(&mut options, &prevname, &prevext, name, &as_type)?;
337    def_macro(
338      T_CS!(s!("\\{}.{}-h@@k", name, as_type)),
339      None,
340      options.after,
341      None,
342    )?;
343  }
344  // No `else` branch: Perl Package.pm L2580-2611 only mutates
345  // \@currname/\@currext inside the handleoptions=true block. The
346  // handleoptions=false path mirrors plain LaTeX `\input`, which leaves
347  // them untouched. Mutating them here breaks \@currnamestack
348  // discipline: a subsequent inner \RequirePackage's \@pushfilename
349  // captures the leaked name instead of the empty initial-state value,
350  // and expl3-code.tex's \__file_tmp:w stack walk over-reads.
351  // Witnesses: 0805.4519 (inputenc+ansinew), 1705.00041
352  // (\usetikzlibrary{calligraphy}+spath3+expl3).
353
354  if !current_options.is_empty() {
355    assign_value(
356      &s!("{}_loaded_with_options", filename),
357      current_options,
358      Some(Scope::Global),
359    );
360  }
361
362  // Track loaded files in \@filelist BEFORE loading (Perl: Package.pm calls
363  // \@addtofilelist before reading the file, so \@filelist is available inside)
364  if options.handleoptions && lookup_definition(&T_CS!("\\@addtofilelist"))?.is_some() {
365    digest(Tokens!(
366      T_CS!("\\@addtofilelist"),
367      T_BEGIN!(),
368      Explode!(filename),
369      T_END!()
370    ))?;
371  }
372
373  // Skip loading entirely if already loaded (unless reloadable)
374  // This prevents double-loading when e.g. smfart calls load_class("amsart")
375  // after the binding already set the _loaded flag.
376  // Per OXIDIZED_DESIGN #23: gate by the load path's flag — same
377  // path-aware logic as the early-skip above. Allows a binding to
378  // load its same-named raw counterpart via `noltxml=>1`.
379  if !options.reloadable && already_handled(&filename) {
380    if this_frame_announces {
381      note_end(&s!("Loading {:?} definitions", filename));
382      crate::state::assign_value(
383        &banner_key,
384        crate::common::store::Stored::None,
385        Some(crate::state::Scope::Global),
386      );
387    }
388    return Ok(());
389  }
390
391  // Catch Fatal errors during binding loading (e.g., token limit exceeded during
392  // expl3 kernel loading). Convert to non-fatal so document processing continues.
393  let is_binding = if options.noltxml {
394    false
395  } else {
396    match _load_binding(false, &filename, options.reloadable).and_then(|ext| {
397      if ext {
398        Ok(true)
399      } else {
400        _load_binding(true, &filename, options.reloadable)
401      }
402    }) {
403      Ok(v) => v,
404      Err(e) => {
405        Error!(
406          "unexpected",
407          &filename,
408          s!("Error loading binding for '{}': {}", filename, e)
409        );
410        // Mark as loaded even on error to prevent re-loading via raw path
411        assign_value(&s!("{filename}_loaded"), true, Some(Scope::Global));
412        false
413      },
414    }
415  };
416  let mut is_found_raw = false;
417  if is_binding {
418    // We found and loaded a binding successfully, mark it as such.
419    // Perl Package.pm::loadLTXML L2315-2316 sets TWO flags: `$request`_loaded
420    // (e.g. `color.sty_loaded`) AND `$ltxname`_loaded (`color.sty.ltxml_loaded`),
421    // where `.ltxml` is the suffix of the Perl binding file. Rust's port
422    // keeps only the former — `.ltxml` is not a suffix in the Rust world, so
423    // binding-vs-raw-tex distinction is queryable via `*_loaded` directly.
424    // See OXIDIZED_DESIGN.md. Callers of the legacy `.ltxml_loaded` form
425    // must be migrated to `_loaded`.
426    // Per OXIDIZED_DESIGN #23: binding success → `<filename>_loaded`.
427    // Raw load tracks separately via `<filename>_raw_loaded` (see
428    // load_tex_definitions). The `_found_loaded` Rust-only flag is
429    // dropped — read sites check `_loaded || _raw_loaded` instead.
430    let loaded_flag = format!("{filename}_loaded");
431    assign_value(&loaded_flag, true, Some(Scope::Global));
432    // Also set the Perl-equivalent `<filename>.ltxml_loaded` flag —
433    // some callers (e.g. require_package's deps-scan gate) need to
434    // distinguish binding-loaded from raw-loaded. Without this,
435    // paper-bundled .sty files that load natbib (which has a
436    // binding) re-trigger deps-scan on natbib.sty, which re-finds
437    // \usepackage{natbib} in natbib.sty's own warning text, looping
438    // infinitely. Witness 2111.01269 (TIMEOUT from natbib deps loop).
439    assign_value(&s!("{filename}.ltxml_loaded"), true, Some(Scope::Global));
440    // Perl L2326: Let(T_CS('\ver@'.$trequest), T_CS('\fmtversion'), 'global');
441    // Set \ver@name.ext to \fmtversion so LaTeX's \RequirePackage guard works.
442    // Without this, \RequirePackage date checks fail and packages get re-loaded.
443    if options.handleoptions {
444      let ver_cs = T_CS!(s!("\\ver@{}", filename));
445      if lookup_definition(&ver_cs).ok().flatten().is_none() {
446        let fmtversion_cs = T_CS!("\\fmtversion");
447        let_i(&ver_cs, &fmtversion_cs, Some(Scope::Global));
448      }
449    }
450  } else {
451    // We're inverting the control flow, because it is near-instant to check whether we have an
452    // available binding dispatcher, in both contributed and core binding names
453    // Now that we have ensured there is no compiled target of this name, we can start the file
454    // system search dance, call to kpsewhich, etc.
455    //
456    // Perl Package.pm FindFile search order (L2109-2139):
457    //   1. .ltxml binding (handled above by load_binding/load_external_binding)
458    //   2. Raw TeX in search paths, BUT only if INTERPRETING_DEFINITIONS is true (i.e. we're inside
459    //      recursive loading from another raw TeX file)
460    //   3. FindFile_fallback — strip version suffixes, find generic .ltxml binding (e.g.
461    //      icml2024.sty → icml.sty.ltxml)
462    //   4. Raw TeX in search paths (without INTERPRETING_DEFINITIONS gate)
463    //   5. kpsewhich
464    //
465    // This ordering ensures versioned-package fallback bindings take priority
466    // over raw .sty files that may contain layout checks (like ICML's \ifdim
467    // page-margin checks) that produce spurious warnings.
468    let interpreting = lookup_bool_sym(crate::pin!("INTERPRETING_DEFINITIONS"));
469
470    // Step 2: If we're already interpreting raw TeX definitions, look for the file directly.
471    // Perl Package.pm L2117-2119: `pathname_find($file, paths => $paths)` —
472    // LOCAL PATHS ONLY, no kpsewhich. Rust must mirror this: kpsewhich
473    // here would short-circuit Step 3 (fallback ltxml) for any TeX-Live-
474    // shipped raw file. Witness: `\RequirePackage{caption3}` from raw
475    // floatrow.sty — Perl finds caption3.sty NOT in user paths, falls
476    // through to Step 3 → caption.sty.ltxml. Rust with kpsewhich here
477    // returned the real caption3.sty from TL, raw-loading it and
478    // triggering the `\DeclareCaptionFormat{hang}[#1#2#3\par]{...}`
479    // PARAM-leak cascade (arXiv:2506.19291: Rust=30 vs Perl=2).
480    let found_raw = if interpreting && !options.notex {
481      find_file(
482        &filename,
483        Some(FindFileOptions {
484          forbid_ltxml:      options.noltxml,
485          notex:             false,
486          ext_type:          options.extension.as_ref().cloned(),
487          search_paths_only: true,
488        }),
489      )
490    } else {
491      None
492    };
493
494    // Step 3: Try fallback (strip version suffixes / dir prefix) before raw TeX.
495    // Perl Package.pm L2118-2121: FindFile_fallback.
496    //
497    // Design policy: bindings ALWAYS win over local raw .sty/.cls files.
498    // The `.rs` bindings are hand-tuned for the conversion, so if a
499    // fallback name resolves to a registered binding we dispatch there
500    // unconditionally. Raw TeX is the last-resort path (Step 4).
501    //
502    // Two flavors are recorded via [`FallbackKind`] for informational
503    // log messages only — both always fire when the binding exists:
504    //   - Versioned: suffix/prefix actually stripped (Perl-faithful). Drivers: 1206.0536 (mysvjour3
505    //     → svjour3), astro-ph0005021 (./aaspp4 → ./aaspp — aaspp4.sty ships locally with plain-TeX
506    //     `\startdata`; the engine's alignment-aware binding still wins, matching Perl).
507    //   - BasenameOnly: only directory prefix removed. Rust-specific extension keyed to our
508    //     contrib-binding registry. Drivers: 2105.02087 (misc/ieeetran → IEEEtran binding);
509    //     2405.18387 (assets/equations → equations binding, because we ship a tuned binding for
510    //     this name).
511    let found_raw = if found_raw.is_some() {
512      found_raw
513    } else if !options.noltxml {
514      if let Some((fallback, _kind)) = find_file_fallback(name, &as_type) {
515        Info!(
516          "fallback",
517          name,
518          s!("Interpreted as versioned package, falling back to {fallback}")
519        );
520        // Load the fallback binding — use reloadable since we already marked original as "loaded"
521        let ext_suffix = if as_type == "sty" { ".sty" } else { ".cls" };
522        let fallback_name = fallback.trim_end_matches(ext_suffix).to_string();
523        // Forward the original options + after-hook so fallback bindings see
524        // user-supplied class/package options (Perl-faithful: in Perl FindFile
525        // returns a path and the caller's options/after stay attached to the
526        // ORIGINAL `\@currname`-frame). Without this, `\documentstyle[epsfig]{mn1}`
527        // fell back to mn.cls.ltxml with empty options → mn.cls's option-handler
528        // never saw `epsfig` → `\compat@loadpackages` after-hook never fired
529        // → `\psfig` undefined. Witness: astro-ph0002213.
530        let fb_result = input_definitions(&fallback_name, InputDefinitionOptions {
531          extension: Some(Cow::Borrowed(if as_type == "sty" { "sty" } else { "cls" })),
532          options: original_options,
533          after: original_after,
534          handleoptions: options.handleoptions,
535          noerror: true,
536          reloadable: true,
537          ..InputDefinitionOptions::default()
538        });
539        if fb_result.is_ok() {
540          assign_value(&s!("{filename}_loaded"), true, Some(Scope::Global));
541          // NOTE: do NOT set `{filename}.ltxml_loaded` here. The
542          // fallback name is a DIFFERENT binding (e.g. article for
543          // myclass); the original `{filename}` (myclass.cls) has
544          // no binding. Setting it would suppress the downstream
545          // deps-scan check that picks up myclass's
546          // \RequirePackage{caption}.
547        }
548        None // fallback handled the loading; no raw file to load
549      } else {
550        None
551      }
552    } else {
553      None
554    };
555
556    // Step 4: Raw TeX in search paths (without INTERPRETING_DEFINITIONS gate)
557    // Perl Package.pm L2122-2125
558    //
559    // Per OXIDIZED_DESIGN #23: gate by `_raw_loaded` only — when a binding
560    // explicitly loads its raw counterpart via `noltxml=>1`, the binding's
561    // own `_loaded` flag is already set, but we MUST still proceed.
562    //
563    // EXCEPTION: if Step 3 (fallback ltxml binding) just succeeded, Perl's
564    // `if/elsif` flow (Package.pm:2118-2125) RETURNS on success and skips
565    // the raw-tex branch entirely. Rust's port uses sequential `let`
566    // bindings, so we must explicitly check `_loaded` here. Without this
567    // gate, `\RequirePackage{caption2}` loads `caption.sty.ltxml` via
568    // `find_file_fallback` (caption2 → caption strips trailing digit) AND
569    // then ALSO loads raw `caption2.sty`, which fires its
570    // `\@ifpackageloaded{caption}` mutual-exclusivity error. Same pattern
571    // applies to any package whose name ends in `[vV]?[-_.\d]+` and whose
572    // unsuffixed form has its own .ltxml binding.
573    let found_raw = if found_raw.is_some() {
574      found_raw
575    } else if lookup_bool(&s!("{filename}_loaded")) {
576      // Fallback ltxml binding already loaded — don't double-load the raw.
577      None
578    } else if !options.notex && (options.reloadable || !lookup_bool(&s!("{filename}_raw_loaded"))) {
579      // Perl Package.pm L2121-2125 + L2131-2136: combined raw-search
580      // step. Tries local paths first, then kpsewhich. Mirrors Perl's
581      // Step 4 (`!interpreting` local raw) PLUS Step 5 (kpsewhich
582      // unconditionally — note Perl's kpsewhich block lacks the
583      // interpreting gate). The previous `!interpreting` guard here
584      // was wrong: Step 2 now uses `search_paths_only=true`, so
585      // under interpreting=true we still need kpsewhich for raw
586      // files that have no fallback ltxml binding.
587      find_file(
588        &filename,
589        Some(FindFileOptions {
590          forbid_ltxml:      options.noltxml,
591          notex:             false,
592          ext_type:          options.extension.as_ref().cloned(),
593          search_paths_only: options.searchpaths_only,
594        }),
595      )
596    } else {
597      None
598    };
599
600    if let Some(file) = found_raw {
601      is_found_raw = true;
602      // The raw load itself sets `<filename>_raw_loaded` via
603      // load_tex_definitions (per OXIDIZED_DESIGN #23). Read sites
604      // check `_loaded || _raw_loaded` to detect "any load happened".
605      load_tex_definitions(
606        &filename,
607        &file,
608        options.reloadable,
609        options.at_letter,
610        grandparent_in_expl3,
611      )?;
612    } else if !lookup_bool(&s!("{filename}_loaded")) && !lookup_bool(&s!("{filename}_raw_loaded")) {
613      if options.noerror {
614        // With noerror: don't mark as loaded and return Err so callers can
615        // try fallback names (e.g. tikzlibrary → pgflibrary). Matches Perl's
616        // InputDefinitions which returns undef on not-found even with noerror=>1.
617        if this_frame_announces {
618          note_end(&s!("Loading {:?} definitions", filename));
619          crate::state::assign_value(
620            &banner_key,
621            crate::common::store::Stored::None,
622            Some(crate::state::Scope::Global),
623          );
624        }
625        return Err(s!("File not found: {}", filename).into());
626      }
627      // Perl Package.pm L2679 / L2715: maybeRequireDependencies($name, $type)
628      // is invoked when InputDefinitions returned undef ($success false).
629      // We mirror that here in the miss-handler, which is the only point
630      // where we know neither binding nor raw load occurred. Doing the
631      // dependency-scan BEFORE marking `_load_attempted` keeps the call
632      // exactly once-per-package and lets paper-local `.sty` files
633      // (e.g. jinstpub.sty bundling natbib + amsmath dependencies) wire
634      // up their transitively-bound prerequisites even when raw .sty
635      // loading is disabled (`INCLUDE_STYLES=false`, the default).
636      let scan_type =
637        options
638          .extension
639          .as_deref()
640          .unwrap_or(if options.as_class { "cls" } else { "sty" });
641      maybe_require_dependencies(name, scan_type);
642      // Rust-only retry guard: prevents re-attempting a missing file in
643      // a loop (raw TeX repeatedly calling \RequirePackage). Use a
644      // dedicated `_load_attempted` flag — NOT `_loaded` — so the
645      // post-input_definitions success check in `require_package`
646      // remains honest about whether anything actually loaded.
647      assign_value(&s!("{filename}_load_attempted"), true, Some(Scope::Global));
648      Warn!(
649        "missing_file",
650        name,
651        s!(
652          "Can't find binding or file for '{filename}'. \
653          No dispatcher entry and no raw file found on disk."
654        )
655      );
656    }
657  }
658
659  if options.handleoptions {
660    if is_binding || is_found_raw {
661      digest(T_CS!(s!("\\{name}.{as_type}-h@@k")))?;
662    }
663    // Always restore @currname/@currext and pop filename stack,
664    // even when no binding was found, to keep the stack balanced.
665    // Note: @popfilename uses \gdef to restore @currname/@currext from the stack,
666    // so it takes precedence. We also set them with def_macro as a fallback
667    // (matches Perl Package.pm lines 2635-2637).
668    if !prevname.is_empty() {
669      def_macro(
670        T_CS!("\\@currname"),
671        None,
672        Tokens!(ExplodeText!(prevname)),
673        None,
674      )?;
675    }
676    if !prevext.is_empty() {
677      def_macro(
678        T_CS!("\\@currext"),
679        None,
680        Tokens!(ExplodeText!(prevext)),
681        None,
682      )?;
683    }
684    // Perl-faithful: Package.pm:2637 —
685    //   Digest(($pushpop ? T_CS('\@popfilename') : T_CS('\lx@popfilename')));
686    // Pair with the dispatched push above. Using `\@popfilename` (dump's
687    // expl3-wrapped) when both push/pop are defined; else `\lx@popfilename`
688    // (LaTeXML safe internal). The push site re-checks `\@pushfilename` and
689    // `\@popfilename` definedness independently (state may have changed
690    // mid-load); here we re-check too rather than threading a flag.
691    let pop_use_expl = lookup_definition(&T_CS!("\\@pushfilename"))?.is_some()
692      && lookup_definition(&T_CS!("\\@popfilename"))?.is_some();
693    if pop_use_expl {
694      digest(T_CS!("\\@popfilename"))?;
695    } else {
696      digest(T_CS!("\\lx@popfilename"))?;
697    }
698    // Verify @currname was correctly restored, and force-fix if not
699    let restored_name = if lookup_definition(&T_CS!("\\@currname"))?.is_some() {
700      do_expand(T_CS!("\\@currname"))?.to_string()
701    } else {
702      String::new()
703    };
704    if !prevname.is_empty() && restored_name != prevname {
705      // @popfilename may have popped a stale entry; force correct value
706      def_macro(
707        T_CS!("\\@currname"),
708        None,
709        Tokens!(ExplodeText!(prevname)),
710        Some(ExpandableOptions {
711          scope: Some(Scope::Global),
712          ..ExpandableOptions::default()
713        }),
714      )?;
715    }
716    if !prevext.is_empty() {
717      let restored_ext = if lookup_definition(&T_CS!("\\@currext"))?.is_some() {
718        do_expand(T_CS!("\\@currext"))?.to_string()
719      } else {
720        String::new()
721      };
722      if restored_ext != prevext {
723        def_macro(
724          T_CS!("\\@currext"),
725          None,
726          Tokens!(ExplodeText!(prevext)),
727          Some(ExpandableOptions {
728            scope: Some(Scope::Global),
729            ..ExpandableOptions::default()
730          }),
731        )?;
732      }
733    }
734    reset_options()?;
735  }
736  // No handleoptions=false cleanup needed: we never mutated
737  // \@currname/\@currext on that path (matching Perl).
738  if this_frame_announces {
739    note_end(&s!("Loading {:?} definitions", filename));
740    crate::state::assign_value(
741      &banner_key,
742      crate::common::store::Stored::None,
743      Some(crate::state::Scope::Global),
744    );
745  }
746  Ok(())
747}

749/// loads a binding from the main binding dispatcher, if available+found
750pub fn load_binding(file: &str) -> Result<bool> { _load_binding(true, file, false) }
751/// loads a binding from an external binding dispatcher, if available+found
752pub fn load_external_binding(file: &str) -> Result<bool> { _load_binding(false, file, false) }
753// in the spirit of Perl's Package::loadLTXML
754fn _load_binding(internal: bool, request: &str, reloadable: bool) -> Result<bool> {
755  // Perl loadLTXML L2311-2313: skip if already loaded, unless reloadable
756  // (e.g. `\inputencoding{cp1251}` re-invokes cp1251.def to re-register
757  // DeclareInputText mappings after `set_input_encoding` reset them).
758  // OXIDIZED_DESIGN #23: binding load gates ONLY on the binding-specific
759  // `_loaded` flag (set on success below). A prior raw load
760  // (`_raw_loaded`) does NOT preclude the binding from loading — they
761  // are independent paths. Mirrors Perl `loadLTXML` (Package.pm L2311).
762  let loaded_key = s!("{request}_loaded");
763  if !reloadable && lookup_bool(&loaded_key) {
764    return Ok(true);
765  }
766
767  // Re-entrance guard for binding loads: track which bindings are
768  // currently mid-load on this thread, so that if a binding body
769  // transitively calls require_package(SAME_NAME) (e.g. via
770  // \citet → OmniBus closure → require_package(natbib) firing
771  // during natbib's own ProcessOptions chain), we short-circuit
772  // instead of re-entering the dispatcher and looping. Task #260.
773  thread_local! {
774    static IN_PROGRESS: std::cell::RefCell<rustc_hash::FxHashSet<String>> =
775      std::cell::RefCell::new(rustc_hash::FxHashSet::default());
776  }
777  let request_key = request.to_string();
778  let already_loading = IN_PROGRESS.with(|s| s.borrow().contains(&request_key));
779  if already_loading {
780    Warn!(
781      "recursion",
782      &request_key,
783      s!(
784        "Binding-load re-entrance for '{}' (transitive require_package \
785         during its own LoadDefinitions). Short-circuiting to break the \
786         loop; binding's pending side-effects (DefMacro, Let, etc.) \
787         will still complete in the outer frame.",
788        request_key
789      )
790    );
791    return Ok(true);
792  }
793
794  let taken_dispatcher = if internal {
795    get_bindings_dispatch()
796  } else {
797    get_extra_bindings_dispatch()
798  };
799  match taken_dispatcher {
800    Some(ref dispatcher) => {
801      // Perl `Package.pm:loadLTXML L2318` wraps the binding-load body in
802      // `local $UNLOCKED = 1`, allowing bindings to override prior
803      // (locked) definitions. The guard auto-pops on drop.
804      let _unlock_guard = crate::common::local_assignments::local_state_unlocked_guard(true);
805      // Mark in-progress for the duration of this dispatcher call.
806      IN_PROGRESS.with(|s| {
807        s.borrow_mut().insert(request_key.clone());
808      });
809      struct InProgressGuard(String);
810      impl Drop for InProgressGuard {
811        fn drop(&mut self) {
812          let key = self.0.clone();
813          IN_PROGRESS.with(|s| {
814            s.borrow_mut().remove(&key);
815          });
816        }
817      }
818      let _in_progress_guard = InProgressGuard(request_key.clone());
819      let result_opt = dispatcher(request);
820      match result_opt {
821        Some(result) => {
822          // Here and only here we are certain we have binding support.
823          // Preemptively mark as loaded to avoid recursion.
824
825          // Mark binding as loaded (raw `<request>_raw_loaded` is tracked
826          // separately by load_tex_definitions). Per OXIDIZED_DESIGN #23.
827          assign_value(&loaded_key, true, Some(Scope::Global));
828          match result {
829            Ok(()) => Ok(true),
830            Err(e) => Err(e),
831          }
832        },
833        None => Ok(false),
834      }
835    },
836    None => Ok(false),
837  }
838}

840// Factor out handling and passing loading options from input_content,
841// to simplify main routine
842fn before_input_handle_options(
843  options: &mut InputDefinitionOptions,
844  prevname: &str,
845  prevext: &str,
846  name: &str,
847  as_type: &str,
848) -> Result<()> {
849  // Perl-faithful translation of Package.pm:2578-2591:
850  //
851  //   my $pushpop = LookupDefinition(T_CS('\@pushfilename'))
852  //              && LookupDefinition(T_CS('\@popfilename'));
853  //   if ($pushpop) {
854  //     Digest(Tokens(T_CS('\@pushfilename'),
855  //         T_BEGIN, T_END, T_BEGIN, T_END, T_BEGIN, Explode($name), T_END));
856  //   } else {
857  //     Digest(T_CS('\lx@pushfilename'));
858  //   }
859  //
860  // The 3 trailing brace-arg pairs `{}{}{name}` feed
861  // `\@expl@push@filename@aux@@` (which the dump's `\@pushfilename`
862  // body chains into) — that aux takes 3 args. Without them it reads
863  // 3 garbage tokens from the input stream, corrupting the
864  // `\g__hook_name_stack_seq` push. Subsequent `\@popfilename`
865  // then sees an empty/corrupt seq, fires `\msg_error:nn{hooks}{extra-pop-label}`,
866  // whose `\use:e` (=`\edef`) chain expands `\q_no_value` and triggers
867  // recursion-detect. See docs/sandbox_failures_SYNC_STATUS.md
868  // "\q_no_value cascade" for the full investigation.
869  let push_defined = lookup_definition(&T_CS!("\\@pushfilename"))?.is_some();
870  let pop_defined = lookup_definition(&T_CS!("\\@popfilename"))?.is_some();
871  if push_defined && pop_defined {
872    let mut pushtoks = vec![
873      T_CS!("\\@pushfilename"),
874      T_BEGIN!(),
875      T_END!(),
876      T_BEGIN!(),
877      T_END!(),
878      T_BEGIN!(),
879    ];
880    pushtoks.extend(Explode!(name));
881    pushtoks.push(T_END!());
882    digest(Tokens::new(pushtoks))?;
883  } else {
884    digest(T_CS!("\\lx@pushfilename"))?;
885  }
886
887  // For \RequirePackageWithOptions, pass the options from the outer class/style to the inner one.
888  if let Some(with_options_to_pass) = options.withoptions.take() {
889    if !prevname.is_empty() && has_value(&s!("opt@{}.{}", prevname, prevext)) {
890      // Only pass those class options that are declared by the package!
891      let mut topass = Vec::new();
892      with_vecdeque("@declaredoptions", |vdq_opt| {
893        if let Some(declared_options) = vdq_opt {
894          for op in with_options_to_pass.into_iter() {
895            if declared_options.iter().any(|x| {
896              if let Stored::String(val) = x {
897                arena::with(*val, |str| str == op)
898              } else {
899                false
900              }
901            }) {
902              topass.push(op)
903            }
904          }
905        }
906      });
907      if !topass.is_empty() {
908        pass_options(name, as_type, topass)?;
909      }
910    }
911  }
912  // Use letter-catcode (`ExplodeText`) for `\@currext` / `\@currname` so
913  // they match `\@pkgextension`-style build-time-tokenized macros under
914  // `\ifx`. Without this the catcodes diverge — `\@pkgextension` from a
915  // compile-time `DefMacro!("\\@pkgextension", "sty")` tokenizes "sty"
916  // as letters (default LaTeX catcode 11), but the previous `Explode!`
917  // used here produces OTHER catcode tokens, so kvoptions's
918  // `\ifx\@currext\@pkgextension` always returned false — vendor
919  // `\PackageError{kvoptions}{\ProcessLocalKeyvalOptions is intended
920  // for packages only}` then fired on every package that uses kvoptions
921  // (rerunfilecheck reaches this via the hyperref backend `.def` chain).
922  // Witnesses: arXiv:cond-mat/9611206, math/9904040, math/9904041.
923  def_macro(
924    T_CS!("\\@currname"),
925    None,
926    Tokens!(ExplodeText!(name)),
927    None,
928  )?;
929  def_macro(
930    T_CS!("\\@currext"),
931    None,
932    Tokens!(ExplodeText!(as_type)),
933    None,
934  )?;
935  // reset options (Note reset & pass were in opposite order in LoadClass ????)
936  reset_options()?;
937  pass_options(name, as_type, options.options.clone())?;
938
939  // Note which packages are pretending to be classes.
940  if options.as_class {
941    push_value("@masquerading@as@class", arena::pin(name))?;
942  }
943  let current_opt_val = with_vecdeque(&s!("opt@{}.{}", name, as_type), |vdq_opt| match vdq_opt {
944    Some(vdq) => {
945      let mut pieces = String::new();
946      for x in vdq.iter() {
947        if let Stored::String(val) = x {
948          arena::with(*val, |str| pieces.push_str(str));
949        }
950        pieces.push(',');
951      }
952      pieces.pop();
953      pieces
954    },
955    None => String::new(),
956  });
957  def_macro(
958    T_CS!(s!("\\opt@{}.{}", name, as_type)),
959    None,
960    Tokens!(Explode!(current_opt_val)),
961    None,
962  )?;
963  Ok(())
964}

976/// to the document (as opposed to pure definitions).

979/// and/or digestion will pull Tokens from that Mouth until it is
980/// exhausted, or closed.

983/// the TeX material explicitly, rather than referencing a file.
984/// In this case, the `literal` pseudo-protocal may be used.

987  match filepath {
988    // TODO: type => $options{type}, noltxml => 1

990    None => {

992      // elsif (!$options{noerror}) { Error('missing_file', $request, ..., ...); }`
993      // Recoverable Error, NOT Fatal. Pre-fix, the Rust port emitted a
994      // `fatal!(Package, MissingFile)` that terminated the conversion on any
995      // missing-but-non-critical input — over-fatal-izing relative to Perl.
996      if !options.noerror {
997        Error!(
998          "missing_file",
999          request,
1000          format!("Can't find TeX file {request}")
1001        );
1002      }
1003      Ok(())
1004    },

1008/// This is essentially the `\input` equivalent
1009///
1010/// we are most likely expecting to get actual content,
1011/// (possibly with definitions included, as well)
1012/// but might actually be getting pure definitions,
1013/// (like a proper style file)
1014/// in which case we may really want to load a binding.
1015/// Note that generic style files (non-latex) often have a .tex extension.
1016pub fn input(request: &str, options: InputOptions) -> Result<()> {
1017  // unwrap if in quotes \input{"file name"} — Perl parity:
1018  // `$request =~ s/^("+)(.+)\g1$/$2/;` (single-pass strip of a matching
1019  // leading+trailing run of quotes). The previous `while` loop checked
1020  // the unchanged `request`, which spun forever on any quoted input
1021  // since the replacement only touches `clean_req`.
1022  let clean_req = QUOTE_WRAPPED.replace(request, "$1");
1023  // HEURISTIC! First check if equivalent style file, but only under very specific circumstances
1024  // if pathname_is_literaldata(request) {
1025  //   let (dir, name, ftype) = pathname_split(request);
1026  //   let file = name;
1027  //   if !ftype.is_empty() {
1028  //     file += format!(".{}",ftype);
1029  //   }
1030  //   let path;
1031  //   // Firstly, check if we are going to OVERRIDE the requested raw .tex file
1032  //   // with a latexml binding to a style file.
1033  //   if ((dir.is_empty() && (ftype.is_empty() || (ftype == "tex"))  // No SPECIFIC directory, but
1034  // a raw tex file.       // AND, in preamble; SHOULD be style file, OR also if we can't find the
1035  // raw file.     && (LookupValue!("inPreamble") || !FindFile(file))
1036  //     && (path = FindFile(name, type => 'sty', notex => 1))) { // AND there IS such a style file
1037  //     Info!("ignore", request, stomach.get_gullet(),
1038  //       s!("Ignoring input of tex {}, using package {} instead", request, name));
1039  //     RequirePackage!(name); // Then override, assuming we'll find name as a package file!
1040  //     return;
1041  //   }
1042  // }
1043  // // Next special case: If we were currently reading a "known" style or binding file,
1044  // // then this file, even if .tex, must also be definitions rather than content.!!(?)
1045  // Check for *.latexml source-level bindings first — these are always handled
1046  // as definitions regardless of INTERPRETING_DEFINITIONS state.
1047  // Mirrors Perl's automatic .latexml file loading mechanism.
1048  if clean_req.ends_with(".latexml") {
1049    return input_definitions(&clean_req, InputDefinitionOptions::default());
1050  }
1051  if lookup_bool_sym(crate::pin!("INTERPRETING_DEFINITIONS")) {
1052    // Split a binding extension off the request so input_definitions sees
1053    // (name, extension) — matches Perl Package.pm `FindFile` / `Input`
1054    // semantics. Without the split, `find_file_fallback` runs with
1055    // `ext_type=""` and reconstructs `"<base>."` (no extension), which
1056    // never matches a registered binding. Witness: hep-ph9911514 — the
1057    // raw-loaded `elsartwb.sty` issues `\input elsart12\@ptsize.sty` →
1058    // `\input{elsart12.sty}`; the version-strip fallback (elsart12 →
1059    // elsart) needs `ext_type="sty"` to reconstruct `"elsart.sty"` for
1060    // the binding lookup. Perl recovers `\ack` cleanly via this path; the
1061    // earlier Rust port dropped the extension and the fallback never
1062    // resolved.
1063    let has_dir = clean_req.contains('/') || clean_req.contains('\\');

1716/// Perl Package.pm L2759-2796: maybeRequireDependencies
1717/// When a package/class file has no binding AND raw TeX loading is disabled,
1718/// scan the raw file for \RequirePackage/\usepackage/\LoadClass declarations

1720/// fallback that gives us the dependency chain without interpreting raw TeX.

1722// (L2759-L2796). Scan a raw .sty/.cls file for transitive
1723// `\RequirePackage`, `\usepackage`, and (for classes) `\LoadClass`
1724// declarations and route them through `require_package` / `load_class`

1726// file has no .ltxml binding.

1736  }

1747  // A candidate the regex finds in the text but that is NOT in that set is one

1765  // environment, so a `\usepackage`/`\RequirePackage` inside it is NEVER loaded
1766  // by LaTeX. The dep-scan must not anticipate it. Same "more-robust than Perl"
1767  // rationale as the macro-def-body skip below. Witness 1901.05713: thesis.sty
1768  // has a commented-out `\usepackage{hyperref}` inside `\begin{comment}`, which
1769  // the scan otherwise loaded — tripping cleveref's "must be loaded after
1770  // hyperref" `\AtBeginDocument` order-check (hyperref appears loaded though the

1777  // collected order matches Perl's call order to `$collect`.

1779    Lazy::new(|| Regex::new(r"\\RequirePackage\s*(?:\[([^\]]*)\])?\s*\{([^\}]*)\}").unwrap());
1780  static USE_RE: Lazy<Regex> =
1781    Lazy::new(|| Regex::new(r"\\usepackage\s*(?:\[([^\]]*)\])?\s*\{([^\}]*)\}").unwrap());
1782  static CLS_RE: Lazy<Regex> =
1783    Lazy::new(|| Regex::new(r"\\LoadClass\s*(?:\[([^\]]*)\])?\s*\{([^\}]*)\}").unwrap());

1785  // `{` — i.e. the brace that follows opens the macro BODY. Used to detect a
1786  // `\usepackage` that lives in a deferred macro body (loads only when the
1787  // macro is later expanded) vs one inside a load-time conditional.

1795  // macro NAME (without backslash). A `\def\<m>{… \RequirePackage{P} …}` body
1796  // is only truly deferred if `\<m>` is never invoked; the AMS-class idiom
1797  // `\def\@tempa{\RequirePackage{amsmath}}…\@tempa` invokes it immediately and
1798  // so DOES load P (witness ijnam.cls → amsmath → `{aligned}`, 1911.03415).

1803  // is BARE — `FindFile` glues on `.$type` itself per L2073-2076.

1805    file,
1806    Some(FindFileOptions {
1807      ext_type: Some(Cow::Owned(ext_type.to_string())),

1809      ..FindFileOptions::default()
1810    }),

1817  // their .cls inline via filecontents miss the dep-scan and downstream

1822    cached
1823  } else {

1825    // files (ISO-8859 with vendor copyright headers, e.g. cpc-hepnp.cls

1848  // `\usepackage` / `\RequirePackage` that sits in a DEFERRED macro-definition
1849  // body (a `\newcommand` / `\def`-family body) — it loads only if that macro

1851  // inside LOAD-TIME conditionals (`\IfFileExists{X.sty}{\usepackage{X}}`,
1852  // `\@ifundefined{..}{..}`, `\if…\usepackage…\fi`) DO execute during raw-load

1854  // a normally raw-loaded file it doesn't dep-scan at all, and the bundled deps

1856  //   - 1506.06200 — categorytheory.sty `\newcommand{\usediagrams}{\usepackage

1861  //     filter wrongly skipped this conditional; the def-body check is precise.
1862  // A `\usepackage` is deferred iff ANY enclosing `{…}` group is opened directly
1863  // by a `\newcommand`/`\def` definition header.
1864  // Is `\<name>` INVOKED (not merely defined) somewhere in the file? A bare
1865  // control-sequence occurrence not directly preceded by a `\…def`/`\let`
1866  // introducer counts as an invocation. Used to tell an executed deferred-load
1867  // idiom (`\def\@tempa{…\RequirePackage{P}…}\@tempa`) from a never-called one.

1872      let at = from + rel;
1873      from = at + pat.len();

1875      let after_ok = code[at + pat.len()..]
1876        .chars()
1877        .next()
1878        .is_none_or(|c| !c.is_ascii_alphabetic() && c != '@');
1879      if !after_ok {
1880        continue;
1881      }
1882      // Not a (re)definition: the chars just before `\name` aren't `…def`/`…let`.
1883      let before = code[at.saturating_sub(8)..at].trim_end();

1885        continue;
1886      }
1887      return true;
1888    }
1889    false
1890  };

1892    let bytes = code.as_bytes();
1893    let mut stack: Vec<usize> = Vec::new();
1894    let mut i = 0usize;
1895    while i < start {
1896      match bytes[i] {
1897        b'\\' => {
1898          i += 2;
1899          continue;
1900        },
1901        b'{' => stack.push(i),
1902        b'}' => {
1903          stack.pop();
1904        },
1905        _ => {},
1906      }
1907      i += 1;
1908    }

1910      let lo = ob.saturating_sub(400);
1911      let window = code.get(lo..ob).unwrap_or(&code[..ob]);
1912      // A `\def\<m>{…}` body only defers if `\<m>` is never invoked; an invoked
1913      // scratch macro (`\@tempa`) runs its body at load, so its require loads.
1914      if let Some(caps) = DEF_NAME_RE.captures(window) {
1915        return !is_invoked(&caps[1]);
1916      }
1917      // `\newcommand`/`\DeclareRobustCommand` user-command bodies stay deferred
1918      // (witness 1506.06200 `\newcommand{\usediagrams}{\usepackage{diagrams}}`).
1919      DEF_BODY_HEADER_RE.is_match(window)
1920    })
1921  };
1922  let top_level = |cap: &regex::Captures| -> bool {
1923    cap.get(0).map(|m| !in_macro_def_body(m.start())).unwrap_or(true)
1924  };
1925

1927  // here (was: drop a package `\RequirePackage`'d / `\usepackage`'d with two or
1928  // more DIFFERENT option sets, as the signature of a deferred require inside a
1929  // `\def`/`\DeclareOption` body, e.g. aa.cls's
1930  // `\DeclareOption{ascii}{\def\aa@inputenc{\RequirePackage[ascii]{inputenc}}}`
1931  // …`{utf8}{…[utf8]…}`). It over-fired on a package required in BOTH arms of a
1932  // load-time `\if…\else…\fi` with different options — mutually exclusive, so
1933  // exactly one branch loads and the package MUST be kept. Witness: rist.cls's
1934  // `\ifpdf \RequirePackage[pdftex,…]{hyperref} \else \RequirePackage[dvipdfm,…]
1935  // {hyperref} \fi` (1912.00781: hyperref dropped → `\url` undefined). The
1936  // heuristic was also provably REDUNDANT: it only ever saw TOP-LEVEL requires
1937  // (the captures feeding it were `top_level`-filtered), but aa.cls's inputenc
1938  // lives in a `\def`/`\DeclareOption` BODY — already dropped by the def-body
1939  // `top_level` skip below, so it never entered the conflicting set at all. Perl
1940  // has no such gate (L2767-2774 is a plain dedup); the def-body `top_level` skip
1941  // plus the executed-set gate cover the legitimate cases faithfully.
1942  static OPT_SPLIT: Lazy<Regex> = Lazy::new(|| Regex::new(r"\s*,\s*").unwrap());
1943

1955      // candidate whose `\usepackage` never executed anywhere — it was inside a
1956      // false conditional the raw-load skipped.

1959      }

1965    }
1966  };

1988        continue;
1989      }

1994    }

2001      .iter()
2002      .map(|(n, _)| n.as_str())
2003      .chain(packages.iter().map(|(n, _)| n.as_str()))

2006      "dependencies",

2042      }),
2043    )

2045    {

2048        .map(|s| OPT_SPLIT.split(s).map(|x| x.to_string()).collect())
2049        .unwrap_or_default();
2050      let _ = require_package(&pkg, RequireOptions {
2051        options: opts,
2052        ..RequireOptions::default()
2053      });
2054    }

2405/// Strip version/arxiv suffixes from package names to find existing bindings.
2406/// Returns the fallback filename (with extension) and which kind of fallback fired.
2407pub fn find_file_fallback(name: &str, ext_type: &str) -> Option<(String, FallbackKind)> {
2408  use regex::Regex;
2409  // Suffixes with separator (Perl @find_fallback_suffixes)
2410  let suffix_rx = Regex::new(
2411    r"(?i)[._-](arx|arxiv|conference|workshop|tmp|alternate|preprint|fixed|[vV]?[-_.\d]+|old|new|final|clean|mine|priv|rev|mod|modified|edited|custom|altered|rtx)$"
2412  ).ok()?;
2413  // Glued suffixes without separator
2414  let glued_rx = Regex::new(r"(?i)([vV]?[-_.\d]+|arxiv)$").ok()?;
2415  // Prefixes. Perl Package.pm L2182: `^((?:rw|my|preprint)[-_.]?)` —
2416  // separator is OPTIONAL, so `mysvjour3` strips to `svjour3` (not just
2417  // `mysvjour`). Caught on arxiv 1206.0536 (\documentclass{mysvjour3}).
2418  let prefix_rx = Regex::new(r"(?i)^((?:rw|my|preprint)[-_.]?)").ok()?;
2419
2420  // Strip a leading directory path (Perl Package.pm L2167-2170: FindFile_fallback
2421  // calls `pathname_name($name)` first, so e.g. `\documentclass{./sty/IEEEtran}`
2422  // routes the basename `IEEEtran` through the binding-name registry. Without
2423  // this, `IEEEtran.cls.ltxml` is missed because `./sty/IEEEtran.cls.ltxml`
2424  // never matches the @ltxml_paths registry. Driver paper: arXiv:1308.6663.
2425  let basename = pathname::file_name(name);
2426  let mut base = if basename.is_empty() {
2427    name.to_string()
2428  } else {
2429    basename
2430  };
2431  let dir_stripped = base != name;
2432  let mut suffix_stripped = false;
2433  // Iteratively strip suffixes, then glued, then prefixes.
2434  //
2435  // Perl's FindFile_fallback (Package.pm:2174) version-strips the FULL `$file`
2436  // *with its directory prefix intact*, so the prefix regex `^(rw|my|preprint)`
2437  // — anchored at the very START of the string — never matches a name that
2438  // begins with a directory (e.g. `sty/myunits` starts with `sty/`, not `my`).
2439  // We strip the directory FIRST (to allow a basename-exact binding match like
2440  // `misc/ieeetran` → IEEEtran), so we must NOT then apply the `^`-anchored
2441  // prefix strip to the basename, or `sty/myunits` wrongly becomes `units` and
2442  // loads the stock units.sty instead of the paper-local myunits.sty (which
2443  // defines `\T`/`\fC`/`\Cm` via its `\defUnit` mechanism). Witness 1702.05093.
2444  // The suffix/glued strips ARE `$`-anchored, so they still match the tail of a
2445  // dir-prefixed name in Perl (`./aaspp4` → `./aaspp`); keep applying them.
2446  loop {
2447    if let Some(m) = suffix_rx.find(&base) {
2448      base = base[..m.start()].to_string();
2449      suffix_stripped = true;
2450      continue;
2451    }
2452    if let Some(m) = glued_rx.find(&base) {
2453      base = base[..m.start()].to_string();
2454      suffix_stripped = true;
2455      continue;
2456    }
2457    if !dir_stripped {
2458      if let Some(m) = prefix_rx.find(&base) {
2459        base = base[m.end()..].to_string();
2460        suffix_stripped = true;
2461        continue;
2462      }
2463    }
2464    break;
2465  }
2466
2467  if !suffix_stripped && !dir_stripped {
2468    return None;
2469  }
2470  if base.is_empty() || base == name {
2471    return None;
2472  }
2473
2474  let kind = if suffix_stripped {
2475    FallbackKind::Versioned
2476  } else {
2477    FallbackKind::BasenameOnly
2478  };
2479
2480  let fallback_filename = format!("{base}.{ext_type}");
2481  // Check if fallback binding exists
2482  if load_binding(&fallback_filename).unwrap_or(false) {
2483    // Binding exists but was loaded by the check — it's OK, the caller will mark loaded
2484    Some((fallback_filename, kind))
2485  } else if load_external_binding(&fallback_filename).unwrap_or(false) {
2486    Some((fallback_filename, kind))
2487  } else {
2488    None
2489  }
2490}

2492fn find_file_aux(file: &str, options: &FindFileOptions) -> Option<String> {

2494  let cached = lookup_string(&s!("{}_contents", file));
2495  if !cached.is_empty() {
2496    return Some(file.to_string());
2497  }
2498  if pathname::is_absolute(file) {

2522    // (1) we prefer .ltxml bindings, if present

2532    // has an entry for this file, consider it "found". This is how Perl's
2533    // FindFile discovers pgfsys-latexml.def.ltxml etc.

2542    //     (1709.05096 / AIAA → 60s wall-clock SIGABRT in the autoload- trapped-by-abstract loop).

2544    // `notex=true` (i.e. caller wants binding-only search, not a real

2548    // and `\ifeof` returns true, masking the file as missing. Mirrors
2549    // Perl `pathname_find`: only `noltxml=>0,notex=>1` returns binding
2550    // names; the disk-search variant only resolves real files.
2551    // Triggered by 2026-04-26 t1enc.def-cascade investigation: raw
2552    // fonttext.ltx's `\input  {t1enc.def}` opens via raw `\openin` /
2553    // `\IfFileExists`; without this gate find_file returned literal
2554    // "t1enc.def" → empty mouth → kernel's `\@missingfileerror` → 1M
2555    // TooManyErrors during latex.ltx dump-build.
2556    if !options.forbid_ltxml && options.notex {
2557      if lookup_bool(&s!("{file}_binding_available")) {
2558        return Some(file.to_string());
2559      }
2560      // Check the compile-time binding registries from latexml_package and
2561      // latexml_contrib for ANY (name, ext) pair that matches `file` (split
2562      // on the FIRST `.`, mirroring `dispatch()`'s split rule so multi-dot
2563      // names like `pgfmath.code.tex` resolve as `("pgfmath", "code.tex")`).
2564      if let Some((base, ext)) = file.split_once('.') {
2565        // Perl pathname_find L383-389: strict-case first, then case-insensitive
2566        // fallback (mirrors the dispatcher's lookup). Without this, requests
2567        // like `find_file("jhep.cls", notex=true)` would miss `("JHEP","cls")`
2568        // entries that derive from Perl's `JHEP.cls.ltxml` filename.
2569        let exact = crate::state::get_binding_names()
2570          .iter()
2571          .any(|slice| slice.iter().any(|(n, e)| *n == base && *e == ext));
2572        let nocase = exact
2573          || crate::state::get_binding_names().iter().any(|slice| {
2574            slice
2575              .iter()
2576              .any(|(n, e)| n.eq_ignore_ascii_case(base) && e.eq_ignore_ascii_case(ext))
2577          });
2578        if nocase {
2579          return Some(file.to_string());
2580        }
2581      }
2582    } else if !options.forbid_ltxml {
2583      // Narrow notex=false (disk-search) fallback: ONLY honor explicit
2584      // `<file>_binding_available` runtime flags, NOT the broad compile-time
2585      // registry. Mirrors Perl `\openin` calling default-args FindFile —
2586      // see `TeX_FileIO.pool.ltxml:50-64` "we SHOULD find an .ltxml version!"
2587      //
2588      // Use case: pgf.sty's pgfsys.code.tex `\pgfutil@InputIfFileExists{\pgfsysdriver}`
2589      // → `\pgfutil@IfFileExists{pgfsys-latexml.def}` → `\openin` with notex=false.
2590      // The openin impl in tex_file_io.rs creates an empty Mouth on
2591      // Mouth::create failure, so \ifeof=false → pgf inputs the driver.
2592      // pgf_sty.rs sets `pgfsys-latexml.def_binding_available=true` to
2593      // enable this; without the flag we don't fake-find arbitrary binding
2594      // names (which would re-introduce the t1enc.def `\@missingfileerror`
2595      // cascade documented above the notex=true branch).
2596      if lookup_bool(&s!("{file}_binding_available")) {
2597        return Some(file.to_string());
2598      }
2599    }
2600    // Perl L2123-2125: `elsif !notex && !interpreting && pathname_find($file,
2601    // paths=>$paths)` — search local paths for raw TeX.
2602    // (Rust does not yet honour `INTERPRETING_DEFINITIONS` — minor TODO,
2603    //  acknowledged in the audit.)
2604    if !options.notex {
2605      if let Some(path) = pathname::find(file, PathnameFindOptions {
2606        paths: Some(paths),
2607        ..PathnameFindOptions::default()
2608      }) {
2609        return Some(path);
2610      }
2611    }
2612    // Perl L2131-2136: build kpsewhich candidate list:
2613    //   @candidates = ( "$file.ltxml" if !noltxml && !nopaths,
2614    //                   $file        if !notex );
2615    //   if (!searchpaths_only) && pathname_kpsewhich(@candidates) → -f $result
2616    // Perl gates the kpsewhich call only on `!searchpaths_only`; `notex`
2617    // and `noltxml` instead control which candidate names are tried.
2618    if options.search_paths_only {
2619      return None;
2620    }
2621    let mut candidates: Vec<String> = Vec::new();
2622    if !options.forbid_ltxml {
2623      // Perl `!nopaths` (REMOTE_REQUEST) gate not yet modeled in Rust;
2624      // we always include the .ltxml candidate when ltxml is allowed.
2625      candidates.push(s!("{}.ltxml", file));
2626    }
2627    if !options.notex {
2628      candidates.push(file.to_string());
2629    }
2630    if candidates.is_empty() {
2631      return None;
2632    }
2633    let refs: Vec<&str> = candidates.iter().map(|s| s.as_str()).collect();
2634    match pathname::kpsewhich(&refs) {
2635      // Perl L2136: `(-f $result ? $result : undef)` — re-confirm existence.
2636      Some(p) if Path::new(&p).exists() => Some(p),
2637      _ => None,
2638    }
2639  }
2640}

2690/// Define a named color (Perl: DefColor).
2691/// Stores as color_{name} and also defines \\color@{name} macro.
2692pub fn def_color(
2693  name: &str,
2694  color: &crate::common::color::Color,
2695  scope: Option<Scope>,
2696) -> Result<()> {
2697  use crate::common::color;
2698  // Check ifglobalcolors — Perl: $scope='global' if lookupDefinition(\ifglobalcolors) &&
2699  // IfCondition(\ifglobalcolors) Guard with lookup first: xcolor may not be loaded (e.g.
2700  // colordvi-only documents).
2701  //
2702  // ALSO force global when `color_force_global` is set: color_sty.rs's lazy
2703  // dvipsnam.def loader (recovering a dropped `\usepackage[dvipsnames]{color}`
2704  // option, e.g. when hyperref preloaded `color` first) fires from WITHIN a
2705  // grouped digestion (`\textcolor{Blue}{…}` / a listings keywordstyle), so the
2706  // 68 dvips colors would otherwise be defined in that local group and revert
2707  // before the next `\color{…}` — making only the FIRST color resolve. Perl
2708  // loads dvipsnam at the preamble (top level), so its colors persist; the flag
2709  // reproduces that global effect. Witness 1705.06183.
2710  let effective_scope = if (lookup_definition(&T_CS!("\\ifglobalcolors"))?.is_some()
2711    && if_condition(&T_CS!("\\ifglobalcolors"))? == Some(true))
2712    || lookup_bool("color_force_global")
2713  {
2714    Some(Scope::Global)
2715  } else {
2716    scope
2717  };
2718  // Store in state as "model c1 c2 ..."
2719  let stored = color.to_stored();
2720  assign_value(
2721    &s!("color_{name}"),
2722    Stored::String(arena::pin(stored)),
2723    effective_scope,
2724  );
2725  // Define \\color@{name} macro for reversion
2726  // Perl: DefMacroI('\\color@'.$name, undef,
2727  //   '\relax\relax{model spec}{model}{spec_commas}')
2728  let model = color.model();
2729  let comps = color.components();
2730  let spec_parts: Vec<String> = comps.iter().map(|c| color::format_component(*c)).collect();
2731  let spec_space = spec_parts.join(" ");
2732  let spec_comma = spec_parts.join(",");
2733  let model_spec = s!("\\relax\\relax{{{model} {spec_space}}}{{{model}}}{{{spec_comma}}}");
2734  def_macro(
2735    T_CS!(s!("\\\\color@{name}")),
2736    None,
2737    crate::mouth::tokenize_internal(&model_spec),
2738    Some(crate::definition::expandable::ExpandableOptions {
2739      scope: effective_scope,
2740      ..Default::default()
2741    }),
2742  )?;
2743  Ok(())
2744}

2847/// Creates a single `Tokens` representing a TeX invocation of the lead `token` over a list of
2848/// arguments.
2849///
2850/// Note: currently this is near the `Mouth` representation of data, and deals purely with tokens.
2851/// A more generic version of this method may be able to support `ArgWrap` for the argument list.
2852/// Indeed, the return type may also be lifted to a new generic ArgWrap::Invocation, if there was
2853/// benefit.
2854pub fn build_invocation<T: Into<Token>>(token: T, args: Vec<Option<Tokens>>) -> Result<Tokens> {
2855  let token: Token = token.into();
2856  // Note: token may have been \let to another defn!
2857  if let Some(defn) = lookup_definition(&token)? {
2858    let mut invoked_tokens = vec![token];
2859    if let Some(params) = defn.get_parameters() {
2860      invoked_tokens.extend(params.revert_arguments(args)?);
2861    }
2862    Ok(Tokens::new(invoked_tokens))
2863  } else {
2864    let message = s!("Can't invoke {:?}; it is undefined", token.stringify());
2865    token.with_cs_name(|csname| {
2866      Error!("undefined", csname, message);
2867      Ok(())
2868    })?;
2869    let mut invoked_tokens = vec![token];
2870    // DefConstructor!(token, convert_latex_args(args.len(), 0),
2871    // sub { LaTeXML::Core::Stomach::makeError($_[0], 'undefined', token); });
2872    let wrapped_args: Vec<Token> = args
2873      .into_iter()
2874      .flat_map(|arg_opt| {
2875        let mut wrapped = vec![T_BEGIN!()];
2876        if let Some(arg) = arg_opt {
2877          wrapped.extend(arg.unlist());
2878        }
2879        wrapped.push(T_END!());
2880        wrapped
2881      })
2882      .collect();
2883    invoked_tokens.extend(wrapped_args);
2884    Ok(Tokens::new(invoked_tokens))
2885  }
2886}

2977/// Decode a codepoint using the fontmap for a given font and/or encoding (Perl: FontDecode).
2978/// Returns the decoded glyph (if any) and the possibly-adjusted font.
2979pub fn font_decode(
2980  code: i32,
2981  encoding_opt: Option<&str>,
2982  font_opt: Option<Rc<Font>>,
2983) -> (Option<char>, Option<Rc<Font>>) {
2984  if code < 0 {
2985    return (None, font_opt);
2986  }
2987  let font = font_opt.unwrap_or_else(|| lookup_font().unwrap());
2988  let encoding = match encoding_opt {
2989    Some(enc) => enc.to_string(),
2990    None => font
2991      .get_encoding()
2992      .map_or_else(|| "OT1".to_string(), |c| c.to_string()),
2993  };
2994  let map = load_font_map(&encoding);
2995  // Check for family-specific map. Use with_value to avoid cloning the
2996  // Stored envelope when the From<&Stored>→Option<Fontmap> impl only
2997  // needs the enum variant discriminant + an Rc bump.
2998  let (effective_map, _effective_enc) = if let Some(family) = font.get_family() {
2999    let fam_key = s!("{}_{}_fontmap", encoding, family);
3000    let fam_map: Option<Fontmap> = with_value(&fam_key, |v| v.and_then(|s| s.into()));
3001    if let Some(fm) = fam_map {
3002      (Some(fm), s!("{}_{}", encoding, family))
3003    } else {
3004      (map, encoding)
3005    }
3006  } else {
3007    (map, encoding)
3008  };
3009  let glyph = effective_map
3010    .as_ref()
3011    .and_then(|m| m.get(code as usize).copied().flatten());
3012  // In-math alphanumeric mathstyle handling is done in decodeMathChar instead
3013  (glyph, Some(font))
3014}

3016/// Decode a string using the fontmap for a given encoding (Perl: FontDecodeString).
3017/// If `implicit` is true, codepoints missing from the map decode to themselves.
3018pub fn font_decode_string(string: &str, encoding_opt: Option<&str>, implicit: bool) -> String {
3019  let font = lookup_font().unwrap();
3020  let encoding = match encoding_opt {
3021    Some(enc) => enc.to_string(),
3022    None => font
3023      .get_encoding()
3024      .map_or_else(|| "OT1".to_string(), |c| c.to_string()),
3025  };
3026  let map = load_font_map(&encoding);
3027  // Check for family-specific map — same with_value motivation as above.
3028  let effective_map = if let Some(family) = font.get_family() {
3029    let fam_key = s!("{}_{}_fontmap", encoding, family);
3030    let fam_map: Option<Fontmap> = with_value(&fam_key, |v| v.and_then(|s| s.into()));
3031    fam_map.or(map)
3032  } else {
3033    map
3034  };
3035  let input_enc = lookup_string("INPUT_ENCODING");
3036  let map_max: usize = if input_enc == "utf8" { 128 } else { 256 };
3037  // Also limit for short font maps
3038  let map_max = if let Some(ref m) = effective_map {
3039    if m.len() < map_max { m.len() } else { map_max }
3040  } else {
3041    map_max
3042  };
3043
3044  let mut result = String::new();
3045  for ch in string.chars() {
3046    let code = ch as usize;
3047    if implicit {
3048      if let Some(ref m) = effective_map {
3049        if code < map_max {
3050          if let Some(Some(glyph)) = m.get(code) {
3051            result.push(*glyph);
3052          }
3053        } else {
3054          result.push(ch);
3055        }
3056      } else {
3057        result.push(ch);
3058      }
3059    } else if let Some(ref m) = effective_map {
3060      if let Some(Some(glyph)) = m.get(code) {
3061        result.push(*glyph);
3062      }
3063    }
3064  }
3065  result
3066}
