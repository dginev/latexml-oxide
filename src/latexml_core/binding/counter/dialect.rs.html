1//! # Counters
2//!
3//! This is modelled on LaTeX's counter mechanisms, but since it also
4//! provides support for ID's, even where there is no visible reference number,
5//! it is defined in general.
6//! These id's should be both unique, and parallel the visible reference numbers
7//! (as much as possible).  Also, for consistency, we add id's to unnumbered
8//! document elements (eg from \section*); this requires an additional counter
9//! (eg. UNsection) and  mechanisms to track it.
10

35/// configuration for new_counter
36#[derive(Default)]
37pub struct NewCounterOptions<'ct> {
38  /// specifies a prefix to be used in formatting ID's for document structure elements
39  /// counted by this counter.  Ie. subsection 3 in section 2 might get: id="S2.SS3"
40  pub idprefix: &'ct str,
41  /// specifies that the ID is composed from $idwithin's ID,, even though
42  /// the counter isn't numbered within it.  (mainly to avoid duplicated ids)
43  pub idwithin: &'ct str,
44  /// a list of counters that correspond to scopes which are "inside" this one.
45  /// Whenever any definitions scoped to this counter are deactivated,
46  /// the inner counter's scopes are also deactivated.
47  // NOTE: I'm not sure this is even a sensible implementation,
48  // or why inner should be different than the counters reset by incrementing this counter.
49  pub nested: Vec<&'ct str>,
50}

52/// Defines a new counter named $ctr.
53/// If `within` is defined, `ctr` will be reset whenever `within` is incremented.
54pub fn new_counter(ctr: &str, within: &str, options_opt: Option<NewCounterOptions>) -> Result<()> {
55  let unctr = s!("UN{ctr}"); // UNctr is counter for generating ID's for UN-numbered items.
56  if !within.is_empty()
57    && within != "document"
58    && lookup_definition(&T_CS!(s!("\\c@{within}")))?.is_none()
59  {
60    new_counter(within, "", None)?;
61  }
62  let cctr = s!("\\c@{ctr}");
63  let clctr = s!("\\cl@{ctr}");
64  let cunctr = s!("\\c@{unctr}");
65  let clunctr = s!("\\cl@{unctr}");
66  // Perl Package.pm L660-672: Check if counter already defined. Skip register if already a
67  // Register. Warn if previously defined as something other than \relax.
68  let cs_cctr = T_CS!(&cctr);
69  let prev_defn = lookup_definition(&cs_cctr)?;
70  if let Some(ref defn) = prev_defn {
71    if defn.is_register() {
72      // Counter already exists as a register — fine, just continue (may change within/nesting)
73    } else {
74      // Warn unless the previous definition was \relax
75      let relax_meaning = lookup_meaning(&T_RELAX!());
76      let prev_meaning = lookup_meaning(&cs_cctr);
77      if prev_meaning != relax_meaning {
78        Warn!(
79          "unexpected",
80          &cctr,
81          s!("Counter {} was already defined; redefining", cctr)
82        );
83      }
84      def_register(
85        cs_cctr,
86        None,
87        Number::new(0),
88        Some(RegisterOptions {
89          allocate: Some(String::from("\\count")),
90          ..RegisterOptions::default()
91        }),
92      )?;
93    }
94  } else {
95    def_register(
96      cs_cctr,
97      None,
98      Number::new(0),
99      Some(RegisterOptions {
100        allocate: Some(String::from("\\count")),
101        ..RegisterOptions::default()
102      }),
103    )?;
104  }
105  after_assignment();
106  if !has_value(&clctr) {
107    assign_value(&clctr, Tokens!(), Some(Scope::Global));
108  }
109  def_register(T_CS!(&cunctr), None, Number::new(0), None)?;
110  if !has_value(&clunctr) {
111    assign_value(&clunctr, Tokens!(), Some(Scope::Global));
112  }
113
114  if !within.is_empty() {
115    let clwithin = s!("\\cl@{within}");
116    let clunwithin = s!("\\cl@UN{within}");
117    let x = if let Some(cl) = lookup_tokens(&clwithin) {
118      cl.unlist()
119    } else {
120      Vec::new()
121    };
122    let mut clwithin_tokens = vec![T_CS!(ctr), T_CS!(&unctr)];
123    clwithin_tokens.extend(x);
124    assign_value(
125      &clwithin,
126      Stored::Tokens(Tokens::new(clwithin_tokens)),
127      Some(Scope::Global),
128    );
129
130    let mut unx = if let Some(clun) = lookup_tokens(&clunwithin) {
131      clun.unlist()
132    } else {
133      Vec::new()
134    };
135    let mut clunwithin_tokens = vec![T_CS!(unctr)];
136    clunwithin_tokens.append(&mut unx);
137
138    assign_value(
139      &clunwithin,
140      Stored::Tokens(Tokens::new(clunwithin_tokens)),
141      Some(Scope::Global),
142    )
143  }
144
145  if let Some(ref options) = options_opt
146    && !options.nested.is_empty()
147  {
148    assign_value(
149      &s!("nested_counters_{}", ctr),
150      options.nested.clone(),
151      Some(Scope::Global),
152    )
153  }
154
155  // default is equivalent to \arabic{ctr}, but w/o using the LaTeX macro!
156  let ctr_string = ctr.to_string();
157  def_macro(
158    T_CS!(s!("\\the{}", ctr)),
159    None,
160    Some(ExpansionBody::Closure(Rc::new(move |_args| {
161      let counter_value = counter_value(&ctr_string)?.value_of();
162      Ok(Tokens::new(ExplodeText!(counter_value)))
163    }))),
164    Some(ExpandableOptions {
165      scope: Some(Scope::Global),
166      ..ExpandableOptions::default()
167    }),
168  )?;
169  let p_ctr_cs = T_CS!(&s!("\\p@{}", ctr));
170  if lookup_definition(&p_ctr_cs)?.is_none() {
171    def_macro(
172      p_ctr_cs,
173      None,
174      Tokens::default(),
175      Some(ExpandableOptions {
176        scope: Some(Scope::Global),
177        ..ExpandableOptions::default()
178      }),
179    )?;
180  }
181
182  let mut prefix = match options_opt {
183    None => String::new(),
184    Some(ref opt) => opt.idprefix.to_string(),
185  };
186  if !prefix.is_empty() {
187    assign_value(
188      &s!("@ID@prefix@{}", ctr),
189      prefix.clone(),
190      Some(Scope::Global),
191    );
192  } else {
193    prefix = lookup_string(&s!("@ID@prefix@{}", ctr));
194    if prefix.is_empty() {
195      prefix = ctr.to_string();
196    }
197  }
198  prefix = clean_id(&prefix);
199
200  if !prefix.is_empty() {
201    let idwithin = match options_opt {
202      Some(ref opts) => {
203        if opts.idwithin.is_empty() {
204          within
205        } else {
206          opts.idwithin
207        }
208      },
209      None => within,
210    }
211    .to_string();
212
213    let ctr_string = ctr.to_string();
214    let thectrid = s!("\\the{}@ID", ctr);
215    if !idwithin.is_empty() {
216      def_macro(
217        T_CS!(thectrid),
218        None,
219        Some(ExpansionBody::Closure(Rc::new(move |_args| {
220          // Perl Package.pm L696 probes `\lx@empty`, NOT `\@empty` — and
221          // deliberately so: `\@empty` is LaTeX-pool-only (latex_base.rs
222          // aliases it to `\lx@empty`), while `\lx@empty` is engine-level
223          // (base_schema.rs) and exists in PLAIN TeX documents too. With
224          // `\@empty` here, a plain-TeX document's `\the<ctr>@ID` probe
225          // compared `\thedocument@ID` (an alias of `\lx@empty`) against an
226          // UNDEFINED `\@empty` → unequal → the \else branch re-expands the
227          // parent's @ID formatter, spinning the gullet into a 4-token
228          // expansion cycle (caught by the cycle guard as a phantom
229          // `Fatal:Timeout:Recursion`, swallowed inside math-parser
230          // semantics — `create_xmrefs`/`get_xmarg_id`). Witness
231          // math0402448 (plain TeX + 3464 formulae): "Conversion failed:
232          // 1 fatal error" with no Fatal: line in the log.
233          Ok(mouth::tokenize_internal(&s!(
234            "\\expandafter\\ifx\\csname the{}@ID\\endcsname\\lx@empty\\else\\csname the{}@ID\\endcsname.\\fi {}\\csname @{}@ID\\endcsname",
235            idwithin,
236            idwithin,
237            prefix,
238            ctr_string
239          )))
240        }))),
241        Some(ExpandableOptions {
242          scope: Some(Scope::Global),
243          ..ExpandableOptions::default()
244        }),
245      )?;
246    } else {
247      def_macro(
248        T_CS!(thectrid),
249        None,
250        Some(ExpansionBody::Closure(Rc::new(move |_args| {
251          Ok(mouth::tokenize_internal(&s!(
252            "{prefix}\\csname @{ctr_string}@ID\\endcsname",
253          )))
254        }))),
255        Some(ExpandableOptions {
256          scope: Some(Scope::Global),
257          ..ExpandableOptions::default()
258        }),
259      )?;
260    }
261    def_macro(
262      T_CS!(s!("\\@{}@ID", ctr)),
263      None,
264      Some(ExpansionBody::Tokens(Tokens!(T_OTHER!("0")))),
265      Some(ExpandableOptions {
266        scope: Some(Scope::Global),
267        ..ExpandableOptions::default()
268      }),
269    )?;
270  }
271
272  Ok(())
273}
274/// Fetches the value associated with the counter C<$ctr>.
275pub fn counter_value(ctr: &str) -> Result<Number> {
276  match lookup_register(&s!("\\c@{ctr}"), Vec::new())? {
277    None => {
278      let message = s!("Counter {} was not defined; assuming 0", ctr);
279      Warn!("undefined", ctr, message);
280      Ok(Number::new(0))
281    },
282    Some(value) => Ok(Number::new(value.value_of())),
283  }
284}
285/// increments a named counter by a `Number`
286pub fn add_to_counter(ctr: &str, value: Number) -> Result<()> {
287  let v = counter_value(ctr)?.add(value);
288  assign_register(&s!("\\c@{ctr}"), v.into(), Some(Scope::Global), Vec::new())?;
289  after_assignment();
290  let id_cs = T_CS!(s!("\\@{ctr}@ID"));
291  def_macro(
292    id_cs,
293    None,
294    Tokens::new(Explode!(v.value_of())),
295    Some(ExpandableOptions {
296      scope: Some(Scope::Global),
297      ..ExpandableOptions::default()
298    }),
299  )
300}

302/// Analog of `\stepcounter`, steps the counter and returns the expansion of
303/// `\the$ctr`  Usually you should use `ref_step_counter(ctr)` instead.
304pub fn step_counter(ctr: &str, noreset: bool) -> Result<()> {
305  let value = counter_value(ctr)?;
306  let newvalue = value.add(Number::new(1));
307  let c_ctr = s!("\\c@{ctr}");
308  assign_register(&c_ctr, newvalue.into(), Some(Scope::Global), Vec::new())?;
309  after_assignment();
310  let token_value = Tokens::new(Explode!(newvalue.value_of()));
311  def_macro(
312    T_CS!(s!("\\@{ctr}@ID")),
313    None,
314    token_value,
315    Some(ExpandableOptions {
316      scope: Some(Scope::Global),
317      ..ExpandableOptions::default()
318    }),
319  )?;
320
321  // and reset any within counters!
322  if !noreset && let Some(nested) = lookup_tokens(&s!("\\cl@{ctr}")) {
323    for c in nested.unlist() {
324      reset_counter(&c)?;
325    }
326  }
327  Ok(())
328}

330/// Analog of `\refstepcounter`, steps the counter and returns a hash
331/// containing the keys `refnum` and `id`.
332///
333/// This makes it
334/// suitable for use in a `properties` option to constructors.
335/// The `id` is generated in parallel with the reference number
336/// to assist debugging.
337// TODO: Maybe these should be specialized types in Rust, rather than hashmaps?
338pub fn ref_step_counter(ctype: &str, noreset: bool) -> Result<HashMap<Stored>> {
339  // Defensive: under some upstream conditions the {} parameter reader pulls a
340  // trailing `\par` (or similar trailing CS) into a counter-type identifier
341  // before it reaches us (Cluster A: math0010095, hep-ph0204075). Strip it
342  // here so the downstream `\csname @<ctype>...@ID\endcsname` and similar
343  // constructions stay well-formed. We strip the same well-known sentinels
344  // as latex_constructs::strip_trailing_cs.
345  let ctype = {
346    let mut s = ctype;
347    for tail in ["\\par", "\\@startsection@hook", "\\relax"] {
348      if let Some(stripped) = s.strip_suffix(tail) {
349        s = stripped;
350        break;
351      }
352    }
353    s
354  };
355  let ctr = with_mapping("counter_for_type", ctype, |meaning| match meaning {
356    Some(Stored::String(ctr)) => arena::to_string(*ctr),
357    _ => ctype.to_string(),
358  });
359  step_counter(&ctr, noreset)?;
360  maybe_preempt_refnum(&ctr, false);
361
362  let the_ctr_id = s!("\\the{ctr}@ID");
363  let the_ctr = s!("\\the{ctr}");
364
365  let has_id: bool = match lookup_definition(&T_CS!(&the_ctr_id))? {
366    Some(iddef) => {
367      if let Some(params) = iddef.get_parameters() {
368        params.get_num_args() == 0
369      } else {
370        true
371      }
372    },
373    _ => false,
374  };
375
376  let the_ctr_cs = T_CS!(&the_ctr);
377  let the_ctr_id_cs = T_CS!(&the_ctr_id);
378  def_macro(
379    T_CS!("\\@currentlabel"),
380    None,
381    the_ctr_cs,
382    Some(ExpandableOptions {
383      scope: Some(Scope::Global),
384      ..ExpandableOptions::default()
385    }),
386  )?;
387  if has_id {
388    def_macro(
389      T_CS!("\\@currentID"),
390      None,
391      the_ctr_id_cs,
392      Some(ExpandableOptions {
393        scope: Some(Scope::Global),
394        ..ExpandableOptions::default()
395      }),
396    )?;
397  }
398
399  let id = if has_id {
400    digest_literal(Tokens!(T_CS!(&the_ctr_id)))?.to_string()
401  } else {
402    String::new()
403  };
404
405  let refnum = digest_text(Tokens!(T_CS!(&the_ctr)))?;
406  let invocation;
407  {
408    invocation = build_invocation(T_CS!("\\lx@make@tags"), vec![Some(Tokens!(T_OTHER!(
409      ctype
410    )))])?;
411  }
412
413  let tags = stomach::digest(invocation)?;
414
415  // Any scopes activated for previous value of this counter (& any nested counters) must be
416  // removed. This may also include scopes activated for \label
417  deactivate_counter_scope(arena::pin(&ctr));
418
419  // And install the scope (if any) for this reference number.
420  assign_value("current_counter", ctr.clone(), Some(Scope::Local));
421
422  let scope = arena::pin(format!("{ctr}:{refnum}"));
423  let mut receiver = VecDeque::new();
424  receiver.push_front(Stored::String(scope));
425  assign_value(
426    &s!("scopes_for_counter:{ctr}"),
427    receiver,
428    Some(Scope::Local),
429  );
430  activate_scope(scope);
431
432  Ok(stored_map!(
433    "tags" => Stored::Digested(tags),
434    "id" => Stored::String(arena::pin(id))
435  ))
436}

438/// Internal: Use a label-derived reference number and/or ID
439/// instead of the traditional counter based ones.
440/// Since the \label{} determins the reference number and ID,
441/// we MUST sniff out the label BEFORE we call RefStepCounter/RefStepID !!!!!
442/// (see MaybePeekLabel below; and also MaybeNoteLabel for use within
443/// captions & certain equation environments)
444/// Assign a sub to LABEL_MAPPING_HOOK: &sub($label,$counter,$norefnum)
445/// to return the desired refnum and id for a given object.
446fn maybe_preempt_refnum(ctr: &str, norefnum: bool) {
447  if let Some(mapper) = get_label_mapping_hook() {
448    let hj_refnum = T_CS!(s!("\\_PREEMPTED_REFNUM_{ctr}"));
449    let hj_id = T_CS!(s!("\\_PREEMPTED_ID_{ctr}"));
450    // First, restore the \the<ctr> and \the<ctr>@ID macros to defaults
451    if !norefnum && has_meaning(&hj_refnum) {
452      let_i(&T_CS!(s!("\\the{ctr}")), &hj_refnum, Some(Scope::Global));
453    }
454    if has_meaning(&hj_id) {
455      let_i(&T_CS!(s!("\\the{ctr}@ID")), &hj_id, Some(Scope::Global));
456    }
457    let label = lookup_string("PEEKED_LABEL");
458    let (fixedrefnum, fixedid) = mapper(&label, ctr, norefnum);
459    if let Some(refnum) = fixedrefnum
460      && !norefnum
461    {
462      if !has_meaning(&hj_refnum) {
463        // Save for later
464        let_i(&hj_refnum, &T_CS!(s!("\\the{ctr}")), Some(Scope::Global));
465      }
466      let _ = def_macro(
467        T_CS!(s!("\\the{ctr}")),
468        None,
469        ExpansionBody::Tokens(Tokens::new(Explode!(&refnum))),
470        Some(ExpandableOptions {
471          scope: Some(Scope::Global),
472          ..Default::default()
473        }),
474      );
475    }
476    if let Some(id) = fixedid {
477      if !has_meaning(&hj_id) {
478        // Save for later
479        let_i(&hj_id, &T_CS!(s!("\\the{ctr}@ID")), Some(Scope::Global));
480      }
481      let _ = def_macro(
482        T_CS!(s!("\\the{ctr}@ID")),
483        None,
484        ExpansionBody::Tokens(Tokens::new(Explode!(&id))),
485        Some(ExpandableOptions {
486          scope: Some(Scope::Global),
487          ..Default::default()
488        }),
489      );
490    }
491    remove_value("PEEKED_LABEL"); // CONSUME the label
492    assign_value(
493      "PROCESSED_LABEL",
494      Stored::String(arena::pin(label)),
495      Some(Scope::Global),
496    );
497  }
498}

500/// Use to peek for FOLLOWING \label{...} to support label-derived reference numbers
501/// (Perl: MaybePeekLabel)
502pub fn maybe_peek_label() -> Result<()> {
503  if get_label_mapping_hook().is_some() {
504    let peek = crate::gullet::read_non_space()?;
505    if let Some(ref token) = peek {
506      if x_equals(token, &T_CS!("\\label")) {
507        begin_semiverbatim(None);
508        let arg = crate::gullet::read_arg(crate::gullet::ExpansionLevel::Off)?;
509        end_semiverbatim()?;
510        let arg_str = arg.to_string();
511        let label = clean_label(&arg_str, Some("")).into_owned();
512        assign_value(
513          "PEEKED_LABEL",
514          Stored::String(arena::pin(&label)),
515          Some(Scope::Global),
516        );
517        // Put back the arg wrapped in braces so \label can re-read it
518        crate::gullet::unread(Tokens!(T_BEGIN!(), arg, T_END!()));
519      } else {
520        remove_value("PROCESSED_LABEL");
521        remove_value("PEEKED_LABEL");
522      }
523    }
524    if let Some(token) = peek {
525      crate::gullet::unread_one(token);
526    }
527  }
528  Ok(())
529}

531/// Use to note a discovered label to support label-derived refererence numbers
532/// Can by used by \label, among others. Note we only record the label
533/// if it hasn't already been peeked, and consumed.
534pub fn maybe_note_label(label: &str) {
535  if get_label_mapping_hook().is_some() {
536    let label = clean_label(label, Some(""));
537    let processed = lookup_string("PROCESSED_LABEL");
538    if processed.is_empty() || processed != label {
539      // Only if not already processed
540      remove_value("PROCESSED_LABEL");
541      assign_value(
542        "PEEKED_LABEL",
543        Stored::String(arena::pin(label)),
544        Some(Scope::Global),
545      );
546    }
547  }
548}

550fn deactivate_counter_scope(ctr: SymStr) {
551  let (scopes_for_counter, nested_counters) = arena::with(ctr, |cstr| {
552    (
553      s!("scopes_for_counter:{cstr}"),
554      s!("nested_counters_{cstr}"),
555    )
556  });
557  // with_value avoids the outer Stored::clone by reading through a
558  // borrow; we still collect the scope SymStrs (Copy) or panic-pointers
559  // into owned Vecs to outlive the borrow.
560  let scope_syms: Vec<SymStr> = with_value(&scopes_for_counter, |v| match v {
561    Some(Stored::VecDequeStored(stored_scopes)) => stored_scopes
562      .iter()
563      .map(|s| match s {
564        Stored::String(scope) => *scope,
565        _ => panic!("assignment scopes should be stored as strings, got: {s:?}"),
566      })
567      .collect(),
568    _ => Vec::new(),
569  });
570  for scope in scope_syms {
571    deactivate_scope(scope);
572  }
573
574  // TODO: if we ever want to unshift from the nested_counters, we'll need to also use
575  // Stored::VecDequeStored for them.
576  let inner_ctrs: Vec<SymStr> = with_value(&nested_counters, |v| match v {
577    Some(Stored::Strings(stored_counters)) => stored_counters.iter().copied().collect(),
578    _ => Vec::new(),
579  });
580  for inner_ctr in inner_ctrs {
581    deactivate_counter_scope(inner_ctr);
582  }
583}

585/// For UN-numbered units.
586/// Like `RefStepCounter`, but only steps the "uncounter",
587/// and returns only the id;  This is useful for unnumbered cases
588/// of objects that normally get both a refnum and id.
589pub fn ref_step_id(ctype: &str) -> Result<HashMap<Stored>> {
590  let ctr = with_mapping("counter_for_type", ctype, |mapping| match mapping {
591    Some(map) => map.to_string(),
592    None => ctype.to_string(),
593  });
594  let unctr = s!("UN{ctr}");
595  // Perl Package.pm L863-864 ("Avoid fatals..."): if `\c@UN<ctr>` isn't
596  // defined as a register, `NewCounter(ctr)` creates both `\c@<ctr>` and
597  // `\c@UN<ctr>` and the associated `\the<ctr>@ID` expansion. Without it,
598  // unnumbered-section callers like `\specialsection*{}` (which amsart
599  // Let's to `\chapter*{}` even though amsart has no chapter counter) hit
600  // Error:undefined:\thechapter@ID because the counter was never created.
601  let unctr_cmd = s!("\\c@{unctr}");
602  let unctr_defined = lookup_register(&unctr_cmd, Vec::new())
603    .ok()
604    .flatten()
605    .is_some();
606  if !unctr_defined {
607    let _ = new_counter(&ctr, "document", None);
608  }
609  step_counter(&unctr, false)?;
610  maybe_preempt_refnum(&ctr, true);
611  let cunctr_val = lookup_number(&s!("\\c@{unctr}"))
612    .unwrap_or_default()
613    .value_of();
614  def_macro(
615    T_CS!(s!("\\@{ctr}@ID")),
616    None,
617    Tokens!(T_OTHER!("x"), Explode!(cunctr_val)),
618    Some(ExpandableOptions {
619      scope: Some(Scope::Global),
620      ..ExpandableOptions::default()
621    }),
622  )?;
623
624  let the_ctr_id = s!("\\the{ctr}@ID");
625  def_macro(T_CS!("\\@currentID"), None, T_CS!(&the_ctr_id), None)?;
626  Ok(stored_map!("id" =>
627    clean_id(&digest_literal(T_CS!(the_ctr_id))?.to_string())))
628}

630/// Recycle the last ID without incrementing (Perl: RefCurrentID)
631/// Useful if the last ID-ed box got pruned.
632pub fn ref_current_id(ctype: &str) -> Result<HashMap<Stored>> {
633  let ctr = with_mapping("counter_for_type", ctype, |mapping| match mapping {
634    Some(map) => map.to_string(),
635    None => ctype.to_string(),
636  });
637  let the_ctr_id = s!("\\the{ctr}@ID");
638  let id = clean_id(&digest_literal(T_CS!(the_ctr_id))?.to_string());
639  Ok(stored_map!("id" => id))
640}

642/// Resets the counter `ctr` to zero.
643pub fn reset_counter(ctr: &Token) -> Result<()> {
644  let (c_ctr, c_un_ctr, ctr_id) =
645    ctr.with_str(|ctr| (s!("\\c@{ctr}"), s!("\\c@UN{ctr}"), s!("\\@{ctr}@ID")));
646  assign_register(
647    &c_ctr,
648    Number::new(0).into(),
649    Some(Scope::Global),
650    Vec::new(),
651  )?;
652  if !ctr.with_str(|cstr| cstr.starts_with("UN")) {
653    // but not UN
654    assign_register(
655      &c_un_ctr,
656      Number::new(0).into(),
657      Some(Scope::Global),
658      Vec::new(),
659    )?;
660  }
661  def_macro(
662    T_CS!(ctr_id),
663    None,
664    Tokens!(T_OTHER!("0")),
665    Some(ExpandableOptions {
666      scope: Some(Scope::Global),
667      ..ExpandableOptions::default()
668    }),
669  )?;
670  // and reset any within counters!
671  if let Some(nested) = lookup_tokens(&s!("\\cl@{ctr}")) {
672    for c in nested.unlist() {
673      reset_counter(&c)?;
674    }
675  }
676  Ok(())
677}

679/// Create id, and tags for an itemize type \item
680pub fn ref_step_item_counter(tag_opt: Option<&Tokens>) -> Result<HashMap<Stored>> {
681  let counter = lookup_string("itemcounter");
682  let n = lookup_int("itemization_items");
683  assign_value("itemization_items", n + 1, None);
684  let mut attr: HashMap<Stored> = HashMap::default();
685  if n > 0
686    && let Some(sep) = lookup_dimension("\\itemsep")
687  {
688    let default_opt = lookup_dimension("\\lx@default@itemsep");
689    if default_opt.is_none() || sep.value_of() != default_opt.unwrap().value_of() {
690      attr.insert("itemsep", sep.into());
691    }
692  }
693
694  let mut result = if let Some(tag) = tag_opt {
695    let mut props = ref_step_id(&counter)?;
696    if tag.is_empty() {
697      return Ok(props);
698    }
699    let formatter = if counter.starts_with("@desc") {
700      T_CS!("\\descriptionlabel")
701    } else {
702      T_CS!("\\makelabel")
703    };
704    let counter_name = s!("\\{counter}name");
705    let typename = if is_defined(&counter_name) {
706      T_CS!(counter_name)
707    } else {
708      T_CS!("\\itemtyperefname")
709    };
710
711    let mut tag_tokens = vec![
712      T_BEGIN!(),
713      T_CS!("\\let"),
714      T_CS!(s!("\\the{counter}")),
715      T_CS!("\\@empty"),
716      T_CS!("\\def"),
717      T_CS!(s!("\\fnum@{counter}")),
718      T_BEGIN!(),
719      formatter,
720      T_BEGIN!(),
721    ];
722    // TODO: Another iffy clone...
723    let reverted_tag = (*tag).clone().revert();
724    tag_tokens.extend(reverted_tag.clone());
725    tag_tokens.extend(vec![
726      T_END!(),
727      T_END!(),
728      T_CS!("\\def"),
729      T_CS!(s!("\\typerefnum@{counter}")),
730      T_BEGIN!(),
731      typename,
732      T_SPACE!(),
733    ]);
734    tag_tokens.extend(reverted_tag);
735    tag_tokens.push(T_END!());
736    tag_tokens.extend(
737      build_invocation(T_CS!("\\lx@make@tags"), vec![Some(Tokens!(T_OTHER!(
738        counter
739      )))])?
740      .unlist(),
741    );
742    tag_tokens.push(T_END!());
743
744    let tags = stomach::digest(tag_tokens)?;
745    if !tags.is_empty()? {
746      props.insert("tags", tags.into());
747    }
748    props
749  } else {
750    ref_step_counter(&counter, false)?
751  };
752  for (k, v) in attr.into_iter() {
753    result.insert_sym(k, v);
754  }
755  Ok(result)
756}

758/// configuration for begin_itemize
759#[derive(Debug, Default, Clone)]
760pub struct BeginItemizeOptions {
761  /// disable nested id suffix based on stacking level
762  pub nolevel:     bool,
763  /// enumitem series
764  pub series:      Option<Tokens>,
765  /// start at a custom value
766  pub start:       Option<Number>,
767  /// enumitem resume?
768  pub resume:      Option<String>,
769  /// enumitem resume* ?
770  pub resume_star: Option<String>,
771}

773/// Prepare for an list (itemize/enumerate/description/etc)
774/// by determining the right counter (level)
775/// and binding the right \item ( \$type@item, if $type is defined)
776pub fn begin_itemize(
777  itype: &str,
778  counter: Option<&str>,
779  options: BeginItemizeOptions,
780) -> Result<HashMap<Stored>> {
781  // The list-type and level of the *containing* list (if any!)
782  let outercounter = lookup_string("itemcounter");
783  let outerlevel = if !outercounter.is_empty() {
784    lookup_int(&s!("{outercounter}level"))
785  } else {
786    0
787  };
788  let counter = counter.unwrap_or("@item");
789  let listlevel = lookup_int("itemization_level") + 1; // level for this list overall
790  let level = lookup_int(&s!("{counter}level")) + // level for lists of specific type
791    (if options.nolevel { 0 } else { 1 });
792  AssignRegister!(
793    "\\itemsep",
794    lookup_dimension("\\lx@default@itemsep")
795      .unwrap_or_default()
796      .into()
797  );
798  assign_value("itemization_level", listlevel, None);
799  assign_value(&s!("{counter}level"), level, None);
800  assign_value("itemization_items", 0, None);
801  let listpostfix = roman!(listlevel).to_string();
802  let postfix = roman!(level).to_string();
803  let mut usecounter = counter.to_string();
804  if !options.nolevel && !postfix.is_empty() {
805    usecounter.push_str(&postfix);
806  }
807  if !itype.is_empty() {
808    let itype_cs = T_CS!(s!("\\{itype}@item"));
809    let_i(&T_CS!("\\item"), &itype_cs, None);
810  }
811  // In case within odd environment.
812  let_i(&T_CS!("\\par"), &T_CS!("\\lx@normal@par"), None);
813  def_macro(
814    T_CS!("\\@listctr"),
815    None,
816    Tokens!(Explode!(usecounter)),
817    None,
818  )?;
819  // Now arrange that this list's id's are relative to the current (outer) item (if any)
820  // And that the items within this list's id's are relative to this (new) list.
821  assign_value("itemcounter", Stored::String(arena::pin(&usecounter)), None);
822  let listcounter = s!("@itemize{listpostfix}");
823  if lookup_definition(&T_CS!(s!("\\c@{listcounter}")))?.is_none() {
824    //Create new list counters as needed
825    new_counter(&listcounter, "", None)?;
826  }
827  if !outercounter.is_empty() {
828    // Make this list's ID relative to outer list's ID
829    let outerusecounter = s!("{outercounter}{}", roman!(outerlevel).to_string());
830    let thectr = s!("\\the{listcounter}@ID");
831    let theexpansion = s!("\\the{outerusecounter}@ID.I\\arabic{{{listcounter}}}");
832    def_macro(
833      T_CS!(thectr),
834      None,
835      mouth::tokenize_internal(&theexpansion),
836      None,
837    )?;
838
839    // AND reset this list's counter when the outer item is stepped
840    let mut cl_toks = vec![T_CS!(&listcounter)];
841    let cl_name = s!("\\cl@{outerusecounter}");
842    let existing = with_value(&cl_name, |v| match v {
843      Some(Stored::Tokens(tks)) => tks.clone().unlist(),
844      _ => Vec::new(),
845    });
846    cl_toks.extend(existing);
847    assign_value(
848      &cl_name,
849      Stored::Tokens(Tokens::new(cl_toks)),
850      Some(Scope::Global),
851    );
852  }
853  // format the id of \item's relative to the id of this list.
854  // Perl: Tokens(T_CS('\the' . $listcounter . '@ID'), T_OTHER('.i'),
855  //              T_CS('\@' . $usecounter . '@ID'))
856  // — build the Tokens array directly from 3 explicit tokens rather than
857  // round-tripping through a string tokenizer. This matters when `usecounter`
858  // contains digits (e.g. "count1" from \usecounter{count1}): `tokenize_internal`
859  // on a literal string "\@count1@ID" would split into \@count + "1" + "@" + "ID"
860  // because TeX's CS reader stops at digits (digits are catcode OTHER even in
861  // the style table). Perl uses T_CS($name) to build a single CS token directly
862  // by name, bypassing tokenization.
863  let useexp = Tokens::new(vec![
864    T_CS!(s!("\\the{listcounter}@ID")),
865    T_OTHER!(".i"),
866    T_CS!(s!("\\@{usecounter}@ID")),
867  ]);
868  def_macro(T_CS!(s!("\\the{usecounter}@ID")), None, useexp, None)?;
869
870  let mut series = if let Some(s) = options.series {
871    s.to_string()
872  } else {
873    String::new()
874  };
875  if let Some(start) = options.start {
876    SetCounter!(usecounter, start);
877    add_to_counter(&usecounter, Number(-1))?;
878  } else if let Some(s) = match options.resume {
879    Some(s) => Some(s),
880    None => options.resume_star,
881  } {
882    if s != "noseries" {
883      series = s.clone();
884      let last_val = lookup_int(&s!("enumitem_series_{s}_last"));
885      if last_val != 0 {
886        SetCounter!(usecounter, Number(last_val));
887      }
888    }
889  } else {
890    reset_counter(&T_OTHER!(&usecounter))?;
891  }
892
893  let mut rsc = ref_step_counter(&s!("@itemize{listpostfix}"), false)?;
894  rsc.insert("counter", usecounter.into());
895  rsc.insert("series", series.into());
896  Ok(rsc)
897}

899/// Set the itemization style for a given level.
900/// Perl: setItemizationStyle($stuff, $level)
901/// If $level is not given, uses the current @itemlevel.
902/// Defines \labelitem$level to $stuff.
903pub fn set_itemization_style(stuff: Option<&Tokens>, level: Option<i32>) -> Result<()> {
904  if let Some(stuff) = stuff {
905    if stuff.is_empty() {
906      return Ok(());
907    }
908    let level = level.unwrap_or_else(|| lookup_int("@itemlevel").max(0) as i32);
909    let level_str = roman_aux(level);
910    let cs_name = s!("\\labelitem{level_str}");
911    def_macro(T_CS!(&cs_name), None, stuff.clone(), None)?;
912  }
913  Ok(())
914}

916/// Set the enumeration style for a given level.
917/// Perl: setEnumerationStyle($stuff, $level)
918/// Parses the style tokens to detect A/a/I/i/1 patterns
919/// and defines \theenum$level and \labelenum$level accordingly.
920pub fn set_enumeration_style(stuff: Option<&Tokens>, level: Option<i32>) -> Result<()> {
921  if let Some(stuff) = stuff {
922    if stuff.is_empty() {
923      return Ok(());
924    }
925    let level = level.unwrap_or_else(|| lookup_int("enumlevel").max(0) as i32);
926    let level_str = roman_aux(level);
927    // Iterate the borrowed token slice — no clone needed, only reads.
928    let tokens = stuff.unlist_ref();
929    let mut out: Vec<Token> = Vec::with_capacity(tokens.len());
930    let ctr = T_OTHER!(s!("enum{level_str}"));
931    let mut i = 0;
932    while i < tokens.len() {
933      let t = tokens[i];
934      if t.get_catcode() == Catcode::BEGIN {
935        // Copy braced groups verbatim
936        out.push(t);
937        let mut brlevel = 1i32;
938        i += 1;
939        while brlevel > 0 && i < tokens.len() {
940          let tt = tokens[i];
941          if tt.get_catcode() == Catcode::BEGIN {
942            brlevel += 1;
943          } else if tt.get_catcode() == Catcode::END {
944            brlevel -= 1;
945          }
946          out.push(tt);
947          i += 1;
948        }
949      } else {
950        let ch = char::from_u32(t.get_charcode()).unwrap_or('\0');
951        let cat = t.get_catcode();
952        match (ch, cat) {
953          ('A', Catcode::LETTER) => {
954            // \Alph{enum$level}
955            def_macro(
956              T_CS!(s!("\\theenum{level_str}")),
957              None,
958              Tokens::new(vec![T_CS!("\\Alph"), T_BEGIN!(), ctr, T_END!()]),
959              None,
960            )?;
961            out.push(T_CS!(s!("\\theenum{level_str}")));
962          },
963          ('a', Catcode::LETTER) => {
964            // \alph{enum$level}
965            def_macro(
966              T_CS!(s!("\\theenum{level_str}")),
967              None,
968              Tokens::new(vec![T_CS!("\\alph"), T_BEGIN!(), ctr, T_END!()]),
969              None,
970            )?;
971            out.push(T_CS!(s!("\\theenum{level_str}")));
972          },
973          ('I', Catcode::LETTER) => {
974            // \Roman{enum$level}
975            def_macro(
976              T_CS!(s!("\\theenum{level_str}")),
977              None,
978              Tokens::new(vec![T_CS!("\\Roman"), T_BEGIN!(), ctr, T_END!()]),
979              None,
980            )?;
981            out.push(T_CS!(s!("\\theenum{level_str}")));
982          },
983          ('i', Catcode::LETTER) => {
984            // \roman{enum$level}
985            def_macro(
986              T_CS!(s!("\\theenum{level_str}")),
987              None,
988              Tokens::new(vec![T_CS!("\\roman"), T_BEGIN!(), ctr, T_END!()]),
989              None,
990            )?;
991            out.push(T_CS!(s!("\\theenum{level_str}")));
992          },
993          ('1', Catcode::OTHER) => {
994            // \arabic{enum$level}
995            def_macro(
996              T_CS!(s!("\\theenum{level_str}")),
997              None,
998              Tokens::new(vec![T_CS!("\\arabic"), T_BEGIN!(), ctr, T_END!()]),
999              None,
1000            )?;
1001            out.push(T_CS!(s!("\\theenum{level_str}")));
1002          },
1003          _ => {
1004            out.push(t);
1005          },
1006        }
1007        i += 1;
1008      }
1009    }
1010    // Define \labelenum$level = { out }
1011    let mut label_tokens = vec![T_BEGIN!()];
1012    label_tokens.extend(out);
1013    label_tokens.push(T_END!());
1014    def_macro(
1015      T_CS!(s!("\\labelenum{level_str}")),
1016      None,
1017      Tokens::new(label_tokens),
1018      None,
1019    )?;
1020  }
1021  Ok(())
1022}
