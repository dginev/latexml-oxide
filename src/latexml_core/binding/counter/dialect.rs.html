1//! # Counters
2//!
3//! This is modelled on LaTeX's counter mechanisms, but since it also
4//! provides support for ID's, even where there is no visible reference number,
5//! it is defined in general.
6//! These id's should be both unique, and parallel the visible reference numbers
7//! (as much as possible).  Also, for consistency, we add id's to unnumbered
8//! document elements (eg from \section*); this requires an additional counter
9//! (eg. UNsection) and  mechanisms to track it.
10

34/// configuration for new_counter
35#[derive(Default)]
36pub struct NewCounterOptions<'ct> {
37  /// specifies a prefix to be used in formatting ID's for document structure elements
38  /// counted by this counter.  Ie. subsection 3 in section 2 might get: id="S2.SS3"
39  pub idprefix: &'ct str,
40  /// specifies that the ID is composed from $idwithin's ID,, even though
41  /// the counter isn't numbered within it.  (mainly to avoid duplicated ids)
42  pub idwithin: &'ct str,
43  /// a list of counters that correspond to scopes which are "inside" this one.
44  /// Whenever any definitions scoped to this counter are deactivated,
45  /// the inner counter's scopes are also deactivated.
46  // NOTE: I'm not sure this is even a sensible implementation,
47  // or why inner should be different than the counters reset by incrementing this counter.
48  pub nested: Vec<&'ct str>,
49}

51/// Defines a new counter named $ctr.
52/// If `within` is defined, `ctr` will be reset whenever `within` is incremented.
53pub fn new_counter(ctr: &str, within: &str, options_opt: Option<NewCounterOptions>) -> Result<()> {
54  let unctr = s!("UN{ctr}"); // UNctr is counter for generating ID's for UN-numbered items.
55  if !within.is_empty()
56    && within != "document"
57    && lookup_definition(&T_CS!(s!("\\c@{within}")))?.is_none()
58  {
59    new_counter(within, "", None)?;
60  }
61  let cctr = s!("\\c@{ctr}");
62  let clctr = s!("\\cl@{ctr}");
63  let cunctr = s!("\\c@{unctr}");
64  let clunctr = s!("\\cl@{unctr}");
65  def_register(
66    T_CS!(&cctr),
67    None,
68    Number::new(0),
69    Some(RegisterOptions {
70      allocate: Some(String::from("\\count")),
71      ..RegisterOptions::default()
72    }),
73  )?;
74  after_assignment();
75  if !has_value(&clctr) {
76    state::assign_value(&clctr, Tokens!(), Some(Scope::Global));
77  }
78  def_register(T_CS!(&cunctr), None, Number::new(0), None)?;
79  if !has_value(&clunctr) {
80    state::assign_value(&clunctr, Tokens!(), Some(Scope::Global));
81  }
82
83  if !within.is_empty() {
84    let clwithin = s!("\\cl@{within}");
85    let clunwithin = s!("\\cl@UN{within}");
86    let x = if let Some(cl) = state::lookup_tokens(&clwithin) {
87      cl.unlist()
88    } else {
89      Vec::new()
90    };
91    let mut clwithin_tokens = vec![T_CS!(ctr), T_CS!(&unctr)];
92    clwithin_tokens.extend(x);
93    state::assign_value(
94      &clwithin,
95      Stored::Tokens(Tokens::new(clwithin_tokens)),
96      Some(Scope::Global),
97    );
98
99    let mut unx = if let Some(clun) = state::lookup_tokens(&clunwithin) {
100      clun.unlist()
101    } else {
102      Vec::new()
103    };
104    let mut clunwithin_tokens = vec![T_CS!(unctr)];
105    clunwithin_tokens.append(&mut unx);
106
107    state::assign_value(
108      &clunwithin,
109      Stored::Tokens(Tokens::new(clunwithin_tokens)),
110      Some(Scope::Global),
111    )
112  }
113
114  if let Some(ref options) = options_opt {
115    if !options.nested.is_empty() {
116      state::assign_value(
117        &s!("nested_counters_{}", ctr),
118        options.nested.clone(),
119        Some(Scope::Global),
120      )
121    }
122  }
123
124  // default is equivalent to \arabic{ctr}, but w/o using the LaTeX macro!
125  let ctr_string = ctr.to_string();
126  def_macro(
127    T_CS!(s!("\\the{}", ctr)),
128    None,
129    Some(ExpansionBody::Closure(Rc::new(move |_args| {
130      let counter_value = counter_value(&ctr_string)?.value_of();
131      Ok(Tokens::new(ExplodeText!(counter_value)))
132    }))),
133    Some(ExpandableOptions {
134      scope: Some(Scope::Global),
135      ..ExpandableOptions::default()
136    }),
137  )?;
138  let p_ctr_cs = T_CS!(&s!("\\p@{}", ctr));
139  if lookup_definition(&p_ctr_cs)?.is_none() {
140    def_macro(
141      p_ctr_cs,
142      None,
143      Tokens::default(),
144      Some(ExpandableOptions {
145        scope: Some(Scope::Global),
146        ..ExpandableOptions::default()
147      }),
148    )?;
149  }
150
151  let mut prefix = match options_opt {
152    None => String::new(),
153    Some(ref opt) => opt.idprefix.to_string(),
154  };
155  if !prefix.is_empty() {
156    state::assign_value(
157      &s!("@ID@prefix@{}", ctr),
158      prefix.to_string(),
159      Some(Scope::Global),
160    );
161  } else {
162    prefix = state::lookup_string(&s!("@ID@prefix@{}", ctr));
163    if prefix.is_empty() {
164      prefix = ctr.to_string();
165    }
166  }
167  prefix = clean_id(&prefix);
168
169  if !prefix.is_empty() {
170    let idwithin = match options_opt {
171      Some(ref opts) => {
172        if opts.idwithin.is_empty() {
173          within
174        } else {
175          opts.idwithin
176        }
177      },
178      None => within,
179    }
180    .to_string();
181
182    let ctr_string = ctr.to_string();
183    let thectrid = s!("\\the{}@ID", ctr);
184    if !idwithin.is_empty() {
185      def_macro(
186        T_CS!(thectrid),
187        None,
188        Some(ExpansionBody::Closure(Rc::new(move |_args| {
189          Ok(mouth::tokenize_internal(&s!(
190            "\\expandafter\\ifx\\csname the{}@ID\\endcsname\\@empty\\else\\csname the{}@ID\\endcsname.\\fi {}\\csname @{}@ID\\endcsname",
191            idwithin,
192            idwithin,
193            prefix,
194            ctr_string
195          )))
196        }))),
197        Some(ExpandableOptions {
198          scope: Some(Scope::Global),
199          ..ExpandableOptions::default()
200        }),
201      )?;
202    } else {
203      def_macro(
204        T_CS!(thectrid),
205        None,
206        Some(ExpansionBody::Closure(Rc::new(move |_args| {
207          Ok(mouth::tokenize_internal(&s!(
208            "{prefix}\\csname @{ctr_string}@ID\\endcsname",
209          )))
210        }))),
211        Some(ExpandableOptions {
212          scope: Some(Scope::Global),
213          ..ExpandableOptions::default()
214        }),
215      )?;
216    }
217    def_macro(
218      T_CS!(s!("\\@{}@ID", ctr)),
219      None,
220      Some(ExpansionBody::Tokens(Tokens!(T_OTHER!("0")))),
221      Some(ExpandableOptions {
222        scope: Some(Scope::Global),
223        ..ExpandableOptions::default()
224      }),
225    )?;
226  }
227
228  Ok(())
229}
230/// Fetches the value associated with the counter C<$ctr>.
231pub fn counter_value(ctr: &str) -> Result<Number> {
232  match state::lookup_register(&s!("\\c@{ctr}"), Vec::new())? {
233    None => {
234      let message = s!("Counter {} was not defined; assuming 0", ctr);
235      Warn!("undefined", ctr, message);
236      Ok(Number::new(0))
237    },
238    Some(value) => Ok(Number::new(value.value_of())),
239  }
240}
241/// increments a named counter by a `Number`
242pub fn add_to_counter(ctr: &str, value: Number) -> Result<()> {
243  let v = counter_value(ctr)?.add(value);
244  state::assign_register(&s!("\\c@{ctr}"), v.into(), Some(Scope::Global), Vec::new())?;
245  after_assignment();
246  let id_cs = T_CS!(s!("\\@{ctr}@ID"));
247  def_macro(
248    id_cs,
249    None,
250    Tokens::new(Explode!(v.value_of())),
251    Some(ExpandableOptions {
252      scope: Some(Scope::Global),
253      ..ExpandableOptions::default()
254    }),
255  )
256}

258/// Analog of `\stepcounter`, steps the counter and returns the expansion of
259/// `\the$ctr`  Usually you should use `ref_step_counter(ctr)` instead.
260pub fn step_counter(ctr: &str, noreset: bool) -> Result<()> {
261  let value = counter_value(ctr)?;
262  let newvalue = value.add(Number::new(1));
263  let c_ctr = s!("\\c@{ctr}");
264  state::assign_register(&c_ctr, newvalue.into(), Some(Scope::Global), Vec::new())?;
265  state::after_assignment();
266  let token_value = Tokens::new(Explode!(newvalue.value_of()));
267  def_macro(
268    T_CS!(s!("\\@{ctr}@ID")),
269    None,
270    token_value,
271    Some(ExpandableOptions {
272      scope: Some(Scope::Global),
273      ..ExpandableOptions::default()
274    }),
275  )?;
276
277  // and reset any within counters!
278  if !noreset {
279    if let Some(nested) = state::lookup_tokens(&s!("\\cl@{ctr}")) {
280      for c in nested.unlist() {
281        reset_counter(&c)?;
282      }
283    }
284  }
285  Ok(())
286}

288/// Analog of `\refstepcounter`, steps the counter and returns a hash
289/// containing the keys `refnum` and `id`.
290///
291/// This makes it
292/// suitable for use in a `properties` option to constructors.
293/// The `id` is generated in parallel with the reference number
294/// to assist debugging.
295// TODO: Maybe these should be specialized types in Rust, rather than hashmaps?
296pub fn ref_step_counter(ctype: &str, noreset: bool) -> Result<HashMap<Stored>> {
297  let ctr = with_mapping("counter_for_type", ctype, |meaning| match meaning {
298    Some(Stored::String(ctr)) => arena::to_string(*ctr),
299    _ => ctype.to_string(),
300  });
301  step_counter(&ctr, noreset)?;
302  maybe_preempt_refnum(&ctr, false);
303
304  let the_ctr_id = s!("\\the{ctr}@ID");
305  let the_ctr = s!("\\the{ctr}");
306
307  let has_id: bool = if let Some(iddef) = state::lookup_definition(&T_CS!(&the_ctr_id))? {
308    if let Some(params) = iddef.get_parameters() {
309      params.get_num_args() == 0
310    } else {
311      true
312    }
313  } else {
314    false
315  };
316
317  let the_ctr_cs = T_CS!(&the_ctr);
318  let the_ctr_id_cs = T_CS!(&the_ctr_id);
319  def_macro(
320    T_CS!("\\@currentlabel"),
321    None,
322    the_ctr_cs,
323    Some(ExpandableOptions {
324      scope: Some(Scope::Global),
325      ..ExpandableOptions::default()
326    }),
327  )?;
328  if has_id {
329    def_macro(
330      T_CS!("\\@currentID"),
331      None,
332      the_ctr_id_cs,
333      Some(ExpandableOptions {
334        scope: Some(Scope::Global),
335        ..ExpandableOptions::default()
336      }),
337    )?;
338  }
339
340  let id = if has_id {
341    digest_literal(Tokens!(T_CS!(&the_ctr_id)))?.to_string()
342  } else {
343    String::new()
344  };
345
346  let refnum = digest_text(Tokens!(T_CS!(&the_ctr)))?;
347  let invocation;
348  {
349    invocation = build_invocation(T_CS!("\\lx@make@tags"), vec![Some(Tokens!(T_OTHER!(
350      ctype
351    )))])?;
352  }
353
354  let tags = stomach::digest(invocation)?;
355
356  // Any scopes activated for previous value of this counter (& any nested counters) must be
357  // removed. This may also include scopes activated for \label
358  deactivate_counter_scope(arena::pin(&ctr));
359
360  // And install the scope (if any) for this reference number.
361  state::assign_value("current_counter", ctr.to_string(), Some(Scope::Local));
362
363  let scope = arena::pin(format!("{ctr}:{refnum}"));
364  let mut receiver = VecDeque::new();
365  receiver.push_front(Stored::String(scope));
366  state::assign_value(
367    &s!("scopes_for_counter:{ctr}"),
368    receiver,
369    Some(Scope::Local),
370  );
371  state::activate_scope(scope);
372
373  Ok(stored_map!(
374    "tags" => Stored::Digested(tags),
375    "id" => Stored::String(arena::pin(id))
376  ))
377}

420/// Use to note a discovered label to support label-derived refererence numbers
421/// Can by used by \label, among others. Note we only record the label
422/// if it hasn't already been peeked, and consumed.
423pub fn maybe_note_label(label: &str) {
424  if has_value("LABEL_MAPPING_HOOK") {
425    let label = clean_label(label, Some(""));
426    let processed = state::lookup_string("PROCESSED_LABEL");
427    if processed.is_empty() || processed != label {
428      // Only if not already processed
429      state::remove_value("PROCESSED_LABEL");
430      state::assign_value(
431        "PEEKED_LABEL",
432        Stored::String(arena::pin(label)),
433        Some(Scope::Global),
434      );
435    }
436  }
437}

439fn deactivate_counter_scope(ctr: SymStr) {
440  let (scopes_for_counter, nested_counters) = arena::with(ctr, |cstr| {
441    (
442      s!("scopes_for_counter:{cstr}"),
443      s!("nested_counters_{cstr}"),
444    )
445  });
446  let scopes = lookup_value(&scopes_for_counter);
447  if let Some(Stored::VecDequeStored(stored_scopes)) = scopes {
448    for scope_stored in stored_scopes {
449      if let Stored::String(scope) = scope_stored {
450        state::deactivate_scope(scope);
451      } else {
452        panic!("assignmenet scopes should be stored as strings, got: {scope_stored:?}");
453      }
454    }
455  }
456
457  // TODO: if we ever want to unshift from the nested_counters, we'll need to also use
458  // Stored::VecDequeStored for them.
459  let nested = lookup_value(&nested_counters);
460  if let Some(Stored::Strings(stored_counters)) = nested {
461    for inner_ctr in &*stored_counters {
462      deactivate_counter_scope(*inner_ctr);
463    }
464  }
465}

467/// For UN-numbered units.
468/// Like `RefStepCounter`, but only steps the "uncounter",
469/// and returns only the id;  This is useful for unnumbered cases
470/// of objects that normally get both a refnum and id.
471pub fn ref_step_id(ctype: &str) -> Result<HashMap<Stored>> {
472  let ctr = with_mapping("counter_for_type", ctype, |mapping| match mapping {
473    Some(map) => map.to_string(),
474    None => ctype.to_string(),
475  });
476  let unctr = s!("UN{ctr}");
477  step_counter(&unctr, false)?;
478  maybe_preempt_refnum(&ctr, true);
479  let cunctr_val = lookup_number(&s!("\\c@{unctr}"))
480    .unwrap_or_default()
481    .value_of();
482  def_macro(
483    T_CS!(s!("\\@{ctr}@ID")),
484    None,
485    Tokens!(T_OTHER!("x"), Explode!(cunctr_val)),
486    Some(ExpandableOptions {
487      scope: Some(Scope::Global),
488      ..ExpandableOptions::default()
489    }),
490  )?;
491
492  let the_ctr_id = s!("\\the{ctr}@ID");
493  def_macro(T_CS!("\\@currentID"), None, T_CS!(&the_ctr_id), None)?;
494  Ok(stored_map!("id" =>
495    clean_id(&digest_literal(T_CS!(the_ctr_id))?.to_string())))
496}

498/// Resets the counter `ctr` to zero.
499pub fn reset_counter(ctr: &Token) -> Result<()> {
500  let (c_ctr, c_un_ctr, ctr_id) =
501    ctr.with_str(|ctr| (s!("\\c@{ctr}"), s!("\\c@UN{ctr}"), s!("\\@{ctr}@ID")));
502  state::assign_register(
503    &c_ctr,
504    Number::new(0).into(),
505    Some(Scope::Global),
506    Vec::new(),
507  )?;
508  if !ctr.with_str(|cstr| cstr.starts_with("UN")) {
509    // but not UN
510    state::assign_register(
511      &c_un_ctr,
512      Number::new(0).into(),
513      Some(Scope::Global),
514      Vec::new(),
515    )?;
516  }
517  def_macro(
518    T_CS!(ctr_id),
519    None,
520    Tokens!(T_OTHER!("0")),
521    Some(ExpandableOptions {
522      scope: Some(Scope::Global),
523      ..ExpandableOptions::default()
524    }),
525  )?;
526  // and reset any within counters!
527  if let Some(nested) = state::lookup_tokens(&s!("\\cl@{ctr}")) {
528    for c in nested.unlist() {
529      reset_counter(&c)?;
530    }
531  }
532  Ok(())
533}

535/// Create id, and tags for an itemize type \item
536pub fn ref_step_item_counter(tag_opt: Option<&Tokens>) -> Result<HashMap<Stored>> {
537  let counter = state::lookup_string("itemcounter");
538  let n = lookup_int("itemization_items");
539  state::assign_value("itemization_items", n + 1, None);
540  let mut attr: HashMap<Stored> = HashMap::default();
541  if n > 0 {
542    if let Some(sep) = lookup_dimension("\\itemsep") {
543      let default_opt = lookup_dimension("\\lx@default@itemsep");
544      if default_opt.is_none() || sep.value_of() != default_opt.unwrap().value_of() {
545        attr.insert("itemsep", sep.into());
546      }
547    }
548  }
549
550  let mut result = if let Some(tag) = tag_opt {
551    let mut props = ref_step_id(&counter)?;
552    if tag.is_empty() {
553      return Ok(props);
554    }
555    let formatter = if counter.starts_with("@desc") {
556      T_CS!("\\descriptionlabel")
557    } else {
558      T_CS!("\\makelabel")
559    };
560    let counter_name = s!("\\{counter}name");
561    let typename = if is_defined(&counter_name) {
562      T_CS!(counter_name)
563    } else {
564      T_CS!("\\itemtyperefname")
565    };
566
567    let mut tag_tokens = vec![
568      T_BEGIN!(),
569      T_CS!("\\let"),
570      T_CS!(s!("\\the{counter}")),
571      T_CS!("\\@empty"),
572      T_CS!("\\def"),
573      T_CS!(s!("\\fnum@{counter}")),
574      T_BEGIN!(),
575      formatter,
576      T_BEGIN!(),
577    ];
578    // TODO: Another iffy clone...
579    let reverted_tag = (*tag).clone().revert();
580    tag_tokens.extend(reverted_tag.clone());
581    tag_tokens.extend(vec![
582      T_END!(),
583      T_END!(),
584      T_CS!("\\def"),
585      T_CS!(s!("\\typerefnum@{counter}")),
586      T_BEGIN!(),
587      typename,
588      T_SPACE!(),
589    ]);
590    tag_tokens.extend(reverted_tag);
591    tag_tokens.push(T_END!());
592    tag_tokens.extend(
593      build_invocation(T_CS!("\\lx@make@tags"), vec![Some(Tokens!(T_OTHER!(
594        counter
595      )))])?
596      .unlist(),
597    );
598    tag_tokens.push(T_END!());
599
600    let tags = stomach::digest(tag_tokens)?;
601    if !tags.is_empty()? {
602      props.insert("tags", tags.into());
603    }
604    props
605  } else {
606    ref_step_counter(&counter, false)?
607  };
608  for (k, v) in attr.into_iter() {
609    result.insert_sym(k, v);
610  }
611  Ok(result)
612}

629/// Prepare for an list (itemize/enumerate/description/etc)
630/// by determining the right counter (level)
631/// and binding the right \item ( \$type@item, if $type is defined)
632pub fn begin_itemize(
633  itype: &str,
634  counter: Option<&str>,
635  options: BeginItemizeOptions,
636) -> Result<HashMap<Stored>> {
637  // The list-type and level of the *containing* list (if any!)
638  let outercounter = state::lookup_string("itemcounter");
639  let outerlevel = if !outercounter.is_empty() {
640    lookup_int(&s!("{outercounter}level"))
641  } else {
642    0
643  };
644  let counter = counter.unwrap_or("@item");
645  let listlevel = lookup_int("itemization_level") + 1; // level for this list overall
646  let level = lookup_int(&s!("{counter}level")) + // level for lists of specific type
647    (if options.nolevel { 0 } else { 1 });
648  AssignRegister!(
649    "\\itemsep",
650    lookup_dimension("\\lx@default@itemsep")
651      .unwrap_or_default()
652      .into()
653  );
654  state::assign_value("itemization_level", listlevel, None);
655  state::assign_value(&s!("{counter}level"), level, None);
656  state::assign_value("itemization_items", 0, None);
657  let listpostfix = roman!(listlevel).to_string();
658  let postfix = roman!(level).to_string();
659  let mut usecounter = counter.to_string();
660  if !options.nolevel && !postfix.is_empty() {
661    usecounter.push_str(&postfix);
662  }
663  if !itype.is_empty() {
664    let itype_cs = T_CS!(s!("\\{itype}@item"));
665    state::let_i(&T_CS!("\\item"), &itype_cs, None);
666  }
667  // In case within odd environment.
668  state::let_i(&T_CS!("\\par"), &T_CS!("\\lx@normal@par"), None);
669  def_macro(
670    T_CS!("\\@listctr"),
671    None,
672    Tokens!(Explode!(usecounter)),
673    None,
674  )?;
675  // Now arrange that this list's id's are relative to the current (outer) item (if any)
676  // And that the items within this list's id's are relative to this (new) list.
677  state::assign_value("itemcounter", Stored::String(arena::pin(&usecounter)), None);
678  let listcounter = s!("@itemize{listpostfix}");
679  if lookup_definition(&T_CS!(s!("\\c@{listcounter}")))?.is_none() {
680    //Create new list counters as needed
681    new_counter(&listcounter, "", None)?;
682  }
683  if !outercounter.is_empty() {
684    // Make this list's ID relative to outer list's ID
685    let outerusecounter = s!("{outercounter}{}", roman!(outerlevel).to_string());
686    let thectr = s!("\\the{listcounter}@ID");
687    let theexpansion = s!("\\the{outerusecounter}@ID.I\\arabic{{{listcounter}}}");
688    def_macro(
689      T_CS!(thectr),
690      None,
691      mouth::tokenize_internal(&theexpansion),
692      None,
693    )?;
694
695    // AND reset this list's counter when the outer item is stepped
696    let mut cl_toks = vec![T_CS!(&listcounter)];
697    let cl_name = s!("\\cl@{outerusecounter}");
698    if let Some(Stored::Tokens(tks)) = lookup_value(&cl_name) {
699      cl_toks.extend(tks.unlist());
700    }
701    state::assign_value(
702      &cl_name,
703      Stored::Tokens(Tokens::new(cl_toks)),
704      Some(Scope::Global),
705    );
706  }
707  // format the id of \item's relative to the id of this list
708  let useexp = mouth::tokenize_internal(&s!("\\the{listcounter}@ID.i\\@{usecounter}@ID"));
709  def_macro(T_CS!(s!("\\the{usecounter}@ID")), None, useexp, None)?;
710
711  let mut series = if let Some(s) = options.series {
712    s.to_string()
713  } else {
714    String::new()
715  };
716  if let Some(start) = options.start {
717    SetCounter!(usecounter, start);
718    add_to_counter(&usecounter, Number(-1))?;
719  } else if let Some(s) = match options.resume {
720    Some(s) => Some(s),
721    None => options.resume_star,
722  } {
723    if s != "noseries" {
724      series = s;
725      // TODO:
726      // SetCounter!(usecounter,
727      //   lookup_int(&s!("enumitem_series_{}_last",series)),
728      //   state);
729    }
730  } else {
731    reset_counter(&T_OTHER!(&usecounter))?;
732  }
733
734  let mut rsc = ref_step_counter(&s!("@itemize{listpostfix}"), false)?;
735  rsc.insert("counter", usecounter.into());
736  rsc.insert("series", series.into());
737  Ok(rsc)
738}

740/// Copies the current id, tags, and inlist counter values into whatsit properties
741pub fn rescue_caption_counters(captype: &str, whatsit: &mut Whatsit) {
742  let tagskey = &s!("{captype}_tags");
743  if let Some(tags) = state::remove_value(tagskey) {
744    state::assign_value(tagskey, false, Some(Scope::Global));
745    whatsit.set_property("tags", tags);
746  }
747  let idkey = s!("{captype}_id");
748  if let Some(id) = state::remove_value(&idkey) {
749    state::assign_value(&idkey, false, Some(Scope::Global));
750    whatsit.set_property("id", id);
751  }
752  let inlistkey = s!("{captype}_inlist");
753  if let Some(inlist) = state::remove_value(&inlistkey) {
754    state::assign_value(&inlistkey, false, Some(Scope::Global));
755    whatsit.set_property("inlist", inlist);
756  }
