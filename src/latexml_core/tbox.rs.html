22/// Box is a Rust keyword, so we use "Tbox" instead, as in "TeX Box"
23#[derive(Debug, Clone)]
24pub struct Tbox {
25  /// plain-text content
26  pub text:       SymStr,
27  /// associated font for `text`
28  pub font:       Rc<Font>,
29  /// source location where the box originated (`None` = unknown)
30  pub locator:    Option<Locator>,
31  /// misc properties, such as sizing information
32  pub properties: HashMap<Stored>,
33  /// a Tokens list containing the TeX that created (or could have) the Tbox.
34  pub tokens:     Tokens,
35}
36
37impl Default for Tbox {
38  fn default() -> Self {
39    Tbox {
40      text:       pin!(""),
41      font:       Rc::new(Font::text_default()),
42      locator:    None,
43      properties: HashMap::default(),
44      tokens:     Tokens!(),
45    }
46  }
47}

49impl PartialEq for Tbox {

66impl Tbox {
67  /// creates a new Tbox.
68  /// If `font_opt` or `locator_opt` are None, they are obtained from the
69  /// currently active state? Note that `text` can
70  /// be empty, which contributes nothing to the generated document,
71  /// but does record the TeX code (in the tokens).
72  pub fn new(
73    text: SymStr,
74    font_opt: Option<Rc<Font>>,
75    locator_opt: Option<Locator>,
76    tokens_opt: Tokens,
77    mut properties: HashMap<Stored>,
78  ) -> Self {
79    // A text box always has a source position (Perl parity): use the passed
80    // locator, else the gullet's current one. Stored as `Some`.
81    let locator = Some(locator_opt.unwrap_or_else(gullet::get_locator));
82    let mut font = match font_opt {
83      Some(f) => f,
84      None => lookup_font().unwrap(),
85    };
86    let empty_sym = pin!("");
87    let tokens = if text != empty_sym && tokens_opt.is_empty() {
88      Tokens!(Token { text, code: Catcode::OTHER, #[cfg(feature = "token-locators")] loc: 0 })
89    } else {
90      tokens_opt
91    };
92
93    // Perl: if ((!defined $properties{isSpace}) && (defined $string) && ($string =~ /^\s*$/)) {
94    //         $properties{isSpace} = 1; }
95    // Auto-mark all-whitespace text as isSpace (matches Perl Box() behavior)
96    if !properties.contains_key("isSpace") && text != empty_sym {
97      let is_all_ws = arena::with(text, |s| {
98        !s.is_empty() && s.chars().all(|c| c.is_whitespace())
99      });
100      if is_all_ws {
101        properties.insert("isSpace", Stored::Bool(true));
102      }
103    }
104
105    if properties.contains_key("isSpace")
106      && (properties.contains_key("width")
107        || properties.contains_key("height")
108        || properties.contains_key("depth"))
109    {
110      properties
111        .entry("width")
112        .or_insert_with(|| Stored::Dimension(Dimension::default()));
113      properties
114        .entry("height")
115        .or_insert_with(|| Stored::Dimension(Dimension::default()));
116      properties
117        .entry("depth")
118        .or_insert_with(|| Stored::Dimension(Dimension::default()));
119    }
120    if crate::state::lookup_bool_sym(crate::pin!("IN_MATH")) {
121      properties.insert("mode", Stored::String(pin!("math")));
122      if text != empty_sym {
123        with_value(
124          &arena::with(text, |text_str| s!("math_token_attributes_{}", text_str)),
125          |value_opt| {
126            if let Some(Stored::HashString(attr)) = value_opt {
127              for (key, value) in attr.iter() {
128                properties
129                  .entry(key)
130                  .or_insert_with(|| Stored::String(arena::pin(value)));
131              }
132            }
133          },
134        );
135      }
136      font = Rc::new(arena::with(text, |text_str| font.specialize(text_str)));
137    }
138    Tbox {
139      text,
140      font,
141      locator,
142      properties,
143      tokens,
144    }
145  }
146  /// checks if the text content is empty
147  pub fn is_empty(&self) -> bool {
148    // 1. A space-like thing
149    // 2. empty (or whitespace) text content
150    self.get_property_bool("isEmpty")
151      || self.get_property_bool("isSpace")
152      || arena::with(self.text, |text| text.trim().is_empty())
153  }
154
155  /// Whether this box is in math mode.
156  /// Perl: Box.pm::isMath (L79-81): `($mode || 'restricted_horizontal') =~ /math$/`.
157  /// Matches "math", "inline_math", "display_math".
158  pub fn is_math(&self) -> bool {
159    match self.properties.get("mode") {
160      Some(Stored::String(s)) => arena::with(*s, |m| m.ends_with("math")),
161      _ => false,
162    }
163  }
164
165  /// Batch-insert properties. Equivalent to calling `set_property` for each
166  /// entry, but avoids re-looking up the properties map per call.
167  /// Perl: Box.pm::setProperties (L171-175).
168  pub fn set_properties<I>(&mut self, entries: I)
169  where I: IntoIterator<Item = (&'static str, Stored)> {
170    for (key, value) in entries {
171      self.properties.insert(key, value);
172    }
173  }
174
175  /// height + depth of this box (the TeX "total height").
176  /// Returns Dimension::default() (0pt) when either component is absent.
177  /// Perl: Box.pm::getTotalHeight (L202-210).
178  pub fn total_height(&self) -> Dimension {
179    let h = match self.properties.get("height") {
180      Some(Stored::Dimension(d)) => d.0,
181      _ => 0,
182    };
183    let d = match self.properties.get("depth") {
184      Some(Stored::Dimension(d)) => d.0,
185      _ => 0,
186    };
187    Dimension(h + d)
188  }
189}

191impl BoxOps for Tbox {
192  fn get_tokens(&self) -> Option<&Tokens> { Some(&self.tokens) }
193  fn get_properties(&self) -> &HashMap<Stored> { &self.properties }
194  fn get_property(&self, key: &str) -> Option<Cow<'_, Stored>> {
195    let props = &self.properties;
196    if key == "isSpace" {
197      match props.get(key) {
198        Some(value) => Some(Cow::Owned(value.clone())),
199        None => {
200          let tex = self
201            .get_tokens()
202            .map(|tks| tks.clone().untex())
203            .unwrap_or_default(); // !
204          if !tex.is_empty() && tex.chars().all(char::is_whitespace) {
205            // Check the TeX code, not (just) the string!
206            Some(Cow::Owned(Stored::Bool(true)))
207          } else {
208            None
209          }
210        },
211      }
212    } else {
213      props.get(key).map(|v| Cow::Owned(v.clone()))
214    }
215  }
216  fn with_properties<R, FnR>(&self, caller: FnR) -> R
217  where FnR: FnOnce(&HashMap<Stored>) -> R {
218    caller(&self.properties)
219  }
220  fn get_properties_mut(&mut self) -> &mut HashMap<Stored> { &mut self.properties }
221  fn get_string(&self) -> Result<Cow<'_, str>> {
222    // TODO: Should we switch these to symbols? are they used often?
223    Ok(Cow::Owned(arena::with(self.text, |text| text.to_string())))
224  }
225
226  fn be_absorbed(&self, document: &mut Document) -> Result<Vec<Node>> {
227    let text = self.get_string()?;
228    let font = &self.font;
229    let mode = match self.properties.get("mode") {
230      Some(Stored::String(s)) => *s,
231      _ => pin!("text"),
232    };
233
234    if !text.is_empty() {
235      // Perl Box::isMath: `mode =~ /math$/` matches "math" / "inline_math" / "display_math".
236      let mode_is_math = arena::with(mode, |m| m.ends_with("math"));
237      if mode_is_math {
238        // Perl: DefMath ?#isMath — in text context, produce plain text.
239        // Check if we're inside a math element by walking up the DOM.
240        // This handles \And in author frontmatter (text) while preserving
241        // \times inside XMText within math.
242        let in_math_context = {
243          let mut node = document.node.clone();
244          let mut found = false;
245          loop {
246            let qname = crate::document::get_node_qname(&node);
247            let hit = arena::with(qname, |s| s.contains("XM") || s.contains("Math"));
248            if hit {
249              found = true;
250              break;
251            }
252            match node.get_parent() {
253              Some(parent) => node = parent,
254              None => break,
255            }
256          }
257          found
258        };
259        if in_math_context {
260          Ok(vec![document.insert_math_token(
261            &text,
262            Stored::cast_to_string_hash(&self.properties),
263            Some(font),
264          )?])
265        } else {
266          // Text fallback: produce plain text (Perl's ?#isMath else branch)
267          match document.open_text(&text, font)? {
268            None => Ok(Vec::new()),
269            Some(node) => Ok(vec![node]),
270          }
271        }
272      } else {
273        match document.open_text(&text, font)? {
274          None => Ok(Vec::new()),
275          Some(node) => Ok(vec![node]),
276        }
277      }
278    } else {
279      Ok(Vec::new())
280    }
281  }
282
283  fn get_font(&self) -> Result<Option<Cow<'_, Font>>> { Ok(Some(Cow::Borrowed(&self.font))) }
284
285  fn compute_size(&self, options: HashMap<Stored>) -> Result<(Dimension, Dimension, Dimension)> {
286    match self.get_property("body") { Some(body_stored) => {
287      if let Stored::Digested(ref body) = *body_stored {
288        body.compute_size(options)
289      } else {
290        panic!("the stored 'body' property should always be a Stored::Digested enum case.");
291      }
292    } _ => {
293      Ok(self.font.compute_string_size(&self.get_string()?, options))
294    }}
295  }
296}
