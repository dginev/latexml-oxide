21/// Box is a Rust keyword, so we use "Tbox" instead, as in "TeX Box"
22#[derive(Debug, Clone)]
23pub struct Tbox {
24  /// plain-text content
25  pub text:       SymStr,
26  /// associated font for `text`
27  pub font:       Rc<Font>,
28  /// source location where the box originated
29  pub locator:    Locator,
30  /// misc properties, such as sizing information
31  pub properties: HashMap<Stored>,
32  /// a Tokens list containing the TeX that created (or could have) the Tbox.
33  pub tokens:     Tokens,
34}
35
36impl Default for Tbox {
37  fn default() -> Self {
38    Tbox {
39      text:       *EMPTY_SYM,
40      font:       Rc::new(Font::text_default()),
41      locator:    Locator::default(),
42      properties: HashMap::default(),
43      tokens:     Tokens!(),
44    }
45  }
46}

48impl PartialEq for Tbox {
49  // Should this compare fonts too?

65impl Tbox {
66  /// creates a new Tbox.
67  /// If `font_opt` or `locator_opt` are None, they are obtained from the
68  /// currently active state? Note that `text` can
69  /// be empty, which contributes nothing to the generated document,
70  /// but does record the TeX code (in the tokens).
71  pub fn new(
72    text: SymStr,
73    font_opt: Option<Rc<Font>>,
74    locator_opt: Option<Locator>,
75    tokens_opt: Tokens,
76    mut properties: HashMap<Stored>,
77  ) -> Self {
78    let locator = locator_opt.unwrap_or_else(gullet::get_locator);
79    let mut font = match font_opt {
80      Some(f) => f,
81      None => lookup_font().unwrap(),
82    };
83    let empty_sym = *EMPTY_SYM;
84    let tokens = if text != empty_sym && tokens_opt.is_empty() {
85      Tokens!(Token { text, code: Catcode::OTHER })
86    } else {
87      tokens_opt
88    };
89
90    // Perl: if ((!defined $properties{isSpace}) && (defined $string) && ($string =~ /^\s*$/)) {
91    //         $properties{isSpace} = 1; }
92    // Auto-mark all-whitespace text as isSpace (matches Perl Box() behavior)
93    if !properties.contains_key("isSpace") && text != empty_sym {
94      let is_all_ws = arena::with(text, |s| !s.is_empty() && s.chars().all(|c| c.is_whitespace()));
95      if is_all_ws {
96        properties.insert("isSpace", Stored::Bool(true));
97      }
98    }
99
100    if properties.contains_key("isSpace")
101      && (properties.contains_key("width")
102        || properties.contains_key("height")
103        || properties.contains_key("depth"))
104    {
105      properties
106        .entry("width")
107        .or_insert_with(|| Stored::Dimension(Dimension::default()));
108      properties
109        .entry("height")
110        .or_insert_with(|| Stored::Dimension(Dimension::default()));
111      properties
112        .entry("depth")
113        .or_insert_with(|| Stored::Dimension(Dimension::default()));
114    }
115    if lookup_bool("IN_MATH") {
116      properties.insert("mode", Stored::String(*MATH_SYM));
117      if text != empty_sym {
118        with_value(
119          &arena::with(text, |text_str| s!("math_token_attributes_{}", text_str)),
120          |value_opt| {
121            if let Some(Stored::HashString(attr)) = value_opt {
122              for (key, value) in attr.iter() {
123                properties
124                  .entry(key)
125                  .or_insert_with(|| Stored::String(arena::pin(value)));
126              }
127            }
128          },
129        );
130      }
131      font = Rc::new(arena::with(text, |text_str| font.specialize(text_str)));
132    }
133    Tbox {
134      text,
135      font,
136      locator,
137      properties,
138      tokens,
139    }
140  }
141  /// checks if the text content is empty
142  pub fn is_empty(&self) -> bool {
143    // 1. A space-like thing
144    // 2. empty (or whitespace) text content
145    self.get_property_bool("isEmpty")
146      || self.get_property_bool("isSpace")
147      || arena::with(self.text, |text| text.trim().is_empty())
148  }
149}

151impl BoxOps for Tbox {
152  fn get_tokens(&self) -> Option<&Tokens> { Some(&self.tokens) }
153  fn get_properties(&self) -> &HashMap<Stored> { &self.properties }
154  fn get_property(&self, key: &str) -> Option<Cow<'_, Stored>> {
155    let props = &self.properties;
156    if key == "isSpace" {
157      match props.get(key) {
158        Some(value) => Some(Cow::Owned(value.clone())),
159        None => {
160          let tex = self
161            .get_tokens()
162            .map(|tks| tks.clone().untex())
163            .unwrap_or_default(); // !
164          if !tex.is_empty() && tex.chars().all(char::is_whitespace) {
165            // Check the TeX code, not (just) the string!
166            Some(Cow::Owned(Stored::Bool(true)))
167          } else {
168            None
169          }
170        },
171      }
172    } else {
173      props.get(key).map(|v| Cow::Owned(v.clone()))
174    }
175  }
176  fn with_properties<R, FnR>(&self, caller: FnR) -> R
177  where FnR: FnOnce(&HashMap<Stored>) -> R {
178    caller(&self.properties)
179  }
180  fn get_properties_mut(&mut self) -> &mut HashMap<Stored> { &mut self.properties }
181  fn get_string(&self) -> Result<Cow<'_, str>> {
182    // TODO: Should we switch these to symbols? are they used often?
183    Ok(Cow::Owned(arena::with(self.text, |text| text.to_string())))
184  }
185
186  fn be_absorbed(&self, document: &mut Document) -> Result<Vec<Node>> {
187    let text = self.get_string()?;
188    let font = &self.font;
189    let mode = match self.properties.get("mode") {
190      Some(Stored::String(s)) => *s,
191      _ => *TEXT_SYM,
192    };
193
194    if !text.is_empty() {
195      if mode == *MATH_SYM {
196        // Perl: DefMath ?#isMath — in text context, produce plain text.
197        // Check if we're inside a math element by walking up the DOM.
198        // This handles \And in author frontmatter (text) while preserving
199        // \times inside XMText within math.
200        let in_math_context = {
201          let mut node = document.node.clone();
202          let mut found = false;
203          loop {
204            let qname = crate::document::get_node_qname(&node);
205            let name = arena::to_string(qname);
206            if name.contains("XM") || name.contains("Math") {
207              found = true;
208              break;
209            }
210            match node.get_parent() {
211              Some(parent) => node = parent,
212              None => break,
213            }
214          }
215          found
216        };
217        if in_math_context {
218          Ok(vec![document.insert_math_token(
219            &text,
220            Stored::cast_to_string_hash(&self.properties),
221            Some(font),
222          )?])
223        } else {
224          // Text fallback: produce plain text (Perl's ?#isMath else branch)
225          match document.open_text(&text, font)? {
226            None => Ok(Vec::new()),
227            Some(node) => Ok(vec![node]),
228          }
229        }
230      } else {
231        match document.open_text(&text, font)? {
232          None => Ok(Vec::new()),
233          Some(node) => Ok(vec![node]),
234        }
235      }
236    } else {
237      Ok(Vec::new())
238    }
239  }
240
241  fn get_font(&self) -> Result<Option<Cow<'_, Font>>> { Ok(Some(Cow::Borrowed(&self.font))) }
242
243  fn compute_size(&self, options: HashMap<Stored>) -> Result<(Dimension, Dimension, Dimension)> {
244    if let Some(body_stored) = self.get_property("body") {
245      if let Stored::Digested(ref body) = *body_stored {
246        body.compute_size(options)
247      } else {
248        panic!("the stored 'body' property should always be a Stored::Digested enum case.");
249      }
250    } else {
251      Ok(self.font.compute_string_size(&self.get_string()?, options))
252    }
253  }
254}
