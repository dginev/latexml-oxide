9pub type LigatureClosure = Rc<dyn Fn(&str) -> String>;
10pub type FontTestClosure = Rc<dyn Fn(&Font) -> bool>;
11pub type LigatureMatcher =
12  Rc<dyn Fn(&mut Document, &mut Node) -> Result<Option<(usize, String, MathLigatureOptions)>>>;

14#[derive(Debug, Default, Clone, PartialEq, Eq)]
15pub struct MathLigatureOptions {
16  pub role:    Option<String>,
17  pub name:    Option<String>,
18  pub meaning: Option<String>,
19}
20
21impl MathLigatureOptions {
22  pub fn sorted_each(&self) -> [(&str, Option<&String>); 3] {
23    [
24      ("meaning", self.meaning.as_ref()),
25      ("name", self.name.as_ref()),
26      ("role", self.role.as_ref()),
27    ]
28  }
29}

31#[derive(Clone, Default)]
32pub struct Ligature {
33  pub id:        usize,
34  pub regex:     Option<String>,
35  pub code:      Option<LigatureClosure>,
36  pub font_test: Option<FontTestClosure>,
37  pub matcher:   Option<LigatureMatcher>,
38}
39
40impl fmt::Debug for Ligature {
41  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result { write!(f, "{:?}", self.regex) }
42}
43
44impl PartialEq for Ligature {
45  fn eq(&self, other: &Ligature) -> bool { self.id == other.id }
46}

52  #[test]

60  #[test]
61  fn math_ligature_options_equality() {
62    let a = MathLigatureOptions {
63      role:    Some("ADDOP".into()),
64      name:    None,
65      meaning: Some("plus".into()),
66    };
67    let b = MathLigatureOptions {
68      role:    Some("ADDOP".into()),
69      name:    None,
70      meaning: Some("plus".into()),
71    };
72    let c = MathLigatureOptions {
73      role:    Some("RELOP".into()), // differs
74      name:    None,
75      meaning: Some("plus".into()),
76    };
77    assert_eq!(a, b);
78    assert_ne!(a, c);
79  }

81  #[test]
82  fn math_ligature_sorted_each_fixed_order() {
83    // Perl parity: output always ordered (meaning, name, role).
84    let m = MathLigatureOptions {
85      role:    Some("r".into()),
86      name:    Some("n".into()),
87      meaning: Some("m".into()),
88    };
89    let ordered = m.sorted_each();
90    assert_eq!(ordered[0].0, "meaning");
91    assert_eq!(ordered[1].0, "name");
92    assert_eq!(ordered[2].0, "role");
93  }

95  #[test]
96  fn math_ligature_sorted_each_none_values_preserved() {
97    // sorted_each reports None when a field is None.
98    let m = MathLigatureOptions::default();
99    let ordered = m.sorted_each();
100    for (_, v) in ordered {
101      assert!(v.is_none());
102    }
103  }
