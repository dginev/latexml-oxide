116/// Configuration for the Core processing
117#[derive(Default)]
118pub struct CoreOptions {
119  // First, state::related options:
120  /// a custom schema-induced model (default is `None`) for the final XML
121  pub model:            Option<Model>,
122  /// default is 0, sub-zero values are quiet, positive values are verbose
123  pub verbosity:        Option<i32>,
124  /// strict error-reporting (is this still used?)
125  pub strict:           Option<bool>,
126  /// toggle preserving comments in the XML on/off
127  pub include_comments: Option<bool>,
128  /// toggle loading raw .sty modules on/off
129  pub include_styles:   Option<bool>,
130  /// disable math parsing (enabled by default)
131  pub nomathparse:      Option<bool>,
132  /// an optional, fixed id prefix for all xml:id attributes
133  pub documentid:       Option<String>,
134  /// the list of paths used for loading TeX sources and packages
135  pub search_paths:     Option<Vec<String>>,
136  /// the list of paths used for loading graphics assets
137  pub graphics_paths:   Option<Vec<String>>,
138  /// set an explicit encoding of the input text
139  pub input_encoding:   Option<String>,
140  /// a list of package names to preload before processing start
141  pub preload:          Option<Vec<String>>,
142}

144impl Core {
145  /// instantiate a new Core processor
146  pub fn new(options: CoreOptions) -> Self {
147    let preload = options.preload.unwrap_or_default();
148    // pass on the state::options, defaults are handled in state::new
149    let state_options = StateOptions {
150      model: options.model,
151      verbosity: options.verbosity,
152      strict: options.strict,
153      include_comments: options.include_comments,
154      documentid: options.documentid,
155      search_paths: options.search_paths,
156      graphics_paths: options.graphics_paths,
157      include_styles: options.include_styles,
158      input_encoding: options.input_encoding,
159      nomathparse: options.nomathparse,
160      ..StateOptions::default()
161    };
162    stomach::set_stomach(Stomach::default());
163    set_state(State::new(state_options));
164    Core { preload }
165  }

167/// Common operations for Box-like (digested) data
168pub trait BoxOps: Object {
169  /// If composite, unwrap into the contained digested objects (or return self)
170  fn unlist(&self) -> Vec<Digested> { Vec::new() }
171  fn unlist_ref(&self) -> Vec<Cow<'_, Digested>> { Vec::new() }
172  /// absorb the current object into the `Document` XML - returning the corresponding nodes
173  fn be_absorbed(&self, document: &mut Document) -> Result<Vec<Node>>;
174  /// be_absorbed but with allowed side-effects on the carrier (for `Alignment` only)
175  fn be_absorbed_mut(&mut self, _document: &mut Document) -> Result<Vec<Node>> {
176    self.be_absorbed(_document)
177  }
178  /// build a string representation of the underlying digested data
179  fn get_string(&self) -> Result<Cow<'_, str>>;
180  /// get the underlying tokens (preceding digestion)
181  fn get_tokens(&self) -> Option<&Tokens> { None }
182  /// deprecated: get the map of named properties. This can not be usable as long as we have any
183  /// data behind a RefCell wrapper.
184  /// Use `with_properties` instead.
185  fn get_properties(&self) -> &HashMap<Stored> { &NO_PROPERTIES }
186
187  /// execute a function using this object's named properties
188  fn with_properties<R, FnR>(&self, caller: FnR) -> R
189  where FnR: FnOnce(&HashMap<Stored>) -> R;
190  /// get a mutable reference to the map of named properties
191  fn get_properties_mut(&mut self) -> &mut HashMap<Stored> {
192    panic!("get_properties_mut called on type without mutable properties");
193  }
194  /// set a named property (allows all `Stored` types for values)
195  fn set_property<T: Into<Stored>>(&mut self, key: &str, value: T) {
196    self.get_properties_mut().insert(key, value.into());
197  }
198  /// get a single named property (with special "isSpace" check)
199  fn get_property(&self, key: &str) -> Option<Cow<'_, Stored>> {
200    self.with_properties(|props| {
201      if key == "isSpace" {
202        match props.get(key) {
203          Some(value) => Some(Cow::Owned(value.clone())),
204          None => {
205            let tex = self
206              .get_tokens()
207              .map(|tks| tks.clone().untex())
208              .unwrap_or_default(); // !
209            if !tex.is_empty() && tex.chars().all(char::is_whitespace) {
210              // Check the TeX code, not (just) the string!
211              Some(Cow::Owned(Stored::Bool(true)))
212            } else {
213              None
214            }
215          },
216        }
217      } else {
218        props.get(key).map(|v| Cow::Owned(v.clone()))
219      }
220    })
221  }
222  fn get_property_string(&self, key: &str) -> String {
223    self
224      .get_property(key)
225      .map(|v| v.to_string())
226      .unwrap_or_default()
227  }
228  /// get a mutable reference to a single named property (does NOT have the "isSpace" check)
229  fn get_property_mut(&mut self, key: &str) -> Option<&mut Stored> {
230    self.get_properties_mut().get_mut(key)
231  }
232  /// checks if a property key has been set
233  fn has_property(&self, key: &str) -> bool {
234    self.with_properties(|props| props.contains_key(key))
235  }
236  /// obtains a boolean property value (false unless `Stored::Bool`)
237  fn get_property_bool(&self, key: &str) -> bool {
238    self.with_properties(|props| matches!(props.get(key), Some(Stored::Bool(true))))
239  }
240  /// obtains the "body" of a digested object which captured it
241  fn get_body(&self) -> Result<Option<Digested>> {
242    Error!(
243      "boxops",
244      "get_body",
245      "Generic BoxOps::get_body should never be called!"
246    );
247    Ok(None)
248  }
249  /// gets the associated font, if any
250  fn get_font(&self) -> Result<Option<Cow<'_, Font>>>;
251  /// sets an associated font
252  fn set_font(&mut self, _font: Rc<Font>) { /* no-op for types without font */
253  }
254  /// sets a "width" property, for sizing
255  fn set_width<T: Into<Stored>>(&mut self, width: T) { self.set_property("width", width); }
256
257  // For the dimensions of boxes, we'll store the (lazily) computed size as:
258  //    cached_width, cached_height, cached_depth
259  // and the explicitly requested/assigned size as
260  //    width, height, depth.
261  // Generally speaking, an XML element should only get width, height, depth
262  // attributes when they were explicitly set.
263  // However, when requesting the size of a box, you'd get either (w/ explicit size overriding)
264
265  /// gets the "width" property value, if any
266  fn get_width(&self, options: Option<HashMap<Stored>>) -> Result<Option<RegisterValue>> {
267    if !self.has_property("width") && !self.has_property("cached_width") {
268      // TODO: Restore caching?
269      // self.compute_size_store(options.unwrap_or_default())?
270      let (w, ..) = self.compute_size(options.unwrap_or_default())?;
271      return Ok(Some(RegisterValue::Dimension(w)));
272    }
273
274    // Convert MuGlue/MuDimension widths to pt (1mu = font_size/18). `\the\wd`
275    // is a dimension query — the result must be Dimension-typed. Without
276    // this conversion `\hbox{\,}` width came back as `MuGlue(3mu)` and
277    // formatted as `3.0pt` (raw mu treated as pt) instead of `1.66663pt`.
278    // Order of ops (div by 18 then mul by fs) matches Perl integer
279    // truncation: see `mu_to_pt_value` in store.rs.
280    fn coerce_mu(val: &Stored) -> Option<RegisterValue> {
281      match val {
282        Stored::MuGlue(g) => {
283          let fs = crate::state::lookup_font()
284            .and_then(|f| f.get_size())
285            .unwrap_or(10.0);
286          let muwidth = (fs * crate::common::numeric_ops::UNITY_F64 / 18.0) as i64;
287          let pt_scaled =
288            (g.value_of() as f64 * muwidth as f64 / crate::common::numeric_ops::UNITY_F64).trunc();
289          Some(RegisterValue::Dimension(
290            crate::common::dimension::Dimension::new(pt_scaled as i64),
291          ))
292        },
293        Stored::MuDimension(d) => {
294          let fs = crate::state::lookup_font()
295            .and_then(|f| f.get_size())
296            .unwrap_or(10.0);
297          let pt_scaled = (d.value_of() / 18) as f64 * fs;
298          Some(RegisterValue::Dimension(
299            crate::common::dimension::Dimension::new(pt_scaled as i64),
300          ))
301        },
302        _ => val.into(),
303      }
304    }
305    Ok(match self.get_property("width") {
306      Some(val) => coerce_mu(&val),
307      None => match self.get_property("cached_width") {
308        Some(val) => coerce_mu(&val),
309        None => Some(RegisterValue::Dimension(Dimension::default())),
310      },
311    })
312  }
313  /// sets a "height" property value, for sizing
314  fn set_height<T: Into<Stored>>(&mut self, width: T) { self.set_property("height", width); }
315  /// gets the "height" property value, if any.
316  /// Checks "height", then "cached_height", then computes from font if needed.
317  fn get_height(&self) -> Option<RegisterValue> {
318    match self.get_property("height") {
319      Some(val) => (&*val).into(),
320      None => match self.get_property("cached_height") {
321        Some(val) => (&*val).into(),
322        None => {
323          if let Ok((_, h, _)) = self.compute_size(HashMap::default()) {
324            Some(RegisterValue::Dimension(h))
325          } else {
326            Some(RegisterValue::Dimension(Dimension::default()))
327          }
328        },
329      },
330    }
331  }
332  /// sets a "depth" property value, for sizing
333  fn set_depth<T: Into<Stored>>(&mut self, width: T) { self.set_property("depth", width); }
334  /// gets the "depth" property value, if any.
335  /// Checks "depth", then "cached_depth", then computes from font if needed.
336  fn get_depth(&self) -> Option<RegisterValue> {
337    match self.get_property("depth") {
338      Some(val) => (&*val).into(),
339      None => match self.get_property("cached_depth") {
340        Some(val) => (&*val).into(),
341        None => {
342          if let Ok((_, _, d)) = self.compute_size(HashMap::default()) {
343            Some(RegisterValue::Dimension(d))
344          } else {
345            Some(RegisterValue::Dimension(Dimension::default()))
346          }
347        },
348      },
349    }
350  }
351  /// gets the box size as a triple of (width, height, depth)
352  /// the generic implementation is immutable and will recompute the size on each call
353  /// see `Digested::get_size` for a variant with interior mutability which caches the box size
354  fn get_size(
355    &mut self,
356    options: Option<HashMap<Stored>>,
357  ) -> Result<(
358    Dimension,
359    Dimension,
360    Dimension,
361    Dimension,
362    Dimension,
363    Dimension,
364  )> {
365    // TODO: Reintroduce caching?
366    if !(self.has_property("cached_width")
367      && self.has_property("cached_height")
368      && self.has_property("cached_depth"))
369    {
370      self.compute_size_and_cache(options.unwrap_or_default())?;
371    }
372    self.with_properties(|props| {
373      let (width, height, depth, cached_width, cached_height, cached_depth) = (
374        props.get("width"),
375        props.get("height"),
376        props.get("depth"),
377        props.get("cached_width"),
378        props.get("cached_height"),
379        props.get("cached_depth"),
380      );
381
382      // eprintln!("SIZE of {} {}", std::any::type_name::<Self>(), self.get_string()?);
383      //     . "\n preassigned: " . _showsize($$props{width},  $$props{height},  $$props{depth})
384      //     . "\n calculated : " . _showsize($$props{cached_width}, $$props{cached_height},
385      // $$props{cached_depth})     . "\n w/options " . join(',', map { $_ . "=" .
386      // ToString($options{$_}); } sort keys %options)     . "\n =>: " .
387      // _showsize($$props{width} || $$props{cached_width}, $$props{height}
388      // || $$props{cached_height}, $$props{depth} || $$props{cached_depth})     . "\n   Of " .
389      // ToString($self)) if $LaTeXML::DEBUG{size};
390      // Helper: extract a Dimension from a Stored value.
391      // Handles Dimension directly, plus Glue/MuGlue/MuDimension by extracting the base value.
392      // MuGlue/MuDimension values are in scaled mu (1mu = font_size/18);
393      // convert to scaled pt using the current font size.
394      fn stored_to_dim(s: Option<&Stored>) -> Option<Dimension> {
395        match s {
396          Some(Stored::Dimension(d)) => Some(*d),
397          Some(Stored::Glue(g)) => Some(Dimension::new(g.value_of())),
398          Some(Stored::MuGlue(g)) => {
399            // Convert mu to pt: 1mu = font_size / 18
400            let fs = crate::state::lookup_font()
401              .and_then(|f| f.get_size())
402              .unwrap_or(10.0);
403            let muwidth = (fs * crate::common::numeric_ops::UNITY_F64 / 18.0) as i64;
404            let pt_scaled = (g.value_of() as f64 * muwidth as f64
405              / crate::common::numeric_ops::UNITY_F64)
406              .trunc();
407            Some(Dimension::new(pt_scaled as i64))
408          },
409          Some(Stored::MuDimension(d)) => {
410            let fs = crate::state::lookup_font()
411              .and_then(|f| f.get_size())
412              .unwrap_or(10.0);
413            let mu_val = d.value_of() as f64;
414            let pt_scaled = mu_val * fs / 18.0;
415            Some(Dimension::new(pt_scaled as i64))
416          },
417          _ => None,
418        }
419      }
420      Ok((
421        stored_to_dim(width).unwrap_or_else(|| stored_to_dim(cached_width).unwrap_or_default()),
422        stored_to_dim(height).unwrap_or_else(|| stored_to_dim(cached_height).unwrap_or_default()),
423        stored_to_dim(depth).unwrap_or_else(|| stored_to_dim(cached_depth).unwrap_or_default()),
424        stored_to_dim(cached_width).unwrap_or_else(|| stored_to_dim(width).unwrap_or_default()),
425        stored_to_dim(cached_height).unwrap_or_else(|| stored_to_dim(height).unwrap_or_default()),
426        stored_to_dim(cached_depth).unwrap_or_else(|| stored_to_dim(depth).unwrap_or_default()),
427      ))
428    })
429  }
430
431  /// computes and caches (via named properties) the size of a box-like object
432  fn compute_size_and_cache(
433    &mut self,
434    mut options: HashMap<Stored>,
435  ) -> Result<(Dimension, Dimension, Dimension)> {
436    for key in ["width", "height", "depth", "vattach", "layout"] {
437      if let Some(v) = self.get_property(key) {
438        options.insert(key, v.into_owned());
439      }
440    }
441
442    let (w, h, d) = self.compute_size(options)?;
443
444    if !self.has_property("cached_width") {
445      self.set_property("cached_width", w);
446    }
447    if !self.has_property("cached_height") {
448      self.set_property("cached_height", h);
449    }
450    if !self.has_property("cached_depth") {
451      self.set_property("cached_depth", d);
452    }
453    Ok((w, h, d))
454  }
455
456  /// computes and returns the size of a box-like object
457  fn compute_size(&self, options: HashMap<Stored>) -> Result<(Dimension, Dimension, Dimension)>;
458}
