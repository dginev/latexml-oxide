25/// Represents a digested object that can generate arbitrary elements in the XML Document.
26#[derive(Clone)]
27pub struct Whatsit {
28  /// arguments
29  pub args:           Vec<Option<Digested>>,
30  /// additional properties, such as font information or sizing
31  pub properties:     HashMap<Stored>,
32  /// the definition responsible for creating this object
33  pub definition:     Rc<dyn Definition>,
34  /// cached tokens for reverting back
35  ///  (note that the "reversion" _property_ is currently also used)
36  pub reversion:      Option<Tokens>,
37  /// special-case reversion tokens for whatsits representing Dual math structures
38  pub dual_reversion: Option<HashMap<Tokens>>,
39  /// point of origin in the source file (`None` = not recorded; set under
40  /// `--source-map` at constructor digest, Perl `Constructor.pm` L106)
41  pub locator:        Option<Locator>,
42}
43
44impl Default for Whatsit {
45  fn default() -> Self {
46    Whatsit {
47      args:           Vec::new(),
48      properties:     HashMap::default(),
49      definition:     Rc::new(Expandable::default()),
50      reversion:      None,
51      dual_reversion: None,
52      locator:        None,
53    }
54  }
55}
56impl PartialEq for Whatsit {
57  fn eq(&self, other: &Whatsit) -> bool {
58    // identical definition, argument list and body
59    *self.definition == *other.definition
60      && self.args == other.args
61      && if let Some(Stored::Digested(body1)) = self.properties.get("body") {
62        if let Some(Stored::Digested(body2)) = other.properties.get("body") {
63          *body1 == *body2
64        } else {
65          false
66        }
67      } else {
68        !other.properties.contains_key("body")
69      }
70  }
71}

83  /// A Whatsit is empty if it is marked empty, or space-like, or has an empty body.
84  pub fn is_empty(&self) -> Result<bool> {
85    Ok(
86      // 1. A space-like thing
87      // 2. An environment-like structure with an empty body
88      // TODO: For now it is difficult to pass in a state with an initialized TeX.pool.
89      self.get_property_bool("isEmpty")
90        || self.get_property_bool("isSpace")
91        || (self.get_definition().get_cs_name() == "Begin"
92          && match self.get_body()? {
93            Some(b) => b
94              .unlist_ref()
95              .iter()
96              .all(|inner| inner.is_empty().unwrap_or(false)),
97            None => true,
98          }),
99    )
100  }
101  /// sets a pre-assembled HashMap of properties
102  pub fn set_properties(&mut self, props: HashMap<Stored>) {
103    for (key, value) in props {
104      self.properties.insert_sym(key, value);
105    }

107  /// accessor for the definition which built this Whatsit
108  pub fn get_definition(&self) -> Rc<dyn Definition> { Rc::clone(&self.definition) }
109  /// accessor for the argument at index `n` (starting from 1)
110  /// Access argument at 1-based index `n` (matching Perl's `$whatsit->getArg(n)`).
111  /// Returns None for n == 0 (defensive — Perl convention uses 1-based indexing).
112  pub fn get_arg(&self, n: usize) -> Option<&Digested> {
113    if n == 0 {
114      log::warn!("get_arg(0) called — Perl convention uses 1-based indexing");
115      return None;
116    }
117    match self.args.get(n - 1) {
118      Some(Some(opt)) => Some(opt),
119      _ => None,
120    }
121  }
122  /// Mutably borrow argument at 1-based index `n` (matching Perl's `$whatsit->getArg(n)`).
123  /// Panics if n == 0 — use 1-based indexing.
124  pub fn get_arg_mut(&mut self, n: usize) -> Option<&mut Digested> {
125    assert!(
126      n > 0,
127      "get_arg_mut() uses 1-based indexing (Perl convention). Use get_arg_mut(1) for the first argument."
128    );
129    match self.args.get_mut(n - 1) {
130      Some(Some(opt)) => Some(opt),
131      _ => None,
132    }
133  }

141      Some(Stored::Digested(trailer)) => Some(trailer.clone()),
142      _ => None,

145  /// Sets the body of the `whatsit` to the boxes in `body`.
146  /// The last box in `body` is assumed to represent the `trailer`, that is the result of the
147  /// invocation that closed the environment or math.  It is stored separately in the properties
148  /// under "trailer".
149  pub fn set_body(&mut self, mut body: Vec<Digested>) {
150    let trailer_opt = body.pop();
151    // Perl: get mode from whatsit's own properties (not just isMath binary)
152    let mode_opt: Option<String> = self.get_property("mode").and_then(|p| match &*p {
153      Stored::String(s) => Some(arena::to_string(*s)),
154      _ => None,
155    });
156    let mut list = List::new(body);
157    // Set mode from whatsit's own mode property (Perl: $mode from $$self{properties}{mode})
158    if let Some(ref mode_str) = mode_opt {
159      list.set_property("mode", Stored::String(arena::pin(mode_str)));
160      if mode_str.contains("math") {
161        list.mode = Some(TexMode::Math);
162      }
163    } else if self.is_math() {
164      list.mode = Some(TexMode::Math);
165    }
166    self.properties.insert("body", Digested::from(list).into());
167    if let Some(digested) = trailer_opt {
168      self.properties.insert("trailer", digested.clone().into());
169      // And copy any otherwise undefined properties from the trailer
170      // Perl: copies properties from trailer (typically a Whatsit for \end{...})
171      match digested.data() {
172        DigestedData::Whatsit(trailer) => {
173          let trailer_val = trailer.borrow();
174          let props = trailer_val.get_properties();
175          for (prop, value) in props {
176            self
177              .properties
178              .entry_sym(*prop)
179              .or_insert_with(|| value.clone());
180          }
181        },
182        DigestedData::TBox(tbox) => {
183          let tbox_val = tbox.borrow();
184          let props = tbox_val.get_properties();
185          for (prop, value) in props {
186            self
187              .properties
188              .entry_sym(*prop)
189              .or_insert_with(|| value.clone());
190          }
191        },
192        DigestedData::List(list) => {
193          let list_val = list.borrow();
194          let props = list_val.get_properties();
195          for (prop, value) in props {
196            self
197              .properties
198              .entry_sym(*prop)
199              .or_insert_with(|| value.clone());
200          }
201        },
202        _ => {},
203      }
204      // TODO: Perl line 84 — create locator range from self to trailer
205      // $$self{properties}{locator} = Locator->newRange($self->getLocator, $trailer->getLocator);
206    }
207  }

209  /// Like Tokens-substituteParameters, but substitutes in the Whatsit's arguments OR properties!
210  /// #<digit> is the standard TeX positional argument
211  /// # followed by a T_OTHER(propname) specifies the property propname!!
212  fn substitute_parameters(&self, spec: Tokens) -> Result<Vec<Token>> {
213    // TODO: This is kind of unfortunate -- I am not sure what are the reasonable "entryways" into
214    // the Whatsit substituteParameters. For Expandable we now have guarantees that "#,i" has
215    // been mapped into a single T_ARG(#i), but not here. so for now run on each call?
216    let mut in_toks = VecDeque::from(spec.unlist());
217    let args = self.get_args();
218    let props = &self.properties;
219    // Pre-size: `result` is at least as long as the template; args
220    // substitute 1:1 or 1:N. Modest over-allocation beats repeated
221    // doublings on reversion of large whatsits.
222    let mut result = Vec::with_capacity(in_toks.len());
223    while let Some(token) = in_toks.pop_front() {
224      if token.get_catcode() != Catcode::ARG {
225        // Non '#'; copy it
226        result.push(token);
227      } else {
228        let arg_opt = token.with_str(|s| {
229          let n = s.parse::<usize>().unwrap() - 1;
230          if n < args.len() {
231            args[n].clone()
232          } else if n < 10 {
233            // `#N` where N ≤ 10 but fewer args were passed.
234            // Perl returns undef; we return None so the arg is simply omitted
235            // from the reversion stream. Fixes out-of-bounds panic when a
236            // reversion template references more params than the call-site
237            // supplied (sandbox paper 0803.4485).
238            None
239          } else {
240            match props.get(s) {
241              Some(Stored::Digested(v)) => Some((*v).clone()),
242              Some(other) => {
243                panic!("unexpected prop in substitute_parameters, needed Digested, got: {other:?}")
244              },
245              None => None,
246            }
247          }
248        });
249        if let Some(arg) = arg_opt {
250          result.extend(arg.revert()?.unlist());
251        }
252      }
253    }
254    Ok(result)
255  }
256}

258impl fmt::Debug for Whatsit {
259  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
260    write!(f, "Whatsit[")?;
261    let mut pieces = Vec::new();
262    pieces.push(
263      self
264        .get_definition()
265        .get_cs()
266        .with_cs_name(ToString::to_string),
267    );
268    for arg_opt in self.get_args() {
269      if let Some(arg) = arg_opt {
270        pieces.push(arg.stringify());
271      } else {
272        pieces.push(String::new());
273      }
274    }
275    if self.properties.contains_key("body") {
276      pieces.push(self.properties.get("body").unwrap().to_string());
277      if let Some(trailer) = self.properties.get("trailer") {
278        pieces.push(trailer.to_string());
279      }
280    }
281    write!(f, "{}]", pieces.join(","))
282  }
283}

294  fn revert(&self) -> Result<Tokens> {
295    // WARNING: Forbidden knowledge?
296    // (2) caching the reversion (which is a big performance boost)
297    let saved_opt = if let Some(this_branch) = get_dual_branch() {
298      if let Some(ref dual_reversion) = self.dual_reversion {
299        dual_reversion.get(this_branch).cloned()
300      } else {
301        self.reversion.clone()
302      }
303    } else {
304      self.reversion.clone()
305    };
306    if let Some(saved) = saved_opt {
307      return Ok(saved);
308    }
309
310    let mut tokens = Vec::new();
311    let defn = &self.definition;
312    if defn.get_reversion_spec().is_none()
313      && let Some(Stored::Digested(digested)) = self.properties.get("alignment")
314      && let DigestedData::Alignment(alignment) = digested.data()
315    {
316      return alignment.borrow().revert();
317    }
318    // Find the appropriate reversion spec;
319    // content_reversion or presntation_reversion if on dual branch
320    // or (general) reversion, or the reversion from the definition
321    let spec_opt = if let Some(rev) = self.properties.get("reversion") {
322      match rev {
323        Stored::Tokens(tks) => Some(Cow::Owned(Reversion::Tokens(tks.clone()))),
324        Stored::Reversion(rev) => Some(Cow::Borrowed(rev)),
325        other => panic!("TODO: Unexpected reversion directive {other:?}"),
326      }
327    } else {
328      defn.get_reversion_spec().map(Cow::Owned)
329    };
330    let mut is_closure = false;
331    match spec_opt.as_deref() {
332      Some(Reversion::Closure(spec)) => {
333        is_closure = true;
334        let spec_tokens = spec(self, self.get_args()).unwrap();
335        tokens = self.substitute_parameters(spec_tokens)?;
336      },
337      Some(Reversion::Tokens(spec)) => {
338        if !spec.is_empty() {
339          tokens = self.substitute_parameters(spec.clone())?;
340        }
341      },
342      None => {
343        if let Some(alias) = defn.get_alias() {
344          if !alias.is_empty() {
345            // Use From<&str> which maps single characters to their proper catcodes
346            // (e.g. "$" -> T_MATH!(), "{" -> T_BEGIN!(), etc.)
347            // This matches Perl's coerceCS which calls TokenizeInternal for single chars.
348            tokens.push(Token::from(alias.as_str()));
349          }
350        } else {
351          tokens.push(defn.get_cs().into_owned());
352        }
353        if let Some(parameters) = defn.get_parameters() {
354          // Use revert_digested_arguments which checks for digested_reversion
355          // closures on each parameter, allowing parameter types like BoxSpecification
356          // to format their reversion from the structured digested data.
357          // Perl: push(@tokens, $parameters->revertArguments($self->getArgs));
358          tokens.extend(parameters.revert_digested_arguments(self.get_args())?)
359        }
360      },
361    };
362
363    if !is_closure && let Some(body) = self.get_body()? {
364      tokens.extend(body.revert()?.unlist());
365      if let Some(trailer) = self.get_trailer() {
366        tokens.extend(trailer.revert()?.unlist());
367      }
368    }
369
370    // Now cache it, in case it's needed again
371    // TODO: DG: We can't yet cache reversions, because we lack mutability on .revert()
372    //       should we reorganize? is it worth it?
373    //
374    // if let Some(this_branch) = state!().get_dual_branch() {
375    //   if self.dual_reversion.is_none() {
376    //     self.dual_reversion = Some(HashMap::default());
377    //   }
378    //   self.dual_reversion.as_mut().unwrap()
379    //     .insert(this_branch.to_string(), Tokens::new(tokens.clone()));
380    // } else {
381    //   self.reversion = Some(Tokens::new(tokens.clone()));
382    // }
383    Ok(Tokens::new(tokens))
384  }
385}

432  fn compute_size(
433    &self,
434    mut options: HashMap<Stored>,
435  ) -> Result<(Dimension, Dimension, Dimension)> {
436    let defn = self.get_definition();
437    match defn.get_sizer() {
438      Some(sizer) => sizer(self),
439      _ => {
440        if self.has_property("cached_width") || self.has_property("cached_height") {
441          // Perl: when after_digest sets cached dimensions (e.g. image_graphicx_sizer),
442          // compute_size should return them instead of falling through to body/args sum.
443          let w = match self.get_property("cached_width").as_deref() {
444            Some(Stored::Dimension(d)) => *d,
445            _ => Dimension::default(),
446          };
447          let h = match self.get_property("cached_height").as_deref() {
448            Some(Stored::Dimension(d)) => *d,
449            _ => Dimension::default(),
450          };
451          let d = match self.get_property("cached_depth").as_deref() {
452            Some(Stored::Dimension(d)) => *d,
453            _ => Dimension::default(),
454          };
455          Ok((w, h, d))
456        } else {
457          // Nothing specified? use #body if any, else sum all box args
458          // Perl: Whatsit.pm L252-255 — if body exists, pass it to computeBoxesSize
459          // which unlists it internally (Font.pm L650-651). We replicate by extracting
460          // properties from the body (mode, vattach, width) into options, then unlisting.
461          let mut boxes = Vec::new();
462          if let Some(body_stored) = self.get_property("body")
463            && let Stored::Digested(ref body) = *body_stored
464          {
465            // Perl: computeBoxesSize reads mode/vattach/width from $boxes before unlisting
466            for key in &["mode", "vattach", "width"] {
467              if options.get(key).is_none()
468                && let Some(prop) = body.get_property(key)
469              {
470                options.insert(key, (*prop).clone());
471              }
472            }
473            let unlist_boxes = body.unlist();
474            boxes.extend(unlist_boxes);
475          }
476          if boxes.is_empty() {
477            // no body
478            for arg in self.args.iter().flatten() {
479              boxes.extend(arg.unlist());
480            }
481          }
482          let font = match *self.get_property("font").unwrap() {
483            Stored::Font(ref sf) => sf.clone(),
484            _ => lookup_font().unwrap(),
485          };
486          font.compute_boxes_size(&boxes, options)
487        }
488      },
489    }
490  }
