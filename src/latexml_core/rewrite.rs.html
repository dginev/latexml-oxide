13// Defining Rewrite rules that act on the DOM

15#[derive(Clone, Default)]
16pub struct RewriteOptions {
17  pub label:        Option<String>,
18  pub scope:        Option<Scope>,
19  pub xpath:        Option<String>,
20  pub on_match:     Option<Tokens>,
21  pub attributes:   Option<String>,
22  pub replace:      Option<RewriteReplaceClosure>,
23  pub regexp:       Option<String>,
24  pub select:       Option<String>,
25  pub select_count: Option<usize>,
26}
27impl fmt::Debug for RewriteOptions {
28  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result { write!(f, "<RewriteOptions>") }
29}
30impl PartialEq for RewriteOptions {
31  fn eq(&self, other: &RewriteOptions) -> bool { self.select == other.select }
32}
33#[derive(Debug, Clone, Copy, PartialEq, Eq)]
34pub enum RewriteOperator {
35  // only uncompiled:
36  Label,
37  Scope,
38  Xpath,
39  Match,
40  // with available actions:
41  Regexp,
42  Attributes,
43  Action,
44  Replace,
45  Test,
46  MultiSelect,
47  Select,
48  Ignore,
49  Trace,
50}
51#[derive(Clone)]
52pub enum RewritePattern {
53  String(String),
54  Scope(Scope),
55  Tokens(Tokens),
56  Closure(RewriteReplaceClosure),
57}
58impl fmt::Debug for RewritePattern {
59  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
60    match self {
61      RewritePattern::String(x) => write!(f, "{x:?}"),
62      RewritePattern::Scope(x) => write!(f, "{x:?}"),
63      RewritePattern::Tokens(x) => write!(f, "{x:?}"),
64      RewritePattern::Closure(_) => write!(f, "<Rewrite Replacement Closure>"),
65    }
66  }
67}
68#[derive(Debug, Clone)]
69pub struct RewriteClause {
70  compiled: bool,
71  op:       RewriteOperator,
72  pattern:  RewritePattern,
73}

79}
80impl PartialEq for Rewrite {

84impl Rewrite {
85  pub fn new(_kind: &str, mut options: RewriteOptions) -> Self {
86    use RewriteOperator::*;
87    let mut clauses = Vec::new();
88    // collect the non-compiling, early phase clauses from the options
89    if let Some(xpath) = options.select.take() {
90      clauses.push(RewriteClause {
91        compiled: true,
92        op:       Select,
93        pattern:  RewritePattern::String(xpath),
94      })
95    }
96    // collect the actionable clauses from the options
97    if let Some(label) = options.label.take() {
98      clauses.push(RewriteClause {
99        compiled: false,
100        op:       Label,
101        pattern:  RewritePattern::String(label),
102      });
103    }
104    if let Some(scope) = options.scope.take() {
105      clauses.push(RewriteClause {
106        compiled: false,
107        op:       Scope,
108        pattern:  RewritePattern::Scope(scope),
109      });
110    }
111    if let Some(xpath) = options.xpath.take() {
112      clauses.push(RewriteClause {
113        compiled: false,
114        op:       Xpath,
115        pattern:  RewritePattern::String(xpath),
116      });
117    }
118    if let Some(tokens) = options.on_match.take() {
119      clauses.push(RewriteClause {
120        compiled: false,
121        op:       Match,
122        pattern:  RewritePattern::Tokens(tokens),
123      });
124    }
125    if let Some(replace) = options.replace.take() {
126      clauses.push(RewriteClause {
127        compiled: false,
128        op:       Replace,
129        pattern:  RewritePattern::Closure(replace),
130      });
131    }
132    if let Some(r) = options.regexp.take() {
133      clauses.push(RewriteClause {
134        compiled: false,
135        op:       Regexp,
136        pattern:  RewritePattern::String(r),
137      });
138    }
139    Rewrite { options, clauses }
140  }

142  pub fn compile_clauses(&mut self, document: &mut Document) {
143    let current_clauses: Vec<RewriteClause> = self.clauses.drain(..).collect();
144    let mut new_clauses: Vec<RewriteClause> = Vec::new();
145    for clause in current_clauses {
146      if !clause.compiled {
147        new_clauses.push(self.compile_clause(document, clause));
148      } else {
149        new_clauses.push(clause);
150      }
151    }
152    self.clauses = new_clauses;
153  }

215  pub fn invoke(&mut self, document: &mut Document, root: &Node) -> Result<()> {
216    // Debug(('=' x 40)) if $LaTeXML::DEBUG{rewrite};
217    // What goes into self.clauses ???
218    let clauses = self.clauses.iter().collect();
219    self.apply_clause(document, root, 0, clauses)?;
220    Ok(())
221  }
222  // Rewrite spec as input
223  //   scope  => $scope  : a scope like "section:1.2.3" or "label:eq.one"; translated to xpath
224  //   select => $xpath  : selects subtrees based on xpath expression.
225  //   match  => $code   : called on $document and current $node: tests current node, returns
226  // $nnodes, if match   match  => $string : Treats as TeX, converts Box, then DOM tree, to xpath
227  //                      (The matching top-level nodes will be replaced, if replace is the next
228  // op.)   replace=> $code   : removes the current $nnodes, calls $code with $document and
229  // removed nodes   replace=> $string : removes $nnodes
230  //                       Treats $string as TeX, converts to Box and inserts to replace
231  //                       the removed nodes.
232  //   attributes=>$hash : adds data from hash as attributes to the current node.
233  //   regexp  => $string: apply regexp (subst) to all text nodes in/under the current node.
234
235  // Compiled rewrite spec:
236  //   select => $xpath  : operate on nodes selected by $xpath.
237  //   test   => $code   : Calls $code on $document and current $node.
238  //                       Returns number of nodes matched.
239  //   replace=> $code   : removes the current $nnodes, calls $code on them.
240  //   action => $code   : invoke $code on current $node, without removing them.
241  //   regexp  => $string: apply regexp (subst) to all text nodes in/under the current node.
242
243  fn apply_clause(
244    &self,
245    document: &mut Document,
246    tree: &Node,
247    nmatched: usize,
248    mut clauses: VecDeque<&RewriteClause>,
249  ) -> Result<()> {
250    use RewriteOperator::*;
251    if let Some(RewriteClause { compiled: _, op, pattern }) = clauses.pop_front() {
252      match op {
253        Select => {
254          // my ($xpath, $nnodes, @wilds) = @$pattern;
255          if let RewritePattern::String(xpath) = pattern {
256            let matches = document.findnodes(xpath, Some(tree));
257            // Debug("Rewrite selecting \"$xpath\" => " . scalar(@matches) . " matches") if
258            // $LaTeXML::DEBUG{rewrite};
259            for node in matches {
260              // next unless node.get_owner_document()->isSameNode($tree->ownerDocument); # If still
261              // attached to original document!
262              if node.has_attribute("_matched") {
263                continue;
264              }
265              // let w = mark_wildcards(node, wilds);
266              self.apply_clause(
267                document,
268                &node,
269                self.options.select_count.unwrap_or(1),
270                clauses.clone(),
271              )?;
272              // unmark_wildcards(node, w);
273            }
274          } else {
275            // unsupported rewrite pattern? should never happen.
276            todo!()
277          }
278        },
279        Replace => {
280          //     Debug("Rewrite replace at " . $tree->toString . " using $pattern")
281          // if $LaTeXML::DEBUG{rewrite};
282          let mut parent = tree.get_parent().unwrap();
283          // Remove & separate nodes to be replaced, and sibling nodes following them.
284          let mut following = VecDeque::new(); // Collect the matching and following nodes
285          while let Some(mut sib) = parent.get_last_child() {
286            sib.unbind_node();
287            if *tree == sib {
288              following.push_front(sib);
289              break;
290            } else {
291              following.push_front(sib);
292            }
293          }
294          let mut replaced = Vec::new();
295          for _idx in 0..nmatched {
296            // Remove the nodes to be replaced
297            if let Some(popped) = following.pop_front() {
298              replaced.push(popped);
299            } else {
300              todo!(); // should we report an error if nmatched was too large?
301            }
302          }
303          for rnode in replaced.iter() {
304            document.unrecord_node_ids(rnode);
305          }
306          // Carry out the operation, inserting whatever nodes.
307          document.set_node(&parent);
308          let point_opt = parent.get_last_child();
309          if let RewritePattern::Closure(closure) = pattern {
310            closure(document, replaced.iter_mut().collect())?; // Carry out the insertion.
311          }
312
313          // Now collect the newly inserted nodes for any needed patching
314          let inserted = if let Some(point) = point_opt {
315            let mut ins_queue = VecDeque::new();
316            let mut sibs = parent.get_child_nodes();
317            while let Some(sib) = sibs.pop() {
318              if sib == point {
319                break;
320              }
321              ins_queue.push_front(sib);
322            }
323            ins_queue.into_iter().collect::<Vec<Node>>()
324          } else {
325            parent.get_child_nodes()
326          };
327
328          // Now make any adjustments to the new nodes
329          for ins in inserted.iter() {
330            document.record_node_ids(ins)?;
331          }
332          // TODO: Can we avoid this clone?
333          let font = document.get_node_font(tree).clone();
334          // the font of the matched node
335          for ins in inserted.iter() {
336            // Copy the non-semantic parts of font to the replacement
337            document.merge_node_font_rec(ins, &font)?;
338          }
339          // Now, replace the following nodes.
340          for mut follow_node in following {
341            parent.add_child(&mut follow_node)?;
342          }
343        },
344        _ => todo!(),
345      }
346    }
347
348    Ok(())
349  }
