1//! Reader for Rust-native kernel dump files (produced by dump_writer.rs).
2//!
3//! Loads a dump file produced by `latexml_oxide --init=latex.ltx --dest=dump`
4//! and replays the state assignments into the engine.
5//!
6//! **Loading policy:** `M` and `V` entries replay with Perl-style global
7//! assignment semantics, matching `Core/Dumper.pm`'s `I()` / `V()` helpers.
8//! Runtime-state filters below are narrow exceptions for entries that should
9//! never be useful from a format dump.
10//!
11//! Format: tab-separated lines:
12//!   V\tKEY\tTYPE\tDATA             — value assignment
13//!   M\tKEY\tE\tCS\tNARGS\tFLAGS\tTOKENS — Expandable definition
14//!   M\tKEY\tN                       — None meaning (undefined)
15//!   M\tKEY\tT\tCC:TEXT              — Token meaning (let-assignment)
16//!   C\tCHAR\tCC\tVALUE             — catcode assignment
17//!   LC\tCHAR\tCH\tVALUE            — lccode assignment
18//!   UC\tCHAR\tCH\tVALUE            — uccode assignment
19//!   SC\tCHAR\tCH\tVALUE            — sfcode assignment
20//!   DC\tCHAR\tCH\tVALUE            — delcode assignment
21//!   MC\tCHAR\tCH\tVALUE            — mathcode assignment
22

33/// Load a Rust-native dump file into the current State.
34/// Returns the number of entries loaded.
35pub fn load_native_dump(path: &Path) -> Result<usize, String> {
36  let content = std::fs::read_to_string(path)
37    .map_err(|e| format!("Failed to read dump file {}: {}", path.display(), e))?;
38  let count = load_from_str_internal(&content, &path.display().to_string())?;
39  Ok(count)
40}

42/// Load dump data from a string (used by embedded dump modules).

44pub fn load_from_str(content: &str) -> Result<usize, String> {
45  load_from_str_internal(content, "<embedded>")

56// Per-load context used to attach a nominal Locator to dump-installed
57// Expandables. Matches Perl #aaacdba2 (2026): dump-loaded definitions
58// should be traceable to the dump file + line, not report the arena's
59// internal location. Thread-local so concurrent loads (there are none
60// today, but the state is cooperative) don't clobber each other.
61thread_local! {
62  static CURRENT_LOAD_CTX: std::cell::Cell<Option<(crate::common::arena::SymStr, u32)>> =
63    const { std::cell::Cell::new(None) };
64  /// PA/MPA aliases whose target wasn't defined at dump-load time.
65  /// Populated by `load_meaning`'s PA arm, drained by
66  /// `flush_deferred_aliases()` after `_constructs` finishes.
67  static DEFERRED_ALIASES: std::cell::RefCell<Vec<(Token, Token)>> =
68    const { std::cell::RefCell::new(Vec::new()) };
69}
70
71/// Replay any PA/MPA aliases that were deferred during dump load
72/// because their target was not yet defined. Call once after the
73/// post-dump definition pass (`_constructs`) has loaded.
74/// Returns `(applied, skipped)`.
75pub fn flush_deferred_aliases() -> (usize, usize) {
76  let pending: Vec<(Token, Token)> =
77    DEFERRED_ALIASES.with(|cell| std::mem::take(&mut *cell.borrow_mut()));
78  let mut applied = 0usize;
79  let mut skipped = 0usize;
80  for (cs_tok, target_tok) in pending {
81    // Target still undefined — the alias's target must be defined
82    // in some source we never load (e.g. expl3 intarrays that the
83    // short-circuit skips). Leave the key undefined; the engine's
84    // undefined-CS handler will cope at runtime.
85    if !state::has_meaning(&target_tok) {
86      skipped += 1;
87      continue;
88    }
89    // Perl `Lt()` parity: look up target's meaning, write it at
90    // alias key via `assign_internal('meaning', ..., 'global')`.
91    if let Some(meaning) = state::lookup_meaning(&target_tok) {
92      state::assign_internal(
93        TableName::Meaning,
94        cs_tok.get_cs_name(),
95        meaning,
96        Some(Scope::Global),
97      );
98      applied += 1;
99    } else {
100      skipped += 1;
101    }
102  }
103  (applied, skipped)
104}

106fn current_dump_locator() -> crate::common::locator::Locator {
107  if let Some((source, lineno)) = CURRENT_LOAD_CTX.with(|c| c.get()) {
108    crate::common::locator::Locator {
109      source,
110      from_line: lineno,
111      to_line: lineno,
112      from_column: 1,
113      to_column: 1,
114    }
115  } else {
116    crate::common::locator::Locator::default()
117  }
118}

120fn load_from_str_internal(content: &str, source_name: &str) -> Result<usize, String> {
121  let mut count = 0;
122  let mut skipped = 0;
123  let mut errors = 0;
124  let source_sym = crate::common::arena::pin(source_name);
125
126  for (lineno, line) in content.lines().enumerate() {
127    // Trim only CR (from CRLF line endings); `lines()` already strips LF.
128    // Do NOT use `trim()` here — it strips trailing tabs, which are part of
129    // the tab-separated format for entries with empty trailing fields (e.g.
130    // E-entries with empty body: `E\t<cs>\t<nargs>\t<flags>\t`).
131    let line = line.trim_end_matches('\r');
132    if line.is_empty() || line.starts_with('#') {
133      continue;
134    }
135
136    CURRENT_LOAD_CTX.with(|c| c.set(Some((source_sym, (lineno + 1) as u32))));
137
138    match parse_and_load(line) {
139      Ok(true) => count += 1,
140      Ok(false) => skipped += 1,
141      Err(e) => {
142        errors += 1;
143        if errors <= 10 {
144          log::warn!(
145            "[dump_reader] Line {}: {}: {}",
146            lineno + 1,
147            e,
148            &line[..line.len().min(80)]
149          );
150        }
151      },
152    }
153  }
154
155  if errors > 10 {
156    log::warn!("[dump_reader] ... and {} more errors", errors - 10);
157  }
158  log::info!(
159    "[dump_reader] Loaded {} entries from {} ({} skipped, {} errors)",
160    count,
161    source_name,
162    skipped,
163    errors
164  );
165
166  Ok(count)
167}

169/// Parse a single dump line and load it. Returns Ok(true) if loaded,
170/// Ok(false) if filtered (e.g. corrupt MC/DC), Err on parse error.
171fn parse_and_load(line: &str) -> Result<bool, String> {
172  let parts: Vec<&str> = line.splitn(3, '\t').collect();
173  if parts.len() < 2 {
174    return Err("Too few fields".into());
175  }
176
177  let table = parts[0];
178  // Key decode: Cow borrows the original &str when no `%` escape is
179  // present (the overwhelming majority). Saves a per-line allocation
180  // for the ~25k dump entries that have plain CS-name keys.
181  let key_cow: std::borrow::Cow<'_, str> = if parts[1].contains('%') {
182    std::borrow::Cow::Owned(url_decode(parts[1]))
183  } else {
184    std::borrow::Cow::Borrowed(parts[1])
185  };
186  let key = key_cow.as_ref();
187  let data = if parts.len() > 2 { parts[2] } else { "" };
188
189  match table {
190    // V: Value entries (registers, fontdimen, font metadata).
191    // Add-only policy: only loads if key has no existing value.
192    //
193    // Skip MAX_ERRORS: it was set to 1_000_000 in `ini_tex.rs` during
194    // dump-build (to let raw latex.ltx run through transient errors)
195    // and got captured into the dump. Loading that into a regular
196    // conversion lets runaway error cascades (e.g. AmS-TeX `\cases`
197    // mis-parse → 1M `\hbox`/`&` errors per paper) bypass the 10000
198    // default cap. Filter at read time so existing dumps are clean.
199    "V" if key == "MAX_ERRORS" => Ok(false),
200    "V" => load_value(key, data),
201    // M: Meaning entries (Expandable, Let-alias, Register, etc.).
202    //
203    // Perl-faithful: `plain_dump.pool.ltxml` and `latex_dump.pool.ltxml`
204    // emit one `I(...)` per Meaning entry, which is
205    // `assign_internal($STATE, 'meaning', $cs, $def, 'global')` —
206    // unconditional global write. No admission gate, no skip-if-defined.
207    // Match it: route every M entry to `load_meaning` directly.
208    "M" => load_meaning(key, data),
209    // LC/UC: case-mapping codes — safe, always load
210    "LC" => load_lccode(key, data),
211    "UC" => load_uccode(key, data),
212    // SC: space factor codes — safe, always load
213    "SC" => load_sfcode(key, data),
214    // C: catcodes — only for non-ASCII (>127). ASCII catcodes are set by
215    // the engine; loading from dump would conflict.
216    "C" => {
217      let ch = decode_char_key(key);
218      if ch.is_some_and(|c| c as u32 > 127) {
219        load_catcode(key, data)
220      } else {
221        Ok(false)
222      }
223    },
224    // Perl `Core/Dumper.pm:dump_mathcode/dump_delcode` write MC/DC for
225    // every state-set entry; the matching reader is unconditional apply
226    // (CLAUDE.md "Unconditional dump apply"). plain.tex / latex.ltx need
227    // letter mathcodes and `\delcode\(="0028300` etc. replayed from dump
228    // so `\cal abc` (cmsy fam 2) and delimited symbols decode correctly
229    // — without this, `decode_math_char` never fires for letters in the
230    // dump path and they get default-decoded to ASCII (no meaning/role).
231    "MC" => load_mathcode(key, data),
232    "DC" => load_delcode(key, data),
233    _ => Ok(false),
234  }
235}

237/// V entries to unconditionally skip (runtime state, never useful from dump).
238const SKIP_VALUE_KEYS: &[&str] = &[
239  "INTERPRETING_DEFINITIONS",
240  "if_count",
241  "absorb_count",
242  "if_stack",
243  "INCLUDE_COMMENTS",
244  "INCLUDE_STYLES",
245  "INPUT_ENCODING",
246  "CURRENT_INPUT_ENCODING",
247  "SUPPRESS_UNEXPECTED_ERRORS",
248  "SUPPRESS_UNDEFINED_ERRORS",
249  // Upstream Perl IGNORED_SYMBOLS (TeX_Job.pool.ltxml): runtime-only
250  // tables that re-populate from the engine — can't meaningfully round-
251  // trip through the dump.
252  "DOCUMENT_REWRITE_RULES",
253  "PARAMETER_TYPES",
254  "TAG_PROPERTIES",
255  "MATH_LIGATURES",
256  "TEXT_LIGATURES",
257];

259/// V entry key prefixes to skip.
260const SKIP_VALUE_PREFIXES: &[&str] = &["input_file:", "output_file:", "texsys"];
261
262/// V entry key substrings to skip.
263///
264/// Note: `_loaded` / `_raw_loaded` flags are present in the dump (correctly,
265/// since `--init=latex.ltx` sees expl3-code.tex, hyphenation patterns, and
266/// hundreds of other raw-loaded files). But carrying them through into state
267/// at dump-load time blows up in practice:
268///
269///  - Hyphenation `loadhyph-*.tex_loaded` flags make subsequent raw-loading of babel's language.def
270///    skip files that set `\l@<lang>` registers our engine then discovers aren't present,
271///    triggering a flood of error recovery that can consume gigabytes of RAM.
272///  - `expl3.ltx_loaded=1` plus `expl3.sty_loaded=` NOT being set means `\usepackage{expl3}`
273///    doesn't short-circuit AT the .sty layer, but the raw .ltx re-load now enters a stranger code
274///    path with partial flags.
275///
276/// The proper fix, tracked as the "dump/_base mutual-exclusivity" item in
277/// SYNC_STATUS D0, is to have exactly ONE loading path (dump-cache or raw-load)
278/// active at a time, mirroring Perl's `LoadFormat` branching. Until that lands,
279/// keep the skip list conservative so mixed paths don't trigger recovery loops.
280const SKIP_VALUE_CONTAINS: &[&str] = &[
281  "_loaded", /* Package loading flags — see doc comment above.
282             * Substring also matches `_raw_loaded` (OXIDIZED_DESIGN #23). */
283];

285/// Load a value entry: V\tKEY\tTYPE\tDATA
286///
287/// Uses add-only policy: only loads if the key does not already have a value.
288/// This ensures compiled engine state takes priority over dump state.
289fn load_value(key: &str, data: &str) -> Result<bool, String> {
290  // Skip unconditional keys
291  for skip in SKIP_VALUE_KEYS {
292    if key == *skip {
293      return Ok(false);
294    }
295  }
296  // Skip by prefix
297  for prefix in SKIP_VALUE_PREFIXES {
298    if key.starts_with(prefix) {
299      return Ok(false);
300    }
301  }
302  // Skip by substring.
303  for substr in SKIP_VALUE_CONTAINS {
304    if key.contains(substr) {
305      return Ok(false);
306    }
307  }
308
309  // Perl `V()` parity (`Core/Dumper.pm` L59): every dumped Value entry
310  // maps to `assign_internal($STATE, 'value', $key, $val, 'global')` —
311  // unconditional global write. No skip-if-defined.
312
313  let parts: Vec<&str> = data.splitn(2, '\t').collect();
314  if parts.is_empty() {
315    return Err("Missing value type".into());
316  }
317
318  let value = match parts[0] {
319    "N" => return Ok(false), // Don't load None values (would erase existing)
320    "B" => Stored::Bool(parts.get(1).map(|s| *s == "1").unwrap_or(false)),
321    "I" => {
322      let n: i64 = parts
323        .get(1)
324        .unwrap_or(&"0")
325        .parse()
326        .map_err(|e| format!("Bad int: {}", e))?;
327      Stored::Int(n)
328    },
329    // "Nm": Stored::Number marker (distinct from "I" Stored::Int) —
330    // see dump_writer's Number serializer for rationale.
331    "Nm" => {
332      let n: i64 = parts
333        .get(1)
334        .unwrap_or(&"0")
335        .parse()
336        .map_err(|e| format!("Bad number: {}", e))?;
337      Stored::Number(crate::common::number::Number(n))
338    },
339    "S" => Stored::from(url_decode(parts.get(1).unwrap_or(&""))),
340    "F" => {
341      // Stored::Font — written by dump_writer's Stored::Font arm.
342      // Format: F\tname=...\x1fsize=...\x1ffamily=...\x1f...
343      // Each unit-separator-delimited segment is `key=urlencoded_value`.
344      // Mirrors Perl `dump_font` (Core/Dumper.pm L281-284).
345      use crate::common::font::Font;
346      use std::borrow::Cow;
347      let mut font = Font::default();
348      for kv in parts.get(1).unwrap_or(&"").split('\x1f') {
349        if let Some((k, v)) = kv.split_once('=') {
350          let v_dec = url_decode(v);
351          match k {
352            "name" => font.name = Some(Cow::Owned(v_dec)),
353            "family" => font.family = Some(Cow::Owned(v_dec)),
354            "series" => font.series = Some(Cow::Owned(v_dec)),
355            "shape" => font.shape = Some(Cow::Owned(v_dec)),
356            "encoding" => font.encoding = Some(Cow::Owned(v_dec)),
357            "language" => font.language = Some(Cow::Owned(v_dec)),
358            "mathstyle" => font.mathstyle = Some(Cow::Owned(v_dec)),
359            "opacity" => font.opacity = Some(Cow::Owned(v_dec)),
360            "size" => font.size = v_dec.parse().ok(),
361            "scale" => font.scale = v_dec.parse().ok(),
362            "emph" => font.emph = Some(v_dec == "1"),
363            "scripted" => font.scripted = Some(v_dec == "1"),
364            "mathstylestep" => font.mathstylestep = v_dec.parse().ok(),
365            "flags" => font.flags = v_dec.parse().ok(),
366            _ => {},
367          }
368        }
369      }
370      Stored::Font(std::rc::Rc::new(font))
371    },
372    "CH" => {
373      let n: u16 = parts
374        .get(1)
375        .unwrap_or(&"0")
376        .parse()
377        .map_err(|e| format!("Bad charcode: {}", e))?;
378      Stored::Charcode(n)
379    },
380    "CC" => {
381      let n: u8 = parts
382        .get(1)
383        .unwrap_or(&"0")
384        .parse()
385        .map_err(|e| format!("Bad catcode: {}", e))?;
386      Stored::Catcode(Catcode::from(n))
387    },
388    "T" => {
389      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
390      Stored::Token(tok)
391    },
392    "TK" => {
393      let toks = parse_token_list(parts.get(1).unwrap_or(&""))?;
394      Stored::Tokens(Tokens::from(toks))
395    },
396    "D" => {
397      let n: i64 = parts
398        .get(1)
399        .unwrap_or(&"0")
400        .parse()
401        .map_err(|e| format!("Bad dimension: {}", e))?;
402      Stored::Dimension(crate::common::dimension::Dimension(n))
403    },
404    "G" => Stored::Glue(parse_glue(parts.get(1).unwrap_or(&"0"))?),
405    "MD" => {
406      let n: i64 = parts
407        .get(1)
408        .unwrap_or(&"0")
409        .parse()
410        .map_err(|e| format!("Bad mudimension: {}", e))?;
411      Stored::MuDimension(crate::common::mudimension::MuDimension(n))
412    },
413    "MG" => Stored::MuGlue(parse_muglue(parts.get(1).unwrap_or(&"0"))?),
414    "VD" => return Ok(false), // Don't load empty VecDeque (runtime state)
415    _ => return Ok(false),    // Unknown value type
416  };
417
418  // Perl `V()` (`Core/Dumper.pm` L59):
419  //   sub V { State::assign_internal($STATE,'value',$_[0],$_[1],'global'); }
420  // Direct table mutation, no dialect.
421  state::assign_internal(
422    TableName::Value,
423    arena::pin(key),
424    value,
425    Some(Scope::Global),
426  );
427  Ok(true)
428}

430/// Load a meaning entry: M\tKEY\tTYPE\t...
431///
432/// Uses add-only policy: skip if the CS already has a meaning.
433/// Additionally, only loads "safe" definitions — those that won't interfere
434/// with the compiled engine's processing during normal LaTeX operation:
435/// - expl3 internals (contain `:`) — safe because `:` is OTHER under normal catcodes
436/// - Private LaTeX internals (contain `@`) — only invoked by other macros
437/// - Skip all "public" macros that could be invoked during normal expansion and might reference
438///   hooks/primitives not supported by our engine
439fn load_meaning(key: &str, data: &str) -> Result<bool, String> {
440  let cs_tok = Token {
441    text: arena::pin(key),
442    code: Catcode::CS,
443  };
444
445  // Perl `I(...)` parity (`Core/Dumper.pm` L67): every dumped Meaning
446  // entry maps to `assign_internal($STATE, 'meaning', $cs, $def,
447  // 'global')` — unconditional global write. No skip-if-defined, no
448  // admission filter.
449
450  let parts: Vec<&str> = data.splitn(2, '\t').collect();
451  if parts.is_empty() {
452    return Err("Missing meaning type".into());
453  }
454
455  match parts[0] {
456    "N" => {
457      // None meaning — skip (don't define as undefined)
458      Ok(false)
459    },
460    "E" => {
461      // Expandable: E\tCSNAME\tNARGS\tFLAGS\tTOKENS[\tPROTO[\tV3_PARAMS]]
462      //
463      // Three historical shapes, read in fallback order:
464      //   v3 — 6th field present: structured Parameter records; bypasses
465      //        parse_parameters entirely. Only format that round-trips
466      //        Until:/Match: with catcoded delimiter tokens intact.
467      //   v2 — 5th field present: url-decoded prototype string fed to
468      //        parse_parameters. Good for {} / [] / DefToken / simple
469      //        typed params; loses brace-in-delimiter forms.
470      //   v1 — nargs only: "{}".repeat(nargs), all params flattened to
471      //        Plain. Kept as last resort so ancient dumps still load.
472      let eparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(6, '\t').collect();
473      if eparts.len() < 4 {
474        return Err("Incomplete Expandable entry".into());
475      }
476
477      // eparts[0] is the alias-cs from the dump (Perl-side: the cs of
478      // the Definition object that this entry was let-aliased from).
479      //
480      // We propagate the alias ONLY when the target is a known deferred
481      // command (`\unexpanded`, `\the`, `\detokenize`, `\showthe`) — that
482      // narrow case is what makes `\exp_not:n {…}` inside `\edef` bodies
483      // correctly skip re-expansion (Perl `Gullet.pm:505`'s DEFERRED
484      // path), preserving `\__seq_item:n {…}` inside `\seq_gpush:Nn`'s
485      // `\unexpanded`-wrapped body. Without this, the seq stack stays
486      // empty after push, leading to `extra-pop-label` and the
487      // `\q_no_value` recursion cascade during `\@pushfilename`.
488      //
489      // We DON'T propagate alias for the ~1k other Lt-aliased entries
490      // (e.g. `\bool_if_exist:NTF` → `\cs_if_exist:NTF`) — those would
491      // change `defn.get_cs_name()`'s return value, which feeds into
492      // many lookup paths and triggers infinite-loop regressions in
493      // `\@nil` handling, etc. Keep blast radius tight.
494      const DEFERRED_NAMES: &[&str] = &["\\unexpanded", "\\the", "\\detokenize", "\\showthe"];
495      let alias_decoded = url_decode(eparts[0]);
496      let is_alias_diff = cs_tok.with_cs_name(|s| s != alias_decoded.as_str());
497      let alias_for_traits = if is_alias_diff && DEFERRED_NAMES.contains(&alias_decoded.as_str()) {
498        Some(alias_decoded)
499      } else {
500        None
501      };
502      let nargs: usize = eparts[1].parse().unwrap_or(0);
503      let flags = eparts[2];
504      let tok_data = eparts[3];
505      let proto_opt = eparts
506        .get(4)
507        .map(|s| url_decode(s))
508        .filter(|s| !s.is_empty());
509      let v3_opt = eparts.get(5).filter(|s| !s.is_empty());
510
511      let is_long = flags.contains('L');
512      let is_protected = flags.contains('P');
513
514      let expansion = parse_token_list(tok_data)?;
515
516      // Build parameter spec, preferring v3 structured → v2 proto →
517      // v1 nargs-repeat fallback. init_flag=true for both fallbacks:
518      // state is live at runtime so Parameter::init() can resolve
519      // readers via PARAMETER_TYPES.
520      let paramlist = if let Some(v3) = v3_opt {
521        match parse_parameters_v3(v3) {
522          Ok(pl) => pl,
523          Err(_) => proto_opt
524            .as_ref()
525            .and_then(|p| crate::common::def_parser::parse_parameters(p, &cs_tok, true).ok())
526            .flatten()
527            .or_else(|| {
528              if nargs > 0 {
529                let fallback = "{}".repeat(nargs);
530                crate::common::def_parser::parse_parameters(&fallback, &cs_tok, true)
531                  .ok()
532                  .flatten()
533              } else {
534                None
535              }
536            }),
537        }
538      } else {
539        // v2 path: no v3 field, fall back to proto-parsing (with the
540        // original silent-degrade-to-nargs behavior).
541        match proto_opt {
542          Some(proto) => match crate::common::def_parser::parse_parameters(&proto, &cs_tok, true) {
543            Ok(pl) => pl,
544            Err(_) if nargs > 0 => {
545              let fallback = "{}".repeat(nargs);
546              crate::common::def_parser::parse_parameters(&fallback, &cs_tok, true)
547                .map_err(|e| format!("Param parse fallback: {}", e))?
548            },
549            Err(_) => None,
550          },
551          None if nargs > 0 => {
552            let proto = "{}".repeat(nargs);
553            crate::common::def_parser::parse_parameters(&proto, &cs_tok, true)
554              .map_err(|e| format!("Param parse: {}", e))?
555          },
556          None => None,
557        }
558      };
559
560      let options = Some(ExpandableOptions {
561        long: is_long,
562        protected: is_protected,
563        nopack_parameters: true, // tokens already have ARG catcode
564        alias: alias_for_traits,
565        ..ExpandableOptions::default()
566      });
567
568      let expansion_body = Tokens::from(expansion).into();
569      match Expandable::new(cs_tok, paramlist, Some(expansion_body), options) {
570        Ok(mut exp) => {
571          // Perl #aaacdba2: stamp dump-loaded definitions with a
572          // nominal Locator pointing at the dump file + line. Helps
573          // diagnostics attribute errors to the dump source rather
574          // than the arena's compile-site default.
575          exp.locator = current_dump_locator();
576          // Perl `I()` (`Core/Dumper.pm` L67):
577          //   sub I { State::assign_internal($STATE,'meaning',
578          //           $_[0]->getCSName, $_[0], 'global'); }
579          // Direct table mutation — no `:locked` gate, no add-only.
580          // CONFIRMED via probe (2026-04-27): Perl dump load bypasses
581          // the :locked gate. `installDefinition` (State.pm L502-517)
582          // checks :locked and refuses; `assign_internal` (State.pm
583          // L140) does not. Dumper's `I` shorthand calls `assign_internal`
584          // directly, so locked defs ARE silently overwritten by dump.
585          // Rust matches: this code calls `state::assign_internal`, not
586          // `install_definition`. Verified `\hidewidth` and `\leavevmode`
587          // get overwritten by dump entries despite earlier bootstrap defs.
588          state::assign_internal(
589            TableName::Meaning,
590            cs_tok.get_cs_name(),
591            Stored::from(exp),
592            Some(Scope::Global),
593          );
594          Ok(true)
595        },
596        Err(e) => Err(format!("Expandable creation failed: {}", e)),
597      }
598    },
599    "T" => {
600      // Token meaning. Perl `Im()` (`Core/Dumper.pm` L66):
601      //   sub Im { State::assign_internal($STATE,'meaning',
602      //            $_[0], $_[1], 'global'); }
603      // Direct write — no `\let`-chase, no chain follow.
604      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
605      state::assign_internal(
606        TableName::Meaning,
607        cs_tok.get_cs_name(),
608        Stored::Token(tok),
609        Some(Scope::Global),
610      );
611      Ok(true)
612    },
613    "FD" => {
614      // FontDef: `FD\t<font_id>` — Perl `dump_primitive` (Core/Dumper.pm L383-389)
615      // emits this for `\font`-defined primitives. Install a Primitive whose
616      // `before_digest` mirrors `LaTeXML::Core::Definition::FontDef::invoke`
617      // (FontDef.pm L38-45):
618      //   1. lookup the fontinfo hash at <font_id>
619      //   2. assignValue(current_FontDef => $cs)
620      //   3. merge the font into $STATE->lookupValue('font')
621      // The fontinfo Stored::Font rides through the dump as a `V` entry with
622      // `F\t...` payload (see Stored::Font arm in dump_writer + the `F` arm
623      // in parse_value above).
624      use crate::definition::BeforeDigestClosure;
625      use crate::definition::primitive::Primitive;
626      let font_id_raw = url_decode(parts.get(1).unwrap_or(&""));
627      let font_id_pin = arena::pin(&font_id_raw);
628      let font_id_str = font_id_raw.clone();
629      let cs_for_fontdef = cs_tok;
630      let merge_closure: BeforeDigestClosure = std::rc::Rc::new(move || {
631        crate::state::assign_value("current_FontDef", Stored::Token(cs_for_fontdef), None);
632        if let Some(Stored::Font(f)) = crate::state::lookup_value(&font_id_str) {
633          crate::binding::content::merge_font((*f).clone());
634        }
635        Ok(Vec::new())
636      });
637      let prim = Primitive {
638        cs: cs_tok,
639        before_digest: vec![merge_closure],
640        font_id: Some(font_id_pin),
641        ..Primitive::default()
642      };
643      state::assign_internal(
644        TableName::Meaning,
645        cs_tok.get_cs_name(),
646        Stored::Primitive(std::rc::Rc::new(prim)),
647        Some(Scope::Global),
648      );
649      Ok(true)
650    },
651    "PA" | "MPA" => {
652      // Primitive alias: PA\t<target_cs> — the entry's meaning is an
653      // Rc<Primitive> whose own cs is <target_cs>. If <target_cs> == key
654      // this is the "primary" entry (already provided by compiled bindings
655      // in _base.rs etc.); skip. Otherwise, replay `\let <key> <target>`
656      // so the Rc<Primitive> is shared just as it was when the dump was
657      // generated. This is how \tex_let:D, \tex_def:D, etc. survive the
658      // dump — without this the expl3.sty short-circuit guard
659      // `\ifx\csname tex_let:D\endcsname\relax` never fires and the 36k
660      // lines of expl3-code.tex get reprocessed on every run.
661      let target_cs_raw = url_decode(parts.get(1).unwrap_or(&""));
662      if target_cs_raw == key {
663        return Ok(false);
664      }
665      let target_tok = Token {
666        text: arena::pin(&target_cs_raw),
667        code: Catcode::CS,
668      };
669      // Perl `Lt()` (`Core/Dumper.pm` L69-72):
670      //   sub Lt { my $d = State::lookupDefinition($STATE, T_CS($_[1]));
671      //            State::assign_internal($STATE,'meaning',$_[0],$d,'global'); }
672      // Look up the target's current Meaning entry, then writes that
673      // very Stored value at the alias key. Sharing the Rc preserves
674      // identity (e.g. \let\tex_let:D\let keeps the same Primitive Rc).
675      //
676      // If the target is not yet defined (load order has _constructs
677      // running after the dump for some let-aliases — e.g.
678      // `\let\a=\@tabacckludge`), defer until flush_deferred_aliases().
679      if !state::has_meaning(&target_tok) {
680        DEFERRED_ALIASES.with(|cell| {
681          cell.borrow_mut().push((cs_tok, target_tok));
682        });
683        return Ok(false);
684      }
685      let target_meaning = state::lookup_meaning(&target_tok);
686      if let Some(meaning) = target_meaning {
687        state::assign_internal(
688          TableName::Meaning,
689          cs_tok.get_cs_name(),
690          meaning,
691          Some(Scope::Global),
692        );
693      }
694      Ok(true)
695    },
696    "R" => {
697      // Register: R\tCS\tTYPE\tVALUE[\tMATHGLYPH][\tADDRESS]
698      // rparts[0] (internal CS name) is redundant with the outer key —
699      // same reasoning as the E arm; we skip the decode + alloc.
700      // ADDRESS field is a url-encoded address-slot key for allocated
701      // registers (Perl `\newcount\m@ne` → address='\count22'). When
702      // absent, address defaults to the CS name. Without this, dump_reader
703      // wrote `\m@ne`'s -1 to its CS-name slot, but `\m@ne`'s actual
704      // address (`\count22`) held the default 0 — `\settabs 20\columns`
705      // looped infinitely because `\m@ne == 0` never advanced `\count@`
706      // toward 0 in `\loop\ifnum\count@>\z@\@nother\repeat`.
707      let rparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(5, '\t').collect();
708      if rparts.len() < 3 {
709        return Err("Incomplete Register entry".into());
710      }
711      let rtype = rparts[1];
712      let value_str = rparts[2];
713      let mathglyph = rparts
714        .get(3)
715        .filter(|s| !s.is_empty())
716        .and_then(|s| s.parse::<u32>().ok())
717        .and_then(char::from_u32);
718      let dump_address: Option<String> = rparts
719        .get(4)
720        .filter(|s| !s.is_empty())
721        .map(|s| url_decode(s));
722      // For register-aliases (M-line key != register's internal cs), the
723      // storage slot lives at the cs name, not the alias key. e.g.
724      //   M  \tex_endlinechar:D  R  \endlinechar  N  0
725      // means "\tex_endlinechar:D" is meaning-installed but the underlying
726      // register storage is at "\endlinechar". Without this, assignments
727      // through the alias (\tex_endlinechar:D = 32) write to a separate
728      // slot and the real \endlinechar stays unchanged — breaking
729      // \ExplSyntaxOn's `\tex_endlinechar:D = 32 \scan_stop:` line, which
730      // in turn breaks the entire dump-path expl3 whitespace handling
731      // (8 expl3 tests). Mirror Perl's address-via-internal-cs semantics.
732      let internal_cs_decoded = url_decode(rparts[0]);
733      let dump_address: Option<String> = dump_address.or_else(|| {
734        if internal_cs_decoded != *key && !internal_cs_decoded.is_empty() {
735          Some(internal_cs_decoded)
736        } else {
737          None
738        }
739      });
740
741      use crate::common::number::Number;
742      use crate::definition::register::{Register, RegisterType, RegisterValue};
743
744      let (reg_type, reg_value) = match rtype {
745        "N" | "CD" => {
746          let n: i64 = value_str.parse().unwrap_or(0);
747          let rt = if rtype == "CD" {
748            RegisterType::CharDef
749          } else {
750            RegisterType::Number
751          };
752          (rt, Some(RegisterValue::Number(Number::new(n))))
753        },
754        "D" => {
755          let n: i64 = value_str.parse().unwrap_or(0);
756          (
757            RegisterType::Dimension,
758            Some(RegisterValue::Dimension(
759              crate::common::dimension::Dimension(n),
760            )),
761          )
762        },
763        "G" => (
764          RegisterType::Glue,
765          Some(RegisterValue::Glue(parse_glue(value_str)?)),
766        ),
767        "MG" => (
768          RegisterType::MuGlue,
769          Some(RegisterValue::MuGlue(parse_muglue(value_str)?)),
770        ),
771        "TK" => {
772          // Token register: value is comma-separated token list, or "0" for empty
773          let toks = if value_str == "0" || value_str.is_empty() {
774            Vec::new()
775          } else {
776            parse_token_list(value_str)?
777          };
778          (
779            RegisterType::Tokens,
780            Some(RegisterValue::Tokens(Tokens::from(toks))),
781          )
782        },
783        _ => return Ok(false),
784      };
785
786      // Perl-parity with def_register (binding/def/dialect.rs): store the
787      // initial value at the Register's `address` slot AND set `default`, so
788      // a subsequent `value_of` lookup — which reads state::with_value(address)
789      // and falls back to `default` — actually sees the dump's initial value.
790      // CharDefs read their immediate `value` field instead, so we skip the
791      // storage write for them.
792      let mut reg = Register {
793        cs: cs_tok,
794        register_type: reg_type,
795        value: reg_value.clone(),
796        default: if matches!(reg_type, RegisterType::CharDef) {
797          None
798        } else {
799          reg_value.clone()
800        },
801        mathglyph,
802        locator: current_dump_locator(),
803        ..Register::default()
804      };
805      // Set address: prefer dump-supplied address (allocated registers),
806      // fall back to CS name (direct registers like `\count1`).
807      let has_explicit_address = dump_address.is_some();
808      reg.address = dump_address.unwrap_or_else(|| key.to_string());
809      if !matches!(reg_type, RegisterType::CharDef) {
810        if let Some(ref rv) = reg_value {
811          // Perl `R(...)` register dump-restore: the address slot
812          // gets the initial value via `assign_internal('value', ...,
813          // 'global')`. Mirror Perl `def_register` behavior: when the
814          // address is allocated (different from CS) AND already has a
815          // value (from an earlier V entry), DO NOT overwrite — the V
816          // entry holds the runtime value (e.g. `\m@ne`'s `\count22 =
817          // -1`), and the Register's `value` field is just the default
818          // (typically 0). Without this guard, the M entry resets
819          // `\count22` to 0, breaking `\settabs 20\columns` (loops
820          // because `\m@ne` reads as 0 instead of -1, so
821          // `\advance\count@\m@ne` doesn't decrement).
822          let should_assign = !has_explicit_address || !state::has_value(&reg.address);
823          if should_assign {
824            state::assign_internal(
825              TableName::Value,
826              arena::pin(&reg.address),
827              rv.clone(),
828              Some(Scope::Global),
829            );
830          }
831        }
832      }
833      // Perl `I(...)` for the Register meaning entry — direct
834      // `assign_internal('meaning', ..., 'global')`, bypassing the
835      // `:locked` and add-only checks of install_definition.
836      state::assign_internal(
837        TableName::Meaning,
838        cs_tok.get_cs_name(),
839        Stored::from(reg),
840        Some(Scope::Global),
841      );
842      Ok(true)
843    },
844    _ => Ok(false),
845  }
846}

848/// Decode a character key from the dump. Handles:
849/// - Single characters: "A", "è"
850/// - URL-encoded control chars: "%19" (→ char 0x19), "%0A" (→ char 0x0A)
851fn decode_char_key(key: &str) -> Option<char> {
852  let decoded = url_decode(key);
853  decoded.chars().next()
854}

856/// Char-keyed table key: dump uses the single character as the key.
857/// `assign_internal` wants a SymStr — pin the single-char string.
858fn char_key(ch: char) -> crate::common::arena::SymStr {
859  let mut buf = [0u8; 4];
860  arena::pin(ch.encode_utf8(&mut buf))
861}

863/// Load a catcode entry: C\tCHAR\tCC\tVALUE.
864/// Perl `Cc()` (`Core/Dumper.pm` L60): `assign_internal('catcode', ..., 'global')`.
865fn load_catcode(key: &str, data: &str) -> Result<bool, String> {
866  let ch = decode_char_key(key).ok_or_else(|| format!("Bad catcode char: {}", key))?;
867  let parts: Vec<&str> = data.splitn(2, '\t').collect();
868  if parts.len() < 2 || parts[0] != "CC" {
869    return Err(format!("Bad catcode data: {}", data));
870  }
871  let cc: u8 = parts[1]
872    .parse()
873    .map_err(|e| format!("Bad catcode value: {}", e))?;
874  state::assign_internal(
875    TableName::Catcode,
876    char_key(ch),
877    Stored::Catcode(Catcode::from(cc)),
878    Some(Scope::Global),
879  );
880  Ok(true)
881}

973/// Parse a single token from "CC:TEXT" format
974fn parse_token(s: &str) -> Result<Token, String> {
975  let (cc_str, text) = s.split_once(':').ok_or("Missing ':' in token")?;
976  let cc: u8 = cc_str.parse().map_err(|e| format!("Bad CC: {}", e))?;
977  // Fast path: most token text fields have no `%` escapes, so pin the
978  // &str directly — avoids the String allocation url_decode would make
979  // even on its own fast path. Parsing the expl3 kernel alone produces
980  // hundreds of thousands of token entries; every `to_owned()` avoided
981  // here matters.
982  let text_sym = if text.contains('%') {
983    arena::pin(url_decode(text))
984  } else {
985    arena::pin(text)
986  };
987  Ok(Token {
988    text: text_sym,
989    code: Catcode::from(cc),
990  })
991}

993/// Parse comma-separated token list
994fn parse_token_list(s: &str) -> Result<Vec<Token>, String> {
995  if s.is_empty() {
996    return Ok(Vec::new());
997  }
998  s.split(',').map(parse_token).collect()
999}

1001/// Decode the v3 structured Parameters encoding emitted by
1002/// `dump_writer::serialize_parameters_v3` (see that function's docstring
1003/// and `docs/DUMP_FORMAT_PERL_ANALYSIS.md` for the layout).
1004///
1005/// Returns `Ok(None)` for an empty record (no parameters); `Ok(Some(ps))`
1006/// on success; `Err` if any record is malformed. Each Parameter is
1007/// constructed via `Parameter::new(name, spec, Some(extras))`, which
1008/// calls `init()` — the reader function is resolved against the live
1009/// PARAMETER_TYPES table, mirroring the runtime path.
1010pub(crate) fn parse_parameters_v3(
1011  v3: &str,
1012) -> Result<Option<crate::parameter::Parameters>, String> {
1013  if v3.is_empty() {
1014    return Ok(None);
1015  }
1016  let mut params = Vec::new();
1017  for record in v3.split('\x1e') {
1018    // <name>\x1f<spec>\x1f<flags>\x1f<extras>
1019    let fields: Vec<&str> = record.splitn(4, '\x1f').collect();
1020    if fields.len() != 4 {
1021      return Err(format!(
1022        "v3 Parameter record has {} fields, expected 4",
1023        fields.len()
1024      ));
1025    }
1026    let name = url_decode(fields[0]);
1027    let spec = url_decode(fields[1]);
1028    let flags = fields[2];
1029    let extras_str = fields[3];
1030
1031    let extras = if extras_str.is_empty() {
1032      Vec::new()
1033    } else {
1034      extras_str
1035        .split('\x1d')
1036        .map(|tok_list| parse_token_list(tok_list).map(crate::tokens::Tokens::new))
1037        .collect::<Result<Vec<_>, _>>()?
1038    };
1039
1040    let mut param = crate::parameter::Parameter::new(name, spec, Some(extras))
1041      .map_err(|e| format!("Parameter::new failed: {}", e))?;
1042
1043    // Apply flags after construction — Parameter::new + init() handle
1044    // the reader side, but novalue/optional are struct-level booleans
1045    // that the spec-driven init() may or may not have set (e.g. "Optional"
1046    // prefix auto-sets optional, but we want explicit round-trip).
1047    for flag in flags.split(';').filter(|s| !s.is_empty()) {
1048      match flag {
1049        "n=1" => param.novalue = true,
1050        "o=1" => param.optional = true,
1051        _ => {
1052          // Unknown flag — ignore for forward compat; future flags
1053          // added to the writer shouldn't break older readers.
1054        },
1055      }
1056    }
1057    params.push(param);
1058  }
1059  if params.is_empty() {
1060    Ok(None)
1061  } else {
1062    Ok(Some(crate::parameter::Parameters::new(params)))
1063  }
1064}

1066pub(crate) fn url_decode(s: &str) -> String {
1067  // Fast path: the overwhelming majority of dump entries have no `%`
1068  // escapes in their key or proto fields, so a single memcpy via
1069  // `to_owned()` beats char-by-char iteration for ~19k key loads.
1070  if !s.contains('%') {
1071    return s.to_owned();
1072  }
1073  let mut result = String::with_capacity(s.len());
1074  let mut chars = s.chars();
1075  while let Some(ch) = chars.next() {
1076    if ch == '%' {
1077      let hex: String = chars.by_ref().take(2).collect();
1078      if let Ok(byte) = u8::from_str_radix(&hex, 16) {
1079        result.push(byte as char);
1080      }
1081    } else {
1082      result.push(ch);
1083    }
1084  }
1085  result
1086}

1088/// Parse a serialized Glue value: "skip,pN,pfN,mN,mfN"
1089fn parse_glue(s: &str) -> Result<crate::common::glue::Glue, String> {
1090  use crate::common::glue::{FillCode, Glue};
1091  let mut skip = 0i64;
1092  let mut plus = None;
1093  let mut pfill = None;
1094  let mut minus = None;
1095  let mut mfill = None;
1096  for (i, part) in s.split(',').enumerate() {
1097    if i == 0 {
1098      skip = part.parse().map_err(|e| format!("Bad glue skip: {}", e))?;
1099    } else if let Some(rest) = part.strip_prefix("pf") {
1100      pfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
1101    } else if let Some(rest) = part.strip_prefix('p') {
1102      plus = Some(rest.parse().map_err(|e| format!("Bad glue plus: {}", e))?);
1103    } else if let Some(rest) = part.strip_prefix("mf") {
1104      mfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
1105    } else if let Some(rest) = part.strip_prefix('m') {
1106      minus = Some(rest.parse().map_err(|e| format!("Bad glue minus: {}", e))?);
1107    }
1108  }
1109  Ok(Glue {
1110    skip,
1111    plus,
1112    pfill,
1113    minus,
1114    mfill,
1115  })
1116}
