1//! Reader for Rust-native kernel dump files (produced by dump_writer.rs).
2//!
3//! Loads a dump file produced by `latexml_oxide --init=latex.ltx --dest=dump`
4//! and replays the state assignments into the engine.
5//!
6//! **Loading policy:** The dump loads AFTER the compiled engine definitions.
7//! Meanings (M entries) use add-only policy: skip if the CS is already defined.
8//! Values (V entries) use add-only policy: skip if the key already has a value.
9//! This ensures compiled engine semantics (constructors, etc.) take priority
10//! over raw TeX definitions from the dump, matching Perl's approach where
11//! `latex_constructs` overrides the dump.
12//!
13//! Format: tab-separated lines:
14//!   V\tKEY\tTYPE\tDATA             — value assignment
15//!   M\tKEY\tE\tCS\tNARGS\tFLAGS\tTOKENS — Expandable definition
16//!   M\tKEY\tN                       — None meaning (undefined)
17//!   M\tKEY\tT\tCC:TEXT              — Token meaning (let-assignment)
18//!   C\tCHAR\tCC\tVALUE             — catcode assignment
19//!   LC\tCHAR\tCH\tVALUE            — lccode assignment
20//!   UC\tCHAR\tCH\tVALUE            — uccode assignment
21//!   SC\tCHAR\tCH\tVALUE            — sfcode assignment
22//!   DC\tCHAR\tCH\tVALUE            — delcode assignment
23//!   MC\tCHAR\tCH\tVALUE            — mathcode assignment
24

36/// Load a Rust-native dump file into the current State.
37/// Returns the number of entries loaded.
38pub fn load_native_dump(path: &Path) -> Result<usize, String> {
39  let content = std::fs::read_to_string(path)
40    .map_err(|e| format!("Failed to read dump file {}: {}", path.display(), e))?;
41  let count = load_from_str_internal(&content, &path.display().to_string())?;
42  Ok(count)
43}

45/// Load dump data from a string (used by embedded dump modules).
46/// Returns the number of entries loaded.
47pub fn load_from_str(content: &str) -> Result<usize, String> {
48  load_from_str_internal(content, "<embedded>")

51fn load_from_str_internal(content: &str, source_name: &str) -> Result<usize, String> {
52  let mut count = 0;
53  let mut skipped = 0;
54  let mut errors = 0;
55
56  for (lineno, line) in content.lines().enumerate() {
57    let line = line.trim();
58    if line.is_empty() || line.starts_with('#') {
59      continue;
60    }
61
62    match parse_and_load(line) {
63      Ok(true) => count += 1,
64      Ok(false) => skipped += 1,
65      Err(e) => {
66        errors += 1;
67        if errors <= 10 {
68          log::warn!(
69            "[dump_reader] Line {}: {}: {}",
70            lineno + 1,
71            e,
72            &line[..line.len().min(80)]
73          );
74        }
75      }
76    }
77  }
78
79  if errors > 10 {
80    log::warn!("[dump_reader] ... and {} more errors", errors - 10);
81  }
82  log::info!(
83    "[dump_reader] Loaded {} entries from {} ({} skipped, {} errors)",
84    count,
85    source_name,
86    skipped,
87    errors
88  );
89
90  Ok(count)
91}

93/// Parse a single dump line and load it. Returns Ok(true) if loaded,
94/// Ok(false) if skipped (already defined or filtered), Err on parse error.
95fn parse_and_load(line: &str) -> Result<bool, String> {
96  let parts: Vec<&str> = line.splitn(3, '\t').collect();
97  if parts.len() < 2 {
98    return Err("Too few fields".into());
99  }
100
101  let table = parts[0];
102  let key = url_decode(parts[1]);
103  let data = if parts.len() > 2 { parts[2] } else { "" };
104
105  match table {
106    // V: Value entries (registers, fontdimen, font metadata).
107    // Add-only policy: only loads if key has no existing value.
108    "V" => load_value(&key, data),
109    // M: Meaning entries (expandable definitions).
110    // Load @-internal entries (with '@' in name) whose expansions don't
111    // reference the expl3 hook system. These are LaTeX kernel internals
112    // like \@fontswitch, \@thmcounter, etc. that raw TeX classes need.
113    // Skip public macros and expl3 entries (with ':') — too risky.
114    "M" => {
115      let name = key.trim_start_matches('\\');
116      let is_at_internal = name.contains('@') && !name.contains(':');
117      if is_at_internal && !data.contains("\\\\hook") && !data.contains("16:\\hook") {
118        load_meaning(&key, data)
119      } else {
120        Ok(false)
121      }
122    },
123    // LC/UC: case-mapping codes — safe, always load
124    "LC" => load_lccode(&key, data),
125    "UC" => load_uccode(&key, data),
126    // SC: space factor codes — safe, always load
127    "SC" => load_sfcode(&key, data),
128    // C: catcodes — only for non-ASCII (>127). ASCII catcodes are set by
129    // the engine; loading from dump would conflict.
130    "C" => {
131      let ch = decode_char_key(&key);
132      if ch.is_some_and(|c| c as u32 > 127) {
133        load_catcode(&key, data)
134      } else {
135        Ok(false)
136      }
137    }
138    // MC/DC: mathcodes and delcodes from the dump are corrupted by expl3
139    // format initialization (e.g., mathcode('v')=618 maps to '|').
140    // The engine sets correct math/delcodes. Skip.
141    "MC" | "DC" => Ok(false),
142    _ => Ok(false),
143  }
144}

146/// V entries to unconditionally skip (runtime state, never useful from dump).
147const SKIP_VALUE_KEYS: &[&str] = &[
148  "INTERPRETING_DEFINITIONS",
149  "if_count",
150  "absorb_count",
151  "if_stack",
152  "INCLUDE_COMMENTS",
153  "INCLUDE_STYLES",
154  "INPUT_ENCODING",
155  "CURRENT_INPUT_ENCODING",
156  "SUPPRESS_UNEXPECTED_ERRORS",
157  "SUPPRESS_UNDEFINED_ERRORS",
158];
159
160/// V entry key prefixes to skip.
161const SKIP_VALUE_PREFIXES: &[&str] = &[
162  "input_file:",
163  "output_file:",
164  "texsys",
165];
166
167/// V entry key substrings to skip.
168const SKIP_VALUE_CONTAINS: &[&str] = &[
169  "_loaded",       // Package loading flags — must be set by actual loading
170  "_found_loaded", // Package found+loaded flags
171];

173/// Load a value entry: V\tKEY\tTYPE\tDATA
174///
175/// Uses add-only policy: only loads if the key does not already have a value.
176/// This ensures compiled engine state takes priority over dump state.
177fn load_value(key: &str, data: &str) -> Result<bool, String> {
178  // Skip unconditional keys
179  for skip in SKIP_VALUE_KEYS {
180    if key == *skip {
181      return Ok(false);
182    }
183  }
184  // Skip by prefix
185  for prefix in SKIP_VALUE_PREFIXES {
186    if key.starts_with(prefix) {
187      return Ok(false);
188    }
189  }
190  // Skip by substring
191  for substr in SKIP_VALUE_CONTAINS {
192    if key.contains(substr) {
193      return Ok(false);
194    }
195  }
196
197  // Add-only policy: don't overwrite any existing value.
198  // This preserves engine-configured state (e.g., \everymath, \everypar,
199  // named skips, etc.) while filling in gaps from the dump (e.g., expl3
200  // fontdimen intarrays, font metadata).
201  if state::has_value(key) {
202    return Ok(false);
203  }
204
205  let parts: Vec<&str> = data.splitn(2, '\t').collect();
206  if parts.is_empty() {
207    return Err("Missing value type".into());
208  }
209
210  let value = match parts[0] {
211    "N" => return Ok(false), // Don't load None values (would erase existing)
212    "B" => Stored::Bool(parts.get(1).map(|s| *s == "1").unwrap_or(false)),
213    "I" => {
214      let n: i64 = parts
215        .get(1)
216        .unwrap_or(&"0")
217        .parse()
218        .map_err(|e| format!("Bad int: {}", e))?;
219      Stored::Int(n)
220    }
221    "S" => Stored::from(url_decode(parts.get(1).unwrap_or(&""))),
222    "CH" => {
223      let n: u16 = parts
224        .get(1)
225        .unwrap_or(&"0")
226        .parse()
227        .map_err(|e| format!("Bad charcode: {}", e))?;
228      Stored::Charcode(n)
229    }
230    "CC" => {
231      let n: u8 = parts
232        .get(1)
233        .unwrap_or(&"0")
234        .parse()
235        .map_err(|e| format!("Bad catcode: {}", e))?;
236      Stored::Catcode(Catcode::from(n))
237    }
238    "T" => {
239      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
240      Stored::Token(tok)
241    }
242    "TK" => {
243      let toks = parse_token_list(parts.get(1).unwrap_or(&""))?;
244      Stored::Tokens(Tokens::from(toks))
245    }
246    "D" => {
247      let n: i64 = parts.get(1).unwrap_or(&"0").parse()
248        .map_err(|e| format!("Bad dimension: {}", e))?;
249      Stored::Dimension(crate::common::dimension::Dimension(n))
250    }
251    "G" => {
252      Stored::Glue(parse_glue(parts.get(1).unwrap_or(&"0"))?)
253    }
254    "MD" => {
255      let n: i64 = parts.get(1).unwrap_or(&"0").parse()
256        .map_err(|e| format!("Bad mudimension: {}", e))?;
257      Stored::MuDimension(crate::common::mudimension::MuDimension(n))
258    }
259    "MG" => {
260      Stored::MuGlue(parse_muglue(parts.get(1).unwrap_or(&"0"))?)
261    }
262    "VD" => return Ok(false), // Don't load empty VecDeque (runtime state)
263    _ => return Ok(false),    // Unknown value type
264  };
265
266  state::assign_value(key, value, Some(Scope::Global));
267  Ok(true)
268}

270/// Load a meaning entry: M\tKEY\tTYPE\t...
271///
272/// Uses add-only policy: skip if the CS already has a meaning.
273/// Additionally, only loads "safe" definitions — those that won't interfere
274/// with the compiled engine's processing during normal LaTeX operation:
275/// - expl3 internals (contain `:`) — safe because `:` is OTHER under normal catcodes
276/// - Private LaTeX internals (contain `@`) — only invoked by other macros
277/// - Skip all "public" macros that could be invoked during normal expansion
278///   and might reference hooks/primitives not supported by our engine
279#[allow(dead_code)]
280fn load_meaning(key: &str, data: &str) -> Result<bool, String> {
281  let cs_tok = Token {
282    text: arena::pin(key),
283    code: Catcode::CS,
284  };
285
286  // Add-only policy: don't override ANY existing definition.
287  if state::has_meaning(&cs_tok) {
288    return Ok(false);
289  }
290
291  // Safety filter: only load definitions that won't interfere with normal
292  // LaTeX processing. Public macros from the dump (like \document, \hook,
293  // \UseOneTimeHook) can reference expl3 hooks and internal state that
294  // our engine doesn't fully support, causing cascading errors.
295  //
296  // Safe: expl3 internals (with `:` or `__`), LaTeX internals (with `@`)
297  // Unsafe: public macros without `:` or `@` (e.g., \document, \hook)
298  let name = key.trim_start_matches('\\');
299  let is_internal = name.contains(':') || name.contains('@');
300  if !is_internal {
301    // This is a "public" macro. Skip it — our engine either already defines
302    // it (caught by has_meaning above) or it's a raw TeX definition that
303    // would interfere with our LaTeXML-specific processing.
304    return Ok(false);
305  }
306
307  let parts: Vec<&str> = data.splitn(2, '\t').collect();
308  if parts.is_empty() {
309    return Err("Missing meaning type".into());
310  }
311
312  match parts[0] {
313    "N" => {
314      // None meaning — skip (don't define as undefined)
315      Ok(false)
316    }
317    "E" => {
318      // Expandable: E\tCSNAME\tNARGS\tFLAGS\tTOKENS
319      let eparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(4, '\t').collect();
320      if eparts.len() < 4 {
321        return Err("Incomplete Expandable entry".into());
322      }
323
324      let _cs_name = url_decode(eparts[0]);
325      let nargs: usize = eparts[1].parse().unwrap_or(0);
326      let flags = eparts[2];
327      let tok_data = eparts[3];
328
329      let is_long = flags.contains('L');
330      let is_protected = flags.contains('P');
331
332      let expansion = parse_token_list(tok_data)?;
333
334      // Build parameter spec from nargs
335      let paramlist = if nargs > 0 {
336        let proto = "{}".repeat(nargs);
337        crate::common::def_parser::parse_parameters(&proto, &cs_tok, false)
338          .map_err(|e| format!("Param parse: {}", e))?
339      } else {
340        None
341      };
342
343      let options = Some(ExpandableOptions {
344        long: is_long,
345        protected: is_protected,
346        nopack_parameters: true, // tokens already have ARG catcode
347        ..ExpandableOptions::default()
348      });
349
350      let expansion_body = Tokens::from(expansion).into();
351      match Expandable::new(cs_tok, paramlist, Some(expansion_body), options) {
352        Ok(exp) => {
353          state::install_definition(exp, Some(Scope::Global));
354          Ok(true)
355        }
356        Err(e) => Err(format!("Expandable creation failed: {}", e)),
357      }
358    }
359    "T" => {
360      // Token meaning (let-assignment)
361      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
362      state::assign_meaning(&cs_tok, tok, Some(Scope::Global));
363      Ok(true)
364    }
365    "R" => {
366      // Register: R\tCS\tTYPE\tVALUE[\tMATHGLYPH]
367      let rparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(4, '\t').collect();
368      if rparts.len() < 3 {
369        return Err("Incomplete Register entry".into());
370      }
371      let _cs_name = url_decode(rparts[0]);
372      let rtype = rparts[1];
373      let value_str = rparts[2];
374      let mathglyph = rparts.get(3).and_then(|s| s.parse::<u32>().ok()).and_then(char::from_u32);
375
376      use crate::definition::register::{Register, RegisterType, RegisterValue};
377      use crate::common::number::Number;
378
379      let (reg_type, reg_value) = match rtype {
380        "N" | "CD" => {
381          let n: i64 = value_str.parse().unwrap_or(0);
382          let rt = if rtype == "CD" { RegisterType::CharDef } else { RegisterType::Number };
383          (rt, Some(RegisterValue::Number(Number::new(n))))
384        }
385        "D" => {
386          let n: i64 = value_str.parse().unwrap_or(0);
387          (RegisterType::Dimension, Some(RegisterValue::Dimension(
388            crate::common::dimension::Dimension(n))))
389        }
390        "G" => {
391          (RegisterType::Glue, Some(RegisterValue::Glue(
392            parse_glue(value_str)?)))
393        }
394        "MG" => {
395          (RegisterType::MuGlue, Some(RegisterValue::MuGlue(
396            parse_muglue(value_str)?)))
397        }
398        "TK" => {
399          // Token register: value is comma-separated token list, or "0" for empty
400          let toks = if value_str == "0" || value_str.is_empty() {
401            Vec::new()
402          } else {
403            parse_token_list(value_str)?
404          };
405          (RegisterType::Tokens, Some(RegisterValue::Tokens(
406            Tokens::from(toks))))
407        }
408        _ => return Ok(false),
409      };
410
411      let mut reg = Register {
412        cs: cs_tok,
413        register_type: reg_type,
414        value: reg_value,
415        mathglyph,
416        ..Register::default()
417      };
418      // Set address from CS name
419      reg.address = key.to_string();
420      state::install_definition(reg, Some(Scope::Global));
421      Ok(true)
422    }
423    _ => Ok(false),
424  }
425}

427/// Decode a character key from the dump. Handles:
428/// - Single characters: "A", "è"
429/// - URL-encoded control chars: "%19" (→ char 0x19), "%0A" (→ char 0x0A)
430fn decode_char_key(key: &str) -> Option<char> {
431  let decoded = url_decode(key);
432  decoded.chars().next()
433}

435/// Load a catcode entry: C\tCHAR\tCC\tVALUE
436fn load_catcode(key: &str, data: &str) -> Result<bool, String> {
437  let ch = decode_char_key(key).ok_or_else(|| format!("Bad catcode char: {}", key))?;
438  let parts: Vec<&str> = data.splitn(2, '\t').collect();
439  if parts.len() < 2 || parts[0] != "CC" {
440    return Err(format!("Bad catcode data: {}", data));
441  }
442  let cc: u8 = parts[1].parse().map_err(|e| format!("Bad catcode value: {}", e))?;
443  state::assign_catcode(ch, Catcode::from(cc), Some(Scope::Global));
444  Ok(true)
445}

509/// Parse a single token from "CC:TEXT" format
510fn parse_token(s: &str) -> Result<Token, String> {
511  let (cc_str, text) = s.split_once(':').ok_or("Missing ':' in token")?;
512  let cc: u8 = cc_str.parse().map_err(|e| format!("Bad CC: {}", e))?;
513  Ok(Token {
514    text: arena::pin(url_decode(text)),
515    code: Catcode::from(cc),
516  })
517}
518
519/// Parse comma-separated token list
520fn parse_token_list(s: &str) -> Result<Vec<Token>, String> {
521  if s.is_empty() {
522    return Ok(Vec::new());
523  }
524  s.split(',').map(parse_token).collect()
525}

527fn url_decode(s: &str) -> String {
528  let mut result = String::with_capacity(s.len());
529  let mut chars = s.chars();
530  while let Some(ch) = chars.next() {
531    if ch == '%' {
532      let hex: String = chars.by_ref().take(2).collect();
533      if let Ok(byte) = u8::from_str_radix(&hex, 16) {
534        result.push(byte as char);
535      }
536    } else {
537      result.push(ch);
538    }
539  }
540  result
541}

543/// Parse a serialized Glue value: "skip,pN,pfN,mN,mfN"
544fn parse_glue(s: &str) -> Result<crate::common::glue::Glue, String> {
545  use crate::common::glue::{FillCode, Glue};
546  let mut skip = 0i64;
547  let mut plus = None;
548  let mut pfill = None;
549  let mut minus = None;
550  let mut mfill = None;
551  for (i, part) in s.split(',').enumerate() {
552    if i == 0 {
553      skip = part.parse().map_err(|e| format!("Bad glue skip: {}", e))?;
554    } else if let Some(rest) = part.strip_prefix("pf") {
555      pfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
556    } else if let Some(rest) = part.strip_prefix('p') {
557      plus = Some(rest.parse().map_err(|e| format!("Bad glue plus: {}", e))?);
558    } else if let Some(rest) = part.strip_prefix("mf") {
559      mfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
560    } else if let Some(rest) = part.strip_prefix('m') {
561      minus = Some(rest.parse().map_err(|e| format!("Bad glue minus: {}", e))?);
562    }
563  }
564  Ok(Glue { skip, plus, pfill, minus, mfill })
565}

568fn parse_muglue(s: &str) -> Result<crate::common::muglue::MuGlue, String> {

570  use crate::common::muglue::MuGlue;
571  let mut skip = 0i64;
572  let mut plus = None;
573  let mut pfill = None;
574  let mut minus = None;
575  let mut mfill = None;
576  for (i, part) in s.split(',').enumerate() {
577    if i == 0 {

579    } else if let Some(rest) = part.strip_prefix("pf") {
580      pfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
581    } else if let Some(rest) = part.strip_prefix('p') {
582      plus = Some(rest.parse().map_err(|e| format!("Bad muglue plus: {}", e))?);
583    } else if let Some(rest) = part.strip_prefix("mf") {
584      mfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
585    } else if let Some(rest) = part.strip_prefix('m') {
586      minus = Some(rest.parse().map_err(|e| format!("Bad muglue minus: {}", e))?);
587    }
588  }
589  Ok(MuGlue { skip, plus, pfill, minus, mfill })
590}
