1//! Reader for Rust-native kernel dump files (produced by dump_writer.rs).
2//!
3//! Loads a dump file produced by `latexml_oxide --init=latex.ltx --dest=dump`
4//! and replays the state assignments into the engine.
5//!
6//! **Loading policy:** The dump loads AFTER the compiled engine definitions.
7//! Meanings (M entries) use add-only policy: skip if the CS is already defined.
8//! Values (V entries) use add-only policy: skip if the key already has a value.
9//! This ensures compiled engine semantics (constructors, etc.) take priority
10//! over raw TeX definitions from the dump, matching Perl's approach where
11//! `latex_constructs` overrides the dump.
12//!
13//! Format: tab-separated lines:
14//!   V\tKEY\tTYPE\tDATA             — value assignment
15//!   M\tKEY\tE\tCS\tNARGS\tFLAGS\tTOKENS — Expandable definition
16//!   M\tKEY\tN                       — None meaning (undefined)
17//!   M\tKEY\tT\tCC:TEXT              — Token meaning (let-assignment)
18//!   C\tCHAR\tCC\tVALUE             — catcode assignment
19//!   LC\tCHAR\tCH\tVALUE            — lccode assignment
20//!   UC\tCHAR\tCH\tVALUE            — uccode assignment
21//!   SC\tCHAR\tCH\tVALUE            — sfcode assignment
22//!   DC\tCHAR\tCH\tVALUE            — delcode assignment
23//!   MC\tCHAR\tCH\tVALUE            — mathcode assignment

36/// Load a Rust-native dump file into the current State.
37/// Returns the number of entries loaded.
38pub fn load_native_dump(path: &Path) -> Result<usize, String> {
39  let content = std::fs::read_to_string(path)
40    .map_err(|e| format!("Failed to read dump file {}: {}", path.display(), e))?;
41  let count = load_from_str_internal(&content, &path.display().to_string())?;
42  Ok(count)
43}

45/// Load dump data from a string (used by embedded dump modules).
46/// Returns the number of entries loaded.
47pub fn load_from_str(content: &str) -> Result<usize, String> {
48  load_from_str_internal(content, "<embedded>")
49}

51// Per-load context used to attach a nominal Locator to dump-installed
52// Expandables. Matches Perl #aaacdba2 (2026): dump-loaded definitions
53// should be traceable to the dump file + line, not report the arena's
54// internal location. Thread-local so concurrent loads (there are none
55// today, but the state is cooperative) don't clobber each other.
56thread_local! {
57  static CURRENT_LOAD_CTX: std::cell::Cell<Option<(crate::common::arena::SymStr, u32)>> =
58    const { std::cell::Cell::new(None) };
59}
60
61fn current_dump_locator() -> crate::common::locator::Locator {
62  if let Some((source, lineno)) = CURRENT_LOAD_CTX.with(|c| c.get()) {
63    crate::common::locator::Locator {
64      source,
65      from_line: lineno,
66      to_line: lineno,
67      from_column: 0,
68      to_column: 0,
69    }
70  } else {
71    crate::common::locator::Locator::default()
72  }
73}

75fn load_from_str_internal(content: &str, source_name: &str) -> Result<usize, String> {
76  let mut count = 0;
77  let mut skipped = 0;
78  let mut errors = 0;
79  let source_sym = crate::common::arena::pin(source_name);
80
81  for (lineno, line) in content.lines().enumerate() {
82    // Trim only CR (from CRLF line endings); `lines()` already strips LF.
83    // Do NOT use `trim()` here — it strips trailing tabs, which are part of
84    // the tab-separated format for entries with empty trailing fields (e.g.
85    // E-entries with empty body: `E\t<cs>\t<nargs>\t<flags>\t`).
86    let line = line.trim_end_matches('\r');
87    if line.is_empty() || line.starts_with('#') {
88      continue;
89    }
90
91    CURRENT_LOAD_CTX.with(|c| c.set(Some((source_sym, (lineno + 1) as u32))));
92
93    match parse_and_load(line) {
94      Ok(true) => count += 1,
95      Ok(false) => skipped += 1,
96      Err(e) => {
97        errors += 1;
98        if errors <= 10 {
99          log::warn!(
100            "[dump_reader] Line {}: {}: {}",
101            lineno + 1,
102            e,
103            &line[..line.len().min(80)]
104          );
105        }
106      }
107    }
108  }
109
110  if errors > 10 {
111    log::warn!("[dump_reader] ... and {} more errors", errors - 10);
112  }
113  log::info!(
114    "[dump_reader] Loaded {} entries from {} ({} skipped, {} errors)",
115    count,
116    source_name,
117    skipped,
118    errors
119  );
120
121  Ok(count)
122}

124/// Parse a single dump line and load it. Returns Ok(true) if loaded,
125/// Ok(false) if skipped (already defined or filtered), Err on parse error.
126fn parse_and_load(line: &str) -> Result<bool, String> {
127  let parts: Vec<&str> = line.splitn(3, '\t').collect();
128  if parts.len() < 2 {
129    return Err("Too few fields".into());
130  }
131
132  let table = parts[0];
133  // Key decode: Cow borrows the original &str when no `%` escape is
134  // present (the overwhelming majority). Saves a per-line allocation
135  // for the ~25k dump entries that have plain CS-name keys.
136  let key_cow: std::borrow::Cow<'_, str> = if parts[1].contains('%') {
137    std::borrow::Cow::Owned(url_decode(parts[1]))
138  } else {
139    std::borrow::Cow::Borrowed(parts[1])
140  };
141  let key = key_cow.as_ref();
142  let data = if parts.len() > 2 { parts[2] } else { "" };
143
144  match table {
145    // V: Value entries (registers, fontdimen, font metadata).
146    // Add-only policy: only loads if key has no existing value.
147    "V" => load_value(key, data),
148    // M: Meaning entries (expandable definitions + primitive aliases).
149    //
150    // Current policy: only route `@`-internal entries (whose body does
151    // not reference the expl3 hook system) to `load_meaning`. Those
152    // include `@`-internal PA/MPA aliases, which DO get consumed via
153    // the PA arm of `load_meaning`. The @-internal gate is still the
154    // right safety fence: it keeps us out of the expl3 short-circuit
155    // hazard zone.
156    //
157    // `:`-style expl3 Meanings and public-CS PAs (like
158    // `\tex_let:D → \let`) remain gated off because enabling them
159    // causes two failure modes, both observed in earlier experiments:
160    //
161    //   - PA alone: `\tex_let:D` becomes let-aliased to `\let` via
162    //     the dump → `expl3.sty`'s own guard fires → raw
163    //     `\input expl3-code.tex` is skipped → post-guard code hits
164    //     `\__kernel_dependency_version_check:Nn`, `\ProcessOptions`,
165    //     `\keys_define:nn { sys }`, which are `:`-style macros we
166    //     don't load → undefined-CS recovery loop (60 s timeout,
167    //     memory climbing, SIGTERM-by-watchdog).
168    //
169    //   - PA + `:`-style M: the `:`-style bodies themselves trigger
170    //     the same pattern via cross-references.
171    //
172    // Both must be unblocked TOGETHER, in coordination with
173    // `expl3_sty.rs` short-circuiting its whole `load_definitions`
174    // when the dump already supplies expl3 state. See SYNC_STATUS
175    // D0 (d.5).
176    "M" => {
177      let name = key.trim_start_matches('\\');
178      let is_at_internal = name.contains('@') && !name.contains(':');
179      // Safe additional gate: public CharDef/Register entries (payload
180      // starts with `R\t…`). These set a character code or register value
181      // and never chain into expl3/hook machinery, so they can't trigger
182      // the cascade the expl3 short-circuit is guarding against. Allows
183      // plain-TeX math chardefs like `\ldotp`, `\cdotp`, `\intop` to load
184      // without opening the door to public Expandable bodies.
185      let is_public_register = data.starts_with("R\t");
186      if (is_at_internal || is_public_register)
187        && !data.contains("\\\\hook") && !data.contains("16:\\hook") {
188        load_meaning(key, data)
189      } else {
190        Ok(false)
191      }
192    },
193    // LC/UC: case-mapping codes — safe, always load
194    "LC" => load_lccode(key, data),
195    "UC" => load_uccode(key, data),
196    // SC: space factor codes — safe, always load
197    "SC" => load_sfcode(key, data),
198    // C: catcodes — only for non-ASCII (>127). ASCII catcodes are set by
199    // the engine; loading from dump would conflict.
200    "C" => {
201      let ch = decode_char_key(key);
202      if ch.is_some_and(|c| c as u32 > 127) {
203        load_catcode(key, data)
204      } else {
205        Ok(false)
206      }
207    }
208    // MC/DC: mathcodes and delcodes from the dump are corrupted by expl3
209    // format initialization (e.g., mathcode('v')=618 maps to '|').
210    // The engine sets correct math/delcodes. Skip.
211    "MC" | "DC" => Ok(false),
212    _ => Ok(false),
213  }
214}

216/// V entries to unconditionally skip (runtime state, never useful from dump).
217const SKIP_VALUE_KEYS: &[&str] = &[
218  "INTERPRETING_DEFINITIONS",
219  "if_count",
220  "absorb_count",
221  "if_stack",
222  "INCLUDE_COMMENTS",
223  "INCLUDE_STYLES",
224  "INPUT_ENCODING",
225  "CURRENT_INPUT_ENCODING",
226  "SUPPRESS_UNEXPECTED_ERRORS",
227  "SUPPRESS_UNDEFINED_ERRORS",
228  // Upstream Perl IGNORED_SYMBOLS (TeX_Job.pool.ltxml): runtime-only
229  // tables that re-populate from the engine — can't meaningfully round-
230  // trip through the dump.
231  "DOCUMENT_REWRITE_RULES",
232  "PARAMETER_TYPES",
233  "TAG_PROPERTIES",
234  "MATH_LIGATURES",
235  "TEXT_LIGATURES",
236];

238/// V entry key prefixes to skip.
239const SKIP_VALUE_PREFIXES: &[&str] = &[
240  "input_file:",
241  "output_file:",
242  "texsys",
243];

245/// V entry key substrings to skip.
246///
247/// Note: `_loaded` / `_found_loaded` flags are present in the dump (correctly,
248/// since `--init=latex.ltx` sees expl3-code.tex, hyphenation patterns, and
249/// hundreds of other raw-loaded files). But carrying them through into state
250/// at dump-load time blows up in practice:
251///
252///  - Hyphenation `loadhyph-*.tex_loaded` flags make subsequent raw-loading of
253///    babel's language.def skip files that set `\l@<lang>` registers our
254///    engine then discovers aren't present, triggering a flood of error
255///    recovery that can consume gigabytes of RAM.
256///  - `expl3.ltx_loaded=1` plus `expl3.sty_loaded=` NOT being set means
257///    `\usepackage{expl3}` doesn't short-circuit AT the .sty layer, but the
258///    raw .ltx re-load now enters a stranger code path with partial flags.
259///
260/// The proper fix, tracked as the "dump/_base mutual-exclusivity" item in
261/// SYNC_STATUS D0, is to have exactly ONE loading path (dump-cache or raw-load)
262/// active at a time, mirroring Perl's `LoadFormat` branching. Until that lands,
263/// keep the skip list conservative so mixed paths don't trigger recovery loops.
264const SKIP_VALUE_CONTAINS: &[&str] = &[
265  "_loaded",       // Package loading flags — see doc comment above
266  "_found_loaded", // Package found+loaded flags
267];

269/// Load a value entry: V\tKEY\tTYPE\tDATA
270///
271/// Uses add-only policy: only loads if the key does not already have a value.
272/// This ensures compiled engine state takes priority over dump state.
273fn load_value(key: &str, data: &str) -> Result<bool, String> {
274  // Skip unconditional keys
275  for skip in SKIP_VALUE_KEYS {
276    if key == *skip {
277      return Ok(false);
278    }
279  }
280  // Skip by prefix
281  for prefix in SKIP_VALUE_PREFIXES {
282    if key.starts_with(prefix) {
283      return Ok(false);
284    }
285  }
286  // Skip by substring.
287  for substr in SKIP_VALUE_CONTAINS {
288    if key.contains(substr) {
289      return Ok(false);
290    }
291  }
292
293  // Add-only policy: don't overwrite any existing value.
294  // This preserves engine-configured state (e.g., \everymath, \everypar,
295  // named skips, etc.) while filling in gaps from the dump (e.g., expl3
296  // fontdimen intarrays, font metadata).
297  if state::has_value(key) {
298    return Ok(false);
299  }
300
301  let parts: Vec<&str> = data.splitn(2, '\t').collect();
302  if parts.is_empty() {
303    return Err("Missing value type".into());
304  }
305
306  let value = match parts[0] {
307    "N" => return Ok(false), // Don't load None values (would erase existing)
308    "B" => Stored::Bool(parts.get(1).map(|s| *s == "1").unwrap_or(false)),
309    "I" => {
310      let n: i64 = parts
311        .get(1)
312        .unwrap_or(&"0")
313        .parse()
314        .map_err(|e| format!("Bad int: {}", e))?;
315      Stored::Int(n)
316    }
317    "S" => Stored::from(url_decode(parts.get(1).unwrap_or(&""))),
318    "CH" => {
319      let n: u16 = parts
320        .get(1)
321        .unwrap_or(&"0")
322        .parse()
323        .map_err(|e| format!("Bad charcode: {}", e))?;
324      Stored::Charcode(n)
325    }
326    "CC" => {
327      let n: u8 = parts
328        .get(1)
329        .unwrap_or(&"0")
330        .parse()
331        .map_err(|e| format!("Bad catcode: {}", e))?;
332      Stored::Catcode(Catcode::from(n))
333    }
334    "T" => {
335      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
336      Stored::Token(tok)
337    }
338    "TK" => {
339      let toks = parse_token_list(parts.get(1).unwrap_or(&""))?;
340      Stored::Tokens(Tokens::from(toks))
341    }
342    "D" => {
343      let n: i64 = parts.get(1).unwrap_or(&"0").parse()
344        .map_err(|e| format!("Bad dimension: {}", e))?;
345      Stored::Dimension(crate::common::dimension::Dimension(n))
346    }
347    "G" => {
348      Stored::Glue(parse_glue(parts.get(1).unwrap_or(&"0"))?)
349    }
350    "MD" => {
351      let n: i64 = parts.get(1).unwrap_or(&"0").parse()
352        .map_err(|e| format!("Bad mudimension: {}", e))?;
353      Stored::MuDimension(crate::common::mudimension::MuDimension(n))
354    }
355    "MG" => {
356      Stored::MuGlue(parse_muglue(parts.get(1).unwrap_or(&"0"))?)
357    }
358    "VD" => return Ok(false), // Don't load empty VecDeque (runtime state)
359    _ => return Ok(false),    // Unknown value type
360  };
361
362  state::assign_value(key, value, Some(Scope::Global));
363  Ok(true)
364}

366/// Load a meaning entry: M\tKEY\tTYPE\t...
367///
368/// Uses add-only policy: skip if the CS already has a meaning.
369/// Additionally, only loads "safe" definitions — those that won't interfere
370/// with the compiled engine's processing during normal LaTeX operation:
371/// - expl3 internals (contain `:`) — safe because `:` is OTHER under normal catcodes
372/// - Private LaTeX internals (contain `@`) — only invoked by other macros
373/// - Skip all "public" macros that could be invoked during normal expansion
374///   and might reference hooks/primitives not supported by our engine
375fn load_meaning(key: &str, data: &str) -> Result<bool, String> {
376  let cs_tok = Token {
377    text: arena::pin(key),
378    code: Catcode::CS,
379  };
380
381  // Add-only policy: don't override ANY existing definition.
382  if state::has_meaning(&cs_tok) {
383    return Ok(false);
384  }
385
386  // Safety filter: only load definitions that won't interfere with normal
387  // LaTeX processing. Public macros from the dump (like \document, \hook,
388  // \UseOneTimeHook) can reference expl3 hooks and internal state that
389  // our engine doesn't fully support, causing cascading errors.
390  //
391  // Safe: expl3 internals (with `:` or `__`), LaTeX internals (with `@`),
392  //       AND public Register/CharDef entries (they set char codes and
393  //       can't chain into expl3 cascades).
394  // Unsafe: public Expandable macros without `:` or `@` (e.g., \document,
395  //         \hook) — their bodies reference the hook system we don't
396  //         fully support. `_base.rs` + `_constructs.rs` already define
397  //         the public CSes the engine cares about; public-CS Expandable
398  //         dump entries are redundant.
399  let name = key.trim_start_matches('\\');
400  let is_internal = name.contains(':') || name.contains('@');
401  let is_public_register = data.starts_with("R\t");
402  if !is_internal && !is_public_register {
403    return Ok(false);
404  }
405
406  let parts: Vec<&str> = data.splitn(2, '\t').collect();
407  if parts.is_empty() {
408    return Err("Missing meaning type".into());
409  }
410
411  match parts[0] {
412    "N" => {
413      // None meaning — skip (don't define as undefined)
414      Ok(false)
415    }
416    "E" => {
417      // Expandable: E\tCSNAME\tNARGS\tFLAGS\tTOKENS[\tPROTO[\tV3_PARAMS]]
418      //
419      // Three historical shapes, read in fallback order:
420      //   v3 — 6th field present: structured Parameter records; bypasses
421      //        parse_parameters entirely. Only format that round-trips
422      //        Until:/Match: with catcoded delimiter tokens intact.
423      //   v2 — 5th field present: url-decoded prototype string fed to
424      //        parse_parameters. Good for {} / [] / DefToken / simple
425      //        typed params; loses brace-in-delimiter forms.
426      //   v1 — nargs only: "{}".repeat(nargs), all params flattened to
427      //        Plain. Kept as last resort so ancient dumps still load.
428      let eparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(6, '\t').collect();
429      if eparts.len() < 4 {
430        return Err("Incomplete Expandable entry".into());
431      }
432
433      // Note: eparts[0] is the internal CS name carried by the E
434      // serialization (the Expandable's `self.cs`), which for non-alias
435      // entries matches `key`. We don't decode it here — the outer key
436      // is already parsed above into `cs_tok`, and `Expandable::new`
437      // doesn't need a distinct internal name.
438      let nargs: usize = eparts[1].parse().unwrap_or(0);
439      let flags = eparts[2];
440      let tok_data = eparts[3];
441      let proto_opt = eparts.get(4).map(|s| url_decode(s)).filter(|s| !s.is_empty());
442      let v3_opt = eparts.get(5).filter(|s| !s.is_empty());
443
444      let is_long = flags.contains('L');
445      let is_protected = flags.contains('P');
446
447      let expansion = parse_token_list(tok_data)?;
448
449      // Build parameter spec, preferring v3 structured → v2 proto →
450      // v1 nargs-repeat fallback. init_flag=true for both fallbacks:
451      // state is live at runtime so Parameter::init() can resolve
452      // readers via PARAMETER_TYPES.
453      let paramlist = if let Some(v3) = v3_opt {
454        match parse_parameters_v3(v3) {
455          Ok(pl) => pl,
456          Err(_) => proto_opt
457            .as_ref()
458            .and_then(|p| crate::common::def_parser::parse_parameters(p, &cs_tok, true).ok())
459            .flatten()
460            .or_else(|| {
461              if nargs > 0 {
462                let fallback = "{}".repeat(nargs);
463                crate::common::def_parser::parse_parameters(&fallback, &cs_tok, true)
464                  .ok()
465                  .flatten()
466              } else {
467                None
468              }
469            }),
470        }
471      } else {
472        // v2 path: no v3 field, fall back to proto-parsing (with the
473        // original silent-degrade-to-nargs behavior).
474        match proto_opt {
475          Some(proto) => match crate::common::def_parser::parse_parameters(&proto, &cs_tok, true) {
476            Ok(pl) => pl,
477            Err(_) if nargs > 0 => {
478              let fallback = "{}".repeat(nargs);
479              crate::common::def_parser::parse_parameters(&fallback, &cs_tok, true)
480                .map_err(|e| format!("Param parse fallback: {}", e))?
481            }
482            Err(_) => None,
483          },
484          None if nargs > 0 => {
485            let proto = "{}".repeat(nargs);
486            crate::common::def_parser::parse_parameters(&proto, &cs_tok, true)
487              .map_err(|e| format!("Param parse: {}", e))?
488          }
489          None => None,
490        }
491      };
492
493      let options = Some(ExpandableOptions {
494        long: is_long,
495        protected: is_protected,
496        nopack_parameters: true, // tokens already have ARG catcode
497        ..ExpandableOptions::default()
498      });
499
500      let expansion_body = Tokens::from(expansion).into();
501      match Expandable::new(cs_tok, paramlist, Some(expansion_body), options) {
502        Ok(mut exp) => {
503          // Perl #aaacdba2: stamp dump-loaded definitions with a
504          // nominal Locator pointing at the dump file + line. Helps
505          // diagnostics attribute errors to the dump source rather
506          // than the arena's compile-site default.
507          exp.locator = current_dump_locator();
508          state::install_definition(exp, Some(Scope::Global));
509          Ok(true)
510        }
511        Err(e) => Err(format!("Expandable creation failed: {}", e)),
512      }
513    }
514    "T" => {
515      // Token meaning (let-assignment)
516      let tok = parse_token(parts.get(1).unwrap_or(&""))?;
517      state::assign_meaning(&cs_tok, tok, Some(Scope::Global));
518      Ok(true)
519    }
520    "PA" | "MPA" => {
521      // Primitive alias: PA\t<target_cs> — the entry's meaning is an
522      // Rc<Primitive> whose own cs is <target_cs>. If <target_cs> == key
523      // this is the "primary" entry (already provided by compiled bindings
524      // in _base.rs etc.); skip. Otherwise, replay `\let <key> <target>`
525      // so the Rc<Primitive> is shared just as it was when the dump was
526      // generated. This is how \tex_let:D, \tex_def:D, etc. survive the
527      // dump — without this the expl3.sty short-circuit guard
528      // `\ifx\csname tex_let:D\endcsname\relax` never fires and the 36k
529      // lines of expl3-code.tex get reprocessed on every run.
530      let target_cs_raw = url_decode(parts.get(1).unwrap_or(&""));
531      if target_cs_raw == key {
532        return Ok(false);
533      }
534      let target_tok = Token {
535        text: arena::pin(&target_cs_raw),
536        code: Catcode::CS,
537      };
538      // Skip silently if the target isn't in bindings — rare and means
539      // the alias points at a CS only defined during raw-load that we
540      // also lost. The engine's undefined-CS handler will cope at runtime.
541      if !state::has_meaning(&target_tok) {
542        return Ok(false);
543      }
544      state::let_i(&cs_tok, &target_tok, Some(Scope::Global));
545      Ok(true)
546    }
547    "R" => {
548      // Register: R\tCS\tTYPE\tVALUE[\tMATHGLYPH]
549      // rparts[0] (internal CS name) is redundant with the outer key —
550      // same reasoning as the E arm; we skip the decode + alloc.
551      let rparts: Vec<&str> = parts.get(1).unwrap_or(&"").splitn(4, '\t').collect();
552      if rparts.len() < 3 {
553        return Err("Incomplete Register entry".into());
554      }
555      let rtype = rparts[1];
556      let value_str = rparts[2];
557      let mathglyph = rparts.get(3).and_then(|s| s.parse::<u32>().ok()).and_then(char::from_u32);
558
559      use crate::definition::register::{Register, RegisterType, RegisterValue};
560      use crate::common::number::Number;
561
562      let (reg_type, reg_value) = match rtype {
563        "N" | "CD" => {
564          let n: i64 = value_str.parse().unwrap_or(0);
565          let rt = if rtype == "CD" { RegisterType::CharDef } else { RegisterType::Number };
566          (rt, Some(RegisterValue::Number(Number::new(n))))
567        }
568        "D" => {
569          let n: i64 = value_str.parse().unwrap_or(0);
570          (RegisterType::Dimension, Some(RegisterValue::Dimension(
571            crate::common::dimension::Dimension(n))))
572        }
573        "G" => {
574          (RegisterType::Glue, Some(RegisterValue::Glue(
575            parse_glue(value_str)?)))
576        }
577        "MG" => {
578          (RegisterType::MuGlue, Some(RegisterValue::MuGlue(
579            parse_muglue(value_str)?)))
580        }
581        "TK" => {
582          // Token register: value is comma-separated token list, or "0" for empty
583          let toks = if value_str == "0" || value_str.is_empty() {
584            Vec::new()
585          } else {
586            parse_token_list(value_str)?
587          };
588          (RegisterType::Tokens, Some(RegisterValue::Tokens(
589            Tokens::from(toks))))
590        }
591        _ => return Ok(false),
592      };
593
594      // Perl-parity with def_register (binding/def/dialect.rs): store the
595      // initial value at the Register's `address` slot AND set `default`, so
596      // a subsequent `value_of` lookup — which reads state::with_value(address)
597      // and falls back to `default` — actually sees the dump's initial value.
598      // CharDefs read their immediate `value` field instead, so we skip the
599      // storage write for them.
600      let mut reg = Register {
601        cs: cs_tok,
602        register_type: reg_type,
603        value: reg_value.clone(),
604        default: if matches!(reg_type, RegisterType::CharDef) { None } else { reg_value.clone() },
605        mathglyph,
606        locator: current_dump_locator(),
607        ..Register::default()
608      };
609      // Set address from CS name
610      reg.address = key.to_string();
611      if !matches!(reg_type, RegisterType::CharDef) {
612        if let Some(ref rv) = reg_value {
613          state::assign_value(&reg.address, rv.clone(), Some(Scope::Global));
614        }
615      }
616      state::install_definition(reg, Some(Scope::Global));
617      Ok(true)
618    }
619    _ => Ok(false),
620  }
621}

623/// Decode a character key from the dump. Handles:
624/// - Single characters: "A", "è"
625/// - URL-encoded control chars: "%19" (→ char 0x19), "%0A" (→ char 0x0A)
626fn decode_char_key(key: &str) -> Option<char> {
627  let decoded = url_decode(key);
628  decoded.chars().next()
629}

631/// Load a catcode entry: C\tCHAR\tCC\tVALUE
632fn load_catcode(key: &str, data: &str) -> Result<bool, String> {
633  let ch = decode_char_key(key).ok_or_else(|| format!("Bad catcode char: {}", key))?;
634  let parts: Vec<&str> = data.splitn(2, '\t').collect();
635  if parts.len() < 2 || parts[0] != "CC" {
636    return Err(format!("Bad catcode data: {}", data));
637  }
638  let cc: u8 = parts[1].parse().map_err(|e| format!("Bad catcode value: {}", e))?;
639  state::assign_catcode(ch, Catcode::from(cc), Some(Scope::Global));
640  Ok(true)
641}

680// "MC"/"DC" arm in parse_entry returns Ok(false) because the dumped values

682// eventually harvest clean delcode/mathcode data, restore them from git

685/// Parse a single token from "CC:TEXT" format
686fn parse_token(s: &str) -> Result<Token, String> {
687  let (cc_str, text) = s.split_once(':').ok_or("Missing ':' in token")?;
688  let cc: u8 = cc_str.parse().map_err(|e| format!("Bad CC: {}", e))?;
689  // Fast path: most token text fields have no `%` escapes, so pin the
690  // &str directly — avoids the String allocation url_decode would make
691  // even on its own fast path. Parsing the expl3 kernel alone produces
692  // hundreds of thousands of token entries; every `to_owned()` avoided
693  // here matters.
694  let text_sym = if text.contains('%') {
695    arena::pin(url_decode(text))
696  } else {
697    arena::pin(text)
698  };
699  Ok(Token {
700    text: text_sym,
701    code: Catcode::from(cc),
702  })
703}
704
705/// Parse comma-separated token list
706fn parse_token_list(s: &str) -> Result<Vec<Token>, String> {
707  if s.is_empty() {
708    return Ok(Vec::new());
709  }
710  s.split(',').map(parse_token).collect()
711}

713/// Decode the v3 structured Parameters encoding emitted by
714/// `dump_writer::serialize_parameters_v3` (see that function's docstring
715/// and `docs/DUMP_FORMAT_PERL_ANALYSIS.md` for the layout).
716///
717/// Returns `Ok(None)` for an empty record (no parameters); `Ok(Some(ps))`
718/// on success; `Err` if any record is malformed. Each Parameter is
719/// constructed via `Parameter::new(name, spec, Some(extras))`, which
720/// calls `init()` — the reader function is resolved against the live
721/// PARAMETER_TYPES table, mirroring the runtime path.
722pub(crate) fn parse_parameters_v3(v3: &str) -> Result<Option<crate::parameter::Parameters>, String> {
723  if v3.is_empty() {
724    return Ok(None);
725  }
726  let mut params = Vec::new();
727  for record in v3.split('\x1e') {
728    // <name>\x1f<spec>\x1f<flags>\x1f<extras>
729    let fields: Vec<&str> = record.splitn(4, '\x1f').collect();
730    if fields.len() != 4 {
731      return Err(format!("v3 Parameter record has {} fields, expected 4", fields.len()));
732    }
733    let name = url_decode(fields[0]);
734    let spec = url_decode(fields[1]);
735    let flags = fields[2];
736    let extras_str = fields[3];
737
738    let extras = if extras_str.is_empty() {
739      Vec::new()
740    } else {
741      extras_str
742        .split('\x1d')
743        .map(|tok_list| {
744          parse_token_list(tok_list)
745            .map(crate::tokens::Tokens::new)
746        })
747        .collect::<Result<Vec<_>, _>>()?
748    };
749
750    let mut param = crate::parameter::Parameter::new(name, spec, Some(extras))
751      .map_err(|e| format!("Parameter::new failed: {}", e))?;
752
753    // Apply flags after construction — Parameter::new + init() handle
754    // the reader side, but novalue/optional are struct-level booleans
755    // that the spec-driven init() may or may not have set (e.g. "Optional"
756    // prefix auto-sets optional, but we want explicit round-trip).
757    for flag in flags.split(';').filter(|s| !s.is_empty()) {
758      match flag {
759        "n=1" => param.novalue = true,
760        "o=1" => param.optional = true,
761        _ => {
762          // Unknown flag — ignore for forward compat; future flags
763          // added to the writer shouldn't break older readers.
764        }
765      }
766    }
767    params.push(param);
768  }
769  if params.is_empty() {
770    Ok(None)
771  } else {
772    Ok(Some(crate::parameter::Parameters::new(params)))
773  }
774}

776pub(crate) fn url_decode(s: &str) -> String {
777  // Fast path: the overwhelming majority of dump entries have no `%`
778  // escapes in their key or proto fields, so a single memcpy via
779  // `to_owned()` beats char-by-char iteration for ~19k key loads.
780  if !s.contains('%') {
781    return s.to_owned();
782  }
783  let mut result = String::with_capacity(s.len());
784  let mut chars = s.chars();
785  while let Some(ch) = chars.next() {
786    if ch == '%' {
787      let hex: String = chars.by_ref().take(2).collect();
788      if let Ok(byte) = u8::from_str_radix(&hex, 16) {
789        result.push(byte as char);
790      }
791    } else {
792      result.push(ch);
793    }
794  }
795  result
796}

798/// Parse a serialized Glue value: "skip,pN,pfN,mN,mfN"
799fn parse_glue(s: &str) -> Result<crate::common::glue::Glue, String> {
800  use crate::common::glue::{FillCode, Glue};
801  let mut skip = 0i64;
802  let mut plus = None;
803  let mut pfill = None;
804  let mut minus = None;
805  let mut mfill = None;
806  for (i, part) in s.split(',').enumerate() {
807    if i == 0 {
808      skip = part.parse().map_err(|e| format!("Bad glue skip: {}", e))?;
809    } else if let Some(rest) = part.strip_prefix("pf") {
810      pfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
811    } else if let Some(rest) = part.strip_prefix('p') {
812      plus = Some(rest.parse().map_err(|e| format!("Bad glue plus: {}", e))?);
813    } else if let Some(rest) = part.strip_prefix("mf") {
814      mfill = FillCode::new(rest.parse::<usize>().unwrap_or(0));
815    } else if let Some(rest) = part.strip_prefix('m') {
816      minus = Some(rest.parse().map_err(|e| format!("Bad glue minus: {}", e))?);
817    }
818  }
819  Ok(Glue { skip, plus, pfill, minus, mfill })
820}
