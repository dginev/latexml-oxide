39impl Display for ArgWrap {
40  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
41    match self {
42      ArgWrap::Token(t) => write!(f, "{t}"),
43      ArgWrap::Tokens(ts) => write!(f, "{ts}"),
44      ArgWrap::Number(n) => write!(f, "{n}"),
45      ArgWrap::Float(fl) => write!(f, "{fl}"),
46      ArgWrap::Dimension(d) => write!(f, "{d}"),
47      ArgWrap::Glue(gl) => write!(f, "{gl}"),
48      ArgWrap::MuGlue(mugl) => write!(f, "{mugl}"),
49      ArgWrap::MuDimension(mudim) => write!(f, "{mudim}"),
50      ArgWrap::KV(kv) => write!(f, "{kv}"),
51      ArgWrap::AlignmentTemplate(at) => write!(f, "{at}"),
52      ArgWrap::Pair(p) => write!(f, "{p}"),
53      ArgWrap::RegisterDefinition(dbox) => write!(f, "({},{:?})", dbox.0, dbox.1),
54      ArgWrap::None => write!(f, "None"),
55    }
56  }
57}

59impl Object for ArgWrap {
60  fn get_locator(&self) -> Option<Locator> {
61    // Most argument wrappers carry no intrinsic source locator; a few delegate
62    // to their contained value. (Patterns qualified with `ArgWrap::` so the
63    // body's `None` is unambiguously `Option::None`, not the `ArgWrap::None`
64    // variant.)
65    match self {
66      ArgWrap::Glue(t) => t.get_locator(),
67      ArgWrap::MuGlue(t) => t.get_locator(),
68      ArgWrap::MuDimension(t) => t.get_locator(),
69      ArgWrap::KV(kv) => kv.get_locator(),
70      _ => None,
71    }
72  }
73  fn be_digested(self) -> Result<Digested> {
74    use ArgWrap::*;
75    // TODO: Should we just "do nothing" for the None cases, instead of panicking?
76    match self {
77      Token(t) => t.be_digested(),
78      Tokens(t) => t.be_digested(),
79      Number(t) => t.be_digested(),
80      Float(t) => t.be_digested(),
81      Dimension(t) => t.be_digested(),
82      Glue(t) => t.be_digested(),
83      MuGlue(t) => t.be_digested(),
84      MuDimension(t) => t.be_digested(),
85      KV(kv) => kv.be_digested(),
86      Pair(p) => p.be_digested(),
87      None => Ok(Digested::default()),
88      AlignmentTemplate(_) => Ok(Digested::default()), // templates don't digest directly
89      RegisterDefinition(_) => Ok(Digested::default()), // register defs don't digest directly
90    }
91  }
92  fn revert(&self) -> Result<Tokens> {
93    use ArgWrap::*;
94    match self {
95      Token(t) => Ok(Tokens!(*t)),
96      Tokens(t) => Ok(t.clone()),
97      Number(t) => t.revert(),
98      Float(t) => t.revert(),
99      Dimension(t) => t.revert(),
100      Glue(t) => t.revert(),
101      MuGlue(t) => t.revert(),
102      MuDimension(t) => t.revert(),
103      KV(kv) => kv.revert(),
104      Pair(p) => p.revert(),
105      None => Ok(Tokens!()),
106      AlignmentTemplate(_) => Ok(Tokens!()), // templates don't revert directly
107      RegisterDefinition(_) => Ok(Tokens!()), // register defs don't revert directly
108    }
109  }
110}

117  /// Zero-alloc equivalent of `self.to_string() == target` for keyword
118  /// checks in DefMacro bodies. Forwards to `Token::with_str` /
119  /// `Tokens::eq_text` for the Token/Tokens variants (covering ~all
120  /// keyword-argument checks in practice); for other variants falls
121  /// back to `to_string() == target` since the Display impl allocates
122  /// anyway. See `Tokens::eq_text` for the walk semantics.
123  pub fn eq_text(&self, target: &str) -> bool {
124    match self {
125      ArgWrap::Tokens(tks) => tks.eq_text(target),
126      ArgWrap::Token(t) => t.with_str(|s| s == target),
127      other => other.to_string() == target,
128    }
129  }
130
131  /// Zero-alloc `self.to_string().starts_with(prefix)` for Tokens/Token
132  /// variants; falls back to full to_string for others.
133  pub fn starts_with_text(&self, prefix: &str) -> bool {
134    match self {
135      ArgWrap::Tokens(tks) => tks.starts_with_text(prefix),
136      ArgWrap::Token(t) => t.with_str(|s| s.starts_with(prefix)),
137      other => other.to_string().starts_with(prefix),
138    }
139  }

146  pub fn owned_tokens(self) -> Option<Tokens> {
147    match self {

159  pub fn try_to_token(self) -> Result<Token> {
160    match self {
161      ArgWrap::Token(t) => Ok(t),
162      ArgWrap::Tokens(tks) => {
163        let mut list = tks.unlist();
164        if list.is_empty() {
165          Err("try_to_token: empty Tokens".into())
166        } else {
167          Ok(list.remove(0))
168        }
169      },
170      _ => Err(
171        s!(
172          "Hard assumption for Token argument failed. Got instead: {:?}",
173          self
174        )
175        .into(),
176      ),
177    }
178  }

205  pub fn as_tokens(&self) -> Result<Option<Cow<'_, Tokens>>> {
206    use ArgWrap::*;
207    let result = match self {
208      Token(t) => Some(Cow::Owned(Tokens!(*t))),
209      Tokens(tks) => Some(Cow::Borrowed(tks)),
210      Number(_) | Float(_) | Dimension(_) | Glue(_) | MuGlue(_) | MuDimension(_) | KV(_)
211      | Pair(_) => Some(Cow::Owned(self.revert()?)),
212      None => Some(Cow::Borrowed(NO_BORROWED_TOKENS)),
213      AlignmentTemplate(_) | RegisterDefinition(_) => Some(Cow::Owned(Tokens!())),
214    };
215    Ok(result)
216  }

403  /// Borrow the token slice backing an ArgWrap::Tokens, falling back

448impl From<Stored> for Result<ArgWrap> {
449  // A maintenance detail here is that whenever a Parameter can read a new concrete type of data,
450  // then ArgWrap will have a new variant, which must then be possible to store in Stored,
451  // and possible to cast back from storage into ArgWrap
452  fn from(t: Stored) -> Result<ArgWrap> {
453    Ok(match t {
454      Stored::Tokens(t) => ArgWrap::Tokens(t),
455      Stored::Token(t) => ArgWrap::Token(t),
456      Stored::MuDimension(d) => ArgWrap::MuDimension(d),
457      Stored::Glue(g) => ArgWrap::Glue(g),
458      Stored::MuGlue(g) => ArgWrap::MuGlue(g),
459      Stored::Number(n) => ArgWrap::Number(n),
460      Stored::Float(f) => ArgWrap::Float(f),
461      Stored::Dimension(d) => ArgWrap::Dimension(d),
462      // we could just map "_" to None, but it is safer to enumerate, to avoid missing
463      // meaningful cases.
464      Stored::None => ArgWrap::None,
465      Stored::Mouth(_)
466      | Stored::Primitive(_)
467      | Stored::Bool(_)
468      | Stored::Parameter(_)
469      | Stored::MathPrimitive(_)
470      | Stored::Conditional(_)
471      | Stored::Constructor(_)
472      | Stored::Charcode(_)
473      | Stored::Expandable(_)
474      | Stored::Ligature(_)
475      | Stored::HashStored(_)
476      | Stored::HashTagData(_)
477      | Stored::FrontmatterRaw(_)
478      | Stored::HashString(_)
479      | Stored::Digested(_)
480      | Stored::FontDirective(_)
481      | Stored::Register(_)
482      | Stored::Rewrite(_)
483      | Stored::Stash(_)
484      | Stored::Fontmap(_)
485      | Stored::Int(_)
486      | Stored::String(_)
487      | Stored::Strings(_)
488      | Stored::Node(_)
489      | Stored::IfFrame(_)
490      | Stored::Font(_)
491      | Stored::Reversion(_)
492      | Stored::Catcode(_)
493      | Stored::Locator(_)
494      | Stored::VecDequeStored(_)
495      | Stored::VecDigested(_)
496      | Stored::Chars(_)
497      | Stored::KeyVal(_)
498      | Stored::KeyVals(_)
499      | Stored::Template(_) => {
500        Error!(
501          "stored",
502          "type",
503          "Failed to cast to argument; Found stored:",
504          t
505        );
506        ArgWrap::None
507      },
508    })
509  }

624}

628  fn try_from(aw: ArgWrap) -> Result<KeyVals> { aw.try_to_keyvals() }
629}

631impl From<ArgWrap> for Option<KeyVals> {
632  fn from(aw: ArgWrap) -> Option<KeyVals> {
633    match aw {
634      ArgWrap::KV(kv) => Some(*kv),
635      _ => None,
636    }
637  }
638}

640impl From<ArgWrap> for Template {
641  fn from(aw: ArgWrap) -> Template {
642    match aw {
643      ArgWrap::AlignmentTemplate(t) => *t,
644      other => panic!("illegal auto-cast to alignment::Template on {other:?}"),
645    }
646  }
647}
