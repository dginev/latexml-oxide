87  fn invoke_primitive(&self) -> Result<Vec<Digested>> {
88    Debug!("primitive invoke for {:?}", self.cs);
89    // my $profiled = $state->lookupValue('PROFILING') && ($LaTeXML::CURRENT_TOKEN || $$self{cs});
90    // my $tracing = $state->lookupValue('tracingcommands');
91    // LaTeXML::Core::Definition::startProfiling($profiled, 'digest') if $profiled;
92    // print STDERR '{' . $self->tracingCSName . "}\n" if $tracing;
93    let mut invoked_boxes: Vec<Digested> = self.execute_before_digest()?;
94    let args = self.read_arguments()?;
95    // print STDERR $self->tracingArgs(@args) . "\n" if $tracing && @args;
96    match self.replacement {
97      Some(PrimitiveBody::Closure(ref closure)) => invoked_boxes.extend(closure(args)?),
98      Some(PrimitiveBody::String(symbol)) => {
99        let cs_token = self
100          .alias
101          .as_ref()
102          .map(|alias| T_CS!(alias))
103          .unwrap_or(self.cs);
104        let box_tokens = vec![cs_token];
105        if let Some(ref _params) = self.paramlist {
106          todo!(); // we need to generalize the revert functions to take ArgWrap-typed arguments
107          // box_tokens.extend(params.revert_arguments(args)?);
108        }
109        let box_props = SymHashMap::default();
110        invoked_boxes.push(Digested::from(Tbox::new(
111          symbol,
112          None,
113          None,
114          Tokens::new(box_tokens),
115          box_props,
116        )));
117      },
118      None => {
119        // Perl: Box(undef, undef, undef, Tokens($self->getCSorAlias, ...), isEmpty => 1)
120        // Even with no replacement, Perl creates a Box with the CS as reversion and isEmpty flag.
121        // This is essential for font switches (\rm, \it, etc.) to appear in tex attributes.
122        let cs_token = self
123          .alias
124          .as_ref()
125          .map(|alias| T_CS!(alias))
126          .unwrap_or(self.cs);
127        let box_tokens = vec![cs_token];
128        // TODO: add revert_arguments for ArgWrap type when needed
129        let mut box_props = SymHashMap::default();
130        box_props.insert("isEmpty", Stored::Bool(true));
131        invoked_boxes.push(Digested::from(Tbox::new(
132          *EMPTY_SYM,
133          None,
134          None,
135          Tokens::new(box_tokens),
136          box_props,
137        )));
138      },
139    }
140    if !self.after_digest.is_empty() {
141      // optimize to avoid needless generation of whatsits
142      let mut w = Whatsit::default();
143      let after_boxes = self.execute_after_digest(&mut w)?;
144      invoked_boxes.extend(after_boxes);
145    }
146
147    // LaTeXML::Core::Definition::stopProfiling($profiled, 'digest') if $profiled;
148    Ok(invoked_boxes)
149  }
