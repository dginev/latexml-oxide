95  /// Expand the expandable control sequence. This should be carried out by the Gullet.
96  fn invoke(&self, once_only: bool) -> Result<Tokens> {
97    // shortcut for "trivial" macros; but only if not tracing & profiling!!!!
98    let _tracing = lookup_int("tracingmacros") > 0;
99    let _profiled = lookup_bool("PROFILING");
100    match &self.expansion {
101      Some(ExpansionBody::Closure(closure)) => {
102        // Harder to emulate \tracingmacros here.
103        let args = if let Some(ref parms) = self.paramlist {
104          parms.read_arguments(Some(self))?
105        } else {
106          Vec::new()
107        };
108        // Profiling: not implemented (Perl: startProfiling($profiled, 'expand'))
109        let result = closure(args)?;
110        // Tracing: Perl prints tracingCSName ==> tracetoString(result)
111        // Not implemented — silently skip to avoid panic on \tracingmacros=1
112        Ok(result)
113      },
114      Some(ExpansionBody::Tokens(tokens)) => {
115        let result = if self.paramlist.is_none() {
116          // Case: Trivial macro
117          // Profiling: not implemented (Perl: startProfiling($profiled, 'expand'))
118          // Tracing: Perl prints tracingCSName -> tracetoString(expansion)
119          // Not implemented — silently skip to avoid panic on \tracingmacros=1
120          // For trivial expansion, make sure we don't get \cs or \relax\cs direct recursion!
121          let is_recursion = if !once_only {
122            let token_vec = tokens.unlist_ref();
123            let t0_opt = token_vec.first();
124            let t1_opt = token_vec.get(1);
125            if let Some(t0) = t0_opt {
126              if t0 == &self.cs {
127                true
128              } else if let Some(t1) = t1_opt {
129                t1 == &self.cs && t0 == &T_CS!("\\protect")
130              } else {
131                false
132              }
133            } else {
134              false
135            }
136          } else {
137            false
138          };
139          if is_recursion {
140            // TODO: port the Error
141            // if (!$onceonly && $$self{cs}) {
142            //   my ($t0, $t1) = ($etype eq 'LaTeXML::Core::Tokens'
143            //     ? ($$expansion[0], $$expansion[1]) : ($expansion, undef));
144            //   if ($t0 && ($t0->equals($$self{cs})
145            //       || ($t1 && $t1->equals($$self{cs}) && $t0->equals(T_CS('\protect'))))) {
146            Error!(
147              "recursion",
148              self.cs,
149              format!("Token {} expands into itself!", self.cs),
150              "defining as empty"
151            );
152            //     $expansion = TokensI(); } }
153            Tokens!()
154          } else {
155            tokens.clone()
156          }
157        } else {
158          let args = if let Some(ref parms) = self.paramlist {
159            parms.read_arguments(Some(self))?
160          } else {
161            Vec::new()
162          };
163          if self.has_cc_arg {
164            // Do we actually need to substitute the args in?
165            let mut args_tks = Vec::new();
166            for arg in args.iter() {
167              args_tks.push(arg.as_tokens()?);
168            }
169            tokens.substitute_parameters(args_tks.as_slice())
170          } else {
171            tokens.clone()
172          }
173        };
174        // Profiling: Perl appends T_MARKER(profiled) for exclusive profiling
175        // Not implemented — silently skip
176        Ok(result)
177      },
178      None => {
179        // we always need to read the arguments, for e.g. things like \@gobble
180        if let Some(ref parms) = self.paramlist {
181          parms.read_arguments(Some(self))?;
182        }
183        Ok(NO_TOKENS)
184      },
185    }
186  }

201impl Expandable {
202  pub fn new(
203    cs: Token,
204    paramlist: Option<Parameters>,
205    mut expansion_opt: Option<ExpansionBody>,
206    traits: Option<ExpandableOptions>,
207  ) -> Result<Self> {
208    let traits = traits.unwrap_or_default();
209    if !traits.nopack_parameters {
210      if let Some(ExpansionBody::Tokens(expansion_tokens)) = expansion_opt {
211        // Perl: Fatal if expansion is unbalanced (mismatched {/})
212        if !expansion_tokens.is_balanced() {
213          Error!(
214            "misdefined",
215            cs,
216            s!("Expansion of '{}' has unbalanced {{}}", cs),
217            "skipping pack_parameters"
218          );
219          // Store as-is without packing
220          expansion_opt = Some(ExpansionBody::Tokens(expansion_tokens));
221        } else {
222          expansion_opt = Some(ExpansionBody::Tokens(expansion_tokens.pack_parameters()?));
223        }
224      }
225    }
226    let has_cc_arg = match expansion_opt {
227      Some(ExpansionBody::Tokens(ref tks)) => tks
228        .unlist_ref()
229        .iter()
230        .any(|t| t.get_catcode() == Catcode::ARG),
231      _ => false,
232    };
233    // simplify: treat empty tokens as None
234    let expansion = match expansion_opt {
235      Some(ExpansionBody::Tokens(tks)) if tks.is_empty() => None,
236      real_body => real_body,
237    };
238
239    Ok(Expandable {
240      cs,
241      paramlist,
242      expansion,
243      // locator           => $source->getLocator,
244      is_protected: traits.protected || get_prefix("protected"),
245      is_outer: traits.outer || get_prefix("outer"),
246      is_long: traits.long || get_prefix("long"),
247      has_cc_arg,
248      ..Expandable::default()
249    })
250  }
