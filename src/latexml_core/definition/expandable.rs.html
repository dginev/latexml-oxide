101  /// Expand the expandable control sequence. This should be carried out by the Gullet.
102  fn invoke(&self, once_only: bool) -> Result<Tokens> {
103    // Perl shortcut for "trivial" macros that were tracing- or
104    // profiling-aware. Neither tracing nor profiling is implemented
105    // in the Rust port (the returned `_tracing` / `_profiled` values
106    // were discarded), so the two state lookups were pure overhead
107    // on every macro expansion (\~350k calls in si.tex alone per
108    // callgrind). Removed — re-introduce only alongside the actual
109    // tracing/profiling features if/when they land.
110    match &self.expansion {
111      Some(ExpansionBody::Closure(closure)) => {
112        // Harder to emulate \tracingmacros here.
113        let args = if let Some(ref parms) = self.paramlist {
114          parms.read_arguments(Some(self))?
115        } else {
116          Vec::new()
117        };
118        // Profiling: not implemented (Perl: startProfiling($profiled, 'expand'))
119        let result = closure(args)?;
120        // Tracing: Perl prints tracingCSName ==> tracetoString(result)
121        // Not implemented — silently skip to avoid panic on \tracingmacros=1
122        Ok(result)
123      },
124      Some(ExpansionBody::Tokens(tokens)) => {
125        let result = if self.paramlist.is_none() {
126          // Case: Trivial macro
127          // Profiling: not implemented (Perl: startProfiling($profiled, 'expand'))
128          // Tracing: Perl prints tracingCSName -> tracetoString(expansion)
129          // Not implemented — silently skip to avoid panic on \tracingmacros=1
130          // For trivial expansion, make sure we don't get \cs or \relax\cs direct recursion!
131          let is_recursion = if !once_only {
132            let token_vec = tokens.unlist_ref();
133            let t0_opt = token_vec.first();
134            let t1_opt = token_vec.get(1);
135            if let Some(t0) = t0_opt {
136              if t0 == &self.cs {
137                true
138              } else if let Some(t1) = t1_opt {
139                t1 == &self.cs && t0 == &T_CS!("\\protect")
140              } else {
141                false
142              }
143            } else {
144              false
145            }
146          } else {
147            false
148          };
149          if is_recursion {
150            // Self-referential macros like \pgfkeys@mainstop are used as sentinels
151            // in PGF. They should never be expanded — only compared via \ifx.
152            // When detected, return the token itself (not empty) so \ifx can still match.
153            // Perl: $expansion = TokensI() [empty], but we return self to preserve \ifx semantics.
154            // Note: returning self would cause re-expansion in normal contexts, but
155            // the recursion check prevents infinite loops (returns empty on second hit).
156            Error!(
157              "recursion",
158              self.cs,
159              format!("Token {} expands into itself!", self.cs),
160              "defining as empty"
161            );
162            Tokens!()
163          } else {
164            tokens.clone()
165          }
166        } else {
167          let args = if let Some(ref parms) = self.paramlist {
168            parms.read_arguments(Some(self))?
169          } else {
170            Vec::new()
171          };
172          if self.has_cc_arg {
173            // Do we actually need to substitute the args in?
174            // Pre-size: one entry per argument; avoids Vec doublings on
175            // macros with many args.
176            let mut args_tks = Vec::with_capacity(args.len());
177            for arg in args.iter() {
178              args_tks.push(arg.as_tokens()?);
179            }
180            tokens.substitute_parameters(args_tks.as_slice())
181          } else {
182            tokens.clone()
183          }
184        };
185        // Profiling: Perl appends T_MARKER(profiled) for exclusive profiling
186        // Not implemented — silently skip
187        Ok(result)
188      },
189      None => {
190        // we always need to read the arguments, for e.g. things like \@gobble
191        if let Some(ref parms) = self.paramlist {
192          parms.read_arguments(Some(self))?;
193        }
194        Ok(NO_TOKENS)
195      },
196    }
197  }

212impl Expandable {
213  pub fn new(
214    cs: Token,
215    paramlist: Option<Parameters>,
216    mut expansion_opt: Option<ExpansionBody>,
217    traits: Option<ExpandableOptions>,
218  ) -> Result<Self> {
219    let traits = traits.unwrap_or_default();
220    if !traits.nopack_parameters {
221      if let Some(ExpansionBody::Tokens(expansion_tokens)) = expansion_opt {
222        // Perl: Fatal if expansion is unbalanced (mismatched {/})
223        if !expansion_tokens.is_balanced() {
224          Error!(
225            "misdefined",
226            cs,
227            s!("Expansion of '{}' has unbalanced {{}}", cs),
228            "skipping pack_parameters"
229          );
230          // Store as-is without packing
231          expansion_opt = Some(ExpansionBody::Tokens(expansion_tokens));
232        } else {
233          expansion_opt = Some(ExpansionBody::Tokens(expansion_tokens.pack_parameters()?));
234        }
235      }
236    }
237    let has_cc_arg = match expansion_opt {
238      Some(ExpansionBody::Tokens(ref tks)) => tks
239        .unlist_ref()
240        .iter()
241        .any(|t| t.get_catcode() == Catcode::ARG),
242      _ => false,
243    };
244    // simplify: treat empty tokens as None
245    let expansion = match expansion_opt {
246      Some(ExpansionBody::Tokens(tks)) if tks.is_empty() => None,
247      real_body => real_body,
248    };
249
250    Ok(Expandable {
251      cs,
252      paramlist,
253      expansion,
254      // locator           => $source->getLocator,
255      is_protected: traits.protected || get_prefix("protected"),
256      is_outer: traits.outer || get_prefix("outer"),
257      is_long: traits.long || get_prefix("long"),
258      has_cc_arg,
259      alias: traits.alias,
260      ..Expandable::default()
261    })
262  }
263}
