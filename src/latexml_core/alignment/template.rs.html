102        Math => 0.0,
103        Integer => 0.1,
104        MathAltText => 0.2,
105        _ => 0.75,
106      },
107      Integer => match other {

217    let save_between = config.save_between.unwrap_or_default(); // `between` comes before `before`!
218    for column in columns.iter_mut() {
219      column.empty = true;
220    }
221    for v in repeated.iter_mut() {
222      v.empty = true;
223    }
224
225    Template {
226      columns,
227      pseudorow,
228      repeating,
229      repeated,
230      non_repeating,
231      save_before,
232      save_between,

236      padding: None,

256  /// Perl Template.pm L113-118: finish
257  /// Appends \lx@intercol to last column's after unless disabled_intercolumn
258  pub fn finish(&mut self) {
259    let last = if self.repeating {
260      self.repeated.last_mut()
261    } else {
262      self.columns.last_mut()
263    };
264    if let Some(prev) = last {
265      if !self.disabled_intercolumn {
266        let mut after = prev.after.clone().unwrap_or_default().unlist();
267        after.push(T_CS!("\\lx@intercol"));
268        prev.after = Some(Tokens::new(after));
269      }
270    }
271  }

273  // These add material before & after the current column
274  pub fn add_before_column(&mut self, mut new: VecDeque<Token>) {
275    let current_sb = self.save_before.drain(..);
276    new.extend(current_sb);
277    self.save_before = new; // NOTE: goes all the way to front!
278  }
279  // NOTE: \lx@column@trimright should ONLY be added to LaTeX tabular style templates!!!!
280  // NOT \halign style templates!
281  pub fn add_after_column(&mut self, new: Vec<Token>) {
282    if let Some(current_column) = self.columns.last_mut() {
283      let current_after = current_column.after.clone().unwrap_or_default().unlist();
284      current_column.after = Some(Tokens!(T_CS!("\\lx@column@trimright"), new, current_after));
285    }
286  }

288  // Perl Template.pm L65-74: addBetweenColumn
289  pub fn add_between_column(&mut self, tokens: Vec<Token>) {
290    if let Some(current_column) = self.columns.last_mut() {
291      let mut combined = Vec::new();
292      let current_after = current_column.after.clone().unwrap_or_default().unlist();
293      combined.extend(current_after);
294      // Perl L69-70: prepend \lx@intercol unless disabled_intercolumn
295      if !self.disabled_intercolumn {
296        combined.push(T_CS!("\\lx@intercol"));
297      }
298      combined.extend(tokens);
299      current_column.after = Some(Tokens::new(combined));
300    } else {
301      self.save_between.extend(tokens);
302    }
303  }

305  // Perl Template.pm L82-110: addColumn
306  pub fn add_column(&mut self, mut col: Cell) {
307    // Perl L85-87: append \lx@intercol to previous column's after unless disabled_intercolumn
308    if let Some(prev) = if self.repeating {
309      self.repeated.last_mut()
310    } else {
311      self.columns.last_mut()
312    } {
313      if !self.disabled_intercolumn {
314        let mut after = prev.after.clone().unwrap_or_default().unlist();
315        after.push(T_CS!("\\lx@intercol"));
316        prev.after = Some(Tokens::new(after));
317      }
318    }
319    // Perl L88-95: build before from save_between + \lx@intercol + properties before + save_before
320    let mut before = Vec::new();
321    if !self.save_between.is_empty() {
322      before.extend(self.save_between.clone());
323    }
324    // Perl L90: push \lx@intercol unless disabled_intercolumn
325    if !self.disabled_intercolumn {
326      before.push(T_CS!("\\lx@intercol"));
327    }
328    // Perl L91: delete disabled_intercolumn
329    self.disabled_intercolumn = false;
330
331    if let Some(prop_before) = col.before {
332      before.extend(prop_before.unlist());
333    }
334    if !self.save_before.is_empty() {
335      before.extend(self.save_before.clone());
336    }
337    col.before = if !before.is_empty() {
338      Some(Tokens::new(before))
339    } else {
340      None
341    };
342    let mut after = vec![T_CS!("\\lx@column@trimright")];
343    if let Some(prop_after) = col.after {
344      after.extend(prop_after.unlist());
345    }
346    col.after = if after.is_empty() {
347      None
348    } else {
349      Some(Tokens::new(after))
350    };
351    col.empty = true;
352    self.save_between = VecDeque::new();
353    self.save_before = VecDeque::new();
354
355    if self.repeating {
356      self.non_repeating = self.columns.len();
357      self.repeated.push(col);
358    } else {
359      self.columns.push(col);
360    }
361  }

363  pub fn get_column_mut(&mut self, n: usize) -> Option<&mut Cell> {
364    let all_columns = self.columns.len();
365    if (n > all_columns) && self.repeating {
366      let rep = &self.repeated;
367      let m = rep.len();
368      if m > 0 {
369        for i in all_columns..n {
370          let dup = rep[(i - self.non_repeating) % m].clone();
371          self.columns.push(dup);
372        }
373      }
374    }
375    if n > 0 {
376      self.columns.get_mut(n - 1)
377    } else {
378      None
379    }
380  }
