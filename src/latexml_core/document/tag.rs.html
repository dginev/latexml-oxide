9/// Perl's Tag afterOpen/afterClose closures receive ($document, $node, $box).
10/// The $box is the Digested (whatsit) associated with the node, retrieved via getNodeBox.
11pub type TagConstructionClosure =
12  Rc<dyn Fn(&mut Document, &mut Node, Option<&Digested>) -> Result<()>>;
13pub type TagData = (String, Option<HashMap<String, String>>, Digested);
14
15// Specify the properties of a Node tag.
16pub enum TagOptionName {
17  AfterOpen,
18  AfterOpenEarly,
19  AfterOpenLate,
20  AfterClose,
21  AfterCloseEarly,
22  AfterCloseLate,
23}
24
25impl TagOptionName {
26  pub fn all() -> Vec<TagOptionName> {
27    use self::TagOptionName::*;
28    vec![
29      AfterOpen,
30      AfterOpenEarly,
31      AfterOpenLate,
32      AfterClose,
33      AfterCloseEarly,
34      AfterCloseLate,
35    ]
36  }
37  pub fn is_prepend(&self) -> bool {
38    use self::TagOptionName::*;
39    matches!(*self, AfterOpenEarly | AfterCloseEarly)
40  }
41  pub fn is_append(&self) -> bool {
42    use self::TagOptionName::*;
43    matches!(
44      *self,
45      AfterOpen | AfterClose | AfterOpenLate | AfterCloseLate
46    )
47  }
48}

50#[derive(Clone, Default)]
51pub struct TagOptions {
52  pub auto_open:         Option<bool>,
53  pub auto_close:        Option<bool>,
54  pub after_open:        Option<Vec<TagConstructionClosure>>,
55  pub after_close:       Option<Vec<TagConstructionClosure>>,
56  pub after_open_early:  Option<Vec<TagConstructionClosure>>,
57  pub after_close_early: Option<Vec<TagConstructionClosure>>,
58  pub after_open_late:   Option<Vec<TagConstructionClosure>>,
59  pub after_close_late:  Option<Vec<TagConstructionClosure>>,
60}
61
62impl TagOptions {
63  pub fn get(&self, name: &TagOptionName) -> Option<&Vec<TagConstructionClosure>> {
64    use self::TagOptionName::*;
65    match *name {
66      AfterOpen => self.after_open.as_ref(),
67      AfterOpenEarly => self.after_open_early.as_ref(),
68      AfterOpenLate => self.after_open_late.as_ref(),
69      AfterClose => self.after_close.as_ref(),
70      AfterCloseEarly => self.after_close_early.as_ref(),
71      AfterCloseLate => self.after_close_late.as_ref(),
72    }
73  }
74
75  pub fn get_mut(&mut self, name: &TagOptionName) -> Option<&mut Vec<TagConstructionClosure>> {
76    use self::TagOptionName::*;
77    match *name {
78      AfterOpen => self.after_open.as_mut(),
79      AfterOpenEarly => self.after_open_early.as_mut(),
80      AfterOpenLate => self.after_open_late.as_mut(),
81      AfterClose => self.after_close.as_mut(),
82      AfterCloseEarly => self.after_close_early.as_mut(),
83      AfterCloseLate => self.after_close_late.as_mut(),
84    }
85  }
86
87  pub fn remove(&mut self, name: &TagOptionName) -> Vec<TagConstructionClosure> {
88    use self::TagOptionName::*;
89    let field = match *name {
90      AfterOpen => &mut self.after_open,
91      AfterOpenEarly => &mut self.after_open_early,
92      AfterOpenLate => &mut self.after_open_late,
93      AfterClose => &mut self.after_close,
94      AfterCloseEarly => &mut self.after_close_early,
95      AfterCloseLate => &mut self.after_close_late,
96    };
97    match field {
98      Some(ref mut vec) => std::mem::take(vec),
99      None => Vec::new(),
100    }
101  }
102
103  pub fn set(&mut self, name: &TagOptionName, value: Option<Vec<TagConstructionClosure>>) {
104    use self::TagOptionName::*;
105    match *name {
106      AfterOpen => {
107        self.after_open = value;
108      },
109      AfterOpenEarly => {
110        self.after_open_early = value;
111      },
112      AfterOpenLate => {
113        self.after_open_late = value;
114      },
115      AfterClose => {
116        self.after_close = value;
117      },
118      AfterCloseEarly => {
119        self.after_close_early = value;
120      },
121      AfterCloseLate => {
122        self.after_close_late = value;
123      },
124    }
125  }
126
127  pub fn prepend(&mut self, name: &TagOptionName, mut value: Vec<TagConstructionClosure>) {
128    let drained: Vec<TagConstructionClosure> = match self.get_mut(name) {
129      Some(vec) => std::mem::take(vec),
130      None => Vec::new(),
131    };
132
133    value.extend(drained);
134
135    self.set(name, Some(value));
136  }
137
138  pub fn append(&mut self, name: &TagOptionName, value: Vec<TagConstructionClosure>) {
139    if self.get(name).is_none() {
140      // initialize if needed
141      self.set(name, Some(Vec::new()));
142    }
143
144    if let Some(vec) = self.get_mut(name) {
145      vec.extend(value);
146    }
147  }
