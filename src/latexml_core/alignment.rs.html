1//! # Representation of aligned structures
2//! An "Alignment" is an array/tabular construct as:
3//!  `<tabular><tr><td>...`
4//! or, for math mode
5//!   `<XMArray><XMRow><XMCell>...`
6//! (where initially, each XMCell will contain an XMArg to indicate
7//! individual parsing of each cell's content is desired)
8//!
9//! An Alignment object is a sort of fake Whatsit;
10//! It takes some magic to sneak it into the Digestion stream
11//! (see TeX.pool \lx@begin@alignment), but it needs to be created
12//! BEFORE the contents of the alignment are digested,
13//! since we stuff a lot of information into it
14//! (row, column boxes, borders, spacing, etc...)
15//! But once it has been captured, it should otherwise act
16//! like a Whatsit and be responsible for construction (be_absorbed),
17//! and sizing estimation (computeSize)
18//!
19//! Ultimately, this should be better tied into DefConstructor
20//! because an Alignment currently doesn't know what CS created it (debugging!);
21//! Also, it would better connect the things being constructed, reversion, etc.

49/// token-locators: source span of an alignment cell (its content's locator).
50/// `tabular`/`tr`/`td` are opened before their content's `box_to_absorb` is set,
51/// so the absorb explicitly stamps each with its cell/row/table span via
52/// `Document::set_current_box_locator`. See docs/SOURCE_PROVENANCE.md §3.1.3.
53#[cfg(feature = "token-locators")]
54fn cell_loc(cell: &Cell) -> Option<crate::common::locator::Locator> {
55  cell
56    .boxes
57    .as_ref()
58    .and_then(|b| b.get_locator())
59    .filter(|l| l.from_line != 0)
60}

62/// Union (first `from` → last `to`) of a row's cell spans.
63#[cfg(feature = "token-locators")]
64fn row_span(row: &Row) -> Option<crate::common::locator::Locator> {
65  row
66    .get_columns()
67    .iter()
68    .filter_map(cell_loc)
69    .reduce(|a, b| crate::common::locator::Locator::new_range(a, b).unwrap_or(a))
70}

233  pub fn add_line(&mut self, border: &str, cols: Vec<usize>) {
234    if let Some(row_idx) = self.current_row {
235      let Some(row) = self.rows.get_mut(row_idx) else {
236        return;
237      };
238      self.current_column = 1;
239      if !cols.is_empty() {
240        // Perl Alignment.pm:128-130 — `$row->column($c)` returns undef
241        // for out-of-range column index and autovivifies a discarded
242        // temp hash, so the assignment silently no-ops. Match that
243        // here: skip indices that don't map to a real column instead
244        // of panicking on `.unwrap()`. Witness: 0708.2784 with a
245        // `\hline`/`\cline`-style line referencing a column past the
246        // tabular's column count.
247        for c in cols {
248          if let Some(colspec) = row.get_column_mut(c) {
249            colspec.border.push_str(border);
250          }
251        }
252      } else {
253        for colspec in row.get_columns_mut() {
254          colspec.border.push_str(border)
255        }
256      }
257    }
258  }

260  pub fn next_column(&mut self) -> Result<Option<&mut Cell>> {
261    if self.current_row.is_none() {
262      return Ok(None);
263    }
264    self.current_column += 1;
265    let current_row = self.rows.get_mut(self.current_row.unwrap()).unwrap();
266    if current_row.get_column_mut(self.current_column).is_some() {
267      Ok(current_row.get_column_mut(self.current_column))
268    } else {
269      // Perl: Error then add fallback column with align=center
270      Error!("unexpected", "&", "Extra alignment tab '&'");
271      let fallback = Cell {
272        align: Some(Align::Center),
273        ..Cell::default()
274      };
275      current_row.add_column(fallback);
276      Ok(current_row.get_column_mut(self.current_column))
277    }
278  }

398  pub fn start_row(&mut self, pseudorow: bool) -> Result<()> {
399    self.new_row();
400    bgroup(); // Grouping around ROW!
401    if pseudorow {
402      self.current_row_mut().unwrap().set_pseudo()
403    } else {
404      // Store row number before digest — row hooks may need it (e.g. \rowcolors)
405      // and cannot re-borrow the alignment which is already mutably borrowed.
406      let row_num = self.current_row_number();
407      assign_value("alignmentRowNumber", row_num as i32, None);
408      let row_before = stomach::digest(T_CS!("\\lx@alignment@row@before"))?;
409      push_box_list(row_before);
410    }
411    self.in_row = true;
412    assign_value("alignmentStartColumn", 0, None); // ???
413    Ok(())
414  }

427  pub fn start_column(&mut self, pseudorow: bool) -> Result<()> {
428    if !self.in_row {
429      self.start_row(pseudorow)?;
430    } else if pseudorow {
431      self.current_row_mut().unwrap().set_pseudo();
432    }
433    bgroup(); // Grouping around CELL!
434    // Note: a VERY round-about way of tracking the column spanning!
435    assign_value("alignmentStartColumn", self.current_column_number(), None);
436    // Propagate `?` so a TooManyErrors Fatal (e.g. from a runaway
437    // `&` storm in a malformed alignment, paper 1112.6246) actually
438    // aborts rather than getting silently swallowed by `let _ =`.
439    let _colspec = self.next_column()?;
440    set_align_group_count(1000000);
441    self.in_column = true;
442    Ok(())
443  }

508  fn be_absorbed(&self, _document: &mut Document) -> Result<Vec<Node>> {
509    // Alignments must be absorbed via `be_absorbed_mut` because
510    // `normalize_alignment` mutates the carrier (rearranging rows,
511    // applying border specs, etc). The `Digested::be_absorbed` dispatch
512    // for `DigestedData::Alignment` correctly calls `be_absorbed_mut`;
513    // this immutable path should never be reached in practice. Surface
514    // a meaningful error rather than a bare `todo!()` panic if an
515    // unexpected caller lands here.
516    fatal!(
517      Internal,
518      Misdefined,
519      "Alignment::be_absorbed called — use be_absorbed_mut instead \
520       (alignment rearrangement requires &mut self)"
521    );
522  }
523  fn be_absorbed_mut(&mut self, document: &mut Document) -> Result<Vec<Node>> {
524    let ismath = self.is_math;
525    normalize_alignment(self)?;
526    // token-locators: the whole table's span (union of all cell spans), stamped
527    // on the `tabular` element below. Computed before the mutable `rows` borrow.
528    #[cfg(feature = "token-locators")]
529    let table_span = self
530      .rows
531      .iter()
532      .filter_map(row_span)
533      .reduce(|a, b| crate::common::locator::Locator::new_range(a, b).unwrap_or(a));
534    let rows = &mut self.rows;
535    if rows.is_empty() {
536      return Ok(Vec::new());
537    }
538
539    // Guard via the absorb limit to avoid infinite loops (Perl L478-483)
540    let absorb_limit = crate::state::lookup_int("absorb_limit");
541    if absorb_limit > 0 {
542      let mut absorb_count = crate::state::lookup_int("absorb_count");
543      absorb_count += 1;
544      crate::state::assign_value(
545        "absorb_count",
546        absorb_count,
547        Some(crate::state::Scope::Global),
548      );
549      if absorb_count > absorb_limit {
550        fatal!(
551          Timeout,
552          Convert,
553          s!(
554            "Whatsit absorb limit of {} exceeded, infinite loop?",
555            absorb_limit
556          )
557        );
558      }
559    }
560
561    // We _should_ attach boxes to the alignment and rows,
562    // but (ATM) we"ve only got sensible boxes for the cells.
563    let mut attrs = HashMap::default();
564    std::mem::swap(&mut attrs, &mut self.xml_attributes);
565    // Perl Alignment.pm L311-316: pass dimension data to openContainer callback
566    // Serialized as px-value strings for callbacks that need positioning (tikz matrices).
567    if let Some(w) = self.cached_width {
568      attrs.insert("cwidth".to_string(), format!("{}", w.px_value(None)));
569    }
570    if let Some(h) = self.cached_height {
571      attrs.insert("cheight".to_string(), format!("{}", h.px_value(None)));
572    }
573    if let Some(d) = self.cached_depth {
574      attrs.insert("cdepth".to_string(), format!("{}", d.px_value(None)));
575    }
576    let open_container_fn = &self.open_container;
577    // token-locators: stamp the `tabular` with the table span before it opens.
578    #[cfg(feature = "token-locators")]
579    document.set_current_box_locator(table_span);
580    open_container_fn(document, attrs)?;
581
582    for row in rows {
583      // token-locators: stamp this `tr` with the row's span before it opens.
584      #[cfg(feature = "token-locators")]
585      document.set_current_box_locator(row_span(row));
586      let vpad_opt = row.get_padding().copied();
587      // Perl Alignment.pm L319-324: pass position/size to openRow callback
588      let mut open_row_attrs = HashMap::default();
589      for (k, v) in &row.properties {
590        open_row_attrs.insert(k.clone(), v.clone());
591      }
592      if let Some(x) = row.x {
593        open_row_attrs.insert("x".to_string(), Stored::Dimension(x));
594      }
595      if let Some(y) = row.y {
596        open_row_attrs.insert("y".to_string(), Stored::Dimension(y));
597      }
598      if let Some(w) = row.cached_width {
599        open_row_attrs.insert("cwidth".to_string(), Stored::Dimension(w));
600      }
601      if let Some(h) = row.cached_height {
602        open_row_attrs.insert("cheight".to_string(), Stored::Dimension(h));
603      }
604      if let Some(d) = row.cached_depth {
605        open_row_attrs.insert("cdepth".to_string(), Stored::Dimension(d));
606      }
607      let open_row_fn = &self.open_row;
608      open_row_fn(document, open_row_attrs)?;
609      for before in row.before.iter() {
610        document.absorb(before, None)?;
611      }
612      for cell in row.get_columns_mut().iter_mut() {
613        if cell.skipped {
614          continue;
615        }
616        // Normalize the border attribute
617        // Perl: join(' ', sort(map { split(/ */, $_) } $$cell{border}));
618        //       $border =~ s/(.) \1/$1$1/g;

620          cell.border.chars().filter(|c| !c.is_whitespace()).collect();
621        border_chars.sort_unstable();

623        for (idx, &c) in border_chars.iter().enumerate() {
624          border.push(c);
625          // Space between different consecutive chars, no space between same chars
626          if idx + 1 < border_chars.len() && border_chars[idx + 1] != c {
627            border.push(' ');
628          }
629        }
630        let open_column_fn = &self.open_column;
631        let mut cell_attrs = HashMap::default();
632        // Perl Alignment.pm L358-359: pass position/size to openColumn callback
633        if let Some(x) = cell.x {
634          cell_attrs.insert("x".to_string(), format!("{}", x.px_value(None)));
635        }
636        if let Some(y) = cell.y {
637          cell_attrs.insert("y".to_string(), format!("{}", y.px_value(None)));
638        }
639        if let Some(w) = cell.cached_width {
640          cell_attrs.insert("cwidth".to_string(), format!("{}", w.px_value(None)));
641        }
642        if let Some(h) = cell.cached_height {
643          cell_attrs.insert("cheight".to_string(), format!("{}", h.px_value(None)));
644        }
645        if let Some(d) = cell.cached_depth {
646          cell_attrs.insert("cdepth".to_string(), format!("{}", d.px_value(None)));
647        }
648        // Perl: always passes align attribute (Alignment.pm L350)
649        if let Some(ref align) = cell.align {
650          cell_attrs.insert(String::from("align"), align.name());
651        }
652        if let Some(ref vattach) = cell.vattach {
653          cell_attrs.insert(String::from("vattach"), vattach.clone());
654        }
655        if let Some(w) = cell.width {
656          cell_attrs.insert(String::from("width"), w.to_attribute());
657        }
658        if let Some(vpad) = vpad_opt {
659          cell_attrs.insert(String::from("cssstyle"), s!("padding-bottom: {vpad}"));
660        }
661        // colortbl: backgroundcolor from \columncolor/\cellcolor
662        if let Some(ref bg) = cell.backgroundcolor {
663          cell_attrs.insert(String::from("backgroundcolor"), bg.clone());
664        }
665        // Perl: colspan/rowspan attributes for spanning cells
666        if cell.colspan.unwrap_or(1) != 1 {
667          cell_attrs.insert(String::from("colspan"), cell.colspan.unwrap().to_string());
668        }
669        if cell.rowspan.unwrap_or(1) != 1 {
670          cell_attrs.insert(String::from("rowspan"), cell.rowspan.unwrap().to_string());
671        }
672        if !border.is_empty() {
673          cell_attrs.insert(String::from("border"), border);
674        }
675        if cell.thead_in_column || cell.thead_in_row {
676          let mut thead = String::new();
677          if cell.thead_in_column {
678            thead.push_str(Axis::Column.name());
679            if cell.thead_in_row {
680              thead.push(' ');
681            }
682          }
683          if cell.thead_in_row {
684            thead.push_str(Axis::Row.name());
685          }
686          if !thead.is_empty() {
687            cell_attrs.insert(String::from("thead"), thead);
688          }
689        }
690        // Perl Alignment.pm L332-347: ltx_nopad_l/ltx_nopad_r CSS classes
691        // Based on lspaces/rspaces width relative to 0.2em threshold
692        let mut classes: Vec<String> = Vec::new();
693        let empty = cell.empty;
694        // Perl: $$cell{boxes} — truthy when boxes field is defined (even if empty List)
695        let has_boxes = cell.boxes.is_some();
696        let mut pre_absorb: Option<Digested> = None;
697        let mut post_absorb: Option<Digested> = None;
698        if !ismath {
699          // 0.2em ≈ 131072 scaled points; 1.5em ≈ 983040 scaled points (at 10pt font)
700          let threshold_02em: i64 = 131072;
701          let threshold_15em: i64 = 983040;
702          // Perl: $lpad = ($$cell{lspaces} ? $$cell{lspaces}->getWidth->valueOf : 0)
703          // Note: In Perl, lspaces is populated from \lx@intercol (isSpace, width=tabcolsep)
704          // during cell content extraction. Rust doesn't populate lspaces, so we approximate:
705          // - If template has \lx@intercol → there IS intercolumn padding (assume 0.2em)
706          // - Else if template has \hfil/\hfill → centering fill, treat as padding (prevents
707          //   incorrect ltx_nopad_l for regular centered/right-aligned columns)
708          // - Else → no padding (assume 0, enables ltx_nopad_l)
709          // Perl L338-339: lpad/rpad from extracted lspaces/rspaces width.
710          // When lspaces/rspaces are populated (from cell extraction), use their
711          // actual width. When None, use template heuristic as fallback.
712          // Perl L338-339: lpad/rpad from lspaces/rspaces width.
713          // In Perl, lspaces is populated from the left-scan of digested cell boxes.
714          // The left-scan encounters template-injected boxes (vrules, intercol spaces,
715          // fills) and extracts spacing. When lspaces is undef, Perl returns 0.
716          // In Rust, lspaces is not always populated by extraction (extraction stores
717          // None for empty lspaces). We use intercol_reachable_in_before to distinguish:
718          // - Regular columns (|l|): \vrule then \lx@intercol → reachable → threshold
719          // - @{text} columns: text then \lx@intercol → NOT reachable → 0
720          //
721          // KNOWN LIMITATION: this fallback over-reports padding for some
722          // numprint cells (`\lx@intercol\nprt@begin\ignorespaces`-style
723          // before): Perl's extracted lspaces would be undef there → lpad=0 →
724          // ltx_nopad_l added; our heuristic returns threshold_02em → no nopad_l.
725          // Naively removing the heuristic regresses 21 other tabular tests
726          // because Rust's extraction doesn't always populate lspaces for
727          // cases where Perl's would (`|l|`, `|c|`, `|r|`, p/m/X, etc.). The
728          // proper fix is to make lspaces extraction reliable to match Perl's
729          // left-scan; until then, this heuristic is load-bearing.
730          let lpad = cell
731            .lspaces
732            .as_ref()
733            .and_then(|ls| ls.get_width(None).ok().flatten())
734            .map(|rv| rv.value_of())
735            .unwrap_or_else(|| {
736              if intercol_reachable_in_before(&cell.before) || template_has_fill(&cell.before) {
737                threshold_02em
738              } else {
739                0
740              }
741            });
742          let rpad = cell
743            .rspaces
744            .as_ref()
745            .and_then(|rs| rs.get_width(None).ok().flatten())
746            .map(|rv| rv.value_of())
747            .unwrap_or_else(|| {
748              // Perl: rpad from rspaces. When not extracted, use template.
749              // Only the after tokens determine right padding.
750              if template_has_intercol(&cell.after) {
751                threshold_02em
752              } else {
753                0
754              }
755            });
756          // Perl L340-341: ltx_nopad_l, unless math mode (Perl: `unless $ismath`)
757          if !ismath && (!empty || has_boxes) && lpad < threshold_02em {
758            classes.push("ltx_nopad_l".to_string());
759          } else if lpad < threshold_15em {
760            // In math mode, absorb named spacing (like \quad) as XMHint content
761            // even when below the 1.5em threshold. This preserves XMHint for
762            // the math parser to convert to lpadding.
763            if ismath && cell.lspaces.is_some() {
764              pre_absorb = cell.lspaces.take();
765            }
766          } else {
767            pre_absorb = cell.lspaces.take();
768          }
769          // Perl L344-345: ltx_nopad_r, unless math mode (Perl: `unless $ismath`)
770          if !ismath && (!empty || has_boxes) && rpad < threshold_02em {
771            classes.push("ltx_nopad_r".to_string());
772          } else if rpad < threshold_15em {
773            // do nothing — use CSS default padding
774          } else {
775            post_absorb = cell.rspaces.take();
776          }
777        }
778        if let Some(ref cell_class) = cell.class {
779          classes.insert(0, cell_class.clone());
780        }
781        let class_str: String = classes
782          .into_iter()
783          .filter(|s| !s.is_empty())
784          .collect::<Vec<_>>()
785          .join(" ");
786        if !class_str.is_empty() {
787          cell_attrs.insert(String::from("class"), class_str);
788        }
789        //       # Which properties do we expose to the constructor?
790        //       x      => $$cell{x}, y => $$cell{y},
791        //       cached_width => $$cell{cached_width}, cached_height => $$cell{cached_height},
792        // cached_depth => $$cell{cached_depth})
793        // token-locators: stamp this `td` with the cell's content span before it
794        // opens (its content's box_to_absorb is set just below, after the open).
795        #[cfg(feature = "token-locators")]
796        document.set_current_box_locator(cell_loc(cell));
797        cell.cell = open_column_fn(document, cell_attrs)?;
798        // Perl L362: absorb cell content only if !skippable (not just !empty)
799        if !cell.skippable {
800          let box_ref = cell.boxes.as_ref().unwrap();
801          // local $LaTeXML::BOX
802          document.set_box_to_absorb(Some(box_ref.clone()));
803          // Perl wraps cell content in XMArg for math alignments, but NOT for _Capture_ columns
804          // (_Capture_ is not in the schema, so Perl's openElement validation prevents XMArg there)
805          let cur_qname = crate::document::get_node_qname(document.get_node());
806          let wrap_xmarg =
807            ismath && !crate::common::arena::with(cur_qname, |s| s.ends_with("_Capture_"));
808          if wrap_xmarg {
809            // Hacky!
810            document.open_element("ltx:XMArg", Some(string_map!("rule" => "Anything")), None)?;
811          }
812          // Perl L365: absorb pre-spacing (lspaces > 1.5em)
813          if let Some(ref pre) = pre_absorb {
814            document.absorb(pre, None)?;
815          }
816          // In math mode, absorb lspaces as content (creates XMHint for \quad etc.)
817          // This is needed for the math parser to convert XMHint → lpadding.
818          if ismath && pre_absorb.is_none() {
819            if let Some(ref lsp) = cell.lspaces {
820              document.absorb(lsp, None)?;
821            }
822          }
823          document.absorb(box_ref, None)?;
824          // Perl L367: absorb post-spacing (rspaces > 1.5em)
825          if let Some(ref post) = post_absorb {
826            document.absorb(post, None)?;
827          }
828          if wrap_xmarg {
829            // Hacky!
830            document.close_element("ltx:XMArg")?;
831          }
832          // expire local $LaTeXML::BOX
833          document.expire_box_to_absorb();
834        } else if let Some(ref boxes) = cell.boxes {
835          // Cell is skippable but may contain preserved boxes (e.g. \label wrapped
836          // in \lx@hidden@noalign with alignmentPreserve=true). These boxes need
837          // to be absorbed so their constructors run (e.g. \label sets labels= on
838          // the parent equation element via float_to_label).
839          // In Perl, \hfil from the template contributes cell width, making such
840          // cells non-skippable. In Rust, \hfil doesn't contribute width.
841          for item in boxes.unlist() {
842            if item.get_property_bool("alignmentPreserve") {
843              document.absorb(&item, None)?;
844            }
845          }
846        }
847        let close_column_fn = &self.close_column;
848        close_column_fn(document)?;
849      }
850      for after in row.after.iter() {
851        document.absorb(after, None)?;
852      }
853      let close_row_fn = &self.close_row;
854      close_row_fn(document)?;
855    }
856    let close_container_fn = &self.close_container;
857    let node_opt = close_container_fn(document)?;
858
859    // If we're not nested inside another tabular
860    // [This should be an afterConstruct somewhere?]
861    // If requested to guess headers & we're not nested inside another tabular
862    if let Some(mut node) = node_opt {
863      if document
864        .findnodes("ancestor::ltx:tabular", Some(&node))
865        .is_empty()
866      {
867        let hashead = !document
868          .findnodes("descendant::ltx:td[@thead]", Some(&node))
869          .is_empty();
870        // If requested && no cells are already marked as being thead, apply heuristic
871        let guess_headers = self
872          .properties
873          .get("guess_headers")
874          .map(|v| !matches!(v, Stored::Bool(false)))
875          .unwrap_or(false);
876        if guess_headers && !hashead {
877          guess_alignment_headers(document, &mut node, self)?;
878        }
879        // Otherwise, if not a math array, group thead & tbody rows
880        // TODO: Re-design asking the outer Whatsit about "!body->isMath"
881        else if hashead && !ismath {
882          // in case already marked w/thead|tbody
883          alignment_regroup_rows(document, &node)?;
884        }
885      }
886      Ok(vec![node])
887    } else {
888      Ok(Vec::new())
889    }
890  }
891}

921/// a reader for the Template parameter type
922pub fn read_alignment_template() -> Result<Template> {
923  gullet::skip_spaces()?;
924  local_build_template(Template::default());
925  let mut tokens = vec![T_BEGIN!()];
926  let mut nopens = 0;
927  while let Some(open) = gullet::read_token()? {
928    if open.get_catcode() == Catcode::BEGIN {
929      nopens += 1;
930    } else {
931      gullet::unread_one(open);
932      break;
933    }
934  }
935  while let Some(op) = gullet::read_token()? {
936    let cc = op.get_catcode();
937    if cc == Catcode::SPACE {
938    } else if cc == Catcode::END {
939      let mut last_op = op;
940      nopens -= 1;
941      while nopens > 0 {
942        if let Some(next_op) = gullet::read_token()? {
943          last_op = next_op;
944          if last_op.get_catcode() != Catcode::END {
945            break;
946          }
947        } else {
948          break;
949        }
950        nopens -= 1;
951      }
952      if nopens <= 0 {
953        break;
954      }
955      gullet::unread_one(last_op);
956    } else if let Some(defn) = lookup_expandable(&T_CS!(s!("\\NC@rewrite@{op}")), None)? {
957      let invoked = defn.invoke(true)?;
958      gullet::unread(invoked);
959    } else if cc == Catcode::BEGIN {
960      let balanced_arg = gullet::read_balanced(ExpansionLevel::Off, false, false)?;
961      if !balanced_arg.is_empty() {
962        gullet::unread(balanced_arg);
963      }
964    } else {
965      Warn!("unexpected", op, s!("Unrecognized tabular template {op:?}"));
966    }
967    if nopens <= 0 {
968      break;
969    }
970  }
971  tokens.push(T_END!());
972  with_current_build_template(|template_opt| {
973    let t = template_opt.unwrap();
974    t.set_reversion(Tokens::new(tokens));
975    // Perl Alignment.pm L912: $BUILD_TEMPLATE->finish
976    t.finish();
977  });
978  Ok(take_build_template().unwrap())
979}

997//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
998// Experimental alignment heading heuristications.
999//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1000// We attempt to recognize patterns of rows/columns that indicate which might be headers.
1001// We'll characterize the cells by alignment, content and borders.
1002// Then, assuming that headers will be first and be noticably `different' from data lines,
1003// and also that the data lines will have similar structure,  we'll attempt to
1004// recognize groups of header lines and groups data lines, possibly alternating.
1005
1006/// Check whether a template token list contains fill/spacing commands
1007/// like \hfil, \hfill, \hskip, \lx@intercol. Previously used as fallback
1008/// for lpad/rpad; now superseded by template_has_intercol for better @{} handling.
1009#[allow(dead_code)]
1010fn template_has_fill(tokens: &Option<Tokens>) -> bool {
1011  if let Some(ref toks) = tokens {
1012    for tok in toks.unlist_ref() {
1013      let s = tok.to_string();
1014      if s == "\\hfil" || s == "\\hfill" || s == "\\hskip" {
1015        return true;
1016      }
1017    }
1018  }
1019  false
1020}
1021
1022/// Check if \lx@intercol in the before tokens is reachable by the left-scan.
1023/// The left-scan skips: isVerticalRule (\vrule), \relax, isFill (\hfil/\hfill),
1024/// isSpace, isHorizontalRule, alignmentSkippable, Comment.
1025/// It STOPS at real content (like text from @{text}).
1026/// For `\vrule\relax\lx@intercol\hfil`: reachable (vrule is skippable).
1027/// For `@{1}\lx@intercol\hfil`: NOT reachable ("1" blocks the scan).
1028fn intercol_reachable_in_before(tokens: &Option<Tokens>) -> bool {
1029  if let Some(ref toks) = tokens {
1030    for tok in toks.unlist_ref() {
1031      let s = tok.to_string();
1032      if s == "\\lx@intercol" || s.contains("intercol") {
1033        return true;
1034      }
1035      // These are skippable by the left-scan in Perl's extractAlignmentColumn
1036      if s == "\\vrule"
1037        || s == "\\relax"
1038        || s == "\\hfil"
1039        || s == "\\hfill"
1040        || s == "\\hskip"
1041        || s == "\\lx@column@trimright"
1042      {
1043        continue;
1044      }
1045      // Any other token blocks the scan
1046      return false;
1047    }
1048  }
1049  false
1050}
1051
1052/// Check if template tokens contain \lx@intercol (intercolumn spacing).
1053/// Unlike template_has_fill, this ignores \hfil/\hfill which are alignment fill.
1054/// \lx@intercol indicates actual intercolumn padding; \hfil is just centering.
1055/// For @{}c@{} columns, \lx@intercol is disabled but \hfil remains.
1056fn template_has_intercol(tokens: &Option<Tokens>) -> bool {
1057  if let Some(ref toks) = tokens {
1058    for tok in toks.unlist_ref() {
1059      let s = tok.to_string();
1060      if s == "\\lx@intercol" || s.contains("intercol") {
1061        return true;
1062      }
1063    }
1064  }
1065  false
1066}

1153//======================================================================
1154// Regroup the rows into thead, tbody & tfoot
1155// Any leading rows, all of whose cells have attribute thead should be in thead.
1156// UNLESS any of them have a rowspan that extends PAST the end of the thead!!!!
1157// trailing rows marked as thead go into tfoot.
1158fn alignment_regroup_rows(document: &mut Document, table: &Node) -> Result<()> {
1159  let mut rows = document.findnodes("ltx:tr", Some(table));
1160  // `heads` is bounded by the initial thead-candidate rows; pre-size
1161  // to `rows.len()` as a conservative upper bound.
1162  let mut heads = Vec::with_capacity(rows.len());
1163  let mut maxreach = 0;
1164  // Scan initial rows as potential thead
1165  while !rows.is_empty() {
1166    let cells = document.findnodes("ltx:td", Some(&rows[0]));
1167    // Non header cells, done.
1168    if cells
1169      .iter()
1170      .any(|cell| cell.get_attribute("thead").is_none())
1171    {
1172      break;
1173    }
1174    let line = heads.len();
1175    heads.push(rows.remove(0));
1176    for cell in cells {
1177      // Malformed/non-numeric rowspan silently degrades to 0 — matches Perl's
1178      // lax numeric coercion and prevents crashes on unusual input XML.
1179      let this_rowspan = cell
1180        .get_attribute("rowspan")
1181        .and_then(|v| v.parse::<usize>().ok())
1182        .unwrap_or(0)
1183        + line;
1184      if this_rowspan > maxreach {
1185        maxreach = this_rowspan;
1186      }
1187    }
1188  }
1189  if maxreach > heads.len() {
1190    // rowspan crossed over thead boundary! Put head rows back at the FRONT of body rows.
1191    heads.append(&mut rows);
1192    rows = heads;
1193    heads = Vec::new();
1194  }
1195  // scan trailing rows as potential tfoot
1196  let mut foots = VecDeque::new();
1197  while !rows.is_empty() {
1198    let cells = document.findnodes("ltx:td", Some(rows.last().unwrap()));
1199    // Non header cells, done.
1200    if cells
1201      .iter()
1202      .any(|cell| cell.get_attribute("thead").is_none())
1203    {
1204      break;
1205    }
1206    foots.push_front(rows.pop().unwrap())
1207  }
1208  if !heads.is_empty() {
1209    document.wrap_nodes("ltx:thead", heads)?;
1210  }
1211  if !rows.is_empty() {
1212    document.wrap_nodes("ltx:tbody", rows)?;
1213  }
1214  if !foots.is_empty() {
1215    document.wrap_nodes("ltx:tfoot", foots.into_iter().collect())?;
1216  }
1217  Ok(())
1218}

1249      // col.content_class, col.border);

1252      } else if col.cell.is_some() {

1254      } else {
1255        0
1256      });
1257      let (mut border_top, mut border_bottom, mut border_left, mut border_right) = (0, 0, 0, 0);

1270      }
1271      if (border_right > 0) || (border_left > 0) {
1272        v = true;
1273      }
1274      col.border_top = Some(border_top);
1275      col.border_bottom = Some(border_bottom);

1278    }
1279    // pad the columns out.
1280    let to_pad = ncols - this_row_len;
1281    if to_pad > 0 {

1284          align: Some(Align::Center),
1285          cell_type: Some('d'),
1286          content_class: Some(ColumnSpec::Empty),
1287          content_length: Some(0),
1288          ..Cell::default()
1289        };
1290        cols.push(col);
1291      }
1292    }
1293  }
1294  // DG: cache assignments, and execute in post-loop, so that we can avoid indexing arithmetic
1295  let mut outer_border_right_assignments = Vec::new();
1296  let mut outer_border_bottom_assignments = Vec::new();

1298  // to the cells they span over. Deferred to avoid borrow conflicts across rows.
1299  #[allow(clippy::type_complexity)]
1300  let mut rowspan_propagation: Vec<(
1301    usize,
1302    usize,

1306  )> = Vec::new();

1309    let row = &mut alignment.rows[r];
1310    let cols = row.get_columns_mut();

1312      let rs = cols[c].rowspan.unwrap_or(1);
1313      let cs = cols[c].colspan.unwrap_or(1);
1314      let ca = cols[c].align.clone();

1316      let cl = cols[c].content_length;
1317      let rb = cols[c].border_right;
1318

1323        row_reach.align = ca.clone();
1324        row_reach.content_class = cc;
1325        row_reach.content_length = cl;
1326      }
1327      // Perl L1073-1077: copy characterizations to rowspan-covered cells
1328      for sr in 1..rs {
1329        for sc in 0..cs {
1330          rowspan_propagation.push((r + sr, c + sc, ca.clone(), cc, cl));
1331        }
1332      }
1333
1334      // move the outer borders
1335      for sr in 0..rs {
1336        outer_border_right_assignments.push((r + sr, c + cs - 1, rb));
1337      }
1338      for sc in 0..cs {

1344  for (row_idx, col_idx, align, content_class, content_length) in rowspan_propagation.into_iter() {

1348        cols[col_idx].align = align;
1349        cols[col_idx].content_class = content_class;
1350        cols[col_idx].content_length = content_length;
1351      }
1352    }
1353  }
1354  // Apply the collected outer border assignments
1355  for (row_idx, col_idx, value) in outer_border_right_assignments.into_iter() {
1356    if let Some(row) = alignment.rows.get_mut(row_idx) {
1357      let cols = row.get_columns_mut();
1358      if col_idx < cols.len() {
1359        cols[col_idx].border_right = value;
1360      }
1361    }
1362  }
1363  for (row_idx, col_idx, value) in outer_border_bottom_assignments.into_iter() {
1364    if let Some(row) = alignment.rows.get_mut(row_idx) {
1365      let cols = row.get_columns_mut();
1366      if col_idx < cols.len() {
1367        cols[col_idx].border_bottom = value;
1368      }
1369    }
1370  }
1371  // Now, do some border massaging...
1372  // Empty-alignment guard: if ncols==0 (no columns in any row), there are no
1373  // borders to massage. Skip the whole block to avoid out-of-bounds panics on
1374  // `cols[0]` (witness: astro-ph0006087, garmire.tex deluxetable input).
1375  if ncols == 0 {
1376    return;
1377  }
1378  for row in alignment.rows.iter_mut() {
1379    let cols = row.get_columns_mut();
1380    cols[0].border_left = Some(if v { 1 } else { 0 });
1381    if ncols > 1 {
1382      if cols[1].border_left.unwrap_or(0) > 0 {
1383        cols[0].border_right = cols[1].border_left;
1384      }
1385      if cols[ncols - 2].border_right.unwrap_or(0) > 0 {
1386        cols[ncols - 1].border_left = cols[ncols - 2].border_right;
1387      }
1388    }
1389    cols[ncols - 1].border_right = Some(if v { 1 } else { 0 });
1390  }
1391  let nrows = alignment.rows.len();
1392  for c in 0..ncols {
1393    alignment.rows[0].get_columns_mut()[c].border_top = Some(if h { 1 } else { 0 });
1394    if nrows > 1 {
1395      if let Some(bt) = alignment.rows[1].get_columns_mut()[c].border_top {
1396        if bt > 0 {
1397          // only set if border is inked
1398          alignment.rows[0].get_columns_mut()[c].border_bottom = Some(bt);
1399        }
1400      }
1401      if let Some(bb) = alignment.rows[nrows - 2].get_columns_mut()[c].border_bottom {

1406      }
1407    }
1408    alignment.rows[nrows - 1].get_columns_mut()[c].border_bottom = Some(if h { 1 } else { 0 });
1409  }
1410  // Note: This propagation doesn't exist in Perl's Alignment.pm.
1411  // But removing it breaks fonts_test (guessTableHeaders). Needs careful audit.
1412  for r in 1..nrows - 1 {
1413    for c in 1..ncols - 1 {
1414      if let Some(bb) = alignment.rows[r - 1].get_columns_mut()[c].border_bottom {
1415        if bb > 0 {
1416          // only set if border is inked
1417          alignment.rows[r].get_columns_mut()[c].border_top = Some(bb);
1418        }
1419      }
1420      if let Some(bt) = alignment.rows[r + 1].get_columns_mut()[c].border_top {
1421        if bt > 0 {
1422          // only set if border is inked
1423          alignment.rows[r].get_columns_mut()[c].border_bottom = Some(bt);
1424        }
1425      }
1426      if let Some(br) = alignment.rows[r].get_columns_mut()[c - 1].border_right {
1427        if br > 0 {
1428          // only set if border is inked
1429          alignment.rows[r].get_columns_mut()[c].border_left = Some(br);
1430        }
1431      }
1432      if let Some(bl) = alignment.rows[r].get_columns_mut()[c + 1].border_left {
1433        if bl > 0 {
1434          // only set if border is inked
1435          alignment.rows[r].get_columns_mut()[c].border_right = Some(bl);

1486/// Return one of: i(nteger), t(ext), m(ath), ? (unknown) or '_' (empty) (or some combination)
1487///  or 'mx' for alternating text & math.
1488fn classify_alignment_cell(xcell: &Node) -> ColumnSpec {
1489  let content = xcell.get_content();
1490  let mut inferred_classes: Vec<ColumnSpec> = Vec::new();
1491  // Perl L1123: /^[\s\d]+$/ — Perl \d is ASCII-only (0-9).
1492  // Also include mathematical double-struck digits (U+1D7D8-U+1D7E1, 𝟘-𝟡) since these
1493  // are font-decoded equivalents of ASCII 0-9 in blackboard bold fonts. In Perl, these
1494  // appear as ASCII digits with font attributes; in Rust they're Unicode codepoints.
1495  // Exclude circled/enclosed numerals (❶❷❸ U+2776-, ①②③ U+2460-) which are symbols.
1496  if !content.is_empty()
1497    && content
1498      .chars()
1499      .all(|c| c.is_whitespace() || c.is_ascii_digit() || ('\u{1D7D8}'..='\u{1D7E1}').contains(&c))
1500  {
1501    inferred_classes.push(ColumnSpec::Integer);
1502  } else {
1503    let mut nodes = xcell.get_child_nodes();
1504    while !nodes.is_empty() {
1505      let ch = nodes.remove(0);
1506      match ch.get_type() {
1507        Some(NodeType::TextNode) => {
1508          let text = ch.get_content();
1509          if !(text.chars().all(|c| c.is_whitespace())
1510            || (inferred_classes.first() == Some(&ColumnSpec::Math)
1511              && SINGLE_PUNCT.is_match(&text)))
1512          {
1513            inferred_classes.push(ColumnSpec::Text);
1514          }
1515        },
1516        Some(NodeType::ElementNode) => {
1517          with_node_qname(&ch, |chtag| match chtag {
1518            "ltx:text" => {
1519              // Perl L1136-1137: $class .= 't' unless $class eq 't'
1520              // Only skip if the LAST class was also Text (not just first).
1521              // This preserves "tt" for cells with two text elements.
1522              if inferred_classes.last() != Some(&ColumnSpec::Text) {
1523                inferred_classes.push(ColumnSpec::Text);
1524              }
1525            },
1526            "ltx:graphics" => {
1527              if inferred_classes.first() != Some(&ColumnSpec::Graphics) {
1528                inferred_classes.push(ColumnSpec::Graphics);
1529              }
1530            },
1531            "ltx:Math" => {
1532              if inferred_classes.first() != Some(&ColumnSpec::Math) {
1533                inferred_classes.push(ColumnSpec::Math);
1534              }
1535            },
1536            "ltx:XMText" => {
1537              if inferred_classes.first() != Some(&ColumnSpec::Text) {
1538                inferred_classes.push(ColumnSpec::Text);
1539              }
1540            },
1541            "ltx:XMArg" | "ltx:inline-block" | "ltx:p" => {
1542              // Transparent containers: look through to classify children.
1543              // Perl's beAbsorbed creates <text> directly in td; Rust wraps in
1544              // <inline-block><p> from {turn}/{rotate}. Treat these as transparent
1545              // so the classification matches Perl's view of the cell content.
1546              let mut children = ch.get_child_nodes();
1547              children.append(&mut nodes);
1548              nodes = children;
1549            },
1550            other if other.starts_with("ltx:XM") => {
1551              if inferred_classes.first() != Some(&ColumnSpec::Math) {
1552                inferred_classes.push(ColumnSpec::Math);
1553              }
1554            },
1555            _ => {
1556              if inferred_classes.is_empty() {
1557                inferred_classes.push(ColumnSpec::Unknown);
1558              }
1559            },
1560          })
1561        },
1562        _ => {},
1563      }
1564    }
1565  }
1566
1567  // check if we have alternating math-and-text or text-and-math (only if 2+ classes)
1568  if inferred_classes.len() > 1 {
1569    let mut alt_peekable = inferred_classes.iter().peekable();
1570    let mut is_alternating = true;
1571    while let Some(c) = alt_peekable.next() {
1572      match c {
1573        ColumnSpec::Math | ColumnSpec::Integer => {
1574          if let Some(peek) = alt_peekable.peek() {
1575            if !matches!(peek, ColumnSpec::Text) {
1576              is_alternating = false;
1577              break;
1578            }
1579          }
1580        },
1581        ColumnSpec::Text => {
1582          if let Some(peek) = alt_peekable.peek() {
1583            if !matches!(peek, ColumnSpec::Math | ColumnSpec::Integer) {
1584              is_alternating = false;
1585              break;
1586            }
1587          }
1588        },
1589        _ => {
1590          is_alternating = false;
1591          break;
1592        },
1593      }
1594    }
1595    if is_alternating {
1596      inferred_classes = vec![ColumnSpec::MathAltText];
1597    }
1598  }
1599  // Default to empty and return
1600  if inferred_classes.is_empty() {
1601    ColumnSpec::Empty
1602  } else if inferred_classes.len() == 1 {
1603    inferred_classes[0]
1604  } else {
1605    // Perl L1151: multi-class detection.
1606    // "tt" (all Text) → MultiText, mixed math+text → MathAltText
1607    let all_text = inferred_classes
1608      .iter()
1609      .all(|c| matches!(c, ColumnSpec::Text));
1610    if all_text {
1611      ColumnSpec::MultiText
1612    } else {
1613      ColumnSpec::Unknown
1614    }
1615  }
1616}

1618//======================================================================
1619// Scan pairs of rows/columns attempting to recognize differences that
1620// might indicate which are headers and which are data.
1621// Warning: This section is full of "magic numbers"
1622// guessed by sampling various test cases.
1623
1624const MIN_ALIGNMENT_DATA_LINES: usize = 1; //  (or 2?) [CONSTANT]
1625const MAX_ALIGNMENT_HEADER_LINES: usize = 4; // [CONSTANT]
1626
1627// We expect to find header lines at the beginning, noticably different from the eventual data
1628// lines. Both header lines and data lines can consist of several neighboring lines.
1629// Check that header lines are `similar' to each other.  So, the strategy is to look
1630// for a `hump' in the line differences and consider blocks containing these lines to be potential
1631// headers.
1632
1633fn alignment_characterize_lines(
1634  document: &mut Document,
1635  axis: Axis,
1636  reversed: bool,
1637  lines: &mut [Vec<&mut Cell>],
1638) -> Result<()> {
1639  let n = lines.len();
1640  if n < 2 {
1641    return Ok(());
1642  }
1643  // eprintln!("Characterizing {n} {}", if axis == Axis::Row {"rows"} else {"columns"});
1644
1645  // Establish a scale of differences for the table.
1646  let (mut max_diff, mut min_diff, _avg_diff) = (0.0, 99999999.0, 0.0);
1647  for l in 0..n - 1 {
1648    let d = alignment_compare(axis, true, reversed, l, l + 1, lines);
1649    // eprintln!("  compare({l},{}) = {d}", l + 1);
1650    // avg_diff += d;
1651    if d > max_diff {
1652      max_diff = d;
1653    }
1654    if d < min_diff {
1655      min_diff = d;
1656    }
1657  }
1658  // avg_diff = avg_diff / (n - 1) as f64;
1659  if max_diff < 0.05 {
1660    // virtually no differences.
1661    return Ok(());
1662  }
1663  if (n > 2) && ((max_diff - min_diff) < max_diff * 0.5) {
1664    // differences too similar to establish pattern
1665    return Ok(());
1666  }
1667  let tab_threshold = min_diff + 0.3 * (max_diff - min_diff);
1668
1669  // eprintln!("Differences {min_diff} -- {max_diff} => threshold = {tab_threshold}");
1670  // Find the first hump in differences. These are candidates for header lines.
1671  // eprintln!("Scanning for headers");
1672  let (minh, mut maxh) = (1, 1);
1673  let mut diff;
1674  loop {
1675    diff = alignment_compare(axis, true, reversed, maxh - 1, maxh, lines);
1676    if diff >= tab_threshold {
1677      break;
1678    }
1679    maxh += 1;
1680  }
1681  if maxh > MAX_ALIGNMENT_HEADER_LINES {
1682    // too many before even finding diffs? give up!
1683    return Ok(());
1684  }
1685  while alignment_compare(axis, true, reversed, maxh, maxh + 1, lines) > tab_threshold {
1686    maxh += 1;
1687  }
1688  if maxh > MAX_ALIGNMENT_HEADER_LINES {
1689    maxh = MAX_ALIGNMENT_HEADER_LINES;
1690  }
1691  // eprintln!("Found from {minh}--{maxh} potential headers");
1692
1693  let nn = lines[0].len() - 1;
1694  // The sets of lines 1--$minh, .. 1--$maxh are potential headers.
1695  for nh in (minh..=maxh).rev() {
1696    // Check whether the set 1..$nh is plausable.
1697    let heads = alignment_test_headers(nh, tab_threshold, axis, lines);
1698    if !heads.is_empty() {
1699      // Now, change all cells marked as header from td => th.
1700      for h in heads {
1701        for (i, cell) in lines[h].iter_mut().enumerate() {
1702          cell.cell_type = Some('h');
1703          if let Some(ref mut xcell) = cell.cell {
1704            // But NOT empty cells on outer edges.
1705            // Perl: !$$cell{l} is falsy for both undef AND 0.
1706            if (cell.content_class == Some(ColumnSpec::Empty))
1707              && ((i == 0
1708                && (if axis == Axis::Row {
1709                  cell.border_left.unwrap_or(0) == 0
1710                } else {
1711                  cell.border_top.unwrap_or(0) == 0
1712                }))
1713                || (i == nn
1714                  && (if axis == Axis::Row {
1715                    cell.border_right.unwrap_or(0) == 0
1716                  } else {
1717                    cell.border_bottom.unwrap_or(0) == 0
1718                  })))
1719            {
1720            } else {
1721              document.add_ss_values(xcell, "thead", axis.marker_name())?;
1722            }
1723          }
1724        }
1725      }
1726      return Ok(());
1727    }
1728  }
1729  Ok(())
1730}

1732/// Test whether `nhead` lines makes a good fit for the headers
1733fn alignment_test_headers(
1734  nhead: usize,
1735  tab_threshold: f64,
1736  axis: Axis,
1737  lines: &[Vec<&mut Cell>],
1738) -> Vec<usize> {
1739  // eprintln!("Testing {nhead} headers with threshold {tab_threshold} for axis {:?}", axis);
1740  let mut heads: Vec<usize> = (0..nhead).collect(); // The indices of heading lines.
1741  let mut head_length = alignment_max_content_length(0, 0, nhead - 1, lines);
1742  let mut next_line = nhead; // Start from the end of the proposed headings.
1743
1744  // Watch out for the assumed header being really data that is a repeated pattern.
1745  let nrep = lines.len() / nhead;
1746  if nhead > 1 {
1747    //   Debug("Check for apparent header repeated $nrep times") if $LaTeXML::DEBUG{alignment};
1748    let mut matched = true;
1749    for r in 1..nrep {
1750      matched =
1751        matched && alignment_match_head(0, r * nhead, nhead, tab_threshold, axis, lines) > 0;
1752    }
1753    //   Debug("Repeated headers: " . ($matched ? "Matched=> Fail" : "Nomatch => Succeed"))
1754    //     if $LaTeXML::DEBUG{alignment};
1755    // eprintln!("  repeated pattern check: matched={matched}");
1756    if matched {
1757      return Vec::new();
1758    }
1759  }
1760
1761  // And find a following grouping of data lines.
1762  let ndata = alignment_skip_data(next_line, tab_threshold, axis, lines);
1763  // eprintln!("  ndata={ndata} from next_line={next_line}");
1764  if ndata < nhead {
1765    // ???? Well, maybe if _really_ convincing???
1766    return Vec::new();
1767  }
1768  if (ndata < nhead) && (ndata < 2) {
1769    return Vec::new();
1770  }
1771  // Check that the content of the headers isn't dramatically larger than the content in the data
1772  let mut data_length = alignment_max_content_length(0, next_line, next_line + ndata - 1, lines);
1773  next_line += ndata;
1774
1775  let mut nd;
1776  // If there are more lines, they should match either the previous data block, or the head/data
1777  // pattern.
1778  while next_line < lines.len() {
1779    // First try to match a repeat of the 1st data block;
1780    // This would be the case when groups of data have borders around them.
1781    // Could want to match a variable number of datalines, but they should be similar!!!??!?!?
1782    nd = if ndata > 1 {
1783      alignment_match_data(nhead, next_line, ndata, tab_threshold, axis, lines)
1784    } else {
1785      0
1786    };
1787    // eprintln!("  while: next_line={next_line}, nd={nd}");
1788    if nd > 0 {
1789      data_length = alignment_max_content_length(data_length, next_line, next_line + nd - 1, lines);
1790      next_line += nd;
1791    }
1792    // Else, try to match the first header block; less common.
1793    else if alignment_match_head(0, next_line, nhead, tab_threshold, axis, lines) > 0 {
1794      // eprintln!("  matched head at next_line={next_line}");
1795      for idx in next_line..next_line + nhead {
1796        heads.push(idx);
1797      }
1798      head_length =
1799        alignment_max_content_length(head_length, next_line, next_line + nhead - 1, lines);
1800      next_line += nhead;
1801      nd = alignment_match_data(nhead, next_line, ndata, tab_threshold, axis, lines);
1802      if nd == 0 {
1803        return Vec::new();
1804      }
1805      data_length = alignment_max_content_length(data_length, next_line, next_line + nd - 1, lines);
1806      next_line += nd;
1807    } else {
1808      // eprintln!("  no match at next_line={next_line} => fail");
1809      return Vec::new();
1810    }
1811  }
1812  // Header content seems too large relative to data?
1813  // eprintln!("  header content = {head_length}; data content = {data_length}");
1814  if (head_length > 10) && (head_length > 4 * data_length) {
1815    //   Debug("header content too much longer than data content")
1816    //     if $LaTeXML::DEBUG{alignment};
1817    return Vec::new();
1818  }
1819  // Or if a header cell has "large" content?
1820  if head_length >= 1000 {
1821    // Or if a header cell has "large" content?
1822    //   Debug("header content too large")
1823    //     if $LaTeXML::DEBUG{alignment};
1824    return Vec::new();
1825  }
1826
1827  // eprintln!("  Succeeded with {nhead} headers: {heads:?}");
1828  heads
1829}

1831fn alignment_match_head(
1832  p1: usize,
1833  p2: usize,
1834  nhead: usize,
1835  tab_threshold: f64,
1836  axis: Axis,
1837  tablines: &[Vec<&mut Cell>],
1838) -> usize {
1839  let nh = alignment_match_lines(p1, p2, nhead, tab_threshold, axis, tablines);
1840  let ok = nhead == nh;
1841  // Debug("Matched $nh header lines => " . ($ok ? "Succeed" : "Failed")) if
1842  // $LaTeXML::DEBUG{alignment};
1843  if ok { nhead } else { 0 }
1844}

1846fn alignment_match_data(
1847  p1: usize,
1848  p2: usize,
1849  n: usize,
1850  tab_threshold: f64,
1851  axis: Axis,
1852  tablines: &[Vec<&mut Cell>],
1853) -> usize {
1854  let nd = alignment_match_lines(p1, p2, n, tab_threshold, axis, tablines);
1855  let ok = (nd as f64 * 1.0) / n as f64 > 0.66;
1856  //   Debug("Matched $nd data lines => " . ($ok ? "Succeed" : "Failed"))
1857  //     if $LaTeXML::DEBUG{alignment};
1858  if ok { nd } else { 0 }
1859}

1861// Match the $n lines starting at $i2 to those starting at $i1.
1862fn alignment_match_lines(
1863  p1: usize,
1864  p2: usize,
1865  n: usize,
1866  tab_threshold: f64,
1867  axis: Axis,
1868  tablines: &[Vec<&mut Cell>],
1869) -> usize {
1870  let max_n = tablines.len();
1871  for i in 0..n {
1872    if (p1 + i >= max_n)
1873      || (p2 + i >= max_n)
1874      || alignment_compare(axis, false, false, p1 + i, p2 + i, tablines) >= tab_threshold
1875    {
1876      return i;
1877    }
1878  }
1879  n
1880}

1882/// Skip through a block of lines starting at $i that appear to be data, returning the number of
1883/// lines. We'll assume the 1st line is data, compare it to following lines,
1884/// but also accept `continuation' data lines.
1885///
1886/// Note: Perl's continuation-line logic (L1336-1339) is effectively dead code:
1887/// `scalar($::TABLINES[0])` evaluates to an array ref's memory address (huge number),
1888/// making `0.4 * huge` very large, so `count_empty <= huge` is always true.
1889/// The condition `($n < 2) || true` = true, so the `last if` simplifies to just
1890/// `last if diff >= threshold`. We match this behavior.
1891fn alignment_skip_data(
1892  i: usize,
1893  tab_threshold: f64,
1894  axis: Axis,
1895  tablines: &[Vec<&mut Cell>],
1896) -> usize {
1897  let tab_lines_length = tablines.len();
1898  if i >= tab_lines_length {
1899    return 0;
1900  }
1901  let _header_width = if !tablines.is_empty() {
1902    tablines[0].len()
1903  } else {
1904    1
1905  };
1906  let mut n = 1;
1907  while i + n < tab_lines_length {
1908    if alignment_compare(axis, true, false, i + n - 1, i + n, tablines) >= tab_threshold {
1909      // TODO: Perl Alignment.pm L1337-1339 has continuation line check here.
1910      // Applying it changes behavior for fonts/bbold tables (false positive headers).
1911      // Need to investigate further.
1912      break;
1913    }
1914    n += 1;
1915  }
1916  if n >= MIN_ALIGNMENT_DATA_LINES { n } else { 0 }
1917}

1919/// Return the maximum "content length" for lines from $from to $to.
1920fn alignment_max_content_length(
1921  mut length: usize,
1922  from: usize,
1923  to: usize,
1924  tablines: &[Vec<&mut Cell>],
1925) -> usize {
1926  for item in tablines.iter().take(to + 1).skip(from) {
1927    let mut l = 0;
1928    for cell in item.iter() {
1929      l += cell.content_length.unwrap_or(0);
1930    }
1931    if l > length {
1932      length = l;
1933    }
1934  }
1935  length
1936}

1940/// Compare two lines along `Axis` (0=row,1=column), returning a measure of the difference.
1941/// The borders are compared differently if
1942///  `for_adjacency`: we adjacent lines that might belong to the same block,
1943///  otherwise    : comparing two lines that ought to have identical patterns (eg. in a repeated
1944/// block)
1945fn alignment_compare(
1946  axis: Axis,
1947  for_adjacency: bool,
1948  reversed: bool,
1949  p1: usize,
1950  p2: usize,
1951  lines: &[Vec<&mut Cell>],
1952) -> f64 {
1953  let max_guard = lines.len();
1954  if p1 >= max_guard || p2 >= max_guard {
1955    return 0.0;
1956  }
1957  let line1 = &lines[p1];
1958  let line2 = &lines[p2];
1959  if line1.is_empty() && line2.is_empty() {
1960    return 0.0;
1961  } else if line1.is_empty() || line2.is_empty() {
1962    return 99999.0;
1963  }
1964  let ncells = line1.len();
1965  let mut diff = 0.0;
1966
1967  for (cell1, cell2) in line1.iter().zip(line2.iter()) {
1968    // Annoying test avoids warnings if cells inconsistent; likely due to incorrect row/col spans
1969    if cell1.content_class.is_none()
1970      || cell2.content_class.is_none()
1971      || cell1.border_left.is_none()
1972      || cell2.border_left.is_none()
1973      || cell1.border_right.is_none()
1974      || cell2.border_right.is_none()
1975      || cell1.border_bottom.is_none()
1976      || cell2.border_bottom.is_none()
1977      || cell1.border_top.is_none()
1978      || cell2.border_top.is_none()
1979    {
1980      continue;
1981    }
1982    if cell1.align != cell2.align
1983      && cell1.content_class != Some(ColumnSpec::Empty)
1984      && cell2.content_class != Some(ColumnSpec::Empty)
1985    {
1986      diff += 0.75;
1987    }
1988    let d = cell1
1989      .content_class
1990      .as_ref()
1991      .unwrap()
1992      .difference_heuristic(cell2.content_class.as_ref().unwrap());
1993    if d > 0.0 {
1994      diff += d;
1995    }
1996    // compare certain edges
1997    if for_adjacency {
1998      // Compare edges for adjacent rows of potentially different purpose
1999      let mut inner_diffs = 0.0;
2000      if axis == Axis::Row {
2001        if cell1.border_right != cell2.border_right {
2002          inner_diffs += 1.0;
2003        }
2004        if cell1.border_left != cell2.border_left {
2005          inner_diffs += 1.0;
2006        }
2007      } else {
2008        if cell1.border_top != cell2.border_top {
2009          inner_diffs += 1.0;
2010        }
2011        if cell1.border_bottom != cell2.border_bottom {
2012          inner_diffs += 1.0;
2013        }
2014      };
2015      diff += 0.3 * inner_diffs;
2016      // Penalty for apparent divider between.
2017      let pedge = if axis == Axis::Row {
2018        if reversed {
2019          BorderSpec::Top
2020        } else {
2021          BorderSpec::Bottom
2022        }
2023      } else if reversed {
2024        BorderSpec::Left
2025      } else {
2026        BorderSpec::Right
2027      };
2028      let border1_pedge = cell1.border_at(pedge);
2029      let border2_pedge = cell2.border_at(pedge);
2030      if let Some(b1p) = border1_pedge {
2031        if b1p > 0 && (border1_pedge != border2_pedge) {
2032          diff += (b1p as i64 - border2_pedge.unwrap_or(0) as i64).abs() as f64;
2033        }
2034      }
2035    } else {
2036      // Compare edges for rows from diff places for potential similarity
2037      let mut inner_diffs = 0.0;
2038      if cell1.border_right != cell2.border_right {
2039        inner_diffs += 1.0;
2040      }
2041      if cell1.border_left != cell2.border_left {
2042        inner_diffs += 1.0;
2043      }
2044      if cell1.border_top != cell2.border_top {
2045        inner_diffs += 1.0;
2046      }
2047      if cell1.border_bottom != cell2.border_bottom {
2048        inner_diffs += 1.0;
2049      }
2050      diff += 0.3 * inner_diffs;
2051    }
2052  }
2053  diff /= ncells as f64;
2054  // eprintln!("alignment_compare: {p1} - {p2} => {diff};");
2055  // Debug("$p1-$p2 => $diff; ") if $LaTeXML::DEBUG{alignment};
2056  diff
2057}
