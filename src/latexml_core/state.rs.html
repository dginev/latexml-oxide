72/// installation scope in the state_tables
73#[derive(Debug, Clone, Copy, PartialEq, Eq)]
74pub enum Scope {
75  /// globally visible, does not expire
76  Global,
77  /// globally visible, but expires at the end of the current group
78  Local,
79  /// a named scope - visible only when explicitly activated
80  Named(SymStr),
81  /// in-place: replace the value in the frame it was last bound in, WITHOUT
82  /// recording a new undo entry (or globally, at the locked base frame, if it
83  /// was never bound). Perl `State.pm:175` `$scope eq 'inplace'` ("Special case
84  /// for `\box` & friends"). This is Knuth's "same level" reassignment / what
85  /// @xworld21 tentatively called `scope => 'definition'`: the binding keeps its
86  /// save-stack level, so a mutation rides exactly as long as the current
87  /// binding — persisting past the current group if the binding was made above
88  /// it, reverting with the group if it was made locally. Distinct from Global
89  /// (which promotes + wipes lower-frame undo) and Local (which pushes an outer
90  /// binding down a level). The Value-table fast path is `assign_value_inplace`.
91  InPlace,
92}

456impl State {
457  pub fn new(options: StateOptions) -> Self {
458    use crate::token::Catcode::*;
459
460    // Setup default catcodes.
461    let catcode_profile = match options.catcodes {
462      None => Catcodes::Standard,
463      Some(cp) => cp,
464    };
465
466    let mut catcodes: HashMap<char, Catcode> = HashMap::default();
467    match catcode_profile {
468      Catcodes::Standard | Catcodes::Style => {
469        catcodes.insert('\\', ESCAPE);
470        catcodes.insert('{', BEGIN);
471        catcodes.insert('}', END);
472        catcodes.insert('$', MATH);
473        catcodes.insert('&', ALIGN);
474        catcodes.insert('\r', EOL);
475        catcodes.insert('#', PARAM);
476        catcodes.insert('^', SUPER);
477        catcodes.insert('_', SUB);
478        catcodes.insert(' ', SPACE);
479        catcodes.insert('\t', SPACE);
480        catcodes.insert('%', COMMENT);
481        catcodes.insert('~', ACTIVE);
482        // NUL (`\^^@`, U+0000): Perl LaTeXML's default is catcode 12 (OTHER),
483        // NOT the TeXbook's 9 (IGNORE). We follow Perl (ground truth) so that
484        // `\^^@`/`` `^^@ `` reads code 0 (TeXbook 9 would *drop* the NUL token,
485        // making `` `^^@ `` skip to the next token — `\relax` etc. — and return
486        // a bogus code; xint's `\romannumeral`&&@` expansion idiom needs 0).
487        // Real-world bbl files (e.g. astro-ph0004127's spie4012-01a.bbl) carry
488        // stray NULs from BibTeX `\"u`-mangling; as OTHER they become harmless
489        // literal chars (stripped at XML serialization), matching Perl —
490        // crucially NOT ESCAPE, so no bogus `\uninger`-style CS forms. An
491        // explicit `\catcode`^^Q=9` (user/package) is still honored; only the
492        // *default* changes.
493        catcodes.insert('\0', OTHER);
494        catcodes.insert('\u{000c}', ACTIVE);
495        for c in "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ".chars() {
496          catcodes.insert(c, LETTER);
497        }
498      },
499      Catcodes::None => {},
500    }
501    if catcode_profile == Catcodes::Style {
502      catcodes.insert('@', LETTER);
503    }
504
505    let mut value_table = HashMap::default();
506    let mut specials_vdq = VecDeque::new();
507    specials_vdq.push_front(Stored::Chars(Box::new([
508      '^', '_', '~', '&', '$', '#', '\'',
509    ])));
510    value_table.insert(arena::pin_static("SPECIALS"), specials_vdq);
511
512    let mut catcodes_typed: Table = HashMap::default();
513    for (k, v) in catcodes {
514      let mut vdq = VecDeque::new();
515      vdq.push_front(Stored::Catcode(v));
516      catcodes_typed.insert(arena::pin_char(k), vdq);
517    }
518
519    // Basic defaults
520    if let Some(model) = options.model {
521      model::set_model(model);
522    };
523    let verbosity = options.verbosity.unwrap_or(0);
524    // let strict = options.strict.unwrap_or(false);
525    // INCLUDE_COMMENTS: Perl defaults to true (Core.pm L143).
526    // T_COMMENT tokens are now properly converted to XML comment nodes
527    // via Document::insert_comment using raw libxml2 FFI.
528    // Note: Only set when explicitly specified, because STY_STATE/STD_STATE
529    // use default options and state rotation (swap) would overwrite the
530    // main state's INCLUDE_COMMENTS value.
531    let include_comments = options.include_comments;
532    // let include_styles = options.include_styles.unwrap_or(false);
533    let nomathparse = options.nomathparse.unwrap_or(false);
534    let source_map = options.source_map.unwrap_or(false);
535
536    // Perl Core.pm L50-52: `SEARCHPATHS => [map { pathname_absolute(...) } '.',
537    // @searchpaths]` — the current working directory is ALWAYS searched, and
538    // FIRST (it takes precedence over the `--path` dirs). A `--path` entry
539    // ending in `//` is the kpsewhich recursive-search marker; preserve it
540    // across canonicalization (strip it, canonicalize the base, re-append) so
541    // `candidate_pathnames` can expand the marker to the whole subtree.
542    let search_paths: VecDeque<String> = std::iter::once(String::from("."))
543      .chain(options.search_paths.into_iter().flatten())
544      .map(|p| match p.strip_suffix("//") {
545        Some(base) => format!("{}//", pathname::absolute(&pathname::canonical(base))),
546        None => pathname::absolute(&pathname::canonical(&p)),
547      })
548      .collect();
549    let graphics_paths = match options.graphics_paths {
550      None => VecDeque::new(),
551      Some(paths) => paths
552        .iter()
553        .map(|p| pathname::absolute(&pathname::canonical(p)))
554        .collect(),
555    };
556
557    let mut state = State {
558      value: value_table,
559      catcode: catcodes_typed,
560      verbosity,
561      // strict,
562      // include_comments,
563      search_paths,
564      graphics_paths,
565      // include_styles,
566      input_encoding: options.input_encoding,
567      nomathparse,
568      source_map,
569      ..State::default()
570    };
571    // INITEX-equivalent defaults — mirror Perl `State.pm:128-137`.
572    // Sets letter/digit mathcodes (class 7, family 1 for letters / 0 for digits),
573    // upper/lowercase mappings, and sfcode=999 for uppercase letters. Without
574    // these, dump-load path leaves letter mathcodes unset (plain.dump.txt only
575    // captures the 57 plain.tex OVERRIDES), so `\cal abc` math falls through
576    // the text path and loses meaning/role attributes. NODUMP path used to set
577    // these via plain_base.rs L17-41 only — not Perl-faithful since INITEX
578    // owns these (TeXbook ch.17 p309). Setting them here makes both paths
579    // consistent and matches Perl's State::new behaviour.
580    for c in b'0'..=b'9' {
581      state.assign_internal(
582        TableName::Mathcode,
583        arena::pin_char(c as char),
584        Stored::Charcode(0x7000 + c as u16),
585        None,
586      );
587    }
588    for c in b'a'..=b'z' {
589      let big = c - 32;
590      state.assign_internal(
591        TableName::Mathcode,
592        arena::pin_char(c as char),
593        Stored::Charcode(0x7100 + c as u16),
594        None,
595      );
596      state.assign_internal(
597        TableName::Mathcode,
598        arena::pin_char(big as char),
599        Stored::Charcode(0x7100 + big as u16),

601      );
602      state.assign_internal(
603        TableName::Uccode,
604        arena::pin_char(c as char),

606        None,
607      );
608      state.assign_internal(
609        TableName::Lccode,
610        arena::pin_char(big as char),
611        Stored::Charcode(c as u16),
612        None,
613      );
614      state.assign_internal(
615        TableName::Sfcode,
616        arena::pin_char(big as char),
617        Stored::Charcode(999),
618        None,
619      );
620    }
621    // TODO: should these be *fields* in state or really as in Perl - globally assigned values?
622    state.assign_value(
623      "DOCUMENTID",
624      options.documentid.unwrap_or_default(),
625      Some(Scope::Global),
626    );
627    // Perl Core.pm L143: assignValue(INCLUDE_COMMENTS => ..., 'global')
628    // Only set when explicitly specified (not for STY_STATE/STD_STATE defaults)
629    if let Some(ic) = include_comments {
630      state.assign_value("INCLUDE_COMMENTS", ic, Some(Scope::Global));
631    }
632    // Perl Core.pm L47: INCLUDE_PATH_PIS (default true) — emit searchpath PIs
633    state.assign_value("INCLUDE_PATH_PIS", true, Some(Scope::Global));
634    // Perl Core.pm L43: STRICT (default false)
635    if let Some(strict) = options.strict {
636      state.assign_value("STRICT", strict, Some(Scope::Global));
637    }

639    // and INCLUDE_CLASSES (both default 0), so it raw-loads .sty packages AND
640    // .cls classes from the search path. Faithful to Perl — its own
641    // `# … accept both classes and styles?` hedge notwithstanding, the code
642    // sets both, and INCLUDE_CLASSES is what unlocks raw .cls in LoadClass.
643    if let Some(include_styles) = options.include_styles {
644      state.assign_value("INCLUDE_STYLES", include_styles, Some(Scope::Global));
645      state.assign_value("INCLUDE_CLASSES", include_styles, Some(Scope::Global));
646    }
647    // Perl Core.pm L62: NOMATHPARSE
648    state.assign_value("NOMATHPARSE", nomathparse, Some(Scope::Global));
649    // Perl Core.pm L61: PERL_INPUT_ENCODING (default utf-8)
650    let enc = state.input_encoding.as_deref().unwrap_or("utf-8");
651    state.assign_value(
652      "PERL_INPUT_ENCODING",
653      Stored::String(arena::pin(enc)),
654      Some(Scope::Global),
655    );
656
657    // Perl Core.pm L53: $state->assignValue(GRAPHICSPATHS => [map {…} @{$opts{graphicspaths}}])
658    // Mirror with a VecDequeStored of String entries; subsequent push/unshift
659    // operations in `\graphicspath`, `\svgpath`, and Core.pm-equivalent source
660    // directory prepends will append/prepend to this same list.
661    if !state.graphics_paths.is_empty() {
662      let vdq: VecDeque<Stored> = state
663        .graphics_paths
664        .iter()
665        .map(|p| Stored::String(arena::pin(p)))
666        .collect();
667      state.assign_internal(
668        TableName::Value,
669        arena::pin("GRAPHICSPATHS"),
670        Stored::VecDequeStored(vdq),
671        Some(Scope::Global),
672      );
673    }
674
675    state
676  }

711  /// Perl DumpFile equivalent: Take a snapshot of the current state.
712  /// Returns a HashMap mapping (table_name, key) → Stored value.
713  /// Only captures the front (current) value of each key.
714  /// Used before processing latex.ltx to diff what changed.
715  pub fn snapshot(&self) -> rustc_hash::FxHashMap<(TableName, SymStr), Stored> {
716    let tables = [
717      TableName::Value,
718      TableName::Meaning,
719      TableName::Catcode,
720      TableName::Mathcode,
721      TableName::Sfcode,
722      TableName::Lccode,
723      TableName::Uccode,
724      TableName::Delcode,
725    ];
726    let mut snap = rustc_hash::FxHashMap::default();
727    for &tname in &tables {
728      let table = self.table(tname);
729      for (key, values) in table {
730        if let Some(front) = values.front() {
731          snap.insert((tname, *key), front.clone());
732        }
733      }
734    }
735    snap
736  }

738  /// Perl DumpFile equivalent: Compute the diff between current state and a snapshot.
739  /// Returns only entries that CHANGED since the snapshot was taken.
740  /// Skips entries that contain closures (Primitive, Constructor, Conditional)
741  /// since those can't be serialized — they come from Rust engine code.
742  pub fn diff_from_snapshot(
743    &self,
744    snap: &rustc_hash::FxHashMap<(TableName, SymStr), Stored>,
745  ) -> Vec<(TableName, SymStr, Stored)> {
746    let tables = [
747      TableName::Value,
748      TableName::Meaning,
749      TableName::Catcode,
750      TableName::Mathcode,
751      TableName::Sfcode,
752      TableName::Lccode,
753      TableName::Uccode,
754      TableName::Delcode,
755    ];
756    let mut diff = Vec::new();
757    for &tname in &tables {
758      let table = self.table(tname);
759      for (key, values) in table {
760        if let Some(current) = values.front() {
761          let key_pair = (tname, *key);
762          let changed = match snap.get(&key_pair) {
763            None => true, // new entry
764            Some(prev) => {
765              // Compare string representations (cheap approximation of Perl's dump-based diff)
766              format!("{:?}", current) != format!("{:?}", prev)
767            },
768          };
769          if changed && is_serializable(current) {
770            diff.push((tname, *key, current.clone()));
771          }
772        }
773      }
774    }
775    diff
776  }

787  pub(crate) fn assign_internal(
788    &mut self,
789    table_name: TableName,
790    key: SymStr,
791    value: Stored,
792    mut scope_opt: Option<Scope>,
793  ) {
794    // hotcode lookupDefinition for \globaldefs,
795    // since this is called extremely often and should be highly standardized.
796    // TeX semantics: positive → all assignments global, negative → \global
797    // ignored, zero → no override. `\globaldefs` is a Number register, so the
798    // stored variant is `Stored::Number`, NOT `Stored::Int` — Perl's `==`
799    // coerces both, Rust must unwrap explicitly. Perl `State.pm:144-151` uses
800    // strict `==1`/`==-1`; we slightly broaden to TeX's sign-based rule
801    // (matches behavior for the canonical `\globaldefs=1`/`\globaldefs=-1`
802    // uses while also handling rare `\globaldefs=2` etc).
803    // `Scope::Named(_)` and `Scope::InPlace` are preserved per Perl's "ONLY
804    // override global/local/undef" rule (State.pm:146 — `$scope ne 'global' &&
805    // $scope ne 'local'` short-circuits the override): an in-place mutation of
806    // the existing binding is NOT re-scoped by `\globaldefs`.
807    // Without this: pgfplots' `\pgfplots@pop@next@legend`
808    // (`\def\foo{{\globaldefs=1 \let\x=\relax}}`) silently drops the `\let`
809    // on group exit, leaving `\pgfplots@curlegend`/`@curplotlist` undefined
810    // and looping `\pgfplots@createlegend` at the digest wall-clock cap.
811    let preserve = matches!(scope_opt, Some(Scope::Named(_) | Scope::InPlace));
812    if !preserve
813      && let Some(globaldefs) = self.value.get(&pin!("\\globaldefs"))
814      && let Some(global_value) = globaldefs.front()
815    {
816      let int_value: i64 = match *global_value {
817        Stored::Int(v) => v,
818        Stored::Number(n) => n.0,
819        _ => 0,
820      };
821      if int_value > 0 {
822        scope_opt = Some(Scope::Global);
823      } else if int_value < 0 {
824        scope_opt = Some(Scope::Local);
825      }
826    }
827    // TRACE: watch for cleanup:w
828    // regular check, local scope is default, unless a global prefix is set
829    let scope = match scope_opt {
830      Some(s) => s,
831      None => {
832        if self.get_prefix("global") {
833          Scope::Global
834        } else {
835          Scope::Local
836        }
837      },
838    };
839    match scope {
840      Scope::Global => {
841        let mut undo_count = 0;
842
843        // Remove bindings made in all frames down-to & including the next lower locked frame
844        let mut last_frame = None;
845        for frame in &mut self.undo {
846          let is_locked = frame.locked;
847          let frame_table = frame.table_mut(table_name);
848          if let Some(n) = frame_table.remove(&key) {
849            undo_count += n;
850          }
851          last_frame = Some(frame);
852          if is_locked {
853            break;
854          }
855        }
856        // whatever is left -- if anything -- should be bindings below the locked frame.
857        if let Some(frame) = last_frame {
858          frame.table_mut(table_name).insert(key, 1); // Note that there's only one
859          // value in the stack, now
860        }
861
862        // Undo the bindings, if `key` was bound in this frame
863        let state_table = self.table_mut(table_name);
864        if let Some(defs) = state_table.get_mut(&key) {
865          for _ in 1..=undo_count {
866            defs.pop_front();
867          }
868        }
869
870        let table_entry = state_table.entry(key).or_default();
871        table_entry.push_front(value);
872      },
873      Scope::Local => {
874        // Again, split the logic as 1) bookkeeping in undo, then 2) operations in state_tables
875        let mut is_replace = false;
876        // 1. Undo mutable logic
877        if let Some(current_frame) = self.undo.front_mut() {
878          let current_frame_table = current_frame.table_mut(table_name);
879
880          is_replace = current_frame_table.get(&key).unwrap_or(&0) > &0;
881          if is_replace { // If the value was previously assigned in this frame
882            // we do this in 2.1, then proceed to 2.2
883          } else {
884            // Otherwise, push new value & set 1 to be undone
885            current_frame_table.insert(key, 1);
886            //  And push new binding in 2.2
887          }
888        }
889        // 2. state_table mutable logic
890        let state_table = self.table_mut(table_name);
891        let defs = state_table.entry(key).or_default();
892        if is_replace {
893          // 2.1. Replace the value, i.e. remove existing one
894          defs.pop_front();
895        }
896        // 2.2 Add new value
897        defs.push_front(value);
898      },
899      Scope::InPlace => {
900        // Perl `State.pm:175`: replace the front value in the frame it was last
901        // bound in, adding NO undo entry, so the mutation keeps the binding's
902        // save-stack level (the `\box` / same-level semantics). If the key was
903        // never bound, seed it at the locked base frame (Perl's "push globally"
904        // fallback). Mirrors the Value-table `assign_value_inplace` fast path.
905        let state_table = self.table_mut(table_name);
906        if let Some(defs) = state_table.get_mut(&key)
907          && let Some(front) = defs.front_mut()
908        {
909          *front = value;
910        } else {
911          state_table.entry(key).or_default().push_front(value);
912          for frame in &mut self.undo {
913            if frame.locked {
914              frame.table_mut(table_name).insert(key, 1);
915              break;
916            }
917          }
918        }
919      },
920      Scope::Named(scope_name) => {
921        // initialize stash if empty
922        let needs_init = match self.stash.get(&scope_name) {
923          None => true,
924          Some(v) => v.is_empty(),
925        };
926        if needs_init {
927          self.assign_internal(
928            TableName::Stash,
929            scope_name,
930            Stored::Stash(Vec::new()),
931            Some(Scope::Global),
932          );
933        }
934        if let Some(Stored::Stash(stash)) =
935          self.stash.get_mut(&scope_name).as_mut().unwrap().get_mut(0)
936        {
937          stash.push((table_name, key, value.clone()));
938        }
939        let has_active = match self.stash_active.get(&scope_name) {
940          None => false,
941          Some(v) => !v.is_empty(),
942        };
943        if has_active {
944          self.assign_internal(table_name, key, value, Some(Scope::Local));
945        }
946      },
947    }
948  }

1177          "ignore",

1188/// Generate a stub definition for an undefined control-sequence,
1189/// along with appropriate error messge.
1190pub fn generate_error_stub(token: &Token) -> Result<Token> {
1191  let cs = token.with_cs_name(ToString::to_string);
1192  // Perl-faithful counter leniency. A `\c@<ctr>` control sequence is, by
1193  // LaTeX convention, the count register backing counter `<ctr>`. When code
1194  // reads an *undefined* one in a number/register context (e.g.
1195  // `\setcounter{x}{\value{y}}` or `\algrestore`/`\ContinuedFloat` reading
1196  // `\c@subalgorithm@save`), Perl does NOT raise a hard "undefined control
1197  // sequence" error — its counter machinery warns "Counter '<ctr>' was not
1198  // defined; assuming 0" (Package.pm L712) and treats it as 0. Without this,
1199  // `read_x_token` expands the bare undefined `\c@<ctr>` through the generic
1200  // <ltx:ERROR/> path below and the run gains a spurious error. Mirror Perl:
1201  // warn and define the register as 0 so the reader sees a register value,
1202  // not an undefined CS. Same category/message as `counter::dialect::
1203  // counter_value`. Witness 1910.02851 (`\algrestore{RLZFactorization}` +
1204  // `\ContinuedFloat` → `\c@subalgorithm@save`); Perl rc=0.
1205  if let Some(ctr) = cs.strip_prefix("\\c@") {
1206    if !lookup_bool("SUPPRESS_UNDEFINED_ERRORS") {
1207      Warn!(
1208        "undefined",
1209        ctr,
1210        s!("Counter '{}' was not defined; assuming 0", ctr)
1211      );
1212    }
1213    crate::binding::def::dialect::def_register(*token, None, Number::new(0), None)?;
1214    return Ok(*token);
1215  }
1216  // Gate the undefined-CS summary tally by SUPPRESS_UNDEFINED_ERRORS so it
1217  // matches the `Error!` gate at L1021 below — during expl3-code.tex raw
1218  // load with thousands of forward-references we install the ERROR stub
1219  // without polluting the user-facing summary count. See
1220  // project_kernel_dump_parity.md "iow_wrap residual" for full diagnosis.
1221  if !lookup_bool("SUPPRESS_UNDEFINED_ERRORS") {
1222    note_status(LogStatus::Undefined, Some(&cs));
1223  }
1224  // To minimize chatter, go ahead and define it...
1225  if cs.starts_with("\\if") {
1226    // Apparently an \ifsomething ???
1227    let name = cs.replace("\\if", "");
1228    // Perl `generateErrorStub` (State.pm L539-540) passes the recovery note
1229    // as a SEPARATE Error detail, so `generateMessage` renders it on its own
1230    // indented line — not merged into the primary message. Match that (and
1231    // the already-correct stomach.rs path) so the cortex `details`/log first
1232    // line is just "...is not defined." like Perl.
1233    Error!(
1234      "undefined",
1235      token,
1236      s!("The token {} is not defined.", token.stringify()),
1237      "Defining it now as with \\newif"
1238    );
1239    install_definition(
1240      Expandable::new(
1241        T_CS!(s!("\\{}true", name)),
1242        None,
1243        Some(s!("\\let{}\\iftrue", cs).into()),
1244        None,
1245      )?,
1246      Some(Scope::Global),
1247    );
1248    install_definition(
1249      Expandable::new(
1250        T_CS!(s!("\\{}false", name)),
1251        None,
1252        Some(s!("\\let{}\\iffalse", cs).into()),
1253        None,
1254      )?,
1255      Some(Scope::Global),
1256    );
1257    let_i(token, &T_CS!("\\iffalse"), Some(Scope::Global));
1258  } else {
1259    // Allow suppression of undefined errors during bulk loading (e.g., expl3-code.tex)
1260    // where forward references are later resolved by post-load fixups.
1261    if !lookup_bool("SUPPRESS_UNDEFINED_ERRORS") {
1262      Error!(
1263        "undefined",
1264        token,
1265        s!("The token {} is not defined.", token.stringify()),
1266        "Defining it now as <ltx:ERROR/>"
1267      );
1268    }
1269    install_definition(
1270      Constructor {
1271        cs: *token,
1272        replacement: Some(Rc::new(move |document, _args, _props| {
1273          document.make_error("undefined", &cs)
1274        })),
1275        ..Constructor::default()
1276      },
1277      //TODO: sizer => "X"),
1278      Some(Scope::Global),
1279    );
1280  }
1281  Ok(*token)
1282}

1338/// Sym-keyed variant of `assign_value_inplace` — skip the per-call
1339/// `arena::pin(key)` for hot callers with a pre-pinned SymStr.
1340pub fn assign_value_inplace_sym(key_sym: SymStr, value: impl Into<Stored>) {
1341  let value = value.into();
1342  let state = &mut *state_mut!();
1343  let table = &mut state.value;
1344  if let Some(vvec) = table.get_mut(&key_sym)
1345    && let Some(front) = vvec.front_mut()
1346  {
1347    *front = value;
1348    return;
1349  }
1350  // If the value was never assigned, push globally (matching Perl behavior)
1351  let vvec = table.entry(key_sym).or_default();
1352  vvec.push_front(value);
1353  // Find the locked frame and record the undo there
1354  for frame in &mut state.undo {
1355    if frame.locked {
1356      frame.table_mut(TableName::Value).insert(key_sym, 1);
1357      break;
1358    }
1359  }
1360}

1407/// manage a (global) list of values
1408pub fn push_value<T: Into<Stored>>(key: &str, value: T) -> Result<()> {
1409  let key_sym = arena::pin(key);
1410  let value = value.into();
1411  // Capture any BUG-path message, but raise the Error! *after* the state_mut!()
1412  // borrow is dropped — Error! reads MAX_ERRORS, and a live mutable borrow there
1413  // panics "RefCell already mutably borrowed" (tikz-cd 2001.08973).
1414  let bug: Option<String> = {
1415    let mut state = state_mut!();
1416    if !state.value.contains_key(&key_sym) {
1417      state.assign_internal(
1418        TableName::Value,
1419        key_sym,
1420        Stored::VecDequeStored(VecDeque::new()),
1421        Some(Scope::Global),
1422      );
1423    }
1424    match state.value.get_mut(&key_sym).unwrap().front_mut() {
1425      Some(&mut Stored::VecDequeStored(ref mut front)) => {
1426        front.push_back(value);
1427        None
1428      },
1429      // auto-vivify, if None
1430      Some(ref mut field) if matches!(field, Stored::None) => {
1431        let mut new_vdq = VecDeque::new();
1432        new_vdq.push_back(value);
1433        **field = Stored::VecDequeStored(new_vdq);
1434        None
1435      },
1436      // Convert Strings (immutable array) to VecDequeStored for push — matches Perl auto-vivification
1437      Some(ref mut field) if matches!(field, Stored::Strings(_)) => {
1438        let existing: VecDeque<Stored> = if let Stored::Strings(strings) = &**field {
1439          strings.iter().map(|s| Stored::String(*s)).collect()
1440        } else {
1441          VecDeque::new()
1442        };
1443        let mut new_vdq = existing;
1444        new_vdq.push_back(value);
1445        **field = Stored::VecDequeStored(new_vdq);
1446        None
1447      },
1448      other => Some(s!(
1449        "BUG: Tried to push_value into an unsupported Stored field! Field was: {other:?}"
1450      )),
1451    }
1452  };
1453  if let Some(message) = bug {
1454    // Lowercase category for consistency with engine convention.
1455    Error!("state", "Stored", message);
1456  }
1457  Ok(())
1458}
1459/// pops the last value in a named `Stored::VecDequeStored` queue, if any
1460pub fn pop_value(key: &str) -> Result<Option<Stored>> {
1461  let key_sym = arena::pin(key);
1462  // Compute the pop result under the borrow, then raise the BUG Error! *after*
1463  // dropping it — Error! reads MAX_ERRORS, which panics under a live mutable
1464  // borrow (mirrors push_value; tikz-cd 2001.08973).
1465  let popped: std::result::Result<Option<Stored>, ()> = {
1466    let mut state = state_mut!();
1467    if !state.value.contains_key(&key_sym) {
1468      state.assign_internal(
1469        TableName::Value,
1470        key_sym,
1471        Stored::VecDequeStored(VecDeque::new()),
1472        Some(Scope::Global),
1473      );
1474    }
1475    if let Some(&mut Stored::VecDequeStored(ref mut front)) =
1476      state.value.get_mut(&key_sym).unwrap().front_mut()
1477    {
1478      Ok(front.pop_back())
1479    } else {
1480      Err(())
1481    }
1482  };
1483  match popped {
1484    Ok(v) => Ok(v),
1485    Err(()) => {
1486      Error!(
1487        "State",
1488        "Stored",
1489        "BUG: Tried to pop_value from a non-vecdeque value key!"
1490      );
1491      Ok(None)
1492    },
1493  }
1494}

1703      // Release the state borrow first, then pass the interned &str directly
1704      // into tokenize_internal via the re-entrant arena — avoids materializing
1705      // an owned String just to tokenize.
1706      let sym = *sym;
1707      drop(state);
1708      arena::with(sym, |astr| Some(mouth::tokenize_internal(astr)))
1709    },
1710    Some(Stored::VecDequeStored(v)) => {
1711      // Reverting the queue to Tokens routes each String item through
1712      // `mouth::tokenize_internal`, which takes a *mutable* STATE borrow — so
1713      // clone the queue and drop the immutable `state` borrow first (mirrors
1714      // the `Stored::String` branch above). Without this, LookupTokens on a
1715      // VecDequeStored key (e.g. "class_options") panics "RefCell already
1716      // borrowed" (#314).
1717      let vdq = v.clone();
1718      drop(state);
1719      Stored::VecDequeStored(vdq).into()
1720    },
1721    _ => None,
1722  }
1723}

1835/// Faithful port of Perl `LookupDimension` (`Package.pm` L1371-1393, as
1836/// widened by upstream PR #2829): try to turn the argument into a Dimension,
1837/// recognizing strings, registers, ….
1838///
1839/// * a string that looks like an obvious dimension (`/^[0-9+-.]\w\w+$/`,
1840///   e.g. `"3pt"` — but NOT `"0.4pt"`, whose `.` fails `\w`) parses directly;
1841/// * otherwise the string is tokenized: a single token that resolves to a
1842///   register returns its value ("easy and proper case");
1843/// * a multi-token sequence is read as a dimension from a fresh mouth;
1844/// * anything else warns (`expected:register`) unless `noerror`, and yields
1845///   `None` (Perl returns undef).
1846///
1847/// NOTE the #2829 semantics change carried over faithfully: a single token
1848/// whose definition is a MACRO (e.g. a document that `\def`s `\jot`) no
1849/// longer reads its body as a dimension — it now falls through to the warn
1850/// branch. (Perl's digested-Box coercion branch has no Rust equivalent here:
1851/// all our callers pass strings.)
1852pub fn lookup_dimension_cs(cs: &str, noerror: bool) -> Option<Dimension> {
1853  use std::str::FromStr;
1854  // Obvious dimension string? (Perl: /^[0-9\+\-\.]\w\w+$/)
1855  let mut chars = cs.chars();
1856  let obvious = matches!(chars.next(), Some(c) if c.is_ascii_digit() || matches!(c, '+' | '-' | '.'))
1857    && cs.chars().count() >= 3
1858    && chars.all(|c| c.is_alphanumeric() || c == '_');
1859  if obvious && let Ok(d) = Dimension::from_str(cs) {
1860    return Some(d);
1861  }
1862  let tokens = mouth::tokenize_internal(cs);
1863  let toks = tokens.unlist();
1864  if toks.len() == 1 {
1865    match lookup_definition(&toks[0]) {
1866      Ok(Some(defn)) if defn.is_register() => {
1867        // Easy (and proper) case.
1868        return defn.value_of(Vec::new()).map(|rv| Dimension::from(&rv));
1869      },
1870      // Defined but not a register (a `\def`-ized length): fall through and
1871      // read its body as a dimension. NB this is a deliberate DIVERGENCE
1872      // from post-#2829 Perl, which unintentionally LOST this path in the
1873      // rewrite (a single macro token falls to the warn branch upstream) —
1874      // see KNOWN_PERL_ERRORS #41. Real arXiv papers `\def\arraycolsep{...}`
1875      // (cluster regressions cover this); pre-#2829 Perl read the body.
1876      Ok(Some(_)) => {},
1877      // Undefined single token: warn like Perl and yield nothing.
1878      _ => {
1879        if !noerror {
1880          let message = s!("The control sequence '{}' is not a register", cs);
1881          Warn!("expected", "register", message);
1882        }
1883        return None;
1884      },
1885    }
1886  }
1887  // Read the token sequence (a defined single CS expands here, exactly like
1888  // Perl's readingFromMouth) as a dimension from a fresh mouth; an
1889  // unreadable sequence warns Missing-number inside read_dimension and
1890  // yields Dimension(0), matching Perl.
1891  gullet::reading_from_mouth(mouth::Mouth::default(), move || {
1892    gullet::unread(Tokens::new(toks));
1893    gullet::read_dimension()
1894  })
1895  .ok()
1896}

1963pub fn unshift_value<T: Into<Stored>>(key: &str, values: Vec<T>) {
1964  let values_iter = values.into_iter().map(Into::into);
1965  let key_sym = arena::pin(key);
1966  let mut state = state_mut!();
1967  if !state.value.contains_key(&key_sym) {
1968    state.assign_internal(
1969      TableName::Value,
1970      key_sym,
1971      Stored::VecDequeStored(VecDeque::new()),
1972      Some(Scope::Global),
1973    )
1974  }
1975  let receiver = state.value.get_mut(&key_sym).unwrap().front_mut();
1976  if let Some(&mut Stored::VecDequeStored(ref mut front)) = receiver {
1977    for value in values_iter.rev() {
1978      // preserving order unshift, as Perl's
1979      front.push_front(value)
1980    }
1981  } else if receiver.is_none() || matches!(receiver, Some(Stored::None)) {
1982    // Key doesn't exist yet — create a new VecDequeStored via the existing borrow
1983    let mut vd = VecDeque::new();
1984    for value in values_iter {
1985      vd.push_back(value);
1986    }
1987    state.assign_internal(
1988      TableName::Value,
1989      key_sym,
1990      Stored::VecDequeStored(vd),
1991      Some(Scope::Global),
1992    );
1993  } else {
1994    // Wrong type — warn but don't panic
1995    Warn!(
1996      "unexpected",
1997      "unshift_value",
1998      s!(
1999        "unshift_value expects VecDequeStored receiver for key {:?}, got: {:?}",
2000        key,
2001        receiver.map(|r| std::mem::discriminant(r))
2002      )
2003    );
2004  }
2005}

2249/// Closure-based variant of `lookup_meaning` — avoids the per-call
2250/// `Stored::clone()` when the caller only needs to *inspect* the
2251/// meaning (e.g. extract a CS Token from an Expandable/Primitive
2252/// definition). Stored::clone is ~1% of total instructions on
2253/// siunitx-heavy fixtures (5M+ calls per run, each cloning a full
2254/// Stored enum). This helper borrows the stored value instead.
2255///
2256/// For non-CS/ACTIVE tokens, passes `Some(Stored::Token(*token))` —
2257/// matching lookup_meaning's fallback semantics. Note this requires
2258/// a single stack allocation of Stored::Token (Copy), not a heap
2259/// clone.
2260pub fn with_meaning<R>(token: &Token, f: impl FnOnce(Option<&Stored>) -> R) -> R {
2261  let state = state!();
2262  if token.get_catcode().is_active_or_cs() && token.text != pin!("") {
2263    match state.meaning.get(&meaning_key(token)) {

2292/// $meaning should be a definition (for defining active control sequences)
2293/// or another token, for \let
2294pub fn assign_meaning<T: Into<Stored>>(token: &Token, meaning: T, scope: Option<Scope>) {
2295  let mut meaning = meaning.into();
2296  // short-circuit guard to avoid e.g. T_MATH let to itself
2297  if let Stored::Token(ref mt) = meaning
2298    && token == mt
2299  {
2300    return;
2301  }
2302  // For \let chains: if the target token has an expandable/primitive definition,
2303  // store that definition directly instead of the Token indirection.
2304  // This ensures `\let \foo \bar` where \bar is expandable makes \foo expandable too.
2305  // Follow at most 50 \let links to avoid cycles.
2306  if let Stored::Token(ref target) = meaning {
2307    let mut current = *target;
2308    for _ in 0..50 {
2309      match lookup_meaning(&current) {
2310        Some(Stored::Token(next)) => {
2311          current = next; // follow chain
2312        },
2313        Some(Stored::None) | None => break, // dead end — keep as Token
2314        Some(defn) => {
2315          // Found a real definition — use it directly
2316          meaning = defn;
2317          break;
2318        },
2319      }
2320    }
2321  }
2322  let csname_sym = token.pin_cs_name();
2323  state_mut!().assign_internal(TableName::Meaning, csname_sym, meaning, scope);
2324}

2422/// Recognizes mathactive tokens in math mode and also looks for
2423/// cs that have been let to other `executable' tokens.
2424/// Returns a definition object, or a "self inserting" token.
2425/// Used for digestion.
2426pub fn lookup_digestable_definition(token: &Token) -> Option<Stored> {
2427  let cc = token.get_catcode();
2428  let t_sym = token.get_sym();
2429  let is_active_or_cs = cc.is_active_or_cs();
2430  let lookup_sym = if is_active_or_cs
2431    || ((cc == Catcode::LETTER || (cc == Catcode::OTHER))
2432      && lookup_bool_sym(crate::pin!("IN_MATH"))
2433      && (lookup_mathcode_sym(t_sym).unwrap_or(0) == 0x8000))
2434  {
2435    // `\special_relax`-family tokens digest under the bare `\special_relax` no-op.
2436    meaning_key(token)
2437  } else {
2438    // Use cached SymStr from `Catcode::name_sym` instead of re-interning
2439    // `cc.name()` (a &'static str) on every non-active-or-cs token —
2440    // saves a hashmap probe per token on the digest hot path.
2441    cc.name_sym()
2442  };
2443  // Debug!("Looking up digestable {:?}", lookupname);
2444  let state = state!();
2445  let entry_opt = state.meaning.get(&lookup_sym);
2446  if lookup_sym != pin!("") && entry_opt.is_some() && !entry_opt.as_ref().unwrap().is_empty() {
2447    // Debug!("Found definition for: {:?}", lookupname);
2448    if let Some(entry) = entry_opt
2449      && let Some(front) = entry.front()
2450    {
2451      if let Stored::Token(t) = front {
2452        if let Some(lookup_name) = t.get_executable_primitive_name() {
2453          let lookup_sym = arena::pin(lookup_name);
2454          if let Some(retry_entry) = state!().meaning.get(&lookup_sym) {
2455            // special case,
2456            // If a cs has been let to an executable token, lookup ITS defn.
2457            return retry_entry.front().cloned();
2458          }
2459        }
2460        // Also follow \let chains for CS tokens: if \foo is \let to \bar,
2461        // resolve \bar's definition. This handles expl3 aliases like
2462        // \tex_long:D → \long, \tex_gdef:D → \gdef.
2463        if t.get_catcode() == Catcode::CS
2464          && let Some(target_entry) = state.meaning.get(&t.text)
2465          && let Some(target_front) = target_entry.front()
2466          && !matches!(target_front, Stored::Token(_) | Stored::None)
2467        {
2468          return Some(target_front.clone());
2469        }
2470      }
2471      // Perl State.pm:474 lookupDigestableDefinition: the guard
2472      // `($defn = $$entry[0])` is FALSE when the entry's value is undef, so
2473      // execution falls through to `return $token` (self-inserting) for a
2474      // LETTER/OTHER token and to `return undef` for an active/CS one. A
2475      // math-active LETTER/OTHER character whose active meaning was `\let`
2476      // to an undefined CS hits exactly this case — e.g. braket-style
2477      // `\Pr{A|B}`: the macro body does `\mathcode`\|=32768 \let|\SetVert`
2478      // with `\SetVert` itself undefined (neither our nor Perl's braket
2479      // binding defines it), leaving `|`'s meaning an explicit
2480      // `Stored::None`. Returning `Some(Stored::None)` here routed the `|`
2481      // to generateErrorStub ("The token T_OTHER[|] is not defined"); Perl
2482      // instead self-inserts the literal char. Mirror Perl: a None-valued
2483      // entry for a non-active/CS (math-active) char self-inserts; active/CS
2484      // tokens still fall to the `None` return below. Witness 1602.01342.
2485      if matches!(front, Stored::None) && !is_active_or_cs {
2486        return Some(token.into());
2487      }
2488      // if a regular definition, just return.
2489      return Some(front.clone());
2490    }
2491  } else if is_active_or_cs {
2492    return None;
2493  }
2494  Some(token.into())
2495}

2550/// Hoist every Meaning binding installed at the topmost frame since
2551/// `pre_snapshot` was taken to GLOBAL scope. Idempotent: keys already
2552/// in `pre_snapshot` are skipped. Operates on the Meaning table only —
2553/// callers that need to promote Value/Catcode/etc. should add parallel
2554/// helpers (none required so far).
2555pub fn hoist_top_frame_meaning_delta(pre_snapshot: &[SymStr]) {
2556  let pre: rustc_hash::FxHashSet<SymStr> = pre_snapshot.iter().copied().collect();
2557  let new_keys: Vec<SymStr> = {
2558    let state = state!();
2559    state
2560      .undo
2561      .front()
2562      .map(|f| {
2563        f.meaning
2564          .keys()
2565          .copied()
2566          .filter(|k| !pre.contains(k))
2567          .collect()
2568      })
2569      .unwrap_or_default()
2570  };
2571  for key in new_keys {
2572    let current = {
2573      let state = state!();
2574      state
2575        .meaning
2576        .get(&key)
2577        .and_then(|stack| stack.front().cloned())
2578    };

2580      // Direct re-bind via assign_internal so we don't need to round-trip a
2581      // full Token. The Meaning table is keyed by SymStr (the CS name);
2582      // any future read via `assign_meaning(token, ...)` would reach the
2583      // same cell. Scope::Global removes higher-frame undo entries and
2584      // installs at the lowest non-locked frame.
2585      state_mut!().assign_internal(TableName::Meaning, key, value, Some(Scope::Global));
2586    }
2587  }

2720/// Activates all stashed definitions for the named scope. No-op if the scope is already active.
2721pub fn activate_scope(scope: SymStr) {
2722  let mut state = state_mut!();
2723  // do not re-activate if already active.
2724  if let Some(stash_active_entry) = state.stash_active.get(&scope)
2725    && !stash_active_entry.is_empty()
2726  {
2727    return;
2728  }
2729
2730  state.assign_internal(
2731    TableName::StashActive,
2732    scope,
2733    Stored::Bool(true),
2734    Some(Scope::Local),
2735  );
2736  // Also, we need to take ownership of the stashed data, so that we can assign it.
2737  // TODO: Potential to optimize?
2738  // Also x2, we are using a shared "Stored" interface for all data that passes through
2739  // assign_internal, but that causes both uncertainty and overhead in the Stash table
2740  // specifically. TODO x2: Maybe a more ambitious refactor will separate out the Stash logic
2741  // and use "StashTable" directly instead of Stored::Stash(StashTable) ?
2742
2743  let mut actions = Vec::new();
2744
2745  if let Some(Some(Stored::Stash(defns))) = state.stash.get(&scope).map(|x| x.iter().next()) {
2746    for (table_name, key, value) in defns {
2747      // copy the values out from the stashed defns, so that Rust
2748      // is calm we are borrowing safely.
2749
2750      actions.push((*table_name, key.to_owned(), value.clone()));
2751    }
2752  }
2753  // Here we ALWAYS push the stashed values into the table
2754  // since they may be popped off by deactivateScope
2755  for (table_name, key, value) in actions {
2756    let frame = &mut state.undo[0];
2757    let frame_table = frame.table_mut(table_name);
2758    let entry = frame_table.entry(key).or_insert(0);
2759    *entry += 1; // Note that this many values must be undone
2760    let key_table = state.table_mut(table_name).entry(key).or_default();
2761    key_table.push_front(value); // And push new binding.
2762  }
2763}

2769/// Removes any definitions that were associated with the named `scope`.

2771pub fn deactivate_scope(scope: SymStr) {

2774    None => false,
2775    Some(v) => !v.is_empty(),
2776  };
2777  if !scope_exists {
2778    return;
2779  }
2780
2781  state.assign_internal(
2782    TableName::StashActive,
2783    scope,
2784    Stored::Bool(false),
2785    Some(Scope::Global),
2786  );
2787
2788  let mut collected = Vec::new();
2789  if let Some(Some(Stored::Stash(defns))) = state.stash.get(&scope).map(|x| x.iter().next()) {
2790    for (table_name, key, value) in defns {
2791      collected.push((table_name.to_owned(), key.to_owned(), value.to_owned()));
2792    }
2793  }
2794
2795  for (table_name, key, value) in collected {
2796    let front_is_value = if let Some(table_entry_peek) = state.table(table_name).get(&key) {

2845/// convert a unit name into a `f64` scaling factor over `sp`
2846pub fn convert_unit(unit_arg: &str) -> f64 {

2848  // Font-relative units fall back to 10pt metrics when no current font is

2850  // built-in default font; matching with a static fallback is cheaper
2851  // than forcing every caller to ensure a font frame exists.
2852  let font_metric =
2853    |getter: fn(&Font) -> i64| -> f64 { lookup_font().map(|f| getter(&f) as f64).unwrap_or(0.0) };
2854  match unit.as_str() {
2855    "em" => font_metric(|f| f.get_em_width()),
2856    "ex" => font_metric(|f| f.get_ex_height()),
2857    "mu" => font_metric(|f| f.get_mu_width()),
2858    u => match UNITS.get(u) {
2859      Some(sp) => *sp,
2860      None => {
2861        let message = s!("Illegal unit of measure {:?}, assuming pt.", u);
2862        Warn!("expected", "<unit>", message);
2863        *UNITS.get("pt").unwrap()
2864      },
2865    },
2866  }
2867}

2869/// Convert a unit name into the exact TeX `(num, den)` fraction such that a
2870/// dimension of `value` units is `floor(round(value·65536)·num/den)` scaled
2871/// points (see `numeric_ops::fixpoint_unit`).
2872///
2873/// Physical units use TeX's `set_conversion(num)(denom)` fractions verbatim
2874/// (tex.web §458, lines 9020-9032): `in=7227/100, pc=12/1, cm=7227/254,
2875/// mm=7227/2540, bp=7227/7200, dd=1238/1157, cc=14856/1157`, plus `pt=1/1` and
2876/// `sp=1/65536`. `px` follows LaTeXML in aliasing `bp`. Font-relative units
2877/// (`em`/`ex`/`mu`) return `(metric_sp, 65536)`, matching tex.web §8983's
2878/// `nx_plus_y(_, v, xn_over_d(v, f, 65536))` for internal units. This is the
2879/// exact-integer counterpart of [`convert_unit`]; each physical entry satisfies
2880/// `convert_unit(u) == 65536·num/den`.
2881pub fn convert_unit_ratio(unit_arg: &str) -> (i64, i64) {
2882  let unit = unit_arg.to_lowercase();
2883  let font_metric =
2884    |getter: fn(&Font) -> i64| -> i64 { lookup_font().map(|f| getter(&f)).unwrap_or(0) };
2885  // UNITY == 65536 sp per pt; font-relative and `sp` units convert via the
2886  // `floor(fix·v/UNITY)` path (tex.web §8983 nx_plus_y/xn_over_d).
2887  match unit.as_str() {
2888    "em" => (font_metric(|f| f.get_em_width()), UNITY),
2889    "ex" => (font_metric(|f| f.get_ex_height()), UNITY),
2890    "mu" => (font_metric(|f| f.get_mu_width()), UNITY),
2891    "pt" => (1, 1),
2892    "pc" => (12, 1),
2893    "in" => (7227, 100),
2894    "bp" | "px" => (7227, 7200),
2895    "cm" => (7227, 254),
2896    "mm" => (7227, 2540),
2897    "dd" => (1238, 1157),
2898    "cc" => (14856, 1157),
2899    "sp" => (1, UNITY),
2900    u => {
2901      let message = s!("Illegal unit of measure {:?}, assuming pt.", u);
2902      Warn!("expected", "<unit>", message);
2903      (1, 1)
2904    },
2905  }
2906}

3444/// Check whether a Stored value can be serialized for the kernel dump.
3445/// Values containing closures (Primitive, Constructor, Conditional, etc.)
3446/// cannot be serialized — they come from Rust engine code, not the dump.
3447/// This matches Perl's DumpFile which only serializes Expandable macros.
3448pub fn is_serializable(stored: &Stored) -> bool {
3449  use Stored::*;
3450  match stored {
3451    // Data types: always serializable
3452    None | Bool(_) | String(_) | Charcode(_) | Int(_) | Catcode(_) => true,
3453    Token(_) | Tokens(_) | Number(_) | Float(_) => true,
3454    Glue(_) | MuGlue(_) | Dimension(_) | MuDimension(_) => true,
3455    Reversion(_) | KeyVal(_) => true,
3456    Chars(_) | Strings(_) => true,
3457    // Expandable: serializable when body is Tokens OR None (regular
3458    // macros). Closure-bodied Expandables (e.g. `\expandafter`,
3459    // `\unexpanded`, `\the` — defined via `DefMacro!` with a closure
3460    // body) ALSO pass — dump_writer's `serialize_stored` emits a PA
3461    // alias to the canonical CS so `\let \tex_expandafter:D
3462    // \expandafter`-style aliases survive the dump. (Bug C parity fix
3463    // — see project_kernel_dump_tdd.md.) The writer's add-only policy
3464    // at load time skips entries whose key is already defined in the
3465    // compiled engine, so primary CSes don't double-bind.
3466    Expandable(_) => true,
3467    // Register: serializable (stores value + type, no closures)
3468    Register(_) => true,
3469    // Font: serializable (data only)
3470    Font(_) => true,
3471    // Primitives/MathPrimitives/Conditionals: the CLOSURE can't be
3472    // serialized, but each carries its own canonical CS name. If the
3473    // entry's key differs from that canonical CS, this is a `\let`-alias
3474    // we CAN capture (as a "PA" pointer) so the dump reader replays the
3475    // `\let` at load time. dump_writer returns the PA tag; dump_reader
3476    // re-applies via state::let_i. This is how \tex_let:D, \tex_def:D,
3477    // \tex_ifx:D, \if_meaning:w, and the hundreds of other expl3-renamed
3478    // primitives + conditionals survive the dump without needing to re-run
3479    // 36k lines of expl3-code.tex.
3480    //
3481    // Returning true here only means "pass to dump_writer"; the writer's
3482    // serialize_stored emits the PA target. Self-aliases (primary CSes
3483    // not yet aliased anywhere) typically don't appear in the diff because
3484    // they're in the pre-snapshot — but if they do, the dump reader skips
3485    // them by comparing key to target.
3486    Primitive(_) | MathPrimitive(_) | Conditional(_) => true,
3487    // Constructor: same logic as Primitive/Conditional. Constructors carry a
3488    // closure body the dump can't serialize, BUT they each carry a canonical
3489    // CS field. When the entry key differs from that CS, it's a `\let`-alias
3490    // (e.g. `\let \tex_par:D \par` where `\par` is itself a `Let!` alias to
3491    // `\lx@normal@par` — a Constructor). dump_writer emits `PA\t<cs>`;
3492    // dump_reader replays via `state::let_i`. Mirrors Perl's writer:
3493    // `dump_constructor` is undefined in `Dumper.pm`, but TeX_Job.pool
3494    // `DumpFile`'s let-detection branch (L184-198) catches the (key !=
3495    // value->getCSName) case and emits `Lt(key, letkey)`. Without this,
3496    // `\tex_par:D`, `\tex_cr:D`, `\tex_noindent:D`, etc. drop from the dump
3497    // because diff_from_snapshot filters them before the writer's
3498    // Constructor arm sees them.
3499    Constructor(_) => true,
3500    // Collections: serializable if contents are
3501    VecDequeStored(_) | HashStored(_) | HashString(_) => true,
3502    // Everything else: skip for safety
3503    _ => false,
3504  }
3505}

3530    RefCell::new(rustc_hash::FxHashMap::default());
