1//! Image helpers — port of `LaTeXML::Util::Image`.
2//!
3//! Perl counterpart: `lib/LaTeXML/Util/Image.pm`.
4//!
5//! Provides filesystem search for image candidates, minimal header-based
6//! image size detection (PNG / JPEG / EPS) and the graphicx `sizer` that
7//! converts keyval option strings into box dimensions. The Rust port is
8//! intentionally narrower than the Perl original — Image::Magick is not
9//! used at all; LaTeXML::Post::Graphics carries out any heavy-duty image
10//! operations in a post-processing pass.
11

21/// Perl: `image_candidates($path)` (Util::Image L43-57).
22///
23/// Returns comma-separated list of candidate paths for `path`, searching
24/// GRAPHICSPATHS + SEARCHPATHS + SOURCEDIRECTORY. Paths are returned
25/// relative to SOURCEDIRECTORY when possible, matching the Perl
26/// `pathname_relative($_, $base)` post-filter.
27pub fn image_candidates(path: &str) -> String {
28  let path = path.trim().trim_matches('"');
29  if path.is_empty() {
30    return String::new();
31  }
32  let mut search_dirs: Vec<String> = state::get_graphics_paths();
33  search_dirs.extend(state::get_search_paths());
34  let source_dir = state::lookup_string("SOURCEDIRECTORY");
35  if !source_dir.is_empty() {
36    search_dirs.push(source_dir.clone());
37  }
38  if search_dirs.is_empty() {
39    search_dirs.push(".".to_string());
40  }
41
42  let mut candidates: Vec<String> = Vec::new();
43  let path_obj = Path::new(path);
44  let has_extension = path_obj.extension().is_some();
45  let source_path = if source_dir.is_empty() {
46    None
47  } else {
48    Some(PathBuf::from(&source_dir))
49  };
50
51  for dir in &search_dirs {
52    let base = PathBuf::from(dir).join(path);
53    if has_extension {
54      if base.exists() {
55        let rel = match &source_path {
56          Some(sp) => base
57            .strip_prefix(sp)
58            .unwrap_or(&base)
59            .to_string_lossy()
60            .to_string(),
61          None => base.to_string_lossy().to_string(),
62        };
63        candidates.push(rel);
64      }
65    } else {
66      // Search for path with any extension
67      let parent = base.parent().unwrap_or(Path::new("."));
68      let stem = base
69        .file_name()
70        .map(|s| s.to_string_lossy().to_string())
71        .unwrap_or_default();
72      if let Ok(entries) = std::fs::read_dir(parent) {
73        for entry in entries.flatten() {
74          let fname = entry.file_name().to_string_lossy().to_string();
75          if let Some(dot_pos) = fname.find('.') {
76            if fname[..dot_pos] == stem {
77              let full = entry.path();
78              let rel = match &source_path {
79                Some(sp) => full
80                  .strip_prefix(sp)
81                  .unwrap_or(&full)
82                  .to_string_lossy()
83                  .to_string(),
84                None => full.to_string_lossy().to_string(),
85              };
86              candidates.push(rel);
87            }
88          }
89        }
90      }
91    }
92  }
93
94  // If no candidates found and path has no extension, try common image extensions
95  // (matching Perl's pathname_findall with types => ['*'] which tries all known types)
96  if candidates.is_empty() && !has_extension {
97    // Perl typically returns just the first match (png)
98    if let Some(ext) = ["png", "jpg", "jpeg", "gif", "pdf", "eps", "svg"].first() {
99      let with_ext = format!("{path}.{ext}");
100      candidates.push(with_ext);
101    }
102  }
103
104  // Deduplicate while preserving order
105  let mut seen = std::collections::HashSet::new();
106  candidates.retain(|c| seen.insert(c.clone()));
107
108  if candidates.is_empty() {
109    path.to_string()
110  } else {
111    candidates.join(",")
112  }
113}

115/// Perl: `image_graphicx_sizer($whatsit)` (Util::Image L259-272).
116///
117/// Reads image dimensions from `candidates`, applies the `options` string
118/// (graphicx keyvals: width/height/totalheight/scale/keepaspectratio) and
119/// writes back `cached_width`, `cached_height`, `cached_depth` on the
120/// whatsit so downstream getSize() consumers (pgf, tikz) see the correct
121/// box dimensions.
122pub fn image_graphicx_sizer(whatsit: &mut Whatsit) {
123  let dpi_val = state::lookup_int("DPI");
124  let dpi = if dpi_val > 0 { dpi_val as f64 } else { 100.0 }; // Perl: our $DPI = 100
125  let candidates = whatsit
126    .get_property("candidates")
127    .map(|c| c.to_string())
128    .unwrap_or_default();
129  let options = whatsit
130    .get_property("options")
131    .map(|c| c.to_string())
132    .unwrap_or_default();
133
134  // Try to read actual image dimensions from file
135  let mut img_w: f64 = 0.0;
136  let mut img_h: f64 = 0.0;
137  let source_dir = state::lookup_string("SOURCEDIRECTORY");
138  for candidate in candidates.split(',') {
139    let candidate = candidate.trim();
140    if candidate.is_empty() {
141      continue;
142    }
143    let full_path = if Path::new(candidate).is_absolute() {
144      PathBuf::from(candidate)
145    } else if !source_dir.is_empty() {
146      PathBuf::from(&source_dir).join(candidate)
147    } else {
148      PathBuf::from(candidate)
149    };
150    if let Some((w, h)) = read_image_dimensions(&full_path) {
151      img_w = w as f64;
152      img_h = h as f64;
153      break;
154    }
155  }
156
157  if img_w <= 0.0 || img_h <= 0.0 {
158    return;
159  }
160
161  // Apply graphicx options (height, width, scale, keepaspectratio)
162  // Perl: image_graphicx_size applies parsed transformations
163  let dppt = dpi / 72.27; // dots per point
164  let mut w = img_w;
165  let mut h = img_h;
166
167  // Parse options string for simple cases
168  // Perl: image_graphicx_parse uses to_bp() to convert dimensions to big points (1/72 inch)
169  let mut req_w: Option<f64> = None; // in bp (big points)
170  let mut req_h: Option<f64> = None; // in bp
171  let mut keep_ratio = false;
172  let mut scale: Option<f64> = None;
173
174  for opt in options.split(',') {
175    let opt = opt.trim();
176    if let Some(val) = opt.strip_prefix("width=") {
177      if let Ok(dim) = <Dimension as std::str::FromStr>::from_str(val.trim()) {
178        // to_bp: convert pt to bp (1bp = 1/72 inch, 1pt = 1/72.27 inch)
179        req_w = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
180      }
181    } else if let Some(val) = opt.strip_prefix("height=") {
182      if let Ok(dim) = <Dimension as std::str::FromStr>::from_str(val.trim()) {
183        req_h = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
184      }
185    } else if let Some(val) = opt.strip_prefix("totalheight=") {
186      if let Ok(dim) = <Dimension as std::str::FromStr>::from_str(val.trim()) {
187        req_h = Some(dim.value_of() as f64 / 65536.0 * 72.0 / 72.27);
188      }
189    } else if opt.starts_with("keepaspectratio") {
190      keep_ratio = true;
191    } else if let Some(val) = opt.strip_prefix("scale=") {
192      scale = val.trim().parse::<f64>().ok();
193    }
194  }
195
196  // Apply transformations (matching Perl image_graphicx_size logic)
197  if let Some(s) = scale {
198    w = (w * s).ceil();
199    h = (h * s).ceil();
200  }
201  if req_w.is_some() || req_h.is_some() {
202    let target_w = req_w.map(|rw| rw * dppt);
203    let target_h = req_h.map(|rh| rh * dppt);
204    if keep_ratio {
205      match (target_w, target_h) {
206        (Some(tw), Some(th)) => {
207          // Both specified with keepaspectratio: use the more restrictive
208          if w > 0.0 && h > 0.0 {
209            if tw / w < th / h {
210              let th2 = h * tw / w;
211              w = tw;
212              h = th2;
213            } else {
214              let tw2 = w * th / h;
215              w = tw2;
216              h = th;
217            }
218          }
219        },
220        (Some(tw), None) => {
221          if w > 0.0 {
222            h = h * tw / w;
223            w = tw;
224          }
225        },
226        (None, Some(th)) => {
227          if h > 0.0 {
228            w = w * th / h;
229            h = th;
230          }
231        },
232        (None, None) => {},
233      }
234    } else {
235      if let Some(tw) = target_w {
236        w = tw;
237      }
238      if let Some(th) = target_h {
239        h = th;
240      }
241    }
242  }
243  // Perl: ceil pixel dimensions after applying transforms
244  w = w.ceil();
245  h = h.ceil();
246
247  // Convert pixel dimensions back to points, then to scaled points (sp)
248  let width_pt = w * 72.27 / dpi;
249  let height_pt = h * 72.27 / dpi;
250
251  // Perl: Dimension($w * 72.27 / $dpi . 'pt') — parses via TeX fixed-point arithmetic
252  let w_dim =
253    <Dimension as std::str::FromStr>::from_str(&format!("{width_pt}pt")).unwrap_or_default();
254  let h_dim =
255    <Dimension as std::str::FromStr>::from_str(&format!("{height_pt}pt")).unwrap_or_default();
256  whatsit.set_property("cached_width", Stored::Dimension(w_dim));
257  whatsit.set_property("cached_height", Stored::Dimension(h_dim));
258  whatsit.set_property("cached_depth", Stored::Dimension(Dimension::default()));
259}

261/// Read image dimensions (width, height) in pixels from a file.
262/// Supports PNG, JPEG, and EPS (PostScript BoundingBox).
263///
264/// This is a narrow replacement for `Image::Size::imgsize` (Perl
265/// `image_size` at Util::Image L86-97). Only a few formats are needed
266/// for typical arXiv graphics inclusions — anything else returns `None`
267/// so the caller skips sizing (mirroring Perl's `return unless $w`).
268pub fn read_image_dimensions(path: &std::path::Path) -> Option<(u32, u32)> {
269  use std::io::Read;
270  let mut file = std::fs::File::open(path).ok()?;
271  let mut header = [0u8; 32];
272  file.read_exact(&mut header).ok()?;
273
274  // PNG: signature + IHDR chunk
275  if &header[0..8] == b"\x89PNG\r\n\x1a\n" {
276    let width = u32::from_be_bytes([header[16], header[17], header[18], header[19]]);
277    let height = u32::from_be_bytes([header[20], header[21], header[22], header[23]]);
278    return Some((width, height));
279  }
280
281  // JPEG: look for SOF marker
282  if header[0] == 0xFF && header[1] == 0xD8 {
283    // Read the full file for JPEG parsing
284    let mut data = header.to_vec();
285    file.read_to_end(&mut data).ok()?;
286    let mut i = 2;
287    while i + 9 < data.len() {
288      if data[i] != 0xFF {
289        break;
290      }
291      let marker = data[i + 1];
292      // SOF markers: 0xC0-0xCF (except 0xC4 DHT, 0xC8 JPG, 0xCC DAC)
293      if (0xC0..=0xCF).contains(&marker) && marker != 0xC4 && marker != 0xC8 && marker != 0xCC {
294        let height = u16::from_be_bytes([data[i + 5], data[i + 6]]) as u32;
295        let width = u16::from_be_bytes([data[i + 7], data[i + 8]]) as u32;
296        return Some((width, height));
297      }
298      let len = u16::from_be_bytes([data[i + 2], data[i + 3]]) as usize;
299      i += 2 + len;
300    }
301  }
302
303  // EPS: PostScript BoundingBox comment. Perl: LaTeXML::Util::Image reads
304  // the leading `%%BoundingBox: llx lly urx ury` (values in bp, 1bp=1/72").
305  // `%%HiResBoundingBox:` is preferred when present (float precision). We
306  // read the first ~8KB since BoundingBox can be deferred (`(atend)` form
307  // is also valid but would require scanning the tail; skip that).
308  if (header[0] == b'%' && (header[1] == b'!' || header[1] == b'%'))
309    || (header.starts_with(b"\xc5\xd0\xd3\xc6"))
310  // EPS with binary preview header
311  {
312    let mut data = header.to_vec();
313    // Read up to 32KB — BoundingBox typically in first few hundred bytes
314    let mut extra = [0u8; 32768];
315    let n = file.read(&mut extra).ok().unwrap_or(0);
316    data.extend_from_slice(&extra[..n]);
317    // If DOS EPSI binary preview: first 4 bytes are C5 D0 D3 C6, next 4
318    // little-endian is offset to the PostScript section. Skip to it.
319    let text_start = if data.starts_with(b"\xc5\xd0\xd3\xc6") && data.len() >= 8 {
320      u32::from_le_bytes([data[4], data[5], data[6], data[7]]) as usize
321    } else {
322      0
323    };
324    let text = std::str::from_utf8(data.get(text_start..)?).ok()?;
325    // Prefer HiResBoundingBox (float) over BoundingBox (int).
326    let mut found: Option<(f64, f64, f64, f64)> = None;
327    for line in text.lines() {
328      let trimmed = line.trim_start();
329      let rest = if let Some(r) = trimmed.strip_prefix("%%HiResBoundingBox:") {
330        // HiRes wins — take and stop searching.
331        parse_bbox(r).inspect(|&b| {
332          found = Some(b);
333        })
334      } else if found.is_none() {
335        trimmed
336          .strip_prefix("%%BoundingBox:")
337          .and_then(parse_bbox)
338          .inspect(|&b| {
339            found = Some(b);
340          })
341      } else {
342        None
343      };
344      if rest.is_some() && trimmed.starts_with("%%HiResBoundingBox:") {
345        break;
346      }
347    }
348    if let Some((llx, lly, urx, ury)) = found {
349      let w = (urx - llx).max(0.0);
350      let h = (ury - lly).max(0.0);
351      if w > 0.0 && h > 0.0 {
352        // EPS BoundingBox is in bp (1bp = 1/72"). Return as pixels at the
353        // same bp-per-pixel rate the caller expects (it divides by dppt =
354        // dpi/72.27 downstream). Using 1:1 means callers get bp-sized
355        // pixels, consistent with Perl's `image_size` returning bp for
356        // EPS (LaTeXML::Util::Image::image_size L45-L60).
357        return Some((w.round() as u32, h.round() as u32));
358      }
359    }
360  }
361
362  None
363}

365/// Parse `"llx lly urx ury"` from a BoundingBox comment body.
366pub fn parse_bbox(rest: &str) -> Option<(f64, f64, f64, f64)> {
367  let mut it = rest.split_whitespace();
368  let llx = it.next()?.parse::<f64>().ok()?;
369  let lly = it.next()?.parse::<f64>().ok()?;
370  let urx = it.next()?.parse::<f64>().ok()?;
371  let ury = it.next()?.parse::<f64>().ok()?;
372  Some((llx, lly, urx, ury))
373}
