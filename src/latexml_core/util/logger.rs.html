5// ANSI SGR escape sequences (drop-in replacement for the
6// unmaintained `ansi_term` crate; bytes match `ansi_term::Colour::*.paint(...)`).
7const ANSI_RESET: &str = "\x1b[0m";
8const ANSI_GREEN: &str = "\x1b[32m";
9const ANSI_YELLOW: &str = "\x1b[33m";
10const ANSI_RED: &str = "\x1b[31m";
11const ANSI_WHITE: &str = "\x1b[37m";
12
13fn paint(color: &str, text: &str) -> String { format!("{color}{text}{ANSI_RESET}") }

15/// Whether to emit ANSI color escapes on stderr. Colors are a convenience for
16/// an interactive terminal ONLY; when stderr is redirected to a file or pipe
17/// (the canvas/auto-upgrade path: `cortex_worker ... > log.txt 2>&1`) they are
18/// noise that breaks line-anchored error parsing — a naive `grep '^Error:'`
19/// matches `\x1b[31mError:` ZERO times and silently reports "0 errors" on a
20/// failed paper (the false-negative that masked real Rust-only regressions; see
21/// CLAUDE.md "canvas signal integrity"). So: colorize iff stderr is a TTY and
22/// `NO_COLOR` is unset. Cached once — stderr's terminal-ness can't change
23/// mid-process. Note the captured LOG_BUFFER (`.latexml.log`) is already
24/// ANSI-stripped independently; this makes the *redirected stderr* match it.
25fn stderr_use_color() -> bool {
26  use std::io::IsTerminal;
27  use std::sync::OnceLock;
28  static USE_COLOR: OnceLock<bool> = OnceLock::new();
29  *USE_COLOR
30    .get_or_init(|| std::io::stderr().is_terminal() && std::env::var_os("NO_COLOR").is_none())
31}

36/// Thread-local log capture buffer. When enabled, log messages are
37/// appended here (without ANSI colors) in addition to stderr.
38#[thread_local]
39static LOG_BUFFER: RefCell<Option<String>> = RefCell::new(None);
40
41/// Start capturing log output into the buffer (Perl: bind_log).
42pub fn bind_log() { *LOG_BUFFER.borrow_mut() = Some(String::new()); }
43
44/// Flush and return the captured log output, stopping capture (Perl: flush_log).
45pub fn flush_log() -> String { LOG_BUFFER.borrow_mut().take().unwrap_or_default() }

47/// Strip ANSI escape sequences from a string for log file output.
48fn strip_ansi(s: &str) -> String {
49  // Match ESC[ ... m sequences
50  let mut result = String::with_capacity(s.len());
51  let mut in_escape = false;
52  for c in s.chars() {
53    if in_escape {
54      if c == 'm' {
55        in_escape = false;
56      }
57    } else if c == '\x1b' {
58      in_escape = true;
59    } else {
60      result.push(c);
61    }
62  }
63  result
64}

90impl log::Log for LatexmlLogger {
91  fn enabled(&self, metadata: &Metadata) -> bool { metadata.level() <= max_level() }
92
93  fn log(&self, record: &Record) {
94    if self.enabled(record.metadata()) {
95      let record_target = record.target();
96      let details = record.args();
97      if record_target == "note" {
98        // simple print here
99        print_stderr!("{}", details.to_string());
100        return;
101      }
102      let category_object = if record_target.is_empty() {
103        "" // "unknown:unknown" ???
104      } else {
105        record_target
106      };
107      // Following the reporting syntax at: http://dlmf.nist.gov/LaTeXML/manual/errorcodes/
108      let severity = if category_object.starts_with("Fatal:") {
109        ""
110      } else {
111        match record.level() {
112          Level::Info => "Info",
113          Level::Warn => "Warn",
114          Level::Error => "Error",
115          Level::Debug => "Debug",
116          Level::Trace => "Trace",
117        }
118      };
119
120      let message = if severity.is_empty() {
121        s!("{} ", category_object)
122      } else {
123        s!("{}:{} ", severity, category_object)
124      };
125      let painted_message = match record.level() {
126        Level::Info => message,
127        Level::Warn => paint(ANSI_YELLOW, &message),
128        Level::Error => paint(ANSI_RED, &message),
129        Level::Debug => paint(ANSI_GREEN, &message),
130        _ => paint(ANSI_WHITE, &message),
131      } + &details.to_string();
132
133      // Capture to log buffer if active (strip ANSI for clean log text)
134      if let Ok(mut buf) = LOG_BUFFER.try_borrow_mut() {
135        if let Some(ref mut log) = *buf {
136          log.push_str(&strip_ansi(&painted_message));
137          log.push('\n');
138        }
139      }
140
141      // Use `\n` (not `\r`) to guarantee each log line starts on a fresh
142      // line in both TTY and file output. The previous `\r` prefix made
143      // log lines visually overlay any in-flight progress indicator like
144      // `(Loading "foo.sty" definitions... )` — convenient in a terminal
145      // but produced `(...)<CR>Error:...` byte sequences in log files,
146      // breaking line-anchored counts in canvas harnesses
147      // (`grep -cE '^...Error:'` silently returned 0 even when errors
148      // were present). Trade-off: progress indicators in a TTY no longer
149      // get overwritten, but they were not really self-erasing anyway
150      // (they always emitted ` )` to close their parens), so the visual
151      // change is small.
152      // Colorize for an interactive terminal only; when stderr is redirected
153      // to a file/pipe, emit the ANSI-stripped text so on-disk logs stay
154      // grep-clean (matches the captured `.latexml.log` buffer above).
155      if stderr_use_color() {
156        println_stderr!("\n{}", painted_message);
157      } else {
158        println_stderr!("\n{}", strip_ansi(&painted_message));
159      }
160    }
161  }
162
163  fn flush(&self) {}
164}

166/// initialize the logger at a given verbosity `level`
167///
168/// Returns the underlying `SetLoggerError` if another `log` global logger
169/// is already installed (e.g. an embedder set up `tracing-log` first).
170/// Callers can decide whether to ignore that — the in-process `bind_log` /
171/// `flush_log` buffers are independent of the `log` crate sink and keep
172/// working either way.
173pub fn init(level: LevelFilter) -> Result<(), SetLoggerError> {
174  log::set_logger(&LOGGER)?;
175  log::set_max_level(level);
176  Ok(())
177}

183  #[test]
184  fn strip_ansi_removes_color_codes() {
185    // ESC [ ... m should vanish; other content preserved.
186    let red = "\x1b[31mhello\x1b[0m";

199    assert_eq!(strip_ansi(s), "error: warning");

208  #[test]
209  fn strip_ansi_handles_incomplete_escape() {
210    // Unterminated ESC[ sequence — we should not hang.
211    // Current impl: scan until 'm' is found. If never found, consumes
212    // the rest of the input. Document that behavior.
213    let s = "\x1b[1;31m hello";
214    // The scan consumes characters until 'm' is found → the 'm' in the
215    // escape closes, then " hello" remains.
216    assert_eq!(strip_ansi(s), " hello");
217  }

219  #[test]
220  fn bind_log_and_flush_log_roundtrip() {
221    // Before bind_log, flush_log returns empty.
222    // After bind_log, the buffer is active but empty until a log
223    // message arrives. Since we can't easily exercise the Log impl
224    // without initializing a global logger, just verify the
225    // capture-buffer lifecycle primitives.
226    let before = flush_log();
227    assert!(before.is_empty(), "no active buffer → empty flush");
228
229    bind_log();
230    let after = flush_log();
231    assert!(
232      after.is_empty(),
233      "empty buffer is still empty after bind/flush with no log traffic"
234    );
235  }
