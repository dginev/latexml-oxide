5// ANSI SGR escape sequences (drop-in replacement for the
6// unmaintained `ansi_term` crate; bytes match `ansi_term::Colour::*.paint(...)`).
7const ANSI_RESET: &str = "\x1b[0m";
8const ANSI_GREEN: &str = "\x1b[32m";
9const ANSI_YELLOW: &str = "\x1b[33m";
10const ANSI_RED: &str = "\x1b[31m";
11const ANSI_WHITE: &str = "\x1b[37m";

13fn paint(color: &str, text: &str) -> String { format!("{color}{text}{ANSI_RESET}") }

15/// Whether to emit ANSI color escapes on stderr. Colors are a convenience for
16/// an interactive terminal ONLY; when stderr is redirected to a file or pipe
17/// (the canvas/auto-upgrade path: `cortex_worker ... > log.txt 2>&1`) they are
18/// noise that breaks line-anchored error parsing — a naive `grep '^Error:'`
19/// matches `\x1b[31mError:` ZERO times and silently reports "0 errors" on a
20/// failed paper (the false-negative that masked real Rust-only regressions; see
21/// CLAUDE.md "canvas signal integrity"). So: colorize iff stderr is a TTY and
22/// `NO_COLOR` is unset. Cached once — stderr's terminal-ness can't change
23/// mid-process. Note the captured LOG_BUFFER (`.latexml.log`) is already
24/// ANSI-stripped independently; this makes the *redirected stderr* match it.
25fn stderr_use_color() -> bool {
26  use std::{io::IsTerminal, sync::OnceLock};
27  static USE_COLOR: OnceLock<bool> = OnceLock::new();
28  *USE_COLOR
29    .get_or_init(|| std::io::stderr().is_terminal() && std::env::var_os("NO_COLOR").is_none())
30}

35/// Thread-local log capture buffer. When enabled, log messages are
36/// appended here (without ANSI colors) in addition to stderr.
37#[thread_local]
38static LOG_BUFFER: RefCell<Option<String>> = RefCell::new(None);
39
40/// Start capturing log output into the buffer (Perl: bind_log).
41pub fn bind_log() { *LOG_BUFFER.borrow_mut() = Some(String::new()); }
42
43/// Flush and return the captured log output, stopping capture (Perl: flush_log).
44pub fn flush_log() -> String { LOG_BUFFER.borrow_mut().take().unwrap_or_default() }

46/// Diagnostics captured from a worker thread by [`capture`], for the main thread
47/// to fold back in via [`replay_captured`]. Carries both the already-formatted
48/// log text AND the `REPORT` count deltas — `LOG_BUFFER` and `REPORT` are BOTH
49/// `#[thread_local]`, so forwarding only the text would still leave
50/// `status_code` blind to a worker's failures.
51pub struct CapturedDiagnostics {
52  pub log:    String,
53  pub counts: crate::common::error::ReportCounts,
54}
55
56/// Run `f` on the CURRENT (worker) thread with diagnostic capture. Binds a fresh
57/// thread-local log buffer for the duration so any `Error!`/`Warn!`/`Info!` `f`
58/// emits (directly or deep inside a conversion helper) is recorded instead of
59/// lost, and snapshots the worker's `REPORT` counters afterward. The returned
60/// [`CapturedDiagnostics`] is replayed on the main thread by [`replay_captured`]
61/// after the worker is joined, so the messages reach the bound `cortex.log` and
62/// the failures register in `status_code`.
63///
64/// Assumes the worker thread has no pre-bound buffer (the spawned post-processing
65/// pool threads start clean); it does not save/restore a prior binding.
66///
67/// INVARIANTS (unenforced by types; guard the fleet's canonical signal):
68/// - one `capture` per thread lifetime — `bind_log()` CLOBBERS any pre-bound
69///   buffer, and `snapshot_report_counts` does not reset, so reusing capture
70///   on a pooled thread would drop earlier text and double-merge counts;
71/// - callers must `replay_captured` the result on the MAIN thread exactly
72///   once (a panicking worker never returns, losing its pre-panic capture —
73///   the real-time stderr echo retains it, and the caller's worker_panicked
74///   Error keeps status from reading clean).
75pub fn capture<R>(f: impl FnOnce() -> R) -> (R, CapturedDiagnostics) {
76  debug_assert!(
77    LOG_BUFFER
78      .try_borrow()
79      .map(|b| b.is_none())
80      .unwrap_or(false),
81    "logger::capture on a thread with a pre-bound buffer — pooled-thread reuse?"
82  );
83  bind_log();
84  let result = f();
85  let log = flush_log();
86  let counts = crate::common::error::snapshot_report_counts();
87  (result, CapturedDiagnostics { log, counts })
88}

90/// Fold worker-thread diagnostics (from [`capture`]) into the main thread: append
91/// the captured log text to the bound `LOG_BUFFER` and merge the count deltas
92/// into the main `REPORT`. Call on the MAIN thread, in a deterministic order
93/// (e.g. worker/job order), after the workers join. The worker already echoed
94/// each line to the shared stderr fd in real time, so this does NOT re-print to
95/// stderr — it only repairs the captured log + status tally.
96pub fn replay_captured(d: CapturedDiagnostics) {
97  debug_assert!(
98    LOG_BUFFER.try_borrow().is_ok(),
99    "replay_captured: LOG_BUFFER contended — captured text would be dropped"
100  );
101  if !d.log.is_empty()
102    && let Ok(mut buf) = LOG_BUFFER.try_borrow_mut()
103    && let Some(ref mut log) = *buf
104  {
105    // The captured text is already per-record newline-terminated; just make
106    // sure it starts on a fresh line so it can't glue onto an in-flight note.
107    if !log.is_empty() && !log.ends_with('\n') {
108      log.push('\n');
109    }
110    log.push_str(&d.log);
111  }
112  crate::common::error::merge_report_counts(d.counts);
113}

115/// Strip ANSI escape sequences from a string for log file output.
116fn strip_ansi(s: &str) -> String {
117  // Match ESC[ ... m sequences
118  let mut result = String::with_capacity(s.len());
119  let mut in_escape = false;
120  for c in s.chars() {
121    if in_escape {
122      if c == 'm' {
123        in_escape = false;
124      }
125    } else if c == '\x1b' {
126      in_escape = true;
127    } else {
128      result.push(c);
129    }
130  }
131  result
132}

134/// Append a progress note to the capture buffer as INLINE flowing text — the
135/// faithful Perl LaTeXML / tex.web terminal-progress format. `note_begin`
136/// carries a leading '\n' so each stage opens on a fresh line; `note_end`
137/// (` )`) and `note_progress` (`[1][2]…`, `N formulae …`) append inline so a
138/// load's closing paren and the page markers stay on the SAME line as their
139/// opener, and nested closes chain (`… 0.00 sec) 0.05 sec)`). A leading '\n'
140/// is collapsed against an existing trailing '\n' (or buffer start) so we never
141/// emit a blank line. Replaces the old unconditional `push('\n')` per note,
142/// which put every `)` on its own line and doubled newlines into blank lines
143/// (the reported `.latexml.log` noise on corpora.latexml.rs).
144fn append_note(buf: &mut String, note: &str) {
145  if note.starts_with('\n') && (buf.is_empty() || buf.ends_with('\n')) {
146    buf.push_str(&note[1..]);
147  } else {
148    buf.push_str(note);
149  }
150}

176impl log::Log for LatexmlLogger {
177  fn enabled(&self, metadata: &Metadata) -> bool { metadata.level() <= max_level() }
178
179  fn log(&self, record: &Record) {
180    if self.enabled(record.metadata()) {
181      let record_target = record.target();
182      let details = record.args();
183      if record_target == "note" {
184        let note = details.to_string();
185        // A note (e.g. `(Loading foo.sty… )`) is a live progress indicator — but when a capture
186        // buffer is active it must ALSO land there so it reaches the flushed `cortex.log` and
187        // CorTeX's `loaded_file` parser, which anchors on `^(Loading …`. `append_note` keeps it
188        // INLINE (Perl-faithful: `(Loading X… )` on one line, `[1][2]…` chained) while preserving
189        // the `^(Loading` anchor — every `note_begin` carries a leading '\n', so each load still
190        // opens at line start. The following `Info:/Warning:` record re-asserts its own line break
191        // (see the diagnostic-record path below), so the note can't glue onto its anchor.
192        if let Ok(mut buf) = LOG_BUFFER.try_borrow_mut()
193          && let Some(ref mut log) = *buf
194        {
195          append_note(log, &strip_ansi(&note));
196        }
197        print_stderr!("{}", note);
198        return;
199      }
200      let category_object = if record_target.is_empty() {
201        "" // "unknown:unknown" ???
202      } else {
203        record_target
204      };
205      // Following the reporting syntax at: https://math.nist.gov/~BMiller/LaTeXML/manual/errorcodes/
206      // The severity word is the FULL Perl LaTeXML token (Info/Warning/Error/Fatal) — consumers
207      // (CorTeX's log parser, the --server LSP) key on it, so it must match Perl exactly. In
208      // particular WARN must serialize as `Warning` (not the abbreviated `Warn`): CorTeX maps an
209      // unrecognized severity to Info, so `Warn:` silently misfiled every warning (see
210      // LaTeXML/lib/LaTeXML/Common/Error.pm: `"Warning:" . $category . …`).
211      let severity = if category_object.starts_with("Fatal:") {
212        ""
213      } else {
214        match record.level() {
215          Level::Info => "Info",
216          Level::Warn => "Warning",
217          Level::Error => "Error",
218          Level::Debug => "Debug",
219          Level::Trace => "Trace",
220        }
221      };
222
223      let message = if severity.is_empty() {
224        s!("{} ", category_object)
225      } else {
226        s!("{}:{} ", severity, category_object)
227      };
228      let painted_message = match record.level() {
229        Level::Info => message,
230        Level::Warn => paint(ANSI_YELLOW, &message),
231        Level::Error => paint(ANSI_RED, &message),
232        Level::Debug => paint(ANSI_GREEN, &message),
233        _ => paint(ANSI_WHITE, &message),
234      } + &details.to_string();
235
236      // Capture to log buffer if active (strip ANSI for clean log text).
237      // A diagnostic record (Info/Warning/Error/Fatal) must start on a fresh
238      // line so CorTeX's line-anchored parser (^Error:/^Warning:/^Info:)
239      // matches and the record never glues onto an in-flight progress note
240      // (notes no longer force a trailing newline — see append_note).
241      if let Ok(mut buf) = LOG_BUFFER.try_borrow_mut()
242        && let Some(ref mut log) = *buf
243      {
244        if !log.is_empty() && !log.ends_with('\n') {
245          log.push('\n');
246        }
247        log.push_str(&strip_ansi(&painted_message));
248        // Exactly one trailing newline — a multi-detail message (e.g.
249        // `Info:…loaded …\n\tat …\n\tIn …`) already ends with '\n', so an
250        // unconditional push would double it into a blank line before the next
251        // `(Loading …` note.
252        if !log.ends_with('\n') {
253          log.push('\n');
254        }
255      }
256
257      // Use `\n` (not `\r`) to guarantee each log line starts on a fresh
258      // line in both TTY and file output. The previous `\r` prefix made
259      // log lines visually overlay any in-flight progress indicator like
260      // `(Loading "foo.sty" definitions... )` — convenient in a terminal
261      // but produced `(...)<CR>Error:...` byte sequences in log files,
262      // breaking line-anchored counts in canvas harnesses
263      // (`grep -cE '^...Error:'` silently returned 0 even when errors
264      // were present). Trade-off: progress indicators in a TTY no longer
265      // get overwritten, but they were not really self-erasing anyway
266      // (they always emitted ` )` to close their parens), so the visual
267      // change is small.
268      // Colorize for an interactive terminal only; when stderr is redirected
269      // to a file/pipe, emit the ANSI-stripped text so on-disk logs stay
270      // grep-clean (matches the captured `.latexml.log` buffer above).
271      if stderr_use_color() {
272        println_stderr!("\n{}", painted_message);
273      } else {
274        println_stderr!("\n{}", strip_ansi(&painted_message));
275      }
276    }
277  }
278
279  fn flush(&self) {}
280}

282/// initialize the logger at a given verbosity `level`
283///
284/// Returns the underlying `SetLoggerError` if another `log` global logger
285/// is already installed (e.g. an embedder set up `tracing-log` first).
286/// Callers can decide whether to ignore that — the in-process `bind_log` /
287/// `flush_log` buffers are independent of the `log` crate sink and keep
288/// working either way.
289pub fn init(level: LevelFilter) -> Result<(), SetLoggerError> {
290  log::set_logger(&LOGGER)?;
291  log::set_max_level(level);
292  Ok(())
293}

299  #[test]
300  fn append_note_inline_and_no_blank_lines() {
301    // note_begin opens a fresh line; note_end / note_progress stay inline.
302    let mut b = String::new();
303    append_note(&mut b, "\n(Loading keyval.sty..."); // note_begin (buffer empty: no leading blank)
304    append_note(&mut b, " )"); // note_end inline
305    assert_eq!(b, "(Loading keyval.sty... )");
306    append_note(&mut b, "\n(Loading graphics.sty..."); // mid-line: keep the break
307    append_note(&mut b, " )");
308    assert_eq!(b, "(Loading keyval.sty... )\n(Loading graphics.sty... )");
309    // page markers chain inline
310    append_note(&mut b, "\n410 formulae ...");
311    append_note(&mut b, "[1]");
312    append_note(&mut b, "[2]");
313    assert!(b.ends_with("410 formulae ...[1][2]"));
314    // a leading '\n' note after a buffer already at line-start collapses (no blank line)
315    let mut c = String::from("Info:foo\n");
316    append_note(&mut c, "\n(Building...");
317    assert_eq!(c, "Info:foo\n(Building...");
318  }
