5// ANSI SGR escape sequences (drop-in replacement for the
6// unmaintained `ansi_term` crate; bytes match `ansi_term::Colour::*.paint(...)`).
7const ANSI_RESET: &str = "\x1b[0m";
8const ANSI_GREEN: &str = "\x1b[32m";
9const ANSI_YELLOW: &str = "\x1b[33m";
10const ANSI_RED: &str = "\x1b[31m";
11const ANSI_WHITE: &str = "\x1b[37m";
12
13fn paint(color: &str, text: &str) -> String { format!("{color}{text}{ANSI_RESET}") }

15struct LatexmlLogger;
16static LOGGER: LatexmlLogger = LatexmlLogger;

18/// Thread-local log capture buffer. When enabled, log messages are
19/// appended here (without ANSI colors) in addition to stderr.
20#[thread_local]
21static LOG_BUFFER: RefCell<Option<String>> = RefCell::new(None);
22
23/// Start capturing log output into the buffer (Perl: bind_log).
24pub fn bind_log() { *LOG_BUFFER.borrow_mut() = Some(String::new()); }
25
26/// Flush and return the captured log output, stopping capture (Perl: flush_log).
27pub fn flush_log() -> String { LOG_BUFFER.borrow_mut().take().unwrap_or_default() }
28
29/// Strip ANSI escape sequences from a string for log file output.
30fn strip_ansi(s: &str) -> String {
31  // Match ESC[ ... m sequences
32  let mut result = String::with_capacity(s.len());
33  let mut in_escape = false;
34  for c in s.chars() {
35    if in_escape {
36      if c == 'm' {
37        in_escape = false;
38      }
39    } else if c == '\x1b' {
40      in_escape = true;
41    } else {
42      result.push(c);
43    }
44  }
45  result
46}

72impl log::Log for LatexmlLogger {
73  fn enabled(&self, metadata: &Metadata) -> bool { metadata.level() <= max_level() }
74
75  fn log(&self, record: &Record) {
76    if self.enabled(record.metadata()) {
77      let record_target = record.target();
78      let details = record.args();
79      if record_target == "note" {
80        // simple print here
81        print_stderr!("{}", details.to_string());
82        return;
83      }
84      let category_object = if record_target.is_empty() {
85        "" // "unknown:unknown" ???
86      } else {
87        record_target
88      };
89      // Following the reporting syntax at: http://dlmf.nist.gov/LaTeXML/manual/errorcodes/
90      let severity = if category_object.starts_with("Fatal:") {
91        ""
92      } else {
93        match record.level() {
94          Level::Info => "Info",
95          Level::Warn => "Warn",
96          Level::Error => "Error",
97          Level::Debug => "Debug",
98          Level::Trace => "Trace",
99        }
100      };
101
102      let message = if severity.is_empty() {
103        s!("{} ", category_object)
104      } else {
105        s!("{}:{} ", severity, category_object)
106      };
107      let painted_message = match record.level() {
108        Level::Info => message,
109        Level::Warn => paint(ANSI_YELLOW, &message),
110        Level::Error => paint(ANSI_RED, &message),
111        Level::Debug => paint(ANSI_GREEN, &message),
112        _ => paint(ANSI_WHITE, &message),
113      } + &details.to_string();
114
115      // Capture to log buffer if active (strip ANSI for clean log text)
116      if let Ok(mut buf) = LOG_BUFFER.try_borrow_mut() {
117        if let Some(ref mut log) = *buf {
118          log.push_str(&strip_ansi(&painted_message));
119          log.push('\n');
120        }
121      }
122
123      println_stderr!("\r{}", painted_message);
124    }
125  }
126
127  fn flush(&self) {}
128}

130/// initialize the logger at a given verbosity `level`
131///
132/// Returns the underlying `SetLoggerError` if another `log` global logger
133/// is already installed (e.g. an embedder set up `tracing-log` first).
134/// Callers can decide whether to ignore that — the in-process `bind_log` /
135/// `flush_log` buffers are independent of the `log` crate sink and keep
136/// working either way.
137pub fn init(level: LevelFilter) -> Result<(), SetLoggerError> {
138  log::set_logger(&LOGGER)?;
139  log::set_max_level(level);
140  Ok(())
141}

143#[cfg(test)]
144mod tests {

147  #[test]
148  fn strip_ansi_removes_color_codes() {
149    // ESC [ ... m should vanish; other content preserved.
150    let red = "\x1b[31mhello\x1b[0m";
151    assert_eq!(strip_ansi(red), "hello");
152  }
153
154  #[test]
155  fn strip_ansi_noop_on_plain() {
156    assert_eq!(strip_ansi("plain text"), "plain text");
157    assert_eq!(strip_ansi(""), "");
158  }
159
160  #[test]
161  fn strip_ansi_multiple_sequences() {
162    let s = "\x1b[31merror:\x1b[0m \x1b[33mwarning\x1b[0m";
163    assert_eq!(strip_ansi(s), "error: warning");
164  }
165
166  #[test]
167  fn strip_ansi_preserves_unicode() {
168    let s = "\x1b[31mαβγ\x1b[0m";
169    assert_eq!(strip_ansi(s), "αβγ");
170  }
171
172  #[test]
173  fn strip_ansi_handles_incomplete_escape() {
174    // Unterminated ESC[ sequence — we should not hang.
175    // Current impl: scan until 'm' is found. If never found, consumes
176    // the rest of the input. Document that behavior.
177    let s = "\x1b[1;31m hello";
178    // The scan consumes characters until 'm' is found → the 'm' in the
179    // escape closes, then " hello" remains.
180    assert_eq!(strip_ansi(s), " hello");
181  }
182
183  #[test]
184  fn bind_log_and_flush_log_roundtrip() {
185    // Before bind_log, flush_log returns empty.
186    // After bind_log, the buffer is active but empty until a log
187    // message arrives. Since we can't easily exercise the Log impl
188    // without initializing a global logger, just verify the
189    // capture-buffer lifecycle primitives.
190    let before = flush_log();
191    assert!(before.is_empty(), "no active buffer → empty flush");
192
193    bind_log();
194    let after = flush_log();
195    assert!(
196      after.is_empty(),
197      "empty buffer is still empty after bind/flush with no log traffic"
198    );
199  }
