10/// Thread-local log capture buffer. When enabled, log messages are
11/// appended here (without ANSI colors) in addition to stderr.
12#[thread_local]
13static LOG_BUFFER: RefCell<Option<String>> = RefCell::new(None);
14
15/// Start capturing log output into the buffer (Perl: bind_log).
16pub fn bind_log() { *LOG_BUFFER.borrow_mut() = Some(String::new()); }
17
18/// Flush and return the captured log output, stopping capture (Perl: flush_log).
19pub fn flush_log() -> String { LOG_BUFFER.borrow_mut().take().unwrap_or_default() }

21/// Strip ANSI escape sequences from a string for log file output.
22fn strip_ansi(s: &str) -> String {
23  // Match ESC[ ... m sequences
24  let mut result = String::with_capacity(s.len());
25  let mut in_escape = false;
26  for c in s.chars() {
27    if in_escape {
28      if c == 'm' {
29        in_escape = false;
30      }
31    } else if c == '\x1b' {
32      in_escape = true;
33    } else {
34      result.push(c);
35    }
36  }
37  result
38}

64impl log::Log for LatexmlLogger {
65  fn enabled(&self, metadata: &Metadata) -> bool { metadata.level() <= max_level() }
66
67  fn log(&self, record: &Record) {
68    if self.enabled(record.metadata()) {
69      let record_target = record.target();
70      let details = record.args();
71      if record_target == "note" {
72        // simple print here
73        print_stderr!("{}", details.to_string());
74        return;
75      }
76      let category_object = if record_target.is_empty() {
77        "" // "unknown:unknown" ???
78      } else {
79        record_target
80      };
81      // Following the reporting syntax at: http://dlmf.nist.gov/LaTeXML/manual/errorcodes/
82      let severity = if category_object.starts_with("Fatal:") {
83        ""
84      } else {
85        match record.level() {
86          Level::Info => "Info",
87          Level::Warn => "Warn",
88          Level::Error => "Error",
89          Level::Debug => "Debug",
90          Level::Trace => "Trace",
91        }
92      };
93
94      let message = if severity.is_empty() {
95        s!("{} ", category_object)
96      } else {
97        s!("{}:{} ", severity, category_object)
98      };
99      let painted_message = match record.level() {
100        Level::Info => Style::default().paint(message),
101        Level::Warn => Yellow.paint(message),
102        Level::Error => Red.paint(message),
103        Level::Debug => Green.paint(message),
104        _ => White.paint(message),
105      }
106      .to_string()
107        + &details.to_string();
108
109      // Capture to log buffer if active (strip ANSI for clean log text)
110      if let Ok(mut buf) = LOG_BUFFER.try_borrow_mut() {
111        if let Some(ref mut log) = *buf {
112          log.push_str(&strip_ansi(&painted_message));
113          log.push('\n');
114        }
115      }
116
117      println_stderr!("\r{}", painted_message);
118    }
119  }
120
121  fn flush(&self) {}
122}

124/// initialize the logger at a given verbosity `level`
125pub fn init(level: LevelFilter) -> Result<(), SetLoggerError> {
126  log::set_logger(&LOGGER).unwrap();
127  log::set_max_level(level);
128  Ok(())
129}

131#[cfg(test)]
132mod tests {
133  use super::*;

135  #[test]
136  fn strip_ansi_removes_color_codes() {
137    // ESC [ ... m should vanish; other content preserved.
138    let red = "\x1b[31mhello\x1b[0m";
139    assert_eq!(strip_ansi(red), "hello");
140  }
141
142  #[test]
143  fn strip_ansi_noop_on_plain() {
144    assert_eq!(strip_ansi("plain text"), "plain text");
145    assert_eq!(strip_ansi(""), "");
146  }
147
148  #[test]
149  fn strip_ansi_multiple_sequences() {
150    let s = "\x1b[31merror:\x1b[0m \x1b[33mwarning\x1b[0m";
151    assert_eq!(strip_ansi(s), "error: warning");
152  }
153
154  #[test]
155  fn strip_ansi_preserves_unicode() {
156    let s = "\x1b[31mαβγ\x1b[0m";
157    assert_eq!(strip_ansi(s), "αβγ");
158  }
159
160  #[test]
161  fn strip_ansi_handles_incomplete_escape() {
162    // Unterminated ESC[ sequence — we should not hang.
163    // Current impl: scan until 'm' is found. If never found, consumes
164    // the rest of the input. Document that behavior.
165    let s = "\x1b[1;31m hello";
166    // The scan consumes characters until 'm' is found → the 'm' in the
167    // escape closes, then " hello" remains.
168    assert_eq!(strip_ansi(s), " hello");
169  }
170
171  #[test]
172  fn bind_log_and_flush_log_roundtrip() {
173    // Before bind_log, flush_log returns empty.
174    // After bind_log, the buffer is active but empty until a log
175    // message arrives. Since we can't easily exercise the Log impl
176    // without initializing a global logger, just verify the
177    // capture-buffer lifecycle primitives.
178    let before = flush_log();
179    assert!(before.is_empty(), "no active buffer → empty flush");
180
181    bind_log();
182    let after = flush_log();
183    assert!(
184      after.is_empty(),
185      "empty buffer is still empty after bind/flush with no log traffic"
186    );
187  }
