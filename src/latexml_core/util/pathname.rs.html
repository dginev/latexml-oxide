56/// checks if the path is a conforming URL string
57pub fn is_url(path: &str) -> bool { URL_RE.is_match(path) }
58/// checks if the path starts with the "literal:" protocol
59pub fn is_literaldata(data: &str) -> bool { data.starts_with(LITERAL_PROTOCOL) }

61/// check whether a pathname is reloadable as a TeX definition
62pub fn is_reloadable(pathname: &str) -> bool {
63  let (_dir, _name, ext) = split(pathname);
64  // babel.sty exception:
65  // we know the same .ldf file may be reloaded with a different option,
66  // to load an adjacently defined language, so allow that.

69/// absolute paths start with the filesystem root - check if this is one
70pub fn is_absolute(path: &str) -> bool { Path::new(&canonical(path)).is_absolute() }
71/// convert a (possibly relative) file path to an absolute one
72pub fn absolute(path: &str) -> String {
73  Path::new(path)
74    .canonicalize()
75    .unwrap()
76    .to_string_lossy()
77    .to_string()
78}

80/// Split the pathname into components (dir,name,type).
81/// If pathname is absolute, dir starts with volume or '/'
82pub fn split(pathname: &str) -> (String, String, String) {
83  let canonical_pathname = canonical(pathname);
84  let canonical_path = Path::new(&canonical_pathname);
85  let pathdir = match canonical_path.parent() {
86    Some(dir) => dir.to_string_lossy().to_string(),
87    None => String::new(),
88  };
89  let name = match canonical_path.file_name() {
90    Some(n) => n.to_string_lossy().to_string(),
91    None => String::new(),
92  };
93  let pathname_ext = match canonical_path.extension() {
94    Some(e) => e.to_string_lossy().to_string(),
95    None => String::new(),
96  }
97  .to_lowercase();
98  (pathdir, name, pathname_ext)
99}

102pub fn url_split(url: &str) -> (&str, &str) {
103  if let Some(caps) = URL_RE.captures(url) {
104    (
105      caps.get(1).map_or("", |m| m.as_str()),
106      caps.get(2).map_or("", |m| m.as_str()),
107    )
108  } else {
109    (url, "index.tex") // Well, what other default makes sense?
110  }
111}

146/// Note that this returns ONLY recognized protocols!
147pub fn protocol(pathname: &str) -> String {
148  if let Some(cap) = PROTOCOL_RE.captures(pathname) {
149    cap.get(1).map_or(String::new(), |m| m.as_str().to_string())
150  } else if is_literaldata(pathname) {
151    "literal".to_string()
152  } else {
153    "file".to_string()
154  }
155}

157/// combine a directory and a base name into a full path
158pub fn concat(dir: &str, file: &str) -> String {
159  if dir.is_empty() {
160    file.to_owned()
161  } else if file.is_empty() || file == "." {
162    dir.to_owned()
163  } else {
164    let mut path = PathBuf::from(dir);
165    path.push(file);
166    canonical(&path.to_string_lossy())
167  }
168}

170/// It's presumably cheep to concatinate all the pathnames,
171/// relative to the cost of testing for files,
172/// and this simplifies overall.
173pub fn candidate_pathnames(pathname: &str, options: PathnameFindOptions) -> Vec<String> {
174  let mut dirs: Vec<String> = Vec::new();
175  let canonical_pathname = if pathname != "*" {
176    canonical(pathname)
177  } else {
178    pathname.to_owned()
179  };
180
181  let (pathdir, name, pathname_ext) = split(&canonical_pathname);
182
183  let cwd = cwd();
184
185  // generate the set of search paths we'll use.
186  if is_absolute(&canonical_pathname) {
187    dirs.push(pathdir.clone());
188  } else if let Some(paths) = options.paths {
189    for p in paths {
190      // Complete the search paths by prepending current dir to relative paths,
191      let pp_base = if is_absolute(&p) {
192        canonical(&p)
193      } else {
194        concat(&cwd, &p)
195      };
196      let pp = concat(&pp_base, &pathdir);
197      // but only include each dir ONCE
198      if !dirs.contains(&pp) {
199        dirs.push(pp);
200      }
201    }
202  }
203  // Always have the current directory!
204  let from_cwd = concat(&cwd, &pathdir);
205  if !dirs.contains(&from_cwd) {
206    dirs.push(from_cwd);
207  }
208
209  // TODO: The use of INSTALLDIRS should be rethought entirely, as Rust currently doesn't have a
210  // native concept of "installing" a crate and its resources there either needs to be a
211  // more sophisticated build process, OR, a compile step that translates all model/binding
212  // dependencies into rust code. which would allow bundling them side-by-side with the
213  // main application.
214
215  // And, if installation dir specified, append it.
216  if let Some(subdir) = options.installation_subdir {
217    // dirs.extend((*INSTALLDIRS).iter().map(|dir| concat(dir, &subdir)));
218    let full_subdir = concat(&cwd, &subdir);
219    if Path::new(&full_subdir).exists() {
220      dirs.push(full_subdir);
221    } else {
222      let full_subdir_oneup = concat(&s!("{}/..", cwd), &subdir);
223      if Path::new(&full_subdir_oneup).exists() {
224        dirs.push(full_subdir_oneup);
225      }
226    }
227  }
228  // extract the desired extensions.
229  let mut exts = Vec::new();
230  if let Some(ext_vec) = options.extensions {
231    for ext in ext_vec {
232      if ext.is_empty() || pathname_ext == ext.to_lowercase() {
233        exts.push(String::new());
234      } else if ext == "*" {
235        exts.push(s!(".*"));
236        exts.push(String::new());
237      } else {
238        exts.push(s!(".{}", &ext));
239      }
240    }
241  }
242  if exts.is_empty() {
243    exts.push(String::new());
244  }
245
246  let mut paths = Vec::new();
247  // Now, combine; precedence to leading directories.
248  for dir in &dirs {
249    for ext in &exts {
250      if name == "*" {
251        // Unfortunately, we've got to test the file system NOW...
252        //       if ext == ".*" {    // everything
253        // //         opendir(DIR, $dir) or next;
254        // //         push(@paths, map { concat($dir, $_) } grep { !/^\./ } readdir(DIR));
255        // //         closedir(DIR);
256        //       } else {
257        // //         opendir(DIR, $dir) or next;    // ???
258        // //         push(@paths, map { concat($dir, $_) } grep { /\Q$ext\E$/ } readdir(DIR));
259        // //         closedir(DIR); } }
260        //       }
261        //     } else if ext == ".*" { // Unfortunately, we've got to test the file system NOW...
262        // //       opendir(DIR, $dir) or next;      // ???
263        // //       push(@paths, map { concat($dir, $_) } grep { /^\Q$name\E\.\w+$/ } readdir(DIR));
264        // //       closedir(DIR);
265      } else {
266        paths.push(concat(dir, &(name.clone() + ext)));
267      }
268    }
269  }
270  paths
271}

273/// find the requested `pathname` using the `options` search configuration
274pub fn find(pathname: &str, options: PathnameFindOptions) -> Option<String> {
275  if !pathname.is_empty() {
276    let paths = candidate_pathnames(pathname, options);
277    for path in paths {
278      if Path::new(&path).exists() {
279        return Some(path);
280      }
281    }
282  }
283  None
284}

330/// search for a list of candidate names via the external `kpsewhich` utility
331/// returning the first path that is found
332pub fn kpsewhich(candidates: &[&str]) -> Option<String> {
333  if let Some(ref kpse) = *KPSE.lock().unwrap() {
334    for candidate in candidates {
335      if let Some(path) = kpse.find_file(candidate) {
336        return Some(path);
337      }
338    }
339  }
340  None
341}
