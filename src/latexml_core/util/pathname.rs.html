26static HOME_TILDE: &str = "~";
27static HOME_PATH: Lazy<String> = Lazy::new(|| match std::env::var_os("HOME") {

31static PROTOCOL_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"(https|http|ftp):").unwrap());
32// Match Perl LaTeXML's permissive filename behavior: filenames may
33// contain commas, parens, ampersands, etc. that some user paths legitimately
34// use (e.g. `\input{5-Ack,terms}` resolving to `5-Ack,terms.tex`). Only
35// flag genuinely dangerous patterns: shell metacharacters that would
36// enable command injection via kpathsea or `\openin`. Driver: 2308.13679
37// `\input{5-Ack,terms}`. Mirrors Perl's missing nasty-check (Perl simply
38// passes filenames to kpathsea without a pre-filter).
39static PATHNAME_IS_NASTY_RE: Lazy<Regex> =
40  Lazy::new(|| Regex::new(r#"[`$;|<>"\x00\n\r]"#).unwrap());
41// TODO: This is very pragmatic for now, we ought to use a real URL path library long-term
42static URL_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^\w+://(.+)/([^/]+)$").unwrap());
43// Perl `pathname_is_url` is `$pathname =~ /^($PROTOCOL_RE)/` — the protocol is
44// ANCHORED at the start. The previous `is_url` used `URL_RE` (`^\w+://…`),
45// whose leading `\w+` (which includes `_`) matches a filename PREFIX like
46// `myers_http`, so a JabRef `\bibAnnoteFile{myers_http://…/welcome.html_2014}`
47// key (a filename, NOT a URL) read as a URL → `find_file` returned "exists" →
48// `\IfFileExists` took its true branch → the `_` in the key got typeset in
49// text mode ("Script _ can only appear in math mode"; witness 1509.01434).
50// Match Perl: only `http`/`https`/`ftp` followed by `:` at the very start.
51static URL_PREFIX_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(?:https|http|ftp):").unwrap());

53/// Process-global kpathsea handle (cross-thread shared, NOT
54/// `thread_local!`).
55///
56/// **Why an exception to the project's no-`Mutex` rule:**
57/// `kpathsea-rs` wraps a C library that maintains process-wide global
58/// state. Calling `Kpaths::new()` more than once per process (as the
59/// per-thread `thread_local!` pattern would do) re-runs the C-side
60/// init and can corrupt internal tables / leak file descriptors;
61/// concretely on systems without TeXLive installed, the second init
62/// fails with "Can't get directory of program name" and previously
63/// crashed `06_cluster_regressions`. The Mutex here is necessary to
64/// guarantee single-init AND single-active-call semantics for the
65/// underlying non-thread-safe C API. Same class of carve-out as
66/// `latexml_core::watchdog::PRE_EXIT_HOOK`. See
67/// `feedback_no_mutex_use_thread_local` in user memory for the
68/// general rule.
69static KPSE: Lazy<Mutex<Option<Kpaths>>> = Lazy::new(|| Mutex::new(Kpaths::new().ok()));

71/// Force-initialize the kpathsea global state and warm up the per-
72/// format suffix tables.
73///
74/// **Why:** `Kpaths::find_file` lazily inits the kpse format-info
75/// table for each format type the first time a matching filename is
76/// looked up. The chain is
77/// `find_file → guess_format_from_filename → kpathsea_init_format →
78/// kpathsea_init_db → kpathsea_cnf_get → hash_insert_normalized`,
79/// taking ~30-40 ms total across the first dozen lookups. Profile
80/// data on 1910.01256 attributes ~3.5% of wall to that chain.
81///
82/// **What this does:** acquires the `KPSE` mutex once and runs a
83/// single `find_file` probe per common file format. Each probe
84/// guarantees `kpathsea_init_format` runs for that format type, so
85/// every subsequent real lookup hits the post-init fast path.
86///
87/// **Concurrency:** safe to invoke on a background thread spawned at
88/// process start. `KPSE` is process-global (`Lazy<Mutex<…>>`), so the
89/// init done on a background thread is visible to the main thread.
90/// The Mutex briefly serializes the prewarm against the main thread's
91/// first real lookup, but dump load + arg parsing take >50 ms before
92/// digest reaches its first package resolution, by which point the
93/// prewarm is usually finished. Idempotent: re-entry while in flight
94/// is a no-op (lock contention only).
95pub fn prewarm_kpathsea() {
96  let kpse_guard = KPSE.lock().unwrap();
97  let Some(ref kpse) = *kpse_guard else {
98    return;
99  };
100  for sentinel in &[
101    "warmup_lxoxide.tex",
102    "warmup_lxoxide.sty",
103    "warmup_lxoxide.cls",
104    "warmup_lxoxide.def",
105    "warmup_lxoxide.bst",
106    "warmup_lxoxide.fmt",
107    "warmup_lxoxide.afm",
108    "warmup_lxoxide.tfm",
109    "warmup_lxoxide.pfb",
110    "warmup_lxoxide.enc",
111    "warmup_lxoxide.map",
112  ] {
113    // catch_unwind defends against kpathsea-0.2.3 overflow bug (see
114    // kpsewhich docs).
115    let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| kpse.find_file(sentinel)));
116  }
117}

167pub fn is_absolute(path: &str) -> bool { Path::new(&canonical(path)).is_absolute() }
168/// convert a (possibly relative) file path to an absolute one
169///
170/// `std::fs::canonicalize` requires the path to exist; many callers hand
171/// us paths that haven't been resolved yet (e.g. `\import{subdir}{f.sty}`
172/// constructs `subdir/f.sty` before `find_file` probes other dirs).
173/// Mirror Perl's `Cwd::abs_path`-style behavior: produce a lexically
174/// absolute path joined against `current_dir()` when the input is
175/// relative, then run it through our `canonical()` to collapse `.`/`..`
176/// components.
177///
178/// Panics only if `current_dir()` itself fails — that means the cwd was
179/// deleted out from under us, which we cannot safely resolve a relative
180/// path against (and silently returning the input could let a relative
181/// file reference target an attacker-controlled path).
182pub fn absolute(path: &str) -> String {
183  let p = Path::new(path);
184  let joined: PathBuf = if p.is_absolute() {
185    p.to_path_buf()
186  } else {
187    let cwd = std::env::current_dir()
188      .expect("cannot make path absolute: current_dir() failed");
189    cwd.join(p)
190  };
191  canonical(&joined.to_string_lossy())
192}

194/// Split the pathname into components (dir,name,type).
195/// If pathname is absolute, dir starts with volume or '/'
196pub fn split(pathname: &str) -> (String, String, String) {
197  let canonical_pathname = canonical(pathname);
198  let canonical_path = Path::new(&canonical_pathname);
199  let pathdir = match canonical_path.parent() {
200    Some(dir) => dir.to_string_lossy().to_string(),
201    None => String::new(),
202  };
203  let name = match canonical_path.file_stem() {
204    Some(n) => n.to_string_lossy().to_string(),
205    None => String::new(),
206  };
207  // Perl pathname_split preserves case: `$name =~ s/\.([^\.]+)$//`
208  let pathname_ext = match canonical_path.extension() {
209    Some(e) => e.to_string_lossy().to_string(),
210    None => String::new(),
211  };
212  (pathdir, name, pathname_ext)
213}

227/// Canonicalize a pathname by simplifying redundant separators, `.` and `..` components.
228/// Matches Perl's pathname_canonical from Pathname.pm.
229pub fn canonical(pathname: &str) -> String {
230  if is_literaldata(pathname) {
231    return pathname.to_owned();
232  }
233  // Don't call is_absolute, etc, here, cause THEY call US!
234  let home_path: &str = &HOME_PATH;
235
236  let mut pathname = if pathname.starts_with(HOME_TILDE) {
237    pathname.replacen(HOME_TILDE, home_path, 1)
238  } else {
239    pathname.to_string()
240  };
241
242  // Handle URL prefix: strip protocol://host before normalizing path
243  let url_prefix = if let Some(caps) = CANONICAL_URL_RE.captures(&pathname) {
244    let prefix = caps.get(1).unwrap().as_str().to_string();
245    pathname = pathname[prefix.len()..].to_string();
246    Some(prefix)
247  } else {
248    None
249  };
250
251  // Perl: $pathname =~ s|/\./|/|g;
252  while pathname.contains("/./") {
253    pathname = pathname.replace("/./", "/");
254  }
255  // Perl: while ($pathname =~ s|/(?!\.\./)[^/]+/\.\.(/|$)|$1|) { }
256  // Collapse /foo/.. patterns but not /../..
257  // Implemented without lookahead since the regex crate doesn't support it.
258  loop {
259    let mut changed = false;
260    // Find /component/.. where component is not ".."
261    if let Some(dotdot_pos) = pathname.find("/..") {
262      // Check this is actually /../ or /..$ (end of string)
263      let after = dotdot_pos + 3;
264      if after == pathname.len() || pathname.as_bytes().get(after) == Some(&b'/') {
265        // Find the preceding component: look backwards from dotdot_pos for '/'
266        if dotdot_pos > 0 {
267          let prefix = &pathname[..dotdot_pos];
268          if let Some(slash_pos) = prefix.rfind('/') {
269            let component = &pathname[slash_pos + 1..dotdot_pos];
270            if component != ".." && !component.is_empty() {
271              // Replace /component/.. with the trailing part
272              let trail = &pathname[after..];
273              pathname = format!("{}{}", &pathname[..slash_pos], trail);
274              changed = true;
275            }
276          } else if prefix != ".." {
277            // No leading slash, e.g. "foo/.."
278            let trail = &pathname[after..];
279            pathname = if let Some(stripped) = trail.strip_prefix('/') {
280              stripped.to_string()
281            } else {
282              trail.to_string()
283            };
284            if pathname.is_empty() {
285              pathname = ".".to_string();
286            }
287            changed = true;
288          }
289        }
290      }
291    }
292    if !changed {
293      break;
294    }
295  }
296  // Perl: $pathname =~ s|^\./(.)|$1|; — reduce ./foo to foo, but preserve ./
297  if pathname.starts_with("./") && pathname.len() > 2 {
298    pathname = pathname[2..].to_string();
299  }
300  match url_prefix {
301    Some(prefix) => format!("{}{}", prefix, pathname),
302    None => pathname,
303  }
304}

306/// Note that this returns ONLY recognized protocols!
307pub fn protocol(pathname: &str) -> String {
308  if let Some(cap) = PROTOCOL_RE.captures(pathname) {
309    cap.get(1).map_or(String::new(), |m| m.as_str().to_string())
310  } else if is_literaldata(pathname) {
311    "literal".to_string()
312  } else {
313    "file".to_string()
314  }
315}

317/// combine a directory and a base name into a full path
318pub fn concat(dir: &str, file: &str) -> String {
319  if dir.is_empty() {
320    file.to_owned()
321  } else if file.is_empty() || file == "." {
322    dir.to_owned()
323  } else {
324    let mut path = PathBuf::from(dir);
325    path.push(file);
326    canonical(&path.to_string_lossy())
327  }
328}

330/// It's presumably cheep to concatinate all the pathnames,
331/// relative to the cost of testing for files,
332/// and this simplifies overall.
333pub fn candidate_pathnames(pathname: &str, options: PathnameFindOptions) -> Vec<String> {
334  let mut dirs: Vec<String> = Vec::new();
335  let canonical_pathname = if pathname != "*" {
336    canonical(pathname)
337  } else {
338    pathname.to_owned()
339  };
340
341  let (pathdir, name_stem, pathname_ext) = split(&canonical_pathname);
342  // Perl: $name .= '.' . $type if (defined $type) && ($type ne '');
343  // Re-attach the extension to the name, as Perl does after split
344  let name = if !pathname_ext.is_empty() {
345    format!("{}.{}", name_stem, pathname_ext)
346  } else {
347    name_stem
348  };
349
350  let cwd = cwd();
351
352  // generate the set of search paths we'll use.
353  if is_absolute(&canonical_pathname) {
354    dirs.push(pathdir.clone());
355  } else if let Some(paths) = options.paths {
356    for p in paths {
357      // Complete the search paths by prepending current dir to relative paths,
358      let pp_base = if is_absolute(&p) {
359        canonical(&p)
360      } else {
361        concat(&cwd, &p)
362      };
363      let pp = concat(&pp_base, &pathdir);
364      // but only include each dir ONCE
365      if !dirs.contains(&pp) {
366        dirs.push(pp);
367      }
368    }
369  }
370  // Perl: push(@dirs, pathname_concat($cwd, $pathdir)) unless @dirs;
371  // Only add cwd if no search paths were given (fallback)
372  if dirs.is_empty() {
373    let from_cwd = concat(&cwd, &pathdir);
374    if !dirs.contains(&from_cwd) {
375      dirs.push(from_cwd);
376    }
377  }
378
379  // TODO: The use of INSTALLDIRS should be rethought entirely, as Rust currently doesn't have a
380  // native concept of "installing" a crate and its resources there either needs to be a
381  // more sophisticated build process, OR, a compile step that translates all model/binding
382  // dependencies into rust code. which would allow bundling them side-by-side with the
383  // main application.
384
385  // And, if installation dir specified, append it.
386  if let Some(subdir) = options.installation_subdir {
387    // dirs.extend((*INSTALLDIRS).iter().map(|dir| concat(dir, &subdir)));
388    let full_subdir = concat(&cwd, &subdir);
389    if Path::new(&full_subdir).exists() {
390      dirs.push(full_subdir);
391    } else {
392      let full_subdir_oneup = concat(&s!("{}/..", cwd), &subdir);
393      if Path::new(&full_subdir_oneup).exists() {
394        dirs.push(full_subdir_oneup);
395      }
396    }
397  }
398  // extract the desired extensions.
399  // Perl: the extensions from `types` option are applied to the already-reassembled name.
400  // Since name already has its extension, matching an existing extension pushes '' (exact match).
401  let mut exts = Vec::new();
402  if let Some(ext_vec) = options.extensions {
403    for ext in ext_vec {
404      if ext.is_empty() {
405        exts.push(String::new());
406      } else if ext == "*" {
407        exts.push(s!(".*"));
408        exts.push(String::new());
409      } else if !pathname_ext.is_empty() && pathname_ext.eq_ignore_ascii_case(&ext) {
410        // Perl Pathname.pm L353: `if ($pathname =~ /\.\Q$ext\E$/i)` — /i
411        // makes this a case-insensitive extension match; either case of
412        // file extension matches either case of requested type.
413        exts.push(String::new());
414        // Also push the extension itself (Perl pushes both)
415        exts.push(format!(".{}", ext));
416      } else {
417        exts.push(format!(".{}", ext));
418      }
419    }
420  }
421  if exts.is_empty() {
422    exts.push(String::new());
423  }
424
425  let mut paths = Vec::new();
426  // Now, combine; precedence to leading directories.
427  for dir in &dirs {
428    for ext in &exts {
429      if name == "*" {
430        // TODO: wildcard directory listing support
431      } else {
432        paths.push(concat(dir, &(name.clone() + ext)));
433      }
434    }
435  }
436  paths
437}

439/// find the requested `pathname` using the `options` search configuration.
440/// Mirrors Perl `pathname_find` (LaTeXML/Util/Pathname.pm L376-392): directory
441/// search with strict-case match preferred, falling back to a
442/// case-insensitive directory scan. The fallback is required for arxiv
443/// papers shipping uppercase filenames (e.g. `PASJ95.STY` referenced as
444/// `PASJ95.sty`) — Perl's regex pair pushes both strict and `/i` matches
445/// and returns the strict ones if any exist, otherwise the case-insensitive
446/// matches. kpsewhich is the caller's responsibility (see
447/// `LaTeXML::Package::FindFile_aux`).
448pub fn find(pathname: &str, options: PathnameFindOptions) -> Option<String> {
449  if pathname.is_empty() {
450    return None;
451  }
452  let paths = candidate_pathnames(pathname, options);
453  // Pass 1: strict-case existence check (the fast path; matches Perl's
454  // `$local_file =~ m/$regex/` strict regex).
455  for path in &paths {
456    if Path::new(path).exists() {
457      return Some(path.clone());
458    }
459  }
460  // Pass 2: case-insensitive directory scan (Perl's `/i` regex fallback).
461  // Only fired when no strict match existed; mirrors Perl's
462  // `return @paths ? @paths : @nocase_paths` ordering.
463  for path in &paths {
464    let p = Path::new(path);
465    let dir = match p.parent() {
466      Some(d) if !d.as_os_str().is_empty() => d,
467      _ => Path::new("."),
468    };
469    let target = match p.file_name().and_then(|n| n.to_str()) {
470      Some(n) => n,
471      None => continue,
472    };
473    let entries = match std::fs::read_dir(dir) {
474      Ok(e) => e,
475      Err(_) => continue,
476    };
477    for entry in entries.flatten() {
478      if let Some(name) = entry.file_name().to_str() {
479        if name.eq_ignore_ascii_case(target) {
480          return entry.path().to_str().map(String::from);
481        }
482      }
483    }
484  }
485  None
486}

532/// Compose a pathname from dir, name, type components.
533/// Port of Perl's pathname_make(%pieces).
534pub fn make(dir: Option<&str>, name: Option<&str>, ext: Option<&str>) -> String {
535  let mut result = String::new();
536  if let Some(d) = dir {
537    result.push_str(d);
538  }
539  if let Some(n) = name {
540    if !result.is_empty() && !result.ends_with('/') {
541      result.push('/');
542    }
543    result.push_str(n);
544  }
545  if let Some(t) = ext {
546    if !t.is_empty() {
547      result.push('.');
548      result.push_str(t);
549    }
550  }
551  canonical(&result)
552}

554/// Make a pathname relative to a base directory.
555/// Port of Perl's pathname_relative($pathname, $base).
556pub fn relative(pathname: &str, base: &str) -> String {
557  let canonical_pathname = canonical(pathname);
558  if base.is_empty() || !is_absolute(&canonical_pathname) {
559    return canonical_pathname;
560  }
561  let canonical_base = canonical(base);
562  let path = Path::new(&canonical_pathname);
563  let base_path = Path::new(&canonical_base);
564  match path.strip_prefix(base_path) {
565    Ok(rel) => rel.to_string_lossy().to_string(),
566    Err(_) => canonical_pathname,
567  }
568}

570/// Find all matching files (like pathname_findall).
571/// Port of Perl's pathname_findall($pathname, %options).
572pub fn findall(pathname: &str, options: PathnameFindOptions) -> Vec<String> {
573  candidate_pathnames(pathname, options)
574}

576/// search for a list of candidate names via the external `kpsewhich` utility
577/// returning the first path that is found
578pub fn kpsewhich(candidates: &[&str]) -> Option<String> {
579  if let Some(ref kpse) = *KPSE.lock().unwrap() {
580    for candidate in candidates {
581      // kpathsea-0.2.3 panics with "attempt to subtract with overflow" in
582      // `guess_format_from_filename` (lib.rs:92) when `filename.len()` is
583      // shorter than some alt_suffix the format-table holds (the L73 normal-
584      // suffix loop has a `filename.len() > suffix.len()` guard but the L92
585      // alt_suffix loop does NOT). User input like `\usepackage[opt]{}`
586      // produces a `.sty` candidate (empty stem) which trips this. Pre-filter
587      // those: a basename starting with `.` and containing only an extension
588      // is bogus to look up. The catch_unwind below remains as defense-in-
589      // depth. Witnesses: 0711.2664 (`.sty`), cs0503041 (`.sty`).
590      let basename = candidate.rsplit(['/', '\\']).next().unwrap_or(candidate);
591      if basename.starts_with('.') && !basename[1..].contains('.') {
592        continue;
593      }
594      let result =
595        std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| kpse.find_file(candidate)));
596      if let Ok(Some(path)) = result {
597        return Some(path);
598      }
599    }
600  }
601  None
602}

615  fn is_url_schemes() {
616    // Perl `pathname_is_url`: `=~ /^($PROTOCOL_RE)/` — the protocol must be
