1//! Per-job telemetry: phase wall times, counts, and resource peaks.
2//!
3//! See `docs/performance/TELEMETRY.md` for the design contract.
4//!
5//! Default-on instrumentation. Coarse phase wrappers cost ~20ns each
6//! (one `Instant::now()` call); the math-parse histogram update is
7//! the only per-formula instrumentation and is a single atomic
8//! increment of one of 9 `u32` slots.
9//!
10//! Thread-local state. Aggregate at end-of-process via `take()`.
11//! All times in microseconds; counts in their natural unit.
12
13use std::{cell::RefCell, time::Instant};
14
15/// Coarse phase enum. 17 values; bumping requires updating
16/// `Telemetry::write_json` and `tools/perf_phase_summary.py`.
17///
18/// Phase ordering reflects the conversion pipeline order
19/// (Bootstrap → Digest → Build → Rewrite → MathParse →
20/// PostXmlParse → PostScan → Bibliography → Crossref → Graphics →
21/// MathImages → MathmlPres → MathmlCont → Split → Xslt →
22/// Html5Fixups → Serialize) so flat dumps read in execution order.
23#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
24#[repr(u8)]
25pub enum Phase {
26  Bootstrap = 0,
27  Digest = 1,
28  Build = 2,
29  Rewrite = 3,
30  MathParse = 4,
31  /// Parses the XML emitted by core into a `PostDocument`.
32  /// Size-proportional; material on large papers.
33  PostXmlParse = 5,
34  /// Scan phase of latexml_post: ID assignment, label resolution, etc.
35  PostScan = 6,
36  Bibliography = 7,
37  Crossref = 8,
38  /// External-tool dispatch for `\includegraphics` (mutool / pdftocairo / convert / gs).
39  Graphics = 9,
40  /// External-tool dispatch for picture/latex/math image rendering
41  /// (`picture_images.rs` + `latex_images.rs` + `math_images.rs`).
42  MathImages = 10,
43  MathmlPres = 11,
44  MathmlCont = 12,
45  /// Document splitting (only when `--split` is on; otherwise 0).
46  Split = 13,
47  Xslt = 14,
48  /// Final HTML tweaks after XSLT (asset paths, header/footer).
49  Html5Fixups = 15,
50  Serialize = 16,
51}

79/// Math-parse time bucket boundaries in microseconds.
80/// `bucket(us)` returns 0..=8.
81const BUCKET_BOUNDS_US: [u64; 8] = [500, 1_000, 2_000, 5_000, 10_000, 20_000, 50_000, 100_000];
82
83fn bucket_for(us: u64) -> usize {
84  for (i, b) in BUCKET_BOUNDS_US.iter().enumerate() {
85    if us < *b {
86      return i;
87    }
88  }
89  8
90}

168thread_local! {
169  static STATE: RefCell<Telemetry> = RefCell::new(Telemetry::default());
170  // Phase stack: each entry is (phase, started_at). Time accrues only
171  // to the innermost (top-of-stack) phase.
172  static STACK: RefCell<Vec<(Phase, Instant)>> = const { RefCell::new(Vec::new()) };
173}
174
175/// Begin a phase. Subsequent time accrues to this phase until the
176/// matching `phase_exit()` (or the `PhaseGuard` returned by
177/// [`phase`]) drops.
178pub fn phase_enter(p: Phase) {
179  let now = Instant::now();
180  STACK.with(|s| {
181    let mut stack = s.borrow_mut();
182    // If there's a parent phase, charge accumulated wall to it
183    // before we steal the clock.
184    if let Some((parent, started)) = stack.last_mut() {
185      let dt = now.saturating_duration_since(*started).as_micros() as u64;
186      let parent = *parent;
187      STATE.with(|st| st.borrow_mut().phase_us[parent as usize] += dt);
188      *started = now;
189    }
190    stack.push((p, now));
191  });
192}
193
194/// End the innermost phase.
195pub fn phase_exit() {
196  let now = Instant::now();
197  STACK.with(|s| {
198    let mut stack = s.borrow_mut();
199    let (p, started) = stack
200      .pop()
201      .expect("telemetry::phase_exit called without matching phase_enter");
202    let dt = now.saturating_duration_since(started).as_micros() as u64;
203    STATE.with(|st| st.borrow_mut().phase_us[p as usize] += dt);
204    // Reset parent's start so it doesn't double-count our time.
205    if let Some((_, started_parent)) = stack.last_mut() {
206      *started_parent = now;
207    }
208  });
209}
210
211/// RAII guard returned by [`phase`]. Calls `phase_exit` on drop.
212pub struct PhaseGuard {
213  _private: (),
214}
215
216impl Drop for PhaseGuard {
217  fn drop(&mut self) { phase_exit(); }
218}
219
220/// Convenience: `let _g = telemetry::phase(Phase::Digest);`
221pub fn phase(p: Phase) -> PhaseGuard {
222  phase_enter(p);
223  PhaseGuard { _private: () }
224}

235/// Record one math parse: total time and number of successful parses
236/// returned (the Marpa parser may produce multiple ASF derivations
237/// for one input). Updates the histogram bucket for the elapsed time.
238pub fn record_math_parse(us: u64, parses: u32) {
239  STATE.with(|s| {
240    let mut t = s.borrow_mut();
241    t.math_parse_attempts += 1;
242    t.math_parse_count += parses as u64;
243    t.math_parse_buckets[bucket_for(us)] += 1;
244  });
245}

293/// Take the current telemetry record, replacing it with a fresh
294/// default. Use at end-of-process to serialize the result.
295pub fn take() -> Telemetry { STATE.with(|s| std::mem::take(&mut *s.borrow_mut())) }

302fn write_json_string(out: &mut String, s: &str) {
303  out.push('"');
304  for c in s.chars() {
305    match c {
306      '"' => out.push_str("\\\""),
307      '\\' => out.push_str("\\\\"),
308      '\n' => out.push_str("\\n"),
309      '\r' => out.push_str("\\r"),
310      '\t' => out.push_str("\\t"),
311      c if (c as u32) < 0x20 => {
312        use std::fmt::Write;
313        write!(out, "\\u{:04x}", c as u32).unwrap();
314      },
315      c => out.push(c),
316    }
317  }
318  out.push('"');
319}

321impl Telemetry {
322  /// Serialize as a single-line JSON object (suitable for JSONL).
323  /// Hand-written to avoid pulling serde into latexml_core.
324  // The `field!` macro always writes `first = false` after emitting; on the
325  // very last field that final write is naturally dead. Silence the warning.
326  #[allow(unused_assignments)]
327  pub fn to_json_line(&self) -> String {
328    use std::fmt::Write;
329    let mut s = String::with_capacity(1024);
330    s.push('{');
331
332    let mut first = true;
333    macro_rules! field {
334      ($name:literal, $val:expr_2021) => {{
335        if !first {
336          s.push(',');
337        }
338        first = false;
339        s.push('"');
340        s.push_str($name);
341        s.push_str("\":");
342        write!(s, "{}", $val).unwrap();
343      }};
344    }
345    macro_rules! field_str {
346      ($name:literal, $val:expr_2021) => {{
347        if !first {
348          s.push(',');
349        }
350        first = false;
351        s.push('"');
352        s.push_str($name);
353        s.push_str("\":");
354        write_json_string(&mut s, $val);
355      }};
356    }
357    macro_rules! field_array_u64 {
358      ($name:literal, $arr:expr_2021) => {{
359        if !first {
360          s.push(',');
361        }
362        first = false;
363        s.push('"');
364        s.push_str($name);
365        s.push_str("\":[");
366        for (i, v) in $arr.iter().enumerate() {
367          if i > 0 {
368            s.push(',');
369          }
370          write!(s, "{}", v).unwrap();
371        }
372        s.push(']');
373      }};
374    }
375    macro_rules! field_array_u32 {
376      ($name:literal, $arr:expr_2021) => {{
377        field_array_u64!($name, $arr);
378      }};
379    }
380
381    field_str!("paper_id", &self.paper_id);
382    field_str!("git_sha", &self.git_sha);
383    field_str!("cmdline", &self.cmdline);
384    field_str!("host", &self.host);
385    field!("timeout_s", self.timeout_s);
386    field!("schema_version", self.schema_version);
387    field!("wall_us", self.wall_us);
388    field_array_u64!("phase_us", &self.phase_us);
389    // Per-phase aliases for grep convenience
390    for (i, val) in self.phase_us.iter().enumerate() {
391      let phase = match i {
392        0 => Phase::Bootstrap,
393        1 => Phase::Digest,
394        2 => Phase::Build,
395        3 => Phase::Rewrite,
396        4 => Phase::MathParse,
397        5 => Phase::PostXmlParse,
398        6 => Phase::PostScan,
399        7 => Phase::Bibliography,
400        8 => Phase::Crossref,
401        9 => Phase::Graphics,
402        10 => Phase::MathImages,
403        11 => Phase::MathmlPres,
404        12 => Phase::MathmlCont,
405        13 => Phase::Split,
406        14 => Phase::Xslt,
407        15 => Phase::Html5Fixups,
408        16 => Phase::Serialize,
409        _ => unreachable!(),
410      };
411      s.push_str(",\"phase_");
412      s.push_str(phase.as_str());
413      s.push_str("_us\":");
414      write!(s, "{}", val).unwrap();
415    }
416    field!("formulae", self.formulae);
417    field!("math_parse_attempts", self.math_parse_attempts);
418    field!("math_parse_count", self.math_parse_count);
419    field_array_u32!("math_parse_buckets", &self.math_parse_buckets);
420    field!("graphics_assets", self.graphics_assets);
421    field!("graphics_subprocess_count", self.graphics_subprocess_count);
422    field!("db_objects", self.db_objects);
423    field!("output_bytes", self.output_bytes);
424    field!("warnings", self.warnings);
425    field!("errors", self.errors);
426    field!("fatal_errors", self.fatal_errors);
427    field!("external_tool_count", self.external_tool_count);
428    field!("max_rss_kb", self.max_rss_kb);
429    field!("child_user_us", self.child_user_us);
430    field!("child_sys_us", self.child_sys_us);
431    field_str!("category", &self.category);
432    field!("exit_code", self.exit_code);
433
434    s.push('}');
435    s
436  }
437}

447  #[test]
448  fn bucket_boundaries() {
449    assert_eq!(bucket_for(0), 0);
450    assert_eq!(bucket_for(499), 0);
451    assert_eq!(bucket_for(500), 1);
452    assert_eq!(bucket_for(999), 1);
453    assert_eq!(bucket_for(1_000), 2);
454    assert_eq!(bucket_for(99_999), 7);
455    assert_eq!(bucket_for(100_000), 8);
456    assert_eq!(bucket_for(1_000_000), 8);
457  }

459  #[test]
460  fn nested_phases_charge_innermost_only() {
461    let _g_outer = phase(Phase::Bootstrap);
462    sleep(Duration::from_micros(200));
463    {
464      let _g_inner = phase(Phase::Digest);
465      sleep(Duration::from_micros(500));
466    }
467    sleep(Duration::from_micros(200));
468    drop(_g_outer);
469
470    let t = take();
471    // Bootstrap should have accumulated time outside the Digest scope.
472    assert!(
473      t.phase_us[Phase::Bootstrap as usize] >= 300,
474      "bootstrap got {}",
475      t.phase_us[0]
476    );
477    // Digest should have ~500us (with slack for sleep imprecision).
478    assert!(
479      t.phase_us[Phase::Digest as usize] >= 400,
480      "digest got {}",
481      t.phase_us[Phase::Digest as usize]
482    );
483  }

486  fn json_round_trip_basic_fields() {
487    set_paper_id("0901.0001");

490    set_category("ok");
491    set_exit_code(0);
