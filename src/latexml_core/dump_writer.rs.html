1//! Writer for Rust-native kernel dump files.
2//!
3//! Serializes the State diff (from snapshot/diff_from_snapshot) into a
4//! compact text format that can be loaded back efficiently.
5//!
6//! This is the "DumpFile" equivalent from Perl's TeX_Job.pool.ltxml.
7//! The workflow follows Perl's Makefile.PL "make formats" scaffold:
8//!   1. take_snapshot() — before processing latex.ltx
9//!   2. Process latex.ltx through the engine
10//!   3. diff_snapshot() — find what changed
11//!   4. write_dump() — serialize the diff
12//!   5. At runtime: load_dump() restores the state
13

25/// Write a state diff to a dump file.
26/// Returns the number of entries successfully written.
27///
28/// Writes in three ordered sections, matching Perl's
29/// `TeX_Job.pool.ltxml::DumpFile` which separates
30/// `@cmds_early` / `@cmds` / `@cmds_late`:
31///
32///   1. **cmds_early** — M:PA / M:MPA let-aliases whose **target pre-existed** in the bootstrap
33///      snapshot (e.g. `\tex_let:D → \let`, `\tex_def:D → \def`). Applied first because their
34///      targets are always available (they're in the bootstrap pool we loaded unconditionally
35///      before this dump).
36///   2. **cmds** (regular) — V / M:E / M:T / R / C / LC / UC / SC / MC / DC entries: data,
37///      expandable definitions, registers, codes.
38///   3. **cmds_late** — M:PA / M:MPA let-aliases whose **target is defined by this dump**. Applied
39///      last so the target is installed before the alias fires.
40///
41/// The early/late split requires the caller (ini_tex) to have staged a
42/// snapshot via `state::stage_snapshot("bootstrap")`. If no snapshot is
43/// available, all aliases are emitted as `late` (safe — they just wait
44/// until after regular entries).
45pub fn write_dump(
46  path: &Path,
47  entries: &[(TableName, crate::common::arena::SymStr, Stored)],
48) -> Result<usize, String> {
49  let mut file =
50    std::fs::File::create(path).map_err(|e| format!("Failed to create dump file: {}", e))?;
51
52  writeln!(file, "# latexml-oxide kernel dump v3").ok();
53  writeln!(
54    file,
55    "# Generated by dump_writer, {} entries",
56    entries.len()
57  )
58  .ok();
59  writeln!(file, "# Format: <table>\\t<key>\\t<data>").ok();
60  writeln!(file, "#   <table> = M | V | C | LC | UC | SC | MC | DC").ok();
61  writeln!(file, "#   <data> depends on <table>:").ok();
62  writeln!(
63    file,
64    "#     M/E:   E\\t<cs>\\t<nargs>\\t<flags>\\t<body>\\t<proto>\\t<v3_params>"
65  )
66  .ok();
67  writeln!(
68    file,
69    "#            v3_params encodes each Parameter structurally so Until:/Match:"
70  )
71  .ok();
72  writeln!(
73    file,
74    "#            delimiter tokens round-trip (see docs/DUMP_FORMAT_PERL_ANALYSIS.md)."
75  )
76  .ok();
77  writeln!(
78    file,
79    "#     M/PA:  PA\\t<target_cs>   (MPA for math primitives)"
80  )
81  .ok();
82  writeln!(file, "#     M/T:   T\\t<meaning_token>").ok();
83  writeln!(
84    file,
85    "#     M/R:   R\\t<cs>\\t<rtype>\\t<value>[\\t<mathglyph>]"
86  )
87  .ok();
88  writeln!(
89    file,
90    "#     M/N:   N                     (None-meaning sentinel)"
91  )
92  .ok();
93  writeln!(
94    file,
95    "#     V:     <type>\\t<value>       (B,I,S,CH,T,TK,N,F,G,MG,D,MD,VD)"
96  )
97  .ok();
98  writeln!(file, "#     C:                  CC\\t<u8>   (catcode)").ok();
99  writeln!(
100    file,
101    "#     DC/LC/UC/SC/MC:     CH\\t<u32>  (char-indexed table entries)"
102  )
103  .ok();
104  writeln!(file, "#   <flags> = subset of {{L, P}} (long, protected)").ok();
105  writeln!(
106    file,
107    "#   All keys and proto are url-encoded (%09 for tab etc.); token body"
108  )
109  .ok();
110  writeln!(
111    file,
112    "#   tokens use the form <catcode_int>:<url_encoded_text> joined by commas."
113  )
114  .ok();
115
116  // Read the bootstrap snapshot so we can classify each alias's target
117  // as pre-existing (→ early) or defined-during-load (→ late).
118  let bootstrap_snap = crate::state::get_staged_snapshot("bootstrap");
119
120  // Pre-serialize + partition. Skipped keys (ver@..., unserializable
121  // values) fall out naturally.
122  let mut early_aliases: Vec<(String, String, String)> = Vec::new();
123  let mut regular: Vec<(String, String, String)> = Vec::new();
124  let mut late_aliases: Vec<(String, String, String)> = Vec::new();
125  let mut skipped = 0usize;
126
127  // expl3 implements intarrays by stashing values in `\fontdimen<idx>\<font>`
128  // slots, picking `cmr10` at various tiny `at <N>sp` instantiations to
129  // get one font instance per intarray. Each slot is normally written as
130  // an individual `V\tfontdimen_fontinfo_<font> at <Nsp>_<idx>\tD\t<val>`
131  // record; that produces ~89k V-records (≈40% of the dump) for an
132  // initialized expl3 + LaTeX kernel. We group them by (font,size) and
133  // emit a single `IA` record per intarray with the values RLE-encoded
134  // — same in-memory state after replay, ~10× smaller on disk.
135  // See `docs/PERL_LOADFORMAT_AUDIT.md` ("Fontdimen/intarray storage").
136  let mut fontdimen_groups: HashMap<String, Vec<(u32, i64)>> = HashMap::default();
137
138  for (table, key, value) in entries {
139    let table_code = table_to_code(*table);
140    let key_str = arena::with(*key, |s| s.to_string());
141
142    // Skip \ver@ package version markers (runtime metadata, not definitions)
143    if key_str.starts_with("\\ver@") {
144      skipped += 1;
145      continue;
146    }
147
148    // Skip MAX_ERRORS — `ini_tex.rs:189` raises it to 1_000_000 during
149    // dump-build (so raw latex.ltx + expl3-code.tex can run through
150    // transient errors). It must NOT leak into runtime conversions:
151    // a 1M cap lets runaway error cascades (e.g. AmS-TeX `\cases` mis-
152    // parse → 1M `\hbox`/`&` per paper, math0205073) bypass the
153    // 10000 default. dump_reader has a parallel filter for the
154    // already-shipped dumps; this filter keeps future regenerations
155    // clean.
156    if matches!(*table, TableName::Value) && key_str == "MAX_ERRORS" {
157      skipped += 1;
158      continue;
159    }
160
161    // Skip \@currname / \@currext file-IO bookkeeping. These are
162    // assigned per-document during `\input` by `read_input_file_recursive`
163    // (see `binding/content.rs:262-263`) to the literal filename's
164    // tokens, so the snapshot captures the LAST opened file's name
165    // ("plain.tex" / "latex.ltx"). Perl's `TeX_FileIO.pool.ltxml:28-29`
166    // initializes them via `Let('\@currname','\lx@empty')` before any
167    // file load, and Perl's plain_dump.pool.ltxml omits them — so
168    // post-dump they remain at the `\lx@empty` baseline. Matching that
169    // behavior here keeps us file-IO-state-agnostic in the dump.
170    if matches!(*table, TableName::Meaning) && (key_str == "\\@currname" || key_str == "\\@currext")
171    {
172      skipped += 1;
173      continue;
174    }
175
176    // Skip token-list register VALUES whose body contains a self-`\the`
177    // reference. latex.ltx L10564-10580 sets `\frozen@everymath` =
178    // `{... \the\everymath}` then allocates a NEW `\everymath` via
179    // `\newtoks`. Our `\newtoks` doesn't fully detach the new register's
180    // slot from the frozen alias's, so the body lands on `\everymath`'s
181    // value slot — `\the\everymath` then expands recursively at math
182    // entry, blowing the token limit. Until the `\newtoks`/`\let`
183    // slot-aliasing is fixed at the dialect level, the safest cure is
184    // to drop the corrupted self-referential capture; runtime
185    // re-evaluation of latex.ltx's hook chain (or LaTeXML's own
186    // `\everymath`/`\everydisplay` setup) repopulates correctly.
187    if matches!(*table, TableName::Value)
188      && matches!(
189        key_str.as_str(),
190        "\\everymath"
191          | "\\everydisplay"
192          | "\\everyhbox"
193          | "\\everyvbox"
194          | "\\everycr"
195          | "\\everyjob"
196          | "\\everypar"
197          | "\\everyeof"
198      )
199    {
200      // Confirm the loop pattern before dropping (don't suppress
201      // valid hook contributions on engines with sound aliasing).
202      if let Stored::Tokens(ref tks) = value {
203        let body = tks.unlist_ref();
204        let needle_cs = key_str.clone();
205        let has_self_the = body.iter().any(|t| t.with_str(|s| s == "\\the"))
206          && body.iter().any(|t| t.with_str(|s| s == needle_cs));
207        if has_self_the {
208          skipped += 1;
209          continue;
210        }
211      }
212    }
213
214    // Perl #2771 (2026-03-13) + upstream IGNORED_SYMBOLS: these are
215    // runtime-only Value entries — control-flow counters, and large
216    // registered-rule tables that can't meaningfully round-trip through
217    // the dump (they hold closures, cross-refs, or re-populate during
218    // engine init anyway). Match Perl's TeX_Job.pool.ltxml list.
219    if matches!(*table, TableName::Value)
220      && matches!(
221        key_str.as_str(),
222        "if_count"
223          | "absorb_count"
224          | "if_stack"
225          | "DOCUMENT_REWRITE_RULES"
226          | "PARAMETER_TYPES"
227          | "TAG_PROPERTIES"
228          | "MATH_LIGATURES"
229          | "TEXT_LIGATURES"
230      )
231    {
232      skipped += 1;
233      continue;
234    }
235
236    // Perl IGNORED_SYMBOLS: meaning:\lnot, meaning:\to are skipped because
237    // pre-2017 TeXlive let-aliased \lnot → \neg and \to → \rightarrow,
238    // which would gratuitously diverge tests across TL versions.
239    // Mirror Perl `TeX_Job.pool.ltxml` IGNORED_SYMBOLS L104-107.
240    if matches!(*table, TableName::Meaning) && matches!(key_str.as_str(), "\\lnot" | "\\to") {
241      skipped += 1;
242      continue;
243    }
244
245    // Intarray slot consolidation — see fontdimen_groups comment above.
246    if matches!(*table, TableName::Value) {
247      if let Some((prefix, idx)) = parse_fontdimen_key(&key_str) {
248        if let Stored::Dimension(d) = value {
249          fontdimen_groups
250            .entry(prefix.to_string())
251            .or_default()
252            .push((idx, d.0));
253          continue;
254        }
255      }
256    }
257
258    let Some(serialized) = serialize_stored(value) else {
259      skipped += 1;
260      continue;
261    };
262
263    // A let-alias entry is an M row whose serialized form starts with
264    // `PA\t` or `MPA\t`. Nothing else produces those prefixes.
265    let is_alias = matches!(*table, TableName::Meaning)
266      && (serialized.starts_with("PA\t") || serialized.starts_with("MPA\t"));
267
268    let row = (table_code.to_string(), url_encode(&key_str), serialized);
269    if is_alias {
270      // Extract the target CS from "PA\t<target>" / "MPA\t<target>"
271      // (we already filtered self-aliases in serialize_stored, so the
272      //  target is always a different CS than the key).
273      let target_raw = row.2.split('\t').nth(1).unwrap_or("");
274      let target = crate::dump_reader::url_decode(target_raw);
275      let target_sym = arena::pin(&target);
276      // Target pre-existed iff it was in the bootstrap meaning table.
277      let pre_existed = bootstrap_snap
278        .as_ref()
279        .is_some_and(|snap| snap.contains_key(&(TableName::Meaning, target_sym)));
280      if pre_existed {
281        early_aliases.push(row);
282      } else {
283        late_aliases.push(row);
284      }
285    } else {
286      regular.push(row);
287    }
288  }
289
290  // Emit one IA record per intarray group. Fall back to individual V
291  // records for any non-dense group (defensive: dump_reader only knows
292  // how to expand contiguous 1..N runs).
293  let mut ia_count = 0usize;
294  let mut ia_fallback_v = 0usize;
295  for (prefix, mut slots) in fontdimen_groups.into_iter() {
296    slots.sort_by_key(|s| s.0);
297    let dense = slots
298      .iter()
299      .enumerate()
300      .all(|(i, (idx, _))| *idx == (i as u32 + 1));
301    if !dense {
302      eprintln!(
303        "[dump_writer] non-dense intarray {:?} ({} slots) — emitting as individual V records",
304        prefix,
305        slots.len()
306      );
307      for (idx, val) in &slots {
308        let key = format!("{}_{}", prefix, idx);
309        regular.push(("V".to_string(), url_encode(&key), format!("D\t{}", val)));
310        ia_fallback_v += 1;
311      }
312      continue;
313    }
314    let values: Vec<i64> = slots.into_iter().map(|(_, v)| v).collect();
315    let rle = rle_encode_i64(&values);
316    let body = format!("{}\t{}", values.len(), rle);
317    regular.push(("IA".to_string(), url_encode(&prefix), body));
318    ia_count += 1;
319  }
320  if ia_count > 0 || ia_fallback_v > 0 {
321    eprintln!(
322      "[dump_writer] intarray consolidation: {} IA records, {} V fallbacks",
323      ia_count, ia_fallback_v
324    );
325  }
326
327  writeln!(
328    file,
329    "# Section 0: early let-aliases ({}) — target in bootstrap",
330    early_aliases.len()
331  )
332  .ok();
333  for (t, k, v) in &early_aliases {
334    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
335  }
336  writeln!(file, "# Section 1: regular entries ({})", regular.len()).ok();
337  for (t, k, v) in &regular {
338    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
339  }
340  writeln!(
341    file,
342    "# Section 2: late let-aliases ({}) — target in Section 1",
343    late_aliases.len()
344  )
345  .ok();
346  for (t, k, v) in &late_aliases {
347    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
348  }
349
350  let count = early_aliases.len() + regular.len() + late_aliases.len();
351  writeln!(file, "# Written: {}, Skipped: {}", count, skipped).ok();
352  eprintln!(
353    "[dump_writer] Wrote {} entries to {} ({} skipped; {} early + {} regular + {} late)",
354    count,
355    path.display(),
356    skipped,
357    early_aliases.len(),
358    regular.len(),
359    late_aliases.len(),
360  );
361
362  Ok(count)
363}

365fn table_to_code(t: TableName) -> &'static str {
366  match t {
367    TableName::Meaning => "M",
368    TableName::Value => "V",
369    TableName::Catcode => "C",
370    TableName::Mathcode => "MC",
371    TableName::Sfcode => "SC",
372    TableName::Lccode => "LC",
373    TableName::Uccode => "UC",
374    TableName::Delcode => "DC",
375    TableName::Stash => "ST",
376    TableName::StashActive => "SA",
377  }
378}

380/// Serialize a Stored value to a type-tag + data string.
381/// Returns None if the value can't be serialized.
382fn serialize_stored(stored: &Stored) -> Option<String> {
383  match stored {
384    Stored::None => Some("N".to_string()),
385    Stored::Bool(b) => Some(format!("B\t{}", if *b { "1" } else { "0" })),
386    Stored::Int(i) => Some(format!("I\t{}", i)),
387    Stored::String(s) => Some(format!(
388      "S\t{}",
389      url_encode(&arena::with(*s, |s| s.to_string()))
390    )),
391    Stored::Charcode(c) => Some(format!("CH\t{}", c)),
392    Stored::Catcode(cc) => Some(format!("CC\t{}", u8::from(*cc))),
393    Stored::Token(t) => Some(format!("T\t{}", serialize_token(t))),
394    Stored::Tokens(tks) => {
395      let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
396      Some(format!("TK\t{}", tok_strs.join(",")))
397    },
398    Stored::Expandable(exp) => {
399      use crate::common::object::Object;
400      let cs_name = exp.get_cs().with_str(url_encode);
401      let nargs = exp.get_num_args();
402      let mut flags = String::new();
403      if exp.is_long {
404        flags.push('L');
405      }
406      if exp.is_protected() {
407        flags.push('P');
408      }
409      // Two parallel parameter encodings:
410      //
411      //   - `proto` (v2, 5th field) — `Parameters::stringify()`, a flat space-separated prototype
412      //     string. Round-trips simple specs but not `Until:\end{verbatim}` or other
413      //     brace-in-delimiter forms.
414      //   - `v3_params` (v3, 6th field) — per-Parameter structured record carrying (name, spec,
415      //     flags, extras). Modeled on Perl's `P(type, spec, extra=>[T(...)])`; see
416      //     `docs/DUMP_FORMAT_PERL_ANALYSIS.md`. Bypasses `parse_parameters` at load time so
417      //     delimited params round-trip intact.
418      //
419      // We emit both so older readers still work, and newer readers can
420      // prefer v3 transparently.
421      let proto = exp
422        .get_parameters()
423        .map(|p| p.stringify())
424        .unwrap_or_default();
425      let proto_encoded = url_encode(&proto);
426      let v3_params = exp
427        .get_parameters()
428        .map(serialize_parameters_v3)
429        .unwrap_or_default();
430      match exp.get_expansion() {
431        Some(crate::definition::ExpansionBody::Tokens(tks)) => {
432          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
433          Some(format!(
434            "E\t{}\t{}\t{}\t{}\t{}\t{}",
435            cs_name,
436            nargs,
437            flags,
438            tok_strs.join(","),
439            proto_encoded,
440            v3_params
441          ))
442        },
443        // No body at all (e.g. `DefMacro!("\\@gobble{}", None)`): serializes
444        // as an E-entry with empty token body. The reader reconstructs an
445        // Expandable with empty Tokens + the original paramlist, which at
446        // expansion time reads and discards its arguments — semantically
447        // identical to the no-body form's runtime behavior (expandable.rs
448        // `invoke` matches `None` → reads args, returns NO_TOKENS).
449        None => Some(format!(
450          "E\t{}\t{}\t{}\t\t{}\t{}",
451          cs_name, nargs, flags, proto_encoded, v3_params
452        )),
453        // Closure-based body — can't serialize the closure itself, BUT
454        // if the dump entry is a `\let`-alias to a closure-Expandable
455        // primitive (e.g. `\let \tex_expandafter:D \expandafter`), we
456        // CAN capture it via PA. Mirrors the `Stored::Primitive` /
457        // `Stored::Conditional` arms below. Without this, every
458        // `\let \X \expandafter`-style alias is silently dropped from
459        // the dump — the cause of `\tex_expandafter:D`,
460        // `\tex_unexpanded:D`, `\tex_the:D` being missing in
461        // `latex.dump.txt` even though expl3-code.tex L357+ aliases them.
462        //
463        // The dump_reader's add-only policy at load-time skips entries
464        // whose key is already defined (the canonical CS lives in the
465        // compiled engine), so self-aliases are no-ops. Real let-aliases
466        // (key != cs) replay via `state::let_i`.
467        Some(crate::definition::ExpansionBody::Closure(_)) => Some(format!("PA\t{}", cs_name)),
468      }
469    },
470    // Closure-based primitives can't be serialized directly — but if the entry's
471    // CS differs from the primitive's own CS, it's a `\let`-alias, and we CAN
472    // capture the alias so the dump reader can replay `\let <key> <target>` at
473    // load time. This is how `\tex_let:D`, `\tex_def:D`, and hundreds of other
474    // expl3-renamed primitives stay reachable across dump boundaries (the
475    // defining assignment `\let \tex_let:D \let` in expl3-code.tex needs to
476    // survive the dump for `\usepackage{expl3}`'s guard `\ifx\csname
477    // tex_let:D\endcsname\relax` to short-circuit).
478    //
479    // NOTE: the KEY we're saving IS the dump entry's first column; the VALUE
480    // we write is the primitive's OWN CS name. Dump reader compares these:
481    // if equal, it's the "primary" primitive (already in bindings) and skipped;
482    // if different, it's an alias — replay `\let` at load time.
483    Stored::Primitive(p) => {
484      // Perl `Core/Dumper.pm::dump_primitive` (L383-389): if a primitive has
485      // a `font` directive AND no replacement closure (i.e. defined by `\font`),
486      // emit `FD(<cs>, <fontID>)` instead of generic Primitive serialization.
487      // `font_id` is set on `\font`-defined primitives by `engine/tex_fonts.rs`'s
488      // post-define hook (the Rust path) — it carries the value-table key under
489      // which the Stored::Font lives. The reader installs a synthesized
490      // Primitive whose before_digest looks up that font and merges, mirroring
491      // `LaTeXML::Core::Definition::FontDef::invoke` (FontDef.pm L38-45).
492      if let Some(fid) = p.font_id {
493        let fid_str = arena::with(fid, |s| s.to_string());
494        return Some(format!("FD\t{}", url_encode(&fid_str)));
495      }
496      let target_cs = p.cs.with_str(url_encode);
497      Some(format!("PA\t{}", target_cs))
498    },
499    Stored::Font(f) => {
500      // Perl `dump_font` (Core/Dumper.pm L281-284) emits `F(... components ...)`.
501      // Rust's `Font` is a flat options struct, so we serialize the non-None
502      // fields as `key=value` pairs separated by `\x1f` (unit-separator), with
503      // values url-encoded for tab/newline-safety. Reader inverts in dump_reader.rs.
504      let mut parts = Vec::with_capacity(8);
505      if let Some(ref name) = f.name {
506        parts.push(format!("name={}", url_encode(name)));
507      }
508      if let Some(size) = f.size {
509        parts.push(format!("size={}", size));
510      }
511      if let Some(ref family) = f.family {
512        parts.push(format!("family={}", url_encode(family)));
513      }
514      if let Some(ref series) = f.series {
515        parts.push(format!("series={}", url_encode(series)));
516      }
517      if let Some(ref shape) = f.shape {
518        parts.push(format!("shape={}", url_encode(shape)));
519      }
520      if let Some(ref encoding) = f.encoding {
521        parts.push(format!("encoding={}", url_encode(encoding)));
522      }
523      if let Some(ref language) = f.language {
524        parts.push(format!("language={}", url_encode(language)));
525      }
526      if let Some(ref mathstyle) = f.mathstyle {
527        parts.push(format!("mathstyle={}", url_encode(mathstyle)));
528      }
529      if let Some(ref opacity) = f.opacity {
530        parts.push(format!("opacity={}", url_encode(opacity)));
531      }
532      if let Some(scale) = f.scale {
533        parts.push(format!("scale={}", scale));
534      }
535      if let Some(b) = f.emph {
536        parts.push(format!("emph={}", if b { 1 } else { 0 }));
537      }
538      if let Some(b) = f.scripted {
539        parts.push(format!("scripted={}", if b { 1 } else { 0 }));
540      }
541      if let Some(step) = f.mathstylestep {
542        parts.push(format!("mathstylestep={}", step));
543      }
544      if let Some(flags) = f.flags {
545        parts.push(format!("flags={}", flags));
546      }
547      Some(format!("F\t{}", parts.join("\x1f")))
548    },
549    Stored::MathPrimitive(p) => {
550      let target_cs = p.cs.with_str(url_encode);
551      Some(format!("MPA\t{}", target_cs))
552    },
553    // Conditionals — same logic as Primitives. \ifx, \ifnum, \if etc. are
554    // stored as Stored::Conditional with their canonical CS field. When
555    // expl3-code.tex does `\let \if_meaning:w \ifx`, the new entry shares
556    // the same Conditional Rc, with cs = \ifx. Capturing this as a PA
557    // alias lets the dump reader replay `\let \if_meaning:w \ifx` at load
558    // time. Without this, every expl3 conditional alias is silently lost
559    // from the dump and `\if_meaning:w`-style invocations fail at runtime.
560    Stored::Conditional(c) => {
561      let target_cs = c.cs.with_str(url_encode);
562      Some(format!("PA\t{}", target_cs))
563    },
564    // Constructors — `\cr`, `\noalign`, `\mathchoice`, etc. defined via
565    // `DefConstructor!` carry replacement / before_digest / after_digest
566    // closures that the dump format can't serialize. But the canonical
567    // CS lives in the compiled engine, so a `\let \tex_cr:D \cr`-style
568    // alias can round-trip as `PA\t<cs>` and replay via `state::let_i`.
569    // Self-aliases are no-ops at load (key already defined).
570    Stored::Constructor(c) => {
571      let target_cs = c.cs.with_str(url_encode);
572      Some(format!("PA\t{}", target_cs))
573    },
574    Stored::Register(reg) => {
575      // Register: serialize as R\tCS\tTYPE\tVALUE[\tMATHGLYPH]
576      //
577      // Closure-backed registers can't be serialized directly — but if the
578      // dump entry is a `\let`-alias to a closure-Register primitive
579      // (e.g. `\let \tex_dimexpr:D \dimexpr`), we CAN capture it via PA.
580      // Mirrors the `Stored::Primitive` / closure-`Expandable` arms above.
581      // Without this every `\let \tex_dimexpr:D \dimexpr`-style alias is
582      // silently dropped from the dump — the cause of `\tex_dimexpr:D`,
583      // `\tex_cr:D`, `\tex_dp:D`, `\tex_ht:D`, `\tex_wd:D` (and ~430
584      // engine-specific others) being missing in `latex.dump.txt` even
585      // though expl3-code.tex L280-700 aliases them. The dump_reader's
586      // add-only policy at load-time skips entries whose key is already
587      // defined (the canonical CS lives in the compiled engine), so
588      // self-aliases are no-ops. Real let-aliases (key != cs) replay via
589      // `state::let_i` — so the engine-only register's behavior follows.
590      if reg.getter.is_some() || reg.setter.is_some() {
591        let target_cs = reg.cs.with_str(url_encode);
592        return Some(format!("PA\t{}", target_cs));
593      }
594      let cs_name = reg.cs.with_str(url_encode);
595      let rtype = match reg.register_type {
596        crate::definition::register::RegisterType::Number => "N",
597        crate::definition::register::RegisterType::Dimension => "D",
598        crate::definition::register::RegisterType::Glue => "G",
599        crate::definition::register::RegisterType::MuGlue => "MG",
600        crate::definition::register::RegisterType::Tokens => "TK",
601        crate::definition::register::RegisterType::CharDef => "CD",
602        _ => return Option::None,
603      };
604      let value_str = match &reg.value {
605        Some(crate::definition::register::RegisterValue::Number(n)) => n.value_of().to_string(),
606        Some(crate::definition::register::RegisterValue::Dimension(d)) => d.value_of().to_string(),
607        Some(crate::definition::register::RegisterValue::Glue(g)) => {
608          let mut s = g.skip.to_string();
609          if let Some(p) = g.plus {
610            s.push_str(&format!(",p{}", p));
611          }
612          if let Some(ref pf) = g.pfill {
613            s.push_str(&format!(",pf{}", fillcode_index(pf)));
614          }
615          if let Some(m) = g.minus {
616            s.push_str(&format!(",m{}", m));
617          }
618          if let Some(ref mf) = g.mfill {
619            s.push_str(&format!(",mf{}", fillcode_index(mf)));
620          }
621          s
622        },
623        Some(crate::definition::register::RegisterValue::Tokens(tks)) => {
624          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
625          tok_strs.join(",")
626        },
627        _ => "0".to_string(),
628      };
629      // Serialize address if different from cs name. Allocated registers
630      // (`\newcount\m@ne`/`\newdimen\p@`/etc.) point at low-level `\count<n>`/
631      // `\dimen<n>` slots whose runtime values live there, NOT at the alias
632      // CS name. Without the address, dump_reader would `assign_internal` at
633      // the CS name's slot (default value), and `\the\m@ne` would yield 0
634      // instead of -1. Mirror Perl's `R(C(...),undef,...,address=>'\\count22')`
635      // serialization (LaTeXML/Core/Dumper.pm). Format extension:
636      // `R\t<cs>\t<rtype>\t<value>\t<mathglyph_or_empty>\t<address>`.
637      // If address == cs_name we still serialize an empty 6th field for
638      // forward-compat parsers; the reader accepts both 4/5/6-field forms.
639      let mut s = format!("R\t{}\t{}\t{}", cs_name, rtype, value_str);
640      let mathglyph_str = match reg.mathglyph {
641        Some(glyph) => format!("{}", glyph as u32),
642        None => String::new(),
643      };
644      let cs_decoded = arena::with(reg.cs.get_sym(), |s| s.to_string());
645      let address_field = if reg.address.is_empty() || reg.address == cs_decoded {
646        String::new()
647      } else {
648        url_encode(&reg.address)
649      };
650      // Only emit extended fields when needed (keep current format size for
651      // the common case)
652      if !mathglyph_str.is_empty() || !address_field.is_empty() {
653        s.push_str(&format!("\t{}", mathglyph_str));
654      }
655      if !address_field.is_empty() {
656        s.push_str(&format!("\t{}", address_field));
657      }
658      Some(s)
659    },
660    // "Nm" (Number) — distinct from "I" (raw Int) so the reader knows
661    // to install Stored::Number, matching the type expected by register
662    // slots. Round-tripping Number through "I" silently downgraded
663    // values to Stored::Int and broke `\count` register reads.
664    Stored::Number(n) => Some(format!("Nm\t{}", n.value_of())),
665    Stored::Float(f) => Some(format!("F\t{}", f.0)),
666    Stored::Dimension(d) => Some(format!("D\t{}", d.0)),
667    Stored::Glue(g) => {
668      let mut s = format!("G\t{}", g.skip);
669      if let Some(p) = g.plus {
670        s.push_str(&format!(",p{}", p));
671      }
672      if let Some(ref pf) = g.pfill {
673        s.push_str(&format!(",pf{}", fillcode_index(pf)));
674      }
675      if let Some(m) = g.minus {
676        s.push_str(&format!(",m{}", m));
677      }
678      if let Some(ref mf) = g.mfill {
679        s.push_str(&format!(",mf{}", fillcode_index(mf)));
680      }
681      Some(s)
682    },
683    Stored::MuDimension(d) => Some(format!("MD\t{}", d.0)),
684    Stored::MuGlue(g) => {
685      let mut s = format!("MG\t{}", g.skip);
686      if let Some(p) = g.plus {
687        s.push_str(&format!(",p{}", p));
688      }
689      if let Some(ref pf) = g.pfill {
690        s.push_str(&format!(",pf{}", fillcode_index(pf)));
691      }
692      if let Some(m) = g.minus {
693        s.push_str(&format!(",m{}", m));
694      }
695      if let Some(ref mf) = g.mfill {
696        s.push_str(&format!(",mf{}", fillcode_index(mf)));
697      }
698      Some(s)
699    },
700    Stored::VecDequeStored(vd) if vd.is_empty() => Some("VD\t".to_string()),
701    _ => Option::None,
702  }
703}

720/// Serialize a `Parameters` list in the v3 structured format.
721///
722/// Layout (ASCII control-byte delimited so nothing collides with
723/// url-encoded printable content):
724/// - `\x1e` between Parameters (RS, Record Separator)
725/// - `\x1f` between fields of a single Parameter (US, Unit Separator)
726/// - `\x1d` between Tokens lists inside `extras` (GS, Group Separator)
727///
728/// Per-parameter fields: `<name>\x1f<spec>\x1f<flags>\x1f<extras>`
729///   - `<name>` — `Parameter.name` (Perl's `type`), url-encoded.
730///   - `<spec>` — `Parameter.spec`, url-encoded.
731///   - `<flags>` — semicolon-separated `n=1` (novalue) / `o=1` (optional).
732///   - `<extras>` — `<Tokens1>\x1d<Tokens2>…` where each `<TokensN>` is the same comma-joined
733///     `<catcode>:<text>` encoding used for E-entry bodies. Empty extras → empty field.
734///
735/// The reader (`dump_reader::parse_parameters_v3`) hands the decoded
736/// `(name, spec, Some(extras))` triple directly to `Parameter::new`,
737/// bypassing `parse_parameters` entirely — so `Until:\end{verbatim}`
738/// and `Match:…` round-trip with their catcoded delimiter tokens intact.
739pub(crate) fn serialize_parameters_v3(params: &crate::parameter::Parameters) -> String {
740  let mut out = String::new();
741  for (i, p) in params.get_parameters().iter().enumerate() {
742    if i > 0 {
743      out.push('\x1e');
744    }
745    let name = arena::with(p.name, url_encode);
746    let spec = arena::with(p.spec, url_encode);
747    let mut flags = String::new();
748    if p.novalue {
749      flags.push_str("n=1");
750    }
751    if p.optional {
752      if !flags.is_empty() {
753        flags.push(';');
754      }
755      flags.push_str("o=1");
756    }
757    let extras = p
758      .extra
759      .iter()
760      .map(|tks| {
761        tks
762          .clone()
763          .unlist()
764          .iter()
765          .map(serialize_token)
766          .collect::<Vec<_>>()
767          .join(",")
768      })
769      .collect::<Vec<_>>()
770      .join("\x1d");
771    out.push_str(&name);
772    out.push('\x1f');
773    out.push_str(&spec);
774    out.push('\x1f');
775    out.push_str(&flags);
776    out.push('\x1f');
777    out.push_str(&extras);
778  }
779  out
780}
781
782fn url_encode(s: &str) -> String {
783  let mut result = String::with_capacity(s.len());
784  for ch in s.chars() {
785    match ch {
786      '\t' => result.push_str("%09"),
787      '\n' => result.push_str("%0A"),
788      '\r' => result.push_str("%0D"),
789      '%' => result.push_str("%25"),
790      ',' => result.push_str("%2C"),
791      _ if ch.is_ascii_control() => {
792        result.push_str(&format!("%{:02X}", ch as u8));
793      },
794      _ => result.push(ch),
795    }
796  }
797  result
798}

800/// Recognize expl3's intarray-as-fontdimen storage keys, e.g.
801/// `fontdimen_fontinfo_cmr10 at 15sp_12737` → ("fontdimen_fontinfo_cmr10 at 15sp", 12737).
802/// Returns None for other Value keys (so the regular V-record path applies).
803fn parse_fontdimen_key(key: &str) -> Option<(&str, u32)> {
804  // Cheap gate first to keep the hot path fast on non-fontdimen keys.
805  if !key.starts_with("fontdimen_fontinfo_") {
806    return None;
807  }
808  // The last `_<digits>` tail is the slot index; everything before is the
809  // per-intarray prefix. Be defensive: an all-letter tail (e.g. a future
810  // non-indexed `fontdimen_fontinfo_*` key) should not match.
811  let last_us = key.rfind('_')?;
812  let (prefix, tail) = (&key[..last_us], &key[last_us + 1..]);
813  let index: u32 = tail.parse().ok()?;
814  Some((prefix, index))
815}

817/// Run-length encode a slice of i64 values as a comma-separated list:
818/// each run is either `<v>` (single) or `<v>x<n>` (count ≥ 2). Decoder
819/// in `dump_reader::rle_decode_i64` is the inverse.
820fn rle_encode_i64(values: &[i64]) -> String {
821  let mut out = String::new();
822  let mut i = 0;
823  while i < values.len() {
824    let v = values[i];
825    let mut count = 1usize;
826    while i + count < values.len() && values[i + count] == v {
827      count += 1;
828    }
829    if !out.is_empty() {
830      out.push(',');
831    }
832    if count == 1 {
833      out.push_str(&v.to_string());
834    } else {
835      out.push_str(&format!("{}x{}", v, count));
836    }
837    i += count;
838  }
839  out
840}
