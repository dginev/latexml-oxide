1//! Writer for Rust-native kernel dump files.
2//!
3//! Serializes the State diff (from snapshot/diff_from_snapshot) into a
4//! compact text format that can be loaded back efficiently.
5//!
6//! This is the "DumpFile" equivalent from Perl's TeX_Job.pool.ltxml.
7//! The workflow follows Perl's Makefile.PL "make formats" scaffold:
8//!   1. take_snapshot() — before processing latex.ltx
9//!   2. Process latex.ltx through the engine
10//!   3. diff_snapshot() — find what changed
11//!   4. write_dump() — serialize the diff
12//!   5. At runtime: load_dump() restores the state
13

15use std::path::Path;

23/// Write a state diff to a dump file.
24/// Returns the number of entries successfully written.
25///
26/// Writes in three ordered sections, matching Perl's
27/// `TeX_Job.pool.ltxml::DumpFile` which separates
28/// `@cmds_early` / `@cmds` / `@cmds_late`:
29///
30///   1. **cmds_early** — M:PA / M:MPA let-aliases whose **target pre-existed** in the bootstrap
31///      snapshot (e.g. `\tex_let:D → \let`, `\tex_def:D → \def`). Applied first because their
32///      targets are always available (they're in the bootstrap pool we loaded unconditionally
33///      before this dump).
34///   2. **cmds** (regular) — V / M:E / M:T / R / C / LC / UC / SC / MC / DC entries: data,
35///      expandable definitions, registers, codes.
36///   3. **cmds_late** — M:PA / M:MPA let-aliases whose **target is defined by this dump**. Applied
37///      last so the target is installed before the alias fires.
38///
39/// The early/late split requires the caller (ini_tex) to have staged a
40/// snapshot via `state::stage_snapshot("bootstrap")`. If no snapshot is
41/// available, all aliases are emitted as `late` (safe — they just wait
42/// until after regular entries).
43pub fn write_dump(
44  path: &Path,
45  entries: &[(TableName, crate::common::arena::SymStr, Stored)],
46) -> Result<usize, String> {
47  let mut file =
48    std::fs::File::create(path).map_err(|e| format!("Failed to create dump file: {}", e))?;
49
50  writeln!(file, "# latexml-oxide kernel dump v3").ok();
51  writeln!(
52    file,
53    "# Generated by dump_writer, {} entries",
54    entries.len()
55  )
56  .ok();
57  writeln!(file, "# Format: <table>\\t<key>\\t<data>").ok();
58  writeln!(file, "#   <table> = M | V | C | LC | UC | SC | MC | DC").ok();
59  writeln!(file, "#   <data> depends on <table>:").ok();
60  writeln!(
61    file,
62    "#     M/E:   E\\t<cs>\\t<nargs>\\t<flags>\\t<body>\\t<proto>\\t<v3_params>"
63  )
64  .ok();
65  writeln!(
66    file,
67    "#            v3_params encodes each Parameter structurally so Until:/Match:"
68  )
69  .ok();
70  writeln!(
71    file,
72    "#            delimiter tokens round-trip (see docs/DUMP_FORMAT_PERL_ANALYSIS.md)."
73  )
74  .ok();
75  writeln!(
76    file,
77    "#     M/PA:  PA\\t<target_cs>   (MPA for math primitives)"
78  )
79  .ok();
80  writeln!(file, "#     M/T:   T\\t<meaning_token>").ok();
81  writeln!(
82    file,
83    "#     M/R:   R\\t<cs>\\t<rtype>\\t<value>[\\t<mathglyph>]"
84  )
85  .ok();
86  writeln!(
87    file,
88    "#     M/N:   N                     (None-meaning sentinel)"
89  )
90  .ok();
91  writeln!(
92    file,
93    "#     V:     <type>\\t<value>       (B,I,S,CH,T,TK,N,F,G,MG,D,MD,VD)"
94  )
95  .ok();
96  writeln!(file, "#     C:                  CC\\t<u8>   (catcode)").ok();
97  writeln!(
98    file,
99    "#     DC/LC/UC/SC/MC:     CH\\t<u32>  (char-indexed table entries)"
100  )
101  .ok();
102  writeln!(file, "#   <flags> = subset of {{L, P}} (long, protected)").ok();
103  writeln!(
104    file,
105    "#   All keys and proto are url-encoded (%09 for tab etc.); token body"
106  )
107  .ok();
108  writeln!(
109    file,
110    "#   tokens use the form <catcode_int>:<url_encoded_text> joined by commas."
111  )
112  .ok();
113
114  // Read the bootstrap snapshot so we can classify each alias's target
115  // as pre-existing (→ early) or defined-during-load (→ late).
116  let bootstrap_snap = crate::state::get_staged_snapshot("bootstrap");
117
118  // Pre-serialize + partition. Skipped keys (ver@..., unserializable
119  // values) fall out naturally.
120  let mut early_aliases: Vec<(String, String, String)> = Vec::new();
121  let mut regular: Vec<(String, String, String)> = Vec::new();
122  let mut late_aliases: Vec<(String, String, String)> = Vec::new();
123  let mut skipped = 0usize;
124
125  for (table, key, value) in entries {
126    let table_code = table_to_code(*table);
127    let key_str = arena::with(*key, |s| s.to_string());
128
129    // Skip \ver@ package version markers (runtime metadata, not definitions)
130    if key_str.starts_with("\\ver@") {
131      skipped += 1;
132      continue;
133    }
134
135    // Skip \@currname / \@currext file-IO bookkeeping. These are
136    // assigned per-document during `\input` by `read_input_file_recursive`
137    // (see `binding/content.rs:262-263`) to the literal filename's
138    // tokens, so the snapshot captures the LAST opened file's name
139    // ("plain.tex" / "latex.ltx"). Perl's `TeX_FileIO.pool.ltxml:28-29`
140    // initializes them via `Let('\@currname','\lx@empty')` before any
141    // file load, and Perl's plain_dump.pool.ltxml omits them — so
142    // post-dump they remain at the `\lx@empty` baseline. Matching that
143    // behavior here keeps us file-IO-state-agnostic in the dump.
144    if matches!(*table, TableName::Meaning) && (key_str == "\\@currname" || key_str == "\\@currext")
145    {
146      skipped += 1;
147      continue;
148    }
149
150    // Skip token-list register VALUES whose body contains a self-`\the`
151    // reference. latex.ltx L10564-10580 sets `\frozen@everymath` =
152    // `{... \the\everymath}` then allocates a NEW `\everymath` via
153    // `\newtoks`. Our `\newtoks` doesn't fully detach the new register's
154    // slot from the frozen alias's, so the body lands on `\everymath`'s
155    // value slot — `\the\everymath` then expands recursively at math
156    // entry, blowing the token limit. Until the `\newtoks`/`\let`
157    // slot-aliasing is fixed at the dialect level, the safest cure is
158    // to drop the corrupted self-referential capture; runtime
159    // re-evaluation of latex.ltx's hook chain (or LaTeXML's own
160    // `\everymath`/`\everydisplay` setup) repopulates correctly.
161    if matches!(*table, TableName::Value)
162      && matches!(
163        key_str.as_str(),
164        "\\everymath"
165          | "\\everydisplay"
166          | "\\everyhbox"
167          | "\\everyvbox"
168          | "\\everycr"
169          | "\\everyjob"
170          | "\\everypar"
171          | "\\everyeof"
172      )
173    {
174      // Confirm the loop pattern before dropping (don't suppress
175      // valid hook contributions on engines with sound aliasing).
176      if let Stored::Tokens(ref tks) = value {
177        let body = tks.unlist_ref();
178        let needle_cs = key_str.to_string();
179        let has_self_the = body.iter().any(|t| t.with_str(|s| s == "\\the"))
180          && body.iter().any(|t| t.with_str(|s| s == needle_cs));
181        if has_self_the {
182          skipped += 1;
183          continue;
184        }
185      }
186    }
187
188    // Perl #2771 (2026-03-13) + upstream IGNORED_SYMBOLS: these are
189    // runtime-only Value entries — control-flow counters, and large
190    // registered-rule tables that can't meaningfully round-trip through
191    // the dump (they hold closures, cross-refs, or re-populate during
192    // engine init anyway). Match Perl's TeX_Job.pool.ltxml list.
193    if matches!(*table, TableName::Value)
194      && matches!(
195        key_str.as_str(),
196        "if_count"
197          | "absorb_count"
198          | "if_stack"
199          | "DOCUMENT_REWRITE_RULES"
200          | "PARAMETER_TYPES"
201          | "TAG_PROPERTIES"
202          | "MATH_LIGATURES"
203          | "TEXT_LIGATURES"
204      )
205    {
206      skipped += 1;
207      continue;
208    }
209
210    // Perl IGNORED_SYMBOLS: meaning:\lnot, meaning:\to are skipped because
211    // pre-2017 TeXlive let-aliased \lnot → \neg and \to → \rightarrow,
212    // which would gratuitously diverge tests across TL versions.
213    // Mirror Perl `TeX_Job.pool.ltxml` IGNORED_SYMBOLS L104-107.
214    if matches!(*table, TableName::Meaning) && matches!(key_str.as_str(), "\\lnot" | "\\to") {
215      skipped += 1;
216      continue;
217    }
218
219    let Some(serialized) = serialize_stored(value) else {
220      skipped += 1;
221      continue;
222    };
223
224    // A let-alias entry is an M row whose serialized form starts with
225    // `PA\t` or `MPA\t`. Nothing else produces those prefixes.
226    let is_alias = matches!(*table, TableName::Meaning)
227      && (serialized.starts_with("PA\t") || serialized.starts_with("MPA\t"));
228
229    let row = (table_code.to_string(), url_encode(&key_str), serialized);
230    if is_alias {
231      // Extract the target CS from "PA\t<target>" / "MPA\t<target>"
232      // (we already filtered self-aliases in serialize_stored, so the
233      //  target is always a different CS than the key).
234      let target_raw = row.2.split('\t').nth(1).unwrap_or("");
235      let target = crate::dump_reader::url_decode(target_raw);
236      let target_sym = arena::pin(&target);
237      // Target pre-existed iff it was in the bootstrap meaning table.
238      let pre_existed = bootstrap_snap
239        .as_ref()
240        .is_some_and(|snap| snap.contains_key(&(TableName::Meaning, target_sym)));
241      if pre_existed {
242        early_aliases.push(row);
243      } else {
244        late_aliases.push(row);
245      }
246    } else {
247      regular.push(row);
248    }
249  }
250
251  writeln!(
252    file,
253    "# Section 0: early let-aliases ({}) — target in bootstrap",
254    early_aliases.len()
255  )
256  .ok();
257  for (t, k, v) in &early_aliases {
258    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
259  }
260  writeln!(file, "# Section 1: regular entries ({})", regular.len()).ok();
261  for (t, k, v) in &regular {
262    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
263  }
264  writeln!(
265    file,
266    "# Section 2: late let-aliases ({}) — target in Section 1",
267    late_aliases.len()
268  )
269  .ok();
270  for (t, k, v) in &late_aliases {
271    writeln!(file, "{}\t{}\t{}", t, k, v).map_err(|e| format!("Write error: {}", e))?;
272  }
273
274  let count = early_aliases.len() + regular.len() + late_aliases.len();
275  writeln!(file, "# Written: {}, Skipped: {}", count, skipped).ok();
276  eprintln!(
277    "[dump_writer] Wrote {} entries to {} ({} skipped; {} early + {} regular + {} late)",
278    count,
279    path.display(),
280    skipped,
281    early_aliases.len(),
282    regular.len(),
283    late_aliases.len(),
284  );
285
286  Ok(count)
287}

289fn table_to_code(t: TableName) -> &'static str {
290  match t {
291    TableName::Meaning => "M",
292    TableName::Value => "V",
293    TableName::Catcode => "C",
294    TableName::Mathcode => "MC",
295    TableName::Sfcode => "SC",
296    TableName::Lccode => "LC",
297    TableName::Uccode => "UC",
298    TableName::Delcode => "DC",
299    TableName::Stash => "ST",
300    TableName::StashActive => "SA",
301  }
302}

304/// Serialize a Stored value to a type-tag + data string.
305/// Returns None if the value can't be serialized.
306fn serialize_stored(stored: &Stored) -> Option<String> {
307  match stored {
308    Stored::None => Some("N".to_string()),
309    Stored::Bool(b) => Some(format!("B\t{}", if *b { "1" } else { "0" })),
310    Stored::Int(i) => Some(format!("I\t{}", i)),
311    Stored::String(s) => Some(format!(
312      "S\t{}",
313      url_encode(&arena::with(*s, |s| s.to_string()))
314    )),
315    Stored::Charcode(c) => Some(format!("CH\t{}", c)),
316    Stored::Catcode(cc) => Some(format!("CC\t{}", u8::from(*cc))),
317    Stored::Token(t) => Some(format!("T\t{}", serialize_token(t))),
318    Stored::Tokens(tks) => {
319      let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
320      Some(format!("TK\t{}", tok_strs.join(",")))
321    },
322    Stored::Expandable(exp) => {
323      use crate::common::object::Object;
324      let cs_name = exp.get_cs().with_str(url_encode);
325      let nargs = exp.get_num_args();
326      let mut flags = String::new();
327      if exp.is_long {
328        flags.push('L');
329      }
330      if exp.is_protected() {
331        flags.push('P');
332      }
333      // Two parallel parameter encodings:
334      //
335      //   - `proto` (v2, 5th field) — `Parameters::stringify()`, a flat space-separated prototype
336      //     string. Round-trips simple specs but not `Until:\end{verbatim}` or other
337      //     brace-in-delimiter forms.
338      //   - `v3_params` (v3, 6th field) — per-Parameter structured record carrying (name, spec,
339      //     flags, extras). Modeled on Perl's `P(type, spec, extra=>[T(...)])`; see
340      //     `docs/DUMP_FORMAT_PERL_ANALYSIS.md`. Bypasses `parse_parameters` at load time so
341      //     delimited params round-trip intact.
342      //
343      // We emit both so older readers still work, and newer readers can
344      // prefer v3 transparently.
345      let proto = exp
346        .get_parameters()
347        .map(|p| p.stringify())
348        .unwrap_or_default();
349      let proto_encoded = url_encode(&proto);
350      let v3_params = exp
351        .get_parameters()
352        .map(serialize_parameters_v3)
353        .unwrap_or_default();
354      match exp.get_expansion() {
355        Some(crate::definition::ExpansionBody::Tokens(tks)) => {
356          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
357          Some(format!(
358            "E\t{}\t{}\t{}\t{}\t{}\t{}",
359            cs_name,
360            nargs,
361            flags,
362            tok_strs.join(","),
363            proto_encoded,
364            v3_params
365          ))
366        },
367        // No body at all (e.g. `DefMacro!("\\@gobble{}", None)`): serializes
368        // as an E-entry with empty token body. The reader reconstructs an
369        // Expandable with empty Tokens + the original paramlist, which at
370        // expansion time reads and discards its arguments — semantically
371        // identical to the no-body form's runtime behavior (expandable.rs
372        // `invoke` matches `None` → reads args, returns NO_TOKENS).
373        None => Some(format!(
374          "E\t{}\t{}\t{}\t\t{}\t{}",
375          cs_name, nargs, flags, proto_encoded, v3_params
376        )),
377        // Closure-based body — can't serialize the closure itself, BUT
378        // if the dump entry is a `\let`-alias to a closure-Expandable
379        // primitive (e.g. `\let \tex_expandafter:D \expandafter`), we
380        // CAN capture it via PA. Mirrors the `Stored::Primitive` /
381        // `Stored::Conditional` arms below. Without this, every
382        // `\let \X \expandafter`-style alias is silently dropped from
383        // the dump — the cause of `\tex_expandafter:D`,
384        // `\tex_unexpanded:D`, `\tex_the:D` being missing in
385        // `latex.dump.txt` even though expl3-code.tex L357+ aliases them.
386        //
387        // The dump_reader's add-only policy at load-time skips entries
388        // whose key is already defined (the canonical CS lives in the
389        // compiled engine), so self-aliases are no-ops. Real let-aliases
390        // (key != cs) replay via `state::let_i`.
391        Some(crate::definition::ExpansionBody::Closure(_)) => Some(format!("PA\t{}", cs_name)),
392      }
393    },
394    // Closure-based primitives can't be serialized directly — but if the entry's
395    // CS differs from the primitive's own CS, it's a `\let`-alias, and we CAN
396    // capture the alias so the dump reader can replay `\let <key> <target>` at
397    // load time. This is how `\tex_let:D`, `\tex_def:D`, and hundreds of other
398    // expl3-renamed primitives stay reachable across dump boundaries (the
399    // defining assignment `\let \tex_let:D \let` in expl3-code.tex needs to
400    // survive the dump for `\usepackage{expl3}`'s guard `\ifx\csname
401    // tex_let:D\endcsname\relax` to short-circuit).
402    //
403    // NOTE: the KEY we're saving IS the dump entry's first column; the VALUE
404    // we write is the primitive's OWN CS name. Dump reader compares these:
405    // if equal, it's the "primary" primitive (already in bindings) and skipped;
406    // if different, it's an alias — replay `\let` at load time.
407    Stored::Primitive(p) => {
408      // Perl `Core/Dumper.pm::dump_primitive` (L383-389): if a primitive has
409      // a `font` directive AND no replacement closure (i.e. defined by `\font`),
410      // emit `FD(<cs>, <fontID>)` instead of generic Primitive serialization.
411      // `font_id` is set on `\font`-defined primitives by `engine/tex_fonts.rs`'s
412      // post-define hook (the Rust path) — it carries the value-table key under
413      // which the Stored::Font lives. The reader installs a synthesized
414      // Primitive whose before_digest looks up that font and merges, mirroring
415      // `LaTeXML::Core::Definition::FontDef::invoke` (FontDef.pm L38-45).
416      if let Some(fid) = p.font_id {
417        let fid_str = arena::with(fid, |s| s.to_string());
418        return Some(format!("FD\t{}", url_encode(&fid_str)));
419      }
420      let target_cs = p.cs.with_str(url_encode);
421      Some(format!("PA\t{}", target_cs))
422    },
423    Stored::Font(f) => {
424      // Perl `dump_font` (Core/Dumper.pm L281-284) emits `F(... components ...)`.
425      // Rust's `Font` is a flat options struct, so we serialize the non-None
426      // fields as `key=value` pairs separated by `\x1f` (unit-separator), with
427      // values url-encoded for tab/newline-safety. Reader inverts in dump_reader.rs.
428      let mut parts = Vec::with_capacity(8);
429      if let Some(ref name) = f.name {
430        parts.push(format!("name={}", url_encode(name)));
431      }
432      if let Some(size) = f.size {
433        parts.push(format!("size={}", size));
434      }
435      if let Some(ref family) = f.family {
436        parts.push(format!("family={}", url_encode(family)));
437      }
438      if let Some(ref series) = f.series {
439        parts.push(format!("series={}", url_encode(series)));
440      }
441      if let Some(ref shape) = f.shape {
442        parts.push(format!("shape={}", url_encode(shape)));
443      }
444      if let Some(ref encoding) = f.encoding {
445        parts.push(format!("encoding={}", url_encode(encoding)));
446      }
447      if let Some(ref language) = f.language {
448        parts.push(format!("language={}", url_encode(language)));
449      }
450      if let Some(ref mathstyle) = f.mathstyle {
451        parts.push(format!("mathstyle={}", url_encode(mathstyle)));
452      }
453      if let Some(ref opacity) = f.opacity {
454        parts.push(format!("opacity={}", url_encode(opacity)));
455      }
456      if let Some(scale) = f.scale {
457        parts.push(format!("scale={}", scale));
458      }
459      if let Some(b) = f.emph {
460        parts.push(format!("emph={}", if b { 1 } else { 0 }));
461      }
462      if let Some(b) = f.scripted {
463        parts.push(format!("scripted={}", if b { 1 } else { 0 }));
464      }
465      if let Some(step) = f.mathstylestep {
466        parts.push(format!("mathstylestep={}", step));
467      }
468      if let Some(flags) = f.flags {
469        parts.push(format!("flags={}", flags));
470      }
471      Some(format!("F\t{}", parts.join("\x1f")))
472    },
473    Stored::MathPrimitive(p) => {
474      let target_cs = p.cs.with_str(url_encode);
475      Some(format!("MPA\t{}", target_cs))
476    },
477    // Conditionals — same logic as Primitives. \ifx, \ifnum, \if etc. are
478    // stored as Stored::Conditional with their canonical CS field. When
479    // expl3-code.tex does `\let \if_meaning:w \ifx`, the new entry shares
480    // the same Conditional Rc, with cs = \ifx. Capturing this as a PA
481    // alias lets the dump reader replay `\let \if_meaning:w \ifx` at load
482    // time. Without this, every expl3 conditional alias is silently lost
483    // from the dump and `\if_meaning:w`-style invocations fail at runtime.
484    Stored::Conditional(c) => {
485      let target_cs = c.cs.with_str(url_encode);
486      Some(format!("PA\t{}", target_cs))
487    },
488    // Constructors — `\cr`, `\noalign`, `\mathchoice`, etc. defined via
489    // `DefConstructor!` carry replacement / before_digest / after_digest
490    // closures that the dump format can't serialize. But the canonical
491    // CS lives in the compiled engine, so a `\let \tex_cr:D \cr`-style
492    // alias can round-trip as `PA\t<cs>` and replay via `state::let_i`.
493    // Self-aliases are no-ops at load (key already defined).
494    Stored::Constructor(c) => {
495      let target_cs = c.cs.with_str(url_encode);
496      Some(format!("PA\t{}", target_cs))
497    },
498    Stored::Register(reg) => {
499      // Register: serialize as R\tCS\tTYPE\tVALUE[\tMATHGLYPH]
500      //
501      // Closure-backed registers can't be serialized directly — but if the
502      // dump entry is a `\let`-alias to a closure-Register primitive
503      // (e.g. `\let \tex_dimexpr:D \dimexpr`), we CAN capture it via PA.
504      // Mirrors the `Stored::Primitive` / closure-`Expandable` arms above.
505      // Without this every `\let \tex_dimexpr:D \dimexpr`-style alias is
506      // silently dropped from the dump — the cause of `\tex_dimexpr:D`,
507      // `\tex_cr:D`, `\tex_dp:D`, `\tex_ht:D`, `\tex_wd:D` (and ~430
508      // engine-specific others) being missing in `latex.dump.txt` even
509      // though expl3-code.tex L280-700 aliases them. The dump_reader's
510      // add-only policy at load-time skips entries whose key is already
511      // defined (the canonical CS lives in the compiled engine), so
512      // self-aliases are no-ops. Real let-aliases (key != cs) replay via
513      // `state::let_i` — so the engine-only register's behavior follows.
514      if reg.getter.is_some() || reg.setter.is_some() {
515        let target_cs = reg.cs.with_str(url_encode);
516        return Some(format!("PA\t{}", target_cs));
517      }
518      let cs_name = reg.cs.with_str(url_encode);
519      let rtype = match reg.register_type {
520        crate::definition::register::RegisterType::Number => "N",
521        crate::definition::register::RegisterType::Dimension => "D",
522        crate::definition::register::RegisterType::Glue => "G",
523        crate::definition::register::RegisterType::MuGlue => "MG",
524        crate::definition::register::RegisterType::Tokens => "TK",
525        crate::definition::register::RegisterType::CharDef => "CD",
526        _ => return Option::None,
527      };
528      let value_str = match &reg.value {
529        Some(crate::definition::register::RegisterValue::Number(n)) => n.value_of().to_string(),
530        Some(crate::definition::register::RegisterValue::Dimension(d)) => d.value_of().to_string(),
531        Some(crate::definition::register::RegisterValue::Glue(g)) => {
532          let mut s = g.skip.to_string();
533          if let Some(p) = g.plus {
534            s.push_str(&format!(",p{}", p));
535          }
536          if let Some(ref pf) = g.pfill {
537            s.push_str(&format!(",pf{}", fillcode_index(pf)));
538          }
539          if let Some(m) = g.minus {
540            s.push_str(&format!(",m{}", m));
541          }
542          if let Some(ref mf) = g.mfill {
543            s.push_str(&format!(",mf{}", fillcode_index(mf)));
544          }
545          s
546        },
547        Some(crate::definition::register::RegisterValue::Tokens(tks)) => {
548          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
549          tok_strs.join(",")
550        },
551        _ => "0".to_string(),
552      };
553      // Serialize address if different from cs name. Allocated registers
554      // (`\newcount\m@ne`/`\newdimen\p@`/etc.) point at low-level `\count<n>`/
555      // `\dimen<n>` slots whose runtime values live there, NOT at the alias
556      // CS name. Without the address, dump_reader would `assign_internal` at
557      // the CS name's slot (default value), and `\the\m@ne` would yield 0
558      // instead of -1. Mirror Perl's `R(C(...),undef,...,address=>'\\count22')`
559      // serialization (LaTeXML/Core/Dumper.pm). Format extension:
560      // `R\t<cs>\t<rtype>\t<value>\t<mathglyph_or_empty>\t<address>`.
561      // If address == cs_name we still serialize an empty 6th field for
562      // forward-compat parsers; the reader accepts both 4/5/6-field forms.
563      let mut s = format!("R\t{}\t{}\t{}", cs_name, rtype, value_str);
564      let mathglyph_str = match reg.mathglyph {
565        Some(glyph) => format!("{}", glyph as u32),
566        None => String::new(),
567      };
568      let cs_decoded = arena::with(reg.cs.get_sym(), |s| s.to_string());
569      let address_field = if reg.address.is_empty() || reg.address == cs_decoded {
570        String::new()
571      } else {
572        url_encode(&reg.address)
573      };
574      // Only emit extended fields when needed (keep current format size for
575      // the common case)
576      if !mathglyph_str.is_empty() || !address_field.is_empty() {
577        s.push_str(&format!("\t{}", mathglyph_str));
578      }
579      if !address_field.is_empty() {
580        s.push_str(&format!("\t{}", address_field));
581      }
582      Some(s)
583    },
584    // "Nm" (Number) — distinct from "I" (raw Int) so the reader knows
585    // to install Stored::Number, matching the type expected by register
586    // slots. Round-tripping Number through "I" silently downgraded
587    // values to Stored::Int and broke `\count` register reads.
588    Stored::Number(n) => Some(format!("Nm\t{}", n.value_of())),
589    Stored::Float(f) => Some(format!("F\t{}", f.0)),
590    Stored::Dimension(d) => Some(format!("D\t{}", d.0)),
591    Stored::Glue(g) => {
592      let mut s = format!("G\t{}", g.skip);
593      if let Some(p) = g.plus {
594        s.push_str(&format!(",p{}", p));
595      }
596      if let Some(ref pf) = g.pfill {
597        s.push_str(&format!(",pf{}", fillcode_index(pf)));
598      }
599      if let Some(m) = g.minus {
600        s.push_str(&format!(",m{}", m));
601      }
602      if let Some(ref mf) = g.mfill {
603        s.push_str(&format!(",mf{}", fillcode_index(mf)));
604      }
605      Some(s)
606    },
607    Stored::MuDimension(d) => Some(format!("MD\t{}", d.0)),
608    Stored::MuGlue(g) => {
609      let mut s = format!("MG\t{}", g.skip);
610      if let Some(p) = g.plus {
611        s.push_str(&format!(",p{}", p));
612      }
613      if let Some(ref pf) = g.pfill {
614        s.push_str(&format!(",pf{}", fillcode_index(pf)));
615      }
616      if let Some(m) = g.minus {
617        s.push_str(&format!(",m{}", m));
618      }
619      if let Some(ref mf) = g.mfill {
620        s.push_str(&format!(",mf{}", fillcode_index(mf)));
621      }
622      Some(s)
623    },
624    Stored::VecDequeStored(vd) if vd.is_empty() => Some("VD\t".to_string()),
625    _ => Option::None,
626  }
627}

629fn fillcode_index(fc: &crate::common::glue::FillCode) -> usize {
630  use crate::common::glue::FillCode::*;
631  match fc {
632    Fil => 1,
633    Fill => 2,
634    Filll => 3,
635  }

638fn serialize_token(t: &crate::token::Token) -> String {
639  let cc: u8 = t.get_catcode().into();
640  let text = t.with_str(url_encode);
641  format!("{}:{}", cc, text)
642}
643
644/// Serialize a `Parameters` list in the v3 structured format.
645///
646/// Layout (ASCII control-byte delimited so nothing collides with
647/// url-encoded printable content):
648/// - `\x1e` between Parameters (RS, Record Separator)
649/// - `\x1f` between fields of a single Parameter (US, Unit Separator)
650/// - `\x1d` between Tokens lists inside `extras` (GS, Group Separator)
651///
652/// Per-parameter fields: `<name>\x1f<spec>\x1f<flags>\x1f<extras>`
653///   - `<name>` — `Parameter.name` (Perl's `type`), url-encoded.
654///   - `<spec>` — `Parameter.spec`, url-encoded.
655///   - `<flags>` — semicolon-separated `n=1` (novalue) / `o=1` (optional).
656///   - `<extras>` — `<Tokens1>\x1d<Tokens2>…` where each `<TokensN>` is the same comma-joined
657///     `<catcode>:<text>` encoding used for E-entry bodies. Empty extras → empty field.
658///
659/// The reader (`dump_reader::parse_parameters_v3`) hands the decoded
660/// `(name, spec, Some(extras))` triple directly to `Parameter::new`,
661/// bypassing `parse_parameters` entirely — so `Until:\end{verbatim}`
662/// and `Match:…` round-trip with their catcoded delimiter tokens intact.
663pub(crate) fn serialize_parameters_v3(params: &crate::parameter::Parameters) -> String {
664  let mut out = String::new();
665  for (i, p) in params.get_parameters().iter().enumerate() {
666    if i > 0 {
667      out.push('\x1e');
668    }
669    let name = arena::with(p.name, url_encode);
670    let spec = arena::with(p.spec, url_encode);
671    let mut flags = String::new();
672    if p.novalue {
673      flags.push_str("n=1");
674    }
675    if p.optional {
676      if !flags.is_empty() {
677        flags.push(';');
678      }
679      flags.push_str("o=1");
680    }
681    let extras = p
682      .extra
683      .iter()
684      .map(|tks| {
685        tks
686          .clone()
687          .unlist()
688          .iter()
689          .map(serialize_token)
690          .collect::<Vec<_>>()
691          .join(",")
692      })
693      .collect::<Vec<_>>()
694      .join("\x1d");
695    out.push_str(&name);
696    out.push('\x1f');
697    out.push_str(&spec);
698    out.push('\x1f');
699    out.push_str(&flags);
700    out.push('\x1f');
701    out.push_str(&extras);
702  }
703  out
704}
705
706fn url_encode(s: &str) -> String {
707  let mut result = String::with_capacity(s.len());
708  for ch in s.chars() {
709    match ch {
710      '\t' => result.push_str("%09"),
711      '\n' => result.push_str("%0A"),
712      '\r' => result.push_str("%0D"),
713      '%' => result.push_str("%25"),
714      ',' => result.push_str("%2C"),
715      _ if ch.is_ascii_control() => {
716        result.push_str(&format!("%{:02X}", ch as u8));
717      },
718      _ => result.push(ch),
719    }
720  }
721  result
722}

731  /// Construct a Parameter without calling `init()` — `init()` needs live

743  /// A Parameters list with two `Plain {}` args encodes as two records
744  /// separated by RS, each `Plain<US>{}<US><US>`. Verifies baseline layout.
745  #[test]
746  fn v3_encoding_plain() {
747    let ps = Parameters::new(vec![
748      raw_param("Plain", "{}", vec![]),
749      raw_param("Plain", "{}", vec![]),
750    ]);
751    let s = serialize_parameters_v3(&ps);

753  }

755  /// `Until:<delim>` with a brace-containing delimiter. The test covers
756  /// the case that livelocked 00_tokenize under v2: the delimiter tokens
757  /// must serialize as catcoded tokens (not as the raw spec string), so
758  /// `{`/`}` survive as CC_BEGIN / CC_END rather than being re-parsed as
759  /// spec-level braces.
760  #[test]
761  fn v3_encoding_until_with_braces() {
762    let delim = Tokens::new(vec![
763      Token {
764        text: arena::pin("\\end"),
765        code: Catcode::CS,
766      },
767      Token {
768        text: arena::pin("{"),
769        code: Catcode::BEGIN,
770      },
771      Token {
772        text: arena::pin("verbatim"),
773        code: Catcode::LETTER,
774      },
775      Token {
776        text: arena::pin("}"),
777        code: Catcode::END,
778      },
779    ]);
780    let ps = Parameters::new(vec![raw_param("Until", "Until:\\end{verbatim}", vec![
781      delim,
782    ])]);
783    let s = serialize_parameters_v3(&ps);
784    // Expected: Until<US>Until:\end{verbatim}<US><US>16:\end,1:{,11:verbatim,2:}
785    assert!(s.starts_with("Until\x1fUntil:\\end{verbatim}\x1f\x1f"));
786    assert!(s.contains("16:\\end,1:{,11:verbatim,2:}"));
