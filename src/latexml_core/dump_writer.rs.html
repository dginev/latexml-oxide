1//! Writer for Rust-native kernel dump files.
2//!
3//! Serializes the State diff (from snapshot/diff_from_snapshot) into a
4//! compact text format that can be loaded back efficiently.
5//!
6//! This is the "DumpFile" equivalent from Perl's TeX_Job.pool.ltxml.
7//! The workflow follows Perl's Makefile.PL "make formats" scaffold:
8//!   1. take_snapshot() — before processing latex.ltx
9//!   2. Process latex.ltx through the engine
10//!   3. diff_snapshot() — find what changed
11//!   4. write_dump() — serialize the diff
12//!   5. At runtime: load_dump() restores the state
13

23/// Write a state diff to a dump file.
24/// Returns the number of entries successfully written.
25///
26/// Writes in three ordered sections, matching Perl's
27/// `TeX_Job.pool.ltxml::DumpFile` which separates
28/// `@cmds_early` / `@cmds` / `@cmds_late`:
29///
30///   1. **cmds_early** — M:PA / M:MPA let-aliases whose **target
31///      pre-existed** in the bootstrap snapshot (e.g. `\tex_let:D → \let`,
32///      `\tex_def:D → \def`). Applied first because their targets are
33///      always available (they're in the bootstrap pool we loaded
34///      unconditionally before this dump).
35///   2. **cmds** (regular) — V / M:E / M:T / R / C / LC / UC / SC / MC /
36///      DC entries: data, expandable definitions, registers, codes.
37///   3. **cmds_late** — M:PA / M:MPA let-aliases whose **target is
38///      defined by this dump**. Applied last so the target is installed
39///      before the alias fires.
40///
41/// The early/late split requires the caller (ini_tex) to have staged a
42/// snapshot via `state::stage_snapshot("bootstrap")`. If no snapshot is
43/// available, all aliases are emitted as `late` (safe — they just wait
44/// until after regular entries).
45pub fn write_dump(
46  path: &Path,
47  entries: &[(TableName, crate::common::arena::SymStr, Stored)],
48) -> Result<usize, String> {
49  let mut file = std::fs::File::create(path)
50    .map_err(|e| format!("Failed to create dump file: {}", e))?;
51
52  writeln!(file, "# latexml-oxide kernel dump v3").ok();
53  writeln!(file, "# Generated by dump_writer, {} entries", entries.len()).ok();
54  writeln!(file, "# Format: <table>\\t<key>\\t<data>").ok();
55  writeln!(file, "#   <table> = M | V | C | LC | UC | SC | MC | DC").ok();
56  writeln!(file, "#   <data> depends on <table>:").ok();
57  writeln!(file, "#     M/E:   E\\t<cs>\\t<nargs>\\t<flags>\\t<body>\\t<proto>\\t<v3_params>").ok();
58  writeln!(file, "#            v3_params encodes each Parameter structurally so Until:/Match:").ok();
59  writeln!(file, "#            delimiter tokens round-trip (see docs/DUMP_FORMAT_PERL_ANALYSIS.md).").ok();
60  writeln!(file, "#     M/PA:  PA\\t<target_cs>   (MPA for math primitives)").ok();
61  writeln!(file, "#     M/T:   T\\t<meaning_token>").ok();
62  writeln!(file, "#     M/R:   R\\t<cs>\\t<rtype>\\t<value>[\\t<mathglyph>]").ok();
63  writeln!(file, "#     M/N:   N                     (None-meaning sentinel)").ok();
64  writeln!(file, "#     V:     <type>\\t<value>       (B,I,S,CH,T,TK,N,F,G,MG,D,MD,VD)").ok();
65  writeln!(file, "#     C:                  CC\\t<u8>   (catcode)").ok();
66  writeln!(file, "#     DC/LC/UC/SC/MC:     CH\\t<u32>  (char-indexed table entries)").ok();
67  writeln!(file, "#   <flags> = subset of {{L, P}} (long, protected)").ok();
68  writeln!(file, "#   All keys and proto are url-encoded (%09 for tab etc.); token body").ok();
69  writeln!(file, "#   tokens use the form <catcode_int>:<url_encoded_text> joined by commas.").ok();
70
71  // Read the bootstrap snapshot so we can classify each alias's target
72  // as pre-existing (→ early) or defined-during-load (→ late).
73  let bootstrap_snap = crate::state::get_staged_snapshot("bootstrap");
74
75  // Pre-serialize + partition. Skipped keys (ver@..., unserializable
76  // values) fall out naturally.
77  let mut early_aliases: Vec<(String, String, String)> = Vec::new();
78  let mut regular: Vec<(String, String, String)> = Vec::new();
79  let mut late_aliases: Vec<(String, String, String)> = Vec::new();
80  let mut skipped = 0usize;
81
82  for (table, key, value) in entries {
83    let table_code = table_to_code(*table);
84    let key_str = arena::with(*key, |s| s.to_string());
85
86    // Skip \ver@ package version markers (runtime metadata, not definitions)
87    if key_str.starts_with("\\ver@") {
88      skipped += 1;
89      continue;
90    }
91
92    // Perl #2771 (2026-03-13) + upstream IGNORED_SYMBOLS: these are
93    // runtime-only Value entries — control-flow counters, and large
94    // registered-rule tables that can't meaningfully round-trip through
95    // the dump (they hold closures, cross-refs, or re-populate during
96    // engine init anyway). Match Perl's TeX_Job.pool.ltxml list.
97    if matches!(*table, TableName::Value)
98      && matches!(
99        key_str.as_str(),
100        "if_count"
101          | "absorb_count"
102          | "if_stack"
103          | "DOCUMENT_REWRITE_RULES"
104          | "PARAMETER_TYPES"
105          | "TAG_PROPERTIES"
106          | "MATH_LIGATURES"
107          | "TEXT_LIGATURES"
108      )
109    {
110      skipped += 1;
111      continue;
112    }
113
114    let Some(serialized) = serialize_stored(value) else {
115      skipped += 1;
116      continue;
117    };
118
119    // A let-alias entry is an M row whose serialized form starts with
120    // `PA\t` or `MPA\t`. Nothing else produces those prefixes.
121    let is_alias = matches!(*table, TableName::Meaning)
122      && (serialized.starts_with("PA\t") || serialized.starts_with("MPA\t"));
123
124    let row = (table_code.to_string(), url_encode(&key_str), serialized);
125    if is_alias {
126      // Extract the target CS from "PA\t<target>" / "MPA\t<target>"
127      // (we already filtered self-aliases in serialize_stored, so the
128      //  target is always a different CS than the key).
129      let target_raw = row.2.split('\t').nth(1).unwrap_or("");
130      let target = crate::dump_reader::url_decode(target_raw);
131      let target_sym = arena::pin(&target);
132      // Target pre-existed iff it was in the bootstrap meaning table.
133      let pre_existed = bootstrap_snap.as_ref().is_some_and(|snap| {
134        snap.contains_key(&(TableName::Meaning, target_sym))
135      });
136      if pre_existed {
137        early_aliases.push(row);
138      } else {
139        late_aliases.push(row);
140      }
141    } else {
142      regular.push(row);
143    }
144  }
145
146  writeln!(file, "# Section 0: early let-aliases ({}) — target in bootstrap",
147    early_aliases.len()).ok();
148  for (t, k, v) in &early_aliases {
149    writeln!(file, "{}\t{}\t{}", t, k, v)
150      .map_err(|e| format!("Write error: {}", e))?;
151  }
152  writeln!(file, "# Section 1: regular entries ({})", regular.len()).ok();
153  for (t, k, v) in &regular {
154    writeln!(file, "{}\t{}\t{}", t, k, v)
155      .map_err(|e| format!("Write error: {}", e))?;
156  }
157  writeln!(file, "# Section 2: late let-aliases ({}) — target in Section 1",
158    late_aliases.len()).ok();
159  for (t, k, v) in &late_aliases {
160    writeln!(file, "{}\t{}\t{}", t, k, v)
161      .map_err(|e| format!("Write error: {}", e))?;
162  }
163
164  let count = early_aliases.len() + regular.len() + late_aliases.len();
165  writeln!(file, "# Written: {}, Skipped: {}", count, skipped).ok();
166  eprintln!(
167    "[dump_writer] Wrote {} entries to {} ({} skipped; {} early + {} regular + {} late)",
168    count,
169    path.display(),
170    skipped,
171    early_aliases.len(),
172    regular.len(),
173    late_aliases.len(),
174  );
175
176  Ok(count)
177}

179fn table_to_code(t: TableName) -> &'static str {
180  match t {
181    TableName::Meaning => "M",
182    TableName::Value => "V",
183    TableName::Catcode => "C",
184    TableName::Mathcode => "MC",
185    TableName::Sfcode => "SC",
186    TableName::Lccode => "LC",
187    TableName::Uccode => "UC",
188    TableName::Delcode => "DC",
189    TableName::Stash => "ST",
190    TableName::StashActive => "SA",
191  }
192}

194/// Serialize a Stored value to a type-tag + data string.
195/// Returns None if the value can't be serialized.
196fn serialize_stored(stored: &Stored) -> Option<String> {
197  match stored {
198    Stored::None => Some("N".to_string()),
199    Stored::Bool(b) => Some(format!("B\t{}", if *b { "1" } else { "0" })),
200    Stored::Int(i) => Some(format!("I\t{}", i)),
201    Stored::String(s) => Some(format!("S\t{}", url_encode(&arena::with(*s, |s| s.to_string())))),
202    Stored::Charcode(c) => Some(format!("CH\t{}", c)),
203    Stored::Catcode(cc) => Some(format!("CC\t{}", u8::from(*cc))),
204    Stored::Token(t) => Some(format!("T\t{}", serialize_token(t))),
205    Stored::Tokens(tks) => {
206      let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
207      Some(format!("TK\t{}", tok_strs.join(",")))
208    }
209    Stored::Expandable(exp) => {
210      use crate::common::object::Object;
211      let cs_name = exp.get_cs().with_str(url_encode);
212      let nargs = exp.get_num_args();
213      let mut flags = String::new();
214      if exp.is_long { flags.push('L'); }
215      if exp.is_protected() { flags.push('P'); }
216      // Two parallel parameter encodings:
217      //
218      //   - `proto` (v2, 5th field) — `Parameters::stringify()`, a flat
219      //     space-separated prototype string. Round-trips simple specs but
220      //     not `Until:\end{verbatim}` or other brace-in-delimiter forms.
221      //   - `v3_params` (v3, 6th field) — per-Parameter structured record
222      //     carrying (name, spec, flags, extras). Modeled on Perl's
223      //     `P(type, spec, extra=>[T(...)])`; see
224      //     `docs/DUMP_FORMAT_PERL_ANALYSIS.md`. Bypasses `parse_parameters`
225      //     at load time so delimited params round-trip intact.
226      //
227      // We emit both so older readers still work, and newer readers can
228      // prefer v3 transparently.
229      let proto = exp.get_parameters()
230        .map(|p| p.stringify())
231        .unwrap_or_default();
232      let proto_encoded = url_encode(&proto);
233      let v3_params = exp.get_parameters()
234        .map(serialize_parameters_v3)
235        .unwrap_or_default();
236      match exp.get_expansion() {
237        Some(crate::definition::ExpansionBody::Tokens(tks)) => {
238          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
239          Some(format!(
240            "E\t{}\t{}\t{}\t{}\t{}\t{}",
241            cs_name, nargs, flags, tok_strs.join(","), proto_encoded, v3_params
242          ))
243        }
244        // No body at all (e.g. `DefMacro!("\\@gobble{}", None)`): serializes
245        // as an E-entry with empty token body. The reader reconstructs an
246        // Expandable with empty Tokens + the original paramlist, which at
247        // expansion time reads and discards its arguments — semantically
248        // identical to the no-body form's runtime behavior (expandable.rs
249        // `invoke` matches `None` → reads args, returns NO_TOKENS).
250        None => Some(format!("E\t{}\t{}\t{}\t\t{}\t{}", cs_name, nargs, flags, proto_encoded, v3_params)),
251        // Closure-based — can't serialize.
252        Some(crate::definition::ExpansionBody::Closure(_)) => Option::None,
253      }
254    }
255    // Closure-based primitives can't be serialized directly — but if the entry's
256    // CS differs from the primitive's own CS, it's a `\let`-alias, and we CAN
257    // capture the alias so the dump reader can replay `\let <key> <target>` at
258    // load time. This is how `\tex_let:D`, `\tex_def:D`, and hundreds of other
259    // expl3-renamed primitives stay reachable across dump boundaries (the
260    // defining assignment `\let \tex_let:D \let` in expl3-code.tex needs to
261    // survive the dump for `\usepackage{expl3}`'s guard `\ifx\csname
262    // tex_let:D\endcsname\relax` to short-circuit).
263    //
264    // NOTE: the KEY we're saving IS the dump entry's first column; the VALUE
265    // we write is the primitive's OWN CS name. Dump reader compares these:
266    // if equal, it's the "primary" primitive (already in bindings) and skipped;
267    // if different, it's an alias — replay `\let` at load time.
268    Stored::Primitive(p) => {
269      let target_cs = p.cs.with_str(url_encode);
270      Some(format!("PA\t{}", target_cs))
271    }
272    Stored::MathPrimitive(p) => {
273      let target_cs = p.cs.with_str(url_encode);
274      Some(format!("MPA\t{}", target_cs))
275    }
276    Stored::Register(reg) => {
277      // Register: serialize as R\tCS\tTYPE\tVALUE[\tMATHGLYPH]
278      // Only serialize simple registers without getter/setter closures
279      if reg.getter.is_some() || reg.setter.is_some() {
280        return Option::None; // Has closures, can't serialize
281      }
282      let cs_name = reg.cs.with_str(url_encode);
283      let rtype = match reg.register_type {
284        crate::definition::register::RegisterType::Number => "N",
285        crate::definition::register::RegisterType::Dimension => "D",
286        crate::definition::register::RegisterType::Glue => "G",
287        crate::definition::register::RegisterType::MuGlue => "MG",
288        crate::definition::register::RegisterType::Tokens => "TK",
289        crate::definition::register::RegisterType::CharDef => "CD",
290        _ => return Option::None,
291      };
292      let value_str = match &reg.value {
293        Some(crate::definition::register::RegisterValue::Number(n)) => n.value_of().to_string(),
294        Some(crate::definition::register::RegisterValue::Dimension(d)) => d.value_of().to_string(),
295        Some(crate::definition::register::RegisterValue::Glue(g)) => {
296          let mut s = g.skip.to_string();
297          if let Some(p) = g.plus { s.push_str(&format!(",p{}", p)); }
298          if let Some(ref pf) = g.pfill { s.push_str(&format!(",pf{}", fillcode_index(pf))); }
299          if let Some(m) = g.minus { s.push_str(&format!(",m{}", m)); }
300          if let Some(ref mf) = g.mfill { s.push_str(&format!(",mf{}", fillcode_index(mf))); }
301          s
302        }
303        Some(crate::definition::register::RegisterValue::Tokens(tks)) => {
304          let tok_strs: Vec<String> = tks.unlist_ref().iter().map(serialize_token).collect();
305          tok_strs.join(",")
306        }
307        _ => "0".to_string(),
308      };
309      let mut s = format!("R\t{}\t{}\t{}", cs_name, rtype, value_str);
310      if let Some(glyph) = reg.mathglyph {
311        s.push_str(&format!("\t{}", glyph as u32));
312      }
313      Some(s)
314    }
315    Stored::Number(n) => Some(format!("I\t{}", n.value_of())),
316    Stored::Float(f) => Some(format!("F\t{}", f.0)),
317    Stored::Dimension(d) => Some(format!("D\t{}", d.0)),
318    Stored::Glue(g) => {
319      let mut s = format!("G\t{}", g.skip);
320      if let Some(p) = g.plus { s.push_str(&format!(",p{}", p)); }
321      if let Some(ref pf) = g.pfill { s.push_str(&format!(",pf{}", fillcode_index(pf))); }
322      if let Some(m) = g.minus { s.push_str(&format!(",m{}", m)); }
323      if let Some(ref mf) = g.mfill { s.push_str(&format!(",mf{}", fillcode_index(mf))); }
324      Some(s)
325    }
326    Stored::MuDimension(d) => Some(format!("MD\t{}", d.0)),
327    Stored::MuGlue(g) => {
328      let mut s = format!("MG\t{}", g.skip);
329      if let Some(p) = g.plus { s.push_str(&format!(",p{}", p)); }
330      if let Some(ref pf) = g.pfill { s.push_str(&format!(",pf{}", fillcode_index(pf))); }
331      if let Some(m) = g.minus { s.push_str(&format!(",m{}", m)); }
332      if let Some(ref mf) = g.mfill { s.push_str(&format!(",mf{}", fillcode_index(mf))); }
333      Some(s)
334    }
335    Stored::VecDequeStored(vd) if vd.is_empty() => Some("VD\t".to_string()),
336    _ => Option::None,
337  }
338}

345fn serialize_token(t: &crate::token::Token) -> String {
346  let cc: u8 = t.get_catcode().into();
347  let text = t.with_str(url_encode);
348  format!("{}:{}", cc, text)
349}

351/// Serialize a `Parameters` list in the v3 structured format.
352///
353/// Layout (ASCII control-byte delimited so nothing collides with
354/// url-encoded printable content):
355/// - `\x1e` between Parameters (RS, Record Separator)
356/// - `\x1f` between fields of a single Parameter (US, Unit Separator)
357/// - `\x1d` between Tokens lists inside `extras` (GS, Group Separator)
358///
359/// Per-parameter fields: `<name>\x1f<spec>\x1f<flags>\x1f<extras>`
360///   - `<name>` — `Parameter.name` (Perl's `type`), url-encoded.
361///   - `<spec>` — `Parameter.spec`, url-encoded.
362///   - `<flags>` — semicolon-separated `n=1` (novalue) / `o=1` (optional).
363///   - `<extras>` — `<Tokens1>\x1d<Tokens2>…` where each `<TokensN>` is
364///     the same comma-joined `<catcode>:<text>` encoding used for E-entry
365///     bodies. Empty extras → empty field.
366///
367/// The reader (`dump_reader::parse_parameters_v3`) hands the decoded
368/// `(name, spec, Some(extras))` triple directly to `Parameter::new`,
369/// bypassing `parse_parameters` entirely — so `Until:\end{verbatim}`
370/// and `Match:…` round-trip with their catcoded delimiter tokens intact.
371pub(crate) fn serialize_parameters_v3(params: &crate::parameter::Parameters) -> String {
372  let mut out = String::new();
373  for (i, p) in params.get_parameters().iter().enumerate() {
374    if i > 0 {
375      out.push('\x1e');
376    }
377    let name = arena::with(p.name, url_encode);
378    let spec = arena::with(p.spec, url_encode);
379    let mut flags = String::new();
380    if p.novalue {
381      flags.push_str("n=1");
382    }
383    if p.optional {
384      if !flags.is_empty() {
385        flags.push(';');
386      }
387      flags.push_str("o=1");
388    }
389    let extras = p
390      .extra
391      .iter()
392      .map(|tks| {
393        tks.clone()
394          .unlist()
395          .iter()
396          .map(serialize_token)
397          .collect::<Vec<_>>()
398          .join(",")
399      })
400      .collect::<Vec<_>>()
401      .join("\x1d");
402    out.push_str(&name);
403    out.push('\x1f');
404    out.push_str(&spec);
405    out.push('\x1f');
406    out.push_str(&flags);
407    out.push('\x1f');
408    out.push_str(&extras);
409  }
410  out
411}

413fn url_encode(s: &str) -> String {
414  let mut result = String::with_capacity(s.len());
415  for ch in s.chars() {
416    match ch {
417      '\t' => result.push_str("%09"),
418      '\n' => result.push_str("%0A"),
419      '\r' => result.push_str("%0D"),
420      '%' => result.push_str("%25"),
421      ',' => result.push_str("%2C"),
422      _ if ch.is_ascii_control() => {
423        result.push_str(&format!("%{:02X}", ch as u8));
424      }
425      _ => result.push(ch),
426    }
427  }
428  result
429}

438  /// Construct a Parameter without calling `init()` — `init()` needs live
439  /// state (PARAMETER_TYPES table), which we don't set up in unit tests.

442    Parameter {
443      name: arena::pin(name),
444      spec: arena::pin(spec),
445      extra,
446      ..Parameter::default()
447    }

450  /// A Parameters list with two `Plain {}` args encodes as two records
451  /// separated by RS, each `Plain<US>{}<US><US>`. Verifies baseline layout.

462  /// `Until:<delim>` with a brace-containing delimiter. The test covers
463  /// the case that livelocked 00_tokenize under v2: the delimiter tokens
464  /// must serialize as catcoded tokens (not as the raw spec string), so
465  /// `{`/`}` survive as CC_BEGIN / CC_END rather than being re-parsed as
466  /// spec-level braces.
467  #[test]
468  fn v3_encoding_until_with_braces() {
469    let delim = Tokens::new(vec![
470      Token { text: arena::pin("\\end"), code: Catcode::CS },
471      Token { text: arena::pin("{"), code: Catcode::BEGIN },
472      Token { text: arena::pin("verbatim"), code: Catcode::LETTER },
473      Token { text: arena::pin("}"), code: Catcode::END },
474    ]);
475    let ps = Parameters::new(vec![
476      raw_param("Until", "Until:\\end{verbatim}", vec![delim]),
477    ]);
478    let s = serialize_parameters_v3(&ps);
479    // Expected: Until<US>Until:\end{verbatim}<US><US>16:\end,1:{,11:verbatim,2:}
480    assert!(s.starts_with("Until\x1fUntil:\\end{verbatim}\x1f\x1f"));
481    assert!(s.contains("16:\\end,1:{,11:verbatim,2:}"));
482  }
