1//! Wall-clock watchdog that forcibly aborts the process after a deadline.
2//!
3//! The existing `stomach::check_timeout()` is a cooperative mechanism — it only
4//! fires when the digestion loop polls it. That leaves tight native loops
5//! (Marpa precompute / parse, libxml2 post-processing, FFI calls into libxslt,
6//! ...) completely unguarded: a 60-second timeout can easily turn into 10
7//! minutes if control never returns to the digestion loop.
8//!
9//! This module provides a main-level `Watchdog` that spawns a dedicated thread
10//! at construction, wakes after the specified number of seconds, and — if the
11//! watchdog has not yet been cancelled — prints a message and calls
12//! `std::process::abort()`. That guarantees the process dies within
13//! `timeout + poll_interval` of the configured deadline, regardless of what
14//! the main thread is doing.
15//!
16//! # Design notes
17//!
18//! - Uses `Arc<AtomicBool>` for cancellation. Polling every 100 ms keeps the
19//!   cancellation latency low without burning CPU.
20//! - `Drop` on the `Watchdog` handle cancels the watchdog thread, so RAII
21//!   usage (`let _wd = Watchdog::new(secs)`) is sufficient.
22//! - We use `std::process::abort()` rather than `panic!` because panic may
23//!   unwind or be caught by a surrounding `catch_unwind`, which would defeat
24//!   the safety guarantee. `abort()` delivers `SIGABRT` and always terminates
25//!   the process.
26//! - The existing cooperative `stomach::check_timeout()` path is retained: on
27//!   most conversions it fires before the hard abort, giving callers a nice
28//!   `Err(Fatal)` with proper error propagation. The watchdog is a safety
29//!   net for the pathological cases where cooperative polling doesn't happen.
30
31use std::sync::Arc;
32use std::sync::atomic::{AtomicBool, Ordering};
33use std::thread;
34use std::time::{Duration, Instant};
35
36/// Handle to a watchdog thread. Cancels on drop.
37///
38/// `Watchdog::new(0)` is a no-op — produces a handle that does nothing. This
39/// lets call-sites set a watchdog conditionally without special-casing the
40/// "no timeout" branch.
41pub struct Watchdog {
42  cancelled: Arc<AtomicBool>,
43}
44
45impl Watchdog {
46  /// Create a new watchdog. `timeout_secs = 0` disables the watchdog.
47  ///
48  /// The watchdog thread polls `cancelled` every `poll_interval` and aborts
49  /// the process if the deadline is reached without cancellation.
50  pub fn new(timeout_secs: u64) -> Self {
51    let cancelled = Arc::new(AtomicBool::new(false));
52    if timeout_secs > 0 {
53      let c = cancelled.clone();
54      thread::Builder::new()
55        .name("latexml-watchdog".to_string())
56        .spawn(move || Self::run(c, timeout_secs))
57        .expect("watchdog thread spawn failed");
58    }
59    Self { cancelled }
60  }
61
62  fn run(cancelled: Arc<AtomicBool>, timeout_secs: u64) {
63    let deadline = Instant::now() + Duration::from_secs(timeout_secs);
64    let poll_interval = Duration::from_millis(100);
65    while Instant::now() < deadline {
66      if cancelled.load(Ordering::Relaxed) {
67        return; // cancelled: graceful exit.
68      }
69      thread::sleep(poll_interval);
70    }
71    // One last check — avoid racing a cancellation that lands during the
72    // final sleep.
73    if cancelled.load(Ordering::Relaxed) {
74      return;
75    }
76    eprintln!(
77      "latexml-oxide: main-level wall-clock timeout after {}s — aborting process",
78      timeout_secs
79    );
80    std::process::abort();
81  }
82
83  /// Explicitly cancel the watchdog. Idempotent.
84  pub fn cancel(&self) { self.cancelled.store(true, Ordering::Relaxed); }
85}
86
87impl Drop for Watchdog {
88  fn drop(&mut self) { self.cancel(); }
89}
