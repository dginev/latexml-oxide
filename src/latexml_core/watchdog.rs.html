1//! Wall-clock watchdog that forcibly aborts the process after a deadline.
2//!
3//! The existing `stomach::check_timeout()` is a cooperative mechanism — it only
4//! fires when the digestion loop polls it. That leaves tight native loops
5//! (Marpa precompute / parse, libxml2 post-processing, FFI calls into libxslt,
6//! ...) completely unguarded: a 60-second timeout can easily turn into 10
7//! minutes if control never returns to the digestion loop.
8//!
9//! This module provides a main-level `Watchdog` that spawns a dedicated thread
10//! at construction, wakes after the specified number of seconds, and — if the
11//! watchdog has not yet been cancelled — prints a message and calls
12//! `std::process::abort()`. That guarantees the process dies within
13//! `timeout + poll_interval` of the configured deadline, regardless of what
14//! the main thread is doing.
15//!
16//! # Design notes
17//!
18//! - Uses `Arc<AtomicBool>` for cancellation. Polling every 100 ms keeps the cancellation latency
19//!   low without burning CPU.
20//! - `Drop` on the `Watchdog` handle cancels the watchdog thread, so RAII usage (`let _wd =
21//!   Watchdog::new(secs)`) is sufficient.
22//! - We use `std::process::abort()` rather than `panic!` because panic may unwind or be caught by a
23//!   surrounding `catch_unwind`, which would defeat the safety guarantee. `abort()` delivers
24//!   `SIGABRT` and always terminates the process.
25//! - The existing cooperative `stomach::check_timeout()` path is retained: on most conversions it
26//!   fires before the hard abort, giving callers a nice `Err(Fatal)` with proper error propagation.
27//!   The watchdog is a safety net for the pathological cases where cooperative polling doesn't
28//!   happen.
29//!
30//! # Resource limits
31//!
32//! [`Watchdog::with_limits`] guards **both** a wall-clock deadline and a
33//! resident-memory ceiling — the two defenses any executable that converts
34//! arbitrary input needs. It is the shared guard reused by both
35//! `cortex_worker` (in-process, one paper per process) and the
36//! `latexml_oxide --server` LSP (run inside each forked body child, which
37//! self-terminates on breach so the parent reaps it via pipe EOF). The exit
38//! codes are distinct so a supervising parent can tell them apart:
39//! `124` = wall-clock timeout, `137` = memory ceiling.
40//!
41//! # Portability
42//!
43//! The wall-clock guard is portable (`std::thread` + `Instant`). The RAM guard
44//! samples RSS from `/proc/self/status` and is therefore **Linux-only** today;
45//! on other platforms [`process_rss_kb`] returns `None` and the memory ceiling
46//! is silently inactive (the time guard still works). LONG-TERM the guards
47//! should be fully portable — macOS via `task_info(TASK_BASIC_INFO)` and
48//! Windows via `GetProcessMemoryInfo` — so every supported OS gets the same
49//! reliable time + RAM defenses. Tracked as a portability follow-up.
50

52  sync::{
53    Arc,
54    atomic::{AtomicBool, Ordering},
55  },
56  thread,
57  time::{Duration, Instant},

60/// Current resident set size of this process in KiB, or `None` if it can't be
61/// determined (non-Linux, or `/proc` unavailable). Reads `VmRSS` from
62/// `/proc/self/status`. Cheap enough to poll a few times a second.
63pub fn process_rss_kb() -> Option<u64> {
64  let status = std::fs::read_to_string("/proc/self/status").ok()?;
65  for line in status.lines() {
66    if let Some(rest) = line.strip_prefix("VmRSS:") {
67      return rest.split_whitespace().next()?.parse::<u64>().ok();
68    }
69  }
70  None
71}

73/// Exit code used when the wall-clock deadline is exceeded (standard `timeout`).
74pub const EXIT_TIMEOUT: i32 = 124;
75/// Exit code used when the memory ceiling is exceeded (128 + SIGKILL).
76pub const EXIT_OOM: i32 = 137;
77
78/// Handle to a watchdog thread. Cancels on drop.
79///
80/// `Watchdog::new(0)` is a no-op — produces a handle that does nothing. This
81/// lets call-sites set a watchdog conditionally without special-casing the
82/// "no timeout" branch.
83/// Optional hook to run from the watchdog thread immediately before
84/// `exit(124)`. Used by `cortex_worker --standalone` to write a
85/// structured `Status:conversion:3` placeholder to `--output` so the
86/// timeout produces a usable failure artifact instead of a missing
87/// file. Set once at startup via `set_pre_exit_hook`; the hook is
88/// invoked exactly once. Zero overhead on the happy path — only the
89/// watchdog firing reads it.
90type PreExitHook = Box<dyn FnOnce() + Send + 'static>;
91
92static PRE_EXIT_HOOK: std::sync::OnceLock<std::sync::Mutex<Option<PreExitHook>>> =
93  std::sync::OnceLock::new();
94
95pub fn set_pre_exit_hook(hook: PreExitHook) {
96  let cell = PRE_EXIT_HOOK.get_or_init(|| std::sync::Mutex::new(None));
97  if let Ok(mut guard) = cell.lock() {
98    *guard = Some(hook);
99  }
100}
101
102fn run_pre_exit_hook() {
103  if let Some(cell) = PRE_EXIT_HOOK.get()
104    && let Ok(mut guard) = cell.lock()
105    && let Some(hook) = guard.take()
106  {
107    hook();
108  }
109}

111pub struct Watchdog {
112  cancelled: Arc<AtomicBool>,
113}
114
115impl Watchdog {
116  /// Create a wall-clock-only watchdog. `timeout_secs = 0` disables it.
117  /// Equivalent to [`Watchdog::with_limits(timeout_secs, 0)`].
118  pub fn new(timeout_secs: u64) -> Self { Self::with_limits(timeout_secs, 0) }
119
120  /// Create a watchdog guarding a wall-clock deadline **and** a resident-memory
121  /// ceiling. `timeout_secs = 0` disables the time guard; `max_rss_kb = 0`
122  /// disables the memory guard. With both `0` this is a no-op handle.
123  ///
124  /// The thread polls `cancelled`, the deadline, and RSS every `poll_interval`.
125  /// On a time breach it exits [`EXIT_TIMEOUT`]; on a memory breach,
126  /// [`EXIT_OOM`]. The memory guard is inactive where [`process_rss_kb`]
127  /// returns `None` (non-Linux); see the module portability note.
128  pub fn with_limits(timeout_secs: u64, max_rss_kb: u64) -> Self {
129    let cancelled = Arc::new(AtomicBool::new(false));
130    if timeout_secs > 0 || max_rss_kb > 0 {
131      let c = cancelled.clone();
132      thread::Builder::new()
133        .name("latexml-watchdog".to_string())
134        .spawn(move || Self::run(c, timeout_secs, max_rss_kb))
135        .expect("watchdog thread spawn failed");
136    }
137    Self { cancelled }
138  }
139
140  fn run(cancelled: Arc<AtomicBool>, timeout_secs: u64, max_rss_kb: u64) {
141    let deadline = (timeout_secs > 0).then(|| Instant::now() + Duration::from_secs(timeout_secs));
142    let poll_interval = Duration::from_millis(100);
143    loop {
144      if cancelled.load(Ordering::Relaxed) {
145        return; // cancelled: graceful exit.
146      }
147      if let Some(deadline) = deadline
148        && Instant::now() >= deadline
149      {
150        if cancelled.load(Ordering::Relaxed) {
151          return;
152        }
153        eprintln!(
154          "Fatal:timeout:wallclock latexml-oxide: main-level wall-clock timeout after {timeout_secs}s — exiting process"
155        );
156        // Run the optional pre-exit hook (e.g. cortex_worker writing a
157        // structured Status:conversion:3 placeholder to its --output path)
158        // BEFORE exiting. The hook is invoked at most once per process.
159        run_pre_exit_hook();
160        // `std::process::exit(124)` instead of `abort()`: the watchdog must
161        // terminate the whole process (the worker thread is presumed wedged
162        // in a tight loop that won't observe a cooperative cancel), but
163        // `abort()` produces a "Aborted (core dumped)" SIGABRT trace from
164        // the shell. `exit(124)` (standard timeout exit code) runs atexit
165        // handlers, flushes stderr, and leaves a clean exit signal the
166        // parent harness can interpret as "paper timed out" without
167        // conflating it with a Rust panic / memory corruption. Witnesses:
168        // 2602.11915, 2604.11500, 2604.13944, hep-ph9205242, q-alg9604005,
169        // q-alg9605003, q-alg9605028 — the 7 "Aborted" rows in the
170        // 2026-05-13 588-paper sweep.
171        std::process::exit(EXIT_TIMEOUT);
172      }
173      if max_rss_kb > 0
174        && let Some(rss) = process_rss_kb()
175        && rss > max_rss_kb
176      {
177        if cancelled.load(Ordering::Relaxed) {
178          return;
179        }
180        eprintln!(
181          "Fatal:oom:rss latexml-oxide: resident memory {}MB exceeded the {}MB ceiling — exiting process",
182          rss / 1024,
183          max_rss_kb / 1024
184        );
185        run_pre_exit_hook();
186        std::process::exit(EXIT_OOM);
187      }
188      thread::sleep(poll_interval);
189    }
190  }
191
192  /// Explicitly cancel the watchdog. Idempotent.
193  pub fn cancel(&self) { self.cancelled.store(true, Ordering::Relaxed); }
194}
195
196impl Drop for Watchdog {
197  fn drop(&mut self) { self.cancel(); }
198}

200#[cfg(test)]
201mod tests {
202  use super::*;
203
204  #[test]
205  fn watchdog_zero_timeout_is_noop() {
206    // timeout_secs=0 should NOT spawn a thread and NOT abort.
207    let w = Watchdog::new(0);
208    assert!(
209      !w.cancelled.load(Ordering::Relaxed),
210      "initial cancelled state is false"
211    );
212    // Dropping is safe — there's no live thread to interact with.
213    drop(w);
214  }
215
216  #[test]
217  fn watchdog_cancel_is_idempotent() {
218    let w = Watchdog::new(60);
219    w.cancel();
220    assert!(w.cancelled.load(Ordering::Relaxed));
221    // Calling again is a no-op.
222    w.cancel();
223    assert!(w.cancelled.load(Ordering::Relaxed));
224  }
225
226  #[test]
227  fn watchdog_drop_cancels() {
228    let cancelled_ref = {
229      let w = Watchdog::new(60);
230      // Grab a reference to the atomic so we can inspect post-drop.
231      w.cancelled.clone()
232    }; // w dropped here
233    assert!(
234      cancelled_ref.load(Ordering::Relaxed),
235      "drop should set cancelled=true"
236    );
237  }
238
239  #[test]
240  fn watchdog_explicit_cancel_before_drop() {
241    // Pre-drop cancellation is also reflected on the clone.
242    let w = Watchdog::new(60);
243    let cancelled_ref = w.cancelled.clone();
244    w.cancel();
245    assert!(cancelled_ref.load(Ordering::Relaxed));
246    // Explicit drop after cancel remains idempotent.
247    drop(w);
248    assert!(cancelled_ref.load(Ordering::Relaxed));
249  }
250
251  #[test]
252  fn watchdog_long_timeout_doesnt_fire_quickly() {
253    // 60-second timeout shouldn't fire during a 50 ms sleep.
254    let _w = Watchdog::new(60);
255    thread::sleep(Duration::from_millis(50));
256    // If the watchdog had fired, we'd be dead. We made it here → fine.
257  }
258}
