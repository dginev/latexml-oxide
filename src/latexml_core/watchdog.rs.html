1//! Wall-clock watchdog that forcibly aborts the process after a deadline.
2//!
3//! The existing `stomach::check_timeout()` is a cooperative mechanism — it only
4//! fires when the digestion loop polls it. That leaves tight native loops
5//! (Marpa precompute / parse, libxml2 post-processing, FFI calls into libxslt,
6//! ...) completely unguarded: a 60-second timeout can easily turn into 10
7//! minutes if control never returns to the digestion loop.
8//!
9//! This module provides a main-level `Watchdog` that spawns a dedicated thread
10//! at construction, wakes after the specified number of seconds, and — if the
11//! watchdog has not yet been cancelled — prints a message and calls
12//! `std::process::abort()`. That guarantees the process dies within
13//! `timeout + poll_interval` of the configured deadline, regardless of what
14//! the main thread is doing.
15//!
16//! # Design notes
17//!
18//! - Uses `Arc<AtomicBool>` for cancellation. Polling every 100 ms keeps the cancellation latency
19//!   low without burning CPU.
20//! - `Drop` on the `Watchdog` handle cancels the watchdog thread, so RAII usage (`let _wd =
21//!   Watchdog::new(secs)`) is sufficient.
22//! - We use `std::process::abort()` rather than `panic!` because panic may unwind or be caught by a
23//!   surrounding `catch_unwind`, which would defeat the safety guarantee. `abort()` delivers
24//!   `SIGABRT` and always terminates the process.
25//! - The existing cooperative `stomach::check_timeout()` path is retained: on most conversions it
26//!   fires before the hard abort, giving callers a nice `Err(Fatal)` with proper error propagation.
27//!   The watchdog is a safety net for the pathological cases where cooperative polling doesn't
28//!   happen.
29
30use std::sync::Arc;
31use std::sync::atomic::{AtomicBool, Ordering};
32use std::thread;
33use std::time::{Duration, Instant};
34
35/// Handle to a watchdog thread. Cancels on drop.
36///
37/// `Watchdog::new(0)` is a no-op — produces a handle that does nothing. This
38/// lets call-sites set a watchdog conditionally without special-casing the
39/// "no timeout" branch.
40/// Optional hook to run from the watchdog thread immediately before
41/// `exit(124)`. Used by `cortex_worker --standalone` to write a
42/// structured `Status:conversion:3` placeholder to `--output` so the
43/// timeout produces a usable failure artifact instead of a missing
44/// file. Set once at startup via `set_pre_exit_hook`; the hook is
45/// invoked exactly once. Zero overhead on the happy path — only the
46/// watchdog firing reads it.
47type PreExitHook = Box<dyn FnOnce() + Send + 'static>;
48
49static PRE_EXIT_HOOK: std::sync::OnceLock<std::sync::Mutex<Option<PreExitHook>>> =
50  std::sync::OnceLock::new();
51
52pub fn set_pre_exit_hook(hook: PreExitHook) {
53  let cell = PRE_EXIT_HOOK.get_or_init(|| std::sync::Mutex::new(None));
54  if let Ok(mut guard) = cell.lock() {
55    *guard = Some(hook);
56  }
57}
58
59fn run_pre_exit_hook() {
60  if let Some(cell) = PRE_EXIT_HOOK.get() {
61    if let Ok(mut guard) = cell.lock() {
62      if let Some(hook) = guard.take() {
63        hook();
64      }
65    }
66  }
67}
68
69pub struct Watchdog {
70  cancelled: Arc<AtomicBool>,
71}
72
73impl Watchdog {
74  /// Create a new watchdog. `timeout_secs = 0` disables the watchdog.
75  ///
76  /// The watchdog thread polls `cancelled` every `poll_interval` and aborts
77  /// the process if the deadline is reached without cancellation.
78  pub fn new(timeout_secs: u64) -> Self {
79    let cancelled = Arc::new(AtomicBool::new(false));
80    if timeout_secs > 0 {
81      let c = cancelled.clone();
82      thread::Builder::new()
83        .name("latexml-watchdog".to_string())
84        .spawn(move || Self::run(c, timeout_secs))
85        .expect("watchdog thread spawn failed");
86    }
87    Self { cancelled }
88  }
89
90  fn run(cancelled: Arc<AtomicBool>, timeout_secs: u64) {
91    let deadline = Instant::now() + Duration::from_secs(timeout_secs);
92    let poll_interval = Duration::from_millis(100);
93    while Instant::now() < deadline {
94      if cancelled.load(Ordering::Relaxed) {
95        return; // cancelled: graceful exit.
96      }
97      thread::sleep(poll_interval);
98    }
99    // One last check — avoid racing a cancellation that lands during the
100    // final sleep.
101    if cancelled.load(Ordering::Relaxed) {
102      return;
103    }
104    eprintln!(
105      "Fatal:timeout:wallclock latexml-oxide: main-level wall-clock timeout after {}s — exiting process",
106      timeout_secs
107    );
108    // Run the optional pre-exit hook (e.g. cortex_worker writing a
109    // structured Status:conversion:3 placeholder to its --output path)
110    // BEFORE `exit(124)`. The hook is invoked at most once per process.
111    run_pre_exit_hook();
112    // `std::process::exit(124)` instead of `abort()`: the watchdog must
113    // terminate the whole process (the worker thread is presumed wedged
114    // in a tight loop that won't observe a cooperative cancel), but
115    // `abort()` produces a "Aborted (core dumped)" SIGABRT trace from
116    // the shell. `exit(124)` (standard timeout exit code) runs atexit
117    // handlers, flushes stderr, and leaves a clean exit signal the
118    // parent harness can interpret as "paper timed out" without
119    // conflating it with a Rust panic / memory corruption. Witnesses:
120    // 2602.11915, 2604.11500, 2604.13944, hep-ph9205242, q-alg9604005,
121    // q-alg9605003, q-alg9605028 — the 7 "Aborted" rows in the
122    // 2026-05-13 588-paper sweep.
123    std::process::exit(124);
124  }
125
126  /// Explicitly cancel the watchdog. Idempotent.
127  pub fn cancel(&self) { self.cancelled.store(true, Ordering::Relaxed); }
128}
129
130impl Drop for Watchdog {
131  fn drop(&mut self) { self.cancel(); }
132}

134#[cfg(test)]
135mod tests {
136  use super::*;
137
138  #[test]
139  fn watchdog_zero_timeout_is_noop() {
140    // timeout_secs=0 should NOT spawn a thread and NOT abort.
141    let w = Watchdog::new(0);
142    assert!(
143      !w.cancelled.load(Ordering::Relaxed),
144      "initial cancelled state is false"
145    );
146    // Dropping is safe — there's no live thread to interact with.
147    drop(w);
148  }
149
150  #[test]
151  fn watchdog_cancel_is_idempotent() {
152    let w = Watchdog::new(60);
153    w.cancel();
154    assert!(w.cancelled.load(Ordering::Relaxed));
155    // Calling again is a no-op.
156    w.cancel();
157    assert!(w.cancelled.load(Ordering::Relaxed));
158  }
159
160  #[test]
161  fn watchdog_drop_cancels() {
162    let cancelled_ref = {
163      let w = Watchdog::new(60);
164      // Grab a reference to the atomic so we can inspect post-drop.
165      w.cancelled.clone()
166    }; // w dropped here
167    assert!(
168      cancelled_ref.load(Ordering::Relaxed),
169      "drop should set cancelled=true"
170    );
171  }
172
173  #[test]
174  fn watchdog_explicit_cancel_before_drop() {
175    // Pre-drop cancellation is also reflected on the clone.
176    let w = Watchdog::new(60);
177    let cancelled_ref = w.cancelled.clone();
178    w.cancel();
179    assert!(cancelled_ref.load(Ordering::Relaxed));
180    // Explicit drop after cancel remains idempotent.
181    drop(w);
182    assert!(cancelled_ref.load(Ordering::Relaxed));
183  }
184
185  #[test]
186  fn watchdog_long_timeout_doesnt_fire_quickly() {
187    // 60-second timeout shouldn't fire during a 50 ms sleep.
188    let _w = Watchdog::new(60);
189    thread::sleep(Duration::from_millis(50));
190    // If the watchdog had fired, we'd be dead. We made it here → fine.
191  }
192}
