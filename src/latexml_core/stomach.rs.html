8// Conversion timeout: thread-local deadline. When set, digest loops check it.
9thread_local! {
10  static CONVERSION_DEADLINE: Cell<Option<Instant>> = const { Cell::new(None) };
11}
12
13/// Set a conversion timeout (seconds from now). 0 = no timeout.
14pub fn set_timeout(seconds: u64) {
15  if seconds > 0 {
16    CONVERSION_DEADLINE.with(|d| d.set(Some(Instant::now() + std::time::Duration::from_secs(seconds))));
17  } else {
18    CONVERSION_DEADLINE.with(|d| d.set(None));
19  }
20}
21
22/// Check if conversion has timed out. Returns Err if deadline exceeded.
23pub fn check_timeout() -> Result<()> {
24  CONVERSION_DEADLINE.with(|d| {
25    if let Some(deadline) = d.get() {
26      if Instant::now() > deadline {
27        fatal!(Timeout, Convert, "Conversion timed out!");
28      }
29    }
30    Ok(())
31  })
32}

119/// Adds a new stack frame for a TeX group.
120pub fn push_stack_frame(nobox: bool) {
121  let current_token = get_current_token().unwrap_or_else(|| T_CS!("\\relax"));
122  push_frame();
123  assign_value(
124    "beforeAfterGroup",
125    Stored::VecDequeStored(VecDeque::new()),
126    Some(Scope::Local),
127  ); // ALWAYS bind this!
128  assign_value(
129    "afterGroup",
130    Stored::VecDequeStored(VecDeque::new()),
131    Some(Scope::Local),
132  ); // ALWAYS bind this!
133  assign_value("afterAssignment", Stored::None, Some(Scope::Local)); // ALWAYS bind this!
134  assign_value("groupNonBoxing", nobox, Some(Scope::Local)); // ALWAYS bind this!
135  assign_value("groupInitiator", current_token, Some(Scope::Local));
136  assign_value(
137    "groupInitiatorLocator",
138    gullet::get_locator(),
139    Some(Scope::Local),
140  );
141  if !nobox {
142    // For begingroup/endgroup
143    stomach_mut!().boxing.push(current_token)
144  }
145}
146/// Execute tokens stored on beforeAfterGroup (if any); done before popping a stack frame.
147/// Perl: sub executeBeforeAfterGroup (Stomach.pm lines 286-295)
148pub fn execute_before_after_group() -> Result<()> {
149  if let Some(Stored::VecDequeStored(beforeafter)) = remove_value("beforeAfterGroup") {
150    if !beforeafter.is_empty() {
151      let mut result = Vec::new();
152      for beforeafter_frame in beforeafter.into_iter() {
153        match beforeafter_frame {
154          Stored::Tokens(frametoks) => result.push(frametoks.be_digested()?),
155          Stored::Token(frametok) => result.push(frametok.be_digested()?),
156          _ => {
157            // Unexpected value type in beforeAfterGroup — skip silently
158            // rather than panic (could occur with non-standard TeX constructs)
159          },
160        }
161      }
162      // TODO
163      // if (my ($x) = grep { !$_->isaBox } @result) {
164      // Error('misdefined', $x, $self, "Expected a Box|List|Whatsit, but got '" . Stringify($x) .
165      // "'"); @result = (makeMisdefinedError(@result)); }
166      {
167        stomach_mut!().box_list.extend(result);
168      }
169    }
170  }
171  Ok(())
172}

174/// Removes the last/current stack frame, ending a TeX group
175pub fn pop_stack_frame(nobox: bool) -> Result<()> {
176  let after = remove_value("afterGroup");
177  execute_before_after_group()?;
178  pop_frame()?;
179  if !nobox {
180    {
181      stomach_mut!().boxing.pop(); // For begingroup/endgroup
182    }
183  }
184  if let Some(Stored::VecDequeStored(after_entries)) = after {
185    for entry in after_entries.into_iter().rev() {
186      match entry {
187        Stored::Tokens(t) => gullet::unread(t),
188        Stored::Token(t) => gullet::unread_one(t),
189        other => panic!(r"\aftergroup should be used with tokens, got instead: {other:?}"),
190      };
191    }
192  }
193  Ok(())
194}

196/// explain the current frame
197pub fn current_frame_message() -> String {
198  let target = if is_value_bound("MODE", Some(0)) {
199    // SET mode in CURRENT frame ?
200    Cow::Owned(s!("mode-switch to {}", lookup_string("MODE")))
201  } else if lookup_bool("groupNonBoxing") {
202    // Current frame is a non-boxing group?
203    Cow::Borrowed("non-boxing group")
204  } else {
205    Cow::Borrowed("boxing group")
206  };
207
208  let initiator = if let Some(t) = lookup_token("groupInitiator") {
209    t.stringify()
210  } else {
211    String::new()
212  };
213  let locator = lookup_string("groupInitiatorLocator");
214  s!(
215    "current frame is {} due to {} {}",
216    target,
217    initiator,
218    locator
219  )
220}

233/// End a level of binding by popping the last stack frame,
234/// undoing whatever bindings appeared there, and also
235/// decrementing the level of boxing.
236pub fn egroup() -> Result<()> {
237  if is_value_bound("BOUND_MODE", Some(0)) {
238    // Last stack frame was a mode switch!?!?!
239    // Don't pop if there's an error; maybe we'll recover?
240    Error!(
241      "unexpected",
242      get_current_token().unwrap(),
243      s!(
244        "Attempt to close a group that switched to mode {}; {}",
245        lookup_string("MODE"),
246        current_frame_message()
247      )
248    );
249  } else if lookup_bool("groupNonBoxing") {
250    // or group was opened with \begingroup
251    Error!(
252      "unexpected",
253      get_current_token().unwrap(),
254      s!("Attempt to close boxing group; {}", current_frame_message())
255    );
256  } else {
257    // Don't pop if there's an error; maybe we'll recover?
258    pop_stack_frame(false)?;
259  }
260  // Perl's egroup does NOT touch $ALIGN_STATE — tracked at scan level only.
261  Ok(())
262}
263/// Begin a new level of binding by pushing a new stack frame.
264pub fn begingroup() { push_stack_frame(true); }
265/// End a level of binding by popping the last stack frame,
266/// undoing whatever bindings appeared there.
267pub fn endgroup() -> Result<()> {
268  if is_value_bound("BOUND_MODE", Some(0)) {
269    // Last stack frame was a mode switch!?!?!
270    // Don't pop if there's an error; maybe we'll recover?
271    Error!(
272      "unexpected",
273      get_current_token().unwrap().to_string(),
274      s!(
275        "Attempt to close a group that switched to mode {}; {}",
276        lookup_string("MODE"),
277        current_frame_message()
278      )
279    );
280  } else if !lookup_bool("groupNonBoxing") {
281    // or group was opened with \bgroup
282    Error!(
283      "unexpected",
284      get_current_token().unwrap().to_string(),
285      s!(
286        "Attempt to close non-boxing group; {}",
287        current_frame_message()
288      )
289    );
290  } else {
291    pop_stack_frame(true)?;
292  }
293  Ok(())
294}

311  assign_value("MODE", arena::pin(bound_mode), Some(Scope::Local));
312  assign_value("IN_MATH", ismath, Some(Scope::Local));
313  if mode == prevmode {
314  } else if ismath {
315    let curfont = lookup_font().unwrap();
316    // When entering math mode, we set the font to the default math font,
317    // and save the text font for any embedded text.

352  }

356/// Map user-facing mode names to internal bound mode names.
357/// Perl: our %bindable_mode = (text => 'restricted_horizontal', ...)
358fn bindable_mode(umode: &str) -> Option<&'static str> {
359  match umode {
360    "text" | "restricted_horizontal" => Some("restricted_horizontal"),
361    "vertical" | "internal_vertical" => Some("internal_vertical"),
362    "math" | "inline_math" => Some("math"),
363    "display_math" => Some("display_math"),
364    _ => None,
365  }
366}

368/// Begin processing in `mode`; one of "text", "display-math" or "inline-math".
369/// This also begins a new level of grouping and switches to a font
370/// appropriate for the mode.
371/// If `noframe` is true, skip pushing a stack frame (e.g. for \begin{document}).
372/// Perl: sub beginMode (Stomach.pm lines 474-517)
373pub fn begin_mode(mode: &str) -> Result<()> {
374  begin_mode_opt(mode, false)
375}
376/// Like `begin_mode`, but with an explicit `noframe` option.
377/// When `noframe` is true, no stack frame is pushed (the caller already did bgroup).
378pub fn begin_mode_opt(mode: &str, noframe: bool) -> Result<()> {
379  if let Some(bound_mode) = bindable_mode(mode) {
380    if !noframe {
381      push_stack_frame(false); // Effectively bgroup
382    }
383    // Perl: $STATE->assignValue(BOUND_MODE => $mode, 'local');
384    assign_value("BOUND_MODE", arena::pin(bound_mode), Some(Scope::Local));
385    set_mode(bound_mode)?;
386    // Perl Stomach.pm lines 504-507: inject \everymath or \everydisplay tokens
387    // Display math gets \everydisplay, inline math gets \everymath (not both).
388    if bound_mode.contains("math") {
389      let is_display = bound_mode == "display_math";
390      let reg_name = if is_display { "\\everydisplay" } else { "\\everymath" };
391      if let Some(RegisterValue::Tokens(toks)) = lookup_register(reg_name, Vec::new())? {
392        let toks = toks.unlist();
393        if !toks.is_empty() {
394          gullet::unread(Tokens::new(toks));
395        }
396      }
397    }
398    Ok(())
399  } else {
400    Warn!("unexpected", mode, s!("Cannot enter {mode} mode"));
401    Ok(())
402  }
403}
404/// End processing in `mode`; an error is signalled if `stomach` is not
405/// currently in `mode`.  This also ends a level of grouping.
406/// Perl: sub endMode (Stomach.pm lines 522-541)
407pub fn end_mode(mode: &str) -> Result<()> {
408  end_mode_opt(mode, false)
409}
410/// Like `end_mode`, but with an explicit `noframe` option.
411/// When `noframe` is true, executeBeforeAfterGroup is run but the stack frame is not popped.
412pub fn end_mode_opt(mode: &str, noframe: bool) -> Result<()> {
413  if let Some(bound_mode) = bindable_mode(mode) {
414    // Perl checks BOUND_MODE, not MODE, for the frame check
415    if !is_value_bound("BOUND_MODE", Some(0))
416      || (lookup_string("BOUND_MODE") != bound_mode)
417    {
418      // Last stack frame was NOT a mode switch, or was a switch to a different mode.
419      // Log the error but attempt recovery: if the current MODE matches what we expect,
420      // pop intervening non-mode frames to find our mode-switch frame.
421      // This prevents cascading errors from eating the rest of the document.
422      let current_mode = lookup_string("MODE");
423      let message = s!(
424        "Attempt to end mode `{}` in `{}`",
425        mode,
426        current_mode
427      );
428      let category = match get_current_token() {
429        Some(ref token) => token.to_string(),
430        None => String::from("mode"),
431      };
432      Error!("unexpected", category, &message);
433
434      // Recovery: pop non-mode frames until we find our mode-switch frame (max 4 levels)
435      for _ in 0..4 {
436        if is_value_bound("BOUND_MODE", Some(0))
437          && lookup_string("BOUND_MODE") == bound_mode
438        {
439          // Found the matching mode-switch frame — pop it
440          if bound_mode.ends_with("vertical") {
441            leave_horizontal_internal();
442          }
443          pop_stack_frame(false)?;
444          return Ok(());
445        }
446        if is_value_bound("BOUND_MODE", Some(0)) {
447          // Different mode-switch frame — don't pop past it
448          break;
449        }
450        // Pop this non-mode frame to get closer to our mode-switch frame
451        pop_stack_frame(false)?;
452      }
453      // If we couldn't find the frame, at least the error was logged
454    } else {
455      // Perl: leaveHorizontal_internal($self) if $mode =~ /vertical$/;
456      if bound_mode.ends_with("vertical") {
457        leave_horizontal_internal();
458      }
459      if noframe {
460        // No pop, but at least do beforeAfterGroup
461        execute_before_after_group()?;
462      } else {
463        pop_stack_frame(false)?;
464      }
465    }
466  } else {
467    Warn!("unexpected", mode, s!("Cannot end {mode} mode"));
468  }
469  Ok(())
470}

487/// Resume vertical mode by executing \par, in TeX-like fashion.
488/// Perl: sub leaveHorizontal
489pub fn leave_horizontal() -> Result<()> {
490  let mode = lookup_string("MODE");
491  let bound = lookup_string("BOUND_MODE");
492  if mode == "horizontal" && bound.ends_with("vertical") {
493    // This needs to be an invisible, and slightly gentler, \par
494    assign_value("INTERNAL_PAR", true, Some(Scope::Local));
495    let par_result = invoke_token(&T_CS!("\\par"))?;
496    push_box_list_vec(par_result);
497    assign_value("INTERNAL_PAR", false, Some(Scope::Local));
498  }
499  Ok(())

502/// Resume vertical mode internally: reset mode without firing \par.
503/// Used within argument digestion, e.g. endMode for vertical modes.
504/// Perl: sub leaveHorizontal_internal
505pub fn leave_horizontal_internal() {
506  let mode = lookup_string("MODE");
507  let bound = lookup_string("BOUND_MODE");
508  if mode == "horizontal" && bound.ends_with("vertical") {
509    repack_horizontal();
510    assign_value_inplace("MODE", arena::pin(&bound));
511  }

514/// Repack recently digested horizontal items into single horizontal List.
515/// Note that TeX would have done paragraph line-breaking, resulting in essentially
516/// a vertical list.
517/// Perl: sub repackHorizontal (Stomach.pm lines 440-454)
518pub fn repack_horizontal() {
519  let mut stomach = stomach_mut!();
520  let mut para: Vec<Digested> = Vec::new();
521  let mut keep = false;
522
523  loop {
524    let should_pop = if let Some(item) = stomach.box_list.last() {
525      let mode_str = item
526        .get_property("mode")
527        .map(|v| v.to_string())
528        .unwrap_or_else(|| "horizontal".to_string());
529      if mode_str == "horizontal"
530        || mode_str == "restricted_horizontal"
531        || mode_str == "math"
532      {
533        // if ONLY horizontal mode spaces, we can prune them; it just makes an empty ltx:p
534        if mode_str != "horizontal" || !item.get_property_bool("isSpace") {
535          keep = true;
536        }
537        true
538      } else {
539        false
540      }
541    } else {
542      false
543    };
544
545    if should_pop {
546      para.push(stomach.box_list.pop().unwrap());
547    } else {
548      break;
549    }
550  }
551
552  // Items were popped in reverse order, so reverse them back
553  para.reverse();
554
555  if keep {
556    let mut list = List::new(para);
557    list.mode = Some(TexMode::Text); // "horizontal" in Perl
558    // Perl: List(@para, mode => 'horizontal') — set mode property string
559    // This is needed for compute_boxes_size vertical layout to detect paragraph Lists
560    list.set_property("mode", Stored::String(arena::pin_static("horizontal")));
561    // Perl: $list->setProperty(width => LookupRegister('\hsize')) if $mode eq 'horizontal';
562    if let Some(hsize) = lookup_dimension("\\hsize") {
563      list.set_property("width", hsize);
564    }
565    stomach.box_list.push(Digested::from(list));
566  }
567}

606/// Digest a list of tokens independent from any current Gullet.
607/// Typically used to digest arguments to primitives or constructors.
608/// Returns a List containing the digested material.
609pub fn digest<T: Into<Tokens>>(tokens: T) -> Result<Digested> {
610  let tokens: Tokens = tokens.into();
611  if tokens.is_empty() {
612    return Ok(Digested::default());
613  }
614  gullet::reading_from_mouth(Mouth::default(), || {
615    gullet::unread(tokens);
616    clear_prefixes(); // prefixes shouldn't apply here.
617    let mode = if lookup_bool("IN_MATH") {
618      TexMode::Math
619    } else {
620      TexMode::Text
621    };
622    let initdepth = stomach!().boxing.len();
623    let depth = initdepth;
624    new_local_box_list();
625    while let Some(token) = match gullet::get_pending_comment() {
626      Some(comment) => Some(comment),
627      None => gullet::read_x_token(Some(true), false, None)?,
628    } {
629      // Done if we run out of tokens
630      let invoked = invoke_token(&token)?;
631      extend_box_list(invoked);
632
633      if initdepth > stomach!().boxing.len() {
634        // if we've closed the initial mode.
635        break;
636      }
637      if initdepth < depth {
638        // TODO
639        fatal!(Internal, EoF, "We've fallen off the end, somehow !?!?!?");
640        //     Fatal('internal', '<EOF>', self,
641        //       "We've fallen off the end, somehow!?!?!",
642        //       "Last token " . ToString($LaTeXML::CURRENT_TOKEN)
643        //         . " (Boxing depth was $initdepth, now $depth: Boxing generated by "
644        //         . join(', ', map { ToString($_) } @{ $self{boxing} }))
645        //       if $initdepth < $depth;
646      }
647    }
648
649    let mut digested_list = List::new(expire_local_box_list());
650    digested_list.mode = Some(mode);
651    digested_list.into()
652  })
653}

655/// Return the digested `List` after reading and digesting a body from the its Gullet.
656/// The body extends until the current level of boxing or environment is closed.
657pub fn digest_next_body(terminal_opt: Option<Token>) -> Result<Vec<Digested>> {
658  let start_location = { gullet::get_locator() };
659
660  let init_depth = { stomach!().boxing.len() };
661  let mut found_token = false;
662  let mut found_terminal = false;
663  new_local_box_list();
664  let alignment_opt = lookup_alignment();
665  // TODO: bookkeep for "expected" warning
666  //let mut aug = Vec::new();
667
668  // try reading a executable token
669  while let Some(token) = match gullet::get_pending_comment() {
670    Some(comment) => Some(comment),
671    None => gullet::read_x_token(Some(true), false, None)?,
672  } {
673    // Check conversion timeout
674    check_timeout()?;
675    // done if we run out of tokens
676    found_token = true;
677    // first, check for alignment case
678    // Perl #2775: only fire at the original alignment nesting level,
679    // not inside deeper boxing groups (e.g. \vbox inside a tabular cell).
680    if alignment_opt.is_some()
681      && !stomach!().box_list.is_empty()
682      && (stomach!().boxing.len() <= init_depth)
683      && (token == T_ALIGN!()
684        || token == T_CS!("\\cr")
685        || token == T_CS!("\\lx@hidden@cr")
686        || token == T_CS!("\\lx@hidden@crcr"))
687    {
688      gullet::unread_one(token);
689      return Ok(expire_local_box_list());
690    }
691    // normal case
692    let invoked = invoke_token(&token)?;
693    extend_box_list(invoked);
694
695    if let Some(ref terminal) = terminal_opt {
696      if &token == terminal {
697        found_terminal = true;
698        break;
699      }
700    }
701    if init_depth > stomach!().boxing.len() {
702      break;
703    }
704  }
705
706  if let Some(ref terminal) = terminal_opt {
707    if !found_terminal {
708      let message = s!(
709        "body should have ended with {:?}. current body started at {:?}",
710        terminal,
711        start_location
712      );
713      Warn!("expected", terminal, message);
714    }
715  }
716  // and add a Dummy `trailer' if none explicit.
717  if !found_token {
718    push_box_list(Digested::from(Tbox::default()));
719    // info!(target:"digest_next_body","no_token");
720  }
721  Ok(expire_local_box_list())
722}

724/// a convenience function for including chunks of raw TeX (or LaTeX) code
725/// It is useful for copying portions of the normal
726/// implementation that can be handled simply using macros and primitives.
727pub fn raw_tex(text: &str) -> Result<()> {
728  // It could be as simple as this, except if catcodes get changed, it's too late!!!
729  //  Digest(TokenizeInternal($text));
730  let raw_tex_mouth = Mouth::new(
731    text,
732    Some(MouthOptions {
733      fordefinitions: true,
734      at_letter: true,
735      ..MouthOptions::default()
736    }),
737  )?;
738  gullet::reading_from_mouth(raw_tex_mouth, || -> Result<()> {
739    while let Some(token) = gullet::read_x_token(Some(false), false, None)? {
740      if token.get_catcode() != Catcode::SPACE {
741        invoke_token(&token)?;
742      }
743    }
744    Ok(())
745  })?;
746  Ok(())
747}

749/// Invoke a token
750///
751/// If it is a primitive or constructor, the definition will be invoked,
752/// possibly arguments will be parsed from the Gullet.
753/// Otherwise, the token is simply digested: turned into an appropriate box.
754/// Returns a list of boxes/whatsits.
755pub fn invoke_token<'a>(input_token: &'a Token) -> Result<Vec<Digested>> {
756  let mut maybe_token: Option<Cow<'a, Token>> = Some(Cow::Borrowed(input_token));
757  // Overly complex, but want to avoid recursion/stack
758  let mut result: Vec<Digested> = Vec::new();
759  // INVOKE:
760  while maybe_token.is_some() {
761    // TODO: This is silly, switch to an owned input_token (it is Copy as of recently).
762    let token = maybe_token.take().unwrap().into_owned();
763    // info!(target:"invoke_token", "{:?}", token);
764    // RAII guard: auto-pops current_token on scope exit (even on early return/panic)
765    let _token_guard = local_current_token_guard(token);
766    {
767      stomach_mut!().token_stack.push(token);
768    }
769    if { stomach!().token_stack.len() } > MAXSTACK {
770      fatal!(
771        Stomach,
772        Recursion,
773        s!(
774          "Excessive recursion(?): Tokens on stack: {:?}",
775          stomach!().token_stack
776        )
777      );
778    }
779    result = Vec::new();
780
781    // Rust notes: It would be ideal if we could unify the cases for (Primtive, Constructor,
782    // MathPrimitive), as well as (Expandable, Conditional) since the
783    // API is identical. However, as the types are different, Rust
784    // constrains us here, we need separate match arms for each
785    // distinctly typed enum case.
786    let digestable_def = lookup_digestable_definition(&token);
787    match digestable_def {
788      None | Some(Stored::None) => {
789        result = invoke_token_undefined(&token)?;
790      },
791      Some(Stored::Token(meaning)) => {
792        // Common case
793        let cc = meaning.get_catcode();
794        if cc == Catcode::CS {
795          result = invoke_token_undefined(&token)?;
796        } else if cc.is_absorbable() {
797          if let Some(digested) = invoke_token_simple(meaning)? {
798            result.push(digested);
799          }
800        } else {
801          // Perl L187-189: deactivate T_ALIGN to prevent error flood in tables
802          if token.get_catcode() == Catcode::ALIGN {
803            if let Some(relax_meaning) = lookup_meaning(&T_CS!("\\relax")) {
804              assign_meaning(&token, relax_meaning, Some(Scope::Local));
805            }
806          }
807          let message = s!(
808            "The token {:?} (catcode {:?}) should never reach Stomach!",
809            token,
810            cc
811          );
812          Error!("misdefined", token, &message);
813          if let Some(digested) = invoke_token_simple(meaning)? {
814            result.push(digested);
815          }
816        }
817      },
818      Some(Stored::Expandable(meaning)) => {
819        // A math-active character will (typically) be a macro,
820        // but it isn't expanded in the gullet, but later when digesting, in math mode
821        // (? I think)
822        let invoked_meaning = meaning.invoke(false)?;
823        if !invoked_meaning.is_empty() {
824          {
825            gullet::unread(invoked_meaning);
826          }
827        }
828        // replace the token by it's expansion!!!
829        maybe_token = gullet::read_x_token(None, false, None)?.map(Cow::Owned);
830        {
831          stomach_mut!().token_stack.pop();
832        }
833        drop(_token_guard); // expire current token via RAII
834        continue;
835      },
836      Some(Stored::Conditional(meaning)) => {
837        // Conditionals are "expandable", use the regular invoke.
838        let invoked_meaning = meaning.invoke(false)?;
839        gullet::unread(invoked_meaning);
840        maybe_token = gullet::read_x_token(None, false, None)?.map(Cow::Owned);
841        {
842          stomach_mut!().token_stack.pop();
843        }
844        drop(_token_guard); // expire current token via RAII
845        continue;
846      },
847      Some(Stored::Constructor(meaning)) => {
848        // Otherwise, a normal primitive or constructor
849        result = meaning.invoke_primitive()?;
850        if !meaning.is_prefix() {
851          clear_prefixes(); // Clear prefixes unless we just set one.
852        }
853      },
854      Some(Stored::Primitive(meaning)) => {
855        // Otherwise, a normal primitive or constructor
856        result = meaning.invoke_primitive()?;
857        if !meaning.is_prefix() {
858          clear_prefixes(); // Clear prefixes unless we just set one.
859        }
860      },
861      Some(Stored::MathPrimitive(meaning)) => {
862        // Copy of regular Primitive
863        // Otherwise, a normal primitive or constructor
864        result = meaning.invoke_primitive()?;
865        if !meaning.is_prefix() {
866          clear_prefixes(); // Clear prefixes unless we just set one.
867        }
868      },
869      Some(Stored::Register(meaning)) => {
870        // Registers are special primitives
871        result = meaning.invoke_primitive()?;
872        if !meaning.is_prefix() {
873          clear_prefixes(); // Clear prefixes unless we just set one.
874        }
875      },
876      meaning => {
877        // Perl: Error + makeMisdefinedError (non-fatal). Don't crash.
878        Error!("misdefined", token,
879          s!("Unexpected object in Stomach: {:?}", meaning));
880      },
881    }
882    // _token_guard drops here, auto-expiring current token
883    break;
884  }
885  stomach_mut!().token_stack.pop();
886  Ok(result)
887}

889fn invoke_token_undefined(token: &Token) -> Result<Vec<Digested>> {
890  let cs = token.with_cs_name(|cs| String::from(cs));
891  note_status(LogStatus::Undefined, Some(&cs));
892
893  // To minimize chatter, go ahead and define it...
894  if cs.starts_with("\\if") {
895    // Apparently an \ifsomething ???
896    let name = cs.replace("\\if", "");
897    let message = s!("The token {} is not defined.", token.stringify());
898    Error!(
899      "undefined",
900      token,
901      &message,
902      "Defining it now as with \\newif"
903    );
904    // install stub definitions for new conditional
905    install_definition(
906      Expandable::new(
907        T_CS!(s!("\\{}true", name)),
908        None,
909        Tokens!(T_CS!("\\let"), T_CS!(&cs), T_CS!("\\iftrue")).into(),
910        None,
911      )?,
912      None,
913    );
914    install_definition(
915      Expandable::new(
916        T_CS!(s!("\\{}false", name)),
917        None,
918        Tokens!(T_CS!("\\let"), T_CS!(cs), T_CS!("\\iffalse")).into(),
919        None,
920      )?,
921      None,
922    );
923
924    let_i(token, &T_CS!("\\iffalse"), None);
925    gullet::unread_one(*token); // Retry
926    Ok(Vec::new())
927  } else {
928    let message = s!("The token {} is not defined.", token.stringify());
929    Error!(
930      "undefined",
931      token,
932      &message,
933      "Defining it now as <ltx:ERROR/>"
934    );
935    install_definition(
936      Constructor {
937        cs: *token,
938        paramlist: None,
939        replacement: Some(Rc::new(move |document, _args, _props| {
940          document.make_error("undefined", &cs)
941        })),
942        ..Constructor::default()
943      },
944      Some(Scope::Global),
945    );
946    // Perl: unread the token and return empty, so the outer loop re-reads
947    // and dispatches through the normal path (with the newly installed stub).
948    // This ensures gullet-level side effects (filtering, expansion) are applied.
949    gullet::unread_one(*token);
950    Ok(Vec::new())
951  }
952}

954fn invoke_token_simple(meaning: Token) -> Result<Option<Digested>> {
955  let cc = meaning.get_catcode();
956  let font = lookup_font();
957  match cc {
958    Catcode::SPACE => {
959      clear_prefixes(); // Perl Stomach.pm line 234: prefixes shouldn't apply here.
960      // Perl: if($STATE->lookupValue('MODE') =~ /(?:math|vertical)$/) { return (); }
961      let mode = lookup_string("MODE");
962      if mode.ends_with("math") || mode.ends_with("vertical") {
963        Ok(None)
964      } else {
965        enter_horizontal();
966        Ok(Some(Digested::from(Tbox::new(
967          meaning.get_sym(),
968          font,
969          None,
970          Tokens!(meaning),
971          HashMap::default(),
972        ))))
973      }
974    },
975    Catcode::COMMENT => {
976      // Perl Stomach.pm lines 241-244: decode comment via font encoding
977      let decoded = font::decode_string(meaning.get_sym(), None, true);
978      let comment = arena::with(decoded, |s| {
979        // However, spaces normally would have be digested away as positioning...
980        // Replace NBSP + combining strikethrough (OT1 space position) with actual space
981        s.replace("\u{00A0}\u{0335}", " ")
982      });
983      // Perl: returns LaTeXML::Core::Comment->new($comment)
984      // which gets absorbed as an XML comment node via Document::insertComment
985      Ok(Some(Digested::from(Comment(comment))))
986    },
987    _ => {
988      clear_prefixes(); // Perl Stomach.pm line 247: prefixes shouldn't apply here.
989      // Perl: check mathcode for IN_MATH characters (Stomach.pm lines 248-251)
990      // In Perl, all math chars go through decodeMathChar which decodes via
991      // the font encoding. In Rust, Tbox::new already handles IN_MATH:
992      // it sets mode="math", looks up math_token_attributes for role/meaning/name,
993      // and specializes the font. This produces the correct LaTeXML-level properties.
994      // The mathchar parsing handles non-ASCII chars needing font map lookup.
995      // TODO: Use for chars where font-encoding glyph differs from input.
996      // Perl L248-257: if IN_MATH && mathcode → decodeMathChar (math box)
997      // else → enterHorizontal + text box (covers non-math AND math-but-no-mathcode)
998      if lookup_bool("IN_MATH") {
999        if let Some(mathcode) = lookup_mathcode_sym(&meaning.get_sym()) {
1000          return crate::common::mathchar::decode_math_char_for_stomach(mathcode, meaning);
1001        }
1002      }
1003      // Fallthrough: either not in math, or in math but no mathcode
1004      enter_horizontal();
1005      let text = font::decode_string(meaning.get_sym(), None, true);
1006      Ok(Some(Digested::from(Tbox::new(
1007        text,
1008        None,
1009        None,
1010        Tokens!(meaning),   // tokens
1011        HashMap::default(), // properties
1012      ))))
1013    },
1014  }
1015}

1026/// ScriptLevel is similar to boxing level, but relative to current Math mode's level
