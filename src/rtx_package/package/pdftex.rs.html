     Ordering::Greater => Tokens!(T_OTHER!("1")),
     Ordering::Equal => Tokens!(T_OTHER!("0")),
     Ordering::Less => Tokens!(T_OTHER!("-"), T_OTHER!("1"))
    }
  });
