1//! Document-subtree extraction helpers — port of `LaTeXML::Util::Pack`'s
2//! `get_math` and `get_embeddable` (Perl Pack.pm L247-313).
3//!
4//! These are the implementations behind the `--whatsout fragment` and
5//! `--whatsout math` CLI modes. They run AFTER all post-processing on
6//! the final XML/HTML document and return the subtree the user actually
7//! wanted (an embeddable inline snippet, or just the math).
8//!
9//! Companion modules:
10//! * [`crate::pack`] bundles the chosen output into a zip archive.
11//! * [`crate::writer`] serializes it to a file or stdout.
12//!
13//! ## What's not ported (intentional gaps)
14//!
15//! Perl `get_math` falls through to inline an SVG when the math node's
16//! `imagesrc` ends in `.svg` — for our pipeline the SVG is already a
17//! referenced resource bundled by `pack_archive`, so the inline path
18//! has no caller. Doc string flags the gap so a future visitor knows
19//! it's intentional.
20//!
21//! Perl `get_embeddable` copies namespace declarations and RDFa
22//! attributes from the document root onto the extracted node. libxml-rs
23//! exposes `Node::set_attribute` for the RDFa half cheaply; namespace
24//! re-binding requires FFI into `xmlSetNs` and is deferred (the
25//! extracted subtree usually inherits its namespaces fine as long as
26//! the consumer doesn't re-parse it standalone).
27

32/// Output extraction mode — port of Perl `LaTeXML::Util::Pack`'s
33/// `whatsout` option (Pack.pm L323-345). Selects which subtree of the
34/// post-processed document to serialize and ship to the user.
35///
36/// * [`Whatsout::Document`] — full document, no extraction (default).
37/// * [`Whatsout::Fragment`] — embeddable HTML snippet via
38///   [`get_embeddable`].
39/// * [`Whatsout::Math`] — math subtree (or fallback) via [`get_math`].
40#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
41pub enum Whatsout {
42  #[default]
43  Document,
44  Fragment,
45  Math,
46}
47
48impl Whatsout {
49  /// Parse a CLI string into the matching variant. Returns `None` for
50  /// unrecognized values; callers typically fall back to `Document`.
51  /// Mirrors Perl `pack_collection`'s string-tag dispatch — accepts
52  /// `archive` for backward-compat but maps it to `Document` (archive
53  /// bundling is the `pack::pack_archive` concern, not extraction).
54  pub fn from_cli(s: &str) -> Option<Self> {
55    match s {
56      "document" | "archive" => Some(Whatsout::Document),
57      "fragment" => Some(Whatsout::Fragment),
58      "math" => Some(Whatsout::Math),
59      _ => None,
60    }
61  }
62}
63
64/// Apply the requested [`Whatsout`] extraction to `doc` and return the
65/// serialized subtree. Returns the full document for
66/// [`Whatsout::Document`] (the default no-op) or when extraction finds
67/// no candidate node.
68///
69/// Wraps the two `get_*` helpers + libxml `node_to_string` so callers
70/// don't have to thread the inner [`libxml::tree::Document`] through.
71pub fn serialize_whatsout(doc: &PostDocument, mode: Whatsout) -> String {
72  match mode {
73    Whatsout::Document => doc.to_xml_string(),
74    Whatsout::Fragment => get_embeddable(doc)
75      .map(|n| doc.get_document().node_to_string(&n))
76      .unwrap_or_else(|| doc.to_xml_string()),
77    Whatsout::Math => get_math(doc)
78      .map(|n| doc.get_document().node_to_string(&n))
79      .unwrap_or_else(|| doc.to_xml_string()),
80  }
81}

83/// XPath that matches the `<math>` (HTML5 / MathML) and `<Math>`
84/// (legacy LaTeXML pre-MathML) elements anywhere in the document,
85/// regardless of namespace prefix.
86const MATH_XPATH: &str = "//*[local-name()='math' or local-name()='Math']";
87
88/// XPath for math-as-image fallback: `<img class="ltx_Math …">`.
89const MATH_IMG_XPATH: &str = "//*[local-name()='img' and contains(@class,'ltx_Math')]";
90
91/// XPath for embeddable fragment root: `<div class="ltx_document">`.
92/// Perl literally uses `contains(@class, "ltx_document")` so any
93/// element whose `class` includes the substring matches.
94const EMBEDDABLE_XPATH: &str = "//*[contains(@class,'ltx_document')]";
95
96/// RDFa attributes that Perl `get_embeddable` copies from the document
97/// root onto the extracted node (Perl Pack.pm L309).
98const RDFA_ATTRS: &[&str] = &[
99  "prefix", "property", "content", "resource", "about", "typeof", "rel", "rev", "datatype",
100];
101
102/// Class-name pattern that a single-child `<div>` must match for the
103/// unwrap loop to descend into it. Perl regex `/^ltx_(page_(main|content)|document|para|header)$/`.
104fn is_unwrappable_div_class(class: &str) -> bool {
105  matches!(
106    class,
107    "ltx_page_main"
108      | "ltx_page_content"
109      | "ltx_document"
110      | "ltx_para"
111      | "ltx_header"
112  )
113}

115/// Local-name predicate for the "this <p> is purely inline content"
116/// check in `get_embeddable`: child names matching `math|text|span`.
117fn is_inline_child(name: &str) -> bool {
118  // Perl uses `=~ /math|text|span/` (substring match anywhere in the
119  // node name). Match the substring semantics exactly.
120  name.contains("math") || name.contains("text") || name.contains("span")
121}

123/// Extract the math subtree(s) from a post-processed document, mirroring
124/// Perl `LaTeXML::Util::Pack::get_math` (Pack.pm L247-280).
125///
126/// * If exactly one `<math>` (or `<Math>`) is present, return it.
127/// * If multiple, return their **least common ancestor** by walking
128///   up from the first one until its descendant-count matches the
129///   document's total. Unwraps trailing `<tr>` / `<td>` so the LCA
130///   isn't a table cell.
131/// * If no math nodes at all, fall through to a math-image (`<img
132///   class="ltx_Math">`) XPath; if that's also empty, fall through
133///   to [`get_embeddable`] so callers get a useful node either way.
134pub fn get_math(doc: &PostDocument) -> Option<Node> {
135  let math_nodes = doc.findnodes(MATH_XPATH);
136  let math_count = math_nodes.len();
137
138  if math_count == 0 {
139    let img_nodes = doc.findnodes(MATH_IMG_XPATH);
140    if img_nodes.is_empty() {
141      return get_embeddable(doc);
142    }
143    return img_nodes.into_iter().next();
144  }
145
146  let mut math = math_nodes.into_iter().next()?;
147  if math_count > 1 {
148    // Walk up until the subtree under `math` contains every math node.
149    // Perl re-runs the same XPath relative to the current candidate
150    // and adds 1 when the candidate itself is a math node. We use
151    // `findnodes_at` (libxml's `node_evaluate`) for context-scoped
152    // XPath — `findnodes_foreign` falls back to a manual traverser
153    // that doesn't support `local-name()` predicates.
154    let descendant_math_xpath = format!(".{MATH_XPATH}");
155    let mut found = 0;
156    while found != math_count {
157      found = doc.findnodes_at(&descendant_math_xpath, Some(&math)).len();
158      if math.get_name().eq_ignore_ascii_case("math") {
159        found += 1;
160      }
161      if found != math_count {
162        match math.get_parent() {
163          Some(p) => math = p,
164          None => break,
165        }
166      }
167    }
168    // Don't anchor on a table cell — climb out of `<tr>` / `<td>`
169    // (Perl `while ($math->nodeName =~ '^t[rd]$')`).
170    while is_table_row_or_cell(&math.get_name()) {
171      match math.get_parent() {
172        Some(p) => math = p,
173        None => break,
174      }
175    }
176  }
177
178  Some(math)
179}

181/// Extract an embeddable HTML fragment from a post-processed document,
182/// mirroring Perl `LaTeXML::Util::Pack::get_embeddable` (Pack.pm L282-313).
183///
184/// * Find the first `<div class="ltx_document">`.
185/// * Unwrap as long as the current node is a `<div>` with exactly one
186///   child, an unwrappable wrapper class (`ltx_page_main`,
187///   `ltx_page_content`, `ltx_document`, `ltx_para`, `ltx_header`),
188///   and no inline `style` attribute.
189/// * If the resulting node is a `<p>` whose every child is inline-
190///   compatible (local-name contains `math` / `text` / `span`), rename
191///   it to `<span class="text">` so it stays inline-embeddable.
192/// * Copy RDFa attributes (`prefix`, `property`, `content`, `resource`,
193///   `about`, `typeof`, `rel`, `rev`, `datatype`) from the document
194///   root onto the extracted node so the snippet retains its semantic
195///   annotations.
196/// * Namespace declarations from the root are NOT propagated (libxml-rs
197///   gap — see module docs).
198///
199/// If no `ltx_document` element is found, returns the document root,
200/// matching Perl's `return $embeddable || $doc`.
201pub fn get_embeddable(doc: &PostDocument) -> Option<Node> {
202  let root = doc.get_document_element()?;
203  let mut embeddable = doc.findnodes(EMBEDDABLE_XPATH).into_iter().next().unwrap_or_else(|| root.clone());
204
205  // Unwrap nested single-child div wrappers.
206  loop {
207    if embeddable.get_name() != "div" {
208      break;
209    }
210    let children = embeddable.get_child_nodes();
211    if children.len() != 1 {
212      break;
213    }
214    let class = embeddable.get_attribute("class").unwrap_or_default();
215    if !is_unwrappable_div_class(&class) {
216      break;
217    }
218    if embeddable.get_attribute("style").is_some() {
219      break;
220    }
221    match embeddable.get_first_child() {
222      Some(c) => embeddable = c,
223      None => break,
224    }
225  }
226
227  // `<p>` with all-inline children → rename to `<span class="text">`.
228  if embeddable.get_name() == "p" {
229    let children = embeddable.get_child_nodes();
230    if !children.is_empty() && children.iter().all(|c| is_inline_child(&c.get_name())) {
231      let _ = embeddable.set_name("span");
232      let _ = embeddable.set_attribute("class", "text");
233    }
234  }
235
236  // Copy RDFa attributes from doc root.
237  for attr in RDFA_ATTRS {
238    if let Some(value) = root.get_attribute(attr) {
239      let _ = embeddable.set_attribute(attr, &value);
240    }
241  }
242
243  Some(embeddable)
244}

246fn is_table_row_or_cell(name: &str) -> bool {
247  matches!(name, "tr" | "td" | "TR" | "TD")
248}

273  #[test]
274  fn get_embeddable_unwraps_single_child_wrappers_to_inline_span() {
275    // ltx_document (1 child div) → ltx_page_main (1 child p) → p
276    // (terminal: not in unwrap-class allowlist). The <p>'s single
277    // child is the text "Hello world" whose node-name is "text"
278    // (matches inline pattern) → final promotion to span.
279    let xml = r#"<html><body><div class="ltx_document"><div class="ltx_page_main"><p>Hello world</p></div></div></body></html>"#;
280    let d = doc(xml);
281    let node = get_embeddable(&d).expect("some node");
282    assert_eq!(node.get_name(), "span");
283    assert_eq!(node.get_attribute("class").as_deref(), Some("text"));
284  }
