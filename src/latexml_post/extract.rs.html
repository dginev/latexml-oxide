1//! Document-subtree extraction helpers — port of `LaTeXML::Util::Pack`'s
2//! `get_math` and `get_embeddable` (Perl Pack.pm L247-313).
3//!
4//! These are the implementations behind the `--whatsout fragment` and
5//! `--whatsout math` CLI modes. They run AFTER all post-processing on
6//! the final XML/HTML document and return the subtree the user actually
7//! wanted (an embeddable inline snippet, or just the math). The
8//! `--whatsout archive` mode keeps the full document (no extraction) but
9//! flags the caller to wrap it into a zip — see [`Whatsout::is_archive`]
10//! and [`crate::pack::pack_archive`].
11//!
12//! Companion modules:
13//! * [`crate::pack`] bundles the chosen output into a zip archive.
14//! * [`crate::writer`] serializes it to a file or stdout.
15//!
16//! ## What's not ported (intentional gaps)
17//!
18//! Perl `get_math` falls through to inline an SVG when the math node's
19//! `imagesrc` ends in `.svg` — for our pipeline the SVG is already a
20//! referenced resource bundled by `pack_archive`, so the inline path
21//! has no caller. Doc string flags the gap so a future visitor knows
22//! it's intentional.
23//!
24//! Perl `get_embeddable` copies namespace declarations and RDFa
25//! attributes from the document root onto the extracted node. libxml-rs
26//! exposes `Node::set_attribute` for the RDFa half cheaply; namespace
27//! re-binding requires FFI into `xmlSetNs` and is deferred (the
28//! extracted subtree usually inherits its namespaces fine as long as
29//! the consumer doesn't re-parse it standalone).
30

35/// Output extraction mode — port of Perl `LaTeXML::Util::Pack`'s
36/// `whatsout` option (Pack.pm L320-345). Selects which subtree of the
37/// post-processed document to serialize and ship to the user.
38///
39/// * [`Whatsout::Document`] — full document, no extraction (default).
40/// * [`Whatsout::Fragment`] — embeddable HTML snippet via [`get_embeddable`].
41/// * [`Whatsout::Math`] — math subtree (or fallback) via [`get_math`].
42/// * [`Whatsout::Archive`] — full document, but bundled into a zip by [`crate::pack::pack_archive`]
43///   (Pack.pm L326-331 + `Pack/Zip.pm::get_archive`). Extraction is a no-op; the archiving happens
44///   in the binary's output stage.
45#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
46pub enum Whatsout {
47  #[default]
48  Document,
49  Fragment,
50  Math,
51  Archive,
52}
53
54impl Whatsout {
55  /// Parse a CLI string into the matching variant. Returns `None` for
56  /// unrecognized values; callers typically fall back to `Document`.
57  /// Mirrors Perl `pack_collection`'s string-tag dispatch. Perl matches
58  /// archive with `/^archive/`, so the `archive::zip` /
59  /// `archive::zip::perl` long forms (Pack/Zip.pm L118-122) also map to
60  /// [`Whatsout::Archive`].
61  pub fn from_cli(s: &str) -> Option<Self> {
62    match s {
63      "document" => Some(Whatsout::Document),
64      "fragment" => Some(Whatsout::Fragment),
65      "math" => Some(Whatsout::Math),
66      _ if s.starts_with("archive") => Some(Whatsout::Archive),
67      _ => None,
68    }
69  }
70
71  /// Whether this mode bundles the output into a zip archive (Perl
72  /// `pack_collection` `whatsout =~ /^archive/`). The binary's output
73  /// stage branches on this to call [`crate::pack::pack_archive`].
74  pub fn is_archive(self) -> bool { matches!(self, Whatsout::Archive) }
75
76  /// Whether selecting this mode forces post-processing. Perl
77  /// `Config.pm` L454: any `whatsout` other than `document` implies
78  /// `post = 1` (fragment/math need the extraction stage; archive needs
79  /// the full post-processed document to bundle).
80  pub fn requires_post(self) -> bool { !matches!(self, Whatsout::Document) }
81}

83/// Apply the requested [`Whatsout`] extraction to `doc` and return the
84/// serialized subtree. Returns the full document for
85/// [`Whatsout::Document`] (the default no-op) or when extraction finds
86/// no candidate node.
87///
88/// Wraps the two `get_*` helpers + libxml `node_to_string` so callers
89/// don't have to thread the inner [`libxml::tree::Document`] through.
90pub fn serialize_whatsout(doc: &PostDocument, mode: Whatsout) -> String {
91  match mode {
92    // Archive ships the FULL document; the zip wrapping is a separate
93    // output-stage concern (`pack::pack_archive`), not extraction.
94    Whatsout::Document | Whatsout::Archive => doc.to_xml_string(),
95    Whatsout::Fragment => get_embeddable(doc)
96      .map(|n| doc.get_document().node_to_string(&n))
97      .unwrap_or_else(|| doc.to_xml_string()),
98    Whatsout::Math => get_math(doc)
99      .map(|n| doc.get_document().node_to_string(&n))
100      .unwrap_or_else(|| doc.to_xml_string()),
101  }
102}

104/// XPath that matches the `<math>` (HTML5 / MathML) and `<Math>`
105/// (legacy LaTeXML pre-MathML) elements anywhere in the document,
106/// regardless of namespace prefix.
107const MATH_XPATH: &str = "//*[local-name()='math' or local-name()='Math']";
108
109/// XPath for math-as-image fallback: `<img class="ltx_Math …">`.
110const MATH_IMG_XPATH: &str = "//*[local-name()='img' and contains(@class,'ltx_Math')]";
111
112/// XPath for embeddable fragment root: `<div class="ltx_document">`.
113/// Perl literally uses `contains(@class, "ltx_document")` so any
114/// element whose `class` includes the substring matches.
115const EMBEDDABLE_XPATH: &str = "//*[contains(@class,'ltx_document')]";
116
117/// RDFa attributes that Perl `get_embeddable` copies from the document
118/// root onto the extracted node (Perl Pack.pm L309).
119const RDFA_ATTRS: &[&str] = &[
120  "prefix", "property", "content", "resource", "about", "typeof", "rel", "rev", "datatype",
121];
122
123/// Class-name pattern that a single-child `<div>` must match for the
124/// unwrap loop to descend into it. Perl regex `/^ltx_(page_(main|content)|document|para|header)$/`.
125fn is_unwrappable_div_class(class: &str) -> bool {
126  matches!(
127    class,
128    "ltx_page_main" | "ltx_page_content" | "ltx_document" | "ltx_para" | "ltx_header"
129  )
130}

132/// Local-name predicate for the "this <p> is purely inline content"
133/// check in `get_embeddable`: child names matching `math|text|span`.
134fn is_inline_child(name: &str) -> bool {
135  // Perl uses `=~ /math|text|span/` (substring match anywhere in the
136  // node name). Match the substring semantics exactly.
137  name.contains("math") || name.contains("text") || name.contains("span")
138}

140/// Extract the math subtree(s) from a post-processed document, mirroring
141/// Perl `LaTeXML::Util::Pack::get_math` (Pack.pm L247-280).
142///
143/// * If exactly one `<math>` (or `<Math>`) is present, return it.
144/// * If multiple, return their **least common ancestor** by walking up from the first one until its
145///   descendant-count matches the document's total. Unwraps trailing `<tr>` / `<td>` so the LCA
146///   isn't a table cell.
147/// * If no math nodes at all, fall through to a math-image (`<img class="ltx_Math">`) XPath; if
148///   that's also empty, fall through to [`get_embeddable`] so callers get a useful node either way.
149pub fn get_math(doc: &PostDocument) -> Option<Node> {
150  let math_nodes = doc.findnodes(MATH_XPATH);
151  let math_count = math_nodes.len();
152
153  if math_count == 0 {
154    let img_nodes = doc.findnodes(MATH_IMG_XPATH);
155    if img_nodes.is_empty() {
156      return get_embeddable(doc);
157    }
158    return img_nodes.into_iter().next();
159  }
160
161  let mut math = math_nodes.into_iter().next()?;
162  if math_count > 1 {
163    // Walk up until the subtree under `math` contains every math node.
164    // Perl re-runs the same XPath relative to the current candidate
165    // and adds 1 when the candidate itself is a math node. We use
166    // `findnodes_at` (libxml's `node_evaluate`) for context-scoped
167    // XPath — `findnodes_foreign` falls back to a manual traverser
168    // that doesn't support `local-name()` predicates.
169    let descendant_math_xpath = format!(".{MATH_XPATH}");
170    let mut found = 0;
171    while found != math_count {
172      found = doc.findnodes_at(&descendant_math_xpath, Some(&math)).len();
173      if math.get_name().eq_ignore_ascii_case("math") {
174        found += 1;
175      }
176      if found != math_count {
177        match math.get_parent() {
178          Some(p) => math = p,
179          None => break,
180        }
181      }
182    }
183    // Don't anchor on a table cell — climb out of `<tr>` / `<td>`
184    // (Perl `while ($math->nodeName =~ '^t[rd]$')`).
185    while is_table_row_or_cell(&math.get_name()) {
186      match math.get_parent() {
187        Some(p) => math = p,
188        None => break,
189      }
190    }
191  }
192
193  Some(math)
194}

196/// Extract an embeddable HTML fragment from a post-processed document,
197/// mirroring Perl `LaTeXML::Util::Pack::get_embeddable` (Pack.pm L282-313).
198///
199/// * Find the first `<div class="ltx_document">`.
200/// * Unwrap as long as the current node is a `<div>` with exactly one child, an unwrappable wrapper
201///   class (`ltx_page_main`, `ltx_page_content`, `ltx_document`, `ltx_para`, `ltx_header`), and no
202///   inline `style` attribute.
203/// * If the resulting node is a `<p>` whose every child is inline- compatible (local-name contains
204///   `math` / `text` / `span`), rename it to `<span class="text">` so it stays inline-embeddable.
205/// * Copy RDFa attributes (`prefix`, `property`, `content`, `resource`, `about`, `typeof`, `rel`,
206///   `rev`, `datatype`) from the document root onto the extracted node so the snippet retains its
207///   semantic annotations.
208/// * Namespace declarations from the root are NOT propagated (libxml-rs gap — see module docs).
209///
210/// If no `ltx_document` element is found, returns the document root,
211/// matching Perl's `return $embeddable || $doc`.
212pub fn get_embeddable(doc: &PostDocument) -> Option<Node> {
213  let root = doc.get_document_element()?;
214  let mut embeddable = doc
215    .findnodes(EMBEDDABLE_XPATH)
216    .into_iter()
217    .next()
218    .unwrap_or_else(|| root.clone());
219
220  // Unwrap nested single-child div wrappers.
221  loop {
222    if embeddable.get_name() != "div" {
223      break;
224    }
225    let children = embeddable.get_child_nodes();
226    if children.len() != 1 {
227      break;
228    }
229    let class = embeddable.get_attribute("class").unwrap_or_default();
230    if !is_unwrappable_div_class(&class) {
231      break;
232    }
233    if embeddable.get_attribute("style").is_some() {
234      break;
235    }
236    match embeddable.get_first_child() {
237      Some(c) => embeddable = c,
238      None => break,
239    }
240  }
241
242  // `<p>` with all-inline children → rename to `<span class="text">`.
243  if embeddable.get_name() == "p" {
244    let children = embeddable.get_child_nodes();
245    if !children.is_empty() && children.iter().all(|c| is_inline_child(&c.get_name())) {
246      let _ = embeddable.set_name("span");
247      let _ = embeddable.set_attribute("class", "text");
248    }
249  }
250
251  // Copy RDFa attributes from doc root.
252  for attr in RDFA_ATTRS {
253    if let Some(value) = root.get_attribute(attr) {
254      let _ = embeddable.set_attribute(attr, &value);
255    }
256  }
257
258  Some(embeddable)
259}
