14const OM_URI: &str = "http://www.openmath.org/OpenMath";
15const OM_MIMETYPE: &str = "application/openmath+xml";

17/// OpenMath converter table entry.

20/// OpenMath post-processor.

61impl MathProcessor for OpenMath {
62  fn convert_node(&self, doc: &PostDocument, xmath: &Node) -> Option<MathConversion> {
63    let children = element_children(xmath);
64    let xml = if children.len() == 1 {
65      om_expr(doc, &children[0])
66    } else {
67      om_unparsed(doc, &children)
68    };
69
70    Some(MathConversion {
71      processor_name: self.name.clone(),
72      mimetype: Some(OM_MIMETYPE.to_string()),
73      xml: Some(xml),
74      string: None,
75      src: None,
76      width: None,
77      height: None,
78      depth: None,
79    })
80  }
81
82  fn combine_parallel(
83    &self,
84    _doc: &PostDocument,
85    _xmath: &Node,
86    primary: MathConversion,
87    secondaries: Vec<MathConversion>,
88  ) -> MathConversion {
89    let mut attr_children = Vec::new();
90
91    for secondary in &secondaries {
92      let mimetype = secondary.mimetype.as_deref().unwrap_or("unknown");
93      attr_children.push(NodeData::Element {
94        tag: "om:OMS".to_string(),
95        attributes: Some(HashMap::from([
96          ("cd".to_string(), "Alternate".to_string()),
97          ("name".to_string(), mimetype.to_string()),
98        ])),
99        children: vec![],
100      });
101
102      if mimetype == OM_MIMETYPE {
103        if let Some(ref xml) = secondary.xml {
104          attr_children.push(xml.clone());
105        }
106      } else if let Some(ref xml) = secondary.xml {
107        attr_children.push(NodeData::Element {
108          tag: "om:OMFOREIGN".to_string(),
109          attributes: None,
110          children: vec![xml.clone()],
111        });
112      } else if let Some(ref string) = secondary.string {
113        attr_children.push(NodeData::Element {
114          tag: "om:OMSTR".to_string(),
115          attributes: None,
116          children: vec![NodeData::Text(string.clone())],
117        });
118      }
119    }
120
121    if let Some(ref xml) = primary.xml {
122      attr_children.push(xml.clone());
123    }
124
125    MathConversion {
126      processor_name: self.name.clone(),
127      mimetype: Some(OM_MIMETYPE.to_string()),
128      xml: Some(NodeData::Element {
129        tag: "om:OMATTR".to_string(),
130        attributes: None,
131        children: attr_children,
132      }),
133      string: None,
134      src: None,
135      width: None,
136      height: None,
137      depth: None,
138    }
139  }
140
141  fn outer_wrapper(&self, _doc: &PostDocument, _xmath: &Node, conversion: NodeData) -> NodeData {
142    NodeData::Element {
143      tag: "om:OMOBJ".to_string(),
144      attributes: None,
145      children: vec![conversion],
146    }
147  }
148
149  fn raw_id_suffix(&self) -> &str {
150    ".om"
151  }
152
153  fn is_secondary(&self) -> bool {
154    self.is_secondary
155  }
156
157  fn can_convert(&self, _doc: &PostDocument, math: &Node) -> bool {
158    math_is_parsed(math)
159  }
160
161  fn preprocess(&self, _doc: &PostDocument, _nodes: &[Node]) {
162    // Register om namespace (would need &mut doc)
163    log::trace!("OpenMath: would register om namespace");
164  }

168// OpenMath expression conversion
169
170/// Convert an XMath element node to OpenMath.
171///
172/// Port of `om_expr`.
173fn om_expr(doc: &PostDocument, node: &Node) -> NodeData {
174  // Realize XMRef nodes
175  let real_node = if doc.get_qname(node).as_deref() == Some("ltx:XMRef") {
176    if let Some(idref) = node.get_attribute("idref") {
177      doc.find_node_by_id(&idref).cloned()
178    } else {
179      None
180    }
181  } else {
182    Some(node.clone())
183  };
184
185  match real_node {
186    Some(ref n) => om_expr_aux(doc, n),
187    None => om_error("Missing Subexpression"),
188  }
189}
190
191/// Core OpenMath expression conversion.
192///
193/// Port of `om_expr_aux`.
194fn om_expr_aux(doc: &PostDocument, node: &Node) -> NodeData {
195  let tag = match doc.get_qname(node) {
196    Some(t) => t,
197    None => return om_error("Missing Subexpression"),
198  };
199
200  match tag.as_str() {
201    "ltx:XMWrap" | "ltx:XMArg" => {
202      let children = element_children(node);
203      if children.len() == 1 {
204        om_expr(doc, &children[0])
205      } else {
206        om_unparsed(doc, &children)
207      }
208    }
209    "ltx:XMDual" => {
210      let children = element_children(node);
211      if !children.is_empty() {
212        om_expr(doc, &children[0]) // Content branch
213      } else {
214        om_error("Empty XMDual")
215      }
216    }
217    "ltx:XMApp" => {
218      let children = element_children(node);
219      if children.is_empty() {
220        return om_error("Missing Operator");
221      }
222      // Generic application
223      let mut oma_children = Vec::new();
224      for child in &children {
225        oma_children.push(om_expr(doc, child));
226      }
227      NodeData::Element {
228        tag: "om:OMA".to_string(),
229        attributes: None,
230        children: oma_children,
231      }
232    }
233    "ltx:XMTok" => {
234      if let Some(meaning) = node.get_attribute("meaning") {
235        let cd = node.get_attribute("omcd").unwrap_or_else(|| "latexml".to_string());
236        NodeData::Element {
237          tag: "om:OMS".to_string(),
238          attributes: Some(HashMap::from([
239            ("name".to_string(), meaning),
240            ("cd".to_string(), cd),
241          ])),
242          children: vec![],
243        }
244      } else {
245        // Variable
246        let name = node.get_content();
247        let name = if name.trim().is_empty() {
248          node.get_attribute("name").unwrap_or_else(|| "?".to_string())
249        } else {
250          name
251        };
252        NodeData::Element {
253          tag: "om:OMV".to_string(),
254          attributes: Some(HashMap::from([("name".to_string(), name)])),
255          children: vec![],
256        }
257      }
258    }
259    "ltx:XMHint" => {
260      // Hints are ignored in OpenMath
261      NodeData::Text(String::new())
262    }
263    "ltx:XMText" => {
264      let text = node.get_content();
265      NodeData::Element {
266        tag: "om:OMSTR".to_string(),
267        attributes: None,
268        children: vec![NodeData::Text(text)],
269      }
270    }
271    _ => {
272      let text = node.get_content();
273      NodeData::Element {
274        tag: "om:OMSTR".to_string(),
275        attributes: None,
276        children: vec![NodeData::Text(text)],
277      }
278    }
279  }
280}

282/// Convert unparsed (multiple) nodes to OpenMath error expression.
283fn om_unparsed(doc: &PostDocument, nodes: &[Node]) -> NodeData {
284  if nodes.is_empty() {
285    return om_error("Missing Subexpression");
286  }
287
288  let mut children = vec![NodeData::Element {
289    tag: "om:OMS".to_string(),
290    attributes: Some(HashMap::from([
291      ("cd".to_string(), "ambiguous".to_string()),
292      ("name".to_string(), "fragments".to_string()),
293    ])),
294    children: vec![],
295  }];
296
297  for node in nodes {
298    let tag = doc.get_qname(node).unwrap_or_default();
299    if tag == "ltx:XMHint" {
300      continue;
301    }
302    children.push(om_expr_aux(doc, node));
303  }
304
305  NodeData::Element {
306    tag: "om:OME".to_string(),
307    attributes: None,
308    children,
309  }
310}

312/// Create an OpenMath error element.
313fn om_error(msg: &str) -> NodeData {
314  NodeData::Element {
315    tag: "om:OME".to_string(),
316    attributes: None,
317    children: vec![
318      NodeData::Element {
319        tag: "om:OMS".to_string(),
320        attributes: Some(HashMap::from([
321          ("name".to_string(), "unexpected".to_string()),
322          ("cd".to_string(), "moreerrors".to_string()),
323        ])),
324        children: vec![],
325      },
326      NodeData::Element {
327        tag: "om:OMSTR".to_string(),
328        attributes: None,
329        children: vec![NodeData::Text(msg.to_string())],
330      },
331    ],
332  }
