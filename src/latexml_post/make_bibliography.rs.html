1//! Bibliography generation processor.
2//!
3//! Port of `LaTeXML::Post::MakeBibliography` (818 lines of Perl).
4//! Collects bibliographic entries from `.bib.xml` files and the ObjectDB,
5//! formats them according to the bibliography style (numeric, author-year, alpha),
6//! and fills in `ltx:bibliography` elements with `ltx:biblist` + `ltx:bibitem`.
7//!
8//! Pipeline:
9//! 1. Find bibliography sources (xml files, bib files, literals)
10//! 2. Scan bibentry elements for cited keys (via BIBLABEL:* in ObjectDB)
11//! 3. Transitively include entries cited from within included entries
12//! 4. Extract names, dates, titles for sorting
13//! 5. Detect duplicate author+year pairs → assign suffixes (a, b, c...)
14//! 6. Format each entry into ltx:bibitem with ltx:tags + ltx:bibblock sections
15//! 7. Optionally split by initial letter
16

26/// Citation style.
27#[derive(Debug, Clone, PartialEq)]
28pub enum CitationStyle {
29  /// Numeric: [1], [2], ...
30  Numbers,
31  /// Author-Year: Author (Year)
32  AuthorYear,
33  /// Alphabetic: [ABC24]
34  Alpha,
35}

37/// A collected bibliography entry with metadata.
38///
39/// Port of the `%entries` hash entries in `getBibEntries`.
40#[derive(Debug)]
41struct BibEntryData {
42  bib_key: String,
43  cited_key: Option<String>,
44  sort_key: String,
45  initial: String,
46  author_year: String,
47  suffix: Option<String>,
48  /// Author names for display (short form: "Smith et al").
49  authors_short: String,
50  /// Full author names.
51  authors_full: String,
52  /// Sort-form of author names.
53  sort_names: String,
54  year: String,
55  title: String,
56  /// BibTeX type (article, book, inproceedings, etc.).
57  entry_type: String,
58  /// Reference style (number within bibliography).
59  number: u32,
60  /// IDs that cite this entry (from outside bibliography).
61  referrers: HashSet<String>,
62  /// Bib keys that cite this entry (from other bib entries).
63  bibreferrers: HashSet<String>,
64  /// Keys cited from within this entry.
65  citations: Vec<String>,
66  /// The bibentry XML node (from .bib.xml), if available.
67  bibentry: Option<Node>,
68}

70impl BibEntryData {
71  /// Get the normalized bibliography type for CSS class.
72  fn bib_type(&self) -> &str {
73    if self.entry_type.is_empty() { "misc" } else { &self.entry_type }
74  }

76  /// Get the canonical format type (for FMT_SPEC mapping).
77  ///
78  /// Port of `%FMT_SPEC` aliases.
79  fn format_type(&self) -> &str {
80    match self.entry_type.as_str() {
81      "article" => "article",
82      "book" | "periodical" | "collection" | "proceedings"
83      | "manual" | "misc" | "unpublished" | "booklet" => "book",
84      "incollection" | "collection.article" | "proceedings.article"
85      | "inproceedings" | "inbook" => "incollection",
86      "report" | "techreport" => "report",
87      "thesis" | "mastersthesis" | "phdthesis" => "thesis",
88      "website" | "online" => "website",
89      "software" => "software",
90      _ => "book", // Default fallback
91    }
92  }
93}

118  /// Load bibliography source documents.
119  ///
120  /// Port of `getBibliographies`.
121  /// Locates .bib.xml files from:
122  ///   - Command-line options (overrides)
123  ///   - //ltx:bibliography[@files] attribute
124  fn get_bibliographies(&self, doc: &PostDocument) -> Vec<PostDocument> {
125    let mut bibnames: Vec<String> = Vec::new();
126    let mut from_bibliography = false;
127
128    // Use command-line bibliographies if explicitly given
129    if !self.bibliographies.is_empty() {
130      bibnames = self.bibliographies.clone();
131    } else {
132      // Otherwise, read from the bibliography element's files attribute
133      if let Some(bibnode) = doc.findnode("//ltx:bibliography") {
134        let files = bibnode.get_attribute("files")
135          .or_else(|| {
136            bibnode.get_parent().and_then(|p| p.get_attribute("files"))
137          });
138        if let Some(f) = files {
139          from_bibliography = true;
140          bibnames = f.split(',').map(|s| s.trim().to_string()).collect();
141        }
142      }
143    }
144
145    let search_paths = doc.get_search_paths();
146    let mut bibs: Vec<PostDocument> = Vec::new();
147
148    for bib in &bibnames {
149      let mut loaded = false;
150
151      // Try as .xml file
152      if bib.ends_with(".xml") {
153        if let Some(path) = find_file(bib, search_paths) {
154          match PostDocument::new_from_file(&path, PostDocumentOptions {
155            source_directory: Some(".".to_string()),
156            ..PostDocumentOptions::default()
157          }) {
158            Ok(bibdoc) => { bibs.push(bibdoc); loaded = true; }
159            Err(e) => log::warn!("Failed to load bibliography '{}': {}", bib, e),
160          }
161        }
162      }
163      // Try as .bib or from \bibliography command
164      else if bib.ends_with(".bib") || bib.ends_with(".bib.xml") || from_bibliography {
165        let xmlbib = if from_bibliography && !bib.ends_with(".bib") {
166          format!("{}.bib", bib)
167        } else {
168          bib.clone()
169        };
170        // Look for pre-compiled .bib.xml
171        let xml_candidate = if xmlbib.ends_with(".xml") {
172          xmlbib.clone()
173        } else {
174          format!("{}.xml", xmlbib)
175        };
176        if let Some(path) = find_file(&xml_candidate, search_paths) {
177          match PostDocument::new_from_file(&path, PostDocumentOptions {
178            source_directory: Some(".".to_string()),
179            ..PostDocumentOptions::default()
180          }) {
181            Ok(bibdoc) => { bibs.push(bibdoc); loaded = true; }
182            Err(e) => log::warn!("Failed to load bibliography '{}': {}", path, e),
183          }
184        }
185      }
186
187      // If not loaded yet, try raw .bib file and convert it
188      if !loaded {
189        let bib_file = if from_bibliography && !bib.ends_with(".bib") {
190          format!("{}.bib", bib)
191        } else {
192          bib.clone()
193        };
194        if let Some(bib_path) = find_file(&bib_file, search_paths) {
195          match convert_bib_file_to_xml(&bib_path) {
196            Ok(bibdoc) => { bibs.push(bibdoc); loaded = true; }
197            Err(e) => log::warn!("Failed to convert bibliography '{}': {}", bib_path, e),
198          }
199        }
200      }
201
202      if !loaded {
203        log::info!("Couldn't find usable bibliography for '{}'", bib);
204      }
205    }
206
207    log::info!("MakeBibliography: using {} bibliographies", bibs.len());
208    bibs
209  }

211  /// Collect all cited bibliography entries.
212  ///
213  /// Port of `getBibEntries`.
214  /// Scans BIBLABEL:* entries in ObjectDB, resolves to ID:* entries,
215  /// extracts author/year/title, transitively includes cited-from-cited entries,
216  /// assigns suffixes for duplicate author+year pairs.
217  fn get_bib_entries(
218    &self,
219    doc: &PostDocument,
220    bib_node: &Node,
221  ) -> (HashMap<String, BibEntryData>, Vec<PostDocument>) {
222    let lists_str = bib_node.get_attribute("lists").unwrap_or_else(|| "bibliography".to_string());
223    let lists: Vec<&str> = lists_str.split_whitespace().collect();
224
225    // Step 1: Scan bibliography source documents for ltx:bibentry elements.
226    // Build a map: lc(bibkey) → { bibkey, bibentry, citations }
227    // Import bibentry nodes into the main document so that XPath queries
228    // (which use the main document's namespace context) work correctly.
229    let mut entries: HashMap<String, BibEntryData> = HashMap::new();
230    let bib_docs = self.get_bibliographies(doc);
231    for bibdoc in &bib_docs {
232      for bibentry in bibdoc.findnodes("//ltx:bibentry") {
233        let bibkey = match bibentry.get_attribute("key") {
234          Some(k) => k,
235          None => continue,
236        };
237        let lc_key = bibkey.to_lowercase();
238        // Extract citations from within this bibentry
239        let citations: Vec<String> = bibdoc.findnodes_at(".//@bibrefs", Some(&bibentry))
240          .iter()
241          .filter_map(|n| {
242            let val = n.get_content();
243            if val.is_empty() { None } else { Some(val) }
244          })
245          .flat_map(|s| s.split(',').map(String::from).collect::<Vec<_>>())
246          .filter(|s| !s.is_empty())
247          .collect();
248
249        let imported = bibentry.clone();
250
251        entries.insert(lc_key, BibEntryData {
252          bib_key: bibkey,
253          cited_key: None,
254          sort_key: String::new(),
255          initial: String::new(),
256          author_year: String::new(),
257          suffix: None,
258          authors_short: String::new(),
259          authors_full: String::new(),
260          sort_names: String::new(),
261          year: String::new(),
262          title: String::new(),
263          entry_type: String::new(),
264          number: 0,
265          referrers: HashSet::new(),
266          bibreferrers: HashSet::new(),
267          citations,
268          bibentry: Some(imported),
269        });
270      }
271    }
272
273    // Step 2: Collect all cited bibliography keys from BIBLABEL entries in ObjectDB.
274    // Note referrers (from outside the bibliography).
275    let cite_star = lists.iter().any(|list| {
276      self.db.lookup(&format!("BIBLABEL:{}:*", list)).is_some()
277    });
278
279    let mut queue: Vec<String> = Vec::new();
280    for db_key in self.db.get_keys() {
281      if !db_key.starts_with("BIBLABEL:") { continue; }
282      let parts: Vec<&str> = db_key.splitn(3, ':').collect();
283      if parts.len() < 3 { continue; }
284      let (list, bibkey) = (parts[1], parts[2]);
285      if !lists.contains(&list) { continue; }
286
287      let lc_key = bibkey.to_lowercase();
288      if let Some(bentry) = self.db.lookup(db_key) {
289        let has_refs = bentry.get_value("referrers").map(|v| v.is_truthy()).unwrap_or(false);
290        if has_refs {
291          // Filter referrers: walk up parent chain, skip if inside ltx:bibitem
292          if let Some(crate::object_db::Value::Hash(refs)) = bentry.get_value("referrers") {
293            for ref_id in refs.keys() {
294              let mut rid = ref_id.clone();
295              let mut is_from_bib = false;
296              while let Some(entry) = self.db.lookup(&format!("ID:{}", rid)) {
297                let entry_type = entry.get_string("type").unwrap_or("");
298                if entry_type == "ltx:bibitem" {
299                  is_from_bib = true;
300                  break;
301                }
302                match entry.get_string("parent").map(String::from) {
303                  Some(parent) => rid = parent,
304                  None => break,
305                }
306              }
307              if !is_from_bib {
308                // Check for case mismatch
309                if let Some(existing) = entries.get(&lc_key) {
310                  if let Some(ref prev_key) = existing.cited_key {
311                    if prev_key != bibkey {
312                      log::warn!("Case mismatch in bib key '{}' vs '{}'", prev_key, bibkey);
313                    }
314                  }
315                }
316                let entry = entries.entry(lc_key.clone()).or_insert_with(|| BibEntryData {
317                  bib_key: bibkey.to_string(),
318                  cited_key: None,
319                  sort_key: String::new(),
320                  initial: String::new(),
321                  author_year: String::new(),
322                  suffix: None,
323                  authors_short: String::new(),
324                  authors_full: String::new(),
325                  sort_names: String::new(),
326                  year: String::new(),
327                  title: String::new(),
328                  entry_type: String::new(),
329                  number: 0,
330                  referrers: HashSet::new(),
331                  bibreferrers: HashSet::new(),
332                  citations: Vec::new(),
333                  bibentry: None,
334                });
335                entry.cited_key = Some(bibkey.to_string());
336                entry.referrers.insert(ref_id.clone());
337              }
338            }
339          }
340          if entries.get(&lc_key).map(|e| !e.referrers.is_empty()).unwrap_or(false) {
341            queue.push(bibkey.to_string());
342          }
343        } else if cite_star {
344          queue.push(bibkey.to_string());
345        }
346      }
347    }
348
349    // Step 3: Process queue — transitively include cited entries.
350    // For each key, extract names/year/title from bibentry XML.
351    let mut seen: HashSet<String> = HashSet::new();
352    let mut included: HashMap<String, BibEntryData> = HashMap::new();
353    let mut missing_keys: Vec<String> = Vec::new();
354
355    while let Some(bibkey) = queue.pop() {
356      if seen.contains(&bibkey) || bibkey == "*" { continue; }
357      seen.insert(bibkey.clone());
358      let lc_key = bibkey.to_lowercase();
359
360      if let Some(mut entry) = entries.remove(&lc_key) {
361        // Extract metadata from bibentry XML node (if available)
362        if let Some(ref bibentry) = entry.bibentry {
363          let (sort_names, short_names, _full_names) = extract_names(doc, bibentry);
364          entry.sort_names = sort_names.clone();
365          entry.authors_short = short_names;
366
367          let names_for_sort = if sort_names.is_empty() {
368            // Try bib-key or bib-title as fallback
369            if let Some(key_node) = PostDocument::findnodes_foreign("ltx:bib-key", bibentry).into_iter().next() {
370              key_node.get_content()
371            } else if let Some(title_node) = PostDocument::findnodes_foreign("ltx:bib-title", bibentry).into_iter().next() {
372              title_node.get_content()
373            } else {
374              bibkey.clone()
375            }
376          } else {
377            sort_names
378          };
379
380          // Year
381          let date_content = PostDocument::findnodes_foreign("ltx:bib-date[@role='publication']", bibentry).into_iter().next()
382            .map(|n| n.get_content())
383            .unwrap_or_default();
384          let year = extract_four_digit_year(&date_content);
385          entry.year = year.clone();
386
387          // Title
388          let title = PostDocument::findnodes_foreign("ltx:bib-title", bibentry).into_iter().next()
389            .map(|n| n.get_content())
390            .unwrap_or_default();
391          entry.title = title.clone();
392
393          // Type
394          let entry_type = bibentry.get_attribute("type").unwrap_or_else(|| "misc".to_string());
395          entry.entry_type = entry_type;
396
397          // Author+year for suffix detection
398          entry.author_year = format!("{}.{}", names_for_sort, year);
399          entry.initial = PostDocument::initial(&names_for_sort, true);
400
401          // Sort key
402          let sort_key = format!("{}.{}.{}.{}",
403            names_for_sort, year, title, bibkey).to_lowercase();
404          entry.sort_key = sort_key.clone();
405
406          // Enqueue transitive citations
407          let citations = entry.citations.clone();
408          for c in &citations {
409            queue.push(c.clone());
410          }
411          included.insert(sort_key, entry);
412        } else {
413          // No bibentry XML — use ObjectDB metadata
414          let id = self.find_bib_id(&bibkey, &lists);
415          if let Some(id) = id {
416            let id_key = format!("ID:{}", id);
417            let authors = self.db.lookup(&id_key)
418              .and_then(|e| e.get_value("authors").map(|v| v.to_string()))
419              .unwrap_or_default();
420            let full_authors = self.db.lookup(&id_key)
421              .and_then(|e| e.get_value("fullauthors").map(|v| v.to_string()))
422              .unwrap_or_else(|| authors.clone());
423            let year = self.db.lookup(&id_key)
424              .and_then(|e| e.get_value("year").map(|v| v.to_string()))
425              .unwrap_or_default();
426            let title = self.db.lookup(&id_key)
427              .and_then(|e| e.get_value("title").map(|v| v.to_string()))
428              .unwrap_or_default();
429            let entry_type = self.db.lookup(&id_key)
430              .and_then(|e| e.get_value("type").map(|v| v.to_string()))
431              .unwrap_or_else(|| "misc".to_string());
432
433            let year_short = extract_four_digit_year(&year);
434            let names = if authors.is_empty() { bibkey.clone() } else { authors.clone() };
435            let author_year = format!("{}.{}", names, year_short);
436            let initial = PostDocument::initial(&names, true);
437            let sort_key = format!("{}.{}.{}.{}", names, year_short, title, bibkey).to_lowercase();
438
439            entry.authors_short = authors;
440            entry.authors_full = full_authors;
441            entry.sort_names = names;
442            entry.year = year_short;
443            entry.title = title;
444            entry.entry_type = entry_type;
445            entry.author_year = author_year;
446            entry.initial = initial;
447            entry.sort_key = sort_key.clone();
448
449            included.insert(sort_key, entry);
450          } else {
451            missing_keys.push(bibkey);
452          }
453        }
454      } else {
455        // Not found in entries map
456        missing_keys.push(bibkey);
457      }
458    }
459
460    if !missing_keys.is_empty() {
461      log::warn!("Missing bibkeys: {}", missing_keys.join(", "));
462    }
463
464    // Step 4: Note bibreferrers — for each included entry's citations,
465    // mark the cited entry as having this entry as a bibreferrer.
466    let citations_map: Vec<(String, Vec<String>)> = included.values()
467      .map(|e| (e.bib_key.clone(), e.citations.clone()))
468      .collect();
469    for (bibkey, citations) in &citations_map {
470      for cited in citations {
471        let lc = cited.to_lowercase();
472        // Find which sort_key corresponds to this lc bibkey
473        for entry in included.values_mut() {
474          if entry.bib_key.to_lowercase() == lc {
475            entry.bibreferrers.insert(bibkey.clone());
476          }
477        }
478      }
479    }
480
481    log::info!(
482      "MakeBibliography: {} bibentries, {} cited",
483      entries.len() + included.len(),
484      included.len()
485    );
486
487    // Step 5: Sort and detect duplicate author+year → assign suffixes.
488    let mut sorted_keys: Vec<String> = included.keys().cloned().collect();
489    sorted_keys.sort();
490
491    // Port of suffix detection: track by author_year, assign suffixes when duplicated.
492    let mut ay_last: HashMap<String, String> = HashMap::new(); // ay → last sort_key with this ay
493    for key in &sorted_keys {
494      if let Some(entry) = included.get(key) {
495        let ay = entry.author_year.clone();
496        if let Some(prev_key) = ay_last.get(&ay) {
497          let prev_key = prev_key.clone();
498          // Previous entry with same ay needs a suffix too
499          if let Some(prev) = included.get_mut(&prev_key) {
500            if prev.suffix.is_none() {
501              prev.suffix = Some(radix_alpha(1));
502            }
503          }
504          let prev_counter = included.get(&prev_key)
505            .and_then(|p| p.suffix.as_ref())
506            .map(|s| suffix_to_counter(s))
507            .unwrap_or(1);
508          if let Some(e) = included.get_mut(key) {
509            e.suffix = Some(radix_alpha(prev_counter + 1));
510          }
511        }
512        ay_last.insert(ay, key.clone());
513      }
514    }
515
516    // Step 6: Remove sort ERROR nodes from bibentries
517    for entry in included.values() {
518      if let Some(ref bibentry) = entry.bibentry {
519        let sort_errors = PostDocument::findnodes_foreign(".//ltx:ERROR[@class='sort']", bibentry);
520        for mut sortnode in sort_errors {
521          sortnode.unlink();
522        }
523      }
524    }
525
526    // Assign numbers in sort order
527    let mut number = 0u32;
528    for key in &sorted_keys {
529      number += 1;
530      if let Some(entry) = included.get_mut(key) {
531        entry.number = number;
532      }
533    }
534
535    (included, bib_docs)
536  }

538  /// Find the ID for a bibliography key in the ObjectDB.
539  fn find_bib_id(&self, bibkey: &str, lists: &[&str]) -> Option<String> {
540    for list in lists {
541      let bkey = format!("BIBLABEL:{}:{}", list, bibkey);
542      if let Some(bentry) = self.db.lookup(&bkey) {
543        if let Some(id) = bentry.get_string("id") {
544          return Some(id.to_string());
545        }
546      }
547    }
548    None
549  }

551  /// Format a bibliography list.
552  ///
553  /// Port of `makeBibliographyList`.
554  fn make_bibliography_list(
555    &self,
556    doc: &PostDocument,
557    bib_id: &str,
558    initial: Option<&str>,
559    entries: &HashMap<String, BibEntryData>,
560    style: &CitationStyle,
561  ) -> NodeData {
562    let id = if let Some(init) = initial {
563      format!("{}.L1.{}", bib_id, init)
564    } else {
565      format!("{}.L1", bib_id)
566    };
567
568    let mut sorted_keys: Vec<&String> = entries.keys().collect();
569    sorted_keys.sort();
570    let items: Vec<NodeData> = sorted_keys.iter()
571      .filter_map(|key| entries.get(*key))
572      .map(|entry| self.format_bib_entry(doc, bib_id, entry, style))
573      .collect();
574
575    NodeData::Element {
576      tag: "ltx:biblist".to_string(),
577      attributes: Some(HashMap::from([("xml:id".to_string(), id)])),
578      children: items,
579    }
580  }

582  /// Format a single bibentry into a bibitem.
583  ///
584  /// Port of `formatBibEntry`.
585  fn format_bib_entry(
586    &self,
587    doc: &PostDocument,
588    bib_id: &str,
589    entry: &BibEntryData,
590    style: &CitationStyle,
591  ) -> NodeData {
592    // ID generation: match Perl's $id =~ s/^bib//; $id = $bibid . $id;
593    let id = if let Some(ref bibentry) = entry.bibentry {
594      let orig_id = bibentry.get_attribute("xml:id").unwrap_or_default();
595      if orig_id.is_empty() {
596        // No xml:id on bibentry (e.g. from raw .bib parsing) — use number
597        format!("{}.bib{}", bib_id, entry.number)
598      } else {
599        let stripped = orig_id.strip_prefix("bib").unwrap_or(&orig_id);
600        format!("{}{}", bib_id, stripped)
601      }
602    } else {
603      format!("{}.bib{}", bib_id, entry.number)
604    };
605
606    let cited_key = entry.cited_key.as_deref().unwrap_or(&entry.bib_key);
607    let mut children = Vec::new();
608
609    // --- Tags ---
610    let mut tags = Vec::new();
611
612    // Number tag
613    tags.push(NodeData::Element {
614      tag: "ltx:tag".to_string(),
615      attributes: Some(HashMap::from([
616        ("role".to_string(), "number".to_string()),
617        ("class".to_string(), "ltx_bib_number".to_string()),
618      ])),
619      children: vec![NodeData::Text(entry.number.to_string())],
620    });
621
622    // Authors/fullauthors tags — extracted from bibentry XML if available
623    let (author_tag_nodes, has_names, has_key, has_year, has_typetag) =
624      self.build_author_year_tags(doc, entry);
625    tags.extend(author_tag_nodes);
626
627    // Refnum tag: depends on citation style
628    let mut effective_style = style.clone();
629    // Perl: $style = 'numbers' unless (@names || $keytag) && (@year || $typetag)
630    if !((has_names || has_key) && (has_year || has_typetag)) {
631      effective_style = CitationStyle::Numbers;
632    }
633
634    let mut skip_first_block = false;
635    match effective_style {
636      CitationStyle::Numbers => {
637        tags.push(NodeData::Element {
638          tag: "ltx:tag".to_string(),
639          attributes: Some(HashMap::from([
640            ("role".to_string(), "refnum".to_string()),
641            ("class".to_string(), "ltx_bib_key".to_string()),
642            ("open".to_string(), "[".to_string()),
643            ("close".to_string(), "]".to_string()),
644          ])),
645          children: vec![NodeData::Text(entry.number.to_string())],
646        });
647      }
648      CitationStyle::Alpha => {
649        // AY-style: abbreviation from author names + 2-digit year
650        let aa = self.make_alpha_label(doc, entry);
651        let yy = if entry.year.len() >= 4 {
652          entry.year[2..4].to_string()
653        } else {
654          entry.year.clone()
655        };
656        let suffix = entry.suffix.as_deref().unwrap_or("");
657        tags.push(NodeData::Element {
658          tag: "ltx:tag".to_string(),
659          attributes: Some(HashMap::from([
660            ("role".to_string(), "refnum".to_string()),
661            ("class".to_string(), "ltx_bib_abbrv".to_string()),
662            ("open".to_string(), "[".to_string()),
663            ("close".to_string(), "]".to_string()),
664          ])),
665          children: vec![NodeData::Text(format!("{}{}{}", aa, yy, suffix))],
666        });
667      }
668      CitationStyle::AuthorYear => {
669        // Author-Year style: "Author (Year)" — skip first block (redundant)
670        skip_first_block = true;
671        let suffix = entry.suffix.as_deref().unwrap_or("");
672        let author_text = if !entry.authors_short.is_empty() {
673          entry.authors_short.clone()
674        } else {
675          entry.bib_key.clone()
676        };
677        let year_text = if !entry.year.is_empty() {
678          format!("{}{}", entry.year, suffix)
679        } else {
680          String::new()
681        };
682        let mut refnum_children = vec![NodeData::Text(author_text)];
683        if !year_text.is_empty() {
684          refnum_children.push(NodeData::Text(format!(" ({})", year_text)));
685        }
686        tags.push(NodeData::Element {
687          tag: "ltx:tag".to_string(),
688          attributes: Some(HashMap::from([
689            ("role".to_string(), "refnum".to_string()),
690            ("class".to_string(), "ltx_bib_author-year".to_string()),
691          ])),
692          children: refnum_children,
693        });
694      }
695    }
696
697    if !tags.is_empty() {
698      children.push(NodeData::Element {
699        tag: "ltx:tags".to_string(),
700        attributes: None,
701        children: tags,
702      });
703    }
704
705    // --- Content blocks ---
706    let blocks = self.format_blocks(doc, entry, skip_first_block);
707    children.extend(blocks);
708
709    // --- Cited-by block ---
710    let mut citedby: Vec<NodeData> = Vec::new();
711    let mut sorted_referrers: Vec<&String> = entry.referrers.iter().collect();
712    sorted_referrers.sort();
713    for ref_id in &sorted_referrers {
714      citedby.push(NodeData::Element {
715        tag: "ltx:ref".to_string(),
716        attributes: Some(HashMap::from([
717          ("idref".to_string(), ref_id.to_string()),
718          ("show".to_string(), "typerefnum".to_string()),
719        ])),
720        children: vec![],
721      });
722    }
723    if !entry.bibreferrers.is_empty() {
724      let mut sorted_bibrefs: Vec<&String> = entry.bibreferrers.iter().collect();
725      sorted_bibrefs.sort();
726      citedby.push(NodeData::Element {
727        tag: "ltx:bibref".to_string(),
728        attributes: Some(HashMap::from([
729          ("bibrefs".to_string(), sorted_bibrefs.iter().map(|s| s.as_str()).collect::<Vec<_>>().join(",")),
730          ("show".to_string(), "refnum".to_string()),
731        ])),
732        children: vec![],
733      });
734    }
735    if !citedby.is_empty() {
736      let conjoined = PostDocument::conjoin(
737        crate::document::Conjunction::Simple(",\n".to_string()),
738        citedby,
739      );
740      let mut block_children = vec![NodeData::Text("Cited by: ".to_string())];
741      block_children.extend(conjoined);
742      block_children.push(NodeData::Text(".".to_string()));
743      children.push(NodeData::Element {
744        tag: "ltx:bibblock".to_string(),
745        attributes: Some(HashMap::from([("class".to_string(), "ltx_bib_cited".to_string())])),
746        children: block_children,
747      });
748    }
749
750    NodeData::Element {
751      tag: "ltx:bibitem".to_string(),
752      attributes: Some(HashMap::from([
753        ("xml:id".to_string(), id),
754        ("key".to_string(), cited_key.to_string()),
755        ("type".to_string(), entry.entry_type.clone()),
756        ("class".to_string(), format!("ltx_bib_{}", entry.bib_type())),
757      ])),
758      children,
759    }
760  }

762  /// Build author/year/key/title/type tags from bibentry XML.
763  ///
764  /// Returns: (tag nodes, has_names, has_key, has_year, has_typetag)
765  fn build_author_year_tags(
766    &self,
767    doc: &PostDocument,
768    entry: &BibEntryData,
769  ) -> (Vec<NodeData>, bool, bool, bool, bool) {
770    let mut tags = Vec::new();
771    let mut has_names = false;
772    let mut has_key = false;
773    let mut has_year = false;
774    let mut has_typetag = false;
775
776    if let Some(ref bibentry) = entry.bibentry {
777      // Author surnames from bibentry XML
778      let mut surnames: Vec<Node> = doc.findnodes_at(
779        "ltx:bib-name[@role='author']/ltx:surname", Some(bibentry));
780      if surnames.is_empty() {
781        surnames = doc.findnodes_at(
782          "ltx:bib-name[@role='editor']/ltx:surname", Some(bibentry));
783      }
784
785      if surnames.len() > 2 {
786        has_names = true;
787        // Short: first author + et al.
788        let first_text = surnames[0].get_content();
789        tags.push(NodeData::Element {
790          tag: "ltx:tag".to_string(),
791          attributes: Some(HashMap::from([
792            ("role".to_string(), "authors".to_string()),
793            ("class".to_string(), "ltx_bib_author".to_string()),
794          ])),
795          children: vec![
796            NodeData::Text(first_text),
797            NodeData::Element {
798              tag: "ltx:text".to_string(),
799              attributes: Some(HashMap::from([("class".to_string(), "ltx_bib_etal".to_string())])),
800              children: vec![NodeData::Text(" et al.".to_string())],
801            },
802          ],
803        });
804        // Full: all names
805        let mut full_children: Vec<NodeData> = Vec::new();
806        for (i, surname) in surnames.iter().enumerate() {
807          if i > 0 && i < surnames.len() - 1 {
808            full_children.push(NodeData::Text(", ".to_string()));
809          } else if i == surnames.len() - 1 {
810            full_children.push(NodeData::Text(" and ".to_string()));
811          }
812          full_children.push(NodeData::Text(surname.get_content()));
813        }
814        tags.push(NodeData::Element {
815          tag: "ltx:tag".to_string(),
816          attributes: Some(HashMap::from([
817            ("role".to_string(), "fullauthors".to_string()),
818            ("class".to_string(), "ltx_bib_author".to_string()),
819          ])),
820          children: full_children,
821        });
822      } else if surnames.len() == 2 {
823        has_names = true;
824        tags.push(NodeData::Element {
825          tag: "ltx:tag".to_string(),
826          attributes: Some(HashMap::from([
827            ("role".to_string(), "authors".to_string()),
828            ("class".to_string(), "ltx_bib_author".to_string()),
829          ])),
830          children: vec![
831            NodeData::Text(surnames[0].get_content()),
832            NodeData::Text(" and ".to_string()),
833            NodeData::Text(surnames[1].get_content()),
834          ],
835        });
836      } else if !surnames.is_empty() {
837        has_names = true;
838        tags.push(NodeData::Element {
839          tag: "ltx:tag".to_string(),
840          attributes: Some(HashMap::from([
841            ("role".to_string(), "authors".to_string()),
842            ("class".to_string(), "ltx_bib_author".to_string()),
843          ])),
844          children: vec![NodeData::Text(surnames[0].get_content())],
845        });
846      }
847
848      // Key tag
849      if let Some(key_node) = PostDocument::findnodes_foreign("ltx:bib-key", bibentry).into_iter().next() {
850        has_key = true;
851        tags.push(NodeData::Element {
852          tag: "ltx:tag".to_string(),
853          attributes: Some(HashMap::from([
854            ("role".to_string(), "key".to_string()),
855            ("class".to_string(), "ltx_bib_key".to_string()),
856          ])),
857          children: vec![NodeData::Text(key_node.get_content())],
858        });
859      }
860
861      // Year tag
862      if let Some(date_node) = PostDocument::findnodes_foreign("ltx:bib-date[@role='publication']", bibentry).into_iter().next() {
863        has_year = true;
864        let year_text = extract_four_digit_year(&date_node.get_content());
865        let suffix = entry.suffix.as_deref().unwrap_or("");
866        tags.push(NodeData::Element {
867          tag: "ltx:tag".to_string(),
868          attributes: Some(HashMap::from([
869            ("role".to_string(), "year".to_string()),
870            ("class".to_string(), "ltx_bib_year".to_string()),
871          ])),
872          children: vec![NodeData::Text(format!("{}{}", year_text, suffix))],
873        });
874      }
875
876      // Type tag
877      if let Some(type_node) = PostDocument::findnodes_foreign("ltx:bib-type", bibentry).into_iter().next() {
878        has_typetag = true;
879        tags.push(NodeData::Element {
880          tag: "ltx:tag".to_string(),
881          attributes: Some(HashMap::from([
882            ("role".to_string(), "bibtype".to_string()),
883            ("class".to_string(), "ltx_bib_type".to_string()),
884          ])),
885          children: vec![NodeData::Text(type_node.get_content())],
886        });
887      }
888
889      // Title tag
890      if let Some(title_node) = PostDocument::findnodes_foreign("ltx:bib-title", bibentry).into_iter().next() {
891        tags.push(NodeData::Element {
892          tag: "ltx:tag".to_string(),
893          attributes: Some(HashMap::from([
894            ("role".to_string(), "title".to_string()),
895            ("class".to_string(), "ltx_bib_title".to_string()),
896          ])),
897          children: vec![NodeData::Text(title_node.get_content())],
898        });
899      }
900    } else {
901      // No bibentry XML — use ObjectDB metadata strings
902      if !entry.authors_short.is_empty() {
903        has_names = true;
904        tags.push(NodeData::Element {
905          tag: "ltx:tag".to_string(),
906          attributes: Some(HashMap::from([
907            ("role".to_string(), "authors".to_string()),
908            ("class".to_string(), "ltx_bib_author".to_string()),
909          ])),
910          children: vec![NodeData::Text(entry.authors_short.clone())],
911        });
912        if entry.authors_full != entry.authors_short {
913          tags.push(NodeData::Element {
914            tag: "ltx:tag".to_string(),
915            attributes: Some(HashMap::from([
916              ("role".to_string(), "fullauthors".to_string()),
917              ("class".to_string(), "ltx_bib_author".to_string()),
918            ])),
919            children: vec![NodeData::Text(entry.authors_full.clone())],
920          });
921        }
922      }
923      if !entry.year.is_empty() {
924        has_year = true;
925        let suffix = entry.suffix.as_deref().unwrap_or("");
926        tags.push(NodeData::Element {
927          tag: "ltx:tag".to_string(),
928          attributes: Some(HashMap::from([
929            ("role".to_string(), "year".to_string()),
930            ("class".to_string(), "ltx_bib_year".to_string()),
931          ])),
932          children: vec![NodeData::Text(format!("{}{}", entry.year, suffix))],
933        });
934      }
935      if !entry.title.is_empty() {
936        tags.push(NodeData::Element {
937          tag: "ltx:tag".to_string(),
938          attributes: Some(HashMap::from([
939            ("role".to_string(), "title".to_string()),
940            ("class".to_string(), "ltx_bib_title".to_string()),
941          ])),
942          children: vec![NodeData::Text(entry.title.clone())],
943        });
944      }
945    }
946
947    (tags, has_names, has_key, has_year, has_typetag)
948  }

950  /// Generate alphabetic label for AY/alpha citation style.
951  ///
952  /// Port of the alpha refnum logic in `formatBibEntry`.
953  fn make_alpha_label(&self, doc: &PostDocument, entry: &BibEntryData) -> String {
954    if let Some(ref bibentry) = entry.bibentry {
955      let mut surnames: Vec<Node> = doc.findnodes_at(
956        "ltx:bib-name[@role='author']/ltx:surname", Some(bibentry));
957      if surnames.is_empty() {
958        surnames = doc.findnodes_at(
959          "ltx:bib-name[@role='editor']/ltx:surname", Some(bibentry));
960      }
961      if surnames.len() > 1 {
962        let mut aa: String = surnames.iter()
963          .map(|n| n.get_content().chars().next().unwrap_or('?').to_string())
964          .collect();
965        if aa.len() > 3 {
966          aa = format!("{}+", &aa[..3]);
967        }
968        aa.to_uppercase()
969      } else if !surnames.is_empty() {
970        let text = surnames[0].get_content();
971        text.chars().take(3).collect::<String>().to_uppercase()
972      } else {
973        entry.bib_key.chars().take(3).collect::<String>().to_uppercase()
974      }
975    } else {
976      // Fallback: use author short name
977      if !entry.authors_short.is_empty() {
978        entry.authors_short.split_whitespace()
979          .filter_map(|w| w.chars().next())
980          .map(|c| c.to_uppercase().to_string())
981          .collect::<Vec<_>>()
982          .join("")
983      } else {
984        entry.bib_key.chars().take(3).collect::<String>().to_uppercase()
985      }
986    }
987  }

989  /// Format content blocks using the FMT_SPEC table.
990  ///
991  /// Port of the block formatting loop in `formatBibEntry` + `%FMT_SPEC`.
992  fn format_blocks(
993    &self,
994    doc: &PostDocument,
995    entry: &BibEntryData,
996    skip_first: bool,
997  ) -> Vec<NodeData> {
998    let format_type = entry.format_type();
999    let block_specs = get_fmt_spec(format_type);
1000    let mut blocks = Vec::new();
1001
1002    for (i, block_spec) in block_specs.iter().enumerate() {
1003      if skip_first && i == 0 { continue; }
1004
1005      let mut items: Vec<NodeData> = Vec::new();
1006      for field_spec in block_spec {
1007        let (nodes_found, negated) = if let Some(ref bibentry) = entry.bibentry {
1008          let xpath = field_spec.xpath.trim_start_matches('!').trim();
1009          let negated = field_spec.xpath.starts_with('!');
1010          if xpath == "true" {
1011            (true, false)
1012          } else {
1013            let found = !PostDocument::findnodes_foreign(xpath, bibentry).is_empty();
1014            (found, negated)
1015          }
1016        } else {
1017          // No bibentry — try to match from metadata
1018          let found = match_metadata_field(field_spec.xpath, entry);
1019          (found, field_spec.xpath.starts_with('!'))
1020        };
1021
1022        // Check condition
1023        if field_spec.xpath != "true" {
1024          if negated {
1025            if nodes_found { continue; }
1026          } else {
1027            if !nodes_found { continue; }
1028          }
1029        }
1030
1031        // Add punctuation if there are preceding items
1032        if !field_spec.punct.is_empty() && !items.is_empty() {
1033          items.push(NodeData::Text(field_spec.punct.to_string()));
1034        }
1035        // Pre-text
1036        if !field_spec.pre.is_empty() {
1037          items.push(NodeData::Text(field_spec.pre.to_string()));
1038        }
1039        // Content (wrapped in ltx:text with class)
1040        if !field_spec.class.is_empty() {
1041          let content = if let Some(ref bibentry) = entry.bibentry {
1042            let xpath = field_spec.xpath.trim_start_matches('!').trim();
1043            if xpath == "true" {
1044              Vec::new()
1045            } else {
1046              let nodes = PostDocument::findnodes_foreign(xpath, bibentry);
1047              apply_formatter(doc, field_spec.formatter, &nodes)
1048            }
1049          } else {
1050            get_metadata_content(field_spec.xpath, entry)
1051          };
1052          if !content.is_empty() {
1053            items.push(NodeData::Element {
1054              tag: "ltx:text".to_string(),
1055              attributes: Some(HashMap::from([
1056                ("class".to_string(), format!("ltx_bib_{}", field_spec.class)),
1057              ])),
1058              children: content,
1059            });
1060          }
1061        }
1062        // Post-text
1063        if !field_spec.post.is_empty() {
1064          items.push(NodeData::Text(field_spec.post.to_string()));
1065        }
1066      }
1067
1068      if !items.is_empty() {
1069        blocks.push(make_bibblock("", &items));
1070      }
1071    }
1072
1073    // META_BLOCK: Note and External Links
1074    if let Some(ref bibentry) = entry.bibentry {
1075      // Note block
1076      if !PostDocument::findnodes_foreign("ltx:bib-note", bibentry).is_empty() {
1077        let notes = PostDocument::findnodes_foreign("ltx:bib-note", bibentry);
1078        let content: Vec<NodeData> = notes.iter()
1079          .map(|n| NodeData::Text(n.get_content()))
1080          .collect();
1081        if !content.is_empty() {
1082          let mut items = vec![NodeData::Text("Note: ".to_string())];
1083          items.push(NodeData::Element {
1084            tag: "ltx:text".to_string(),
1085            attributes: Some(HashMap::from([
1086              ("class".to_string(), "ltx_bib_note".to_string()),
1087            ])),
1088            children: content,
1089          });
1090          blocks.push(make_bibblock("", &items));
1091        }
1092      }
1093      // External Links block
1094      let links_xpath = "ltx:bib-links | ltx:bib-review | ltx:bib-identifier | ltx:bib-url";
1095      let link_nodes = PostDocument::findnodes_foreign(links_xpath, bibentry);
1096      if !link_nodes.is_empty() {
1097        let link_items = format_links(doc, &link_nodes);
1098        if !link_items.is_empty() {
1099          let mut items = vec![NodeData::Text("External Links: ".to_string())];
1100          items.push(NodeData::Element {
1101            tag: "ltx:text".to_string(),
1102            attributes: Some(HashMap::from([
1103              ("class".to_string(), "ltx_bib_links".to_string()),
1104            ])),
1105            children: link_items,
1106          });
1107          blocks.push(make_bibblock("", &items));
1108        }
1109      }
1110    }
1111
1112    blocks
1113  }
1114}

1116impl Processor for MakeBibliography {
1117  fn get_name(&self) -> &str { &self.name }
1118
1119  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
1120    doc.findnodes("//ltx:bibliography")
1121  }
1122
1123  fn process(&mut self, mut doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
1124    for bib in &nodes {
1125      // Skip if already populated
1126      if !doc.findnodes_at(".//ltx:bibitem", Some(bib)).is_empty() {
1127        continue;
1128      }
1129
1130      // Read citation style from element attributes
1131      let citestyle_str = bib.get_attribute("citestyle").unwrap_or_else(|| "numbers".to_string());
1132      let style = match citestyle_str.as_str() {
1133        "AY" | "authoryear" | "author-year" => CitationStyle::AuthorYear,
1134        "alpha" | "Alpha" => CitationStyle::Alpha,
1135        _ => CitationStyle::Numbers,
1136      };
1137
1138      // bib_docs must be kept alive as long as entries (bibentry Nodes reference them)
1139      let (entries, _bib_docs) = self.get_bib_entries(&doc, bib);
1140      if entries.is_empty() {
1141        log::info!("MakeBibliography: no entries to process");
1142        continue;
1143      }
1144
1145      let bib_id = bib.get_attribute("xml:id")
1146        .or_else(|| doc.get_document_element().and_then(|r| r.get_attribute("xml:id")))
1147        .unwrap_or_else(|| "bib".to_string());
1148
1149      if self.split {
1150        // Split by initial letter
1151        let mut by_initial: HashMap<String, HashMap<String, &BibEntryData>> = HashMap::new();
1152        for (key, entry) in &entries {
1153          by_initial.entry(entry.initial.clone()).or_default().insert(key.clone(), entry);
1154        }
1155        let mut initials: Vec<&String> = by_initial.keys().collect();
1156        initials.sort();
1157        for initial in initials {
1158          // Build a subset HashMap for this initial
1159          let subset: HashMap<String, BibEntryData> = by_initial[initial].iter()
1160            .map(|(k, e)| (k.clone(), clone_entry(e)))
1161            .collect();
1162          let biblist = self.make_bibliography_list(&doc, &bib_id, Some(initial), &subset, &style);
1163          let mut bib_mut = bib.clone();
1164          doc.add_nodes(&mut bib_mut, &[biblist]);
1165        }
1166      } else {
1167        let biblist = self.make_bibliography_list(&doc, &bib_id, None, &entries, &style);
1168        let mut bib_mut = bib.clone();
1169        doc.add_nodes(&mut bib_mut, &[biblist]);
1170      }
1171
1172      log::info!("MakeBibliography: formatted {} entries", entries.len());
1173
1174      // Register formatted bibitems in ObjectDB so CrossRef can resolve citations.
1175      // Port of Perl's approach where bibitems are registered during Scan,
1176      // but here we must register them after MakeBibliography creates them.
1177      let lists_str = bib.get_attribute("lists").unwrap_or_else(|| "bibliography".to_string());
1178      for entry in entries.values() {
1179        let cited_key = entry.cited_key.as_deref().unwrap_or(&entry.bib_key);
1180        // Compute the same ID as format_bib_entry
1181        let bibitem_id = if let Some(ref bibentry) = entry.bibentry {
1182          let orig_id = bibentry.get_attribute("xml:id").unwrap_or_default();
1183          if orig_id.is_empty() {
1184            format!("{}.bib{}", bib_id, entry.number)
1185          } else {
1186            let stripped = orig_id.strip_prefix("bib").unwrap_or(&orig_id);
1187            format!("{}{}", bib_id, stripped)
1188          }
1189        } else {
1190          format!("{}.bib{}", bib_id, entry.number)
1191        };
1192
1193        // Register BIBLABEL:{list}:{key} → id
1194        for list in lists_str.split_whitespace() {
1195          let label_key = format!("BIBLABEL:{}:{}", list, cited_key);
1196          self.db.register(&label_key, vec![
1197            ("id", crate::object_db::Value::from(bibitem_id.as_str())),
1198          ]);
1199        }
1200
1201        // Register ID:{id} with type, location, and number for CrossRef URL generation
1202        let location = doc.site_relative_destination().unwrap_or_default();
1203        self.db.register(&format!("ID:{}", bibitem_id), vec![
1204          ("type", crate::object_db::Value::from("ltx:bibitem")),
1205          ("location", crate::object_db::Value::from(location.as_str())),
1206          ("fragid", crate::object_db::Value::from(bibitem_id.as_str())),
1207          ("number", crate::object_db::Value::from(entry.number.to_string().as_str())),
1208        ]);
1209      }
1210    }
1211
1212    // Remove any remaining bibentry elements (they've been converted to bibitems)
1213    let bibentries = doc.findnodes("//ltx:bibentry");
1214    if !bibentries.is_empty() {
1215      doc.remove_nodes(&bibentries);
1216    }
1217
1218    // Remove empty biblists
1219    let biblists = doc.findnodes("//ltx:biblist");
1220    let empty_lists: Vec<Node> = biblists.into_iter()
1221      .filter(|n| {
1222        n.get_first_child().map(|c| {
1223          let mut has_element = false;
1224          let mut current = Some(c);
1225          while let Some(ref node) = current {
1226            if node.get_type() == Some(libxml::tree::NodeType::ElementNode) {
1227              has_element = true;
1228              break;
1229            }
1230            current = node.get_next_sibling();
1231          }
1232          !has_element
1233        }).unwrap_or(true)
1234      })
1235      .collect();
1236    if !empty_lists.is_empty() {
1237      doc.remove_nodes(&empty_lists);
1238    }
1239
1240    Ok(vec![doc])
1241  }
1242}

1462/// Apply a formatter function to the given nodes.
1463///
1464/// Port of the various `do_*` functions.
1465fn apply_formatter(doc: &PostDocument, formatter: Formatter, nodes: &[Node]) -> Vec<NodeData> {
1466  match formatter {
1467    Formatter::Any => {
1468      nodes.iter().map(|n| NodeData::Text(n.get_content())).collect()
1469    }
1470    Formatter::Authors => {
1471      format_author_nodes(doc, nodes)
1472    }
1473    Formatter::EditorsA => {
1474      let mut result = format_author_nodes(doc, nodes);
1475      let suffix = if nodes.len() > 1 { " (Eds.)" } else { " (Ed.)" };
1476      result.push(NodeData::Text(suffix.to_string()));
1477      result
1478    }
1479    Formatter::EditorsB => {
1480      let mut result = vec![NodeData::Text("(".to_string())];
1481      result.extend(format_author_nodes(doc, nodes));
1482      let suffix = if nodes.len() > 1 { " Eds.)" } else { " Ed.)" };
1483      result.push(NodeData::Text(suffix.to_string()));
1484      result
1485    }
1486    Formatter::Year => {
1487      let suffix = ""; // Suffix is handled elsewhere
1488      let content: Vec<NodeData> = nodes.iter()
1489        .map(|n| {
1490          let text = n.get_content();
1491          let year = extract_four_digit_year(&text);
1492          NodeData::Text(year)
1493        })
1494        .collect();
1495      let mut result = vec![NodeData::Text(" (".to_string())];
1496      result.extend(content);
1497      result.push(NodeData::Text(format!("{})", suffix)));
1498      result
1499    }
1500    Formatter::Type => {
1501      let content: Vec<NodeData> = nodes.iter()
1502        .map(|n| NodeData::Text(n.get_content()))
1503        .collect();
1504      let mut result = vec![NodeData::Text("(".to_string())];
1505      result.extend(content);
1506      result.push(NodeData::Text(")".to_string()));
1507      result
1508    }
1509    Formatter::Title => {
1510      nodes.iter().map(|n| NodeData::Text(n.get_content())).collect()
1511    }
1512    Formatter::ThesisType => {
1513      nodes.iter().map(|n| NodeData::Text(n.get_content())).collect()
1514    }
1515    Formatter::Edition => {
1516      let mut result: Vec<NodeData> = nodes.iter()
1517        .map(|n| NodeData::Text(n.get_content()))
1518        .collect();
1519      result.push(NodeData::Text(" edition".to_string()));
1520      result
1521    }
1522    Formatter::Pages => {
1523      let mut result = vec![NodeData::Text("pp.\u{00A0}".to_string())]; // Non-breaking space
1524      result.extend(nodes.iter().map(|n| NodeData::Text(n.get_content())));
1525      result
1526    }
1527    Formatter::CrossRef => {
1528      // Port of do_crossref
1529      if let Some(node) = nodes.first() {
1530        if let Some(bibrefs) = node.get_attribute("bibrefs") {
1531          return vec![NodeData::Element {
1532            tag: "ltx:cite".to_string(),
1533            attributes: None,
1534            children: vec![NodeData::Element {
1535              tag: "ltx:bibref".to_string(),
1536              attributes: Some(HashMap::from([
1537                ("bibrefs".to_string(), bibrefs),
1538                ("show".to_string(), "title, author".to_string()),
1539              ])),
1540              children: vec![],
1541            }],
1542          }];
1543        }
1544      }
1545      Vec::new()
1546    }
1547    Formatter::Links => {
1548      format_links(doc, nodes)
1549    }
1550    Formatter::None => Vec::new(),
1551  }
1552}

1554/// Format author name nodes.
1555///
1556/// Port of `do_names` / `do_name`.
1557fn format_author_nodes(_doc: &PostDocument, name_nodes: &[Node]) -> Vec<NodeData> {
1558  let mut result: Vec<NodeData> = Vec::new();
1559  let mut names: Vec<Node> = name_nodes.to_vec();
1560
1561  // Check for "others" sentinel (et al.)
1562  let etal = names.last()
1563    .map(|n| n.get_content().trim() == "others")
1564    .unwrap_or(false);
1565  if etal {
1566    names.pop();
1567  }
1568
1569  let sep = if names.len() > 2 { ", " } else { " " };
1570
1571  for (i, name) in names.iter().enumerate() {
1572    if i > 0 {
1573      result.push(NodeData::Text(sep.to_string()));
1574      if !etal && i == names.len() - 1 {
1575        result.push(NodeData::Text("and ".to_string()));
1576      }
1577    }
1578    // Format single name: initials + surname
1579    if let Some(givenname) = PostDocument::findnodes_foreign("ltx:givenname", name).into_iter().next() {
1580      let given_text = givenname.get_content();
1581      let initials: String = given_text.split_whitespace()
1582        .map(|word| {
1583          if word.ends_with('.') {
1584            format!("{} ", word)
1585          } else if let Some(first) = word.chars().next() {
1586            format!("{}. ", first)
1587          } else {
1588            String::new()
1589          }
1590        })
1591        .collect();
1592      result.push(NodeData::Text(initials));
1593    }
1594    if let Some(surname) = PostDocument::findnodes_foreign("ltx:surname", name).into_iter().next() {
1595      result.push(NodeData::Text(surname.get_content()));
1596    }
1597  }
1598
1599  if etal {
1600    result.push(NodeData::Text(sep.to_string()));
1601    result.push(NodeData::Element {
1602      tag: "ltx:text".to_string(),
1603      attributes: Some(HashMap::from([("class".to_string(), "ltx_bib_etal".to_string())])),
1604      children: vec![NodeData::Text("et al.".to_string())],
1605    });
1606  }
1607
1608  result
1609}

1622
1623    match tag.as_str() {
1624      "ltx:bib-identifier" | "ltx:bib-review" => {
1625        if let Some(href) = href {
1626          links.push(NodeData::Element {
1627            tag: "ltx:ref".to_string(),
1628            attributes: Some(HashMap::from([
1629              ("href".to_string(), href),
1630              ("class".to_string(), format!("{} ltx_bib_external", scheme)),
1631            ])),
1632            children: vec![NodeData::Text(content_text)],
1633          });
1634        } else {
1635          links.push(NodeData::Element {
1636            tag: "ltx:text".to_string(),
1637            attributes: Some(HashMap::from([
1638              ("class".to_string(), format!("{} ltx_bib_external", scheme)),
1639            ])),
1640            children: vec![NodeData::Text(content_text)],
1641          });
1642        }
1643      }
1644      "ltx:bib-links" => {
1645        links.push(NodeData::Element {
1646          tag: "ltx:text".to_string(),
1647          attributes: Some(HashMap::from([
1648            ("class".to_string(), "ltx_bib_external".to_string()),
1649          ])),
1650          children: vec![NodeData::Text(content_text)],
1651        });
1652      }
1653      "ltx:bib-url" => {
1654        if let Some(href) = href {
1655          links.push(NodeData::Element {
1656            tag: "ltx:ref".to_string(),
1657            attributes: Some(HashMap::from([
1658              ("href".to_string(), href),
1659              ("class".to_string(), "ltx_bib_external".to_string()),
1660            ])),
1661            children: vec![NodeData::Text(content_text)],
1662          });
1663        }
1664      }
1665      _ => {}
1666    }
1667  }
1668
1669  // Join with ",\n"
1670  if links.len() > 1 {
1671    let mut result = Vec::new();
1672    for (i, link) in links.into_iter().enumerate() {
1673      if i > 0 {
1674        result.push(NodeData::Text(",\n".to_string()));
1675      }
1676      result.push(link);
1677    }
1678    result
1679  } else {
1680    links
1681  }
1682}

1687/// Extract author names from a bibentry node.
1688///
1689/// Port of the name extraction logic in `getBibEntries`.
1690/// Returns (sort_names, short_names, full_names).
1691fn extract_names(doc: &PostDocument, bibentry: &Node) -> (String, String, String) {
1692  let mut name_nodes: Vec<Node> = PostDocument::findnodes_foreign("ltx:bib-name[@role='author']", bibentry);
1693  if name_nodes.is_empty() {
1694    name_nodes = PostDocument::findnodes_foreign("ltx:bib-name[@role='editor']", bibentry);
1695  }
1696
1697  if name_nodes.is_empty() {
1698    // Try bib-key
1699    if let Some(key_node) = PostDocument::findnodes_foreign("ltx:bib-key", bibentry).into_iter().next() {
1700      let text = key_node.get_content();
1701      return (text.clone(), text.clone(), text);
1702    }
1703    // Try bib-title
1704    if let Some(title_node) = PostDocument::findnodes_foreign("ltx:bib-title", bibentry).into_iter().next() {
1705      let text = title_node.get_content();
1706      return (text.clone(), text.clone(), text);
1707    }
1708    return (String::new(), String::new(), String::new());
1709  }
1710
1711  // Sort names: "Surname Givenname" for each
1712  let sort_names: String = name_nodes.iter()
1713    .map(|n| get_name_text(doc, n))
1714    .collect::<Vec<_>>()
1715    .join(" ");
1716
1717  // Short names: surnames only, with "et al" for >2
1718  let surnames: Vec<String> = name_nodes.iter()
1719    .filter_map(|n| PostDocument::findnodes_foreign("ltx:surname", n).into_iter().next().map(|s| s.get_content()))
1720    .collect();
1721
1722  let short_names = if surnames.len() > 2 {
1723    format!("{} et al", surnames[0])
1724  } else if surnames.len() == 2 {
1725    format!("{} and {}", surnames[0], surnames[1])
1726  } else if !surnames.is_empty() {
1727    surnames[0].clone()
1728  } else {
1729    String::new()
1730  };
1731
1732  let full_names = surnames.join(", ");
1733  (sort_names, short_names, full_names)
1734}

1810

1812    .map(|name| format_single_name(name.trim()))

1821        result.push_str("and ");

1823    }
1824    result.push_str(name);
1825  }
1826  if has_etal {
1827    result.push_str(sep);
1828    result.push_str("et al.");
1829  }
1830  result
1831}

1918/// Parse a raw `.bib` file into BibTeX entries.
1919///
1920/// Handles `@type{key, field = {value}, field = "value", field = number}`.
1921/// Supports nested braces in values and string concatenation with `#`.
1922fn parse_bibtex(input: &str) -> Vec<BibEntry> {
1923  let mut entries = Vec::new();
1924  let chars: Vec<char> = input.chars().collect();
1925  let len = chars.len();
1926  let mut i = 0;
1927
1928  while i < len {
1929    // Skip to next @
1930    if chars[i] != '@' { i += 1; continue; }
1931    i += 1; // skip @
1932
1933    // Read entry type
1934    let type_start = i;
1935    while i < len && chars[i].is_alphanumeric() { i += 1; }
1936    let entry_type = chars[type_start..i].iter().collect::<String>().to_lowercase();
1937
1938    // Skip @string, @preamble, @comment
1939    if entry_type == "string" || entry_type == "preamble" || entry_type == "comment" {
1940      // Skip to matching brace
1941      while i < len && chars[i] != '{' && chars[i] != '(' { i += 1; }
1942      if i < len {
1943        i += 1;
1944        let open = if chars[i - 1] == '{' { '{' } else { '(' };
1945        let close = if open == '{' { '}' } else { ')' };
1946        let mut depth = 1;
1947        while i < len && depth > 0 {
1948          if chars[i] == open { depth += 1; }
1949          if chars[i] == close { depth -= 1; }
1950          i += 1;
1951        }
1952      }
1953      continue;
1954    }
1955
1956    // Skip whitespace
1957    while i < len && chars[i].is_whitespace() { i += 1; }
1958
1959    // Opening brace or paren
1960    if i >= len || (chars[i] != '{' && chars[i] != '(') { continue; }
1961    let close_ch = if chars[i] == '{' { '}' } else { ')' };
1962    i += 1;
1963
1964    // Skip whitespace
1965    while i < len && chars[i].is_whitespace() { i += 1; }
1966
1967    // Read citation key (until comma or whitespace)
1968    let key_start = i;
1969    while i < len && chars[i] != ',' && chars[i] != close_ch && !chars[i].is_whitespace() {
1970      i += 1;
1971    }
1972    let key = chars[key_start..i].iter().collect::<String>().trim().to_string();
1973
1974    // Skip to comma
1975    while i < len && chars[i] != ',' && chars[i] != close_ch { i += 1; }
1976    if i < len && chars[i] == ',' { i += 1; }
1977
1978    // Read fields
1979    let mut fields = Vec::new();
1980    loop {
1981      // Skip whitespace and commas
1982      while i < len && (chars[i].is_whitespace() || chars[i] == ',') { i += 1; }
1983      if i >= len || chars[i] == close_ch { break; }
1984
1985      // Read field name
1986      let fname_start = i;
1987      while i < len && chars[i] != '=' && !chars[i].is_whitespace()
1988        && chars[i] != close_ch { i += 1; }
1989      let fname = chars[fname_start..i].iter().collect::<String>().trim().to_lowercase();
1990
1991      // Skip whitespace and =
1992      while i < len && (chars[i].is_whitespace() || chars[i] == '=') { i += 1; }
1993      if i >= len || chars[i] == close_ch { break; }
1994
1995      // Read field value
1996      let value = read_bib_value(&chars, &mut i, close_ch);
1997      if !fname.is_empty() {
1998        fields.push((fname, value));
1999      }
2000
2001      // Skip trailing comma
2002      while i < len && chars[i].is_whitespace() { i += 1; }
2003      if i < len && chars[i] == ',' { i += 1; }
2004    }
2005
2006    // Skip closing brace
2007    if i < len && chars[i] == close_ch { i += 1; }
2008
2009    if !key.is_empty() {
2010      entries.push(BibEntry { entry_type, key, fields });
2011    }
2012  }
2013
2014  entries
2015}

2017/// Read a BibTeX field value (braced, quoted, or bare number/string).
2018fn read_bib_value(chars: &[char], i: &mut usize, _entry_close: char) -> String {
2019  let len = chars.len();
2020  let mut result = String::new();
2021
2022  loop {
2023    while *i < len && chars[*i].is_whitespace() { *i += 1; }
2024    if *i >= len { break; }
2025
2026    if chars[*i] == '{' {
2027      // Braced value — handle nested braces
2028      *i += 1;
2029      let mut depth = 1;
2030      while *i < len && depth > 0 {
2031        if chars[*i] == '{' { depth += 1; }
2032        else if chars[*i] == '}' {
2033          depth -= 1;
2034          if depth == 0 { *i += 1; break; }
2035        }
2036        result.push(chars[*i]);
2037        *i += 1;
2038      }
2039    } else if chars[*i] == '"' {
2040      // Quoted value
2041      *i += 1;
2042      while *i < len && chars[*i] != '"' {
2043        if chars[*i] == '{' {
2044          // Nested braces in quoted strings
2045          *i += 1;
2046          let mut depth = 1;
2047          while *i < len && depth > 0 {
2048            if chars[*i] == '{' { depth += 1; }
2049            else if chars[*i] == '}' { depth -= 1; if depth == 0 { *i += 1; break; } }
2050            result.push(chars[*i]);
2051            *i += 1;
2052          }
2053        } else {
2054          result.push(chars[*i]);
2055          *i += 1;
2056        }
2057      }
2058      if *i < len && chars[*i] == '"' { *i += 1; }
2059    } else if chars[*i].is_alphanumeric() {
2060      // Bare word or number
2061      while *i < len && (chars[*i].is_alphanumeric() || chars[*i] == '-' || chars[*i] == '_') {
2062        result.push(chars[*i]);
2063        *i += 1;
2064      }
2065    } else {
2066      break;
2067    }
2068
2069    // Check for # concatenation
2070    while *i < len && chars[*i].is_whitespace() { *i += 1; }
2071    if *i < len && chars[*i] == '#' {
2072      *i += 1;
2073      continue;
2074    }
2075    break;
2076  }
2077
2078  result
2079}

2081/// Strip outer braces from a BibTeX field value.
2082/// "{My Title}" → "My Title"
2083fn strip_braces(s: &str) -> String {
2084  let mut result = String::with_capacity(s.len());
2085  for c in s.chars() {
2086    if c != '{' && c != '}' {
2087      result.push(c);
2088    }
2089  }
2090  result

2093/// Parse BibTeX author field into individual author names.
2094/// "Lastname, Firstname and Lastname2, Firstname2" → vec of (surname, givenname)
2095fn parse_bib_authors(authors_str: &str) -> Vec<(String, String)> {
2096  let mut result = Vec::new();
2097  // Split on " and " (case insensitive)
2098  let parts: Vec<&str> = authors_str.split(" and ").collect();
2099  for part in parts {
2100    let part = part.trim();
2101    if part.is_empty() { continue; }
2102    let clean = strip_braces(part);
2103    if let Some((surname, given)) = clean.split_once(',') {
2104      result.push((surname.trim().to_string(), given.trim().to_string()));
2105    } else {
2106      // "Firstname Lastname" format — last word is surname
2107      let words: Vec<&str> = clean.split_whitespace().collect();
2108      if words.len() >= 2 {
2109        let surname = words.last().unwrap().to_string();
2110        let given = words[..words.len()-1].join(" ");
2111        result.push((surname, given));
2112      } else if words.len() == 1 {
2113        result.push((words[0].to_string(), String::new()));
2114      }
2115    }
2116  }
2117  result
2118}
