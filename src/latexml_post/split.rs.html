1//! Document splitting processor.
2//!
3//! Port of `LaTeXML::Post::Split`.
4//! Splits a document into multiple pages based on an XPath expression
5//! that identifies section-level elements to extract as separate documents.

14/// Page naming strategy for split documents.
15#[derive(Debug, Clone)]
16pub enum SplitNaming {
17  /// Use xml:id attribute
18  Id,
19  /// Use xml:id, relative to parent
20  IdRelative,
21  /// Use labels attribute
22  Label,
23  /// Use labels, relative to parent
24  LabelRelative,
25}
26
27/// A tree node used to track the hierarchy of pages during splitting.
28struct PageEntry {
29  node:     Node,
30  id:       Option<String>,
31  upid:     Option<String>,
32  name:     String,
33  children: Vec<PageEntry>,
34  document: Option<PostDocument>,
35}

37/// Split post-processor: splits a document into multiple pages.
38///
39/// Port of `LaTeXML::Post::Split`.
40pub struct Split {
41  name:                 String,
42  /// XPath expression to find elements that become pages.
43  split_xpath:          String,
44  /// Naming strategy for page files.
45  split_naming:         SplitNaming,
46  /// Whether to suppress navigation links.
47  no_navigation:        bool,
48  /// Counter for unnamed pages.
49  unnamed_page_counter: u32,
50}
51
52impl Split {
53  pub fn new(split_xpath: &str, split_naming: SplitNaming, no_navigation: bool) -> Self {
54    Split {
55      name: "Split".to_string(),
56      split_xpath: split_xpath.to_string(),
57      split_naming,
58      no_navigation,
59      unnamed_page_counter: 0,
60    }
61  }

63  /// Get the nodes that will become separate pages.
64  fn get_pages(&self, doc: &PostDocument) -> Vec<Node> { doc.findnodes(&self.split_xpath) }

66  /// Generate a name for an unnamed page.
67  fn generate_unnamed_page_name(&mut self) -> String {
68    self.unnamed_page_counter += 1;
69    format!("FOO{}", self.unnamed_page_counter)
70  }

72  /// Sort pages into a tree hierarchy.
73  ///
74  /// Port of Perl `presortPages`.
75  /// If a page is a descendant of another page, it becomes a child in the tree.
76  fn presort_pages(
77    tree: &mut PageEntry,
78    haschildren: &mut HashMap<String, bool>,
79    pages: Vec<Node>,
80  ) {
81    // We track the "current" position in the tree by maintaining a path of ancestors.
82    // Since we can't have mutable borrows at multiple tree levels simultaneously,
83    // we use an index-based approach to walk the tree.
84    let mut path: Vec<usize> = Vec::new(); // indices into children arrays
85
86    for page in pages {
87      // Walk back up the tree until we find an ancestor of `page`
88      loop {
89        let current_node = Self::get_node_at(tree, &path);
90        if is_child(&page, &current_node) {
91          break;
92        }
93        if path.is_empty() {
94          break;
95        }
96        path.pop();
97      }
98
99      let current_node = Self::get_node_at(tree, &path);
100      let current_id = get_xml_id(&current_node);
101      let localname = current_node.get_name();
102      haschildren.insert(localname, true);
103
104      let page_id = get_xml_id(&page);
105      let entry = PageEntry {
106        node:     page,
107        id:       page_id,
108        upid:     current_id,
109        name:     String::new(),
110        children: Vec::new(),
111        document: None,
112      };
113
114      // Add as child of current position
115      let parent = Self::get_entry_at_mut(tree, &path);
116      parent.children.push(entry);
117      let new_idx = parent.children.len() - 1;
118
119      // Go "down" — following pages may be children of this one
120      path.push(new_idx);
121    }
122  }
123
124  /// Get the node at a given path in the tree.
125  fn get_node_at(tree: &PageEntry, path: &[usize]) -> Node {
126    let mut current = tree;
127    for &idx in path {
128      current = &current.children[idx];
129    }
130    current.node.clone()
131  }
132
133  /// Get a mutable reference to the entry at a given path.
134  fn get_entry_at_mut<'a>(tree: &'a mut PageEntry, path: &[usize]) -> &'a mut PageEntry {
135    let mut current = tree;
136    for &idx in path {
137      current = &mut current.children[idx];
138    }
139    current
140  }

142  /// Compute destination pathnames for each page in the tree.
143  ///
144  /// Port of Perl `prenamePages`.
145  fn prename_pages(
146    &mut self,
147    doc: &PostDocument,
148    tree: &mut PageEntry,
149    haschildren: &HashMap<String, bool>,
150  ) {
151    for i in 0..tree.children.len() {
152      let (parent_name, parent_node) = (tree.name.clone(), tree.node.clone());
153      let child = &tree.children[i];
154      let child_localname = child.node.get_name();
155      let recursive = haschildren.get(&child_localname).copied().unwrap_or(false);
156      let name = self.get_page_name(doc, &child.node, &parent_node, &parent_name, recursive);
157      tree.children[i].name = name;
158    }
159    // Recurse into children
160    for child in &mut tree.children {
161      self.prename_pages(doc, child, haschildren);
162    }
163  }

165  /// Process a sequence of page entries, removing them from the document
166  /// and generating sub-documents for each.
167  ///
168  /// Port of Perl `processPages`.
169  fn process_pages(
170    &mut self,
171    doc: &mut PostDocument,
172    entries: &mut Vec<PageEntry>,
173  ) -> Vec<PostDocument> {
174    // Before any document surgery, copy inheritable attributes.
175    let mut intoc = false;
176    for entry in entries.iter() {
177      let node = &entry.node;
178      if let Some(inlist) = node.get_attribute("inlist") {
179        if inlist.contains("toc") {
180          intoc = true;
181        }
182      }
183      // Copy xml:lang and backgroundcolor from ancestors
184      for attr in &["xml:lang", "backgroundcolor"] {
185        let xpath = format!("ancestor-or-self::*[@{}][1]", attr);
186        if let Some(anc) = doc.findnode_at(&xpath, node) {
187          if let Some(val) = anc.get_attribute(attr) {
188            let mut node_mut = node.clone();
189            node_mut.set_attribute(attr, &val).ok();
190          }
191        }
192      }
193    }
194
195    let mut docs = Vec::new();
196    while !entries.is_empty() {
197      let parent = match entries[0].node.get_parent() {
198        Some(p) => p,
199        None => {
200          entries.remove(0);
201          continue;
202        },
203      };
204
205      // Remove page & ALL following siblings (backwards).
206      let mut removed: Vec<Node> = Vec::new();
207      loop {
208        let last = parent.get_last_child();
209        match last {
210          Some(mut sib) => {
211            sib.unlink_node();
212            removed.insert(0, sib.clone());
213            if sib == entries[0].node {
214              break;
215            }
216          },
217          None => break,
218        }
219      }
220
221      // Build TOC from adjacent nodes being extracted.
222      let mut toc: Vec<NodeData> = Vec::new();
223
224      // Process a sequence of adjacent pages that share the same parent.
225      while !entries.is_empty() && !removed.is_empty() && entries[0].node == removed[0] {
226        let mut entry = entries.remove(0);
227        let page = entry.node.clone();
228
229        // If any pages go in toc, assume siblings should too
230        if intoc && page.get_attribute("inlist").is_none() {
231          let mut page_mut = page.clone();
232          page_mut.set_attribute("inlist", "toc").ok();
233        }
234
235        // Remove this page from the removed list and from the document's ID cache
236        let removed_node = removed.remove(0);
237        doc.remove_nodes(&[removed_node]);
238
239        // Build TOC entry
240        if let Some(id) = get_xml_id(&page) {
241          let mut toc_attrs = HashMap::new();
242          toc_attrs.insert("idref".to_string(), id);
243          toc_attrs.insert("show".to_string(), "toctitle".to_string());
244          let tocentry = NodeData::Element {
245            tag:        "ltx:tocentry".to_string(),
246            attributes: None,
247            children:   vec![NodeData::Element {
248              tag:        "ltx:ref".to_string(),
249              attributes: Some(toc_attrs),
250              children:   vec![],
251            }],
252          };
253          toc.push(tocentry);
254        }
255
256        // Process children pages BEFORE this page (Perl: "Due to the way document building works")
257        let mut child_docs = self.process_pages(doc, &mut entry.children);
258
259        // Create sub-document from the extracted page
260        let subdoc = doc.new_document(page, &entry.name);
261        entry.document = Some(subdoc);
262        // Take the document out to push to the results
263        docs.push(entry.document.take().unwrap());
264        docs.append(&mut child_docs);
265      }
266
267      // Add TOC to reflect the extracted pages
268      if !toc.is_empty() {
269        // Only add if parent doesn't already have a TOC with lists='toc'
270        let has_toc = !doc
271          .findnodes_at("descendant::ltx:TOC[@lists='toc']", Some(&parent))
272          .is_empty();
273        if !has_toc {
274          let parent_type = parent.get_name();
275          let mut toclist_attrs = HashMap::new();
276          toclist_attrs.insert("class".to_string(), format!("ltx_toclist_{}", parent_type));
277          let toc_node = NodeData::Element {
278            tag:        "ltx:TOC".to_string(),
279            attributes: None,
280            children:   vec![NodeData::Element {
281              tag:        "ltx:toclist".to_string(),
282              attributes: Some(toclist_attrs),
283              children:   toc,
284            }],
285          };
286          let mut parent_mut = parent.clone();
287          doc.add_nodes(&mut parent_mut, &[toc_node]);
288        }
289      }
290
291      // Re-add remaining siblings
292      let mut parent_mut = parent;
293      for mut child in removed {
294        parent_mut.add_child(&mut child).ok();
295      }
296    }
297    docs
298  }

300  /// Add navigation elements to all documents in the tree.
301  ///
302  /// Port of Perl `addNavigation`.
303  fn add_navigation(entry: &mut PageEntry, nav_nodes: &[Node]) {
304    if let Some(ref mut doc) = entry.document {
305      if let Some(mut root) = doc.get_document_element() {
306        let nav_data: Vec<NodeData> = nav_nodes
307          .iter()
308          .map(|n| NodeData::XmlNode(n.clone()))
309          .collect();
310        doc.add_nodes(&mut root, &nav_data);
311      }
312    }
313    for child in &mut entry.children {
314      Self::add_navigation(child, nav_nodes);
315    }
316  }

318  /// Compute the destination pathname for a page.
319  ///
320  /// Port of `Split::getPageName`.
321  fn get_page_name(
322    &mut self,
323    doc: &PostDocument,
324    page: &Node,
325    parent: &Node,
326    parent_path: &str,
327    recursive: bool,
328  ) -> String {
329    let attr = match self.split_naming {
330      SplitNaming::Id | SplitNaming::IdRelative => "xml:id",
331      SplitNaming::Label | SplitNaming::LabelRelative => "labels",
332    };
333
334    let mut name = if attr == "xml:id" {
335      get_xml_id(page).unwrap_or_default()
336    } else {
337      page.get_attribute(attr).unwrap_or_default()
338    };
339
340    // Truncate to first label, strip LABEL: prefix
341    if let Some(first) = name.split_whitespace().next() {
342      name = first.to_string();
343    }
344    if let Some(stripped) = name.strip_prefix("LABEL:") {
345      name = stripped.to_string();
346    }
347
348    if name.is_empty() {
349      if attr == "labels" {
350        if let Some(id) = get_xml_id(page) {
351          log::info!(
352            "Using '{}' to create page pathname, instead of missing '{}'",
353            id,
354            attr
355          );
356          name = id;
357        } else {
358          name = self.generate_unnamed_page_name();
359          log::info!(
360            "Using '{}' to create page pathname, instead of missing '{}'",
361            name,
362            attr
363          );
364        }
365      } else {
366        name = self.generate_unnamed_page_name();
367        log::info!(
368          "Using '{}' to create page pathname, instead of missing '{}'",
369          name,
370          attr
371        );
372      }
373    }
374
375    // Relative naming: strip parent prefix
376    let as_dir = match self.split_naming {
377      SplitNaming::IdRelative | SplitNaming::LabelRelative => {
378        let parent_attr = if attr == "xml:id" {
379          get_xml_id(parent)
380        } else {
381          parent.get_attribute(attr)
382        };
383        if let Some(pname) = parent_attr {
384          let pname = pname.split_whitespace().next().unwrap_or("");
385          let pname = pname.strip_prefix("LABEL:").unwrap_or(pname);
386          if let Some(rest) = name.strip_prefix(pname) {
387            let rest = rest.trim_start_matches(['.', '_', ':']);
388            if !rest.is_empty() {
389              name = rest.to_string();
390            }
391          }
392        }
393        recursive
394      },
395      _ => false,
396    };
397
398    // Sanitize colons
399    name = name.replace(':', "_");
400
401    let ext = doc
402      .get_destination_extension()
403      .unwrap_or_else(|| "xml".to_string());
404    let parent_dir = Path::new(parent_path)
405      .parent()
406      .and_then(|p| p.to_str())
407      .unwrap_or(".");
408
409    // Normalize empty parent_dir to "."
410    let parent_dir = if parent_dir.is_empty() {
411      "."
412    } else {
413      parent_dir
414    };
415
416    if as_dir {
417      format!("{}/{}/index.{}", parent_dir, name, ext)
418    } else {
419      format!("{}/{}.{}", parent_dir, name, ext)
420    }
421  }

424impl Processor for Split {
425  fn get_name(&self) -> &str { &self.name }
426
427  fn process(&mut self, mut doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
428    let root = match nodes.into_iter().next() {
429      Some(r) => r,
430      None => return Ok(vec![doc]),
431    };
432
433    // Ensure root has an ID (Writer will remove TEMPORARY_DOCUMENT_ID)
434    let mut root_mut = root;
435    if get_xml_id(&root_mut).is_none() {
436      root_mut
437        .set_attribute("xml:id", "TEMPORARY_DOCUMENT_ID")
438        .ok();
439    }
440
441    let pages = self.get_pages(&doc);
442    // Filter out the root node itself (Perl: grep { $_->parentNode->parentNode })
443    let pages: Vec<Node> = pages
444      .into_iter()
445      .filter(|p| p.get_parent().and_then(|pp| pp.get_parent()).is_some())
446      .collect();
447
448    if pages.is_empty() {
449      log::info!("[not split]");
450      return Ok(vec![doc]);
451    }
452
453    // Save and remove navigation elements
454    let nav_nodes: Vec<Node> = doc.findnodes("descendant::ltx:navigation");
455    if !nav_nodes.is_empty() {
456      doc.remove_nodes(&nav_nodes);
457    }
458
459    // Build the page tree
460    let root_id = get_xml_id(&root_mut);
461    let root_dest = doc.get_destination().unwrap_or("").to_string();
462    let mut tree = PageEntry {
463      node:     root_mut,
464      id:       root_id,
465      upid:     None,
466      name:     root_dest,
467      children: Vec::new(),
468      document: Some(doc),
469    };
470
471    let mut haschildren = HashMap::new();
472    Self::presort_pages(&mut tree, &mut haschildren, pages);
473
474    // Take doc out so we can pass &PostDocument and &mut tree without borrow conflict
475    let doc_tmp = tree.document.take().unwrap();
476    self.prename_pages(&doc_tmp, &mut tree, &haschildren);
477    tree.document = Some(doc_tmp);
478
479    // Process pages: extract and create sub-documents
480    let mut doc = tree.document.take().unwrap();
481    let mut docs = vec![];
482    let mut child_docs = self.process_pages(&mut doc, &mut tree.children);
483
484    // Restore navigation to all documents
485    if !nav_nodes.is_empty() && !self.no_navigation {
486      // Put doc back into tree for navigation distribution
487      tree.document = Some(doc);
488      Self::add_navigation(&mut tree, &nav_nodes);
489      doc = tree.document.take().unwrap();
490
491      // Also add nav to child docs
492      for child_doc in &mut child_docs {
493        if let Some(mut root) = child_doc.get_document_element() {
494          let nav_data: Vec<NodeData> = nav_nodes
495            .iter()
496            .map(|n| NodeData::XmlNode(n.clone()))
497            .collect();
498          child_doc.add_nodes(&mut root, &nav_data);
499        }
500      }
501    }
502
503    docs.insert(0, doc);
504    docs.append(&mut child_docs);
505
506    let n = docs.len();
507    log::info!(
508      "{}",
509      if n > 1 {
510        format!(" [Split into {} pages]", n)
511      } else {
512        "[not split]".to_string()
513      }
514    );
515
516    Ok(docs)
517  }
518}

520/// Check if `child` is a descendant of `ancestor`.
521fn is_child(child: &Node, ancestor: &Node) -> bool {
522  let mut parent = child.get_parent();
523  while let Some(ref p) = parent {
524    if *p == *ancestor {
525      return true;
526    }
527    parent = p.get_parent();
528  }
529  false
