1//! Index generation processor.
2//!
3//! Port of `LaTeXML::Post::MakeIndex` (504 lines of Perl).
4//! Collects INDEX:* entries from the ObjectDB, builds a tree of index entries
5//! grouped by initial letter, and fills in `ltx:index` and `ltx:glossary` elements.
6//! Supports permuted indexes, splitting by initial, see-also references,
7//! range-style page references, and glossary entry formatting.
8

17/// Index tree node.
18#[derive(Debug)]
19struct IndexTree {
20  id:               String,
21  key:              Option<String>,
22  full_key:         Option<String>,
23  phrase:           Option<String>,
24  phrase_text:      Option<String>,
25  full_phrase_text: Option<String>,
26  subtrees:         HashMap<String, IndexTree>,
27  referrers:        HashMap<String, HashMap<String, bool>>,
28  see_also:         Vec<String>,
29}

40      subtrees:         HashMap::new(),

54/// MakeIndex post-processor.
55///

64impl MakeIndex {
65  pub fn new(db: ObjectDB, split: bool, permuted: bool) -> Self {
66    MakeIndex {
67      name: "MakeIndex".to_string(),
68      db,
69      split,
70      permuted,
71    }
72  }

74  /// Build the index tree from ObjectDB INDEX:* entries.
75  ///
76  /// Port of `MakeIndex::build_tree`.
77  fn build_tree(&self, index_id: &str) -> Option<(IndexTree, HashMap<String, String>)> {
78    let keys: Vec<String> = self
79      .db
80      .get_keys()
81      .into_iter()
82      .filter(|k| k.starts_with("INDEX:"))
83      .cloned()
84      .collect();
85    if keys.is_empty() {
86      return None;
87    }
88    log::info!("MakeIndex: {} entries", keys.len());
89
90    let mut all_phrases: HashMap<String, String> = HashMap::new();
91    let mut tree = IndexTree::new(index_id);
92
93    for key in &keys {
94      if let Some(entry) = self.db.lookup(key) {
95        let phrase_keys: Vec<&str> = key
96          .strip_prefix("INDEX:")
97          .unwrap_or("")
98          .split(':')
99          .filter(|s| !s.is_empty())
100          .collect();
101        if phrase_keys.is_empty() {
102          continue;
103        }
104
105        if self.permuted {
106          // Cyclic permutations of phrase keys
107          for perm in cyclic_permute(&phrase_keys) {
108            if self.split {
109              let init = initial_letter(perm[0]);
110              let subtree = tree.subtrees.entry(init.clone()).or_insert_with(|| {
111                let mut st = IndexTree::new(&tree.id);
112                st.phrase = Some(init);
113                st
114              });
115              add_tree_rec(subtree, &perm, &mut all_phrases, entry);
116            } else {
117              add_tree_rec(&mut tree, &perm, &mut all_phrases, entry);
118            }
119          }
120        } else if self.split {
121          let init = initial_letter(phrase_keys[0]);
122          let subtree = tree.subtrees.entry(init.clone()).or_insert_with(|| {
123            let mut st = IndexTree::new(&tree.id);
124            st.phrase = Some(init);
125            st
126          });
127          add_tree_rec(subtree, &phrase_keys, &mut all_phrases, entry);
128        } else {
129          add_tree_rec(&mut tree, &phrase_keys, &mut all_phrases, entry);
130        }
131      }
132    }
133    Some((tree, all_phrases))
134  }

136  /// Generate XML for an index list from a tree.
137  fn make_index_list(
138    &self,
139    all_phrases: &HashMap<String, String>,
140    tree: &IndexTree,
141  ) -> Option<NodeData> {
142    if tree.subtrees.is_empty() {
143      return None;
144    }
145    let mut sorted_keys: Vec<&String> = tree.subtrees.keys().collect();
146    sorted_keys.sort_by(|a, b| a.to_lowercase().cmp(&b.to_lowercase()).then(a.cmp(b)));
147
148    let entries: Vec<NodeData> = sorted_keys
149      .iter()
150      .filter_map(|key| {
151        tree
152          .subtrees
153          .get(*key)
154          .map(|st| self.make_index_entry(all_phrases, st))
155      })
156      .collect();
157    if entries.is_empty() {
158      return None;
159    }
160    Some(NodeData::Element {
161      tag:        "ltx:indexlist".to_string(),
162      attributes: None,
163      children:   entries,
164    })
165  }

167  /// Generate a single index entry with sub-entries, references, and see-also.
168  ///
169  /// Port of `MakeIndex::makeIndexEntry`.
170  fn make_index_entry(&self, all_phrases: &HashMap<String, String>, tree: &IndexTree) -> NodeData {
171    let mut children = Vec::new();
172
173    // Phrase
174    if let Some(ref phrase) = tree.phrase {
175      children.push(NodeData::Element {
176        tag:        "ltx:indexphrase".to_string(),
177        attributes: tree
178          .key
179          .as_ref()
180          .map(|k| HashMap::from([("key".to_string(), k.clone())])),
181        children:   vec![NodeData::Text(phrase.clone())],
182      });
183    }
184
185    // Referrer links (combined with range handling)
186    let mut links = Vec::new();
187    if !tree.referrers.is_empty() {
188      links.extend(self.combine_index_entries(&tree.referrers));
189    }
190
191    // See-also links
192    for see_text in &tree.see_also {
193      if !links.is_empty() {
194        links.push(NodeData::Text(", ".to_string()));
195      }
196      // Try to find the referenced phrase in allphrases
197      if let Some(target_id) = all_phrases
198        .get(see_text)
199        .or_else(|| all_phrases.get(&see_text.to_lowercase()))
200      {
201        links.push(NodeData::Element {
202          tag:        "ltx:ref".to_string(),
203          attributes: Some(HashMap::from([("idref".to_string(), target_id.clone())])),
204          children:   vec![NodeData::Text(see_text.clone())],
205        });
206      } else {
207        links.push(NodeData::Element {
208          tag:        "ltx:text".to_string(),
209          attributes: Some(HashMap::from([("font".to_string(), "italic".to_string())])),
210          children:   vec![NodeData::Text(see_text.clone())],
211        });
212      }
213    }
214
215    if !links.is_empty() {
216      children.push(NodeData::Element {
217        tag:        "ltx:indexrefs".to_string(),
218        attributes: None,
219        children:   links,
220      });
221    }
222
223    // Sub-entries
224    if let Some(sublist) = self.make_index_list(all_phrases, tree) {
225      children.push(sublist);
226    }
227
228    NodeData::Element {
229      tag: "ltx:indexentry".to_string(),
230      attributes: if tree.id.is_empty() {
231        None
232      } else {
233        Some(HashMap::from([("xml:id".to_string(), tree.id.clone())]))
234      },
235      children,
236    }
237  }

239  /// Combine index entry referrers into a comma-separated list with range support.
240  ///
241  /// Port of `combineIndexEntries`.
242  fn combine_index_entries(&self, refs: &HashMap<String, HashMap<String, bool>>) -> Vec<NodeData> {
243    let mut ids: Vec<&String> = refs.keys().collect();
244    ids.sort_by(|a, b| a.to_lowercase().cmp(&b.to_lowercase()).then(a.cmp(b)));
245
246    let mut links = Vec::new();
247    let mut i = 0;
248    while i < ids.len() {
249      if !links.is_empty() {
250        links.push(NodeData::Text(", ".to_string()));
251      }
252      let id = ids[i];
253      let styles = refs.get(id).unwrap();
254
255      // Check for range start
256      if styles.contains_key("rangestart") {
257        let start_id = id;
258        let mut end_id = id;
259        let mut level = 1i32;
260        i += 1;
261        while i < ids.len() && level > 0 {
262          end_id = ids[i];
263          if let Some(s) = refs.get(end_id) {
264            if s.contains_key("rangestart") {
265              level += 1;
266            }
267            if s.contains_key("rangeend") {
268              level -= 1;
269            }
270          }
271          i += 1;
272        }
273        // Range: start–end
274        links.push(NodeData::Element {
275          tag:        "ltx:text".to_string(),
276          attributes: None,
277          children:   vec![
278            self.make_index_ref(start_id, styles),
279            NodeData::Text("\u{2014}".to_string()), // em-dash
280            self.make_index_ref(end_id, refs.get(end_id).unwrap_or(styles)),
281          ],
282        });
283      } else {
284        links.push(self.make_index_ref(id, styles));
285        i += 1;
286      }
287    }
288    links
289  }

291  /// Make a single index reference link.
292  ///
293  /// Port of `makeIndexRefs`.
294  fn make_index_ref(&self, id: &str, styles: &HashMap<String, bool>) -> NodeData {
295    let style: Vec<&String> = styles
296      .keys()
297      .filter(|s| *s != "rangestart" && *s != "rangeend")
298      .collect();
299    let primary_style = style.first().map(|s| s.as_str()).unwrap_or("normal");
300
301    let ref_node = NodeData::Element {
302      tag:        "ltx:ref".to_string(),
303      attributes: Some(HashMap::from([
304        ("idref".to_string(), id.to_string()),
305        ("show".to_string(), "typerefnum".to_string()),
306      ])),
307      children:   vec![],
308    };
309
310    if primary_style != "normal" {
311      NodeData::Element {
312        tag:        "ltx:text".to_string(),
313        attributes: Some(HashMap::from([(
314          "font".to_string(),
315          primary_style.to_string(),
316        )])),
317        children:   vec![ref_node],
318      }
319    } else {
320      ref_node
321    }
322  }

324  /// Get glossary entries from the ObjectDB.
325  ///
326  /// Port of `MakeIndex::getGlossaryEntries`.
327  fn get_glossary_entries(&self, lists: &str, glossary_id: &str) -> Vec<GlossaryEntry> {
328    let list_set: std::collections::HashSet<&str> = lists.split(',').collect();
329    let mut entries = Vec::new();
330
331    for db_key in self.db.get_keys() {
332      if !db_key.starts_with("GLOSSARY:") {
333        continue;
334      }
335      let parts: Vec<&str> = db_key.splitn(3, ':').collect();
336      if parts.len() < 3 {
337        continue;
338      }
339      let list = parts[1];
340      let key = parts[2];
341      if !list_set.contains(list) {
342        continue;
343      }
344
345      if let Some(entry) = self.db.lookup(db_key) {
346        // Check if it has referrers
347        let has_refs = entry
348          .get_value("referrers")
349          .map(|v| v.is_truthy())
350          .unwrap_or(false);
351        if !has_refs {
352          continue;
353        }
354
355        let sort_key = entry.get_string("phrase:sort").unwrap_or(key).to_string();
356        let initial = sort_key
357          .chars()
358          .next()
359          .filter(|c| c.is_ascii_alphabetic())
360          .map(|c| c.to_uppercase().to_string())
361          .unwrap_or_else(|| "*".to_string());
362        let id = format!("{}.{}", glossary_id, key);
363        let term = entry.get_string("phrase:name").unwrap_or(key).to_string();
364        let desc = entry
365          .get_string("phrase:description")
366          .unwrap_or("")
367          .to_string();
368
369        entries.push(GlossaryEntry {
370          initial,
371          sort_key,
372          formatted: NodeData::Element {
373            tag:        "ltx:glossaryentry".to_string(),
374            attributes: Some(HashMap::from([
375              ("lists".to_string(), lists.to_string()),
376              ("xml:id".to_string(), id),
377              ("key".to_string(), key.to_string()),
378            ])),
379            children:   vec![
380              NodeData::Element {
381                tag:        "ltx:glossaryphrase".to_string(),
382                attributes: Some(HashMap::from([
383                  ("role".to_string(), "label".to_string()),
384                  ("key".to_string(), key.to_string()),
385                ])),
386                children:   vec![NodeData::Text(term)],
387              },
388              NodeData::Element {
389                tag:        "ltx:glossaryphrase".to_string(),
390                attributes: Some(HashMap::from([(
391                  "role".to_string(),
392                  "definition".to_string(),
393                )])),
394                children:   vec![NodeData::Text(desc)],
395              },
396            ],
397          },
398        });
399      }
400    }
401    entries.sort_by(|a, b| a.sort_key.cmp(&b.sort_key));
402    entries
403  }

405  /// Generate a glossary list.
406  ///
407  /// Port of `MakeIndex::makeGlossaryList`.
408  fn make_glossary_list(&self, entries: &[GlossaryEntry]) -> NodeData {

417impl Processor for MakeIndex {
418  fn get_name(&self) -> &str { &self.name }
419
420  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
421    doc.findnodes("//ltx:index[not(ltx:indexlist)] | //ltx:glossary[not(ltx:glossarylist)]")
422  }
423
424  fn process(&mut self, mut doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
425    for node in &nodes {
426      let tag = doc.get_qname(node).unwrap_or_default();
427      let id = node.get_attribute("xml:id").unwrap_or_default();
428
429      if tag == "ltx:index" {
430        if let Some((tree, all_phrases)) = self.build_tree(&id) {
431          if let Some(index_list) = self.make_index_list(&all_phrases, &tree) {
432            let mut node_mut = node.clone();
433            doc.add_nodes(&mut node_mut, &[index_list]);
434          }
435        }
436      } else if tag == "ltx:glossary" {
437        let lists = node
438          .get_attribute("lists")
439          .unwrap_or_else(|| "glossary".to_string());
440        let entries = self.get_glossary_entries(&lists, &id);
441        if !entries.is_empty() {
442          let glist = self.make_glossary_list(&entries);
443          let mut node_mut = node.clone();
444          doc.add_nodes(&mut node_mut, &[glist]);
445        }
446      }
447    }
448    Ok(vec![doc])
449  }
450}

455fn add_tree_rec(
456  tree: &mut IndexTree,
457  phrase_keys: &[&str],
458  all_phrases: &mut HashMap<String, String>,
459  entry: &crate::object_db::Entry,
460) {
461  if phrase_keys.is_empty() {
462    // Leaf: record referrers and see_also
463    if let Some(Value::Hash(refs)) = entry.get_value("referrers") {
464      for (k, v) in refs {
465        let styles = tree.referrers.entry(k.clone()).or_default();
466        styles.insert(v.to_string(), true);
467      }
468    }
469    if let Some(Value::List(see_items)) = entry.get_value("see_also") {
470      for item in see_items {
471        tree.see_also.push(item.to_string());
472      }
473    }
474    return;
475  }
476  let key = phrase_keys[0];
477  let rest = &phrase_keys[1..];
478  let key_id = get_index_key_id(key);
479  let parent_key = tree.full_key.clone().unwrap_or_default();
480  let full_key = if parent_key.is_empty() {
481    key.to_string()
482  } else {
483    format!("{}.{}", parent_key, key)
484  };
485  let parent_phrase = tree.full_phrase_text.clone().unwrap_or_default();
486  let full_phrase = if parent_phrase.is_empty() {
487    key.to_string()
488  } else {
489    format!("{} {}", parent_phrase, key)
490  };
491
492  let tree_id = tree.id.clone();
493  let subtree = tree.subtrees.entry(key.to_string()).or_insert_with(|| {
494    let id = format!("{}.{}", tree_id, key_id);
495    all_phrases.insert(full_key.clone(), id.clone());
496    all_phrases.insert(full_key.to_lowercase(), id.clone());
497    all_phrases.insert(full_phrase.clone(), id.clone());
498    all_phrases.insert(full_phrase.to_lowercase(), id.clone());
499    let mut st = IndexTree::new(&id);
500    st.key = Some(key.to_string());
501    st.full_key = Some(full_key.clone());
502    st.phrase = Some(key.to_string());
503    st.phrase_text = Some(key.to_string());
504    st.full_phrase_text = Some(full_phrase);
505    st
506  });
507  add_tree_rec(subtree, rest, all_phrases, entry);
508}

510fn initial_letter(key: &str) -> String {
511  let decomposed: String = key.nfd().collect();
512  match decomposed.trim().chars().next() {
513    Some(c) if c.is_ascii_alphabetic() => c.to_uppercase().to_string(),
514    _ => "*".to_string(),
515  }
516}
517
518fn get_index_key_id(key: &str) -> String {
519  key
520    .nfd()
521    .collect::<String>()
522    .chars()
523    .filter(|c| c.is_ascii_alphanumeric())
524    .collect()
525}

527/// Generate cyclic permutations of a slice.
528fn cyclic_permute<T: Clone>(items: &[T]) -> Vec<Vec<T>> {
529  if items.len() <= 1 {
530    return vec![items.to_vec()];
531  }
532  (0..items.len())
533    .map(|i| {
534      let mut perm = items[i..].to_vec();
535      perm.extend_from_slice(&items[..i]);
536      perm
537    })
538    .collect()
539}
