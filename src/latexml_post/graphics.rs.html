175  /// Find the graphics source file for a node.
176  ///
177  /// Port of `Graphics::findGraphicFile`.
178  fn find_graphic_file(
179    &self,
180    _doc: &PostDocument,
181    node: &Node,
182    search_paths: &[String],
183  ) -> Option<String> {
184    let source = node.get_attribute("graphic")?;
185
186    // Check candidates attribute first (comma-separated list of found files)
187    // Perl: findGraphicFile checks each candidate path, resolving relative to search paths.
188    if let Some(candidates) = node.get_attribute("candidates") {
189      // Pick the best candidate by desirability
190      let mut best: Option<(String, i32)> = None;
191      for path in candidates.split(',') {
192        let path = path.trim();
193        if path.is_empty() {
194          continue;
195        }
196        // Try the path directly, then in each search directory
197        let resolved = if Path::new(path).exists() {
198          Some(path.to_string())
199        } else {
200          search_paths.iter().find_map(|sp| {
201            let candidate = format!("{}/{}", sp, path);
202            if Path::new(&candidate).exists() {
203              Some(candidate)
204            } else {
205              None
206            }
207          })
208        };
209        if let Some(resolved_path) = resolved {
210          let ext = Path::new(&resolved_path)
211            .extension()
212            .and_then(|e| e.to_str())
213            .unwrap_or("")
214            .to_lowercase();
215          let props = self.type_properties.get(&ext);
216          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
217          let is_same_type = props
218            .and_then(|p| p.destination_type.as_ref())
219            .map(|dt| dt == &ext)
220            .unwrap_or(false);
221          let desirability = if is_same_type { 10 } else { d };
222          if best.as_ref().is_none_or(|(_, bd)| desirability > *bd) {
223            best = Some((resolved_path, desirability));
224          }
225        }
226      }
227      if let Some((path, _)) = best {
228        return Some(path);
229      }
230    }
231
232    // Search for the file in search paths
233    let path = Path::new(&source);
234    let name = path.file_stem().and_then(|s| s.to_str()).unwrap_or(&source);
235    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or("");
236
237    let file_base = if dir.is_empty() {
238      name.to_string()
239    } else {
240      format!("{}/{}", dir, name)
241    };
242
243    // Try each type in search paths
244    let mut best_desirability: i32 = -1;
245    let mut best_path: Option<String> = None;
246
247    let types: Vec<String> = self
248      .graphics_types
249      .iter()
250      .flat_map(|t| vec![t.clone(), t.to_uppercase()])
251      .collect();
252
253    for search_path in search_paths {
254      // Try without extension first (source might already have one)
255      let candidate = if search_path.is_empty() {
256        source.clone()
257      } else {
258        format!("{}/{}", search_path, source)
259      };
260      if Path::new(&candidate).exists() {
261        let ext = Path::new(&candidate)
262          .extension()
263          .and_then(|e| e.to_str())
264          .unwrap_or("")
265          .to_lowercase();
266        let props = self.type_properties.get(&ext);
267        let d = props.map(|p| p.desirability as i32).unwrap_or(5);
268        if d > best_desirability {
269          best_desirability = d;
270          best_path = Some(candidate);
271        }
272      }
273      // Try each extension
274      for ext in &types {
275        let candidate = format!("{}/{}.{}", search_path, file_base, ext);
276        if Path::new(&candidate).exists() {
277          let props = self.type_properties.get(&ext.to_lowercase());
278          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
279          let is_same_type = props
280            .and_then(|p| p.destination_type.as_ref())
281            .map(|dt| dt == &ext.to_lowercase())
282            .unwrap_or(false);
283          let desirability = if is_same_type { 10 } else { d };
284          if desirability > best_desirability {
285            best_desirability = desirability;
286            best_path = Some(candidate);
287          }
288        }
289      }
290    }
291
292    best_path
293  }

295  /// Set the image source attributes on a graphics node.
296  ///
297  /// Port of `Graphics::setGraphicSrc`.
298  fn set_graphic_src(node: &mut Node, src: &str, width: Option<u32>, height: Option<u32>) {
299    node.set_attribute("imagesrc", src).ok();
300    // HTML width/height are in pixels (unitless)
301    if let Some(w) = width {
302      node.set_attribute("imagewidth", &w.to_string()).ok();
303    }
304    if let Some(h) = height {
305      node.set_attribute("imageheight", &h.to_string()).ok();
306    }
307    // Set aspect ratio class
308    if let (Some(w), Some(h)) = (width, height) {
309      let class = if w as f64 > 1.24 * h as f64 {
310        "ltx_img_landscape"
311      } else if h as f64 > 1.24 * w as f64 {
312        "ltx_img_portrait"
313      } else {
314        "ltx_img_square"
315      };
316      let existing = node.get_attribute("class").unwrap_or_default();
317      let new_class = if existing.is_empty() {
318        class.to_string()
319      } else {
320        format!("{} {}", existing, class)
321      };
322      node.set_attribute("class", &new_class).ok();
323    }
324  }

339  /// Read a named parameter from processing instructions.
340  /// Port of Perl's `Graphics::getParameter`.
341  /// Checks both direct PI (`<?latexml DPI="100"?>`) and
342  /// latexml.sty package options (`<?latexml package="latexml" options="magnify=1.2"?>`).
343  fn get_parameter(&self, doc: &PostDocument, param: &str) -> Option<f64> {
344    let direct_re = regex::Regex::new(&format!(
345      r#"^\s*{}\s*=\s*[\"']?([\d.]+)[\"']?\s*$"#,
346      regex::escape(param)
347    ))
348    .ok()?;
349    let options_re =
350      regex::Regex::new(r#"package\s*=\s*[\"']latexml[\"'].*options\s*=\s*[\"'](.*?)[\"']"#)
351        .ok()?;
352    let param_in_options_re =
353      regex::Regex::new(&format!(r#"\b{}\s*=\s*([\d.]+)"#, regex::escape(param))).ok()?;
354
355    for pi in doc.findnodes(".//processing-instruction('latexml')") {
356      let text = pi.get_content();
357      if let Some(cap) = direct_re.captures(&text) {
358        return cap[1].parse().ok();
359      }
360      if let Some(cap) = options_re.captures(&text) {
361        if let Some(inner) = param_in_options_re.captures(&cap[1]) {
362          return inner[1].parse().ok();
363        }
364      }
365    }
366    None
367  }

378  /// Parse graphicx options and apply transforms to image dimensions.
379  /// Port of Perl's `getTransform` + `image_graphicx_trivial`.
380  ///
381  /// Handles: scale=N, width=Npt, height=Npt, keepaspectratio
382  fn apply_graphicx_transforms(raw_w: u32, raw_h: u32, options: &str, dpi: u32) -> (u32, u32) {
383    let dppt = dpi as f64 / 72.27; // dots per point
384    let mut w = raw_w as f64;
385    let mut h = raw_h as f64;
386
387    // Parse options as key=value pairs
388    let mut scale: Option<f64> = None;
389    let mut target_width: Option<f64> = None;
390    let mut target_height: Option<f64> = None;
391    let mut keep_aspect = false;
392
393    for opt in options.split(',') {
394      let opt = opt.trim();
395      if let Some((key, val)) = opt.split_once('=') {
396        let key = key.trim();
397        let val = val.trim();
398        match key {
399          "scale" => {
400            scale = val.parse::<f64>().ok();
401          },
402          "width" => {
403            // Parse dimension: "345.0pt" or "345pt" or bare number
404            let val = val.trim_end_matches("pt").trim_end_matches("px");
405            target_width = val.parse::<f64>().ok();
406          },
407          "height" => {
408            let val = val.trim_end_matches("pt").trim_end_matches("px");
409            target_height = val.parse::<f64>().ok();
410          },
411          "keepaspectratio" => {
412            keep_aspect = val == "true" || val == "1" || val.is_empty();
413          },
414          _ => {},
415        }
416      } else if opt == "keepaspectratio" {
417        keep_aspect = true;
418      }
419    }
420
421    // Apply transforms (matching Perl's image_graphicx_trivial)
422    if let Some(s) = scale {
423      w *= s;
424      h *= s;
425    } else if target_width.is_some() || target_height.is_some() {
426      let tw = target_width.unwrap_or(w / dppt);
427      let th = target_height.unwrap_or(h / dppt);
428      if keep_aspect {
429        // Preserve aspect ratio: use the more constraining dimension
430        let scale_w = tw / (w / dppt);
431        let scale_h = th / (h / dppt);
432        let s = scale_w.min(scale_h);
433        w = (w / dppt * s * dppt).ceil();
434        h = (h / dppt * s * dppt).ceil();
435      } else {
436        w = (tw * dppt).ceil();
437        h = (th * dppt).ceil();
438      }
439    }
440
441    (w.max(1.0) as u32, h.max(1.0) as u32)
442  }

444  /// Copy a source image to the destination directory, preserving relative paths.
445  /// Returns the destination path (relative to dest_dir).
446  fn copy_to_destination(source: &str, source_dir: &str, dest_dir: &str) -> Option<String> {
447    // Compute relative path of source from source_dir
448    let source_path = Path::new(source);
449    let source_base = Path::new(source_dir);
450    let rel_path = source_path.strip_prefix(source_base).unwrap_or(source_path);
451
452    // Build absolute destination path
453    let abs_dest = PathBuf::from(dest_dir).join(rel_path);
454
455    // Create parent directories if needed
456    if let Some(parent) = abs_dest.parent() {
457      std::fs::create_dir_all(parent).ok()?;
458    }
459
460    // Copy the file (skip if same path)
461    let source_canonical = std::fs::canonicalize(source).ok();
462    let dest_canonical = std::fs::canonicalize(&abs_dest).ok();
463    if source_canonical != dest_canonical || dest_canonical.is_none() {
464      std::fs::copy(source, &abs_dest).ok()?;
465    }
466
467    // Return relative path for imagesrc attribute
468    Some(rel_path.to_string_lossy().to_string())
469  }

473  fn parse_page_option(options: &str) -> Option<u32> {
474    for opt in options.split(',') {

487  /// Try to convert a PDF to plain SVG via `inkscape`, preserving vector
488  /// content. Returns `true` on success. Tracks upstream
489  /// brucemiller/LaTeXML#902.
490  ///
491  /// Caller decides when to attempt this — typically only for PDF sources
492  /// below a file-size threshold, because inkscape on raster-embedded PDFs
493  /// produces massive output (>100 MB) and can take 40+ seconds
494  /// (measured: Fade.pdf 1.7 MB → 46 s / 102 MB SVG vs `convert` 1.4 s /
495  /// 61 KB PNG).
496  ///
497  /// `page` is 1-based (graphicx convention); converted to 0-based for
498  /// inkscape's `--pdf-page`.
499  ///
500  /// Guarded by a **hard timeout** (15 s default; see
501  /// `inkscape_timeout_secs`). Pathological small-PDF cases have been
502  /// observed — if inkscape is still running after the deadline we SIGKILL
503  /// it and return `false` so the caller falls back to ImageMagick. The
504  /// timeout is generous enough (15 s) for all well-behaved small vector
505  /// plots and strict enough to prevent the 46 s+ runaway behaviour seen
506  /// on Fade.pdf-class inputs.
507  fn convert_image_svg(source: &str, dest: &str, page: Option<u32>) -> bool {
508    let mut cmd = std::process::Command::new("inkscape");
509    cmd
510      .arg("--export-type=svg")
511      .arg("--export-plain-svg")
512      .arg(format!("--export-filename={}", dest));
513    if let Some(p) = page {
514      cmd.arg(format!("--pdf-page={}", p.saturating_sub(1)));
515    }
516    cmd.arg(source);
517    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
518    match Self::run_with_timeout(cmd, timeout) {
519      Some(status) => status.success() && Path::new(dest).exists(),
520      None => {
521        log::warn!(
522          "Graphics: inkscape SVG conversion for {} exceeded {} s — killed",
523          source,
524          timeout.as_secs()
525        );
526        // Best-effort cleanup of a partial output.
527        let _ = std::fs::remove_file(dest);
528        false
529      },
530    }
531  }

533  /// Hard timeout (seconds) for the `inkscape` subprocess. Overridable via
534  /// the `LATEXML_INKSCAPE_TIMEOUT_SECS` environment variable for
535  /// debugging; defaults to 15 s — enough for all benign vector-authored
536  /// plots we've measured (< 1 s typical), strict enough to cut off the
537  /// Fade.pdf-class 40 s+ runaway cases.
538  fn inkscape_timeout_secs() -> u64 { INKSCAPE_TIMEOUT_SECS.unwrap_or(15) }

540  /// Run `cmd` and enforce a wall-clock timeout. Returns `Some(status)` on
541  /// clean exit, `None` if the child was killed on timeout or spawn
542  /// failed. Polls every 50 ms — cheap compared to the subprocess cost.
543  fn run_with_timeout(
544    mut cmd: std::process::Command,
545    timeout: std::time::Duration,
546  ) -> Option<std::process::ExitStatus> {
547    // Redirect stdio so a slow inkscape doesn't block on a full pipe.
548    cmd
549      .stdout(std::process::Stdio::null())
550      .stderr(std::process::Stdio::null());
551    let mut child = cmd.spawn().ok()?;
552    let start = std::time::Instant::now();
553    loop {
554      match child.try_wait() {
555        Ok(Some(status)) => return Some(status),
556        Ok(None) => {
557          if start.elapsed() >= timeout {
558            let _ = child.kill();
559            let _ = child.wait();
560            return None;
561          }
562          std::thread::sleep(std::time::Duration::from_millis(50));
563        },
564        Err(_) => {
565          let _ = child.kill();
566          let _ = child.wait();
567          return None;
568        },
569      }
570    }
571  }

573  /// Parse SVG viewBox ("minX minY width height") and return (width, height).
574  /// Falls back to `width`/`height` root attributes if viewBox is missing.
575  /// Returns None on parse failure so callers can skip dimension attributes.
576  fn read_svg_dimensions(path: &str) -> Option<(u32, u32)> {
577    let content = std::fs::read_to_string(path).ok()?;
578    // Look at just the root <svg> opening tag (first ~2 KB is enough).
579    // We must skip the optional `<?xml … ?>` prolog and any `<!-- … -->`
580    // or `<!DOCTYPE …>` preamble — otherwise `find('>')` matches the
581    // prolog's `?>` instead of the root tag.
582    let head = &content[..content.len().min(2048)];
583    let svg_start = head.find("<svg")?;
584    let svg_rest = &head[svg_start..];
585    let svg_tag_end = svg_rest.find('>')?;
586    let root = &svg_rest[..=svg_tag_end];
587    let extract = |attr: &str| -> Option<String> {
588      let key = format!("{}=\"", attr);
589      let start = root.find(&key)? + key.len();
590      let rest = &root[start..];
591      let end = rest.find('"')?;
592      Some(rest[..end].to_string())
593    };
594    let parse_dim = |s: &str| -> Option<f64> {
595      let s = s.trim();
596      // Strip trailing unit if present (pt, px, mm, etc.)
597      let numeric_end = s
598        .find(|c: char| !c.is_ascii_digit() && c != '.' && c != '-')
599        .unwrap_or(s.len());
600      s[..numeric_end].parse::<f64>().ok()
601    };
602    if let Some(vb) = extract("viewBox") {
603      let parts: Vec<&str> = vb.split_whitespace().collect();
604      if parts.len() == 4 {
605        if let (Some(w), Some(h)) = (parse_dim(parts[2]), parse_dim(parts[3])) {
606          return Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32));
607        }
608      }
609    }
610    let w = extract("width").as_deref().and_then(parse_dim);
611    let h = extract("height").as_deref().and_then(parse_dim);
612    match (w, h) {
613      (Some(w), Some(h)) => Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32)),
614      _ => None,
615    }
616  }
617
618  /// Decide whether the vector-SVG path should be attempted for this PDF
619  /// source. File-size heuristic: small PDFs (< threshold KB) are nearly
620  /// always vector-authored plots; large PDFs typically contain embedded
621  /// rasters that inkscape re-renders as absurdly large SVG (empirical
622  /// measurement in round-17 — see upstream brucemiller/LaTeXML#902).
623  fn should_try_svg_path(source: &str, threshold_kb: u32) -> bool {
624    if threshold_kb == 0 {
625      return false;
626    }
627    if !source.to_lowercase().ends_with(".pdf") {
628      return false;
629    }
630    match std::fs::metadata(source) {
631      Ok(md) => md.len() <= (threshold_kb as u64) * 1024,
632      Err(_) => false,
633    }
634  }
635
636  fn raster_density_for_source(source: &str) -> u32 {
637    let source_lc = source.to_lowercase();
638    let is_postscript =
639      source_lc.ends_with(".eps") || source_lc.ends_with(".ps") || source_lc.ends_with(".ai");
640    let is_pdf = source_lc.ends_with(".pdf");
641    let page_box = if is_postscript {
642      read_postscript_bounding_box(source)
643    } else if is_pdf {
644      read_pdf_page_box(source)
645    } else {
646      None
647    };
648    let Some((w_pt, h_pt)) = page_box else {
649      return Self::DEFAULT_RASTER_DENSITY;
650    };
651    let max_pt = w_pt.max(h_pt);
652    if max_pt <= 0.0 {
653      return Self::DEFAULT_RASTER_DENSITY;
654    }
655
656    let max_density = ((Self::MAX_RASTER_DIMENSION_PX as f64) * 72.0 / max_pt).floor() as u32;
657    Self::DEFAULT_RASTER_DENSITY.min(max_density.max(1))
658  }
659
660  fn should_try_eps_pdf_path(source: &str, page: Option<u32>) -> bool {
661    page.is_none() && source.to_lowercase().ends_with(".eps")
662  }
663
664  /// Some EPS files make ImageMagick/Ghostscript spend tens of seconds in
665  /// direct rasterization. Converting EPS to a cropped PDF first and then
666  /// rasterizing the PDF via poppler is much faster for those cases, while
667  /// still falling back to ImageMagick if either helper is unavailable.
668  fn convert_eps_via_pdf(source: &str, dest: &str, density: u32) -> bool {
669    let dest_path = Path::new(dest);
670    let parent = dest_path.parent().unwrap_or_else(|| Path::new("."));
671    let stem = dest_path
672      .file_name()
673      .and_then(|s| s.to_str())
674      .unwrap_or("image");
675    let unique = std::time::SystemTime::now()
676      .duration_since(std::time::UNIX_EPOCH)
677      .map(|d| d.as_nanos())
678      .unwrap_or(0);
679    let tmp_pdf = parent.join(format!(".{}.{}.pdf", stem, unique));
680    let tmp_prefix = parent.join(format!(".{}.{}.pdftocairo", stem, unique));
681    let tmp_png = PathBuf::from(format!("{}.png", tmp_prefix.to_string_lossy()));
682    let timeout = std::time::Duration::from_secs(20);
683
684    let cleanup = |tmp_pdf: &Path, tmp_png: &Path| {
685      let _ = std::fs::remove_file(tmp_pdf);
686      let _ = std::fs::remove_file(tmp_png);
687    };
688
689    let mut ps2pdf = std::process::Command::new("ps2pdf");
690    ps2pdf.arg("-dEPSCrop").arg(source).arg(&tmp_pdf);
691    let ps2pdf_ok = Self::run_with_timeout(ps2pdf, timeout)
692      .map(|status| status.success())
693      .unwrap_or(false)
694      && tmp_pdf.exists();
695    if !ps2pdf_ok {
696      cleanup(&tmp_pdf, &tmp_png);
697      return false;
698    }
699
700    let mut pdftocairo = std::process::Command::new("pdftocairo");
701    pdftocairo
702      .arg("-singlefile")
703      .arg("-png")
704      .arg("-r")
705      .arg(density.to_string())
706      .arg(&tmp_pdf)
707      .arg(&tmp_prefix);
708    let cairo_ok = Self::run_with_timeout(pdftocairo, timeout)
709      .map(|status| status.success())
710      .unwrap_or(false)
711      && tmp_png.exists();
712    if !cairo_ok {
713      cleanup(&tmp_pdf, &tmp_png);
714      return false;
715    }
716
717    let _ = std::fs::remove_file(dest);
718    let installed = std::fs::rename(&tmp_png, dest)
719      .or_else(|_| std::fs::copy(&tmp_png, dest).map(|_| ()))
720      .is_ok()
721      && dest_path.exists();
722    cleanup(&tmp_pdf, &tmp_png);
723    installed
724  }
725
726  /// Convert a graphics file using ImageMagick's `convert` command.
727  /// Perl: image_graphicx_complex via Image::Magick / convert CLI.
728  /// `page` is 1-based (graphicx convention); converted to 0-based for ImageMagick.
729  fn convert_image(source: &str, dest: &str, _dpi: u32, page: Option<u32>) -> bool {
730    // Build the source argument with optional page selector
731    // Perl: image_read reads "$source[$page]" where $page = ($page // 1) - 1
732    let source_arg = if let Some(p) = page {
733      format!("{}[{}]", source, p.saturating_sub(1))
734    } else {
735      // No page specified: use [0] for PDFs to avoid converting all pages
736      if source.to_lowercase().ends_with(".pdf") {
737        format!("{}[0]", source)
738      } else {
739        source.to_string()
740      }
741    };
742    // Shell out to convert (matching Perl's approach)
743    // -define pdf:use-cropbox=true matches Perl's Image::Magick option (line 466)
744    let density = Self::raster_density_for_source(source);
745    if Self::should_try_eps_pdf_path(source, page)
746      && Self::convert_eps_via_pdf(source, dest, density)
747    {
748      return true;
749    }
750    let result = std::process::Command::new("convert")
751      .arg("-define")
752      .arg("pdf:use-cropbox=true")
753      .arg("-density")
754      .arg(density.to_string())
755      .arg(&source_arg)
756      .arg(dest)
757      .output();
758    match result {
759      Ok(output) => output.status.success(),
760      Err(_) => false,
761    }
762  }
763}

765fn read_postscript_bounding_box(source: &str) -> Option<(f64, f64)> {
766  let content = std::fs::read_to_string(source).ok()?;
767  for line in content.lines().take(80) {
768    let Some(rest) = line.strip_prefix("%%BoundingBox:") else {
769      continue;
770    };
771    let mut vals = rest
772      .split_whitespace()
773      .filter_map(|s| s.parse::<f64>().ok());
774    let (Some(x0), Some(y0), Some(x1), Some(y1)) =
775      (vals.next(), vals.next(), vals.next(), vals.next())
776    else {
777      return None;
778    };
779    let w = (x1 - x0).abs();
780    let h = (y1 - y0).abs();
781    return Some((w, h));
782  }
783  None
784}
785
786fn read_pdf_page_box(source: &str) -> Option<(f64, f64)> {
787  let bytes = std::fs::read(source).ok()?;
788  let content = String::from_utf8_lossy(&bytes);
789  parse_pdf_page_box(&content, &PDF_CROP_BOX_RE)
790    .or_else(|| parse_pdf_page_box(&content, &PDF_MEDIA_BOX_RE))
791}
792
793fn parse_pdf_page_box(content: &str, re: &regex::Regex) -> Option<(f64, f64)> {
794  let captures = re.captures(content)?;
795  let x0 = captures.get(1)?.as_str().parse::<f64>().ok()?;
796  let y0 = captures.get(2)?.as_str().parse::<f64>().ok()?;
797  let x1 = captures.get(3)?.as_str().parse::<f64>().ok()?;
798  let y1 = captures.get(4)?.as_str().parse::<f64>().ok()?;
799  Some(((x1 - x0).abs(), (y1 - y0).abs()))
800}

802impl Processor for Graphics {
803  fn get_name(&self) -> &str { &self.name }
804
805  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
806    doc.findnodes("//ltx:graphics[not(@imagesrc)]")
807  }
808
809  fn process(&mut self, doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
810    let mut search_paths = self.find_graphics_paths(&doc);
811    search_paths.extend(doc.get_search_paths().iter().cloned());
812    // Also add source directory
813    let source_dir = doc.get_source_directory().to_string();
814    if !source_dir.is_empty() && !search_paths.contains(&source_dir) {
815      search_paths.push(source_dir.clone());
816    }
817    // Add current directory as fallback
818    if !search_paths.contains(&".".to_string()) {
819      search_paths.push(".".to_string());
820    }
821
822    let dest_dir = doc.get_destination_directory().unwrap_or(".").to_string();
823    // Read DPI/magnify/zoomout from processing instructions (set by latexml.sty)
824    let dpi = self
825      .get_parameter(&doc, "DPI")
826      .map(|v| v as u32)
827      .or(self.dpi)
828      .unwrap_or(100);
829    let magnify = self.get_parameter(&doc, "magnify").unwrap_or(self.magnify);
830    let _zoomout = self.get_parameter(&doc, "zoomout").unwrap_or(self.zoomout);
831    // Perl: effective DPI = DPI * magnify / zoomout (used for scale-to transforms)
832    let effective_dpi = ((dpi as f64) * magnify / _zoomout) as u32;
833    let n_to_process = nodes.len();
834    latexml_core::telemetry::set_graphics_assets(n_to_process as u32);
835
836    // Counter for generating unique resource names (like Perl's generateResourcePathname)
837    let mut resource_counter: u32 = 0;
838
839    // Two-phase plan so the slow per-image `convert` subprocess and
840    // `read_image_dimensions` calls can run in parallel without touching
841    // the libxml DOM off-thread.
842    //
843    // Phase 1 (serial): read each node's attributes, resolve source path,
844    // decide the conversion kind, and allocate resource-name counters.
845    //   - `Plan::NotFound` — apply fallback on the main thread later
846    //   - `Plan::Copy { .. }` — apply on the main thread later (cheap)
847    //   - `Plan::Convert { .. }` — independent; run in parallel.
848    // Phase 2 (parallel): run convert_image + read_image_dimensions for
849    // `Plan::Convert` entries. Produces `JobResult`s keyed by node index.
850    // Phase 3 (serial): apply DOM mutations on the main thread in original
851    // node order so attribute writes happen on the libxml-owning thread.
852    enum Plan {
853      NotFound {
854        idx:     usize,
855        graphic: String,
856      },
857      Copy {
858        idx:     usize,
859        source:  String,
860        options: String,
861      },
862      Convert {
863        idx:     usize,
864        options: String,
865        job_id:  usize,
866      },
867    }
868    struct ConvertJob {
869      job_id:       usize,
870      source:       String,
871      page:         Option<u32>,
872      rel_dest:     String,
873      abs_dest_str: String,
874      /// `Some((rel_svg, abs_svg_str))` when the worker should
875      /// first attempt the inkscape-SVG path and only fall back
876      /// to `convert` on failure. `None` means the classic
877      /// raster-only path.
878      svg_paths:    Option<(String, String)>,
879    }
880    struct ConvertOutcome {
881      job_id:   usize,
882      /// Path to write into `imagesrc`; `None` if both convert and copy-fallback failed.
883      imagesrc: Option<String>,
884      /// Raw (pre-transform) dimensions read from whichever file we ended up with.
885      raw_dims: Option<(u32, u32)>,
886    }
887
888    let mut plans: Vec<Plan> = Vec::with_capacity(n_to_process);
889    let mut convert_jobs: Vec<ConvertJob> = Vec::new();
890    let mut convert_job_ids: HashMap<(String, Option<u32>, String), usize> = HashMap::new();
891    let mut convert_source_counts: HashMap<String, u32> = HashMap::new();
892    for (idx, node) in nodes.iter().enumerate() {
893      let options = node.get_attribute("options").unwrap_or_default();
894      let page = Self::parse_page_option(&options);
895      let Some(source) = self.find_graphic_file(&doc, node, &search_paths) else {
896        let graphic = node
897          .get_attribute("graphic")
898          .unwrap_or_else(|| "none".to_string());
899        plans.push(Plan::NotFound { idx, graphic });
900        continue;
901      };
902      let src_ext = Path::new(&source)
903        .extension()
904        .and_then(|e| e.to_str())
905        .unwrap_or("")
906        .to_lowercase();
907      let props = self.type_properties.get(&src_ext).cloned();
908      let dest_type = props
909        .as_ref()
910        .and_then(|p| p.destination_type.as_ref())
911        .cloned()
912        .unwrap_or(src_ext.clone());
913      let needs_conversion = dest_type != src_ext;
914      let has_page = page.is_some();
915      if needs_conversion || has_page {
916        let job_key = (source.clone(), page, options.clone());
917        let job_id = if let Some(&job_id) = convert_job_ids.get(&job_key) {
918          job_id
919        } else {
920          let prior_source_jobs = convert_source_counts.get(&source).copied().unwrap_or(0);
921          let dest_name = if has_page || prior_source_jobs > 0 {
922            resource_counter += 1;
923            format!("x{}", resource_counter)
924          } else {
925            Path::new(&source)
926              .file_stem()
927              .and_then(|s| s.to_str())
928              .unwrap_or("image")
929              .to_string()
930          };
931          convert_source_counts.insert(source.clone(), prior_source_jobs + 1);
932          // Vector-SVG path: opt-in for small PDFs only. We prepare an
933          // alternate `.svg` destination path alongside the normal raster
934          // destination so the worker can try inkscape first, then fall
935          // back. The file-size heuristic gates this — see
936          // `should_try_svg_path`.
937          let try_svg = Self::should_try_svg_path(&source, self.svg_threshold_kb);
938          let rel_dest = format!("{}.{}", dest_name, dest_type);
939          let abs_dest = PathBuf::from(&dest_dir).join(&rel_dest);
940          if let Some(parent) = abs_dest.parent() {
941            std::fs::create_dir_all(parent).ok();
942          }
943          let abs_dest_str = abs_dest.to_string_lossy().to_string();
944          let svg_paths = if try_svg {
945            let rel_svg = format!("{}.svg", dest_name);
946            let abs_svg = PathBuf::from(&dest_dir).join(&rel_svg);
947            let abs_svg_str = abs_svg.to_string_lossy().to_string();
948            Some((rel_svg, abs_svg_str))
949          } else {
950            None
951          };
952          let job_id = convert_jobs.len();
953          convert_jobs.push(ConvertJob {
954            job_id,
955            source: source.clone(),
956            page,
957            rel_dest,
958            abs_dest_str,
959            svg_paths,
960          });
961          convert_job_ids.insert(job_key, job_id);
962          job_id
963        };
964        plans.push(Plan::Convert { idx, options, job_id });
965      } else {
966        plans.push(Plan::Copy { idx, source, options });
967      }
968    }
969
970    // Phase 2: parallel conversions. Bounded worker count to avoid
971    // oversubscribing when many images are in flight. `convert` itself
972    // is single-threaded per invocation, so the ceiling is useful CPU
973    // parallelism — capped at a reasonable limit to avoid fork/memory
974    // storms with many-image papers.
975    let convert_count = convert_jobs.len();
976    let worker_cap = std::thread::available_parallelism()
977      .map(|n| n.get())
978      .unwrap_or(4)
979      .clamp(1, 8);
980    let n_workers = convert_count.min(worker_cap).max(1);
981    let source_dir_ref = source_dir.as_str();
982    let dest_dir_ref = dest_dir.as_str();
983    let mut outcomes: Vec<ConvertOutcome> = Vec::with_capacity(convert_count);
984    if convert_count > 0 {
985      use std::sync::Mutex;
986      use std::sync::atomic::{AtomicU32, AtomicUsize, Ordering};
987      let next = AtomicUsize::new(0);
988      let out = Mutex::new(Vec::<ConvertOutcome>::with_capacity(convert_count));
989      // Subprocess tally: telemetry's thread_local! STATE is per-thread,
990      // and worker threads exit before `phase_us[graphics]` aggregation,
991      // so worker increments would be lost. Accumulate in a shared
992      // AtomicU32 here and merge into telemetry once the scope joins.
993      // One increment per `Self::convert_image_svg` / `Self::convert_image`
994      // call (the EPS-via-PDF internal pair counts as one).
995      let subproc_count = AtomicU32::new(0);
996      let subproc_ref = &subproc_count;
997      // Unique conversion jobs only. Repeated nodes with the same
998      // source/page/options share one subprocess result, while distinct
999      // options keep separate outputs.
1000      let jobs: Vec<&ConvertJob> = convert_jobs.iter().collect();
1001      std::thread::scope(|s| {
1002        for _ in 0..n_workers {
1003          s.spawn(|| {
1004            loop {
1005              let i = next.fetch_add(1, Ordering::Relaxed);
1006              if i >= jobs.len() {
1007                break;
1008              }
1009              let ConvertJob {
1010                job_id,
1011                source,
1012                page,
1013                rel_dest,
1014                abs_dest_str,
1015                svg_paths,
1016              } = jobs[i];
1017              // Try vector-SVG path first if requested for this source.
1018              // On success, pick dimensions from the SVG viewBox.
1019              let svg_outcome = if let Some((rel_svg, abs_svg)) = svg_paths {
1020                subproc_ref.fetch_add(1, Ordering::Relaxed);
1021                if Self::convert_image_svg(source, abs_svg, *page) {
1022                  let raw_dims = Self::read_svg_dimensions(abs_svg);
1023                  Some(ConvertOutcome {
1024                    job_id: *job_id,
1025                    imagesrc: Some(rel_svg.clone()),
1026                    raw_dims,
1027                  })
1028                } else {
1029                  log::warn!(
1030                    "Graphics: inkscape SVG path failed for {}, falling back to convert",
1031                    source
1032                  );
1033                  None
1034                }
1035              } else {
1036                None
1037              };
1038              let converted = if svg_outcome.is_none() {
1039                subproc_ref.fetch_add(1, Ordering::Relaxed);
1040                Self::convert_image(source, abs_dest_str, dpi, *page)
1041              } else {
1042                false
1043              };
1044              let outcome = if let Some(o) = svg_outcome {
1045                o
1046              } else if converted {
1047                ConvertOutcome {
1048                  job_id:   *job_id,
1049                  imagesrc: Some(rel_dest.clone()),
1050                  raw_dims: Self::read_image_dimensions(abs_dest_str),
1051                }
1052              } else {
1053                log::warn!("Graphics: Failed to convert {} to {}", source, abs_dest_str);
1054                if let Some(rel) = Self::copy_to_destination(source, source_dir_ref, dest_dir_ref) {
1055                  ConvertOutcome {
1056                    job_id:   *job_id,
1057                    imagesrc: Some(rel),
1058                    raw_dims: Self::read_image_dimensions(source),
1059                  }
1060                } else {
1061                  ConvertOutcome {
1062                    job_id:   *job_id,
1063                    imagesrc: None,
1064                    raw_dims: None,
1065                  }
1066                }
1067              };
1068              out.lock().unwrap().push(outcome);
1069            }
1070          });
1071        }
1072      });
1073      outcomes = out.into_inner().unwrap();
1074      outcomes.sort_by_key(|o| o.job_id);
1075      latexml_core::telemetry::add_graphics_subprocess(subproc_count.load(Ordering::Relaxed));
1076    }
1077    let outcomes_by_job: HashMap<usize, ConvertOutcome> =
1078      outcomes.into_iter().map(|o| (o.job_id, o)).collect();
1079
1080    // Phase 3: serial DOM mutations. Preserves original node order.
1081    let apply_transforms =
1082      |options: &str, raw_dims: Option<(u32, u32)>| -> (Option<u32>, Option<u32>) {
1083        match raw_dims {
1084          Some((w, h)) if !options.is_empty() => {
1085            let (tw, th) = Self::apply_graphicx_transforms(w, h, options, effective_dpi);
1086            (Some(tw), Some(th))
1087          },
1088          Some((w, h)) => (Some(w), Some(h)),
1089          None => (None, None),
1090        }
1091      };
1092    for plan in &plans {
1093      match plan {
1094        Plan::NotFound { idx, graphic } => {
1095          log::warn!("Graphics: No source found for {}", graphic);
1096          let mut node_mut = nodes[*idx].clone();
1097          node_mut.set_attribute("imagesrc", graphic).ok();
1098        },
1099        Plan::Copy { idx, source, options } => {
1100          let mut node_mut = nodes[*idx].clone();
1101          if let Some(rel) = Self::copy_to_destination(source, &source_dir, &dest_dir) {
1102            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
1103            Self::set_graphic_src(&mut node_mut, &rel, w, h);
1104          } else {
1105            let rel_path = Path::new(source)
1106              .strip_prefix(&source_dir)
1107              .unwrap_or(Path::new(source));
1108            let rel_str = rel_path.to_string_lossy().to_string();
1109            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
1110            Self::set_graphic_src(&mut node_mut, &rel_str, w, h);
1111          }
1112        },
1113        Plan::Convert { idx, options, job_id } => {
1114          if let Some(out) = outcomes_by_job.get(job_id) {
1115            let mut node_mut = nodes[*idx].clone();
1116            if let Some(imagesrc) = &out.imagesrc {
1117              let (w, h) = apply_transforms(options, out.raw_dims);
1118              Self::set_graphic_src(&mut node_mut, imagesrc, w, h);
1119            }
1120          }
1121        },
1122      }
1123    }
1124
1125    log::info!(
1126      "Graphics {} {} to process",
1127      doc.get_destination().unwrap_or("?"),
1128      n_to_process
1129    );
1130    Ok(vec![doc])
1131  }
1132}

1201  /// File-size heuristic: PDF under threshold triggers SVG attempt,
1202  /// large PDF does not, non-PDF is always skipped, threshold=0 disables.
1203  #[test]
1204  fn should_try_svg_path_heuristic() {
1205    let tmp = std::env::temp_dir().join("latexml_graphics_svg_gate_test");
1206    std::fs::create_dir_all(&tmp).unwrap();
1207    let small_pdf = tmp.join("small.pdf");
1208    let big_pdf = tmp.join("big.pdf");
1209    let png = tmp.join("raster.png");
1210    std::fs::write(&small_pdf, vec![0u8; 10 * 1024]).unwrap(); // 10 KB
1211    std::fs::write(&big_pdf, vec![0u8; 500 * 1024]).unwrap(); // 500 KB
1212    std::fs::write(&png, vec![0u8; 10 * 1024]).unwrap(); // PNG, irrelevant size
1213
1214    // threshold = 0 → always false.
1215    assert!(!Graphics::should_try_svg_path(
1216      small_pdf.to_str().unwrap(),
1217      0
1218    ));
1219    // Under threshold → true.
1220    assert!(Graphics::should_try_svg_path(
1221      small_pdf.to_str().unwrap(),
1222      200
1223    ));
1224    // Over threshold → false.
1225    assert!(!Graphics::should_try_svg_path(
1226      big_pdf.to_str().unwrap(),
1227      200
1228    ));
1229    // Non-PDF → always false even under threshold.
1230    assert!(!Graphics::should_try_svg_path(png.to_str().unwrap(), 200));
1231    // Missing file → false, not panic.
1232    assert!(!Graphics::should_try_svg_path("/no/such/file.pdf", 200));
1233
1234    std::fs::remove_dir_all(&tmp).ok();
1235  }

1331    use crate::document::{PostDocument, PostDocumentOptions};
1332    use std::os::unix::fs::PermissionsExt;
1333
1334    let tmp = std::env::temp_dir().join(format!("latexml_graphics_dedupe_{}", std::process::id()));
1335    std::fs::create_dir_all(&tmp).unwrap();
1336    let source = tmp.join("plot.ai");
1337    std::fs::write(&source, "%!PS-Adobe-3.0\n%%BoundingBox: 0 0 100 100\n").unwrap();
1338    let log = tmp.join("convert.log");
1339    let fake_convert = tmp.join("convert");
1340    std::fs::write(
1341      &fake_convert,
1342      "#!/bin/sh\nprintf '%s\\n' \"$*\" >> \"$LATEXML_FAKE_CONVERT_LOG\"\nexit 0\n",
1343    )
1344    .unwrap();
1345    let mut perms = std::fs::metadata(&fake_convert).unwrap().permissions();
1346    perms.set_mode(0o755);
1347    std::fs::set_permissions(&fake_convert, perms).unwrap();
1348
1349    let old_path = std::env::var("PATH").unwrap_or_default();
1350    let _path_guard = EnvGuard::set("PATH", &format!("{}:{}", tmp.display(), old_path));
1351    let _log_guard = EnvGuard::set("LATEXML_FAKE_CONVERT_LOG", log.to_str().unwrap());
1352    let xml = format!(
1353      r#"<?xml version="1.0"?>
1354<document xmlns="http://dlmf.nist.gov/LaTeXML" xml:id="d">
1355  <graphics graphic="plot.ai" candidates="{0}" options="width=20pt"/>
1356  <graphics graphic="plot.ai" candidates="{0}" options="width=40pt"/>
1357  <graphics graphic="plot.ai" candidates="{0}" options="width=20pt"/>
1358</document>"#,
1359      source.display()
1360    );
1361    let doc_opts = PostDocumentOptions {
1362      destination: Some(tmp.join("out.html").display().to_string()),
1363      source_directory: Some(tmp.display().to_string()),
1364      ..Default::default()
1365    };
1366    let doc = PostDocument::new_from_string(&xml, doc_opts).unwrap();

1372    let out = docs[0].to_xml_string();
