16// Process-once cached env var (see WISDOM #56 — getenv hot-path race).
17// Parsed-and-validated at init: only positive integer values are
18// honored; everything else (unset, empty, "0", malformed) leaves
19// `INKSCAPE_TIMEOUT_SECS` at None and the caller falls back to the
20// 15-second default in `inkscape_timeout_secs`.
21static INKSCAPE_TIMEOUT_SECS: LazyLock<Option<u64>> = LazyLock::new(|| {
22  std::env::var("LATEXML_INKSCAPE_TIMEOUT_SECS")
23    .ok()
24    .and_then(|s| s.parse::<u64>().ok())
25    .filter(|&n| n > 0)
26});

158  /// Find the graphics source file for a node.
159  ///
160  /// Port of `Graphics::findGraphicFile`.
161  fn find_graphic_file(
162    &self,
163    _doc: &PostDocument,
164    node: &Node,
165    search_paths: &[String],
166  ) -> Option<String> {
167    let source = node.get_attribute("graphic")?;
168
169    // Check candidates attribute first (comma-separated list of found files)
170    // Perl: findGraphicFile checks each candidate path, resolving relative to search paths.
171    if let Some(candidates) = node.get_attribute("candidates") {
172      // Pick the best candidate by desirability
173      let mut best: Option<(String, i32)> = None;
174      for path in candidates.split(',') {
175        let path = path.trim();
176        if path.is_empty() {
177          continue;
178        }
179        // Try the path directly, then in each search directory
180        let resolved = if Path::new(path).exists() {
181          Some(path.to_string())
182        } else {
183          search_paths.iter().find_map(|sp| {
184            let candidate = format!("{}/{}", sp, path);
185            if Path::new(&candidate).exists() {
186              Some(candidate)
187            } else {
188              None
189            }
190          })
191        };
192        if let Some(resolved_path) = resolved {
193          let ext = Path::new(&resolved_path)
194            .extension()
195            .and_then(|e| e.to_str())
196            .unwrap_or("")
197            .to_lowercase();
198          let props = self.type_properties.get(&ext);
199          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
200          let is_same_type = props
201            .and_then(|p| p.destination_type.as_ref())
202            .map(|dt| dt == &ext)
203            .unwrap_or(false);
204          let desirability = if is_same_type { 10 } else { d };
205          if best.as_ref().is_none_or(|(_, bd)| desirability > *bd) {
206            best = Some((resolved_path, desirability));
207          }
208        }
209      }
210      if let Some((path, _)) = best {
211        return Some(path);
212      }
213    }
214
215    // Search for the file in search paths
216    let path = Path::new(&source);
217    let name = path.file_stem().and_then(|s| s.to_str()).unwrap_or(&source);
218    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or("");
219
220    let file_base = if dir.is_empty() {
221      name.to_string()
222    } else {
223      format!("{}/{}", dir, name)
224    };
225
226    // Try each type in search paths
227    let mut best_desirability: i32 = -1;
228    let mut best_path: Option<String> = None;
229
230    let types: Vec<String> = self
231      .graphics_types
232      .iter()
233      .flat_map(|t| vec![t.clone(), t.to_uppercase()])
234      .collect();
235
236    for search_path in search_paths {
237      // Try without extension first (source might already have one)
238      let candidate = if search_path.is_empty() {
239        source.clone()
240      } else {
241        format!("{}/{}", search_path, source)
242      };
243      if Path::new(&candidate).exists() {
244        let ext = Path::new(&candidate)
245          .extension()
246          .and_then(|e| e.to_str())
247          .unwrap_or("")
248          .to_lowercase();
249        let props = self.type_properties.get(&ext);
250        let d = props.map(|p| p.desirability as i32).unwrap_or(5);
251        if d > best_desirability {
252          best_desirability = d;
253          best_path = Some(candidate);
254        }
255      }
256      // Try each extension
257      for ext in &types {
258        let candidate = format!("{}/{}.{}", search_path, file_base, ext);
259        if Path::new(&candidate).exists() {
260          let props = self.type_properties.get(&ext.to_lowercase());
261          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
262          let is_same_type = props
263            .and_then(|p| p.destination_type.as_ref())
264            .map(|dt| dt == &ext.to_lowercase())
265            .unwrap_or(false);
266          let desirability = if is_same_type { 10 } else { d };
267          if desirability > best_desirability {
268            best_desirability = desirability;
269            best_path = Some(candidate);
270          }
271        }
272      }
273    }
274
275    best_path
276  }

278  /// Set the image source attributes on a graphics node.
279  ///
280  /// Port of `Graphics::setGraphicSrc`.
281  fn set_graphic_src(node: &mut Node, src: &str, width: Option<u32>, height: Option<u32>) {
282    node.set_attribute("imagesrc", src).ok();
283    // HTML width/height are in pixels (unitless)
284    if let Some(w) = width {
285      node.set_attribute("imagewidth", &w.to_string()).ok();
286    }
287    if let Some(h) = height {
288      node.set_attribute("imageheight", &h.to_string()).ok();
289    }
290    // Set aspect ratio class
291    if let (Some(w), Some(h)) = (width, height) {
292      let class = if w as f64 > 1.24 * h as f64 {
293        "ltx_img_landscape"
294      } else if h as f64 > 1.24 * w as f64 {
295        "ltx_img_portrait"
296      } else {
297        "ltx_img_square"
298      };
299      let existing = node.get_attribute("class").unwrap_or_default();
300      let new_class = if existing.is_empty() {
301        class.to_string()
302      } else {
303        format!("{} {}", existing, class)
304      };
305      node.set_attribute("class", &new_class).ok();
306    }
307  }

311    let re = regex::Regex::new(r#"^\s*graphicspath\s*=\s*[\"'](.*?)[\"']\s*$"#).unwrap();
312    let mut paths = Vec::new();

322  /// Read a named parameter from processing instructions.
323  /// Port of Perl's `Graphics::getParameter`.
324  /// Checks both direct PI (`<?latexml DPI="100"?>`) and
325  /// latexml.sty package options (`<?latexml package="latexml" options="magnify=1.2"?>`).
326  fn get_parameter(&self, doc: &PostDocument, param: &str) -> Option<f64> {
327    let direct_re = regex::Regex::new(&format!(
328      r#"^\s*{}\s*=\s*[\"']?([\d.]+)[\"']?\s*$"#,
329      regex::escape(param)
330    ))
331    .ok()?;
332    let options_re =
333      regex::Regex::new(r#"package\s*=\s*[\"']latexml[\"'].*options\s*=\s*[\"'](.*?)[\"']"#)
334        .ok()?;
335    let param_in_options_re =
336      regex::Regex::new(&format!(r#"\b{}\s*=\s*([\d.]+)"#, regex::escape(param))).ok()?;
337
338    for pi in doc.findnodes(".//processing-instruction('latexml')") {
339      let text = pi.get_content();
340      if let Some(cap) = direct_re.captures(&text) {
341        return cap[1].parse().ok();
342      }
343      if let Some(cap) = options_re.captures(&text) {
344        if let Some(inner) = param_in_options_re.captures(&cap[1]) {
345          return inner[1].parse().ok();
346        }
347      }
348    }
349    None
350  }

361  /// Parse graphicx options and apply transforms to image dimensions.
362  /// Port of Perl's `getTransform` + `image_graphicx_trivial`.
363  ///
364  /// Handles: scale=N, width=Npt, height=Npt, keepaspectratio
365  fn apply_graphicx_transforms(raw_w: u32, raw_h: u32, options: &str, dpi: u32) -> (u32, u32) {
366    let dppt = dpi as f64 / 72.27; // dots per point
367    let mut w = raw_w as f64;
368    let mut h = raw_h as f64;
369
370    // Parse options as key=value pairs
371    let mut scale: Option<f64> = None;
372    let mut target_width: Option<f64> = None;
373    let mut target_height: Option<f64> = None;
374    let mut keep_aspect = false;
375
376    for opt in options.split(',') {
377      let opt = opt.trim();
378      if let Some((key, val)) = opt.split_once('=') {
379        let key = key.trim();
380        let val = val.trim();
381        match key {
382          "scale" => {
383            scale = val.parse::<f64>().ok();
384          },
385          "width" => {
386            // Parse dimension: "345.0pt" or "345pt" or bare number
387            let val = val.trim_end_matches("pt").trim_end_matches("px");
388            target_width = val.parse::<f64>().ok();
389          },
390          "height" => {
391            let val = val.trim_end_matches("pt").trim_end_matches("px");
392            target_height = val.parse::<f64>().ok();
393          },
394          "keepaspectratio" => {
395            keep_aspect = val == "true" || val == "1" || val.is_empty();
396          },
397          _ => {},
398        }
399      } else if opt == "keepaspectratio" {
400        keep_aspect = true;
401      }
402    }
403
404    // Apply transforms (matching Perl's image_graphicx_trivial)
405    if let Some(s) = scale {
406      w *= s;
407      h *= s;
408    } else if target_width.is_some() || target_height.is_some() {
409      let tw = target_width.unwrap_or(w / dppt);
410      let th = target_height.unwrap_or(h / dppt);
411      if keep_aspect {
412        // Preserve aspect ratio: use the more constraining dimension
413        let scale_w = tw / (w / dppt);
414        let scale_h = th / (h / dppt);
415        let s = scale_w.min(scale_h);
416        w = (w / dppt * s * dppt).ceil();
417        h = (h / dppt * s * dppt).ceil();
418      } else {
419        w = (tw * dppt).ceil();
420        h = (th * dppt).ceil();
421      }
422    }
423
424    (w.max(1.0) as u32, h.max(1.0) as u32)
425  }

427  /// Copy a source image to the destination directory, preserving relative paths.
428  /// Returns the destination path (relative to dest_dir).
429  fn copy_to_destination(source: &str, source_dir: &str, dest_dir: &str) -> Option<String> {
430    // Compute relative path of source from source_dir
431    let source_path = Path::new(source);
432    let source_base = Path::new(source_dir);
433    let rel_path = source_path.strip_prefix(source_base).unwrap_or(source_path);
434
435    // Build absolute destination path
436    let abs_dest = PathBuf::from(dest_dir).join(rel_path);
437
438    // Create parent directories if needed
439    if let Some(parent) = abs_dest.parent() {
440      std::fs::create_dir_all(parent).ok()?;
441    }
442
443    // Copy the file (skip if same path)
444    let source_canonical = std::fs::canonicalize(source).ok();
445    let dest_canonical = std::fs::canonicalize(&abs_dest).ok();
446    if source_canonical != dest_canonical || dest_canonical.is_none() {
447      std::fs::copy(source, &abs_dest).ok()?;
448    }
449
450    // Return relative path for imagesrc attribute
451    Some(rel_path.to_string_lossy().to_string())
452  }

454  /// Extract `page=N` from graphicx options string.
455  /// Returns 1-based page number (matching graphicx convention), or None.
456  fn parse_page_option(options: &str) -> Option<u32> {
457    for opt in options.split(',') {
458      let opt = opt.trim();
459      if let Some((key, val)) = opt.split_once('=') {
460        if key.trim() == "page" {
461          // Strip braces: page={2} → 2
462          let val = val.trim().trim_matches('{').trim_matches('}');

464        }
465      }
466    }

470  /// Try to convert a PDF to plain SVG via `inkscape`, preserving vector
471  /// content. Returns `true` on success. Tracks upstream
472  /// brucemiller/LaTeXML#902.
473  ///
474  /// Caller decides when to attempt this — typically only for PDF sources
475  /// below a file-size threshold, because inkscape on raster-embedded PDFs
476  /// produces massive output (>100 MB) and can take 40+ seconds
477  /// (measured: Fade.pdf 1.7 MB → 46 s / 102 MB SVG vs `convert` 1.4 s /
478  /// 61 KB PNG).
479  ///
480  /// `page` is 1-based (graphicx convention); converted to 0-based for
481  /// inkscape's `--pdf-page`.
482  ///
483  /// Guarded by a **hard timeout** (15 s default; see
484  /// `inkscape_timeout_secs`). Pathological small-PDF cases have been
485  /// observed — if inkscape is still running after the deadline we SIGKILL
486  /// it and return `false` so the caller falls back to ImageMagick. The
487  /// timeout is generous enough (15 s) for all well-behaved small vector
488  /// plots and strict enough to prevent the 46 s+ runaway behaviour seen
489  /// on Fade.pdf-class inputs.
490  fn convert_image_svg(source: &str, dest: &str, page: Option<u32>) -> bool {
491    let mut cmd = std::process::Command::new("inkscape");
492    cmd
493      .arg("--export-type=svg")
494      .arg("--export-plain-svg")
495      .arg(format!("--export-filename={}", dest));
496    if let Some(p) = page {
497      cmd.arg(format!("--pdf-page={}", p.saturating_sub(1)));
498    }
499    cmd.arg(source);
500    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
501    match Self::run_with_timeout(cmd, timeout) {
502      Some(status) => status.success() && Path::new(dest).exists(),
503      None => {
504        log::warn!(
505          "Graphics: inkscape SVG conversion for {} exceeded {} s — killed",
506          source,
507          timeout.as_secs()
508        );
509        // Best-effort cleanup of a partial output.
510        let _ = std::fs::remove_file(dest);
511        false
512      },
513    }
514  }

516  /// Hard timeout (seconds) for the `inkscape` subprocess. Overridable via
517  /// the `LATEXML_INKSCAPE_TIMEOUT_SECS` environment variable for
518  /// debugging; defaults to 15 s — enough for all benign vector-authored
519  /// plots we've measured (< 1 s typical), strict enough to cut off the
520  /// Fade.pdf-class 40 s+ runaway cases.
521  fn inkscape_timeout_secs() -> u64 { INKSCAPE_TIMEOUT_SECS.unwrap_or(15) }

523  /// Run `cmd` and enforce a wall-clock timeout. Returns `Some(status)` on
524  /// clean exit, `None` if the child was killed on timeout or spawn
525  /// failed. Polls every 50 ms — cheap compared to the subprocess cost.
526  fn run_with_timeout(
527    mut cmd: std::process::Command,
528    timeout: std::time::Duration,
529  ) -> Option<std::process::ExitStatus> {
530    // Redirect stdio so a slow inkscape doesn't block on a full pipe.
531    cmd
532      .stdout(std::process::Stdio::null())
533      .stderr(std::process::Stdio::null());
534    let mut child = cmd.spawn().ok()?;
535    let start = std::time::Instant::now();
536    loop {
537      match child.try_wait() {
538        Ok(Some(status)) => return Some(status),
539        Ok(None) => {
540          if start.elapsed() >= timeout {
541            let _ = child.kill();
542            let _ = child.wait();
543            return None;
544          }
545          std::thread::sleep(std::time::Duration::from_millis(50));
546        },
547        Err(_) => {
548          let _ = child.kill();
549          let _ = child.wait();
550          return None;
551        },
552      }
553    }
554  }

556  /// Parse SVG viewBox ("minX minY width height") and return (width, height).
557  /// Falls back to `width`/`height` root attributes if viewBox is missing.
558  /// Returns None on parse failure so callers can skip dimension attributes.
559  fn read_svg_dimensions(path: &str) -> Option<(u32, u32)> {
560    let content = std::fs::read_to_string(path).ok()?;
561    // Look at just the root <svg> opening tag (first ~2 KB is enough).
562    // We must skip the optional `<?xml … ?>` prolog and any `<!-- … -->`
563    // or `<!DOCTYPE …>` preamble — otherwise `find('>')` matches the
564    // prolog's `?>` instead of the root tag.
565    let head = &content[..content.len().min(2048)];
566    let svg_start = head.find("<svg")?;
567    let svg_rest = &head[svg_start..];
568    let svg_tag_end = svg_rest.find('>')?;
569    let root = &svg_rest[..=svg_tag_end];
570    let extract = |attr: &str| -> Option<String> {
571      let key = format!("{}=\"", attr);
572      let start = root.find(&key)? + key.len();
573      let rest = &root[start..];
574      let end = rest.find('"')?;
575      Some(rest[..end].to_string())
576    };
577    let parse_dim = |s: &str| -> Option<f64> {
578      let s = s.trim();
579      // Strip trailing unit if present (pt, px, mm, etc.)
580      let numeric_end = s
581        .find(|c: char| !c.is_ascii_digit() && c != '.' && c != '-')
582        .unwrap_or(s.len());
583      s[..numeric_end].parse::<f64>().ok()
584    };
585    if let Some(vb) = extract("viewBox") {
586      let parts: Vec<&str> = vb.split_whitespace().collect();
587      if parts.len() == 4 {
588        if let (Some(w), Some(h)) = (parse_dim(parts[2]), parse_dim(parts[3])) {
589          return Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32));
590        }
591      }
592    }
593    let w = extract("width").as_deref().and_then(parse_dim);
594    let h = extract("height").as_deref().and_then(parse_dim);
595    match (w, h) {
596      (Some(w), Some(h)) => Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32)),
597      _ => None,
598    }
599  }

601  /// Decide whether the vector-SVG path should be attempted for this PDF
602  /// source. File-size heuristic: small PDFs (< threshold KB) are nearly
603  /// always vector-authored plots; large PDFs typically contain embedded
604  /// rasters that inkscape re-renders as absurdly large SVG (empirical
605  /// measurement in round-17 — see upstream brucemiller/LaTeXML#902).
606  fn should_try_svg_path(source: &str, threshold_kb: u32) -> bool {
607    if threshold_kb == 0 {
608      return false;
609    }
610    if !source.to_lowercase().ends_with(".pdf") {
611      return false;
612    }
613    match std::fs::metadata(source) {
614      Ok(md) => md.len() <= (threshold_kb as u64) * 1024,
615      Err(_) => false,
616    }
617  }
618
619  /// Convert a graphics file using ImageMagick's `convert` command.
620  /// Perl: image_graphicx_complex via Image::Magick / convert CLI.
621  /// `page` is 1-based (graphicx convention); converted to 0-based for ImageMagick.
622  fn convert_image(source: &str, dest: &str, _dpi: u32, page: Option<u32>) -> bool {
623    // Build the source argument with optional page selector
624    // Perl: image_read reads "$source[$page]" where $page = ($page // 1) - 1
625    let source_arg = if let Some(p) = page {
626      format!("{}[{}]", source, p.saturating_sub(1))
627    } else {
628      // No page specified: use [0] for PDFs to avoid converting all pages
629      if source.to_lowercase().ends_with(".pdf") {
630        format!("{}[0]", source)
631      } else {
632        source.to_string()
633      }
634    };
635    // Shell out to convert (matching Perl's approach)
636    // -define pdf:use-cropbox=true matches Perl's Image::Magick option (line 466)
637    let result = std::process::Command::new("convert")
638      .arg("-define")
639      .arg("pdf:use-cropbox=true")
640      .arg("-density")
641      .arg("150")
642      .arg(&source_arg)
643      .arg(dest)
644      .output();
645    match result {
646      Ok(output) => output.status.success(),
647      Err(_) => false,
648    }
649  }
650}

652impl Processor for Graphics {
653  fn get_name(&self) -> &str { &self.name }
654
655  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
656    doc.findnodes("//ltx:graphics[not(@imagesrc)]")
657  }
658
659  fn process(&mut self, doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
660    let mut search_paths = self.find_graphics_paths(&doc);
661    search_paths.extend(doc.get_search_paths().iter().cloned());
662    // Also add source directory
663    let source_dir = doc.get_source_directory().to_string();
664    if !source_dir.is_empty() && !search_paths.contains(&source_dir) {
665      search_paths.push(source_dir.clone());
666    }
667    // Add current directory as fallback
668    if !search_paths.contains(&".".to_string()) {
669      search_paths.push(".".to_string());
670    }
671
672    let dest_dir = doc.get_destination_directory().unwrap_or(".").to_string();
673    // Read DPI/magnify/zoomout from processing instructions (set by latexml.sty)
674    let dpi = self
675      .get_parameter(&doc, "DPI")
676      .map(|v| v as u32)
677      .or(self.dpi)
678      .unwrap_or(100);
679    let magnify = self.get_parameter(&doc, "magnify").unwrap_or(self.magnify);
680    let _zoomout = self.get_parameter(&doc, "zoomout").unwrap_or(self.zoomout);
681    // Perl: effective DPI = DPI * magnify / zoomout (used for scale-to transforms)
682    let effective_dpi = ((dpi as f64) * magnify / _zoomout) as u32;
683    let n_to_process = nodes.len();
684
685    // Counter for generating unique resource names (like Perl's generateResourcePathname)
686    let mut resource_counter: u32 = 0;
687
688    // Two-phase plan so the slow per-image `convert` subprocess and
689    // `read_image_dimensions` calls can run in parallel without touching
690    // the libxml DOM off-thread.
691    //
692    // Phase 1 (serial): read each node's attributes, resolve source path,
693    // decide the conversion kind, and allocate resource-name counters.
694    //   - `Plan::NotFound` — apply fallback on the main thread later
695    //   - `Plan::Copy { .. }` — apply on the main thread later (cheap)
696    //   - `Plan::Convert { .. }` — independent; run in parallel.
697    // Phase 2 (parallel): run convert_image + read_image_dimensions for
698    // `Plan::Convert` entries. Produces `JobResult`s keyed by node index.
699    // Phase 3 (serial): apply DOM mutations on the main thread in original
700    // node order so attribute writes happen on the libxml-owning thread.
701    enum Plan {
702      NotFound {
703        idx:     usize,
704        graphic: String,
705      },
706      Copy {
707        idx:     usize,
708        source:  String,
709        options: String,
710      },
711      Convert {
712        idx:          usize,
713        source:       String,
714        options:      String,
715        page:         Option<u32>,
716        rel_dest:     String,
717        abs_dest_str: String,
718        /// `Some((rel_svg, abs_svg_str))` when the worker should
719        /// first attempt the inkscape-SVG path and only fall back
720        /// to `convert` on failure. `None` means the classic
721        /// raster-only path.
722        svg_paths:    Option<(String, String)>,
723      },
724    }
725    struct ConvertOutcome {
726      idx:      usize,
727      /// Path to write into `imagesrc`; `None` if both convert and copy-fallback failed.
728      imagesrc: Option<String>,
729      /// Raw (pre-transform) dimensions read from whichever file we ended up with.
730      raw_dims: Option<(u32, u32)>,
731      /// Options passed to transforms (captured once to avoid a DOM read off-thread).
732      options:  String,
733    }
734
735    let mut plans: Vec<Plan> = Vec::with_capacity(n_to_process);
736    for (idx, node) in nodes.iter().enumerate() {
737      let options = node.get_attribute("options").unwrap_or_default();
738      let page = Self::parse_page_option(&options);
739      let Some(source) = self.find_graphic_file(&doc, node, &search_paths) else {
740        let graphic = node
741          .get_attribute("graphic")
742          .unwrap_or_else(|| "none".to_string());
743        plans.push(Plan::NotFound { idx, graphic });
744        continue;
745      };
746      let src_ext = Path::new(&source)
747        .extension()
748        .and_then(|e| e.to_str())
749        .unwrap_or("")
750        .to_lowercase();
751      let props = self.type_properties.get(&src_ext).cloned();
752      let dest_type = props
753        .as_ref()
754        .and_then(|p| p.destination_type.as_ref())
755        .cloned()
756        .unwrap_or(src_ext.clone());
757      let needs_conversion = dest_type != src_ext;
758      let has_page = page.is_some();
759      if needs_conversion || has_page {
760        let dest_name = if has_page {
761          resource_counter += 1;
762          format!("x{}", resource_counter)
763        } else {
764          Path::new(&source)
765            .file_stem()
766            .and_then(|s| s.to_str())
767            .unwrap_or("image")
768            .to_string()
769        };
770        // Vector-SVG path: opt-in for small PDFs only. We prepare an
771        // alternate `.svg` destination path alongside the normal raster
772        // destination so the worker can try inkscape first, then fall
773        // back. The file-size heuristic gates this — see
774        // `should_try_svg_path`.
775        let try_svg = Self::should_try_svg_path(&source, self.svg_threshold_kb);
776        let rel_dest = format!("{}.{}", dest_name, dest_type);
777        let abs_dest = PathBuf::from(&dest_dir).join(&rel_dest);
778        if let Some(parent) = abs_dest.parent() {
779          std::fs::create_dir_all(parent).ok();
780        }
781        let abs_dest_str = abs_dest.to_string_lossy().to_string();
782        let svg_paths = if try_svg {
783          let rel_svg = format!("{}.svg", dest_name);
784          let abs_svg = PathBuf::from(&dest_dir).join(&rel_svg);
785          let abs_svg_str = abs_svg.to_string_lossy().to_string();
786          Some((rel_svg, abs_svg_str))
787        } else {
788          None
789        };
790        plans.push(Plan::Convert {
791          idx,
792          source,
793          options,
794          page,
795          rel_dest,
796          abs_dest_str,
797          svg_paths,
798        });
799      } else {
800        plans.push(Plan::Copy { idx, source, options });
801      }
802    }
803
804    // Phase 2: parallel conversions. Bounded worker count to avoid
805    // oversubscribing when many images are in flight. `convert` itself
806    // is single-threaded per invocation, so the ceiling is useful CPU
807    // parallelism — capped at a reasonable limit to avoid fork/memory
808    // storms with many-image papers.
809    let convert_count = plans
810      .iter()
811      .filter(|p| matches!(p, Plan::Convert { .. }))
812      .count();
813    let worker_cap = std::thread::available_parallelism()
814      .map(|n| n.get())
815      .unwrap_or(4)
816      .clamp(1, 8);
817    let n_workers = convert_count.min(worker_cap).max(1);
818    let source_dir_ref = source_dir.as_str();
819    let dest_dir_ref = dest_dir.as_str();
820    let mut outcomes: Vec<ConvertOutcome> = Vec::with_capacity(convert_count);
821    if convert_count > 0 {
822      use std::sync::Mutex;
823      use std::sync::atomic::{AtomicUsize, Ordering};
824      let next = AtomicUsize::new(0);
825      let out = Mutex::new(Vec::<ConvertOutcome>::with_capacity(convert_count));
826      // Collect just the Convert entries into a fresh Vec so worker index
827      // access is O(1).
828      let jobs: Vec<&Plan> = plans
829        .iter()
830        .filter(|p| matches!(p, Plan::Convert { .. }))
831        .collect();
832      std::thread::scope(|s| {
833        for _ in 0..n_workers {
834          s.spawn(|| {
835            loop {
836              let i = next.fetch_add(1, Ordering::Relaxed);
837              if i >= jobs.len() {
838                break;
839              }
840              let Plan::Convert {
841                idx,
842                source,
843                options,
844                page,
845                rel_dest,
846                abs_dest_str,
847                svg_paths,
848              } = jobs[i]
849              else {
850                unreachable!()
851              };
852              // Try vector-SVG path first if requested for this source.
853              // On success, pick dimensions from the SVG viewBox.
854              let svg_outcome = if let Some((rel_svg, abs_svg)) = svg_paths {
855                if Self::convert_image_svg(source, abs_svg, *page) {
856                  let raw_dims = Self::read_svg_dimensions(abs_svg);
857                  Some(ConvertOutcome {
858                    idx: *idx,
859                    imagesrc: Some(rel_svg.clone()),
860                    raw_dims,
861                    options: options.clone(),
862                  })
863                } else {
864                  log::warn!(
865                    "Graphics: inkscape SVG path failed for {}, falling back to convert",
866                    source
867                  );
868                  None
869                }
870              } else {
871                None
872              };
873              let outcome = if let Some(o) = svg_outcome {
874                o
875              } else if Self::convert_image(source, abs_dest_str, dpi, *page) {
876                ConvertOutcome {
877                  idx:      *idx,
878                  imagesrc: Some(rel_dest.clone()),
879                  raw_dims: Self::read_image_dimensions(abs_dest_str),
880                  options:  options.clone(),
881                }
882              } else {
883                log::warn!("Graphics: Failed to convert {} to {}", source, abs_dest_str);
884                if let Some(rel) = Self::copy_to_destination(source, source_dir_ref, dest_dir_ref) {
885                  ConvertOutcome {
886                    idx:      *idx,
887                    imagesrc: Some(rel),
888                    raw_dims: Self::read_image_dimensions(source),
889                    options:  options.clone(),
890                  }
891                } else {
892                  ConvertOutcome {
893                    idx:      *idx,
894                    imagesrc: None,
895                    raw_dims: None,
896                    options:  options.clone(),
897                  }
898                }
899              };
900              out.lock().unwrap().push(outcome);
901            }
902          });
903        }
904      });
905      outcomes = out.into_inner().unwrap();
906      outcomes.sort_by_key(|o| o.idx);
907    }
908    let mut outcome_iter = outcomes.into_iter().peekable();
909
910    // Phase 3: serial DOM mutations. Preserves original node order.
911    let apply_transforms =
912      |options: &str, raw_dims: Option<(u32, u32)>| -> (Option<u32>, Option<u32>) {
913        match raw_dims {
914          Some((w, h)) if !options.is_empty() => {
915            let (tw, th) = Self::apply_graphicx_transforms(w, h, options, effective_dpi);
916            (Some(tw), Some(th))
917          },
918          Some((w, h)) => (Some(w), Some(h)),
919          None => (None, None),
920        }
921      };
922    for plan in &plans {
923      match plan {
924        Plan::NotFound { idx, graphic } => {
925          log::warn!("Graphics: No source found for {}", graphic);
926          let mut node_mut = nodes[*idx].clone();
927          node_mut.set_attribute("imagesrc", graphic).ok();
928        },
929        Plan::Copy { idx, source, options } => {
930          let mut node_mut = nodes[*idx].clone();
931          if let Some(rel) = Self::copy_to_destination(source, &source_dir, &dest_dir) {
932            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
933            Self::set_graphic_src(&mut node_mut, &rel, w, h);
934          } else {
935            let rel_path = Path::new(source)
936              .strip_prefix(&source_dir)
937              .unwrap_or(Path::new(source));
938            let rel_str = rel_path.to_string_lossy().to_string();
939            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
940            Self::set_graphic_src(&mut node_mut, &rel_str, w, h);
941          }
942        },
943        Plan::Convert { idx, .. } => {
944          // Pull the matching outcome. Outcomes are sorted by idx and each
945          // Convert plan has a unique idx, so peek-and-consume is correct.
946          if let Some(out) = outcome_iter.peek() {
947            if out.idx == *idx {
948              let out = outcome_iter.next().unwrap();
949              let mut node_mut = nodes[*idx].clone();
950              if let Some(imagesrc) = out.imagesrc {
951                let (w, h) = apply_transforms(&out.options, out.raw_dims);
952                Self::set_graphic_src(&mut node_mut, &imagesrc, w, h);
953              }
954            }
955          }
956        },
957      }
958    }
959
960    log::info!(
961      "Graphics {} {} to process",
962      doc.get_destination().unwrap_or("?"),
963      n_to_process
964    );
965    Ok(vec![doc])
966  }
967}

1016  fn should_try_svg_path_heuristic() {

1020    let big_pdf = tmp.join("big.pdf");
1021    let png = tmp.join("raster.png");
1022    std::fs::write(&small_pdf, vec![0u8; 10 * 1024]).unwrap(); // 10 KB
1023    std::fs::write(&big_pdf, vec![0u8; 500 * 1024]).unwrap(); // 500 KB
1024    std::fs::write(&png, vec![0u8; 10 * 1024]).unwrap(); // PNG, irrelevant size
1025
1026    // threshold = 0 → always false.
1027    assert!(!Graphics::should_try_svg_path(
1028      small_pdf.to_str().unwrap(),
1029      0

1033      small_pdf.to_str().unwrap(),

1038      big_pdf.to_str().unwrap(),
1039      200

1043    // Missing file → false, not panic.
