209  /// Find the graphics source file for a node.
210  ///
211  /// Port of `Graphics::findGraphicFile`.
212  fn find_graphic_file(
213    &self,
214    _doc: &PostDocument,
215    node: &Node,
216    search_paths: &[String],
217  ) -> Option<String> {
218    let source = node.get_attribute("graphic")?;
219
220    // Check candidates attribute first (comma-separated list of found files)
221    // Perl: findGraphicFile checks each candidate path, resolving relative to search paths.
222    if let Some(candidates) = node.get_attribute("candidates") {
223      // Pick the best candidate by desirability
224      let mut best: Option<(String, i32)> = None;
225      for path in candidates.split(',') {
226        let path = path.trim();
227        if path.is_empty() {
228          continue;
229        }
230        // Try the path directly, then in each search directory
231        let resolved = if Path::new(path).exists() {
232          Some(path.to_string())
233        } else {
234          search_paths.iter().find_map(|sp| {
235            let candidate = format!("{}/{}", sp, path);
236            if Path::new(&candidate).exists() {
237              Some(candidate)
238            } else {
239              None
240            }
241          })
242        };
243        if let Some(resolved_path) = resolved {
244          let ext = Path::new(&resolved_path)
245            .extension()
246            .and_then(|e| e.to_str())
247            .unwrap_or("")
248            .to_lowercase();
249          // Skip candidates whose extension is NOT a known graphics type.
250          // Perl's `findGraphicFile` re-searches with `types =>
251          // getGraphicsSourceTypes` (Post/Graphics.pm L150-151), which excludes
252          // non-graphics types — notably `.pdf_tex`, the inkscape "PDF+LaTeX"
253          // wrapper that is `\input`'d (it itself does `\includegraphics{grid}`),
254          // NOT a raster/vector image. `image_candidates` is deliberately
255          // unfiltered (matching Perl's `types => ['*']`), so a sibling
256          // `grid.pdf_tex` lands in the `candidates` attribute next to the real
257          // `grid.pdf`/`grid.eps`; without this filter it sorts first and gets
258          // picked, then fails to convert (`pdf_tex` has no destination_type).
259          // Empty ext is kept (the file may carry no extension but known content).
260          // Witness 1907.12308 (`\input{grid.pdf_tex}` → `\includegraphics{grid}`).
261          if !ext.is_empty() && !self.graphics_types.iter().any(|t| t == &ext) {
262            continue;
263          }
264          let props = self.type_properties.get(&ext);
265          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
266          let is_same_type = props
267            .and_then(|p| p.destination_type.as_ref())
268            .map(|dt| dt == &ext)
269            .unwrap_or(false);
270          let desirability = if is_same_type { 10 } else { d };
271          if best.as_ref().is_none_or(|(_, bd)| desirability > *bd) {
272            best = Some((resolved_path, desirability));
273          }
274        }
275      }
276      if let Some((path, _)) = best {
277        return Some(path);
278      }
279    }
280
281    // Search for the file in search paths
282    let path = Path::new(&source);
283    let name = path.file_stem().and_then(|s| s.to_str()).unwrap_or(&source);
284    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or("");
285
286    let file_base = if dir.is_empty() {
287      name.to_string()
288    } else {
289      format!("{}/{}", dir, name)
290    };
291
292    // Try each type in search paths
293    let mut best_desirability: i32 = -1;
294    let mut best_path: Option<String> = None;
295
296    let types: Vec<String> = self
297      .graphics_types
298      .iter()
299      .flat_map(|t| vec![t.clone(), t.to_uppercase()])
300      .collect();
301
302    for search_path in search_paths {
303      // Try without extension first (source might already have one)
304      let candidate = if search_path.is_empty() {
305        source.clone()
306      } else {
307        format!("{}/{}", search_path, source)
308      };
309      if Path::new(&candidate).exists() {
310        let ext = Path::new(&candidate)
311          .extension()
312          .and_then(|e| e.to_str())
313          .unwrap_or("")
314          .to_lowercase();
315        let props = self.type_properties.get(&ext);
316        let d = props.map(|p| p.desirability as i32).unwrap_or(5);
317        if d > best_desirability {
318          best_desirability = d;
319          best_path = Some(candidate);
320        }
321      }
322      // Try each extension
323      for ext in &types {
324        let candidate = format!("{}/{}.{}", search_path, file_base, ext);
325        if Path::new(&candidate).exists() {
326          let props = self.type_properties.get(&ext.to_lowercase());
327          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
328          let is_same_type = props
329            .and_then(|p| p.destination_type.as_ref())
330            .map(|dt| dt == &ext.to_lowercase())
331            .unwrap_or(false);
332          let desirability = if is_same_type { 10 } else { d };
333          if desirability > best_desirability {
334            best_desirability = desirability;
335            best_path = Some(candidate);
336          }
337        }
338      }
339    }
340
341    best_path
342  }

344  /// Set the image source attributes on a graphics node.
345  ///
346  /// Port of `Graphics::setGraphicSrc`.
347  fn set_graphic_src(node: &mut Node, src: &str, width: Option<u32>, height: Option<u32>) {
348    node.set_attribute("imagesrc", src).ok();
349    // HTML width/height are in pixels (unitless)
350    if let Some(w) = width {
351      node.set_attribute("imagewidth", &w.to_string()).ok();
352    }
353    if let Some(h) = height {
354      node.set_attribute("imageheight", &h.to_string()).ok();
355    }
356    // Set aspect ratio class
357    if let (Some(w), Some(h)) = (width, height) {
358      let class = if w as f64 > 1.24 * h as f64 {
359        "ltx_img_landscape"
360      } else if h as f64 > 1.24 * w as f64 {
361        "ltx_img_portrait"
362      } else {
363        "ltx_img_square"
364      };
365      let existing = node.get_attribute("class").unwrap_or_default();
366      let new_class = if existing.is_empty() {
367        class.to_string()
368      } else {
369        format!("{} {}", existing, class)
370      };
371      node.set_attribute("class", &new_class).ok();
372    }
373  }

375  /// Find graphicspath from processing instructions.
376  fn find_graphics_paths(&self, doc: &PostDocument) -> Vec<String> {
377    // Perl `Post/Graphics.pm:91`:
378    //   [map { pathname_canonical($_) }
379    //    $self->findGraphicsPaths($doc), $doc->getSearchPaths]
380    // — the search paths are the union of graphicspath PIs PLUS the
381    // document's own search paths (typically the source directory). The
382    // prior Rust port included only the PI half, which left every paper
383    // with raw `.ps`/`.eps`/etc. files in the source directory (and no
384    // explicit `\graphicspath{...}`) emitting `Error:expected:source`
385    // for every figure, even though the source files are present.
386    // Driver: astro-ph0002170 (8 .ps figures in the zip, all "not found").
387    use std::sync::LazyLock;
388    static GRAPHICSPATH_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
389      regex::Regex::new(r#"^\s*graphicspath\s*=\s*[\"'](.*?)[\"']\s*$"#).unwrap()
390    });
391    let mut paths = Vec::new();
392    for pi in doc.findnodes(".//processing-instruction('latexml')") {
393      let text = pi.get_content();
394      if let Some(cap) = GRAPHICSPATH_RE.captures(&text) {
395        paths.push(cap[1].to_string());
396      }
397    }
398    paths.extend(doc.get_search_paths().iter().cloned());
399    paths
400  }

402  /// Read a named parameter from processing instructions.
403  /// Port of Perl's `Graphics::getParameter`.
404  /// Checks both direct PI (`<?latexml DPI="100"?>`) and
405  /// latexml.sty package options (`<?latexml package="latexml" options="magnify=1.2"?>`).
406  fn get_parameter(&self, doc: &PostDocument, param: &str) -> Option<f64> {
407    let direct_re = regex::Regex::new(&format!(
408      r#"^\s*{}\s*=\s*[\"']?([\d.]+)[\"']?\s*$"#,
409      regex::escape(param)
410    ))
411    .ok()?;
412    let options_re =
413      regex::Regex::new(r#"package\s*=\s*[\"']latexml[\"'].*options\s*=\s*[\"'](.*?)[\"']"#)
414        .ok()?;
415    let param_in_options_re =
416      regex::Regex::new(&format!(r#"\b{}\s*=\s*([\d.]+)"#, regex::escape(param))).ok()?;
417
418    for pi in doc.findnodes(".//processing-instruction('latexml')") {
419      let text = pi.get_content();
420      if let Some(cap) = direct_re.captures(&text) {
421        return cap[1].parse().ok();
422      }
423      if let Some(cap) = options_re.captures(&text) {
424        if let Some(inner) = param_in_options_re.captures(&cap[1]) {
425          return inner[1].parse().ok();
426        }
427      }
428    }
429    None
430  }

442  /// to one of {0, 90, 180, 270} when within 5° of those targets,
443  /// otherwise the raw float (rotation of arbitrary angles is
444  /// handled separately and is more complex due to bounding-box
445  /// changes).
446  fn parse_angle_option(options: &str) -> Option<f64> {
447    for opt in options.split(',') {
448      let opt = opt.trim();
449      if let Some((key, val)) = opt.split_once('=') {
450        if key.trim() == "angle" {
451          return val.trim().parse::<f64>().ok();
452        }
453      }
454    }
455    None

458  /// Parse graphicx options and apply transforms to image dimensions.
459  /// Port of Perl's `getTransform` + `image_graphicx_trivial`.
460  ///
461  /// Handles: scale=N, width=Npt, height=Npt, keepaspectratio, angle=N.
462  fn apply_graphicx_transforms(raw_w: u32, raw_h: u32, options: &str, dpi: u32) -> (u32, u32) {
463    let dppt = dpi as f64 / 72.27; // dots per point
464
465    // angle=N — for axis-aligned rotations (90, -90, 180, 270, ...),
466    // swap width/height so the HTML's imagewidth/imageheight attrs
467    // match the rotated physical image. Driver: 1303.5091 Figs 5-7
468    // use `[angle=90,scale=0.75]` and rendered upside-down without
469    // this swap.
470    let angle = Self::parse_angle_option(options).unwrap_or(0.0);
471    let rot_mod = ((angle.rem_euclid(360.0) + 0.5).floor() as i32 % 360 + 360) % 360;
472    let (mut w, mut h) = if rot_mod == 90 || rot_mod == 270 {
473      (raw_h as f64, raw_w as f64)
474    } else {
475      (raw_w as f64, raw_h as f64)
476    };
477
478    // Parse options as key=value pairs
479    let mut scale: Option<f64> = None;
480    let mut target_width: Option<f64> = None;
481    let mut target_height: Option<f64> = None;
482    let mut keep_aspect = false;
483
484    for opt in options.split(',') {
485      let opt = opt.trim();
486      if let Some((key, val)) = opt.split_once('=') {
487        let key = key.trim();
488        let val = val.trim();
489        match key {
490          "scale" => {
491            scale = val.parse::<f64>().ok();
492          },
493          "width" => {
494            // Parse dimension: "345.0pt" or "345pt" or bare number
495            let val = val.trim_end_matches("pt").trim_end_matches("px");
496            target_width = val.parse::<f64>().ok();
497          },
498          "height" => {
499            let val = val.trim_end_matches("pt").trim_end_matches("px");
500            target_height = val.parse::<f64>().ok();
501          },
502          "keepaspectratio" => {
503            keep_aspect = val == "true" || val == "1" || val.is_empty();
504          },
505          _ => {},
506        }
507      } else if opt == "keepaspectratio" {
508        keep_aspect = true;
509      }
510    }
511
512    // Apply transforms (matching Perl's image_graphicx_trivial).
513    //
514    // graphicx semantics (texbook L182 of graphics.dtx):
515    //   * scale=S         — both dims × S
516    //   * width=W only    — preserve aspect: height auto-scales by W/raw_w
517    //   * height=H only   — preserve aspect: width auto-scales by H/raw_h
518    //   * width=W height=H (no keepaspectratio)
519    //                     — stretch independently
520    //   * +keepaspectratio — use the more constraining dimension
521    //
522    // Earlier Rust port did `th = target_height.unwrap_or(h / dppt)` in the
523    // width-only path, which fed the unscaled raw height back through and
524    // emitted `width=W height=raw_pixels` — visibly wrong (square sources
525    // displayed as 4:1 ribbons; witness astro-ph0005397 Fig 11 sfh_burst).
526    if let Some(s) = scale {
527      w *= s;
528      h *= s;
529    } else {
530      match (target_width, target_height) {
531        (Some(tw), Some(th)) if keep_aspect => {
532          let scale_w = tw / (w / dppt);
533          let scale_h = th / (h / dppt);
534          let s = scale_w.min(scale_h);
535          w = (w / dppt * s * dppt).ceil();
536          h = (h / dppt * s * dppt).ceil();
537        },
538        (Some(tw), Some(th)) => {
539          w = (tw * dppt).ceil();
540          h = (th * dppt).ceil();
541        },
542        (Some(tw), None) => {
543          // width-only: auto-scale height proportionally.
544          let s = tw / (w / dppt);
545          w = (tw * dppt).ceil();
546          h = (h * s).ceil();
547        },
548        (None, Some(th)) => {
549          // height-only: auto-scale width proportionally.
550          let s = th / (h / dppt);
551          h = (th * dppt).ceil();
552          w = (w * s).ceil();
553        },
554        (None, None) => {
555          // No dimension hints — keep raw pixel size.
556        },
557      }
558    }
559
560    (w.max(1.0) as u32, h.max(1.0) as u32)
561  }

563  /// Physically rotate a rasterized image via `convert -rotate N`.
564  /// Called after the rasterizer produces a PNG when graphicx
565  /// options include `angle=N`. Returns true on success.
566  /// Pre-condition: `dest` exists. Post-condition: `dest` is
567  /// in-place rotated.
568  /// Content fingerprint for graphics-asset deduplication. SipHash
569  /// (`std::collections::hash_map::DefaultHasher`) over the file's
570  /// raw bytes. Returns `None` if the file can't be opened.
571  ///
572  /// Collisions are theoretically possible but astronomically unlikely
573  /// for paper-sized graphics dirs (worst case: a few hundred files
574  /// per paper, all under 50 MB). We use a u64 hash as the dedup key.
575  ///
576  /// Purpose: byte-identical files `x1.pdf` and `x2.pdf` (e.g. shared
577  /// figures across subsections, or duplicated by the author) should
578  /// share one rasterized PNG/SVG in the output bundle. Both `<img>`
579  /// tags then reference the first-seen filename's stem — saves both
580  /// conversion time AND output-bundle disk space.
581  fn hash_file_content(path: &str) -> Option<u64> {
582    use std::hash::Hasher;
583    use std::io::Read;
584    let mut file = std::fs::File::open(path).ok()?;
585    let mut hasher = std::collections::hash_map::DefaultHasher::new();
586    let mut buf = [0u8; 65536];
587    loop {
588      let n = file.read(&mut buf).ok()?;
589      if n == 0 {
590        break;
591      }
592      hasher.write(&buf[..n]);
593    }
594    Some(hasher.finish())
595  }

633  /// Returns the destination path (relative to dest_dir).

653    }
654

656    Some(rel_path.to_string_lossy().to_string())

675  /// Try to convert a PDF to plain SVG via `inkscape`, preserving vector
676  /// content. Returns `true` on success. Tracks upstream
677  /// brucemiller/LaTeXML#902.
678  ///
679  /// Caller decides when to attempt this — typically only for PDF sources
680  /// below a file-size threshold, because inkscape on raster-embedded PDFs
681  /// produces massive output (>100 MB) and can take 40+ seconds
682  /// (measured: Fade.pdf 1.7 MB → 46 s / 102 MB SVG vs `convert` 1.4 s /
683  /// 61 KB PNG).
684  ///
685  /// `page` is 1-based (graphicx convention); converted to 0-based for
686  /// inkscape's `--pdf-page`.
687  ///
688  /// Guarded by a **hard timeout** (15 s default; see
689  /// `inkscape_timeout_secs`). Pathological small-PDF cases have been
690  /// observed — if inkscape is still running after the deadline we SIGKILL
691  /// it and return `false` so the caller falls back to ImageMagick. The
692  /// timeout is generous enough (15 s) for all well-behaved small vector
693  /// plots and strict enough to prevent the 46 s+ runaway behaviour seen
694  /// on Fade.pdf-class inputs.
695  fn convert_image_svg(source: &str, dest: &str, page: Option<u32>) -> bool {
696    // Try fast vector rasterizers in order of measured speed +
697    // gzip-compressibility on the canvas slow-tail. Each is gated by
698    // `MAX_SVG_OUTPUT_BYTES`; pathological vector-heavy PDFs (e.g.
699    // R-Graphics `W.pdf`) can emit >100 MB SVG which we discard so
700    // the caller falls back to raster.
701    //
702    // Order (subprocess; library license doesn't propagate):
703    //   1. mutool (MuPDF CLI) — fastest, plus ~4× more
704    //      gzip-compressible SVG output than pdftocairo (1.5 MB vs
705    //      6.0 MB gz on matplotlib scatter).
706    //   2. pdftocairo (poppler) — universally available with TeX Live.
707    //      20-40× faster than inkscape on benign vector PDFs.
708    //   3. inkscape — last vector resort. Some PDFs that fail
709    //      poppler still render via inkscape (but it can time out).
710    if Self::convert_image_svg_mutool(source, dest, page) {
711      return true;
712    }
713    if Self::convert_image_svg_pdftocairo(source, dest, page) {
714      return true;
715    }
716    let mut cmd = std::process::Command::new("inkscape");
717    cmd
718      .arg("--export-type=svg")
719      .arg("--export-plain-svg")
720      .arg(format!("--export-filename={}", dest));
721    if let Some(p) = page {
722      cmd.arg(format!("--pdf-page={}", p.saturating_sub(1)));
723    }
724    cmd.arg(source);
725    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
726    match Self::run_with_timeout(cmd, timeout) {
727      Some(status) => {
728        if !(status.success() && Path::new(dest).exists()) {
729          return false;
730        }
731        // Reject pathological inkscape output that explodes to >100 MB
732        // — keep the dest hole open so the worker falls back to raster.
733        if Self::svg_output_too_large(dest) {
734          let _ = std::fs::remove_file(dest);
735          return false;
736        }
737        true
738      },
739      None => {
740        // Subprocess wall-clock timeout; class=`shell` mirrors Perl
741        // LaTeXImages.pm:293 `Error('shell', $cmd, …)`.
742        Warn!(
743          "shell", "inkscape",
744          "Graphics: inkscape SVG conversion for {} exceeded {} s — killed",
745          source,
746          timeout.as_secs()
747        );
748        // Best-effort cleanup of a partial output.
749        let _ = std::fs::remove_file(dest);
750        false
751      },
752    }
753  }

768  /// `mutool convert -F svg` (MuPDF CLI) — first-choice SVG vector
769  /// converter. Faster than `pdftocairo -svg` on vector-heavy PDFs
770  /// (~2×), AND produces output that gzip-compresses ~4× better when
771  /// served as `.svgz`. Subprocess invocation — no MuPDF code linked.
772  ///
773  /// Measured 2026-05-12 on matplotlib AugmentedMSRA10K…pos.pdf:
774  ///   pdftocairo -svg: 1.17 s, 29.9 MB raw, 6.0 MB gz
775  ///   mutool convert:  0.52 s, 29.7 MB raw, 1.5 MB gz
776  ///
777  /// mutool's `convert` emits one file per page via a printf-style
778  /// pattern. We use `%d` to capture the requested page and rename
779  /// the result to the caller's `dest`.
780  fn convert_image_svg_mutool(source: &str, dest: &str, page: Option<u32>) -> bool {
781    let dest_path = Path::new(dest);
782    let parent = dest_path.parent().unwrap_or_else(|| Path::new("."));
783    let stem = dest_path
784      .file_name()
785      .and_then(|s| s.to_str())
786      .unwrap_or("image");
787    let unique = std::time::SystemTime::now()
788      .duration_since(std::time::UNIX_EPOCH)
789      .map(|d| d.as_nanos())
790      .unwrap_or(0);
791    let tmp_pattern = parent.join(format!(".{}.{}.mutool_svg%d.svg", stem, unique));
792    let tmp_pattern_str = tmp_pattern.to_string_lossy().to_string();
793    let p1 = page.map(|p| p.max(1)).unwrap_or(1);
794    let tmp_actual = parent.join(format!(
795      ".{}.{}.mutool_svg{}.svg",
796      stem, unique, p1
797    ));
798    let cleanup = || {
799      let _ = std::fs::remove_file(&tmp_actual);
800    };
801
802    let mut cmd = std::process::Command::new("mutool");
803    cmd
804      .arg("convert")
805      .arg("-F")
806      .arg("svg")
807      .arg("-o")
808      .arg(&tmp_pattern_str)
809      .arg(source)
810      .arg(p1.to_string());
811    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
812    let ok = Self::run_with_timeout(cmd, timeout)
813      .map(|status| status.success())
814      .unwrap_or(false)
815      && tmp_actual.exists();
816    if !ok {
817      cleanup();
818      return false;
819    }
820    if std::fs::metadata(&tmp_actual)
821      .map(|md| md.len() > Self::MAX_SVG_OUTPUT_BYTES)
822      .unwrap_or(true)
823    {
824      cleanup();
825      return false;
826    }
827    let _ = std::fs::remove_file(dest);
828    let installed = std::fs::rename(&tmp_actual, dest)
829      .or_else(|_| std::fs::copy(&tmp_actual, dest).map(|_| ()))
830      .is_ok()
831      && dest_path.exists();
832    cleanup();
833    installed
834  }

882  /// Run `cmd` and enforce a wall-clock timeout. Returns `Some(status)` on
883  /// clean exit, `None` if the child was killed on timeout or spawn
884  /// failed. Polls every 50 ms — cheap compared to the subprocess cost.
885  ///
886  /// On Unix, each child runs in its OWN session (setsid via pre_exec)
887  /// so a timeout kill targets the entire process group with `killpg`.
888  /// Without that, ImageMagick's `convert` was spawning `gs` and dying
889  /// on SIGKILL while leaving gs orphaned — those gs processes held on
890  /// for 10+ minutes per pathological PDF and stalled large sandbox
891  /// runs. The same hardening protects inkscape / pdftocairo / ps2pdf.
892  fn run_with_timeout(
893    mut cmd: std::process::Command,
894    timeout: std::time::Duration,
895  ) -> Option<std::process::ExitStatus> {
896    // Redirect stdio so a slow inkscape doesn't block on a full pipe.
897    cmd
898      .stdout(std::process::Stdio::null())
899      .stderr(std::process::Stdio::null());
900    #[cfg(unix)]
901    {
902      use std::os::unix::process::CommandExt;
903      // SAFETY: setsid(2) is async-signal-safe and is the documented way
904      // to make a child process group leader between fork() and exec().
905      unsafe {
906        cmd.pre_exec(|| {
907          // SAFETY: same as above — async-signal-safe call permitted here.
908          if libc::setsid() == -1 {
909            // Fall back: setpgid(0, 0). If both fail we proceed anyway.
910            let _ = libc::setpgid(0, 0);
911          }
912          // Die-with-watcher: `setsid` detaches this converter from every
913          // process group, so if the process running `run_with_timeout` is
914          // itself killed (e.g. the LSP server SIGKILLs a preempted body
915          // child mid-post-processing), nothing would ever time out or kill
916          // a runaway gs/inkscape — the exact orphan pathology this
917          // function's group-kill solves, reintroduced one level up.
918          // PR_SET_PDEATHSIG makes the kernel SIGKILL the converter when its
919          // spawning thread dies; it survives execve, so setting it here
920          // (between fork and exec) covers the exec'd tool. Linux-only —
921          // elsewhere the orphan window simply remains. prctl is
922          // async-signal-safe. Guard the fork→prctl race: if the watcher
923          // died before prctl took effect we are already reparented (to
924          // init or a subreaper) — exit instead of running unwatched.
925          #[cfg(target_os = "linux")]
926          {
927            libc::prctl(libc::PR_SET_PDEATHSIG, libc::SIGKILL);
928            if libc::getppid() == 1 {
929              libc::_exit(127);
930            }
931          }
932          Ok(())
933        });
934      }
935    }
936    let mut child = cmd.spawn().ok()?;
937    let pid = child.id() as i32;
938    let kill_group = || {
939      #[cfg(unix)]
940      {
941        // SIGTERM the whole group first (graceful), then SIGKILL after
942        // a brief grace if the leader is still alive. This matches what
943        // `timeout(1) --kill-after` does for the bench script's outer
944        // guard.
945        // SAFETY: killpg(2) on a known pid is documented + safe.
946        unsafe {
947          libc::killpg(pid, libc::SIGTERM);
948        }
949        std::thread::sleep(std::time::Duration::from_millis(200));
950        unsafe {
951          libc::killpg(pid, libc::SIGKILL);
952        }
953      }
954      #[cfg(not(unix))]
955      {
956        // Non-Unix platforms: best-effort PID kill only.
957        let _ = pid;
958      }
959    };
960    let start = std::time::Instant::now();
961    loop {
962      match child.try_wait() {
963        Ok(Some(status)) => return Some(status),
964        Ok(None) => {
965          if start.elapsed() >= timeout {
966            kill_group();
967            let _ = child.kill();
968            let _ = child.wait();
969            return None;
970          }
971          std::thread::sleep(std::time::Duration::from_millis(50));
972        },
973        Err(_) => {
974          kill_group();
975          let _ = child.kill();
976          let _ = child.wait();
977          return None;
978        },
979      }
980    }
981  }

983  /// Parse SVG viewBox ("minX minY width height") and return (width, height).
984  /// Falls back to `width`/`height` root attributes if viewBox is missing.
985  /// Returns None on parse failure so callers can skip dimension attributes.
986  fn read_svg_dimensions(path: &str) -> Option<(u32, u32)> {
987    let content = std::fs::read_to_string(path).ok()?;
988    // Look at just the root <svg> opening tag (first ~2 KB is enough).
989    // We must skip the optional `<?xml … ?>` prolog and any `<!-- … -->`
990    // or `<!DOCTYPE …>` preamble — otherwise `find('>')` matches the
991    // prolog's `?>` instead of the root tag.
992    // UTF-8-safe slice: if the 2048-byte mark falls mid-codepoint, walk
993    // forward to the next char boundary so the slice is always valid.
994    // Witness: 1307.4573 (xfig-pstex_t paper with multi-byte chars in
995    // SVG preamble metadata) — previously FATAL_101 panic at
996    // graphics.rs:957 from `&content[..2048]` cutting a UTF-8 sequence.
997    let head_end = {
998      let mut end = content.len().min(2048);
999      while end < content.len() && !content.is_char_boundary(end) {
1000        end += 1;
1001      }
1002      end
1003    };
1004    let head = &content[..head_end];
1005    let svg_start = head.find("<svg")?;
1006    let svg_rest = &head[svg_start..];
1007    let svg_tag_end = svg_rest.find('>')?;
1008    let root = &svg_rest[..=svg_tag_end];
1009    let extract = |attr: &str| -> Option<String> {
1010      let key = format!("{}=\"", attr);
1011      let start = root.find(&key)? + key.len();
1012      let rest = &root[start..];
1013      let end = rest.find('"')?;
1014      Some(rest[..end].to_string())
1015    };
1016    let parse_dim = |s: &str| -> Option<f64> {
1017      let s = s.trim();
1018      // Strip trailing unit if present (pt, px, mm, etc.)
1019      let numeric_end = s
1020        .find(|c: char| !c.is_ascii_digit() && c != '.' && c != '-')
1021        .unwrap_or(s.len());
1022      s[..numeric_end].parse::<f64>().ok()
1023    };
1024    if let Some(vb) = extract("viewBox") {
1025      let parts: Vec<&str> = vb.split_whitespace().collect();
1026      if parts.len() == 4 {
1027        if let (Some(w), Some(h)) = (parse_dim(parts[2]), parse_dim(parts[3])) {
1028          return Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32));
1029        }
1030      }
1031    }
1032    let w = extract("width").as_deref().and_then(parse_dim);
1033    let h = extract("height").as_deref().and_then(parse_dim);
1034    match (w, h) {
1035      (Some(w), Some(h)) => Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32)),
1036      _ => None,
1037    }
1038  }

1040  /// Decide whether the vector-SVG path should be attempted for this PDF
1041  /// source.
1042  ///
1043  /// Two modes, in priority order:
1044  ///
1045  /// 1. **Explicit threshold** (`threshold_kb > 0`): legacy
1046  ///    `--graphics-svg-threshold-kb N` behaviour — try SVG for PDFs at
1047  ///    most `N` KB, regardless of content. Preserved for back-compat
1048  ///    and as the manual override on canvases where the auto-detector
1049  ///    misclassifies.
1050  /// 2. **Auto-detect default** (`threshold_kb == 0`): scan the PDF
1051  ///    header for `/Subtype /Image` and `/Subtype/Image` (the two
1052  ///    typical formattings of an image XObject declaration). If
1053  ///    NONE is present in the first 256 KB and the file size is at
1054  ///    most 500 KB → try SVG. This is the per-paper relief case
1055  ///    documented in PERFORMANCE.md (130× speedup on the 41 KB
1056  ///    pgfplots fixture).
1057  ///
1058  /// Both modes can be globally disabled via
1059  /// `LATEXML_GRAPHICS_VECTOR_AUTO_OFF=1` (auto-detect only — leaves
1060  /// the explicit-threshold path active).
1061  ///
1062  /// Safety net: any false positive falls back to ImageMagick when
1063  /// inkscape emits >`MAX_SVG_OUTPUT_BYTES` (8 MB) of SVG, so a misread
1064  /// raster PDF degrades to "tried SVG, got too-big output, used
1065  /// convert" instead of a stuck pipeline.
1066  fn should_try_svg_path(source: &str, threshold_kb: u32) -> bool {
1067    if !source.to_lowercase().ends_with(".pdf") {
1068      return false;
1069    }
1070    if threshold_kb > 0 {
1071      // Legacy explicit-threshold path: bytes-only decision.
1072      return match std::fs::metadata(source) {
1073        Ok(md) => md.len() <= (threshold_kb as u64) * 1024,
1074        Err(_) => false,
1075      };
1076    }
1077    // Auto-detect path. Honour the opt-out.
1078    if Self::vector_auto_detect_disabled() {
1079      return false;
1080    }
1081    let len = match std::fs::metadata(source) {
1082      Ok(md) => md.len(),
1083      Err(_) => return false,
1084    };
1085    // Hard upper bound: even if the detector misses an image
1086    // somewhere deeper in the file, a 500 KB cap keeps the
1087    // worst-case wasted inkscape work bounded (~1-2 s before the
1088    // 8 MB output cap kicks in or the conversion finishes anyway).
1089    const AUTO_MAX_BYTES: u64 = 500 * 1024;
1090    if len > AUTO_MAX_BYTES {
1091      return false;
1092    }
1093    !Self::pdf_has_image_xobject(source).unwrap_or(true)
1094  }

1119  /// guarded by `should_try_svg_path`'s outer size check) the entire
1120  /// stream fits comfortably within the limit. Pure-vector PDFs scan
1121  /// in well under a millisecond on modern hardware.
1122  fn pdf_has_image_xobject(source: &str) -> Option<bool> {
1123    use std::io::Read;
1124    const SCAN_LIMIT: usize = 256 * 1024;
1125    let mut f = std::fs::File::open(source).ok()?;
1126    let mut buf = vec![0u8; SCAN_LIMIT];
1127    let n = f.read(&mut buf).ok()?;

1129    // Both spelling variants seen in the wild: `/Subtype /Image` (PDFs
1130    // from inkscape / cairo / latex+dvips) and `/Subtype/Image` (more
1131    // common in pdflatex output and ImageMagick-produced PDFs).
1132    Some(twoway_contains(head, b"/Subtype /Image") || twoway_contains(head, b"/Subtype/Image"))

1149      || source_lc.ends_with(".ai");

1170  /// Returns true when the PS / EPS file's DSC header declares
1171  /// `%%Orientation: Landscape`. PGPLOT and a handful of older
1172  /// scientific renderers emit landscape PS files with a portrait
1173  /// `%%BoundingBox` — content is drawn rotated 90° on the page, and
1174  /// PS-level renderers (gs, IM) ignore the Orientation hint, producing
1175  /// visibly upside-down output. ps2pdf is the one tool in the chain
1176  /// that honors the comment by writing `/Rotate 90` into the PDF
1177  /// header; pdftocairo then renders the rotated PDF correctly.
1178  ///
1179  /// Witness: astro-ph0103041 NickMorgan.fig2.ps. Only the first ~80
1180  /// lines of the DSC prologue are scanned because all conforming PS
1181  /// files emit `%%Orientation:` early.
1182  fn postscript_is_landscape(source: &str) -> bool {
1183    let Ok(file) = std::fs::File::open(source) else {
1184      return false;
1185    };
1186    use std::io::BufRead;
1187    let reader = std::io::BufReader::new(file);
1188    for line in reader.lines().take(80).map_while(Result::ok) {
1189      if let Some(rest) = line.strip_prefix("%%Orientation:") {
1190        return rest.trim().eq_ignore_ascii_case("Landscape");
1191      }
1192    }
1193    false
1194  }

1413  /// Direct Ghostscript rasterization for EPS/PS sources, bypassing
1414  /// ImageMagick's wrapper.
1415  ///
1416  /// `convert` for EPS already shells out to `gs` internally, so by
1417  /// invoking `gs` ourselves we save the IM read-pipeline overhead
1418  /// (~50-200 ms per image on the canvas). gs's `Rotate` direction is
1419  /// CCW — same as graphicx and IM — so this matches the Perl
1420  /// `image_graphicx_complex` semantics exactly. No /Rotate metadata
1421  /// is produced (gs writes PNG/JPG directly), so we don't inherit
1422  /// the rotation-mismatch bug from the disabled ps2pdf path.
1423  fn convert_eps_via_gs(source: &str, dest: &str, density: u32) -> bool {
1424    let dest_path = Path::new(dest);
1425    let parent = dest_path.parent().unwrap_or_else(|| Path::new("."));
1426    let stem = dest_path
1427      .file_name()
1428      .and_then(|s| s.to_str())
1429      .unwrap_or("image");
1430    let unique = std::time::SystemTime::now()
1431      .duration_since(std::time::UNIX_EPOCH)
1432      .map(|d| d.as_nanos())
1433      .unwrap_or(0);
1434    // gs picks its output extension from the device, not the path, so
1435    // we pass it whatever name and rename atomically at the end.
1436    let tmp = parent.join(format!(".{}.{}.gs", stem, unique));
1437    let timeout = std::time::Duration::from_secs(30);
1438    let dest_lc = dest.to_lowercase();
1439    // pngalpha matches IM's `ps:alpha` delegate (the default EPS→PNG
1440    // path). It produces an RGBA PNG where blank canvas is transparent
1441    // — important for plot backgrounds matching document background.
1442    // png16m would force a white background regardless of source.
1443    let device = if dest_lc.ends_with(".jpg") || dest_lc.ends_with(".jpeg") {
1444      "jpeg"
1445    } else {
1446      "pngalpha"
1447    };
1448
1449    let mut cmd = std::process::Command::new("gs");
1450    cmd
1451      .arg("-q")
1452      .arg("-dNOPAUSE")
1453      .arg("-dBATCH")
1454      .arg("-dSAFER")
1455      // Antialiasing — IM passes these through its delegate by
1456      // default. Without them gs produces aliased, jagged output
1457      // that's visibly worse than `convert`. 4 = max quality;
1458      // 2 = balanced; 1 = off. Matches IM's delegate.xml defaults.
1459      .arg("-dTextAlphaBits=4")
1460      .arg("-dGraphicsAlphaBits=4")
1461      // Render the entire page in memory rather than band-by-band.
1462      // Eliminates seam artifacts on large pages. Mirrors IM's
1463      // delegate.xml: -dMaxBitmap=500000000.
1464      .arg("-dMaxBitmap=500000000")
1465      .arg("-dAlignToPixels=0")
1466      .arg("-dGridFitTT=2")
1467      // -dEPSCrop here means "honor the EPS BoundingBox when rendering"
1468      // (a gs rendering flag), NOT the ps2pdf flag that injected
1469      // /Rotate in the earlier disabled path. gs writing PNG never
1470      // produces PDF metadata, so this is safe.
1471      .arg("-dEPSCrop");
1472    // When the EPS declares a `%%BoundingBox`, force the device
1473    // page-size to match it AND lock it via `-dFIXEDMEDIA`. Some EPS
1474    // files (notably `pswrite`-output, e.g. AFPL Ghostscript-generated
1475    // figures like astro-ph0503029/figure7.eps) embed their own
1476    // `setpagedevice` calls that override `-dEPSCrop` and force a full
1477    // US-Letter page (612 × 792 pt), so the content lands at the bottom
1478    // of a 1020 × 1320 px canvas with a 968-pixel blank above it.
1479    // `-dFIXEDMEDIA` makes gs ignore the embedded `setpagedevice` and
1480    // honour our explicit dimensions. Witness: astro-ph0503029 fig 7.
1481    //
1482    // When the BoundingBox is offset from origin (e.g.
1483    // `%%BoundingBox: 117 242 524 567` in hep-ph0608319/data6.ps),
1484    // FIXED page-size alone isn't enough — content drawn at PS
1485    // (117, 242) lands OUTSIDE a (407, 325) page. Translate via
1486    // PostScript `-c "<x0_neg> <y0_neg> translate"` BEFORE the EPS
1487    // file is interpreted, shifting the content to origin (0, 0).
1488    // PS `-c` snippet executes after the device init but before the
1489    // file load. Witness: hep-ph0608319/data6.ps (`(atend)` header,
1490    // real BBox `117 242 524 567`) — without translate the rendered
1491    // 992 × 1403 letter page has a tiny content blob; with translate
1492    // we get a tight 407 × 325 pt crop matching what convert produces.
1493    // Device init flags must precede `-c` / `-f` because gs `-f`
1494    // takes the NEXT argument as a file to interpret; anything after
1495    // `-f` is no longer parsed as an option.
1496    cmd
1497      .arg(format!("-sDEVICE={}", device))
1498      .arg(format!("-r{}", density))
1499      .arg(format!("-sOutputFile={}", tmp.display()));
1500    let bbox_full = read_postscript_bounding_box_full(source);
1501    if let Some((x0, y0, w_pt, h_pt)) = bbox_full {
1502      let w = w_pt.max(1.0).ceil() as u32;
1503      let h = h_pt.max(1.0).ceil() as u32;
1504      cmd
1505        .arg("-dFIXEDMEDIA")
1506        .arg(format!("-dDEVICEWIDTHPOINTS={}", w))
1507        .arg(format!("-dDEVICEHEIGHTPOINTS={}", h));
1508      if x0.abs() > 0.5 || y0.abs() > 0.5 {
1509        cmd
1510          .arg("-c")
1511          .arg(format!("{} {} translate", -x0, -y0))
1512          .arg("-f");
1513      }
1514    }
1515    cmd.arg(source);
1516    let gs_ok = Self::run_with_timeout(cmd, timeout)
1517      .map(|s| s.success())
1518      .unwrap_or(false)
1519      && tmp.exists();
1520    if !gs_ok {
1521      let _ = std::fs::remove_file(&tmp);
1522      return false;
1523    }
1524    let _ = std::fs::remove_file(dest);
1525    let installed = std::fs::rename(&tmp, dest)
1526      .or_else(|_| std::fs::copy(&tmp, dest).map(|_| ()))
1527      .is_ok()
1528      && dest_path.exists();
1529    let _ = std::fs::remove_file(&tmp);
1530    installed
1531  }

1533  /// Convert a graphics file using ImageMagick's `convert` command.
1534  /// Perl: image_graphicx_complex via Image::Magick / convert CLI.
1535  /// `page` is 1-based (graphicx convention); converted to 0-based for ImageMagick.
1536  fn convert_image(source: &str, dest: &str, _dpi: u32, page: Option<u32>) -> bool {
1537    // Build the source argument with optional page selector
1538    // Perl: image_read reads "$source[$page]" where $page = ($page // 1) - 1
1539    let source_arg = if let Some(p) = page {
1540      format!("{}[{}]", source, p.saturating_sub(1))
1541    } else {
1542      // No page specified: use [0] for PDFs to avoid converting all pages
1543      if source.to_lowercase().ends_with(".pdf") {
1544        format!("{}[0]", source)
1545      } else {
1546        source.to_string()
1547      }
1548    };
1549    // Shell out to convert (matching Perl's approach)
1550    // -define pdf:use-cropbox=true matches Perl's Image::Magick option (line 466)
1551    let density = Self::raster_density_for_source(source);
1552    if Self::should_try_eps_pdf_path(source, page)
1553      && Self::convert_eps_via_pdf(source, dest, density)
1554    {
1555      return true;
1556    }
1557    // Fast EPS/PS path: skip the ImageMagick wrapper and call `gs`
1558    // directly. Same renderer, same CCW Rotate convention; ~50-200 ms
1559    // saved per image. Falls through to `convert` on any failure.
1560    // Only attempted when no page selector is present (EPS is
1561    // single-page; PS multi-page handled by `convert`'s `[N]` syntax).
1562    //
1563    // EXCEPT: when the PS file declares `%%Orientation: Landscape` in
1564    // its header comments — typical of PGPLOT output, e.g.
1565    // astro-ph0103041 NickMorgan.fig2.ps. Direct gs / convert ignore
1566    // the Orientation comment and render at literal portrait BBox
1567    // coordinates, producing visibly upside-down output. ps2pdf
1568    // honors %%Orientation and writes `/Rotate 90` into the resulting
1569    // PDF; pdftocairo then honors the PDF rotation and emits
1570    // correctly-oriented landscape pixels. Route those through the
1571    // pdf-intermediate path. The earlier disabled `should_try_eps_pdf_path`
1572    // was about an ORTHOGONAL bug (graphicx angle= compounding with
1573    // ps2pdf-injected /Rotate on portrait files) — the 1303.5091 EPS
1574    // files have NO `%%Orientation:` comment, so they go through the
1575    // gs path as before.
1576    let src_lc = source.to_lowercase();
1577    let is_postscript = src_lc.ends_with(".eps")
1578      || src_lc.ends_with(".epsi")
1579      || src_lc.ends_with(".epsf")
1580      || src_lc.ends_with(".ps");
1581    if is_postscript && page.is_none() {
1582      if Self::postscript_is_landscape(source)
1583        && Self::convert_eps_via_pdf(source, dest, density)
1584      {
1585        return true;
1586      }
1587      if Self::convert_eps_via_gs(source, dest, density) {
1588        return true;
1589      }
1590    }
1591    // For PDF sources, try fast subprocess rasterizers in measured-
1592    // speed order. Subprocess (not linked) so library license doesn't
1593    // propagate — same legal pattern as invoking `git` or `ffmpeg`.
1594    // In-process Rust crates were evaluated 2026-05-12 and rejected
1595    // (mupdf-rs AGPL, poppler-rs GPL, pdfium-render single-threaded).
1596    //
1597    //   1. mutool (MuPDF CLI) — ~1.7× faster than pdftocairo on the
1598    //      canvas slow-tail (matplotlib/pgfplots scatter PDFs).
1599    //   2. pdftocairo (poppler) — universally available with TeX Live;
1600    //      25× faster than convert/gs.
1601    //   3. convert/gs — last-resort, hard-timeout-bounded.
1602    if Self::should_try_pdf_cairo_path(source) && dest.to_lowercase().ends_with(".png") {
1603      if Self::convert_pdf_via_mutool(source, dest, density, page) {
1604        return true;
1605      }
1606      if Self::convert_pdf_via_pdftocairo(source, dest, density, page) {
1607        return true;
1608      }
1609    }
1610    // Wall-clock timeout to bound `gs`-via-`convert` runaways on
1611    // pathological PDFs (raster-heavy or with broken xref tables).
1612    // Matches the inkscape path's defensive bound; without this, an
1613    // arbitrary `convert` invocation could run for minutes and stall
1614    // the entire post-processing phase. 60 s is enough for any
1615    // reasonably-sized graphic; tune via `LATEXML_CONVERT_TIMEOUT_SECS`.
1616    //
1617    // Crucially: `run_with_timeout` puts convert in its own process
1618    // group via setsid+pre_exec (Unix), so killing convert on timeout
1619    // also kills the gs grandchild. Without that, gs orphaned by a
1620    // dying convert kept running 10+ min and stalled the sandbox.
1621    let mut cmd = std::process::Command::new("convert");
1622    cmd
1623      .arg("-define")
1624      .arg("pdf:use-cropbox=true")
1625      .arg("-density")
1626      .arg(density.to_string())
1627      .arg(&source_arg)
1628      .arg(dest);
1629    let timeout = std::time::Duration::from_secs(Self::convert_timeout_secs());
1630    match Self::run_with_timeout(cmd, timeout) {
1631      // Mirror the original `cmd.output()` semantics: report success based
1632      // on exit status alone, not on whether `dest` was actually written.
1633      // (The fake-convert test fixture exits 0 without producing a file.)
1634      Some(status) => status.success(),
1635      None => {
1636        Warn!(
1637          "shell", "convert",
1638          "Graphics: convert/gs for {} exceeded {} s — killed",
1639          source,
1640          timeout.as_secs()
1641        );
1642        let _ = std::fs::remove_file(dest);
1643        false
1644      },
1645    }
1646  }
1647
1648  /// Hard timeout (seconds) for the `convert` subprocess. Mirrors
1649  /// `inkscape_timeout_secs`; default 60 s. Override via
1650  /// `LATEXML_CONVERT_TIMEOUT_SECS` for debugging.
1651  fn convert_timeout_secs() -> u64 { CONVERT_TIMEOUT_SECS.unwrap_or(60) }
1652}

1690/// Returns the EPS BoundingBox as `(x0, y0, w, h)` where (x0, y0) is
1691/// the lower-left corner of the content in PS coords and (w, h) is
1692/// the content extent. Callers needing only the extent can ignore the
1693/// origin via `_`-destructuring or `.map(|(_, _, w, h)| (w, h))`.
1694///
1695/// Handles three DSC variants:
1696///  1. `%%BoundingBox: x0 y0 x1 y1` in the header (most files).
1697///  2. `%%BoundingBox: (atend)` in the header, real values in the
1698///     Trailer at end-of-file (some HIGZ, PAW, certain pswrite output).
1699///  3. `%%HiResBoundingBox: x0.x y0.y x1.x y1.y` — used when literal
1700///     `%%BoundingBox:` is missing.
1701fn read_postscript_bounding_box_full(source: &str) -> Option<(f64, f64, f64, f64)> {
1702  let content = std::fs::read_to_string(source).ok()?;
1703  let mut header_lines = content.lines().take(80);
1704  let mut atend = false;
1705  let mut hi_res: Option<(f64, f64, f64, f64)> = None;
1706  for line in &mut header_lines {
1707    if let Some(rest) = line.strip_prefix("%%BoundingBox:") {
1708      let rest_trim = rest.trim();
1709      if rest_trim.eq_ignore_ascii_case("(atend)") {
1710        atend = true;
1711        continue;
1712      }
1713      if let Some(b) = parse_bbox_quadruple(rest) {
1714        return Some(b);
1715      }
1716    } else if let Some(rest) = line.strip_prefix("%%HiResBoundingBox:") {
1717      hi_res = hi_res.or_else(|| parse_bbox_quadruple(rest));
1718    }
1719  }
1720  if atend {
1721    // Scan the last ~80 lines for a Trailer-section BoundingBox.
1722    let tail: Vec<&str> = content.lines().rev().take(80).collect();
1723    for line in tail {
1724      if let Some(rest) = line.strip_prefix("%%BoundingBox:") {
1725        let rest_trim = rest.trim();
1726        if rest_trim.eq_ignore_ascii_case("(atend)") {
1727          continue;
1728        }
1729        if let Some(b) = parse_bbox_quadruple(rest) {
1730          return Some(b);
1731        }
1732      }
1733    }
1734  }
1735  hi_res
1736}
1737
1738fn parse_bbox_quadruple(s: &str) -> Option<(f64, f64, f64, f64)> {
1739  let mut vals = s.split_whitespace().filter_map(|s| s.parse::<f64>().ok());
1740  let (Some(x0), Some(y0), Some(x1), Some(y1)) =
1741    (vals.next(), vals.next(), vals.next(), vals.next())
1742  else {
1743    return None;
1744  };
1745  let w = (x1 - x0).abs();
1746  let h = (y1 - y0).abs();
1747  Some((x0, y0, w, h))
1748}

1750/// Legacy width/height-only accessor for callers that don't need the
1751/// origin offset. Use `read_postscript_bounding_box_full` when you
1752/// need to translate the content to PS origin (0, 0).
1753fn read_postscript_bounding_box(source: &str) -> Option<(f64, f64)> {
1754  read_postscript_bounding_box_full(source).map(|(_, _, w, h)| (w, h))
1755}

1757fn read_pdf_page_box(source: &str) -> Option<(f64, f64)> {
1758  let bytes = std::fs::read(source).ok()?;
1759  // Fast-fail: most modern PDFs (matplotlib, pgfplots, …) compress
1760  // their page dictionary inside an object stream, so `/MediaBox` and
1761  // `/CropBox` never appear as raw bytes. Skip the UTF-8 conversion
1762  // (which iterates Utf8Chunks across the entire file) when no
1763  // candidate token is present. Measured 2026-05-12 on 1910.01256:
1764  // ~10 ms saved across the 5-PDF graphics phase.
1765  let has_crop = memchr_find(&bytes, b"/CropBox").is_some();
1766  let has_media = memchr_find(&bytes, b"/MediaBox").is_some();
1767  if !has_crop && !has_media {
1768    return None;
1769  }
1770  let content = String::from_utf8_lossy(&bytes);
1771  parse_pdf_page_box(&content, &PDF_CROP_BOX_RE)
1772    .or_else(|| parse_pdf_page_box(&content, &PDF_MEDIA_BOX_RE))
1773}

1775/// Byte-level substring search (no UTF-8 conversion). Std-only —
1776/// avoids pulling in `memchr` for one call site.
1777fn memchr_find(haystack: &[u8], needle: &[u8]) -> Option<usize> {
1778  if needle.is_empty() || needle.len() > haystack.len() {
1779    return None;
1780  }
1781  let first = needle[0];
1782  let mut i = 0;
1783  while i + needle.len() <= haystack.len() {
1784    if haystack[i] == first && &haystack[i..i + needle.len()] == needle {
1785      return Some(i);
1786    }
1787    i += 1;
1788  }
1789  None
1790}

1792fn parse_pdf_page_box(content: &str, re: &regex::Regex) -> Option<(f64, f64)> {
1793  let captures = re.captures(content)?;
1794  let x0 = captures.get(1)?.as_str().parse::<f64>().ok()?;
1795  let y0 = captures.get(2)?.as_str().parse::<f64>().ok()?;
1796  let x1 = captures.get(3)?.as_str().parse::<f64>().ok()?;
1797  let y1 = captures.get(4)?.as_str().parse::<f64>().ok()?;
1798  Some(((x1 - x0).abs(), (y1 - y0).abs()))
1799}

1801impl Processor for Graphics {
1802  fn get_name(&self) -> &str { &self.name }
1803
1804  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
1805    doc.findnodes("//ltx:graphics[not(@imagesrc)]")
1806  }
1807
1808  fn process(&mut self, doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
1809    let mut search_paths = self.find_graphics_paths(&doc);
1810    search_paths.extend(doc.get_search_paths().iter().cloned());
1811    // Also add source directory
1812    let source_dir = doc.get_source_directory().to_string();
1813    if !source_dir.is_empty() && !search_paths.contains(&source_dir) {
1814      search_paths.push(source_dir.clone());
1815    }
1816    // Add current directory as fallback
1817    if !search_paths.contains(&".".to_string()) {
1818      search_paths.push(".".to_string());
1819    }
1820
1821    let dest_dir = doc.get_destination_directory().unwrap_or(".").to_string();
1822    // Read DPI/magnify/zoomout from processing instructions (set by latexml.sty)
1823    let dpi = self
1824      .get_parameter(&doc, "DPI")
1825      .map(|v| v as u32)
1826      .or(self.dpi)
1827      .unwrap_or(100);
1828    let magnify = self.get_parameter(&doc, "magnify").unwrap_or(self.magnify);
1829    let _zoomout = self.get_parameter(&doc, "zoomout").unwrap_or(self.zoomout);
1830    // Perl: effective DPI = DPI * magnify / zoomout (used for scale-to transforms)
1831    let effective_dpi = ((dpi as f64) * magnify / _zoomout) as u32;
1832    let n_to_process = nodes.len();
1833    latexml_core::telemetry::set_graphics_assets(n_to_process as u32);
1834
1835    // Counter for generating unique resource names (like Perl's generateResourcePathname)
1836    let mut resource_counter: u32 = 0;
1837
1838    // Two-phase plan so the slow per-image `convert` subprocess and
1839    // `read_image_dimensions` calls can run in parallel without touching
1840    // the libxml DOM off-thread.
1841    //
1842    // Phase 1 (serial): read each node's attributes, resolve source path,
1843    // decide the conversion kind, and allocate resource-name counters.
1844    //   - `Plan::NotFound` — apply fallback on the main thread later
1845    //   - `Plan::Copy { .. }` — apply on the main thread later (cheap)
1846    //   - `Plan::Convert { .. }` — independent; run in parallel.
1847    // Phase 2 (parallel): run convert_image + read_image_dimensions for
1848    // `Plan::Convert` entries. Produces `JobResult`s keyed by node index.
1849    // Phase 3 (serial): apply DOM mutations on the main thread in original
1850    // node order so attribute writes happen on the libxml-owning thread.
1851    enum Plan {
1852      NotFound {
1853        idx:     usize,
1854        graphic: String,
1855      },
1856      Copy {
1857        idx:     usize,
1858        source:  String,
1859        options: String,
1860      },
1861      Convert {
1862        idx:     usize,
1863        options: String,
1864        job_id:  usize,
1865      },
1866    }
1867    struct ConvertJob {
1868      job_id:       usize,
1869      source:       String,
1870      page:         Option<u32>,
1871      rel_dest:     String,
1872      abs_dest_str: String,
1873      /// `Some((rel_svg, abs_svg_str))` when the worker should
1874      /// first attempt the inkscape-SVG path and only fall back
1875      /// to `convert` on failure. `None` means the classic
1876      /// raster-only path.
1877      svg_paths:    Option<(String, String)>,
1878    }
1879    struct ConvertOutcome {
1880      job_id:   usize,
1881      /// Path to write into `imagesrc`; `None` if both convert and copy-fallback failed.
1882      imagesrc: Option<String>,
1883      /// Raw (pre-transform) dimensions read from whichever file we ended up with.
1884      raw_dims: Option<(u32, u32)>,
1885    }
1886
1887    let mut plans: Vec<Plan> = Vec::with_capacity(n_to_process);
1888    let mut convert_jobs: Vec<ConvertJob> = Vec::new();
1889    // Dedup key uses content-hash when readable, else falls back to
1890    // (source-path, page, options). Two byte-identical files with the
1891    // same options share one conversion + one output bundle entry.
1892    // The first-seen source's stem names the dest. Both <img> tags
1893    // end up pointing to that same rel_dest.
1894    #[derive(Hash, Eq, PartialEq)]
1895    enum JobKey {
1896      Hashed(u64, Option<u32>, String),
1897      Pathy(String, Option<u32>, String),
1898    }
1899    let mut convert_job_ids: HashMap<JobKey, usize> = HashMap::default();
1900    // Plan::Copy uses the same (hash, options) dedup so byte-identical
1901    // raster sources point at one output. `options` is part of the key
1902    // because angle= mutates the dest in-place — different rotations of
1903    // the same source need different dest files.
1904    #[derive(Hash, Eq, PartialEq)]
1905    enum CopyKey {
1906      Hashed(u64, String),
1907      Pathy(String, String),
1908    }
1909    let mut copy_dedup: HashMap<CopyKey, String> = HashMap::default();
1910    let mut convert_source_counts: HashMap<String, u32> = HashMap::default();
1911    for (idx, node) in nodes.iter().enumerate() {
1912      let options = node.get_attribute("options").unwrap_or_default();
1913      let page = Self::parse_page_option(&options);
1914      let Some(source) = self.find_graphic_file(&doc, node, &search_paths) else {
1915        let graphic = node
1916          .get_attribute("graphic")
1917          .unwrap_or_else(|| "none".to_string());
1918        plans.push(Plan::NotFound { idx, graphic });
1919        continue;
1920      };
1921      let src_ext = Path::new(&source)
1922        .extension()
1923        .and_then(|e| e.to_str())
1924        .unwrap_or("")
1925        .to_lowercase();
1926      let props = self.type_properties.get(&src_ext).cloned();
1927      let dest_type = props
1928        .as_ref()
1929        .and_then(|p| p.destination_type.as_ref())
1930        .cloned()
1931        .unwrap_or_else(|| src_ext.clone());
1932      let needs_conversion = dest_type != src_ext;
1933      let has_page = page.is_some();
1934      if needs_conversion || has_page {
1935        let content_hash = Self::hash_file_content(&source);
1936        let job_key = match content_hash {
1937          Some(h) => JobKey::Hashed(h, page, options.clone()),
1938          None => JobKey::Pathy(source.clone(), page, options.clone()),
1939        };
1940        let job_id = if let Some(&job_id) = convert_job_ids.get(&job_key) {
1941          job_id
1942        } else {
1943          let prior_source_jobs = convert_source_counts.get(&source).copied().unwrap_or(0);
1944          let dest_name = if has_page || prior_source_jobs > 0 {
1945            resource_counter += 1;
1946            format!("x{}", resource_counter)
1947          } else {
1948            Path::new(&source)
1949              .file_stem()
1950              .and_then(|s| s.to_str())
1951              .unwrap_or("image")
1952              .to_string()
1953          };
1954          convert_source_counts.insert(source.clone(), prior_source_jobs + 1);
1955          // Vector-SVG path: opt-in for small PDFs only. We prepare an
1956          // alternate `.svg` destination path alongside the normal raster
1957          // destination so the worker can try inkscape first, then fall
1958          // back. The file-size heuristic gates this — see
1959          // `should_try_svg_path`.
1960          let try_svg = Self::should_try_svg_path(&source, self.svg_threshold_kb);
1961          let rel_dest = format!("{}.{}", dest_name, dest_type);
1962          let abs_dest = PathBuf::from(&dest_dir).join(&rel_dest);
1963          if let Some(parent) = abs_dest.parent() {
1964            std::fs::create_dir_all(parent).ok();
1965          }
1966          let abs_dest_str = abs_dest.to_string_lossy().to_string();
1967          let svg_paths = if try_svg {
1968            let rel_svg = format!("{}.svg", dest_name);
1969            let abs_svg = PathBuf::from(&dest_dir).join(&rel_svg);
1970            let abs_svg_str = abs_svg.to_string_lossy().to_string();
1971            Some((rel_svg, abs_svg_str))
1972          } else {
1973            None
1974          };
1975          let job_id = convert_jobs.len();
1976          convert_jobs.push(ConvertJob {
1977            job_id,
1978            source: source.clone(),
1979            page,
1980            rel_dest,
1981            abs_dest_str,
1982            svg_paths,
1983          });
1984          convert_job_ids.insert(job_key, job_id);
1985          job_id
1986        };
1987        plans.push(Plan::Convert { idx, options, job_id });
1988      } else {
1989        plans.push(Plan::Copy { idx, source, options });
1990      }
1991    }
1992
1993    // Phase 2: parallel conversions. Bounded worker count to avoid
1994    // oversubscribing when many images are in flight. `convert` itself
1995    // is single-threaded per invocation, so the ceiling is useful CPU
1996    // parallelism — capped at a reasonable limit to avoid fork/memory
1997    // storms with many-image papers.
1998    let convert_count = convert_jobs.len();
1999    // Worker cap controls fork-fan-out of mutool / pdftocairo / convert.
2000    // Each spawn pulls libgs + libpoppler + libpng into a fresh
2001    // address space (~30 ms ambient), so on graphics-heavy papers
2002    // (e.g. LHCb 2402.01336 with 17 unique PDFs) sub-batches at
2003    // cap = 8 added wasted batch boundaries on a 28-CPU host. 22 is
2004    // a measured sweet spot on 28-core machines under the canvas
2005    // sweep workload: high enough to one-shot the typical tail-
2006    // paper graphics fan, low enough to leave headroom for the
2007    // outer cortex_worker pool (12-16 workers) without the kernel
2008    // scheduler thrashing — 12 × 32 ≈ 384 inflight subprocs trips
2009    // the internal 60 s watchdog and produces the "sweep flake"
2010    // pattern in stages 2/4/5. 12 × 22 ≈ 264 stays under the
2011    // measured starvation threshold. The 1910.01256 mini-bench
2012    // (5 PDFs) is unaffected because it is already <= cap.
2013    let worker_cap = std::thread::available_parallelism()
2014      .map(|n| n.get())
2015      .unwrap_or(4)
2016      .clamp(1, 22);
2017    let n_workers = convert_count.min(worker_cap).max(1);
2018    let source_dir_ref = source_dir.as_str();
2019    let dest_dir_ref = dest_dir.as_str();
2020    let mut outcomes: Vec<ConvertOutcome> = Vec::with_capacity(convert_count);
2021    if convert_count > 0 {
2022      use std::sync::atomic::{AtomicU32, AtomicUsize, Ordering};
2023      let next = AtomicUsize::new(0);
2024      // Subprocess tally: telemetry's thread_local! STATE is per-thread,
2025      // and worker threads exit before `phase_us[graphics]` aggregation,
2026      // so worker increments would be lost. Accumulate in a shared
2027      // AtomicU32 here and merge into telemetry once the scope joins.
2028      // One increment per `Self::convert_image_svg` / `Self::convert_image`
2029      // call (the EPS-via-PDF internal pair counts as one).
2030      let subproc_count = AtomicU32::new(0);
2031      let subproc_ref = &subproc_count;
2032      // Unique conversion jobs only. Repeated nodes with the same
2033      // source/page/options share one subprocess result, while distinct
2034      // options keep separate outputs.
2035      let jobs: Vec<&ConvertJob> = convert_jobs.iter().collect();
2036      // Each worker accumulates into a thread-local Vec returned from
2037      // its closure; the main thread merges them after scope join. No
2038      // shared mutable state during the parallel phase — replaces the
2039      // previous `Mutex<Vec<…>>` per project policy (thread_local-only
2040      // for in-memory state, no `Mutex`).
2041      // R35.C: spawn workers with a small (2 MB) stack via
2042      // `spawn_scoped`, which returns Result; if a spawn fails with
2043      // EAGAIN/WouldBlock (canvas + 6 GB ulimit can run out of address
2044      // space on graphics-heavy papers — witness hep-ph0012156, 12778
2045      // formulas, R35.C), drop the failure and let the surviving
2046      // workers pick up the remaining jobs via the shared `next`
2047      // counter. If every spawn fails, run all jobs on the current
2048      // thread instead of crashing.
2049      let worker_outcomes: Vec<Vec<ConvertOutcome>> = std::thread::scope(|s| {
2050        let handles: Vec<_> = (0..n_workers)
2051          .filter_map(|_| {
2052            std::thread::Builder::new()
2053              .stack_size(2 * 1024 * 1024)
2054              .spawn_scoped(s, || {
2055              let mut local = Vec::<ConvertOutcome>::new();
2056              loop {
2057              let i = next.fetch_add(1, Ordering::Relaxed);
2058              if i >= jobs.len() {
2059                break;
2060              }
2061              let ConvertJob {
2062                job_id,
2063                source,
2064                page,
2065                rel_dest,
2066                abs_dest_str,
2067                svg_paths,
2068              } = jobs[i];
2069              // Try vector-SVG path first if requested for this source.
2070              // The cache layer (graphics_cache) hardlinks/copies a
2071              // matching cached output before any subprocess fires and
2072              // round-trips the dimensions through a .dims sidecar so
2073              // hits skip the `read_*_dimensions` re-measure too.
2074              // Misses fall through to the real conversion + measure
2075              // and write back on success. Disable via
2076              // LATEXML_GRAPHICS_CACHE_OFF=1.
2077              let svg_outcome = if let Some((rel_svg, abs_svg)) = svg_paths {
2078                let svg_key = crate::graphics_cache::RenderKey {
2079                  page:    *page,
2080                  density: 0,
2081                  ext:     "svg",
2082                };
2083                let svg_res = crate::graphics_cache::with_cache_dims(
2084                  source,
2085                  abs_svg,
2086                  svg_key,
2087                  || {
2088                    subproc_ref.fetch_add(1, Ordering::Relaxed);
2089                    Self::convert_image_svg(source, abs_svg, *page)
2090                  },
2091                  || Self::read_svg_dimensions(abs_svg)
2092                    .map(|(w, h)| crate::graphics_cache::CachedDims { width: w, height: h }),
2093                );
2094                match svg_res {
2095                  crate::graphics_cache::ConvertResult::Ok { dims } => Some(ConvertOutcome {
2096                    job_id: *job_id,
2097                    imagesrc: Some(rel_svg.clone()),
2098                    raw_dims: dims.map(|d| (d.width, d.height)),
2099                  }),
2100                  crate::graphics_cache::ConvertResult::Failed => {
2101                    Warn!(
2102                      "shell", "inkscape",
2103                      "Graphics: inkscape SVG path failed for {}, falling back to convert",
2104                      source
2105                    );
2106                    None
2107                  },
2108                }
2109              } else {
2110                None
2111              };
2112              let raster_res = if svg_outcome.is_none() {
2113                let raster_key = crate::graphics_cache::RenderKey {
2114                  page:    *page,
2115                  density: Self::raster_density_for_source(source),
2116                  ext:     ext_from_path(abs_dest_str),
2117                };

2119                  source,
2120                  abs_dest_str,
2121                  raster_key,
2122                  || {
2123                    subproc_ref.fetch_add(1, Ordering::Relaxed);
2124                    Self::convert_image(source, abs_dest_str, dpi, *page)
2125                  },
2126                  || Self::read_image_dimensions(abs_dest_str)
2127                    .map(|(w, h)| crate::graphics_cache::CachedDims { width: w, height: h }),
2128                )
2129              } else {
2130                crate::graphics_cache::ConvertResult::Failed
2131              };

2133                o

2137                  imagesrc: Some(rel_dest.clone()),
2138                  raw_dims: raster_res.dims().map(|d| (d.width, d.height)),
2139                }
2140              } else {
2141                // Final-failure: every conversion path exhausted. Promoted
2142                // from warn → Error 2026-05-08 because we want all
2143                // images to convert successfully, and a silent warning
2144                // hides regressions in the rasterizer chain.
2145                // Error class/object mirror Perl Graphics.pm:274
2146                // `Error('imageprocessing', $source, …)` so the
2147                // harness aggregates with engine/package emissions.
2148                Error!(
2149                  "imageprocessing", source,
2150                  "Graphics: Failed to convert {} to {}", source, abs_dest_str
2151                );

2153                  ConvertOutcome {
2154                    job_id:   *job_id,
2155                    imagesrc: Some(rel),
2156                    raw_dims: Self::read_image_dimensions(source),
2157                  }
2158                } else {
2159                  ConvertOutcome {
2160                    job_id:   *job_id,
2161                    imagesrc: None,
2162                    raw_dims: None,
2163                  }
2164                }
2165              };
2166              local.push(outcome);
2167            }
2168              local
2169            })
2170            .ok()
2171          })
2172          .collect();
2173        // Note: if EVERY spawn failed (extreme memory pressure), no
2174        // jobs run and graphics will be missing from the output. That
2175        // is much less destructive than panicking the whole worker
2176        // and losing the entire conversion. Surviving workers always
2177        // race for the same `next` counter, so a single survivor is
2178        // enough to complete all jobs.
2179        handles.into_iter().map(|h| h.join().unwrap()).collect()
2180      });
2181      for v in worker_outcomes {
2182        outcomes.extend(v);
2183      }
2184      outcomes.sort_by_key(|o| o.job_id);
2185      latexml_core::telemetry::add_graphics_subprocess(subproc_count.load(Ordering::Relaxed));
2186    }
2187    let outcomes_by_job: HashMap<usize, ConvertOutcome> =
2188      outcomes.into_iter().map(|o| (o.job_id, o)).collect();

2190    // Phase 3: serial DOM mutations. Preserves original node order.
2191    let apply_transforms =
2192      |options: &str, raw_dims: Option<(u32, u32)>| -> (Option<u32>, Option<u32>) {
2193        match raw_dims {
2194          Some((w, h)) if !options.is_empty() => {
2195            let (tw, th) = Self::apply_graphicx_transforms(w, h, options, effective_dpi);
2196            (Some(tw), Some(th))
2197          },
2198          Some((w, h)) => (Some(w), Some(h)),
2199          None => (None, None),
2200        }
2201      };
2202    for plan in &plans {
2203      match plan {
2204        Plan::NotFound { idx, graphic } => {
2205          // Perl `Post/Graphics.pm:216` uses Warn level. An earlier
2206          // Rust-only promotion to Error (2026-05-08) was motivated by
2207          // "we want all images to convert", but the real driver of the
2208          // not-found cases on the canvas is the missing
2209          // `doc.get_search_paths()` half of `find_graphics_paths`
2210          // (just fixed above), not actual missing files. With sources
2211          // findable, this branch hits only when the .tex literally
2212          // references a non-existent file — exactly the case Perl
2213          // emits at Warn level. Restore Perl-faithful Warn.
2214          Warn!(
2215            "expected", "source",
2216            "No graphic source found; skipping (source was '{}')", graphic
2217          );
2218          let mut node_mut = nodes[*idx].clone();
2219          node_mut.set_attribute("imagesrc", graphic).ok();
2220        },
2221        Plan::Copy { idx, source, options } => {
2222          let mut node_mut = nodes[*idx].clone();
2223          // Content-hash dedup: if a byte-identical source with the
2224          // same options was already copied (and rotated), point this
2225          // node at the same rel. Avoids both duplicate I/O and a
2226          // duplicate output file in the bundle. Fall back to source-
2227          // path keying when the file can't be hashed.
2228          let hash_opt = Self::hash_file_content(source);
2229          let key = match hash_opt {
2230            Some(h) => CopyKey::Hashed(h, options.clone()),
2231            None => CopyKey::Pathy(source.clone(), options.clone()),
2232          };
2233          let rel = if let Some(existing) = copy_dedup.get(&key) {
2234            existing.clone()
2235          } else {
2236            let rel_opt = Self::copy_to_destination(source, &source_dir, &dest_dir);
2237            let rel = rel_opt.unwrap_or_else(|| {
2238              Path::new(source)
2239                .strip_prefix(&source_dir)
2240                .unwrap_or_else(|_| Path::new(source))
2241                .to_string_lossy()
2242                .to_string()
2243            });
2244            // Plan::Copy fires only for raster sources (web-native PNG
2245            // / JPG / GIF) where `dest_type == src_ext`. graphicx
2246            // `angle=` rotation IS meaningful here — the source carries
2247            // no PDF /Rotate metadata to pre-rotate from. Apply via
2248            // convert.
2249            // Perl semantics (Util/Image.pm:image_graphicx_complex
2250            // L390-394): IM `Rotate` with `degrees => -$a1` — graphicx
2251            // angle is CCW; convert -rotate is CW; negate to match.
2252            let angle = Self::parse_angle_option(options).unwrap_or(0.0);
2253            if angle.abs() > 0.5 {
2254              let dest_full = PathBuf::from(&dest_dir).join(&rel);
2255              Self::rotate_image_inplace(&dest_full.to_string_lossy(), -angle);
2256            }
2257            copy_dedup.insert(key, rel.clone());
2258            rel
2259          };
2260          let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
2261          Self::set_graphic_src(&mut node_mut, &rel, w, h);
2262        },
2263        Plan::Convert { idx, options, job_id } => {
2264          if let Some(out) = outcomes_by_job.get(job_id) {
2265            let mut node_mut = nodes[*idx].clone();
2266            if let Some(imagesrc) = &out.imagesrc {
2267              // Plan::Convert handles non-raster sources (EPS, PS, PDF,
2268              // AI). With ps2pdf's /Rotate-injection path disabled (see
2269              // should_try_eps_pdf_path), all of these now go through
2270              // ImageMagick `convert` (or pdftocairo for plain .pdf),
2271              // neither of which pre-applies graphicx rotation. So
2272              // apply the graphicx angle uniformly here.
2273              //
2274              // Perl semantics (Util/Image.pm:image_graphicx_complex
2275              // L390-394): `image_internalop('Rotate', degrees => -$a1)`.
2276              // ImageMagick Rotate is CCW (matches graphicx); from CLI
2277              // it's CW → pass -angle to match Perl's intent.
2278              let angle = Self::parse_angle_option(options).unwrap_or(0.0);
2279              if angle.abs() > 0.5 {
2280                let dest_full = PathBuf::from(&dest_dir).join(imagesrc);
2281                Self::rotate_image_inplace(&dest_full.to_string_lossy(), -angle);
2282              }
2283              let (w, h) = apply_transforms(options, out.raw_dims);
2284              Self::set_graphic_src(&mut node_mut, imagesrc, w, h);
2285            }
2286          }
2287        },
2288      }
2289    }
2290
2291    Info!(
2292      "graphics", "process",
2293      "Graphics {} {} to process",
2294      doc.get_destination().unwrap_or("?"),
2295      n_to_process
2296    );
2297    Ok(vec![doc])
2298  }
