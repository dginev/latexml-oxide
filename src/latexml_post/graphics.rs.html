185  /// Find the graphics source file for a node.
186  ///
187  /// Port of `Graphics::findGraphicFile`.
188  fn find_graphic_file(
189    &self,
190    _doc: &PostDocument,
191    node: &Node,
192    search_paths: &[String],
193  ) -> Option<String> {
194    let source = node.get_attribute("graphic")?;
195
196    // Check candidates attribute first (comma-separated list of found files)
197    // Perl: findGraphicFile checks each candidate path, resolving relative to search paths.
198    if let Some(candidates) = node.get_attribute("candidates") {
199      // Pick the best candidate by desirability
200      let mut best: Option<(String, i32)> = None;
201      for path in candidates.split(',') {
202        let path = path.trim();
203        if path.is_empty() {
204          continue;
205        }
206        // Try the path directly, then in each search directory
207        let resolved = if Path::new(path).exists() {
208          Some(path.to_string())
209        } else {
210          search_paths.iter().find_map(|sp| {
211            let candidate = format!("{}/{}", sp, path);
212            if Path::new(&candidate).exists() {
213              Some(candidate)
214            } else {
215              None
216            }
217          })
218        };
219        if let Some(resolved_path) = resolved {
220          let ext = Path::new(&resolved_path)
221            .extension()
222            .and_then(|e| e.to_str())
223            .unwrap_or("")
224            .to_lowercase();
225          let props = self.type_properties.get(&ext);
226          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
227          let is_same_type = props
228            .and_then(|p| p.destination_type.as_ref())
229            .map(|dt| dt == &ext)
230            .unwrap_or(false);
231          let desirability = if is_same_type { 10 } else { d };
232          if best.as_ref().is_none_or(|(_, bd)| desirability > *bd) {
233            best = Some((resolved_path, desirability));
234          }
235        }
236      }
237      if let Some((path, _)) = best {
238        return Some(path);
239      }
240    }
241
242    // Search for the file in search paths
243    let path = Path::new(&source);
244    let name = path.file_stem().and_then(|s| s.to_str()).unwrap_or(&source);
245    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or("");
246
247    let file_base = if dir.is_empty() {
248      name.to_string()
249    } else {
250      format!("{}/{}", dir, name)
251    };
252
253    // Try each type in search paths
254    let mut best_desirability: i32 = -1;
255    let mut best_path: Option<String> = None;
256
257    let types: Vec<String> = self
258      .graphics_types
259      .iter()
260      .flat_map(|t| vec![t.clone(), t.to_uppercase()])
261      .collect();
262
263    for search_path in search_paths {
264      // Try without extension first (source might already have one)
265      let candidate = if search_path.is_empty() {
266        source.clone()
267      } else {
268        format!("{}/{}", search_path, source)
269      };
270      if Path::new(&candidate).exists() {
271        let ext = Path::new(&candidate)
272          .extension()
273          .and_then(|e| e.to_str())
274          .unwrap_or("")
275          .to_lowercase();
276        let props = self.type_properties.get(&ext);
277        let d = props.map(|p| p.desirability as i32).unwrap_or(5);
278        if d > best_desirability {
279          best_desirability = d;
280          best_path = Some(candidate);
281        }
282      }
283      // Try each extension
284      for ext in &types {
285        let candidate = format!("{}/{}.{}", search_path, file_base, ext);
286        if Path::new(&candidate).exists() {
287          let props = self.type_properties.get(&ext.to_lowercase());
288          let d = props.map(|p| p.desirability as i32).unwrap_or(0);
289          let is_same_type = props
290            .and_then(|p| p.destination_type.as_ref())
291            .map(|dt| dt == &ext.to_lowercase())
292            .unwrap_or(false);
293          let desirability = if is_same_type { 10 } else { d };
294          if desirability > best_desirability {
295            best_desirability = desirability;
296            best_path = Some(candidate);
297          }
298        }
299      }
300    }
301
302    best_path
303  }

305  /// Set the image source attributes on a graphics node.
306  ///
307  /// Port of `Graphics::setGraphicSrc`.
308  fn set_graphic_src(node: &mut Node, src: &str, width: Option<u32>, height: Option<u32>) {
309    node.set_attribute("imagesrc", src).ok();
310    // HTML width/height are in pixels (unitless)
311    if let Some(w) = width {
312      node.set_attribute("imagewidth", &w.to_string()).ok();
313    }
314    if let Some(h) = height {
315      node.set_attribute("imageheight", &h.to_string()).ok();
316    }
317    // Set aspect ratio class
318    if let (Some(w), Some(h)) = (width, height) {
319      let class = if w as f64 > 1.24 * h as f64 {
320        "ltx_img_landscape"
321      } else if h as f64 > 1.24 * w as f64 {
322        "ltx_img_portrait"
323      } else {
324        "ltx_img_square"
325      };
326      let existing = node.get_attribute("class").unwrap_or_default();
327      let new_class = if existing.is_empty() {
328        class.to_string()
329      } else {
330        format!("{} {}", existing, class)
331      };
332      node.set_attribute("class", &new_class).ok();
333    }
334  }

349  /// Read a named parameter from processing instructions.
350  /// Port of Perl's `Graphics::getParameter`.
351  /// Checks both direct PI (`<?latexml DPI="100"?>`) and
352  /// latexml.sty package options (`<?latexml package="latexml" options="magnify=1.2"?>`).
353  fn get_parameter(&self, doc: &PostDocument, param: &str) -> Option<f64> {
354    let direct_re = regex::Regex::new(&format!(
355      r#"^\s*{}\s*=\s*[\"']?([\d.]+)[\"']?\s*$"#,
356      regex::escape(param)
357    ))
358    .ok()?;
359    let options_re =
360      regex::Regex::new(r#"package\s*=\s*[\"']latexml[\"'].*options\s*=\s*[\"'](.*?)[\"']"#)
361        .ok()?;
362    let param_in_options_re =
363      regex::Regex::new(&format!(r#"\b{}\s*=\s*([\d.]+)"#, regex::escape(param))).ok()?;
364
365    for pi in doc.findnodes(".//processing-instruction('latexml')") {
366      let text = pi.get_content();
367      if let Some(cap) = direct_re.captures(&text) {
368        return cap[1].parse().ok();
369      }
370      if let Some(cap) = options_re.captures(&text) {
371        if let Some(inner) = param_in_options_re.captures(&cap[1]) {
372          return inner[1].parse().ok();
373        }
374      }
375    }
376    None
377  }

388  /// Parse graphicx options and apply transforms to image dimensions.
389  /// Port of Perl's `getTransform` + `image_graphicx_trivial`.
390  ///
391  /// Handles: scale=N, width=Npt, height=Npt, keepaspectratio
392  fn apply_graphicx_transforms(raw_w: u32, raw_h: u32, options: &str, dpi: u32) -> (u32, u32) {
393    let dppt = dpi as f64 / 72.27; // dots per point
394    let mut w = raw_w as f64;
395    let mut h = raw_h as f64;
396
397    // Parse options as key=value pairs
398    let mut scale: Option<f64> = None;
399    let mut target_width: Option<f64> = None;
400    let mut target_height: Option<f64> = None;
401    let mut keep_aspect = false;
402
403    for opt in options.split(',') {
404      let opt = opt.trim();
405      if let Some((key, val)) = opt.split_once('=') {
406        let key = key.trim();
407        let val = val.trim();
408        match key {
409          "scale" => {
410            scale = val.parse::<f64>().ok();
411          },
412          "width" => {
413            // Parse dimension: "345.0pt" or "345pt" or bare number
414            let val = val.trim_end_matches("pt").trim_end_matches("px");
415            target_width = val.parse::<f64>().ok();
416          },
417          "height" => {
418            let val = val.trim_end_matches("pt").trim_end_matches("px");
419            target_height = val.parse::<f64>().ok();
420          },
421          "keepaspectratio" => {
422            keep_aspect = val == "true" || val == "1" || val.is_empty();
423          },
424          _ => {},
425        }
426      } else if opt == "keepaspectratio" {
427        keep_aspect = true;
428      }
429    }
430
431    // Apply transforms (matching Perl's image_graphicx_trivial)
432    if let Some(s) = scale {
433      w *= s;
434      h *= s;
435    } else if target_width.is_some() || target_height.is_some() {
436      let tw = target_width.unwrap_or(w / dppt);
437      let th = target_height.unwrap_or(h / dppt);
438      if keep_aspect {
439        // Preserve aspect ratio: use the more constraining dimension
440        let scale_w = tw / (w / dppt);
441        let scale_h = th / (h / dppt);
442        let s = scale_w.min(scale_h);
443        w = (w / dppt * s * dppt).ceil();
444        h = (h / dppt * s * dppt).ceil();
445      } else {
446        w = (tw * dppt).ceil();
447        h = (th * dppt).ceil();
448      }
449    }
450
451    (w.max(1.0) as u32, h.max(1.0) as u32)
452  }

454  /// Copy a source image to the destination directory, preserving relative paths.
455  /// Returns the destination path (relative to dest_dir).
456  fn copy_to_destination(source: &str, source_dir: &str, dest_dir: &str) -> Option<String> {
457    // Compute relative path of source from source_dir
458    let source_path = Path::new(source);
459    let source_base = Path::new(source_dir);
460    let rel_path = source_path.strip_prefix(source_base).unwrap_or(source_path);
461
462    // Build absolute destination path
463    let abs_dest = PathBuf::from(dest_dir).join(rel_path);
464
465    // Create parent directories if needed
466    if let Some(parent) = abs_dest.parent() {
467      std::fs::create_dir_all(parent).ok()?;
468    }
469
470    // Copy the file (skip if same path)
471    let source_canonical = std::fs::canonicalize(source).ok();
472    let dest_canonical = std::fs::canonicalize(&abs_dest).ok();
473    if source_canonical != dest_canonical || dest_canonical.is_none() {
474      std::fs::copy(source, &abs_dest).ok()?;
475    }
476
477    // Return relative path for imagesrc attribute
478    Some(rel_path.to_string_lossy().to_string())
479  }

497  /// Try to convert a PDF to plain SVG via `inkscape`, preserving vector
498  /// content. Returns `true` on success. Tracks upstream
499  /// brucemiller/LaTeXML#902.
500  ///
501  /// Caller decides when to attempt this — typically only for PDF sources
502  /// below a file-size threshold, because inkscape on raster-embedded PDFs
503  /// produces massive output (>100 MB) and can take 40+ seconds
504  /// (measured: Fade.pdf 1.7 MB → 46 s / 102 MB SVG vs `convert` 1.4 s /
505  /// 61 KB PNG).
506  ///
507  /// `page` is 1-based (graphicx convention); converted to 0-based for
508  /// inkscape's `--pdf-page`.
509  ///
510  /// Guarded by a **hard timeout** (15 s default; see
511  /// `inkscape_timeout_secs`). Pathological small-PDF cases have been
512  /// observed — if inkscape is still running after the deadline we SIGKILL
513  /// it and return `false` so the caller falls back to ImageMagick. The
514  /// timeout is generous enough (15 s) for all well-behaved small vector
515  /// plots and strict enough to prevent the 46 s+ runaway behaviour seen
516  /// on Fade.pdf-class inputs.
517  fn convert_image_svg(source: &str, dest: &str, page: Option<u32>) -> bool {
518    // First try poppler's `pdftocairo -svg`. Empirically 20-40× faster
519    // than inkscape on benign vector PDFs (e.g. matplotlib/pgfplots
520    // exports: 0.02 s vs 0.5 s) and produces equivalent or smaller SVG.
521    // We accept its output only if the result fits under
522    // `MAX_SVG_OUTPUT_BYTES`; pathological vector-heavy PDFs (e.g.
523    // R-Graphics `W.pdf`) can otherwise emit >100 MB SVG. On rejection
524    // or failure we fall through to inkscape, then to PNG raster.
525    if Self::convert_image_svg_pdftocairo(source, dest, page) {
526      return true;
527    }
528    let mut cmd = std::process::Command::new("inkscape");
529    cmd
530      .arg("--export-type=svg")
531      .arg("--export-plain-svg")
532      .arg(format!("--export-filename={}", dest));
533    if let Some(p) = page {
534      cmd.arg(format!("--pdf-page={}", p.saturating_sub(1)));
535    }
536    cmd.arg(source);
537    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
538    match Self::run_with_timeout(cmd, timeout) {
539      Some(status) => {
540        if !(status.success() && Path::new(dest).exists()) {
541          return false;
542        }
543        // Reject pathological inkscape output that explodes to >100 MB
544        // — keep the dest hole open so the worker falls back to raster.
545        if Self::svg_output_too_large(dest) {
546          let _ = std::fs::remove_file(dest);
547          return false;
548        }
549        true
550      },
551      None => {
552        log::warn!(
553          "Graphics: inkscape SVG conversion for {} exceeded {} s — killed",
554          source,
555          timeout.as_secs()
556        );
557        // Best-effort cleanup of a partial output.
558        let _ = std::fs::remove_file(dest);
559        false
560      },
561    }
562  }

564  /// Maximum acceptable SVG output size from a vector conversion. Above
565  /// this we discard the SVG and force the raster fallback — it's nearly
566  /// always cheaper to ship a 30 KB PNG than a 100 MB SVG even when both
567  /// are technically valid. Tuned from observed cases: well-behaved
568  /// matplotlib plots are ~500 KB - 2 MB; W.pdf-class explodes to
569  /// 70-115 MB across all known PDF→SVG tools.
570  const MAX_SVG_OUTPUT_BYTES: u64 = 8 * 1024 * 1024; // 8 MB
571
572  fn svg_output_too_large(path: &str) -> bool {
573    std::fs::metadata(path)
574      .map(|md| md.len() > Self::MAX_SVG_OUTPUT_BYTES)
575      .unwrap_or(false)
576  }

578  /// `pdftocairo -svg` rasterizes the page's vector content to SVG via
579  /// poppler/cairo. Much faster than inkscape on the kind of vector PDFs
580  /// matplotlib/pgfplots produce. Returns true ONLY if the output is
581  /// reasonably-sized; otherwise we discard and let the caller try
582  /// inkscape (which sometimes simplifies further).
583  fn convert_image_svg_pdftocairo(source: &str, dest: &str, page: Option<u32>) -> bool {
584    let mut cmd = std::process::Command::new("pdftocairo");
585    cmd.arg("-svg");
586    if let Some(p) = page {
587      let p1 = p.max(1);
588      cmd
589        .arg("-f")
590        .arg(p1.to_string())
591        .arg("-l")
592        .arg(p1.to_string());
593    } else {
594      cmd.arg("-f").arg("1").arg("-l").arg("1");
595    }
596    cmd.arg(source).arg(dest);
597    let timeout = std::time::Duration::from_secs(Self::inkscape_timeout_secs());
598    match Self::run_with_timeout(cmd, timeout) {
599      Some(status) => {
600        if !(status.success() && Path::new(dest).exists()) {
601          let _ = std::fs::remove_file(dest);
602          return false;
603        }

624  /// Run `cmd` and enforce a wall-clock timeout. Returns `Some(status)` on
625  /// clean exit, `None` if the child was killed on timeout or spawn
626  /// failed. Polls every 50 ms — cheap compared to the subprocess cost.
627  fn run_with_timeout(
628    mut cmd: std::process::Command,
629    timeout: std::time::Duration,
630  ) -> Option<std::process::ExitStatus> {
631    // Redirect stdio so a slow inkscape doesn't block on a full pipe.
632    cmd
633      .stdout(std::process::Stdio::null())
634      .stderr(std::process::Stdio::null());
635    let mut child = cmd.spawn().ok()?;
636    let start = std::time::Instant::now();
637    loop {
638      match child.try_wait() {
639        Ok(Some(status)) => return Some(status),
640        Ok(None) => {
641          if start.elapsed() >= timeout {
642            let _ = child.kill();
643            let _ = child.wait();
644            return None;
645          }
646          std::thread::sleep(std::time::Duration::from_millis(50));
647        },
648        Err(_) => {
649          let _ = child.kill();
650          let _ = child.wait();
651          return None;
652        },
653      }
654    }
655  }

657  /// Parse SVG viewBox ("minX minY width height") and return (width, height).
658  /// Falls back to `width`/`height` root attributes if viewBox is missing.
659  /// Returns None on parse failure so callers can skip dimension attributes.
660  fn read_svg_dimensions(path: &str) -> Option<(u32, u32)> {
661    let content = std::fs::read_to_string(path).ok()?;
662    // Look at just the root <svg> opening tag (first ~2 KB is enough).
663    // We must skip the optional `<?xml … ?>` prolog and any `<!-- … -->`
664    // or `<!DOCTYPE …>` preamble — otherwise `find('>')` matches the
665    // prolog's `?>` instead of the root tag.
666    let head = &content[..content.len().min(2048)];
667    let svg_start = head.find("<svg")?;
668    let svg_rest = &head[svg_start..];
669    let svg_tag_end = svg_rest.find('>')?;
670    let root = &svg_rest[..=svg_tag_end];
671    let extract = |attr: &str| -> Option<String> {
672      let key = format!("{}=\"", attr);
673      let start = root.find(&key)? + key.len();
674      let rest = &root[start..];
675      let end = rest.find('"')?;
676      Some(rest[..end].to_string())
677    };
678    let parse_dim = |s: &str| -> Option<f64> {
679      let s = s.trim();
680      // Strip trailing unit if present (pt, px, mm, etc.)
681      let numeric_end = s
682        .find(|c: char| !c.is_ascii_digit() && c != '.' && c != '-')
683        .unwrap_or(s.len());
684      s[..numeric_end].parse::<f64>().ok()
685    };
686    if let Some(vb) = extract("viewBox") {
687      let parts: Vec<&str> = vb.split_whitespace().collect();
688      if parts.len() == 4 {
689        if let (Some(w), Some(h)) = (parse_dim(parts[2]), parse_dim(parts[3])) {
690          return Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32));
691        }
692      }
693    }
694    let w = extract("width").as_deref().and_then(parse_dim);
695    let h = extract("height").as_deref().and_then(parse_dim);
696    match (w, h) {
697      (Some(w), Some(h)) => Some((w.round().max(1.0) as u32, h.round().max(1.0) as u32)),
698      _ => None,
699    }
700  }

759  /// Rasterize a PDF directly via `pdftocairo --png`. Much faster than
760  /// `convert`/Ghostscript for vector-heavy PDFs. Returns true only when
761  /// the destination file was actually written.
762  fn convert_pdf_via_pdftocairo(source: &str, dest: &str, density: u32, page: Option<u32>) -> bool {
763    let dest_path = Path::new(dest);
764    let parent = dest_path.parent().unwrap_or_else(|| Path::new("."));
765    let stem = dest_path
766      .file_name()
767      .and_then(|s| s.to_str())
768      .unwrap_or("image");
769    let unique = std::time::SystemTime::now()
770      .duration_since(std::time::UNIX_EPOCH)
771      .map(|d| d.as_nanos())
772      .unwrap_or(0);
773    let tmp_prefix = parent.join(format!(".{}.{}.pdftocairo", stem, unique));
774    let tmp_png = PathBuf::from(format!("{}.png", tmp_prefix.to_string_lossy()));
775    let timeout = std::time::Duration::from_secs(20);
776
777    let cleanup = |tmp_png: &Path| {
778      let _ = std::fs::remove_file(tmp_png);
779    };
780
781    let mut pdftocairo = std::process::Command::new("pdftocairo");
782    pdftocairo
783      .arg("-singlefile")
784      .arg("-png")
785      .arg("-r")
786      .arg(density.to_string());
787    // graphicx page is 1-based; pdftocairo also uses 1-based.
788    if let Some(p) = page {
789      let p1 = p.max(1);
790      pdftocairo
791        .arg("-f")
792        .arg(p1.to_string())
793        .arg("-l")
794        .arg(p1.to_string());
795    } else {
796      // Default to first page (matches Perl/`convert` `[0]` behavior).
797      pdftocairo.arg("-f").arg("1").arg("-l").arg("1");
798    }
799    pdftocairo.arg(source).arg(&tmp_prefix);
800
801    let cairo_ok = Self::run_with_timeout(pdftocairo, timeout)
802      .map(|status| status.success())
803      .unwrap_or(false)
804      && tmp_png.exists();
805    if !cairo_ok {
806      cleanup(&tmp_png);
807      return false;
808    }
809
810    let _ = std::fs::remove_file(dest);
811    let installed = std::fs::rename(&tmp_png, dest)
812      .or_else(|_| std::fs::copy(&tmp_png, dest).map(|_| ()))
813      .is_ok()
814      && dest_path.exists();
815    cleanup(&tmp_png);
816    installed
817  }

881  /// Convert a graphics file using ImageMagick's `convert` command.
882  /// Perl: image_graphicx_complex via Image::Magick / convert CLI.
883  /// `page` is 1-based (graphicx convention); converted to 0-based for ImageMagick.
884  fn convert_image(source: &str, dest: &str, _dpi: u32, page: Option<u32>) -> bool {
885    // Build the source argument with optional page selector
886    // Perl: image_read reads "$source[$page]" where $page = ($page // 1) - 1
887    let source_arg = if let Some(p) = page {
888      format!("{}[{}]", source, p.saturating_sub(1))
889    } else {
890      // No page specified: use [0] for PDFs to avoid converting all pages
891      if source.to_lowercase().ends_with(".pdf") {
892        format!("{}[0]", source)
893      } else {
894        source.to_string()
895      }
896    };
897    // Shell out to convert (matching Perl's approach)
898    // -define pdf:use-cropbox=true matches Perl's Image::Magick option (line 466)
899    let density = Self::raster_density_for_source(source);
900    if Self::should_try_eps_pdf_path(source, page)
901      && Self::convert_eps_via_pdf(source, dest, density)
902    {
903      return true;
904    }
905    // For PDF sources, prefer poppler's `pdftocairo --png` over
906    // ImageMagick's `convert`-via-Ghostscript. Empirically 25× faster on
907    // vector-heavy PDFs (e.g. R-Graphics output), and avoids tens-of-seconds
908    // gs runaways. Falls through to the convert/gs path if pdftocairo is
909    // unavailable or fails for any reason.
910    if Self::should_try_pdf_cairo_path(source)
911      && dest.to_lowercase().ends_with(".png")
912      && Self::convert_pdf_via_pdftocairo(source, dest, density, page)
913    {
914      return true;
915    }
916    // Wall-clock timeout to bound `gs`-via-`convert` runaways on
917    // pathological PDFs (raster-heavy or with broken xref tables).
918    // Matches the inkscape path's defensive bound; without this, an
919    // arbitrary `convert` invocation could run for minutes and stall
920    // the entire post-processing phase. 60 s is enough for any
921    // reasonably-sized graphic; tune via `LATEXML_CONVERT_TIMEOUT_SECS`.
922    let mut cmd = std::process::Command::new("convert");
923    cmd
924      .arg("-define")
925      .arg("pdf:use-cropbox=true")
926      .arg("-density")
927      .arg(density.to_string())
928      .arg(&source_arg)
929      .arg(dest);
930    let timeout = std::time::Duration::from_secs(Self::convert_timeout_secs());
931    match Self::run_with_timeout(cmd, timeout) {
932      // Mirror the original `cmd.output()` semantics: report success based
933      // on exit status alone, not on whether `dest` was actually written.
934      // (The fake-convert test fixture exits 0 without producing a file.)
935      Some(status) => status.success(),
936      None => {
937        log::warn!(
938          "Graphics: convert/gs for {} exceeded {} s — killed",
939          source,
940          timeout.as_secs()
941        );
942        let _ = std::fs::remove_file(dest);
943        false
944      },
945    }
946  }

954fn read_postscript_bounding_box(source: &str) -> Option<(f64, f64)> {
955  let content = std::fs::read_to_string(source).ok()?;
956  for line in content.lines().take(80) {
957    let Some(rest) = line.strip_prefix("%%BoundingBox:") else {
958      continue;
959    };
960    let mut vals = rest
961      .split_whitespace()
962      .filter_map(|s| s.parse::<f64>().ok());
963    let (Some(x0), Some(y0), Some(x1), Some(y1)) =
964      (vals.next(), vals.next(), vals.next(), vals.next())
965    else {
966      return None;
967    };
968    let w = (x1 - x0).abs();
969    let h = (y1 - y0).abs();
970    return Some((w, h));
971  }
972  None
973}

977  let content = String::from_utf8_lossy(&bytes);

982fn parse_pdf_page_box(content: &str, re: &regex::Regex) -> Option<(f64, f64)> {
983  let captures = re.captures(content)?;
984  let x0 = captures.get(1)?.as_str().parse::<f64>().ok()?;
985  let y0 = captures.get(2)?.as_str().parse::<f64>().ok()?;
986  let x1 = captures.get(3)?.as_str().parse::<f64>().ok()?;
987  let y1 = captures.get(4)?.as_str().parse::<f64>().ok()?;
988  Some(((x1 - x0).abs(), (y1 - y0).abs()))
989}

998  fn process(&mut self, doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
999    let mut search_paths = self.find_graphics_paths(&doc);
1000    search_paths.extend(doc.get_search_paths().iter().cloned());
1001    // Also add source directory
1002    let source_dir = doc.get_source_directory().to_string();
1003    if !source_dir.is_empty() && !search_paths.contains(&source_dir) {
1004      search_paths.push(source_dir.clone());
1005    }
1006    // Add current directory as fallback
1007    if !search_paths.contains(&".".to_string()) {
1008      search_paths.push(".".to_string());
1009    }
1010
1011    let dest_dir = doc.get_destination_directory().unwrap_or(".").to_string();
1012    // Read DPI/magnify/zoomout from processing instructions (set by latexml.sty)
1013    let dpi = self
1014      .get_parameter(&doc, "DPI")
1015      .map(|v| v as u32)
1016      .or(self.dpi)
1017      .unwrap_or(100);
1018    let magnify = self.get_parameter(&doc, "magnify").unwrap_or(self.magnify);
1019    let _zoomout = self.get_parameter(&doc, "zoomout").unwrap_or(self.zoomout);
1020    // Perl: effective DPI = DPI * magnify / zoomout (used for scale-to transforms)
1021    let effective_dpi = ((dpi as f64) * magnify / _zoomout) as u32;
1022    let n_to_process = nodes.len();
1023    latexml_core::telemetry::set_graphics_assets(n_to_process as u32);
1024
1025    // Counter for generating unique resource names (like Perl's generateResourcePathname)
1026    let mut resource_counter: u32 = 0;
1027
1028    // Two-phase plan so the slow per-image `convert` subprocess and
1029    // `read_image_dimensions` calls can run in parallel without touching
1030    // the libxml DOM off-thread.
1031    //
1032    // Phase 1 (serial): read each node's attributes, resolve source path,
1033    // decide the conversion kind, and allocate resource-name counters.
1034    //   - `Plan::NotFound` — apply fallback on the main thread later
1035    //   - `Plan::Copy { .. }` — apply on the main thread later (cheap)
1036    //   - `Plan::Convert { .. }` — independent; run in parallel.
1037    // Phase 2 (parallel): run convert_image + read_image_dimensions for
1038    // `Plan::Convert` entries. Produces `JobResult`s keyed by node index.
1039    // Phase 3 (serial): apply DOM mutations on the main thread in original
1040    // node order so attribute writes happen on the libxml-owning thread.
1041    enum Plan {
1042      NotFound {
1043        idx:     usize,
1044        graphic: String,
1045      },
1046      Copy {
1047        idx:     usize,
1048        source:  String,
1049        options: String,
1050      },
1051      Convert {
1052        idx:     usize,
1053        options: String,
1054        job_id:  usize,
1055      },
1056    }
1057    struct ConvertJob {
1058      job_id:       usize,
1059      source:       String,
1060      page:         Option<u32>,
1061      rel_dest:     String,
1062      abs_dest_str: String,
1063      /// `Some((rel_svg, abs_svg_str))` when the worker should
1064      /// first attempt the inkscape-SVG path and only fall back
1065      /// to `convert` on failure. `None` means the classic
1066      /// raster-only path.
1067      svg_paths:    Option<(String, String)>,
1068    }
1069    struct ConvertOutcome {
1070      job_id:   usize,
1071      /// Path to write into `imagesrc`; `None` if both convert and copy-fallback failed.
1072      imagesrc: Option<String>,
1073      /// Raw (pre-transform) dimensions read from whichever file we ended up with.
1074      raw_dims: Option<(u32, u32)>,
1075    }
1076
1077    let mut plans: Vec<Plan> = Vec::with_capacity(n_to_process);
1078    let mut convert_jobs: Vec<ConvertJob> = Vec::new();
1079    let mut convert_job_ids: HashMap<(String, Option<u32>, String), usize> = HashMap::new();
1080    let mut convert_source_counts: HashMap<String, u32> = HashMap::new();
1081    for (idx, node) in nodes.iter().enumerate() {
1082      let options = node.get_attribute("options").unwrap_or_default();
1083      let page = Self::parse_page_option(&options);
1084      let Some(source) = self.find_graphic_file(&doc, node, &search_paths) else {
1085        let graphic = node
1086          .get_attribute("graphic")
1087          .unwrap_or_else(|| "none".to_string());
1088        plans.push(Plan::NotFound { idx, graphic });
1089        continue;
1090      };
1091      let src_ext = Path::new(&source)
1092        .extension()
1093        .and_then(|e| e.to_str())
1094        .unwrap_or("")
1095        .to_lowercase();
1096      let props = self.type_properties.get(&src_ext).cloned();
1097      let dest_type = props
1098        .as_ref()
1099        .and_then(|p| p.destination_type.as_ref())
1100        .cloned()
1101        .unwrap_or(src_ext.clone());
1102      let needs_conversion = dest_type != src_ext;
1103      let has_page = page.is_some();
1104      if needs_conversion || has_page {
1105        let job_key = (source.clone(), page, options.clone());
1106        let job_id = if let Some(&job_id) = convert_job_ids.get(&job_key) {
1107          job_id
1108        } else {
1109          let prior_source_jobs = convert_source_counts.get(&source).copied().unwrap_or(0);
1110          let dest_name = if has_page || prior_source_jobs > 0 {
1111            resource_counter += 1;
1112            format!("x{}", resource_counter)
1113          } else {
1114            Path::new(&source)
1115              .file_stem()
1116              .and_then(|s| s.to_str())
1117              .unwrap_or("image")
1118              .to_string()
1119          };
1120          convert_source_counts.insert(source.clone(), prior_source_jobs + 1);
1121          // Vector-SVG path: opt-in for small PDFs only. We prepare an
1122          // alternate `.svg` destination path alongside the normal raster
1123          // destination so the worker can try inkscape first, then fall
1124          // back. The file-size heuristic gates this — see
1125          // `should_try_svg_path`.
1126          let try_svg = Self::should_try_svg_path(&source, self.svg_threshold_kb);
1127          let rel_dest = format!("{}.{}", dest_name, dest_type);
1128          let abs_dest = PathBuf::from(&dest_dir).join(&rel_dest);
1129          if let Some(parent) = abs_dest.parent() {
1130            std::fs::create_dir_all(parent).ok();
1131          }
1132          let abs_dest_str = abs_dest.to_string_lossy().to_string();
1133          let svg_paths = if try_svg {
1134            let rel_svg = format!("{}.svg", dest_name);
1135            let abs_svg = PathBuf::from(&dest_dir).join(&rel_svg);
1136            let abs_svg_str = abs_svg.to_string_lossy().to_string();
1137            Some((rel_svg, abs_svg_str))
1138          } else {
1139            None
1140          };
1141          let job_id = convert_jobs.len();
1142          convert_jobs.push(ConvertJob {
1143            job_id,
1144            source: source.clone(),
1145            page,
1146            rel_dest,
1147            abs_dest_str,
1148            svg_paths,
1149          });
1150          convert_job_ids.insert(job_key, job_id);
1151          job_id
1152        };
1153        plans.push(Plan::Convert { idx, options, job_id });
1154      } else {
1155        plans.push(Plan::Copy { idx, source, options });
1156      }
1157    }
1158
1159    // Phase 2: parallel conversions. Bounded worker count to avoid
1160    // oversubscribing when many images are in flight. `convert` itself
1161    // is single-threaded per invocation, so the ceiling is useful CPU
1162    // parallelism — capped at a reasonable limit to avoid fork/memory
1163    // storms with many-image papers.
1164    let convert_count = convert_jobs.len();
1165    let worker_cap = std::thread::available_parallelism()
1166      .map(|n| n.get())
1167      .unwrap_or(4)
1168      .clamp(1, 8);
1169    let n_workers = convert_count.min(worker_cap).max(1);
1170    let source_dir_ref = source_dir.as_str();
1171    let dest_dir_ref = dest_dir.as_str();
1172    let mut outcomes: Vec<ConvertOutcome> = Vec::with_capacity(convert_count);
1173    if convert_count > 0 {
1174      use std::sync::Mutex;
1175      use std::sync::atomic::{AtomicU32, AtomicUsize, Ordering};
1176      let next = AtomicUsize::new(0);
1177      let out = Mutex::new(Vec::<ConvertOutcome>::with_capacity(convert_count));
1178      // Subprocess tally: telemetry's thread_local! STATE is per-thread,
1179      // and worker threads exit before `phase_us[graphics]` aggregation,
1180      // so worker increments would be lost. Accumulate in a shared
1181      // AtomicU32 here and merge into telemetry once the scope joins.
1182      // One increment per `Self::convert_image_svg` / `Self::convert_image`
1183      // call (the EPS-via-PDF internal pair counts as one).
1184      let subproc_count = AtomicU32::new(0);
1185      let subproc_ref = &subproc_count;
1186      // Unique conversion jobs only. Repeated nodes with the same
1187      // source/page/options share one subprocess result, while distinct
1188      // options keep separate outputs.
1189      let jobs: Vec<&ConvertJob> = convert_jobs.iter().collect();
1190      std::thread::scope(|s| {
1191        for _ in 0..n_workers {
1192          s.spawn(|| {
1193            loop {
1194              let i = next.fetch_add(1, Ordering::Relaxed);
1195              if i >= jobs.len() {
1196                break;
1197              }
1198              let ConvertJob {
1199                job_id,
1200                source,
1201                page,
1202                rel_dest,
1203                abs_dest_str,
1204                svg_paths,
1205              } = jobs[i];
1206              // Try vector-SVG path first if requested for this source.
1207              // On success, pick dimensions from the SVG viewBox.
1208              let svg_outcome = if let Some((rel_svg, abs_svg)) = svg_paths {
1209                subproc_ref.fetch_add(1, Ordering::Relaxed);
1210                if Self::convert_image_svg(source, abs_svg, *page) {
1211                  let raw_dims = Self::read_svg_dimensions(abs_svg);
1212                  Some(ConvertOutcome {
1213                    job_id: *job_id,
1214                    imagesrc: Some(rel_svg.clone()),
1215                    raw_dims,
1216                  })
1217                } else {
1218                  log::warn!(
1219                    "Graphics: inkscape SVG path failed for {}, falling back to convert",
1220                    source
1221                  );
1222                  None
1223                }
1224              } else {
1225                None
1226              };
1227              let converted = if svg_outcome.is_none() {
1228                subproc_ref.fetch_add(1, Ordering::Relaxed);
1229                Self::convert_image(source, abs_dest_str, dpi, *page)
1230              } else {
1231                false
1232              };
1233              let outcome = if let Some(o) = svg_outcome {
1234                o
1235              } else if converted {
1236                ConvertOutcome {
1237                  job_id:   *job_id,
1238                  imagesrc: Some(rel_dest.clone()),
1239                  raw_dims: Self::read_image_dimensions(abs_dest_str),
1240                }
1241              } else {
1242                log::warn!("Graphics: Failed to convert {} to {}", source, abs_dest_str);
1243                if let Some(rel) = Self::copy_to_destination(source, source_dir_ref, dest_dir_ref) {
1244                  ConvertOutcome {
1245                    job_id:   *job_id,
1246                    imagesrc: Some(rel),
1247                    raw_dims: Self::read_image_dimensions(source),
1248                  }
1249                } else {
1250                  ConvertOutcome {
1251                    job_id:   *job_id,
1252                    imagesrc: None,
1253                    raw_dims: None,
1254                  }
1255                }
1256              };
1257              out.lock().unwrap().push(outcome);
1258            }
1259          });
1260        }
1261      });
1262      outcomes = out.into_inner().unwrap();
1263      outcomes.sort_by_key(|o| o.job_id);
1264      latexml_core::telemetry::add_graphics_subprocess(subproc_count.load(Ordering::Relaxed));
1265    }
1266    let outcomes_by_job: HashMap<usize, ConvertOutcome> =
1267      outcomes.into_iter().map(|o| (o.job_id, o)).collect();
1268
1269    // Phase 3: serial DOM mutations. Preserves original node order.
1270    let apply_transforms =
1271      |options: &str, raw_dims: Option<(u32, u32)>| -> (Option<u32>, Option<u32>) {
1272        match raw_dims {
1273          Some((w, h)) if !options.is_empty() => {
1274            let (tw, th) = Self::apply_graphicx_transforms(w, h, options, effective_dpi);
1275            (Some(tw), Some(th))
1276          },
1277          Some((w, h)) => (Some(w), Some(h)),
1278          None => (None, None),
1279        }
1280      };
1281    for plan in &plans {
1282      match plan {
1283        Plan::NotFound { idx, graphic } => {
1284          log::warn!("Graphics: No source found for {}", graphic);
1285          let mut node_mut = nodes[*idx].clone();
1286          node_mut.set_attribute("imagesrc", graphic).ok();
1287        },
1288        Plan::Copy { idx, source, options } => {
1289          let mut node_mut = nodes[*idx].clone();
1290          if let Some(rel) = Self::copy_to_destination(source, &source_dir, &dest_dir) {
1291            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
1292            Self::set_graphic_src(&mut node_mut, &rel, w, h);
1293          } else {
1294            let rel_path = Path::new(source)
1295              .strip_prefix(&source_dir)
1296              .unwrap_or(Path::new(source));
1297            let rel_str = rel_path.to_string_lossy().to_string();
1298            let (w, h) = apply_transforms(options, Self::read_image_dimensions(source));
1299            Self::set_graphic_src(&mut node_mut, &rel_str, w, h);
1300          }
1301        },
1302        Plan::Convert { idx, options, job_id } => {
1303          if let Some(out) = outcomes_by_job.get(job_id) {
1304            let mut node_mut = nodes[*idx].clone();
1305            if let Some(imagesrc) = &out.imagesrc {
1306              let (w, h) = apply_transforms(options, out.raw_dims);
1307              Self::set_graphic_src(&mut node_mut, imagesrc, w, h);
1308            }
1309          }
1310        },
1311      }
1312    }
1313
1314    log::info!(
1315      "Graphics {} {} to process",
1316      doc.get_destination().unwrap_or("?"),
1317      n_to_process
1318    );
1319    Ok(vec![doc])
1320  }
1321}
