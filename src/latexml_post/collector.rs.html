1//! Abstract collector base for content generation processors.
2//!
3//! Port of `LaTeXML::Post::Collector` (113 lines of Perl).
4//! Base class for processors that collect information from multiple documents
5//! and build derived content (indexes, bibliographies, etc.).
6//! Supports splitting collected content into sub-documents by initial letter.
7
8use libxml::tree::Node;
9use std::collections::HashMap;
10use std::path::Path;
11
12use crate::document::{NodeData, PostDocument};
13use crate::processor::{ProcessResult, Processor};

15/// Abstract collector post-processor.
16///
17/// Port of `LaTeXML::Post::Collector`.
18/// Subclasses (MakeIndex, MakeBibliography) implement the actual `process` method.
19pub struct Collector {
20  name: String,
21  resource_directory: Option<String>,
22  resource_prefix: Option<String>,
23  /// Optional scanner to rescan generated content.
24  /// In Perl: `$$self{scanner}->process($doc, $$self{scanner}->toProcess($doc))`
25  has_scanner: bool,
26}
27
28impl Collector {
29  pub fn new(name: &str) -> Self {
30    Collector {
31      name: name.to_string(),
32      resource_directory: None,
33      resource_prefix: None,
34      has_scanner: false,
35    }
36  }
37
38  /// Set whether this collector has an attached scanner for rescanning.
39  pub fn with_scanner(mut self) -> Self {
40    self.has_scanner = true;
41    self
42  }
43}

45/// Given collected content broken into portions by initial letter,
46/// fill in the main document with the first sub-collection,
47/// and create new documents for the rest.
48///
49/// Port of `Collector::makeSubCollectionDocuments`.
50///
51/// The `collections` map has: initial → XML data for that section.
52/// The first sub-collection fills the existing `root` element;
53/// each subsequent one gets a new sub-document.
54pub fn make_sub_collection_documents(
55  doc: &mut PostDocument,
56  root: &Node,
57  collections: &HashMap<String, Vec<NodeData>>,
58) -> Vec<PostDocument> {
59  let mut initials: Vec<&String> = collections.keys().collect();
60  initials.sort();
61
62  if initials.is_empty() {
63    return vec![];
64  }
65
66  let _root_tag = doc.get_qname(root).unwrap_or_else(|| "ltx:index".to_string());
67  let root_id = root.get_attribute("xml:id").unwrap_or_default();
68
69  // Build (id, initial) pairs for each sub-collection
70  let ids: Vec<(String, &str)> = initials.iter().enumerate().map(|(i, init)| {
71    if i == 0 {
72      (root_id.clone(), init.as_str())
73    } else {
74      (format!("{}.{}", root_id, init), init.as_str())
75    }
76  }).collect();
77
78  // For the first sub-collection, fill the main document's root element
79  if let Some(first_init) = initials.first() {
80    if let Some(data) = collections.get(*first_init) {
81      // Build TOC linking all sub-collections
82      let toc_entries: Vec<NodeData> = ids.iter().enumerate().map(|(i, (id, init))| {
83        if i == 0 {
84          NodeData::Element {
85            tag: "ltx:tocentry".to_string(),
86            attributes: None,
87            children: vec![NodeData::Text(init.to_string())],
88          }
89        } else {
90          NodeData::Element {
91            tag: "ltx:tocentry".to_string(),
92            attributes: None,
93            children: vec![NodeData::Element {
94              tag: "ltx:ref".to_string(),
95              attributes: Some(HashMap::from([
96                ("idref".to_string(), id.clone()),
97                ("show".to_string(), "refnum".to_string()),
98              ])),
99              children: vec![NodeData::Text(init.to_string())],
100            }],
101          }
102        }
103      }).collect();
104
105      let toc = NodeData::Element {
106        tag: "ltx:TOC".to_string(),
107        attributes: Some(HashMap::from([("format".to_string(), "veryshort".to_string())])),
108        children: vec![NodeData::Element {
109          tag: "ltx:toclist".to_string(),
110          attributes: None,
111          children: toc_entries,
112        }],
113      };
114
115      let mut root_mut = root.clone();
116      let mut content = vec![toc];
117      content.extend(data.clone());
118      doc.add_nodes(&mut root_mut, &content);
119    }
120  }
121
122  // For subsequent sub-collections, we'd create new documents
123  // This requires PostDocument::newDocument which needs more infrastructure
124  log::info!(
125    "Collector: {} sub-collections by initial: {:?}",
126    initials.len(),
127    initials
128  );
129
130  // NOTE: Sub-documents for initials[1..] require PostDocument::newDocument
131  //   which creates XML documents from element roots with proper ID remapping.
132  vec![]
133}

135/// Compute a page name for a sub-collection document.
136///
137/// If the main document is "index.html", use just the initial as the name.
138/// Otherwise, append the initial to the document name.
139///
140/// Port of `Collector::getPageName`.
141pub fn get_page_name(doc: &PostDocument, initial: &str) -> String {
142  if let Some(dest) = doc.get_destination() {
143    let path = Path::new(dest);
144    let stem = path.file_stem().and_then(|s| s.to_str()).unwrap_or("doc");
145    let ext = path.extension().and_then(|e| e.to_str()).unwrap_or("xml");
146    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or(".");
147    let name = if stem == "index" {
148      initial.to_string()
149    } else {
150      format!("{}.{}", stem, initial)
151    };
152    format!("{}/{}.{}", dir, name, ext)
153  } else {
154    format!("{}.xml", initial)
155  }
156}

158impl Processor for Collector {
159  fn get_name(&self) -> &str {
160    &self.name
161  }
162
163  fn process(&mut self, doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult {
164    log::warn!("Abstract Collector::process called");
165    Ok(vec![doc])
166  }
