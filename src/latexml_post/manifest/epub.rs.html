1//! EPUB manifest creation.
2//!
3//! Port of `LaTeXML::Post::Manifest::Epub` (252 lines of Perl).
4//! Creates the EPUB 3.2 package structure:
5//! - `mimetype` file
6//! - `META-INF/container.xml`
7//! - `OPS/content.opf` (spine + manifest)
8//! - Indexes all content files with correct media types
9
10use std::fs;
11use std::path::Path;

13/// EPUB 3.2 container.xml content.
14const CONTAINER_XML: &str = r#"<?xml version="1.0"?>
15<container version="1.0" xmlns="urn:oasis:names:tc:opendocument:xmlns:container">
16    <rootfiles>
17        <rootfile full-path="OPS/content.opf" media-type="application/oebps-package+xml"/>
18   </rootfiles>
19</container>"#;

21/// Core Media Types as per EPUB 3.2 spec.
22fn core_media_type(ext: &str) -> &'static str {
23  match ext.to_lowercase().as_str() {
24    "gif" => "image/gif",
25    "jpg" | "jpeg" => "image/jpeg",
26    "png" => "image/png",
27    "svg" => "image/svg+xml",
28    "mp3" => "audio/mpeg",
29    "mp4" | "mpg4" => "audio/mp4",
30    "css" => "text/css",
31    "ttf" => "font/ttf",
32    "otf" => "font/otf",
33    "woff" => "font/woff",
34    "woff2" => "font/woff2",
35    "xhtml" => "application/xhtml+xml",
36    "js" => "text/javascript",
37    "ncx" => "application/x-dtbncx+xml",
38    "smi" | "smil" => "application/smil+xml",
39    "pls" => "application/pls+xml",
40    _ => "application/octet-stream",
41  }
42}

44/// EPUB manifest builder.
45///
46/// Port of `LaTeXML::Post::Manifest::Epub`.
47pub struct EpubManifest {
48  site_directory:    String,
49  unique_identifier: Option<String>,
50}
51
52impl EpubManifest {
53  pub fn new(site_directory: &str) -> Self {
54    EpubManifest {
55      site_directory:    site_directory.to_string(),
56      unique_identifier: None,
57    }
58  }
59
60  /// Initialize the EPUB directory structure.
61  ///
62  /// Port of `Epub::initialize`.
63  pub fn initialize(
64    &mut self,
65    _title: &str,
66    _authors: &[String],
67    _language: &str,
68  ) -> Result<(), String> {
69    let dir = &self.site_directory;
70
71    // 1. Create mimetype file
72    let mime_path = format!("{}/mimetype", dir);
73    fs::write(&mime_path, "application/epub+zip")
74      .map_err(|e| format!("Couldn't write mimetype: {}", e))?;
75
76    // 2. Create META-INF/container.xml
77    let meta_inf = format!("{}/META-INF", dir);
78    fs::create_dir_all(&meta_inf).map_err(|e| format!("Couldn't create META-INF: {}", e))?;
79    fs::write(format!("{}/container.xml", meta_inf), CONTAINER_XML)
80      .map_err(|e| format!("Couldn't write container.xml: {}", e))?;
81
82    // 3. Create OPS directory
83    let ops_dir = format!("{}/OPS", dir);
84    fs::create_dir_all(&ops_dir).map_err(|e| format!("Couldn't create OPS: {}", e))?;
85
86    // Generate a UUID for the publication
87    self.unique_identifier = Some(format!("urn:uuid:{}", generate_uuid()));
88
89    Ok(())
90  }

92  /// Add a document to the EPUB spine.
93  ///
94  /// Port of `Epub::process` per-document loop.
95  pub fn add_document(
96    &self,
97    destination: &str,
98    has_math: bool,
99    has_svg: bool,
100    has_nav: bool,
101  ) -> SpineEntry {
102    let path = Path::new(destination);
103    let name = path.file_stem().and_then(|s| s.to_str()).unwrap_or("doc");
104    let ext = path.extension().and_then(|e| e.to_str()).unwrap_or("xhtml");
105    let item_id = url_to_id(&format!("{}.{}", name, ext));
106
107    let mut properties = Vec::new();
108    if has_math {
109      properties.push("mathml");
110    }
111    if has_svg {
112      properties.push("svg");
113    }
114    if has_nav {
115      properties.push("nav");
116    }
117
118    SpineEntry {
119      id:         item_id,
120      href:       format!("{}.{}", name, ext),
121      media_type: "application/xhtml+xml".to_string(),
122      properties: if properties.is_empty() {
123        None
124      } else {
125        Some(properties.join(" "))
126      },
127    }
128  }

130  /// Generate the content.opf XML as a string.
131  ///
132  /// Port of `Epub::finalize`.
133  pub fn generate_opf(
134    &self,
135    title: &str,
136    authors: &[String],
137    language: &str,
138    spine: &[SpineEntry],
139    resources: &[ResourceEntry],
140  ) -> String {
141    let uid = self
142      .unique_identifier
143      .as_deref()
144      .unwrap_or("urn:uuid:00000000-0000-0000-0000-000000000000");
145    let now = chrono_like_now();
146
147    let mut xml = String::new();
148    xml.push_str("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
149    xml.push_str("<package xmlns=\"http://www.idpf.org/2007/opf\" unique-identifier=\"pub-id\" version=\"3.0\">\n");
150
151    // Metadata
152    xml.push_str("  <metadata xmlns:dc=\"http://purl.org/dc/elements/1.1/\">\n");
153    xml.push_str(&format!("    <dc:title>{}</dc:title>\n", escape_xml(title)));
154    for author in authors {
155      xml.push_str(&format!(
156        "    <dc:creator>{}</dc:creator>\n",
157        escape_xml(author)
158      ));
159    }
160    xml.push_str(&format!("    <dc:language>{}</dc:language>\n", language));
161    xml.push_str(&format!(
162      "    <meta property=\"dcterms:modified\">{}</meta>\n",
163      now
164    ));
165    xml.push_str(&format!(
166      "    <dc:identifier id=\"pub-id\">{}</dc:identifier>\n",
167      uid
168    ));
169    xml.push_str("  </metadata>\n");
170
171    // Manifest
172    xml.push_str("  <manifest>\n");
173    for entry in spine {
174      xml.push_str(&format!(
175        "    <item id=\"{}\" href=\"{}\" media-type=\"{}\"",
176        entry.id, entry.href, entry.media_type
177      ));
178      if let Some(ref props) = entry.properties {
179        xml.push_str(&format!(" properties=\"{}\"", props));
180      }
181      xml.push_str("/>\n");
182    }
183    for res in resources {
184      xml.push_str(&format!(
185        "    <item id=\"{}\" href=\"{}\" media-type=\"{}\"/>\n",
186        res.id, res.href, res.media_type
187      ));
188    }
189    xml.push_str("  </manifest>\n");
190
191    // Spine
192    xml.push_str("  <spine>\n");
193    for entry in spine {
194      xml.push_str(&format!("    <itemref idref=\"{}\"/>\n", entry.id));
195    }
196    xml.push_str("  </spine>\n");
197    xml.push_str("</package>\n");
198
199    xml
200  }
201}

220/// Convert a URL/filename to a valid NCName for use as an XML id.
221///
222/// Port of `url_id`.
223fn url_to_id(name: &str) -> String {
224  let mut result = String::from("_");
225  for ch in name.chars() {
226    if ch.is_ascii_alphanumeric() || ch == '_' || ch == '-' || ch == '.' {
227      result.push(ch);
228    } else {
229      result.push_str(&format!("_x{:X}_", ch as u32));
230    }
231  }
232  result
233}

235/// Generate a UUID v4 string.
236fn generate_uuid() -> String {
237  // Simple random UUID v4 (no external dependency)
238  use std::time::{SystemTime, UNIX_EPOCH};
239  let seed = SystemTime::now()
240    .duration_since(UNIX_EPOCH)
241    .unwrap_or_default()
242    .as_nanos();
243  // LCG-based pseudo-random for simplicity
244  let mut state = seed as u64;
245  let mut bytes = [0u8; 16];
246  for b in &mut bytes {
247    state = state
248      .wrapping_mul(6364136223846793005)
249      .wrapping_add(1442695040888963407);
250    *b = (state >> 33) as u8;
251  }
252  bytes[6] = (bytes[6] & 0x0F) | 0x40; // version 4
253  bytes[8] = (bytes[8] & 0x3F) | 0x80; // variant 1
254  format!(
255    "{:02x}{:02x}{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
256    bytes[0],
257    bytes[1],
258    bytes[2],
259    bytes[3],
260    bytes[4],
261    bytes[5],
262    bytes[6],
263    bytes[7],
264    bytes[8],
265    bytes[9],
266    bytes[10],
267    bytes[11],
268    bytes[12],
269    bytes[13],
270    bytes[14],
271    bytes[15]
272  )
273}

275/// Simple XML escaping.
276fn escape_xml(s: &str) -> String {
277  s.replace('&', "&amp;")
278    .replace('<', "&lt;")
279    .replace('>', "&gt;")
280    .replace('"', "&quot;")
281}

283/// Generate an ISO 8601 timestamp (CCYY-MM-DDThh:mm:ssZ).
284fn chrono_like_now() -> String {
285  use std::time::{SystemTime, UNIX_EPOCH};
286  let secs = SystemTime::now()
287    .duration_since(UNIX_EPOCH)
288    .unwrap_or_default()
289    .as_secs();
290  // Simple UTC timestamp computation (no chrono dependency)
291  let days = secs / 86400;
292  let time_of_day = secs % 86400;
293  let hours = time_of_day / 3600;
294  let minutes = (time_of_day % 3600) / 60;
295  let seconds = time_of_day % 60;
296  // Rough date from epoch days (good enough for timestamps)
297  let (year, month, day) = days_to_ymd(days as i64);
298  format!(
299    "{:04}-{:02}-{:02}T{:02}:{:02}:{:02}Z",
300    year, month, day, hours, minutes, seconds
301  )
302}

304/// Convert days since Unix epoch to (year, month, day).
305fn days_to_ymd(days: i64) -> (i64, u32, u32) {
306  // Algorithm from https://howardhinnant.github.io/date_algorithms.html
307  let z = days + 719468;
308  let era = if z >= 0 { z } else { z - 146096 } / 146097;
309  let doe = (z - era * 146097) as u32;
310  let yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
311  let y = yoe as i64 + era * 400;
312  let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
313  let mp = (5 * doy + 2) / 153;
314  let d = doy - (153 * mp + 2) / 5 + 1;
315  let m = if mp < 10 { mp + 3 } else { mp - 9 };
316  (y + if m <= 2 { 1 } else { 0 }, m, d)
317}
