1//! Cross-reference resolution processor.
2//!
3//! Port of `LaTeXML::Post::CrossRef` (946 lines of Perl).
4//! Resolves cross-references (`ltx:ref`, `ltx:bibref`, etc.) by looking up
5//! referenced IDs in the ObjectDB and filling in the reference text,
6//! titles, and navigation links.

15/// Sectional element types that appear in TOCs.
16const NORMAL_TOC_TYPES: &[&str] = &[
17  "ltx:document", "ltx:part", "ltx:chapter", "ltx:section",
18  "ltx:subsection", "ltx:subsubsection", "ltx:paragraph", "ltx:subparagraph",
19  "ltx:index", "ltx:bibliography", "ltx:glossary", "ltx:appendix",
20];
21
22/// Fallback fields when a requested ref show key is not found.
23fn ref_fallbacks(key: &str) -> &'static [&'static str] {
24  match key {
25    "typerefnum" => &["refnum"],
26    "toctitle" => &["title", "toccaption"],
27    "title" => &["toccaption"],
28    "rawtoctitle" => &["toctitle", "title", "toccaption"],
29    "rawtitle" => &["title", "toccaption"],
30    _ => &[],
31  }
32}

34/// URL style for cross-references.
35#[derive(Debug, Clone)]
36pub enum UrlStyle {
37  /// Use file.html#fragment
38  File,
39  /// Use server-side paths (strip trailing index.ext)
40  Server,
41  /// Negotiated: strip file extension and trailing index
42  Negotiated,
43}

45/// CrossRef post-processor.
46///
47/// Port of `LaTeXML::Post::CrossRef`.
48pub struct CrossRef {
49  name: String,
50  /// Reference to the shared ObjectDB.
51  pub db: ObjectDB,
52  /// URL style for cross-references.
53  url_style: UrlStyle,
54  /// File extension used for output (e.g. "html", "xml").
55  extension: String,
56  /// Default show format for TOC refs.
57  toc_show: String,
58  /// Default show format for regular refs.
59  ref_show: String,
60  /// Minimum useful content length for refs.
61  min_ref_length: usize,
62  /// Join string between parent+child ref text.
63  ref_join: String,
64  /// Type of navigation TOC to add (e.g. "context").
65  navigation_toc: Option<String>,
66  /// Track missing references for reporting.
67  missing: HashMap<String, HashMap<String, HashMap<String, u32>>>,
68}

96  /// Note a missing reference.
97  fn note_missing(&mut self, severity: &str, ref_type: &str, key: &str) {
98    self.missing
99      .entry(severity.to_string())
100      .or_default()
101      .entry(ref_type.to_string())

108  /// Generate a URL for a referenced ID.
109  ///
110  /// Port of `CrossRef::generateURL`.
111  fn generate_url(&mut self, doc: &PostDocument, id: &str) -> Option<String> {
112    let entry = self.db.lookup(&format!("ID:{}", id))?;
113    let location = entry.get_string("location")?;
114
115    let doc_location = doc.site_relative_destination().unwrap_or_default();
116    let mut url = relative_url(location, &doc_location);
117
118    // Apply URL style
119    match self.url_style {
120      UrlStyle::Server => {
121        let index_suffix = format!("index.{}", self.extension);
122        if url.ends_with(&index_suffix) {
123          let prefix = &url[..url.len() - index_suffix.len()];
124          url = if prefix.is_empty() { "./".to_string() } else { prefix.to_string() };
125        }
126      }
127      UrlStyle::Negotiated => {
128        let ext_suffix = format!(".{}", self.extension);
129        if url.ends_with(&ext_suffix) {
130          url = url[..url.len() - ext_suffix.len()].to_string();
131        }
132        if url.ends_with("/index") {
133          url = url[..url.len() - 5].to_string();
134        }
135      }
136      UrlStyle::File => {}
137    }
138
139    if url.is_empty() {
140      url = ".".to_string();
141    }
142
143    // Add fragment ID
144    let fragid = entry.get_string("fragid").map(String::from);
145    let loc = location.to_string();
146    if let Some(fid) = fragid {
147      if url == "." || loc == doc_location {
148        url = String::new();
149      }
150      url = format!("{}#{}", url, fid);
151    } else if loc == doc_location {
152      url = String::new();
153    }
154
155    Some(url)
156  }

158  /// Generate a title string for a referenced ID, traversing parents for context.
159  ///
160  /// Port of `CrossRef::generateTitle`.
161  fn generate_title(&self, _doc: &PostDocument, id: &str, shown: &str) -> Option<String> {
162    let mut current_id = id.to_string();
163    let mut result = String::new();
164    let mut prefix = String::new();
165    let mut shown_so_far = shown.to_string();
166
167    while let Some(entry) = self.db.lookup(&format!("ID:{}", current_id)) {
168      let mut pieces = Vec::new();
169      let mut is_dup = false;
170
171      // Try title, then typerefnum, then refnum
172      if let Some(title_val) = entry.get_value("title") {
173        if title_val.is_truthy() {
174          is_dup = shown_so_far.contains("title");
175          pieces.push(title_val.to_string());
176        }
177      }
178      if pieces.is_empty() {
179        let has_type = entry.get_value("tag:creftypecap")
180          .or_else(|| entry.get_value("tag:creftype"));
181        let has_refnum = entry.get_value("refnum");
182        if has_type.is_some() && has_refnum.is_some() {
183          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
184          if let Some(t) = has_type { pieces.push(t.to_string()); }
185          if let Some(r) = has_refnum { pieces.push(r.to_string()); }
186        } else if let Some(tr) = entry.get_value("typerefnum") {
187          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
188          pieces.push(tr.to_string());
189        } else if let Some(r) = has_refnum {
190          is_dup = shown_so_far.contains("refnum");
191          pieces.push(r.to_string());
192        }
193      }
194
195      if is_dup {
196        prefix = "In ".to_string();
197        shown_so_far.clear();
198      } else {
199        let title = pieces.join(" ");
200        let title = title.trim();
201        if !title.is_empty() {
202          result.push_str(&prefix);
203          prefix = self.ref_join.clone();
204          result.push_str(title);
205        }
206      }
207
208      // Walk to parent for more context
209      match entry.get_string("parent").map(String::from) {
210        Some(pid) => current_id = pid,
211        None => break,
212      }
213    }
214
215    if result.is_empty() { None } else { Some(result) }
216  }

218  /// Generate a title for the document itself.
219  ///
220  /// Port of `CrossRef::generateDocumentTitle`.
221  fn generate_document_title(&self, doc: &PostDocument) -> Option<String> {
222    // Try to generate from the document's root ID
223    if let Some(docid) = doc.get_document_element().and_then(|r| r.get_attribute("xml:id")) {
224      let title = self.generate_title(doc, &docid, "toctitle");
225      if title.as_ref().map(|t| !t.is_empty()).unwrap_or(false) {
226        return title;
227      }
228    }
229    // Fallback: look for a title element in the document
230    if let Some(node) = doc.findnode("//ltx:title | //ltx:toctitle | //ltx:caption | //ltx:toccaption") {
231      let text = get_text_content_node(&node);
232      if !text.is_empty() {
233        return Some(text);
234      }
235    }
236    None
237  }

239  /// Generate content for a glossary reference.
240  ///
241  /// Port of `CrossRef::generateGlossaryRefTitle`.
242  fn generate_glossary_ref_title(
243    &self,
244    entry_key: &str,
245    show: &str,
246  ) -> Vec<NodeData> {
247    let entry = match self.db.lookup(entry_key) {
248      Some(e) => e,
249      None => return vec![],
250    };
251
252    let phrase_key = format!("phrase:{}", show);
253    if let Some(val) = entry.get_value(&phrase_key) {
254      return vec![NodeData::Element {
255        tag: "ltx:text".to_string(),
256        attributes: Some(HashMap::from([(
257          "class".to_string(),
258          format!("ltx_glossary_{}", show),
259        )])),
260        children: vec![NodeData::Text(val.to_string())],
261      }];
262    }
263
264    // Handle -plural and -indefinite suffixes
265    if let Some(base_show) = show.strip_suffix("-plural") {
266      let base_key = format!("phrase:{}", base_show);
267      if let Some(val) = entry.get_value(&base_key) {
268        return vec![NodeData::Element {
269          tag: "ltx:text".to_string(),
270          attributes: Some(HashMap::from([(
271            "class".to_string(),
272            format!("ltx_glossary_{}", show),
273          )])),
274          children: vec![NodeData::Text(format!("{}s", val))],
275        }];
276      }
277    }
278    if let Some(base_show) = show.strip_suffix("-indefinite") {
279      let base_key = format!("phrase:{}", base_show);
280      if let Some(val) = entry.get_value(&base_key) {
281        let text = val.to_string();
282        let article = if text.starts_with(|c: char| "aeiouAEIOU".contains(c)) {
283          "an "
284        } else {
285          "a "
286        };
287        return vec![NodeData::Element {
288          tag: "ltx:text".to_string(),
289          attributes: Some(HashMap::from([(
290            "class".to_string(),
291            format!("ltx_glossary_{}", show),
292          )])),
293          children: vec![
294            NodeData::Text(article.to_string()),
295            NodeData::Text(text),
296          ],
297        }];
298      }
299    }
300
301    vec![]
302  }

320  /// Generate reference content for a given ID and show pattern.
321  ///
322  /// Port of `CrossRef::generateRef`.
323  fn generate_ref(&mut self, _doc: &PostDocument, req_id: &str, req_show: &str) -> Vec<NodeData> {
324    let show_options = if !req_show.contains("title") {
325      vec![req_show.to_string(), "title".to_string()]
326    } else {
327      vec![req_show.to_string(), "refnum".to_string()]
328    };
329
330    for show in &show_options {
331      let mut stuff = Vec::new();
332      let mut id = req_id.to_string();
333      let mut pending = String::new();
334      loop {
335        let entry_exists = self.db.lookup(&format!("ID:{}", id)).is_some();
336        if !entry_exists {
337          break;
338        }
339        let s = self.generate_ref_aux(&id, show);
340        if !s.is_empty() {
341          if !pending.is_empty() {
342            stuff.push(NodeData::Text(pending.clone()));
343          }
344          stuff.extend(s);
345          if self.check_ref_content(&stuff) {
346            return stuff;
347          }
348          pending = self.ref_join.clone();
349        }
350        let parent = self.db.lookup(&format!("ID:{}", id))
351          .and_then(|e| e.get_string("parent").map(String::from));
352        match parent {
353          Some(pid) => id = pid,
354          None => break,
355        }
356      }
357      if !stuff.is_empty() {
358        return stuff;
359      }
360    }
361
362    self.note_missing("info", "Usable title for ID", req_id);
363    vec![NodeData::Text(req_id.to_string())]
364  }

366  /// Generate ref content from a single DB entry.
367  fn generate_ref_aux(&self, id: &str, show: &str) -> Vec<NodeData> {
368    let entry = match self.db.lookup(&format!("ID:{}", id)) {
369      Some(e) => e,
370      None => return vec![],
371    };
372
373    let mut stuff = Vec::new();
374    let mut ok = false;
375    let mut remaining = show.to_string();
376
377    while !remaining.is_empty() {
378      if remaining.starts_with(|c: char| c.is_alphanumeric()) {
379        let keyword: String = remaining.chars().take_while(|c| c.is_alphanumeric()).collect();
380        remaining = remaining[keyword.len()..].to_string();
381        let key = keyword.to_lowercase();
382        let class = if key.contains("title") { "ltx_ref_title" } else { "ltx_ref_tag" };
383
384        let mut keys_to_try = vec![key.clone(), format!("tag:{}", key)];
385        keys_to_try.extend(ref_fallbacks(&key).iter().map(|s| s.to_string()));
386
387        for k in &keys_to_try {
388          if let Some(val) = entry.get_value(k) {
389            if val.is_truthy() {
390              ok = true;
391              let text = val.to_string();
392              stuff.push(NodeData::Element {
393                tag: "ltx:text".to_string(),
394                attributes: Some(HashMap::from([("class".to_string(), class.to_string())])),
395                children: vec![NodeData::Text(text)],
396              });
397              break;
398            }
399          }
400        }
401      } else if remaining.starts_with('{') {
402        if let Some(end) = remaining[1..].find('}') {
403          let literal = &remaining[1..1 + end];
404          if !literal.is_empty() {
405            stuff.push(NodeData::Text(literal.to_string()));
406          }
407          remaining = remaining[2 + end..].to_string();
408        } else {
409          remaining.clear();
410        }
411      } else if remaining.starts_with('~') {
412        remaining = remaining[1..].to_string();
413        if !stuff.is_empty() {
414          stuff.push(NodeData::Text("\u{00A0}".to_string()));
415        }
416      } else if remaining.starts_with(|c: char| c.is_whitespace()) {
417        let ws: String = remaining.chars().take_while(|c| c.is_whitespace()).collect();
418        remaining = remaining[ws.len()..].to_string();
419        if !stuff.is_empty() {
420          stuff.push(NodeData::Text(ws));
421        }
422      } else {
423        let sym: String = remaining.chars()
424          .take_while(|c| !c.is_alphanumeric() && *c != '{' && *c != '~')
425          .collect();
426        remaining = remaining[sym.len()..].to_string();
427        stuff.push(NodeData::Text(sym));
428      }
429    }
430
431    if ok { stuff } else { vec![] }
432  }

434  /// Check if ref content is "good enough".
435  fn check_ref_content(&self, stuff: &[NodeData]) -> bool {
436    let text = text_content(stuff);
437    let cleaned = text.replace("in ", "");
438    cleaned.chars().any(|c| c.is_alphanumeric())
439  }

444  fn fill_in_relations(&mut self, doc: &mut PostDocument) {
445    let root_id = match doc.get_document_element().and_then(|r| r.get_attribute("xml:id")) {
446      Some(id) => id,
447      None => return,
448    };
449
450    let mut current_id = root_id;
451    let mut rel = "up".to_string();
452    loop {
453      let parent_id = self.db.lookup(&format!("ID:{}", current_id))
454        .and_then(|e| e.get_string("parent").map(String::from));
455      match parent_id {
456        Some(pid) => {
457          let has_title = self.db.lookup(&format!("ID:{}", pid))
458            .and_then(|e| e.get_value("title"))
459            .map(|v| v.is_truthy())
460            .unwrap_or(false);
461          if has_title {
462            doc.add_navigation(&rel, &pid);
463            rel = format!("{} up", rel);
464          }
465          current_id = pid;
466        }
467        None => break,
468      }
469    }
470  }

472  fn fill_in_tocs(&mut self, doc: &mut PostDocument) {
473    let tocs = doc.findnodes("descendant::ltx:TOC[not(ltx:toclist)]");
474    for toc in &tocs {
475      let id = doc.get_document_element()
476        .and_then(|r| r.get_attribute("xml:id"))
477        .unwrap_or_default();
478      let show = toc.get_attribute("show").unwrap_or_else(|| self.toc_show.clone());
479
480      let list = self.gen_toc(&id, &show);
481      if !list.is_empty() {
482        let toclist = NodeData::Element {
483          tag: "ltx:toclist".to_string(),
484          attributes: None,
485          children: list,
486        };
487        let mut toc_mut = toc.clone();
488        doc.add_nodes(&mut toc_mut, &[toclist]);
489      }
490    }
491  }

493  fn gen_toc(&self, id: &str, show: &str) -> Vec<NodeData> {
494    let entry = match self.db.lookup(&format!("ID:{}", id)) {
495      Some(e) => e,
496      None => return vec![],
497    };
498
499    let children = entry.get_children();
500    let kids: Vec<NodeData> = children
501      .iter()
502      .flat_map(|child_id| self.gen_toc(child_id, show))
503      .collect();
504
505    let entry_type = entry.get_string("type").unwrap_or("");
506    let is_toc_type = NORMAL_TOC_TYPES.contains(&entry_type);
507    let in_toc = entry.get_value("inlist")
508      .map(|v| match v {
509        Value::Hash(h) => h.contains_key("toc"),
510        _ => false,
511      })
512      .unwrap_or(false);
513
514    if is_toc_type && in_toc {
515      let type_name = entry_type.strip_prefix("ltx:").unwrap_or(entry_type);
516      let mut toc_children = vec![NodeData::Element {
517        tag: "ltx:ref".to_string(),
518        attributes: Some(HashMap::from([
519          ("show".to_string(), show.to_string()),
520          ("idref".to_string(), id.to_string()),
521        ])),
522        children: vec![],
523      }];
524      if !kids.is_empty() {
525        toc_children.push(NodeData::Element {
526          tag: "ltx:toclist".to_string(),
527          attributes: Some(HashMap::from([(
528            "class".to_string(),
529            format!("ltx_toclist_{}", type_name),
530          )])),
531          children: kids,
532        });
533      }
534      vec![NodeData::Element {
535        tag: "ltx:tocentry".to_string(),
536        attributes: Some(HashMap::from([(
537          "class".to_string(),
538          format!("ltx_tocentry_{}", type_name),
539        )])),
540        children: toc_children,
541      }]
542    } else {
543      kids
544    }
545  }

547  fn fill_in_frags(&self, doc: &PostDocument) {
548    for node in &doc.findnodes("//*[@xml:id]") {
549      // NOTE: get_attribute("xml:id") may not work in all libxml2 builds;
550      // get_property("id") is more reliable for xml: namespace attributes.
551      // TODO: Fix in rust-libxml — get_attribute should handle xml: prefix.
552      let id = node.get_attribute("xml:id")
553        .or_else(|| node.get_property("id"));
554      if let Some(id) = id {
555        if let Some(entry) = self.db.lookup(&format!("ID:{}", id)) {
556          if let Some(fragid) = entry.get_string("fragid") {
557            let mut n = node.clone();
558            n.set_attribute("fragid", fragid).ok();
559          }
560        }
561      }
562    }
563  }

565  fn fill_in_refs(&mut self, doc: &mut PostDocument) {
566    let mut refs = doc.findnodes("//*[@idref]");
567    refs.extend(doc.findnodes("//*[@labelref]"));
568    for ref_node in &refs {
569      let tag = doc.get_qname(ref_node).unwrap_or_default();
570      if tag == "ltx:XMRef" {
571        continue;
572      }
573
574      let mut ref_mut = ref_node.clone();
575      let mut id = ref_node.get_attribute("idref");
576      let show = ref_node.get_attribute("show").unwrap_or_else(|| self.ref_show.clone());
577
578      if id.is_none() {
579        if let Some(label) = ref_node.get_attribute("labelref") {
580          if let Some(entry) = self.db.lookup(&label) {
581            if let Some(resolved_id) = entry.get_string("id") {
582              ref_mut.set_attribute("idref", resolved_id).ok();
583              id = Some(resolved_id.to_string());
584            }
585          }
586          if id.is_none() {
587            self.note_missing("warn", "Target for Label", &label);
588            PostDocument::add_class(&mut ref_mut, "ltx_missing_label");
589          }
590        }
591      }
592
593      if let Some(ref id_str) = id {
594        if ref_mut.get_attribute("href").is_none() {
595          if let Some(url) = self.generate_url(doc, id_str) {
596            ref_mut.set_attribute("href", &url).ok();
597          }
598        }
599        if ref_mut.get_attribute("title").is_none() {
600          if let Some(titlestring) = self.generate_title(doc, id_str, &show) {
601            ref_mut.set_attribute("title", &titlestring).ok();
602          }
603        }
604        if ref_mut.get_first_child().is_none() && tag != "ltx:graphics" && tag != "ltx:picture" {
605          let content = self.generate_ref(doc, id_str, &show);
606          doc.add_nodes(&mut ref_mut, &content);
607        }
608      }
609    }
610  }

612  fn fill_in_glossaryrefs(&mut self, doc: &mut PostDocument) {
613    for ref_node in &doc.findnodes("descendant::ltx:glossaryref") {
614      let mut ref_mut = ref_node.clone();
615      let key = ref_node.get_attribute("key").unwrap_or_default();
616      let list = ref_node.get_attribute("inlist").unwrap_or_default();
617
618      let gkey = format!("GLOSSARY:{}:{}", list, key);
619      if let Some(entry) = self.db.lookup(&gkey) {
620        if let Some(id) = entry.get_string("id") {
621          ref_mut.set_attribute("idref", id).ok();
622        }
623      } else {
624        self.note_missing("warn", "Glossary Entry for key", &key);
625      }
626
627      if ref_mut.get_first_child().is_none() {
628        doc.add_nodes(&mut ref_mut, &[NodeData::Text(key.clone())]);
629        PostDocument::add_class(&mut ref_mut, "ltx_missing");
630      }
631    }
632  }
633
634  fn fill_in_bibrefs(&mut self, doc: &mut PostDocument) {
635    let bibrefs = doc.findnodes("//ltx:bibref");
636    for bibref in &bibrefs {
637      let keys_str = bibref.get_attribute("bibrefs").unwrap_or_default();
638      let _show = bibref.get_attribute("show").unwrap_or_else(|| "refnum".to_string());
639      let sep = bibref.get_attribute("separator").unwrap_or_else(|| ",".to_string());
640      let lists_str = bibref.get_attribute("inlist").unwrap_or_else(|| "bibliography".to_string());
641      let lists: Vec<&str> = lists_str.split_whitespace().collect();
642
643      let mut refs: Vec<NodeData> = Vec::new();
644      for key in keys_str.split(',').filter(|k| !k.is_empty()) {
645        let mut found_id = None;
646        for list in &lists {
647          let bkey = format!("BIBLABEL:{}:{}", list, key);
648          if let Some(bentry) = self.db.lookup(&bkey) {
649            found_id = bentry.get_string("id").map(String::from);
650            if found_id.is_some() { break; }
651          }
652        }
653        if !refs.is_empty() {
654          refs.push(NodeData::Text(format!("{} ", sep)));
655        }
656        if let Some(id) = found_id {
657          let mut attrs = HashMap::new();
658          attrs.insert("idref".to_string(), id.clone());
659          if let Some(url) = self.generate_url(doc, &id) {
660            attrs.insert("href".to_string(), url);
661          }
662          // Perl: use 'number' field for numeric citations (bare number without brackets).
663          // The 'refnum' field includes brackets like "[13]", causing double brackets [[13]].
664          let refnum = self.db.lookup(&format!("ID:{}", id))
665            .and_then(|e| e.get_value("number").or_else(|| e.get_value("refnum")))
666            .map(|v| v.to_string())
667            .unwrap_or_else(|| key.to_string());
668          refs.push(NodeData::Element {
669            tag: "ltx:ref".to_string(),
670            attributes: Some(attrs),
671            children: vec![NodeData::Text(refnum)],
672          });
673        } else {
674          self.note_missing("warn", "Entry for citation", key);
675          refs.push(NodeData::Element {
676            tag: "ltx:ref".to_string(),
677            attributes: Some(HashMap::from([
678              ("idref".to_string(), key.to_string()),
679              ("class".to_string(), "ltx_missing_citation".to_string()),
680            ])),
681            children: vec![NodeData::Text(key.to_string())],
682          });
683        }
684      }
685      if !refs.is_empty() {
686        doc.replace_node(bibref, &refs);
687      }
688    }
689  }
690
691  fn fill_in_mathlinks(&mut self, doc: &PostDocument) {
692    for sym in &doc.findnodes("descendant::*[@decl_id or @meaning]") {
693      let tag = doc.get_qname(sym).unwrap_or_default();
694      if tag == "ltx:XMRef" || sym.get_attribute("href").is_some() {
695        continue;
696      }
697      let entry_key = sym.get_attribute("decl_id")
698        .map(|did| format!("DECLARATION:local:{}", did))
699        .or_else(|| sym.get_attribute("meaning").map(|m| format!("DECLARATION:global:{}", m)));
700      let parent_id = entry_key.as_ref()
701        .and_then(|ek| self.db.lookup(ek))
702        .and_then(|entry| entry.get_string("parent").map(String::from));
703      if let Some(pid) = parent_id {
704        if let Some(url) = self.generate_url(doc, &pid) {
705          let mut sym_mut = sym.clone();
706          sym_mut.set_attribute("href", &url).ok();
707        }
708      }
709    }
710  }
711
712  fn report_missing(&self) {
713    for (severity, types) in &self.missing {
714      for (ref_type, items) in types {
715        let keys: Vec<&String> = items.keys().collect();
716        let msg = format!("Missing {}: {}", ref_type, keys.iter().map(|s| s.as_str()).collect::<Vec<_>>().join(","));
717        match severity.as_str() {
718          "error" => log::error!("{}", msg),
719          "warn" => log::warn!("{}", msg),
720          _ => log::info!("{}", msg),
721        }
722      }
723    }
724  }
725}

727impl Processor for CrossRef {
728  fn get_name(&self) -> &str {
729    &self.name
730  }
731
732  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
733    match doc.get_document_element() {
734      Some(el) => vec![el],
735      None => vec![],
736    }
737  }
738
739  fn process(&mut self, mut doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult {
740    self.missing.clear();
741
742    // Generate document title and add navigation
743    let doc_title = self.generate_document_title(&doc);
744    let navtoc = self.navigation_toc.clone();
745
746    if (navtoc.is_some() || doc_title.is_some())
747      && doc.findnode("//ltx:navigation").is_none() {
748        if let Some(mut root) = doc.get_document_element() {
749          doc.add_nodes(&mut root, &[NodeData::Element {
750            tag: "ltx:navigation".to_string(),
751            attributes: None,
752            children: vec![],
753          }]);
754        }
755      }
756    if let Some(ref format) = navtoc {
757      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
758        doc.add_nodes(&mut nav, &[NodeData::Element {
759          tag: "ltx:TOC".to_string(),
760          attributes: Some(HashMap::from([("format".to_string(), format.clone())])),
761          children: vec![],
762        }]);
763      }
764    }
765    if let Some(ref title) = doc_title {
766      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
767        doc.add_nodes(&mut nav, &[NodeData::Element {
768          tag: "ltx:title".to_string(),
769          attributes: None,
770          children: vec![NodeData::Text(title.clone())],
771        }]);
772      }
773    }
774
775    self.fill_in_relations(&mut doc);
776    self.fill_in_tocs(&mut doc);
777    self.fill_in_frags(&doc);
778    self.fill_in_glossaryrefs(&mut doc);
779    self.fill_in_refs(&mut doc);
780    self.fill_in_bibrefs(&mut doc);
781    self.fill_in_mathlinks(&doc);
782    self.copy_resources(&doc);
783    self.report_missing();
784    Ok(vec![doc])
785  }
786}

791fn relative_url(target: &str, base: &str) -> String {
792  if target == base {
793    return ".".to_string();
794  }
795  let target_parts: Vec<&str> = target.split('/').collect();
796  let base_parts: Vec<&str> = base.split('/').collect();
797  let common = target_parts.iter().zip(base_parts.iter())
798    .take_while(|(a, b)| a == b)
799    .count();
800  let mut result = String::new();
801  for _ in common..base_parts.len().saturating_sub(1) {
802    result.push_str("../");
803  }
804  result.push_str(&target_parts[common..].join("/"));
805  if result.is_empty() { ".".to_string() } else { result }
806}

808/// Get text content from an XML node, normalizing whitespace.
809///
810/// Port of `getTextContent`.
811fn get_text_content_node(node: &Node) -> String {
812  let text = node.get_content();
813  let trimmed = text.trim();
814  // Normalize whitespace
815  trimmed.split_whitespace().collect::<Vec<_>>().join(" ")
816}

818fn text_content(nodes: &[NodeData]) -> String {
819  nodes.iter().map(|n| match n {
820    NodeData::Text(s) => s.clone(),
821    NodeData::Element { children, .. } => text_content(children),
822    NodeData::XmlNode(n) => n.get_content(),
823  }).collect::<Vec<_>>().join("")
824}
