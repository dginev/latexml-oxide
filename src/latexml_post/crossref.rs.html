1//! Cross-reference resolution processor.
2//!
3//! Port of `LaTeXML::Post::CrossRef` (946 lines of Perl).
4//! Resolves cross-references (`ltx:ref`, `ltx:bibref`, etc.) by looking up
5//! referenced IDs in the ObjectDB and filling in the reference text,
6//! titles, and navigation links.

15/// Sectional element types that appear in TOCs.
16const NORMAL_TOC_TYPES: &[&str] = &[
17  "ltx:document",
18  "ltx:part",
19  "ltx:chapter",
20  "ltx:section",
21  "ltx:subsection",
22  "ltx:subsubsection",
23  "ltx:paragraph",
24  "ltx:subparagraph",
25  "ltx:index",
26  "ltx:bibliography",
27  "ltx:glossary",
28  "ltx:appendix",
29];

31/// Fallback fields when a requested ref show key is not found.
32fn ref_fallbacks(key: &str) -> &'static [&'static str] {
33  match key {
34    "typerefnum" => &["refnum"],
35    "toctitle" => &["title", "toccaption"],
36    "title" => &["toccaption"],
37    "rawtoctitle" => &["toctitle", "title", "toccaption"],
38    "rawtitle" => &["title", "toccaption"],
39    _ => &[],
40  }
41}

43/// URL style for cross-references.
44#[derive(Debug, Clone)]
45pub enum UrlStyle {
46  /// Use file.html#fragment
47  File,
48  /// Use server-side paths (strip trailing index.ext)
49  Server,
50  /// Negotiated: strip file extension and trailing index
51  Negotiated,
52}

54/// CrossRef post-processor.
55///
56/// Port of `LaTeXML::Post::CrossRef`.
57pub struct CrossRef {
58  name:           String,
59  /// Reference to the shared ObjectDB.
60  pub db:         ObjectDB,
61  /// URL style for cross-references.
62  url_style:      UrlStyle,
63  /// File extension used for output (e.g. "html", "xml").
64  extension:      String,
65  /// Default show format for TOC refs.
66  toc_show:       String,
67  /// Default show format for regular refs.
68  ref_show:       String,
69  /// Minimum useful content length for refs.
70  min_ref_length: usize,
71  /// Join string between parent+child ref text.
72  ref_join:       String,
73  /// Type of navigation TOC to add (e.g. "context").
74  navigation_toc: Option<String>,
75  /// Track missing references for reporting.
76  missing:        HashMap<String, HashMap<String, HashMap<String, u32>>>,
77}

109    self
110      .missing
111      .entry(severity.to_string())
112      .or_default()
113      .entry(ref_type.to_string())

120  /// Generate a URL for a referenced ID.
121  ///
122  /// Port of `CrossRef::generateURL`.
123  fn generate_url(&mut self, doc: &PostDocument, id: &str) -> Option<String> {
124    let entry = self.db.lookup(&format!("ID:{}", id))?;
125    let location = entry.get_string("location")?;
126
127    let doc_location = doc.site_relative_destination().unwrap_or_default();
128    let mut url = relative_url(location, &doc_location);
129
130    // Apply URL style
131    match self.url_style {
132      UrlStyle::Server => {
133        let index_suffix = format!("index.{}", self.extension);
134        if url.ends_with(&index_suffix) {
135          let prefix = &url[..url.len() - index_suffix.len()];
136          url = if prefix.is_empty() {
137            "./".to_string()
138          } else {
139            prefix.to_string()
140          };
141        }
142      },
143      UrlStyle::Negotiated => {
144        let ext_suffix = format!(".{}", self.extension);
145        if url.ends_with(&ext_suffix) {
146          url = url[..url.len() - ext_suffix.len()].to_string();
147        }
148        if url.ends_with("/index") {
149          url = url[..url.len() - 5].to_string();
150        }
151      },
152      UrlStyle::File => {},
153    }
154
155    if url.is_empty() {
156      url = ".".to_string();
157    }
158
159    // Add fragment ID
160    let fragid = entry.get_string("fragid").map(String::from);
161    let loc = location.to_string();
162    if let Some(fid) = fragid {
163      if url == "." || loc == doc_location {
164        url = String::new();
165      }
166      url = format!("{}#{}", url, fid);
167    } else if loc == doc_location {
168      url = String::new();
169    }
170
171    Some(url)
172  }

174  /// Generate a title string for a referenced ID, traversing parents for context.
175  ///
176  /// Port of `CrossRef::generateTitle`.
177  fn generate_title(&self, _doc: &PostDocument, id: &str, shown: &str) -> Option<String> {
178    let mut current_id = id.to_string();
179    let mut result = String::new();
180    let mut prefix = String::new();
181    let mut shown_so_far = shown.to_string();
182
183    while let Some(entry) = self.db.lookup(&format!("ID:{}", current_id)) {
184      let mut pieces = Vec::new();
185      let mut is_dup = false;
186
187      // Try title, then typerefnum, then refnum
188      if let Some(title_val) = entry.get_value("title") {
189        if title_val.is_truthy() {
190          is_dup = shown_so_far.contains("title");
191          pieces.push(title_val.to_string());
192        }
193      }
194      if pieces.is_empty() {
195        let has_type = entry
196          .get_value("tag:creftypecap")
197          .or_else(|| entry.get_value("tag:creftype"));
198        let has_refnum = entry.get_value("refnum");
199        if has_type.is_some() && has_refnum.is_some() {
200          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
201          if let Some(t) = has_type {
202            pieces.push(t.to_string());
203          }
204          if let Some(r) = has_refnum {
205            pieces.push(r.to_string());
206          }
207        } else if let Some(tr) = entry.get_value("typerefnum") {
208          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
209          pieces.push(tr.to_string());
210        } else if let Some(r) = has_refnum {
211          is_dup = shown_so_far.contains("refnum");
212          pieces.push(r.to_string());
213        }
214      }
215
216      if is_dup {
217        prefix = "In ".to_string();
218        shown_so_far.clear();
219      } else {
220        let title = pieces.join(" ");
221        let title = title.trim();
222        if !title.is_empty() {
223          result.push_str(&prefix);
224          prefix = self.ref_join.clone();
225          result.push_str(title);
226        }
227      }
228
229      // Walk to parent for more context
230      match entry.get_string("parent").map(String::from) {
231        Some(pid) => current_id = pid,
232        None => break,
233      }
234    }
235
236    if result.is_empty() {
237      None
238    } else {
239      Some(result)
240    }
241  }

243  /// Generate a title for the document itself.
244  ///
245  /// Port of `CrossRef::generateDocumentTitle`.
246  fn generate_document_title(&self, doc: &PostDocument) -> Option<String> {
247    // Try to generate from the document's root ID
248    if let Some(docid) = doc
249      .get_document_element()
250      .and_then(|r| r.get_attribute("xml:id"))
251    {
252      let title = self.generate_title(doc, &docid, "toctitle");
253      if title.as_ref().map(|t| !t.is_empty()).unwrap_or(false) {
254        return title;
255      }
256    }
257    // Fallback: look for a title element in the document
258    if let Some(node) =
259      doc.findnode("//ltx:title | //ltx:toctitle | //ltx:caption | //ltx:toccaption")
260    {
261      let text = get_text_content_node(&node);
262      if !text.is_empty() {
263        return Some(text);
264      }
265    }
266    None
267  }

269  /// Generate content for a glossary reference.
270  ///
271  /// Port of `CrossRef::generateGlossaryRefTitle`.
272  fn generate_glossary_ref_title(&self, entry_key: &str, show: &str) -> Vec<NodeData> {
273    let entry = match self.db.lookup(entry_key) {
274      Some(e) => e,
275      None => return vec![],
276    };
277
278    let phrase_key = format!("phrase:{}", show);
279    if let Some(val) = entry.get_value(&phrase_key) {
280      return vec![NodeData::Element {
281        tag:        "ltx:text".to_string(),
282        attributes: Some(HashMap::from([(
283          "class".to_string(),
284          format!("ltx_glossary_{}", show),
285        )])),
286        children:   vec![NodeData::Text(val.to_string())],
287      }];
288    }
289
290    // Handle -plural and -indefinite suffixes
291    if let Some(base_show) = show.strip_suffix("-plural") {
292      let base_key = format!("phrase:{}", base_show);
293      if let Some(val) = entry.get_value(&base_key) {
294        return vec![NodeData::Element {
295          tag:        "ltx:text".to_string(),
296          attributes: Some(HashMap::from([(
297            "class".to_string(),
298            format!("ltx_glossary_{}", show),
299          )])),
300          children:   vec![NodeData::Text(format!("{}s", val))],
301        }];
302      }
303    }
304    if let Some(base_show) = show.strip_suffix("-indefinite") {
305      let base_key = format!("phrase:{}", base_show);
306      if let Some(val) = entry.get_value(&base_key) {
307        let text = val.to_string();
308        let article = if text.starts_with(|c: char| "aeiouAEIOU".contains(c)) {
309          "an "
310        } else {
311          "a "
312        };
313        return vec![NodeData::Element {
314          tag:        "ltx:text".to_string(),
315          attributes: Some(HashMap::from([(
316            "class".to_string(),
317            format!("ltx_glossary_{}", show),
318          )])),
319          children:   vec![NodeData::Text(article.to_string()), NodeData::Text(text)],
320        }];
321      }
322    }
323
324    vec![]
325  }

343  /// Generate reference content for a given ID and show pattern.
344  ///
345  /// Port of `CrossRef::generateRef`.
346  fn generate_ref(&mut self, _doc: &PostDocument, req_id: &str, req_show: &str) -> Vec<NodeData> {
347    let show_options = if !req_show.contains("title") {
348      vec![req_show.to_string(), "title".to_string()]
349    } else {
350      vec![req_show.to_string(), "refnum".to_string()]
351    };
352
353    for show in &show_options {
354      let mut stuff = Vec::new();
355      let mut id = req_id.to_string();
356      let mut pending = String::new();
357      loop {
358        let entry_exists = self.db.lookup(&format!("ID:{}", id)).is_some();
359        if !entry_exists {
360          break;
361        }
362        let s = self.generate_ref_aux(&id, show);
363        if !s.is_empty() {
364          if !pending.is_empty() {
365            stuff.push(NodeData::Text(pending.clone()));
366          }
367          stuff.extend(s);
368          if self.check_ref_content(&stuff) {
369            return stuff;
370          }
371          pending = self.ref_join.clone();
372        }
373        let parent = self
374          .db
375          .lookup(&format!("ID:{}", id))
376          .and_then(|e| e.get_string("parent").map(String::from));
377        match parent {
378          Some(pid) => id = pid,
379          None => break,
380        }
381      }
382      if !stuff.is_empty() {
383        return stuff;
384      }
385    }
386
387    self.note_missing("info", "Usable title for ID", req_id);
388    vec![NodeData::Text(req_id.to_string())]
389  }

391  /// Generate ref content from a single DB entry.
392  fn generate_ref_aux(&self, id: &str, show: &str) -> Vec<NodeData> {
393    let entry = match self.db.lookup(&format!("ID:{}", id)) {
394      Some(e) => e,
395      None => return vec![],
396    };
397
398    let mut stuff = Vec::new();
399    let mut ok = false;
400    let mut remaining = show.to_string();
401
402    while !remaining.is_empty() {
403      if remaining.starts_with(|c: char| c.is_alphanumeric()) {
404        let keyword: String = remaining
405          .chars()
406          .take_while(|c| c.is_alphanumeric())
407          .collect();
408        remaining = remaining[keyword.len()..].to_string();
409        let key = keyword.to_lowercase();
410        let class = if key.contains("title") {
411          "ltx_ref_title"
412        } else {
413          "ltx_ref_tag"
414        };
415
416        let mut keys_to_try = vec![key.clone(), format!("tag:{}", key)];
417        keys_to_try.extend(ref_fallbacks(&key).iter().map(|s| s.to_string()));
418
419        for k in &keys_to_try {
420          if let Some(val) = entry.get_value(k) {
421            if val.is_truthy() {
422              ok = true;
423              let text = val.to_string();
424              stuff.push(NodeData::Element {
425                tag:        "ltx:text".to_string(),
426                attributes: Some(HashMap::from([("class".to_string(), class.to_string())])),
427                children:   vec![NodeData::Text(text)],
428              });
429              break;
430            }
431          }
432        }
433      } else if remaining.starts_with('{') {
434        if let Some(end) = remaining[1..].find('}') {
435          let literal = &remaining[1..1 + end];
436          if !literal.is_empty() {
437            stuff.push(NodeData::Text(literal.to_string()));
438          }
439          remaining = remaining[2 + end..].to_string();
440        } else {
441          remaining.clear();
442        }
443      } else if remaining.starts_with('~') {
444        remaining = remaining[1..].to_string();
445        if !stuff.is_empty() {
446          stuff.push(NodeData::Text("\u{00A0}".to_string()));
447        }
448      } else if remaining.starts_with(|c: char| c.is_whitespace()) {
449        let ws: String = remaining
450          .chars()
451          .take_while(|c| c.is_whitespace())
452          .collect();
453        remaining = remaining[ws.len()..].to_string();
454        if !stuff.is_empty() {
455          stuff.push(NodeData::Text(ws));
456        }
457      } else {
458        let sym: String = remaining
459          .chars()
460          .take_while(|c| !c.is_alphanumeric() && *c != '{' && *c != '~')
461          .collect();
462        remaining = remaining[sym.len()..].to_string();
463        stuff.push(NodeData::Text(sym));
464      }
465    }
466
467    if ok { stuff } else { vec![] }
468  }

470  /// Check if ref content is "good enough".
471  fn check_ref_content(&self, stuff: &[NodeData]) -> bool {
472    let text = text_content(stuff);
473    let cleaned = text.replace("in ", "");
474    cleaned.chars().any(|c| c.is_alphanumeric())
475  }

480  fn fill_in_relations(&mut self, doc: &mut PostDocument) {
481    let root_id = match doc
482      .get_document_element()
483      .and_then(|r| r.get_attribute("xml:id"))
484    {
485      Some(id) => id,
486      None => return,
487    };
488
489    let mut current_id = root_id;
490    let mut rel = "up".to_string();
491    loop {
492      let parent_id = self
493        .db
494        .lookup(&format!("ID:{}", current_id))
495        .and_then(|e| e.get_string("parent").map(String::from));
496      match parent_id {
497        Some(pid) => {
498          let has_title = self
499            .db
500            .lookup(&format!("ID:{}", pid))
501            .and_then(|e| e.get_value("title"))
502            .map(|v| v.is_truthy())
503            .unwrap_or(false);
504          if has_title {
505            doc.add_navigation(&rel, &pid);
506            rel = format!("{} up", rel);
507          }
508          current_id = pid;
509        },
510        None => break,
511      }
512    }
513  }

515  fn fill_in_tocs(&mut self, doc: &mut PostDocument) {
516    let tocs = doc.findnodes("descendant::ltx:TOC[not(ltx:toclist)]");
517    for toc in &tocs {
518      let id = doc
519        .get_document_element()
520        .and_then(|r| r.get_attribute("xml:id"))
521        .unwrap_or_default();
522      let show = toc
523        .get_attribute("show")
524        .unwrap_or_else(|| self.toc_show.clone());
525
526      let list = self.gen_toc(&id, &show);
527      if !list.is_empty() {
528        let toclist = NodeData::Element {
529          tag:        "ltx:toclist".to_string(),
530          attributes: None,
531          children:   list,
532        };
533        let mut toc_mut = toc.clone();
534        doc.add_nodes(&mut toc_mut, &[toclist]);
535      }
536    }
537  }

539  fn gen_toc(&self, id: &str, show: &str) -> Vec<NodeData> {
540    let entry = match self.db.lookup(&format!("ID:{}", id)) {
541      Some(e) => e,
542      None => return vec![],
543    };
544
545    let children = entry.get_children();
546    let kids: Vec<NodeData> = children
547      .iter()
548      .flat_map(|child_id| self.gen_toc(child_id, show))
549      .collect();
550
551    let entry_type = entry.get_string("type").unwrap_or("");
552    let is_toc_type = NORMAL_TOC_TYPES.contains(&entry_type);
553    let in_toc = entry
554      .get_value("inlist")
555      .map(|v| match v {
556        Value::Hash(h) => h.contains_key("toc"),
557        _ => false,
558      })
559      .unwrap_or(false);
560
561    if is_toc_type && in_toc {
562      let type_name = entry_type.strip_prefix("ltx:").unwrap_or(entry_type);
563      let mut toc_children = vec![NodeData::Element {
564        tag:        "ltx:ref".to_string(),
565        attributes: Some(HashMap::from([
566          ("show".to_string(), show.to_string()),
567          ("idref".to_string(), id.to_string()),
568        ])),
569        children:   vec![],
570      }];
571      if !kids.is_empty() {
572        toc_children.push(NodeData::Element {
573          tag:        "ltx:toclist".to_string(),
574          attributes: Some(HashMap::from([(
575            "class".to_string(),
576            format!("ltx_toclist_{}", type_name),
577          )])),
578          children:   kids,
579        });
580      }
581      vec![NodeData::Element {
582        tag:        "ltx:tocentry".to_string(),
583        attributes: Some(HashMap::from([(
584          "class".to_string(),
585          format!("ltx_tocentry_{}", type_name),
586        )])),
587        children:   toc_children,
588      }]
589    } else {
590      kids
591    }
592  }

594  fn fill_in_frags(&self, doc: &PostDocument) {
595    // Invert loop: iterate DB entries (~1K on arXiv:0705.0790) and look up
596    // each node via the idcache, instead of iterating every xml:id-bearing
597    // node in the DOM (60K+ on math-heavy papers, ~98% of which map to
598    // XM* descendants with no DB entry). This avoids the XPath `//*[@xml:id]`
599    // result-set clone and 60K hashmap misses per paper.
600    //
601    // Correctness: fragids are only assigned to nodes that have a DB entry;
602    // the old DOM-iteration variant early-exited on `lookup` miss for every
603    // non-DB node, so the outputs match exactly.
604    for key in self.db.get_keys() {
605      let id = match key.strip_prefix("ID:") {
606        Some(rest) => rest,
607        None => continue,
608      };
609      let entry = match self.db.lookup(key) {
610        Some(e) => e,
611        None => continue,
612      };
613      let fragid = match entry.get_string("fragid") {
614        Some(f) => f,
615        None => continue,
616      };
617      if let Some(node) = doc.find_node_by_id(id) {
618        let mut n = node.clone();
619        n.set_attribute("fragid", fragid).ok();
620      }
621    }
622  }

624  fn fill_in_refs(&mut self, doc: &mut PostDocument) {
625    let mut refs = doc.findnodes("//*[@idref]");
626    refs.extend(doc.findnodes("//*[@labelref]"));
627    for ref_node in &refs {
628      let tag = doc.get_qname(ref_node).unwrap_or_default();
629      if tag == "ltx:XMRef" {
630        continue;
631      }
632
633      let mut ref_mut = ref_node.clone();
634      let mut id = ref_node.get_attribute("idref");
635      let show = ref_node
636        .get_attribute("show")
637        .unwrap_or_else(|| self.ref_show.clone());
638
639      if id.is_none() {
640        if let Some(label) = ref_node.get_attribute("labelref") {
641          if let Some(entry) = self.db.lookup(&label) {
642            if let Some(resolved_id) = entry.get_string("id") {
643              ref_mut.set_attribute("idref", resolved_id).ok();
644              id = Some(resolved_id.to_string());
645            }
646          }
647          if id.is_none() {
648            self.note_missing("warn", "Target for Label", &label);
649            PostDocument::add_class(&mut ref_mut, "ltx_missing_label");
650          }
651        }
652      }
653
654      if let Some(ref id_str) = id {
655        if ref_mut.get_attribute("href").is_none() {
656          if let Some(url) = self.generate_url(doc, id_str) {
657            ref_mut.set_attribute("href", &url).ok();
658          }
659        }
660        if ref_mut.get_attribute("title").is_none() {
661          if let Some(titlestring) = self.generate_title(doc, id_str, &show) {
662            ref_mut.set_attribute("title", &titlestring).ok();
663          }
664        }
665        if ref_mut.get_first_child().is_none() && tag != "ltx:graphics" && tag != "ltx:picture" {
666          let content = self.generate_ref(doc, id_str, &show);
667          doc.add_nodes(&mut ref_mut, &content);
668        }
669      }
670    }
671  }
672
673  fn fill_in_glossaryrefs(&mut self, doc: &mut PostDocument) {
674    for ref_node in &doc.findnodes("descendant::ltx:glossaryref") {
675      let mut ref_mut = ref_node.clone();
676      let key = ref_node.get_attribute("key").unwrap_or_default();
677      let list = ref_node.get_attribute("inlist").unwrap_or_default();
678
679      let gkey = format!("GLOSSARY:{}:{}", list, key);
680      if let Some(entry) = self.db.lookup(&gkey) {
681        if let Some(id) = entry.get_string("id") {
682          ref_mut.set_attribute("idref", id).ok();
683        }
684      } else {
685        self.note_missing("warn", "Glossary Entry for key", &key);
686      }
687
688      if ref_mut.get_first_child().is_none() {
689        doc.add_nodes(&mut ref_mut, &[NodeData::Text(key.clone())]);
690        PostDocument::add_class(&mut ref_mut, "ltx_missing");
691      }
692    }
693  }
694
695  fn fill_in_bibrefs(&mut self, doc: &mut PostDocument) {
696    let bibrefs = doc.findnodes("//ltx:bibref");
697    for bibref in &bibrefs {
698      let keys_str = bibref.get_attribute("bibrefs").unwrap_or_default();
699      let _show = bibref
700        .get_attribute("show")
701        .unwrap_or_else(|| "refnum".to_string());
702      let sep = bibref
703        .get_attribute("separator")
704        .unwrap_or_else(|| ",".to_string());
705      let lists_str = bibref
706        .get_attribute("inlist")
707        .unwrap_or_else(|| "bibliography".to_string());
708      let lists: Vec<&str> = lists_str.split_whitespace().collect();
709
710      let mut refs: Vec<NodeData> = Vec::new();
711      for key in keys_str.split(',').filter(|k| !k.is_empty()) {
712        let mut found_id = None;
713        for list in &lists {
714          let bkey = format!("BIBLABEL:{}:{}", list, key);
715          if let Some(bentry) = self.db.lookup(&bkey) {
716            found_id = bentry.get_string("id").map(String::from);
717            if found_id.is_some() {
718              break;
719            }
720          }
721        }
722        if !refs.is_empty() {
723          refs.push(NodeData::Text(format!("{} ", sep)));
724        }
725        if let Some(id) = found_id {
726          let mut attrs = HashMap::new();
727          attrs.insert("idref".to_string(), id.clone());
728          if let Some(url) = self.generate_url(doc, &id) {
729            attrs.insert("href".to_string(), url);
730          }
731          // Perl: use 'number' field for numeric citations (bare number without brackets).
732          // The 'refnum' field includes brackets like "[13]", causing double brackets [[13]].
733          let refnum = self
734            .db
735            .lookup(&format!("ID:{}", id))
736            .and_then(|e| e.get_value("number").or_else(|| e.get_value("refnum")))
737            .map(|v| v.to_string())
738            .unwrap_or_else(|| key.to_string());
739          refs.push(NodeData::Element {
740            tag:        "ltx:ref".to_string(),
741            attributes: Some(attrs),
742            children:   vec![NodeData::Text(refnum)],
743          });
744        } else {
745          self.note_missing("warn", "Entry for citation", key);
746          refs.push(NodeData::Element {
747            tag:        "ltx:ref".to_string(),
748            attributes: Some(HashMap::from([
749              ("idref".to_string(), key.to_string()),
750              ("class".to_string(), "ltx_missing_citation".to_string()),
751            ])),
752            children:   vec![NodeData::Text(key.to_string())],
753          });
754        }
755      }
756      if !refs.is_empty() {
757        doc.replace_node(bibref, &refs);
758      }
759    }
760  }
761
762  fn fill_in_mathlinks(&mut self, doc: &PostDocument) {
763    for sym in &doc.findnodes("descendant::*[@decl_id or @meaning]") {
764      let tag = doc.get_qname(sym).unwrap_or_default();
765      if tag == "ltx:XMRef" || sym.get_attribute("href").is_some() {
766        continue;
767      }
768      let entry_key = sym
769        .get_attribute("decl_id")
770        .map(|did| format!("DECLARATION:local:{}", did))
771        .or_else(|| {
772          sym
773            .get_attribute("meaning")
774            .map(|m| format!("DECLARATION:global:{}", m))
775        });
776      let parent_id = entry_key
777        .as_ref()
778        .and_then(|ek| self.db.lookup(ek))
779        .and_then(|entry| entry.get_string("parent").map(String::from));
780      if let Some(pid) = parent_id {
781        if let Some(url) = self.generate_url(doc, &pid) {
782          let mut sym_mut = sym.clone();
783          sym_mut.set_attribute("href", &url).ok();
784        }
785      }
786    }
787  }
788
789  fn report_missing(&self) {
790    for (severity, types) in &self.missing {
791      for (ref_type, items) in types {
792        let keys: Vec<&String> = items.keys().collect();
793        let msg = format!(
794          "Missing {}: {}",
795          ref_type,
796          keys
797            .iter()
798            .map(|s| s.as_str())
799            .collect::<Vec<_>>()
800            .join(",")
801        );
802        match severity.as_str() {
803          "error" => log::error!("{}", msg),
804          "warn" => log::warn!("{}", msg),
805          _ => log::info!("{}", msg),
806        }
807      }
808    }
809  }
810}

812impl Processor for CrossRef {
813  fn get_name(&self) -> &str { &self.name }
814
815  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
816    match doc.get_document_element() {
817      Some(el) => vec![el],
818      None => vec![],
819    }
820  }
821
822  fn process(&mut self, mut doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult {
823    self.missing.clear();
824
825    // Generate document title and add navigation
826    let doc_title = self.generate_document_title(&doc);
827    let navtoc = self.navigation_toc.clone();
828
829    if (navtoc.is_some() || doc_title.is_some()) && doc.findnode("//ltx:navigation").is_none() {
830      if let Some(mut root) = doc.get_document_element() {
831        doc.add_nodes(&mut root, &[NodeData::Element {
832          tag:        "ltx:navigation".to_string(),
833          attributes: None,
834          children:   vec![],
835        }]);
836      }
837    }
838    if let Some(ref format) = navtoc {
839      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
840        doc.add_nodes(&mut nav, &[NodeData::Element {
841          tag:        "ltx:TOC".to_string(),
842          attributes: Some(HashMap::from([("format".to_string(), format.clone())])),
843          children:   vec![],
844        }]);
845      }
846    }
847    if let Some(ref title) = doc_title {
848      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
849        doc.add_nodes(&mut nav, &[NodeData::Element {
850          tag:        "ltx:title".to_string(),
851          attributes: None,
852          children:   vec![NodeData::Text(title.clone())],
853        }]);
854      }
855    }
856
857    self.fill_in_relations(&mut doc);
858    self.fill_in_tocs(&mut doc);
859    self.fill_in_frags(&doc);
860    self.fill_in_glossaryrefs(&mut doc);
861    self.fill_in_refs(&mut doc);
862    self.fill_in_bibrefs(&mut doc);
863    self.fill_in_mathlinks(&doc);
864    self.copy_resources(&doc);
865    self.report_missing();
866    Ok(vec![doc])
867  }
868}

873fn relative_url(target: &str, base: &str) -> String {
874  if target == base {
875    return ".".to_string();
876  }
877  let target_parts: Vec<&str> = target.split('/').collect();
878  let base_parts: Vec<&str> = base.split('/').collect();
879  let common = target_parts
880    .iter()
881    .zip(base_parts.iter())
882    .take_while(|(a, b)| a == b)
883    .count();
884  let mut result = String::new();
885  for _ in common..base_parts.len().saturating_sub(1) {
886    result.push_str("../");
887  }
888  result.push_str(&target_parts[common..].join("/"));
889  if result.is_empty() {
890    ".".to_string()
891  } else {
892    result
893  }
894}

896/// Get text content from an XML node, normalizing whitespace.
897///
898/// Port of `getTextContent`.
899fn get_text_content_node(node: &Node) -> String {
900  let text = node.get_content();
901  let trimmed = text.trim();
902  // Normalize whitespace
903  trimmed.split_whitespace().collect::<Vec<_>>().join(" ")
904}
905
906fn text_content(nodes: &[NodeData]) -> String {
907  nodes
908    .iter()
909    .map(|n| match n {
910      NodeData::Text(s) => s.clone(),
911      NodeData::Element { children, .. } => text_content(children),
912      NodeData::XmlNode(n) => n.get_content(),
913    })
914    .collect::<Vec<_>>()
915    .join("")
916}
