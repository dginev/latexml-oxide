3//! Port of `LaTeXML::Post::CrossRef` (946 lines of Perl).
4//! Resolves cross-references (`ltx:ref`, `ltx:bibref`, etc.) by looking up

19/// Perl CrossRef.pm `$normaltoctypes` (L202-206): the sectional element types
20/// used by `gentoc_context`'s UPWARD ancestor/sibling enclosure. This is NOT
21/// the normal `gen_toc` path — that filters purely by the TOC's
22/// `select`/`inlist` (issue #291). Deliberately excludes
23/// `ltx:abstract`/`ltx:acknowledgements` (matching Perl exactly) so frontmatter
24/// does not clutter the navigation breadcrumb's sibling rows.
25const NORMAL_TOC_TYPES: &[&str] = &[
26  "ltx:document",
27  "ltx:part",
28  "ltx:chapter",
29  "ltx:section",
30  "ltx:subsection",
31  "ltx:subsubsection",
32  "ltx:paragraph",
33  "ltx:subparagraph",
34  "ltx:index",
35  "ltx:bibliography",
36  "ltx:glossary",
37  "ltx:appendix",
38];

40/// Memoized result of `get_child_page_ids` for one ObjectDB entry: the
41/// distinct descendant page ids, plus a position index so
42/// `find_previous_page_id`/`find_next_page_id` can locate a sibling in O(1)
43/// instead of the Perl pop/shift scan.
44struct ChildPages {
45  ids:      Vec<String>,
46  index_of: HashMap<String, usize>,
47}

49/// Fallback fields when a requested ref show key is not found.
50fn ref_fallbacks(key: &str) -> &'static [&'static str] {
51  match key {
52    "typerefnum" => &["refnum"],
53    "toctitle" => &["title", "toccaption"],
54    "title" => &["toccaption"],
55    "rawtoctitle" => &["toctitle", "title", "toccaption"],
56    "rawtitle" => &["title", "toccaption"],
57    _ => &[],
58  }
59}

61/// URL style for cross-references.
62#[derive(Debug, Clone)]
63pub enum UrlStyle {
64  /// Use file.html#fragment
65  File,
66  /// Use server-side paths (strip trailing index.ext)
67  Server,
68  /// Negotiated: strip file extension and trailing index
69  Negotiated,
70}

72/// CrossRef post-processor.
73///
74/// Port of `LaTeXML::Post::CrossRef`.
75pub struct CrossRef {
76  name:           String,
77  /// Reference to the shared ObjectDB.
78  pub db:         ObjectDB,
79  /// URL style for cross-references.
80  url_style:      UrlStyle,
81  /// File extension used for output (e.g. "html", "xml").
82  extension:      String,
83  /// Default show format for TOC refs.
84  toc_show:       String,
85  /// Default show format for regular refs.
86  ref_show:       String,
87  /// Minimum useful content length for refs.
88  min_ref_length: usize,
89  /// Join string between parent+child ref text.
90  ref_join:       String,
91  /// Type of navigation TOC to add (e.g. "context").
92  navigation_toc: Option<String>,
93  /// Track missing references for reporting.
94  missing:        HashMap<String, HashMap<String, HashMap<String, u32>>>,
95  /// Memoized `get_child_page_ids` results, keyed by entry id. The ObjectDB
96  /// is read-only for the whole CrossRef pass, so a given entry's child pages
97  /// never change — caching them across all split pages turns the per-page
98  /// O(siblings) recomputation (Perl's unpruned `getChildPages`) into O(1)
99  /// lookups, eliminating the `fill_in_relations` O(n²).
100  child_pages:    RefCell<HashMap<String, Rc<ChildPages>>>,
101}

145  /// Generate a URL for a referenced ID.
146  ///
147  /// Port of `CrossRef::generateURL`.
148  fn generate_url(&mut self, doc: &PostDocument, id: &str) -> Option<String> {
149    let entry = self.db.lookup(&format!("ID:{}", id))?;
150    let location = entry.get_string("location")?;
151
152    let doc_location = doc.site_relative_destination().unwrap_or_default();
153    let mut url = relative_url(location, &doc_location);
154
155    // Apply URL style
156    match self.url_style {
157      UrlStyle::Server => {
158        let index_suffix = format!("index.{}", self.extension);
159        if url.ends_with(&index_suffix) {
160          let prefix = &url[..url.len() - index_suffix.len()];
161          url = if prefix.is_empty() {
162            "./".to_string()
163          } else {
164            prefix.to_string()
165          };
166        }
167      },
168      UrlStyle::Negotiated => {
169        let ext_suffix = format!(".{}", self.extension);
170        if url.ends_with(&ext_suffix) {
171          url = url[..url.len() - ext_suffix.len()].to_string();
172        }
173        if url.ends_with("/index") {
174          url = url[..url.len() - 5].to_string();
175        }
176      },
177      UrlStyle::File => {},
178    }
179
180    if url.is_empty() {
181      url = ".".to_string();
182    }
183
184    // Add fragment ID
185    let fragid = entry.get_string("fragid").map(String::from);
186    let loc = location.to_string();
187    if let Some(fid) = fragid {
188      if url == "." || loc == doc_location {
189        url = String::new();
190      }
191      url = format!("{}#{}", url, fid);
192    } else if loc == doc_location {
193      url = String::new();
194    }
195
196    Some(url)
197  }

199  /// Generate a title string for a referenced ID, traversing parents for context.
200  ///
201  /// Port of `CrossRef::generateTitle`.
202  fn generate_title(&self, _doc: &PostDocument, id: &str, shown: &str) -> Option<String> {
203    let mut current_id = id.to_string();
204    let mut result = String::new();
205    let mut prefix = String::new();
206    let mut shown_so_far = shown.to_string();
207
208    while let Some(entry) = self.db.lookup(&format!("ID:{}", current_id)) {
209      let mut pieces = Vec::new();
210      let mut is_dup = false;
211
212      // Try title, then typerefnum, then refnum
213      if let Some(title_val) = entry.get_value("title") {
214        if title_val.is_truthy() {
215          is_dup = shown_so_far.contains("title");
216          pieces.push(title_val.to_string());
217        }
218      }
219      if pieces.is_empty() {
220        let has_type = entry
221          .get_value("tag:creftypecap")
222          .or_else(|| entry.get_value("tag:creftype"));
223        let has_refnum = entry.get_value("refnum");
224        if has_type.is_some() && has_refnum.is_some() {
225          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
226          if let Some(t) = has_type {
227            pieces.push(t.to_string());
228          }
229          if let Some(r) = has_refnum {
230            pieces.push(r.to_string());
231          }
232        } else if let Some(tr) = entry.get_value("typerefnum") {
233          is_dup = shown_so_far.contains("type") && shown_so_far.contains("refnum");
234          pieces.push(tr.to_string());
235        } else if let Some(r) = has_refnum {
236          is_dup = shown_so_far.contains("refnum");
237          pieces.push(r.to_string());
238        }
239      }
240
241      if is_dup {
242        prefix = "In ".to_string();
243        shown_so_far.clear();
244      } else {
245        let title = pieces.join(" ");
246        let title = title.trim();
247        if !title.is_empty() {
248          result.push_str(&prefix);
249          prefix = self.ref_join.clone();
250          result.push_str(title);
251        }
252      }
253
254      // Walk to parent for more context
255      match entry.get_string("parent").map(String::from) {
256        Some(pid) => current_id = pid,
257        None => break,
258      }
259    }
260
261    if result.is_empty() {
262      None
263    } else {
264      Some(result)
265    }
266  }

268  /// Generate a title for the document itself.
269  ///
270  /// Port of `CrossRef::generateDocumentTitle`.
271  fn generate_document_title(&self, doc: &PostDocument) -> Option<String> {
272    // Try to generate from the document's root ID. Use `get_xml_id` so we
273    // pick up ids stored in the xml namespace (Scan's default placement)
274    // as well as the bare "xml:id" attribute form.
275    if let Some(docid) = doc.get_document_element().as_ref().and_then(get_xml_id) {
276      // Perl `generateDocumentTile` (CrossRef.pm L809) calls
277      // `generateTitle($doc, $docid)` with NO `$shown` arg → `$shown=''`. Passing
278      // "toctitle" here is WRONG: `generate_title`'s dup test is `shown.contains("title")`
279      // (Perl `$shown =~ /title/`), and "toctitle" contains "title", so the page's OWN
280      // (deepest) title is falsely flagged a duplicate and dropped — every split section
281      // page's <title> collapsed to "In <parent>" instead of "<section> ‣ <ancestors>".
282      let title = self.generate_title(doc, &docid, "");
283      if title.as_ref().map(|t| !t.is_empty()).unwrap_or(false) {
284        return title;
285      }
286    }
287    // Fallback: look for a title element in the document
288    if let Some(node) =
289      doc.findnode("//ltx:title | //ltx:toctitle | //ltx:caption | //ltx:toccaption")
290    {
291      let text = get_text_content_node(&node);
292      if !text.is_empty() {
293        return Some(text);
294      }
295    }
296    None
297  }

299  /// Generate content for a glossary reference.
300  ///
301  /// Port of `CrossRef::generateGlossaryRefTitle`.
302  fn generate_glossary_ref_title(&self, entry_key: &str, show: &str) -> Vec<NodeData> {
303    let entry = match self.db.lookup(entry_key) {
304      Some(e) => e,
305      None => return vec![],
306    };
307
308    let phrase_key = format!("phrase:{}", show);
309    if let Some(val) = entry.get_value(&phrase_key) {
310      return vec![NodeData::Element {
311        tag:        "ltx:text".to_string(),
312        attributes: Some(HashMap::from_iter([(
313          "class".to_string(),
314          format!("ltx_glossary_{}", show),
315        )])),
316        children:   vec![NodeData::Text(val.to_string())],
317      }];
318    }
319
320    // Handle -plural and -indefinite suffixes
321    if let Some(base_show) = show.strip_suffix("-plural") {
322      let base_key = format!("phrase:{}", base_show);
323      if let Some(val) = entry.get_value(&base_key) {
324        return vec![NodeData::Element {
325          tag:        "ltx:text".to_string(),
326          attributes: Some(HashMap::from_iter([(
327            "class".to_string(),
328            format!("ltx_glossary_{}", show),
329          )])),
330          children:   vec![NodeData::Text(format!("{}s", val))],
331        }];
332      }
333    }
334    if let Some(base_show) = show.strip_suffix("-indefinite") {
335      let base_key = format!("phrase:{}", base_show);
336      if let Some(val) = entry.get_value(&base_key) {
337        let text = val.to_string();
338        let article = if text.starts_with(|c: char| "aeiouAEIOU".contains(c)) {
339          "an "
340        } else {
341          "a "
342        };
343        return vec![NodeData::Element {
344          tag:        "ltx:text".to_string(),
345          attributes: Some(HashMap::from_iter([(
346            "class".to_string(),
347            format!("ltx_glossary_{}", show),
348          )])),
349          children:   vec![NodeData::Text(article.to_string()), NodeData::Text(text)],
350        }];
351      }
352    }
353
354    vec![]
355  }

373  /// Generate reference content for a given ID and show pattern.
374  ///
375  /// Port of `CrossRef::generateRef`.
376  fn generate_ref(&mut self, _doc: &PostDocument, req_id: &str, req_show: &str) -> Vec<NodeData> {
377    let show_options = if !req_show.contains("title") {
378      vec![req_show.to_string(), "title".to_string()]
379    } else {
380      vec![req_show.to_string(), "refnum".to_string()]
381    };
382
383    for show in &show_options {
384      let mut stuff = Vec::new();
385      let mut id = req_id.to_string();
386      let mut pending = String::new();
387      loop {
388        let entry_exists = self.db.lookup(&format!("ID:{}", id)).is_some();
389        if !entry_exists {
390          break;
391        }
392        let s = self.generate_ref_aux(&id, show);
393        if !s.is_empty() {
394          if !pending.is_empty() {
395            stuff.push(NodeData::Text(pending.clone()));
396          }
397          stuff.extend(s);
398          if self.check_ref_content(&stuff) {
399            return stuff;
400          }
401          pending = self.ref_join.clone();
402        }
403        let parent = self
404          .db
405          .lookup(&format!("ID:{}", id))
406          .and_then(|e| e.get_string("parent").map(String::from));
407        match parent {
408          Some(pid) => id = pid,
409          None => break,
410        }
411      }
412      if !stuff.is_empty() {
413        return stuff;
414      }
415    }
416
417    self.note_missing("info", "Usable title for ID", req_id);
418    vec![NodeData::Text(req_id.to_string())]
419  }

421  /// Generate ref content from a single DB entry.
422  fn generate_ref_aux(&self, id: &str, show: &str) -> Vec<NodeData> {
423    let entry = match self.db.lookup(&format!("ID:{}", id)) {
424      Some(e) => e,
425      None => return vec![],
426    };
427
428    let mut stuff = Vec::new();
429    let mut ok = false;
430    let mut remaining = show.to_string();
431
432    while !remaining.is_empty() {
433      if remaining.starts_with(|c: char| c.is_alphanumeric()) {
434        let keyword: String = remaining
435          .chars()
436          .take_while(|c| c.is_alphanumeric())
437          .collect();
438        remaining = remaining[keyword.len()..].to_string();
439        let key = keyword.to_lowercase();
440        let class = if key.contains("title") {
441          "ltx_ref_title"
442        } else {
443          "ltx_ref_tag"
444        };
445
446        let mut keys_to_try = vec![key.clone(), format!("tag:{}", key)];
447        keys_to_try.extend(ref_fallbacks(&key).iter().map(|s| s.to_string()));
448
449        for k in &keys_to_try {
450          if let Some(val) = entry.get_value(k) {
451            if val.is_truthy() {
452              ok = true;
453              let text = val.to_string();
454              stuff.push(NodeData::Element {
455                tag:        "ltx:text".to_string(),
456                attributes: Some(HashMap::from_iter([(
457                  "class".to_string(),
458                  class.to_string(),
459                )])),
460                children:   vec![NodeData::Text(text)],
461              });
462              break;
463            }
464          }
465        }
466      } else if remaining.starts_with('{') {
467        if let Some(end) = remaining[1..].find('}') {
468          let literal = &remaining[1..1 + end];
469          if !literal.is_empty() {
470            stuff.push(NodeData::Text(literal.to_string()));
471          }
472          remaining = remaining[2 + end..].to_string();
473        } else {
474          remaining.clear();
475        }
476      } else if remaining.starts_with('~') {
477        remaining = remaining[1..].to_string();
478        if !stuff.is_empty() {
479          stuff.push(NodeData::Text("\u{00A0}".to_string()));
480        }
481      } else if remaining.starts_with(|c: char| c.is_whitespace()) {
482        let ws: String = remaining
483          .chars()
484          .take_while(|c| c.is_whitespace())
485          .collect();
486        remaining = remaining[ws.len()..].to_string();
487        if !stuff.is_empty() {
488          stuff.push(NodeData::Text(ws));
489        }
490      } else {
491        let sym: String = remaining
492          .chars()
493          .take_while(|c| !c.is_alphanumeric() && *c != '{' && *c != '~')
494          .collect();
495        remaining = remaining[sym.len()..].to_string();
496        stuff.push(NodeData::Text(sym));
497      }
498    }
499
500    if ok { stuff } else { vec![] }
501  }

503  /// Check if ref content is "good enough".
504  fn check_ref_content(&self, stuff: &[NodeData]) -> bool {
505    let text = text_content(stuff);
506    let cleaned = text.replace("in ", "");
507    cleaned.chars().any(|c| c.is_alphanumeric())
508  }

513  fn fill_in_relations(&mut self, doc: &mut PostDocument) {
514    // Same get_xml_id trick as generate_document_title: Scan stores ids
515    // in the xml namespace by default; without this, sub-docs would skip
516    // relation filling and never gain the prev/next/up navigation.
517    let page_id = match doc.get_document_element().as_ref().and_then(get_xml_id) {
518      Some(id) => id,
519      None => return,
520    };
521
522    // 1. up / "up up" / "up up up" — walk ancestors that have a title.
523    let mut current_id = page_id.clone();
524    let mut rel = "up".to_string();
525    let mut topmost = current_id.clone();
526    loop {
527      let parent_id = self
528        .db
529        .lookup(&format!("ID:{}", current_id))
530        .and_then(|e| e.get_string("parent").map(String::from));
531      match parent_id {
532        Some(pid) => {
533          let has_title = self
534            .db
535            .lookup(&format!("ID:{}", pid))
536            .and_then(|e| e.get_value("title"))
537            .map(|v| v.is_truthy())
538            .unwrap_or(false);
539          if has_title {
540            doc.add_navigation(&rel, &pid);
541            rel = format!("{} up", rel);
542          }
543          current_id = pid.clone();
544          topmost = pid;
545        },
546        None => break,
547      }
548    }
549
550    // 2. start — the topmost ancestor (root page), if different from us.
551    if topmost != page_id {
552      if let Some(top_pageid) = self
553        .db
554        .lookup(&format!("ID:{}", topmost))
555        .and_then(|e| e.get_string("pageid").map(String::from))
556      {
557        doc.add_navigation("start", &top_pageid);
558      }
559    }
560
561    // 3. prev / next — walk the page tree.
562    if let Some(prev) = self.find_previous_page_id(&page_id) {
563      doc.add_navigation("prev", &prev);
564    }
565    if let Some(next) = self.find_next_page_id(&page_id) {
566      doc.add_navigation("next", &next);
567    }
568  }

581  /// Resolve `entry_id` to the pageid of the page that *contains* its
582  /// parent. Port of Perl `CrossRef::getParentPage`.
583  fn get_parent_page_id(&self, entry_id: &str) -> Option<String> {
584    let entry = self.db.lookup(&format!("ID:{}", entry_id))?;
585    let pageid = entry.get_string("pageid")?.to_string();
586    let page_entry = self.db.lookup(&format!("ID:{}", pageid))?;
587    let parent_id = page_entry.get_string("parent")?.to_string();
588    let parent_entry = self.db.lookup(&format!("ID:{}", parent_id))?;
589    Some(parent_entry.get_string("pageid")?.to_string())
590  }

592  /// Memoized `get_child_page_ids`. The ObjectDB is immutable for the whole
593  /// CrossRef pass, so a given entry's child-page list is stable and shared
594  /// across every page (see the `child_pages` field). Also records each id's
595  /// position so the sibling finders skip the Perl pop/shift scan.
596  fn child_pages(&self, entry_id: &str) -> Rc<ChildPages> {
597    if let Some(cached) = self.child_pages.borrow().get(entry_id) {
598      return cached.clone();
599    }
600    let ids = self.compute_child_page_ids(entry_id);
601    let mut index_of = HashMap::default();
602    // Last occurrence wins, matching the Perl scan that peels from the end.
603    for (i, id) in ids.iter().enumerate() {
604      index_of.insert(id.clone(), i);
605    }
606    let rc = Rc::new(ChildPages { ids, index_of });
607    self
608      .child_pages
609      .borrow_mut()
610      .insert(entry_id.to_string(), rc.clone());
611    rc
612  }

614  /// Recursively collect distinct child page ids under `entry_id`.
615  /// Port of Perl `CrossRef::getChildPages` (the uncached recursion body;
616  /// recursion reuses the cache via [`child_pages`](Self::child_pages)).
617  fn compute_child_page_ids(&self, entry_id: &str) -> Vec<String> {
618    let entry = match self.db.lookup(&format!("ID:{}", entry_id)) {
619      Some(e) => e,
620      None => return Vec::new(),
621    };
622    let here_pageid = entry.get_string("pageid").map(String::from);
623    let children = entry.get_children();
624    let mut out = Vec::new();
625    for ch in children {
626      let ch_entry = match self.db.lookup(&format!("ID:{}", ch)) {

699  fn fill_in_tocs(&mut self, doc: &mut PostDocument) {

701    // context to documentElement. oxide's `findnodes(None)` defaults to
702    // the XML document node, where libxml2's `descendant::` axis evaluates

705    // user's `\tableofcontents` placeholder is reachable.

708      None => Vec::new(),

712      // assigns xml:id via the xml namespace, which is invisible to a
713      // bare `get_attribute("xml:id")` lookup but is found by

721      // fill_in_tocs L227-231 resolves this via `getRootPage`, walking up the
722      // *page* hierarchy (parent → its pageid → …) and taking the root page's
723      // `pageid`. Default scope (`current` or absent) keeps the current-page id,
724      // so the inline `\tableofcontents` placeholder stays page-local.
725      if toc.get_attribute("scope").as_deref() == Some("global") {

727      }

729        .get_attribute("show")

733      // by `\tableofcontents` from `tocdepth`) restricts which element types

738      let types: Option<HashSet<&str>> = select_attr.as_deref().map(|s| {

815  /// Perl CrossRef.pm gentoc L246-262. Generate the TOC for `id` and its
816  /// children. `localto` (when `Some`) restricts the downward recursion to

823    show: &str,
824    types: Option<&HashSet<&str>>,
825    lists: &HashSet<&str>,
826    localto: Option<&str>,
827    selfid: Option<&str>,
828  ) -> Vec<NodeData> {

830      Some(e) => e,
831      None => return vec![],
832    };

835    // entry lives on the target page.

839    };

871  /// Perl CrossRef.pm gentocentry L268-283. Build one `ltx:tocentry` for an
872  /// entry: the `before < show > after` split (`generateRef_simple` for the

874  /// this is the `selfid`, and a nested `ltx:toclist` of `children`.
875  fn gen_tocentry(

898      show_mid = mid;

918      children:   vec![],

925        tag: "ltx:toclist".to_string(),
926        attributes: Some(HashMap::from_iter([(
927          "class".to_string(),

957  /// Perl CrossRef.pm gentoc_context L288-311. A "context" TOC: the current
958  /// page's own contents (downward, page-local), enclosed upward within its
959  /// ancestors and their sibling sections — the navigation-bar breadcrumb.
960  fn gen_toc_context(
961    &self,
962    id: &str,
963    show: &str,
964    types: Option<&HashSet<&str>>,
965    lists: &HashSet<&str>,
966  ) -> Vec<NodeData> {

968      Some(e) => e,
969      None => return vec![],
970    };

972    // Downward TOC covering items WITHIN the current page (localto = this page's
973    // location; selfid = this id so the current entry is marked ltx_ref_self).
974    let location = start.get_string("location").unwrap_or("").to_string();

978    // of the child we ascended through; its slot in each parent's sibling row is
979    // replaced by the accumulated `navtoc` subtree.

985        Some(e) => e,
986        None => break,
987      };

990      // tocentries, except the one we came from (spliced with `navtoc`).
991      let mut row: Vec<NodeData> = Vec::new();
992      for child_id in parent.get_children() {
993        let child = match self.db.lookup(&format!("ID:{}", child_id)) {
994          Some(e) => e,
995          None => continue,
996        };
997        if !NORMAL_TOC_TYPES.contains(&child.get_string("type").unwrap_or("")) {
998          continue;
999        }
1000        let child_id_val = child.get_string("id").unwrap_or(&child_id);
1001        if child_id_val == came_from {
1002          row.append(&mut navtoc);
1003        } else {
1004          row.push(self.gen_tocentry(child, None, show, Vec::new()));
1005        }
1006      }
1007      navtoc = row;
1008
1009      // gentoc_context L304-306: wrap in the parent's own tocentry, but only if
1010      // the parent passes the type filter AND is itself nested (never wrap the
1011      // top-level document).

1018      if parent_ok && parent_has_parent {

1021
1022      came_from = pid;
1023      parent_id = parent.get_string("parent").map(String::from);
1024    }
1025
1026    navtoc
1027  }

1029  fn fill_in_frags(&self, doc: &PostDocument) {

1035    // every page's entries (tens of thousands) while each page has a handful

1167      // natbib emits show patterns like:
1168      //   "AuthorsPhrase1Year"           → \citep{X} → "Author (Year)"
1169      //   "Authors Phrase1YearPhrase2"  → \citet{X} → "Author (Year)"
1170      //   "refnum"                       → numeric / default
1171      // Anything containing "Author" or "Year" wants the author-year
1172      // text built from the bibentry's `authors`/`year` fields; the
1173      // legacy refnum-only path serves the numeric case.
1174      let want_authoryear = show.contains("Author") || show.contains("Year");
1175      let sep = bibref
1176        .get_attribute("separator")
1177        .unwrap_or_else(|| ",".to_string());
1178      // The lists to search, most-specific first. Perl CrossRef.pm L515 reads
1179      // `inlist || 'bibliography'` — an *exclusive* choice, which strands every
1180      // citation of a document that loads `bibunits`/`chapterbib` but keeps a
1181      // single main `\bibliography`: `\cite` stamps CITE_UNIT=bu0 onto the
1182      // bibref, while the bibitems register under the default `bibliography`
1183      // list, so the unit-only lookup never matches (witness 2303.06077: 93
1184      // bibitems, 93 dangling keys, 0 links).
1185      //
1186      // Perl's own Scan.pm L379-380 spells the intended chain — unit lists
1187      // PLUS the main one ("Citation specifies main 'bibliography', as well as
1188      // any specific others (eg. per chapter)") — and registers the reference
1189      // under both. We follow Scan's convention here so the two agree; the unit
1190      // list still wins, since the search breaks on the first list that yields
1191      // an id. OXIDIZED_DESIGN #59, KNOWN_PERL_ERRORS #50.
1192      let inlist = bibref.get_attribute("inlist").unwrap_or_default();
1193      let mut lists: Vec<&str> = inlist.split_whitespace().collect();
1194      if !lists.contains(&"bibliography") {
1195        lists.push("bibliography");
1196      }
1197
1198      let mut refs: Vec<NodeData> = Vec::new();
1199      for key in keys_str.split(',').filter(|k| !k.is_empty()) {
1200        let mut found_id = None;
1201        for list in &lists {
1202          let bkey = format!("BIBLABEL:{}:{}", list, key);
1203          if let Some(bentry) = self.db.lookup(&bkey) {
1204            found_id = bentry.get_string("id").map(String::from);
1205            if found_id.is_some() {
1206              break;
1207            }
1208          }
1209        }
1210        if !refs.is_empty() {
1211          refs.push(NodeData::Text(format!("{} ", sep)));
1212        }
1213        if let Some(id) = found_id {
1214          let mut attrs = HashMap::default();
1215          attrs.insert("idref".to_string(), id.clone());
1216          if let Some(url) = self.generate_url(doc, &id) {
1217            attrs.insert("href".to_string(), url);
1218          }
1219          // Build the display text: author-year when natbib's `show`
1220          // requests it AND the bibentry has the author/year metadata;
1221          // otherwise fall back to the numeric `number`/`refnum`
1222          // (matches the legacy path).
1223          // Perl: use 'number' field for numeric citations (bare number without brackets).
1224          // The 'refnum' field includes brackets like "[13]", causing double brackets [[13]].
1225          let entry = self.db.lookup(&format!("ID:{}", id));
1226          let display = if want_authoryear {
1227            let authors = entry
1228              .and_then(|e| {
1229                e.get_value("authors")
1230                  .or_else(|| e.get_value("fullauthors"))
1231                  .or_else(|| e.get_value("keytag"))
1232              })
1233              .map(|v| v.to_string())
1234              .map(|s| s.trim().to_string())
1235              .filter(|s| !s.is_empty());
1236            let year = entry
1237              .and_then(|e| e.get_value("year").or_else(|| e.get_value("typetag")))
1238              .map(|v| v.to_string())
1239              .map(|s| s.trim().to_string())
1240              .filter(|s| !s.is_empty());
1241            match (authors, year) {
1242              (Some(a), Some(y)) => {
1243                // `Phrase1`/`Phrase2` in the show string mark where
1244                // open/close paren or yyseparator usually go in Perl's
1245                // bibrefphrase markup. We collapse to a simple
1246                // "Authors Year" form: `\citep` (AuthorsPhrase1Year)
1247                // becomes "Author Year" inside the surrounding
1248                // parens emitted by the citemacro; `\citet`
1249                // (Authors Phrase1YearPhrase2) becomes "Author Year"
1250                // with the macro adding the year-parens. Good enough
1251                // for visual parity with the PDF in the common case.
1252                format!("{} {}", a, y)
1253              },
1254              (Some(a), None) => a,
1255              (None, Some(y)) => y,
1256              (None, None) => {
1257                // No author/year metadata → fall back to refnum.
1258                entry
1259                  .and_then(|e| e.get_value("number").or_else(|| e.get_value("refnum")))
1260                  .map(|v| v.to_string())
1261                  .unwrap_or_else(|| key.to_string())
1262              },
1263            }
1264          } else {
1265            entry
1266              .and_then(|e| e.get_value("number").or_else(|| e.get_value("refnum")))
1267              .map(|v| v.to_string())
1268              .unwrap_or_else(|| key.to_string())
1269          };
1270          refs.push(NodeData::Element {
1271            tag:        "ltx:ref".to_string(),
1272            attributes: Some(attrs),
1273            children:   vec![NodeData::Text(display)],
1274          });
1275        } else {
1276          self.note_missing("warn", "Entry for citation", key);
1277          refs.push(NodeData::Element {
1278            tag:        "ltx:ref".to_string(),
1279            attributes: Some(HashMap::from_iter([
1280              ("idref".to_string(), key.to_string()),
1281              ("class".to_string(), "ltx_missing_citation".to_string()),
1282            ])),
1283            children:   vec![NodeData::Text(key.to_string())],
1284          });
1285        }
1286      }
1287      if !refs.is_empty() {
1288        doc.replace_node(bibref, &refs);
1289      }
1290    }
1291  }

1293  fn fill_in_mathlinks(&mut self, doc: &PostDocument) {
1294    for sym in &doc.findnodes("descendant::*[@decl_id or @meaning]") {
1295      let tag = doc.get_qname(sym).unwrap_or_default();
1296      if tag == "ltx:XMRef" || sym.get_attribute("href").is_some() {
1297        continue;
1298      }
1299      let entry_key = sym
1300        .get_attribute("decl_id")
1301        .map(|did| format!("DECLARATION:local:{}", did))
1302        .or_else(|| {
1303          sym
1304            .get_attribute("meaning")
1305            .map(|m| format!("DECLARATION:global:{}", m))
1306        });
1307      let parent_id = entry_key
1308        .as_ref()
1309        .and_then(|ek| self.db.lookup(ek))
1310        .and_then(|entry| entry.get_string("parent").map(String::from));
1311      if let Some(pid) = parent_id {
1312        if let Some(url) = self.generate_url(doc, &pid) {
1313          let mut sym_mut = sym.clone();
1314          sym_mut.set_attribute("href", &url).ok();
1315        }
1316      }
1317    }
1318  }
1319
1320  fn report_missing(&self) {
1321    for (severity, types) in &self.missing {
1322      for (ref_type, items) in types {
1323        let keys: Vec<&String> = items.keys().collect();
1324        let msg = format!(
1325          "Missing {}: {}",
1326          ref_type,
1327          keys
1328            .iter()
1329            .map(|s| s.as_str())
1330            .collect::<Vec<_>>()
1331            .join(",")
1332        );
1333        // Perl CrossRef.pm L72-75: structured Error/Warn/Info with
1334        // class='expected', object='ids'. Use harness-friendly target.
1335        match severity.as_str() {
1336          "error" => Error!("expected", "ids", "{}", msg),
1337          "warn" => Warn!("expected", "ids", "{}", msg),
1338          _ => Info!("expected", "ids", "{}", msg),
1339        }
1340      }
1341    }
1342  }

1345impl Processor for CrossRef {
1346  fn get_name(&self) -> &str { &self.name }
1347
1348  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
1349    match doc.get_document_element() {
1350      Some(el) => vec![el],
1351      None => vec![],
1352    }
1353  }
1354
1355  fn process(&mut self, mut doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult {
1356    self.missing.clear();
1357
1358    // Generate document title and add navigation
1359    let doc_title = self.generate_document_title(&doc);
1360    let navtoc = self.navigation_toc.clone();
1361
1362    if (navtoc.is_some() || doc_title.is_some()) && doc.findnode("//ltx:navigation").is_none() {
1363      if let Some(mut root) = doc.get_document_element() {
1364        doc.add_nodes(&mut root, &[NodeData::Element {
1365          tag:        "ltx:navigation".to_string(),
1366          attributes: None,
1367          children:   vec![],
1368        }]);
1369      }
1370    }
1371    if let Some(ref format) = navtoc {
1372      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
1373        // Perl CrossRef.pm L50: `['ltx:TOC', {format => $navtoc}]` — format
1374        // ONLY, no `scope`. `fill_in_tocs` then defaults scope to `current`, so
1375        // this TOC is built relative to THIS page's document element. On a split
1376        // document that yields a per-page navigation breadcrumb via
1377        // `gen_toc_context` (each page's own contents enclosed within its
1378        // ancestors + their siblings), rather than one identical global tree.
1379        doc.add_nodes(&mut nav, &[NodeData::Element {
1380          tag:        "ltx:TOC".to_string(),
1381          attributes: Some(HashMap::from_iter([("format".to_string(), format.clone())])),
1382          children:   vec![],
1383        }]);
1384      }
1385    }
1386    if let Some(ref title) = doc_title {
1387      if let Some(mut nav) = doc.findnode("//ltx:navigation") {
1388        doc.add_nodes(&mut nav, &[NodeData::Element {
1389          tag:        "ltx:title".to_string(),
1390          attributes: None,
1391          children:   vec![NodeData::Text(title.clone())],
1392        }]);
1393      }
1394    }
1395
1396    self.fill_in_relations(&mut doc);
1397    self.fill_in_tocs(&mut doc);
1398    self.fill_in_frags(&doc);
1399    self.fill_in_glossaryrefs(&mut doc);
1400    self.fill_in_refs(&mut doc);
1401    self.fill_in_bibrefs(&mut doc);
1402    self.fill_in_mathlinks(&doc);
1403    self.copy_resources(&doc);
1404    self.report_missing();
1405    Ok(vec![doc])
1406  }
1407}

1412fn relative_url(target: &str, base: &str) -> String {
1413  if target == base {
1414    return ".".to_string();
1415  }
1416  let target_parts: Vec<&str> = target.split('/').collect();
1417  let base_parts: Vec<&str> = base.split('/').collect();
1418  let common = target_parts
1419    .iter()
1420    .zip(base_parts.iter())
1421    .take_while(|(a, b)| a == b)
1422    .count();
1423  let mut result = String::new();
1424  for _ in common..base_parts.len().saturating_sub(1) {
1425    result.push_str("../");
1426  }
1427  result.push_str(&target_parts[common..].join("/"));
1428  if result.is_empty() {
1429    ".".to_string()
1430  } else {
1431    result
1432  }
1433}

1435/// Get text content from an XML node, normalizing whitespace.
1436///
1437/// Port of `getTextContent`.
1438fn get_text_content_node(node: &Node) -> String {
1439  let text = node.get_content();
1440  let trimmed = text.trim();
1441  // Normalize whitespace
1442  trimmed.split_whitespace().collect::<Vec<_>>().join(" ")
1443}
1444
1445fn text_content(nodes: &[NodeData]) -> String {
1446  nodes
1447    .iter()
1448    .map(|n| match n {
1449      NodeData::Text(s) => s.clone(),
1450      NodeData::Element { children, .. } => text_content(children),
1451      NodeData::XmlNode(n) => n.get_content(),
1452    })
1453    .collect::<Vec<_>>()
1454    .join("")
1455}
