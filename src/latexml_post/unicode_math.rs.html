1//! Unicode math conversion processor.
2//!
3//! Port of `LaTeXML::Post::UnicodeMath` (435 lines of Perl).
4//! Converts XMath nodes to Unicode mathematical notation strings.
5//! Attempts compliance with UTN#28 (Unicode Technical Note on Plain Text Math).
6//!
7//! Used as:
8//! 1. A standalone text math format
9//! 2. A secondary format within MathML `m:annotation`
10//! 3. A utility for converting math to plain text (e.g. for title attributes)
11
12use libxml::tree::Node;
13
14use crate::document::{PostDocument, element_children};
15use crate::math_processor::{MathConversion, MathProcessor};
16use crate::processor::{ProcessResult, Processor};
17
18const UNICODE_MATH_MIMETYPE: &str = "application/x-unicodemath";
19
20// Precedence levels (matching Perl)
21const PREC_RELOP: i32 = 1;
22const PREC_ADDOP: i32 = 2;
23const PREC_MULOP: i32 = 3;
24const PREC_SCRIPTOP: i32 = 4;
25const PREC_SYMBOL: i32 = 10;
26
27/// UnicodeMath post-processor.
28///
29/// Port of `LaTeXML::Post::UnicodeMath`.
30pub struct UnicodeMath {
31  name:         String,
32  is_secondary: bool,
33}
34
35impl Default for UnicodeMath {
36  fn default() -> Self { Self::new() }
37}
38
39impl UnicodeMath {
40  pub fn new() -> Self {
41    UnicodeMath {
42      name:         "UnicodeMath".to_string(),
43      is_secondary: false,
44    }
45  }
46}

48impl Processor for UnicodeMath {
49  fn get_name(&self) -> &str { &self.name }
50
51  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
52    doc.findnodes("//ltx:Math[not(ancestor::ltx:Math)]")
53  }
54
55  fn process(&mut self, doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult { Ok(vec![doc]) }
56}

58impl MathProcessor for UnicodeMath {
59  fn convert_node(&self, doc: &PostDocument, xmath: &Node) -> Option<MathConversion> {
60    let math = xmath.get_parent()?;
61    let (string, _prec) = unimath_internal(doc, &math);
62    Some(MathConversion {
63      processor_name: self.name.clone(),
64      mimetype:       Some(UNICODE_MATH_MIMETYPE.to_string()),
65      xml:            None,
66      string:         Some(string),
67      src:            None,
68      width:          None,
69      height:         None,
70      depth:          None,
71    })
72  }
73
74  fn raw_id_suffix(&self) -> &str { ".muni" }
75
76  fn is_secondary(&self) -> bool { self.is_secondary }
77}

79/// Public entry point: convert a Math/XMath node to Unicode string.
80///
81/// Port of `unicodemath($doc, $node)`.
82pub fn unicodemath(doc: &PostDocument, node: &Node) -> String {
83  let (uni, _prec) = unimath_internal(doc, node);
84  uni
85}

87// ======================================================================
88// Precedence-based infix roles
89
90fn infix_prec(role: &str) -> Option<i32> {
91  match role {
92    "ADDOP" | "BINOP" => Some(PREC_ADDOP),
93    "MULOP" | "MIDDLE" | "COMPOSEOP" | "MODIFIEROP" => Some(PREC_MULOP),
94    "RELOP" | "METARELOP" | "ARROW" => Some(PREC_RELOP),
95    _ => None,
96  }
97}

99/// Get the operator role, following embellished operators.
100///
101/// Port of `getOperatorRole`.
102fn get_operator_role(doc: &PostDocument, node: &Node) -> Option<String> {
103  if let Some(role) = node.get_attribute("role") {
104    return Some(role);
105  }
106  if doc.is_qname(node, "ltx:XMApp") {
107    let children = element_children(node);
108    if children.len() >= 2 {
109      let op_role = children[0].get_attribute("role").unwrap_or_default();
110      if matches!(
111        op_role.as_str(),
112        "SUPERSCRIPTOP" | "SUBSCRIPTOP" | "OVERACCENT" | "UNDERACCENT" | "MODIFIER" | "MODIFIEROP"
113      ) {
114        return get_operator_role(doc, &children[1]);
115      }
116    }
117  }
118  None
119}

127/// Convert a node, returning (string, precedence).
128///
129/// Port of `unimath_internal`.
130fn unimath_internal(doc: &PostDocument, node: &Node) -> (String, i32) {
131  let tag = doc.get_qname(node).unwrap_or_default();
132  let role = node.get_attribute("role").unwrap_or_default();
133
134  match tag.as_str() {
135    "ltx:Math" | "ltx:XMath" => unimath_map(doc, &element_children(node)),
136    "ltx:XMDual" => {
137      let children = element_children(node);
138      if children.len() >= 2 {
139        unimath_internal(doc, &children[1]) // Presentation branch
140      } else {
141        unimath_error("Empty XMDual")
142      }
143    },
144    "ltx:XMWrap" | "ltx:XMArg" => unimath_map(doc, &element_children(node)),
145    "ltx:XMApp" => {
146      let children = element_children(node);
147      if children.is_empty() {
148        return unimath_error("Missing Operator");
149      }
150      let op = &children[0];
151      let args = &children[1..];
152
153      // Handle floating/post scripts
154      if role.contains("SUBSCRIPT") {
155        return unimath_sub(doc, None, op);
156      }
157      if role.contains("SUPERSCRIPT") {
158        return unimath_sup(doc, None, op);
159      }
160
161      // Realize operator and dispatch
162      let rop = realize(doc, op).unwrap_or_else(|| op.clone());
163      let op_role = get_operator_role(doc, &rop).unwrap_or_default();
164      let meaning = rop.get_attribute("meaning").unwrap_or_default();
165
166      // Dispatch by role/meaning
167      match (op_role.as_str(), meaning.as_str()) {
168        (_, "formulae") | (_, "multirelation") => unimath_map(doc, args),
169        (_, "limit-from") | (_, "annotated") => unimath_prefix(doc, op, args),
170        (_, "square-root") => {
171          if !args.is_empty() {
172            let inner = unimath_nested(doc, &args[0], PREC_MULOP);
173            (format!("\u{221A}{}", inner), PREC_MULOP)
174          } else {
175            ("\u{221A}".to_string(), PREC_MULOP)
176          }
177        },
178        (_, "nth-root") => {
179          if args.len() >= 2 {
180            let (n, _) = unimath_internal(doc, &args[1]);
181            let base = unimath_nested(doc, &args[0], PREC_MULOP);
182            let op_str = match n.as_str() {
183              "2" => "\u{221A}".to_string(),
184              "3" => "\u{221B}".to_string(),
185              "4" => "\u{221C}".to_string(),
186              _ => format!("\\root {}\\of", n),
187            };
188            (format!("{}{}", op_str, base), PREC_MULOP)
189          } else {
190            unimath_prefix(doc, op, args)
191          }
192        },
193        (_, "continued-fraction") => unimath_error("continued fraction"),
194        ("FRACOP", _) => {
195          if args.len() >= 2 {
196            let thickness = children[0].get_attribute("thickness");
197            if thickness.is_some() {
198              // Binomial-like
199              let num = unimath_nested(doc, &args[0], 0);
200              let den = unimath_nested(doc, &args[1], 0);
201              (format!("({}\u{00A6}{})", num, den), PREC_SYMBOL)
202            } else {
203              let num = unimath_nested(doc, &args[0], PREC_MULOP);
204              let den = unimath_nested(doc, &args[1], PREC_MULOP);
205              (format!("{}/{}", num, den), 1)
206            }
207          } else {
208            unimath_prefix(doc, op, args)
209          }
210        },
211        ("SUPERSCRIPTOP", _) => {
212          if args.len() >= 2 {
213            unimath_sup(doc, Some(&args[0]), &args[1])
214          } else {
215            unimath_prefix(doc, op, args)
216          }
217        },
218        ("SUBSCRIPTOP", _) => {
219          if args.len() >= 2 {
220            unimath_sub(doc, Some(&args[0]), &args[1])
221          } else {
222            unimath_prefix(doc, op, args)
223          }
224        },
225        ("OVERACCENT", _) => {
226          if !args.is_empty() {
227            unimath_overaccent(doc, op, &args[0])
228          } else {
229            unimath_prefix(doc, op, args)
230          }
231        },
232        ("UNDERACCENT", _) => {
233          if !args.is_empty() {
234            unimath_underaccent(doc, op, &args[0])
235          } else {
236            unimath_prefix(doc, op, args)
237          }
238        },
239        ("POSTFIX", _) => {
240          if !args.is_empty() {
241            let (op_str, _) = unimath_internal(doc, op);
242            let base = unimath_nested(doc, &args[0], PREC_MULOP);
243            (format!("{}{}", base, op_str), PREC_MULOP)
244          } else {
245            unimath_prefix(doc, op, args)
246          }
247        },
248        ("ENCLOSE", _) => {
249          if !args.is_empty() {
250            (unimath_nested(doc, &args[0], PREC_SYMBOL), PREC_SYMBOL)
251          } else {
252            (String::new(), PREC_SYMBOL)
253          }
254        },
255        _ => {
256          // Check if it's an infix role
257          if let Some(prec) = infix_prec(&op_role) {
258            unimath_infix_with_prec(doc, op, args, prec)
259          } else {
260            // Default: prefix
261            unimath_prefix(doc, op, args)
262          }
263        },
264      }
265    },
266    "ltx:XMTok" => {
267      let meaning = node.get_attribute("meaning").unwrap_or_default();
268      if meaning == "absent" {
269        return (String::new(), PREC_SYMBOL);
270      }
271      let text = stylize_content(node);
272      (text, PREC_SYMBOL)
273    },
274    "ltx:XMHint" => (String::new(), 0),
275    "ltx:XMArray" => {
276      let rows: Vec<String> = element_children(node)
277        .iter()
278        .map(|row| {
279          element_children(row)
280            .iter()
281            .map(|cell| unimath_nested(doc, cell, 0))
282            .collect::<Vec<_>>()
283            .join("&")
284        })
285        .collect();
286      (format!("\u{25A0}({})", rows.join("@")), 0)
287    },
288    "ltx:XMText" => unimath_text(node),
289    "ltx:XMRef" => {
290      if let Some(target) = realize(doc, node) {
291        unimath_internal(doc, &target)
292      } else {
293        unimath_error("Unresolved XMRef")
294      }
295    },
296    "ltx:ERROR" => unimath_error(&node.get_content()),
297    _ => unimath_text(node),
298  }
299}

301/// Convert and wrap in braces if precedence is too low.
302///
303/// Port of `unimath_nested`.
304fn unimath_nested(doc: &PostDocument, node: &Node, prec: i32) -> String {
305  let (string, iprec) = unimath_internal(doc, node);
306  if iprec >= prec {
307    string
308  } else {
309    format!("{{{}}}", string)
310  }
311}

313/// Combine conversion of multiple nodes.
314///
315/// Port of `unimath_map`.
316fn unimath_map(doc: &PostDocument, args: &[Node]) -> (String, i32) {
317  let mut oprec = 0;
318  // If wrapped in OPEN...CLOSE, it's a symbol-level expression
319  if args.len() > 1 {
320    let first_role = args
321      .first()
322      .and_then(|n| n.get_attribute("role"))
323      .unwrap_or_default();
324    let last_role = args
325      .last()
326      .and_then(|n| n.get_attribute("role"))
327      .unwrap_or_default();
328    if first_role == "OPEN" && last_role == "CLOSE" {
329      oprec = PREC_SYMBOL;
330    }
331  }
332  let result: String = args.iter().map(|a| unimath_nested(doc, a, 0)).collect();
333  (result, oprec)
334}

336/// Prefix application: op arg1 arg2 ...
337///
338/// Port of `unimath_prefix`.
339fn unimath_prefix(doc: &PostDocument, op: &Node, args: &[Node]) -> (String, i32) {
340  if args.is_empty() {
341    return (String::new(), PREC_SYMBOL);
342  }
343  let op_str = unimath_nested(doc, op, 0);
344  let args_str: String = args
345    .iter()
346    .map(|a| unimath_nested(doc, a, PREC_SYMBOL))
347    .collect();
348  (format!("{}{}", op_str, args_str), PREC_SYMBOL)
349}

351/// Infix application with explicit precedence.
352///
353/// Port of `unimath_infix`.
354fn unimath_infix_with_prec(
355  doc: &PostDocument,
356  op: &Node,
357  args: &[Node],
358  prec: i32,
359) -> (String, i32) {
360  if args.is_empty() {
361    return (String::new(), PREC_SYMBOL);
362  }
363  let opuni = unimath_nested(doc, op, prec);
364  if args.len() == 1 {
365    // Single arg = prefix
366    let arg = unimath_nested(doc, &args[0], prec);
367    (format!("{}{}", opuni, arg), prec)
368  } else {
369    let mut items = vec![unimath_nested(doc, &args[0], prec)];
370    for arg in &args[1..] {
371      items.push(opuni.clone());
372      items.push(unimath_nested(doc, arg, prec));
373    }
374    (items.join(""), prec)
375  }
376}

378/// Subscript: base_script or _{script}base (pre).
379///
380/// Port of `unimath_sub`.
381fn unimath_sub(doc: &PostDocument, base: Option<&Node>, script: &Node) -> (String, i32) {
382  let ubase = base
383    .map(|b| unimath_nested(doc, b, PREC_SCRIPTOP))
384    .unwrap_or_default();
385  let (uscript, prec) = unimath_internal(doc, script);
386  let uscript = if prec < PREC_SCRIPTOP {
387    format!("{{{}}}", uscript)
388  } else {
389    uscript
390  };
391  (format!("{}_{}", ubase, uscript), PREC_SCRIPTOP)
392}

394/// Superscript: base^script or ^{script}base (pre).
395///
396/// Port of `unimath_sup`.
397fn unimath_sup(doc: &PostDocument, base: Option<&Node>, script: &Node) -> (String, i32) {
398  let ubase = base
399    .map(|b| unimath_nested(doc, b, PREC_SCRIPTOP))
400    .unwrap_or_default();
401  let (uscript, prec) = unimath_internal(doc, script);
402  let uscript = if prec < PREC_SCRIPTOP {
403    format!("{{{}}}", uscript)
404  } else {
405    uscript
406  };
407  (format!("{}^{}", ubase, uscript), PREC_SCRIPTOP)
408}

410/// Over-accent: combining character above.
411///
412/// Port of `unimath_overaccent`.
413fn unimath_overaccent(doc: &PostDocument, op: &Node, base: &Node) -> (String, i32) {
414  let acc = op.get_content();
415  let combining = overaccent_combining(&acc);
416  let (mut ubase, _) = unimath_internal(doc, base);
417  if ubase.chars().count() > 1 {
418    ubase = format!("({})", ubase);
419  }
420  let accent_str = combining.unwrap_or_else(|| format!("\u{252C}{}", acc));
421  (format!("{}{}", ubase, accent_str), PREC_SCRIPTOP)
422}

438/// Convert an XMTok's content to styled Unicode text.
439///
440/// Port of `stylizeContent` (simplified — full version needs unicode_convert).
441fn stylize_content(node: &Node) -> String {
442  let role = node
443    .get_attribute("role")
444    .unwrap_or_else(|| "ID".to_string());
445  let text = node.get_content();
446  if text.is_empty() {
447    // Fallback for empty tokens
448    static DEFAULT_CONTENT: &[(&str, &str)] = &[
449      ("MULOP", "\u{2062}"), // INVISIBLE TIMES
450      ("ADDOP", "\u{2064}"), // INVISIBLE PLUS
451      ("PUNCT", "\u{2063}"), // INVISIBLE SEPARATOR
452    ];
453    for (r, default) in DEFAULT_CONTENT {
454      if role == *r {
455        return default.to_string();
456      }
457    }
458    node
459      .get_attribute("name")
460      .or_else(|| node.get_attribute("meaning"))
461      .unwrap_or(role)
462  } else {
463    text
464  }
465}

467/// Text content in quotes.
468///
469/// Port of `unimath_text`.
470fn unimath_text(node: &Node) -> (String, i32) {
471  let text = node.get_content();
472  (format!("\"{}\"", text), PREC_SYMBOL)
473}

478/// Map over-accent character to combining equivalent.
479fn overaccent_combining(acc: &str) -> Option<String> {
480  let c = acc.chars().next()?;
481  let combining = match c {
482    '^' => '\u{0302}',        // hat
483    '\u{02C7}' => '\u{030C}', // check
484    '~' => '\u{0303}',        // tilde
485    '\u{0084}' => '\u{0301}', // acute
486    '\u{0060}' => '\u{0300}', // grave
487    '\u{02D9}' => '\u{0307}', // dot
488    '\u{00AB}' => '\u{0308}', // ddot
489    '\u{00AF}' => '\u{0304}', // bar/overline
490    '\u{2192}' => '\u{20D7}', // vec
491    '\u{02D8}' => '\u{0306}', // breve
492    'o' => '\u{030A}',        // ring
493    '\u{02DD}' => '\u{030B}', // double acute
494    _ => return None,
495  };
496  Some(combining.to_string())
497}
498
499/// Map under-accent character to combining equivalent.
500fn underaccent_combining(acc: &str) -> Option<String> {
501  let c = acc.chars().next()?;
502  let combining = match c {
503    '\u{00B8}' => '\u{0327}', // cedilla
504    '.' => '\u{0323}',        // dot below
505    '\u{00AF}' => '\u{0331}', // macron below
506    '=' | ',' => '\u{0361}',  // tie / lfhook
507    _ => return None,
508  };
509  Some(combining.to_string())
510}
