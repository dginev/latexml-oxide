1//! Output bundling — port of `LaTeXML::Post::Pack`.
2//!
3//! Separates final-bundle concerns from the post-processing pipeline.
4//! Both `latexml_oxide` and `cortex_worker` previously inlined their own
5//! `pack_output_zip` / `pack_output_zip_with_resources` + `add_dir_to_zip`
6//! copies; this module is the single source of truth so the two binaries
7//! produce byte-identical bundle layouts.
8//!
9//! Bundle layout (zip):
10//! ```text
11//! <html_filename>            — post-processed HTML
12//! <resource_dir>/...         — every non-`.html` file under the staging
13//!                               dir (Graphics-converted PNG/SVG, CSS, …),
14//!                               preserving subdirectories.
15//! <log_filename>             — log text, if `log_filename` is set and
16//!                               `log` is non-empty.
17//! status                     — single-line status string.
18//! telemetry.json             — single-line JSON per-job telemetry, only
19//!                               written when `telemetry_json` is set
20//!                               (cortex_worker canvas runs).
21//! ```
22
23use std::fs::File;
24use std::io::{self, BufWriter, Write};
25use std::path::Path;
26
27use zip::write::SimpleFileOptions;
28use zip::{CompressionMethod, ZipWriter};

30/// Write-buffer size for the zip output. 64 KiB matches the typical
31/// compressed-block size from miniz_oxide/flate2 on HTML+image
32/// content; smaller buffers (8 KiB) cause one syscall per block,
33/// larger (1 MiB) waste RSS without improving throughput. Measured on
34/// 1910.01256 (3 PNG + 2 SVG + 2 CSS + HTML = 1.2 MB zip): unbuffered
35/// → ~70 write() syscalls; 64 KiB buffer → ~12.
36const ZIP_WRITE_BUF: usize = 64 * 1024;
37
38/// Options for [`pack_archive`].
39pub struct PackOptions<'a> {
40  /// Destination zip path.
41  pub zip_path:       &'a str,
42  /// Name to use for the HTML entry inside the zip (e.g. `paper.html`).
43  /// Conventionally `<stem>.html` where `stem` is the source TeX name.
44  pub html_filename:  &'a str,
45  /// Post-processed HTML content.
46  pub html:           &'a str,
47  /// Name for the log entry; pass `None` to skip writing a log entry.
48  pub log_filename:   Option<&'a str>,
49  /// Log content. Skipped if empty even when `log_filename` is set.
50  pub log:            &'a str,
51  /// Single-line status string. Always written as `status`.
52  pub status:         &'a str,
53  /// Resource staging directory (typically a `TempDir`). Every
54  /// non-`.html` file under it is bundled, preserving subdirectories.
55  /// Pass `None` to skip resource bundling.
56  pub resource_dir:   Option<&'a Path>,
57  /// Optional per-job telemetry JSON line. When `Some`, written as
58  /// `telemetry.json` inside the zip. Used by `cortex_worker` canvas
59  /// runs; `benchmark_canvas.sh` extracts this member and appends to
60  /// `<output_dir>/telemetry.jsonl`. See `docs/TELEMETRY.md`.
61  pub telemetry_json: Option<&'a str>,
62}
63
64/// Pack the post-processing outputs into a zip archive.
65///
66/// Returns an `io::Result` rather than `crate::processor::PostError`
67/// because callers (binary mains) are already `Box<dyn Error>`-typed.
68///
69/// **IO performance:** the underlying file is wrapped in a 64 KiB
70/// `BufWriter` before handing it to `ZipWriter`. The zip crate's
71/// internal deflate output is small chunks (per-block from miniz);
72/// without buffering each chunk would be its own `write()` syscall.
73/// Measured ~6× fewer syscalls on 1910.01256 (7 resource files + HTML).
74pub fn pack_archive(opts: &PackOptions) -> io::Result<()> {
75  let file = File::create(opts.zip_path)?;
76  let buf_file = BufWriter::with_capacity(ZIP_WRITE_BUF, file);
77  let mut zip = ZipWriter::new(buf_file);
78  let zip_options = SimpleFileOptions::default().compression_method(CompressionMethod::Deflated);
79
80  // HTML first, so `unzip -l` shows the main artifact at the top.
81  zip
82    .start_file(opts.html_filename, zip_options)
83    .map_err(io_err)?;
84  zip.write_all(opts.html.as_bytes())?;
85
86  // Resource files (Graphics-converted PNG/SVG, injected CSS, etc.).
87  if let Some(dir) = opts.resource_dir {
88    if dir.exists() {
89      add_dir_to_zip(&mut zip, dir, dir, &zip_options)?;
90    }
91  }
92
93  // Log entry.
94  if let Some(log_name) = opts.log_filename {
95    if !opts.log.is_empty() {
96      zip.start_file(log_name, zip_options).map_err(io_err)?;
97      zip.write_all(opts.log.as_bytes())?;
98    }
99  }
100
101  // Status.
102  zip.start_file("status", zip_options).map_err(io_err)?;
103  zip.write_all(opts.status.as_bytes())?;
104
105  // Telemetry (cortex_worker only).
106  if let Some(tjson) = opts.telemetry_json {
107    zip
108      .start_file("telemetry.json", zip_options)
109      .map_err(io_err)?;
110    zip.write_all(tjson.as_bytes())?;
111  }
112
113  zip.finish().map_err(io_err)?;
114  Ok(())
115}

117/// Recursively add files from `dir` to a ZIP archive, preserving the
118/// directory structure relative to `base`. Skips `.html` files because
119/// the post-processed HTML is added separately by [`pack_archive`].
120///
121/// Each source file is wrapped in a 64 KiB `BufReader` to amortise
122/// `read()` syscalls on the input side (the `io::copy` 8 KiB default
123/// chunk would otherwise issue ~ceil(filesize/8K) reads per resource).
124fn add_dir_to_zip<W: Write + io::Seek>(
125  zip: &mut ZipWriter<W>,
126  dir: &Path,
127  base: &Path,
128  options: &SimpleFileOptions,
129) -> io::Result<()> {
130  for entry in std::fs::read_dir(dir)? {
131    let entry = entry?;
132    let path = entry.path();
133    let rel = path.strip_prefix(base).unwrap_or(&path);
134    let name = rel.to_string_lossy().to_string();
135
136    if path.is_dir() {
137      add_dir_to_zip(zip, &path, base, options)?;
138    } else if !name.ends_with(".html") {
139      zip.start_file(&name, *options).map_err(io_err)?;
140      let f = File::open(&path)?;
141      let mut buf_reader = io::BufReader::with_capacity(ZIP_WRITE_BUF, f);
142      std::io::copy(&mut buf_reader, zip)?;
143    }
144  }
145  Ok(())
146}

148/// Convert a `zip::result::ZipError` into an `io::Error` so the caller
149/// signature can stay `io::Result`. The zip crate wraps `io::Error`
150/// already; we just re-wrap unrecognized kinds as `Other`.
151fn io_err(e: zip::result::ZipError) -> io::Error {
152  match e {
153    zip::result::ZipError::Io(inner) => inner,
154    other => io::Error::other(other.to_string()),
155  }
