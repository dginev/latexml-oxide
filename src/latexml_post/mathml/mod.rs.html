27const MML_URI: &str = "http://www.w3.org/1998/Math/MathML";
28const MML_MIMETYPE: &str = "application/mathml-presentation+xml";

31/// MathML post-processor.
32///
33/// Port of `LaTeXML::Post::MathML`.
34/// Handles both Presentation and Content MathML conversion.
35pub struct MathML {
36  name:            String,
37  is_secondary:    bool,
38  /// Whether to produce Content MathML (vs Presentation).
39  content_mathml:  bool,
40  /// Whether to use plane1 Unicode characters for styled identifiers.
41  plane1:          bool,
42  /// Whether to enable line-breaking.
43  linebreaking:    bool,
44  /// Line width for line-breaking.
45  line_width:      u32,
46  /// Whether to keep the XMath nodes alongside the generated MathML.
47  keep_xmath:      bool,
48  /// Whether to emit invisible times (U+2062). When false, replaces with zero-width space.
49  /// Perl: $$MATHPROCESSOR{invisibletimes} — defaults to true.
50  invisible_times: bool,
51  /// Whether to include TeX source annotation in parallel MathML.
52  /// Perl: --mathtex adds <m:annotation encoding='application/x-tex'>
53  mathtex:         bool,
54  /// Whether to add intent=":literal" on all <math> elements.
55  /// ar5iv.sty.ltxml monkey-patches outerWrapper to add this.
56  intent_literal:  bool,
57  /// Parallel-markup secondaries (e.g. a Content-MathML processor under a
58  /// Presentation-MathML primary). Held by the primary rather than registered
59  /// as independent chain passes: during the primary's `process_math_node`,
60  /// each secondary's `convert_node` runs against the still-live XMath and the
61  /// results are folded into one `<m:semantics>` via [`combine_parallel`].
62  /// Port of Perl `MathProcessor`'s primary→secondary parallel model
63  /// (`$$self{parallel}` / `combineParallel`). Empty for a standalone format.
64  secondaries:     Vec<Box<dyn MathProcessor>>,
65}

155impl Processor for MathML {
156  fn get_name(&self) -> &str { &self.name }
157
158  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
159    doc.findnodes("//ltx:Math[not(ancestor::ltx:Math)]")
160  }
161
162  fn process(&mut self, mut doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
163    // Register the MathML namespace so add_nodes can create m: elements
164    doc.add_namespace("m", MML_URI);
165
166    // Process all math nodes
167    process_math(self, &mut doc, nodes, self.keep_xmath)?;
168    Ok(vec![doc])
169  }
170}

172impl MathProcessor for MathML {
173  fn convert_node(&self, doc: &PostDocument, xmath: &Node) -> Option<MathConversion> {
174    // Set invisible_times flag for rendering
175    presentation::set_invisible_times(self.invisible_times);
176
177    let xml = if self.content_mathml {
178      content::convert_to_cmml(doc, xmath)
179    } else {
180      presentation::convert_to_pmml(doc, xmath)
181    };
182
183    let mimetype = if self.content_mathml {
184      CMML_MIMETYPE
185    } else {
186      MML_MIMETYPE
187    };
188
189    // If mathtex is enabled, wrap in <m:semantics> with TeX annotation.
190    // Skip when this primary carries parallel secondaries: `combine_parallel`
191    // then builds the single `<m:semantics>` (primary + content annotation-xml
192    // + the x-tex annotation), so wrapping here would double-nest semantics.
193    let final_xml = if self.mathtex && self.secondaries.is_empty() {
194      let tex_str = xmath
195        .get_parent()
196        .and_then(|p| p.get_attribute("tex"))
197        .unwrap_or_default();
198      if tex_str.is_empty() {
199        xml
200      } else {
201        NodeData::Element {
202          tag:        "m:semantics".to_string(),
203          attributes: None,
204          children:   vec![xml, NodeData::Element {
205            tag:        "m:annotation".to_string(),
206            attributes: Some(HashMap::from_iter([(
207              "encoding".to_string(),
208              "application/x-tex".to_string(),
209            )])),
210            children:   vec![NodeData::Text(tex_str)],
211          }],
212        }
213      }
214    } else {
215      xml
216    };
217
218    Some(MathConversion {
219      processor_name: self.name.clone(),
220      mimetype:       Some(mimetype.to_string()),
221      xml:            Some(final_xml),
222      string:         None,
223      src:            None,
224      width:          None,
225      height:         None,
226      depth:          None,
227    })
228  }
229
230  fn combine_parallel(
231    &self,
232    _doc: &PostDocument,
233    xmath: &Node,
234    primary: MathConversion,
235    secondaries: Vec<MathConversion>,
236  ) -> MathConversion {
237    if secondaries.is_empty() {
238      return primary;
239    }
240
241    // Build m:semantics element with primary + annotation-xml for secondaries
242    let mut children = Vec::new();
243    if let Some(ref xml) = primary.xml {
244      children.push(xml.clone());
245    }
246
247    for secondary in &secondaries {
248      let mimetype = secondary.mimetype.as_deref().unwrap_or("unknown");
249      // Parallel markup names the format via the canonical encoding label
250      // (e.g. `MathML-Content`), not the raw internal mimetype. Port of
251      // `%ENCODINGS` / `encoding_for_mimetype`.
252      let encoding = encoding_for_mimetype(mimetype).to_string();
253      if let Some(ref xml) = secondary.xml {
254        children.push(NodeData::Element {
255          tag:        "m:annotation-xml".to_string(),
256          attributes: Some(HashMap::from_iter([("encoding".to_string(), encoding)])),
257          children:   vec![xml.clone()],
258        });
259      } else if let Some(ref string) = secondary.string {
260        children.push(NodeData::Element {
261          tag:        "m:annotation".to_string(),
262          attributes: Some(HashMap::from_iter([("encoding".to_string(), encoding)])),
263          children:   vec![NodeData::Text(string.clone())],
264        });
265      }
266    }
267
268    // TeX source annotation. In the standalone (no-secondary) path this is added
269    // by `convert_node`; in the parallel path that wrap is skipped, so the
270    // single combined `<m:semantics>` carries the x-tex annotation here.
271    if self.mathtex {
272      let tex_str = xmath
273        .get_parent()
274        .and_then(|p| p.get_attribute("tex"))
275        .unwrap_or_default();
276      if !tex_str.is_empty() {
277        children.push(NodeData::Element {
278          tag:        "m:annotation".to_string(),
279          attributes: Some(HashMap::from_iter([(
280            "encoding".to_string(),
281            "application/x-tex".to_string(),
282          )])),
283          children:   vec![NodeData::Text(tex_str)],
284        });
285      }
286    }
287
288    MathConversion {
289      processor_name: self.name.clone(),
290      mimetype:       Some(MML_MIMETYPE.to_string()),
291      xml:            Some(NodeData::Element {
292        tag: "m:semantics".to_string(),
293        attributes: None,
294        children,
295      }),
296      string:         None,
297      src:            None,
298      width:          None,
299      height:         None,
300      depth:          None,
301    }
302  }

304  fn outer_wrapper(&self, _doc: &PostDocument, xmath: &Node, conversion: NodeData) -> NodeData {
305    let mut attrs = HashMap::default();
306    // Determine display mode and alttext from parent Math element
307    // Port of MathML::outerWrapper (L77-100)
308    if let Some(math) = xmath.get_parent() {
309      let mode = math
310        .get_attribute("mode")
311        .unwrap_or_else(|| "inline".to_string());
312      attrs.insert(
313        "display".to_string(),
314        if mode == "display" {
315          "block".to_string()
316        } else {
317          "inline".to_string()
318        },
319      );
320      if let Some(tex) = math.get_attribute("tex") {
321        attrs.insert("alttext".to_string(), tex);
322      }
323      if let Some(class) = math.get_attribute("class") {
324        attrs.insert("class".to_string(), class);
325      }
326    }
327
328    // ar5iv.sty.ltxml: intent=":literal" for all math elements
329    if self.intent_literal {
330      attrs.insert("intent".to_string(), ":literal".to_string());
331    }
332
333    NodeData::Element {
334      tag:        "m:math".to_string(),
335      attributes: Some(attrs),
336      children:   vec![conversion],
337    }
338  }

371  match mimetype {
372    "application/mathml-presentation+xml" => "MathML-Presentation",
373    "application/mathml-content+xml" => "MathML-Content",
374    "image/svg+xml" => "SVG1.1",
375    _ => mimetype,
376  }

402/// Stylize a token's content for MathML.
403///
404/// Port of `stylizeContent` (~130 lines of Perl).
405/// Given an XMTok node and a target MathML tag, determines:
406/// - The text content (with empty-token defaults)
407/// - The mathvariant (from font, with plane1 mapping)
408/// - Operator dictionary properties (fence, stretchy, etc.)
409/// - Color, background, class, href attributes
410///
411/// Returns (text, attributes_map).
412pub fn stylize_content(node: &Node, target_tag: &str) -> (String, HashMap<String, String>) {
413  let _is_element = true; // Node is always an element in our API
414  let role = node.get_attribute("role").unwrap_or_default();
415  let font = node.get_attribute("font");
416  let size = node.get_attribute("fontsize");
417  let color = node.get_attribute("color");
418  let bgcolor = node.get_attribute("backgroundcolor");
419  let opacity = node.get_attribute("opacity");
420  let class = node.get_attribute("class");
421  let href = node.get_attribute("href");
422  let title = node.get_attribute("title");
423
424  let mut text = node.get_content();
425  let is_token = matches!(target_tag, "m:mi" | "m:mo" | "m:mn");
426
427  // Default token content for invisible operators
428  static DEFAULT_CONTENT: &[(&str, &str)] = &[
429    ("MULOP", "\u{2062}"), // INVISIBLE TIMES
430    ("ADDOP", "\u{2064}"), // INVISIBLE PLUS
431    ("PUNCT", "\u{2063}"), // INVISIBLE SEPARATOR
432  ];
433
434  if text.is_empty() {
435    for (r, default) in DEFAULT_CONTENT {
436      if role == *r {
437        text = default.to_string();
438        break;
439      }
440    }
441    if text.is_empty() {
442      text = node
443        .get_attribute("name")
444        .or_else(|| node.get_attribute("meaning"))
445        .unwrap_or_else(|| role.clone());
446    }
447  }
448
449  // Minus sign normalization (MathML Core prefers Unicode minus)
450  if text == "-" && matches!(role.as_str(), "ADDOP" | "OPERATOR") {
451    text = "\u{2212}".to_string();
452  }
453
454  // Determine mathvariant from font.
455  // Port of Perl stylizeContent lines 689-756.
456  let mut variant: Option<&str> = font.as_deref().map(unicode::unicode_mathvariant);
457  let mut attrs = HashMap::default();
458
459  // Single char mi: italic is default, "normal" must be stated explicitly
460  // Perl L717-719
461  if target_tag == "m:mi" && text.chars().count() == 1 {
462    if variant == Some("italic") {
463      variant = None; // defaults to italic
464    } else if variant.is_none() && font.is_none() {
465      // no font at all — italic is MathML default for single-char mi
466    } else if variant.is_none() {
467      variant = Some("normal"); // font present but no recognized variant → say "normal"
468    }
469  } else if font.is_some() && variant == Some("normal") {
470    // "normal" is default for non-mi tokens; omit
471    variant = None;
472  }
473
474  // Plane 1 Unicode conversion.
475  // Port of Perl stylizeContent lines 729-741.
476  // When plane1=true (default), convert text to Mathematical Alphanumeric Symbols
477  // and clear the mathvariant (the character itself carries the style).
478  if let Some(v) = variant {
479    if target_tag != "m:mtext" {
480      if let Some(u_text) = unicode::unicode_convert(&text, v) {
481        if !u_text.is_empty() || text.is_empty() {
482          text = u_text;
483          variant = None; // Plane 1 char carries the style; no mathvariant needed
484        }
485      }
486    }
487  }
488
489  // Apply remaining variant (not cleared by plane1 conversion)
490  if let Some(v) = variant {
491    if target_tag == "m:mi" && text.chars().count() == 1 {
492      if v != "italic" {
493        attrs.insert("mathvariant".to_string(), v.to_string());
494      }
495    } else if v != "normal" {
496      attrs.insert("mathvariant".to_string(), v.to_string());
497    }
498  } else if target_tag == "m:mi" && text.chars().count() > 1 && font.is_none() {
499    // Multi-char mi without font → normal
500    attrs.insert("mathvariant".to_string(), "normal".to_string());
501  }
502
503  // Invisible/format-only text needs no styling attributes.
504  // Port of Perl L743-744: if ($text =~ /^\p{Format}*$/)
505  let is_format_only = text.chars().all(|c| {
506    use std::char;
507    matches!(char::from_u32(c as u32), Some(ch) if ch.is_control())
508      || matches!(c,
509        '\u{200B}'..='\u{200F}' | '\u{2028}'..='\u{202F}'
510        | '\u{2060}'..='\u{2064}' | '\u{FEFF}' | '\u{00AD}')
511  });
512  if is_format_only {
513    variant = None;
514    // Don't clear font/color etc from attrs — they weren't added yet
515  }
516
517  // Font-based CSS class fallbacks.
518  // Port of Perl L746-756: only when variant was NOT converted to plane1.
519  let mut class_val = class;
520  if let Some(ref f) = font {
521    if !is_format_only {
522      if f.contains("caligraphic") {
523        let c = class_val.as_deref().unwrap_or("");
524        class_val = Some(if c.is_empty() {
525          "ltx_font_mathcaligraphic".to_string()
526        } else {
527          format!("{} ltx_font_mathcaligraphic", c)
528        });
529      } else if f.contains("script") {
530        let c = class_val.as_deref().unwrap_or("");
531        class_val = Some(if c.is_empty() {
532          "ltx_font_mathscript".to_string()
533        } else {
534          format!("{} ltx_font_mathscript", c)
535        });
536      } else if f.contains("fraktur") && text.chars().all(|c| "+-0123456789.".contains(c)) {
537        // Perl L751: fraktur number → oldstyle class
538        let c = class_val.as_deref().unwrap_or("");
539        class_val = Some(if c.is_empty() {
540          "ltx_font_oldstyle".to_string()
541        } else {
542          format!("{} ltx_font_oldstyle", c)
543        });
544      } else if f.contains("smallcaps") {
545        let c = class_val.as_deref().unwrap_or("");
546        class_val = Some(if c.is_empty() {
547          "ltx_font_smallcaps".to_string()
548        } else {
549          format!("{} ltx_font_smallcaps", c)
550        });
551      } else if let Some(v) = variant {
552        if v != "normal" {
553          // Perl L755-756: leftover mathvariant → CSS fallback
554          let c = class_val.as_deref().unwrap_or("");
555          class_val = Some(if c.is_empty() {
556            format!("ltx_mathvariant_{}", v)
557          } else {
558            format!("{} ltx_mathvariant_{}", c, v)
559          });
560        }
561      }
562    }
563  }
564  if let Some(cv) = class_val {
565    if !cv.is_empty() {
566      attrs.insert("class".to_string(), cv);
567    }
568  }
569
570  // Operator-specific properties (mo only) — from MathML Core operator dictionary
571  if target_tag == "m:mo" {
572    let dict_props = operator_dictionary::opdict_lookup(&text, &role);
573
574    // Implied role-based properties
575    let is_fence = matches!(role.as_str(), "OPEN" | "CLOSE" | "MIDDLE");
576    let is_sep = role == "PUNCT";
577    let is_large = matches!(role.as_str(), "SUMOP" | "INTOP");
578    let is_move = matches!(role.as_str(), "SUMOP" | "INTOP" | "BIGOP" | "LIMITOP");
579    let is_symm = is_large || text == "/";
580
581    // Only set attributes that differ from the operator dictionary defaults
582    // (matching Perl's xor-based attribute generation)
583    let stretchy = node.get_attribute("stretchy").as_deref() == Some("true");
584    if stretchy != dict_props.stretchy {
585      attrs.insert(
586        "stretchy".to_string(),
587        if stretchy { "true" } else { "false" }.to_string(),
588      );
589    }
590    if is_fence != dict_props.fence {
591      attrs.insert(
592        "fence".to_string(),
593        if is_fence { "true" } else { "false" }.to_string(),
594      );
595    }
596    if is_sep != dict_props.separator {
597      attrs.insert(
598        "separator".to_string(),
599        if is_sep { "true" } else { "false" }.to_string(),
600      );
601    }
602    if is_large != dict_props.largeop {
603      attrs.insert(
604        "largeop".to_string(),
605        if is_large { "true" } else { "false" }.to_string(),
606      );
607    }
608    if is_large {
609      attrs.insert("_largeop".to_string(), "1".to_string()); // For needsMathStyle
610    }
611    if is_symm && !dict_props.symmetric && (stretchy || dict_props.stretchy) {
612      attrs.insert("symmetric".to_string(), "true".to_string());
613    }
614    if is_move {
615      let pos = node.get_attribute("scriptpos").unwrap_or_default();
616      if pos.starts_with("mid") {
617        attrs.insert("movablelimits".to_string(), "false".to_string());
618      }
619    }
620
621    // Store dictionary spacing for later spacing resolution
622    attrs.insert("_lspace".to_string(), format!("{}em", dict_props.lspace));
623    attrs.insert("_rspace".to_string(), format!("{}em", dict_props.rspace));
624  }
625
626  // Color
627  if let Some(c) = color {
628    attrs.insert("mathcolor".to_string(), c);
629  }
630  if let Some(bg) = bgcolor {
631    attrs.insert("mathbackground".to_string(), bg);
632  }
633  if let Some(op) = opacity {
634    let style_val = format!("opacity:{}", op);
635    attrs.insert("style".to_string(), style_val);
636  }
637
638  // Size
639  if let Some(s) = size {
640    if s != "100%" {
641      attrs.insert("mathsize".to_string(), s);
642    }
643  }
644
645  // Href and title (tokens only)
646  if is_token {
647    if let Some(h) = href {
648      attrs.insert("href".to_string(), h);
649    }
650    if let Some(t) = title {
651      attrs.insert("title".to_string(), t);
652    }
653  }
654
655  (text, attrs)
656}

658/// Convert an XMHint spacing attribute to em value.
659///
660/// Port of `getXMHintSpacing`.
661pub fn get_xm_hint_spacing(width: &str) -> f64 {
662  // Perl (MathML.pm L380-385): /^([\d\.\+\-]+)(pt|mu|em)(\s+plus\s+…)?(\s+minus\s+…)?$/
663  // — a GLUE width ("3.0pt plus 2.0pt minus 1.0pt") contributes its natural
664  // part; the stretch/shrink tails are ignored.
665  let trimmed = width.trim();
666  let base = trimmed
667    .split(" plus ")
668    .next()
669    .unwrap_or(trimmed)
670    .split(" minus ")
671    .next()
672    .unwrap_or(trimmed)
673    .trim();
674  if let Some((num_str, unit)) = base
675    .rfind(|c: char| c.is_ascii_digit() || c == '.')
676    .map(|i| (&base[..=i], base[i + 1..].trim()))
677  {
678    let num: f64 = num_str.parse().unwrap_or(0.0);
679    match unit {
680      "em" => num,
681      "mu" => num / 18.0,
682      "pt" => num / 10.0, // Assuming 10pt font
683      _ => 0.0,
684    }
685  } else {
686    0.0
687  }
688}

690/// Find an inherited attribute by walking up the LaTeXML ancestor chain.
691///
692/// Port of `find_inherited_attribute`.
693pub fn find_inherited_attribute(
694  _doc: &PostDocument,
695  node: &Node,
696  attribute: &str,
697) -> Option<String> {
698  let mut current = Some(node.clone());
699  while let Some(ref n) = current {
700    // Perl getQName returns undef for non-elements → stop. Also guards the
701    // FFI: reading the ns field of a Document node is a misaligned deref.
702    if n.get_type() != Some(libxml::tree::NodeType::ElementNode) {
703      break;
704    }
705    if let Some(ns) = n.get_namespace() {
706      if ns.get_href() != crate::document::LTX_NSURI {
707        break; // Stop at non-LaTeXML elements
708      }
709    }
710    if let Some(val) = n.get_attribute(attribute) {
711      return Some(val);
712    }
713    current = n.get_parent();
714  }
715  None
716}

719// DefMathML converter dispatch table
720//
721// Port of the `%MMLTable_P` / `%MMLTable_C` lookup tables and the
722// 800+ lines of DefMathML declarations in MathML.pm.
723//
724// The Perl pattern is:
725//   DefMathML("Mode:Role:Meaning", \&pmml_handler, \&cmml_handler);
726// Lookup tries: "Mode:Role:Meaning", "Mode:?:Meaning", "Mode:Role:?", "Mode:?:?"
727//
728// In Rust, we encode this as a static table of known role→tag mappings
729// and meaning→element mappings, and the actual dispatch happens in
730// presentation.rs::pmml_apply() and content.rs::cmml().

732/// Presentation MathML tag for a token role.
733///
734/// Port of Token:ROLE:? DefMathML declarations.
735/// These map roles to their default MathML element type.
736pub fn pmml_tag_for_role(role: &str) -> &'static str {
737  match role {
738    // Operators → m:mo
739    "PUNCT" | "PERIOD" | "OPEN" | "CLOSE" | "MIDDLE" | "VERTBAR" | "ARROW" | "OVERACCENT"
740    | "UNDERACCENT" | "ADDOP" | "MULOP" | "BINOP" | "RELOP" | "METARELOP" | "MODIFIEROP"
741    | "COMPOSEOP" | "APPLYOP" | "OPERATOR" | "SUPOP" | "POSTFIX" | "DIFFOP" => "m:mo",
742    // Big operators → m:mo (with largeop)
743    "BIGOP" | "SUMOP" | "INTOP" | "LIMITOP" => "m:mo",
744    // Functions → m:mi (but rendered as operator names)
745    "FUNCTION" | "OPFUNCTION" | "TRIGFUNCTION" => "m:mi",
746    // Numbers → m:mn
747    "NUMBER" => "m:mn",
748    // Identifiers → m:mi (default)
749    _ => "m:mi",
750  }
751}

853/// Wrap items in an mrow, filtering out ignorable items.
854///
855/// Port of `pmml_row` + `filter_row`.
856pub fn pmml_row(items: Vec<NodeData>) -> NodeData {
857  // Filter out ignorable items (those with _ignorable attribute)
858  let filtered: Vec<NodeData> = items
859    .into_iter()
860    .filter(|item| match item {
861      NodeData::Element { attributes, .. } => {
862        if let Some(attrs) = attributes {
863          !attrs.contains_key("_ignorable")
864        } else {
865          true
866        }
867      },
868      _ => true,
869    })
870    .collect();
871
872  if filtered.len() == 1 {
873    filtered.into_iter().next().unwrap()
874  } else {
875    NodeData::Element {
876      tag:        "m:mrow".to_string(),
877      attributes: None,
878      children:   filtered,
879    }
880  }
881}

883/// Parenthesize an expression with open/close delimiters.
884///
885/// Port of `pmml_parenthesize`.
886pub fn pmml_parenthesize(item: NodeData, open: Option<&str>, close: Option<&str>) -> NodeData {
887  if open.is_none() && close.is_none() {
888    return item;
889  }
890
891  let mut children = Vec::new();
892  if let Some(o) = open {
893    children.push(NodeData::Element {
894      tag:        "m:mo".to_string(),
895      attributes: Some(HashMap::from_iter([
896        ("fence".to_string(), "true".to_string()),
897        ("stretchy".to_string(), "true".to_string()),
898      ])),
899      children:   vec![NodeData::Text(o.to_string())],
900    });
901  }
902  children.push(item);
903  if let Some(c) = close {
904    children.push(NodeData::Element {
905      tag:        "m:mo".to_string(),
906      attributes: Some(HashMap::from_iter([
907        ("fence".to_string(), "true".to_string()),
908        ("stretchy".to_string(), "true".to_string()),
909      ])),
910      children:   vec![NodeData::Text(c.to_string())],
911    });
912  }
913
914  NodeData::Element {
915    tag: "m:mrow".to_string(),
916    attributes: None,
917    children,
918  }
919}

921/// Punctuate a list of items with separators.
922///
923/// Port of `pmml_punctuate`.
924pub fn pmml_punctuate(separators: &str, items: Vec<NodeData>) -> NodeData {
925  if items.is_empty() {
926    return NodeData::Element {
927      tag:        "m:mrow".to_string(),
928      attributes: None,
929      children:   vec![],
930    };
931  }
932
933  let mut result = Vec::new();
934  let mut sep_chars: Vec<char> = separators.chars().collect();
935  let last_sep = if sep_chars.is_empty() {
936    ','
937  } else {
938    *sep_chars.last().unwrap()
939  };
940
941  let mut iter = items.into_iter();
942  result.push(iter.next().unwrap());
943
944  for item in iter {
945    let sep = if sep_chars.is_empty() {
946      last_sep
947    } else {
948      sep_chars.remove(0)
949    };
950    result.push(NodeData::Element {
951      tag:        "m:mo".to_string(),
952      attributes: Some(HashMap::from_iter([(
953        "separator".to_string(),
954        "true".to_string(),
955      )])),
956      children:   vec![NodeData::Text(sep.to_string())],
957    });
958    result.push(item);
959  }
960
961  pmml_row(result)
962}

964/// Convert a text node within XMText to Presentation MathML.
965///
966/// Port of `pmml_text_aux`.
967/// Handles text nodes, element nodes (including nested Math),
968/// and preserves font/color attributes through the conversion.
969pub fn pmml_text_aux(doc: &PostDocument, node: &Node) -> Vec<NodeData> {
970  use libxml::tree::NodeType;
971
972  match node.get_type() {
973    Some(NodeType::TextNode) => {
974      let text = node.get_content();
975      // Replace leading/trailing whitespace with NBSP
976      let text = text.trim_start().to_string();
977      let text = if text.is_empty() {
978        "\u{00A0}".to_string()
979      } else {
980        let mut t = text;
981        if t.starts_with(char::is_whitespace) {
982          t = format!("\u{00A0}{}", t.trim_start());
983        }
984        if t.ends_with(char::is_whitespace) {
985          t = format!("{}\u{00A0}", t.trim_end());
986        }
987        t
988      };
989      vec![NodeData::Element {
990        tag:        "m:mtext".to_string(),
991        attributes: None,
992        children:   vec![NodeData::Text(text)],
993      }]
994    },
995    Some(NodeType::ElementNode) => {
996      let tag = doc.get_qname(node).unwrap_or_default();
997      match tag.as_str() {
998        "ltx:Math" => {
999          // Nested math: convert XMath if present
1000          match doc.findnode_at("ltx:XMath", node) {
1001            Some(xmath) => {
1002              vec![presentation::convert_to_pmml(doc, &xmath)]
1003            },
1004            _ => {
1005              vec![]
1006            },
1007          }
1008        },
1009        "ltx:text" => {
1010          // Recurse on children
1011          let mut results = Vec::new();
1012          if let Some(child) = node.get_first_child() {
1013            let mut current = Some(child);
1014            while let Some(ref c) = current {
1015              results.extend(pmml_text_aux(doc, c));
1016              current = c.get_next_sibling();
1017            }
1018          }
1019          results
1020        },
1021        "ltx:picture" => {
1022          // Picture in text: wrap in mtext. Eagerly materialize the picture
1023          // subtree into owned NodeData so the result is not tied to the
1024          // source node's libxml2 lifetime. Perl: MathProcessor.pm
1025          // convertXMTextContent (Post.pm L456-489). A lazy
1026          // `NodeData::XmlNode(node.clone())` here SIGSEGVs in
1027          // `add_xml_node` once the parent XMath is unlinked (its children
1028          // are stripped into a detached document fragment and later
1029          // accesses via the stale rust-libxml wrapper dereference freed
1030          // memory — reproducible on 0710.1208 / 1110.2158 / 1605.07431).
1031          vec![NodeData::Element {
1032            tag:        "m:mtext".to_string(),
1033            attributes: None,
1034            children:   convert_xm_text_content(doc, node, true),
1035          }]
1036        },
1037        _ => {
1038          // Unknown element (e.g. ltx:ref, ltx:bibref, ltx:inline-block,
1039          // …): preserve the raw subtree inside the mtext so the XSLT
1040          // can transform it (ltx:ref → HTML <a>, etc.). Perl
1041          // `pmml_text_aux` (MathML.pm L1063-1073) clones the whole
1042          // node into the returned mtext; we eagerly materialize an
1043          // owned subtree, threading `doc` through so URI→prefix
1044          // resolution recovers the canonical `ltx:` prefix on
1045          // default-namespace elements.
1046          let cloned = rebuild_text_subtree_with_doc(node, true, Some(doc))
1047            .unwrap_or_else(|| NodeData::Text("\u{00A0}".to_string()));
1048          vec![NodeData::Element {
1049            tag:        "m:mtext".to_string(),
1050            attributes: None,
1051            children:   vec![cloned],
1052          }]
1053        },
1054      }
1055    },
1056    _ => vec![],
1057  }
1058}

1060/// Eagerly materialize an XMText-or-picture subtree into owned NodeData.
1061///
1062/// Port of `LaTeXML::Post::MathProcessor::convertXMTextContent`
1063/// (Post.pm L456-489). Walks `node` recursively and rebuilds the subtree
1064/// as owned NodeData, so downstream consumers do not depend on the
1065/// source node's libxml2 lifetime. Internal `_*` attributes and stray
1066/// `xml:id` are dropped (Perl mirrors this); `fragid` would be remapped
1067/// to a fresh id in Perl but MathML::Presentation does not carry a
1068/// processor-level id suffix through this path, so we drop it too and
1069/// let the surrounding MathML ids govern.
1070///
1071/// When `convert_spaces` is true, leading/trailing whitespace on text
1072/// nodes is replaced with NBSP so the rendered MathML does not collapse
1073/// the space. Nested `ltx:Math` would in Perl re-enter the MathML
1074/// converter; we preserve it as a plain element subtree here (same
1075/// limitation as the enclosing pmml_text_aux path: nested math in
1076/// mtext is uncommon, and the XSLT stage can still render it).
1077pub fn convert_xm_text_content(
1078  doc: &PostDocument,
1079  node: &Node,
1080  convert_spaces: bool,
1081) -> Vec<NodeData> {
1082  node
1083    .get_child_nodes()
1084    .iter()
1085    .filter_map(|c| rebuild_text_subtree_with_doc(c, convert_spaces, Some(doc)))
1086    .collect()
1087}

1089/// Rebuild a libxml2 subtree into owned `NodeData`, dropping internal
1090/// `_*`, `xml:id`, and `fragid` attributes. Shared between
1091/// `convert_xm_text_content` (Perl `convertXMTextContent`,
1092/// Post.pm L456-489) and `pmml_text_aux` for cases where Perl
1093/// calls `cloneNode($node, 'nest')` (MathML.pm L1073) — i.e. when an
1094/// unhandled element like `ltx:ref` appears inside a text-mode
1095/// fragment and must survive into the output so the XSLT can
1096/// transform it (e.g. `ltx:ref` → `<a>`).
1097pub fn rebuild_text_subtree(node: &Node, convert_spaces: bool) -> Option<NodeData> {
1098  rebuild_text_subtree_with_doc(node, convert_spaces, None)
1099}

1101/// Same as `rebuild_text_subtree`, but consults the post-document's
1102/// namespace map to resolve elements whose source `xmlns="…"` carries
1103/// an empty prefix. `add_nodes` only emits elements whose tag is
1104/// `prefix:local`; without a prefix the element is dropped with a
1105/// `malformed:namespace` warning. libxml2 reports an empty
1106/// `Namespace::get_prefix()` for the default-namespace branch even
1107/// when the doc has a `ltx:` prefix declared elsewhere, so we
1108/// reverse-lookup the URI in `PostDocument::namespaces` to recover
1109/// the canonical prefix.
1110pub fn rebuild_text_subtree_with_doc(
1111  node: &Node,
1112  convert_spaces: bool,
1113  doc: Option<&PostDocument>,
1114) -> Option<NodeData> {
1115  use libxml::tree::NodeType;
1116  match node.get_type() {
1117    Some(NodeType::TextNode) => {
1118      let mut text = node.get_content();
1119      if convert_spaces {
1120        if text.starts_with(char::is_whitespace) {
1121          text = format!("\u{00A0}{}", text.trim_start());
1122        }
1123        if text.ends_with(char::is_whitespace) {
1124          text = format!("{}\u{00A0}", text.trim_end());
1125        }
1126      }
1127      Some(NodeData::Text(text))
1128    },
1129    Some(NodeType::ElementNode) => {
1130      let tag = {
1131        let local = node.get_name();
1132        match node.get_namespace() {
1133          Some(ns) => {
1134            let prefix = ns.get_prefix();
1135            if !prefix.is_empty() {
1136              format!("{prefix}:{local}")
1137            } else {
1138              // Default-namespace element. add_nodes won't accept a
1139              // tag without prefix — reverse-resolve URI → prefix
1140              // via the post-document's namespace map.
1141              let uri = ns.get_href();
1142              match doc.and_then(|d| {
1143                d.namespaces
1144                  .iter()
1145                  .find(|(p, u)| !p.is_empty() && **u == uri)
1146                  .map(|(p, _)| p.clone())
1147              }) {
1148                Some(p) => format!("{p}:{local}"),
1149                None => local,
1150              }
1151            }
1152          },
1153          None => local,
1154        }
1155      };
1156      // Copy attributes, skipping internal `_*`, `xml:id`, and `fragid`.
1157      // Matches Perl convertXMTextContent (Post.pm L479-483); the
1158      // `fragid → xml:id` remap requires the MathProcessor's IDSuffix
1159      // which this helper does not receive — drop both here rather
1160      // than forge a wrong id.
1161      let mut attrs: HashMap<String, String> = HashMap::default();
1162      for (k, v) in node.get_attributes() {
1163        if k.starts_with('_') || k == "xml:id" || k == "fragid" {
1164          continue;
1165        }
1166        attrs.insert(k, v);
1167      }
1168      let children: Vec<NodeData> = node
1169        .get_child_nodes()
1170        .iter()
1171        .filter_map(|c| rebuild_text_subtree_with_doc(c, convert_spaces, doc))
1172        .collect();
1173      Some(NodeData::Element {
1174        tag,
1175        attributes: if attrs.is_empty() { None } else { Some(attrs) },
1176        children,
1177      })
1178    },
1179    _ => None,
1180  }
1181}
