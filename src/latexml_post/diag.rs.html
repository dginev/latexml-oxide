1//! Diagnostic emission for `latexml_post`.
2//!
3//! Five capitalized macros — `Note!`, `Info!`, `Warn!`, `Error!`,
4//! `Fatal!` — mirror the LaTeXML Perl `Note()`/`Info()`/`Warn()`/
5//! `Error()`/`Fatal()` reporting conventions. Local to this crate so
6//! `latexml_post` does not need to import `latexml_core::common::error`
7//! just to emit diagnostics. The macro names deliberately shadow the
8//! identically-named macros from `latexml_core` — same shape (`(category,
9//! object, …)`), simpler implementation (no `note_status` counter,
10//! no error-cap unwinding, no location trace appended).
11//!
12//! Output format goes through the shared `latexml_core::util::logger`
13//! formatter via `log::info!`/`warn!`/`error!` with an explicit
14//! `target = "<category>:<object>"`, yielding the canonical
15//! `{Severity}:{category}:{object} {message}` line the harness
16//! aggregates from every other stage (engine, package, contrib).
17//!
18//! `Note!` is the lone exception: it bypasses the logger formatter
19//! entirely and writes the bare message to stderr, matching the
20//! prefix-less `Note(…)` output style from Perl LaTeXML.
21//!
22//! `Fatal!` additionally `return`s `Err(PostError::Processing(…))`
23//! so the calling function can early-exit via `?`, mirroring the way
24//! Perl `Fatal()` early-exits via `die`. The crate's `PostError` type
25//! differs from `latexml_core::common::error::Error`, which is why we
26//! cannot reuse the upstream `Fatal!`.
27//!
28//! Convention notes carried over from Perl `LaTeXML::Post::*`:
29//!   * `Error('expected', 'source', …)`        — Graphics.pm:216 (missing source)
30//!   * `Error('imageprocessing', $source, …)`  — Graphics.pm:274 (conversion fail)
31//!   * `Error('expected', 'stylesheet', …)`    — XSLT.pm:36/47 (XSLT setup)
32//!   * `Error('missing-file', $stylesheet, …)` — XSLT.pm:42 (missing XSLT)
33//!   * `Error('expected', 'Image::Magick', …)` — LaTeXImages.pm:128 (env)
34//!   * `Error('I/O', $path, …)`                — LaTeXImages.pm:259 (I/O)
35//!   * `Error('shell', $cmd, …)`               — LaTeXImages.pm:293/328 (subprocess)
36//!   * `Fatal('misdefined', (ref $self), …)`   — Post.pm:177/434 (no-process / abstract)
37//!   * `Fatal('unexpected', $dir, …)`          — Post.pm:701 (bad destdir)
38
39#[macro_export]
40macro_rules! Note {
41  ($input:expr) => {{
42    if log::max_level() >= log::LevelFilter::Info {
43      eprintln!("{}", $input);
44    }
45  }};
46}
47
48#[macro_export]
49macro_rules! Info {
50  ($category:expr, $object:expr, $msg:expr) => {
51    log::info!(target: &format!("{}:{}", $category, $object), "{}", $msg)
52  };
53  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {
54    log::info!(target: &format!("{}:{}", $category, $object), $fmt, $($arg)+)
55  };
56}
57
58#[macro_export]
59macro_rules! Warn {
60  ($category:expr, $object:expr, $msg:expr) => {
61    log::warn!(target: &format!("{}:{}", $category, $object), "{}", $msg)
62  };
63  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {
64    log::warn!(target: &format!("{}:{}", $category, $object), $fmt, $($arg)+)
65  };
66}
67
68#[macro_export]
69macro_rules! Error {
70  ($category:expr, $object:expr, $msg:expr) => {
71    log::error!(target: &format!("{}:{}", $category, $object), "{}", $msg)
72  };
73  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {
74    log::error!(target: &format!("{}:{}", $category, $object), $fmt, $($arg)+)
75  };
76}
77
78#[macro_export]
79macro_rules! Fatal {
80  ($category:expr, $object:expr, $msg:expr) => {{
81    log::error!(target: &format!("Fatal:{}:{}", $category, $object), "{}", $msg);
82    return Err($crate::processor::PostError::Processing(
83      format!("{}:{}: {}", $category, $object, $msg)
84    ));
85  }};
86  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {{
87    let __m = format!($fmt, $($arg)+);
88    log::error!(target: &format!("Fatal:{}:{}", $category, $object), "{}", __m);
89    return Err($crate::processor::PostError::Processing(
90      format!("{}:{}: {}", $category, $object, __m)
91    ));
92  }};
93}
