1//! Diagnostic emission for `latexml_post`.
2//!
3//! Post-processing functions return `PostError`, not the engine's
4//! `latexml_core::common::error::Error`, so we cannot directly reuse
5//! the full `Error!`/`Warn!`/`Fatal!` macros from
6//! `latexml_core::common::error` — those early-return
7//! `Err(LatexmlError)` on max-errors / runaway-loop and would type-mismatch
8//! against `PostError` returns.
9//!
10//! Instead we emit through `log::{error,warn,info}!` with a `target:`
11//! attribute shaped exactly the way `latexml_core::common::error` shapes it.
12//! The shared logger formatter at `latexml_core::util::logger::log` reads
13//! `record.target()` and emits:
14//!
15//!   `{Severity}:{target} {message}`
16//!
17//! With `target = "<class>:<object>"` that yields the canonical
18//! `Error:<class>:<object> <message>` line the harness aggregates from
19//! every other stage (engine, package, contrib).
20//!
21//! Convention notes carried over from Perl `LaTeXML::Post::*`:
22//!   * `Error('expected', 'source', …)`        — Graphics.pm:216 (missing source)
23//!   * `Error('imageprocessing', $source, …)`  — Graphics.pm:274 (conversion fail)
24//!   * `Error('expected', 'stylesheet', …)`    — XSLT.pm:36/47 (XSLT setup)
25//!   * `Error('missing-file', $stylesheet, …)` — XSLT.pm:42 (missing XSLT)
26//!   * `Error('expected', 'Image::Magick', …)` — LaTeXImages.pm:128 (env)
27//!   * `Error('I/O', $path, …)`                — LaTeXImages.pm:259 (I/O)
28//!   * `Error('shell', $cmd, …)`               — LaTeXImages.pm:293/328 (subprocess)
29//!   * `Fatal('misdefined', (ref $self), …)`   — Post.pm:177/434 (no-process / abstract)
30//!   * `Fatal('unexpected', $dir, …)`          — Post.pm:701 (bad destdir)
31
32#[macro_export]
33macro_rules! log_post_error {
34  ($category:expr, $object:expr, $msg:expr) => {
35    log::error!(target: &format!("{}:{}", $category, $object), "{}", $msg)
36  };
37  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {
38    log::error!(target: &format!("{}:{}", $category, $object), $fmt, $($arg)+)
39  };
40}

62/// `log_post_fatal!` emits the same target-prefixed `Fatal:<class>:<object>`
63/// line a Perl `Fatal('class', 'object', …)` would, and then converts to
64/// a `PostError::Processing` so the calling function can early-return via
65/// `?` like Perl's `die`.
66///
67/// The logger's "Fatal:" prefix is matched by
68/// `latexml_core::util::logger::log` line 82 (`starts_with("Fatal:")` →
69/// no severity prefix added, the class-prefix `Fatal:` is the severity).
70#[macro_export]
71macro_rules! log_post_fatal {
72  ($category:expr, $object:expr, $msg:expr) => {{
73    log::error!(target: &format!("Fatal:{}:{}", $category, $object), "{}", $msg);
74    return Err($crate::processor::PostError::Processing(
75      format!("{}:{}: {}", $category, $object, $msg)
76    ));
77  }};
78  ($category:expr, $object:expr, $fmt:expr, $($arg:tt)+) => {{
79    let __m = format!($fmt, $($arg)+);
80    log::error!(target: &format!("Fatal:{}:{}", $category, $object), "{}", __m);
81    return Err($crate::processor::PostError::Processing(
82      format!("{}:{}: {}", $category, $object, __m)
83    ));
84  }};
85}
