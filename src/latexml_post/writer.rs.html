1//! XML file output processor.
2//!
3//! Port of `LaTeXML::Post::Writer`.
4//! Serializes the XML document to a file or stdout,
5//! handling DOCTYPE removal, TEMPORARY_DOCUMENT_ID cleanup,
6//! and HTML vs XML serialization.
7
8use libxml::tree::{Node, SaveOptions};
9use std::fs;
10
11use crate::document::PostDocument;
12use crate::processor::{PostError, ProcessResult, Processor};
13
14/// Output format for the writer.
15#[derive(Debug, Clone, Copy, PartialEq, Eq)]
16pub enum OutputFormat {
17  Xml,
18  Html,
19}
20
21/// Writer post-processor: serializes document to file.
22///
23/// Port of `LaTeXML::Post::Writer`.
24pub struct Writer {
25  name:         String,
26  format:       OutputFormat,
27  omit_doctype: bool,
28  is_html:      bool,
29}
30
31impl Writer {
32  pub fn new(format: Option<OutputFormat>, omit_doctype: bool, is_html: bool) -> Self {
33    Writer {
34      name: "Writer".to_string(),
35      format: format.unwrap_or(OutputFormat::Xml),
36      omit_doctype,
37      is_html,
38    }
39  }
40}
41
42impl Processor for Writer {
43  fn get_name(&self) -> &str { &self.name }
44
45  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
46    match doc.get_document_element() {
47      Some(el) => vec![el],
48      None => vec![],
49    }
50  }
51
52  fn process(&mut self, doc: PostDocument, nodes: Vec<Node>) -> ProcessResult {
53    let mut root = match nodes.into_iter().next() {
54      Some(r) => r,
55      None => return Ok(vec![doc]),
56    };
57
58    // Remove TEMPORARY_DOCUMENT_ID if present (Perl Writer.pm L41-42)
59    if let Some(id) = root.get_attribute("xml:id") {
60      if id == "TEMPORARY_DOCUMENT_ID" {
61        let _ = root.remove_attribute("xml:id");
62      }
63    }
64
65    // Serialize: HTML uses toStringHTML, XML uses toString(1)  (Perl Writer.pm L44-47)
66    let serialized = if self.is_html {
67      doc.get_document().to_string_with_options(SaveOptions {
68        as_html: true,
69        format: true,
70        ..SaveOptions::default()
71      })
72    } else {
73      doc.get_document().to_string_with_options(SaveOptions {
74        format: true,
75        ..SaveOptions::default()
76      })
77    };
78
79    if let Some(destination) = doc.get_destination() {
80      // Create destination directory if needed
81      if let Some(destdir) = doc.get_destination_directory() {
82        fs::create_dir_all(destdir).map_err(|e| {
83          PostError::Io(std::io::Error::new(
84            e.kind(),
85            format!("Couldn't create directory '{}': {}", destdir, e),
86          ))
87        })?;
88      }
89      fs::write(destination, &serialized).map_err(|e| {
90        PostError::Io(std::io::Error::new(
91          e.kind(),
92          format!("Couldn't write '{}': {}", destination, e),
93        ))
94      })?;
95      log::info!("Wrote '{}' ({})", destination, serialized.len());
96    } else {
97      print!("{}", serialized);
98    }
99
100    Ok(vec![doc])
101  }
102}

104#[cfg(test)]
105mod tests {
106  use super::*;
107
108  #[test]
109  fn output_format_variants() {
110    let _ = OutputFormat::Xml;
111    let _ = OutputFormat::Html;
112  }
113
114  #[test]
115  fn output_format_partial_eq() {
116    assert_eq!(OutputFormat::Xml, OutputFormat::Xml);
117    assert_ne!(OutputFormat::Xml, OutputFormat::Html);
118    assert_eq!(OutputFormat::Html, OutputFormat::Html);
119  }
120
121  #[test]
122  fn output_format_copy_clone() {
123    // Copy trait: move-after-use still works.
124    let a = OutputFormat::Xml;
125    let b = a;
126    let _ = a; // still usable due to Copy
127    assert_eq!(a, b);
128  }
129
130  #[test]
131  fn writer_new_default_format_xml() {
132    let w = Writer::new(None, false, false);
133    assert_eq!(w.get_name(), "Writer");
134    assert_eq!(w.format, OutputFormat::Xml);
135    assert!(!w.omit_doctype);
136    assert!(!w.is_html);
137  }
138
139  #[test]
140  fn writer_new_explicit_format() {
141    let w = Writer::new(Some(OutputFormat::Html), true, true);
142    assert_eq!(w.format, OutputFormat::Html);
143    assert!(w.omit_doctype);
144    assert!(w.is_html);
145  }
146
147  #[test]
148  fn writer_get_name_is_writer() {
149    let w = Writer::new(None, false, false);
150    assert_eq!(w.get_name(), "Writer");
151  }
152}
