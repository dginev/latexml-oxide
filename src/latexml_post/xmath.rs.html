1//! XMath pseudo-generator for preserving LaTeXML's parsed math.
2//!
3//! Port of `LaTeXML::Post::XMath`.
4//!
5//! If XMath is the primary representation (or only one), it is left in place.
6//! If secondary, it is cloned with modified IDs and moved.
7//! Must be the last math formatter in the chain when used as secondary,
8//! since XMath removal would break subsequent formatters.
9
10use libxml::tree::Node;
11use rustc_hash::FxHashMap as HashMap;
12
13use crate::document::{NodeData, PostDocument, element_children_iter};
14use crate::math_processor::{MathConversion, MathProcessor};
15use crate::processor::{ProcessResult, Processor};
16
17const XMATH_MIMETYPE: &str = "application/x-latexml";
18
19/// XMath post-processor: preserves or clones XMath as a math representation.
20///
21/// Port of `LaTeXML::Post::XMath`.
22pub struct XMath {
23  name:         String,
24  is_secondary: bool,
25}
26
27impl Default for XMath {
28  fn default() -> Self { Self::new() }
29}
30
31impl XMath {
32  pub fn new() -> Self {
33    XMath {
34      name:         "XMath".to_string(),
35      is_secondary: false,
36    }
37  }
38}
39
40impl Processor for XMath {
41  fn get_name(&self) -> &str { &self.name }
42
43  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
44    doc.findnodes("//ltx:Math[not(ancestor::ltx:Math)]")
45  }
46
47  fn process(&mut self, doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult { Ok(vec![doc]) }
48}
49
50impl MathProcessor for XMath {
51  fn convert_node(&self, _doc: &PostDocument, xmath: &Node) -> Option<MathConversion> {
52    let id_suffix = self.id_suffix();
53    let xml = if !id_suffix.is_empty() {
54      // Secondary: clone the XMath with modified IDs
55      let children: Vec<NodeData> = element_children_iter(xmath)
56        .map(NodeData::XmlNode)
57        .collect();
58      Some(NodeData::Element {
59        tag: "ltx:XMath".to_string(),
60        attributes: Some(HashMap::from_iter([("_sourced".to_string(), "1".to_string())])),
61        children,
62      })
63    } else {
64      // Primary: just reference the existing XMath node
65      Some(NodeData::XmlNode(xmath.clone()))
66    };
67
68    Some(MathConversion {
69      processor_name: self.name.clone(),
70      mimetype: Some(XMATH_MIMETYPE.to_string()),
71      xml,
72      string: None,
73      src: None,
74      width: None,
75      height: None,
76      depth: None,
77    })
78  }
79
80  fn combine_parallel(
81    &self,
82    _doc: &PostDocument,
83    _xmath: &Node,
84    primary: MathConversion,
85    secondaries: Vec<MathConversion>,
86  ) -> MathConversion {
87    let mut alt_children = Vec::new();
88
89    // Primary XML goes first
90    if let Some(ref xml) = primary.xml {
91      alt_children.push(xml.clone());
92    }
93
94    // Add secondaries
95    for secondary in &secondaries {
96      let mimetype = secondary.mimetype.as_deref().unwrap_or("unknown");
97      if mimetype == XMATH_MIMETYPE {
98        if let Some(ref xml) = secondary.xml {
99          alt_children.push(xml.clone());
100        }
101      } else if let Some(ref xml) = secondary.xml {
102        // Other XML: needs wrapping (outerWrapper would be called by the processor)
103        alt_children.push(xml.clone());
104      }
105    }
106
107    MathConversion {
108      processor_name: self.name.clone(),
109      mimetype:       Some(XMATH_MIMETYPE.to_string()),
110      xml:            Some(NodeData::Element {
111        tag:        "_Fragment_".to_string(),
112        attributes: None,
113        children:   alt_children,
114      }),
115      string:         None,
116      src:            None,
117      width:          None,
118      height:         None,
119      depth:          None,
120    }
121  }
122
123  fn raw_id_suffix(&self) -> &str { ".xm" }
124
125  fn is_secondary(&self) -> bool { self.is_secondary }
126}
