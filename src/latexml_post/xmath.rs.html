1//! XMath pseudo-generator for preserving LaTeXML's parsed math.
2//!
3//! Port of `LaTeXML::Post::XMath`.
4//!
5//! If XMath is the primary representation (or only one), it is left in place.
6//! If secondary, it is cloned with modified IDs and moved.
7//! Must be the last math formatter in the chain when used as secondary,
8//! since XMath removal would break subsequent formatters.
9
10use libxml::tree::Node;
11use std::collections::HashMap;
12
13use crate::document::{element_children_iter, NodeData, PostDocument};
14use crate::math_processor::{MathConversion, MathProcessor};
15use crate::processor::{ProcessResult, Processor};
16
17const XMATH_MIMETYPE: &str = "application/x-latexml";
18
19/// XMath post-processor: preserves or clones XMath as a math representation.
20///
21/// Port of `LaTeXML::Post::XMath`.
22pub struct XMath {
23  name: String,
24  is_secondary: bool,
25}
26
27impl Default for XMath {
28    fn default() -> Self {
29        Self::new()
30    }
31}
32
33impl XMath {
34  pub fn new() -> Self {
35    XMath {
36      name: "XMath".to_string(),
37      is_secondary: false,
38    }
39  }
40}
41
42impl Processor for XMath {
43  fn get_name(&self) -> &str {
44    &self.name
45  }
46
47  fn to_process(&self, doc: &PostDocument) -> Vec<Node> {
48    doc.findnodes("//ltx:Math[not(ancestor::ltx:Math)]")
49  }
50
51  fn process(&mut self, doc: PostDocument, _nodes: Vec<Node>) -> ProcessResult {
52    Ok(vec![doc])
53  }
54}
55
56impl MathProcessor for XMath {
57  fn convert_node(&self, _doc: &PostDocument, xmath: &Node) -> Option<MathConversion> {
58    let id_suffix = self.id_suffix();
59    let xml = if !id_suffix.is_empty() {
60      // Secondary: clone the XMath with modified IDs
61      let children: Vec<NodeData> = element_children_iter(xmath)
62        .map(NodeData::XmlNode)
63        .collect();
64      Some(NodeData::Element {
65        tag: "ltx:XMath".to_string(),
66        attributes: Some(HashMap::from([("_sourced".to_string(), "1".to_string())])),
67        children,
68      })
69    } else {
70      // Primary: just reference the existing XMath node
71      Some(NodeData::XmlNode(xmath.clone()))
72    };
73
74    Some(MathConversion {
75      processor_name: self.name.clone(),
76      mimetype: Some(XMATH_MIMETYPE.to_string()),
77      xml,
78      string: None,
79      src: None,
80      width: None,
81      height: None,
82      depth: None,
83    })
84  }
85
86  fn combine_parallel(
87    &self,
88    _doc: &PostDocument,
89    _xmath: &Node,
90    primary: MathConversion,
91    secondaries: Vec<MathConversion>,
92  ) -> MathConversion {
93    let mut alt_children = Vec::new();
94
95    // Primary XML goes first
96    if let Some(ref xml) = primary.xml {
97      alt_children.push(xml.clone());
98    }
99
100    // Add secondaries
101    for secondary in &secondaries {
102      let mimetype = secondary.mimetype.as_deref().unwrap_or("unknown");
103      if mimetype == XMATH_MIMETYPE {
104        if let Some(ref xml) = secondary.xml {
105          alt_children.push(xml.clone());
106        }
107      } else if let Some(ref xml) = secondary.xml {
108        // Other XML: needs wrapping (outerWrapper would be called by the processor)
109        alt_children.push(xml.clone());
110      }
111    }
112
113    MathConversion {
114      processor_name: self.name.clone(),
115      mimetype: Some(XMATH_MIMETYPE.to_string()),
116      xml: Some(NodeData::Element {
117        tag: "_Fragment_".to_string(),
118        attributes: None,
119        children: alt_children,
120      }),
121      string: None,
122      src: None,
123      width: None,
124      height: None,
125      depth: None,
126    }
127  }
128
129  fn raw_id_suffix(&self) -> &str {
130    ".xm"
131  }
132
133  fn is_secondary(&self) -> bool {
134    self.is_secondary
135  }
136}
