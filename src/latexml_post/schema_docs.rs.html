1//! Schema-doc post-processing — visual customizations for `--splitat=section`
2//! schema documentation.
3//!
4//! Three string-level passes run against each split sub-page after the
5//! standard LaTeXML XSLT has produced HTML:
6//!
7//! 1. **Content-model rendering** — pretty-print RelaxNG-style structural
8//!    expressions (`A , B | C?`) from one-line walls into operator-leading
9//!    multi-line layout. Replaces `tools/render-content-models.py`.
10//!
11//! 2. **Definition-card decoration** — promote `schema.X` anchor ids onto
12//!    parent `<dt>` elements, wrap kind words ("Pattern" / "Element" /
13//!    "Attribute" / "Add to") in chip spans, and append `§` permalink
14//!    anchors. Replaces `tools/decorate-definitions.py`.
15//!
16//! 3. **Sidebar item index + module narrative** — collect each page's
17//!    Pattern/Element/Attribute definitions and inject a per-module item
18//!    index into the navbar, and prepend a curated narrative aside above
19//!    the section heading (loaded from a TOML file). Replaces
20//!    `tools/inject-module-sidebar.py`.
21//!
22//! All three passes are idempotent: re-running on already-processed HTML
23//! is a no-op. The driver is `process_page`; `load_summaries` reads the
24//! per-module narrative TOML once.
25

46/// Run the schema-doc passes on a single page's HTML.
47///
48/// Layout: defs are description-list items inside the per-module
49/// section page (`--splitat=section`). No kind-bucket subsections;
50/// Patterns and Elements interleave in source order so cross-refs
51/// between them stay on one page. Long pages get a JS-driven filter
52/// input (browser Ctrl-F still works since items default to visible).
53pub fn process_page(html: &str) -> String {
54  let html = lift_module_narrative(html);
55  let html = render_content_models(&html);
56  let html = decorate_definitions(&html);
57  let html = inject_sidebar_index(&html);
58  let html = inject_theme_switcher(&html);
59  inject_experimental_banner(&html)
60}

62/// Inject the rustdoc-styled Settings popover *markup* — only the
63/// HTML widget that exposes Theme (Light / Dark / Ayu / System) and
64/// the Hide-sidebar toggle. **No `<script>` element is injected from
65/// here**; the runtime
66/// (`resources/javascript/relaxng-schema-rustdoc-theme.js`) is
67/// pulled in via the standard `--javascript=…` flag of `latexml_oxide`,
68/// which the orchestration shell (`tools/generate-scholarly-schema-docs`)
69/// passes alongside `--css=…`. The XSLT then emits a non-deferred
70/// `<script src>` in `<head>` for us — same code path the CSS
71/// `<link>` uses — so `applyTheme()` runs synchronously before paint.
72///
73/// The runtime handles three pieces of behaviour, all on the
74/// pre-existing widget markup this function injects:
75///
76/// 1. Pre-paint application of `data-theme` / `data-pref-*` from
77///    `localStorage`.
78/// 2. Settings popover wiring (radios + checkboxes + click-outside +
79///    system colour-scheme listener) on `DOMContentLoaded`.
80/// 3. The in-page schema-def filter (sticky search above long def
81///    lists) — replaces the prior `inject_filter_script` pass.
82///
83/// Settings widget shape:
84///
85/// * Theme fieldset — 4 radios (Light / Dark / Ayu / System), keyed
86///   to `localStorage["schema-theme"]`.
87/// * Display fieldset — 1 checkbox:
88///
89///   | localStorage key   | `<html>` attribute  | CSS effect |
90///   |--------------------|---------------------|------------|
91///   | `schema-pref-sidebar` | `data-pref-sidebar="on"` | hide `nav.ltx_page_navbar` |
92///
93///   (Sans-serif font swap and content-model wrap were tried but
94///   removed — neither was easy to use, and the results were
95///   marginal compared to the existing layout.)
96///
97/// * Tasteful credit line linking to the rustdoc reference docs.
98///
99/// Other rustdoc settings (auto-hide methods, search single-result
100/// jump, line numbers on examples, deprecation, keyboard shortcuts)
101/// are out of scope — they don't apply to schema docs.
102fn inject_theme_switcher(html: &str) -> String {
103  if html.contains("data-schema-theme-widget") {
104    return html.to_string();
105  }
106  static BODY_OPEN_RE: OnceLock<Regex> = OnceLock::new();
107  let body_open_re = BODY_OPEN_RE.get_or_init(|| Regex::new(r"(?i)<body[^>]*>").unwrap());
108
109  // Settings widget — markup only; behaviour wired up by the JS
110  // runtime fetched via `--javascript=relaxng-schema-rustdoc-theme.js`.
111  let widget = r##"<details class="schema-theme-switcher" data-schema-theme-widget>
112<summary aria-label="Settings" title="Settings"><span class="schema-gear" aria-hidden="true">⚙</span></summary>
113<div class="schema-theme-popover" role="dialog" aria-label="Settings">
114<fieldset>
115<legend>Theme</legend>
116<label><input type="radio" name="schema-theme-radio" value="light"> Light</label>
117<label><input type="radio" name="schema-theme-radio" value="dark"> Dark</label>
118<label><input type="radio" name="schema-theme-radio" value="ayu"> Ayu</label>
119<label><input type="radio" name="schema-theme-radio" value="system"> System</label>
120</fieldset>
121<fieldset class="schema-pref-block">
122<legend>Display</legend>
123<label><input type="checkbox" data-schema-pref="sidebar"> Hide sidebar</label>
124</fieldset>
125<p class="schema-theme-credit">Theme inspired by <a href="https://doc.rust-lang.org/rustdoc/what-is-rustdoc.html" rel="noopener">rustdoc</a>.</p>
126</div>
127</details>"##;
128
129  body_open_re
130    .replace(html, |caps: &regex::Captures| {
131      format!("{}{}", caps.get(0).unwrap().as_str(), widget)
132    })
133    .into_owned()
134}

155    .into_owned()
156}

159/// `\moduleabstract` produces `<ltx:para class="schema_module_narrative">`
160/// which LaTeXML's HTML output renders as a marked
161/// `<div class="ltx_para schema_module_narrative">` for the first
162/// paragraph, *plus* one unmarked `<div class="ltx_para">` per
163/// subsequent paragraph (the marker class doesn't survive across
164/// `\par` breaks inside the macro arg). Trang emits one
165/// `<a:documentation>` per `## comment` block in the source RNC,
166/// and our `extract_docs` joins them with blank lines, so a module
167/// with multiple `## comment` paragraphs lands as multiple `<p>`s
168/// in this run.
169///
170/// The post-pass walks the marked div *and every immediately-
171/// following `<div class="ltx_para">` sibling* up to the next
172/// non-paragraph element (typically the description-list opener),
173/// then folds the whole run into one left-bordered
174/// `<aside class="schema_module_narrative">` block right after the
175/// section heading. Each source paragraph stays in its own `<p>`
176/// inside the aside.
177fn lift_module_narrative(html: &str) -> String {
178  if html.contains(r#"<aside class="schema_module_narrative">"#) {
179    return html.to_string();
180  }
181  static NARRATIVE_OPEN_RE: OnceLock<Regex> = OnceLock::new();
182  static EXTRA_PARA_RE: OnceLock<Regex> = OnceLock::new();
183  static P_RE: OnceLock<Regex> = OnceLock::new();
184  static HEADING_RE: OnceLock<Regex> = OnceLock::new();
185
186  // Step 1: locate the first `schema_module_narrative` div, capturing
187  // the whole `<div …>…</div>` (class order is liberal — either
188  // `ltx_para` or `schema_module_narrative` may come first).
189  let narrative_open_re = NARRATIVE_OPEN_RE.get_or_init(|| {
190    Regex::new(
191      r#"(?s)<div [^>]*class="[^"]*schema_module_narrative[^"]*"[^>]*>.*?</div>"#,
192    )
193    .unwrap()
194  });
195  // Step 2: anchor-at-start regex that matches whitespace + one
196  // additional `<div class="…schema_module_narrative…">…</div>`
197  // sibling — `genschema_oxide`'s lift emits one `\moduleabstract`
198  // per source paragraph so every paragraph carries the marker
199  // class. Walking ONLY marked siblings (not generic `ltx_para`
200  // divs) avoids accidentally consuming the wrapper around the
201  // description list (which is `<div class="ltx_para">` itself in
202  // some splits).
203  let extra_para_re = EXTRA_PARA_RE.get_or_init(|| {
204    Regex::new(
205      r#"(?s)\A\s*<div [^>]*class="[^"]*schema_module_narrative[^"]*"[^>]*>.*?</div>"#,
206    )
207    .unwrap()
208  });
209  // Step 3: pull every `<p class="ltx_p">…</p>` out of the combined
210  // run — they're the paragraphs to splice into the aside.
211  let p_re = P_RE.get_or_init(|| {
212    Regex::new(r#"(?s)<p class="ltx_p[^"]*">.*?</p>"#).unwrap()
213  });
214  let heading_re = HEADING_RE.get_or_init(|| {
215    Regex::new(r#"(?s)(<h1 class="ltx_title ltx_title_section">.*?</h1>)"#).unwrap()
216  });
217
218  let first = match narrative_open_re.find(html) {
219    Some(m) => m,
220    None => return html.to_string(),
221  };
222  // Extend the match through any contiguous trailing
223  // `<div class="ltx_para">` siblings.
224  let mut end = first.end();
225  while end < html.len() {
226    let rest = &html[end..];
227    match extra_para_re.find(rest) {
228      Some(m) => end += m.end(),
229      None => break,
230    }
231  }
232  let block = &html[first.start()..end];
233  let paragraphs: Vec<&str> = p_re.find_iter(block).map(|m| m.as_str()).collect();
234  let inner = paragraphs.join("\n");
235  let aside = format!(
236    r#"<aside class="schema_module_narrative">{}</aside>"#,
237    inner
238  );
239
240  // Strip the original block, then insert the aside right after
241  // the section heading.
242  let mut stripped = String::with_capacity(html.len());
243  stripped.push_str(&html[..first.start()]);
244  stripped.push_str(&html[end..]);
245  let result = heading_re.replace(&stripped, |caps: &regex::Captures| {
246    format!("{}\n{}", &caps[1], aside)
247  });
248  result.into_owned()
249}

253#[derive(Debug)]
254enum Tok {
255  A(String),       // <a>...</a>
256  SpanRef(String), // <span class="ltx_ref ...">...</span> (self-page ref)
257  SpanTt(String),  // <span class="ltx_text ltx_font_typewriter">...</span>
258  SpanLit(String), // <span class="ltx_text ltx_font_italic">...</span>
259  Sup(String),     // <sup class="ltx_sup">[?*+]</sup>
260  LParen,
261  RParen,
262  OpOr,
263  OpAnd,
264  OpSeq,
265}
266
267fn tokenize(s: &str) -> Option<Vec<Tok>> {
268  static RE: OnceLock<Regex> = OnceLock::new();
269  let re = RE.get_or_init(|| {
270    Regex::new(concat!(
271      r#"(?P<a><a\s[^>]*>.*?</a>)"#,
272      r#"|(?P<spanref><span\s+class="ltx_ref\b[^"]*">.*?</span>)"#,
273      r#"|(?P<spantt><span\s+class="ltx_text\s+ltx_font_typewriter">.*?</span>)"#,
274      r#"|(?P<spanlit><span\s+class="ltx_text\s+ltx_font_italic">.*?</span>)"#,
275      r#"|(?P<sup><sup\s+class="ltx_sup">[?*+]</sup>)"#,
276      r"|(?P<lparen>\()",
277      r"|(?P<rparen>\))",
278      r"|(?P<opor>\s*\|\s*)",
279      r"|(?P<opand>\s*(?:&amp;|&)\s*)",
280      r"|(?P<opseq>\s*,\s*)",
281      r"|(?P<ws>\s+)",
282    ))
283    .unwrap()
284  });
285
286  let mut tokens = Vec::new();
287  let mut pos = 0;
288  while pos < s.len() {
289    let m = re.captures_at(s, pos)?;
290    let mat = m.get(0).unwrap();
291    if mat.start() != pos {
292      return None; // unexpected character — refuse to mangle
293    }
294    if let Some(t) = m.name("a") {
295      tokens.push(Tok::A(t.as_str().to_string()));
296    } else if let Some(t) = m.name("spanref") {
297      tokens.push(Tok::SpanRef(t.as_str().to_string()));
298    } else if let Some(t) = m.name("spantt") {
299      tokens.push(Tok::SpanTt(t.as_str().to_string()));
300    } else if let Some(t) = m.name("spanlit") {
301      tokens.push(Tok::SpanLit(t.as_str().to_string()));
302    } else if let Some(t) = m.name("sup") {
303      tokens.push(Tok::Sup(t.as_str().to_string()));
304    } else if m.name("lparen").is_some() {
305      tokens.push(Tok::LParen);
306    } else if m.name("rparen").is_some() {
307      tokens.push(Tok::RParen);
308    } else if m.name("opor").is_some() {
309      tokens.push(Tok::OpOr);
310    } else if m.name("opand").is_some() {
311      tokens.push(Tok::OpAnd);
312    } else if m.name("opseq").is_some() {
313      tokens.push(Tok::OpSeq);
314    } // ws: skip
315    pos = mat.end();
316  }
317  Some(tokens)
318}

320#[derive(Debug)]
321enum Node {
322  Atom { html: String, quantifier: String },
323  Group { op: Option<&'static str>, items: Vec<Node>, quantifier: String },
324}
325
326fn parse(tokens: &[Tok], mut pos: usize) -> (Node, usize) {
327  let mut items: Vec<Node> = Vec::new();
328  let mut op: Option<&'static str> = None;
329  while pos < tokens.len() {
330    match &tokens[pos] {
331      Tok::RParen => return (Node::Group { op, items, quantifier: String::new() }, pos),
332      Tok::LParen => {
333        let (inner, np) = parse(tokens, pos + 1);
334        pos = np;
335        let mut group = inner;
336        if pos < tokens.len() && matches!(tokens[pos], Tok::RParen) {
337          pos += 1;
338        }
339        if let (Node::Group { quantifier, .. }, Some(Tok::Sup(s))) =
340          (&mut group, tokens.get(pos))
341        {
342          *quantifier = s.clone();
343          pos += 1;
344        }
345        items.push(group);
346      },
347      Tok::A(html) | Tok::SpanRef(html) | Tok::SpanTt(html) | Tok::SpanLit(html) => {
348        let mut atom = Node::Atom { html: html.clone(), quantifier: String::new() };
349        pos += 1;
350        if let (Node::Atom { quantifier, .. }, Some(Tok::Sup(s))) = (&mut atom, tokens.get(pos)) {
351          *quantifier = s.clone();
352          pos += 1;
353        }
354        items.push(atom);
355      },
356      Tok::OpOr => {
357        if op.is_none() {
358          op = Some("OpOr");
359        }
360        pos += 1;
361      },
362      Tok::OpAnd => {
363        if op.is_none() {
364          op = Some("OpAnd");
365        }
366        pos += 1;
367      },
368      Tok::OpSeq => {
369        if op.is_none() {
370          op = Some("OpSeq");
371        }
372        pos += 1;
373      },
374      Tok::Sup(_) => {
375        pos += 1;
376      },
377    }
378  }
379  (Node::Group { op, items, quantifier: String::new() }, pos)
380}

382fn op_html(op: &str) -> String {
383  let (class, glyph) = match op {
384    "OpOr" => ("op op-or", "|"),
385    "OpAnd" => ("op op-and", "&"),
386    "OpSeq" => ("op op-seq", ","),
387    _ => ("op", "?"),
388  };
389  format!(r#"<span class="{}">{}</span>"#, class, glyph)
390}

392fn is_short(node: &Node) -> bool {
393  match node {
394    Node::Atom { .. } => true,
395    Node::Group { items, .. } => {
396      !items.iter().any(|c| matches!(c, Node::Group { .. })) && items.len() <= 4
397    },
398  }
399}
400
401fn render(node: &Node, indent: usize) -> String {
402  let pad = "  ".repeat(indent);
403  match node {
404    Node::Atom { html, quantifier } => format!("{}{}", html, quantifier),
405    Node::Group { op, items, quantifier } => {
406      if items.is_empty() {
407        return String::new();
408      }
409      if items.len() == 1 && op.is_none() {
410        return format!("{}{}", render(&items[0], indent), quantifier);
411      }
412      if is_short(node) {
413        let sep = match op {
414          Some(o) => format!(" {} ", op_html(o)),
415          None => " ".to_string(),
416        };
417        let parts: Vec<String> = items.iter().map(|c| render(c, indent)).collect();
418        return format!("({}){}", parts.join(&sep), quantifier);
419      }
420      let inner_pad = "  ".repeat(indent + 1);
421      let op_seg = op.map(op_html).unwrap_or_default();
422      let mut lines = vec![String::from("(")];
423      for (i, c) in items.iter().enumerate() {
424        let prefix = if i == 0 { String::from("  ") } else { format!("{} ", op_seg) };
425        lines.push(format!("{}{}{}", inner_pad, prefix, render(c, indent + 1)));
426      }
427      lines.push(format!("{}){}", pad, quantifier));
428      lines.join("\n")
429    },
430  }
431}

433fn render_content_models(html: &str) -> String {
434  if html.contains(r#"class="schema-content-model""#) {
435    return html.to_string();
436  }
437  static RE: OnceLock<Regex> = OnceLock::new();
438  let re = RE.get_or_init(|| Regex::new(r#"(?s)<p class="ltx_p">(\s*\(.+?)</p>"#).unwrap());
439  re.replace_all(html, |caps: &regex::Captures| {
440    let inner = caps[1].trim();
441    let Some(tokens) = tokenize(inner) else {
442      return caps[0].to_string();
443    };
444    if !matches!(tokens.first(), Some(Tok::LParen)) {
445      return caps[0].to_string();
446    }
447    let (mut ast, mut pos) = parse(&tokens, 1);
448    if !matches!(tokens.get(pos), Some(Tok::RParen)) {
449      return caps[0].to_string();
450    }
451    pos += 1;
452    if let Some(Tok::Sup(s)) = tokens.get(pos) {
453      if let Node::Group { quantifier, .. } = &mut ast {
454        *quantifier = s.clone();
455      }
456      pos += 1;
457    }
458    if pos != tokens.len() {
459      return caps[0].to_string();
460    }
461    let body = render(&ast, 0);
462    format!(r#"<p class="ltx_p"><code class="schema-content-model">{}</code></p>"#, body)
463  })
464  .into_owned()
465}

469/// Decorate each `<dt>` definition heading: promote the
470/// `\hypertarget{schema.<name>}` anchor onto the `<dt>` element,
471/// wrap the kind word in a chip span, and append a `§` permalink.
472///
473/// With defs as description-list items (matching upstream Perl), each
474/// `\elementdef` / `\patterndef` / etc. renders as
475///
476/// ```html
477/// <dt id="I1.ix1" class="ltx_item">
478///   <span class="ltx_tag ltx_tag_item">
479///     <span class="ltx_text ltx_font_bold ltx_font_italic">Element </span>
480///     <span class="ltx_text ltx_font_sansserif ltx_font_bold">name</span>
481///   </span>
482/// </dt>
483/// <dd class="ltx_item">
484///   <p class="ltx_p"><a name="schema.X" id="schema.X" class="ltx_anchor">…doc…</a></p>
485///   …
486/// </dd>
487/// ```
488///
489/// We rewrite the `<dt>` to carry `id="schema.X"` plus a chip + name
490/// + § permalink, and strip the redundant `id=` from the inner anchor.
491fn decorate_definitions(html: &str) -> String {
492  if html.contains("schema-kind-chip") {
493    return html.to_string();
494  }
495  static DT_RE: OnceLock<Regex> = OnceLock::new();
496  static ANCHOR_RE: OnceLock<Regex> = OnceLock::new();
497
498  // Match `<dt class="ltx_item">` with the kicker structure
499  // `<bold-italic>KIND <sansserif>NAME</></></dt>` at any nesting depth
500  // — top-level (e.g. `id="I1.ix3"`) and nested (e.g.
501  // `id="I1.ix3.I3.ix2"`) both qualify. Nested matches are how a
502  // pattern like `ltx.span.elem = element span {...}` exposes its
503  // inner `\elementdef{xhtml:span}` so that `\elementref{xhtml:span}`
504  // cross-refs resolve. Duplicate-id collisions across multiple
505  // nested defs of the same name are guarded below by `seen_ids`.
506  let dt_re = DT_RE.get_or_init(|| {
507    Regex::new(concat!(
508      r#"(?s)<dt id="([^"]+)" class="ltx_item">"#,
509      r#"<span class="ltx_tag ltx_tag_item">"#,
510      r#"<span class="ltx_text ltx_font_bold ltx_font_italic">"#,
511      r"([A-Za-z]+(?:\s+[A-Za-z]+)?)\s+",
512      r#"<span class="ltx_text ltx_font_sansserif[^"]*">"#,
513      "([^<]+)</span>",
514      r"</span></span></dt>",
515    ))
516    .unwrap()
517  });
518  let anchor_re = ANCHOR_RE.get_or_init(|| {
519    Regex::new(
520      r#"<a name="(schema\.[^"]+)" id="schema\.[^"]+" class="ltx_anchor">"#,
521    )
522    .unwrap()
523  });
524
525  let kind_class = |kind: &str| -> Option<&'static str> {
526    match kind {
527      "Pattern" => Some("kind-pattern"),
528      "Element" => Some("kind-element"),
529      "Attribute" => Some("kind-attribute"),
530      "Add to" => Some("kind-pattern-add"),
531      _ => None,
532    }
533  };
534
535  let dts: Vec<regex::Captures<'_>> = dt_re.captures_iter(html).collect();
536  if dts.is_empty() {
537    return html.to_string();
538  }
539  let anchors: Vec<regex::Match<'_>> = anchor_re.find_iter(html).collect();
540
541  let mut rewrites: Vec<(usize, usize, String)> = Vec::new();
542  // The same xhtml:NAME often appears as a nested elementdef under
543  // several wrapping pattern defs on one page (e.g. `xhtml:div`
544  // appears 5x in scaffold.html as the body of distinct ltx.*.elem
545  // patterns). We can only assign `id="schema.xhtml..div"` to one of
546  // them; the rest stay as ordinary nested kicker rows. Pick the
547  // first occurrence — that's what LaTeXML's `\hypertarget` would
548  // also pick.
549  let mut seen_ids: HashSet<String> = HashSet::new();
550  // Skip nested attribute promotion: attribute names (`dir`, `class`,
551  // `id`, …) routinely repeat across patterns and would collide.
552  // Top-level attribute defs don't exist in this schema flavour.
553  let promotable = |kind: &str, depth: usize| -> bool {
554    if depth == 0 {
555      return true; // top-level: always (Pattern/Element/Attribute/Add to).
556    }
557    matches!(kind, "Pattern" | "Element")
558  };
559
560  for (i, dt) in dts.iter().enumerate() {
561    let dt_match = dt.get(0).unwrap();
562    let next_pos = dts.get(i + 1).map(|n| n.get(0).unwrap().start()).unwrap_or(html.len());
563    let raw_id = &dt[1];
564    let kind = &dt[2];
565    let name = &dt[3];
566    let Some(class) = kind_class(kind) else {
567      continue;
568    };
569    // `I1.ix3` is depth-0 (top-level dl), `I1.ix3.I3.ix2` is depth-1
570    // (nested dl), etc. Each `.I\d+.ix\d+` pair past the first counts
571    // one level of nesting.
572    let depth = raw_id.matches(".ix").count().saturating_sub(1);
573    if !promotable(kind, depth) {
574      continue;
575    }
576
577    // Derive the def's anchor id from kind + name. Doing this from
578    // the heading text (rather than searching for a sibling `<a
579    // name="schema.X">`) is robust to empty-doc defs — when the
580    // doc-arg is empty, `\hypertarget{schema.X}{}` produces no anchor
581    // element in the HTML, but the `<dt>` itself still needs the id
582    // so cross-page links to `#schema.X` resolve here. Patternadds
583    // get a separate `schema.add.<name>` so they don't clash with
584    // the canonical def's `schema.<name>`.
585    //
586    // Pass the name through `clean_anchor_name` so the id matches the
587    // hrefs that LaTeXML's `\hyperlink{\cleanhypername{schema.X}}`
588    // emits in `\elementref` / `\patternref` body text — `:` becomes
589    // `..`, otherwise xhtml:foo dt ids never resolve their cross-refs.
590    let cleaned_name = clean_anchor_name(name);
591    let new_id = if kind == "Add to" {
592      format!("schema.add.{}", cleaned_name)
593    } else {
594      format!("schema.{}", cleaned_name)
595    };
596    if !seen_ids.insert(new_id.clone()) {
597      // Another dt already claimed this id on this page — leave the
598      // duplicate as a plain kicker row.
599      continue;
600    }
601
602    let new_dt = format!(
603      concat!(
604        r##"<dt id="{id}" class="ltx_item schema-def">"##,
605        r##"<span class="ltx_tag ltx_tag_item">"##,
606        r##"<span class="schema-kind-chip {class}">{kind}</span>"##,
607        r##"<span class="schema-name">{name}</span>"##,
608        r##"<a class="schema-permalink" href="#{id}" "##,
609        r##"aria-label="permalink to this definition">§</a>"##,
610        r"</span></dt>",
611      ),
612      id = new_id,
613      class = class,
614      kind = kind,
615      name = name,
616    );
617    rewrites.push((dt_match.start(), dt_match.end(), new_dt));
618
619    // If a sibling `<a name="schema.X" id="schema.X">` exists (the
620    // \hypertarget rendering when doc was non-empty), strip its
621    // duplicate `id=` so the page doesn't carry two elements with
622    // the same id. Keep the `name=` so legacy `#name` URLs still
623    // resolve to the inner anchor's position too.
624    let matching = anchors
625      .iter()
626      .find(|a| a.start() >= dt_match.end() && a.start() < next_pos);
627    if let Some(a) = matching {
628      static STRIP_ID_RE: OnceLock<Regex> = OnceLock::new();
629      let strip_id_re = STRIP_ID_RE
630        .get_or_init(|| Regex::new(r#" id="schema\.[^"]+""#).unwrap());
631      let stripped = strip_id_re.replace(a.as_str(), "").into_owned();
632      rewrites.push((a.start(), a.end(), stripped));
633    }
634  }
635
636  rewrites.sort_by_key(|(s, _, _)| std::cmp::Reverse(*s));
637  let mut out = html.to_string();
638  for (s, e, replacement) in rewrites {
639    out.replace_range(s..e, &replacement);
640  }
641  out
642}
643
644// ---------- pass 3: sidebar item index ------------------------------------
645
646/// Collect every decorated `schema-def` on the page, group by kind,
647/// and inject a per-page kind index at the top of the navbar (above
648/// the cross-page `<nav class="ltx_TOC">` module list).
649fn inject_sidebar_index(html: &str) -> String {
650  if html.contains(r#"class="schema_module_index""#) {
651    return html.to_string();
652  }
653  static ITEM_RE: OnceLock<Regex> = OnceLock::new();
654  static NAVBAR_RE: OnceLock<Regex> = OnceLock::new();
655
656  // Matches the post-decorate `<dt class="schema-def">` heading
657  // (chip + name + permalink). Description-list shape, mirroring
658  // upstream Perl `latexmlman.sty`.
659  let item_re = ITEM_RE.get_or_init(|| {
660    Regex::new(concat!(
661      r#"<dt id="([^"]+)" class="ltx_item schema-def">"#,
662      r#"<span class="ltx_tag ltx_tag_item">"#,
663      r#"<span class="schema-kind-chip kind-([a-z-]+)">([^<]+)</span>"#,
664      r#"<span class="schema-name">([^<]+)</span>"#,
665    ))
666    .unwrap()
667  });
668  let navbar_re = NAVBAR_RE.get_or_init(|| {
669    Regex::new(concat!(
670      r#"(?s)(<nav class="ltx_page_navbar">"#,
671      r#"(?:[^<]*<a [^>]+rel="start"[^>]*>.*?</a>)?\s*)"#,
672      r#"(<nav class="ltx_TOC">)"#,
673    ))
674    .unwrap()
675  });
676
677  let mut seen: HashSet<(String, String)> = HashSet::new();
678  // Top-level navbar buckets, in order:
679  //   Patterns are SUBDIVIDED by their last dot-suffix
680  //   ("PATTERNS — ELEM", "PATTERNS — ATTRS", ...) so a long flat list
681  //   becomes a structured outline. Patterns whose name has no dot
682  //   land in the catch-all "PATTERNS — OTHER".
683  // Elements / Attribute / Add to render as single buckets.
684  let kinds_order = ["Pattern", "Element", "Attribute", "Add to"];
685  let kinds_plural: HashMap<&str, &str> = [
686    ("Pattern", "Patterns"),
687    ("Element", "Elements"),
688    ("Attribute", "Attributes"),
689    ("Add to", "Pattern Additions"),
690  ]
691  .iter()
692  .copied()
693  .collect();
694
695  // Insertion-ordered subgroups: for "Pattern", key = suffix. For
696  // every other kind, key = "" (single bucket).
697  let mut by_kind: HashMap<&str, Vec<(String, Vec<(String, String)>)>> = HashMap::new();
698
699  for cap in item_re.captures_iter(html) {
700    let dt_id = cap[1].to_string();
701    let kind = cap[3].to_string();
702    let name = cap[4].to_string();
703    if !kinds_plural.contains_key(kind.as_str()) {
704      continue;
705    }
706    if !seen.insert((kind.clone(), name.clone())) {
707      continue;
708    }
709    let bucket: &str = kinds_order.iter().find(|k| **k == kind.as_str()).copied().unwrap();
710    let subkey = if bucket == "Pattern" {
711      pattern_suffix(&name).unwrap_or("Other").to_string()
712    } else {
713      String::new()
714    };
715    let kind_subgroups = by_kind.entry(bucket).or_default();
716    let pos = kind_subgroups.iter().position(|(k, _)| k == &subkey);
717    let entries = match pos {
718      Some(idx) => &mut kind_subgroups[idx].1,
719      None => {
720        kind_subgroups.push((subkey, Vec::new()));
721        &mut kind_subgroups.last_mut().unwrap().1
722      },
723    };
724    entries.push((name, dt_id));
725  }
726
727  if by_kind.is_empty() {
728    return html.to_string();
729  }
730
731  let mut fragment = String::from(r#"<section class="schema_module_index">"#);
732  for kind in kinds_order {
733    let Some(subgroups) = by_kind.get_mut(kind) else { continue };
734    // Sort subgroups for Patterns alphabetically by suffix, with
735    // "Other" last; non-Pattern kinds keep insertion order (single
736    // empty-key entry).
737    if kind == "Pattern" {
738      subgroups.sort_by(|a, b| match (a.0.as_str(), b.0.as_str()) {
739        ("Other", _) => std::cmp::Ordering::Greater,
740        (_, "Other") => std::cmp::Ordering::Less,
741        (x, y) => x.cmp(y),
742      });
743    }
744    for (suffix, entries) in subgroups {
745      entries.sort_by(|a, b| a.0.cmp(&b.0));
746      let heading = if kind == "Pattern" {
747        format!("{} — {}", kinds_plural[kind], suffix.to_uppercase())
748      } else {
749        kinds_plural[kind].to_string()
750      };
751      fragment.push_str(&format!(
752        r#"<h6 class="schema_index_heading">{}</h6>"#,
753        html_escape(&heading)
754      ));
755      fragment.push_str(r#"<ul class="schema_index_list">"#);
756      for (name, dt_id) in entries.iter() {
757        fragment.push_str(&format!(
758          r##"<li><a href="#{}">{}</a></li>"##,
759          html_escape(dt_id),
760          html_escape(name),
761        ));
762      }
763      fragment.push_str("</ul>");
764    }
765  }
766  fragment.push_str("</section>");
767
768  let in_schema = r#"<h6 class="schema_in_schema">In schema</h6>"#;
769
770  let result = navbar_re.replace(html, |caps: &regex::Captures| {
771    format!("{}{}{}{}", &caps[1], fragment, in_schema, &caps[2])
772  });
773  result.into_owned()
774}
775
776/// Extract a pattern's "suffix" — the last dot-separated segment of
777/// its name. Returns None when the name has no dot (sidebar bucket
778/// then folds it into "Other"). Used to subdivide the PATTERNS bucket
779/// in the navbar so a long flat list becomes
780/// "PATTERNS — ELEM" / "PATTERNS — ATTRS" / etc.
781fn pattern_suffix(name: &str) -> Option<&str> {
782  // Skip obvious namespace-prefix names (we shouldn't see them in
783  // the Pattern bucket, but be defensive).
784  let after_colon = name.rsplit_once(':').map(|(_, t)| t).unwrap_or(name);
785  after_colon.rsplit_once('.').map(|(_, suffix)| suffix)
786}

788// ---------- helpers -------------------------------------------------------
789
790/// Mirror `latexmlman.sty`'s `\cleanhypername` macro for HTML anchor
791/// ids. The macro splits on `:` and rejoins with `..` (because `:`
792/// inside a TeX `\hypertarget` argument is brittle), so a raw name
793/// like `xhtml:header` ends up in HTML hrefs as `xhtml..header`. The
794/// schema-doc decorator builds dt ids from raw names, so without this
795/// transform every `\elementref{xhtml:foo}` / `\patternref{xhtml:foo}`
796/// produced by LaTeXML lands on a non-existent `#schema.xhtml..foo`
797/// while the dt sits at `#schema.xhtml:foo`. Underscores survive as
798/// `_` in both forms; only `:` needs the substitution.
799fn clean_anchor_name(name: &str) -> String {
800  name.replace(':', "..")
801}
802
803fn html_escape(s: &str) -> String {
804  s.replace('&', "&amp;")
805    .replace('<', "&lt;")
806    .replace('>', "&gt;")
807    .replace('"', "&quot;")
808}
