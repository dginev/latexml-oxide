36// Regex for parsing DefMathRewrite calls from .latexml files
37// Matches: DefMathRewrite( ... );

186  /// Restriction: convert_document runs on a single thread, and should never try branching out.
187  fn convert_document(&mut self, digested: Digested) -> Result<Document> {
188    note_begin("Building");
189    let mut document = Document::new();
190    {
191      // TODO: Can we disentangle the ownership to avoid the clone?
192      let paths_stored = state::get_search_paths();
193      let schema_paths = paths_stored
194        .iter()
195        .map(String::as_str)
196        .collect::<Vec<&str>>();
197      let default_model_load = model::with_schema_data(|schema_opt| match schema_opt {
198        None => true,
199        Some(v) => v.last() == Some(&arena::pin_static("LaTeXML")),
200      });
201      if default_model_load {
202        // Compile-time load of model AND indirect model
203        load_model!("LaTeXML");
204      } else {
205        // Eager-load at runtime
206        model::load_schema(schema_paths.as_slice())?; // If needed?
207      }
208      if state::has_search_paths() {
209        {
210          if state::lookup_bool("INCLUDE_COMMENTS") {
211            let paths_string = state::with_search_paths(|paths| {
212              paths
213                .iter()
214                .map(String::as_str)
215                .collect::<Vec<&str>>()
216                .join(",")
217            });
218            let attributes = map! {s!("paths") => paths_string};
219            document.insert_pi("latexml", Some(attributes))?;
220          }
221        }
222      }
223    }
224
225    for preload in &self.preload {
226      if preload.ends_with(".pool") {
227        continue;
228      }
229      let mut options: Option<String> = None;
230      LATEX_OPTION_REGEX.replace_all(preload, |refs: &Captures| -> String {
231        options = Some(refs.get(1).map_or("", |m| m.as_str()).to_string());
232        String::new()
233      });
234      if preload.ends_with(".cls") {
235        CLS_EXT_REGEX.replace_all(preload, "");
236        let attributes = map! {s!("class") => preload.to_string()};
237        document.insert_pi("latexml", Some(attributes))?;
238      } else {
239        STY_EXT_REGEX.replace_all(preload, "");
240        let attributes = map! {s!("package") => preload.to_string()};
241        document.insert_pi("latexml", Some(attributes))?;
242      }
243    }
244    Debug!("Doc absorb: {:?}", digested);
245
246    document.absorb(&digested, None)?;
247    note_end("Building");
248
249    // Load .latexml file if it exists alongside the source .tex file.
250    // Perl does this automatically during initialization; we do it post-build
251    // so the rewrite rules can be compiled against the built document.
252    if let Some(Stored::String(source_sym)) = state::lookup_value("SOURCEFILE") {
253      let source_path = arena::with(source_sym, |s| s.to_string());
254      // Replace .tex extension with .latexml
255      let latexml_path = if source_path.ends_with(".tex") {
256        source_path.replace(".tex", ".latexml")
257      } else {
258        format!("{}.latexml", source_path)
259      };
260      if Path::new(&latexml_path).exists() {
261        let _ = load_latexml_file(&latexml_path);
262      }
263    }
264
265    let has_rewrites = state::has_value("DOCUMENT_REWRITE_RULES");
266    if has_rewrites {
267      note_begin("Rewriting");
268      document.mark_xmnode_visibility()?;
269      document.load_labels_for_rewrite()?;
270      // TODO: What is the right way to do rewrites in a daemon-safe manner?
271      if let Some(Stored::VecDequeStored(rules)) = state::remove_value("DOCUMENT_REWRITE_RULES") {
272        if let Some(root) = document.get_document().get_root_element() {
273          // Step 1: copy the rules locally through Rc, to be able to invoke them with mutable
274          // state. (TODO: obviously, this could be avoided if they never needed mutable
275          // state. When do they?)
276          let mut rewrites = Vec::new();
277          for rule in rules {
278            if let Stored::Rewrite(mut rewrite_rule) = rule {
279              rewrite_rule.compile_clauses(&mut document);
280              rewrites.push(rewrite_rule); // clone the Rc
281            }
282          }
283          // Step 2: invoke the rewrite rules
284          for mut rewrite_rule in rewrites {
285            rewrite_rule.invoke(&mut document, &root)?;
286          }
287        }
288      }
289      note_end("Rewriting");
290    }
291
292    // Apply \lxDeclare declarations: set roles/names/meanings on matching XMTok elements.
293    // Must run BEFORE math parsing so the parser sees the updated roles.
294    apply_lx_declarations(&mut document);
295
296    if !state::get_nomathparse_flag() {
297      let mut parser = MathParser::default();
298      parser.parse_math(&mut document)?;
299      // Post-parse: apply kludge to failed XMath nodes, then mark as unparsed
300      if !parser.failed_xmath_ids.is_empty() {
301        // Collect failed XMath nodes for kludge processing
302        let mut failed_xmaths = Vec::new();
303        for mut math_node in document.findnodes("descendant-or-self::ltx:Math[not(@text)]", None) {
304          for xmath_child in document.findnodes("ltx:XMath", Some(&math_node)) {
305            if parser.failed_xmath_ids.contains(&xmath_child.to_hashable()) {
306              failed_xmaths.push(xmath_child.clone());
307              document.add_class(&mut math_node, "ltx_math_unparsed")?;
308              break;
309            }
310          }
311        }
312        // Apply kludge to each failed XMath — balance OPEN/CLOSE delimiters
313        for mut xmath in failed_xmaths {
314          parse_kludge(&mut xmath, &mut document);
315        }
316      }
317      // Renumber xml:ids inside parsed XMath subtrees to be sequential in document
318      // order. The Marpa parser explores multiple parse alternatives, consuming ID
319      // counter slots for pruned nodes. This pass reassigns IDs post-parse.
320      renumber_math_ids(&mut document);
321    }
322
323    note_begin("Finalizing");
324    document.finalize()?;
325    note_end("Finalizing");
326    // Perl core produces role="UNKNOWN" for single-letter math tokens.
327    // Per-document .latexml files set role="ID" via DefMathRewrite BEFORE parsing.
328    // We do NOT apply a blanket conversion — roles are set by rewrite rules only.
329    // Cleanup unreferenced xml:ids on XMTok elements generated by the math parser.
330    // Must run after finalize (which includes prune_xmduals that may transfer ids).
331    document.cleanup_unreferenced_xmtok_ids();
332    Ok(document)
333  }

335  fn digest_internal(&mut self) -> Result<Digested> {
336    let mut boxes = Vec::new();
337    while gullet::has_more_input() {
338      let next_bodies: Vec<Digested> = stomach::digest_next_body(None)?;
339      // info!(target:"core:digest_next_body", "\n{:?}\n----\n",next_bodies);
340      boxes.extend(next_bodies);
341    }
342    gullet::flush();
343    Ok(Digested::from(List::new(boxes)))
344  }

355  /// Restriction: `digest_file` runs on a single thread, and should never try branching out.
356  fn digest_file(&mut self, mut request: String, options: DigestionOptions) -> Result<Digested> {
357    let mut dir = String::new();
358    let name;
359    // let mut ext = String::new();
360    let mode = match options.mode {
361      None => DigestionMode::TeX,
362      Some(m) => m,
363    };
364
365    if pathname::is_literaldata(&request) {
366      // ext = mode.extension();
367      name = s!("Anonymous String");
368    } else if pathname::is_url(&request) {
369      // ext = mode.extension();
370      name = request.clone();
371    } else {
372      let ext_str = s!(".{}", mode.extension());
373      let request_base = if request.ends_with(&ext_str) {
374        request[0..request.len() - ext_str.len()].to_string()
375      } else {
376        request
377      };
378
379      if let Some(pathname) = pathname::find(&request_base, PathnameFindOptions {
380        extensions: Some(vec![mode.extension(), String::new()]),
381        ..PathnameFindOptions::default()
382      }) {
383        request = pathname;
384        dir = pathname::directory(&request);
385        name = pathname::file_stem(&request);
386      // ext = pathname::extension(&request);
387      } else {
388        let message = s!("Can't find {} file {} ", mode, request_base);
389        fatal!(Core, MissingFile, message);
390      }
391    }
392    note_begin(&s!("Digesting {} {}", mode, name));
393    let main_pool = s!("{}.pool", mode);
394    let noinitialize = options.noinitialize.unwrap_or(false);
395    if !noinitialize {
396      let mut preloads = vec![main_pool];
397      preloads.extend(self.preload.clone());
398      self.initialize_singletons(preloads)?;
399    }
400    {
401      if !pathname::is_literaldata(&request) {
402        state::assign_value("SOURCEFILE", request.clone(), None);
403      }
404      if !dir.is_empty() {
405        state::assign_value("SOURCEDIRECTORY", dir.clone(), None);
406      }
407      state::search_paths_push_front(dir.clone());
408      state::graphics_paths_push_front(dir);
409      state::install_definition(
410        Stored::Expandable(Rc::new(Expandable {
411          cs: T_CS!("\\jobname"),
412          paramlist: None,
413          expansion: Tokens::new(Explode!(name)).into(),
414          ..Expandable::default()
415        })),
416        None,
417      );
418    }
419
420    // Reverse order, since last opened is first read!
421    if let Some(postamble) = options.postamble {
422      self.load_postamble(postamble);
423    }
424
425    {
426      // Make sure the stomach trick is used very *tightly*, always with a surrounding scope.
427      input_content(&request, InputOptions::default())?;
428    }
429
430    if let Some(preamble) = options.preamble {
431      self.load_preamble(preamble);
432    }
433
434    // Now for the Hacky part for BibTeX!!!
435    // if mode == DigestionMode::BibTeX {
436    //   let bib = LaTeXML::Pre::BibTeX->newFromGullet($name, $state->getStomach->getGullet);
437    //   LaTeXML::Package::InputContent("literal:" . $bib->toTeX);
438    // }
439
440    let list = self.digest_internal()?;
441    note_end(&s!("Digesting {} {}", mode, name));
442    Ok(list)
443  }
444}

446/// Load a `.latexml` file alongside a `.tex` source file.
447/// Parses `DefMathRewrite(...)` calls and registers them as rewrite rules.
448/// Perl loads these automatically; this provides the equivalent for Rust tests.
449///
450/// Supported patterns:
451///   - Single character: `match => 'a'` -> XPath on XMTok text content
452///   - Complex patterns (e.g. `\hat{f}`, `f_D`, `f_\WildCard`): skipped
453///   - `scope => 'label:...'`: scoped rewrites via label lookup
454///   - `attributes => { role => 'FUNCTION' }`: sets role (and optionally name/meaning)
455fn load_latexml_file(path: &str) -> Result<()> {
456  use latexml_core::rewrite::{RewriteClause, RewriteOperator, RewritePattern};
457
458  let content = match std::fs::read_to_string(path) {
459    Ok(c) => c,
460    Err(_) => return Ok(()), // File doesn't exist or can't be read
461  };
462
463  for cap in DEF_MATH_REWRITE_RE.captures_iter(&content) {
464    let body = &cap[1];
465
466    // Extract match pattern
467    let match_str = match MATCH_RE.captures(body) {
468      Some(m) => m[1].to_string(),
469      None => continue, // No match clause, skip
470    };
471
472    // Skip complex patterns: anything with \, _, {, } in the match string
473    // These need full token-level matching (digestion + DOM -> XPath) which we
474    // don't implement yet.
475    if match_str.contains('\\') || match_str.contains('_') || match_str.contains('{') {
476      continue;
477    }
478
479    // Build attributes map from the attributes => { ... } section
480    let mut attrs = HashMap::default();
481    if let Some(role_cap) = ROLE_RE.captures(body) {
482      attrs.insert("role".to_string(), role_cap[1].to_string());
483    }
484    if let Some(name_cap) = NAME_ATTR_RE.captures(body) {
485      attrs.insert("name".to_string(), name_cap[1].to_string());
486    }
487    if let Some(meaning_cap) = MEANING_RE.captures(body) {
488      attrs.insert("meaning".to_string(), meaning_cap[1].to_string());
489    }
490    if attrs.is_empty() {
491      continue; // No attributes to set
492    }
493
494    // Build the XPath for this match.
495    // For math mode, Perl's compile_match digests "$a$", builds DOM, then generates:
496    //   descendant-or-self::ltx:XMTok[text()='a'][@_pvis and @_cvis]
497    // We generate the equivalent XPath directly for single-character matches.
498    let xpath = format!(
499      "descendant-or-self::ltx:XMTok[text()='{}'][@_pvis and @_cvis]",
500      match_str
501    );
502
503    // Check for optional scope
504    let scope_str = SCOPE_RE.captures(body).map(|s| s[1].to_string());
505
506    // Build the rewrite rule with pre-compiled clauses
507    let mut clauses = Vec::new();
508
509    // Add scope clause if present (compiled during compile_clauses phase)
510    if let Some(ref scope) = scope_str {
511      clauses.push(RewriteClause::new_uncompiled(
512        RewriteOperator::Scope,
513        RewritePattern::String(scope.clone()),
514      ));
515    }
516
517    // Add match clause (pre-compiled as XPath string)
518    clauses.push(RewriteClause::new_uncompiled(
519      RewriteOperator::Match,
520      RewritePattern::String(xpath),
521    ));
522
523    // Add attributes clause
524    clauses.push(RewriteClause::new_compiled(
525      RewriteOperator::Attributes,
526      RewritePattern::String(String::new()),
527    ));
528
529    let rewrite = Rewrite {
530      options: RewriteOptions {
531        attributes_map: Some(attrs),
532        is_math: true,
533        select_count: Some(1),
534        ..RewriteOptions::default()
535      },
536      clauses,
537    };
538
539    state::push_value("DOCUMENT_REWRITE_RULES", rewrite)?;
540  }
541
542  Ok(())
543}

545/// Apply \lxDeclare declarations to the document.
546/// Simple fast-path: matches single-token patterns in XMTok elements
547/// and sets role/name/meaning attributes.
548fn apply_lx_declarations(document: &mut Document) {
549  let decls_str = match state::lookup_value("LATEXML_DECLARATIONS") {
550    Some(Stored::String(s)) => arena::with(s, |r| r.to_string()),
551    _ => return,
552  };
553  if decls_str.is_empty() {
554    return;
555  }
556
557  // Parse declarations: "token_text\trole\tname\tmeaning" per line
558  let declarations: Vec<(&str, &str, &str, &str)> = decls_str
559    .lines()
560    .filter_map(|line| {
561      let parts: Vec<&str> = line.splitn(4, '\t').collect();
562      if parts.len() == 4 {
563        Some((parts[0], parts[1], parts[2], parts[3]))
564      } else {
565        None
566      }
567    })
568    .collect();
569
570  if declarations.is_empty() {
571    return;
572  }
573
574  // Find all XMTok elements in the document and apply matching declarations
575  let xmtoks = document.findnodes("descendant-or-self::ltx:XMTok", None);
576  for mut tok in xmtoks {
577    let content = tok.get_content();
578    let tok_name = tok.get_attribute("name").unwrap_or_default();
579    for &(pattern, role, name, meaning) in &declarations {
580      // Match by content text, or by XMTok name attribute (for CS patterns like \circ)
581      let matches = content == pattern
582        || (!tok_name.is_empty() && pattern.starts_with('\\') && pattern[1..] == tok_name);
583      if matches {
584        if !role.is_empty() {
585          let _ = tok.set_attribute("role", role);
586        }
587        if !name.is_empty() {
588          let _ = tok.set_attribute("name", name);
589        }
590        if !meaning.is_empty() {
591          let _ = tok.set_attribute("meaning", meaning);
592        }
593        break; // First matching declaration wins
594      }
595    }
596  }
597}

599/// Fallback parser for unparseable math expressions.
600/// Perl: MathParser.pm parse_kludge().
601/// Balances OPEN/CLOSE delimiters by wrapping matched groups in XMWrap.
602/// Uses document.wrap_nodes for proper namespace handling.
603fn parse_kludge(mathnode: &mut libxml::tree::Node, document: &mut Document) {
604  use latexml_math_parser::get_grammatical_role;
605  let children: Vec<libxml::tree::Node> = mathnode
606    .get_child_elements()
607    .into_iter()
608    .filter(|n| n.get_name() != "XMHint")
609    .collect();
610  if children.is_empty() {
611    return;
612  }
613
614  // Phase 1: Find OPEN/CLOSE pairs and wrap each group using wrap_nodes.
615  // We process innermost pairs first (like matching parentheses).
616  // Iterate until no more OPEN/CLOSE pairs are found.
617  let mut changed = true;
618  while changed {
619    changed = false;
620    let elems = mathnode.get_child_elements();
621    // Find the innermost OPEN that has a matching CLOSE
622    let mut open_idx = None;
623    for (i, n) in elems.iter().enumerate() {
624      let role = get_grammatical_role(n);
625      if role == "OPEN" {
626        open_idx = Some(i);
627      } else if role == "CLOSE" && open_idx.is_some() {
628        // Found innermost OPEN..CLOSE pair
629        let start = open_idx.unwrap();
630        let end = i;
631        // Collect the nodes between OPEN and CLOSE (inclusive)
632        let group: Vec<libxml::tree::Node> =
633          elems[start..=end].iter().cloned().collect();
634        if group.len() > 1 {
635          let _ = document.wrap_nodes("ltx:XMWrap", group);
636          changed = true;
637        }
638        break; // Restart after wrapping
639      }
640    }
641    // If we found an unmatched OPEN (no CLOSE), wrap OPEN through end
642    if !changed && open_idx.is_some() {
643      let elems = mathnode.get_child_elements();
644      let start = open_idx.unwrap().min(elems.len().saturating_sub(1));
645      if start < elems.len() {
646        let group: Vec<libxml::tree::Node> =
647          elems[start..].iter().cloned().collect();
648        if group.len() > 1 {
649          let _ = document.wrap_nodes("ltx:XMWrap", group);
650          changed = true;
651        }
652      }
653    }
654  }
655  // Phase 2: Unwrap ALL top-level XMWrap nodes.
656  // Perl: foreach pair, if kludge is XMWrap, extract children (not the wrapper).
657  // This preserves flat structure for simple expressions like <1>, [1], (1).
658  loop {
659    let top_elems = mathnode.get_child_elements();
660    let wrap_opt = top_elems.into_iter().find(|n| n.get_name() == "XMWrap");
661    if let Some(wrap) = wrap_opt {
662      let _ = document.unwrap_nodes(wrap);
663    } else {
664      break;
665    }
666  }
667}

669/// Renumber xml:ids inside parsed XMath subtrees so they are sequential in
670/// document order. The Marpa parser explores multiple parse alternatives,
671/// consuming ID counter slots for pruned nodes (e.g. m1.1, m1.7, m1.12
672/// instead of m1.1, m1.2, m1.3). This pure post-processing pass reassigns
673/// IDs after all pruning is complete.
674///
675/// Optimized: single DFS walk per XMath (not XPath), O(1) parent-prefix
676/// lookup via ID string parsing, and allocation reuse across Math nodes.
677fn renumber_math_ids(document: &mut Document) {
678  let xml_ns = "http://www.w3.org/XML/1998/namespace";
679  let math_nodes = document.findnodes("descendant-or-self::ltx:Math[@text]", None);
680
681  // Reuse allocations across Math nodes
682  let mut id_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
683  let mut idref_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
684  let mut id_map: std::collections::HashMap<String, String> = std::collections::HashMap::new();
685  let mut counters: std::collections::HashMap<String, u32> = std::collections::HashMap::new();
686
687  for mut math_node in math_nodes {
688    let math_id = match math_node.get_attribute_ns("id", xml_ns) {
689      Some(id) => id,
690      None => continue,
691    };
692
693    let xmath_nodes = document.findnodes("ltx:XMath", Some(&math_node));
694    for xmath in xmath_nodes {
695      id_entries.clear();
696      idref_entries.clear();
697      id_map.clear();
698      counters.clear();
699
700      // Single DFS walk collects both xml:id and idref nodes in document order
701      renumber_collect_dfs(&xmath, xml_ns, &mut id_entries, &mut idref_entries);
702      if id_entries.is_empty() {
703        continue;
704      }
705
706      // Build old→new mapping. Parent prefix is derived by string parsing:
707      // old_id "S0.Ex1.m1.3.1" → rfind('.') → parent old_id "S0.Ex1.m1.3"
708      // → look up in id_map for renumbered parent. O(1) per node.
709      let mut any_changed = false;
710      for (_node, old_id) in &id_entries {
711        let parent_new_id = if let Some(dot_pos) = old_id.rfind('.') {
712          let parent_old = &old_id[..dot_pos];
713          id_map
714            .get(parent_old)
715            .cloned()
716            .unwrap_or_else(|| parent_old.to_string())
717        } else {
718          math_id.clone()
719        };
720        let counter = counters.entry(parent_new_id.clone()).or_insert(0);
721        *counter += 1;
722        let new_id = format!("{parent_new_id}.{counter}");
723        if new_id != *old_id {
724          any_changed = true;
725        }
726        id_map.insert(old_id.clone(), new_id);
727      }
728
729      if !any_changed {
730        continue;
731      }
732
733      // Apply new xml:ids in TWO passes to avoid idstore collisions.
734      // A new id like "m1.1" would collide with an old "m1.1" still in the
735      // idstore if we interleave unrecord+record. Strip all first, then assign.
736      let mut nodes_to_update: Vec<(libxml::tree::Node, String)> = Vec::new();
737      for (mut node, old_id) in id_entries.drain(..) {
738        if let Some(new_id) = id_map.get(&old_id) {
739          if new_id != &old_id {
740            document.unrecord_id(&old_id);
741            let _ = node.remove_attribute("xml:id");
742            let _ = node.remove_attribute_ns("id", xml_ns);
743            nodes_to_update.push((node, new_id.clone()));
744          }
745        }
746      }
747      for (mut node, new_id) in nodes_to_update {
748        let _ = document.set_attribute(&mut node, "xml:id", &new_id);
749      }
750
751      // Update idrefs
752      for (mut node, old_idref) in idref_entries.drain(..) {
753        if let Some(new_idref) = id_map.get(&old_idref) {
754          if new_idref != &old_idref {
755            let _ = node.set_attribute("idref", new_idref);
756          }
757        }
758      }
759
760      // Reset _ID_counter__ on the Math node to the final count
761      if let Some(&final_count) = counters.get(&math_id) {
762        let _ = math_node.set_attribute("_ID_counter__", &final_count.to_string());
763      }
764    }
765  }
766}

768/// DFS walk collecting nodes with xml:id and idref attributes in document order.
769/// Stops at nested `Math` elements (which have their own parsing scope).
770fn renumber_collect_dfs(
771  node: &libxml::tree::Node,
772  xml_ns: &str,
773  id_entries: &mut Vec<(libxml::tree::Node, String)>,
774  idref_entries: &mut Vec<(libxml::tree::Node, String)>,
775) {
776  if let Some(id) = node.get_attribute_ns("id", xml_ns) {
777    id_entries.push((node.clone(), id));
778  }
779  if let Some(idref) = node.get_attribute("idref") {
780    idref_entries.push((node.clone(), idref));
781  }
782  for child in node.get_child_elements() {
783    // Skip nested Math elements — they have their own parsing scope
784    // and their IDs (with prefixes like "m") are document-construction IDs.
785    if child.get_name() == "Math" {
786      continue;
787    }
788    renumber_collect_dfs(&child, xml_ns, id_entries, idref_entries);
789  }
790}
