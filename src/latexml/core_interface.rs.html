98/// Parse a preload spec into `(name, ext, options)`.
99///
100/// Mirrors Perl `Core.pm:initializeState` (regexes
101/// `s/^\[([^\]]*)\]//` then `s/\.(\w+)$//`): the option bracket
102/// comes at the *front*, e.g. `[ids,mathlexemes]latexml.sty`.
103/// Defaults to `ext = "sty"` when the spec has no `.<ext>` suffix.
104pub(crate) fn parse_preload_spec(preload: &str) -> (String, String, Vec<String>) {
105  let (base, options) = match preload
106    .strip_prefix('[')
107    .and_then(|rest| rest.find(']').map(|end| (&rest[..end], &rest[end + 1..])))
108  {
109    Some((opts_str, rest)) => {
110      let opts: Vec<String> = opts_str
111        .split(',')
112        .map(|s| s.trim().to_string())
113        .filter(|s| !s.is_empty())
114        .collect();
115      (rest.to_string(), opts)
116    },
117    None => (preload.to_string(), vec![]),
118  };
119  let (name, ext) = match base.rfind('.') {
120    Some(pos) => (base[..pos].to_string(), base[pos + 1..].to_string()),
121    None => (base.clone(), String::from("sty")),
122  };
123  (name, ext, options)
124}

127  fn initialize_singletons(&mut self, preloads: Vec<String>) -> Result<()> {
128    // reset the error REPORT singleton
129    error::initialize_report();
130    // reset localized variables (if_frames, current_token, align state, etc.)
131    latexml_core::common::local_assignments::initialize_localized();
132    // now handle conversion state
133    gullet::initialize_gullet();
134    stomach::initialize_stomach();
135    // should we reset the model also?
136    model::initialize_model();
137    // let paths = state::search_paths;
138    let dump_path = LATEXML_DUMP.clone();
139    state::assign_value("InitialPreloads", true, Some(Scope::Global));
140    for preload in preloads {
141      let (name, ext, options) = parse_preload_spec(&preload);
142      let handleoptions = ext == "sty" || ext == "cls";
143      // Pass package options via state (Perl: \PassOptionsToPackage equivalent).
144      // Match `\PassOptionsToPackage` at latex_constructs.rs L3838-3842: push the
145      // `Vec<String>` through `push_value` so it lands as a `Stored::Strings`
146      // batch inside the `opt@<name>.<ext>` `VecDequeStored`. The batch shape is
147      // what `collect_syms` (binding/content.rs L1157) flattens when
148      // `\ProcessOptions*` enumerates declared options; storing a single
149      // comma-joined `Stored::String("opt1,opt2")` instead silently bypasses
150      // every `DeclareOption!` site, so e.g. dvipsnames/svgnames/x11names
151      // palettes never load — visible as `Error:unexpected:Apricot Can't find
152      // color named 'Apricot'; assuming Black` on a `[dvipsnames]color.sty`
153      // preload.
154      if !options.is_empty() {
155        let opt_key = format!("opt@{name}.{ext}");
156        state::push_value(&opt_key, options)?;
157      }
158      input_definitions(&name, InputDefinitionOptions {
159        extension: Some(ext.into()),
160        handleoptions,
161        ..InputDefinitionOptions::default()
162      })?;
163    }
164    state::assign_value("InitialPreloads", false, Some(Scope::Global));
165
166    // Load kernel dump AFTER pools (provides TeX/LaTeX macros the pools skipped).
167    if let Some(ref dump_path) = dump_path {
168      let path = Path::new(dump_path);
169      if path.exists() {
170        // Rust-native tab-separated format (from --init mode). The
171        // Perl-format `dump_loader` was deleted 2026-04-18 (dead code —
172        // we never consumed Perl-generated dumps).
173        let result = latexml_core::dump_reader::load_native_dump(path);
174        match result {
175          Ok(count) => {
176            eprintln!(
177              "[latexml-oxide] Loaded {} kernel definitions from {}",
178              count,
179              path.display()
180            );
181          },
182          Err(e) => {
183            eprintln!("[latexml-oxide] Warning: failed to load dump: {}", e);
184          },
185        }
186      }
187    }
188    Ok(())
189  }

191  // TODO: We should choose between this function or digest_file, rather than implement twice,
192  // right?
193  fn digest(
194    &mut self,
195    request: String,
196    preamble: Option<String>,
197    postamble: Option<String>,
198    mode: Option<DigestionMode>,
199    _no_init: bool,
200  ) -> Result<Digested> {
201    let mut _ext = match &mode {
202      Some(m) => Some(m.extension()),
203      None => Some(DigestionMode::TeX.extension()),
204    };
205    let mut dir_opt = None;
206
207    // Canonicalize relative paths so `Path::parent()` gives a real directory.
208    // `Path::new("foo.tex").parent()` returns `Some("")` (empty string) which
209    // poisons SEARCHPATHS / SOURCEDIRECTORY: an empty-string search-path
210    // entry resolves files via cwd-name with no normalization, changing the
211    // order in which resource files (e.g. `ts1enc.def` vs `t1enc.def`) are
212    // discovered. Concrete symptom: TS1 fontmap leaks into control-sequence
213    // construction → `cn` characters become `⚮♪` → `\c@cn` undefined →
214    // 381-error cascade (paper 0709.2868). Canonicalizing matches Perl's
215    // `File::Spec->splitpath` behavior which always yields a real directory.
216    let canonical_request = if pathname::is_literaldata(&request) || pathname::is_url(&request) {
217      request.clone()
218    } else {
219      std::fs::canonicalize(&request)
220        .ok()
221        .and_then(|p| p.to_str().map(String::from))
222        .unwrap_or_else(|| request.clone())
223    };
224    let name = if pathname::is_literaldata(&request) {
225      s!("Anonymous String")
226    } else if pathname::is_url(&request) {
227      request.clone()
228    } else {
229      let path = Path::new(&canonical_request);
230      dir_opt = path.parent();
231      match path.file_stem() {
232        None => String::from("missing_name"),
233        Some(pf) => pf.to_str().unwrap().to_string(),
234      }
235    };
236    // else {
237    //   $self->withState(sub {
238    //       Fatal('missing_file', $request, undef, "Can't find $mode file $request"); }); } }
239    // };
240    let digestion_note = s!("Digesting {}", name);
241    note_begin(&digestion_note);
242    // $self->initializestate::$mode . ".pool", @{ $$self{preload} || [] }) unless
243    // $options{noinitialize};
244    if !pathname::is_literaldata(&request) {
245      state::assign_value("SOURCEFILE", arena::pin(&request), None);
246    }
247    if let Some(dir) = dir_opt {
248      let dir_str = dir.to_str().unwrap_or(".");
249      // Perl Core.pm L195-200 unshifts the SOURCE file's directory onto
250      // SEARCHPATHS so subsequent `\input`-style lookups resolve relative
251      // to the main file. When `canonicalize` succeeded `dir_str` is the
252      // absolute parent; if it failed (e.g. file not on disk yet at the
253      // time we resolved — unusual for normal latexml CLI invocations
254      // but possible for `literal:` etc.) `dir_str` may be empty and an
255      // empty entry on SEARCHPATHS is useless. Fall back to CWD in that
256      // case so paper-local files (`\input{Chapter/Abstract}` etc.) are
257      // discoverable. Witness: arXiv:2604.09744, 2603.04457 (papers
258      // bundling subdirectory `\subimport` chains).
259      let resolved_dir = if dir_str.is_empty() {
260        std::env::current_dir()
261          .ok()
262          .and_then(|cwd| cwd.to_str().map(String::from))
263          .unwrap_or_else(|| ".".to_string())
264      } else {
265        dir_str.to_string()
266      };
267      state::assign_value("SOURCEDIRECTORY", arena::pin(&resolved_dir), None);
268      // Perl Core.pm L195-200: `$state->unshiftValue(SEARCHPATHS => $dir)`.
269      // `unshift` puts the source dir at the FRONT so it's the new "lead"
270      // — the same lead `\lx@append@path` reads as the basis for
271      // appended subdir paths. Pushing to BACK (`add_search_path`) left
272      // the lead as whatever the CLI's `--path` provided (typically
273      // `ar5iv-bindings/bindings`); `\subimport{Chapter/}{Abstract}`
274      // then appended Chapter/ to ar5iv-bindings/bindings instead of to
275      // the paper's directory, and `\input{Abstract}` couldn't resolve.
276      // Witness: arXiv:2604.09744, 2603.04457.
277      state::search_paths_push_front(resolved_dir);
278    }
279    //   if defined $dir && !grep { $_ eq $dir } @{ $state->lookupValue('SEARCHPATHS') };
280    // $state->unshiftValue(GRAPHICSPATHS => $dir)
281
282    // if defined $dir && !grep { $_ eq $dir } @{ $state->lookupValue('GRAPHICSPATHS') };
283
284    let name_copy = name.clone();
285    state::install_definition(
286      Stored::Expandable(Rc::new(Expandable {
287        cs: T_CS!("\\jobname"),
288        paramlist: None,
289        expansion: Tokens::new(Explode!(name_copy)).into(),
290        ..Expandable::default()
291      })),
292      None,
293    );
294
295    // Reverse order, since last opened is first read!
296    // (Perl: Core.pm L154-157 in `digestFile`.)
297    if let Some(postamble) = postamble {
298      self.load_postamble(postamble);
299    }
300    input_content(&request, InputOptions::default())?;
301    if let Some(preamble) = preamble {
302      self.load_preamble(preamble);
303    }
304
305    // Now for the Hacky part for BibTeX!!!
306    // Perl `Core.pm` L160-162: drain the .bib mouth via the Pre::BibTeX
307    // parser, register each entry in the bibtex.rs thread-local
308    // registry, and push back a `literal:` wrapper that produces a
309    // `\begin{bibtex@bibliography}...\end{bibtex@bibliography}` block
310    // for the digester to process.
311    if matches!(mode, Some(DigestionMode::BibTeX)) {
312      use latexml_engine::pre_bibtex::PreBibTeX;
313      let mut bib = PreBibTeX::new_from_gullet(&name);
314      match bib.to_tex() {
315        Ok(tex) => {
316          input_content(&s!("literal:{tex}"), InputOptions::default())?;
317        },
318        Err(parse_err) => {
319          Error!(
320            "bibtex",
321            "parse_failed",
322            s!("Failed to parse BibTeX file {}: {:?}", name, parse_err)
323          );
324        },
325      }
326    }
327
328    let list = self.digest_internal()?;
329    note_end(&digestion_note);
330
331    Ok(list)
332  }

341  /// Restriction: convert_document runs on a single thread, and should never try branching out.
342  fn convert_document(&mut self, digested: Digested) -> Result<Document> {
343    note_begin("Building");
344    let mut document = Document::new();
345    {
346      // TODO: Can we disentangle the ownership to avoid the clone?
347      let paths_stored = state::get_search_paths();
348      let schema_paths = paths_stored
349        .iter()
350        .map(String::as_str)
351        .collect::<Vec<&str>>();
352      let default_model_load = model::with_schema_data(|schema_opt| match schema_opt {
353        None => true,
354        Some(v) => v.last() == Some(&pin!("LaTeXML")),
355      });
356      if default_model_load {
357        // Compile-time load of model AND indirect model. Single
358        // shared instantiation lives at `crate::load_latexml_default_model`
359        // so LTO can keep exactly one `_ModelLoader::build_model` in
360        // the final binary (~600 KiB per copy otherwise).
361        crate::load_latexml_default_model();
362      } else {
363        // Eager-load at runtime
364        model::load_schema(schema_paths.as_slice())?; // If needed?
365      }
366      if state::has_search_paths() {
367        {
368          if state::lookup_bool("INCLUDE_COMMENTS") {
369            let paths_string = state::with_search_paths(|paths| {
370              paths
371                .iter()
372                .map(String::as_str)
373                .collect::<Vec<&str>>()
374                .join(",")
375            });
376            let attributes = map! {s!("searchpaths") => paths_string};
377            document.insert_pi("latexml", Some(attributes))?;
378          }
379        }
380      }
381    }
382
383    for preload in &self.preload {
384      if preload.ends_with(".pool") {
385        continue;
386      }
387      let mut options: Option<String> = None;
388      LATEX_OPTION_REGEX.replace_all(preload, |refs: &Captures| -> String {
389        options = Some(refs.get(1).map_or("", |m| m.as_str()).to_string());
390        String::new()
391      });
392      if preload.ends_with(".cls") {
393        CLS_EXT_REGEX.replace_all(preload, "");
394        let attributes = map! {s!("class") => preload.to_string()};
395        document.insert_pi("latexml", Some(attributes))?;
396      } else {
397        STY_EXT_REGEX.replace_all(preload, "");
398        let attributes = map! {s!("package") => preload.to_string()};
399        document.insert_pi("latexml", Some(attributes))?;
400      }
401    }
402    Debug!("Doc absorb: {:?}", digested);
403
404    document.absorb(&digested, None)?;
405    note_end("Building");
406
407    // Load .latexml file if it exists alongside the source .tex file.
408    // Perl does this automatically during initialization; we do it post-build
409    // so the rewrite rules can be compiled against the built document.
410    if let Some(Stored::String(source_sym)) = state::lookup_value("SOURCEFILE") {
411      let source_path = arena::with(source_sym, |s| s.to_string());
412      // Replace .tex extension with .latexml
413      let latexml_path = if source_path.ends_with(".tex") {
414        source_path.replace(".tex", ".latexml")
415      } else {
416        format!("{}.latexml", source_path)
417      };
418      if Path::new(&latexml_path).exists() {
419        let _ = load_latexml_file(&latexml_path);
420      }
421    }
422
423    let has_rewrites = state::has_value("DOCUMENT_REWRITE_RULES");
424    if has_rewrites {
425      let _gp_rewrite = latexml_core::telemetry::phase(latexml_core::telemetry::Phase::Rewrite);
426      note_begin("Rewriting");
427      document.mark_xmnode_visibility()?;
428      document.load_labels_for_rewrite()?;
429      // TODO: What is the right way to do rewrites in a daemon-safe manner?
430      if let Some(Stored::VecDequeStored(rules)) = state::remove_value("DOCUMENT_REWRITE_RULES")
431        && let Some(root) = document.get_document().get_root_element()
432      {
433        // Step 1: copy the rules locally through Rc, to be able to invoke them with mutable
434        // state. (TODO: obviously, this could be avoided if they never needed mutable
435        // state. When do they?)
436        let mut rewrites = Vec::new();
437        for rule in rules {
438          if let Stored::Rewrite(mut rewrite_rule) = rule {
439            rewrite_rule.compile_clauses(&mut document);
440            rewrites.push(rewrite_rule);
441          }
442        }
443        // 31 rules compiled for declare test; XPath matching issue prevents application
444        // Step 2: invoke the rewrite rules
445        // R35.D instrumentation: print per-rule timing if
446        // LATEXML_REWRITE_TIMING=1. Logs BEFORE the rule runs so we can
447        // identify the rule that hangs (the timeout watchdog kills
448        // mid-rule otherwise).
449        let trace_all = std::env::var_os("LATEXML_REWRITE_TIMING").is_some();
450        let n_rules = rewrites.len();
451        for (idx, mut rewrite_rule) in rewrites.into_iter().enumerate() {
452          // Build a useful one-line hint from the rule's options. The
453          // Debug impl on RewriteOptions is `<RewriteOptions>` only,
454          // so reach into the fields directly.
455          let opts = &rewrite_rule.options;
456          let mut xpath_hint = format!(
457            "select={:?} xpath={:?} regexp={:?} scope={:?} label={:?} clauses={}",
458            opts
459              .select
460              .as_deref()
461              .map(|s| s.chars().take(60).collect::<String>()),
462            opts
463              .xpath
464              .as_deref()
465              .map(|s| s.chars().take(60).collect::<String>()),
466            opts
467              .regexp
468              .as_deref()
469              .map(|s| s.chars().take(60).collect::<String>()),
470            opts.scope.as_ref().map(|_| "<scope>"),
471            opts.label.as_deref(),
472            rewrite_rule.clauses.len(),
473          );
474          // Dump compiled clauses by op + pattern preview (helps when
475          // the options struct itself is empty after compile_clauses
476          // moved them into the clauses vec).
477          for (ci, c) in rewrite_rule.clauses.iter().enumerate() {
478            use std::fmt::Write;
479            let _ = write!(
480              xpath_hint,
481              "\n    [{ci}] op={:?} pat={:?}",
482              c.op,
483              match &c.pattern {
484                latexml_core::rewrite::RewritePattern::String(s) =>
485                  format!("Str({})", s.chars().take(120).collect::<String>()),
486                latexml_core::rewrite::RewritePattern::Tokens(_) => "Tokens(..)".into(),
487                latexml_core::rewrite::RewritePattern::Closure(_) => "Closure(..)".into(),
488                latexml_core::rewrite::RewritePattern::NodeList(n) =>
489                  format!("NodeList({})", n.len()),
490                _ => "??".into(),
491              }
492            );
493          }
494          if trace_all {
495            eprintln!(
496              "[rewrite-timing] rule #{}/{} START :: {}",
497              idx, n_rules, xpath_hint
498            );
499            // Flush stderr so it appears even if the rule hangs
500            use std::io::Write;
501            let _ = std::io::stderr().flush();
502          }
503          let started = std::time::Instant::now();
504          rewrite_rule.invoke(&mut document, &root)?;
505          let elapsed = started.elapsed();
506          if trace_all {
507            eprintln!(
508              "[rewrite-timing] rule #{}/{} END {:.2?}",
509              idx, n_rules, elapsed
510            );
511          } else if elapsed > std::time::Duration::from_secs(5) {
512            eprintln!(
513              "[rewrite-timing] rule #{}/{} SLOW {:.2?} :: {}",
514              idx, n_rules, elapsed, xpath_hint
515            );
516          }
517        }
518      }
519      note_end("Rewriting");
520    }
521
522    // Apply \lxDeclare declarations: set roles/names/meanings on matching XMTok elements.
523    // Must run BEFORE math parsing so the parser sees the updated roles.
524    apply_lx_declarations(&mut document);
525
526    if !state::get_nomathparse_flag() {
527      // Telemetry: count formulae and time the whole Marpa parse pass.
528      // Per-formula bucket histogram requires per-call instrumentation
529      // inside latexml_math_parser::parser::parse_math; deferred.
530      let xmath_count = document.findnodes("//ltx:XMath", None).len() as u32;
531      latexml_core::telemetry::set_formulae(xmath_count);
532      let _gp = latexml_core::telemetry::phase(latexml_core::telemetry::Phase::MathParse);
533      let mut parser = MathParser::default();
534      parser.parse_math(&mut document)?;
535      drop(_gp);
536      // Post-parse: mark failed XMath nodes as unparsed.
537      // The parser's parse_kludge already handles OPEN/CLOSE wrapping + script attachment
538      // (parse_kludgeScripts_rec), so we only need to add the unparsed CSS class here.
539      if !parser.failed_xmath_ids.is_empty() {
540        for mut math_node in document.findnodes("descendant-or-self::ltx:Math[not(@text)]", None) {
541          for xmath_child in document.findnodes("ltx:XMath", Some(&math_node)) {
542            if parser.failed_xmath_ids.contains(&xmath_child.to_hashable()) {
543              document.add_class(&mut math_node, "ltx_math_unparsed")?;
544              break;
545            }
546          }
547        }
548      }
549      // Renumber xml:ids inside parsed XMath subtrees to be sequential in document
550      // order. The Marpa parser explores multiple parse alternatives, consuming ID
551      // counter slots for pruned nodes. This pass reassigns IDs post-parse.
552      renumber_math_ids(&mut document);
553      // Fill in \ltx@count@parses markers with actual parse tree counts.
554      // Each marker is <ltx:text _parsetrees_marker="true">0</ltx:text>.
555      // Find the preceding ltx:Math[@_parsetrees] and copy the count.
556      let markers = document.findnodes("//*[@_parsetrees_marker='true']", None);
557      for mut marker in markers {
558        let count = {
559          let preceding = document.findnodes("preceding::ltx:Math[@_parsetrees][1]", Some(&marker));
560          preceding
561            .into_iter()
562            .last()
563            .and_then(|m| m.get_attribute("_parsetrees"))
564            .unwrap_or_else(|| "0".to_string())
565        };
566        // Replace the text content with the actual count
567        for mut child in marker.get_child_nodes() {
568          child.unlink_node();
569        }
570        let _ = marker.append_text(&count);
571        // Remove the marker attribute
572        let _ = marker.remove_attribute("_parsetrees_marker");
573      }
574    }
575
576    note_begin("Finalizing");
577    document.finalize()?;
578    note_end("Finalizing");
579    // Perl core produces role="UNKNOWN" for single-letter math tokens.
580    // Per-document .latexml files set role="ID" via DefMathRewrite BEFORE parsing.
581    // We do NOT apply a blanket conversion — roles are set by rewrite rules only.
582    // Cleanup unreferenced xml:ids on XMTok elements generated by the math parser.
583    // Must run after finalize (which includes prune_xmduals that may transfer ids).
584    document.cleanup_unreferenced_xmtok_ids();
585    Ok(document)
586  }

588  fn digest_internal(&mut self) -> Result<Digested> {
589    let mut boxes = Vec::new();
590    while gullet::has_more_input() {
591      // Perl finishDigestion L219-220: loop consuming input even after errors.
592      // Catch Fatal errors (TooManyErrors, etc.) so we can still produce partial output.
593      match stomach::digest_next_body(None) {
594        Ok(next_bodies) => boxes.extend(next_bodies),
595        Err(e) => {
596          // Re-raise MemoryBudget / wall-clock Timeout (Convert) errors:
597          // those are *resource* failures, not recoverable digestion
598          // hiccups. Catching them here would silently produce empty
599          // output for a runaway-loop paper, masking a real bug and
600          // inflating canvas pass rates with empty conversions.
601          // R35.A: ensure pathological inputs fail loudly (exit 1+)
602          // rather than silently turning into a zero-byte HTML.
603          use latexml_core::common::error::{ErrorCategory, ErrorTarget};
604          if matches!(
605            (&e.target, &e.category),
606            (ErrorTarget::Timeout, ErrorCategory::MemoryBudget)
607              | (ErrorTarget::Timeout, ErrorCategory::Convert)
608              | (ErrorTarget::Timeout, ErrorCategory::TokenLimit)
609              | (ErrorTarget::Timeout, ErrorCategory::PushbackLimit)
610          ) {
611            log::warn!(
612              "digest_internal: resource failure ({:?}/{:?}) — not recovering",
613              e.target,
614              e.category
615            );
616            return Err(e);
617          }
618          // The Err that landed here was raised via `Fatal!` (or
619          // similar), which incremented `LogStatus::Fatal` via
620          // `note_status` in error.rs:353 BUT never emitted the
621          // standard `Fatal:<target>:<category>` log line — that's
622          // normally Error::log_fatal's job, called from the
623          // converter's outer Err handler. We catch the Err here and
624          // continue (recovery), so log_fatal never runs. Without
625          // this explicit emission, papers like arXiv:1903.01633
626          // report `1 fatal error` in the final summary while the
627          // log shows zero `Fatal:` lines — undiagnosable from
628          // logs alone. Emit the Fatal: line directly (not via
629          // log_fatal which would double-increment the counter).
630          let target_str = format!("Fatal:{:?}:{:?} ", e.target, e.category);
631          log::error!(target: &target_str, "{}", e.message);
632          log::warn!("digest_internal: error during recovery digestion: {:?}", e);
633          break;
634        },
635      }
636    }
637    gullet::flush();
638    Ok(Digested::from(List::new(boxes)))
639  }

641  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
642  // Mid-level API.
643
644  // options are currently being evolved to accomodate the Daemon:
645  //    mode  : the processing mode, ie the pool to preload: TeX or BibTeX
646  //    noinitialize : if defined, it does not initialize State.
647  //    preamble = names a tex file (or standard_preamble.tex)
648  //    postamble = names a tex file (or standard_postamble.tex)
649
650  /// Restriction: `digest_file` runs on a single thread, and should never try branching out.
651  fn digest_file(&mut self, mut request: String, options: DigestionOptions) -> Result<Digested> {
652    let mut dir = String::new();
653    let name;
654    // let mut ext = String::new();
655    let mode = match options.mode {
656      None => DigestionMode::TeX,
657      Some(m) => m,
658    };
659
660    if pathname::is_literaldata(&request) {
661      // ext = mode.extension();
662      name = s!("Anonymous String");
663    } else if pathname::is_url(&request) {
664      // ext = mode.extension();
665      name = request.clone();
666    } else {
667      let ext_str = s!(".{}", mode.extension());
668      let request_base = if request.ends_with(&ext_str) {
669        request[0..request.len() - ext_str.len()].to_string()
670      } else {
671        request
672      };
673
674      if let Some(pathname) = pathname::find(&request_base, PathnameFindOptions {
675        extensions: Some(vec![mode.extension(), String::new()]),
676        ..PathnameFindOptions::default()
677      }) {
678        request = pathname;
679        dir = pathname::directory(&request);
680        name = pathname::file_stem(&request);
681        // Perl Core.pm L195-200 unshifts the SOURCE file's directory onto
682        // SEARCHPATHS so subsequent `\input`-style lookups resolve relative
683        // to the main file. When the user invokes with a bare filename
684        // (e.g. `latexml_oxide neurips_2025.tex` from the paper's cwd),
685        // `pathname::find` returns the relative match and `pathname::
686        // directory` returns an empty string — which then gets pushed
687        // onto SEARCHPATHS as a useless entry. Resolve to CWD in that
688        // case so the paper-local Chapter/ etc. are reachable.
689        if dir.is_empty()
690          && let Ok(cwd) = std::env::current_dir()
691        {
692          dir = cwd.to_string_lossy().to_string();
693        }
694      // ext = pathname::extension(&request);
695      } else {
696        let message = s!("Can't find {} file {} ", mode, request_base);
697        fatal!(Core, MissingFile, message);
698      }
699    }
700    note_begin(&s!("Digesting {} {}", mode, name));
701    let main_pool = s!("{}.pool", mode);
702    let noinitialize = options.noinitialize.unwrap_or(false);
703    if !noinitialize {
704      let mut preloads = vec![main_pool];
705      preloads.extend(self.preload.clone());
706      self.initialize_singletons(preloads)?;
707    }
708    {
709      let source_file = if pathname::is_literaldata(&request) {
710        None
711      } else {
712        Some(request.as_str())
713      };
714      establish_source_context(source_file, &name, &dir);
715    }
716
717    // Reverse order, since last opened is first read!
718    if let Some(postamble) = options.postamble {
719      self.load_postamble(postamble);
720    }
721
722    {
723      // Make sure the stomach trick is used very *tightly*, always with a surrounding scope.
724      input_content(&request, InputOptions::default())?;
725    }
726
727    if let Some(preamble) = options.preamble {
728      self.load_preamble(preamble);
729    }
730
731    // Now for the Hacky part for BibTeX!!!
732    // Perl `Core.pm` L160-162: drain the mouth(s) just opened on the
733    // .bib file, run the low-level parser to build a registry of
734    // BibEntry objects, then push a literal wrapper TeX block that
735    // the LaTeX-side digester reads back as
736    //   \begin{bibtex@bibliography}
737    //     \ProcessBibTeXEntry{<key1>}
738    //     ...
739    //   \end{bibtex@bibliography}
740    if matches!(mode, DigestionMode::BibTeX) {
741      use latexml_engine::pre_bibtex::PreBibTeX;
742      let mut bib = PreBibTeX::new_from_gullet(&name);
743      match bib.to_tex() {
744        Ok(tex) => {
745          input_content(&s!("literal:{tex}"), InputOptions::default())?;
746        },
747        Err(parse_err) => {
748          Error!(
749            "bibtex",
750            "parse_failed",
751            s!("Failed to parse BibTeX file {}: {:?}", name, parse_err)
752          );
753        },
754      }
755    }
756
757    let list = self.digest_internal()?;
758    note_end(&s!("Digesting {} {}", mode, name));
759    Ok(list)
760  }
761}

764/// load — `SOURCEFILE`, `SOURCEDIRECTORY`, the front of `SEARCHPATHS`,
765/// `GRAPHICSPATHS`, and `\jobname` (Perl Core.pm L195-200). Shared by

786    state::graphics_paths_push_front(dir.to_string());

790      cs: T_CS!("\\jobname"),
791      paramlist: None,
792      expansion: Tokens::new(Explode!(jobname)).into(),

794    })),
795    None,
796  );

799/// Load a `.latexml` file alongside a `.tex` source file.
800/// Parses `DefMathRewrite(...)` calls and registers them as rewrite rules.
801/// Perl loads these automatically; this provides the equivalent for Rust tests.
802///
803/// Supported patterns:
804///   - Single character: `match => 'a'` -> XPath on XMTok text content
805///   - Complex patterns (e.g. `\hat{f}`, `f_D`, `f_\WildCard`): skipped
806///   - `scope => 'label:...'`: scoped rewrites via label lookup
807///   - `attributes => { role => 'FUNCTION' }`: sets role (and optionally name/meaning)
808fn load_latexml_file(path: &str) -> Result<()> {
809  use latexml_core::rewrite::{RewriteClause, RewriteOperator, RewritePattern};
810
811  let content = match std::fs::read_to_string(path) {
812    Ok(c) => c,
813    Err(_) => return Ok(()), // File doesn't exist or can't be read
814  };
815
816  for cap in DEF_MATH_REWRITE_RE.captures_iter(&content) {
817    let body = &cap[1];
818
819    // Extract match pattern
820    let match_str = match MATCH_RE.captures(body) {
821      Some(m) => m[1].to_string(),
822      None => continue, // No match clause, skip
823    };
824
825    // Build attributes map from the attributes => { ... } section
826    let mut attrs = HashMap::default();
827    if let Some(role_cap) = ROLE_RE.captures(body) {
828      attrs.insert("role".to_string(), role_cap[1].to_string());
829    }
830    if let Some(name_cap) = NAME_ATTR_RE.captures(body) {
831      attrs.insert("name".to_string(), name_cap[1].to_string());
832    }
833    if let Some(meaning_cap) = MEANING_RE.captures(body) {
834      attrs.insert("meaning".to_string(), meaning_cap[1].to_string());
835    }
836    if attrs.is_empty() {
837      continue; // No attributes to set
838    }
839
840    // Check for optional scope
841    let scope_str = SCOPE_RE.captures(body).map(|s| s[1].to_string());
842
843    // Use compile_declare_pattern for all patterns (simple + complex).
844    // The .latexml match strings use the same format as \lxDeclare body_text:
845    //   'f' (simple), 'f_D' (literal subscript), 'f_\WildCard' (wildcard),
846    //   '\hat{f}' (accent), "x^{\prime}" (prime).
847    let pat = latexml_core::rewrite::declare::compile_declare_pattern(&match_str);
848    if pat.xpath.is_empty() {
849      continue; // Unrecognized pattern, skip
850    }
851
852    // For math mode, append visibility check to XPath
853    let xpath = format!("{}[@_pvis and @_cvis]", pat.xpath);
854
855    // Determine select_count based on pattern type
856    let select_count = pat.select_count().or(Some(1usize));
857
858    // Build the rewrite rule
859    let mut clauses = Vec::new();
860
861    // Add scope clause if present
862    if let Some(ref scope) = scope_str {
863      clauses.push(RewriteClause::new_uncompiled(
864        RewriteOperator::Scope,
865        RewritePattern::String(scope.clone()),
866      ));
867    }
868
869    // Add match clause (pre-compiled as XPath string)
870    clauses.push(RewriteClause::new_uncompiled(
871      RewriteOperator::Match,
872      RewritePattern::String(xpath),
873    ));
874
875    // Add attributes clause
876    clauses.push(RewriteClause::new_compiled(
877      RewriteOperator::Attributes,
878      RewritePattern::String(String::new()),
879    ));
880
881    let rewrite = Rewrite {
882      options: RewriteOptions {
883        attributes_map: Some(attrs),
884        is_math: true,
885        select_count,
886        wildcard_paths: pat.wildcard_paths.clone(),
887        declare_filter: Some(pat),
888        ..RewriteOptions::default()
889      },
890      clauses,
891    };
892
893    state::push_value("DOCUMENT_REWRITE_RULES", rewrite)?;
894  }
895
896  Ok(())
897}

899/// Apply \lxDeclare declarations to the document.
900/// Simple fast-path: matches single-token patterns in XMTok elements
901/// and sets role/name/meaning attributes.
902fn apply_lx_declarations(document: &mut Document) {
903  let decls_str = match state::lookup_value("LATEXML_DECLARATIONS") {
904    Some(Stored::String(s)) => arena::with(s, |r| r.to_string()),
905    _ => return,
906  };
907  if decls_str.is_empty() {
908    return;
909  }
910
911  // Parse declarations:
912  // "token_text\trole\tname\tmeaning\tdecl_id\tmatch_font\tscope_prefix".
913  // match_font (font_attribute_string of the digested pattern, e.g.
914  // "italic"/"bold") makes matching font-aware: a plain italic `$x$` declaration
915  // must not annotate a bold `\mathbf{x}` — different fonts denote different
916  // meanings; empty when the pattern carried no distinguishing font.
917  // scope_prefix carries the section gate for scope=section declarations
918  // (INCLUDING untagged ones, which have no decl_id to infer it from).
919  let declarations: Vec<(&str, &str, &str, &str, &str, &str, &str)> = decls_str
920    .lines()
921    .filter_map(|line| {
922      let parts: Vec<&str> = line.splitn(7, '\t').collect();
923      if parts.len() >= 4 {
924        Some((
925          parts[0],
926          parts[1],
927          parts[2],
928          parts[3],
929          *parts.get(4).unwrap_or(&""),
930          *parts.get(5).unwrap_or(&""),
931          *parts.get(6).unwrap_or(&""),
932        ))
933      } else {
934        None
935      }
936    })
937    .collect();
938
939  if declarations.is_empty() {
940    return;
941  }
942
943  // Find all XMTok elements in the document and apply matching declarations.
944  // Skip tokens already marked by the rewrite system (_matched) — these were
945  // handled by subscript/prime/wildcard patterns which take precedence.
946  let xmtoks = document.findnodes("descendant-or-self::ltx:XMTok", None);
947  for mut tok in xmtoks {
948    if tok.has_attribute("_matched") {
949      continue;
950    }
951    let content = tok.get_content();
952    let tok_name = tok.get_attribute("name").unwrap_or_default();
953    // Find the section scope of this token (ancestor section's xml:id)
954    let tok_scope = {
955      let mut scope = String::new();
956      let mut cur = tok.get_parent();
957      while let Some(p) = cur {
958        if p.get_name() == "section" {
959          scope = p
960            .get_property("id")
961            .or_else(|| p.get_attribute("xml:id"))
962            .unwrap_or_default();
963          break;
964        }
965        cur = p.get_parent();
966      }
967      scope
968    };
969
970    for &(pattern, role, name, meaning, decl_id, match_font, scope_prefix) in &declarations {
971      // Match by content text, or by XMTok name attribute (for CS patterns like \circ)
972      let matches = content == pattern
973        || (!tok_name.is_empty() && pattern.starts_with('\\') && pattern[1..] == tok_name);
974      if matches {
975        // Font-class check (mirrors declare_node_matches in the rewrite path):
976        // discriminate only on the meaningful font *classes* — bold vs not,
977        // caligraphic/typewriter family — NOT on an exact font-string match.
978        // Exact equality is wrong here because the declaration's digested frame
979        // font (e.g. "italic") differs from an upright operator token's font
980        // even when they should match (e.g. `$*$` → the ∗ COMPOSEOP token).
981        // A plain/italic declaration rejects bold/caligraphic/typewriter tokens
982        // (so italic `$x$` skips bold `\mathbf{x}`); a declaration that is
983        // itself bold/caligraphic/typewriter requires the token to share it.
984        {
985          let tf = document.get_node_font(&tok);
986          let tok_bold = tf
987            .get_series()
988            .map(|s| s.as_ref() == "bold")
989            .unwrap_or(false);
990          let tok_fam = tf.get_family();
991          let tok_cal = tok_fam
992            .as_ref()
993            .map(|f| f.as_ref() == "caligraphic")
994            .unwrap_or(false);
995          let tok_tt = tok_fam
996            .as_ref()
997            .map(|f| f.as_ref() == "typewriter")
998            .unwrap_or(false);
999          let decl_bold = match_font.contains("bold");
1000          let decl_cal = match_font.contains("caligraphic");
1001          let decl_tt = match_font.contains("typewriter");
1002          if tok_bold != decl_bold || tok_cal != decl_cal || tok_tt != decl_tt {
1003            continue;
1004          }
1005        }
1006        // Scope gate: the explicit scope_prefix (covers UNTAGGED
1007        // scope=section declarations), falling back to the decl_id's section
1008        // prefix for older/tagged lines.
1009        let gate = if !scope_prefix.is_empty() {
1010          scope_prefix
1011        } else {
1012          decl_id.split('.').next().unwrap_or("")
1013        };
1014        if (!decl_id.is_empty() || !scope_prefix.is_empty())
1015          && !gate.is_empty()
1016          && !tok_scope.is_empty()
1017          && tok_scope != gate
1018        {
1019          continue; // Wrong section — skip this declaration
1020        }
1021        if !role.is_empty() {
1022          let _ = tok.set_attribute("role", role);
1023        }
1024        if !name.is_empty() {
1025          let _ = tok.set_attribute("name", name);
1026        }
1027        if !meaning.is_empty() {
1028          let _ = tok.set_attribute("meaning", meaning);
1029        }
1030        if !decl_id.is_empty() {
1031          let _ = tok.set_attribute("decl_id", decl_id);
1032        }
1033        break; // First matching declaration wins
1034      }
1035    }
1036  }
1037}

1039/// Fallback parser for unparseable math expressions.
1040/// Perl: MathParser.pm parse_kludge().
1041/// Balances OPEN/CLOSE delimiters by wrapping matched groups in XMWrap.
1042/// Uses document.wrap_nodes for proper namespace handling.
1043/// Renumber xml:ids inside parsed XMath subtrees so they are sequential in
1044/// document order. The Marpa parser explores multiple parse alternatives,
1045/// consuming ID counter slots for pruned nodes (e.g. m1.1, m1.7, m1.12
1046/// instead of m1.1, m1.2, m1.3). This pure post-processing pass reassigns
1047/// IDs after all pruning is complete.
1048///
1049/// Optimized: single DFS walk per XMath (not XPath), O(1) parent-prefix
1050/// lookup via ID string parsing, and allocation reuse across Math nodes.
1051fn renumber_math_ids(document: &mut Document) {
1052  let xml_ns = "http://www.w3.org/XML/1998/namespace";
1053  let math_nodes = document.findnodes("descendant-or-self::ltx:Math[@text]", None);
1054
1055  // Reuse allocations across Math nodes
1056  let mut id_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
1057  let mut idref_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
1058  let mut id_map: rustc_hash::FxHashMap<String, String> = rustc_hash::FxHashMap::default();
1059  let mut referenced_ids: rustc_hash::FxHashSet<String> = rustc_hash::FxHashSet::default();
1060
1061  for mut math_node in math_nodes {
1062    let math_id = match math_node.get_attribute_ns("id", xml_ns) {
1063      Some(id) => id,
1064      None => continue,
1065    };
1066
1067    let xmath_nodes = document.findnodes("ltx:XMath", Some(&math_node));
1068    for xmath in xmath_nodes {
1069      id_entries.clear();
1070      idref_entries.clear();
1071      id_map.clear();
1072      referenced_ids.clear();
1073
1074      // Single DFS walk collects both xml:id and idref nodes in document order
1075      renumber_collect_dfs(&xmath, xml_ns, &mut id_entries, &mut idref_entries);
1076      if id_entries.is_empty() {
1077        continue;
1078      }
1079
1080      // Collect all referenced IDs (from XMRef idref attributes)
1081      for (_, idref) in &idref_entries {
1082        referenced_ids.insert(idref.clone());
1083      }
1084
1085      // Strip xml:id from XMTok elements that are not referenced by any XMRef.
1086      // The math parser assigns xml:ids to all tokens during parsing, but only
1087      // structural nodes (XMApp, XMDual) and explicitly referenced tokens need them.
1088      // Orphan XMTok ids inflate the renumbering counter causing ID gaps.
1089      {
1090        let mut stripped = false;
1091        for (node, id) in &mut id_entries {
1092          if node.get_name() == "XMTok" && !referenced_ids.contains(id.as_str()) {
1093            document.unrecord_id(id);
1094            let _ = node.remove_attribute("xml:id");
1095            let _ = node.remove_attribute_ns("id", xml_ns);
1096            id.clear(); // mark for removal
1097            stripped = true;
1098          }
1099        }
1100        if stripped {
1101          id_entries.retain(|(_, id)| !id.is_empty());
1102        }
1103      }
1104
1105      if id_entries.is_empty() {
1106        continue;
1107      }
1108
1109      // Build old→new mapping. Flat sequential numbering under the math_id prefix,
1110      // matching Perl's approach of assigning all IDs at the same level.
1111      let mut counter = 0u32;
1112      let mut any_changed = false;
1113      for (_node, old_id) in &id_entries {
1114        counter += 1;
1115        let new_id = format!("{math_id}.{counter}");
1116        if new_id != *old_id {
1117          any_changed = true;
1118        }
1119        id_map.insert(old_id.clone(), new_id);
1120      }
1121
1122      if !any_changed {
1123        continue;
1124      }
1125
1126      // Apply new xml:ids in TWO passes to avoid idstore collisions.
1127      // A new id like "m1.1" would collide with an old "m1.1" still in the
1128      // idstore if we interleave unrecord+record. Strip all first, then assign.
1129      let mut nodes_to_update: Vec<(libxml::tree::Node, String)> = Vec::new();
1130      for (mut node, old_id) in id_entries.drain(..) {
1131        if let Some(new_id) = id_map.get(&old_id)
1132          && new_id != &old_id
1133        {
1134          document.unrecord_id(&old_id);
1135          let _ = node.remove_attribute("xml:id");
1136          let _ = node.remove_attribute_ns("id", xml_ns);
1137          nodes_to_update.push((node, new_id.clone()));
1138        }
1139      }
1140      for (mut node, new_id) in nodes_to_update {
1141        let _ = document.set_attribute(&mut node, "xml:id", &new_id);
1142      }
1143
1144      // Update idrefs
1145      for (mut node, old_idref) in idref_entries.drain(..) {
1146        if let Some(new_idref) = id_map.get(&old_idref)
1147          && new_idref != &old_idref
1148        {
1149          let _ = node.set_attribute("idref", new_idref);
1150        }
1151      }
1152
1153      // Reset _ID_counter__ on the Math node to the final count
1154      let _ = math_node.set_attribute("_ID_counter__", &counter.to_string());
1155    }
1156  }
1157}
1158
1159/// DFS walk collecting nodes with xml:id and idref attributes in document order.
1160/// Stops at nested `Math` elements (which have their own parsing scope).
1161fn renumber_collect_dfs(
1162  node: &libxml::tree::Node,
1163  xml_ns: &str,
1164  id_entries: &mut Vec<(libxml::tree::Node, String)>,
1165  idref_entries: &mut Vec<(libxml::tree::Node, String)>,
1166) {
1167  if let Some(id) = node.get_attribute_ns("id", xml_ns) {
1168    id_entries.push((node.clone(), id));
1169  }
1170  if let Some(idref) = node.get_attribute("idref") {
1171    idref_entries.push((node.clone(), idref));
1172  }
1173  for child in node.get_child_elements() {
1174    // Skip nested Math elements — they have their own parsing scope
1175    if child.get_name() == "Math" {
1176      continue;
1177    }
1178    renumber_collect_dfs(&child, xml_ns, id_entries, idref_entries);
1179  }
1180}
