63  fn initialize_singletons(&mut self, preloads: Vec<String>) -> Result<()> {
64    // reset the error REPORT singleton
65    error::initialize_report();
66    // reset localized variables (if_frames, current_token, align state, etc.)
67    latexml_core::common::local_assignments::initialize_localized();

77    }
78    state::assign_value("InitialPreloads", false, Some(Scope::Global));
79    Ok(())

168  /// Restriction: convert_document runs on a single thread, and should never try branching out.
169  fn convert_document(&mut self, digested: Digested) -> Result<Document> {
170    note_begin("Building");
171    let mut document = Document::new();
172    {
173      // TODO: Can we disentangle the ownership to avoid the clone?
174      let paths_stored = state::get_search_paths();
175      let schema_paths = paths_stored
176        .iter()
177        .map(String::as_str)
178        .collect::<Vec<&str>>();
179      let default_model_load = model::with_schema_data(|schema_opt| match schema_opt {
180        None => true,
181        Some(v) => v.last() == Some(&arena::pin_static("LaTeXML")),
182      });
183      if default_model_load {
184        // Compile-time load of model AND indirect model
185        load_model!("LaTeXML");
186      } else {
187        // Eager-load at runtime
188        model::load_schema(schema_paths.as_slice())?; // If needed?
189      }
190      if state::has_search_paths() {
191        {
192          if state::lookup_bool("INCLUDE_COMMENTS") {
193            let paths_string = state::with_search_paths(|paths| {
194              paths
195                .iter()
196                .map(String::as_str)
197                .collect::<Vec<&str>>()
198                .join(",")
199            });
200            let attributes = map! {s!("paths") => paths_string};
201            document.insert_pi("latexml", Some(attributes))?;
202          }
203        }
204      }
205    }
206
207    for preload in &self.preload {
208      if preload.ends_with(".pool") {
209        continue;
210      }
211      let mut options: Option<String> = None;
212      LATEX_OPTION_REGEX.replace_all(preload, |refs: &Captures| -> String {
213        options = Some(refs.get(1).map_or("", |m| m.as_str()).to_string());
214        String::new()
215      });
216      if preload.ends_with(".cls") {
217        CLS_EXT_REGEX.replace_all(preload, "");
218        let attributes = map! {s!("class") => preload.to_string()};
219        document.insert_pi("latexml", Some(attributes))?;
220      } else {
221        STY_EXT_REGEX.replace_all(preload, "");
222        let attributes = map! {s!("package") => preload.to_string()};
223        document.insert_pi("latexml", Some(attributes))?;
224      }
225    }
226    Debug!("Doc absorb: {:?}", digested);
227
228    document.absorb(&digested, None)?;
229    note_end("Building");
230
231    let has_rewrites = state::has_value("DOCUMENT_REWRITE_RULES");
232    if has_rewrites {
233      note_begin("Rewriting");
234      document.mark_xmnode_visibility()?;
235      document.load_labels_for_rewrite()?;
236      // TODO: What is the right way to do rewrites in a daemon-safe manner?
237      if let Some(Stored::VecDequeStored(rules)) = state::remove_value("DOCUMENT_REWRITE_RULES") {
238        if let Some(root) = document.get_document().get_root_element() {
239          // Step 1: copy the rules locally through Rc, to be able to invoke them with mutable
240          // state. (TODO: obviously, this could be avoided if they never needed mutable
241          // state. When do they?)
242          let mut rewrites = Vec::new();
243          for rule in rules {
244            if let Stored::Rewrite(mut rewrite_rule) = rule {
245              rewrite_rule.compile_clauses(&mut document);
246              rewrites.push(rewrite_rule); // clone the Rc
247            }
248          }
249          // Step 2: invoke the rewrite rules
250          for mut rewrite_rule in rewrites {
251            rewrite_rule.invoke(&mut document, &root)?;
252          }
253        }
254      }
255      note_end("Rewriting");
256    }
257
258    // Apply \lxDeclare declarations: set roles/names/meanings on matching XMTok elements.
259    // Must run BEFORE math parsing so the parser sees the updated roles.
260    apply_lx_declarations(&mut document);
261
262    if !state::get_nomathparse_flag() {
263      let mut parser = MathParser::default();
264      parser.parse_math(&mut document)?;
265    }
266
267    note_begin("Finalizing");
268    document.finalize()?;
269    note_end("Finalizing");
270    // Cleanup unreferenced xml:ids on XMTok elements generated by the math parser.
271    // Must run after finalize (which includes prune_xmduals that may transfer ids).
272    document.cleanup_unreferenced_xmtok_ids();
273    Ok(document)
274  }

276  fn digest_internal(&mut self) -> Result<Digested> {
277    let mut boxes = Vec::new();
278    while gullet::has_more_input() {
279      let next_bodies: Vec<Digested> = stomach::digest_next_body(None)?;
280      // info!(target:"core:digest_next_body", "\n{:?}\n----\n",next_bodies);
281      boxes.extend(next_bodies);
282    }
283    gullet::flush();
284    Ok(Digested::from(List::new(boxes)))
285  }

290  // options are currently being evolved to accomodate the Daemon:
291  //    mode  : the processing mode, ie the pool to preload: TeX or BibTeX
292  //    noinitialize : if defined, it does not initialize State.

296  /// Restriction: `digest_file` runs on a single thread, and should never try branching out.
297  fn digest_file(&mut self, mut request: String, options: DigestionOptions) -> Result<Digested> {
298    let mut dir = String::new();
299    let name;
300    // let mut ext = String::new();
301    let mode = match options.mode {
302      None => DigestionMode::TeX,
303      Some(m) => m,
304    };
305
306    if pathname::is_literaldata(&request) {
307      // ext = mode.extension();
308      name = s!("Anonymous String");
309    } else if pathname::is_url(&request) {
310      // ext = mode.extension();
311      name = request.clone();
312    } else {
313      let ext_str = s!(".{}", mode.extension());
314      let request_base = if request.ends_with(&ext_str) {
315        request[0..request.len() - ext_str.len()].to_string()
316      } else {
317        request
318      };
319
320      if let Some(pathname) = pathname::find(&request_base, PathnameFindOptions {
321        extensions: Some(vec![mode.extension(), String::new()]),
322        ..PathnameFindOptions::default()
323      }) {
324        request = pathname;
325        dir = pathname::directory(&request);
326        name = pathname::file_stem(&request);
327      // ext = pathname::extension(&request);
328      } else {
329        let message = s!("Can't find {} file {} ", mode, request_base);
330        fatal!(Core, MissingFile, message);
331      }
332    }
333    note_begin(&s!("Digesting {} {}", mode, name));
334    let main_pool = s!("{}.pool", mode);
335    let noinitialize = options.noinitialize.unwrap_or(false);
336    if !noinitialize {
337      let mut preloads = vec![main_pool];
338      preloads.extend(self.preload.clone());
339      self.initialize_singletons(preloads)?;
340    }
341    {
342      if !pathname::is_literaldata(&request) {
343        state::assign_value("SOURCEFILE", request.clone(), None);
344      }
345      if !dir.is_empty() {
346        state::assign_value("SOURCEDIRECTORY", dir.clone(), None);
347      }
348      state::search_paths_push_front(dir.clone());
349      state::graphics_paths_push_front(dir);
350      state::install_definition(
351        Stored::Expandable(Rc::new(Expandable {
352          cs: T_CS!("\\jobname"),
353          paramlist: None,
354          expansion: Tokens::new(Explode!(name)).into(),
355          ..Expandable::default()
356        })),
357        None,
358      );
359    }
360
361    // Reverse order, since last opened is first read!
362    if let Some(postamble) = options.postamble {
363      self.load_postamble(postamble);
364    }
365
366    {
367      // Make sure the stomach trick is used very *tightly*, always with a surrounding scope.
368      input_content(&request, InputOptions::default())?;
369    }
370
371    if let Some(preamble) = options.preamble {
372      self.load_preamble(preamble);
373    }
374
375    // Now for the Hacky part for BibTeX!!!
376    // if mode == DigestionMode::BibTeX {
377    //   let bib = LaTeXML::Pre::BibTeX->newFromGullet($name, $state->getStomach->getGullet);
378    //   LaTeXML::Package::InputContent("literal:" . $bib->toTeX);
379    // }
380
381    let list = self.digest_internal()?;
382    note_end(&s!("Digesting {} {}", mode, name));
383    Ok(list)
384  }
385}

387/// Apply \lxDeclare declarations to the document.
388/// Simple fast-path: matches single-token patterns in XMTok elements
389/// and sets role/name/meaning attributes.
390fn apply_lx_declarations(document: &mut Document) {
391  let decls_str = match state::lookup_value("LATEXML_DECLARATIONS") {
392    Some(Stored::String(s)) => arena::with(s, |r| r.to_string()),
393    _ => return,
394  };
395  if decls_str.is_empty() {
396    return;
397  }
398
399  // Parse declarations: "token_text\trole\tname\tmeaning" per line
400  let declarations: Vec<(&str, &str, &str, &str)> = decls_str
401    .lines()
402    .filter_map(|line| {
403      let parts: Vec<&str> = line.splitn(4, '\t').collect();
404      if parts.len() == 4 {
405        Some((parts[0], parts[1], parts[2], parts[3]))
406      } else {
407        None
408      }
409    })
410    .collect();
411
412  if declarations.is_empty() {
413    return;
414  }
415
416  // Find all XMTok elements in the document and apply matching declarations
417  let xmtoks = document.findnodes("descendant-or-self::ltx:XMTok", None);
418  for mut tok in xmtoks {
419    let content = tok.get_content();
420    for &(pattern, role, name, meaning) in &declarations {
421      if content == pattern {
422        if !role.is_empty() {
423          let _ = tok.set_attribute("role", role);
424        }
425        if !name.is_empty() {
426          let _ = tok.set_attribute("name", name);
427        }
428        if !meaning.is_empty() {
429          let _ = tok.set_attribute("meaning", meaning);
430        }
431        break; // First matching declaration wins
432      }
433    }
434  }
