95  fn initialize_singletons(&mut self, preloads: Vec<String>) -> Result<()> {
96    // reset the error REPORT singleton
97    error::initialize_report();
98    // reset localized variables (if_frames, current_token, align state, etc.)
99    latexml_core::common::local_assignments::initialize_localized();
100    // now handle conversion state
101    gullet::initialize_gullet();
102    stomach::initialize_stomach();
103    // should we reset the model also?
104    model::initialize_model();
105    // let paths = state::search_paths;
106    let dump_path = LATEXML_DUMP.clone();
107    state::assign_value("InitialPreloads", true, Some(Scope::Global));
108    for preload in preloads {
109      // Perl: initializeState extracts extension and options from "name.ext[opt1,opt2]"
110      // Parse bracket options: "latexml.sty[nobibtex]" → name="latexml", ext="sty",
111      // options=["nobibtex"]
112      let (preload_base, options) = if let Some(bracket_pos) = preload.find('[') {
113        let base = preload[..bracket_pos].to_string();
114        let opts_str = preload[bracket_pos + 1..].trim_end_matches(']');
115        let opts: Vec<String> = opts_str
116          .split(',')
117          .map(|s| s.trim().to_string())
118          .filter(|s| !s.is_empty())
119          .collect();
120        (base, opts)
121      } else {
122        (preload.clone(), vec![])
123      };
124      let (name, ext) = if let Some(pos) = preload_base.rfind('.') {
125        (
126          preload_base[..pos].to_string(),
127          preload_base[pos + 1..].to_string(),
128        )
129      } else {
130        (preload_base.clone(), String::from("sty"))
131      };
132      let handleoptions = ext == "sty" || ext == "cls";
133      // Pass package options via state (Perl: \PassOptionsToPackage equivalent)
134      if !options.is_empty() {
135        let opt_key = format!("opt@{name}.{ext}");
136        let opt_str = options.join(",");
137        state::assign_value(
138          &opt_key,
139          latexml_core::common::store::Stored::String(latexml_core::common::arena::pin(&opt_str)),
140          Some(Scope::Global),
141        );
142      }
143      input_definitions(&name, InputDefinitionOptions {
144        extension: Some(ext.into()),
145        handleoptions,
146        ..InputDefinitionOptions::default()
147      })?;
148    }
149    state::assign_value("InitialPreloads", false, Some(Scope::Global));
150
151    // Load kernel dump AFTER pools (provides TeX/LaTeX macros the pools skipped).
152    if let Some(ref dump_path) = dump_path {
153      let path = std::path::Path::new(dump_path);
154      if path.exists() {
155        // Rust-native tab-separated format (from --init mode). The
156        // Perl-format `dump_loader` was deleted 2026-04-18 (dead code —
157        // we never consumed Perl-generated dumps).
158        let result = latexml_core::dump_reader::load_native_dump(path);
159        match result {
160          Ok(count) => {
161            eprintln!(
162              "[latexml-oxide] Loaded {} kernel definitions from {}",
163              count,
164              path.display()
165            );
166          },
167          Err(e) => {
168            eprintln!("[latexml-oxide] Warning: failed to load dump: {}", e);
169          },
170        }
171      }
172    }
173    Ok(())
174  }

178  fn digest(
179    &mut self,
180    request: String,
181    _preamble: Option<String>,
182    _postamble: Option<String>,
183    mode: Option<DigestionMode>,
184    _no_init: bool,
185  ) -> Result<Digested> {
186    let mut _ext = match mode {
187      Some(m) => Some(m.extension()),
188      None => Some(DigestionMode::TeX.extension()),
189    };
190    let mut dir_opt = None;
191
192    // Canonicalize relative paths so `Path::parent()` gives a real directory.
193    // `Path::new("foo.tex").parent()` returns `Some("")` (empty string) which
194    // poisons SEARCHPATHS / SOURCEDIRECTORY: an empty-string search-path
195    // entry resolves files via cwd-name with no normalization, changing the
196    // order in which resource files (e.g. `ts1enc.def` vs `t1enc.def`) are
197    // discovered. Concrete symptom: TS1 fontmap leaks into control-sequence
198    // construction → `cn` characters become `⚮♪` → `\c@cn` undefined →
199    // 381-error cascade (paper 0709.2868). Canonicalizing matches Perl's
200    // `File::Spec->splitpath` behavior which always yields a real directory.
201    let canonical_request = if pathname::is_literaldata(&request) || pathname::is_url(&request) {
202      request.clone()
203    } else {
204      std::fs::canonicalize(&request)
205        .ok()
206        .and_then(|p| p.to_str().map(String::from))
207        .unwrap_or_else(|| request.clone())
208    };
209    let name = if pathname::is_literaldata(&request) {
210      s!("Anonymous String")
211    } else if pathname::is_url(&request) {
212      request.clone()
213    } else {
214      let path = Path::new(&canonical_request);
215      dir_opt = path.parent();
216      match path.file_stem() {
217        None => String::from("missing_name"),
218        Some(pf) => pf.to_str().unwrap().to_string(),
219      }
220    };
221    // else {
222    //   $self->withState(sub {
223    //       Fatal('missing_file', $request, undef, "Can't find $mode file $request"); }); } }
224    // };
225    let digestion_note = s!("Digesting {}", name);
226    note_begin(&digestion_note);
227    // $self->initializestate::$mode . ".pool", @{ $$self{preload} || [] }) unless
228    // $options{noinitialize};
229    if !pathname::is_literaldata(&request) {
230      state::assign_value("SOURCEFILE", arena::pin(&request), None);
231    }
232    if let Some(dir) = dir_opt {
233      let dir = dir.to_str().unwrap_or(".");
234      {
235        state::assign_value("SOURCEDIRECTORY", arena::pin(dir), None);
236        state::add_search_path(dir.to_string());
237      }
238    }
239    //   if defined $dir && !grep { $_ eq $dir } @{ $state->lookupValue('SEARCHPATHS') };
240    // $state->unshiftValue(GRAPHICSPATHS => $dir)
241
242    // if defined $dir && !grep { $_ eq $dir } @{ $state->lookupValue('GRAPHICSPATHS') };
243
244    let name_copy = name;
245    state::install_definition(
246      Stored::Expandable(Rc::new(Expandable {
247        cs: T_CS!("\\jobname"),
248        paramlist: None,
249        expansion: Tokens::new(Explode!(name_copy)).into(),
250        ..Expandable::default()
251      })),
252      None,
253    );
254
255    // $self->loadPostamble($options{postamble}) if $options{postamble};
256    input_content(&request, InputOptions::default())?;
257    // $self->loadPreamble($options{preamble}) if $options{preamble};
258
259    // // Now for the Hacky part for BibTeX!!!
260    // if ($mode eq 'BibTeX') {
261    //   my $bib = LaTeXML::Pre::BibTeX->newFromGullet($name, $state->getStomach->getGullet);
262    //   LaTeXML::Package::InputContent("literal:" . $bib->toTeX); }
263
264    let list = self.digest_internal()?;
265    note_end(&digestion_note);
266
267    Ok(list)
268  }

277  /// Restriction: convert_document runs on a single thread, and should never try branching out.
278  fn convert_document(&mut self, digested: Digested) -> Result<Document> {
279    note_begin("Building");
280    let mut document = Document::new();
281    {
282      // TODO: Can we disentangle the ownership to avoid the clone?
283      let paths_stored = state::get_search_paths();
284      let schema_paths = paths_stored
285        .iter()
286        .map(String::as_str)
287        .collect::<Vec<&str>>();
288      let default_model_load = model::with_schema_data(|schema_opt| match schema_opt {
289        None => true,
290        Some(v) => v.last() == Some(&arena::pin_static("LaTeXML")),
291      });
292      if default_model_load {
293        // Compile-time load of model AND indirect model
294        load_model!("LaTeXML");
295      } else {
296        // Eager-load at runtime
297        model::load_schema(schema_paths.as_slice())?; // If needed?
298      }
299      if state::has_search_paths() {
300        {
301          if state::lookup_bool("INCLUDE_COMMENTS") {
302            let paths_string = state::with_search_paths(|paths| {
303              paths
304                .iter()
305                .map(String::as_str)
306                .collect::<Vec<&str>>()
307                .join(",")
308            });
309            let attributes = map! {s!("searchpaths") => paths_string};
310            document.insert_pi("latexml", Some(attributes))?;
311          }
312        }
313      }
314    }
315
316    for preload in &self.preload {
317      if preload.ends_with(".pool") {
318        continue;
319      }
320      let mut options: Option<String> = None;
321      LATEX_OPTION_REGEX.replace_all(preload, |refs: &Captures| -> String {
322        options = Some(refs.get(1).map_or("", |m| m.as_str()).to_string());
323        String::new()
324      });
325      if preload.ends_with(".cls") {
326        CLS_EXT_REGEX.replace_all(preload, "");
327        let attributes = map! {s!("class") => preload.to_string()};
328        document.insert_pi("latexml", Some(attributes))?;
329      } else {
330        STY_EXT_REGEX.replace_all(preload, "");
331        let attributes = map! {s!("package") => preload.to_string()};
332        document.insert_pi("latexml", Some(attributes))?;
333      }
334    }
335    Debug!("Doc absorb: {:?}", digested);
336
337    document.absorb(&digested, None)?;
338    note_end("Building");
339
340    // Load .latexml file if it exists alongside the source .tex file.
341    // Perl does this automatically during initialization; we do it post-build
342    // so the rewrite rules can be compiled against the built document.
343    if let Some(Stored::String(source_sym)) = state::lookup_value("SOURCEFILE") {
344      let source_path = arena::with(source_sym, |s| s.to_string());
345      // Replace .tex extension with .latexml
346      let latexml_path = if source_path.ends_with(".tex") {
347        source_path.replace(".tex", ".latexml")
348      } else {
349        format!("{}.latexml", source_path)
350      };
351      if Path::new(&latexml_path).exists() {
352        let _ = load_latexml_file(&latexml_path);
353      }
354    }
355
356    let has_rewrites = state::has_value("DOCUMENT_REWRITE_RULES");
357    if has_rewrites {
358      note_begin("Rewriting");
359      document.mark_xmnode_visibility()?;
360      document.load_labels_for_rewrite()?;
361      // TODO: What is the right way to do rewrites in a daemon-safe manner?
362      if let Some(Stored::VecDequeStored(rules)) = state::remove_value("DOCUMENT_REWRITE_RULES") {
363        if let Some(root) = document.get_document().get_root_element() {
364          // Step 1: copy the rules locally through Rc, to be able to invoke them with mutable
365          // state. (TODO: obviously, this could be avoided if they never needed mutable
366          // state. When do they?)
367          let mut rewrites = Vec::new();
368          for rule in rules {
369            if let Stored::Rewrite(mut rewrite_rule) = rule {
370              rewrite_rule.compile_clauses(&mut document);
371              rewrites.push(rewrite_rule);
372            }
373          }
374          // 31 rules compiled for declare test; XPath matching issue prevents application
375          // Step 2: invoke the rewrite rules
376          for mut rewrite_rule in rewrites {
377            rewrite_rule.invoke(&mut document, &root)?;
378          }
379        }
380      }
381      note_end("Rewriting");
382    }
383
384    // Apply \lxDeclare declarations: set roles/names/meanings on matching XMTok elements.
385    // Must run BEFORE math parsing so the parser sees the updated roles.
386    apply_lx_declarations(&mut document);
387
388    if !state::get_nomathparse_flag() {
389      let mut parser = MathParser::default();
390      parser.parse_math(&mut document)?;
391      // Post-parse: mark failed XMath nodes as unparsed.
392      // The parser's parse_kludge already handles OPEN/CLOSE wrapping + script attachment
393      // (parse_kludgeScripts_rec), so we only need to add the unparsed CSS class here.
394      if !parser.failed_xmath_ids.is_empty() {
395        for mut math_node in document.findnodes("descendant-or-self::ltx:Math[not(@text)]", None) {
396          for xmath_child in document.findnodes("ltx:XMath", Some(&math_node)) {
397            if parser.failed_xmath_ids.contains(&xmath_child.to_hashable()) {
398              document.add_class(&mut math_node, "ltx_math_unparsed")?;
399              break;
400            }
401          }
402        }
403      }
404      // Renumber xml:ids inside parsed XMath subtrees to be sequential in document
405      // order. The Marpa parser explores multiple parse alternatives, consuming ID
406      // counter slots for pruned nodes. This pass reassigns IDs post-parse.
407      renumber_math_ids(&mut document);
408      // Fill in \ltx@count@parses markers with actual parse tree counts.
409      // Each marker is <ltx:text _parsetrees_marker="true">0</ltx:text>.
410      // Find the preceding ltx:Math[@_parsetrees] and copy the count.
411      let markers = document.findnodes("//*[@_parsetrees_marker='true']", None);
412      for mut marker in markers {
413        let count = {
414          let preceding = document.findnodes("preceding::ltx:Math[@_parsetrees][1]", Some(&marker));
415          preceding
416            .into_iter()
417            .last()
418            .and_then(|m| m.get_attribute("_parsetrees"))
419            .unwrap_or_else(|| "0".to_string())
420        };
421        // Replace the text content with the actual count
422        for mut child in marker.get_child_nodes() {
423          child.unlink_node();
424        }
425        let _ = marker.append_text(&count);
426        // Remove the marker attribute
427        let _ = marker.remove_attribute("_parsetrees_marker");
428      }
429    }
430
431    note_begin("Finalizing");
432    document.finalize()?;
433    note_end("Finalizing");
434    // Perl core produces role="UNKNOWN" for single-letter math tokens.
435    // Per-document .latexml files set role="ID" via DefMathRewrite BEFORE parsing.
436    // We do NOT apply a blanket conversion — roles are set by rewrite rules only.
437    // Cleanup unreferenced xml:ids on XMTok elements generated by the math parser.
438    // Must run after finalize (which includes prune_xmduals that may transfer ids).
439    document.cleanup_unreferenced_xmtok_ids();
440    Ok(document)
441  }

443  fn digest_internal(&mut self) -> Result<Digested> {
444    let mut boxes = Vec::new();
445    while gullet::has_more_input() {
446      // Perl finishDigestion L219-220: loop consuming input even after errors.
447      // Catch Fatal errors (TooManyErrors, etc.) so we can still produce partial output.
448      match stomach::digest_next_body(None) {
449        Ok(next_bodies) => boxes.extend(next_bodies),
450        Err(e) => {
451          log::warn!("digest_internal: error during recovery digestion: {:?}", e);
452          break;
453        },
454      }
455    }
456    gullet::flush();
457    Ok(Digested::from(List::new(boxes)))
458  }

469  /// Restriction: `digest_file` runs on a single thread, and should never try branching out.
470  fn digest_file(&mut self, mut request: String, options: DigestionOptions) -> Result<Digested> {
471    let mut dir = String::new();
472    let name;
473    // let mut ext = String::new();
474    let mode = match options.mode {
475      None => DigestionMode::TeX,
476      Some(m) => m,
477    };
478
479    if pathname::is_literaldata(&request) {
480      // ext = mode.extension();
481      name = s!("Anonymous String");
482    } else if pathname::is_url(&request) {
483      // ext = mode.extension();
484      name = request.clone();
485    } else {
486      let ext_str = s!(".{}", mode.extension());
487      let request_base = if request.ends_with(&ext_str) {
488        request[0..request.len() - ext_str.len()].to_string()
489      } else {
490        request
491      };
492
493      if let Some(pathname) = pathname::find(&request_base, PathnameFindOptions {
494        extensions: Some(vec![mode.extension(), String::new()]),
495        ..PathnameFindOptions::default()
496      }) {
497        request = pathname;
498        dir = pathname::directory(&request);
499        name = pathname::file_stem(&request);
500      // ext = pathname::extension(&request);
501      } else {
502        let message = s!("Can't find {} file {} ", mode, request_base);
503        fatal!(Core, MissingFile, message);
504      }
505    }
506    note_begin(&s!("Digesting {} {}", mode, name));
507    let main_pool = s!("{}.pool", mode);
508    let noinitialize = options.noinitialize.unwrap_or(false);
509    if !noinitialize {
510      let mut preloads = vec![main_pool];
511      preloads.extend(self.preload.clone());
512      self.initialize_singletons(preloads)?;
513    }
514    {
515      if !pathname::is_literaldata(&request) {
516        state::assign_value("SOURCEFILE", request.clone(), None);
517      }
518      if !dir.is_empty() {
519        state::assign_value("SOURCEDIRECTORY", dir.clone(), None);
520      }
521      state::search_paths_push_front(dir.clone());
522      // Perl Core.pm L200:
523      //   $state->unshiftValue(GRAPHICSPATHS => $dir)
524      //     if !grep { $_ eq $dir } @{ $state->lookupValue('GRAPHICSPATHS') };
525      if !state::graphics_paths_contains(&dir) {
526        state::graphics_paths_push_front(dir);
527      }
528      state::install_definition(
529        Stored::Expandable(Rc::new(Expandable {
530          cs: T_CS!("\\jobname"),
531          paramlist: None,
532          expansion: Tokens::new(Explode!(name)).into(),
533          ..Expandable::default()
534        })),
535        None,
536      );
537    }
538
539    // Reverse order, since last opened is first read!
540    if let Some(postamble) = options.postamble {
541      self.load_postamble(postamble);
542    }
543
544    {
545      // Make sure the stomach trick is used very *tightly*, always with a surrounding scope.
546      input_content(&request, InputOptions::default())?;
547    }
548
549    if let Some(preamble) = options.preamble {
550      self.load_preamble(preamble);
551    }
552
553    // Now for the Hacky part for BibTeX!!!
554    // if mode == DigestionMode::BibTeX {
555    //   let bib = LaTeXML::Pre::BibTeX->newFromGullet($name, $state->getStomach->getGullet);
556    //   LaTeXML::Package::InputContent("literal:" . $bib->toTeX);
557    // }
558
559    let list = self.digest_internal()?;
560    note_end(&s!("Digesting {} {}", mode, name));
561    Ok(list)
562  }

565/// Load a `.latexml` file alongside a `.tex` source file.
566/// Parses `DefMathRewrite(...)` calls and registers them as rewrite rules.
567/// Perl loads these automatically; this provides the equivalent for Rust tests.
568///
569/// Supported patterns:
570///   - Single character: `match => 'a'` -> XPath on XMTok text content
571///   - Complex patterns (e.g. `\hat{f}`, `f_D`, `f_\WildCard`): skipped
572///   - `scope => 'label:...'`: scoped rewrites via label lookup
573///   - `attributes => { role => 'FUNCTION' }`: sets role (and optionally name/meaning)
574fn load_latexml_file(path: &str) -> Result<()> {
575  use latexml_core::rewrite::{RewriteClause, RewriteOperator, RewritePattern};
576
577  let content = match std::fs::read_to_string(path) {
578    Ok(c) => c,
579    Err(_) => return Ok(()), // File doesn't exist or can't be read
580  };
581
582  for cap in DEF_MATH_REWRITE_RE.captures_iter(&content) {
583    let body = &cap[1];
584
585    // Extract match pattern
586    let match_str = match MATCH_RE.captures(body) {
587      Some(m) => m[1].to_string(),
588      None => continue, // No match clause, skip
589    };
590
591    // Build attributes map from the attributes => { ... } section
592    let mut attrs = HashMap::default();
593    if let Some(role_cap) = ROLE_RE.captures(body) {
594      attrs.insert("role".to_string(), role_cap[1].to_string());
595    }
596    if let Some(name_cap) = NAME_ATTR_RE.captures(body) {
597      attrs.insert("name".to_string(), name_cap[1].to_string());
598    }
599    if let Some(meaning_cap) = MEANING_RE.captures(body) {
600      attrs.insert("meaning".to_string(), meaning_cap[1].to_string());
601    }
602    if attrs.is_empty() {
603      continue; // No attributes to set
604    }
605
606    // Check for optional scope
607    let scope_str = SCOPE_RE.captures(body).map(|s| s[1].to_string());
608
609    // Use compile_declare_pattern for all patterns (simple + complex).
610    // The .latexml match strings use the same format as \lxDeclare body_text:
611    //   'f' (simple), 'f_D' (literal subscript), 'f_\WildCard' (wildcard),
612    //   '\hat{f}' (accent), "x^{\prime}" (prime).
613    let pat = latexml_package::package::latexml_sty::compile_declare_pattern_pub(&match_str);
614    if pat.xpath.is_empty() {
615      continue; // Unrecognized pattern, skip
616    }
617
618    // Add declare metadata for Rust-side filtering
619    attrs.insert("_declare_type".to_string(), pat.pattern_type.to_string());
620    if let Some(ref b) = pat.base_text {
621      attrs.insert("_declare_base".to_string(), b.clone());
622    }
623    if let Some(ref s) = pat.sub_text {
624      attrs.insert("_declare_sub".to_string(), s.clone());
625    }
626    if let Some(ref a) = pat.accent_name {
627      attrs.insert("_declare_accent".to_string(), a.clone());
628    }
629
630    // For math mode, append visibility check to XPath
631    let xpath = format!("{}[@_pvis and @_cvis]", pat.xpath);
632
633    // Determine select_count based on pattern type
634    let select_count = match pat.pattern_type {
635      "literal_subscript" | "prime" | "subscript" => Some(2usize),
636      "accent" => Some(1usize),
637      _ => Some(1usize),
638    };
639
640    // Build the rewrite rule
641    let mut clauses = Vec::new();
642
643    // Add scope clause if present
644    if let Some(ref scope) = scope_str {
645      clauses.push(RewriteClause::new_uncompiled(
646        RewriteOperator::Scope,
647        RewritePattern::String(scope.clone()),
648      ));
649    }
650
651    // Add match clause (pre-compiled as XPath string)
652    clauses.push(RewriteClause::new_uncompiled(
653      RewriteOperator::Match,
654      RewritePattern::String(xpath),
655    ));
656
657    // Add attributes clause
658    clauses.push(RewriteClause::new_compiled(
659      RewriteOperator::Attributes,
660      RewritePattern::String(String::new()),
661    ));
662
663    let rewrite = Rewrite {
664      options: RewriteOptions {
665        attributes_map: Some(attrs),
666        is_math: true,
667        select_count,
668        wildcard_paths: pat.wildcard_paths,
669        ..RewriteOptions::default()
670      },
671      clauses,
672    };
673
674    state::push_value("DOCUMENT_REWRITE_RULES", rewrite)?;
675  }
676
677  Ok(())
678}

680/// Apply \lxDeclare declarations to the document.
681/// Simple fast-path: matches single-token patterns in XMTok elements
682/// and sets role/name/meaning attributes.
683fn apply_lx_declarations(document: &mut Document) {
684  let decls_str = match state::lookup_value("LATEXML_DECLARATIONS") {
685    Some(Stored::String(s)) => arena::with(s, |r| r.to_string()),
686    _ => return,
687  };
688  if decls_str.is_empty() {
689    return;
690  }
691
692  // Parse declarations: "token_text\trole\tname\tmeaning\tdecl_id" per line
693  let declarations: Vec<(&str, &str, &str, &str, &str)> = decls_str
694    .lines()
695    .filter_map(|line| {
696      let parts: Vec<&str> = line.splitn(5, '\t').collect();
697      if parts.len() >= 4 {
698        Some((
699          parts[0],
700          parts[1],
701          parts[2],
702          parts[3],
703          *parts.get(4).unwrap_or(&""),
704        ))
705      } else {
706        None
707      }
708    })
709    .collect();
710
711  if declarations.is_empty() {
712    return;
713  }
714
715  // Find all XMTok elements in the document and apply matching declarations.
716  // Skip tokens already marked by the rewrite system (_matched) — these were
717  // handled by subscript/prime/wildcard patterns which take precedence.
718  let xmtoks = document.findnodes("descendant-or-self::ltx:XMTok", None);
719  for mut tok in xmtoks {
720    if tok.has_attribute("_matched") {
721      continue;
722    }
723    let content = tok.get_content();
724    let tok_name = tok.get_attribute("name").unwrap_or_default();
725    // Find the section scope of this token (ancestor section's xml:id)
726    let tok_scope = {
727      let mut scope = String::new();
728      let mut cur = tok.get_parent();
729      while let Some(p) = cur {
730        if p.get_name() == "section" {
731          scope = p
732            .get_property("id")
733            .or_else(|| p.get_attribute("xml:id"))
734            .unwrap_or_default();
735          break;
736        }
737        cur = p.get_parent();
738      }
739      scope
740    };
741
742    for &(pattern, role, name, meaning, decl_id) in &declarations {
743      // Match by content text, or by XMTok name attribute (for CS patterns like \circ)
744      let matches = content == pattern
745        || (!tok_name.is_empty() && pattern.starts_with('\\') && pattern[1..] == tok_name);
746      if matches {
747        // Check scope: if decl_id has a section prefix (e.g. "S1" from "S1.XMD1"),
748        // only apply to tokens within that section
749        if !decl_id.is_empty() {
750          if let Some(section_prefix) = decl_id.split('.').next() {
751            if !section_prefix.is_empty() && !tok_scope.is_empty() && tok_scope != section_prefix {
752              continue; // Wrong section — skip this declaration
753            }
754          }
755        }
756        if !role.is_empty() {
757          let _ = tok.set_attribute("role", role);
758        }
759        if !name.is_empty() {
760          let _ = tok.set_attribute("name", name);
761        }
762        if !meaning.is_empty() {
763          let _ = tok.set_attribute("meaning", meaning);
764        }
765        if !decl_id.is_empty() {
766          let _ = tok.set_attribute("decl_id", decl_id);
767        }
768        break; // First matching declaration wins
769      }
770    }
771  }
772}

774/// Fallback parser for unparseable math expressions.
775/// Perl: MathParser.pm parse_kludge().
776/// Balances OPEN/CLOSE delimiters by wrapping matched groups in XMWrap.
777/// Uses document.wrap_nodes for proper namespace handling.
778/// Renumber xml:ids inside parsed XMath subtrees so they are sequential in
779/// document order. The Marpa parser explores multiple parse alternatives,
780/// consuming ID counter slots for pruned nodes (e.g. m1.1, m1.7, m1.12
781/// instead of m1.1, m1.2, m1.3). This pure post-processing pass reassigns
782/// IDs after all pruning is complete.
783///
784/// Optimized: single DFS walk per XMath (not XPath), O(1) parent-prefix
785/// lookup via ID string parsing, and allocation reuse across Math nodes.
786fn renumber_math_ids(document: &mut Document) {
787  let xml_ns = "http://www.w3.org/XML/1998/namespace";
788  let math_nodes = document.findnodes("descendant-or-self::ltx:Math[@text]", None);
789
790  // Reuse allocations across Math nodes
791  let mut id_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
792  let mut idref_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
793  let mut id_map: std::collections::HashMap<String, String> = std::collections::HashMap::new();
794  let mut referenced_ids: std::collections::HashSet<String> = std::collections::HashSet::new();
795
796  for mut math_node in math_nodes {
797    let math_id = match math_node.get_attribute_ns("id", xml_ns) {
798      Some(id) => id,
799      None => continue,
800    };
801
802    let xmath_nodes = document.findnodes("ltx:XMath", Some(&math_node));
803    for xmath in xmath_nodes {
804      id_entries.clear();
805      idref_entries.clear();
806      id_map.clear();
807      referenced_ids.clear();
808
809      // Single DFS walk collects both xml:id and idref nodes in document order
810      renumber_collect_dfs(&xmath, xml_ns, &mut id_entries, &mut idref_entries);
811      if id_entries.is_empty() {
812        continue;
813      }
814
815      // Collect all referenced IDs (from XMRef idref attributes)
816      for (_, idref) in &idref_entries {
817        referenced_ids.insert(idref.clone());
818      }
819
820      // Strip xml:id from XMTok elements that are not referenced by any XMRef.
821      // The math parser assigns xml:ids to all tokens during parsing, but only
822      // structural nodes (XMApp, XMDual) and explicitly referenced tokens need them.
823      // Orphan XMTok ids inflate the renumbering counter causing ID gaps.
824      {
825        let mut stripped = false;
826        for (node, id) in &mut id_entries {
827          if node.get_name() == "XMTok" && !referenced_ids.contains(id.as_str()) {
828            document.unrecord_id(id);
829            let _ = node.remove_attribute("xml:id");
830            let _ = node.remove_attribute_ns("id", xml_ns);
831            id.clear(); // mark for removal
832            stripped = true;
833          }
834        }
835        if stripped {
836          id_entries.retain(|(_, id)| !id.is_empty());
837        }
838      }
839
840      if id_entries.is_empty() {
841        continue;
842      }
843
844      // Build old→new mapping. Flat sequential numbering under the math_id prefix,
845      // matching Perl's approach of assigning all IDs at the same level.
846      let mut counter = 0u32;
847      let mut any_changed = false;
848      for (_node, old_id) in &id_entries {
849        counter += 1;
850        let new_id = format!("{math_id}.{counter}");
851        if new_id != *old_id {
852          any_changed = true;
853        }
854        id_map.insert(old_id.clone(), new_id);
855      }
856
857      if !any_changed {
858        continue;
859      }
860
861      // Apply new xml:ids in TWO passes to avoid idstore collisions.
862      // A new id like "m1.1" would collide with an old "m1.1" still in the
863      // idstore if we interleave unrecord+record. Strip all first, then assign.
864      let mut nodes_to_update: Vec<(libxml::tree::Node, String)> = Vec::new();
865      for (mut node, old_id) in id_entries.drain(..) {
866        if let Some(new_id) = id_map.get(&old_id) {
867          if new_id != &old_id {
868            document.unrecord_id(&old_id);
869            let _ = node.remove_attribute("xml:id");
870            let _ = node.remove_attribute_ns("id", xml_ns);
871            nodes_to_update.push((node, new_id.clone()));
872          }
873        }
874      }
875      for (mut node, new_id) in nodes_to_update {
876        let _ = document.set_attribute(&mut node, "xml:id", &new_id);
877      }
878
879      // Update idrefs
880      for (mut node, old_idref) in idref_entries.drain(..) {
881        if let Some(new_idref) = id_map.get(&old_idref) {
882          if new_idref != &old_idref {
883            let _ = node.set_attribute("idref", new_idref);
884          }
885        }
886      }
887
888      // Reset _ID_counter__ on the Math node to the final count
889      let _ = math_node.set_attribute("_ID_counter__", &counter.to_string());
890    }
891  }
892}

894/// DFS walk collecting nodes with xml:id and idref attributes in document order.
895/// Stops at nested `Math` elements (which have their own parsing scope).
896fn renumber_collect_dfs(
897  node: &libxml::tree::Node,
898  xml_ns: &str,
899  id_entries: &mut Vec<(libxml::tree::Node, String)>,
900  idref_entries: &mut Vec<(libxml::tree::Node, String)>,
901) {
902  if let Some(id) = node.get_attribute_ns("id", xml_ns) {
903    id_entries.push((node.clone(), id));
904  }
905  if let Some(idref) = node.get_attribute("idref") {
906    idref_entries.push((node.clone(), idref));
907  }
908  for child in node.get_child_elements() {
909    // Skip nested Math elements — they have their own parsing scope
910    if child.get_name() == "Math" {
911      continue;
912    }
913    renumber_collect_dfs(&child, xml_ns, id_entries, idref_entries);
914  }
