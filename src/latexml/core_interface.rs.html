36// Regex for parsing DefMathRewrite calls from .latexml files
37// Matches: DefMathRewrite( ... );
38static DEF_MATH_REWRITE_RE: Lazy<Regex> = Lazy::new(|| {

81  fn initialize_singletons(&mut self, preloads: Vec<String>) -> Result<()> {
82    // reset the error REPORT singleton
83    error::initialize_report();
84    // reset localized variables (if_frames, current_token, align state, etc.)
85    latexml_core::common::local_assignments::initialize_localized();
86    // now handle conversion state
87    gullet::initialize_gullet();
88    stomach::initialize_stomach();
89    // should we reset the model also?
90    model::initialize_model();
91    // let paths = state::search_paths;
92    let dump_path = std::env::var("LATEXML_DUMP").ok();
93    state::assign_value("InitialPreloads", true, Some(Scope::Global));
94    for preload in preloads {
95      input_definitions(&preload, InputDefinitionOptions::default())?;
96    }
97    state::assign_value("InitialPreloads", false, Some(Scope::Global));
98
99    // Load kernel dump AFTER pools (provides TeX/LaTeX macros the pools skipped).
100    if let Some(ref dump_path) = dump_path {
101      let path = std::path::Path::new(dump_path);
102      if path.exists() {
103        let result = if dump_path.ends_with(".oxide") {
104          // Rust-native format (from --init mode)
105          latexml_core::dump_reader::load_native_dump(path)
106        } else {
107          // Perl format (from Perl's make formats)
108          latexml_core::dump_loader::load_dump(path)
109        };
110        match result {
111          Ok(count) => {
112            eprintln!(
113              "[latexml-oxide] Loaded {} kernel definitions from {}",
114              count, path.display()
115            );
116          }
117          Err(e) => {
118            eprintln!("[latexml-oxide] Warning: failed to load dump: {}", e);
119          }
120        }
121      }
122    }
123    Ok(())
124  }

212  /// Restriction: convert_document runs on a single thread, and should never try branching out.
213  fn convert_document(&mut self, digested: Digested) -> Result<Document> {
214    note_begin("Building");
215    let mut document = Document::new();
216    {
217      // TODO: Can we disentangle the ownership to avoid the clone?
218      let paths_stored = state::get_search_paths();
219      let schema_paths = paths_stored
220        .iter()
221        .map(String::as_str)
222        .collect::<Vec<&str>>();
223      let default_model_load = model::with_schema_data(|schema_opt| match schema_opt {
224        None => true,
225        Some(v) => v.last() == Some(&arena::pin_static("LaTeXML")),
226      });
227      if default_model_load {
228        // Compile-time load of model AND indirect model
229        load_model!("LaTeXML");
230      } else {
231        // Eager-load at runtime
232        model::load_schema(schema_paths.as_slice())?; // If needed?
233      }
234      if state::has_search_paths() {
235        {
236          if state::lookup_bool("INCLUDE_COMMENTS") {
237            let paths_string = state::with_search_paths(|paths| {
238              paths
239                .iter()
240                .map(String::as_str)
241                .collect::<Vec<&str>>()
242                .join(",")
243            });
244            let attributes = map! {s!("paths") => paths_string};
245            document.insert_pi("latexml", Some(attributes))?;
246          }
247        }
248      }
249    }
250
251    for preload in &self.preload {
252      if preload.ends_with(".pool") {
253        continue;
254      }
255      let mut options: Option<String> = None;
256      LATEX_OPTION_REGEX.replace_all(preload, |refs: &Captures| -> String {
257        options = Some(refs.get(1).map_or("", |m| m.as_str()).to_string());
258        String::new()
259      });
260      if preload.ends_with(".cls") {
261        CLS_EXT_REGEX.replace_all(preload, "");
262        let attributes = map! {s!("class") => preload.to_string()};
263        document.insert_pi("latexml", Some(attributes))?;
264      } else {
265        STY_EXT_REGEX.replace_all(preload, "");
266        let attributes = map! {s!("package") => preload.to_string()};
267        document.insert_pi("latexml", Some(attributes))?;
268      }
269    }
270    Debug!("Doc absorb: {:?}", digested);
271
272    document.absorb(&digested, None)?;
273    note_end("Building");
274
275    // Load .latexml file if it exists alongside the source .tex file.
276    // Perl does this automatically during initialization; we do it post-build
277    // so the rewrite rules can be compiled against the built document.
278    if let Some(Stored::String(source_sym)) = state::lookup_value("SOURCEFILE") {
279      let source_path = arena::with(source_sym, |s| s.to_string());
280      // Replace .tex extension with .latexml
281      let latexml_path = if source_path.ends_with(".tex") {
282        source_path.replace(".tex", ".latexml")
283      } else {
284        format!("{}.latexml", source_path)
285      };
286      if Path::new(&latexml_path).exists() {
287        let _ = load_latexml_file(&latexml_path);
288      }
289    }
290
291    let has_rewrites = state::has_value("DOCUMENT_REWRITE_RULES");
292    if has_rewrites {
293      note_begin("Rewriting");
294      document.mark_xmnode_visibility()?;
295      document.load_labels_for_rewrite()?;
296      // TODO: What is the right way to do rewrites in a daemon-safe manner?
297      if let Some(Stored::VecDequeStored(rules)) = state::remove_value("DOCUMENT_REWRITE_RULES") {
298        if let Some(root) = document.get_document().get_root_element() {
299          // Step 1: copy the rules locally through Rc, to be able to invoke them with mutable
300          // state. (TODO: obviously, this could be avoided if they never needed mutable
301          // state. When do they?)
302          let mut rewrites = Vec::new();
303          for rule in rules {
304            if let Stored::Rewrite(mut rewrite_rule) = rule {
305              rewrite_rule.compile_clauses(&mut document);
306              rewrites.push(rewrite_rule); // clone the Rc
307            }
308          }
309          // Step 2: invoke the rewrite rules
310          for mut rewrite_rule in rewrites {
311            rewrite_rule.invoke(&mut document, &root)?;
312          }
313        }
314      }
315      note_end("Rewriting");
316    }
317
318    // Apply \lxDeclare declarations: set roles/names/meanings on matching XMTok elements.
319    // Must run BEFORE math parsing so the parser sees the updated roles.
320    apply_lx_declarations(&mut document);
321
322    if !state::get_nomathparse_flag() {
323      let mut parser = MathParser::default();
324      parser.parse_math(&mut document)?;
325      // Post-parse: apply kludge to failed XMath nodes, then mark as unparsed
326      if !parser.failed_xmath_ids.is_empty() {
327        // Collect failed XMath nodes for kludge processing
328        let mut failed_xmaths = Vec::new();
329        for mut math_node in document.findnodes("descendant-or-self::ltx:Math[not(@text)]", None) {
330          for xmath_child in document.findnodes("ltx:XMath", Some(&math_node)) {
331            if parser.failed_xmath_ids.contains(&xmath_child.to_hashable()) {
332              failed_xmaths.push(xmath_child.clone());
333              document.add_class(&mut math_node, "ltx_math_unparsed")?;
334              break;
335            }
336          }
337        }
338        // Apply kludge to each failed XMath — balance OPEN/CLOSE delimiters
339        for mut xmath in failed_xmaths {
340          parse_kludge(&mut xmath, &mut document);
341        }
342      }
343      // Renumber xml:ids inside parsed XMath subtrees to be sequential in document
344      // order. The Marpa parser explores multiple parse alternatives, consuming ID
345      // counter slots for pruned nodes. This pass reassigns IDs post-parse.
346      renumber_math_ids(&mut document);
347    }
348
349    note_begin("Finalizing");
350    document.finalize()?;
351    note_end("Finalizing");
352    // Perl core produces role="UNKNOWN" for single-letter math tokens.
353    // Per-document .latexml files set role="ID" via DefMathRewrite BEFORE parsing.
354    // We do NOT apply a blanket conversion — roles are set by rewrite rules only.
355    // Cleanup unreferenced xml:ids on XMTok elements generated by the math parser.
356    // Must run after finalize (which includes prune_xmduals that may transfer ids).
357    document.cleanup_unreferenced_xmtok_ids();
358    Ok(document)
359  }

361  fn digest_internal(&mut self) -> Result<Digested> {
362    let mut boxes = Vec::new();
363    while gullet::has_more_input() {
364      let next_bodies: Vec<Digested> = stomach::digest_next_body(None)?;
365      // info!(target:"core:digest_next_body", "\n{:?}\n----\n",next_bodies);
366      boxes.extend(next_bodies);
367    }
368    gullet::flush();
369    Ok(Digested::from(List::new(boxes)))
370  }

381  /// Restriction: `digest_file` runs on a single thread, and should never try branching out.
382  fn digest_file(&mut self, mut request: String, options: DigestionOptions) -> Result<Digested> {
383    let mut dir = String::new();
384    let name;
385    // let mut ext = String::new();
386    let mode = match options.mode {
387      None => DigestionMode::TeX,
388      Some(m) => m,
389    };
390
391    if pathname::is_literaldata(&request) {
392      // ext = mode.extension();
393      name = s!("Anonymous String");
394    } else if pathname::is_url(&request) {
395      // ext = mode.extension();
396      name = request.clone();
397    } else {
398      let ext_str = s!(".{}", mode.extension());
399      let request_base = if request.ends_with(&ext_str) {
400        request[0..request.len() - ext_str.len()].to_string()
401      } else {
402        request
403      };
404
405      if let Some(pathname) = pathname::find(&request_base, PathnameFindOptions {
406        extensions: Some(vec![mode.extension(), String::new()]),
407        ..PathnameFindOptions::default()
408      }) {
409        request = pathname;
410        dir = pathname::directory(&request);
411        name = pathname::file_stem(&request);
412      // ext = pathname::extension(&request);
413      } else {
414        let message = s!("Can't find {} file {} ", mode, request_base);
415        fatal!(Core, MissingFile, message);
416      }
417    }
418    note_begin(&s!("Digesting {} {}", mode, name));
419    let main_pool = s!("{}.pool", mode);
420    let noinitialize = options.noinitialize.unwrap_or(false);
421    if !noinitialize {
422      let mut preloads = vec![main_pool];
423      preloads.extend(self.preload.clone());
424      self.initialize_singletons(preloads)?;
425    }
426    {
427      if !pathname::is_literaldata(&request) {
428        state::assign_value("SOURCEFILE", request.clone(), None);
429      }
430      if !dir.is_empty() {
431        state::assign_value("SOURCEDIRECTORY", dir.clone(), None);
432      }
433      state::search_paths_push_front(dir.clone());
434      state::graphics_paths_push_front(dir);
435      state::install_definition(
436        Stored::Expandable(Rc::new(Expandable {
437          cs: T_CS!("\\jobname"),
438          paramlist: None,
439          expansion: Tokens::new(Explode!(name)).into(),
440          ..Expandable::default()
441        })),
442        None,
443      );
444    }
445
446    // Reverse order, since last opened is first read!
447    if let Some(postamble) = options.postamble {
448      self.load_postamble(postamble);
449    }
450
451    {
452      // Make sure the stomach trick is used very *tightly*, always with a surrounding scope.
453      input_content(&request, InputOptions::default())?;
454    }
455
456    if let Some(preamble) = options.preamble {
457      self.load_preamble(preamble);
458    }
459
460    // Now for the Hacky part for BibTeX!!!
461    // if mode == DigestionMode::BibTeX {
462    //   let bib = LaTeXML::Pre::BibTeX->newFromGullet($name, $state->getStomach->getGullet);
463    //   LaTeXML::Package::InputContent("literal:" . $bib->toTeX);
464    // }
465
466    let list = self.digest_internal()?;
467    note_end(&s!("Digesting {} {}", mode, name));
468    Ok(list)
469  }
470}

472/// Load a `.latexml` file alongside a `.tex` source file.
473/// Parses `DefMathRewrite(...)` calls and registers them as rewrite rules.
474/// Perl loads these automatically; this provides the equivalent for Rust tests.
475///
476/// Supported patterns:
477///   - Single character: `match => 'a'` -> XPath on XMTok text content
478///   - Complex patterns (e.g. `\hat{f}`, `f_D`, `f_\WildCard`): skipped
479///   - `scope => 'label:...'`: scoped rewrites via label lookup
480///   - `attributes => { role => 'FUNCTION' }`: sets role (and optionally name/meaning)
481fn load_latexml_file(path: &str) -> Result<()> {
482  use latexml_core::rewrite::{RewriteClause, RewriteOperator, RewritePattern};
483
484  let content = match std::fs::read_to_string(path) {
485    Ok(c) => c,
486    Err(_) => return Ok(()), // File doesn't exist or can't be read
487  };
488
489  for cap in DEF_MATH_REWRITE_RE.captures_iter(&content) {
490    let body = &cap[1];
491
492    // Extract match pattern
493    let match_str = match MATCH_RE.captures(body) {
494      Some(m) => m[1].to_string(),
495      None => continue, // No match clause, skip
496    };
497
498    // Skip complex patterns: anything with \, _, {, } in the match string
499    // These need full token-level matching (digestion + DOM -> XPath) which we
500    // don't implement yet.
501    if match_str.contains('\\') || match_str.contains('_') || match_str.contains('{') {
502      continue;
503    }
504
505    // Build attributes map from the attributes => { ... } section
506    let mut attrs = HashMap::default();
507    if let Some(role_cap) = ROLE_RE.captures(body) {
508      attrs.insert("role".to_string(), role_cap[1].to_string());
509    }
510    if let Some(name_cap) = NAME_ATTR_RE.captures(body) {
511      attrs.insert("name".to_string(), name_cap[1].to_string());
512    }
513    if let Some(meaning_cap) = MEANING_RE.captures(body) {
514      attrs.insert("meaning".to_string(), meaning_cap[1].to_string());
515    }
516    if attrs.is_empty() {
517      continue; // No attributes to set
518    }
519
520    // Build the XPath for this match.
521    // For math mode, Perl's compile_match digests "$a$", builds DOM, then generates:
522    //   descendant-or-self::ltx:XMTok[text()='a'][@_pvis and @_cvis]
523    // We generate the equivalent XPath directly for single-character matches.
524    let xpath = format!(
525      "descendant-or-self::ltx:XMTok[text()='{}'][@_pvis and @_cvis]",
526      match_str
527    );
528
529    // Check for optional scope
530    let scope_str = SCOPE_RE.captures(body).map(|s| s[1].to_string());
531
532    // Build the rewrite rule with pre-compiled clauses
533    let mut clauses = Vec::new();
534
535    // Add scope clause if present (compiled during compile_clauses phase)
536    if let Some(ref scope) = scope_str {
537      clauses.push(RewriteClause::new_uncompiled(
538        RewriteOperator::Scope,
539        RewritePattern::String(scope.clone()),
540      ));
541    }
542
543    // Add match clause (pre-compiled as XPath string)
544    clauses.push(RewriteClause::new_uncompiled(
545      RewriteOperator::Match,
546      RewritePattern::String(xpath),
547    ));
548
549    // Add attributes clause
550    clauses.push(RewriteClause::new_compiled(
551      RewriteOperator::Attributes,
552      RewritePattern::String(String::new()),
553    ));
554
555    let rewrite = Rewrite {
556      options: RewriteOptions {
557        attributes_map: Some(attrs),
558        is_math: true,
559        select_count: Some(1),
560        ..RewriteOptions::default()
561      },
562      clauses,
563    };
564
565    state::push_value("DOCUMENT_REWRITE_RULES", rewrite)?;
566  }
567
568  Ok(())
569}

571/// Apply \lxDeclare declarations to the document.
572/// Simple fast-path: matches single-token patterns in XMTok elements
573/// and sets role/name/meaning attributes.
574fn apply_lx_declarations(document: &mut Document) {
575  let decls_str = match state::lookup_value("LATEXML_DECLARATIONS") {
576    Some(Stored::String(s)) => arena::with(s, |r| r.to_string()),
577    _ => return,
578  };
579  if decls_str.is_empty() {
580    return;
581  }
582
583  // Parse declarations: "token_text\trole\tname\tmeaning" per line
584  let declarations: Vec<(&str, &str, &str, &str)> = decls_str
585    .lines()
586    .filter_map(|line| {
587      let parts: Vec<&str> = line.splitn(4, '\t').collect();
588      if parts.len() == 4 {
589        Some((parts[0], parts[1], parts[2], parts[3]))
590      } else {
591        None
592      }
593    })
594    .collect();
595
596  if declarations.is_empty() {
597    return;
598  }
599
600  // Find all XMTok elements in the document and apply matching declarations
601  let xmtoks = document.findnodes("descendant-or-self::ltx:XMTok", None);
602  for mut tok in xmtoks {
603    let content = tok.get_content();
604    let tok_name = tok.get_attribute("name").unwrap_or_default();
605    for &(pattern, role, name, meaning) in &declarations {
606      // Match by content text, or by XMTok name attribute (for CS patterns like \circ)
607      let matches = content == pattern
608        || (!tok_name.is_empty() && pattern.starts_with('\\') && pattern[1..] == tok_name);
609      if matches {
610        if !role.is_empty() {
611          let _ = tok.set_attribute("role", role);
612        }
613        if !name.is_empty() {
614          let _ = tok.set_attribute("name", name);
615        }
616        if !meaning.is_empty() {
617          let _ = tok.set_attribute("meaning", meaning);
618        }
619        break; // First matching declaration wins
620      }
621    }
622  }
623}

625/// Fallback parser for unparseable math expressions.
626/// Perl: MathParser.pm parse_kludge().
627/// Balances OPEN/CLOSE delimiters by wrapping matched groups in XMWrap.
628/// Uses document.wrap_nodes for proper namespace handling.
629fn parse_kludge(mathnode: &mut libxml::tree::Node, document: &mut Document) {
630  use latexml_math_parser::get_grammatical_role;
631  let children: Vec<libxml::tree::Node> = mathnode
632    .get_child_elements()
633    .into_iter()
634    .filter(|n| n.get_name() != "XMHint")
635    .collect();
636  if children.is_empty() {
637    return;
638  }
639
640  // Phase 1: Find OPEN/CLOSE pairs and wrap each group using wrap_nodes.
641  // We process innermost pairs first (like matching parentheses).
642  // Iterate until no more OPEN/CLOSE pairs are found.
643  let mut changed = true;
644  while changed {
645    changed = false;
646    let elems = mathnode.get_child_elements();
647    // Find the innermost OPEN that has a matching CLOSE
648    let mut open_idx = None;
649    for (i, n) in elems.iter().enumerate() {
650      let role = get_grammatical_role(n);
651      if role == "OPEN" {
652        open_idx = Some(i);
653      } else if role == "CLOSE" && open_idx.is_some() {
654        // Found innermost OPEN..CLOSE pair
655        let start = open_idx.unwrap();
656        let end = i;
657        // Collect the nodes between OPEN and CLOSE (inclusive)
658        let group: Vec<libxml::tree::Node> =
659          elems[start..=end].to_vec();
660        if group.len() > 1 {
661          let _ = document.wrap_nodes("ltx:XMWrap", group);
662          changed = true;
663        }
664        break; // Restart after wrapping
665      }
666    }
667    // If we found an unmatched OPEN (no CLOSE), wrap OPEN through end
668    if !changed {
669      if let Some(oi) = open_idx {
670        let elems = mathnode.get_child_elements();
671        let start = oi.min(elems.len().saturating_sub(1));
672        if start < elems.len() {
673          let group: Vec<libxml::tree::Node> =
674            elems[start..].to_vec();
675          if group.len() > 1 {
676            let _ = document.wrap_nodes("ltx:XMWrap", group);
677            changed = true;
678          }
679        }
680      }
681    }
682  }
683  // Phase 2: Unwrap ALL top-level XMWrap nodes.
684  // Perl: foreach pair, if kludge is XMWrap, extract children (not the wrapper).
685  // This preserves flat structure for simple expressions like <1>, [1], (1).
686  loop {
687    let top_elems = mathnode.get_child_elements();
688    let wrap_opt = top_elems.into_iter().find(|n| n.get_name() == "XMWrap");
689    if let Some(wrap) = wrap_opt {
690      let _ = document.unwrap_nodes(wrap);
691    } else {
692      break;
693    }
694  }
695}

697/// Renumber xml:ids inside parsed XMath subtrees so they are sequential in
698/// document order. The Marpa parser explores multiple parse alternatives,
699/// consuming ID counter slots for pruned nodes (e.g. m1.1, m1.7, m1.12
700/// instead of m1.1, m1.2, m1.3). This pure post-processing pass reassigns
701/// IDs after all pruning is complete.
702///
703/// Optimized: single DFS walk per XMath (not XPath), O(1) parent-prefix
704/// lookup via ID string parsing, and allocation reuse across Math nodes.
705fn renumber_math_ids(document: &mut Document) {
706  let xml_ns = "http://www.w3.org/XML/1998/namespace";
707  let math_nodes = document.findnodes("descendant-or-self::ltx:Math[@text]", None);
708
709  // Reuse allocations across Math nodes
710  let mut id_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
711  let mut idref_entries: Vec<(libxml::tree::Node, String)> = Vec::new();
712  let mut id_map: std::collections::HashMap<String, String> = std::collections::HashMap::new();
713  let mut referenced_ids: std::collections::HashSet<String> = std::collections::HashSet::new();
714
715  for mut math_node in math_nodes {
716    let math_id = match math_node.get_attribute_ns("id", xml_ns) {
717      Some(id) => id,
718      None => continue,
719    };
720
721    let xmath_nodes = document.findnodes("ltx:XMath", Some(&math_node));
722    for xmath in xmath_nodes {
723      id_entries.clear();
724      idref_entries.clear();
725      id_map.clear();
726      referenced_ids.clear();
727
728      // Single DFS walk collects both xml:id and idref nodes in document order
729      renumber_collect_dfs(&xmath, xml_ns, &mut id_entries, &mut idref_entries);
730      if id_entries.is_empty() {
731        continue;
732      }
733
734      // Collect all referenced IDs (from XMRef idref attributes)
735      for (_, idref) in &idref_entries {
736        referenced_ids.insert(idref.clone());
737      }
738
739      // Strip xml:id from XMTok elements that are not referenced by any XMRef.
740      // The math parser assigns xml:ids to all tokens during parsing, but only
741      // structural nodes (XMApp, XMDual) and explicitly referenced tokens need them.
742      // Orphan XMTok ids inflate the renumbering counter causing ID gaps.
743      {
744        let mut stripped = false;
745        for (node, id) in &mut id_entries {
746          if node.get_name() == "XMTok" && !referenced_ids.contains(id.as_str()) {
747            document.unrecord_id(id);
748            let _ = node.remove_attribute("xml:id");
749            let _ = node.remove_attribute_ns("id", xml_ns);
750            id.clear(); // mark for removal
751            stripped = true;
752          }
753        }
754        if stripped {
755          id_entries.retain(|(_, id)| !id.is_empty());
756        }
757      }
758
759      if id_entries.is_empty() {
760        continue;
761      }
762
763      // Build old→new mapping. Flat sequential numbering under the math_id prefix,
764      // matching Perl's approach of assigning all IDs at the same level.
765      let mut counter = 0u32;
766      let mut any_changed = false;
767      for (_node, old_id) in &id_entries {
768        counter += 1;
769        let new_id = format!("{math_id}.{counter}");
770        if new_id != *old_id {
771          any_changed = true;
772        }
773        id_map.insert(old_id.clone(), new_id);
774      }
775
776      if !any_changed {
777        continue;
778      }
779
780      // Apply new xml:ids in TWO passes to avoid idstore collisions.
781      // A new id like "m1.1" would collide with an old "m1.1" still in the
782      // idstore if we interleave unrecord+record. Strip all first, then assign.
783      let mut nodes_to_update: Vec<(libxml::tree::Node, String)> = Vec::new();
784      for (mut node, old_id) in id_entries.drain(..) {
785        if let Some(new_id) = id_map.get(&old_id) {
786          if new_id != &old_id {
787            document.unrecord_id(&old_id);
788            let _ = node.remove_attribute("xml:id");
789            let _ = node.remove_attribute_ns("id", xml_ns);
790            nodes_to_update.push((node, new_id.clone()));
791          }
792        }
793      }
794      for (mut node, new_id) in nodes_to_update {
795        let _ = document.set_attribute(&mut node, "xml:id", &new_id);
796      }
797
798      // Update idrefs
799      for (mut node, old_idref) in idref_entries.drain(..) {
800        if let Some(new_idref) = id_map.get(&old_idref) {
801          if new_idref != &old_idref {
802            let _ = node.set_attribute("idref", new_idref);
803          }
804        }
805      }
806
807      // Reset _ID_counter__ on the Math node to the final count
808      let _ = math_node.set_attribute("_ID_counter__", &counter.to_string());
809    }
810  }
811}

813/// DFS walk collecting nodes with xml:id and idref attributes in document order.
814/// Stops at nested `Math` elements (which have their own parsing scope).
815fn renumber_collect_dfs(
816  node: &libxml::tree::Node,
817  xml_ns: &str,
818  id_entries: &mut Vec<(libxml::tree::Node, String)>,
819  idref_entries: &mut Vec<(libxml::tree::Node, String)>,
820) {
821  if let Some(id) = node.get_attribute_ns("id", xml_ns) {
822    id_entries.push((node.clone(), id));
823  }
824  if let Some(idref) = node.get_attribute("idref") {
825    idref_entries.push((node.clone(), idref));
826  }
827  for child in node.get_child_elements() {
828    // Skip nested Math elements — they have their own parsing scope
829    if child.get_name() == "Math" {
830      continue;
831    }
832    renumber_collect_dfs(&child, xml_ns, id_entries, idref_entries);
833  }
834}
