21/// Resolve a runtime `<request>.rhai` binding from the local search paths
22/// (source directory + `--path`) and load it if present. Local-only — never
23/// consults kpsewhich, since `.rhai` files are user-supplied customizations,
24/// not TeX-tree resources (and probing kpsewhich for a non-existent `.rhai`
25/// on every package load would be wasteful). Returns `None` when no such file
26/// exists, so the caller falls through to the compiled dispatchers.
27/// See `docs/script_bindings_plan.md` §7.
28#[cfg(feature = "runtime-bindings")]
29fn rhai_dispatch(request: &str) -> Option<Result<()>> {
30  use latexml_core::binding::content::{FindFileOptions, find_file};
31  let path = find_file(
32    request,
33    Some(FindFileOptions {
34      // Append `.rhai` (so `foo.sty` resolves `foo.sty.rhai`) and search the
35      // local paths only.
36      ext_type: Some("rhai".into()),
37      search_paths_only: true,
38      ..FindFileOptions::default()
39    }),
40  )?;
41  Some(latexml_contrib::script_bindings::load_file(&path).map(|_| ()))
42}

44/// Install the binding-resolution **priority chain** as the single dispatcher
45/// and register the `latexml_package` binding-name registry.
46///
47/// This is the one definition of binding-resolution policy, shared by
48/// [`Converter::initialize_session`] and the integration-test harness
49/// (`util::test::process_texfile`) — so a test exercises the *same* chain a real
50/// conversion does, and local `.rhai` fixtures are discovered identically.
51///
52/// Priority, highest first (installed in ONE slot, so call-site ordering can't
53/// reshuffle it):
54///   1. a local `<request>.rhai` in the search paths — lets a user (or a test
55///      fixture next to its `.tex`) OVERRIDE any compiled binding of the same
56///      name (e.g. `article.cls.rhai` shadows `article_cls`). `runtime-bindings`
57///      only.
58///   2. the `extra` dispatcher (`latexml_contrib` for our binaries) — consulted
59///      before `latexml_package` to preserve the prior external-before-internal
60///      order; the two registries are disjoint, so the order is immaterial.
61///   3. `latexml_package` — core compiled engine bindings.
62pub(crate) fn install_binding_dispatch(extra: Option<BindingDispatcher>) {
63  set_bindings_dispatch(Rc::new(move |request: &str| {
64    #[cfg(feature = "runtime-bindings")]
65    if let Some(result) = rhai_dispatch(request) {
66      return Some(result);
67    }
68    if let Some(extra) = extra.as_ref()
69      && let Some(result) = extra(request)
70    {
71      return Some(result);
72    }
73    latexml_package::dispatch(request)
74  }));
75  // Register every (name, ext) binding pair so `find_file(notex=true)` can
76  // resolve compile-time bindings across all extensions
77  // (.cls/.sty/.def/.pool/code.tex/...). This also feeds `load_class`'s
78  // Perl-parity prefix-match fallback (Package.pm L2702-2706) via the
79  // class-filtered `state::get_class_binding_names()` view. Source of
80  // truth: `latexml_package::BINDINGS`.
81  add_binding_names(latexml_package::binding_names());
82}

122  pub fn initialize_session(&mut self) -> Result<()> {
123    // Install the binding-resolution priority chain (rhai > contrib > package)
124    // — the single source of resolution policy, shared with the integration-test
125    // harness via `install_binding_dispatch`.
126    install_binding_dispatch(self.opts.extra_bindings_dispatch.clone());
127    // Also expose contrib's bindings (memoir / siamltex / scrbook / etc.)
128    // so they participate in the same resolution pool. We unconditionally
129    // register latexml_contrib here because the canonical setup for both
130    // `latexml_oxide` and `cortex_worker` binaries loads it — downstream
131    // embedders that replace the dispatchers can register their own
132    // (name, ext) slice the same way via `add_binding_names`.
133    add_binding_names(latexml_contrib::binding_names());
134    // Prepare LaTeXML object — load mode-specific pool + user preloads.
135    // Perl: $self->initializeState($mode.".pool", @{$$self{preload} || []})
136    // For `--bibtex` (mode = BibTeX), Perl `Common/Config.pm:406`
137    // unshifts ['TeX.pool', 'LaTeX.pool', 'BibTeX.pool'] into the preload
138    // list. `BibTeX.pool` already begins with `LoadPool('LaTeX')` (and
139    // LaTeX with TeX), so we only need the BibTeX entry — the transitive
140    // chain handles the rest, and pool loads are idempotent.
141    let mut preloads = match self.opts.mode {
142      Some(DigestionMode::BibTeX) => vec![s!("TeX.pool"), s!("BibTeX.pool")],
143      _ => vec![s!("TeX.pool")],
144    };
145    preloads.extend(self.core.preload.iter().cloned());
146    self.core.initialize_singletons(preloads)?;
147    self.ready = true;
148    Ok(())
149  }

454  /// Convert in-memory `content` under the source name `name`, producing the
455  /// HTML5-format core XML (the persistent server then post-processes it).
456  /// Unlike [`Converter::convert`] (`literal:` → anonymous source), the source
457  /// is *named*, so `--source-map` stamps its locators. Focused on the
458  /// `Document`/HTML5 path the server uses — no amble wrapping, no TeX/Box
459  /// output formats.
460  pub fn convert_content_with_provenance(
461    mut self,
462    name: &str,
463    content: String,
464  ) -> ConversionResponse {
465    // Load + digest through the shared top-level loader, so the source-context
466    // setup is not duplicated here.
467    let digested = match self.digest_content_with_provenance(name, content) {
468      Ok(d) => d,
469      Err(e) => {
470        if !self.ready {
471          return ConversionResponse {
472            result:      None,
473            log:         self.flush_log(),
474            status:      s!("Initialization failed."),
475            status_code: 3,
476          };
477        }
478        report_mut!().status_code = 3;
479        e.log_fatal();
480        // Salvage whatever digested before the error (mirrors `convert`).
481        match self.core.digest_internal() {
482          Ok(salvaged) if !salvaged.is_empty().unwrap_or(true) => salvaged,
483          _ => Digested::from(List::new(Vec::new())),
484        }
485      },
486    };
487
488    self.runtime.status = get_status_message();
489    self.runtime.status_code = get_status_code();
490
491    let serialized = {
492      let _g = telemetry::phase(Phase::Build);
493      match self.core.convert_document(digested) {
494        Ok(dom) => {
495          let _g = telemetry::phase(Phase::Serialize);
496          dom.serialize_to_string()
497        },
498        Err(e) => {
499          // `Error!` expands into a `Result`-returning context; wrap it the
500          // same way `convert` does so it composes in this `-> ConversionResponse` fn.
501          // Timeout-target resource fatals get the standard `Fatal:` line
502          // (see the sibling handler in `convert` — P1-4).
503          if matches!(e.target, ErrorTarget::Timeout) {
504            e.log_fatal();
505          } else {
506            let message = s!("{:?}", e);
507            let err = || {
508              Error!("document", "convert", message);
509              Ok(())
510            };
511            err().ok();
512          }
513          String::new()
514        },
515      }
516    };
517
518    let log = self.flush_log();
519    ConversionResponse {
520      result: Some(serialized),
521      log,
522      status: self.runtime.status.clone(),
523      status_code: self.runtime.status_code,
524    }
525  }

537  /// Digest in-memory `content` as the **main document** named `name`, leaving
538  /// the thread-local engine state **live**. Used by the persistent server to
539  /// warm a preamble once (then resume body digestion in a fork child over the
540  /// inherited state) and by [`Converter::convert_content_with_provenance`] for
541  /// the in-process path.
542  ///
543  /// This is the in-memory twin of [`crate::core_interface::DigestionAPI::digest_file`]:
544  /// same top-level spine — establish the source context, open the source,
545  /// `digest_internal` — but the content is *supplied* rather than read from
546  /// disk. It shares [`crate::core_interface::establish_source_context`] with
547  /// `digest_file` (so `SOURCEFILE`/`SOURCEDIRECTORY`/`SEARCHPATHS`/
548  /// `GRAPHICSPATHS`/`\jobname` can't drift), making sibling
549  /// `\usepackage`/`\input`/`\includegraphics` of local files resolve. The
550  /// source is opened as a *named* mouth (not the anonymous `literal:`
551  /// protocol) so locators carry `name` — the **provenance** that
552  /// `--source-map` needs (`stamp_source_locator` only stamps
553  /// `.tex`/`.ltx`/`.bbl`/`.bib` user sources). Initializes the session if
554  /// needed; does not finalize a document.
555  pub fn digest_content_with_provenance(
556    &mut self,
557    name: &str,
558    content: String,
559  ) -> Result<Digested> {
560    if !self.ready {
561      self.initialize_session()?;
562    }
563    self.bind_log();
564    // Top-level document load: establish the source context (SOURCEFILE,
565    // SOURCEDIRECTORY, SEARCHPATHS, GRAPHICSPATHS, \jobname) so sibling
566    // \usepackage/\input/\includegraphics of local files resolve. Shared with
567    // `digest_file` via `establish_source_context` so the two can't drift.
568    // (A continuation/nested mouth must NOT do this — see
569    // `open_named_in_memory_mouth`.)
570    let path = std::path::Path::new(name);
571    let dir = path.parent().and_then(|p| p.to_str()).unwrap_or("");
572    let jobname = path.file_stem().and_then(|s| s.to_str()).unwrap_or(name);
573    crate::core_interface::establish_source_context(Some(name), jobname, dir);
574    open_named_in_memory_mouth(name, content)?;
575    self.core.digest_internal()
576  }
577}

579/// Open a gullet mouth over in-memory `content` whose source is named `name`
580/// (a real path/filename). Uses the Mouth's cached-content branch so locators
581/// carry `name` rather than "Anonymous String".
582///
583/// Low-level and *position-agnostic*: it does NOT touch the document-global
584/// `SOURCEDIRECTORY`/`SEARCHPATHS`, so it is safe for a continuation (the
585/// forked child's body over already-inherited state) or a nested include. For
586/// the *main* document load, go through [`Converter::digest_named`], which
587/// installs the document directory first.
588pub fn open_named_in_memory_mouth(name: &str, content: String) -> Result<()> {
589  use latexml_core::{
590    gullet,
591    mouth::{Mouth, MouthOptions},
592  };
593  let mouth = Mouth::create(name, MouthOptions {
594    notes: true,
595    content: Some(content),
596    ..MouthOptions::default()
597  })?;
598  gullet::open_mouth(mouth, true);
599  Ok(())
600}

602/// Resolve the `(preamble, postamble)` to wrap the source in, based on
603/// the requested input chunk size. Faithful port of Perl `LaTeXML.pm`
604/// L165-172 — note both ambles key on **`whatsin`** (not `whatsout`):
605///
606/// * `math` → `\begin{document}\ensuremathfollows` … `\ensuremathpreceeds\end{document}` (magic
607///   math-mode trigger).
608/// * `fragment` → the caller-supplied `preamble`/`postamble`, defaulting to `standard_preamble.tex`
609///   / `standard_postamble.tex`.
610/// * everything else (`document`, `archive`, …) → no wrapping.
611pub(crate) fn resolve_amble(
612  whatsin: &DataSize,
613  preamble: &Option<String>,
614  postamble: &Option<String>,
615) -> (Option<String>, Option<String>) {
616  match whatsin {
617    DataSize::Math => (
618      Some(s!("literal:\\begin{{document}}\\ensuremathfollows")),
619      Some(s!("literal:\\ensuremathpreceeds\\end{{document}}")),
620    ),
621    DataSize::Fragment => (
622      Some(
623        preamble
624          .clone()
625          .unwrap_or_else(|| s!("standard_preamble.tex")),
626      ),
627      Some(
628        postamble
629          .clone()
630          .unwrap_or_else(|| s!("standard_postamble.tex")),
631      ),
632    ),
633    _ => (None, None),
634  }
635}
