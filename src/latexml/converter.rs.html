46  pub fn initialize_session(&mut self) -> Result<()> {
47    // Add default package bindings
48    set_bindings_dispatch(Rc::new(latexml_package::dispatch));
49    // Add additional binding definitions if any
50    if let Some(closure) = &self.opts.extra_bindings_dispatch {
51      set_extra_bindings_dispatch(closure.clone());
52    }
53    // Prepare LaTeXML object
54    self.core.initialize_singletons(vec![s!("TeX.pool")])?;
55    self.ready = true;
56    Ok(())
57  }
