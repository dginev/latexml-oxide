1//! Format dump mode — Rust equivalent of Perl's iniTeX + DumpFile.
2//!
3//! Usage: `latexml_oxide --init=latex.ltx --dest=latex_dump.oxide`
4//!
5//! Follows Perl's make formats scaffold (Makefile.PL + Core.pm::iniTeX):
6//! 1. Initialize the engine (load pools)
7//! 2. Take a snapshot of the state
8//! 3. Process the init file (e.g., latex.ltx) as raw TeX
9//! 4. Compute the diff (what changed)
10//! 5. Write the dump file with changed entries
11//!
12//! The resulting dump can be loaded at runtime to skip re-processing
13//! the LaTeX kernel on every test run.
14

27/// Process an init file and write a format dump.
28/// Perl equivalent: Core.pm::iniTeX → TeX_Job.pool.ltxml::DumpFile
29pub fn dump_format(
30  _converter: &mut Converter,
31  init_file: &str,
32  destination: Option<&str>,
33) -> Result<usize, String> {
34  eprintln!("[ini_tex] Dumping format from {}", init_file);
35
36  // Strict Perl `iniTeX` + `DumpFile` order:
37  //
38  //   Core.pm L168-212 (iniTeX, default mode='Base'):
39  //     initializeState('Base.pool');     # ← Step 1 below
40  //     installDefinition('\jobname', ...);
41  //     installDefinition('\dump', Tokens());  # no-op
42  //     DumpFile($file, $dest);           # ← Step 2 below
43  //
44  //   TeX_Job.pool.ltxml L120-220 (DumpFile):
45  //     LoadPool($name . '_bootstrap');   # ← Step 3 below
46  //     $snap = ...                       # ← Step 4 below
47  //     loadTeXDefinitions($name, ...)    # ← Step 5 below
48  //     diff                              # ← Step 6 below
49  //     write                             # ← Step 7 below
50  //
51  // CRITICAL: Perl iniTeX defaults to `mode='Base'`, so `Base.pool` is
52  // loaded BEFORE any bootstrap. Without it, raw plain.tex / latex.ltx
53  // can't expand any TeX primitive — every `\def`, `\catcode`,
54  // `\let`, `\edef` etc. is undefined and we get an error cascade.
55  // After Base.pool, only `<name>_bootstrap` is loaded — NEVER
56  // `<name>_base`, `<name>_dump`, or `<name>_constructs`. Those
57  // pollute the diff with `:locked` flags, base/constructs
58  // definitions, etc. that the dump should NOT carry.
59
60  // Step 1: Load Base.pool equivalent (Perl `initializeState('Base.pool')`).
61  eprintln!("[ini_tex] Loading Base.pool (Perl `initializeState('Base.pool')`)");
62  if let Err(e) = latexml_package::engine::base::load_definitions() {
63    eprintln!("[ini_tex] base warning: {}", e);
64  }
65
66  // Mark this as init/dump mode so machinery elsewhere (notably
67  // `tex_file_io::\\input`'s LaTeX-style brace-arg auto-load of
68  // `LaTeX.pool`) skips behaviors that would corrupt the dump-build
69  // — see `tex_file_io.rs` for the gate.
70  state::assign_value("INI_TEX_MODE", true, Some(state::Scope::Global));
71
72  // Clear LaTeX/expl3/AmSTeX autoload triggers and `\documentstyle`
73  // installed by `tex.rs` during `prepare_session`. These triggers
74  // pre-define `\makeatletter`, `\documentclass`, etc. — which then
75  // poison the snapshot, causing raw `latex.ltx` at L1798
76  // (`\DeclareRobustCommand\makeatletter`) to hit the "redefining"
77  // branch in `\declare@robustcommand` (L1388), which calls
78  // `\@latex@info{Redefining ...}` — but `\@latex@info` isn't
79  // defined until L1799, triggering an undefined-CS cascade.
80  //
81  // Perl `Core.pm::iniTeX` defaults to `mode='Base'` for dump-build,
82  // so `Base.pool` is loaded but `TeX.pool`'s autoload triggers are
83  // NOT. Mirror that here by clearing them right before the snapshot.
84  for trigger in &[
85    // LaTeX autoload triggers (tex.rs L149-167)
86    "\\documentclass",
87    "\\newcommand",
88    "\\renewcommand",
89    "\\newenvironment",
90    "\\renewenvironment",
91    "\\NeedsTeXFormat",
92    "\\ProvidesPackage",
93    "\\RequirePackage",
94    "\\ProvidesFile",
95    "\\makeatletter",
96    "\\makeatother",
97    "\\begin",
98    "\\listfiles",
99    "\\nofiles",
100    "\\typeout",
101    "\\PassOptionsToPackage",
102    // `\@load@latex@pool` itself
103    "\\@load@latex@pool",
104    // expl3 autoload triggers
105    "\\ExplSyntaxOn",
106    "\\ProvidesExplClass",
107    "\\ProvidesExplPackage",
108    // AmSTeX/amsmath autoload triggers
109    "\\mathfrak",
110    "\\mathbb",
111    "\\Bbb",
112    "\\theoremstyle",
113    "\\numberwithin",
114    "\\align",
115    "\\subequations",
116    "\\multline",
117    "\\curraddr",
118    "\\subjclass",
119    // `\documentstyle` was also defined in tex.rs as a runtime macro
120    "\\documentstyle",
121  ] {
122    state::assign_meaning(
123      &latexml_core::T_CS!(*trigger),
124      latexml_core::common::store::Stored::None,
125      Some(state::Scope::Global),
126    );
127  }
128
129  // Step 2 + 3: install \jobname / \dump (no-op), then load bootstrap.
130  // (Perl Core.pm L204-207 + TeX_Job.pool.ltxml L127-129)
131  let init_lower = init_file.to_ascii_lowercase();
132  let is_plain_init = init_lower.contains("plain");
133
134  if is_plain_init {
135    eprintln!("[ini_tex] Loading plain_bootstrap (mirrors Perl `LoadPool('plain_bootstrap')`)");
136    if let Err(e) = latexml_package::engine::plain_bootstrap::load_definitions() {
137      eprintln!("[ini_tex] plain_bootstrap warning: {}", e);
138    }
139  } else {
140    eprintln!("[ini_tex] Loading latex_bootstrap (mirrors Perl `LoadPool('latex_bootstrap')`)");
141    // latex_bootstrap.rs L11 does `InnerPool!(plain_bootstrap)` itself
142    // (mirrors Perl `LoadPool('plain_bootstrap')` at the top of
143    // latex_bootstrap.pool.ltxml), so plain_bootstrap state is included.
144    if let Err(e) = latexml_package::engine::latex_bootstrap::load_definitions() {
145      eprintln!("[ini_tex] latex_bootstrap warning: {}", e);
146    }
147  }
148
149  // Perl `DumpFile` L132-138: snapshot all tables AFTER the bootstrap pool.
150  let snap = state::take_snapshot();
151  // Re-stage as "bootstrap" so `dump_writer` finds it for let-alias
152  // classification (early/late sections).
153  state::stage_snapshot_value("bootstrap", snap.clone());
154  let snap_size = snap.len();
155  eprintln!(
156    "[ini_tex] Snapshot taken at bootstrap ({} entries)",
157    snap_size
158  );
159
160  // Step 2: Process the init file as raw TeX definitions.
161  // Perl: loadTeXDefinitions($name, $path, type => $type)
162  // This digests the file through the engine, creating definitions.
163  let (_, name, ext) = split_path(init_file);
164  eprintln!("[ini_tex] Loading {} (ext: {})", name, ext);
165
166  // Lift the token limit for format dumps — expl3-code.tex alone uses ~5M tokens.
167  let saved_limit = latexml_core::gullet::set_token_limit(None);
168
169  // In init mode, suppress error/warning output during format loading.
170  // Raw latex.ltx redefines commands already in the compiled engine ("already defined"),
171  // and expl3-code.tex has forward references that produce transient errors.
172  // All these errors are benign — the dump captures the final correct state.
173  // Set LATEXML_INIT_DEBUG=1 to keep errors visible (for debugging the
174  // expl3 cascade — Perl parity target is zero errors during expl3 load).
175  let init_debug = *INIT_DEBUG;
176  let prev_suppress = latexml_core::common::error::set_suppress_log_output(!init_debug);
177
178  // Suppress known expl3 loading errors at the state level too
179  state::assign_value("SUPPRESS_UNDEFINED_ERRORS", !init_debug, None);
180  state::assign_value("SUPPRESS_UNEXPECTED_ERRORS", !init_debug, None);
181
182  // Lift the MAX_ERRORS cap during dump-build. Raw latex.ltx contains
183  // many CSes our engine reports as errors (forward references in
184  // expl3-code.tex, `\@onlypreamble` checks, autoload triggers, etc.).
185  // The default 10000-error cap aborts dump-build before plain.tex's
186  // `\outer\def\newread`, `\loop`, etc. land in the diff. Mirrors Perl
187  // `DumpFile`'s behavior — Perl runs latex.ltx through to `\dump`
188  // regardless of error count.
189  state::assign_value("MAX_ERRORS", 1_000_000_i64, None);
190
191  // Use the full filename with extension for proper file resolution
192  let load_name = if ext.is_empty() {
193    name.clone()
194  } else {
195    format!("{}.{}", name, ext)
196  };
197  let result = input_definitions(&load_name, InputDefinitionOptions {
198    noltxml: true,
199    ..InputDefinitionOptions::default()
200  });
201  if let Err(e) = result {
202    eprintln!("[ini_tex] Warning during loading: {}", e);
203  }
204
205  // Restore limits and suppression
206  latexml_core::gullet::restore_token_limit(saved_limit);
207  latexml_core::common::error::set_suppress_log_output(prev_suppress);
208  state::assign_value("SUPPRESS_UNDEFINED_ERRORS", false, None);
209  state::assign_value("SUPPRESS_UNEXPECTED_ERRORS", false, None);
210
211  // Step 3: Compute the diff — only entries that changed.
212  let diff = state::diff_snapshot(&snap);
213  eprintln!(
214    "[ini_tex] Post-load diff: {} changed entries (from {} pre-dump)",
215    diff.len(),
216    snap_size
217  );
218
219  // Step 4: Write the dump.
220  // Default: write text dump to resources/dumps/<kind>.<YYYY>.dump.txt for
221  // build.rs embedding. With --dest: write to the specified path.
222  let kind = if name.contains("latex") { "latex" } else { "plain" };
223  let ambient_year = latexml_engine::dump_paths::detect_ambient_texlive_year();
224  let (dest, is_text_dump) = match destination {
225    Some(d) if d.ends_with(".rs") => (d.to_string(), false),
226    Some(d) => (d.to_string(), true),
227    None => {
228      let year = ambient_year.ok_or_else(|| {
229        "Could not detect ambient TeXLive year (no kpsewhich/pdflatex). \
230         Pass --dest=<path> to override the dump filename."
231          .to_string()
232      })?;
233      let dump_name = latexml_engine::dump_paths::dump_filename(kind, year);
234      let dump_dir = "resources/dumps";
235      std::fs::create_dir_all(dump_dir)
236        .map_err(|e| format!("Failed to create {}: {}", dump_dir, e))?;
237      (format!("{}/{}", dump_dir, dump_name), true)
238    },
239  };
240
241  if is_text_dump {
242    // Write text format (loaded at runtime via dump_reader::load_from_str)
243    let write_count = latexml_core::dump_writer::write_dump(Path::new(&dest), &diff)?;
244    // Save versioned TeX Live stamp for staleness detection.
245    if let Some(year) = ambient_year {
246      save_texlive_version(year);
247    }
248    eprintln!("[ini_tex] Wrote {} text entries to {}", write_count, dest);
249    eprintln!("Format dump complete: {} entries written", write_count);
250    Ok(write_count)
251  } else {
252    // Write compiled Rust source (legacy format)
253    let tmp = format!("{}.tmp", dest);
254    let _write_count = latexml_core::dump_writer::write_dump(Path::new(&tmp), &diff)?;
255    let rs_count = latexml_core::dump_codegen::generate_rs(Path::new(&tmp), Path::new(&dest))?;
256    let _ = std::fs::remove_file(&tmp);
257    eprintln!(
258      "[ini_tex] Generated {} Rust definitions to {}",
259      rs_count, dest
260    );
261    eprintln!("Format dump complete: {} entries written", rs_count);
262    Ok(rs_count)
263  }
264}

266/// Generate a compiled Rust module from a dump file.
267/// Reads the text dump and produces a .rs file with direct state assignment calls.
268pub fn codegen_from_dump(dump_path: &str, output_path: &str) -> Result<usize, String> {
269  eprintln!("[ini_tex] Generating Rust module from {}", dump_path);
270  let count =
271    latexml_core::dump_codegen::generate_rs(Path::new(dump_path), Path::new(output_path))?;
272  eprintln!("[ini_tex] Generated {} entries to {}", count, output_path);
273  Ok(count)
274}

276fn save_texlive_version(year: u32) {
277  let version = std::process::Command::new("kpsewhich")
278    .arg("--version")
279    .output()
280    .ok()
281    .and_then(|o| {
282      if o.status.success() {
283        String::from_utf8(o.stdout).ok()
284      } else {
285        None
286      }
287    });
288  if let Some(v) = version {
289    let stamp = latexml_engine::dump_paths::version_filename(year);
290    let _ = std::fs::write(format!("resources/dumps/{}", stamp), v.trim());
291  }
292}

294fn split_path(path: &str) -> (String, String, String) {
295  let p = Path::new(path);
296  let dir = p
297    .parent()
298    .map(|d| d.to_string_lossy().to_string())
299    .unwrap_or_default();
300  let stem = p
301    .file_stem()
302    .map(|s| s.to_string_lossy().to_string())
303    .unwrap_or_default();
304  let ext = p
305    .extension()
306    .map(|e| e.to_string_lossy().to_string())
307    .unwrap_or_default();
308  (dir, stem, ext)
309}
