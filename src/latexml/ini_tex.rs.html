1//! Format dump mode — Rust equivalent of Perl's iniTeX + DumpFile.
2//!
3//! Usage: `latexml_oxide --init=latex.ltx --dest=latex_dump.oxide`
4//!
5//! Follows Perl's make formats scaffold (Makefile.PL + Core.pm::iniTeX):
6//! 1. Initialize the engine (load pools)
7//! 2. Take a snapshot of the state
8//! 3. Process the init file (e.g., latex.ltx) as raw TeX
9//! 4. Compute the diff (what changed)
10//! 5. Write the dump file with changed entries
11//!
12//! The resulting dump can be loaded at runtime to skip re-processing
13//! the LaTeX kernel on every test run.
14

23/// Process an init file and write a format dump.
24/// Perl equivalent: Core.pm::iniTeX → TeX_Job.pool.ltxml::DumpFile
25pub fn dump_format(
26  _converter: &mut Converter,
27  init_file: &str,
28  destination: Option<&str>,
29) -> Result<usize, String> {
30  eprintln!("[ini_tex] Dumping format from {}", init_file);
31
32  // Step 1: Take a snapshot of the state BEFORE processing.
33  // Perl: DumpFile takes snapshot, then loads file, then diffs.
34  let snap = state::take_snapshot();
35  let snap_size = snap.len();
36  eprintln!("[ini_tex] Pre-dump snapshot: {} entries", snap_size);
37
38  // Step 2: Process the init file as raw TeX definitions.
39  // Perl: loadTeXDefinitions($name, $path, type => $type)
40  // This digests the file through the engine, creating definitions.
41  let (_, name, ext) = split_path(init_file);
42  eprintln!("[ini_tex] Loading {} (ext: {})", name, ext);
43
44  // Lift the token limit for format dumps — expl3-code.tex alone uses ~5M tokens.
45  let saved_limit = latexml_core::gullet::set_token_limit(None);
46
47  // In init mode, suppress error/warning output during format loading.
48  // Raw latex.ltx redefines commands already in the compiled engine ("already defined"),
49  // and expl3-code.tex has forward references that produce transient errors.
50  // All these errors are benign — the dump captures the final correct state.
51  let prev_suppress = latexml_core::common::error::set_suppress_log_output(true);
52
53  // Suppress known expl3 loading errors at the state level too
54  state::assign_value("SUPPRESS_UNDEFINED_ERRORS", true, None);
55  state::assign_value("SUPPRESS_UNEXPECTED_ERRORS", true, None);
56
57  // Use the full filename with extension for proper file resolution
58  let load_name = if ext.is_empty() { name.clone() } else { format!("{}.{}", name, ext) };
59  let result = input_definitions(
60    &load_name,
61    InputDefinitionOptions {
62      noltxml: true,
63      ..InputDefinitionOptions::default()
64    },
65  );
66  if let Err(e) = result {
67    eprintln!("[ini_tex] Warning during loading: {}", e);
68  }
69
70  // Restore limits and suppression
71  latexml_core::gullet::restore_token_limit(saved_limit);
72  latexml_core::common::error::set_suppress_log_output(prev_suppress);
73  state::assign_value("SUPPRESS_UNDEFINED_ERRORS", false, None);
74  state::assign_value("SUPPRESS_UNEXPECTED_ERRORS", false, None);
75
76  // Step 3: Compute the diff — only entries that changed.
77  let diff = state::diff_snapshot(&snap);
78  eprintln!(
79    "[ini_tex] Post-load diff: {} changed entries (from {} pre-dump)",
80    diff.len(),
81    snap_size
82  );
83
84  // Step 4: Write the dump as native Rust source.
85  let default_dest = "latexml_package/src/engine/latex_dump.rs";
86  let dest = destination.unwrap_or(default_dest);
87
88  // Write to temp intermediate format, then codegen to .rs
89  let tmp = format!("{}.tmp", dest);
90  let _write_count = latexml_core::dump_writer::write_dump(Path::new(&tmp), &diff)?;
91  let rs_count = latexml_core::dump_codegen::generate_rs(Path::new(&tmp), Path::new(dest))?;
92  let _ = std::fs::remove_file(&tmp);
93
94  eprintln!("[ini_tex] Generated {} Rust definitions to {}", rs_count, dest);
95  eprintln!("Format dump complete: {} entries written", rs_count);
96
97  Ok(rs_count)
98}

100/// Generate a compiled Rust module from a dump file.
101/// Reads the text dump and produces a .rs file with direct state assignment calls.
102pub fn codegen_from_dump(dump_path: &str, output_path: &str) -> Result<usize, String> {
103  eprintln!("[ini_tex] Generating Rust module from {}", dump_path);
104  let count = latexml_core::dump_codegen::generate_rs(
105    Path::new(dump_path),
106    Path::new(output_path),
107  )?;
108  eprintln!("[ini_tex] Generated {} entries to {}", count, output_path);
109  Ok(count)
110}

112fn split_path(path: &str) -> (String, String, String) {
113  let p = Path::new(path);
114  let dir = p
115    .parent()
116    .map(|d| d.to_string_lossy().to_string())
117    .unwrap_or_default();
118  let stem = p
119    .file_stem()
120    .map(|s| s.to_string_lossy().to_string())
121    .unwrap_or_default();
122  let ext = p
123    .extension()
124    .map(|e| e.to_string_lossy().to_string())
125    .unwrap_or_default();
126  (dir, stem, ext)
127}
