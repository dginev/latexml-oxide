13pub(crate) fn get_file_path(uri: &str) -> String {
14  let s = uri.strip_prefix("file://").unwrap_or(uri);
15  // Percent-decode at the BYTE level, then reassemble as UTF-8. Decoding
16  // each %XX to `byte as char` would turn multi-byte sequences (e.g.
17  // `%C3%A9` = é) into Latin-1 mojibake — wrong SOURCEDIRECTORY/search
18  // paths for any non-ASCII document path.
19  let bytes = s.as_bytes();
20  let mut decoded: Vec<u8> = Vec::with_capacity(bytes.len());
21  let mut i = 0;
22  while i < bytes.len() {
23    if bytes[i] == b'%' {
24      if let Some(hex) = bytes.get(i + 1..i + 3)
25        && hex.is_ascii()
26        && let Ok(byte) = u8::from_str_radix(std::str::from_utf8(hex).unwrap_or(""), 16)
27      {
28        decoded.push(byte);
29        i += 3;
30        continue;
31      }
32      decoded.push(b'%');
33      i += 1;
34    } else {
35      decoded.push(bytes[i]);
36      i += 1;
37    }
38  }
39  String::from_utf8_lossy(&decoded).into_owned()
40}

42/// Byte ranges of `%`-comments in `text` (from each unescaped `%` to its line
43/// end). A `%` is escaped only when immediately preceded by a backslash that
44/// is itself a control-sequence escape (`\%`); after `\\` (the line-break
45/// control word) a `%` DOES start a comment. Verbatim environments are not
46/// modeled — a preamble `verbatim` containing `\begin{document}` is out of
47/// scope.
48pub(crate) fn comment_spans(text: &str) -> Vec<(usize, usize)> {
49  let bytes = text.as_bytes();
50  let mut spans = Vec::new();
51  let mut i = 0;
52  while i < bytes.len() {
53    match bytes[i] {
54      b'\\' => i += 2, // skip the escaped char (covers \% and \\ alike)
55      b'%' => {
56        let end = bytes[i..]
57          .iter()
58          .position(|&b| b == b'\n')
59          .map(|p| i + p)
60          .unwrap_or(bytes.len());
61        spans.push((i, end));
62        i = end + 1;
63      },
64      _ => i += 1,
65    }
66  }
67  spans
68}

70/// Find the **effective** `\begin{document}` — the first regex match that is
71/// not inside a `%`-comment. A commented-out `\begin{document}` (common in
72/// templates) must not split the preamble: the old `regex.find` cut the text
73/// at the commented occurrence, leaking the rest of the comment line into the
74/// body as content.
75pub(crate) fn find_begin_document(regex: &regex::Regex, text: &str) -> Option<(usize, usize)> {
76  let spans = comment_spans(text);
77  regex
78    .find_iter(text)
79    .find(|m| !spans.iter().any(|&(s, e)| s < m.start() && m.start() < e))
80    .map(|m| (m.start(), m.end()))
81}

83/// Final path component (e.g. `main.tex`). The client lowercases for matching.
84fn basename(path: &str) -> String {
85  std::path::Path::new(path)
86    .file_name()
87    .and_then(|s| s.to_str())
88    .map(String::from)
89    .unwrap_or_else(|| path.to_string())
90}

92/// Resolve the source-map decoder ring: `sources[tag]` is the basename of the
93/// file the integer `tag` (in each `data-sourcepos`) refers to. The main
94/// buffer is digested as a `literal:` source named "Anonymous String"; map
95/// that to the document's own basename so the client can resolve tag 0 back
96/// to the active file. Other tags are real `\input`-ed files. Must be called
97/// while the post-conversion thread-local state is still live.
98pub(crate) fn collect_sources(uri: &str) -> Vec<String> {
99  let self_base = basename(&get_file_path(uri));
100  latexml_core::state::source_table_snapshot()
101    .iter()
102    .map(|sym| {
103      let name = latexml_core::common::arena::with(*sym, |s| s.to_string());
104      if name == "Anonymous String" {
105        self_base.clone()
106      } else {
107        basename(&name)
108      }
109    })
110    .collect()
111}

113/// Post-process the core ltx XML into HTML5 — the form the editor renders.
114/// This runs the same pipeline the CLI and the ar5iv-editor server use
115/// (`run_post_processing` with the embedded `LaTeXML-html5.xsl`), which turns
116/// presentation MathML on and rewrites the source-map `data:sourcepos`
117/// (foreign-namespaced, colon) attributes into the HTML `data-sourcepos`
118/// (dash) the client decodes. Without this the server returned raw core XML.
119pub(crate) fn post_process_html(core_xml: &str, uri: &str) -> String {
120  let file_path = get_file_path(uri);
121  let source_dir = std::path::Path::new(&file_path)
122    .parent()
123    .and_then(|p| p.to_str())
124    .map(String::from);
125  crate::post::run_post_processing(core_xml, &crate::post::PostOptions {
126    pmml:                      true,
127    cmml:                      false,
128    keep_xmath:                false,
129    stylesheet:                Some("resources/XSLT/LaTeXML-html5.xsl"),
130    destination:               None,
131    source_directory:          source_dir.as_deref(),
132    site_directory:            None,
133    search_paths:              &[],
134    // The server returns HTML as a string with no destination — it must never
135    // write CSS/JS resource files to disk (would pollute the cwd). The client
136    // supplies its own preview styling.
137    nodefaultresources:        true,
138    css_files:                 &[],
139    js_files:                  &[],
140    noinvisibletimes:          false,
141    mathtex:                   false,
142    navigationtoc:             None,
143    schemadocs:                false,
144    split:                     false,
145    split_xpath:               None,
146    split_naming:              None,
147    xslt_parameters:           &[],
148    graphics_svg_threshold_kb: 0,
149    graphicimages:             true,
150    timestamp:                 None,
151    icon:                      None,
152    whatsout:                  latexml_post::extract::Whatsout::Document,
153  })
154}

156pub(crate) fn make_config(uri: &str) -> Config {
157  let file_path = get_file_path(uri);
158  let dir_path = std::path::Path::new(&file_path).parent();
159  let mut search_paths = Vec::new();
160  if let Some(parent) = dir_path
161    && let Some(p_str) = parent.to_str()
162    && !p_str.is_empty()
163  {
164    search_paths.push(p_str.to_string());
165  }
166
167  Config {
168    verbosity:               0,
169    format:                  OutputFormat::HTML5,
170    whatsin:                 DataSize::Document,
171    whatsout:                DataSize::Document,
172    preamble:                None,
173    postamble:               None,
174    mode:                    None,
175    bindings_dispatch:       Some(Rc::new(latexml_package::dispatch)),
176    extra_bindings_dispatch: Some(Rc::new(latexml_contrib::dispatch)),
177    // Preload ar5iv.sty: this server backs the ar5iv-editor, and ar5iv.sty
178    // enables raw `.sty` handling so a paper's *local, binding-less* packages
179    // (e.g. a bundled `mystyle.sty`) load instead of being skipped with a
180    // missing-file warning. Mirrors the sandbox/ar5iv conversion workflow.
181    preload:                 Some(vec!["ar5iv.sty".to_string()]),
182    search_paths:            if search_paths.is_empty() {
183      None
184    } else {
185      Some(search_paths)
186    },
187    include_comments:        None,
188    strict:                  None,
189    include_styles:          None,
190    // Math parsing is always ON in the server: the parsed MathML (and its
191    // source-mapped tokens) is one of the features this preview showcases, so
192    // we deliberately do not expose a disable knob here.
193    nomathparse:             None,
194    source_map:              Some(true),
195    // Server input is editor buffer text (already UTF-8); no override needed.
196    inputencoding:           None,
197  }
198}

200/// The *set* of TeX-relevant files in the root's directory. Compared by
201/// key set only (not mtimes) in the warm-cache check: it exists to catch
202/// files APPEARING or DISAPPEARING, which can change `find_file`
203/// resolution (a new local `foo.sty` shadowing the texmf one) — something
204/// the warm-up read-log of *successful opens* cannot see. Content changes
205/// of files the warm-up actually opened are caught precisely by the
206/// read-log's `Disk(mtime)`/`Overlay(version)` pins
207/// (`overlay::warmup_dep_snapshot`); content changes of files it did NOT
208/// open cannot affect a replay of the same preamble. Comparing mtimes
209/// here (as this scan originally did) forced a full preamble re-warm on
210/// every save of a same-directory *body* file — pure waste.
211pub(crate) fn get_directory_dependencies(uri: &str) -> BTreeSet<String> {
212  let mut deps = BTreeSet::new();
213  let file_path = get_file_path(uri);
214  if let Some(parent) = std::path::Path::new(&file_path).parent()
215    && let Ok(entries) = std::fs::read_dir(parent)
216  {
217    for entry in entries.flatten() {
218      let path = entry.path();
219      if !entry.file_type().map(|t| t.is_file()).unwrap_or(false) {
220        continue;
221      }
222      let is_dep = path
223        .extension()
224        .and_then(|e| e.to_str())
225        .map(|e| {
226          matches!(
227            e.to_lowercase().as_str(),
228            "sty" | "cls" | "tex" | "cfg" | "def" | "bib" | "clo"
229          )
230        })
231        .unwrap_or(false);
232      if !is_dep {
233        continue;
234      }
235      if let Some(path_str) = path.to_str() {
236        deps.insert(path_str.to_string());
237      }
238    }
239  }
240  deps
241}

253  #[test]
254  fn file_path_percent_decodes_utf8() {
255    // %C3%A9 = é (two UTF-8 bytes) — the old char-per-byte decode produced
256    // "Ã©" mojibake and broke SOURCEDIRECTORY for non-ASCII paths.
257    assert_eq!(
258      get_file_path("file:///home/u/caf%C3%A9/main.tex"),
259      "/home/u/café/main.tex"
260    );
261    assert_eq!(get_file_path("file:///plain/path.tex"), "/plain/path.tex");
262    // Malformed escapes pass through unmangled.
263    assert_eq!(get_file_path("file:///x%2/y%"), "/x%2/y%");
264    assert_eq!(get_file_path("file:///sp%20ace.tex"), "/sp ace.tex");
265  }

267  #[test]
268  fn begin_document_skips_commented_occurrences() {
269    let re = regex::Regex::new(r"\\begin\s*\{\s*document\s*\}").unwrap();
270    // Commented-out \begin{document} (template style) must not split there.
271    let text = "\\documentclass{article}\n% \\begin{document} not yet!\n\\begin{document}\nBody\n\\end{document}\n";
272    let (start, end) = find_begin_document(&re, text).unwrap();
273    assert_eq!(&text[start..end], "\\begin{document}");
274    assert!(
275      text[..start].contains("not yet!"),
276      "split is AFTER the commented line"
277    );
278    // An escaped \% does not start a comment.
279    let text2 = "100\\% sure\\begin{document}x";
280    let (s2, _) = find_begin_document(&re, text2).unwrap();
281    assert_eq!(s2, "100\\% sure".len());
282    // All occurrences commented → no split (in-process fallback path).
283    assert_eq!(find_begin_document(&re, "% \\begin{document}\n"), None);
284  }
