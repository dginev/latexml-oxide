12// ======================================================================
13// Server — warm preamble cache + in-process fallback conversion.
14// ======================================================================
15
16pub(crate) struct Server {
17  pub(crate) begin_doc_regex:          regex::Regex,
18  /// Per-conversion wall-clock budget in seconds (`--timeout`; 0 disables).
19  pub(crate) timeout_secs:             u64,
20  /// Per-conversion resident-memory ceiling in KiB (`--max-memory`; 0 disables).
21  pub(crate) max_rss_kb:               u64,
22  pub(crate) warmed_uri:               Option<String>,
23  pub(crate) warmed_preamble:          Option<String>,
24  pub(crate) warmed_preamble_digested: Option<latexml_core::digested::Digested>,
25  /// Log captured while digesting the preamble, so preamble diagnostics
26  /// survive across body-only fork conversions (the warmup only re-runs on a
27  /// cache miss).
28  pub(crate) warmed_preamble_log:      String,
29  /// File SET of the root's directory at warm time (appearance /
30  /// disappearance check — see `get_directory_dependencies`). Content
31  /// staleness of opened files is `warmed_source_deps`' job.
32  pub(crate) warmed_dependencies:      BTreeSet<String>,
33  /// Warm-up read-log snapshot: every source the preamble digest opened,
34  /// pinned as Overlay(version)/Disk(mtime) (`overlay::warmup_dep_snapshot`).
35  pub(crate) warmed_source_deps:       Vec<(String, DepState)>,
36  /// Open editor buffers (didOpen/didChange/convert text), keyed by
37  /// absolute path — the unsaved-overlay source of truth.
38  pub(crate) open_buffers:             FxHashMap<String, Buffer>,
39  /// Engine `_contents` keys applied by the last `overlay::apply`, so
40  /// closed buffers get cleared on the next apply.
41  pub(crate) overlay_keys:             Vec<String>,
42  /// Per-directory root-detection cache.
43  pub(crate) root_cache:               RootCache,
44  /// Client-configured root (`initializationOptions.rootDocument`).
45  pub(crate) root_override:            Option<PathBuf>,
46  /// Per root-uri: the file uris diagnostics were last published to, so a
47  /// newly-clean file gets an explicit empty publish (stale squiggles clear).
48  pub(crate) last_published:           FxHashMap<String, Vec<String>>,
49}
50
51impl Server {
52  pub(crate) fn new(timeout_secs: u64, max_rss_kb: u64) -> Self {
53    Server {
54      begin_doc_regex: regex::Regex::new(r"\\begin\s*\{\s*document\s*\}").unwrap(),
55      timeout_secs,
56      max_rss_kb,
57      warmed_uri: None,
58      warmed_preamble: None,
59      warmed_preamble_digested: None,
60      warmed_preamble_log: String::new(),
61      warmed_dependencies: BTreeSet::new(),
62      warmed_source_deps: Vec::new(),
63      open_buffers: FxHashMap::default(),
64      overlay_keys: Vec::new(),
65      root_cache: RootCache::default(),
66      root_override: None,
67      last_published: FxHashMap::default(),
68    }
69  }
70
71  /// Record buffer state from didOpen/didChange/convert. `version` is the
72  /// LSP-provided document version when present; otherwise the previous
73  /// version + 1 (so the dep snapshot still observes the change).
74  pub(crate) fn upsert_buffer(&mut self, path: String, text: String, version: Option<i64>) {
75    let next = version.unwrap_or_else(|| {
76      self
77        .open_buffers
78        .get(&path)
79        .map(|b| b.version + 1)
80        .unwrap_or(1)
81    });
82    self
83      .open_buffers
84      .insert(path, Buffer { version: next, text });
85  }
86
87  /// Re-apply the overlay for the project rooted at `root` onto the engine's
88  /// `_contents` channel. Call after any thread-state reset and before
89  /// digestion / the body fork.
90  pub(crate) fn apply_overlay(&mut self, root: &std::path::Path) {
91    let dir = project_dir(root);
92    self.overlay_keys = overlay_apply(&self.open_buffers, &dir, &self.overlay_keys);
93  }
94
95  /// Drop the warm preamble cache. MUST be called whenever the thread-local
96  /// engine state is reset by an in-process conversion — otherwise a later
97  /// cache-hit fork would inherit a state that no longer matches the cached
98  /// preamble boxes (the bug that made `didChange` corrupt the next
99  /// `latexml/convert`).
100  pub(crate) fn invalidate_cache(&mut self) {
101    self.warmed_uri = None;
102    self.warmed_preamble = None;
103    self.warmed_preamble_digested = None;
104    self.warmed_preamble_log.clear();
105    self.warmed_dependencies.clear();
106    self.warmed_source_deps.clear();
107  }
108
109  /// Full, in-process conversion of `text`. Resets and rebuilds engine state,
110  /// so it invalidates the warm cache. Used as the fallback whenever the
111  /// warm-fork path is unavailable (no `\begin{document}`, fork failure,
112  /// non-Unix). Line numbers are naturally file-relative (the whole document
113  /// is one source), and the source-map table is read back after conversion.
114  pub(crate) fn convert_in_process(&mut self, uri: &str, text: &str) -> ConvertOutput {
115    self.invalidate_cache();
116    latexml_core::state::reset_thread_state();
117    // Overlay must be re-applied after the reset wiped the engine state.
118    let root_path = PathBuf::from(get_file_path(uri));
119    self.overlay_keys.clear();
120    self.apply_overlay(&root_path);
121    // Cooperative wall-clock guard for the in-process path. (There is no child
122    // to reap here — this path runs on the server's own thread — so the hard
123    // RAM/time backstops don't apply; the cooperative deadline + the engine's
124    // RSS fuse are what bound a runaway fallback conversion.)
125    latexml_core::stomach::set_timeout(self.timeout_secs);
126
127    let opts = make_config(uri);
128    let mut converter = Converter::from_config(opts.clone());
129    if let Err(e) = converter.prepare_session(&opts) {
130      return ConvertOutput::error(format!("Fatal: prepare_session failed: {e}"));
131    }
132    converter.bind_log();
133    // Fallback path (no `\begin{document}`, fork failure, or non-Unix). Use a
134    // *named* in-memory source (the document path) so `--source-map` stamps
135    // locators here too, matching the warm-fork path.
136    let resp = converter.convert_content_with_provenance(&get_file_path(uri), text.to_string());
137    let sources = collect_sources(uri);
138    let status_code = resp.status_code as i64;
139    let status = if resp.status.is_empty() {
140      status_label(status_code).to_string()
141    } else {
142      resp.status
143    };
144    let log = resp.log;
145    let diags = parse_log_diagnostics(&log);
146    let html = post_process_html(&resp.result.unwrap_or_default(), uri);
147    ConvertOutput {
148      html,
149      log,
150      diags,
151      sources,
152      root: None, // attributed by the trigger wrappers
153      status,
154      status_code,
155    }
156  }
157}

159/// Publish diagnostics grouped per attributed file uri (multi-file model);
160/// unattributed ones go to the edited buffer's uri. Files diagnosed last
161/// round but clean now get an explicit empty publish so stale squiggles
162/// clear. Shared by the unix and generic server flavors.
163pub(crate) fn publish_grouped_diagnostics(
164  server: &mut Server,
165  root_uri: &str,
166  edited_uri: &str,
167  out: &ConvertOutput,
168  writer: &mut impl std::io::Write,
169) -> std::io::Result<()> {
170  let mut by_uri: FxHashMap<String, Vec<Diag>> = FxHashMap::default();
171  for d in &out.diags {
172    let uri = match d.file.as_deref() {
173      Some(path) => format!("file://{path}"),
174      None => edited_uri.to_string(),
175    };
176    by_uri.entry(uri).or_default().push(d.clone());
177  }
178  let published: Vec<String> = by_uri.keys().cloned().collect();
179  for (uri, diags) in &by_uri {
180    send_message(writer, &publish_diagnostics_notification(uri, diags))?;
181  }
182  // Clear files that had diagnostics last round but none now.
183  if let Some(previous) = server
184    .last_published
185    .insert(root_uri.to_string(), published.clone())
186  {
187    for stale in previous {
188      if !published.contains(&stale) {
189        send_message(writer, &publish_diagnostics_notification(&stale, &[]))?;
190      }
191    }
192  }
193  Ok(())
194}
