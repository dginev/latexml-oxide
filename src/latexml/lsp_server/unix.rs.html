18/// Outcome of one warm-fork conversion attempt.
19enum WarmResult {
20  /// Conversion finished (possibly via in-process fallback).
21  Done(ConvertOutput),
22  /// A newer same-document request superseded this one; the preempting
23  /// message has already been pushed onto the pending queue.
24  Cancelled,
25}
26
27/// Buffered, `poll`-aware reader over a raw fd (stdin = 0). We read with
28/// `libc::read` directly rather than `std::io::Stdin` so that `poll(2)`
29/// readiness and our own user-space buffer stay consistent.
30struct FdReader {
31  fd:  i32,
32  buf: Vec<u8>,
33}
34
35impl FdReader {
36  fn new() -> Self { Self::from_fd(0) }
37
38  /// Reader over an arbitrary fd — the framing logic is fd-agnostic; tests
39  /// drive it over a `pipe(2)` instead of stdin.
40  fn from_fd(fd: i32) -> Self { FdReader { fd, buf: Vec::new() } }
41
42  /// One blocking `read` into the buffer; `Ok(0)` is EOF.
43  fn fill(&mut self) -> std::io::Result<usize> {
44    let mut tmp = [0u8; 8192];
45    loop {
46      let n = unsafe {
47        libc::read(self.fd, tmp.as_mut_ptr() as *mut libc::c_void, tmp.len())
48      };
49      if n < 0 {
50        let err = std::io::Error::last_os_error();
51        if err.raw_os_error() == Some(libc::EINTR) {
52          continue;
53        }
54        return Err(err);
55      }
56      if n == 0 {
57        return Ok(0);
58      }
59      self.buf.extend_from_slice(&tmp[..n as usize]);
60      return Ok(n as usize);
61    }
62  }
63
64  /// Pull one complete frame out of the buffer, if present. A loop (not
65  /// recursion) so a flood of malformed headers can't grow the stack.
66  fn take_frame(&mut self) -> Option<String> {
67    loop {
68      let he = find_subseq(&self.buf, b"\r\n\r\n")?;
69      let cl = parse_content_length(&self.buf[..he]);
70      let body_start = he + 4;
71      match cl {
72        Some(cl) if self.buf.len() >= body_start + cl => {
73          let body: Vec<u8> = self.buf[body_start..body_start + cl].to_vec();
74          self.buf.drain(..body_start + cl);
75          return Some(String::from_utf8_lossy(&body).into_owned());
76        },
77        // Malformed header with no parseable Content-Length: drop it and retry.
78        None => {
79          self.buf.drain(..body_start);
80        },
81        // Header present but body not fully arrived yet.
82        Some(_) => return None,
83      }
84    }
85  }
86
87  /// Blocking read of the next complete message; `None` on EOF. Skips empty
88  /// bodies.
89  fn next_message(&mut self) -> Option<String> {
90    loop {
91      if let Some(frame) = self.take_frame() {
92        if frame.is_empty() {
93          continue;
94        }
95        return Some(frame);
96      }
97      match self.fill() {
98        Ok(0) => return None,
99        Ok(_) => continue,
100        Err(_) => return None,
101      }
102    }
103  }
104}

106fn find_subseq(haystack: &[u8], needle: &[u8]) -> Option<usize> {
107  haystack.windows(needle.len()).position(|w| w == needle)
108}

110/// Number of live threads in this process (Linux: `/proc/self/task`
111/// entries; elsewhere `0` = unknown, which disables the advisory check).
112fn task_count() -> usize {
113  std::fs::read_dir("/proc/self/task")
114    .map(|entries| entries.count())
115    .unwrap_or(0)
116}

118/// Thread count observed when the server started. The `latexml_oxide`
119/// binary runs everything on a dedicated big-stack worker thread while
120/// `main` parks in `join()` — so the baseline is normally 2, and that
121/// parked `main` is fork-benign (a futex wait holds no allocator/logger
122/// lock). What MUST hold is that serving never *grows* the thread count:
123/// any thread spawned during a conversion (a future engine/post
124/// `thread::spawn` that outlives its scope) could hold the malloc lock at
125/// fork time and deadlock the child before its Watchdog arms.
126static BASELINE_THREADS: std::sync::atomic::AtomicUsize =
127  std::sync::atomic::AtomicUsize::new(0);
128
129fn parse_content_length(header: &[u8]) -> Option<usize> {
130  let s = std::str::from_utf8(header).ok()?;
131  for line in s.split("\r\n") {
132    let l = line.trim();
133    if l.to_ascii_lowercase().starts_with("content-length:") {
134      if let Some(v) = l.split(':').nth(1) {
135        if let Ok(n) = v.trim().parse::<usize>() {
136          return Some(n);
137        }
138      }
139    }
140  }
141  None
142}

144/// Reap `pid`, returning its exit code; a signal death maps to the shell
145/// convention `128 + signo` (so an OS OOM-killer SIGKILL reports as `137`,
146/// same as the Watchdog's deliberate memory-ceiling exit — both mean "ran
147/// out of memory" to `finish`). The shared `Watchdog` in the child exits
148/// `124` on timeout / `137` on the memory ceiling.
149fn reap(pid: i32) -> i32 {
150  let mut status = 0i32;
151  unsafe {
152    libc::waitpid(pid, &mut status, 0);
153    if libc::WIFEXITED(status) {
154      libc::WEXITSTATUS(status)
155    } else if libc::WIFSIGNALED(status) {
156      128 + libc::WTERMSIG(status)
157    } else {
158      -1
159    }
160  }
161}

163/// Does this raw message supersede the in-flight conversion of the project
164/// rooted at `current_root`? A newer `latexml/convert` OR
165/// `didChange`/`didOpen` of ANY file in the same project preempts: every
166/// trigger of a project converts the same root, so the in-flight result is
167/// already stale and finishing it would only delay the newer request behind a
168/// wasted compile.
169fn preempts(body: &str, current_root: &std::path::Path) -> bool {
170  match parse_json(body) {
171    Ok(req) => super::message_doc_uri(&req)
172      .map(|uri| {
173        let path = get_file_path(&uri);
174        same_project(current_root, std::path::Path::new(&path))
175      })
176      .unwrap_or(false),
177    Err(_) => false,
178  }
179}

181fn is_exit(body: &str) -> bool {
182  parse_json(body)
183    .ok()
184    .and_then(|r| r.get("method").and_then(|m| m.as_str()).map(str::to_string))
185    .as_deref()
186    == Some("exit")
187}

189/// Block until the body child finishes or is preempted, multiplexing
190/// `{stdin, child-pipe}` on a single thread.
191///
192/// The child self-guards its wall-clock and RAM budgets via the shared
193/// [`latexml_core::watchdog::Watchdog`] (see `run_body_child`), exiting with a
194/// distinct code on breach. So the parent doesn't poll resources — it waits
195/// for the pipe to close and lets `finish` interpret the exit code. The only
196/// active concern here is stdin: a newer same-document `latexml/convert`
197/// preempts (SIGKILL) the in-flight child.
198fn wait_for_child(
199  pid: i32,
200  read_fd: i32,
201  current_root: &std::path::Path,
202  reader: &mut FdReader,
203  pending: &mut VecDeque<String>,
204) -> WarmResult {
205  // Owns `read_fd`; closes it on every return path.
206  let mut pipe = unsafe { std::fs::File::from_raw_fd(read_fd) };
207
208  loop {
209    // 1. Handle every COMPLETE stdin frame already buffered. We never block
210    //    waiting for the rest of a partial frame here — doing so (the old
211    //    `next_message()` call) ignored the child pipe while blocked, and a
212    //    client that stalls mid-frame while the child blocks writing a
213    //    > pipe-capacity payload would deadlock all three parties.
214    while let Some(body) = reader.take_frame() {
215      if body.is_empty() {
216        continue;
217      }
218      if is_exit(&body) {
219        kill_and_reap(pid);
220        std::process::exit(0);
221      }
222      if preempts(&body, current_root) {
223        kill_and_reap(pid);
224        pending.push_back(body);
225        return WarmResult::Cancelled;
226      }
227      // Unrelated message (didClose, shutdown, a different document):
228      // queue it for after this compile and keep waiting.
229      pending.push_back(body);
230    }
231
232    // 2. Poll both fds.
233    let mut fds = [
234      libc::pollfd { fd: 0, events: libc::POLLIN, revents: 0 },
235      libc::pollfd { fd: read_fd, events: libc::POLLIN, revents: 0 },
236    ];
237    let rc = unsafe { libc::poll(fds.as_mut_ptr(), fds.len() as libc::nfds_t, -1) };
238    if rc < 0 {
239      let err = std::io::Error::last_os_error();
240      if err.raw_os_error() == Some(libc::EINTR) {
241        continue;
242      }
243      // poll failed: fall back to a blocking drain of the child.
244      let mut bytes = Vec::new();
245      let _ = pipe.read_to_end(&mut bytes);
246      let code = reap(pid);
247      return finish(&bytes, code);
248    }
249    let pipe_ready = (fds[1].revents & (libc::POLLIN | libc::POLLHUP)) != 0;
250    if pipe_ready {
251      // The child writes its whole payload in one shot at the very end, so
252      // pipe-readable means the compile is essentially done — drain & reap.
253      let mut bytes = Vec::new();
254      let _ = pipe.read_to_end(&mut bytes);
255      let code = reap(pid);
256      return finish(&bytes, code);
257    }
258    if (fds[0].revents & (libc::POLLIN | libc::POLLHUP)) != 0 {
259      // 3. One non-blocking-equivalent fill; complete frames are handled at
260      //    the top of the next iteration.
261      match reader.fill() {
262        // stdin EOF mid-compile: client gone — kill the child and stop.
263        Ok(0) => {
264          kill_and_reap(pid);
265          return WarmResult::Cancelled;
266        },
267        Ok(_) => {},
268        Err(_) => {
269          kill_and_reap(pid);
270          return WarmResult::Cancelled;
271        },
272      }
273    }
274  }
275}

277/// Terminate the in-flight body child: SIGTERM first (default disposition
278/// kills it, but gives any future cleanup handler a chance), a short grace
279/// for it to disappear, then SIGKILL, then reap. The child's graphics
280/// subprocesses are protected separately via `PR_SET_PDEATHSIG` (see
281/// `graphics::run_with_timeout`): when the child dies, the kernel reaps the
282/// `setsid`-detached converter grandchildren that would otherwise be
283/// orphaned with no watcher.
284fn kill_and_reap(pid: i32) {
285  unsafe {
286    libc::kill(pid, libc::SIGTERM);
287  }
288  // Brief grace: poll for exit up to ~50 ms before escalating.
289  for _ in 0..10 {
290    let mut status = 0i32;
291    let r = unsafe { libc::waitpid(pid, &mut status, libc::WNOHANG) };
292    if r == pid {
293      return; // already reaped
294    }
295    std::thread::sleep(std::time::Duration::from_millis(5));
296  }
297  unsafe {
298    libc::kill(pid, libc::SIGKILL);
299  }
300  reap(pid);
301}

303/// Parse the child's pipe payload into a `ConvertOutput`. The parent owns the
304/// preamble (warmup) log and the source-map context, so it merges them here.
305/// `exit_code` is the child's exit status: a child hard-terminated by its
306/// `Watchdog` writes no payload and exits [`watchdog::EXIT_TIMEOUT`] /
307/// [`watchdog::EXIT_OOM`], which we map to a fatal result.
308fn finish(bytes: &[u8], exit_code: i32) -> WarmResult {
309  use latexml_core::watchdog::{EXIT_OOM, EXIT_TIMEOUT};
310  // Threaded back through a thread-local set just before the fork; see
311  // `run_warm`. We stash the preamble log in a cell to avoid widening this
312  // function's signature through the poll machinery.
313  let body_str = String::from_utf8_lossy(bytes).into_owned();
314  match parse_json(&body_str) {
315    Ok(payload) => {
316      if let Some(err) = payload.get("error").and_then(|e| e.as_str()) {
317        WarmResult::Done(ConvertOutput::error(format!("child error: {err}")))
318      } else {
319        let html = payload
320          .get("html")
321          .and_then(|h| h.as_str())
322          .unwrap_or("")
323          .to_string();
324        let body_log = payload
325          .get("log")
326          .and_then(|l| l.as_str())
327          .unwrap_or("")
328          .to_string();
329        let sources = match payload.get("sources") {
330          Some(Value::Array(arr)) => arr
331            .iter()
332            .filter_map(|v| v.as_str().map(String::from))
333            .collect(),
334          _ => Vec::new(),
335        };
336        let combined_log = format!("{}{}", PREAMBLE_LOG.with(|c| c.borrow().clone()), body_log);
337        let diags = parse_log_diagnostics(&combined_log);
338        // Engine status/code reported by the child (0/1/2/3).
339        let status_code = payload.get("statusCode").and_then(|c| c.as_i64()).unwrap_or(0);
340        let status = payload
341          .get("status")
342          .and_then(|s| s.as_str())
343          .map(String::from)
344          .unwrap_or_else(|| status_label(status_code).to_string());
345        WarmResult::Done(ConvertOutput {
346          html,
347          log: combined_log,
348          diags,
349          sources,
350          root: None, // attributed by convert_trigger
351          status,
352          status_code,
353        })
354      }
355    },
356    // No usable payload — the child was hard-terminated (its Watchdog exited
357    // it for a resource breach the cooperative guards didn't catch) or
358    // crashed. Map the exit code to a fatal result.
359    Err(e) => {
360      let out = match exit_code {
361        EXIT_TIMEOUT => {
362          ConvertOutput::failed("timeout", 3, "conversion timed out (hard wall-clock limit)".to_string())
363        },
364        EXIT_OOM => ConvertOutput::failed(
365          "fatal",
366          3,
367          "conversion exceeded the memory ceiling".to_string(),
368        ),
369        0 => ConvertOutput::error(format!("child payload parse error: {e}")),
370        c => ConvertOutput::error(format!("child exited unexpectedly (code {c})")),
371      };
372      WarmResult::Done(out)
373    },
374  }
375}

383/// Fork a child that digests `body`, prepends the warm preamble's digested
384/// boxes, builds the DOM and writes `{html, log, sources}` (or `{error}`)
385/// back over a pipe. Returns `(pid, read_fd)` to the parent.
386fn spawn_body_child(
387  uri: &str,
388  offset_lines: usize,
389  body: &str,
390  warmed: &latexml_core::digested::Digested,
391  timeout_secs: u64,
392  max_rss_kb: u64,
393) -> Result<(i32, i32), String> {
394  let mut fds = [0i32; 2];
395  unsafe {
396    if libc::pipe(fds.as_mut_ptr()) < 0 {
397      return Err("pipe() failed".to_string());
398    }
399  }
400  let (read_fd, write_fd) = (fds[0], fds[1]);
401
402  // INVARIANT: no thread beyond the server-start baseline may be live at
403  // fork time — an extra thread could hold the allocator (or logger) lock
404  // mid-operation, and the forked child would deadlock on its first
405  // allocation, BEFORE its Watchdog spawns, wedging the server on the pipe
406  // forever. Today this holds: graphics workers are `thread::scope`-joined,
407  // `run_with_timeout` is poll-based, and `Watchdog`s are created
408  // child-side only. (The baseline itself is fork-benign: the binary's
409  // `main` parks in `join()` on the worker thread that runs this server —
410  // see BASELINE_THREADS.) The debug assertion catches any future
411  // `thread::spawn` that silently breaks the invariant.
412  #[cfg(debug_assertions)]
413  {
414    let baseline = BASELINE_THREADS.load(std::sync::atomic::Ordering::Relaxed);
415    let now = task_count();
416    debug_assert!(
417      baseline == 0 || now == 0 || now <= baseline,
418      "forking the body child with {now} live threads (server started with {baseline}) — fork-unsafe"
419    );
420  }
421
422  let pid = unsafe { libc::fork() };
423  if pid < 0 {
424    unsafe {
425      libc::close(read_fd);
426      libc::close(write_fd);
427    }
428    return Err("fork() failed".to_string());
429  }
430
431  if pid == 0 {
432    // ---- child ----
433    unsafe {
434      libc::close(read_fd);
435      // Insurance: the child inherits the parent's stdout (fd 1) — the LSP
436      // protocol stream. No engine path writes to stdout today, but a
437      // single stray `println!` would corrupt the framing for every client.
438      // Point the child's fd 1 at /dev/null; its payload goes over the
439      // pipe and its logging over stderr.
440      let devnull = libc::open(c"/dev/null".as_ptr(), libc::O_WRONLY);
441      if devnull >= 0 {
442        libc::dup2(devnull, 1);
443        libc::close(devnull);
444      }
445    }
446    let payload = run_body_child(uri, offset_lines, body, warmed, timeout_secs, max_rss_kb);
447    let bytes = payload.to_string().into_bytes();
448    let mut file = unsafe { std::fs::File::from_raw_fd(write_fd) };
449    let _ = file.write_all(&bytes);
450    let _ = file.flush();
451    drop(file); // close write end → parent sees EOF
452    std::process::exit(0);
453  }
454
455  // ---- parent ----
456  unsafe {
457    libc::close(write_fd);
458  }
459  Ok((pid, read_fd))
460}

462/// Child-side body compilation. State (definitions, mode, fonts, source-map
463/// table) is inherited from the warm parent via copy-on-write, so we work
464/// against a bare `Core` and the thread-local logger — **never**
465/// `Converter::from_config`, whose `Core::new` calls `set_state` and would
466/// wipe the inherited state (undefined `\par`, `lookup_font()` → None).
467/// `offset_lines` blank lines are prepended to the body literal so the child
468/// mouth's line counter is file-relative (fixes diagnostics/`data-sourcepos`
469/// being off by the preamble length).
470fn run_body_child(
471  uri: &str,
472  offset_lines: usize,
473  body: &str,
474  warmed: &latexml_core::digested::Digested,
475  timeout_secs: u64,
476  max_rss_kb: u64,
477) -> Value {
478  // Hard guard for this child, identical to cortex_worker's: a background
479  // thread that exits the child (124 timeout / 137 OOM) if it blows through
480  // its wall-clock or RAM budget — including native hangs (libxslt, Marpa)
481  // the cooperative deadline can't see. The child self-terminates; the parent
482  // reaps it (`finish` maps the exit code). Held for the whole conversion.
483  let _watchdog = latexml_core::watchdog::Watchdog::with_limits(timeout_secs, max_rss_kb);
484  let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
485    // Re-arm the cooperative deadline FRESH from this child's clock. The child
486    // inherits the parent's thread-local CONVERSION_DEADLINE via COW (set
487    // during warm-up); without this the body would run against a stale (often
488    // already-expired) deadline. Digest loops raise Fatal:Timeout (caught
489    // below) for the graceful common case; the Watchdog above is the backstop.
490    latexml_core::stomach::set_timeout(timeout_secs);
491    // Bare Core over the inherited state — does NOT reset thread-local state.
492    let mut core = latexml_core::Core {
493      preload: make_config(uri).preload.unwrap_or_default(),
494    };
495    latexml_core::util::logger::bind_log();
496
497    // Open the body as a *named* in-memory mouth (same path as the preamble)
498    // so its locators are stampable user sources sharing tag 0. The
499    // `offset_lines` blank lines make the body mouth's line counter
500    // file-relative.
501    let body_content = format!("{}{}", "\n".repeat(offset_lines), body);
502    crate::converter::open_named_in_memory_mouth(&get_file_path(uri), body_content)
503      .map_err(|e| format!("input error: {e}"))?;
504
505    let body_digested = core
506      .digest_internal()
507      .map_err(|e| format!("digest error: {e}"))?;
508
509    let mut combined = warmed.unlist();
510    combined.extend(body_digested.unlist());
511    let full = latexml_core::digested::Digested::from(latexml_core::list::List {
512      boxes: combined,
513      ..Default::default()
514    });
515
516    let dom = core
517      .convert_document(full)
518      .map_err(|e| format!("dom conversion error: {e}"))?;
519    let core_xml = dom.serialize_to_string();
520    let sources = collect_sources(uri);
521    // Post-process to HTML5 in the child (the heavy XSLT stays inside the
522    // cancellable/throwaway process).
523    let html = post_process_html(&core_xml, uri);
524    // Engine status (0 ok / 1 warning / 2 error / 3 fatal) — cumulative over
525    // the inherited preamble report plus this body.
526    let status = latexml_core::common::error::get_status_message();
527    let status_code = latexml_core::common::error::get_status_code() as i64;
528    let log = latexml_core::util::logger::flush_log();
529    Ok::<(String, String, Vec<String>, String, i64), String>((
530      html,
531      log,
532      sources,
533      status,
534      status_code,
535    ))
536  }));
537
538  match result {
539    Ok(Ok((html, log, sources, status, status_code))) => jobj(vec![
540      ("html", jstr(html)),
541      ("log", jstr(log)),
542      (
543        "sources",
544        Value::Array(sources.into_iter().map(jstr).collect()),
545      ),
546      ("status", jstr(status)),
547      // Integer (not jnum's float) so the parent's `as_i64()` round-trips.
548      ("statusCode", Value::from(status_code)),
549    ]),
550    Ok(Err(msg)) => jobj(vec![("error", jstr(msg))]),
551    Err(_) => jobj(vec![("error", jstr("child panicked"))]),
552  }
553}

555impl Server {
556  /// Convert `text` via the warm-preamble + fork-body pipeline, falling back
557  /// to in-process conversion when that path is unavailable.
558  fn run_warm(
559    &mut self,
560    uri: &str,
561    text: &str,
562    reader: &mut FdReader,
563    pending: &mut VecDeque<String>,
564  ) -> WarmResult {
565    let Some((_, mat_end)) = find_begin_document(&self.begin_doc_regex, text) else {
566      // No (un-commented) document body boundary — convert the whole thing
567      // in-process.
568      return WarmResult::Done(self.convert_in_process(uri, text));
569    };
570    let preamble = &text[..mat_end];
571    let body = &text[mat_end..];
572    let offset_lines = preamble.matches('\n').count();
573    let deps = get_directory_dependencies(uri);
574    let root_path = std::path::PathBuf::from(get_file_path(uri));
575    let timeout = self.timeout_secs;
576    // Cooperative deadline for the (parent-side) warm-up digest. The forked
577    // body child re-arms its own fresh deadline; see `run_body_child`.
578    latexml_core::stomach::set_timeout(timeout);
579
580    let cache_hit = self.warmed_uri.as_deref() == Some(uri)
581      && self.warmed_preamble.as_deref() == Some(preamble)
582      && self.warmed_preamble_digested.is_some()
583      // Same-dir file SET unchanged (no file appeared/disappeared that
584      // could change find_file resolution — mtimes deliberately not
585      // compared; see `get_directory_dependencies`).
586      && self.warmed_dependencies == deps
587      // Read-log snapshot: every source the warm-up opened must still be at
588      // its pinned Overlay(version)/Disk(mtime) state (multi-file model).
589      && deps_still_current(&self.warmed_source_deps, &self.open_buffers);
590
591    if !cache_hit {
592      log::info!("Warming preamble cache for {uri}");
593      self.invalidate_cache();
594      latexml_core::state::reset_thread_state();
595
596      let opts = make_config(uri);
597      let mut converter = Converter::from_config(opts.clone());
598      if converter.prepare_session(&opts).is_err() {
599        return WarmResult::Done(self.convert_in_process(uri, text));
600      }
601      // The reset wiped the engine state: re-apply the unsaved-buffer overlay
602      // BEFORE the preamble digest, so preamble-time \input / \usepackage of
603      // open buffers see editor state.
604      self.overlay_keys.clear();
605      self.apply_overlay(&root_path);
606      // Name the preamble source after the document path so its locators are
607      // stampable user sources (and share tag 0 with the body).
608      match converter.digest_content_with_provenance(&get_file_path(uri), preamble.to_string()) {
609        Ok(pre) => {
610          self.warmed_preamble_log = converter.flush_log();
611          self.warmed_uri = Some(uri.to_string());
612          self.warmed_preamble = Some(preamble.to_string());
613          self.warmed_preamble_digested = Some(pre);
614          self.warmed_dependencies = deps;
615          // Pin the warm-up read-log (which sources, at which state).
616          self.warmed_source_deps = warmup_dep_snapshot(&self.open_buffers, &root_path);
617        },
618        Err(e) => {
619          log::error!("Preamble warmup failed ({e}); falling back to in-process");
620          return WarmResult::Done(self.convert_in_process(uri, text));
621        },
622      }
623    } else {
624      // Cache hit: refresh the overlay so the BODY fork inherits current
625      // buffer state (body-time \input of an edited chapter). The preamble
626      // deps were just verified unchanged, so the warm state stays valid.
627      self.apply_overlay(&root_path);
628    }
629
630    let warmed = self.warmed_preamble_digested.as_ref().unwrap();
631    // The child self-guards its time + RAM budget via the shared Watchdog.
632    let (pid, read_fd) =
633      match spawn_body_child(uri, offset_lines, body, warmed, timeout, self.max_rss_kb) {
634        Ok(v) => v,
635        Err(e) => {
636          log::error!("{e}; falling back to in-process");
637          return WarmResult::Done(self.convert_in_process(uri, text));
638        },
639      };
640
641    // Hand the preamble log to `finish` (which runs deep inside the poll
642    // loop) without threading it through every signature.
643    PREAMBLE_LOG.with(|c| *c.borrow_mut() = self.warmed_preamble_log.clone());
644
645    match wait_for_child(pid, read_fd, &root_path, reader, pending) {
646      WarmResult::Cancelled => WarmResult::Cancelled,
647      // A child that reported an internal error is reported as-is rather than
648      // silently re-run in-process: re-running would reset the warm cache and
649      // mask the failure. The fallback is reserved for *spawn/transport*
650      // failures (handled above), not engine errors inside the child.
651      done => done,
652    }
653  }
654}

656pub fn run(timeout_secs: u64, max_rss_kb: u64) -> Result<(), Box<dyn std::error::Error>> {
657  // Snapshot the thread-count baseline for the fork-safety debug assertion.
658  BASELINE_THREADS.store(task_count(), std::sync::atomic::Ordering::Relaxed);
659  let mut stdout = std::io::stdout();
660  let mut server = Server::new(timeout_secs, max_rss_kb);
661  let mut reader = FdReader::new();
662  let mut pending: VecDeque<String> = VecDeque::new();
663
664  loop {
665    let body = match pending.pop_front() {
666      Some(b) => b,
667      None => match reader.next_message() {
668        Some(b) => b,
669        None => break, // stdin EOF
670      },
671    };
672    if !dispatch(&mut server, &mut reader, &mut pending, &body, &mut stdout)? {
673      break;
674    }
675  }
676  Ok(())
677}

679/// Handle one message. Returns `false` to stop the server (on `exit`).
680fn dispatch(
681  server: &mut Server,
682  reader: &mut FdReader,
683  pending: &mut VecDeque<String>,
684  body_str: &str,
685  stdout: &mut std::io::Stdout,
686) -> Result<bool, Box<dyn std::error::Error>> {
687  let request = match parse_json(body_str) {
688    Ok(v) => v,
689    Err(e) => {
690      log::error!("Failed to parse incoming JSON: {e}");
691      return Ok(true);
692    },
693  };
694  let id = request.get("id").cloned().unwrap_or(Value::Null);
695  let method = request.get("method").and_then(|m| m.as_str()).unwrap_or("");
696  log::debug!("LSP request: method='{method}', id={id:?}");
697
698  match method {
699    "initialize" => {
700      // Multi-file model: a client-configured project root wins over all
701      // detection (docs/archive/LSP_MULTIFILE_PLAN.md §3A).
702      if let Some(root) = request
703        .get("params")
704        .and_then(|p| p.get("initializationOptions"))
705        .and_then(|o| o.get("rootDocument"))
706        .and_then(|r| r.as_str())
707      {
708        server.root_override = Some(std::path::PathBuf::from(root));
709      }
710      let caps = jobj(vec![(
711        "capabilities",
712        jobj(vec![("textDocumentSync", jnum(1.0))]),
713      )]);
714      send_message(stdout, &response(id, caps))?;
715    },
716    "initialized" => {},
717    "textDocument/didOpen" => {
718      if let Some((uri, text)) = did_open_params(&request) {
719        convert_trigger(server, reader, pending, &uri, &text, doc_version(&request), stdout, None)?;
720      }
721    },
722    "textDocument/didChange" => {
723      if let Some((uri, text)) = did_change_params(&request) {
724        convert_trigger(server, reader, pending, &uri, &text, doc_version(&request), stdout, None)?;
725      }
726    },
727    "textDocument/didClose" => {
728      if let Some(uri) = request
729        .get("params")
730        .and_then(|p| p.get("textDocument"))
731        .and_then(|d| d.get("uri"))
732        .and_then(|u| u.as_str())
733      {
734        // Forget the buffer (the overlay reverts to disk state on the next
735        // apply) and clear its diagnostics.
736        server.open_buffers.remove(&get_file_path(uri));
737        send_message(stdout, &publish_diagnostics_notification(uri, &[]))?;
738      }
739    },
740    "shutdown" => {
741      send_message(stdout, &response(id, Value::Null))?;
742    },
743    "latexml/convert" => {
744      if let (Some(uri), Some(text)) = (
745        request
746          .get("params")
747          .and_then(|p| p.get("uri"))
748          .and_then(|u| u.as_str()),
749        request
750          .get("params")
751          .and_then(|p| p.get("text"))
752          .and_then(|t| t.as_str()),
753      ) {
754        let (uri, text) = (uri.to_string(), text.to_string());
755        convert_trigger(server, reader, pending, &uri, &text, None, stdout, Some(id))?;
756      } else if id != Value::Null {
757        // A request (has an id) MUST be answered — silently dropping it
758        // leaves the client awaiting the response forever.
759        send_message(
760          stdout,
761          &error_response(id, -32602.0, "latexml/convert: missing params.uri/params.text".to_string()),
762        )?;
763      }
764    },
765    "exit" => return Ok(false),
766    other => {
767      if id != Value::Null {
768        send_message(
769          stdout,
770          &error_response(id, -32601.0, format!("Method '{other}' not found")),
771        )?;
772      } else {
773        log::warn!("Unhandled LSP notification: {other}");
774      }
775    },
776  }
777  Ok(true)
778}

780/// LSP `textDocument.version` of a didOpen/didChange payload.
781fn doc_version(request: &Value) -> Option<i64> {
782  request
783    .get("params")
784    .and_then(|p| p.get("textDocument"))
785    .and_then(|d| d.get("version"))
786    .and_then(|v| v.as_i64())
787}
788
789/// One conversion trigger (didOpen / didChange / `latexml/convert`), through
790/// the multi-file pipeline: record the buffer, resolve the project root,
791/// coalesce against newer same-project triggers, convert the ROOT via the
792/// shared warm-fork path, then answer (`request_id`-carrying convert gets the
793/// result object; notifications publish diagnostics).
794#[allow(clippy::too_many_arguments)]
795fn convert_trigger(
796  server: &mut Server,
797  reader: &mut FdReader,
798  pending: &mut VecDeque<String>,
799  uri: &str,
800  text: &str,
801  version: Option<i64>,
802  stdout: &mut std::io::Stdout,
803  request_id: Option<Value>,
804) -> Result<(), Box<dyn std::error::Error>> {
805  let buffer_path = get_file_path(uri);
806  server.upsert_buffer(buffer_path.clone(), text.to_string(), version);
807  let override_root = server.root_override.clone();
808  let root = resolve_root(
809    &mut server.root_cache,
810    override_root.as_deref(),
811    std::path::Path::new(&buffer_path),
812    Some(text),
813  );
814
815  // Coalesce: a newer trigger for this PROJECT already queued makes this
816  // compile stale before it starts.
817  if superseded_in_pending(pending, &root) {
818    if let Some(id) = request_id {
819      send_message(stdout, &response(id, cancelled_result_object()))?;
820    }
821    return Ok(());
822  }
823
824  // The root's text: the edited buffer itself, another open buffer, or disk.
825  let root_str = root.to_string_lossy().into_owned();
826  let root_text = if root_str == buffer_path {
827    text.to_string()
828  } else if let Some(buf) = server.open_buffers.get(&root_str) {
829    buf.text.clone()
830  } else {
831    match std::fs::read_to_string(&root) {
832      Ok(t) => t,
833      Err(e) => {
834        // Degrade to v1 behavior: convert the buffer standalone.
835        log::warn!("cannot read project root {root_str} ({e}); converting the buffer standalone");
836        text.to_string()
837      },
838    }
839  };
840  let root_uri = format!("file://{root_str}");
841
842  match server.run_warm(&root_uri, &root_text, reader, pending) {
843    WarmResult::Done(mut out) => {
844      // Attribute diagnostics to project files (multi-file model).
845      attribute_diag_files(&mut out.diags, &root, &server.open_buffers);
846      out.root = Some(root_str);
847      match request_id {
848        Some(id) => send_message(stdout, &response(id, out.to_result_object()))?,
849        None => publish_grouped_diagnostics(server, &root_uri, uri, &out, stdout)?,
850      }
851    },
852    WarmResult::Cancelled => {
853      if let Some(id) = request_id {
854        send_message(stdout, &response(id, cancelled_result_object()))?;
855      }
856      // A superseded notification publishes nothing — the newer run will.
857    },
858  }
859  Ok(())
860}

864#[cfg(test)]
865mod framing_tests {
866  use super::*;
867
868  /// A `pipe(2)` pair: write fragments into `w`, read frames via FdReader on
869  /// the read end. Closed on drop.
870  struct PipePair {
871    r: i32,
872    w: i32,
873  }
874  impl PipePair {
875    fn new() -> Self {
876      let mut fds = [0i32; 2];
877      assert_eq!(unsafe { libc::pipe(fds.as_mut_ptr()) }, 0);
878      PipePair { r: fds[0], w: fds[1] }
879    }
880    fn write(&self, bytes: &[u8]) {
881      let n = unsafe { libc::write(self.w, bytes.as_ptr() as *const libc::c_void, bytes.len()) };
882      assert_eq!(n, bytes.len() as isize);
883    }
884  }
885  impl Drop for PipePair {
886    fn drop(&mut self) {
887      unsafe {
888        libc::close(self.r);
889        libc::close(self.w);
890      }
891    }
892  }
893
894  fn frame(body: &str) -> Vec<u8> {
895    format!("Content-Length: {}\r\n\r\n{}", body.len(), body).into_bytes()
896  }
897
898  #[test]
899  fn take_frame_handles_fragmented_input() {
900    let pipe = PipePair::new();
901    let mut reader = FdReader::from_fd(pipe.r);
902    let msg = frame(r#"{"method":"initialized"}"#);
903    // Header fragment only: no frame yet.
904    pipe.write(&msg[..10]);
905    reader.fill().unwrap();
906    assert_eq!(reader.take_frame(), None);
907    // Rest of header + partial body: still no frame.
908    let header_len = msg.len() - 24;
909    pipe.write(&msg[10..header_len + 5]);
910    reader.fill().unwrap();
911    assert_eq!(reader.take_frame(), None);
912    // Remainder: one complete frame.
913    pipe.write(&msg[header_len + 5..]);
914    reader.fill().unwrap();
915    assert_eq!(
916      reader.take_frame().as_deref(),
917      Some(r#"{"method":"initialized"}"#)
918    );
919    assert_eq!(reader.take_frame(), None);
920  }
921
922  #[test]
923  fn take_frame_two_frames_one_fill() {
924    let pipe = PipePair::new();
925    let mut reader = FdReader::from_fd(pipe.r);
926    let mut bytes = frame("{\"a\":1}");
927    bytes.extend(frame("{\"b\":2}"));
928    pipe.write(&bytes);
929    reader.fill().unwrap();
930    assert_eq!(reader.take_frame().as_deref(), Some("{\"a\":1}"));
931    assert_eq!(reader.take_frame().as_deref(), Some("{\"b\":2}"));
932    assert_eq!(reader.take_frame(), None);
933  }

936  fn take_frame_skips_malformed_header() {
937    let pipe = PipePair::new();
938    let mut reader = FdReader::from_fd(pipe.r);
939    let mut bytes = b"Garbage-Header: x\r\n\r\n".to_vec();
940    bytes.extend(frame("{\"ok\":true}"));
941    pipe.write(&bytes);
942    reader.fill().unwrap();
943    // The malformed (no Content-Length) header is dropped; the good frame
944    // behind it is returned — in a loop, not recursion.
945    assert_eq!(reader.take_frame().as_deref(), Some("{\"ok\":true}"));
946  }
