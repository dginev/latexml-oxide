1//! Reusable engine presets that don't depend on the test harness.
2//!
3//! Extracted from `util/test.rs` (audit DEP-02, 2026-05-18) so the
4//! `latexmlmath_oxide` standalone binary can build without the test
5//! harness's `glob`/`phf` dependencies. The two helpers here have no
6//! ties to the harness machinery — they just construct a minimal
7//! `Core` and tokenize a single inline formula.
8
9use std::rc::Rc;
10
11use latexml_core::{Core, CoreOptions, document::Document, state};
12use latexml_math_parser::node_to_grammar_lexemes;
13use libxml::tree::Node;
14
15use crate::core_interface::DigestionAPI;
16
17/// Provide a default `Core` engine preloaded with `article.cls`
18/// and `amsmath.sty` — the minimum needed to digest most formulae.
19pub fn new_test_engine() -> Core {
20  let core_engine = Core::new(CoreOptions {
21    preload: Some(
22      ["article.cls", "amsmath.sty"]
23        .map(|x| x.to_string())
24        .to_vec(),
25    ),
26    verbosity: Some(-2),
27    search_paths: None,
28    nomathparse: Some(true),
29    include_comments: Some(false),
30    ..CoreOptions::default()
31  });
32  // Shared model loader — see crate::load_latexml_default_model.
33  crate::load_latexml_default_model();
34  state::set_bindings_dispatch(Rc::new(latexml_package::dispatch));
35  state::add_binding_names(latexml_package::binding_names());
36  core_engine
37}

39/// Simple tokenization of a single formula, without any custom preloads
40/// beyond latex and amsmath.
41pub fn lex_single_tex_formula(
42  tex: &str,
43  latexml: &mut Core,
44) -> (Vec<String>, Vec<Node>, Option<Node>, Document) {
45  let xml_result = latexml.convert_file(format!("literal:\\[ {tex} \\]"));
46  assert!(xml_result.is_ok(), "{:?}", xml_result.err());
47  let mut doc = xml_result.unwrap();
48
49  match doc.findnode("//*[local-name()='XMath']", None) {
50    Some(math) => {
51      let mut idx = 0;
52      let (lexemes, nodes) = node_to_grammar_lexemes(&math, &mut idx);
53      (lexemes, nodes, Some(math), doc)
54    },
55    None => (Vec::new(), Vec::new(), None, doc),
56  }
57}
