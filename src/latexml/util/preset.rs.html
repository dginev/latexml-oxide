1//! Reusable engine presets that don't depend on the test harness.
2//!
3//! Extracted from `util/test.rs` (audit DEP-02, 2026-05-18) so the
4//! `latexmlmath_oxide` standalone binary can build without the test
5//! harness's `glob`/`phf` dependencies. The two helpers here have no
6//! ties to the harness machinery — they just construct a minimal
7//! `Core` and tokenize a single inline formula.

17/// Provide a default `Core` engine preloaded with `article.cls`
18/// and `amsmath.sty` — the minimum needed to digest most formulae.
19pub fn new_test_engine() -> Core {
20  let core_engine = Core::new(CoreOptions {
21    // `LaTeX.pool` FIRST — a class preload must not be the thing that drags the
22    // pool in. `\@pushfilename` changes meaning when the pool (and the kernel dump
23    // behind it) loads: preloading `article.cls` alone pushes the class's filename
24    // with the pre-pool `\@pushfilename` (which never touches
25    // `\g__hook_name_stack_seq`), then pops it with the real expl3 `\@popfilename`
26    // that the pool installed — popping a seq that only ever saw the *inner*
27    // packages' pushes. Hence "LaTeX hooks Error: Extra \PopDefaultHookLabel" on
28    // every latexmlmath_oxide run. Same order ar5iv's preload list uses.
29    // See SYNC_STATUS "`--preload=<cls>` hook-stack imbalance" — the underlying
30    // ordering bug is still open; this preset just stops provoking it.
31    preload: Some(
32      ["LaTeX.pool", "article.cls", "amsmath.sty"]
33        .map(|x| x.to_string())
34        .to_vec(),
35    ),
36    verbosity: Some(-2),
37    search_paths: None,
38    nomathparse: Some(true),
39    include_comments: Some(false),
40    ..CoreOptions::default()
41  });
42  // Shared model loader — see crate::load_latexml_default_model.
43  crate::load_latexml_default_model();
44  state::set_bindings_dispatch(Rc::new(latexml_package::dispatch));
45  state::add_binding_names(latexml_package::binding_names());
46  core_engine
47}

49/// Simple tokenization of a single formula, without any custom preloads
50/// beyond latex and amsmath.
51pub fn lex_single_tex_formula(
52  tex: &str,
53  latexml: &mut Core,
54) -> (Vec<String>, Vec<Node>, Option<Node>, Document) {
55  let xml_result = latexml.convert_file(format!("literal:\\[ {tex} \\]"));
56  assert!(xml_result.is_ok(), "{:?}", xml_result.err());
57  let mut doc = xml_result.unwrap();
58
59  match doc.findnode("//*[local-name()='XMath']", None) {
60    Some(math) => {
61      let mut idx = 0;
62      let (lexemes, nodes) = node_to_grammar_lexemes(&math, &mut idx);
63      (lexemes, nodes, Some(math), doc)
64    },
65    None => (Vec::new(), Vec::new(), None, doc),
66  }
67}
