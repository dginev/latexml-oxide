22pub fn latexml_tests_internal(
23  dirpath: &str,
24  requires: Option<&phf::Map<&str, &str>>,
25  dispatcher_opt: Option<BindingDispatcher>,
26) {
27  if !validate_requirements(dirpath, requires) {
28    return; // test group only if required files are found.
29  }
30  for tex_file in glob(&s!("{}/*.tex", dirpath)).unwrap().flatten() {
31    let name = tex_file.file_stem().unwrap().to_str().unwrap();
32    let xml_file = tex_file.with_extension("xml");
33
34    let tex_file_string = tex_file.to_str().unwrap();
35    let xml_file_str = xml_file.to_str().unwrap();
36    if xml_file.exists() {
37      latexml_ok_internal(tex_file_string, xml_file_str, name, dispatcher_opt.clone());
38    } else {
39      // Skip, these could be tex fragment files.
40    }
41  }
42}

58static INIT_TEST_RSS_CAP: Once = Once::new();
59/// Raise the per-process RSS fuse for the multi-conversion test harness.
60///
61/// `latexml_core::stomach`'s memory budget defaults to **4.5 GB**, sized to
62/// bound a *single* conversion — that low default is load-bearing in production,
63/// where a massively parallel fleet runs many one-paper processes at once and
64/// the aggregate host RSS is `N × cap` (raising it would OOM the machine).
65///
66/// `cargo test` is the one place that runs many conversions in ONE process:
67/// libtest spawns a thread per test, so at high parallelism (e.g. `-j128` on a
68/// many-core box) the process-wide RSS is the *sum* of all in-flight
69/// conversions and trips the single-conversion fuse on otherwise-fine
70/// documents (a false `MemoryBudget` cascade on article/book/report …). So the
71/// harness — not the production default — raises the cap, once, here. An
72/// explicit `LATEXML_RSS_CAP_BYTES` (or `--test-threads=N`) still wins.
73fn init_test_rss_cap() {
74  INIT_TEST_RSS_CAP.call_once(|| {
75    if std::env::var_os("LATEXML_RSS_CAP_BYTES").is_none() {
76      // SAFETY: set exactly once under `Once`, at the very top of every
77      // generated test (before any conversion thread reads the env in
78      // `stomach::check_timeout`). `Once`'s release/acquire ordering
79      // happens-before all those reads, so there is no setenv/getenv race.
80      unsafe {
81        std::env::set_var("LATEXML_RSS_CAP_BYTES", "9000000000");
82      }
83    }
84  });
85}

87// Linker-section trick: register a SIGSEGV handler via `.init_array` so
88// it runs BEFORE `main()` (and therefore before any thread the test
89// harness spawns can crash). `init_logger` was too late — by the time
90// the first test thread called it, sibling threads had already crashed.
91//
92// Gated by `LATEXML_SIGSEGV_TRACE` so the handler is opt-in; signal()
93// + std::backtrace inside a SIGSEGV handler is technically not async-
94// signal-safe (uses heap), but for diagnostic purposes a best-effort
95// stack dump is far more useful than the bare `signal: 11` line cargo
96// reports today.
97#[used]
98#[cfg_attr(target_os = "linux", unsafe(link_section = ".init_array"))]
99static SIGSEGV_INSTALLER: extern "C" fn() = sigsegv_installer;
100
101extern "C" fn sigsegv_installer() {
102  // Read env at process start. SAFETY: nothing has run yet, env is
103  // populated by the kernel/loader.
104  if *SIGSEGV_TRACE {
105    eprintln!("[sigsegv_installer] installing SIGSEGV handler");
106    install_sigsegv_handler();
107  }
108}

110/// Install a SIGSEGV handler that prints the crashing thread's name and a
111/// best-effort backtrace before the kernel terminates the process. This
112/// is a diagnostic-only hook — the handler is not async-signal-safe (it
113/// uses `std::backtrace` and `eprintln!`, which malloc), but for
114/// post-mortem of the libxml2-suspected multi-thread crash that's
115/// observed under `cargo test --release --tests`, even an unsafe
116/// stack print is more useful than the bare `signal: 11` line cargo
117/// reports today.
118fn install_sigsegv_handler() {
119  unsafe extern "C" {
120    fn signal(sig: i32, handler: extern "C" fn(i32)) -> usize;
121    fn raise(sig: i32) -> i32;
122  }
123  // Linux SIGSEGV = 11, SIGBUS = 7, SIGABRT = 6.
124  const SIGSEGV: i32 = 11;
125  const SIGBUS: i32 = 7;
126  const SIGABRT: i32 = 6;
127  const SIG_DFL: usize = 0;
128
129  extern "C" fn handler(sig: i32) {
130    // Capture context synchronously and persist to a per-pid file —
131    // cargo test buffers/discards stderr from binaries that exit by
132    // signal, so eprintln!() never reaches the user. Writing to
133    // `/tmp/latexml_sigsegv_<pid>.txt` survives the kill.
134    let tid = std::thread::current().id();
135    let name = std::thread::current()
136      .name()
137      .unwrap_or("<unnamed>")
138      .to_string();
139    let pid = std::process::id();
140    let path = format!("/tmp/latexml_sigsegv_{pid}.txt");
141    let bt = std::backtrace::Backtrace::force_capture();
142    let exe = std::env::current_exe()
143      .map(|p| p.display().to_string())
144      .unwrap_or_else(|_| "<unknown>".into());
145    let body = format!(
146      "=== SIGSEGV-handler ===\nsignal={sig}\nthread={name:?}\nid={tid:?}\nexe={exe}\npid={pid}\n\n{bt}\n"
147    );
148    let _ = std::fs::write(&path, &body);
149    // Also try eprintln (best effort; usually lost by cargo on signal).
150    eprintln!("{body}");
151    eprintln!("[SIGSEGV-handler] full trace written to {path}");
152    // Reset to default and re-raise so cargo still sees the original signal.
153    unsafe {
154      let raw_dfl: extern "C" fn(i32) = std::mem::transmute(SIG_DFL);
155      signal(sig, raw_dfl);
156      raise(sig);
157    }
158  }
159
160  unsafe {
161    signal(SIGSEGV, handler);
162    signal(SIGBUS, handler);
163    signal(SIGABRT, handler);
164  }
165}

167/// **Intentionally-failing tests** — a *permanent contract* that this input
168/// SHOULD produce errors. The TeX is genuinely ill-formed / pathological, so
169/// erroring is the CORRECT, desired outcome forever — NOT something to "fix".
170/// (The discriminator is the input's validity, NOT whether Perl also errors:
171/// see `ERROR_DEBT` for valid inputs that merely error today.)
172///
173/// The contract is a SOFT, RECOVERABLE error: the harness asserts the **exact**
174/// `Error:` count AND that there was **no `Fatal:`** — the whole point is that
175/// the engine recovers and completes the conversion (graceful degradation),
176/// never crashes. Drift fails BOTH ways: *more*/fatal = a handling regression;
177/// *zero* = we silently STOPPED detecting the bad input. Logged
178/// `[intentional-fail]`.
179const INTENTIONALLY_FAILING: &[(&str, usize, &str)] = &[
180  (
181    "protect_self_ref",
182    1,
183    "intrinsic self-recursion (\\def\\cs{\\protect\\cs} typeset): pdflatex HANGS, \
184     Perl+Rust both emit 1 SOFT error — the recursion guard prevents the hang. \
185     Verified 2026-06-10 (latexml --verbose=1 error, pdflatex=timeout).",
186  ),
187  (
188    "io",
189    2,
190    "deliberate malformed read content: `exists.data` line 21 has an unbalanced \
191     brace (`line { with extra } }`) to verify the engine emits a SOFT, \
192     recoverable error (not Fatal) and completes. pdflatex also errors+recovers \
193     (`! Too many }'s, silently discards }`); Perl+Rust both emit 2 soft errors. \
194     Verified 2026-06-10.",
195  ),
196];
197
198/// **Error debt** — valid input we INTEND to convert cleanly (a desired,
199/// surpass-Perl success), but which errors today. TEMPORARY: each MUST be
200/// driven to zero by improving the Rust core, then removed. The harness
201/// tolerates ANY count (logged `[error-debt]`) and does NOT fail at zero,
202/// because the count is **environment-dependent** for some entries (e.g.
203/// `glossary` errors on one host's datatool/expl3 but converts clean in CI) —
204/// failing at zero would break whichever environment is already clean. When an
205/// entry's `[error-debt] … 0 errors` shows up EVERYWHERE, remove it by review.
206/// Each note records Perl's current behavior (verify with `latexml --verbose`
207/// — `--quiet` HIDES Perl errors). Tracked in `docs/SYNC_STATUS.md`.
208const ERROR_DEBT: &[(&str, &str)] = &[
209  (
210    "figure_mixed_content",
211    "complex figures should convert clean; today: ltx:theorem not allowed in \
212     ltx:figure (Perl also: 1). True fix = schema expansion (theorems in \
213     figures) — tracked as a separate issue.",
214  ),
215  (
216    "glossary",
217    "~50 undefined datatool/expl3 macros (\\__datatool_*, \\DTLinitials, …); \
218     Rust-only (Perl: 0 errors). Root cause: l3regex gap (datatool word/initials \
219     parsing uses \\regex_new:N + capture groups). Large — own effort.",
220  ),

223/// Emit a line to the process's REAL stderr, SURVIVING libtest's per-test
224/// output capture. libtest only intercepts the `print!`/`eprint!` macros and
225/// replays them solely on FAILURE, so a plain `eprintln!` from a PASSING test
226/// is swallowed — which defeats the `[error-debt] … review for removal` and
227/// `[intentional-fail]` notices, whose entire purpose is to be SEEN on a green
228/// run (review m2). A direct `write(2)` bypasses the capture. One syscall per
229/// line is atomic up to PIPE_BUF, so concurrent test threads don't interleave.
230#[cfg(unix)]
231fn note_uncaptured(line: &str) {
232  use std::{io::Write, os::unix::io::FromRawFd};
233  // SAFETY: fd 2 is the process stderr, valid for the whole run. `ManuallyDrop`
234  // stops the `File`'s Drop from `close()`-ing the shared descriptor.
235  let mut f = std::mem::ManuallyDrop::new(unsafe { std::fs::File::from_raw_fd(2) });
236  let _ = f.write_all(format!("{line}\n").as_bytes());
237}
238#[cfg(not(unix))]
239fn note_uncaptured(line: &str) {
240  eprintln!("{line}");
241}

243pub fn latexml_test_single(
244  tex_file_str: &str,
245  name: &str,
246  dirpath: &str,
247  requires: Option<&phf::Map<&str, &str>>,
248  dispatcher_opt: Option<BindingDispatcher>,
249) {
250  init_logger();
251  init_test_rss_cap();
252  if !validate_requirements(dirpath, requires) {
253    return; // test group only if required files are found.
254  }
255  // Suppress log output for any test expected to emit errors (both categories)
256  // so single-test runs stay readable; the gate still counts + classifies them.
257  let suppress = INTENTIONALLY_FAILING.iter().any(|(n, ..)| *n == name)
258    || ERROR_DEBT.iter().any(|(n, _)| *n == name);
259  if suppress {
260    latexml_core::common::error::set_suppress_log_output(true);
261  }
262  let tex_file = PathBuf::from(tex_file_str);
263  let xml_file = tex_file.with_extension("xml");
264  if matches!(xml_file.try_exists(), Ok(true)) {
265    latexml_ok_internal(
266      tex_file_str,
267      &xml_file.to_string_lossy(),
268      name,
269      dispatcher_opt,
270    );
271  } else {
272    // Skip, these could be tex fragment files.
273  }
274  if suppress {
275    latexml_core::common::error::set_suppress_log_output(false);
276  }
277}

287fn latexml_ok_internal(
288  tex_path: &str,
289  xml_path: &str,
290  name: &str,
291  extra_bindings_dispatcher: Option<BindingDispatcher>,
292) {
293  let tex_strings = process_texfile(tex_path, name, extra_bindings_dispatcher);
294  if !tex_strings.is_empty() {
295    let xml_strings = process_xmlfile(xml_path, name);
296    if !xml_strings.is_empty() {
297      let mut found_diff = false;
298      for (lineno, (tex_line, xml_line)) in tex_strings.iter().zip(xml_strings.iter()).enumerate() {
299        if tex_line != xml_line {
300          found_diff = true;
301          eprintln!(
302            "DIFF line {lineno} in {xml_path}:\n  ACTUAL:   {tex_line}\n  EXPECTED: {xml_line}"
303          );
304        }
305      }
306      if tex_strings.len() != xml_strings.len() {
307        found_diff = true;
308        eprintln!(
309          "DIFF length mismatch for {name:?}: actual {} lines, expected {} lines",
310          tex_strings.len(),
311          xml_strings.len()
312        );
313        // Print extra lines
314        let min_len = tex_strings.len().min(xml_strings.len());
315        if tex_strings.len() > min_len {
316          for (i, line) in tex_strings[min_len..].iter().enumerate() {
317            eprintln!("  ACTUAL extra line {}: {line}", min_len + i);
318          }
319        }
320        if xml_strings.len() > min_len {
321          for (i, line) in xml_strings[min_len..].iter().enumerate() {
322            eprintln!("  EXPECTED extra line {}: {line}", min_len + i);
323          }
324        }
325      }
326      if found_diff {
327        panic!("Differences found in {xml_path} — see DIFF lines above");
328      }
329    }
330  }
331}

333/// Returns the list-of-strings form of whatever was requested, if successful,
334/// otherwise empty; and they will have reported the failure
335fn process_texfile(
336  tex_path: &str,
337  name: &str,
338  extra_bindings_dispatcher: Option<BindingDispatcher>,
339) -> Vec<String> {
340  let mut latexml = Core::new(CoreOptions {
341    verbosity: Some(-2),
342    search_paths: None,
343    preload: None,
344    include_comments: Some(false),
345    ..CoreOptions::default()
346  });
347  // Install the SAME binding-resolution priority chain a real conversion uses
348  // (rhai > extra/contrib > package), via the shared helper. This is what lets a
349  // test resolve a local `<pkg>.<ext>.rhai` fixture sitting next to its `.tex`
350  // (found through the source-dir search path) — exactly as the Perl suite
351  // resolves a local `<pkg>.<ext>.ltxml`. The `extra` dispatcher (passed by test
352  // groups that still rely on compiled `latexml_contrib` fixtures) is folded in
353  // as tier 2.
354  crate::converter::install_binding_dispatch(extra_bindings_dispatcher);
355  let r = match latexml.convert_file(tex_path.to_owned()) {
356    Err(e) => panic!("{:?}: Couldn't convert {:?}; {:?}", name, tex_path, e),
357    Ok(doc) => process_ltx_doc(doc, name),
358  };
359  // Drop the engine, then free this thread's accumulated thread-local
360  // state. libtest spawns a fresh thread per test, and the engine's
361  // roots are `#[thread_local]` *attribute* statics, which do NOT run
362  // destructors on thread exit — so without this each test would leak
363  // its ~110 MB engine, accumulating to ~4.9 GB across the suite. The
364  // output is already owned `String`s by now, so no live `SymStr`
365  // survives the reset. See `latexml_core::reset_thread_engine`.
366  // Error gate: every `.tex`/`.xml` regression test is an error-regression
367  // sentinel, not just an XML-shape check. `note_status` counts `Error:`/
368  // `Fatal:` even when log output is off, so this is the canonical signal.
369  // Three contracts:
370  //   • normal test            → MUST be error-clean (n_err == 0).
371  //   • INTENTIONALLY_FAILING  → MUST emit its exact SOFT count, NEVER fatal
372  //                              (graceful recovery is the contract; permanent).
373  //   • ERROR_DEBT             → tolerated (count is env-dependent for some);
374  //                              logged, never fails; manual review for removal.
375  //
376  // Perf (runs on every test): the hot path is two thread-local integer reads
377  // via `get_status` (no allocation, no log-string scan) plus tiny slice
378  // lookups; messages build only on the cold panic path. `convert_file` reset
379  // the report at conversion start, so this count is exactly this conversion's.
380  // Read BEFORE `reset_thread_engine`.
381  use latexml_core::common::error::{LogStatus, get_status};
382  let n_soft = get_status(LogStatus::Error);
383  let n_fatal = get_status(LogStatus::Fatal);
384  let n_err = n_soft + n_fatal;
385  let intentional = INTENTIONALLY_FAILING
386    .iter()
387    .find(|(n, ..)| *n == name)
388    .copied();
389  let debt = ERROR_DEBT.iter().find(|(n, _)| *n == name).copied();
390  // Decide the verdict (and any cold-path message) before tearing down.
391  let verdict: Result<(), String> = match (intentional, debt) {
392    // Permanent contract: exact SOFT-error count, and NEVER fatal — the point is
393    // graceful recovery. Drift fails both ways; a Fatal is always a regression.
394    (Some((_, expect, reason)), _) => {
395      note_uncaptured(&format!(
396        "[intentional-fail] {name}: {n_soft} soft errors, {n_fatal} fatal (expect {expect}, 0) — {reason}"
397      ));
398      if n_fatal > 0 {
399        Err(format!(
400          "{name}: INTENTIONALLY_FAILING must degrade to a SOFT error, but got a Fatal \
401           ({}) — graceful recovery regressed. Reason: {reason}",
402          latexml_core::common::error::get_status_message()
403        ))
404      } else if n_soft == expect {
405        Ok(())
406      } else if n_soft == 0 {
407        Err(format!(
408          "{name}: INTENTIONALLY_FAILING expects {expect} error(s) but got 0 — error \
409           detection regressed (this input must still error). Reason: {reason}"
410        ))
411      } else {
412        Err(format!(
413          "{name}: INTENTIONALLY_FAILING expects exactly {expect} soft error(s), got {n_soft} \
414           ({}) — handling drifted. Reason: {reason}",
415          latexml_core::common::error::get_status_message()
416        ))
417      }
418    },
419    // Temporary debt: tolerate whatever it does today (the count is
420    // environment-dependent for some entries — e.g. `glossary` errors on one
421    // box but converts clean in CI, per the host's datatool/expl3 version), so
422    // the gate does NOT fail at zero. Removal is a manual review step when an
423    // entry is clean EVERYWHERE (the `[error-debt] … 0 errors` log flags it).
424    (None, Some((_, reason))) => {
425      if n_err == 0 {
426        note_uncaptured(&format!(
427          "[error-debt] {name}: 0 errors HERE — clean in this \
428          environment; review for removal once clean everywhere — {reason}"
429        ));
430      } else {
431        note_uncaptured(&format!("[error-debt] {name}: {n_err} errors — {reason}"));
432      }
433      Ok(())
434    },
435    // Normal test: must be error-clean.
436    (None, None) => {
437      if n_err == 0 {
438        Ok(())
439      } else {
440        Err(format!(
441          "{name}: conversion logged errors ({}) — a normal-TeX test must be error-clean. \
442           Fix the engine/binding/specimen. If the input SHOULD error (verify with \
443           bin/latexml --verbose), add to INTENTIONALLY_FAILING; if it should convert \
444           clean but doesn't yet, add to ERROR_DEBT with a SYNC_STATUS entry. See \
445           docs/reproducers/MALFORMED_CLOSE_NUMBERED_2026-06-10.md.",
446          latexml_core::common::error::get_status_message()
447        ))
448      }
449    },
450  };
451  drop(latexml);
452  latexml_core::reset_thread_engine();
453  if let Err(msg) = verdict {
454    panic!("{msg}");
455  }
456  r
457}

459/// Loads the reference XML file as raw text lines, avoiding libxml2
460/// re-serialization which would normalize `<p></p>` to `<p/>`.
461fn process_xmlfile<'a>(xml_path: &'a str, _name: &'a str) -> Vec<String> {
462  match std::fs::read_to_string(xml_path) {
463    Err(e) => panic!("Failed to read XML file {:?}: {:?}", xml_path, e),
464    Ok(contents) => {
465      let mut lines: Vec<String> = contents.split('\n').map(ToString::to_string).collect();
466      // Remove trailing empty line from final newline
467      if lines.last().is_some_and(|l| l.is_empty()) {
468        lines.pop();
469      }
470      lines
471    },
472  }
473}
474fn process_ltx_doc(doc: Document, name: &str) -> Vec<String> {
475  let doc_str = doc.serialize_to_string();
476  if *SAVE_ACTUAL {
477    let path = format!("/tmp/latexml_actual_{name}.xml");
478    std::fs::write(&path, &doc_str).ok();
479    eprintln!("Saved actual XML to {path}");
480    // Also save using libxml's built-in serializer for comparison
481    let path2 = format!("/tmp/latexml_actual_{name}_libxml.xml");
482    let libxml_str = doc
483      .document
484      .to_string_with_options(libxml::tree::SaveOptions {
485        format: true,
486        ..libxml::tree::SaveOptions::default()
487      });
488    std::fs::write(&path2, &libxml_str).ok();
489    eprintln!("Saved libxml XML to {path2}");
490  }
491  let mut lines: Vec<String> = doc_str.split('\n').map(ToString::to_string).collect();
492  // Remove trailing empty line from final newline
493  if lines.last().is_some_and(|l| l.is_empty()) {
494    lines.pop();
495  }
496  lines
497}

537// the project's #1 signal-integrity rule (robust error-log counting).
538
539/// Count inline `Error:<class>:` markers (parity_check.sh's lax pattern, see
540/// feedback_strict_vs_lax_error_grep.md). Errors are emitted INLINE within
541/// `(Building...Error:..)` envelopes, not at line starts.
542pub fn error_count(log: &str) -> usize {
543  log
544    .match_indices("Error:")
545    .filter(|(i, _)| {
546      let tail = &log.as_bytes()[*i + 6..];
547      let n_class = tail.iter().take_while(|b| b.is_ascii_lowercase()).count();
548      n_class > 0 && tail.get(n_class) == Some(&b':')
549    })
550    .count()
551}

553/// True iff a year-versioned latex kernel dump is present in the dev tree.
554/// Without it the engine raw-loads `expl3-code.tex` (degraded mode) and the
555/// error landscape is dominated by unrelated raw-load cascades — dump-gated
556/// tests should SKIP rather than measure the wrong thing. Delegates to the
557/// engine's own dump-name convention so a filename-scheme change cannot make
558/// the tests silently self-skip forever.
559pub fn dump_available() -> bool {
560  let dir = std::path::Path::new(concat!(env!("CARGO_MANIFEST_DIR"), "/../resources/dumps"));
561  !latexml_engine::dump_paths::available_years_in_dir(dir, "latex").is_empty()
562}

564/// True iff `kpsewhich` resolves the named file in the host TeX tree.
565pub fn kpse_has(file: &str) -> bool {
566  std::process::Command::new("kpsewhich")
567    .arg(file)
568    .output()
569    .map(|o| o.status.success() && !o.stdout.is_empty())
570    .unwrap_or(false)
571}

573/// Convert a test fixture with the standard HTML5 config and return the full
574/// response (result/log/status). The shared boilerplate for the standalone
575/// regression tests.
576pub fn convert_fixture(source: &str) -> crate::converter::ConversionResponse {
577  init_test_rss_cap();
578  let _ = latexml_core::util::logger::init(log::LevelFilter::Warn);
579  let cfg = latexml_core::common::Config {
580    format: latexml_core::common::OutputFormat::HTML5,
581    ..latexml_core::common::Config::default()
582  };
583  let mut c = crate::converter::Converter::from_config(cfg);
584  c.initialize_session().expect("initialize");
585  c.convert(source.to_string())
586}

588#[cfg(test)]
589mod exemption_audit {
590  use std::path::{Path, PathBuf};
591
592  use super::{ERROR_DEBT, INTENTIONALLY_FAILING};
593
594  /// Review m1: the exemption tables match on the bare `file_stem`, which is
595  /// NOT unique across the suite's globbed test dirs. A future `glossary.tex`
596  /// (etc.) under a different directory would silently inherit the
597  /// ERROR_DEBT / INTENTIONALLY_FAILING exemption — an ERROR_DEBT collision
598  /// could then MASK a real regression. Guard the invariant the match relies
599  /// on: every exemption key resolves to AT MOST ONE `.tex` across `tests/`.
600  /// (Cheaper and lower-churn than dir-qualifying the keys; if this ever fails,
601  /// rename the fixture or dir-qualify that entry.)
602  #[test]
603  fn exemption_keys_have_unique_stems() {
604    fn collect(dir: &Path, out: &mut Vec<(String, PathBuf)>) {
605      let Ok(rd) = std::fs::read_dir(dir) else {
606        return;
607      };
608      for ent in rd.flatten() {
609        let p = ent.path();
610        if p.is_dir() {
611          collect(&p, out);
612        } else if p.extension().and_then(|e| e.to_str()) == Some("tex")
613          && let Some(stem) = p.file_stem().and_then(|s| s.to_str())
614        {
615          out.push((stem.to_string(), p.clone()));
616        }
617      }
618    }
619    let mut all = Vec::new();
620    collect(Path::new("tests"), &mut all);
621    assert!(
622      !all.is_empty(),
623      "no .tex fixtures found under tests/ — wrong CWD?"
624    );
625
626    let keys = INTENTIONALLY_FAILING
627      .iter()
628      .map(|(n, ..)| *n)
629      .chain(ERROR_DEBT.iter().map(|(n, _)| *n));
630    for key in keys {
631      let hits: Vec<&PathBuf> = all
632        .iter()
633        .filter(|(s, _)| s == key)
634        .map(|(_, p)| p)
635        .collect();
636      assert!(
637        hits.len() <= 1,
638        "exemption key {key:?} matches {} .tex fixtures {:?} — the bare-stem \
639         match would apply the exemption to ALL of them, potentially masking a \
640         regression. Dir-qualify the entry or rename the fixture.",
641        hits.len(),
642        hits
643      );
644    }
645  }
646}
