25pub fn latexml_tests_internal(
26  dirpath: &str,
27  requires: Option<&phf::Map<&str, &str>>,
28  dispatcher_opt: Option<BindingDispatcher>,
29) {
30  if !validate_requirements(dirpath, requires) {
31    return; // test group only if required files are found.
32  }
33  for tex_file in glob(&s!("{}/*.tex", dirpath)).unwrap().flatten() {
34    let name = tex_file.file_stem().unwrap().to_str().unwrap();
35    let xml_file = tex_file.with_extension("xml");
36
37    let tex_file_string = tex_file.to_str().unwrap();
38    let xml_file_str = xml_file.to_str().unwrap();
39    if xml_file.exists() {
40      latexml_ok_internal(tex_file_string, xml_file_str, name, dispatcher_opt.clone());
41    } else {
42      // Skip, these could be tex fragment files.
43    }
44  }
45}

47static INIT_LOGGER: Once = Once::new();
48pub fn init_logger() {
49  INIT_LOGGER.call_once(|| {
50    // Use Off level for clean test output. Error/Warn counting still works
51    // via note_status(); set LATEXML_TEST_LOG=1 to see warnings during debugging.
52    let level = if *TEST_LOG {
53      log::LevelFilter::Warn
54    } else {
55      log::LevelFilter::Off
56    };
57    latexml_core::util::logger::init(level).unwrap();
58  });
59}

61// Linker-section trick: register a SIGSEGV handler via `.init_array` so
62// it runs BEFORE `main()` (and therefore before any thread the test
63// harness spawns can crash). `init_logger` was too late — by the time
64// the first test thread called it, sibling threads had already crashed.
65//
66// Gated by `LATEXML_SIGSEGV_TRACE` so the handler is opt-in; signal()
67// + std::backtrace inside a SIGSEGV handler is technically not async-
68// signal-safe (uses heap), but for diagnostic purposes a best-effort
69// stack dump is far more useful than the bare `signal: 11` line cargo
70// reports today.
71#[used]
72#[cfg_attr(target_os = "linux", link_section = ".init_array")]
73static SIGSEGV_INSTALLER: extern "C" fn() = sigsegv_installer;
74
75extern "C" fn sigsegv_installer() {
76  // Read env at process start. SAFETY: nothing has run yet, env is
77  // populated by the kernel/loader.
78  if *SIGSEGV_TRACE {
79    eprintln!("[sigsegv_installer] installing SIGSEGV handler");
80    install_sigsegv_handler();
81  }
82}

84/// Install a SIGSEGV handler that prints the crashing thread's name and a
85/// best-effort backtrace before the kernel terminates the process. This
86/// is a diagnostic-only hook — the handler is not async-signal-safe (it
87/// uses `std::backtrace` and `eprintln!`, which malloc), but for
88/// post-mortem of the libxml2-suspected multi-thread crash that's
89/// observed under `cargo test --release --tests`, even an unsafe
90/// stack print is more useful than the bare `signal: 11` line cargo
91/// reports today.
92fn install_sigsegv_handler() {
93  extern "C" {
94    fn signal(sig: i32, handler: extern "C" fn(i32)) -> usize;
95    fn raise(sig: i32) -> i32;
96  }
97  // Linux SIGSEGV = 11, SIGBUS = 7, SIGABRT = 6.
98  const SIGSEGV: i32 = 11;
99  const SIGBUS: i32 = 7;
100  const SIGABRT: i32 = 6;
101  const SIG_DFL: usize = 0;
102
103  extern "C" fn handler(sig: i32) {
104    // Capture context synchronously and persist to a per-pid file —
105    // cargo test buffers/discards stderr from binaries that exit by
106    // signal, so eprintln!() never reaches the user. Writing to
107    // `/tmp/latexml_sigsegv_<pid>.txt` survives the kill.
108    let tid = std::thread::current().id();
109    let name = std::thread::current()
110      .name()
111      .unwrap_or("<unnamed>")
112      .to_string();
113    let pid = std::process::id();
114    let path = format!("/tmp/latexml_sigsegv_{pid}.txt");
115    let bt = std::backtrace::Backtrace::force_capture();
116    let exe = std::env::current_exe()
117      .map(|p| p.display().to_string())
118      .unwrap_or_else(|_| "<unknown>".into());
119    let body = format!(
120      "=== SIGSEGV-handler ===\nsignal={sig}\nthread={name:?}\nid={tid:?}\nexe={exe}\npid={pid}\n\n{bt}\n"
121    );
122    let _ = std::fs::write(&path, &body);
123    // Also try eprintln (best effort; usually lost by cargo on signal).
124    eprintln!("{body}");
125    eprintln!("[SIGSEGV-handler] full trace written to {path}");
126    // Reset to default and re-raise so cargo still sees the original signal.
127    unsafe {
128      let raw_dfl: extern "C" fn(i32) = std::mem::transmute(SIG_DFL);
129      signal(sig, raw_dfl);
130      raise(sig);
131    }
132  }
133
134  unsafe {
135    signal(SIGSEGV, handler);
136    signal(SIGBUS, handler);
137    signal(SIGABRT, handler);
138  }
139}

141/// Tests whose TeX input is *known* to produce Error/Warn messages in both
142/// the Perl reference implementation and the Rust port.  We suppress log
143/// output for these so that `cargo test` runs cleanly, while still counting
144/// errors internally (MAX_ERRORS check still fires).
145/// Tests where Perl LaTeXML also produces Error/Warn messages.
146/// ONLY add tests here if verified that Perl `bin/latexml` emits errors on the same input.
147const KNOWN_ERROR_TESTS: &[&str] = &[
148  "io",                   // Perl: 2 errors (mode-switch egroup from \readnext)
149  "figure_mixed_content", // Perl: 1 error (ltx:theorem not allowed in ltx:figure)
150];
151
152pub fn latexml_test_single(
153  tex_file_str: &str,
154  name: &str,
155  dirpath: &str,
156  requires: Option<&phf::Map<&str, &str>>,
157  dispatcher_opt: Option<BindingDispatcher>,
158) {
159  init_logger();
160  if !validate_requirements(dirpath, requires) {
161    return; // test group only if required files are found.
162  }
163  let suppress = KNOWN_ERROR_TESTS.contains(&name);
164  if suppress {
165    latexml_core::common::error::set_suppress_log_output(true);
166  }
167  let tex_file = PathBuf::from(tex_file_str);
168  let xml_file = tex_file.with_extension("xml");
169  if matches!(xml_file.try_exists(), Ok(true)) {
170    latexml_ok_internal(
171      tex_file_str,
172      &xml_file.to_string_lossy(),
173      name,
174      dispatcher_opt,
175    );
176  } else {
177    // Skip, these could be tex fragment files.
178  }
179  if suppress {
180    latexml_core::common::error::set_suppress_log_output(false);
181  }
182}

192fn latexml_ok_internal(
193  tex_path: &str,
194  xml_path: &str,
195  name: &str,
196  extra_bindings_dispatcher: Option<BindingDispatcher>,
197) {
198  let tex_strings = process_texfile(tex_path, name, extra_bindings_dispatcher);
199  if !tex_strings.is_empty() {
200    let xml_strings = process_xmlfile(xml_path, name);
201    if !xml_strings.is_empty() {
202      let mut found_diff = false;
203      for (lineno, (tex_line, xml_line)) in tex_strings.iter().zip(xml_strings.iter()).enumerate() {
204        if tex_line != xml_line {
205          found_diff = true;
206          eprintln!(
207            "DIFF line {lineno} in {xml_path}:\n  ACTUAL:   {tex_line}\n  EXPECTED: {xml_line}"
208          );
209        }
210      }
211      if tex_strings.len() != xml_strings.len() {
212        found_diff = true;
213        eprintln!(
214          "DIFF length mismatch for {name:?}: actual {} lines, expected {} lines",
215          tex_strings.len(),
216          xml_strings.len()
217        );
218        // Print extra lines
219        let min_len = tex_strings.len().min(xml_strings.len());
220        if tex_strings.len() > min_len {
221          for (i, line) in tex_strings[min_len..].iter().enumerate() {
222            eprintln!("  ACTUAL extra line {}: {line}", min_len + i);
223          }
224        }
225        if xml_strings.len() > min_len {
226          for (i, line) in xml_strings[min_len..].iter().enumerate() {
227            eprintln!("  EXPECTED extra line {}: {line}", min_len + i);
228          }
229        }
230      }
231      if found_diff {
232        panic!("Differences found in {xml_path} — see DIFF lines above");
233      }
234    }
235  }
236}

238/// Returns the list-of-strings form of whatever was requested, if successful,
239/// otherwise empty; and they will have reported the failure
240fn process_texfile(
241  tex_path: &str,
242  name: &str,
243  extra_bindings_dispatcher: Option<BindingDispatcher>,
244) -> Vec<String> {
245  let mut latexml = Core::new(CoreOptions {
246    verbosity: Some(-2),
247    search_paths: None,
248    preload: None,
249    include_comments: Some(false),
250    ..CoreOptions::default()
251  });
252  // Add the package bindings
253  state::set_bindings_dispatch(Rc::new(latexml_package::dispatch));
254  state::add_binding_names(latexml_package::binding_names());
255  // If we want to test the latexml_contrib bindings, we need to pass in the additional binding
256  // dispatcher, which makes the contrib bindings visible
257  if let Some(dispatcher) = extra_bindings_dispatcher {
258    state::set_extra_bindings_dispatch(dispatcher);
259  }
260  match latexml.convert_file(tex_path.to_owned()) {
261    Err(e) => panic!("{:?}: Couldn't convert {:?}; {:?}", name, tex_path, e),
262    Ok(doc) => process_ltx_doc(doc, name),
263  }
264}

266/// Loads the reference XML file as raw text lines, avoiding libxml2
267/// re-serialization which would normalize `<p></p>` to `<p/>`.
268fn process_xmlfile<'a>(xml_path: &'a str, _name: &'a str) -> Vec<String> {
269  match std::fs::read_to_string(xml_path) {
270    Err(e) => panic!("Failed to read XML file {:?}: {:?}", xml_path, e),
271    Ok(contents) => {
272      let mut lines: Vec<String> = contents.split('\n').map(ToString::to_string).collect();
273      // Remove trailing empty line from final newline
274      if lines.last().is_some_and(|l| l.is_empty()) {
275        lines.pop();
276      }
277      lines
278    },
279  }
280}
281fn process_ltx_doc(doc: Document, name: &str) -> Vec<String> {
282  let doc_str = doc.serialize_to_string();
283  if *SAVE_ACTUAL {
284    let path = format!("/tmp/latexml_actual_{name}.xml");
285    std::fs::write(&path, &doc_str).ok();
286    eprintln!("Saved actual XML to {path}");
287    // Also save using libxml's built-in serializer for comparison
288    let path2 = format!("/tmp/latexml_actual_{name}_libxml.xml");
289    let libxml_str = doc
290      .document
291      .to_string_with_options(libxml::tree::SaveOptions {
292        format: true,
293        ..libxml::tree::SaveOptions::default()
294      });
295    std::fs::write(&path2, &libxml_str).ok();
296    eprintln!("Saved libxml XML to {path2}");
297  }
298  let mut lines: Vec<String> = doc_str.split('\n').map(ToString::to_string).collect();
299  // Remove trailing empty line from final newline
300  if lines.last().is_some_and(|l| l.is_empty()) {
301    lines.pop();
302  }
303  lines
304}
