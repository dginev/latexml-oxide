1//! Post-processing pipeline API.
2//!
3//! Provides a public interface to the LaTeXML post-processing pipeline
4//! (Scan → Bibliography → CrossRef → Graphics → Split → MathML → XSLT → HTML5 fixups).
5//! Used by both the `latexml_oxide` binary and the `cortex_worker` binary.
6

8  Info, Warn,
9  common::error::{LogStatus, note_progress, note_status},
10  s,
11  telemetry::{self, Phase},
12};

14  document::{PostDocument, PostDocumentOptions},
15  object_db::ObjectDB,
16  processor::Processor,

20// Process-once cached env var (see WISDOM #56 — getenv hot-path race).
21static POST_AUDIT: Lazy<bool> = Lazy::new(|| std::env::var("LATEXML_POST_AUDIT").is_ok());
22
23/// Options for the post-processing pipeline.
24pub struct PostOptions<'a> {
25  pub pmml:                      bool,
26  pub cmml:                      bool,
27  pub keep_xmath:                bool,
28  pub stylesheet:                Option<&'a str>,
29  pub destination:               Option<&'a str>,
30  pub source_directory:          Option<&'a str>,
31  /// Extra resource search paths (the `--path` flag): directories searched
32  /// for `--css`/`--javascript` files (and other post resources) to copy into
33  /// the destination, in addition to the document's own paths, the current
34  /// directory, and the binary's embedded resource table.
35  pub search_paths:              &'a [String],
36  pub nodefaultresources:        bool,
37  pub css_files:                 &'a [String],
38  pub js_files:                  &'a [String],
39  pub noinvisibletimes:          bool,
40  pub mathtex:                   bool,
41  pub navigationtoc:             Option<&'a str>,
42  pub schemadocs:                bool,
43  pub split:                     bool,
44  pub split_xpath:               Option<String>,
45  pub split_naming:              Option<&'a str>,
46  pub xslt_parameters:           &'a [String],
47  /// If > 0, try the vector-SVG converters (mutool → pdftocairo) for PDF
48  /// graphics smaller than this many KB (vector-preservation path). Fall
49  /// back to the raster ImageMagick `convert`/`gs` → PNG path on failure
50  /// or timeout. Tracks upstream brucemiller/LaTeXML#902.
51  pub graphics_svg_threshold_kb: u32,
52  /// Output extraction mode (Perl `LaTeXML::Util::Pack::whatsout`).
53  /// `Document` (default) → serialize the full post-processed
54  /// document; `Fragment` → embeddable HTML snippet via
55  /// `latexml_post::extract::get_embeddable`; `Math` → math subtree
56  /// via `get_math`. Applied to each `PostDocument` in the final
57  /// serialization loop.
58  pub whatsout:                  latexml_post::extract::Whatsout,
59}

61/// Emit a post-processing error: capture it into the active log buffer (so it
62/// reaches `cortex.log`) AND bump the Error counter so `Status:conversion`
63/// reflects post-phase failures (Perl `LaTeXML.pm` L633: `max(core, post)`).
64///
65/// Deliberately NOT the `Error!` macro: that expands to a too-many-errors
66/// `Fatal!(…)` → `return Err(error::Error)`, which only typechecks inside the
67/// digester's `Result<_, error::Error>` functions. The post-processing entry
68/// points return `String`/`Result<_, ()>`, so we log+count directly with no
69/// control-flow side effect.
70fn post_error(object: &str, message: &str) {
71  note_status(LogStatus::Error, None);
72  log::error!(target: &format!("post:{object}"), "{message}");
73}
74
75/// Result of [`run_post_processing_logged`]: the serialized HTML plus the
76/// post-phase log text and status code, mirroring Perl `LaTeXML.pm`'s
77/// `convert_post`, which flushes the log AFTER post and folds the post status
78/// into the final `max(core, post)` verdict (L631-634).
79pub struct PostOutcome {
80  /// Serialized post-processed output (same value as [`run_post_processing`]).
81  pub html:        String,
82  /// Log text captured during the post phase only — Graphics/MathML/XSLT
83  /// `Info!`/`Warn!`/`post_error` lines. Append to the core conversion log.
84  pub log:         String,
85  /// Status code read from the shared REPORT counter AFTER post. Because the
86  /// counter is NOT reset between core and post, this is already the combined
87  /// core+post code; callers still `max()` it with the core code as a
88  /// belt-and-suspenders guard for paths that force a fatal outside REPORT
89  /// (e.g. a `catch_unwind`-trapped panic).
90  pub status_code: usize,
91}

93/// Run post-processing while capturing its log into a fresh thread-local
94/// LOG_BUFFER. `Converter::convert()` already flushed the *core* log into its
95/// own response, leaving the buffer unbound — so without re-binding here every
96/// post-phase `Info!`/`Warn!`/`post_error` (incl. a silent EPS→PNG failure)
97/// reaches stderr only and never the persisted `--log`/`cortex.log`. Perl's
98/// single `flush_log()` after `convert_post` sweeps up the post log for ALL
99/// consumers; this is the Rust equivalent, shared by the `latexml_oxide` and
100/// `cortex_worker` binaries so their logs reach parity. See SYNC_STATUS task 5.
101pub fn run_post_processing_logged(xml: &str, opts: &PostOptions) -> PostOutcome {
102  run_post_processing_impl_logged(PostInput::Xml(xml), opts)
103}
104
105/// File-input variant of [`run_post_processing_logged`]: parses the XML from
106/// disk via libxml2's streaming reader rather than from an in-memory `String`.
107/// Used for already-converted `.xml` inputs, where slurping the file would keep
108/// a whole extra copy of a multi-hundred-MB document resident alongside the DOM.
109pub fn run_post_processing_from_file_logged(path: &str, opts: &PostOptions) -> PostOutcome {
110  run_post_processing_impl_logged(PostInput::File(path), opts)
111}
112
113fn run_post_processing_impl_logged(input: PostInput, opts: &PostOptions) -> PostOutcome {
114  latexml_core::util::logger::bind_log();
115  let html = run_post_processing_impl(input, opts);
116  let log = latexml_core::util::logger::flush_log();
117  let status_code = latexml_core::common::error::get_status_code();
118  PostOutcome { html, log, status_code }
119}

121/// Where the post-pipeline reads its already-converted LaTeXML XML from.
122#[derive(Clone, Copy)]
123enum PostInput<'a> {
124  /// XML already resident in memory (the TeX→XML conversion result). The common
125  /// path — the 2.8M-doc arXiv fleet and every TeX conversion feed this.
126  Xml(&'a str),
127  /// A path to an XML file, parsed via libxml2's streaming file reader
128  /// (`xmlReadIO`). Avoids vivifying a very large document as a Rust `String`
129  /// on top of the ~11× DOM it becomes.
130  File(&'a str),
131}

133/// Run the post-processing pipeline on in-memory XML output.
134///
135/// Executes: Split → Scan → MakeBibliography → CrossRef → Graphics → MathML → XSLT → HTML5 fixups.
136pub fn run_post_processing(xml: &str, opts: &PostOptions) -> String {
137  run_post_processing_impl(PostInput::Xml(xml), opts)
138}

140/// File-input variant of [`run_post_processing`].

142  run_post_processing_impl(PostInput::File(path), opts)

145fn run_post_processing_impl(input: PostInput, opts: &PostOptions) -> String {
146  let PostOptions {
147    pmml,
148    cmml,
149    keep_xmath,
150    stylesheet,
151    destination,
152    source_directory,
153    search_paths,
154    nodefaultresources,
155    css_files,
156    js_files,
157    noinvisibletimes,
158    mathtex,
159    navigationtoc,
160    schemadocs,
161    split,
162    ref split_xpath,
163    split_naming,
164    xslt_parameters,
165    graphics_svg_threshold_kb,
166    whatsout,
167  } = *opts;
168
169  let mut doc_opts = PostDocumentOptions::default();
170  if let Some(dest) = destination {
171    doc_opts.destination = Some(dest.to_string());
172  }
173  if let Some(src_dir) = source_directory {
174    doc_opts.source_directory = Some(src_dir.to_string());
175    let mut sp = doc_opts.searchpaths.take().unwrap_or_default();
176    sp.push(src_dir.to_string());
177    doc_opts.searchpaths = Some(sp);
178  }
179  let audit = *POST_AUDIT;
180  let audit_start = |name: &str| -> Option<(String, std::time::Instant)> {
181    if audit {
182      Some((name.to_string(), std::time::Instant::now()))
183    } else {
184      None
185    }
186  };
187  let audit_end = |started: Option<(String, std::time::Instant)>| {
188    if let Some((name, t0)) = started {
189      let ms = t0.elapsed().as_millis();
190      Info!("audit", "phase", s!("{} took {}ms", name, ms));
191    }
192  };
193
194  telemetry::phase_enter(Phase::PostXmlParse);
195  let t_parse = audit_start("PostDocument parse");
196  // `raw_xml` holds the in-memory serialization — `Some` only for `Xml` input.
197  // `File` input streams from disk and never holds it, so `raw_xml` is `None`
198  // and the three raw-string features (empty-doc substitution, SVG extraction,
199  // ar5iv sniff) plus the on-failure echo are re-derived from the parsed doc.
200  //
201  // Perl LaTeXML.pm L330-336: an empty core result (e.g. after a Fatal) is still
202  // post-processed against a bare <document/> root ("important for utility
203  // features such as packing .zip archives"). Mirror that for empty in-memory
204  // input; a File is never empty (an empty file simply errors on parse below).
205  let raw_xml: Option<&str> = match input {
206    PostInput::Xml(xml) if xml.trim().is_empty() => Some("<document/>"),
207    PostInput::Xml(xml) => Some(xml),
208    PostInput::File(_) => None,
209  };
210  // Best-effort output when a phase bails: echo the original XML for in-memory
211  // input; for File input the source is on disk (and a failure here is itself
212  // the case we are hardening against), so return empty and rely on the logged
213  // Error + status code.
214  let fallback = || raw_xml.map(str::to_string).unwrap_or_default();
215
216  let parsed = match input {
217    // `raw_xml` is `Some` for `Xml` input (the two arms above), so the default
218    // is dead — kept only to avoid an `unwrap`.
219    PostInput::Xml(_) => PostDocument::new_from_string(raw_xml.unwrap_or("<document/>"), doc_opts),
220    PostInput::File(path) => PostDocument::new_from_file(path, doc_opts),
221  };
222  let doc = match parsed {
223    Ok(d) => d,
224    Err(e) => {
225      post_error("parse", &format!("failed to parse XML: {e}"));
226      telemetry::phase_exit();
227      return fallback();
228    },
229  };
230  audit_end(t_parse);
231  telemetry::phase_exit();
232
233  // SVG extraction reads the pre-post XML before any in-tree mutation; do it
234  // once up-front so the regex-based fragment table is valid for every split
235  // sub-document below. In-memory input scans the string; File input locates
236  // pictures via a limit-safe `//ltx:picture` DOM query and serializes only
237  // those (never the whole file) — most large documents have none.
238  let t_svg = audit_start("SVG extraction");
239  let svg_fragments = match raw_xml {
240    Some(xml) => extract_svg_fragments(xml),
241    None => extract_svg_fragments_from_doc(&doc),
242  };
243  audit_end(t_svg);
244
245  // ar5iv sniff: the ar5iv package emits literal-intent MathML. In-memory input
246  // checks the raw string; file input checks the parsed doc's `<?latexml
247  // package="ar5iv"?>` PI. Computed here — before Split moves `doc` into `docs`.
248  let intent_literal = match raw_xml {
249    Some(xml) => xml.contains("package=\"ar5iv"),
250    None => doc
251      .processing_instructions()
252      .iter()
253      .any(|pi| pi.contains("package=\"ar5iv")),
254  };
255
256  // Perl-faithful pipeline order (latexmlpost L223-242):
257  //   Split → Scan → MakeBibliography → CrossRef → Graphics → ...
258  // Split runs FIRST so each downstream pass sees the per-page
259  // destination. With the previous order (Scan before Split), every
260  // entry's `location` was the root document and CrossRef built every
261  // ref as a within-page anchor — the user-visible TOC links pointed
262  // at `#Ch1` instead of `Ch1.html`.
263  //
264  // Phase 1: Split (only attributes time when --split is on)
265  let mut docs: Vec<PostDocument> = if split {
266    telemetry::phase_enter(Phase::Split);
267    let result = if let Some(xpath) = split_xpath {
268      let naming = match split_naming {
269        Some("id") | None => latexml_post::split::SplitNaming::Id,
270        Some("idrelative") => latexml_post::split::SplitNaming::IdRelative,
271        Some("label") => latexml_post::split::SplitNaming::Label,
272        Some("labelrelative") => latexml_post::split::SplitNaming::LabelRelative,
273        Some(other) => {
274          Warn!(
275            "post",
276            "split",
277            format!("Unknown splitnaming '{other}', using 'id'")
278          );
279          latexml_post::split::SplitNaming::Id
280        },
281      };
282      let mut splitter = latexml_post::split::Split::new(xpath, naming, false);
283      let split_nodes = splitter.to_process(&doc);
284      match splitter.process(doc, split_nodes) {
285        Ok(docs) => {
286          if docs.len() > 1 {
287            note_progress(&format!("Split into {} documents", docs.len()));
288          }
289          Ok(docs)
290        },
291        Err(e) => {
292          post_error("split", &format!("Split failed: {e}"));
293          Err(())
294        },
295      }
296    } else {
297      Ok(vec![doc])
298    };
299    telemetry::phase_exit();
300    match result {
301      Ok(ds) => ds,
302      Err(()) => return fallback(),
303    }
304  } else {
305    vec![doc]
306  };
307
308  // Helper: run one processor across every doc in a Vec, mirroring
309  // Perl Post.pm:50-65's `foreach $proc { @newdocs = ... }` loop.
310  // Each processor's per-doc `process` may return multiple docs (only
311  // Split actually does, and we've already run Split above), so the
312  // inner result is flattened back into `docs`.
313  fn run_phase<P: Processor + ?Sized>(
314    docs: Vec<PostDocument>,
315    proc: &mut P,
316    label: &'static str,
317  ) -> Result<Vec<PostDocument>, ()> {
318    let mut out: Vec<PostDocument> = Vec::with_capacity(docs.len());
319    for d in docs {
320      let nodes = proc.to_process(&d);
321      if nodes.is_empty() {
322        out.push(d);
323        continue;
324      }
325      match proc.process(d, nodes) {
326        Ok(processed) => out.extend(processed),
327        Err(e) => {
328          post_error(label, &format!("{label} failed: {e}"));
329          return Err(());
330        },
331      }
332    }
333    Ok(out)
334  }
335
336  // Phase 2: Scan — runs on EACH sub-document so its entries register
337  // the per-page `location` and `pageid`. Single shared ObjectDB so
338  // the later CrossRef pass can resolve cross-doc refs.
339  let mut scanner = latexml_post::scan::Scan::new(ObjectDB::new());
340  telemetry::phase_enter(Phase::PostScan);
341  let t_scan = audit_start("Scan");
342  docs = match run_phase(docs, &mut scanner, "Scan") {
343    Ok(d) => d,
344    Err(()) => {
345      telemetry::phase_exit();
346      return fallback();
347    },
348  };
349  audit_end(t_scan);
350  telemetry::phase_exit();
351
352  // Phase 2b: MakeIndex (Perl LaTeXML.pm L466-470)
353  //   Runs BEFORE MakeBibliography in Perl. Populates `<ltx:indexlist>`
354  //   inside `<ltx:index>` placeholders and `<ltx:glossarylist>` inside
355  //   `<ltx:glossary>` placeholders, using the `GLOSSARY:*` / `INDEX:*`
356  //   entries Scan registered in the ObjectDB. Without this pass, glossary
357  //   sections render empty in HTML (witness: tests/structure/glossary.tex).
358  let mut indexer = latexml_post::make_index::MakeIndex::new(scanner.db, false, false);
359  telemetry::phase_enter(Phase::PostScan);
360  let t_idx = audit_start("MakeIndex");
361  docs = match run_phase(docs, &mut indexer, "MakeIndex") {
362    Ok(d) => d,
363    Err(()) => {
364      telemetry::phase_exit();
365      return fallback();
366    },
367  };
368  audit_end(t_idx);
369  telemetry::phase_exit();
370
371  // Phase 3: MakeBibliography
372  let mut bibmaker = latexml_post::make_bibliography::MakeBibliography::new(indexer.db, false);
373  telemetry::phase_enter(Phase::Bibliography);
374  let t_bib = audit_start("MakeBibliography");
375  docs = match run_phase(docs, &mut bibmaker, "MakeBibliography") {
376    Ok(d) => d,
377    Err(()) => {
378      telemetry::phase_exit();
379      return fallback();
380    },
381  };
382  audit_end(t_bib);
383  telemetry::phase_exit();
384
385  // Phase 4: CrossRef
386  let mut crossref = latexml_post::crossref::CrossRef::new(
387    bibmaker.db,
388    latexml_post::crossref::UrlStyle::File,
389    true,
390  );
391  if let Some(navtoc) = navigationtoc {
392    crossref.set_navigation_toc(navtoc);
393  }
394  telemetry::phase_enter(Phase::Crossref);
395  let t_xref = audit_start("CrossRef");
396  docs = match run_phase(docs, &mut crossref, "CrossRef") {
397    Ok(d) => d,
398    Err(()) => {
399      telemetry::phase_exit();
400      return fallback();
401    },
402  };
403  audit_end(t_xref);
404  telemetry::phase_exit();
405
406  // Phase 5: Graphics
407  let mut graphics_proc = latexml_post::graphics::Graphics::new(None, true)
408    .with_svg_threshold_kb(graphics_svg_threshold_kb);
409  telemetry::phase_enter(Phase::Graphics);
410  let t_gfx = audit_start("Graphics");
411  docs = match run_phase(docs, &mut graphics_proc, "Graphics") {
412    Ok(d) => d,
413    Err(()) => {
414      telemetry::phase_exit();
415      return fallback();
416    },
417  };
418  audit_end(t_gfx);
419  telemetry::phase_exit();
420
421  // Phase 3: MathML + XSLT
422  let mut post = latexml_post::Post::new();
423  let mut processors: Vec<Box<dyn Processor>> = Vec::new();
424
425  // `intent_literal` was computed up-front (before Split consumed `doc`).
426
427  // Parallel P+C markup: when both formats are requested, the Content-MathML
428  // processor is a *secondary* of the Presentation primary, folded into one
429  // `<m:semantics>`/`<m:annotation-xml>` by `combine_parallel` — NOT a second
430  // independent pass (which left the content tree as an orphan `<apply>` sibling
431  // of `<m:semantics>`, rendered as stray text by browsers). Mirrors Perl
432  // `MathProcessor`'s primary→secondary parallel model.
433  if pmml {
434    let mut presentation = latexml_post::mathml::MathML::new_presentation()
435      .with_keep_xmath(keep_xmath)
436      .with_invisible_times(!noinvisibletimes)
437      .with_mathtex(mathtex)
438      .with_intent_literal(intent_literal);
439    if cmml {
440      presentation = presentation.with_secondaries(vec![Box::new(
441        latexml_post::mathml::MathML::new_content()
442          .with_keep_xmath(keep_xmath)
443          .with_invisible_times(!noinvisibletimes)
444          .secondary(),
445      )]);
446    }
447    processors.push(Box::new(presentation));
448  } else if cmml {
449    // Content-only output (no presentation primary).
450    processors.push(Box::new(
451      latexml_post::mathml::MathML::new_content()
452        .with_keep_xmath(keep_xmath)
453        .with_invisible_times(!noinvisibletimes),
454    ));
455  }
456  if let Some(xsl_path) = stylesheet {
457    let mut searchpaths = vec![".".to_string()];
458    if let Ok(exe) = std::env::current_exe()
459      && let Some(project_root) = exe
460        .parent()
461        .and_then(|p| p.parent())
462        .and_then(|p| p.parent())
463    {
464      searchpaths.insert(0, project_root.display().to_string());
465    }
466    // Prepend the user's `--path` directories so `--css`/`--javascript`
467    // resources are found there (and take priority over `.`/project root)
468    // when copy_param_resources searches for them.
469    for p in search_paths.iter().rev() {
470      searchpaths.insert(0, p.clone());
471    }
472    let mut xslt_params = rustc_hash::FxHashMap::default();
473
474    // When `--schemadocs` is on, auto-prepend the rustdoc-styled
475    // theme assets so callers don't need to repeat them on every
476    // invocation. Idempotent: skipped if the user has already
477    // listed the same basename via `--css` / `--javascript`.
478    let prepend = |list: &[String], extra: &str| -> Vec<String> {
479      if list
480        .iter()
481        .any(|p| std::path::Path::new(p).file_name().and_then(|s| s.to_str()) == Some(extra))
482      {
483        list.to_vec()
484      } else {
485        let mut out = Vec::with_capacity(list.len() + 1);
486        out.push(extra.to_string());
487        out.extend_from_slice(list);
488        out
489      }
490    };
491    let css_effective: Vec<String> = if schemadocs {
492      prepend(css_files, latexml_post::schema_docs::THEME_CSS_BASENAME)
493    } else {
494      css_files.to_vec()
495    };
496    let js_effective: Vec<String> = if schemadocs {
497      prepend(js_files, latexml_post::schema_docs::THEME_JS_BASENAME)
498    } else {
499      js_files.to_vec()
500    };
501    if !css_effective.is_empty() {
502      xslt_params.insert(
503        "CSS".to_string(),
504        format!("\"{}\"", css_effective.join("|")),
505      );
506    }
507    if !js_effective.is_empty() {
508      xslt_params.insert(
509        "JAVASCRIPT".to_string(),
510        format!("\"{}\"", js_effective.join("|")),
511      );
512    }
513    if let Some(navtoc) = navigationtoc {
514      xslt_params.insert("NAVIGATIONTOC".to_string(), format!("\"{}\"", navtoc));
515    }
516    for param in xslt_parameters {
517      if let Some((key, value)) = param.split_once('=') {
518        xslt_params.insert(key.to_string(), format!("\"{}\"", value));
519      }
520    }
521    match latexml_post::xslt::XSLT::new(
522      xsl_path,
523      xslt_params,
524      nodefaultresources,
525      None,
526      searchpaths,
527    ) {
528      Ok(xslt) => processors.push(Box::new(xslt)),
529      Err(e) => post_error("xslt", &format!("XSLT error: {e}")),
530    }
531  }
532
533  // process_chain attributes per-processor inside latexml_post::Post::
534  // process_chain (MathmlPres / MathmlCont / Xslt). No outer phase wrap
535  // needed; the inner per-processor guards cover their own time.
536  //
537  // Perl-faithful: pass ALL split docs to ProcessChain at once. Perl's
538  // `Post::ProcessChain_internal` (Post.pm L41-67) seeds `@docs = ($doc)`
539  // and lets each processor return a (possibly multi-element) list which
540  // becomes the input to the next processor. We've already produced the
541  // post-Split list above; ProcessChain just needs to fan MathML/XSLT
542  // across it.
543  let is_html_out = stylesheet.is_some_and(|s| s.contains("html"));
544  let t_chain = audit_start("process_chain");
545  let chain_result = post.process_chain(docs, &mut processors);
546  audit_end(t_chain);
547  let results = match chain_result {
548    Ok(r) => r,
549    Err(e) => {
550      post_error("convert", &format!("Post-processing failed: {e}"));
551      return fallback();
552    },
553  };
554
555  // Serialize + write each doc IMMEDIATELY in the loop (rather than
556  // collecting first then writing). Multi-doc PostDocument cleanup has
557  // a known libxml2 idcache use-after-free at process exit (see the
558  // SVG-processor note above this block) that can intermittently SIGSEGV
559  // after the last doc has been serialized. Writing eagerly inside the
560  // iteration means even if cleanup later trips, every page that finished
561  // post-processing is already on disk. The first doc's content is
562  // returned so the caller can also write it to --dest in non-split mode.
563  let mut main_output: Option<String> = None;
564  let n = results.len();
565  for doc in results.into_iter() {
566    let dest = doc.get_destination().map(String::from);
567    let t_serialize = audit_start("to_xml_string");
568    // `serialize_whatsout` is a no-op (returns full doc) for the
569    // default `Whatsout::Document`; for Fragment / Math it returns
570    // the extracted subtree serialized via libxml `node_to_string`.
571    let output = latexml_post::extract::serialize_whatsout(&doc, whatsout);
572    audit_end(t_serialize);
573    let output = if is_html_out {
574      finalize_html5(output, &svg_fragments)
575    } else {
576      output
577    };
578    let output = if schemadocs && is_html_out {
579      latexml_post::schema_docs::process_page(&output)
580    } else {
581      output
582    };
583    // Write each doc to disk inside the loop. The single-file caller
584    // also writes the first doc's content to --dest; the redundant write
585    // is harmless and ensures the file is on disk even if a later libxml2
586    // cleanup tripwire fires.
587    if let Some(path) = dest.as_deref() {
588      if let Some(parent) = std::path::Path::new(path).parent()
589        && !parent.as_os_str().is_empty()
590      {
591        let _ = std::fs::create_dir_all(parent);
592      }
593      if let Err(e) = std::fs::write(path, &output) {
594        post_error("write", &format!("failed to write page {path}: {e}"));
595      }
596    }
597    if main_output.is_none() {
598      main_output = Some(output);
599    }
600    let _ = n;
601  }
602
603  main_output.unwrap_or_else(fallback)
604}

606/// Re-derive SVG fragments from a parsed document (the file-input equivalent of
607/// [`extract_svg_fragments`]).
608///
609/// `//ltx:picture` is a *typed* descendant query, which libxml2 evaluates
610/// without materializing `descendant-or-self::node()` — so it stays under the
611/// 10M node-set ceiling even on a 600 MB document. Most large documents have no
612/// pictures at all (the fast, zero-serialization path); when they do, only the
613/// picture subtrees are serialized and handed to the shared string extractor,
614/// never the whole file.
615fn extract_svg_fragments_from_doc(doc: &PostDocument) -> Vec<(String, String)> {
616  let pictures = doc.findnodes("//ltx:picture");
617  if pictures.is_empty() {
618    return Vec::new();
619  }
620  let joined: String = pictures.iter().map(|p| doc.node_to_string(p)).collect();
621  extract_svg_fragments(&joined)
622}

624/// Apply HTML5 cleanup (XML prolog strip, void-element fixes) and inject SVG
625/// fragments into empty `ltx_picture` spans. Pulled out of `run_post_processing`
626/// so it can run on every split sub-document, not just the first.
627fn finalize_html5(output: String, svg_fragments: &[(String, String)]) -> String {
628  use std::sync::LazyLock;
629  // Cached at first call — regex compile is the slow part of `Regex::new`,
630  // and finalize_html5 runs on every (sub-)document in the post-pipeline.
631  static XML_PROLOG_RE: LazyLock<regex::Regex> =
632    LazyLock::new(|| regex::Regex::new(r"^<\?xml[^?]*\?>\s*").unwrap());
633  static NON_VOID_SELF_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
634    regex::Regex::new(
635      r"<(span|div|p|a|td|th|tr|section|article|figure|figcaption|pre|code|em|strong|b|i|u|sub|sup|small|cite)(\s[^>]*)?/>",
636    ).unwrap()
637  });
638  static VOID_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
639    regex::Regex::new(r"</(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)>")
640      .unwrap()
641  });
642  static VOID_SELF_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
643    regex::Regex::new(
644      r"<(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)(\s[^>]*?)\s*/>",
645    )
646    .unwrap()
647  });
648
649  let _gp_html5 = telemetry::phase(Phase::Html5Fixups);
650  // Strip <?xml version...?> prolog: HTML5 must NOT have an XML declaration.
651  // libxml2's to_string() includes it by default; we strip it here.
652  let output = XML_PROLOG_RE.replace(&output, "").to_string();
653  let output = NON_VOID_SELF_CLOSE_RE
654    .replace_all(&output, "<$1$2></$1>")
655    .to_string();
656  let output = VOID_CLOSE_RE.replace_all(&output, "").to_string();
657  let mut output = VOID_SELF_CLOSE_RE
658    .replace_all(&output, "<$1$2>")
659    .to_string();
660  if !svg_fragments.is_empty() {
661    for (pic_id, svg_html) in svg_fragments {
662      let pattern = format!(
663        r#"<span id="{}" class="ltx_picture"([^>]*)></span>"#,
664        regex::escape(pic_id)
665      );
666      if let Ok(re) = regex::Regex::new(&pattern) {
667        output = re
668          .replace(&output, |caps: &regex::Captures| {
669            format!(
670              r#"<span id="{}" class="ltx_picture"{}>{}</span>"#,
671              pic_id, &caps[1], svg_html
672            )
673          })
674          .to_string();
675      }
676    }
677  }
678  output
679}

681/// Extract SVG fragments from intermediate LaTeXML XML.
682///
683/// Finds `<picture>` elements, converts their children to inline SVG HTML.
684/// Uses a lightweight regex+string approach (no libxml2) to avoid the
685/// use-after-free crash in PostDocument cleanup.
686///
687/// Returns (picture_id, svg_html) pairs for post-XSLT injection.
688fn extract_svg_fragments(xml: &str) -> Vec<(String, String)> {
689  use std::sync::LazyLock;
690  // Fast-fail: most documents have no `<picture>` elements (tikz / pgf
691  // is uncommon in the canvas). Skip the backtracking lazy-match
692  // regex (`(?s)...(.*?)`) when `<picture` doesn't appear as a literal
693  // substring. `str::contains` is a SIMD-accelerated byte search and
694  // takes microseconds even on ~MB inputs.
695  if !xml.contains("<picture") {
696    return Vec::new();
697  }
698  static PICTURE_RE: LazyLock<regex::Regex> =
699    LazyLock::new(|| regex::Regex::new(r#"(?s)<picture([^>]*)>(.*?)</picture>"#).unwrap());
700  static ID_RE: LazyLock<regex::Regex> =
701    LazyLock::new(|| regex::Regex::new(r#"xml:id="([^"]+)""#).unwrap());
702  static WIDTH_RE: LazyLock<regex::Regex> =
703    LazyLock::new(|| regex::Regex::new(r#"width="([^"]+)""#).unwrap());
704  static HEIGHT_RE: LazyLock<regex::Regex> =
705    LazyLock::new(|| regex::Regex::new(r#"height="([^"]+)""#).unwrap());
706  let mut fragments = Vec::new();
707  let picture_re = &*PICTURE_RE;
708  let id_re = &*ID_RE;
709  let width_re = &*WIDTH_RE;
710  let height_re = &*HEIGHT_RE;
711
712  for pic_caps in picture_re.captures_iter(xml) {
713    let attrs = &pic_caps[1];
714    let content = &pic_caps[2];
715    let id = id_re
716      .captures(attrs)
717      .map(|c| c[1].to_string())
718      .unwrap_or_default();
719    let width = width_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
720    let height = height_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
721
722    if id.is_empty() || content.trim().is_empty() {
723      continue;
724    }
725
726    let w = width.unwrap_or(100.0);
727    let h = height.unwrap_or(100.0);
728
729    // Build inline SVG: coordinate system has y-flip (TeX origin bottom-left, SVG top-left)
730    let mut svg_content = format!(
731      r#"<svg xmlns="http://www.w3.org/2000/svg" version="1.1" width="{w:.2}" height="{h:.2}" overflow="visible">"#,
732    );
733    svg_content.push_str(&format!(
734      r#"<g transform="translate(0,{h:.2}) scale(1,-1)">"#,
735    ));
736
737    // Convert ltx picture children to SVG elements
738    svg_content.push_str(&convert_picture_children_to_svg(content));
739
740    svg_content.push_str("</g></svg>");
741    fragments.push((id, svg_content));
742  }
743  fragments
744}

746/// Convert LaTeXML picture children (g, line, text, circle, etc.) to SVG elements.
747fn convert_picture_children_to_svg(content: &str) -> String {
748  use std::sync::LazyLock;
749  static G_RE: LazyLock<regex::Regex> =
750    LazyLock::new(|| regex::Regex::new(r#"(?s)<g([^>]*)>(.*?)</g>"#).unwrap());
751  static TRANSFORM_RE: LazyLock<regex::Regex> =
752    LazyLock::new(|| regex::Regex::new(r#"transform="([^"]+)""#).unwrap());
753  static LINE_RE: LazyLock<regex::Regex> =
754    LazyLock::new(|| regex::Regex::new(r#"<line\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
755  static CIRCLE_RE: LazyLock<regex::Regex> =
756    LazyLock::new(|| regex::Regex::new(r#"<circle([^/]*)/?>"#).unwrap());
757  static ELLIPSE_RE: LazyLock<regex::Regex> =
758    LazyLock::new(|| regex::Regex::new(r#"<ellipse([^/]*)/?>"#).unwrap());
759  static RECT_RE: LazyLock<regex::Regex> =
760    LazyLock::new(|| regex::Regex::new(r#"<rect([^/]*)/?>"#).unwrap());
761  static POLYGON_RE: LazyLock<regex::Regex> =
762    LazyLock::new(|| regex::Regex::new(r#"<polygon([^/]*)/?>"#).unwrap());
763  static PATH_RE: LazyLock<regex::Regex> =
764    LazyLock::new(|| regex::Regex::new(r#"<path([^/]*)/?>"#).unwrap());
765  static BEZIER_RE: LazyLock<regex::Regex> =
766    LazyLock::new(|| regex::Regex::new(r#"<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
767  static TEXT_RE: LazyLock<regex::Regex> =
768    LazyLock::new(|| regex::Regex::new(r#"(?s)<text([^>]*)>(.*?)</text>"#).unwrap());
769
770  let mut svg = String::new();
771
772  for g_caps in G_RE.captures_iter(content) {
773    let g_attrs = &g_caps[1];
774    let g_content = &g_caps[2];
775
776    // Extract transform
777    let transform = TRANSFORM_RE.captures(g_attrs).map(|c| c[1].to_string());
778
779    if let Some(t) = &transform {
780      svg.push_str(&format!(r#"<g transform="{t}">"#));
781    } else {
782      svg.push_str("<g>");
783    }
784
785    // <line points="x1,y1 x2,y2" stroke="..." stroke-width="..."/>
786    for line_caps in LINE_RE.captures_iter(g_content) {
787      let points = &line_caps[1];
788      let rest_attrs = &line_caps[2];
789      let coords: Vec<&str> = points.split_whitespace().collect();
790      if coords.len() >= 2 {
791        let p1: Vec<&str> = coords[0].split(',').collect();
792        let p2: Vec<&str> = coords[1].split(',').collect();
793        if p1.len() == 2 && p2.len() == 2 {
794          svg.push_str(&format!(
795            r#"<line x1="{}" y1="{}" x2="{}" y2="{}"{}/>"#,
796            p1[0], p1[1], p2[0], p2[1], rest_attrs
797          ));
798        }
799      }
800    }
801
802    // <circle cx="..." cy="..." r="..." .../>
803    for circle_caps in CIRCLE_RE.captures_iter(g_content) {
804      svg.push_str(&format!("<circle{}/>", &circle_caps[1]));
805    }
806
807    // <ellipse cx="..." cy="..." rx="..." ry="..." .../>
808    for ellipse_caps in ELLIPSE_RE.captures_iter(g_content) {
809      svg.push_str(&format!("<ellipse{}/>", &ellipse_caps[1]));
810    }
811
812    // <rect x="..." y="..." width="..." height="..." .../>
813    for rect_caps in RECT_RE.captures_iter(g_content) {
814      svg.push_str(&format!("<rect{}/>", &rect_caps[1]));
815    }
816
817    // <polygon points="..." .../>
818    for polygon_caps in POLYGON_RE.captures_iter(g_content) {
819      svg.push_str(&format!("<polygon{}/>", &polygon_caps[1]));
820    }
821
822    // <path d="..." .../>
823    for path_caps in PATH_RE.captures_iter(g_content) {
824      svg.push_str(&format!("<path{}/>", &path_caps[1]));
825    }
826
827    // <bezier points="x1,y1 x2,y2 x3,y3 x4,y4" .../>
828    // Convert to SVG cubic bezier path
829    for bez_caps in BEZIER_RE.captures_iter(g_content) {
830      let points = &bez_caps[1];
831      let rest = &bez_caps[2];
832      let coords: Vec<&str> = points.split_whitespace().collect();
833      if coords.len() >= 4 {
834        // SVG cubic bezier: M x0,y0 C x1,y1 x2,y2 x3,y3
835        let d = format!(
836          "M {} C {} {} {}",
837          coords[0], coords[1], coords[2], coords[3]
838        );
839        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
840      } else if coords.len() >= 3 {
841        // Quadratic bezier: M x0,y0 Q x1,y1 x2,y2
842        let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
843        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
844      }
845    }
846
847    // <arc .../> — arc segments (rarely used, stub for now)
848    // <wedge .../> — filled wedges (rarely used, stub for now)
849
850    // <text>...</text> — wrap in SVG text with y-flip correction
851    for text_caps in TEXT_RE.captures_iter(g_content) {
852      let text_attrs = &text_caps[1];
853      let text_content = &text_caps[2];
854      svg.push_str(&format!(
855        r#"<g transform="scale(1,-1)"><text{text_attrs}>{text_content}</text></g>"#,
856      ));
857    }
858
859    svg.push_str("</g>");
860  }
861
862  // Also handle direct children not inside <g> (e.g. top-level <bezier>, <line>)
863  // These appear directly inside <picture> without a <g> wrapper
864  let direct_bezier_re =
865    regex::Regex::new(r#"(?m)^\s*<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap();
866  for bez_caps in direct_bezier_re.captures_iter(content) {
867    let points = &bez_caps[1];
868    let rest = &bez_caps[2];
869    let coords: Vec<&str> = points.split_whitespace().collect();
870    if coords.len() >= 4 {
871      let d = format!(
872        "M {} C {} {} {}",
873        coords[0], coords[1], coords[2], coords[3]
874      );
875      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
876    } else if coords.len() >= 3 {
877      let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
878      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
879    }
880  }
881
882  svg
883}

885/// Parse a TeX dimension string (e.g. "100.0pt") to pixels.
886fn parse_tex_dim(s: &str) -> Option<f64> {
887  let s = s.trim();
888  if let Some(rest) = s.strip_suffix("pt") {
889    rest.parse::<f64>().ok().map(|v| v * 96.0 / 72.27)
890  } else if let Some(rest) = s.strip_suffix("px") {
891    rest.parse::<f64>().ok()
892  } else {
893    s.parse::<f64>().ok()
894  }
