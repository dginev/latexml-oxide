1//! Post-processing pipeline API.
2//!
3//! Provides a public interface to the LaTeXML post-processing pipeline
4//! (Scan → Bibliography → CrossRef → Graphics → Split → MathML → XSLT → HTML5 fixups).
5//! Used by both the `latexml_oxide` binary and the `cortex_worker` binary.
6
7use latexml_core::{Info, s};
8use latexml_core::telemetry::{self, Phase};
9use latexml_post::document::{PostDocument, PostDocumentOptions};
10use latexml_post::object_db::ObjectDB;
11use latexml_post::processor::Processor;
12use once_cell::sync::Lazy;
13
14// Process-once cached env var (see WISDOM #56 — getenv hot-path race).
15static POST_AUDIT: Lazy<bool> = Lazy::new(|| std::env::var("LATEXML_POST_AUDIT").is_ok());
16
17/// Options for the post-processing pipeline.
18pub struct PostOptions<'a> {
19  pub pmml:                      bool,
20  pub cmml:                      bool,
21  pub keep_xmath:                bool,
22  pub stylesheet:                Option<&'a str>,
23  pub destination:               Option<&'a str>,
24  pub source_directory:          Option<&'a str>,
25  pub nodefaultresources:        bool,
26  pub css_files:                 &'a [String],
27  pub js_files:                  &'a [String],
28  pub noinvisibletimes:          bool,
29  pub mathtex:                   bool,
30  pub navigationtoc:             Option<&'a str>,
31  pub schemadocs:                bool,
32  pub split:                     bool,
33  pub split_xpath:               Option<String>,
34  pub split_naming:              Option<&'a str>,
35  pub xslt_parameters:           &'a [String],
36  /// If > 0, try `inkscape` for PDF graphics smaller than this many KB
37  /// (vector-preservation path). Fall back to ImageMagick `convert` on
38  /// failure or timeout. Tracks upstream brucemiller/LaTeXML#902.
39  pub graphics_svg_threshold_kb: u32,
40  /// Output extraction mode (Perl `LaTeXML::Util::Pack::whatsout`).
41  /// `Document` (default) → serialize the full post-processed
42  /// document; `Fragment` → embeddable HTML snippet via
43  /// `latexml_post::extract::get_embeddable`; `Math` → math subtree
44  /// via `get_math`. Applied to each `PostDocument` in the final
45  /// serialization loop.
46  pub whatsout:                  latexml_post::extract::Whatsout,
47}
48
49/// Run the post-processing pipeline on XML output.
50///
51/// Executes: Scan → MakeBibliography → CrossRef → Graphics → Split → MathML → XSLT → HTML5 fixups.
52pub fn run_post_processing(xml: &str, opts: &PostOptions) -> String {
53  let PostOptions {
54    pmml,
55    cmml,
56    keep_xmath,
57    stylesheet,
58    destination,
59    source_directory,
60    nodefaultresources,
61    css_files,
62    js_files,
63    noinvisibletimes,
64    mathtex,
65    navigationtoc,
66    schemadocs,
67    split,
68    ref split_xpath,
69    split_naming,
70    xslt_parameters,
71    graphics_svg_threshold_kb,
72    whatsout,
73  } = *opts;
74
75  let mut doc_opts = PostDocumentOptions::default();
76  if let Some(dest) = destination {
77    doc_opts.destination = Some(dest.to_string());
78  }
79  if let Some(src_dir) = source_directory {
80    doc_opts.source_directory = Some(src_dir.to_string());
81    let mut sp = doc_opts.searchpaths.take().unwrap_or_default();
82    sp.push(src_dir.to_string());
83    doc_opts.searchpaths = Some(sp);
84  }
85  let audit = *POST_AUDIT;
86  let audit_start = |name: &str| -> Option<(String, std::time::Instant)> {
87    if audit {
88      Some((name.to_string(), std::time::Instant::now()))
89    } else {
90      None
91    }
92  };
93  let audit_end = |started: Option<(String, std::time::Instant)>| {
94    if let Some((name, t0)) = started {
95      let ms = t0.elapsed().as_millis();
96      Info!("audit", "phase", s!("{} took {}ms", name, ms));
97    }
98  };
99
100  telemetry::phase_enter(Phase::PostXmlParse);
101  let t_parse = audit_start("PostDocument::new_from_string");
102  let doc = match PostDocument::new_from_string(xml, doc_opts) {
103    Ok(d) => d,
104    Err(e) => {
105      eprintln!("Post-processing: failed to parse XML: {}", e);
106      telemetry::phase_exit();
107      return xml.to_string();
108    },
109  };
110  audit_end(t_parse);
111  telemetry::phase_exit();
112
113  // SVG extraction reads the pre-post XML before any in-tree mutation;
114  // do it once up-front so the regex-based fragment table is valid for
115  // every split sub-document below.
116  let t_svg = audit_start("SVG extraction");
117  let svg_fragments = extract_svg_fragments(xml);
118  audit_end(t_svg);
119
120  // Perl-faithful pipeline order (latexmlpost L223-242):
121  //   Split → Scan → MakeBibliography → CrossRef → Graphics → ...
122  // Split runs FIRST so each downstream pass sees the per-page
123  // destination. With the previous order (Scan before Split), every
124  // entry's `location` was the root document and CrossRef built every
125  // ref as a within-page anchor — the user-visible TOC links pointed
126  // at `#Ch1` instead of `Ch1.html`.
127  //
128  // Phase 1: Split (only attributes time when --split is on)
129  let mut docs: Vec<PostDocument> = if split {
130    telemetry::phase_enter(Phase::Split);
131    let result = if let Some(ref xpath) = split_xpath {
132      let naming = match split_naming {
133        Some("id") | None => latexml_post::split::SplitNaming::Id,
134        Some("idrelative") => latexml_post::split::SplitNaming::IdRelative,
135        Some("label") => latexml_post::split::SplitNaming::Label,
136        Some("labelrelative") => latexml_post::split::SplitNaming::LabelRelative,
137        Some(other) => {
138          eprintln!("Unknown splitnaming '{}', using 'id'", other);
139          latexml_post::split::SplitNaming::Id
140        },
141      };
142      let mut splitter = latexml_post::split::Split::new(xpath, naming, false);
143      let split_nodes = splitter.to_process(&doc);
144      match splitter.process(doc, split_nodes) {
145        Ok(docs) => {
146          if docs.len() > 1 {
147            eprintln!("Split into {} documents", docs.len());
148          }
149          Ok(docs)
150        },
151        Err(e) => {
152          eprintln!("Post-processing: Split failed: {}", e);
153          Err(())
154        },
155      }
156    } else {
157      Ok(vec![doc])
158    };
159    telemetry::phase_exit();
160    match result {
161      Ok(ds) => ds,
162      Err(()) => return xml.to_string(),
163    }
164  } else {
165    vec![doc]
166  };
167
168  // Helper: run one processor across every doc in a Vec, mirroring
169  // Perl Post.pm:50-65's `foreach $proc { @newdocs = ... }` loop.
170  // Each processor's per-doc `process` may return multiple docs (only
171  // Split actually does, and we've already run Split above), so the
172  // inner result is flattened back into `docs`.
173  fn run_phase<P: Processor + ?Sized>(
174    docs: Vec<PostDocument>,
175    proc: &mut P,
176    label: &'static str,
177  ) -> Result<Vec<PostDocument>, ()> {
178    let mut out: Vec<PostDocument> = Vec::with_capacity(docs.len());
179    for d in docs {
180      let nodes = proc.to_process(&d);
181      if nodes.is_empty() {
182        out.push(d);
183        continue;
184      }
185      match proc.process(d, nodes) {
186        Ok(processed) => out.extend(processed),
187        Err(e) => {
188          eprintln!("Post-processing: {} failed: {}", label, e);
189          return Err(());
190        },
191      }
192    }
193    Ok(out)
194  }
195
196  // Phase 2: Scan — runs on EACH sub-document so its entries register
197  // the per-page `location` and `pageid`. Single shared ObjectDB so
198  // the later CrossRef pass can resolve cross-doc refs.
199  let mut scanner = latexml_post::scan::Scan::new(ObjectDB::new());
200  telemetry::phase_enter(Phase::PostScan);
201  let t_scan = audit_start("Scan");
202  docs = match run_phase(docs, &mut scanner, "Scan") {
203    Ok(d) => d,
204    Err(()) => {
205      telemetry::phase_exit();
206      return xml.to_string();
207    },
208  };
209  audit_end(t_scan);
210  telemetry::phase_exit();
211
212  // Phase 2b: MakeIndex (Perl LaTeXML.pm L466-470)
213  //   Runs BEFORE MakeBibliography in Perl. Populates `<ltx:indexlist>`
214  //   inside `<ltx:index>` placeholders and `<ltx:glossarylist>` inside
215  //   `<ltx:glossary>` placeholders, using the `GLOSSARY:*` / `INDEX:*`
216  //   entries Scan registered in the ObjectDB. Without this pass, glossary
217  //   sections render empty in HTML (witness: tests/structure/glossary.tex).
218  let mut indexer = latexml_post::make_index::MakeIndex::new(scanner.db, false, false);
219  telemetry::phase_enter(Phase::PostScan);
220  let t_idx = audit_start("MakeIndex");
221  docs = match run_phase(docs, &mut indexer, "MakeIndex") {
222    Ok(d) => d,
223    Err(()) => {
224      telemetry::phase_exit();
225      return xml.to_string();
226    },
227  };
228  audit_end(t_idx);
229  telemetry::phase_exit();
230
231  // Phase 3: MakeBibliography
232  let mut bibmaker = latexml_post::make_bibliography::MakeBibliography::new(indexer.db, false);
233  telemetry::phase_enter(Phase::Bibliography);
234  let t_bib = audit_start("MakeBibliography");
235  docs = match run_phase(docs, &mut bibmaker, "MakeBibliography") {
236    Ok(d) => d,
237    Err(()) => {
238      telemetry::phase_exit();
239      return xml.to_string();
240    },
241  };
242  audit_end(t_bib);
243  telemetry::phase_exit();
244
245  // Phase 4: CrossRef
246  let mut crossref = latexml_post::crossref::CrossRef::new(
247    bibmaker.db,
248    latexml_post::crossref::UrlStyle::File,
249    true,
250  );
251  if let Some(navtoc) = navigationtoc {
252    crossref.set_navigation_toc(navtoc);
253  }
254  telemetry::phase_enter(Phase::Crossref);
255  let t_xref = audit_start("CrossRef");
256  docs = match run_phase(docs, &mut crossref, "CrossRef") {
257    Ok(d) => d,
258    Err(()) => {
259      telemetry::phase_exit();
260      return xml.to_string();
261    },
262  };
263  audit_end(t_xref);
264  telemetry::phase_exit();
265
266  // Phase 5: Graphics
267  let mut graphics_proc = latexml_post::graphics::Graphics::new(None, true)
268    .with_svg_threshold_kb(graphics_svg_threshold_kb);
269  telemetry::phase_enter(Phase::Graphics);
270  let t_gfx = audit_start("Graphics");
271  docs = match run_phase(docs, &mut graphics_proc, "Graphics") {
272    Ok(d) => d,
273    Err(()) => {
274      telemetry::phase_exit();
275      return xml.to_string();
276    },
277  };
278  audit_end(t_gfx);
279  telemetry::phase_exit();
280
281  // Phase 3: MathML + XSLT
282  let mut post = latexml_post::Post::new();
283  let mut processors: Vec<Box<dyn Processor>> = Vec::new();
284
285  let intent_literal = xml.contains("package=\"ar5iv");
286
287  if pmml {
288    processors.push(Box::new(
289      latexml_post::mathml::MathML::new_presentation()
290        .with_keep_xmath(keep_xmath)
291        .with_invisible_times(!noinvisibletimes)
292        .with_mathtex(mathtex)
293        .with_intent_literal(intent_literal),
294    ));
295  }
296  if cmml {
297    processors.push(Box::new(
298      latexml_post::mathml::MathML::new_content()
299        .with_keep_xmath(keep_xmath)
300        .with_invisible_times(!noinvisibletimes),
301    ));
302  }
303  if let Some(xsl_path) = stylesheet {
304    let mut searchpaths = vec![".".to_string()];
305    if let Ok(exe) = std::env::current_exe() {
306      if let Some(project_root) = exe
307        .parent()
308        .and_then(|p| p.parent())
309        .and_then(|p| p.parent())
310      {
311        searchpaths.insert(0, project_root.display().to_string());
312      }
313    }
314    let mut xslt_params = rustc_hash::FxHashMap::default();
315
316    // When `--schemadocs` is on, auto-prepend the rustdoc-styled
317    // theme assets so callers don't need to repeat them on every
318    // invocation. Idempotent: skipped if the user has already
319    // listed the same basename via `--css` / `--javascript`.
320    let prepend = |list: &[String], extra: &str| -> Vec<String> {
321      if list.iter().any(|p| {
322        std::path::Path::new(p).file_name().and_then(|s| s.to_str()) == Some(extra)
323      }) {
324        list.to_vec()
325      } else {
326        let mut out = Vec::with_capacity(list.len() + 1);
327        out.push(extra.to_string());
328        out.extend_from_slice(list);
329        out
330      }
331    };
332    let css_effective: Vec<String> = if schemadocs {
333      prepend(css_files, latexml_post::schema_docs::THEME_CSS_BASENAME)
334    } else {
335      css_files.to_vec()
336    };
337    let js_effective: Vec<String> = if schemadocs {
338      prepend(js_files, latexml_post::schema_docs::THEME_JS_BASENAME)
339    } else {
340      js_files.to_vec()
341    };
342    if !css_effective.is_empty() {
343      xslt_params.insert("CSS".to_string(), format!("\"{}\"", css_effective.join("|")));
344    }
345    if !js_effective.is_empty() {
346      xslt_params.insert(
347        "JAVASCRIPT".to_string(),
348        format!("\"{}\"", js_effective.join("|")),
349      );
350    }
351    if let Some(navtoc) = navigationtoc {
352      xslt_params.insert("NAVIGATIONTOC".to_string(), format!("\"{}\"", navtoc));
353    }
354    for param in xslt_parameters {
355      if let Some((key, value)) = param.split_once('=') {
356        xslt_params.insert(key.to_string(), format!("\"{}\"", value));
357      }
358    }
359    match latexml_post::xslt::XSLT::new(
360      xsl_path,
361      xslt_params,
362      nodefaultresources,
363      None,
364      searchpaths,
365    ) {
366      Ok(xslt) => processors.push(Box::new(xslt)),
367      Err(e) => eprintln!("Post-processing: XSLT error: {}", e),
368    }
369  }
370
371  // process_chain attributes per-processor inside latexml_post::Post::
372  // process_chain (MathmlPres / MathmlCont / Xslt). No outer phase wrap
373  // needed; the inner per-processor guards cover their own time.
374  //
375  // Perl-faithful: pass ALL split docs to ProcessChain at once. Perl's
376  // `Post::ProcessChain_internal` (Post.pm L41-67) seeds `@docs = ($doc)`
377  // and lets each processor return a (possibly multi-element) list which
378  // becomes the input to the next processor. We've already produced the
379  // post-Split list above; ProcessChain just needs to fan MathML/XSLT
380  // across it.
381  let is_html_out = stylesheet.is_some_and(|s| s.contains("html"));
382  let t_chain = audit_start("process_chain");
383  let chain_result = post.process_chain(docs, &mut processors);
384  audit_end(t_chain);
385  let results = match chain_result {
386    Ok(r) => r,
387    Err(e) => {
388      eprintln!("Post-processing failed: {}", e);
389      return xml.to_string();
390    },
391  };
392
393  // Serialize + write each doc IMMEDIATELY in the loop (rather than
394  // collecting first then writing). Multi-doc PostDocument cleanup has
395  // a known libxml2 idcache use-after-free at process exit (see the
396  // SVG-processor note above this block) that can intermittently SIGSEGV
397  // after the last doc has been serialized. Writing eagerly inside the
398  // iteration means even if cleanup later trips, every page that finished
399  // post-processing is already on disk. The first doc's content is
400  // returned so the caller can also write it to --dest in non-split mode.
401  let mut main_output: Option<String> = None;
402  let n = results.len();
403  for doc in results.into_iter() {
404    let dest = doc.get_destination().map(String::from);
405    let t_serialize = audit_start("to_xml_string");
406    // `serialize_whatsout` is a no-op (returns full doc) for the
407    // default `Whatsout::Document`; for Fragment / Math it returns
408    // the extracted subtree serialized via libxml `node_to_string`.
409    let output = latexml_post::extract::serialize_whatsout(&doc, whatsout);
410    audit_end(t_serialize);
411    let output = if is_html_out {
412      finalize_html5(output, &svg_fragments)
413    } else {
414      output
415    };
416    let output = if schemadocs && is_html_out {
417      latexml_post::schema_docs::process_page(&output)
418    } else {
419      output
420    };
421    // Write each doc to disk inside the loop. The single-file caller
422    // also writes the first doc's content to --dest; the redundant write
423    // is harmless and ensures the file is on disk even if a later libxml2
424    // cleanup tripwire fires.
425    if let Some(path) = dest.as_deref() {
426      if let Some(parent) = std::path::Path::new(path).parent() {
427        if !parent.as_os_str().is_empty() {
428          let _ = std::fs::create_dir_all(parent);
429        }
430      }
431      if let Err(e) = std::fs::write(path, &output) {
432        eprintln!("Post-processing: failed to write page {}: {}", path, e);
433      }
434    }
435    if main_output.is_none() {
436      main_output = Some(output);
437    }
438    let _ = n;
439  }
440
441  main_output.unwrap_or_else(|| xml.to_string())
442}

444/// Apply HTML5 cleanup (XML prolog strip, void-element fixes) and inject SVG
445/// fragments into empty `ltx_picture` spans. Pulled out of `run_post_processing`
446/// so it can run on every split sub-document, not just the first.
447fn finalize_html5(output: String, svg_fragments: &[(String, String)]) -> String {
448  use std::sync::LazyLock;
449  // Cached at first call — regex compile is the slow part of `Regex::new`,
450  // and finalize_html5 runs on every (sub-)document in the post-pipeline.
451  static XML_PROLOG_RE: LazyLock<regex::Regex> =
452    LazyLock::new(|| regex::Regex::new(r"^<\?xml[^?]*\?>\s*").unwrap());
453  static NON_VOID_SELF_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
454    regex::Regex::new(
455      r"<(span|div|p|a|td|th|tr|section|article|figure|figcaption|pre|code|em|strong|b|i|u|sub|sup|small|cite)(\s[^>]*)?/>",
456    ).unwrap()
457  });
458  static VOID_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
459    regex::Regex::new(
460      r"</(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)>",
461    ).unwrap()
462  });
463  static VOID_SELF_CLOSE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
464    regex::Regex::new(
465      r"<(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)(\s[^>]*?)\s*/>",
466    ).unwrap()
467  });
468
469  let _gp_html5 = telemetry::phase(Phase::Html5Fixups);
470  // Strip <?xml version...?> prolog: HTML5 must NOT have an XML declaration.
471  // libxml2's to_string() includes it by default; we strip it here.
472  let output = XML_PROLOG_RE.replace(&output, "").to_string();
473  let output = NON_VOID_SELF_CLOSE_RE.replace_all(&output, "<$1$2></$1>").to_string();
474  let output = VOID_CLOSE_RE.replace_all(&output, "").to_string();
475  let mut output = VOID_SELF_CLOSE_RE.replace_all(&output, "<$1$2>").to_string();
476  if !svg_fragments.is_empty() {
477    for (pic_id, svg_html) in svg_fragments {
478      let pattern = format!(
479        r#"<span id="{}" class="ltx_picture"([^>]*)></span>"#,
480        regex::escape(pic_id)
481      );
482      if let Ok(re) = regex::Regex::new(&pattern) {
483        output = re
484          .replace(&output, |caps: &regex::Captures| {
485            format!(
486              r#"<span id="{}" class="ltx_picture"{}>{}</span>"#,
487              pic_id, &caps[1], svg_html
488            )
489          })
490          .to_string();
491      }
492    }
493  }
494  output
495}

497/// Extract SVG fragments from intermediate LaTeXML XML.
498///
499/// Finds `<picture>` elements, converts their children to inline SVG HTML.
500/// Uses a lightweight regex+string approach (no libxml2) to avoid the
501/// use-after-free crash in PostDocument cleanup.
502///
503/// Returns (picture_id, svg_html) pairs for post-XSLT injection.
504fn extract_svg_fragments(xml: &str) -> Vec<(String, String)> {
505  use std::sync::LazyLock;
506  // Fast-fail: most documents have no `<picture>` elements (tikz / pgf
507  // is uncommon in the canvas). Skip the backtracking lazy-match
508  // regex (`(?s)...(.*?)`) when `<picture` doesn't appear as a literal
509  // substring. `str::contains` is a SIMD-accelerated byte search and
510  // takes microseconds even on ~MB inputs.
511  if !xml.contains("<picture") {
512    return Vec::new();
513  }
514  static PICTURE_RE: LazyLock<regex::Regex> = LazyLock::new(|| {
515    regex::Regex::new(r#"(?s)<picture([^>]*)>(.*?)</picture>"#).unwrap()
516  });
517  static ID_RE: LazyLock<regex::Regex> =
518    LazyLock::new(|| regex::Regex::new(r#"xml:id="([^"]+)""#).unwrap());
519  static WIDTH_RE: LazyLock<regex::Regex> =
520    LazyLock::new(|| regex::Regex::new(r#"width="([^"]+)""#).unwrap());
521  static HEIGHT_RE: LazyLock<regex::Regex> =
522    LazyLock::new(|| regex::Regex::new(r#"height="([^"]+)""#).unwrap());
523  let mut fragments = Vec::new();
524  let picture_re = &*PICTURE_RE;
525  let id_re = &*ID_RE;
526  let width_re = &*WIDTH_RE;
527  let height_re = &*HEIGHT_RE;
528
529  for pic_caps in picture_re.captures_iter(xml) {
530    let attrs = &pic_caps[1];
531    let content = &pic_caps[2];
532    let id = id_re
533      .captures(attrs)
534      .map(|c| c[1].to_string())
535      .unwrap_or_default();
536    let width = width_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
537    let height = height_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
538
539    if id.is_empty() || content.trim().is_empty() {
540      continue;
541    }
542
543    let w = width.unwrap_or(100.0);
544    let h = height.unwrap_or(100.0);
545
546    // Build inline SVG: coordinate system has y-flip (TeX origin bottom-left, SVG top-left)
547    let mut svg_content = format!(
548      r#"<svg xmlns="http://www.w3.org/2000/svg" version="1.1" width="{w:.2}" height="{h:.2}" overflow="visible">"#,
549    );
550    svg_content.push_str(&format!(
551      r#"<g transform="translate(0,{h:.2}) scale(1,-1)">"#,
552    ));
553
554    // Convert ltx picture children to SVG elements
555    svg_content.push_str(&convert_picture_children_to_svg(content));
556
557    svg_content.push_str("</g></svg>");
558    fragments.push((id, svg_content));
559  }
560  fragments
561}

563/// Convert LaTeXML picture children (g, line, text, circle, etc.) to SVG elements.
564fn convert_picture_children_to_svg(content: &str) -> String {
565  use std::sync::LazyLock;
566  static G_RE: LazyLock<regex::Regex> =
567    LazyLock::new(|| regex::Regex::new(r#"(?s)<g([^>]*)>(.*?)</g>"#).unwrap());
568  static TRANSFORM_RE: LazyLock<regex::Regex> =
569    LazyLock::new(|| regex::Regex::new(r#"transform="([^"]+)""#).unwrap());
570  static LINE_RE: LazyLock<regex::Regex> =
571    LazyLock::new(|| regex::Regex::new(r#"<line\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
572  static CIRCLE_RE: LazyLock<regex::Regex> =
573    LazyLock::new(|| regex::Regex::new(r#"<circle([^/]*)/?>"#).unwrap());
574  static ELLIPSE_RE: LazyLock<regex::Regex> =
575    LazyLock::new(|| regex::Regex::new(r#"<ellipse([^/]*)/?>"#).unwrap());
576  static RECT_RE: LazyLock<regex::Regex> =
577    LazyLock::new(|| regex::Regex::new(r#"<rect([^/]*)/?>"#).unwrap());
578  static POLYGON_RE: LazyLock<regex::Regex> =
579    LazyLock::new(|| regex::Regex::new(r#"<polygon([^/]*)/?>"#).unwrap());
580  static PATH_RE: LazyLock<regex::Regex> =
581    LazyLock::new(|| regex::Regex::new(r#"<path([^/]*)/?>"#).unwrap());
582  static BEZIER_RE: LazyLock<regex::Regex> =
583    LazyLock::new(|| regex::Regex::new(r#"<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
584  static TEXT_RE: LazyLock<regex::Regex> =
585    LazyLock::new(|| regex::Regex::new(r#"(?s)<text([^>]*)>(.*?)</text>"#).unwrap());
586
587  let mut svg = String::new();
588
589  for g_caps in G_RE.captures_iter(content) {
590    let g_attrs = &g_caps[1];
591    let g_content = &g_caps[2];
592
593    // Extract transform
594    let transform = TRANSFORM_RE.captures(g_attrs).map(|c| c[1].to_string());
595
596    if let Some(t) = &transform {
597      svg.push_str(&format!(r#"<g transform="{t}">"#));
598    } else {
599      svg.push_str("<g>");
600    }
601
602    // <line points="x1,y1 x2,y2" stroke="..." stroke-width="..."/>
603    for line_caps in LINE_RE.captures_iter(g_content) {
604      let points = &line_caps[1];
605      let rest_attrs = &line_caps[2];
606      let coords: Vec<&str> = points.split_whitespace().collect();
607      if coords.len() >= 2 {
608        let p1: Vec<&str> = coords[0].split(',').collect();
609        let p2: Vec<&str> = coords[1].split(',').collect();
610        if p1.len() == 2 && p2.len() == 2 {
611          svg.push_str(&format!(
612            r#"<line x1="{}" y1="{}" x2="{}" y2="{}"{}/>"#,
613            p1[0], p1[1], p2[0], p2[1], rest_attrs
614          ));
615        }
616      }
617    }
618
619    // <circle cx="..." cy="..." r="..." .../>
620    for circle_caps in CIRCLE_RE.captures_iter(g_content) {
621      svg.push_str(&format!("<circle{}/>", &circle_caps[1]));
622    }
623
624    // <ellipse cx="..." cy="..." rx="..." ry="..." .../>
625    for ellipse_caps in ELLIPSE_RE.captures_iter(g_content) {
626      svg.push_str(&format!("<ellipse{}/>", &ellipse_caps[1]));
627    }
628
629    // <rect x="..." y="..." width="..." height="..." .../>
630    for rect_caps in RECT_RE.captures_iter(g_content) {
631      svg.push_str(&format!("<rect{}/>", &rect_caps[1]));
632    }
633
634    // <polygon points="..." .../>
635    for polygon_caps in POLYGON_RE.captures_iter(g_content) {
636      svg.push_str(&format!("<polygon{}/>", &polygon_caps[1]));
637    }
638
639    // <path d="..." .../>
640    for path_caps in PATH_RE.captures_iter(g_content) {
641      svg.push_str(&format!("<path{}/>", &path_caps[1]));
642    }
643
644    // <bezier points="x1,y1 x2,y2 x3,y3 x4,y4" .../>
645    // Convert to SVG cubic bezier path
646    for bez_caps in BEZIER_RE.captures_iter(g_content) {
647      let points = &bez_caps[1];
648      let rest = &bez_caps[2];
649      let coords: Vec<&str> = points.split_whitespace().collect();
650      if coords.len() >= 4 {
651        // SVG cubic bezier: M x0,y0 C x1,y1 x2,y2 x3,y3
652        let d = format!(
653          "M {} C {} {} {}",
654          coords[0], coords[1], coords[2], coords[3]
655        );
656        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
657      } else if coords.len() >= 3 {
658        // Quadratic bezier: M x0,y0 Q x1,y1 x2,y2
659        let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
660        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
661      }
662    }
663
664    // <arc .../> — arc segments (rarely used, stub for now)
665    // <wedge .../> — filled wedges (rarely used, stub for now)
666
667    // <text>...</text> — wrap in SVG text with y-flip correction
668    for text_caps in TEXT_RE.captures_iter(g_content) {
669      let text_attrs = &text_caps[1];
670      let text_content = &text_caps[2];
671      svg.push_str(&format!(
672        r#"<g transform="scale(1,-1)"><text{text_attrs}>{text_content}</text></g>"#,
673      ));
674    }
675
676    svg.push_str("</g>");
677  }
678
679  // Also handle direct children not inside <g> (e.g. top-level <bezier>, <line>)
680  // These appear directly inside <picture> without a <g> wrapper
681  let direct_bezier_re =
682    regex::Regex::new(r#"(?m)^\s*<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap();
683  for bez_caps in direct_bezier_re.captures_iter(content) {
684    let points = &bez_caps[1];
685    let rest = &bez_caps[2];
686    let coords: Vec<&str> = points.split_whitespace().collect();
687    if coords.len() >= 4 {
688      let d = format!(
689        "M {} C {} {} {}",
690        coords[0], coords[1], coords[2], coords[3]
691      );
692      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
693    } else if coords.len() >= 3 {
694      let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
695      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
696    }
697  }
698
699  svg
700}

702/// Parse a TeX dimension string (e.g. "100.0pt") to pixels.
703fn parse_tex_dim(s: &str) -> Option<f64> {
704  let s = s.trim();
705  if let Some(rest) = s.strip_suffix("pt") {
706    rest.parse::<f64>().ok().map(|v| v * 96.0 / 72.27)
707  } else if let Some(rest) = s.strip_suffix("px") {
708    rest.parse::<f64>().ok()
709  } else {
710    s.parse::<f64>().ok()
711  }
712}
