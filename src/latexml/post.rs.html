1//! Post-processing pipeline API.
2//!
3//! Provides a public interface to the LaTeXML post-processing pipeline
4//! (Scan → Bibliography → CrossRef → Graphics → Split → MathML → XSLT → HTML5 fixups).
5//! Used by both the `latexml_oxide` binary and the `cortex_worker` binary.
6
7use latexml_post::document::{PostDocument, PostDocumentOptions};
8use latexml_post::object_db::ObjectDB;
9use latexml_post::processor::Processor;
10use once_cell::sync::Lazy;
11
12// Process-once cached env var (see WISDOM #56 — getenv hot-path race).
13static POST_AUDIT: Lazy<bool> = Lazy::new(|| std::env::var("LATEXML_POST_AUDIT").is_ok());
14
15/// Options for the post-processing pipeline.
16pub struct PostOptions<'a> {
17  pub pmml:                      bool,
18  pub cmml:                      bool,
19  pub keep_xmath:                bool,
20  pub stylesheet:                Option<&'a str>,
21  pub destination:               Option<&'a str>,
22  pub source_directory:          Option<&'a str>,
23  pub nodefaultresources:        bool,
24  pub css_files:                 &'a [String],
25  pub js_files:                  &'a [String],
26  pub noinvisibletimes:          bool,
27  pub mathtex:                   bool,
28  pub navigationtoc:             Option<&'a str>,
29  pub split:                     bool,
30  pub split_xpath:               Option<String>,
31  pub split_naming:              Option<&'a str>,
32  pub xslt_parameters:           &'a [String],
33  /// If > 0, try `inkscape` for PDF graphics smaller than this many KB
34  /// (vector-preservation path). Fall back to ImageMagick `convert` on
35  /// failure or timeout. Tracks upstream brucemiller/LaTeXML#902.
36  pub graphics_svg_threshold_kb: u32,
37}
38
39/// Run the post-processing pipeline on XML output.
40///
41/// Executes: Scan → MakeBibliography → CrossRef → Graphics → Split → MathML → XSLT → HTML5 fixups.
42pub fn run_post_processing(xml: &str, opts: &PostOptions) -> String {
43  let PostOptions {
44    pmml,
45    cmml,
46    keep_xmath,
47    stylesheet,
48    destination,
49    source_directory,
50    nodefaultresources,
51    css_files,
52    js_files,
53    noinvisibletimes,
54    mathtex,
55    navigationtoc,
56    split,
57    ref split_xpath,
58    split_naming,
59    xslt_parameters,
60    graphics_svg_threshold_kb,
61  } = *opts;
62
63  let mut doc_opts = PostDocumentOptions::default();
64  if let Some(dest) = destination {
65    doc_opts.destination = Some(dest.to_string());
66  }
67  if let Some(src_dir) = source_directory {
68    doc_opts.source_directory = Some(src_dir.to_string());
69    let mut sp = doc_opts.searchpaths.take().unwrap_or_default();
70    sp.push(src_dir.to_string());
71    doc_opts.searchpaths = Some(sp);
72  }
73  let audit = *POST_AUDIT;
74  let audit_start = |name: &str| -> Option<(String, std::time::Instant)> {
75    if audit {
76      Some((name.to_string(), std::time::Instant::now()))
77    } else {
78      None
79    }
80  };
81  let audit_end = |started: Option<(String, std::time::Instant)>| {
82    if let Some((name, t0)) = started {
83      let ms = t0.elapsed().as_millis();
84      log::info!("POST_AUDIT phase {} took {}ms", name, ms);
85    }
86  };
87
88  let t_parse = audit_start("PostDocument::new_from_string");
89  let doc = match PostDocument::new_from_string(xml, doc_opts) {
90    Ok(d) => d,
91    Err(e) => {
92      eprintln!("Post-processing: failed to parse XML: {}", e);
93      return xml.to_string();
94    },
95  };
96  audit_end(t_parse);
97
98  // Phase 1: Scan
99  let t_scan = audit_start("Scan");
100  let db = ObjectDB::new();
101  let mut scanner = latexml_post::scan::Scan::new(db);
102  let scan_nodes = scanner.to_process(&doc);
103  let doc = match scanner.process(doc, scan_nodes) {
104    Ok(mut docs) => docs.remove(0),
105    Err(e) => {
106      eprintln!("Post-processing: Scan failed: {}", e);
107      return xml.to_string();
108    },
109  };
110  audit_end(t_scan);
111
112  // Phase 1.5: MakeBibliography
113  let t_bib = audit_start("MakeBibliography");
114  let db = scanner.db;
115  let mut bibmaker = latexml_post::make_bibliography::MakeBibliography::new(db, false);
116  let bib_nodes = bibmaker.to_process(&doc);
117  let doc = if !bib_nodes.is_empty() {
118    match bibmaker.process(doc, bib_nodes) {
119      Ok(mut docs) => docs.remove(0),
120      Err(e) => {
121        eprintln!("Post-processing: MakeBibliography failed: {}", e);
122        return xml.to_string();
123      },
124    }
125  } else {
126    doc
127  };
128  audit_end(t_bib);
129
130  // Phase 2: CrossRef
131  let t_xref = audit_start("CrossRef");
132  let db = bibmaker.db;
133  let mut crossref =
134    latexml_post::crossref::CrossRef::new(db, latexml_post::crossref::UrlStyle::File, true);
135  let xref_nodes = crossref.to_process(&doc);
136  let doc = match crossref.process(doc, xref_nodes) {
137    Ok(mut docs) => docs.remove(0),
138    Err(e) => {
139      eprintln!("Post-processing: CrossRef failed: {}", e);
140      return xml.to_string();
141    },
142  };
143  audit_end(t_xref);
144
145  // Phase 2.5: Graphics
146  let t_gfx = audit_start("Graphics");
147  let mut graphics_proc = latexml_post::graphics::Graphics::new(None, true)
148    .with_svg_threshold_kb(graphics_svg_threshold_kb);
149  let graphics_nodes = graphics_proc.to_process(&doc);
150  let doc = if !graphics_nodes.is_empty() {
151    match graphics_proc.process(doc, graphics_nodes) {
152      Ok(mut docs) => docs.remove(0),
153      Err(e) => {
154        eprintln!("Post-processing: Graphics failed: {}", e);
155        return xml.to_string();
156      },
157    }
158  } else {
159    doc
160  };
161  audit_end(t_gfx);
162
163  // Phase 2.6: SVG (convert ltx:picture children to svg:svg + svg:* elements)
164  // Without this, the XSLT picture template falls back to "as-TeX" mode and
165  // emits an empty span — figures with picture-environment content are lost.
166  // Phase 2.6: SVG
167  // Convert ltx:picture elements to inline SVG for HTML output.
168  //
169  // The latexml_post::svg::SVG processor works correctly but causes a
170  // use-after-free crash in libxml2 during PostDocument cleanup (nodes
171  // unlinked by replace_node are freed but still referenced in the idcache).
172  //
173  // Workaround: extract SVG fragments from the INTERMEDIATE XML using
174  // string processing (no libxml2 involvement), then inject them into
175  // the final HTML AFTER XSLT completes.
176  let t_svg = audit_start("SVG extraction");
177  let svg_fragments = extract_svg_fragments(xml);
178  audit_end(t_svg);
179
180  // Phase 2.75: Split
181  let doc = if split {
182    if let Some(ref xpath) = split_xpath {
183      let naming = match split_naming {
184        Some("id") | None => latexml_post::split::SplitNaming::Id,
185        Some("idrelative") => latexml_post::split::SplitNaming::IdRelative,
186        Some("label") => latexml_post::split::SplitNaming::Label,
187        Some("labelrelative") => latexml_post::split::SplitNaming::LabelRelative,
188        Some(other) => {
189          eprintln!("Unknown splitnaming '{}', using 'id'", other);
190          latexml_post::split::SplitNaming::Id
191        },
192      };
193      let mut splitter = latexml_post::split::Split::new(xpath, naming, false);
194      let split_nodes = splitter.to_process(&doc);
195      match splitter.process(doc, split_nodes) {
196        Ok(mut docs) => {
197          if docs.len() > 1 {
198            eprintln!("Split into {} documents", docs.len());
199          }
200          docs.remove(0)
201        },
202        Err(e) => {
203          eprintln!("Post-processing: Split failed: {}", e);
204          return xml.to_string();
205        },
206      }
207    } else {
208      doc
209    }
210  } else {
211    doc
212  };
213
214  // Phase 3: MathML + XSLT
215  let mut post = latexml_post::Post::new();
216  let mut processors: Vec<Box<dyn Processor>> = Vec::new();
217
218  let intent_literal = xml.contains("package=\"ar5iv");
219
220  if pmml {
221    processors.push(Box::new(
222      latexml_post::mathml::MathML::new_presentation()
223        .with_keep_xmath(keep_xmath)
224        .with_invisible_times(!noinvisibletimes)
225        .with_mathtex(mathtex)
226        .with_intent_literal(intent_literal),
227    ));
228  }
229  if cmml {
230    processors.push(Box::new(
231      latexml_post::mathml::MathML::new_content()
232        .with_keep_xmath(keep_xmath)
233        .with_invisible_times(!noinvisibletimes),
234    ));
235  }
236  if let Some(xsl_path) = stylesheet {
237    let mut searchpaths = vec![".".to_string()];
238    if let Ok(exe) = std::env::current_exe() {
239      if let Some(project_root) = exe
240        .parent()
241        .and_then(|p| p.parent())
242        .and_then(|p| p.parent())
243      {
244        searchpaths.insert(0, project_root.display().to_string());
245      }
246    }
247    let mut xslt_params = std::collections::HashMap::new();
248    if !css_files.is_empty() {
249      xslt_params.insert("CSS".to_string(), format!("\"{}\"", css_files.join("|")));
250    }
251    if !js_files.is_empty() {
252      xslt_params.insert(
253        "JAVASCRIPT".to_string(),
254        format!("\"{}\"", js_files.join("|")),
255      );
256    }
257    if let Some(navtoc) = navigationtoc {
258      xslt_params.insert("NAVIGATIONTOC".to_string(), format!("\"{}\"", navtoc));
259    }
260    for param in xslt_parameters {
261      if let Some((key, value)) = param.split_once('=') {
262        xslt_params.insert(key.to_string(), format!("\"{}\"", value));
263      }
264    }
265    match latexml_post::xslt::XSLT::new(
266      xsl_path,
267      xslt_params,
268      nodefaultresources,
269      None,
270      searchpaths,
271    ) {
272      Ok(xslt) => processors.push(Box::new(xslt)),
273      Err(e) => eprintln!("Post-processing: XSLT error: {}", e),
274    }
275  }
276
277  let t_chain = audit_start("process_chain");
278  let chain_result = post.process_chain(doc, &mut processors);
279  audit_end(t_chain);
280  match chain_result {
281    Ok(results) => {
282      let t_serialize = audit_start("to_xml_string");
283      let output = results[0].to_xml_string();
284      audit_end(t_serialize);
285      if stylesheet.is_some_and(|s| s.contains("html")) {
286        // Strip <?xml version...?> prolog: HTML5 must NOT have an XML declaration.
287        // libxml2's to_string() includes it by default; we strip it here.
288        let output = regex::Regex::new(r"^<\?xml[^?]*\?>\s*")
289          .unwrap()
290          .replace(&output, "")
291          .to_string();
292        let re = regex::Regex::new(
293          r"<(span|div|p|a|td|th|tr|section|article|figure|figcaption|pre|code|em|strong|b|i|u|sub|sup|small|cite)(\s[^>]*)?/>",
294        )
295        .unwrap();
296        let output = re.replace_all(&output, "<$1$2></$1>").to_string();
297        let void_close_re = regex::Regex::new(
298          r"</(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)>",
299        )
300        .unwrap();
301        let output = void_close_re.replace_all(&output, "").to_string();
302        let void_selfclose_re = regex::Regex::new(
303          r"<(br|img|hr|input|meta|link|col|area|base|source|track|wbr|embed|param)(\s[^>]*?)\s*/>",
304        )
305        .unwrap();
306        let mut output = void_selfclose_re.replace_all(&output, "<$1$2>").to_string();
307        // Phase G: inject SVG fragments into empty ltx_picture spans
308        if !svg_fragments.is_empty() {
309          for (pic_id, svg_html) in &svg_fragments {
310            // Replace <span id="ID" class="ltx_picture" style="..."></span>
311            // with <span id="ID" class="ltx_picture" style="...">SVG_CONTENT</span>
312            let pattern = format!(
313              r#"<span id="{}" class="ltx_picture"([^>]*)></span>"#,
314              regex::escape(pic_id)
315            );
316            if let Ok(re) = regex::Regex::new(&pattern) {
317              output = re
318                .replace(&output, |caps: &regex::Captures| {
319                  format!(
320                    r#"<span id="{}" class="ltx_picture"{}>{}</span>"#,
321                    pic_id, &caps[1], svg_html
322                  )
323                })
324                .to_string();
325            }
326          }
327        }
328        output
329      } else {
330        output
331      }
332    },
333    Err(e) => {
334      eprintln!("Post-processing failed: {}", e);
335      xml.to_string()
336    },
337  }
338}

340/// Extract SVG fragments from intermediate LaTeXML XML.
341///
342/// Finds `<picture>` elements, converts their children to inline SVG HTML.
343/// Uses a lightweight regex+string approach (no libxml2) to avoid the
344/// use-after-free crash in PostDocument cleanup.
345///
346/// Returns (picture_id, svg_html) pairs for post-XSLT injection.
347fn extract_svg_fragments(xml: &str) -> Vec<(String, String)> {
348  let mut fragments = Vec::new();
349  // Match <picture ... xml:id="ID" ... width="W" height="H" ...>CONTENT</picture>
350  let picture_re = regex::Regex::new(r#"(?s)<picture([^>]*)>(.*?)</picture>"#).unwrap();
351  let id_re = regex::Regex::new(r#"xml:id="([^"]+)""#).unwrap();
352  let width_re = regex::Regex::new(r#"width="([^"]+)""#).unwrap();
353  let height_re = regex::Regex::new(r#"height="([^"]+)""#).unwrap();
354
355  for pic_caps in picture_re.captures_iter(xml) {
356    let attrs = &pic_caps[1];
357    let content = &pic_caps[2];
358    let id = id_re
359      .captures(attrs)
360      .map(|c| c[1].to_string())
361      .unwrap_or_default();
362    let width = width_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
363    let height = height_re.captures(attrs).and_then(|c| parse_tex_dim(&c[1]));
364
365    if id.is_empty() || content.trim().is_empty() {
366      continue;
367    }
368
369    let w = width.unwrap_or(100.0);
370    let h = height.unwrap_or(100.0);
371
372    // Build inline SVG: coordinate system has y-flip (TeX origin bottom-left, SVG top-left)
373    let mut svg_content = format!(
374      r#"<svg xmlns="http://www.w3.org/2000/svg" version="1.1" width="{w:.2}" height="{h:.2}" overflow="visible">"#,
375    );
376    svg_content.push_str(&format!(
377      r#"<g transform="translate(0,{h:.2}) scale(1,-1)">"#,
378    ));
379
380    // Convert ltx picture children to SVG elements
381    svg_content.push_str(&convert_picture_children_to_svg(content));
382
383    svg_content.push_str("</g></svg>");
384    fragments.push((id, svg_content));
385  }
386  fragments
387}

389/// Convert LaTeXML picture children (g, line, text, circle, etc.) to SVG elements.
390fn convert_picture_children_to_svg(content: &str) -> String {
391  use std::sync::LazyLock;
392  static G_RE: LazyLock<regex::Regex> =
393    LazyLock::new(|| regex::Regex::new(r#"(?s)<g([^>]*)>(.*?)</g>"#).unwrap());
394  static TRANSFORM_RE: LazyLock<regex::Regex> =
395    LazyLock::new(|| regex::Regex::new(r#"transform="([^"]+)""#).unwrap());
396  static LINE_RE: LazyLock<regex::Regex> =
397    LazyLock::new(|| regex::Regex::new(r#"<line\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
398  static CIRCLE_RE: LazyLock<regex::Regex> =
399    LazyLock::new(|| regex::Regex::new(r#"<circle([^/]*)/?>"#).unwrap());
400  static ELLIPSE_RE: LazyLock<regex::Regex> =
401    LazyLock::new(|| regex::Regex::new(r#"<ellipse([^/]*)/?>"#).unwrap());
402  static RECT_RE: LazyLock<regex::Regex> =
403    LazyLock::new(|| regex::Regex::new(r#"<rect([^/]*)/?>"#).unwrap());
404  static POLYGON_RE: LazyLock<regex::Regex> =
405    LazyLock::new(|| regex::Regex::new(r#"<polygon([^/]*)/?>"#).unwrap());
406  static PATH_RE: LazyLock<regex::Regex> =
407    LazyLock::new(|| regex::Regex::new(r#"<path([^/]*)/?>"#).unwrap());
408  static BEZIER_RE: LazyLock<regex::Regex> =
409    LazyLock::new(|| regex::Regex::new(r#"<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap());
410  static TEXT_RE: LazyLock<regex::Regex> =
411    LazyLock::new(|| regex::Regex::new(r#"(?s)<text([^>]*)>(.*?)</text>"#).unwrap());
412
413  let mut svg = String::new();
414
415  for g_caps in G_RE.captures_iter(content) {
416    let g_attrs = &g_caps[1];
417    let g_content = &g_caps[2];
418
419    // Extract transform
420    let transform = TRANSFORM_RE.captures(g_attrs).map(|c| c[1].to_string());
421
422    if let Some(t) = &transform {
423      svg.push_str(&format!(r#"<g transform="{t}">"#));
424    } else {
425      svg.push_str("<g>");
426    }
427
428    // <line points="x1,y1 x2,y2" stroke="..." stroke-width="..."/>
429    for line_caps in LINE_RE.captures_iter(g_content) {
430      let points = &line_caps[1];
431      let rest_attrs = &line_caps[2];
432      let coords: Vec<&str> = points.split_whitespace().collect();
433      if coords.len() >= 2 {
434        let p1: Vec<&str> = coords[0].split(',').collect();
435        let p2: Vec<&str> = coords[1].split(',').collect();
436        if p1.len() == 2 && p2.len() == 2 {
437          svg.push_str(&format!(
438            r#"<line x1="{}" y1="{}" x2="{}" y2="{}"{}/>"#,
439            p1[0], p1[1], p2[0], p2[1], rest_attrs
440          ));
441        }
442      }
443    }
444
445    // <circle cx="..." cy="..." r="..." .../>
446    for circle_caps in CIRCLE_RE.captures_iter(g_content) {
447      svg.push_str(&format!("<circle{}/>", &circle_caps[1]));
448    }
449
450    // <ellipse cx="..." cy="..." rx="..." ry="..." .../>
451    for ellipse_caps in ELLIPSE_RE.captures_iter(g_content) {
452      svg.push_str(&format!("<ellipse{}/>", &ellipse_caps[1]));
453    }
454
455    // <rect x="..." y="..." width="..." height="..." .../>
456    for rect_caps in RECT_RE.captures_iter(g_content) {
457      svg.push_str(&format!("<rect{}/>", &rect_caps[1]));
458    }
459
460    // <polygon points="..." .../>
461    for polygon_caps in POLYGON_RE.captures_iter(g_content) {
462      svg.push_str(&format!("<polygon{}/>", &polygon_caps[1]));
463    }
464
465    // <path d="..." .../>
466    for path_caps in PATH_RE.captures_iter(g_content) {
467      svg.push_str(&format!("<path{}/>", &path_caps[1]));
468    }
469
470    // <bezier points="x1,y1 x2,y2 x3,y3 x4,y4" .../>
471    // Convert to SVG cubic bezier path
472    for bez_caps in BEZIER_RE.captures_iter(g_content) {
473      let points = &bez_caps[1];
474      let rest = &bez_caps[2];
475      let coords: Vec<&str> = points.split_whitespace().collect();
476      if coords.len() >= 4 {
477        // SVG cubic bezier: M x0,y0 C x1,y1 x2,y2 x3,y3
478        let d = format!(
479          "M {} C {} {} {}",
480          coords[0], coords[1], coords[2], coords[3]
481        );
482        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
483      } else if coords.len() >= 3 {
484        // Quadratic bezier: M x0,y0 Q x1,y1 x2,y2
485        let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
486        svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
487      }
488    }
489
490    // <arc .../> — arc segments (rarely used, stub for now)
491    // <wedge .../> — filled wedges (rarely used, stub for now)
492
493    // <text>...</text> — wrap in SVG text with y-flip correction
494    for text_caps in TEXT_RE.captures_iter(g_content) {
495      let text_attrs = &text_caps[1];
496      let text_content = &text_caps[2];
497      svg.push_str(&format!(
498        r#"<g transform="scale(1,-1)"><text{text_attrs}>{text_content}</text></g>"#,
499      ));
500    }
501
502    svg.push_str("</g>");
503  }
504
505  // Also handle direct children not inside <g> (e.g. top-level <bezier>, <line>)
506  // These appear directly inside <picture> without a <g> wrapper
507  let direct_bezier_re =
508    regex::Regex::new(r#"(?m)^\s*<bezier\s+points="([^"]+)"([^/]*)/?>"#).unwrap();
509  for bez_caps in direct_bezier_re.captures_iter(content) {
510    let points = &bez_caps[1];
511    let rest = &bez_caps[2];
512    let coords: Vec<&str> = points.split_whitespace().collect();
513    if coords.len() >= 4 {
514      let d = format!(
515        "M {} C {} {} {}",
516        coords[0], coords[1], coords[2], coords[3]
517      );
518      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
519    } else if coords.len() >= 3 {
520      let d = format!("M {} Q {} {}", coords[0], coords[1], coords[2]);
521      svg.push_str(&format!(r#"<path d="{d}"{rest} fill="none"/>"#));
522    }
523  }
524
525  svg
526}

528/// Parse a TeX dimension string (e.g. "100.0pt") to pixels.
529fn parse_tex_dim(s: &str) -> Option<f64> {
530  let s = s.trim();
531  if let Some(rest) = s.strip_suffix("pt") {
532    rest.parse::<f64>().ok().map(|v| v * 96.0 / 72.27)
533  } else if let Some(rest) = s.strip_suffix("px") {
534    rest.parse::<f64>().ok()
535  } else {
536    s.parse::<f64>().ok()
537  }
538}
