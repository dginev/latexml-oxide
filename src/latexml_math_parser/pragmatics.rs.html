8/// Different supported pragmatics according to which to enforce consistency.
9/// Usually one would use them in increasing generality, iterating over a parse forest
10/// until a sufficiently small number of parses is reached.
11#[derive(Copy, Clone, Debug)]
12pub enum ValidationPragmatics {
13  ConsistentLetterBlocks,
14  ConsistentCase,
15  ConsistentCaseFlat,
16  ConsistentCaseFlatUnstyled,
17  InconsistentNames, /* TODO: Since name consistency is built into the curry levels, using the
18                      * inconsistent names pragma should ensure each LP variable has a globally
19                      * unique name */
20  FencedAtomsAreNotFunctions,
21  FencedLettersAreFunctionArguments,
22  AdjacentNumbersDontMultiply,
23  AdjacentUnfencedScriptsDontApply,
24  AdjacentFunctionsDontUnifyIntoOperator,
25  OpfunctionsAreRarelyArguments,
26  UnfencedLetterArgumentsRequireVisualCues,
27  HigherOrderIDsAreExceptions,
28  HigherOrderInvisibleOpsAreExceptions,
29  StandaloneDiffopsAreNotNumerators,
30  PostfixTermsAreFencedIfSingleArguments,
31  RestrictNumeralFractions,
32  /// Prefer parses where scripts attach to bases rather than floating standalone.
33  MaximizeScriptAttachment,
34  /// An expression should never have `absent` on BOTH the left and right sides
35  /// of a binary/relational operator.
36  NoBilateralAbsent,
37  /// Functions prefer wider absorption: `\log 2x^2` means `log(2*x^2)`,
38  /// not `log(2)*x^2`. When a function can absorb more factors, prefer that parse.
39  /// Implemented by penalizing parses where a function's result is multiplied
40  /// by factors that could have been the function's argument.
41  FunctionsPreferWiderAbsorption,
42  /// Bigops prefer wider absorption: `\int F\times Gdx` means `∫(F×G dx)`,
43  /// not `∫(F)×Gdx`. Perl's moreOpArgFactors absorbs MulOp chains.
44  /// Rejects trees where mulop(bigop_app(narrow), rhs) when rhs is a simple factor.
45  BigopPreferWiderAbsorption,
46  /// Prefer binary ADDOP over unary when it follows a complete term.
47  /// In `-12x^2 - 4xy + 2y`, the interior `-` and `+` should be binary operators,
48  /// not unary prefix applied to the following term. Rejects parses where
49  /// prefix_apply(addop, rhs) appears as a non-initial child of an additive chain.
50  PreferBinaryAddop,
51  /// For invisible-times chains of simple unfenced tokens (e.g., `ppppp`, `xyz`),
52  /// enforce left-associative grouping: `((p·p)·p)·p` not `p·(p·(p·p))`.
53  /// In mathematical practice, consecutive identifiers without explicit operators
54  /// or parentheses form a flat product — associativity doesn't matter semantically,
55  /// and enumerating all groupings causes exponential ambiguity.
56  FlattenSimpleInvisibleTimesChains,
57  /// In `a = b + c + d`, the `=` must be at the outermost level.
58  /// An ADDOP/MULOP cannot have an unfenced RELOP child — that would mean
59  /// treating a relation as a term in an arithmetic expression.
60  /// Exception: fenced relations like `(x=0)` used as modifiers.
61  RelopsAreOutermost,
62}

140  /// Recursively checks a tree at each level for applicable pragmatics.
141  /// All levels must validate for the full tree to be valid.
142  pub fn validate_recursive(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
143    self.validate(tree)?; // top-level check
144    // Recursive check:
145    match tree {
146      XM::Choices(subtrees) => {
147        for subtree in subtrees.iter() {
148          self.validate_recursive(subtree)?
149        }
150      },
151      XM::Apply(Operator(op), args, ..) => {
152        self.validate_recursive(op)?;
153        for arg_subtree in args.trees() {
154          self.validate_recursive(arg_subtree)?;
155        }
156      },
157      XM::Dual(ref content, ref pres, _, _) => {
158        self.validate_recursive(content)?;
159        self.validate_recursive(pres)?;
160      },
161      XM::Wrap(ref items, _, _) => {
162        for item in items.iter() {
163          self.validate_recursive(item)?;
164        }
165      },
166      _ => {},
167    };
168    Ok(())
169  }
170}

172/// Validate a pragmatic class, as indicated by a string name,
173///   against a context dictionary of known objects updating the dictionary when needed
174///
175/// The pragmatics are only used for single-letter values with factor/function role,
176///   relying on common conventions in mathematical syntax:
177///
178/// The two main alphabets are separated into letter blocks, each block
179///   required to coordinate to the same xarith type where each separate style of a letter
180///   (italic/bold;Uppercase/lowercase) stands for a separate pragmatic block.
181///
182/// For example in "Ax^2 + Bx + C" we can obtain a single parse where A,B,C are coefficients
183fn _pragma_letter_blocks(name: &str) -> String {
184  let (base, sep, lexeme) = distill_lexeme(name);
185  let lexeme_pragmatic = if lexeme.len() == 1 {
186    let letter = lexeme.chars().next().unwrap();
187    match _PRAGMATIC_BLOCK_MAP.get(&letter) {
188      Some(block) => block,
189      None => lexeme,
190    }
191  } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
192    match _PRAGMATIC_BLOCK_MAP.get(&greek_letter) {
193      Some(block_name) => block_name,
194      None => lexeme,
195    }
196  } else {
197    lexeme
198  };
199  base.to_owned() + sep + lexeme_pragmatic
200}

202/// Validate a pragmatic class, indicated by the case of a named lexeme
203///
204/// For example in SUx
205fn _pragma_letter_case(name: &str) -> String {
206  let (base, sep, lexeme) = distill_lexeme(name);
207  let lexeme_pragmatic = if lexeme.chars().next().unwrap().is_uppercase() {
208    "U"
209  } else {
210    "l"
211  };
212  base.to_owned() + sep + lexeme_pragmatic
213}

215/// Validate a pragmatic class, indicated by the case of a named lexeme
216/// ALSO disregarding any scripted guards
217/// (i.e. scripted lexemes must coordinate with unscripted ones)
218/// For example in "R S_2"
219fn _pragma_letter_case_flat(name: &str) -> String { _pragma_letter_case(name).replace("sub__", "") }
220
221fn _pragma_letter_case_flat_unstyled(name: &str) -> String {
222  let unstyled_name = match name.rfind('-') {
223    Some(position) => {
224      let (base, trailer) = name.split_at(position);
225      let (_sep, lexeme) = trailer.split_at(1);
226      match base.rfind(':') {
227        Some(position) => {
228          let (role, _discard) = base.split_at(position);
229          format!("{role}:{lexeme}")
230        },
231        None => format!("{base}:{lexeme}"),
232      }
233    },
234    None => name.to_owned(),
235  };
236  _pragma_letter_case_flat(&unstyled_name)
237}

239fn pragma_fenced_atoms_are_not_functions(tree: &XM) -> Result<(), Box<dyn Error>> {
240  if let XM::Apply(Operator(op), ..) = tree {
241    if let XM::Lexeme(ref _lexeme, ref atom_meta) = &**op {
242      if let Some(ref fences) = atom_meta.fenced {
243        if fences.as_str() == "parens" {
244          return Err(
245            "pruning non-argument parenthetical atom, used as LHS of function application"
246              .into(),
247          );
248        }
249      }
250    }
251  }
252  Ok(())
253}

257fn pragma_fenced_letters_are_function_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
258  // Mathematical convention: `f(x)` reads as function application, not
259  // multiplication, when x is a letter. Marpa produces both parses in the
260  // ambiguous forest; this pragma chooses the mathematically-consistent one
261  // unconditionally. MATHPARSER_SPECULATE has no role here — the decision is
262  // a pragmatic preference, not a grammar switch.
263  if let XM::Apply(Operator(op), ref args, ..) = tree {
264    match **op {
265      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
266        if let Some(top_lhs) = args.trees().first() {
267          if let Some(ref fences) = top_lhs.get_meta().fenced {
268            if fences.as_str() == "parens" {
269              if let XM::Lexeme(lhs_name, _) = top_lhs.get_baseline() {
270                if !lhs_name.starts_with("NUMBER") {
271                  return Err(
272                    "pruning non-argument parenthetical atom, used as LHS of invisible times"
273                      .into(),
274                  );
275                }
276              }
277            }
278          }
279
280          // Slightly tricky check -- the top RHS needs to be fenced, but we care about the
281          // "baseline" content being a variable - disregarding scripts.
282          if let Some(top_rhs) = args.trees().get(1) {
283            if let XM::Lexeme(rhs_name, _) = top_rhs.get_baseline() {
284              if let Some(ref fences) = top_rhs.get_meta().fenced {
285                if fences.as_str() == "parens" {
286                  // if the RHS is a number, prune unless the LHS is fenced (things like cycle
287                  // notation)
288                  if !rhs_name.starts_with("NUMBER") {
289                    return Err(
290                      "pruning non-argument parenthetical atom, used as RHS of invisible times"
291                        .into(),
292                    );
293                  } else {
294                    match args.trees().first() {
295                      Some(XM::Lexeme(_, lhs_meta)) if lhs_meta.fenced.is_none() => {
296                        return Err(
297                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
298                           times"
299                            .into(),
300                        );
301                      },
302                      Some(XM::Apply(_, _, _, lhs_meta)) if lhs_meta.fenced.is_none() => {
303                        return Err(
304                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
305                           times"
306                            .into(),
307                        );
308                      },
309                      _ => {},
310                    }
311                  }
312                }
313              }
314            }
315          }
316        }
317      },
318      _ => {},
319    }
320  }
321  Ok(())
322}

324/// If we have two standalone letters in the same, such as "A x" or "F X", prune parses that
325/// interpret them as an application, unless they have a role that clearly indicates the LHS is
326/// intended as a functional. This may need to be applied as an optional filter at the end of
327/// pruning, as there are known, albeit rare, counter-examples ("f x").
328fn pragma_unfenced_letter_arguments_require_visual_cues(tree: &XM) -> Result<(), Box<dyn Error>> {
329  match *tree {
330    XM::Apply(Operator(ref op), ref args, ..) if args.0.len() == 1 => {
331      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
332        if atom_meta.fenced.is_none() {
333          let (arg_base, _sep, lexeme) = distill_lexeme(arg_name);
334          let arg_letter: char = if lexeme.len() == 1 {
335            lexeme.chars().next().unwrap()
336          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
337            greek_letter
338          } else {
339            return Ok(()); // rule only applies to single char cases
340          };
341          if arg_base.starts_with("OPERATOR") {
342            // for now single letter OPERATOR arguments just don't make sense without parentheses
343            // e.g. OPERATOR:italic-delta is a differential, and wouldn't be used carelessly as an
344            // arg
345            return Err("single letter argument with role OPERATOR is unusual, prune.".into());
346          }
347          let op_name = op.base_operator_name();
348          let (base, _sep, lexeme) = distill_lexeme(&op_name);
349          if base.starts_with("FUNCTION") || base.starts_with("OPERATOR") {
350            return Ok(()); // don't doubt dedicated roles
351          }
352
353          let op_letter: char = if lexeme.len() == 1 {
354            lexeme.chars().next().unwrap()
355          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
356            greek_letter
357          } else {
358            return Ok(()); // rule only applies to single char cases
359          };
360          if (op_letter.is_lowercase() && arg_letter.is_lowercase())
361            || (op_letter.is_uppercase() && arg_letter.is_uppercase())
362          {
363            return Err("operator and argument are visually similar, prune.".into());
364          }
365        }
366      }
367    },
368    _ => {},
369  };
370  Ok(())
371}

373/// OPFUNCTION are explicitly marked to be operators in an application, they should not be in an
374/// argument role of a single argument Apply. Note that this pragma still allows OPFUNCTIONs to
375/// appear as list elements.
376fn pragma_opfunctions_are_rarely_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
377  match *tree {
378    XM::Apply(_, ref args, ..) if args.0.len() == 1 => {
379      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
380        if arg_name.starts_with("OPFUNCTION") && atom_meta.fenced.is_none() {
381          return Err("OPFUNCTIONs are rarely arguments, prune.".into());
382        }
383      }
384    },
385    _ => {},
386  }
387  Ok(())
388}

390/// Higher order "ID ( FUNCTION )" constructs occassionally show up in the
391/// current DLMF markup, and should remain valid. But they are exceptions.
392/// Meanwhile, a reading of
393/// "ID FUNCTION" as an ID "applied to" the FUNCTION should be invalid.
394/// For now this looks easiest to achieve via a pragmatic prune,
395/// Mostly since we can't enforce IDs to strictly have "curry=1", rather
396/// we need them with "curry >= 1", to stay a little more lenient.
397fn pragma_higher_order_ids_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
398  if let XM::Apply(ref op, ref args, ..) = tree {
399    if args.0.len() == 1 {
400      match &*op.0 {
401        XM::Lexeme(ref op_name, _) if op_name.starts_with("ID:") => match args.0[0] {
402          Some(XM::Lexeme(ref rhs_name, ref rhs_meta)) => {
403            if rhs_name.starts_with("FUNCTION") && rhs_meta.fenced.is_none() {
404              return Err("ID of a higher order than a FUNCTION is not allowed.".into());
405            }
406            if rhs_name.starts_with("ID:") && rhs_meta.fenced.is_none() {
407              return Err("ID applied to unfenced ID is highly unusual, prune.".into());
408            }
409          },
410          Some(XM::Apply(..)) => {
411            return Err("ID of a higher order shouldn't accept any compound trees.".into());
412          },
413          _ => {},
414        },
415        _ => {},
416      }
417    }
418  }
419  Ok(())
420}

422/// If we see " h f (x,y) " a parse where
423/// "h f" are invisible multiplied, is not our first choice.
424/// A right-associative application h(f(x,y)) is instead.
425/// Prune it out if possible.
426fn pragma_higher_order_invisible_ops_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
427  if let XM::Apply(Operator(op), ref args, ..) = tree {
428    match **op {
429      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
430        let trees = args.trees();
431        if trees.len() == 2 {
432          let lhs = trees[0];
433          let rhs = trees[1];
434          if let XM::Lexeme(ref lhs_name, _) = lhs.get_baseline() {
435            if let XM::Lexeme(ref rhs_name, _) = rhs.get_baseline() {
436              if name_is_functional_or_id(lhs_name) && name_is_functional(rhs_name) {
437                return Err(
438                  "Pruning higher order 'FUNCTION x FUNCTION' parse to give precedence to \
439                   right-associative readings"
440                    .into(),
441                );
442              }
443            }
444          }
445        }
446      },
447      _ => {},
448    }
449  }
450
451  Ok(())
452}

454/// If two numbers are left next to each other, as in "10(5)" it is rarely (never?) the intention
455/// that they are to be multiplied Prune such parses. We have special rules for some notations, such
456/// as "dlmf_range".
457fn pragma_adjacent_numbers_dont_use_invisible_times(tree: &XM) -> Result<(), Box<dyn Error>> {
458  if let XM::Apply(Operator(op), ref args, ..) = tree {
459    match **op {
460      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
461        let arg_trees = args.trees();
462        if arg_trees.len() == 2 {
463          if let Some(lhs) = arg_trees.first() {
464            if lhs.base_operator_name().starts_with("NUMBER") {
465              if let Some(rhs) = arg_trees.get(1) {
466                if rhs.base_operator_name().starts_with("NUMBER") {
467                  return Err(
468                    "pruning two adjacent NUMBERs that used an invisible operator".into(),
469                  );
470                }
471              }
472            }
473          }
474        }
475      },
476      _ => {},
477    }
478  }
479  Ok(())
480}

501/// Constructs such as " e_i z_0 " are rarely applying e(z). This is even a common expectation of
502/// function symbols say " f_i g_j ". This pragma prunes trees that applies two adjacent scripted
503/// constructs.
504fn pragma_adjacent_unfenced_scripts_dont_apply(tree: &XM) -> Result<(), Box<dyn Error>> {
505  if let XM::Apply(Operator(op), ref args, ..) = tree {
506    if args.trees().len() == 1 {
507      if let XM::Apply(Operator(ref op_op), _, _, ref op_meta) = **op {
508        let op_base_name = op_op.base_operator_name();
509        if op_meta.fenced.is_none() && (op_base_name == "unknown.subscript") {
510          // only subscripts on the outer one to avoid pruning e.g. \nabla^2 u_{0,0}
511          if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = args.trees().first() {
512            let arg_op_name = arg_op.base_operator_name();
513            if arg_meta.fenced.is_none()
514              && (arg_op_name == "unknown.subscript" || arg_op_name == "unknown.superscript")
515            {
516              return Err("Prune: adjacent unfenced scripts do not form an application.".into());
517            }
518          }
519        }
520      }
521    }
522  }
523  Ok(())
524}

526/// Adjacent functions don't unify into a single operator
527/// as they are meant to apply right-to-left, one-by-one
528fn pragma_adjacent_functions_dont_unify_into_op(tree: &XM) -> Result<(), Box<dyn Error>> {
529  if let XM::Apply(Operator(op), ..) = tree {
530    if let XM::Apply(Operator(inner_op), inner_args, _, inner_meta) = &**op {
531      if inner_meta.fenced.is_none() {
532        if let XM::Lexeme(name, _) = inner_op.get_baseline() {
533          if name_is_functional_or_id(name) {
534            let inner_trees = inner_args.trees();
535            if inner_trees.len() == 1 {
536              if let XM::Lexeme(rhs_name, _) = inner_args.trees().first().unwrap().get_baseline() {
537                if name_is_functional(rhs_name) {
538                  return Err(
539                    "Two applied FUNCTIONS as operator violates right-associative behavior.".into(),
540                  );
541                }
542              }
543            }
544          }
545        }
546      }
547    }
548  }
549  Ok(())
550}
551/// Postfix
552fn pragma_postfix_terms_are_fenced_if_single_arg(tree: &XM) -> Result<(), Box<dyn Error>> {
553  if let XM::Apply(Operator(op), ref args, ..) = tree {
554    let arg_trees = args.trees();
555    if arg_trees.len() == 1 {
556      if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = arg_trees.first() {
557        if arg_op.base_operator_name().starts_with("POSTFIX") && arg_meta.fenced.is_none() {
558          let op_name = op.base_operator_name();
559          if op_name.starts_with("ID")
560            || op_name.starts_with("UNKNOWN")
561            || op_name.starts_with("FUNCTION")
562            || op_name.starts_with("OPERATOR")
563            || op_name.starts_with("DIFFOP")
564          {
565            return Err("pruning postfix term used as single argument without fences".into());
566          }
567        }
568      }
569    }
570  }
571  Ok(())

574/// Scripts are inherently relational — they modify a base. A standalone floating
575/// script (_b by itself, with base=absent) indicates a pre-script that should
576/// attach to a subsequent base. Prefer parses where scripts are attached to
577/// bases over parses where scripts float independently.
578///
579/// This is a universal mathematical convention: in `_b^a A^c_d`, the `_b^a`
580/// are pre-scripts of A, not standalone subscript/superscript expressions.
581fn pragma_maximize_script_attachment(tree: &XM) -> Result<(), Box<dyn Error>> {
582  // Detect any script-op Apply where the first arg (base) is "absent".
583  // This pattern means a standalone floating script that should be a pre-script.
584  if let XM::Apply(Operator(ref op), ref args, _, _) = tree {
585    if let XM::Token(ref props, _) = **op {
586      if props.role.as_deref().is_some_and(|r|
587        r == "SUBSCRIPTOP" || r == "SUPERSCRIPTOP")
588      {
589        // Check if base (first arg) is "absent" — standalone script
590        if let Some(Some(XM::Token(ref base_props, _))) = args.0.first() {
591          if base_props.meaning.as_deref() == Some("absent") {
592            return Err(
593              "Prune: standalone floating script (base=absent) should attach as pre-script.".into()
594            );
595          }
596        }
597      }
598    }
599  }
600  Ok(())
601}

603/// An expression with `absent` on BOTH sides of a relational/binary operator is
604/// never valid mathematical notation. `absent < a` (prefix relop) or `a > absent`
605/// (postfix relop) can occur, but `absent < a > absent` is two missing operands —
606/// too tortured. This prunes `< a >` from being parsed as a bilateral relation chain,
607/// allowing QM expectation `<a>` to win instead.
608fn pragma_no_bilateral_absent(tree: &XM) -> Result<(), Box<dyn Error>> {
609  if let XM::Apply(_, ref args, ..) = tree {
610    let trees = args.trees();
611    if trees.len() >= 2 {
612      let first_absent = matches!(trees.first(),
613        Some(XM::Token(ref p, _)) if p.meaning.as_deref() == Some("absent"));
614      let last_absent = matches!(trees.last(),
615        Some(XM::Token(ref p, _)) if p.meaning.as_deref() == Some("absent"));
616      if first_absent && last_absent {
617        return Err(
618          "Prune: bilateral absent (absent on both sides) is never valid notation.".into()
619        );
620      }
621    }
622  }
623  Ok(())
624}

626/// Functions prefer wider argument absorption: `\log 2x^2` means `log(2*x^2)`.
627/// Prune parses where a function application is immediately followed by invisible
628/// multiplication with unfenced factors. The competing parse would have the function
629/// absorb those factors as arguments.
630fn pragma_functions_prefer_wider_absorption(tree: &XM) -> Result<(), Box<dyn Error>> {
631  // Reject: invisible_times(function_app(f, narrow_arg), simple_rhs)
632  // This forces the wider parse: function_app(f, narrow_arg * rhs)
633  // Only fires when RHS is a simple factor (Lexeme/Token/Wrap), NOT another
634  // function application — chained functions like sin(πx)*cos(2πy) should stay separate.
635  if let XM::Apply(Operator(op), ref args, ..) = tree {
636    let is_invisible_times = match **op {
637      XM::Lexeme(ref oplexeme, _) => oplexeme.contains("invisible_operator"),
638      XM::Token(ref props, _) => {
639        props.meaning.as_deref() == Some("times")
640          && props.role.as_deref() == Some("MULOP")
641      },
642      _ => false,
643    };
644    if is_invisible_times {
645      let trees = args.trees();
646      if trees.len() == 2 {
647        // LHS is a function application (Apply(function, arg))
648        if let XM::Apply(Operator(ref func_op), ref func_args, _, ref func_meta) = trees[0] {
649          // Check if the function op is FUNCTION/OPFUNCTION/TRIGFUNCTION
650          let func_name = func_op.base_operator_name();
651          if (func_name.starts_with("OPFUNCTION") || func_name.starts_with("TRIGFUNCTION"))
652            && func_meta.fenced.is_none()
653            && func_args.trees().len() == 1
654          {
655            // RHS must be a simple factor — not another function application
656            // or compound expression. Scripted factors (x^2) are simple.
657            let rhs = trees[1];
658            let rhs_is_simple = match rhs {
659              XM::Lexeme(..) | XM::Token(..) | XM::Wrap(..) => true,
660              XM::Apply(Operator(ref rhs_op), _, _, _) => {
661                // Scripted factors (SUPERSCRIPTOP/SUBSCRIPTOP) are simple
662                let rhs_role = match &**rhs_op {
663                  XM::Token(ref props, _) => props.role.as_deref().unwrap_or(""),
664                  XM::Lexeme(ref lex, _) => lex.split(':').next().unwrap_or(""),
665                  _ => "",
666                };
667                rhs_role == "SUPERSCRIPTOP" || rhs_role == "SUBSCRIPTOP"
668              },
669              _ => false,
670            };
671            let rhs_meta = rhs.get_meta();
672            if rhs_is_simple && rhs_meta.fenced.is_none() {
673              return Err(
674                "Prune: function application followed by simple unfenced factor — \
675                 prefer wider absorption.".into()
676              );
677            }
678          }
679        }
680      }
681    }
682  }
683  // Also check N-ary invisible_times chains for bare OPFUNCTION in non-terminal
684  // positions. E.g. Apply(×, [f@(x), d, x]) where d is bare OPFUNCTION at index 1
685  // (not the last). The competing parse Apply(×, [f@(x), d@(x)]) is preferred.
686  // This covers the pattern: ∫ f(x) \diffd x → f@(x) * diffd@(x), not f@(x)*d*x.
687  if let XM::Apply(Operator(op), ref args, ..) = tree {
688    let is_invisible_times = match **op {
689      XM::Lexeme(ref oplexeme, _) => oplexeme.contains("invisible_operator"),
690      XM::Token(ref props, _) => {
691        props.meaning.as_deref() == Some("times")
692          && props.role.as_deref() == Some("MULOP")
693      },
694      _ => false,
695    };
696    if is_invisible_times {
697      let trees = args.trees();
698      // Check non-terminal positions for bare OPFUNCTION/TRIGFUNCTION tokens.
699      // Both types absorb bare arguments via prefix_apply. If they appear as
700      // standalone factors in an invisible_times chain (not the last element),
701      // the competing parse with absorption is preferred.
702      // FUNCTION is excluded — it only absorbs fenced (parenthesized) args,
703      // so `f * x` (invisible_times) is correct for bare FUNCTION.
704      if trees.len() >= 3 {
705        for tree in trees.iter().take(trees.len() - 1) {
706          let is_bare_absorbing_func = match tree {
707            XM::Token(ref props, _) => matches!(
708              props.role.as_deref(),
709              Some("OPFUNCTION") | Some("TRIGFUNCTION")
710            ),
711            XM::Lexeme(ref lex, _) => {
712              lex.starts_with("OPFUNCTION:") || lex.starts_with("TRIGFUNCTION:")
713            },
714            _ => false,
715          };
716          if is_bare_absorbing_func {
717            return Err(
718              "Prune: bare OPFUNCTION/TRIGFUNCTION in N-ary invisible_times chain \
719               (non-terminal) — prefer absorption via prefix_apply."
720                .into(),
721            );
722          }
723        }
724      }
725    }
726  }
727  Ok(())
728}

730/// Bigop prefer wider absorption: reject mulop(bigop_app(narrow), rhs)
731/// when rhs is a simple factor that could have been part of the bigop's argument.
732/// Perl's moreOpArgFactors absorbs MulOp chains into bigop arguments.
733fn pragma_bigop_prefer_wider_absorption(tree: &XM) -> Result<(), Box<dyn Error>> {
734  // Pattern: mulop(bigop_app, simple_rhs) or invisible_times(bigop_app, simple_rhs)
735  if let XM::Apply(Operator(op), ref args, ..) = tree {
736    let is_mulop = match **op {
737      XM::Token(ref props, _) => {
738        props.role.as_deref() == Some("MULOP")
739      },
740      XM::Lexeme(ref lex, _) => {
741        lex.starts_with("MULOP") || lex.contains("invisible_operator")
742      },
743      _ => false,
744    };
745    if is_mulop {
746      let trees = args.trees();
747      if trees.len() == 2 {
748        // LHS is a bigop application (Apply with BIGOP/SUMOP/INTOP/LIMITOP/DIFFOP op)
749        if let XM::Apply(Operator(ref bigop_op), _, _, _) = trees[0] {
750          let bigop_name = bigop_op.base_operator_name();
751          let is_bigop = bigop_name.starts_with("BIGOP")
752            || bigop_name.starts_with("SUMOP")
753            || bigop_name.starts_with("INTOP")
754            || bigop_name.starts_with("LIMITOP")
755            || bigop_name.starts_with("DIFFOP");
756          if is_bigop {
757            // RHS should be a simple factor, not another bigop or function
758            let rhs = trees[1];
759            let rhs_is_simple = match rhs {
760              XM::Lexeme(..) | XM::Token(..) | XM::Wrap(..) => true,
761              XM::Apply(Operator(ref rhs_op), _, _, _) => {
762                let rhs_role = match &**rhs_op {
763                  XM::Token(ref props, _) => props.role.as_deref().unwrap_or(""),
764                  XM::Lexeme(ref lex, _) => lex.split(':').next().unwrap_or(""),
765                  _ => "",
766                };
767                // Scripted factors and invisible_times products are simple
768                rhs_role == "SUPERSCRIPTOP"
769                  || rhs_role == "SUBSCRIPTOP"
770                  || rhs_role == "MULOP"
771                  || rhs_role == "DIFFOP"
772              },
773              _ => false,
774            };
775            if rhs_is_simple {
776              return Err(
777                "Prune: bigop application followed by mulop factor — \
778                 prefer wider bigop absorption."
779                  .into(),
780              );
781            }
782          }
783        }
784      }
785    }
786  }
787  Ok(())
788}

790/// Prefer binary ADDOP interpretation over unary prefix when the ADDOP
791/// is between terms in an additive chain. Rejects parses where a unary prefix
792/// (like `-x`) appears as a non-first argument to an infix ADDOP application.
793/// For `a - b + c`, this rejects `a + (prefix(-,b+c))` in favor of `(a-b)+c`.
794fn pragma_prefer_binary_addop(tree: &XM) -> Result<(), Box<dyn Error>> {
795  // Check: infix ADDOP application where one argument is itself a prefix ADDOP
796  if let XM::Apply(Operator(op), ref args, ..) = tree {
797    let is_addop = match **op {
798      XM::Token(ref props, _) => props.role.as_deref() == Some("ADDOP"),
799      XM::Lexeme(ref lex, _) => lex.starts_with("ADDOP"),
800      _ => false,
801    };
802    if is_addop {
803      // Check arguments: if any non-first argument is a unary prefix ADDOP application,
804      // this parse used unary where binary was more appropriate.
805      let trees = args.trees();
806      for (i, arg) in trees.iter().enumerate() {
807        if i == 0 { continue; } // first arg can legitimately start with unary
808        if is_unary_addop_prefix(arg) {
809          return Err("prefer_binary_addop: non-initial argument is unary ADDOP prefix".into());
810        }
811      }
812    }
813  }
814  Ok(())
815}

817/// Check if a tree is a unary ADDOP prefix application (like -x or +x).
818fn is_unary_addop_prefix(tree: &XM) -> bool {
819  if let XM::Apply(Operator(op), ref args, ..) = tree {
820    let is_addop = match **op {
821      XM::Token(ref props, _) => props.role.as_deref() == Some("ADDOP"),
822      XM::Lexeme(ref lex, _) => lex.starts_with("ADDOP"),
823      _ => false,
824    };
825    if is_addop && args.trees().len() == 1 {
826      return true; // unary prefix: addop(x) with single argument
827    }
828  }
829  false
830}

985/// For invisible-times chains of simple unfenced tokens, enforce left-associative grouping.
986///
987/// Mathematical practice: `pppppp` is always read as a flat product `p·p·p·p·p·p`.
988/// The grammar produces exponentially many associativity groupings (Catalan number growth),
989/// but they are all semantically identical for simple identifier chains.
990///
991/// Rule: if `Apply(invisible_times, [lhs, rhs])` where `rhs` is itself
992/// `Apply(invisible_times, ...)`, AND all leaf operands are simple unfenced identifiers
993/// (no fences, no operators, no numbers), then prune — only left-associative grouping
994/// `Apply(invisible_times, [Apply(invisible_times, [...]), single])` survives.
995fn pragma_flatten_simple_invisible_times(tree: &XM) -> Result<(), Box<dyn Error>> {
996  check_invisible_times_recursive(tree)
997}

999/// Recursively check all subtrees for right-associative invisible-times chains.
1000fn check_invisible_times_recursive(tree: &XM) -> Result<(), Box<dyn Error>> {
1001  if let XM::Apply(Operator(op), ref args, ..) = tree {
1002    if let XM::Lexeme(ref oplexeme, _) = **op {
1003      if oplexeme == "x.invisible_operator" {
1004        let trees = args.trees();
1005        if trees.len() == 2 {
1006          let rhs = trees[1];
1007          // Check if the RHS is itself an invisible-times application
1008          if is_invisible_times_apply(rhs) && all_simple_identifiers(tree) {
1009            return Err(
1010              "Pruning right-associative invisible-times of simple identifier chain. \
1011               Flat left-associative product is the only reasonable reading."
1012                .into(),
1013            );
1014          }
1015        }
1016      }
1017    }
1018    // Recurse into all subtrees
1019    for subtree in args.trees() {
1020      check_invisible_times_recursive(subtree)?;
1021    }
1022  }
1023  Ok(())
1024}

1026/// Check if a tree node is an invisible-times application
1027fn is_invisible_times_apply(tree: &XM) -> bool {
1028  if let XM::Apply(Operator(op), ..) = tree {
1029    if let XM::Lexeme(ref oplexeme, _) = **op {
1030      return oplexeme == "x.invisible_operator";
1031    }
1032  }
1033  false
1034}

1036/// Check if ALL leaf tokens in a tree are simple unfenced atoms
1037/// (identifiers or numbers, no fences, no operators)
1038fn all_simple_identifiers(tree: &XM) -> bool {
1039  match tree {
1040    XM::Lexeme(name, meta) => {
1041      // Simple atom: UNKNOWN, ID, or NUMBER role, no fences
1042      meta.fenced.is_none()
1043        && (name.starts_with("UNKNOWN") || name.starts_with("ID") || name.starts_with("NUMBER"))
1044    },
1045    XM::Apply(Operator(op), ref args, ..) => {
1046      if let XM::Lexeme(ref oplexeme, _) = **op {
1047        // For invisible-times applications, check operator and all args
1048        if oplexeme == "x.invisible_operator" {
1049          return args.trees().iter().all(|a| all_simple_identifiers(a));
1050        }
1051        // Scripted atoms (subscript/superscript of simple identifiers) are also simple
1052        if oplexeme.starts_with("SUBSCRIPTOP")
1053          || oplexeme.starts_with("SUPERSCRIPTOP")
1054          || oplexeme.starts_with("POSTSUPERSCRIPT")
1055          || oplexeme.starts_with("POSTSUBSCRIPT")
1056        {
1057          return args.trees().iter().all(|a| all_simple_identifiers(a));
1058        }
1059      }
1060      false
1061    },
1062    XM::Token(ref props, _) => {
1063      // Token with UNKNOWN/ID/NUMBER role
1064      props.role.as_deref().is_some_and(|r| {
1065        r == "UNKNOWN" || r == "ID" || r == "NUMBER"
1066      }) && props.meaning.as_deref() != Some("absent")
1067    },
1068    _ => false,
1069  }
1070}

1072/// In `a = b + c + d`, RELOP must be at the outermost level.
1073/// Rejects parses where an ADDOP or MULOP contains an unfenced RELOP child.
1074/// Exception: fenced relations like `(x=0)` are allowed as terms.
1075fn pragma_relops_are_outermost(tree: &XM) -> Result<(), Box<dyn Error>> {
1076  check_relops_recursive(tree, false)
1077}
1078
1079fn check_relops_recursive(tree: &XM, inside_addop_or_mulop: bool) -> Result<(), Box<dyn Error>> {
1080  if let XM::Apply(Operator(op), ref args, ..) = tree {
1081    if let XM::Lexeme(ref name, _) = **op {
1082      let is_addop = name.starts_with("ADDOP");
1083      let is_mulop = name.starts_with("MULOP") || name == "x.invisible_operator";
1084      let is_relop = name.starts_with("RELOP");
1085
1086      // If we're inside an addop/mulop and this node is a relop, reject
1087      if inside_addop_or_mulop && is_relop {
1088        return Err(
1089          "Pruning: RELOP found inside ADDOP/MULOP — relations must be at the outermost level"
1090            .into(),
1091        );
1092      }
1093
1094      let child_context = inside_addop_or_mulop || is_addop || is_mulop;
1095      for subtree in args.trees() {
1096        // Don't propagate into fenced subtrees — (x=0) as a term is fine
1097        if !is_fenced(subtree) {
1098          check_relops_recursive(subtree, child_context)?;
1099        }
1100      }
1101    }
1102  }
1103  Ok(())
1104}
