10/// Different supported pragmatics according to which to enforce consistency.
11/// Usually one would use them in increasing generality, iterating over a parse forest
12/// until a sufficiently small number of parses is reached.
13#[derive(Copy, Clone, Debug)]
14pub enum ValidationPragmatics {
15  ConsistentLetterBlocks,
16  ConsistentCase,
17  ConsistentCaseFlat,
18  ConsistentCaseFlatUnstyled,
19  InconsistentNames, /* TODO: Since name consistency is built into the curry levels, using the
20                      * inconsistent names pragma should ensure each LP variable has a globally
21                      * unique name */
22  FencedAtomsAreNotFunctions,
23  FencedLettersAreFunctionArguments,
24  AdjacentNumbersDontMultiply,
25  AdjacentUnfencedScriptsDontApply,
26  AdjacentFunctionsDontUnifyIntoOperator,
27  OpfunctionsAreRarelyArguments,
28  UnfencedLetterArgumentsRequireVisualCues,
29  HigherOrderIDsAreExceptions,
30  HigherOrderInvisibleOpsAreExceptions,
31  StandaloneDiffopsAreNotNumerators,
32  PostfixTermsAreFencedIfSingleArguments,
33  RestrictNumeralFractions,
34  /// Prefer parses where scripts attach to bases rather than floating standalone.
35  MaximizeScriptAttachment,
36  /// An expression should never have `absent` on BOTH the left and right sides
37  /// of a binary/relational operator.
38  NoBilateralAbsent,
39  /// Functions prefer wider absorption: `\log 2x^2` means `log(2*x^2)`,
40  /// not `log(2)*x^2`. When a function can absorb more factors, prefer that parse.
41  /// Implemented by penalizing parses where a function's result is multiplied
42  /// by factors that could have been the function's argument.
43  FunctionsPreferWiderAbsorption,
44  /// Bigops prefer wider absorption: `\int F\times Gdx` means `∫(F×G dx)`,
45  /// not `∫(F)×Gdx`. Perl's moreOpArgFactors absorbs MulOp chains.
46  /// Rejects trees where mulop(bigop_app(narrow), rhs) when rhs is a simple factor.
47  BigopPreferWiderAbsorption,
48  /// Prefer binary ADDOP over unary when it follows a complete term.
49  /// In `-12x^2 - 4xy + 2y`, the interior `-` and `+` should be binary operators,
50  /// not unary prefix applied to the following term. Rejects parses where
51  /// prefix_apply(addop, rhs) appears as a non-initial child of an additive chain.
52  PreferBinaryAddop,
53  /// For invisible-times chains of simple unfenced tokens (e.g., `ppppp`, `xyz`),
54  /// enforce left-associative grouping: `((p·p)·p)·p` not `p·(p·(p·p))`.
55  /// In mathematical practice, consecutive identifiers without explicit operators
56  /// or parentheses form a flat product — associativity doesn't matter semantically,
57  /// and enumerating all groupings causes exponential ambiguity.
58  FlattenSimpleInvisibleTimesChains,
59  /// In `a = b + c + d`, the `=` must be at the outermost level.
60  /// An ADDOP/MULOP cannot have an unfenced RELOP child — that would mean
61  /// treating a relation as a term in an arithmetic expression.
62  /// Exception: fenced relations like `(x=0)` used as modifiers.
63  RelopsAreOutermost,
64}

165  /// Recursively checks a tree at each level for applicable pragmatics.
166  /// All levels must validate for the full tree to be valid.
167  pub fn validate_recursive(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
168    self.validate(tree)?; // top-level check
169    // Recursive check:
170    match tree {
171      XM::Choices(subtrees) => {
172        for subtree in subtrees.iter() {
173          self.validate_recursive(subtree)?
174        }
175      },
176      XM::Apply(Operator(op), args, ..) => {
177        self.validate_recursive(op)?;
178        for arg_subtree in args.trees() {
179          self.validate_recursive(arg_subtree)?;
180        }
181      },
182      XM::Dual(ref content, ref pres, ..) => {
183        self.validate_recursive(content)?;
184        self.validate_recursive(pres)?;
185      },
186      XM::Wrap(ref items, ..) => {
187        for item in items.iter() {
188          self.validate_recursive(item)?;
189        }
190      },
191      _ => {},
192    };
193    Ok(())
194  }

233/// Return true iff `op` is an invisible-times operator head, in either of the
234/// two forms the XM tree produces: `XM::Lexeme("…invisible_operator…")` (the
235/// Marpa lexeme form) or `XM::Token { role: "MULOP", meaning: "times" }`
236/// (the post-`apply_invisible_times` form). Helpers across this file need to
237/// match both — historically several sites matched only the Lexeme form and
238/// silently never fired on real parses.
239fn is_invisible_times_op(op: &XM) -> bool {
240  match op {
241    XM::Lexeme(oplexeme, _) => oplexeme.contains("invisible_operator"),
242    XM::Token(props, _) => {
243      props.meaning.as_deref() == Some("times") && props.role.as_deref() == Some("MULOP")
244    },
245    _ => false,
246  }
247}

249/// Extract the consistency key for the letter-blocks pragma.
250///
251/// The pragmatics are only used for single-letter values with factor/function
252/// role, relying on common conventions in mathematical syntax. The two main
253/// alphabets are separated into letter blocks, each block required to
254/// coordinate to the same xarith type where each separate style of a letter
255/// (italic/bold;Uppercase/lowercase) stands for a separate pragmatic block.
256///
257/// For example in "Ax^2 + Bx + C" we can obtain a single parse where A, B, C
258/// are coefficients (same block) and x is the variable (different block).
259fn _pragma_letter_blocks(name: &str) -> LetterBlocksKey {
260  let (base, sep, lexeme) = distill_lexeme(name);
261  let role_prefix = format!("{base}{sep}");
262  let block = if lexeme.chars().count() == 1 {
263    let letter = lexeme.chars().next().unwrap();
264    _PRAGMATIC_BLOCK_MAP
265      .get(&letter)
266      .cloned()
267      .unwrap_or_else(|| LetterBlock::Standalone(lexeme.to_owned()))
268  } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
269    _PRAGMATIC_BLOCK_MAP
270      .get(&greek_letter)
271      .cloned()
272      .unwrap_or_else(|| LetterBlock::Standalone(lexeme.to_owned()))
273  } else {
274    LetterBlock::Standalone(lexeme.to_owned())
275  };
276  LetterBlocksKey { role_prefix, block }
277}

279/// Extract the consistency key for the letter-case pragma. For example in
280/// "SUx" the key is Lower.
281fn _pragma_letter_case(name: &str) -> LetterCaseKey {
282  let (base, sep, lexeme) = distill_lexeme(name);
283  let role_prefix = format!("{base}{sep}");
284  let case = match lexeme.chars().next() {
285    Some(c) if c.is_uppercase() => LetterCase::Upper,
286    _ => LetterCase::Lower,
287  };
288  LetterCaseKey { role_prefix, case }
289}

291/// Variant of `_pragma_letter_case` that disregards scripted guards
292/// (i.e. scripted lexemes must coordinate with unscripted ones).
293/// For example in "R S_2" R and S_2 should both decide the same case.
294fn _pragma_letter_case_flat(name: &str) -> LetterCaseKey {
295  let key = _pragma_letter_case(name);
296  LetterCaseKey {
297    role_prefix: key.role_prefix.replace("sub__", ""),
298    case:        key.case,
299  }
300}

302/// Variant of `_pragma_letter_case_flat` that also drops style annotations
303/// before the final separator, so italic/bold/upright letters compare as the
304/// same style class.
305fn _pragma_letter_case_flat_unstyled(name: &str) -> LetterCaseKey {
306  // Normalize "ROLE:style-x" → "ROLE:x" so the style annotation drops.
307  let unstyled_name = match name.rfind('-') {
308    Some(position) => {
309      let (base, trailer) = name.split_at(position);
310      let (_sep, lexeme) = trailer.split_at(1);
311      match base.rfind(':') {
312        Some(position) => {
313          let (role, _discard) = base.split_at(position);
314          format!("{role}:{lexeme}")
315        },
316        None => format!("{base}:{lexeme}"),
317      }
318    },
319    None => name.to_owned(),
320  };
321  _pragma_letter_case_flat(&unstyled_name)
322}

324/// Gather a canonical "consistency key" for each single-letter operand of an
325/// invisible-times `Apply`, run the caller's key-extractor on the lexeme name,
326/// and reject if any two peer keys differ. Shared backbone for the four
327/// Consistent* pragmas.
328///
329/// Only fires on `Apply(invisible_operator, …)` — the operator denotes
330/// "juxtaposition" where mathematical convention says peer letters play the
331/// same role (all coefficients, or all variables). Explicit operators
332/// (`a + A`, `a = A`) have their own conventions and are out of scope here.
333///
334/// A "letter operand" is a fenceless `XM::Lexeme` whose distilled lexeme is
335/// either a single char or a Greek name (via `greek_name_to_letter`). Other
336/// operand shapes (numbers, sub-applications, fenced groups) are skipped.
337fn pragma_consistency_via_key<K>(
338  tree: &XM,
339  key_of: fn(&str) -> K,
340  diagnostic: &str,
341) -> Result<(), Box<dyn Error>>
342where
343  K: PartialEq,
344{
345  if let XM::Apply(Operator(op), args, ..) = tree {
346    if !is_invisible_times_op(op) {
347      return Ok(());
348    }
349    // Only fire when there are ≥3 letter operands. A 2-letter chain like
350    // `Ax` is the canonical "coefficient × variable" shape and legitimately
351    // crosses letter blocks (A-E × x-z); rejecting it would prune the right
352    // parse. 3+ operands is where consistency is much more likely than
353    // accidental block-crossing.
354    let mut first_key: Option<K> = None;
355    let mut letter_count = 0usize;
356    let mut keys: Vec<K> = Vec::new();
357    for tree_arg in args.trees() {
358      if let XM::Lexeme(ref name, ref meta) = *tree_arg {
359        if meta.fenced.is_some() {
360          continue;
361        }
362        let (_base, _sep, lexeme) = distill_lexeme(name);
363        let is_letter = lexeme.chars().count() == 1 || greek_name_to_letter(lexeme).is_some();
364        if !is_letter {
365          continue;
366        }
367        letter_count += 1;
368        let key = key_of(name);
369        if first_key.is_none() {
370          first_key = Some(key);
371        } else {
372          keys.push(key);
373        }
374      }
375    }
376    // ≥3 letters required — see comment above on the 2-letter
377    // `Ax`-style coefficient×variable exemption.
378    if letter_count < 3 {
379      return Ok(());
380    }
381    if let Some(ref first) = first_key {
382      for k in &keys {
383        if k != first {
384          return Err(diagnostic.into());
385        }
386      }
387    }
388  }
389  Ok(())
390}

424fn pragma_fenced_atoms_are_not_functions(tree: &XM) -> Result<(), Box<dyn Error>> {
425  if let XM::Apply(Operator(op), ..) = tree {
426    if let XM::Lexeme(ref _lexeme, ref atom_meta) = &**op {
427      if let Some(ref fences) = atom_meta.fenced {
428        if fences.as_str() == "parens" {
429          return Err(
430            "pruning non-argument parenthetical atom, used as LHS of function application".into(),
431          );
432        }
433      }
434    }
435  }
436  Ok(())
437}

439fn pragma_fenced_letters_are_function_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
440  // Mathematical convention: `f(x)` reads as function application, not
441  // multiplication, when x is a letter. Marpa produces both parses in the
442  // ambiguous forest; this pragma chooses the mathematically-consistent one
443  // unconditionally. MATHPARSER_SPECULATE has no role here — the decision is
444  // a pragmatic preference, not a grammar switch.
445  let XM::Apply(Operator(op), ref args, ..) = tree else {
446    return Ok(());
447  };
448  if !is_invisible_times_op(op) {
449    return Ok(());
450  }
451  let trees = args.trees();
452  let Some(top_lhs) = trees.first() else {
453    return Ok(());
454  };
455  if let Some(ref fences) = top_lhs.get_meta().fenced {
456    if fences.as_str() == "parens" {
457      if let XM::Lexeme(lhs_name, _) = top_lhs.get_baseline() {
458        if !lhs_name.starts_with("NUMBER") {
459          return Err(
460            "pruning non-argument parenthetical atom, used as LHS of invisible times".into(),
461          );
462        }
463      }
464    }
465  }
466
467  // Slightly tricky check -- the top RHS needs to be fenced, but we care about the
468  // "baseline" content being a variable - disregarding scripts.
469  if let Some(top_rhs) = trees.get(1) {
470    if let XM::Lexeme(rhs_name, _) = top_rhs.get_baseline() {
471      if let Some(ref fences) = top_rhs.get_meta().fenced {
472        if fences.as_str() == "parens" {
473          // if the RHS is a number, prune unless the LHS is fenced (things like cycle
474          // notation)
475          if !rhs_name.starts_with("NUMBER") {
476            return Err(
477              "pruning non-argument parenthetical atom, used as RHS of invisible times".into(),
478            );
479          } else {
480            match trees.first() {
481              Some(XM::Lexeme(_, lhs_meta)) if lhs_meta.fenced.is_none() => {
482                return Err(
483                  "pruning non-argument parenthetical NUMBER, used as RHS of invisible times"
484                    .into(),
485                );
486              },
487              Some(XM::Apply(_, _, _, lhs_meta)) if lhs_meta.fenced.is_none() => {
488                return Err(
489                  "pruning non-argument parenthetical NUMBER, used as RHS of invisible times"
490                    .into(),
491                );
492              },
493              _ => {},
494            }
495          }
496        }
497      }
498    } else if let Some(prune_reason) = is_dual_fenced_rhs(top_rhs, trees.first().copied()) {
499      return Err(prune_reason.into());
500    }
501  }
502  Ok(())
503}

505/// Detect a parens-fenced `XM::Dual` on the RHS of invisible-times and
506/// classify whether it should be pruned in favor of function-application.
507///
508/// The `fenced` grammar action wraps a parenthesized expression as
509/// `Dual(content_ref, Wrap[OPEN, expr, CLOSE])` without setting the
510/// `Meta::fenced` field on the Dual itself. The `Lexeme`-based check
511/// above misses these. This helper inspects the presentation Wrap
512/// directly:
513///
514/// 1. First and last items of the Wrap are `XM::Token` with role
515///    OPEN/CLOSE and content `(` / `)`.
516/// 2. Inner content (between the parens) is either a non-NUMBER
517///    `Lexeme`, a structured `Apply` (list/vector/formulae), or a
518///    `Dual` whose own content is non-numeric.
519/// 3. Special-case: if the inner content IS a `NUMBER`, fall through
520///    to the legacy "number with non-fenced LHS" exception so the
521///    cycle-notation case `(a,b)(c,d)` doesn't double-prune.
522///
523/// Returns `Some(error_msg)` if the parse should be pruned, `None`
524/// otherwise.
525/// Recognize an `XM::Dual(_, Wrap[OPEN, ..., CLOSE])` shape on the RHS
526/// of invisible-times where the outer delimiters are PARENS, BRACKETS,
527/// or VERTBARS — all of which can legitimately indicate function-app
528/// when preceded by a letter LHS. Returns `Some(_, error_msg)` if the
529/// shape matches.
530///
531/// `kind` is the fence-pair category for diagnostics:
532/// "parens" / "brackets" / "vertbars".
533fn is_dual_fenced_rhs(top_rhs: &XM, top_lhs: Option<&XM>) -> Option<&'static str> {
534  let XM::Dual(_, ref presentation, ..) = top_rhs else {
535    return None;
536  };
537  let XM::Wrap(ref items, ..) = **presentation else {
538    return None;
539  };
540  // Recognize PARENS or BRACKETS delimiters. **Vertbars `|...|`
541  // are deliberately excluded** — K-12 math convention reads
542  // `a|f|b` as `a * |f| * b` (absolute-value multiplication),
543  // NOT as `a@(|f|)` (function application). Generalizing the
544  // "fence implies function-argument" rule to vertbars regressed
545  // 2 legacy tests; the qm_test bra-ket case has its own
546  // expected interpretation that this pragma should not enforce.
547  enum Fence {
548    Parens,
549    Brackets,
550  }
551  // (Marker: vertbar-fenced cases purposely fall through; the qm_test
552  // and similar bra-ket tests pass under a different mechanism
553  // (the QM context resolves differently in soft_prune).
554  let recognize_open = |x: Option<&XM>| -> Option<Fence> {
555    match x {
556      Some(XM::Token(p, _)) => match (p.role.as_deref(), p.content.as_deref()) {
557        (Some("OPEN"), Some("(")) => Some(Fence::Parens),
558        (Some("OPEN"), Some("[")) => Some(Fence::Brackets),
559        _ => None,
560      },
561      Some(XM::Lexeme(name, _)) => {
562        if name.starts_with("OPEN:(:") {
563          Some(Fence::Parens)
564        } else if name.starts_with("OPEN:[:") {
565          Some(Fence::Brackets)
566        } else {
567          None
568        }
569      },
570      _ => None,
571    }
572  };
573  let recognize_close = |x: Option<&XM>, fence: &Fence| -> bool {
574    match x {
575      Some(XM::Token(p, _)) => match (p.role.as_deref(), p.content.as_deref(), fence) {
576        (Some("CLOSE"), Some(")"), Fence::Parens) => true,
577        (Some("CLOSE"), Some("]"), Fence::Brackets) => true,
578        _ => false,
579      },
580      Some(XM::Lexeme(name, _)) => match fence {
581        Fence::Parens => name.starts_with("CLOSE:):"),
582        Fence::Brackets => name.starts_with("CLOSE:]:"),
583      },
584      _ => false,
585    }
586  };
587
588  let Some(fence) = recognize_open(items.first()) else {
589    return None;
590  };
591  if !recognize_close(items.last(), &fence) {
592    return None;
593  }
594  let _ = match fence {
595    Fence::Parens => "parenthetical",
596    Fence::Brackets => "bracketed",
597  };
598
599  // Look at the Dual's content branch — that's the semantic shape.
600  let content = match top_rhs {
601    XM::Dual(c, _, _, _) => &**c,
602    _ => return None,
603  };
604  // Skip the NUMBER case to defer to the legacy NUMBER-exception code.
605  if let XM::Lexeme(inner_name, _) = content.get_baseline() {
606    if inner_name.starts_with("NUMBER") {
607      match top_lhs {
608        Some(XM::Lexeme(_, lhs_meta)) if lhs_meta.fenced.is_none() => {
609          return Some(
610            "pruning non-argument fenced NUMBER (Dual-wrapped), used as RHS of \
611             invisible times",
612          );
613        },
614        Some(XM::Apply(_, _, _, lhs_meta)) if lhs_meta.fenced.is_none() => {
615          return Some(
616            "pruning non-argument fenced NUMBER (Dual-wrapped), used as RHS of \
617             invisible times",
618          );
619        },
620        _ => return None,
621      }
622    }
623  }
624  Some("pruning non-argument fenced Dual atom, used as RHS of invisible times")
625}

926/// Functions prefer wider argument absorption: `\log 2x^2` means `log(2*x^2)`.
927/// Prune parses where a function application is immediately followed by invisible
928/// multiplication with unfenced factors. The competing parse would have the function
929/// absorb those factors as arguments.
930fn pragma_functions_prefer_wider_absorption(tree: &XM) -> Result<(), Box<dyn Error>> {
931  // Reject: invisible_times(function_app(f, narrow_arg), simple_rhs)
932  // This forces the wider parse: function_app(f, narrow_arg * rhs)
933  // Only fires when RHS is a simple factor (Lexeme/Token/Wrap), NOT another
934  // function application — chained functions like sin(πx)*cos(2πy) should stay separate.
935  if let XM::Apply(Operator(op), ref args, ..) = tree {
936    if is_invisible_times_op(op) {
937      let trees = args.trees();
938      if trees.len() == 2 {
939        // LHS is a function application (Apply(function, arg))
940        if let XM::Apply(Operator(ref func_op), ref func_args, _, ref func_meta) = trees[0] {
941          // Check if the function op is FUNCTION/OPFUNCTION/TRIGFUNCTION
942          let func_name = func_op.base_operator_name();
943          if (func_name.starts_with("OPFUNCTION") || func_name.starts_with("TRIGFUNCTION"))
944            && func_meta.fenced.is_none()
945            && func_args.trees().len() == 1
946          {
947            // RHS must be a simple factor — not another function application
948            // or compound expression. Scripted factors (x^2) are simple.
949            let rhs = trees[1];
950            let rhs_is_simple = match rhs {
951              XM::Lexeme(..) | XM::Token(..) | XM::Wrap(..) => true,
952              XM::Apply(Operator(ref rhs_op), ..) => {
953                // Scripted factors (SUPERSCRIPTOP/SUBSCRIPTOP) are simple
954                let rhs_role = match &**rhs_op {
955                  XM::Token(ref props, _) => props.role.as_deref().unwrap_or(""),
956                  XM::Lexeme(ref lex, _) => lex.split(':').next().unwrap_or(""),
957                  _ => "",
958                };
959                rhs_role == "SUPERSCRIPTOP" || rhs_role == "SUBSCRIPTOP"
960              },
961              _ => false,
962            };
963            let rhs_meta = rhs.get_meta();
964            if rhs_is_simple && rhs_meta.fenced.is_none() {
965              return Err(
966                "Prune: function application followed by simple unfenced factor — \
967                 prefer wider absorption."
968                  .into(),
969              );
970            }
971          }
972        }
973      }
974    }
975  }
976  // Also check N-ary invisible_times chains for bare OPFUNCTION in non-terminal
977  // positions. E.g. Apply(×, [f@(x), d, x]) where d is bare OPFUNCTION at index 1
978  // (not the last). The competing parse Apply(×, [f@(x), d@(x)]) is preferred.
979  // This covers the pattern: ∫ f(x) \diffd x → f@(x) * diffd@(x), not f@(x)*d*x.
980  if let XM::Apply(Operator(op), ref args, ..) = tree {
981    if is_invisible_times_op(op) {
982      let trees = args.trees();
983      // Check non-terminal positions for bare OPFUNCTION/TRIGFUNCTION tokens.
984      // Both types absorb bare arguments via prefix_apply. If they appear as
985      // standalone factors in an invisible_times chain (not the last element),
986      // the competing parse with absorption is preferred.
987      // FUNCTION is excluded — it only absorbs fenced (parenthesized) args,
988      // so `f * x` (invisible_times) is correct for bare FUNCTION.
989      if trees.len() >= 3 {
990        for tree in trees.iter().take(trees.len() - 1) {
991          let is_bare_absorbing_func = match tree {
992            XM::Token(ref props, _) => matches!(
993              props.role.as_deref(),
994              Some("OPFUNCTION") | Some("TRIGFUNCTION")
995            ),
996            XM::Lexeme(ref lex, _) => {
997              lex.starts_with("OPFUNCTION:") || lex.starts_with("TRIGFUNCTION:")
998            },
999            _ => false,
1000          };
1001          if is_bare_absorbing_func {
1002            return Err(
1003              "Prune: bare OPFUNCTION/TRIGFUNCTION in N-ary invisible_times chain \
1004               (non-terminal) — prefer absorption via prefix_apply."
1005                .into(),
1006            );
1007          }
1008        }
1009      }
1010    }
1011  }
1012  Ok(())
1013}

1064        }

1073/// (like `-x`) appears as a non-first argument to an infix ADDOP application.
1074/// For `a - b + c`, this rejects `a + (prefix(-,b+c))` in favor of `(a-b)+c`.

1267/// For invisible-times chains of simple unfenced tokens, enforce left-associative grouping.
1268///
1269/// Mathematical practice: `pppppp` is always read as a flat product `p·p·p·p·p·p`.
1270/// The grammar produces exponentially many associativity groupings (Catalan number growth),
1271/// but they are all semantically identical for simple identifier chains.
1272///
1273/// Rule: if `Apply(invisible_times, [lhs, rhs])` where `rhs` is itself
1274/// `Apply(invisible_times, ...)`, AND all leaf operands are simple unfenced identifiers
1275/// (no fences, no operators, no numbers), then prune — only left-associative grouping
1276/// `Apply(invisible_times, [Apply(invisible_times, [...]), single])` survives.
1277fn pragma_flatten_simple_invisible_times(tree: &XM) -> Result<(), Box<dyn Error>> {
1278  check_invisible_times_recursive(tree)
1279}
1280
1281/// Recursively check all subtrees for right-associative invisible-times chains.
1282fn check_invisible_times_recursive(tree: &XM) -> Result<(), Box<dyn Error>> {
1283  if let XM::Apply(Operator(op), ref args, ..) = tree {
1284    if is_invisible_times_op(op) {
1285      let trees = args.trees();
1286      if trees.len() == 2 {
1287        let rhs = trees[1];
1288        // Check if the RHS is itself an invisible-times application
1289        if is_invisible_times_apply(rhs) && all_simple_identifiers(tree) {
1290          return Err(
1291            "Pruning right-associative invisible-times of simple identifier chain. \
1292             Flat left-associative product is the only reasonable reading."
1293              .into(),
1294          );
1295        }
1296      }
1297    }
1298    // Recurse into all subtrees
1299    for subtree in args.trees() {
1300      check_invisible_times_recursive(subtree)?;
1301    }
1302  }
1303  Ok(())
1304}

1306/// Check if a tree node is an invisible-times application, in either operator form.
1307fn is_invisible_times_apply(tree: &XM) -> bool {
1308  if let XM::Apply(Operator(op), ..) = tree {
1309    return is_invisible_times_op(op);
1310  }
1311  false
1312}

1314/// Check if ALL leaf tokens in a tree are simple unfenced atoms
1315/// (identifiers or numbers, no fences, no operators)
1316fn all_simple_identifiers(tree: &XM) -> bool {
1317  match tree {
1318    XM::Lexeme(name, meta) => {
1319      // Simple atom: UNKNOWN, ID, or NUMBER role, no fences
1320      meta.fenced.is_none()
1321        && (name.starts_with("UNKNOWN") || name.starts_with("ID") || name.starts_with("NUMBER"))
1322    },
1323    XM::Apply(Operator(op), ref args, ..) => {
1324      // For invisible-times applications (either operator shape),
1325      // check operator and all args
1326      if is_invisible_times_op(op) {
1327        return args.trees().iter().all(|a| all_simple_identifiers(a));
1328      }
1329      if let XM::Lexeme(ref oplexeme, _) = **op {
1330        // Scripted atoms (subscript/superscript of simple identifiers) are also simple
1331        if oplexeme.starts_with("SUBSCRIPTOP")
1332          || oplexeme.starts_with("SUPERSCRIPTOP")
1333          || oplexeme.starts_with("POSTSUPERSCRIPT")
1334          || oplexeme.starts_with("POSTSUBSCRIPT")
1335        {
1336          return args.trees().iter().all(|a| all_simple_identifiers(a));
1337        }
1338      }
1339      false
1340    },
1341    XM::Token(ref props, _) => {
1342      // Token with UNKNOWN/ID/NUMBER role
1343      props
1344        .role
1345        .as_deref()
1346        .is_some_and(|r| r == "UNKNOWN" || r == "ID" || r == "NUMBER")
1347        && props.meaning.as_deref() != Some("absent")
1348    },
1349    _ => false,
1350  }
1351}

1353/// In `a = b + c + d`, RELOP must be at the outermost level.
1354/// Rejects parses where an ADDOP or MULOP contains an unfenced RELOP child.
1355/// Exception: fenced relations like `(x=0)` are allowed as terms.
1356fn pragma_relops_are_outermost(tree: &XM) -> Result<(), Box<dyn Error>> {
1357  check_relops_recursive(tree, false)
1358}
1359
1360fn check_relops_recursive(tree: &XM, inside_addop_or_mulop: bool) -> Result<(), Box<dyn Error>> {
1361  if let XM::Apply(Operator(op), ref args, ..) = tree {
1362    if let XM::Lexeme(ref name, _) = **op {
1363      let is_addop = name.starts_with("ADDOP");
1364      let is_mulop = name.starts_with("MULOP") || &**name == "x.invisible_operator";
1365      let is_relop = name.starts_with("RELOP");
1366
1367      // If we're inside an addop/mulop and this node is a relop, reject
1368      if inside_addop_or_mulop && is_relop {
1369        return Err(
1370          "Pruning: RELOP found inside ADDOP/MULOP — relations must be at the outermost level"
1371            .into(),
1372        );
1373      }
1374
1375      let child_context = inside_addop_or_mulop || is_addop || is_mulop;
1376      for subtree in args.trees() {
1377        // Don't propagate into fenced subtrees — (x=0) as a term is fine
1378        if !is_fenced(subtree) {
1379          check_relops_recursive(subtree, child_context)?;
1380        }
1381      }
1382    }
1383  }
1384  Ok(())
1385}

1388fn is_fenced(tree: &XM) -> bool {
1389  match tree {
1390    XM::Lexeme(_, ref meta) => meta.fenced.is_some(),
1391    XM::Apply(_, _, _, ref meta) => meta.fenced.is_some(),
1392    _ => false,
1393  }

2024  fn nested_inv_times(letter_names: &[&str]) -> XM {
2025    // Build a right-associative chain `a × (b × (c × d))` under the
2026    // XM::Token{MULOP, times} operator shape.
2027    use crate::semantics::metadata::Meta;
2028    use crate::semantics::tree::{Args, Operator, XProps};
2029    use std::borrow::Cow;
2030
2031    assert!(letter_names.len() >= 2, "need at least 2 operands");
2032    let mut acc = XM::Lexeme(Rc::from(*letter_names.last().unwrap()), Meta::default());
2033    for name in letter_names.iter().rev().skip(1) {
2034      let op_props = XProps {
2035        role: Some(Cow::Borrowed("MULOP")),
2036        meaning: Some(Cow::Borrowed("times")),
2037        content: Some(Cow::Borrowed("\u{2062}")),
2038        ..XProps::default()
2039      };
2040      let op = Operator(Box::new(XM::Token(op_props, Meta::default())));
2041      let args = Args(vec![
2042        Some(XM::Lexeme(Rc::from(*name), Meta::default())),
2043        Some(acc),
2044      ]);
2045      acc = XM::Apply(op, args, XProps::default(), Meta::default());
2046    }
2047    acc
2048  }
