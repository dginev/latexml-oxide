8/// Different supported pragmatics according to which to enforce consistency.
9/// Usually one would use them in increasing generality, iterating over a parse forest
10/// until a sufficiently small number of parses is reached.
11#[derive(Copy, Clone, Debug)]
12pub enum ValidationPragmatics {
13  ConsistentLetterBlocks,
14  ConsistentCase,
15  ConsistentCaseFlat,
16  ConsistentCaseFlatUnstyled,
17  InconsistentNames, /* TODO: Since name consistency is built into the curry levels, using the
18                      * inconsistent names pragma should ensure each LP variable has a globally
19                      * unique name */
20  FencedAtomsAreNotFunctions,
21  FencedLettersAreFunctionArguments,
22  AdjacentNumbersDontMultiply,
23  AdjacentUnfencedScriptsDontApply,
24  AdjacentFunctionsDontUnifyIntoOperator,
25  OpfunctionsAreRarelyArguments,
26  UnfencedLetterArgumentsRequireVisualCues,
27  HigherOrderIDsAreExceptions,
28  HigherOrderInvisibleOpsAreExceptions,
29  StandaloneDiffopsAreNotNumerators,
30  PostfixTermsAreFencedIfSingleArguments,
31  RestrictNumeralFractions,
32  /// Prefer parses where scripts attach to bases rather than floating standalone.
33  MaximizeScriptAttachment,
34  /// An expression should never have `absent` on BOTH the left and right sides
35  /// of a binary/relational operator.
36  NoBilateralAbsent,
37  /// Functions prefer wider absorption: `\log 2x^2` means `log(2*x^2)`,
38  /// not `log(2)*x^2`. When a function can absorb more factors, prefer that parse.
39  /// Implemented by penalizing parses where a function's result is multiplied
40  /// by factors that could have been the function's argument.
41  FunctionsPreferWiderAbsorption,
42  /// Bigops prefer wider absorption: `\int F\times Gdx` means `∫(F×G dx)`,
43  /// not `∫(F)×Gdx`. Perl's moreOpArgFactors absorbs MulOp chains.
44  /// Rejects trees where mulop(bigop_app(narrow), rhs) when rhs is a simple factor.
45  BigopPreferWiderAbsorption,
46}

65  /// optionally, until there is a single parse left. If a pragmatic rule invalidates

68    // Order here is crucial - it shows a decreasing level of adoption of each rule in the

87  /// Employs the selected pragmatic to augment the validation of the core Meta object
88  pub fn validate(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
89    use ValidationPragmatics::*;
90    match self {
91      FencedAtomsAreNotFunctions => pragma_fenced_atoms_are_not_functions(tree),
92      UnfencedLetterArgumentsRequireVisualCues => {
93        pragma_unfenced_letter_arguments_require_visual_cues(tree)
94      },
95
96      OpfunctionsAreRarelyArguments => pragma_opfunctions_are_rarely_arguments(tree),
97
98      FencedLettersAreFunctionArguments => pragma_fenced_letters_are_function_arguments(tree),
99      HigherOrderIDsAreExceptions => pragma_higher_order_ids_are_exceptions(tree),
100      HigherOrderInvisibleOpsAreExceptions => {
101        pragma_higher_order_invisible_ops_are_exceptions(tree)
102      },
103      AdjacentNumbersDontMultiply => pragma_adjacent_numbers_dont_use_invisible_times(tree),
104      AdjacentUnfencedScriptsDontApply => pragma_adjacent_unfenced_scripts_dont_apply(tree),
105      StandaloneDiffopsAreNotNumerators => pragma_standalone_diffops_are_not_numerators(tree),
106      PostfixTermsAreFencedIfSingleArguments => pragma_postfix_terms_are_fenced_if_single_arg(tree),
107      AdjacentFunctionsDontUnifyIntoOperator => pragma_adjacent_functions_dont_unify_into_op(tree),
108      RestrictNumeralFractions => pragma_restrict_numeral_fractions(tree),
109      MaximizeScriptAttachment => pragma_maximize_script_attachment(tree),
110      NoBilateralAbsent => pragma_no_bilateral_absent(tree),
111      FunctionsPreferWiderAbsorption => pragma_functions_prefer_wider_absorption(tree),
112      BigopPreferWiderAbsorption => pragma_bigop_prefer_wider_absorption(tree),
113      _ => Ok(()),
114    }
115  }

117  /// Recursively checks a tree at each level for applicable pragmatics.
118  /// All levels must validate for the full tree to be valid.
119  pub fn validate_recursive(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
120    self.validate(tree)?; // top-level check
121    // Recursive check:
122    match tree {
123      XM::Choices(subtrees) => {
124        for subtree in subtrees.iter() {
125          self.validate_recursive(subtree)?
126        }
127      },
128      XM::Apply(Operator(op), args, ..) => {
129        self.validate_recursive(op)?;
130        for arg_subtree in args.trees() {
131          self.validate_recursive(arg_subtree)?;
132        }
133      },
134      XM::Dual(ref content, ref pres, _, _) => {
135        self.validate_recursive(content)?;
136        self.validate_recursive(pres)?;
137      },
138      XM::Wrap(ref items, _, _) => {
139        for item in items.iter() {
140          self.validate_recursive(item)?;
141        }
142      },
143      _ => {},
144    };
145    Ok(())
146  }
147}

149/// Validate a pragmatic class, as indicated by a string name,
150///   against a context dictionary of known objects updating the dictionary when needed
151///
152/// The pragmatics are only used for single-letter values with factor/function role,
153///   relying on common conventions in mathematical syntax:
154///
155/// The two main alphabets are separated into letter blocks, each block
156///   required to coordinate to the same xarith type where each separate style of a letter
157///   (italic/bold;Uppercase/lowercase) stands for a separate pragmatic block.
158///
159/// For example in "Ax^2 + Bx + C" we can obtain a single parse where A,B,C are coefficients
160fn _pragma_letter_blocks(name: &str) -> String {
161  let (base, sep, lexeme) = distill_lexeme(name);
162  let lexeme_pragmatic = if lexeme.len() == 1 {
163    let letter = lexeme.chars().next().unwrap();
164    match _PRAGMATIC_BLOCK_MAP.get(&letter) {
165      Some(block) => block,
166      None => lexeme,
167    }
168  } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
169    match _PRAGMATIC_BLOCK_MAP.get(&greek_letter) {
170      Some(block_name) => block_name,
171      None => lexeme,
172    }
173  } else {
174    lexeme
175  };
176  base.to_owned() + sep + lexeme_pragmatic
177}

179/// Validate a pragmatic class, indicated by the case of a named lexeme
180///
181/// For example in SUx
182fn _pragma_letter_case(name: &str) -> String {
183  let (base, sep, lexeme) = distill_lexeme(name);
184  let lexeme_pragmatic = if lexeme.chars().next().unwrap().is_uppercase() {
185    "U"
186  } else {
187    "l"
188  };
189  base.to_owned() + sep + lexeme_pragmatic
190}

192/// Validate a pragmatic class, indicated by the case of a named lexeme
193/// ALSO disregarding any scripted guards
194/// (i.e. scripted lexemes must coordinate with unscripted ones)
195/// For example in "R S_2"
196fn _pragma_letter_case_flat(name: &str) -> String { _pragma_letter_case(name).replace("sub__", "") }
197
198fn _pragma_letter_case_flat_unstyled(name: &str) -> String {
199  let unstyled_name = match name.rfind('-') {
200    Some(position) => {
201      let (base, trailer) = name.split_at(position);
202      let (_sep, lexeme) = trailer.split_at(1);
203      match base.rfind(':') {
204        Some(position) => {
205          let (role, _discard) = base.split_at(position);
206          format!("{role}:{lexeme}")
207        },
208        None => format!("{base}:{lexeme}"),
209      }
210    },
211    None => name.to_owned(),
212  };
213  _pragma_letter_case_flat(&unstyled_name)
214}

216fn pragma_fenced_atoms_are_not_functions(tree: &XM) -> Result<(), Box<dyn Error>> {
217  if let XM::Apply(Operator(op), ..) = tree {
218    if let XM::Lexeme(ref _lexeme, ref atom_meta) = **op {
219      if let Some(ref fences) = atom_meta.fenced {
220        if fences.as_str() == "parens" {
221          return Err(
222            "pruning non-argument parenthetical atom, used as LHS of function application".into(),
223          );
224        }
225      }
226    }
227  }
228  Ok(())
229}

231fn pragma_fenced_letters_are_function_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
232  if let XM::Apply(Operator(op), ref args, ..) = tree {
233    match **op {
234      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
235        if let Some(top_lhs) = args.trees().first() {
236          if let Some(ref fences) = top_lhs.get_meta().fenced {
237            if fences.as_str() == "parens" {
238              if let XM::Lexeme(lhs_name, _) = top_lhs.get_baseline() {
239                if !lhs_name.starts_with("NUMBER") {
240                  return Err(
241                    "pruning non-argument parenthetical atom, used as LHS of invisible times"
242                      .into(),
243                  );
244                }
245              }
246            }
247          }
248
249          // Slightly tricky check -- the top RHS needs to be fenced, but we care about the
250          // "baseline" content being a variable - disregarding scripts.
251          if let Some(top_rhs) = args.trees().get(1) {
252            if let XM::Lexeme(rhs_name, _) = top_rhs.get_baseline() {
253              if let Some(ref fences) = top_rhs.get_meta().fenced {
254                if fences.as_str() == "parens" {
255                  // if the RHS is a number, prune unless the LHS is fenced (things like cycle
256                  // notation)
257                  if !rhs_name.starts_with("NUMBER") {
258                    return Err(
259                      "pruning non-argument parenthetical atom, used as RHS of invisible times"
260                        .into(),
261                    );
262                  } else {
263                    match args.trees().first() {
264                      Some(XM::Lexeme(_, lhs_meta)) if lhs_meta.fenced.is_none() => {
265                        return Err(
266                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
267                           times"
268                            .into(),
269                        );
270                      },
271                      Some(XM::Apply(_, _, _, lhs_meta)) if lhs_meta.fenced.is_none() => {
272                        return Err(
273                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
274                           times"
275                            .into(),
276                        );
277                      },
278                      _ => {},
279                    }
280                  }
281                }
282              }
283            }
284          }
285        }
286      },
287      _ => {},
288    }
289  }
290  Ok(())
291}

293/// If we have two standalone letters in the same, such as "A x" or "F X", prune parses that
294/// interpret them as an application, unless they have a role that clearly indicates the LHS is
295/// intended as a functional. This may need to be applied as an optional filter at the end of
296/// pruning, as there are known, albeit rare, counter-examples ("f x").
297fn pragma_unfenced_letter_arguments_require_visual_cues(tree: &XM) -> Result<(), Box<dyn Error>> {
298  match *tree {
299    XM::Apply(Operator(ref op), ref args, ..) if args.0.len() == 1 => {
300      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
301        if atom_meta.fenced.is_none() {
302          let (arg_base, _sep, lexeme) = distill_lexeme(arg_name);
303          let arg_letter: char = if lexeme.len() == 1 {
304            lexeme.chars().next().unwrap()
305          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
306            greek_letter
307          } else {
308            return Ok(()); // rule only applies to single char cases
309          };
310          if arg_base.starts_with("OPERATOR") {
311            // for now single letter OPERATOR arguments just don't make sense without parentheses
312            // e.g. OPERATOR:italic-delta is a differential, and wouldn't be used carelessly as an
313            // arg
314            return Err("single letter argument with role OPERATOR is unusual, prune.".into());
315          }
316          let op_name = op.base_operator_name();
317          let (base, _sep, lexeme) = distill_lexeme(&op_name);
318          if base.starts_with("FUNCTION") || base.starts_with("OPERATOR") {
319            return Ok(()); // don't doubt dedicated roles
320          }
321
322          let op_letter: char = if lexeme.len() == 1 {
323            lexeme.chars().next().unwrap()
324          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
325            greek_letter
326          } else {
327            return Ok(()); // rule only applies to single char cases
328          };
329          if (op_letter.is_lowercase() && arg_letter.is_lowercase())
330            || (op_letter.is_uppercase() && arg_letter.is_uppercase())
331          {
332            return Err("operator and argument are visually similar, prune.".into());
333          }
334        }
335      }
336    },
337    _ => {},
338  };
339  Ok(())
340}

342/// OPFUNCTION are explicitly marked to be operators in an application, they should not be in an
343/// argument role of a single argument Apply. Note that this pragma still allows OPFUNCTIONs to
344/// appear as list elements.
345fn pragma_opfunctions_are_rarely_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
346  match *tree {
347    XM::Apply(_, ref args, ..) if args.0.len() == 1 => {
348      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
349        if arg_name.starts_with("OPFUNCTION") && atom_meta.fenced.is_none() {
350          return Err("OPFUNCTIONs are rarely arguments, prune.".into());
351        }
352      }
353    },
354    _ => {},
355  }
356  Ok(())
357}

359/// Higher order "ID ( FUNCTION )" constructs occassionally show up in the
360/// current DLMF markup, and should remain valid. But they are exceptions.
361/// Meanwhile, a reading of
362/// "ID FUNCTION" as an ID "applied to" the FUNCTION should be invalid.
363/// For now this looks easiest to achieve via a pragmatic prune,
364/// Mostly since we can't enforce IDs to strictly have "curry=1", rather
365/// we need them with "curry >= 1", to stay a little more lenient.
366fn pragma_higher_order_ids_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
367  if let XM::Apply(ref op, ref args, ..) = tree {
368    if args.0.len() == 1 {
369      match &*op.0 {
370        XM::Lexeme(ref op_name, _) if op_name.starts_with("ID:") => match args.0[0] {
371          Some(XM::Lexeme(ref rhs_name, ref rhs_meta)) => {
372            if rhs_name.starts_with("FUNCTION") && rhs_meta.fenced.is_none() {
373              return Err("ID of a higher order than a FUNCTION is not allowed.".into());
374            }
375            if rhs_name.starts_with("ID:") && rhs_meta.fenced.is_none() {
376              return Err("ID applied to unfenced ID is highly unusual, prune.".into());
377            }
378          },
379          Some(XM::Apply(..)) => {
380            return Err("ID of a higher order shouldn't accept any compound trees.".into());
381          },
382          _ => {},
383        },
384        _ => {},
385      }
386    }
387  }
388  Ok(())
389}

391/// If we see " h f (x,y) " a parse where
392/// "h f" are invisible multiplied, is not our first choice.
393/// A right-associative application h(f(x,y)) is instead.
394/// Prune it out if possible.
395fn pragma_higher_order_invisible_ops_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
396  if let XM::Apply(Operator(op), ref args, ..) = tree {
397    match **op {
398      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
399        let trees = args.trees();
400        if trees.len() == 2 {
401          let lhs = trees[0];
402          let rhs = trees[1];
403          if let XM::Lexeme(ref lhs_name, _) = lhs.get_baseline() {
404            if let XM::Lexeme(ref rhs_name, _) = rhs.get_baseline() {
405              if name_is_functional_or_id(lhs_name) && name_is_functional(rhs_name) {
406                return Err(
407                  "Pruning higher order 'FUNCTION x FUNCTION' parse to give precedence to \
408                   right-associative readings"
409                    .into(),
410                );
411              }
412            }
413          }
414        }
415      },
416      _ => {},
417    }
418  }
419
420  Ok(())
421}

423/// If two numbers are left next to each other, as in "10(5)" it is rarely (never?) the intention
424/// that they are to be multiplied Prune such parses. We have special rules for some notations, such
425/// as "dlmf_range".
426fn pragma_adjacent_numbers_dont_use_invisible_times(tree: &XM) -> Result<(), Box<dyn Error>> {
427  if let XM::Apply(Operator(op), ref args, ..) = tree {
428    match **op {
429      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
430        let arg_trees = args.trees();
431        if arg_trees.len() == 2 {
432          if let Some(lhs) = arg_trees.first() {
433            if lhs.base_operator_name().starts_with("NUMBER") {
434              if let Some(rhs) = arg_trees.get(1) {
435                if rhs.base_operator_name().starts_with("NUMBER") {
436                  return Err(
437                    "pruning two adjacent NUMBERs that used an invisible operator".into(),
438                  );
439                }
440              }
441            }
442          }
443        }
444      },
445      _ => {},
446    }
447  }
448  Ok(())
449}

459          if numlexeme.starts_with("DIFFOP") {
460            return Err("pruning standalone diffops are not numerators".into());
461          }
462        }
463      },
464      _ => {},
465    }
466  }
467  Ok(())

470/// Constructs such as " e_i z_0 " are rarely applying e(z). This is even a common expectation of
471/// function symbols say " f_i g_j ". This pragma prunes trees that applies two adjacent scripted
472/// constructs.
473fn pragma_adjacent_unfenced_scripts_dont_apply(tree: &XM) -> Result<(), Box<dyn Error>> {
474  if let XM::Apply(Operator(op), ref args, ..) = tree {
475    if args.trees().len() == 1 {
476      if let XM::Apply(Operator(ref op_op), _, _, ref op_meta) = **op {
477        let op_base_name = op_op.base_operator_name();
478        if op_meta.fenced.is_none() && (op_base_name == "unknown.subscript") {
479          // only subscripts on the outer one to avoid pruning e.g. \nabla^2 u_{0,0}
480          if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = args.trees().first() {
481            let arg_op_name = arg_op.base_operator_name();
482            if arg_meta.fenced.is_none()
483              && (arg_op_name == "unknown.subscript" || arg_op_name == "unknown.superscript")
484            {
485              return Err("Prune: adjacent unfenced scripts do not form an application.".into());
486            }
487          }
488        }
489      }
490    }
491  }
492  Ok(())
493}

495/// Adjacent functions don't unify into a single operator
496/// as they are meant to apply right-to-left, one-by-one
497fn pragma_adjacent_functions_dont_unify_into_op(tree: &XM) -> Result<(), Box<dyn Error>> {
498  if let XM::Apply(Operator(op), ..) = tree {
499    if let XM::Apply(Operator(inner_op), inner_args, _, inner_meta) = &**op {
500      if inner_meta.fenced.is_none() {
501        if let XM::Lexeme(name, _) = inner_op.get_baseline() {
502          if name_is_functional_or_id(name) {
503            let inner_trees = inner_args.trees();
504            if inner_trees.len() == 1 {
505              if let XM::Lexeme(rhs_name, _) = inner_args.trees().first().unwrap().get_baseline() {
506                if name_is_functional(rhs_name) {
507                  return Err(
508                    "Two applied FUNCTIONS as operator violates right-associative behavior.".into(),
509                  );
510                }
511              }
512            }
513          }
514        }
515      }
516    }
517  }
518  Ok(())
519}
520/// Postfix
521fn pragma_postfix_terms_are_fenced_if_single_arg(tree: &XM) -> Result<(), Box<dyn Error>> {
522  if let XM::Apply(Operator(op), ref args, ..) = tree {
523    let arg_trees = args.trees();
524    if arg_trees.len() == 1 {
525      if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = arg_trees.first() {
526        if arg_op.base_operator_name().starts_with("POSTFIX") && arg_meta.fenced.is_none() {
527          let op_name = op.base_operator_name();
528          if op_name.starts_with("ID")
529            || op_name.starts_with("UNKNOWN")
530            || op_name.starts_with("FUNCTION")
531            || op_name.starts_with("OPERATOR")
532            || op_name.starts_with("DIFFOP")
533          {
534            return Err("pruning postfix term used as single argument without fences".into());
535          }
536        }
537      }
538    }
539  }
540  Ok(())
541}

543/// Scripts are inherently relational — they modify a base. A standalone floating
544/// script (_b by itself, with base=absent) indicates a pre-script that should
545/// attach to a subsequent base. Prefer parses where scripts are attached to
546/// bases over parses where scripts float independently.
547///
548/// This is a universal mathematical convention: in `_b^a A^c_d`, the `_b^a`
549/// are pre-scripts of A, not standalone subscript/superscript expressions.
550fn pragma_maximize_script_attachment(tree: &XM) -> Result<(), Box<dyn Error>> {
551  // Detect any script-op Apply where the first arg (base) is "absent".
552  // This pattern means a standalone floating script that should be a pre-script.
553  if let XM::Apply(Operator(ref op), ref args, _, _) = tree {
554    if let XM::Token(ref props, _) = **op {
555      if props.role.as_deref().is_some_and(|r|
556        r == "SUBSCRIPTOP" || r == "SUPERSCRIPTOP")
557      {
558        // Check if base (first arg) is "absent" — standalone script
559        if let Some(Some(XM::Token(ref base_props, _))) = args.0.first() {
560          if base_props.meaning.as_deref() == Some("absent") {
561            return Err(
562              "Prune: standalone floating script (base=absent) should attach as pre-script.".into()
563            );
564          }
565        }
566      }
567    }
568  }
569  Ok(())
570}

572/// An expression with `absent` on BOTH sides of a relational/binary operator is
573/// never valid mathematical notation. `absent < a` (prefix relop) or `a > absent`
574/// (postfix relop) can occur, but `absent < a > absent` is two missing operands —
575/// too tortured. This prunes `< a >` from being parsed as a bilateral relation chain,
576/// allowing QM expectation `<a>` to win instead.
577fn pragma_no_bilateral_absent(tree: &XM) -> Result<(), Box<dyn Error>> {
578  if let XM::Apply(_, ref args, ..) = tree {
579    let trees = args.trees();
580    if trees.len() >= 2 {
581      let first_absent = matches!(trees.first(),
582        Some(XM::Token(ref p, _)) if p.meaning.as_deref() == Some("absent"));
583      let last_absent = matches!(trees.last(),
584        Some(XM::Token(ref p, _)) if p.meaning.as_deref() == Some("absent"));
585      if first_absent && last_absent {
586        return Err(
587          "Prune: bilateral absent (absent on both sides) is never valid notation.".into()
588        );
589      }
590    }
591  }
592  Ok(())
593}
594
595/// Functions prefer wider argument absorption: `\log 2x^2` means `log(2*x^2)`.
596/// Prune parses where a function application is immediately followed by invisible
597/// multiplication with unfenced factors. The competing parse would have the function
598/// absorb those factors as arguments.
599fn pragma_functions_prefer_wider_absorption(tree: &XM) -> Result<(), Box<dyn Error>> {
600  // Reject: invisible_times(function_app(f, narrow_arg), simple_rhs)
601  // This forces the wider parse: function_app(f, narrow_arg * rhs)
602  // Only fires when RHS is a simple factor (Lexeme/Token/Wrap), NOT another
603  // function application — chained functions like sin(πx)*cos(2πy) should stay separate.
604  if let XM::Apply(Operator(op), ref args, ..) = tree {
605    let is_invisible_times = match **op {
606      XM::Lexeme(ref oplexeme, _) => oplexeme.contains("invisible_operator"),
607      XM::Token(ref props, _) => {
608        props.meaning.as_deref() == Some("times")
609          && props.role.as_deref() == Some("MULOP")
610      },
611      _ => false,
612    };
613    if is_invisible_times {
614      let trees = args.trees();
615      if trees.len() == 2 {
616        // LHS is a function application (Apply(function, arg))
617        if let XM::Apply(Operator(ref func_op), ref func_args, _, ref func_meta) = trees[0] {
618          // Check if the function op is FUNCTION/OPFUNCTION/TRIGFUNCTION
619          let func_name = func_op.base_operator_name();
620          if (func_name.starts_with("OPFUNCTION") || func_name.starts_with("TRIGFUNCTION"))
621            && func_meta.fenced.is_none()
622            && func_args.trees().len() == 1
623          {
624            // RHS must be a simple factor — not another function application
625            // or compound expression. Scripted factors (x^2) are simple.
626            let rhs = trees[1];
627            let rhs_is_simple = match rhs {
628              XM::Lexeme(..) | XM::Token(..) | XM::Wrap(..) => true,
629              XM::Apply(Operator(ref rhs_op), _, _, _) => {
630                // Scripted factors (SUPERSCRIPTOP/SUBSCRIPTOP) are simple
631                let rhs_role = match &**rhs_op {
632                  XM::Token(ref props, _) => props.role.as_deref().unwrap_or(""),
633                  XM::Lexeme(ref lex, _) => lex.split(':').next().unwrap_or(""),
634                  _ => "",
635                };
636                rhs_role == "SUPERSCRIPTOP" || rhs_role == "SUBSCRIPTOP"
637              },
638              _ => false,
639            };
640            let rhs_meta = rhs.get_meta();
641            if rhs_is_simple && rhs_meta.fenced.is_none() {
642              return Err(
643                "Prune: function application followed by simple unfenced factor — \
644                 prefer wider absorption.".into()
645              );
646            }
647          }
648        }
649      }
650    }
651  }
652  Ok(())
653}

655/// Bigop prefer wider absorption: reject mulop(bigop_app(narrow), rhs)
656/// when rhs is a simple factor that could have been part of the bigop's argument.
657/// Perl's moreOpArgFactors absorbs MulOp chains into bigop arguments.
658fn pragma_bigop_prefer_wider_absorption(tree: &XM) -> Result<(), Box<dyn Error>> {
659  // Pattern: mulop(bigop_app, simple_rhs) or invisible_times(bigop_app, simple_rhs)
660  if let XM::Apply(Operator(op), ref args, ..) = tree {
661    let is_mulop = match **op {
662      XM::Token(ref props, _) => {
663        props.role.as_deref() == Some("MULOP")
664      },
665      XM::Lexeme(ref lex, _) => {
666        lex.starts_with("MULOP") || lex.contains("invisible_operator")
667      },
668      _ => false,
669    };
670    if is_mulop {
671      let trees = args.trees();
672      if trees.len() == 2 {
673        // LHS is a bigop application (Apply with BIGOP/SUMOP/INTOP/LIMITOP/DIFFOP op)
674        if let XM::Apply(Operator(ref bigop_op), _, _, _) = trees[0] {
675          let bigop_name = bigop_op.base_operator_name();
676          let is_bigop = bigop_name.starts_with("BIGOP")
677            || bigop_name.starts_with("SUMOP")
678            || bigop_name.starts_with("INTOP")
679            || bigop_name.starts_with("LIMITOP")
680            || bigop_name.starts_with("DIFFOP");
681          if is_bigop {
682            // RHS should be a simple factor, not another bigop or function
683            let rhs = trees[1];
684            let rhs_is_simple = match rhs {
685              XM::Lexeme(..) | XM::Token(..) | XM::Wrap(..) => true,
686              XM::Apply(Operator(ref rhs_op), _, _, _) => {
687                let rhs_role = match &**rhs_op {
688                  XM::Token(ref props, _) => props.role.as_deref().unwrap_or(""),
689                  XM::Lexeme(ref lex, _) => lex.split(':').next().unwrap_or(""),
690                  _ => "",
691                };
692                // Scripted factors and invisible_times products are simple
693                rhs_role == "SUPERSCRIPTOP"
694                  || rhs_role == "SUBSCRIPTOP"
695                  || rhs_role == "MULOP"
696                  || rhs_role == "DIFFOP"
697              },
698              _ => false,
699            };
700            if rhs_is_simple {
701              return Err(
702                "Prune: bigop application followed by mulop factor — \
703                 prefer wider bigop absorption."
704                  .into(),
705              );
706            }
707          }
708        }
709      }
710    }
711  }
712  Ok(())
713}

715fn pragma_restrict_numeral_fractions(tree: &XM) -> Result<(), Box<dyn Error>> {
716  if let XM::Apply(Operator(op), ref args, ..) = tree {
717    match **op {
718      XM::Lexeme(ref oplexeme, _) if oplexeme == "arith1.divide" => {
719        let arg_trees = args.trees();
720        if arg_trees.len() == 2 {
721          if let XM::Lexeme(arg1_name, arg1_meta) = arg_trees[0] {
722            if let XM::Lexeme(arg2_name, arg2_meta) = arg_trees[1] {
723              if arg1_name.starts_with("NUMBER")
724                && arg2_name.starts_with("NUMBER")
725                && !arg1_meta.syntax_trace.is_empty()
726                || !arg2_meta.syntax_trace.is_empty()
727              {
728                return Err(
729                  "only tokens are allowed in numeric fractions, derived rules are pruned to \
730                   avoid redundancy."
731                    .into(),
732                );
733              }
734            }
735          }
736        }
737      },
738      _ => {},
739    }
740  }
741  Ok(())
742}

810static _PRAGMATIC_BLOCK_MAP: Lazy<HashMap<char, String>> = Lazy::new(|| {
811  // generally, we can observe that the latin alphabet shares "intent"
812  // in blocks of 3 letter in mathematics,
813  // as a fast-and-loose rule of thumb. a-e is an exception as
814  // a rather stable 5 letter block with shared utility.
815  let mut map = HashMap::default();
816  // |a b c d e | f g h |i j k| |l m n| |o p q| |r s t| |u v w| |x y z|
817  let latin_blocks = [
818    ('a', 'e'),
819    ('f', 'h'),
820    ('i', 'k'),
821    ('l', 'n'),
822    ('o', 'q'),
823    ('r', 't'),
824    ('u', 'w'),
825    ('x', 'z'),
826  ];
827  // |α β γ| δ | ϵ | ζ η θ | ι κ | λ μ ν ξ | ο π ρ | σ τ υ | ϕ χ ψ | ω
828  let greek_blocks = [
829    ('α', 'γ'),
830    ('ζ', 'θ'),
831    ('ι', 'κ'),
832    ('λ', 'ξ'),
833    ('ο', 'ρ'),
834    ('σ', 'υ'),
835    ('ϕ', 'ψ'),
836  ];
837  // | Α Β Γ | Δ | Ε | Ζ Η Θ | Ι Κ | Λ Μ Ν Ξ | Ο Π Ρ | Σ Τ Υ | Φ Χ Ψ | Ω
838  let up_greek_blocks = [
839    ('Α', 'Γ'),
840    ('Ζ', 'Θ'),
841    ('Ι', 'Κ'),
842    ('Λ', 'Ξ'),
843    ('Ο', 'Ρ'),
844    ('Σ', 'Υ'),
845    ('Φ', 'Ψ'),
846  ];
847  for (start, end) in latin_blocks.iter() {
848    let mark = format!("{start}{end}");
849    let up_start = start.to_ascii_uppercase();
850    let up_end = end.to_ascii_uppercase();
851    for c_u8 in (*start as u8)..=(*end as u8) {
852      map.insert(c_u8.into(), mark.clone());
853    }
854    let up_mark = format!("{up_start}{up_end}");
855    for c_u8 in (up_start as u8)..=(up_end as u8) {
856      map.insert(c_u8.into(), up_mark.clone());
857    }
858  }
859  for (start, end) in greek_blocks.iter().chain(up_greek_blocks.iter()) {
860    let mark = format!("{start}{end}");
861    for c_u32 in (*start as u32)..=(*end as u32) {
862      map.insert(std::char::from_u32(c_u32).unwrap(), mark.clone());
863    }
864  }
865  map
866});
