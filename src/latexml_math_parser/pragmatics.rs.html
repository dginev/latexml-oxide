8/// Different supported pragmatics according to which to enforce consistency.
9/// Usually one would use them in increasing generality, iterating over a parse forest
10/// until a sufficiently small number of parses is reached.
11#[derive(Copy, Clone, Debug)]
12pub enum ValidationPragmatics {
13  ConsistentLetterBlocks,
14  ConsistentCase,
15  ConsistentCaseFlat,
16  ConsistentCaseFlatUnstyled,
17  InconsistentNames, /* TODO: Since name consistency is built into the curry levels, using the
18                      * inconsistent names pragma should ensure each LP variable has a globally
19                      * unique name */
20  FencedAtomsAreNotFunctions,
21  FencedLettersAreFunctionArguments,
22  AdjacentNumbersDontMultiply,
23  AdjacentUnfencedScriptsDontApply,
24  AdjacentFunctionsDontUnifyIntoOperator,
25  OpfunctionsAreRarelyArguments,
26  UnfencedLetterArgumentsRequireVisualCues,
27  HigherOrderIDsAreExceptions,
28  HigherOrderInvisibleOpsAreExceptions,
29  StandaloneDiffopsAreNotNumerators,
30  PostfixTermsAreFencedIfSingleArguments,
31  RestrictNumeralFractions,
32}

34impl ValidationPragmatics {
35  /// Pragmatic rules that are *always* strictly enforced
36  pub fn expert_defaults() -> Vec<Self> {
37    use ValidationPragmatics::*;
38    vec![
39      FencedAtomsAreNotFunctions,
40      FencedLettersAreFunctionArguments,
41      UnfencedLetterArgumentsRequireVisualCues,
42      OpfunctionsAreRarelyArguments,
43      AdjacentNumbersDontMultiply,
44      StandaloneDiffopsAreNotNumerators,
45      PostfixTermsAreFencedIfSingleArguments,
46      RestrictNumeralFractions,
47    ]
48  }
49  /// Pragmatic rules that are executed at the end of the parse process,
50  /// optionally, until there is a single parse left. If a pragmatic rule invalidates
51  /// all choices, the rule is skipped.
52  pub fn student_defaults() -> Vec<Self> {
53    // Order here is crucial - it shows a decreasing level of adoption of each rule in the
54    // scientific community, ending with some "crutch"-like rules, as last ditch attempts to discard
55    // parses
56    use ValidationPragmatics::*;
57    vec![
58      HigherOrderIDsAreExceptions,
59      HigherOrderInvisibleOpsAreExceptions,
60      AdjacentUnfencedScriptsDontApply,
61      AdjacentFunctionsDontUnifyIntoOperator,
62      ConsistentLetterBlocks,
63      ConsistentCase,
64      ConsistentCaseFlat,
65      ConsistentCaseFlatUnstyled,
66    ]
67  }

69  /// Employs the selected pragmatic to augment the validation of the core Meta object
70  pub fn validate(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
71    use ValidationPragmatics::*;
72    match self {
73      FencedAtomsAreNotFunctions => pragma_fenced_atoms_are_not_functions(tree),
74      UnfencedLetterArgumentsRequireVisualCues => {
75        pragma_unfenced_letter_arguments_require_visual_cues(tree)
76      },
77
78      OpfunctionsAreRarelyArguments => pragma_opfunctions_are_rarely_arguments(tree),
79
80      FencedLettersAreFunctionArguments => pragma_fenced_letters_are_function_arguments(tree),
81      HigherOrderIDsAreExceptions => pragma_higher_order_ids_are_exceptions(tree),
82      HigherOrderInvisibleOpsAreExceptions => {
83        pragma_higher_order_invisible_ops_are_exceptions(tree)
84      },
85      AdjacentNumbersDontMultiply => pragma_adjacent_numbers_dont_use_invisible_times(tree),
86      AdjacentUnfencedScriptsDontApply => pragma_adjacent_unfenced_scripts_dont_apply(tree),
87      StandaloneDiffopsAreNotNumerators => pragma_standalone_diffops_are_not_numerators(tree),
88      PostfixTermsAreFencedIfSingleArguments => pragma_postfix_terms_are_fenced_if_single_arg(tree),
89      AdjacentFunctionsDontUnifyIntoOperator => pragma_adjacent_functions_dont_unify_into_op(tree),
90      RestrictNumeralFractions => pragma_restrict_numeral_fractions(tree),
91      _ => Ok(()),
92    }
93  }

95  /// Recursively checks a tree at each level for applicable pragmatics.
96  /// All levels must validate for the full tree to be valid.
97  pub fn validate_recursive(&self, tree: &XM) -> Result<(), Box<dyn Error>> {
98    self.validate(tree)?; // top-level check
99    // Recursive check:
100    match tree {
101      XM::Choices(subtrees) => {
102        for subtree in subtrees.iter() {
103          self.validate_recursive(subtree)?
104        }
105      },
106      XM::Apply(Operator(op), args, ..) => {
107        self.validate_recursive(op)?;
108        for arg_subtree in args.trees() {
109          self.validate_recursive(arg_subtree)?;
110        }
111      },
112      _ => {},
113    };
114    Ok(())
115  }
116}

118/// Validate a pragmatic class, as indicated by a string name,
119///   against a context dictionary of known objects updating the dictionary when needed
120///
121/// The pragmatics are only used for single-letter values with factor/function role,
122///   relying on common conventions in mathematical syntax:
123///
124/// The two main alphabets are separated into letter blocks, each block
125///   required to coordinate to the same xarith type where each separate style of a letter
126///   (italic/bold;Uppercase/lowercase) stands for a separate pragmatic block.
127///
128/// For example in "Ax^2 + Bx + C" we can obtain a single parse where A,B,C are coefficients
129fn _pragma_letter_blocks(name: &str) -> String {
130  let (base, sep, lexeme) = distill_lexeme(name);
131  let lexeme_pragmatic = if lexeme.len() == 1 {
132    let letter = lexeme.chars().next().unwrap();
133    match _PRAGMATIC_BLOCK_MAP.get(&letter) {
134      Some(block) => block,
135      None => lexeme,
136    }
137  } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
138    match _PRAGMATIC_BLOCK_MAP.get(&greek_letter) {
139      Some(block_name) => block_name,
140      None => lexeme,
141    }
142  } else {
143    lexeme
144  };
145  base.to_owned() + sep + lexeme_pragmatic
146}
147
148/// Validate a pragmatic class, indicated by the case of a named lexeme
149///
150/// For example in SUx
151fn _pragma_letter_case(name: &str) -> String {
152  let (base, sep, lexeme) = distill_lexeme(name);
153  let lexeme_pragmatic = if lexeme.chars().next().unwrap().is_uppercase() {
154    "U"
155  } else {
156    "l"
157  };
158  base.to_owned() + sep + lexeme_pragmatic
159}

161/// Validate a pragmatic class, indicated by the case of a named lexeme
162/// ALSO disregarding any scripted guards
163/// (i.e. scripted lexemes must coordinate with unscripted ones)
164/// For example in "R S_2"
165fn _pragma_letter_case_flat(name: &str) -> String { _pragma_letter_case(name).replace("sub__", "") }
166
167fn _pragma_letter_case_flat_unstyled(name: &str) -> String {
168  let unstyled_name = match name.rfind('-') {
169    Some(position) => {
170      let (base, trailer) = name.split_at(position);
171      let (_sep, lexeme) = trailer.split_at(1);
172      match base.rfind(':') {
173        Some(position) => {
174          let (role, _discard) = base.split_at(position);
175          format!("{role}:{lexeme}")
176        },
177        None => format!("{base}:{lexeme}"),
178      }
179    },
180    None => name.to_owned(),
181  };
182  _pragma_letter_case_flat(&unstyled_name)
183}

185fn pragma_fenced_atoms_are_not_functions(tree: &XM) -> Result<(), Box<dyn Error>> {
186  if let XM::Apply(Operator(op), ..) = tree {
187    if let XM::Lexeme(ref _lexeme, ref atom_meta) = **op {
188      if let Some(ref fences) = atom_meta.fenced {
189        if fences.as_str() == "parens" {
190          return Err(
191            "pruning non-argument parenthetical atom, used as LHS of function application".into(),
192          );
193        }
194      }
195    }
196  }
197  Ok(())
198}

200fn pragma_fenced_letters_are_function_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
201  if let XM::Apply(Operator(op), ref args, ..) = tree {
202    match **op {
203      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
204        if let Some(top_lhs) = args.trees().first() {
205          if let Some(ref fences) = top_lhs.get_meta().fenced {
206            if fences.as_str() == "parens" {
207              if let XM::Lexeme(lhs_name, _) = top_lhs.get_baseline() {
208                if !lhs_name.starts_with("NUMBER") {
209                  return Err(
210                    "pruning non-argument parenthetical atom, used as LHS of invisible times"
211                      .into(),
212                  );
213                }
214              }
215            }
216          }
217
218          // Slightly tricky check -- the top RHS needs to be fenced, but we care about the
219          // "baseline" content being a variable - disregarding scripts.
220          if let Some(top_rhs) = args.trees().get(1) {
221            if let XM::Lexeme(rhs_name, _) = top_rhs.get_baseline() {
222              if let Some(ref fences) = top_rhs.get_meta().fenced {
223                if fences.as_str() == "parens" {
224                  // if the RHS is a number, prune unless the LHS is fenced (things like cycle
225                  // notation)
226                  if !rhs_name.starts_with("NUMBER") {
227                    return Err(
228                      "pruning non-argument parenthetical atom, used as RHS of invisible times"
229                        .into(),
230                    );
231                  } else {
232                    match args.trees().first() {
233                      Some(XM::Lexeme(_, lhs_meta)) if lhs_meta.fenced.is_none() => {
234                        return Err(
235                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
236                           times"
237                            .into(),
238                        );
239                      },
240                      Some(XM::Apply(_, _, _, lhs_meta)) if lhs_meta.fenced.is_none() => {
241                        return Err(
242                          "pruning non-argument parenthetical NUMBER, used as RHS of invisible \
243                           times"
244                            .into(),
245                        );
246                      },
247                      _ => {},
248                    }
249                  }
250                }
251              }
252            }
253          }
254        }
255      },
256      _ => {},
257    }
258  }
259  Ok(())
260}

262/// If we have two standalone letters in the same, such as "A x" or "F X", prune parses that
263/// interpret them as an application, unless they have a role that clearly indicates the LHS is
264/// intended as a functional. This may need to be applied as an optional filter at the end of
265/// pruning, as there are known, albeit rare, counter-examples ("f x").
266fn pragma_unfenced_letter_arguments_require_visual_cues(tree: &XM) -> Result<(), Box<dyn Error>> {
267  match *tree {
268    XM::Apply(Operator(ref op), ref args, ..) if args.0.len() == 1 => {
269      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
270        if atom_meta.fenced.is_none() {
271          let (arg_base, _sep, lexeme) = distill_lexeme(arg_name);
272          let arg_letter: char = if lexeme.len() == 1 {
273            lexeme.chars().next().unwrap()
274          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
275            greek_letter
276          } else {
277            return Ok(()); // rule only applies to single char cases
278          };
279          if arg_base.starts_with("OPERATOR") {
280            // for now single letter OPERATOR arguments just don't make sense without parentheses
281            // e.g. OPERATOR:italic-delta is a differential, and wouldn't be used carelessly as an
282            // arg
283            return Err("single letter argument with role OPERATOR is unusual, prune.".into());
284          }
285          let op_name = op.base_operator_name();
286          let (base, _sep, lexeme) = distill_lexeme(&op_name);
287          if base.starts_with("FUNCTION") || base.starts_with("OPERATOR") {
288            return Ok(()); // don't doubt dedicated roles
289          }
290
291          let op_letter: char = if lexeme.len() == 1 {
292            lexeme.chars().next().unwrap()
293          } else if let Some(greek_letter) = greek_name_to_letter(lexeme) {
294            greek_letter
295          } else {
296            return Ok(()); // rule only applies to single char cases
297          };
298          if (op_letter.is_lowercase() && arg_letter.is_lowercase())
299            || (op_letter.is_uppercase() && arg_letter.is_uppercase())
300          {
301            return Err("operator and argument are visually similar, prune.".into());
302          }
303        }
304      }
305    },
306    _ => {},
307  };
308  Ok(())
309}

311/// OPFUNCTION are explicitly marked to be operators in an application, they should not be in an
312/// argument role of a single argument Apply. Note that this pragma still allows OPFUNCTIONs to
313/// appear as list elements.
314fn pragma_opfunctions_are_rarely_arguments(tree: &XM) -> Result<(), Box<dyn Error>> {
315  match *tree {
316    XM::Apply(_, ref args, ..) if args.0.len() == 1 => {
317      if let Some(XM::Lexeme(ref arg_name, ref atom_meta)) = args.0[0] {
318        if arg_name.starts_with("OPFUNCTION") && atom_meta.fenced.is_none() {
319          return Err("OPFUNCTIONs are rarely arguments, prune.".into());
320        }
321      }
322    },
323    _ => {},
324  }
325  Ok(())
326}

328/// Higher order "ID ( FUNCTION )" constructs occassionally show up in the
329/// current DLMF markup, and should remain valid. But they are exceptions.
330/// Meanwhile, a reading of
331/// "ID FUNCTION" as an ID "applied to" the FUNCTION should be invalid.
332/// For now this looks easiest to achieve via a pragmatic prune,
333/// Mostly since we can't enforce IDs to strictly have "curry=1", rather
334/// we need them with "curry >= 1", to stay a little more lenient.
335fn pragma_higher_order_ids_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
336  if let XM::Apply(ref op, ref args, ..) = tree {
337    if args.0.len() == 1 {
338      match &*op.0 {
339        XM::Lexeme(ref op_name, _) if op_name.starts_with("ID:") => match args.0[0] {
340          Some(XM::Lexeme(ref rhs_name, ref rhs_meta)) => {
341            if rhs_name.starts_with("FUNCTION") && rhs_meta.fenced.is_none() {
342              return Err("ID of a higher order than a FUNCTION is not allowed.".into());
343            }
344            if rhs_name.starts_with("ID:") && rhs_meta.fenced.is_none() {
345              return Err("ID applied to unfenced ID is highly unusual, prune.".into());
346            }
347          },
348          Some(XM::Apply(..)) => {
349            return Err("ID of a higher order shouldn't accept any compound trees.".into());
350          },
351          _ => {},
352        },
353        _ => {},
354      }
355    }
356  }
357  Ok(())
358}

360/// If we see " h f (x,y) " a parse where
361/// "h f" are invisible multiplied, is not our first choice.
362/// A right-associative application h(f(x,y)) is instead.
363/// Prune it out if possible.
364fn pragma_higher_order_invisible_ops_are_exceptions(tree: &XM) -> Result<(), Box<dyn Error>> {
365  if let XM::Apply(Operator(op), ref args, ..) = tree {
366    match **op {
367      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
368        let trees = args.trees();
369        if trees.len() == 2 {
370          let lhs = trees[0];
371          let rhs = trees[1];
372          if let XM::Lexeme(ref lhs_name, _) = lhs.get_baseline() {
373            if let XM::Lexeme(ref rhs_name, _) = rhs.get_baseline() {
374              if name_is_functional_or_id(lhs_name) && name_is_functional(rhs_name) {
375                return Err(
376                  "Pruning higher order 'FUNCTION x FUNCTION' parse to give precedence to \
377                   right-associative readings"
378                    .into(),
379                );
380              }
381            }
382          }
383        }
384      },
385      _ => {},
386    }
387  }
388
389  Ok(())
390}

392/// If two numbers are left next to each other, as in "10(5)" it is rarely (never?) the intention
393/// that they are to be multiplied Prune such parses. We have special rules for some notations, such
394/// as "dlmf_range".
395fn pragma_adjacent_numbers_dont_use_invisible_times(tree: &XM) -> Result<(), Box<dyn Error>> {
396  if let XM::Apply(Operator(op), ref args, ..) = tree {
397    match **op {
398      XM::Lexeme(ref oplexeme, _) if oplexeme == "x.invisible_operator" => {
399        let arg_trees = args.trees();
400        if arg_trees.len() == 2 {
401          if let Some(lhs) = arg_trees.first() {
402            if lhs.base_operator_name().starts_with("NUMBER") {
403              if let Some(rhs) = arg_trees.get(1) {
404                if rhs.base_operator_name().starts_with("NUMBER") {
405                  return Err(
406                    "pruning two adjacent NUMBERs that used an invisible operator".into(),
407                  );
408                }
409              }
410            }
411          }
412        }
413      },
414      _ => {},
415    }
416  }
417  Ok(())
418}

439/// Constructs such as " e_i z_0 " are rarely applying e(z). This is even a common expectation of
440/// function symbols say " f_i g_j ". This pragma prunes trees that applies two adjacent scripted
441/// constructs.
442fn pragma_adjacent_unfenced_scripts_dont_apply(tree: &XM) -> Result<(), Box<dyn Error>> {
443  if let XM::Apply(Operator(op), ref args, ..) = tree {
444    if args.trees().len() == 1 {
445      if let XM::Apply(Operator(ref op_op), _, _, ref op_meta) = **op {
446        let op_base_name = op_op.base_operator_name();
447        if op_meta.fenced.is_none() && (op_base_name == "unknown.subscript") {
448          // only subscripts on the outer one to avoid pruning e.g. \nabla^2 u_{0,0}
449          if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = args.trees().first() {
450            let arg_op_name = arg_op.base_operator_name();
451            if arg_meta.fenced.is_none()
452              && (arg_op_name == "unknown.subscript" || arg_op_name == "unknown.superscript")
453            {
454              return Err("Prune: adjacent unfenced scripts do not form an application.".into());
455            }
456          }
457        }
458      }
459    }
460  }
461  Ok(())
462}

464/// Adjacent functions don't unify into a single operator
465/// as they are meant to apply right-to-left, one-by-one
466fn pragma_adjacent_functions_dont_unify_into_op(tree: &XM) -> Result<(), Box<dyn Error>> {
467  if let XM::Apply(Operator(op), ..) = tree {
468    if let XM::Apply(Operator(inner_op), inner_args, _, inner_meta) = &**op {
469      if inner_meta.fenced.is_none() {
470        if let XM::Lexeme(name, _) = inner_op.get_baseline() {
471          if name_is_functional_or_id(name) {
472            let inner_trees = inner_args.trees();
473            if inner_trees.len() == 1 {
474              if let XM::Lexeme(rhs_name, _) = inner_args.trees().first().unwrap().get_baseline() {
475                if name_is_functional(rhs_name) {
476                  return Err(
477                    "Two applied FUNCTIONS as operator violates right-associative behavior.".into(),
478                  );
479                }
480              }
481            }
482          }
483        }
484      }
485    }
486  }
487  Ok(())
488}
489/// Postfix
490fn pragma_postfix_terms_are_fenced_if_single_arg(tree: &XM) -> Result<(), Box<dyn Error>> {
491  if let XM::Apply(Operator(op), ref args, ..) = tree {
492    let arg_trees = args.trees();
493    if arg_trees.len() == 1 {
494      if let Some(XM::Apply(Operator(arg_op), _, _, arg_meta)) = arg_trees.first() {
495        if arg_op.base_operator_name().starts_with("POSTFIX") && arg_meta.fenced.is_none() {
496          let op_name = op.base_operator_name();
497          if op_name.starts_with("ID")
498            || op_name.starts_with("UNKNOWN")
499            || op_name.starts_with("FUNCTION")
500            || op_name.starts_with("OPERATOR")
501            || op_name.starts_with("DIFFOP")
502          {
503            return Err("pruning postfix term used as single argument without fences".into());
504          }
505        }
506      }
507    }
508  }
509  Ok(())
510}

512fn pragma_restrict_numeral_fractions(tree: &XM) -> Result<(), Box<dyn Error>> {
513  if let XM::Apply(Operator(op), ref args, ..) = tree {
514    match **op {
515      XM::Lexeme(ref oplexeme, _) if oplexeme == "arith1.divide" => {
516        let arg_trees = args.trees();
517        if arg_trees.len() == 2 {
518          if let XM::Lexeme(arg1_name, arg1_meta) = arg_trees[0] {
519            if let XM::Lexeme(arg2_name, arg2_meta) = arg_trees[1] {
520              if arg1_name.starts_with("NUMBER")
521                && arg2_name.starts_with("NUMBER")
522                && !arg1_meta.syntax_trace.is_empty()
523                || !arg2_meta.syntax_trace.is_empty()
524              {
525                return Err(
526                  "only tokens are allowed in numeric fractions, derived rules are pruned to \
527                   avoid redundancy."
528                    .into(),
529                );
530              }
531            }
532          }
533        }
534      },
535      _ => {},
536    }
537  }
538  Ok(())
539}

596pub fn name_is_functional(name: &str) -> bool {
597  name.starts_with("FUNCTION")
598    || name.starts_with("OPFUNCTION")
599    || name.starts_with("TRIGFUNCTION")
600    || name.starts_with("UNKNOWN")
601}
602
603pub fn name_is_functional_or_id(name: &str) -> bool {
604  name.starts_with("ID") || name_is_functional(name)
605}

607static _PRAGMATIC_BLOCK_MAP: Lazy<HashMap<char, String>> = Lazy::new(|| {
608  // generally, we can observe that the latin alphabet shares "intent"
609  // in blocks of 3 letter in mathematics,
610  // as a fast-and-loose rule of thumb. a-e is an exception as
611  // a rather stable 5 letter block with shared utility.
612  let mut map = HashMap::default();
613  // |a b c d e | f g h |i j k| |l m n| |o p q| |r s t| |u v w| |x y z|
614  let latin_blocks = [
615    ('a', 'e'),
616    ('f', 'h'),
617    ('i', 'k'),
618    ('l', 'n'),
619    ('o', 'q'),
620    ('r', 't'),
621    ('u', 'w'),
622    ('x', 'z'),
623  ];
624  // |α β γ| δ | ϵ | ζ η θ | ι κ | λ μ ν ξ | ο π ρ | σ τ υ | ϕ χ ψ | ω
625  let greek_blocks = [
626    ('α', 'γ'),
627    ('ζ', 'θ'),
628    ('ι', 'κ'),
629    ('λ', 'ξ'),
630    ('ο', 'ρ'),
631    ('σ', 'υ'),
632    ('ϕ', 'ψ'),
633  ];
634  // | Α Β Γ | Δ | Ε | Ζ Η Θ | Ι Κ | Λ Μ Ν Ξ | Ο Π Ρ | Σ Τ Υ | Φ Χ Ψ | Ω
635  let up_greek_blocks = [
636    ('Α', 'Γ'),
637    ('Ζ', 'Θ'),
638    ('Ι', 'Κ'),
639    ('Λ', 'Ξ'),
640    ('Ο', 'Ρ'),
641    ('Σ', 'Υ'),
642    ('Φ', 'Ψ'),
643  ];
644  for (start, end) in latin_blocks.iter() {
645    let mark = format!("{start}{end}");
646    let up_start = start.to_ascii_uppercase();
647    let up_end = end.to_ascii_uppercase();
648    for c_u8 in (*start as u8)..=(*end as u8) {
649      map.insert(c_u8.into(), mark.clone());
650    }
651    let up_mark = format!("{up_start}{up_end}");
652    for c_u8 in (up_start as u8)..=(up_end as u8) {
653      map.insert(c_u8.into(), up_mark.clone());
654    }
655  }
656  for (start, end) in greek_blocks.iter().chain(up_greek_blocks.iter()) {
657    let mark = format!("{start}{end}");
658    for c_u32 in (*start as u32)..=(*end as u32) {
659      map.insert(std::char::from_u32(c_u32).unwrap(), mark.clone());
660    }
661  }
662  map
663});
