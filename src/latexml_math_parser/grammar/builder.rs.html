1use crate::semantics::*;
2use marpa::grammar::Grammar as MarpaGrammar;
3use marpa::result::Result;
4use marpa::tree_builder::TreeBuilder;
5
6pub fn init_grammar() -> Result<(MarpaGrammar, Actions, TreeBuilder)> {
7  // We create a declarative macro language of our own, in the spirit of the Marpa SLIF
8  default_registry!();
9  // Tokens, to be used in rules directly
10  token!(atom ~ "ATOM");
11  token!(unknown ~ "UNKNOWN");
12  token!(id ~ "ID");
13  token!(array ~ "ARRAY");
14  token!(number ~ "NUMBER");
15  token!(punct ~ "PUNCT");
16  token!(period ~ "PERIOD");
17  token!(addop ~ "ADDOP");
18  token!(mulop ~ "MULOP");
19  token!(relop ~ "RELOP");
20  token!(elideop ~ "ELIDEOP");
21  token!(langle_rel = "RELOP:less-than");
22  token!(langle_open = "OPEN:langle");
23  token!(langle = [langle_rel langle_open]);
24  token!(rangle_rel = "RELOP:greater-than");
25  token!(rangle_close = "CLOSE:rangle");
26  token!(rangle =[rangle_rel rangle_close]);
27  token!(vertbar ~ "VERTBAR");
28  token!(singlevertbar = "VERTBAR:|");
29  token!(middle_bar = "MIDDLE:|");
30  token!(middle_parallel = "MIDDLE:parallel-to");
31  token!(midbar = [vertbar middle_bar middle_parallel]);
32  token!(lbrace = "OPEN:{");
33  token!(rbrace = "CLOSE:}");
34  token!(lparen = "OPEN:(");
35  token!(rparen = "CLOSE:)");
36  token!(lbracket = "OPEN:[");
37  token!(rbracket = "CLOSE:]");
38  token!(relop_equals = "RELOP:equals");
39  token!(metarelop ~ "METARELOP");
40  token!(modifierop ~ "MODIFIEROP");
41  token!(modifier ~ "MODIFIER");
42  token!(arrow ~ "ARROW");
43  token!(binop ~ "BINOP");
44  token!(postfix ~ "POSTFIX");
45  token!(function ~ "FUNCTION");
46  token!(opfunction ~ "OPFUNCTION");
47  token!(trigfunction ~ "TRIGFUNCTION");
48  token!(applyop ~ "APPLYOP");
49  token!(composeop ~ "COMPOSEOP");
50  token!(supop ~ "SUPOP");
51  token!(open ~ "OPEN");
52  token!(close ~ "CLOSE");
53  token!(middle ~ "MIDDLE");
54  token!(bigop ~ "BIGOP");
55  token!(sumop ~ "SUMOP");
56  token!(intop ~ "INTOP");
57  token!(limitop ~ "LIMITOP");
58  token!(diffop ~ "DIFFOP");
59  token!(operator ~ "OPERATOR");
60  token!(start_postsubscript ~ "start_POSTSUBSCRIPT");
61  token!(end_postsubscript ~ "end_POSTSUBSCRIPT");
62  token!(start_postsuperscript ~ "start_POSTSUPERSCRIPT");
63  token!(end_postsuperscript ~ "end_POSTSUPERSCRIPT");
64  token!(start_floatsuperscript ~ "start_FLOATSUPERSCRIPT");
65  token!(end_floatsuperscript ~ "end_FLOATSUPERSCRIPT");
66  token!(start_floatsubscript ~ "start_FLOATSUBSCRIPT");
67  token!(end_floatsubscript ~ "end_FLOATSUBSCRIPT");
68
69  rules!(
70    // Factors
71    // opfunction/function/trigfunction are NOT factors — they require arguments.
72    // Standalone usage is handled at the term level (term += function | ...).
73    // `2 \sin` is handled via dedicated tight_term rules below.
74    factor_base = unknown | number | id | atom;
75    factor = factor_base | opfunction;
76    // Terms
77    // Perl: bigop = BIGOP | SUMOP | INTOP | LIMITOP | DIFFOP
78    any_bigop = bigop | sumop | intop | limitop | diffop;
79    // Adjacent bigops compose like higher-order functions:
80    // \int\iint => integral(double-integral), itself a compound operator
81    composed_bigop = any_bigop any_bigop => prefix_apply
82      | any_bigop composed_bigop => prefix_apply;
83
84    // Compound operators: OPERATOR composed with functions/other operators (right-recursive)
85    // D sin => Apply(D, sin), D D sin => Apply(D, Apply(D, sin))
86    // Must end with a function/trigfunction (no bare operator-only compounds)
87    compound_operator = operator trigfunction => prefix_apply
88      | operator function => prefix_apply
89      | operator compound_operator => prefix_apply;
90
91    // tight_term includes single factors (for left-recursive chaining)
92    // and all compound constructs (invisible times, prefix application, etc.)
93    // The `\log x` → `log*x` issue is handled by semantic pruning in
94    // apply_invisible_times, not at the grammar level.
95    tight_term = factor
96      | tight_term factor => apply_invisible_times
97      // Perl MathGrammar L423: POSTFIX (e.g. n!) => Apply(op, term)
98      | tight_term postfix => apply_postfix
99      | function tight_term => prefix_apply
100      | trigfunction tight_term => prefix_apply
101      | any_bigop tight_term => prefix_apply
102      | composed_bigop tight_term => prefix_apply
103      | compound_operator tight_term => prefix_apply
104      | operator factor => prefix_apply
105      | factor_base applyop tight_term => prefix_apply_applyop;
106
107    // Composed functions: f∘g, sin∘cos — these can then be applied as functions
108    // COMPOSEOP operates on function-level operands (curry level 2)
109    // Left-to-right associative (matching Perl): f∘g∘h = (f∘g)∘h
110    composed_term = function composeop function => infix_apply
111      | function composeop trigfunction => infix_apply
112      | function composeop opfunction => infix_apply
113      | trigfunction composeop function => infix_apply
114      | trigfunction composeop trigfunction => infix_apply
115      | trigfunction composeop opfunction => infix_apply
116      | opfunction composeop function => infix_apply
117      | opfunction composeop trigfunction => infix_apply
118      | opfunction composeop opfunction => infix_apply
119      // Left-recursive for left-to-right associativity
120      | composed_term composeop function => infix_apply
121      | composed_term composeop trigfunction => infix_apply
122      | composed_term composeop opfunction => infix_apply;
123
124    // Composed functions can be applied like regular functions
125    tight_term += composed_term tight_term => prefix_apply;
126
127    term = tight_term
128    | term mulop tight_term => infix_apply_nary
129    | term mulop tight_term elideop => infix_apply_and_elide
130    // Fallback: COMPOSEOP on general terms (for non-function-level composition)
131    | term composeop term => infix_apply
132    | operator applyop term => prefix_apply_applyop;
133
134    // Allow standalone functions/trigfunctions/opfunctions as terms
135    // This is needed for (f*g)(x) where f and g are FUNCTION tokens
136    // opfunction here allows standalone \operatorname{R} to parse
137    term += function | trigfunction | opfunction | composed_term;
138    // Allow elideop (\cdots) as a term for chains like y + i + \cdots + y_n
139    // Perl treats cdots as a regular term in addition chains
140    term += elideop;
141
142    // Higher-order operator terms: functions as standalone objects multiplied by factors
143    // `2\sin` = `2 * sin`, `2\sin\cos` = `2 * sin * cos`
144    // These are term-level (not tight_term) so they don't interfere with
145    // function application: `2\sin x` = `2 * sin(x)` (not `(2*sin) * x`)
146    tight_opterm = factor function => apply_invisible_times
147      | factor trigfunction => apply_invisible_times
148      | factor opfunction => apply_invisible_times
149      | tight_opterm function => apply_invisible_times
150      | tight_opterm trigfunction => apply_invisible_times
151      | tight_opterm opfunction => apply_invisible_times;
152    term += tight_opterm;
153
154    // Expressions
155    expression = term
156      | expression addop term => infix_apply_nary
157      | expression addop term elideop => infix_apply_and_elide
158      | addop tight_term => prefix_apply
159      | factor addop => postfix_apply
160      | expression addop => postfix_apply
161      // Perl MathGrammar L236: addExpressionModifier: MODIFIEROP Expression
162      // => Apply(modifierop, expr, expr2). Handles infix `a mod b`.
163      | expression modifierop expression => infix_apply;
164
165    // Formula
166    // Perl MathGrammar L73/236: MODIFIEROP Expression => Apply(mod, Absent, expr)
167    modifier_expression = modifierop expression => modifier_prefix_apply;
168    // Perl: within a Formula, comma-separated expressions after a relop form a list RHS.
169    // e.g. a=b,c,d → a = list(b,c,d), not list(a=b, c, d).
170    // Uses formula_list_apply which rejects items containing relops (those belong at statement level).
171    formula_list = expression punct expression => formula_list_apply
172      | formula_list punct expression => formula_list_apply;
173    // Perl MathGrammar L709-711: Two-part relops (>=, <=, <<, >>)
174    two_part_relop = langle_rel langle_rel => two_part_relop_combine
175      | rangle_rel rangle_rel => two_part_relop_combine
176      | langle_rel relop_equals => two_part_relop_combine
177      | rangle_rel relop_equals => two_part_relop_combine;
178
179    formula = expression
180      | formula relop expression => infix_relation
181      | formula two_part_relop expression => infix_relation
182      | formula relop formula_list => infix_relation
183      | formula relop => postfix_relop
184      | formula arrow expression => infix_relation
185      | arrow expression => prefix_arrow_apply
186      // Perl MathGrammar L81: AnyOp Expression => Apply(AnyOp, Absent(), Expression)
187      // Leading relop with implied absent left operand (e.g. "= e + f + g" in eqnarray)
188      | relop expression => prefix_relop_apply
189      | metarelop expression => prefix_relop_apply
190      | modifier_expression
191      // Perl MathGrammar L236: addExpressionModifier: MODIFIER
192      // Standalone postfix modifier (e.g. `8\pmod{3}` → annotated(8, pmod(3)))
193      | formula modifier => postfix_modifier_apply;
194
195    // Perl MathGrammar: Factor includes preScripted['bigop'] as standalone
196    // So standalone bigops can form statements (needed for list expressions like \int \quad \int)
197    statement = formula
198      | statement metarelop formula => infix_relation
199      | metarelop formula => prefix_metarelop_apply
200      | any_bigop | composed_bigop
201      | operator | compound_operator
202      | function | trigfunction
203      // Bare operators can form comma-separated lists: +,-,×
204      | addop | mulop | relop | arrow;
205
206    end_punct = punct | period;
207    statements = statement
208      | statement end_punct => postfix_embellished
209      | statements punct statement => list_apply
210      // Perl: MorphVertbar — VERTBAR as conditional modifier: x | y,z,t
211      | statement vertbar statements => vertbar_modifier;
212
213    // Extensions, now that we have more category variables defined
214    fenced_factor = lbrace expression rbrace    => fenced
215           | lbracket expression rbracket       => fenced
216           | lparen formula rparen              => fenced
217           | lparen term punct term rparen      => interval
218           | lparen term punct term rbracket    => interval
219           | lbracket term punct term rbracket  => interval
220           | lbracket term punct term rparen  => interval
221           | rbracket term punct term lbracket => interval
222           | singlevertbar expression singlevertbar => fenced
223           // Perl's Fence for comma-separated items in braces: {a,b} and {a,b,c}
224           | lbrace term punct term rbrace => fence
225           | lbrace term punct term punct term rbrace => fence
226           // Perl: {a|b} conditional-set with VERTBAR or MIDDLE separator
227           | lbrace formula singlevertbar formula rbrace => fence
228           | lbrace formula middle_bar formula rbrace => fence
229           | lbrace formula metarelop formula rbrace => fence
230           // Empty fenced expressions: () [] {} ⌊⌋ etc.
231           | lparen rparen => empty_fenced
232           | lbracket rbracket => empty_fenced
233           | lbrace rbrace => empty_fenced
234           | open close => empty_fenced;
235    factor += fenced_factor;
236
237    // UNKNOWN followed by fenced args => function application (Perl: Apply[UNKNOWN atom_args])
238    // f(x) => f@(x), g(a+b) => g@(a+b). Only active when MATHPARSER_SPECULATE is set.
239    // Without speculation, this parse is pruned and Marpa uses invisible-times instead.
240    tight_term += unknown fenced_factor => speculative_prefix_apply;
241    // OPFUNCTION followed by fenced args => function application
242    // \operatorname{cov}(L) => cov@(L). Always treated as application, not multiplication.
243    tight_term += opfunction fenced_factor => prefix_apply;
244    // Perl: OPFUNCTION absorbs barearg (factor chain) just like FUNCTION/TRIGFUNCTION
245    // \log x => log@(x), \operatorname{cov}(L) already handled by fenced_factor rule
246    tight_term += opfunction tight_term => prefix_apply;
247    tight_term += opfunction factor => prefix_apply;
248    // TRIGFUNCTION absorbs bare args: \sin x => sin@(x), \cos\pi => cos@(pi)
249    // Note: trigfunction tight_term already in compound_operator, can't duplicate.
250    tight_term += trigfunction factor => prefix_apply;
251    tight_term += trigfunction fenced_factor => prefix_apply;
252    // Perl IntFactor L640-651: diffd followed by ATOM/UNKNOWN/ID => Apply(DIFFOP(d), var)
253    // Uses existing `unknown` terminal; semantic action checks text is literally "d".
254    // At factor level so it can appear as right operand of invisible_times.
255    factor += unknown factor_base => diffop_apply;
256
257    // Script content
258    postsubarg = start_postsubscript expression end_postsubscript => faux_wrap;
259    postsuperarg = start_postsuperscript expression end_postsuperscript => faux_wrap
260      // TODO: what other kinds of arguments are accepted in scripts? Should we do "anything"?
261      | start_postsuperscript supop end_postsuperscript => faux_wrap;
262    floatsubarg = start_floatsubscript expression end_floatsubscript => faux_wrap;
263    floatsuperarg = start_floatsuperscript expression end_floatsuperscript => faux_wrap;
264    // standalone top-level variants of floating scripts:
265    floatsubscript = start_floatsubscript expression end_floatsubscript => standalone_script;
266    floatsuperscript = start_floatsuperscript expression end_floatsuperscript => standalone_script;
267    // Scripted factors -- avoid adding ambiguity in the left-right order of collection
268    // first ALL left (=float), then right (=post).
269    scripted_factor_l11 = floatsuperarg factor_base => prefix_script
270      | floatsuperarg opfunction => prefix_script;
271    scripted_factor_l12 = floatsubarg factor_base => prefix_script
272      | floatsubarg opfunction => prefix_script;
273    scripted_factor_l1 = scripted_factor_l11 | scripted_factor_l12;
274    scripted_factor_l2 = floatsuperarg scripted_factor_l12 => prefix_script
275      | floatsubarg scripted_factor_l11 => prefix_script;
276
277    scripted_factor_r11 = factor_base postsuperarg => postfix_script
278      | opfunction postsuperarg => postfix_script
279      | any_bigop postsuperarg => postfix_script
280      | scripted_factor_l1 postsuperarg => postfix_script
281      | scripted_factor_l2 postsuperarg => postfix_script
282      | fenced_factor postsuperarg => postfix_script;
283    scripted_factor_r12 = factor_base postsubarg => postfix_script
284      | opfunction postsubarg => postfix_script
285      | any_bigop postsubarg => postfix_script
286      | scripted_factor_l1 postsubarg => postfix_script
287      | scripted_factor_l2 postsubarg => postfix_script
288      | fenced_factor postsubarg => postfix_script;
289    scripted_factor_r1 = scripted_factor_r11 | scripted_factor_r12;
290    scripted_factor_r2 = scripted_factor_r12 postsuperarg => postfix_script
291      | scripted_factor_r11 postsubarg => postfix_script;
292    factor += scripted_factor_l1 | scripted_factor_l2 | scripted_factor_r1 | scripted_factor_r2;
293
294    // Scripted bigops: \int_0^\infty, \sum_{n=1}^N, etc.
295    // These are bigops with post-scripts that still act as prefix operators.
296    // Perl: preScripted['INTOP'] addIntOpArgs / preScripted['bigop'] addOpArgs
297    // Chain scripts: first sub then super, or vice versa (like scripted_factor_r1/r2)
298    scripted_bigop_r1 = any_bigop postsuperarg => postfix_script
299      | any_bigop postsubarg => postfix_script;
300    scripted_bigop = scripted_bigop_r1
301      | scripted_bigop_r1 postsuperarg => postfix_script
302      | scripted_bigop_r1 postsubarg => postfix_script;
303    tight_term += scripted_bigop tight_term => prefix_apply;
304    tight_term += scripted_bigop factor => prefix_apply;
305    // Scripted bigops can also appear as standalone statements
306    statement += scripted_bigop;
307
308    anyop = addop | mulop | relop | arrow | metarelop
309      | bigop | sumop | intop
310      | limitop | diffop | vertbar | supop
311      | modifierop | composed_bigop | operator | compound_operator;
312
313    anyscript = floatsuperscript | floatsubscript;
314
315    anything = statements | anyop | anyscript |
316      anyop anyop => compound_operator_2
317  );
318  // | term_argument postsuperarg tex_argument  => post_script
319  // | term_argument postsubscript tex_argument    => post_script
320  start!(anything);
321
322  // Also prepare the tree builder rules here (for now)
323  Ok((grammar!(), actions!(), builder!()))
324}
