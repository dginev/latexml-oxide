1use crate::semantics::*;
2use marpa::grammar::Grammar as MarpaGrammar;
3use marpa::result::Result;
4use marpa::tree_builder::TreeBuilder;

6// Marpa SLIF-style grammar DSL inside `grammar!()` / `production!()`
7// macros. A handful of long alternation lists (`qm_bracket`,
8// `floatsuperscript`) push past 100 chars; `rustfmt::skip` is applied
9// to the entire builder so the BNF reads as a flat table rather than
10// being re-flowed mid-rule.
11#[rustfmt::skip]
12pub fn init_grammar() -> Result<(MarpaGrammar, Actions, TreeBuilder)> {
13  // We create a declarative macro language of our own, in the spirit of the Marpa SLIF
14  default_registry!();
15  // Tokens, to be used in rules directly
16  token!(atom ~ "ATOM");
17  token!(unknown ~ "UNKNOWN");
18  token!(id ~ "ID");
19  // M4: Specialized tokens for "d" that could be differential operators.
20  // Lexer emits XDIFFUNK/XDIFFID instead of UNKNOWN/ID for "d" content.
21  // These are added as alternatives everywhere unknown/id appear, plus in the diffop rule.
22  token!(diffunk ~ "XDIFFUNK");
23  token!(diffid ~ "XDIFFID");
24  token!(array ~ "ARRAY");
25  token!(number ~ "NUMBER");
26  token!(punct ~ "PUNCT");
27  token!(period ~ "PERIOD");
28  token!(addop ~ "ADDOP");
29  token!(mulop ~ "MULOP");
30  token!(relop ~ "RELOP");
31  token!(elideop ~ "ELIDEOP");
32  token!(langle_rel = "RELOP:less-than");
33  token!(langle_open = "OPEN:langle");
34  token!(langle = [langle_rel langle_open]);
35  token!(rangle_rel = "RELOP:greater-than");
36  token!(rangle_close = "CLOSE:rangle");
37  token!(rangle =[rangle_rel rangle_close]);
38  token!(vertbar ~ "VERTBAR");
39  token!(singlevertbar = "VERTBAR:|");
40  token!(close_pipe = "CLOSE:|");
41  token!(middle_bar = "MIDDLE:|");
42  token!(middle_parallel = "MIDDLE:parallel-to");
43  token!(midbar = [vertbar middle_bar middle_parallel]);
44  token!(lbrace = "OPEN:{");
45  token!(rbrace = "CLOSE:}");
46  token!(lparen = "OPEN:(");
47  token!(rparen = "CLOSE:)");
48  token!(lbracket = "OPEN:[");
49  token!(rbracket = "CLOSE:]");
50  token!(relop_equals = "RELOP:equals");
51  token!(metarelop ~ "METARELOP");
52  token!(modifierop ~ "MODIFIEROP");
53  token!(modifier ~ "MODIFIER");
54  token!(arrow ~ "ARROW");
55  token!(binop ~ "BINOP");
56  token!(postfix ~ "POSTFIX");
57  token!(function ~ "FUNCTION");
58  token!(opfunction ~ "OPFUNCTION");
59  token!(trigfunction ~ "TRIGFUNCTION");
60  token!(applyop ~ "APPLYOP");
61  token!(composeop ~ "COMPOSEOP");
62  token!(supop ~ "SUPOP");
63  // `open`/`close` match GENERIC delimiters (\lfloor, \lceil, \llbracket, etc.)
64  // Specific delimiters (paren, bracket, brace, langle) have their own tokens.
65  // The lexer (util.rs) emits OTHER_OPEN:/OTHER_CLOSE: for generic delimiters
66  // so Marpa doesn't ambiguously match both `open` AND `lparen` for `OPEN:(:N`.
67  token!(open ~ "OTHER_OPEN");
68  token!(close ~ "OTHER_CLOSE");
69  token!(middle ~ "MIDDLE");
70  token!(bigop ~ "BIGOP");
71  token!(sumop ~ "SUMOP");
72  token!(intop ~ "INTOP");
73  token!(limitop ~ "LIMITOP");
74  token!(diffop ~ "DIFFOP");
75  token!(operator ~ "OPERATOR");
76  token!(start_postsubscript ~ "start_POSTSUBSCRIPT");
77  token!(end_postsubscript ~ "end_POSTSUBSCRIPT");
78  token!(start_postsuperscript ~ "start_POSTSUPERSCRIPT");
79  token!(end_postsuperscript ~ "end_POSTSUPERSCRIPT");
80  // Separated bigop script tokens — prevents earley chart competition
81  // between scripted_bigop and scripted_factor rules
82  token!(start_bigopsub ~ "start_BIGOPSUB");
83  token!(end_bigopsub ~ "end_BIGOPSUB");
84  token!(start_bigopsup ~ "start_BIGOPSUP");
85  token!(end_bigopsup ~ "end_BIGOPSUP");
86  token!(start_floatsuperscript ~ "start_FLOATSUPERSCRIPT");
87  token!(end_floatsuperscript ~ "end_FLOATSUPERSCRIPT");
88  token!(start_floatsubscript ~ "start_FLOATSUBSCRIPT");
89  token!(end_floatsubscript ~ "end_FLOATSUBSCRIPT");
90  token!(start_arrow ~ "start_ARROW");
91  token!(end_arrow ~ "end_ARROW");
92
93  rules!(
94      // Factors
95      // opfunction/function/trigfunction are NOT factors — they require arguments.
96      // Standalone usage is handled at the term level (term += function | ...).
97      // `2 \sin` is handled via dedicated tight_term rules below.
98      // Perl MathGrammar L315: ATOM_OR_ID : ATOM | ID | ARRAY
99      // XMArray elements (role="ARRAY") should parse as atoms/factors, like matrices in equations
100      // M4: diffunk/diffid are added as factor_base alternatives so "d" tokens
101      // can appear anywhere unknown/id appear. The diffop rule only uses diffunk/diffid.
102      factor_base = unknown | number | id | atom | array | diffunk | diffid;
103      // Perl MathGrammar L277: OPEN ARRAY CLOSE -> Fence (e.g. \{ array \} or ( array ))
104      // Also handle unmatched delimiters for cases-like patterns.
105      // Perf: `open` is now narrowed to OTHER_OPEN (non-paren/bracket/brace), so
106      // we add specific rules for each standard delimiter.
107      fenced_array = open array close => fenced
108        | open array => open_fenced
109        | array close => close_fenced
110        | lparen array rparen => fenced
111        | lparen array => open_fenced
112        | array rparen => close_fenced
113        | lbracket array rbracket => fenced
114        | lbracket array => open_fenced
115        | array rbracket => close_fenced
116        | lbrace array rbrace => fenced
117        | lbrace array => open_fenced
118        | array rbrace => close_fenced;
119      // FUNCTION and OPFUNCTION are both factors (participate in implicit multiplication).
120      // The distinction is in argument absorption: OPFUNCTION absorbs bare args,
121      // FUNCTION only absorbs fenced (parenthesized) args.
122      factor = factor_base | function | opfunction | fenced_array;
123      // Perl: limit-from@(number, sign) — directional limits: 0+, 1-
124      // A "left-only term": on the left behaves as a term (for comma lists),
125      // on the right terminates at the addop (like expression-level postfix).
126      limit_from_term = factor_base addop => limit_from_apply;
127      // Terms
128      // Perl: bigop = BIGOP | SUMOP | INTOP | LIMITOP | DIFFOP
129      any_bigop = bigop | sumop | intop | limitop | diffop;
130      // Adjacent bigops compose like higher-order functions:
131      // \int\iint => integral(double-integral), itself a compound operator
132      composed_bigop = any_bigop any_bigop => prefix_apply
133        | any_bigop composed_bigop => prefix_apply;
134
135      // Compound operators: OPERATOR composed with functions/other operators (right-recursive)
136      // D sin => Apply(D, sin), D D sin => Apply(D, Apply(D, sin))
137      // Must end with a function/trigfunction (no bare operator-only compounds)
138      compound_operator = operator trigfunction => prefix_apply
139        | operator function => prefix_apply
140        | operator opfunction => prefix_apply
141        | operator compound_operator => prefix_apply;
142
143      // tight_term includes single factors (for left-recursive chaining)
144      // and all compound constructs (invisible times, prefix application, etc.)
145      // The `\log x` → `log*x` issue is handled by semantic pruning in
146      // apply_invisible_times, not at the grammar level.
147      tight_term = factor
148        | tight_term factor => apply_invisible_times
149        // Perl MathGrammar L423: POSTFIX (e.g. n!) => Apply(op, term)
150        | tight_term postfix => apply_postfix
151        // Note: FUNCTION does NOT absorb bare args — only parens or APPLYOP.
152        // `fga` = f*g*a, but `f(a)` = f@(a). OPFUNCTION absorbs: `Fga` = F@(g*a).
153        // FUNCTION only chains via opfunction's factor status or APPLYOP.
154        // trigfunction uses trigbarearg via applied_func (absorbs MulOp chains)
155        // NOTE: bigop rules moved to += section (after `term` is defined) so they
156        // can absorb full term (mulop chains like x² * dx), not just tight_term.
157        | composed_bigop tight_term => prefix_apply
158        | operator factor => prefix_apply
159        | factor_base applyop tight_term => prefix_apply_applyop
160        // Perl: FUNCTION/OPFUNCTION/TRIGFUNCTION + explicit APPLYOP + argument
161        // Handles \lxDeclare-annotated tokens: f⁡(x) where ⁡ is APPLYOP
162        | function applyop tight_term => prefix_apply_applyop
163        | opfunction applyop tight_term => prefix_apply_applyop
164        | trigfunction applyop tight_term => prefix_apply_applyop;
165
166      // Perl MathGrammar L258: Factor moreFactors — consecutive function
167      // applications chain with invisible times.
168      // e.g. \sin x \cos y => sin(x) * cos(y)
169      //
170      // IMPORTANT — TRIGFUNCTION ARGUMENT SCOPING AMBIGUITY:
171      // Perl's trigBarearg absorbs MulOp chains: \sin\pi\times x => sin(π×x).
172      // We deliberately DO NOT implement this absorption. The expression
173      // sin(π)×x vs sin(π×x) is a *legitimate semantic ambiguity* that cannot
174      // be resolved purely by grammar structure. Both parses are valid:
175      //   sin(π)×x = 0×x (evaluating sin at π)
176      //   sin(π×x) = sin of the product
177      // Perl's Parse::RecDescent picks the "absorb" interpretation heuristically.
178      //
179      // FUTURE WORK: To match Perl's output, implement *targeted semantic pruning*
180      // in the parse tree selection phase (semantics/tree.rs) that uses context
181      // cues to prefer one interpretation:
182      //   - If the MulOp is invisible (⁢), prefer absorption (sin 2x → sin(2x))
183      //   - If the MulOp is explicit (×,·), either interpretation is valid
184      //   - If the argument is a known constant (π, e), standalone may be preferred
185      // This is a semantic-level decision, not a grammar-level one.
186      // applied_func and tight_term augmentations moved below trig_arg definition
187
188      // Composed functions: f∘g, sin∘cos — these can then be applied as functions
189      // COMPOSEOP operates on function-level operands (curry level 2)
190      // Left-to-right associative (matching Perl): f∘g∘h = (f∘g)∘h
191      composed_term = function composeop function => infix_apply
192        | function composeop trigfunction => infix_apply
193        | function composeop opfunction => infix_apply
194        | trigfunction composeop function => infix_apply
195        | trigfunction composeop trigfunction => infix_apply
196        | trigfunction composeop opfunction => infix_apply
197        | opfunction composeop function => infix_apply
198        | opfunction composeop trigfunction => infix_apply
199        | opfunction composeop opfunction => infix_apply
200        // Left-recursive for left-to-right associativity
201        | composed_term composeop function => infix_apply
202        | composed_term composeop trigfunction => infix_apply
203        | composed_term composeop opfunction => infix_apply;
204
205      // Composed functions can be applied like regular functions
206      tight_term += composed_term tight_term => prefix_apply;
207
208      term = tight_term
209      | term mulop tight_term => infix_apply_nary
210      | term mulop tight_term elideop => infix_apply_and_elide
211      // Perl: BINOP matches both AddOp and MulOp (ambiguous precedence from \mathbin)
212      | term binop tight_term => infix_apply_nary
213      | term binop tight_term elideop => infix_apply_and_elide
214      // Fallback: COMPOSEOP on general terms (for non-function-level composition)
215      | term composeop term => infix_apply
216      | operator applyop term => prefix_apply_applyop;
217
218
219      // Allow standalone functions/trigfunctions/opfunctions/operators as terms
220      // This is needed for (f*g)(x) where f and g are FUNCTION tokens
221      // opfunction here allows standalone \operatorname{R} to parse
222      // operator as term enables D - 1 (subtraction), D + G (addition)
223      term += function | trigfunction | opfunction | composed_term | operator;
224      // Allow elideop (\cdots) as a term for chains like y + i + \cdots + y_n
225      // Perl treats cdots as a regular term in addition chains
226      term += elideop;
227
228      // Higher-order operator terms: functions as standalone objects multiplied by factors
229      // `2\sin` = `2 * sin`, `2\sin\cos` = `2 * sin * cos`
230      // These are term-level (not tight_term) so they don't interfere with
231      // function application: `2\sin x` = `2 * sin(x)` (not `(2*sin) * x`)
232      tight_opterm = factor function => apply_invisible_times
233        | factor trigfunction => apply_invisible_times
234        | factor opfunction => apply_invisible_times
235        // Consecutive functions multiply: fgh → f·g·h (Perl Factor moreFactors)
236        | function function => apply_invisible_times
237        | function trigfunction => apply_invisible_times
238        | function opfunction => apply_invisible_times
239        | tight_opterm function => apply_invisible_times
240        | tight_opterm trigfunction => apply_invisible_times
241        | tight_opterm opfunction => apply_invisible_times;
242      term += tight_opterm;
243
244      // Expressions
245      expression = term
246        | expression addop term => infix_apply_nary
247        | expression addop term elideop => infix_apply_and_elide
248        | addop tight_term => prefix_apply
249        | factor addop => postfix_apply
250        | expression addop => postfix_apply
251        // Perl MathGrammar L236: addExpressionModifier: MODIFIEROP Expression
252        // => Apply(modifierop, expr, expr2). Handles infix `a mod b`.
253        | expression modifierop expression => infix_apply
254        // Perl MathGrammar L236: addExpressionModifier: MODIFIER
255        // Standalone postfix modifier (e.g. `8\pmod{3}` → annotated(8, pmod(3)))
256        // Placed at expression level so MODIFIER binds BEFORE RELOP.
257        // e.g. `5 ≡ 8 \pmod{3}` → `5 ≡ annotated(8, pmod(3))` not `annotated(5≡8, pmod(3))`
258        | expression modifier => postfix_modifier_apply
259        // Perl MathGrammar L224-233: OPEN relop/modifierop Expression balancedClose
260        // Parenthesized modifier expressions: x(>0) → annotated(x, Fence(>0))
261        | expression lparen relop expression rparen => annotated_fenced_modifier
262        | expression lparen modifierop expression rparen => annotated_fenced_modifier
263        // Perl MathGrammar L223: PUNCT? OPEN relop Expression CLOSE
264        // Semicolon annotation: a;(<e) → annotated(a, absent < e)
265        | expression punct lparen relop expression rparen => annotated_punct_fenced_modifier
266        | expression punct lparen modifierop expression rparen => annotated_punct_fenced_modifier;
267
268      // Formula
269      // Perl MathGrammar L73/236: MODIFIEROP Expression => Apply(mod, Absent, expr)
270      modifier_expression = modifierop expression => modifier_prefix_apply;
271      // Perl: within a Formula, comma-separated expressions after a relop form a list RHS.
272      // e.g. a=b,c,d → a = list(b,c,d), not list(a=b, c, d).
273      // Uses formula_list_apply which rejects items containing relops (those belong at statement level).
274      formula_list = expression punct expression => formula_list_apply
275        | formula_list punct expression => formula_list_apply;
276      // Comma-separated term lists: term, term, term, ...
277      // Used for angle-bracket inner products <x,y>, <a,b,c>, etc.
278      // Also includes limit_from_term for patterns like (1+, 0+, 1-, 0-).
279      term_list = term punct term => list_apply
280        | term_list punct term => list_apply
281        | limit_from_term punct term => list_apply
282        | limit_from_term punct limit_from_term => list_apply
283        | term_list punct limit_from_term => list_apply
284        | term punct limit_from_term => list_apply;
285
286      // Perl MathGrammar L709-711: Two-part relops (>=, <=, <<, >>)
287      two_part_relop = langle_rel langle_rel => two_part_relop_combine
288        | rangle_rel rangle_rel => two_part_relop_combine
289        | langle_rel relop_equals => two_part_relop_combine
290        | rangle_rel relop_equals => two_part_relop_combine;
291
292      formula = expression
293        | formula relop expression => infix_relation
294        | formula two_part_relop expression => infix_relation
295        | formula relop formula_list => infix_relation
296        | formula relop => postfix_relop
297        // Perl moreRelations: `relop moreRelations` — consecutive relops chain without intervening terms
298        // e.g. `A ∈ ∞ ∋` → the ∈ absorbs ∞, then ∋ appends to the chain (no absent)
299        | formula relop relop => consecutive_relop_chain
300        | formula arrow expression => infix_relation
301        | arrow expression => prefix_arrow_apply
302        // Arrow-wrapped content (from amscd XMWrap role="ARROW"):
303        // Parsed as a prefix arrow application on the enclosed content.
304        | start_arrow arrow expression end_arrow => arrow_wrap_apply
305        | start_arrow arrow end_arrow => arrow_wrap_solo
306        // Perl MathGrammar L81: AnyOp Expression => Apply(AnyOp, Absent(), Expression)
307        // Leading relop with implied absent left operand (e.g. "= e + f + g" in eqnarray)
308        | relop expression => prefix_relop_apply
309        | metarelop expression => prefix_relop_apply
310        | modifier_expression;
311
312      // Perl MathGrammar: Factor includes preScripted['bigop'] as standalone
313      // So standalone bigops can form statements (needed for list expressions like \int \quad \int)
314      statement = formula
315        | statement metarelop formula => infix_relation
316        | metarelop formula => prefix_metarelop_apply
317        | any_bigop | composed_bigop
318        | operator | compound_operator
319        | function | trigfunction
320        // Bare operators can form comma-separated lists: +,-,×
321        | addop | mulop | binop | relop | arrow
322  ;
323
324      end_punct = punct | period;
325      statements = statement
326        | statement end_punct => postfix_embellished
327        | statements end_punct => postfix_embellished
328        | statements punct statement => list_apply
329        // Perl MathGrammar L129: endPunct includes PERIOD. Period creates formulae, not list.
330        | statements period statement => formulae_apply
331        // Perl: MorphVertbar — VERTBAR as conditional modifier: x | y,z,t
332        | statement vertbar statements => vertbar_modifier;
333
334      // Perl MathGrammar: Formulae = Formula (endPunct Formula)* → NewFormulae()
335      // Separate nonterminal from expression-level formula_list to avoid ambiguity:
336      // expression-level formula_list uses formula_list_apply (rejects relops),
337      // while formulae uses formulae_apply (accepts full statements).
338      formulae = statement punct statement => formulae_apply
339        | formulae punct statement => formulae_apply
340        // Period also separates formulae
341        | statement period statement => formulae_apply
342        | formulae period statement => formulae_apply;
343
344      // Extensions, now that we have more category variables defined
345      fenced_factor = lbrace expression rbrace    => fenced
346             | lbracket expression rbracket       => fenced
347             | lparen formula rparen              => fenced
348             // METARELOP inside parens: f(a:b), f(a↔b) — colon/arrow as relation in fenced
349             | lparen formula metarelop expression rparen => fence
350             // Parenthesized comma-separated lists: (a,b,c), (a+b, c+d), (1+, 0+, 1-, 0-)
351             // Perf (Fix 3): `lparen term_list rparen` was duplicate with
352             // `lparen formula_list rparen` for non-relational content (both produce
353             // identical `list@(...)` trees). Dropped; formula_list covers
354             // (a,b,c), (a+b,c+d), and (0+,1-) via `factor addop => postfix_apply`
355             // which produces limit-from XM matching limit_from_apply semantics.
356             | lparen formula_list rparen        => fenced
357             // Bracketed and braced comma-separated lists: [a,b,c], {a,b,c}
358             | lbracket formula_list rbracket    => fenced
359             | lbrace formula_list rbrace        => fenced
360             // Angle brackets as delimiters: <x,y> for inner products, etc.
361             // Old typesetting conventions used < > instead of \langle \rangle.
362             // Uses term_list (comma-separated terms) to avoid matching complex
363             // nested expressions. Only fires when content has commas.
364             | langle_rel term_list rangle_rel => fenced
365             // Angle-bracket fencing with \langle/\rangle (distinct from parentheses)
366             // Now that langle_open/rangle_close are NOT remapped to lparen/rparen,
367             // we need explicit rules for angle-bracket fenced expressions.
368             // M8: removed `langle_open expression rangle_close` — subsumed by formula
369             // (every expression is a formula; keeping both creates 2x ambiguity)
370             | langle_open formula rangle_close => fenced
371             | langle_open term_list rangle_close => fenced
372             | langle_open formula_list rangle_close => fenced
373             | langle_open formula metarelop expression rangle_close => fence
374             // Perf/design: interval rules moved out of fenced_factor into
375             // term (see `tight_term += interval_term` below). Math convention:
376             // an interval `(a,b)` is a named mathematical object — a set of
377             // numbers — not a grouping mechanism like `(a+b)`. Treating it as
378             // a term instead of fenced_factor has a clean consequence:
379             // function application `f(x,y)` takes a fenced_factor, so the
380             // interval interpretation of `(x,y)` is pruned naturally; the
381             // list interpretation (from `lparen formula_list rparen`) wins.
382             // QM bra-ket uses langle_open/rangle_close (specific ⟨⟩ tokens),
383             // avoiding ambiguity with relational < > (langle_rel/rangle_rel).
384             // Conditional probability uses lparen/rparen (specific () tokens),
385             // avoiding ambiguity with ket (which requires rangle_close).
386             //
387             // Perl MathGrammar L294: || exp || → norm (must be before |exp| → abs-val)
388             // CatSymbols merges two | into ‖; singlevertbar = VERTBAR:|
389             | singlevertbar singlevertbar expression singlevertbar singlevertbar => norm_fenced
390             | singlevertbar expression singlevertbar => fenced
391             // Dirac ket: |label⟩ — VERTBAR as opening, CLOSE:rangle as closing
392             // Restricted to rangle_close (⟩) to avoid ambiguity with conditional
393             // probability (x|y) where ) is a generic CLOSE but not rangle.
394             // Perl MathGrammar uses RANGLE specifically, not generic CLOSE.
395             // Ket labels: expressions, arrows, operators, relops, etc.
396             | singlevertbar expression rangle_close => qm_ket
397             | singlevertbar arrow rangle_close => qm_ket
398             | singlevertbar metarelop rangle_close => qm_ket
399             | singlevertbar operator rangle_close => qm_ket
400             | singlevertbar any_bigop rangle_close => qm_ket
401             | singlevertbar mulop rangle_close => qm_ket
402             | singlevertbar addop rangle_close => qm_ket
403             | singlevertbar relop rangle_close => qm_ket
404             | singlevertbar modifierop rangle_close => qm_ket
405             // Dirac bra: ⟨label| — OPEN:langle as opening, VERTBAR as closing
406             // Restricted to langle_open to avoid ambiguity with parens.
407             | langle_open expression singlevertbar => qm_bra
408             | langle_open arrow singlevertbar => qm_bra
409             | langle_open metarelop singlevertbar => qm_bra
410             | langle_open operator singlevertbar => qm_bra
411             // Braket: ⟨a|b⟩ → inner-product@(a, b)
412             | langle_open expression singlevertbar expression rangle_close => qm_braket
413             // Bracket: ⟨a|f|b⟩ → quantum-operator-product@(a, f, b)
414             | langle_open expression singlevertbar expression singlevertbar expression rangle_close => qm_bracket
415             // Perl's Fence for comma-separated items in braces: {a,b} and {a,b,c}
416             | lbrace term punct term rbrace => fence
417             | lbrace term punct term punct term rbrace => fence
418             // Perl: {a|b} conditional-set with VERTBAR or MIDDLE separator
419             | lbrace formula singlevertbar formula rbrace => fence
420             | lbrace formula middle_bar formula rbrace => fence
421             | lbrace formula metarelop formula rbrace => fence
422             // Conditional probability: p(a|b) — safe now that ket uses rangle_close
423             // (not generic close), so |y) no longer matches ket pattern.
424             // NOTE: formula_list variants (x,y|z) don't work due to Marpa limitation:
425             // formula_list completion doesn't propagate to singlevertbar continuation.
426             // Perl handles this via recursive descent context. Tracked as known limitation.
427             | lparen formula singlevertbar formula rparen => fence
428             | lparen formula_list singlevertbar formula rparen => fence
429             | lparen formula singlevertbar formula_list rparen => fence
430             // \middle separator: \left(a\middle|b\right) → fenced with separator
431             // MIDDLE tokens are author-explicit (unlike bare |), so unambiguous.
432             // `open`/`close` now only match generic delimiters (OTHER_OPEN/OTHER_CLOSE),
433             // so we also add specific rules for paren/bracket. Not for langle/brace —
434             // those match specific QM/set rules, not this generic fence path.
435             | open formula middle_bar formula close => fence
436             | open formula middle formula close => fence
437             | lparen formula middle_bar formula rparen => fence
438             | lparen formula middle formula rparen => fence
439             | lbracket formula middle_bar formula rbracket => fence
440             | lbracket formula middle formula rbracket => fence
441             // Generic OPEN/CLOSE delimiters: \lfloor...\rfloor, \lceil...\rceil, etc.
442             // Perl MathGrammar: OPEN Expression CLOSE → Fence
443             | open expression close => fenced

505        // rule competed with cascade-via-tight_term in FGHa and doubled
506        // enumeration for every OPFUNCTION+OPFUNCTION pair.
507        // Delimited function application: f(x), f[x], F(x), \sin(x) etc.
508        // Perl: ApplyDelimited creates XMDual(content=Apply(XMRef(f),XMRef(args)),
509        //        presentation=Apply(f, XMWrap(open, args, close))).

512        | function lparen formula rparen => apply_delimited

515        | opfunction lbracket formula rbracket => apply_delimited

530        | lparen term punct term rbracket    => interval
531        | lbracket term punct term rbracket  => interval
532        | lbracket term punct term rparen  => interval
533        | rbracket term punct term lbracket => interval;

536      // UNKNOWN followed by fenced args => function application (Perl: doubtArgs/maybeArgs)
537      // f(x) => f@(x), g(a+b) => g@(a+b). Only active when MATHPARSER_SPECULATE is set.
538      // Without speculation, this parse is pruned and Marpa uses invisible-times instead.
539      // NOTE: ID tokens are multiplicative atoms — NEVER prefix-apply. Only UNKNOWN
540      // tokens get speculative function application. ID always uses invisible-times.
541      tight_term += unknown fenced_factor => speculative_prefix_apply
542        | diffunk fenced_factor => speculative_prefix_apply;
543      // Perf: `tight_term += function fenced_factor => prefix_apply` removed.
544      // It duplicated the applied_func path (function fenced_factor => prefix_apply)
545      // and competed with apply_delimited (function lparen formula rparen =>
546      // apply_delimited) for `f(x)` cases, adding ambiguity with no benefit.
547      // OPFUNCTION bare arg absorption (Perl: addOpArgs barearg + moreargs)
548      // \log 2x^2 => log@(2*x^2). These tight_term rules serve as priority
549      // boosters that ensure cascading opfunction application (FGHa → F@(G@(H@(a))))
550      // wins over `F@(G) * H@(a)`. Removing the direct `tight_term += opfunction
551      // tight_term` rule (even after removing `applied_func = opfunction opfunction`
552      // in commit ffcafc33e) still breaks FGHa — the `applied_func = opfunction
553      // tight_term` + `tight_term += applied_func` lifting path does NOT preserve
554      // Marpa's cascade-over-invisible-times preference. The direct self-recursive
555      // rule is required.

557      tight_term += opfunction factor => prefix_apply;
558      // Perf: removed `opfunction fenced_factor => prefix_apply` — `factor` already
559      // includes fenced_factor, so this rule was a duplicate that caused Marpa to
560      // enumerate the same tree twice for every `\sin(x)` (OPFUNCTION) form.
561      // TRIGFUNCTION absorbs bare args: \sin x => sin@(x), \cos\pi => cos@(pi).
562      // Note: `factor` is used here (not factor_base) to support scripted args
563      // like \sin a^2 (scripted_factor_r1 is in factor but not factor_base).
564      // Narrowing to factor_base breaks \sin a^2 = sin(a^2) parses.
565      tight_term += trigfunction factor => prefix_apply;

567      // More targeted than the previous `compound_operator tight_term` — absorbs only one factor,
568      // not an entire invisible-times chain. Covers fenced_factor too (since factor += fenced_factor).

570      // Perl IntFactor L640-651: diffd followed by ATOM/UNKNOWN/ID => Apply(DIFFOP(d), var)
571      // Semantic action checks text is literally "d" and INTOP context.
572      // At factor level so it can appear as right operand of invisible_times.

574      // M4: Only "d" tokens can be diffops. diffunk/diffid are emitted by the

588      //   - vertbar, supops: statement has no derivation
589      //   - modifierop: statement only has `modifierop formula`, not bare

723        | scripted_bigop_r1 bigopsubarg => postfix_script

765        | floatsuperarg prescripted_bigop => prefix_script
766        | floatsubarg prescripted_bigop => prefix_script;

769      // so they can't incorrectly match bare post-scripts as pre-scripts.

773        | postsubarg prescripted_bigop_inner => prefix_script_pre
774        | postsuperarg prescripted_bigop => prefix_script_pre
775        | postsubarg prescripted_bigop => prefix_script_pre;

777      tight_term += prescripted_bigop factor => prefix_apply;

784        | tight_term singlevertbar postsubarg postsuperarg => eval_at
785        | tight_term singlevertbar postsuperarg postsubarg => eval_at

787        | tight_term close_pipe postsubarg => eval_at
788        | tight_term close_pipe postsubarg postsuperarg => eval_at
789        | tight_term close_pipe postsuperarg postsubarg => eval_at;

792        | bigop | sumop | intop
793        | limitop | diffop | vertbar | supop
794        | modifierop | composed_bigop | operator | compound_operator;

797        // Standalone floating script pairs (no base: {}^c_d or {}_d^c)
798        // Perl: NewScript(NewScript(Absent(), super, 'post'), sub, 'post')
799        | floatsuperscript postsubarg => postfix_script
800        | floatsubscript postsuperarg => postfix_script;
