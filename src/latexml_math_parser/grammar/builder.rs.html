5// Marpa SLIF-style grammar DSL inside `grammar!()` / `production!()`
6// macros. A handful of long alternation lists (`qm_bracket`,
7// `floatsuperscript`) push past 100 chars; `rustfmt::skip` is applied
8// to the entire builder so the BNF reads as a flat table rather than
9// being re-flowed mid-rule.
10#[rustfmt::skip]
11pub fn init_grammar() -> Result<(MarpaGrammar, Actions, TreeBuilder)> {
12  // We create a declarative macro language of our own, in the spirit of the Marpa SLIF
13  default_registry!();
14  // Tokens, to be used in rules directly
15  token!(atom ~ "ATOM");
16  token!(unknown ~ "UNKNOWN");
17  token!(id ~ "ID");
18  // M4: Specialized tokens for "d" that could be differential operators.
19  // Lexer emits XDIFFUNK/XDIFFID instead of UNKNOWN/ID for "d" content.
20  // These are added as alternatives everywhere unknown/id appear, plus in the diffop rule.
21  token!(diffunk ~ "XDIFFUNK");
22  token!(diffid ~ "XDIFFID");
23  token!(array ~ "ARRAY");
24  token!(number ~ "NUMBER");
25  token!(punct ~ "PUNCT");
26  token!(period ~ "PERIOD");
27  // PUNCT with `\quad`-class spacing (rpadding ≥ 5pt). The lexer
28  // (util.rs::punct_followed_by_wide_space) emits these as
29  // `WIDE_PUNCT:,:idx` so the grammar can prefer formulae_apply
30  // for the `formula , \quad condition` arXiv idiom without
31  // enumerating the `list_apply` alternatives that the pragma
32  // would only reject post-hoc. See docs/archive/MATH_AMBIGUITY_AUDIT_2026-05-21.md §2.
33  token!(wide_punct ~ "WIDE_PUNCT");
34  token!(addop ~ "ADDOP");
35  token!(mulop ~ "MULOP");
36  // The `/` divide-MULOP specifically (vs `\times`/`\cdot`/etc.). Used to scope
37  // the bare-bigop fraction rule (`\partial/\partial t`) to division only, so it
38  // does NOT fire on `\partial \times B` (which Perl keeps as a flat product).
39  token!(divide = "MULOP:divide");
40  token!(relop ~ "RELOP");
41  token!(elideop ~ "ELIDEOP");
42  token!(langle_rel = "RELOP:less-than");
43  token!(langle_open = "OPEN:langle");
44  token!(langle = [langle_rel langle_open]);
45  token!(rangle_rel = "RELOP:greater-than");
46  token!(rangle_close = "CLOSE:rangle");
47  token!(rangle =[rangle_rel rangle_close]);
48  token!(vertbar ~ "VERTBAR");
49  token!(singlevertbar = "VERTBAR:|");
50  // `\|` and `\Vert` lex to a single doubled-bar token `VERTBAR:||` (vs four
51  // single `VERTBAR:|` for `||x||`). Used by the `\|x\|` norm rule below.
52  token!(doublevertbar = "VERTBAR:||");
53  // `\left|...\right|` produces VERTBAR tokens tagged `stretchy="true"`.
54  // The lexer (util.rs) further distinguishes those emitted by `\@left`
55  // from those emitted by `\@right` via the `lx@side` property set in
56  // the respective constructors (tex_math.rs). Side-distinct lexemes
57  // (LEFT_STRETCHY_VERTBAR / RIGHT_STRETCHY_VERTBAR) eliminate
58  // combinatorial pairing ambiguity for the kerned-stack norm idioms
59  // (\vertii, \vertiii, …) where multiple identical bars appear in
60  // sequence. `stretchy_vertbar` is the union — used by legacy rules
61  // (e.g. eval_at) that don't care which side a stretchy bar came from.
62  // See docs/archive/MATH_AMBIGUITY_AUDIT_2026-05-21.md §2 (delimiter-pairing) and Task #263.
63  token!(left_stretchy_vertbar ~ "LEFT_STRETCHY_VERTBAR");
64  token!(right_stretchy_vertbar ~ "RIGHT_STRETCHY_VERTBAR");
65  // Fallback for stretchy bars that arrived without a side tag
66  // (legacy DOM input, or a code path that bypassed `\@left`/`\@right`).
67  // The lexer in util.rs emits this when `role_side` is absent.
68  // Rules that legitimately accept any side (eval_at) enumerate both
69  // alternatives explicitly rather than relying on a union token —
70  // the Marpa tree builder doesn't roll up alternation-of-tokens cleanly.
71  token!(undirected_stretchy_vertbar ~ "STRETCHY_VERTBAR");
72  token!(close_pipe = "CLOSE:|");
73  token!(middle_bar = "MIDDLE:|");
74  token!(middle_parallel = "MIDDLE:parallel-to");
75  token!(midbar = [vertbar middle_bar middle_parallel]);
76  token!(lbrace = "OPEN:{");
77  token!(rbrace = "CLOSE:}");
78  token!(lparen = "OPEN:(");
79  token!(rparen = "CLOSE:)");
80  token!(lbracket = "OPEN:[");
81  token!(rbracket = "CLOSE:]");
82  token!(relop_equals = "RELOP:equals");
83  token!(metarelop ~ "METARELOP");
84  token!(modifierop ~ "MODIFIEROP");
85  token!(modifier ~ "MODIFIER");
86  token!(arrow ~ "ARROW");
87  token!(binop ~ "BINOP");
88  token!(postfix ~ "POSTFIX");
89  token!(function ~ "FUNCTION");
90  token!(opfunction ~ "OPFUNCTION");
91  token!(trigfunction ~ "TRIGFUNCTION");
92  token!(applyop ~ "APPLYOP");
93  token!(composeop ~ "COMPOSEOP");
94  token!(supop ~ "SUPOP");
95  // `open`/`close` match GENERIC delimiters (\lfloor, \lceil, \llbracket, etc.)
96  // Specific delimiters (paren, bracket, brace, langle) have their own tokens.
97  // The lexer (util.rs) emits OTHER_OPEN:/OTHER_CLOSE: for generic delimiters
98  // so Marpa doesn't ambiguously match both `open` AND `lparen` for `OPEN:(:N`.
99  token!(open ~ "OTHER_OPEN");
100  token!(close ~ "OTHER_CLOSE");
101  token!(middle ~ "MIDDLE");
102  token!(bigop ~ "BIGOP");
103  token!(sumop ~ "SUMOP");
104  token!(intop ~ "INTOP");
105  token!(limitop ~ "LIMITOP");
106  token!(diffop ~ "DIFFOP");
107  token!(operator ~ "OPERATOR");
108  token!(start_postsubscript ~ "start_POSTSUBSCRIPT");
109  token!(end_postsubscript ~ "end_POSTSUBSCRIPT");
110  token!(start_postsuperscript ~ "start_POSTSUPERSCRIPT");
111  token!(end_postsuperscript ~ "end_POSTSUPERSCRIPT");
112  // Separated bigop script tokens — prevents earley chart competition
113  // between scripted_bigop and scripted_factor rules
114  token!(start_bigopsub ~ "start_BIGOPSUB");
115  token!(end_bigopsub ~ "end_BIGOPSUB");
116  token!(start_bigopsup ~ "start_BIGOPSUP");
117  token!(end_bigopsup ~ "end_BIGOPSUP");
118  token!(start_floatsuperscript ~ "start_FLOATSUPERSCRIPT");
119  token!(end_floatsuperscript ~ "end_FLOATSUPERSCRIPT");
120  token!(start_floatsubscript ~ "start_FLOATSUBSCRIPT");
121  token!(end_floatsubscript ~ "end_FLOATSUBSCRIPT");
122  token!(start_arrow ~ "start_ARROW");
123  token!(end_arrow ~ "end_ARROW");

125  rules!(
126      // Factors
127      // opfunction/function/trigfunction are NOT factors — they require arguments.
128      // Standalone usage is handled at the term level (term += function | ...).
129      // `2 \sin` is handled via dedicated tight_term rules below.
130      // Perl MathGrammar L315: ATOM_OR_ID : ATOM | ID | ARRAY
131      // XMArray elements (role="ARRAY") should parse as atoms/factors, like matrices in equations
132      // M4: diffunk/diffid are added as factor_base alternatives so "d" tokens
133      // can appear anywhere unknown/id appear. The diffop rule only uses diffunk/diffid.
134      factor_base = unknown | number | id | atom | array | diffunk | diffid;
135      // Perl MathGrammar L277: OPEN ARRAY CLOSE -> Fence (e.g. \{ array \} or ( array ))
136      // Also handle unmatched delimiters for cases-like patterns.
137      // Perf: `open` is now narrowed to OTHER_OPEN (non-paren/bracket/brace), so
138      // we add specific rules for each standard delimiter.
139      fenced_array = open array close => fenced
140        | open array => open_fenced
141        | array close => close_fenced
142        | lparen array rparen => fenced
143        | lparen array => open_fenced
144        | array rparen => close_fenced
145        | lbracket array rbracket => fenced
146        | lbracket array => open_fenced
147        | array rbracket => close_fenced
148        | lbrace array rbrace => fenced
149        | lbrace array => open_fenced
150        | array rbrace => close_fenced;
151      // FUNCTION is a factor (participates in implicit multiplication).
152      // OPFUNCTION is intentionally NOT here. Including OPFUNCTION in
153      // `factor` would let `tight_term factor → apply_invisible_times`
154      // admit a bare OPFUNCTION as its LEFT operand — a reading the
155      // pragma `apply_invisible_times: left is OPFUNCTION/.../FUNCTION,
156      // prefer prefix_apply` correctly rejects, but only after the
157      // grammar has enumerated thousands of such derivations
158      // (~52% of trees on complex math-heavy equations such as
159      // 1911.09517 eq.993). Excluding OPFUNCTION from `factor`
160      // pushes the constraint upstream into the grammar: bare
161      // OPFUNCTION cannot anchor an invisible-times chain on the LEFT.
162      // To keep legitimate trailing-OPFUNCTION cases (`c \not`,
163      // `a b \not`, ...), the `tight_term += tight_term opfunction =>
164      // apply_invisible_times` rule below admits OPFUNCTION as a
165      // chain-terminating RIGHT operand. OPFUNCTION as the head of an
166      // applied form (`\log x`, `\sin x`) enters via `applied_func`
167      // (prefix_apply) and via the `tight_term += opfunction tight_term`
168      // and `tight_term += opfunction factor` rules further below.
169      factor = factor_base | function | fenced_array;
170      // Perl: limit-from@(number, sign) — directional limits: 0+, 1-
171      // A "left-only term": on the left behaves as a term (for comma lists),
172      // on the right terminates at the addop (like expression-level postfix).
173      limit_from_term = factor_base addop => limit_from_apply;
174      // Terms
175      // Perl: bigop = BIGOP | SUMOP | INTOP | LIMITOP | DIFFOP
176      any_bigop = bigop | sumop | intop | limitop | diffop;
177      // Adjacent bigops compose like higher-order functions:
178      // \int\iint => integral(double-integral), itself a compound operator
179      composed_bigop = any_bigop any_bigop => prefix_apply
180        | any_bigop composed_bigop => prefix_apply;
181
182      // Compound operators: OPERATOR composed with functions/other operators (right-recursive)
183      // D sin => Apply(D, sin), D D sin => Apply(D, Apply(D, sin))
184      // Must end with a function/trigfunction (no bare operator-only compounds)
185      compound_operator = operator trigfunction => prefix_apply
186        | operator function => prefix_apply
187        | operator opfunction => prefix_apply
188        | operator compound_operator => prefix_apply;
189
190      // tight_term includes single factors (for left-recursive chaining)
191      // and all compound constructs (invisible times, prefix application, etc.)
192      // The `\log x` → `log*x` issue is handled by semantic pruning in
193      // apply_invisible_times, not at the grammar level.
194      tight_term = factor
195        | tight_term factor => apply_invisible_times
196        // Perl MathGrammar L423: POSTFIX (e.g. n!) => Apply(op, term)
197        | tight_term postfix => apply_postfix
198        // Note: FUNCTION does NOT absorb bare args — only parens or APPLYOP.
199        // `fga` = f*g*a, but `f(a)` = f@(a). OPFUNCTION absorbs: `Fga` = F@(g*a).
200        // FUNCTION only chains via opfunction's factor status or APPLYOP.
201        // trigfunction uses trigbarearg via applied_func (absorbs MulOp chains)
202        // NOTE: bigop rules moved to += section (after `term` is defined) so they
203        // can absorb full term (mulop chains like x² * dx), not just tight_term.
204        | composed_bigop tight_term => prefix_apply
205        | operator factor => prefix_apply
206        | factor_base applyop tight_term => prefix_apply_applyop
207        // Perl: FUNCTION/OPFUNCTION/TRIGFUNCTION + explicit APPLYOP + argument
208        // Handles \lxDeclare-annotated tokens: f⁡(x) where ⁡ is APPLYOP
209        | function applyop tight_term => prefix_apply_applyop
210        | opfunction applyop tight_term => prefix_apply_applyop
211        | trigfunction applyop tight_term => prefix_apply_applyop;
212
213      // Perl MathGrammar L258: Factor moreFactors — consecutive function
214      // applications chain with invisible times.
215      // e.g. \sin x \cos y => sin(x) * cos(y)
216      //
217      // IMPORTANT — TRIGFUNCTION ARGUMENT SCOPING AMBIGUITY:
218      // Perl's trigBarearg absorbs MulOp chains: \sin\pi\times x => sin(π×x).
219      // We deliberately DO NOT implement this absorption. The expression
220      // sin(π)×x vs sin(π×x) is a *legitimate semantic ambiguity* that cannot
221      // be resolved purely by grammar structure. Both parses are valid:
222      //   sin(π)×x = 0×x (evaluating sin at π)
223      //   sin(π×x) = sin of the product
224      // Perl's Parse::RecDescent picks the "absorb" interpretation heuristically.
225      //
226      // FUTURE WORK: To match Perl's output, implement *targeted semantic pruning*
227      // in the parse tree selection phase (semantics/tree.rs) that uses context
228      // cues to prefer one interpretation:
229      //   - If the MulOp is invisible (⁢), prefer absorption (sin 2x → sin(2x))
230      //   - If the MulOp is explicit (×,·), either interpretation is valid
231      //   - If the argument is a known constant (π, e), standalone may be preferred
232      // This is a semantic-level decision, not a grammar-level one.
233      // applied_func and tight_term augmentations moved below trig_arg definition
234
235      // Composed functions: f∘g, sin∘cos — these can then be applied as functions
236      // COMPOSEOP operates on function-level operands (curry level 2)
237      // Left-to-right associative (matching Perl): f∘g∘h = (f∘g)∘h
238      composed_term = function composeop function => infix_apply
239        | function composeop trigfunction => infix_apply
240        | function composeop opfunction => infix_apply
241        | trigfunction composeop function => infix_apply
242        | trigfunction composeop trigfunction => infix_apply
243        | trigfunction composeop opfunction => infix_apply
244        | opfunction composeop function => infix_apply
245        | opfunction composeop trigfunction => infix_apply
246        | opfunction composeop opfunction => infix_apply
247        // Left-recursive for left-to-right associativity
248        | composed_term composeop function => infix_apply
249        | composed_term composeop trigfunction => infix_apply
250        | composed_term composeop opfunction => infix_apply;
251
252      // Composed functions can be applied like regular functions
253      tight_term += composed_term tight_term => prefix_apply;
254
255
256      term = tight_term
257      | term mulop tight_term => infix_apply_nary
258      | term mulop tight_term elideop => infix_apply_and_elide
259      // Perl: BINOP matches both AddOp and MulOp (ambiguous precedence from \mathbin)
260      | term binop tight_term => infix_apply_nary
261      | term binop tight_term elideop => infix_apply_and_elide
262      // Fallback: COMPOSEOP on general terms (for non-function-level composition)
263      | term composeop term => infix_apply
264      | operator applyop term => prefix_apply_applyop;
265
266
267      // Allow standalone functions/trigfunctions/opfunctions/operators as terms
268      // This is needed for (f*g)(x) where f and g are FUNCTION tokens
269      // opfunction here allows standalone \operatorname{R} to parse
270      // operator as term enables D - 1 (subtraction), D + G (addition)
271      term += function | trigfunction | opfunction | composed_term | operator;
272      // Allow elideop (\cdots) as a term for chains like y + i + \cdots + y_n
273      // Perl treats cdots as a regular term in addition chains
274      term += elideop;
275
276      // Higher-order operator terms: functions as standalone objects multiplied by factors
277      // `2\sin` = `2 * sin`, `2\sin\cos` = `2 * sin * cos`
278      // These are term-level (not tight_term) so they don't interfere with
279      // function application: `2\sin x` = `2 * sin(x)` (not `(2*sin) * x`)
280      tight_opterm = factor function => apply_invisible_times
281        | factor trigfunction => apply_invisible_times
282        | factor opfunction => apply_invisible_times
283        // Consecutive functions multiply: fgh → f·g·h (Perl Factor moreFactors)
284        | function function => apply_invisible_times
285        | function trigfunction => apply_invisible_times
286        | function opfunction => apply_invisible_times
287        | tight_opterm function => apply_invisible_times
288        | tight_opterm trigfunction => apply_invisible_times
289        | tight_opterm opfunction => apply_invisible_times;
290      term += tight_opterm;
291
292      // Expressions
293      expression = term
294        | expression addop term => infix_apply_nary
295        | expression addop term elideop => infix_apply_and_elide
296        | addop tight_term => prefix_apply
297        | factor addop => postfix_apply
298        | expression addop => postfix_apply
299        // Perl MathGrammar L236: addExpressionModifier: MODIFIEROP Expression
300        // => Apply(modifierop, expr, expr2). Handles infix `a mod b`.
301        | expression modifierop expression => infix_apply
302        // Perl MathGrammar L236: addExpressionModifier: MODIFIER
303        // Standalone postfix modifier (e.g. `8\pmod{3}` → annotated(8, pmod(3)))
304        // Placed at expression level so MODIFIER binds BEFORE RELOP.
305        // e.g. `5 ≡ 8 \pmod{3}` → `5 ≡ annotated(8, pmod(3))` not `annotated(5≡8, pmod(3))`
306        | expression modifier => postfix_modifier_apply
307        // Perl MathGrammar L224-233: OPEN relop/modifierop Expression balancedClose
308        // Parenthesized modifier expressions: x(>0) → annotated(x, Fence(>0))
309        | expression lparen relop expression rparen => annotated_fenced_modifier
310        | expression lparen modifierop expression rparen => annotated_fenced_modifier
311        // Perl MathGrammar L223: PUNCT? OPEN relop Expression CLOSE
312        // Semicolon annotation: a;(<e) → annotated(a, absent < e)
313        | expression punct lparen relop expression rparen => annotated_punct_fenced_modifier
314        | expression punct lparen modifierop expression rparen => annotated_punct_fenced_modifier;
315
316      // Formula
317      // Perl MathGrammar L73/236: MODIFIEROP Expression => Apply(mod, Absent, expr)
318      modifier_expression = modifierop expression => modifier_prefix_apply;
319      // Perl: within a Formula, comma-separated expressions after a relop form a list RHS.
320      // e.g. a=b,c,d → a = list(b,c,d), not list(a=b, c, d).
321      // Uses formula_list_apply which rejects items containing relops (those belong at statement level).
322      formula_list = expression punct expression => formula_list_apply
323        | formula_list punct expression => formula_list_apply;
324
325      // ASF migration item 5 (Option A semantics, 2026-05-19, user-
326      // articulated): `modified_term` is a `tight_term` carrying ONE
327      // relop modifier. Single-relop only — multi-relop chains
328      // (`x < y < 0`) stay at the formula level via the existing
329      // multirelation flattening.
330      //
331      // Action: reuse `infix_relation`. For a single relop call it
332      // produces `Apply(relop, [tight_term, expression])` — the same
333      // shape as `formula relop expression` reduces to for the base
334      // case. So `x = 0` parses identically via either route, ASF /
335      // tree-iter dedup eliminates the duplicate.
336      //
337      // Why this category exists: it enables comma-list contexts
338      // where each item carries its own relop — most notably
339      // function arguments like `P(x = 0, y < 0)`. Today such
340      // input is `ltx_math_unparsed` because `formula_list_apply`
341      // rejects relational items (a comma-list-of-relations is
342      // semantically a different beast from a relop chain RHS).
343      // Modified_term threaded through `list_apply` (not
344      // `formula_list_apply`) gives the desired list-of-relations
345      // shape. Surpass-Perl improvement — the corresponding Perl
346      // grammar via `addEasyArgs` handles this through a different
347      // path with the same effect; here we keep Perl's semantic
348      // outcome.
349      //
350      // See `docs/math/MATH_PARSER_ASF_TIEBREAKING.md` (commit
351      // 5cde377610) for the proposal context.
352      modified_term = tight_term relop expression => infix_relation;
353      // Phase 1: only the all-modified-terms variants. Mixed-content
354      // variants (`modified_term punct expression`, etc.) deferred
355      // until a witness shows them needed, to keep ambiguity growth
356      // tight (the `parse_tree_count_limits` regression test is the
357      // canary).
358      formula_list += modified_term punct modified_term => modified_list_apply
359        | formula_list punct modified_term => modified_list_apply;
360      // Comma-separated term lists: term, term, term, ...
361      // Used for angle-bracket inner products <x,y>, <a,b,c>, etc.
362      // Also includes limit_from_term for patterns like (1+, 0+, 1-, 0-).
363      term_list = term punct term => list_apply
364        | term_list punct term => list_apply
365        | limit_from_term punct term => list_apply
366        | limit_from_term punct limit_from_term => list_apply
367        | term_list punct limit_from_term => list_apply
368        | term punct limit_from_term => list_apply;
369
370      // Perl MathGrammar L709-711: Two-part relops (>=, <=, <<, >>)
371      two_part_relop = langle_rel langle_rel => two_part_relop_combine
372        | rangle_rel rangle_rel => two_part_relop_combine
373        | langle_rel relop_equals => two_part_relop_combine
374        | rangle_rel relop_equals => two_part_relop_combine;
375
376      formula = expression
377        | formula relop expression => infix_relation
378        | formula two_part_relop expression => infix_relation
379        // NOTE: deliberately NO `formula relop formula_list` rule. A bare
380        // (unparenthesized) comma-list is NOT a single expression, so it can
381        // never be a relation operand: `0 < x,y` can only mean
382        // `list(0<x, y)` (the comma splits at the statements level via
383        // `statements punct statement`), never `0 < list(x,y)`. Admitting the
384        // list as a relop RHS produced a spurious parse for every
385        // comma-after-relation, multiplying enumeration combinatorially down a
386        // formula_list (a top ambiguity-explosion source — `1510.03361`'s
387        // 5000-tree-cap equations). Parenthesized lists `(x,y)` remain single
388        // expressions via `lparen formula_list rparen => fenced`.
389        | formula relop => postfix_relop
390        // Perl moreRelations: `relop moreRelations` — consecutive relops chain without intervening terms
391        // e.g. `A ∈ ∞ ∋` → the ∈ absorbs ∞, then ∋ appends to the chain (no absent)
392        | formula relop relop => consecutive_relop_chain
393        | formula arrow expression => infix_relation
394        // A trailing ARROW with no RHS (e.g. `a \to`, `x \mapsto`) — postfix
395        // relation with an absent right operand, like `formula relop =>
396        // postfix_relop`. Perl yields `a to absent`; without this it fell to
397        // ltx_math_unparsed.
398        | formula arrow => postfix_relop
399        | arrow expression => prefix_arrow_apply
400        // Arrow-wrapped content (from amscd XMWrap role="ARROW"):
401        // Parsed as a prefix arrow application on the enclosed content.
402        | start_arrow arrow expression end_arrow => arrow_wrap_apply
403        | start_arrow arrow end_arrow => arrow_wrap_solo
404        // Perl MathGrammar L81: AnyOp Expression => Apply(AnyOp, Absent(), Expression)
405        // Leading relop with implied absent left operand (e.g. "= e + f + g" in eqnarray)
406        | relop expression => prefix_relop_apply
407        | metarelop expression => prefix_relop_apply
408        | modifier_expression;
409
410      // Perl MathGrammar: Factor includes preScripted['bigop'] as standalone
411      // So standalone bigops can form statements (needed for list expressions like \int \quad \int)
412      statement = formula
413        | statement metarelop formula => infix_relation
414        // A METARELOP (e.g. `:`) whose RHS is a `\quad`-separated list, mirroring
415        // the RELOP rule `formula relop formula_list` (line ~375): `a : b \quad c`
416        // → `a colon list@(b, c)`. Without it, only the bare RELOP form parsed
417        // (`a = b \quad c` worked, `a : b \quad c` fell to ltx_math_unparsed) —
418        // common in `\forall x : P \quad Q`-style notation.
419        | statement metarelop formula_list => infix_relation
420        // A trailing METARELOP with no RHS (e.g. `x \mapsto`) — postfix relation
421        // with an absent right operand, mirroring `formula relop => postfix_relop`
422        // for plain relops. Perl yields `x maps-to absent`; without this the
423        // formula fell to ltx_math_unparsed.
424        | statement metarelop => postfix_relop
425        | metarelop formula => prefix_metarelop_apply
426        | any_bigop | composed_bigop
427        | operator | compound_operator
428        | function | trigfunction
429        // Bare operators can form comma-separated lists: +,-,×
430        | addop | mulop | binop | relop | arrow
431  ;
432
433      end_punct = punct | period;
434      statements = statement
435        | statement end_punct => postfix_embellished
436        | statements end_punct => postfix_embellished
437        | statements punct statement => list_apply
