1use crate::semantics::*;
2use marpa::grammar::Grammar as MarpaGrammar;
3use marpa::result::Result;
4use marpa::tree_builder::TreeBuilder;
5
6pub fn init_grammar() -> Result<(MarpaGrammar, Actions, TreeBuilder)> {
7  // We create a declarative macro language of our own, in the spirit of the Marpa SLIF
8  default_registry!();
9  // Tokens, to be used in rules directly
10  token!(atom ~ "ATOM");
11  token!(unknown ~ "UNKNOWN");
12  token!(id ~ "ID");
13  token!(array ~ "ARRAY");
14  token!(number ~ "NUMBER");
15  token!(punct ~ "PUNCT");
16  token!(period ~ "PERIOD");
17  token!(addop ~ "ADDOP");
18  token!(mulop ~ "MULOP");
19  token!(relop ~ "RELOP");
20  token!(elideop ~ "ELIDEOP");
21  token!(langle_rel = "RELOP:less-than");
22  token!(langle_open = "OPEN:langle");
23  token!(langle = [langle_rel langle_open]);
24  token!(rangle_rel = "RELOP:greater-than");
25  token!(rangle_close = "CLOSE:rangle");
26  token!(rangle =[rangle_rel rangle_close]);
27  token!(vertbar ~ "VERTBAR");
28  token!(singlevertbar = "VERTBAR:|");
29  token!(middle_bar = "MIDDLE:|");
30  token!(middle_parallel = "MIDDLE:parallel-to");
31  token!(midbar = [vertbar middle_bar middle_parallel]);
32  token!(lbrace = "OPEN:{");
33  token!(rbrace = "CLOSE:}");
34  token!(lparen = "OPEN:(");
35  token!(rparen = "CLOSE:)");
36  token!(lbracket = "OPEN:[");
37  token!(rbracket = "CLOSE:]");
38  token!(relop_equals = "RELOP:equals");
39  token!(metarelop ~ "METARELOP");
40  token!(modifierop ~ "MODIFIEROP");
41  token!(modifier ~ "MODIFIER");
42  token!(arrow ~ "ARROW");
43  token!(binop ~ "BINOP");
44  token!(postfix ~ "POSTFIX");
45  token!(function ~ "FUNCTION");
46  token!(opfunction ~ "OPFUNCTION");
47  token!(trigfunction ~ "TRIGFUNCTION");
48  token!(applyop ~ "APPLYOP");
49  token!(composeop ~ "COMPOSEOP");
50  token!(supop ~ "SUPOP");
51  token!(open ~ "OPEN");
52  token!(close ~ "CLOSE");
53  token!(middle ~ "MIDDLE");
54  token!(bigop ~ "BIGOP");
55  token!(sumop ~ "SUMOP");
56  token!(intop ~ "INTOP");
57  token!(limitop ~ "LIMITOP");
58  token!(diffop ~ "DIFFOP");
59  token!(operator ~ "OPERATOR");
60  token!(start_postsubscript ~ "start_POSTSUBSCRIPT");
61  token!(end_postsubscript ~ "end_POSTSUBSCRIPT");
62  token!(start_postsuperscript ~ "start_POSTSUPERSCRIPT");
63  token!(end_postsuperscript ~ "end_POSTSUPERSCRIPT");
64  token!(start_floatsuperscript ~ "start_FLOATSUPERSCRIPT");
65  token!(end_floatsuperscript ~ "end_FLOATSUPERSCRIPT");
66  token!(start_floatsubscript ~ "start_FLOATSUBSCRIPT");
67  token!(end_floatsubscript ~ "end_FLOATSUBSCRIPT");
68
69  rules!(
70    // Factors
71    factor_base = unknown | number | id | atom | opfunction;
72    factor = factor_base;
73    // Terms
74    // Perl: bigop = BIGOP | SUMOP | INTOP | LIMITOP | DIFFOP
75    any_bigop = bigop | sumop | intop | limitop | diffop;
76    // Adjacent bigops compose like higher-order functions:
77    // \int\iint => integral(double-integral), itself a compound operator
78    composed_bigop = any_bigop any_bigop => prefix_apply
79      | any_bigop composed_bigop => prefix_apply;
80
81    // Compound operators: OPERATOR composed with functions/other operators (right-recursive)
82    // D sin => Apply(D, sin), D D sin => Apply(D, Apply(D, sin))
83    // Must end with a function/trigfunction (no bare operator-only compounds)
84    compound_operator = operator trigfunction => prefix_apply
85      | operator function => prefix_apply
86      | operator compound_operator => prefix_apply;
87
88    tight_term = factor
89      | tight_term factor => apply_invisible_times
90      // Perl MathGrammar L423: POSTFIX (e.g. n!) => Apply(op, term)
91      | tight_term postfix => apply_postfix
92      | function tight_term => prefix_apply
93      | trigfunction tight_term => prefix_apply
94      | any_bigop tight_term => prefix_apply
95      | composed_bigop tight_term => prefix_apply
96      | compound_operator tight_term => prefix_apply
97      | operator factor => prefix_apply
98      | factor_base applyop tight_term => prefix_apply_applyop;
99
100    term = tight_term
101    | term mulop tight_term => infix_apply_nary
102    | term mulop tight_term elideop => infix_apply_and_elide
103    // Perl: COMPOSEOP creates function composition (f∘g)
104    // Functions can participate as operands of composition
105    | term composeop term => infix_apply
106    | operator applyop term => prefix_apply_applyop;
107
108    // Allow standalone functions/trigfunctions as terms for composition
109    // This is needed for (f*g)(x) where f and g are FUNCTION tokens
110    term += function | trigfunction;
111
112    // Expressions
113    expression = term
114      | expression addop term => infix_apply_nary
115      | expression addop term elideop => infix_apply_and_elide
116      | addop tight_term => prefix_apply
117      | factor addop => postfix_apply
118      // Perl MathGrammar L236: addExpressionModifier: MODIFIEROP Expression
119      // => Apply(modifierop, expr, expr2). Handles infix `a mod b`.
120      | expression modifierop expression => infix_apply;
121
122    // Formula
123    // Perl MathGrammar L73/236: MODIFIEROP Expression => Apply(mod, Absent, expr)
124    modifier_expression = modifierop expression => prefix_apply;
125    // Perl: within a Formula, comma-separated expressions after a relop form a list RHS.
126    // e.g. a=b,c,d → a = list(b,c,d), not list(a=b, c, d).
127    // Uses formula_list_apply which rejects items containing relops (those belong at statement level).
128    formula_list = expression punct expression => formula_list_apply
129      | formula_list punct expression => formula_list_apply;
130    // Perl MathGrammar L709-711: Two-part relops (>=, <=, <<, >>)
131    two_part_relop = langle_rel langle_rel => two_part_relop_combine
132      | rangle_rel rangle_rel => two_part_relop_combine
133      | langle_rel relop_equals => two_part_relop_combine
134      | rangle_rel relop_equals => two_part_relop_combine;
135
136    formula = expression
137      | formula relop expression => infix_relation
138      | formula two_part_relop expression => infix_relation
139      | formula relop formula_list => infix_relation
140      | formula relop => postfix_relop
141      | formula arrow expression => infix_relation
142      | arrow expression => prefix_arrow_apply
143      // Perl MathGrammar L81: AnyOp Expression => Apply(AnyOp, Absent(), Expression)
144      // Leading relop with implied absent left operand (e.g. "= e + f + g" in eqnarray)
145      | relop expression => prefix_relop_apply
146      | modifier_expression;
147
148    // Perl MathGrammar: Factor includes preScripted['bigop'] as standalone
149    // So standalone bigops can form statements (needed for list expressions like \int \quad \int)
150    statement = formula
151      | statement metarelop formula => infix_relation
152      | metarelop formula => prefix_metarelop_apply
153      | any_bigop | composed_bigop
154      | operator | compound_operator
155      | function | trigfunction
156      // Bare operators can form comma-separated lists: +,-,×
157      | addop | mulop | relop | arrow;
158
159    end_punct = punct | period;
160    statements = statement
161      | statement end_punct => postfix_embellished
162      | statements punct statement => list_apply
163      // Perl: MorphVertbar — VERTBAR as conditional modifier: x | y,z,t
164      | statement vertbar statements => vertbar_modifier;
165
166    // Extensions, now that we have more category variables defined
167    fenced_factor = lbrace expression rbrace    => fenced
168           | lbracket expression rbracket       => fenced
169           | lparen formula rparen              => fenced
170           | lparen term punct term rparen      => interval
171           | lparen term punct term rbracket    => interval
172           | lbracket term punct term rbracket  => interval
173           | lbracket term punct term rparen  => interval
174           | rbracket term punct term lbracket => interval
175           | singlevertbar expression singlevertbar => fenced
176           // Perl's Fence for comma-separated items in braces: {a,b} and {a,b,c}
177           | lbrace term punct term rbrace => fence
178           | lbrace term punct term punct term rbrace => fence
179           // Perl: {a|b} conditional-set with VERTBAR or MIDDLE separator
180           | lbrace formula singlevertbar formula rbrace => fence
181           | lbrace formula middle_bar formula rbrace => fence
182           | lbrace formula metarelop formula rbrace => fence;
183    factor += fenced_factor;
184
185    // UNKNOWN followed by fenced args => function application (Perl: Apply[UNKNOWN atom_args])
186    // f(x) => f@(x), g(a+b) => g@(a+b). Only active when MATHPARSER_SPECULATE is set.
187    // Without speculation, this parse is pruned and Marpa uses invisible-times instead.
188    tight_term += unknown fenced_factor => speculative_prefix_apply;
189    // OPFUNCTION followed by fenced args => function application
190    // \operatorname{cov}(L) => cov@(L). Always treated as application, not multiplication.
191    tight_term += opfunction fenced_factor => prefix_apply;
192    // Perl IntFactor L640-651: diffd followed by ATOM/UNKNOWN/ID => Apply(DIFFOP(d), var)
193    // Uses existing `unknown` terminal; semantic action checks text is literally "d".
194    // At factor level so it can appear as right operand of invisible_times.
195    factor += unknown factor_base => diffop_apply;
196
197    // Script content
198    postsubarg = start_postsubscript expression end_postsubscript => faux_wrap;
199    postsuperarg = start_postsuperscript expression end_postsuperscript => faux_wrap
200      // TODO: what other kinds of arguments are accepted in scripts? Should we do "anything"?
201      | start_postsuperscript supop end_postsuperscript => faux_wrap;
202    floatsubarg = start_floatsubscript expression end_floatsubscript => faux_wrap;
203    floatsuperarg = start_floatsuperscript expression end_floatsuperscript => faux_wrap;
204    // standalone top-level variants of floating scripts:
205    floatsubscript = start_floatsubscript expression end_floatsubscript => standalone_script;
206    floatsuperscript = start_floatsuperscript expression end_floatsuperscript => standalone_script;
207    // Scripted factors -- avoid adding ambiguity in the left-right order of collection
208    // first ALL left (=float), then right (=post).
209    scripted_factor_l11 = floatsuperarg factor_base => prefix_script;
210    scripted_factor_l12 = floatsubarg factor_base => prefix_script;
211    scripted_factor_l1 = scripted_factor_l11 | scripted_factor_l12;
212    scripted_factor_l2 = floatsuperarg scripted_factor_l12 => prefix_script
213      | floatsubarg scripted_factor_l11 => prefix_script;
214
215    scripted_factor_r11 = factor_base postsuperarg => postfix_script
216      | scripted_factor_l1 postsuperarg => postfix_script
217      | scripted_factor_l2 postsuperarg => postfix_script
218      | fenced_factor postsuperarg => postfix_script;
219    scripted_factor_r12 = factor_base postsubarg => postfix_script
220      | scripted_factor_l1 postsubarg => postfix_script
221      | scripted_factor_l2 postsubarg => postfix_script
222      | fenced_factor postsubarg => postfix_script;
223    scripted_factor_r1 = scripted_factor_r11 | scripted_factor_r12;
224    scripted_factor_r2 = scripted_factor_r12 postsuperarg => postfix_script
225      | scripted_factor_r11 postsubarg => postfix_script;
226    factor += scripted_factor_l1 | scripted_factor_l2 | scripted_factor_r1 | scripted_factor_r2;
227
228    // Scripted bigops: \int_0^\infty, \sum_{n=1}^N, etc.
229    // These are bigops with post-scripts that still act as prefix operators.
230    // Perl: preScripted['INTOP'] addIntOpArgs / preScripted['bigop'] addOpArgs
231    // Chain scripts: first sub then super, or vice versa (like scripted_factor_r1/r2)
232    scripted_bigop_r1 = any_bigop postsuperarg => postfix_script
233      | any_bigop postsubarg => postfix_script;
234    scripted_bigop = scripted_bigop_r1
235      | scripted_bigop_r1 postsuperarg => postfix_script
236      | scripted_bigop_r1 postsubarg => postfix_script;
237    tight_term += scripted_bigop tight_term => prefix_apply;
238    // Scripted bigops can also appear as standalone statements
239    statement += scripted_bigop;
240
241    anyop = addop | mulop | relop | arrow | metarelop
242      | bigop | sumop | intop
243      | limitop | diffop | vertbar | supop
244      | modifierop | composed_bigop | operator | compound_operator;
245
246    anyscript = floatsuperscript | floatsubscript;
247
248    anything = statements | anyop | anyscript |
249      anyop anyop => compound_operator_2
250  );
251  // | term_argument postsuperarg tex_argument  => post_script
252  // | term_argument postsubscript tex_argument    => post_script
253  start!(anything);
254
255  // Also prepare the tree builder rules here (for now)
256  Ok((grammar!(), actions!(), builder!()))
257}
