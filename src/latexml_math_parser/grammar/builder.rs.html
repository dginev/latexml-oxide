1use crate::semantics::*;
2use marpa::grammar::Grammar as MarpaGrammar;
3use marpa::result::Result;
4use marpa::tree_builder::TreeBuilder;
5
6pub fn init_grammar() -> Result<(MarpaGrammar, Actions, TreeBuilder)> {
7  // We create a declarative macro language of our own, in the spirit of the Marpa SLIF
8  default_registry!();
9  // Tokens, to be used in rules directly
10  token!(atom ~ "ATOM");
11  token!(unknown ~ "UNKNOWN");
12  token!(id ~ "ID");
13  token!(array ~ "ARRAY");
14  token!(number ~ "NUMBER");
15  token!(punct ~ "PUNCT");
16  token!(period ~ "PERIOD");
17  token!(addop ~ "ADDOP");
18  token!(mulop ~ "MULOP");
19  token!(relop ~ "RELOP");
20  token!(elideop ~ "ELIDEOP");
21  token!(langle_rel = "RELOP:less-than");
22  token!(langle_open = "OPEN:langle");
23  token!(langle = [langle_rel langle_open]);
24  token!(rangle_rel = "RELOP:greater-than");
25  token!(rangle_close = "CLOSE:rangle");
26  token!(rangle =[rangle_rel rangle_close]);
27  token!(vertbar ~ "VERTBAR");
28  token!(singlevertbar = "VERTBAR:|");
29  token!(middle_bar = "MIDDLE:|");
30  token!(middle_parallel = "MIDDLE:parallel-to");
31  token!(midbar = [vertbar middle_bar middle_parallel]);
32  token!(lbrace = "OPEN:{");
33  token!(rbrace = "CLOSE:}");
34  token!(lparen = "OPEN:(");
35  token!(rparen = "CLOSE:)");
36  token!(lbracket = "OPEN:[");
37  token!(rbracket = "CLOSE:]");
38  token!(metarelop ~ "METARELOP");
39  token!(modifierop ~ "MODIFIEROP");
40  token!(modifier ~ "MODIFIER");
41  token!(arrow ~ "ARROW");
42  token!(binop ~ "BINOP");
43  token!(postfix ~ "POSTFIX");
44  token!(function ~ "FUNCTION");
45  token!(opfunction ~ "OPFUNCTION");
46  token!(trigfunction ~ "TRIGFUNCTION");
47  token!(applyop ~ "APPLYOP");
48  token!(composeop ~ "COMPOSEOP");
49  token!(supop ~ "SUPOP");
50  token!(open ~ "OPEN");
51  token!(close ~ "CLOSE");
52  token!(middle ~ "MIDDLE");
53  token!(bigop ~ "BIGOP");
54  token!(sumop ~ "SUMOP");
55  token!(intop ~ "INTOP");
56  token!(limitop ~ "LIMITOP");
57  token!(diffop ~ "DIFFOP");
58  token!(operator ~ "OPERATOR");
59  token!(start_postsubscript ~ "start_POSTSUBSCRIPT");
60  token!(end_postsubscript ~ "end_POSTSUBSCRIPT");
61  token!(start_postsuperscript ~ "start_POSTSUPERSCRIPT");
62  token!(end_postsuperscript ~ "end_POSTSUPERSCRIPT");
63  token!(start_floatsuperscript ~ "start_FLOATSUPERSCRIPT");
64  token!(end_floatsuperscript ~ "end_FLOATSUPERSCRIPT");
65  token!(start_floatsubscript ~ "start_FLOATSUBSCRIPT");
66  token!(end_floatsubscript ~ "end_FLOATSUBSCRIPT");
67
68  rules!(
69    // Factors
70    factor_base = unknown | number | id | atom | opfunction;
71    factor = factor_base;
72    // Terms
73    tight_term = factor
74      | tight_term factor => apply_invisible_times
75      // TODO: develop this further
76      | function factor => prefix_apply;
77
78    term = tight_term
79    | term mulop tight_term => infix_apply_nary
80    | term mulop tight_term elideop => infix_apply_and_elide
81    | bigop tight_term => prefix_apply;
82
83    // Expressions
84    expression = term
85      | expression addop term => infix_apply_nary
86      | expression addop term elideop => infix_apply_and_elide
87      | addop tight_term => prefix_apply
88      | factor addop => postfix_apply;
89
90    // Formula
91    formula = expression
92      | formula relop expression => infix_relation;
93
94    statement = formula
95      | statement metarelop formula => infix_relation;
96
97    end_punct = punct | period;
98    statements = statement
99      | statement end_punct => postfix_embellished
100      | statement punct statement => infix_apply;
101
102    // Extensions, now that we have more category variables defined
103    fenced_factor = lbrace expression rbrace    => fenced
104           | lbracket expression rbracket       => fenced
105           | lparen formula rparen              => fenced
106           | lparen term punct term rparen      => interval
107           | lparen term punct term rbracket    => interval
108           | lbracket term punct term rbracket  => interval
109           | lbracket term punct term rparen  => interval
110           | rbracket term punct term lbracket => interval
111           | singlevertbar expression singlevertbar => fenced
112           // Perl's Fence for comma-separated items in braces: {a,b} and {a,b,c}
113           | lbrace term punct term rbrace => fence
114           | lbrace term punct term punct term rbrace => fence;
115    factor += fenced_factor;
116
117
118    // Script content
119    postsubarg = start_postsubscript expression end_postsubscript => faux_wrap;
120    postsuperarg = start_postsuperscript expression end_postsuperscript => faux_wrap
121      // TODO: what other kinds of arguments are accepted in scripts? Should we do "anything"?
122      | start_postsuperscript supop end_postsuperscript => faux_wrap;
123    floatsubarg = start_floatsubscript expression end_floatsubscript => faux_wrap;
124    floatsuperarg = start_floatsuperscript expression end_floatsuperscript => faux_wrap;
125    // standalone top-level variants of floating scripts:
126    floatsubscript = start_floatsubscript expression end_floatsubscript => standalone_script;
127    floatsuperscript = start_floatsuperscript expression end_floatsuperscript => standalone_script;
128    // Scripted factors -- avoid adding ambiguity in the left-right order of collection
129    // first ALL left (=float), then right (=post).
130    scripted_factor_l11 = floatsuperarg factor_base => prefix_script;
131    scripted_factor_l12 = floatsubarg factor_base => prefix_script;
132    scripted_factor_l1 = scripted_factor_l11 | scripted_factor_l12;
133    scripted_factor_l2 = floatsuperarg scripted_factor_l12 => prefix_script
134      | floatsubarg scripted_factor_l11 => prefix_script;
135
136    scripted_factor_r11 = factor_base postsuperarg => postfix_script
137      | scripted_factor_l1 postsuperarg => postfix_script
138      | scripted_factor_l2 postsuperarg => postfix_script
139      | fenced_factor postsuperarg => postfix_script;
140    scripted_factor_r12 = factor_base postsubarg => postfix_script
141      | scripted_factor_l1 postsubarg => postfix_script
142      | scripted_factor_l2 postsubarg => postfix_script
143      | fenced_factor postsubarg => postfix_script;
144    scripted_factor_r1 = scripted_factor_r11 | scripted_factor_r12;
145    scripted_factor_r2 = scripted_factor_r12 postsuperarg => postfix_script
146      | scripted_factor_r11 postsubarg => postfix_script;
147    factor += scripted_factor_l1 | scripted_factor_l2 | scripted_factor_r1 | scripted_factor_r2;
148
149    anyop = addop | mulop | relop | metarelop
150      | bigop | sumop | intop
151      | limitop | diffop | vertbar | supop;
152
153    anyscript = floatsuperscript | floatsubscript;
154
155    anything = statements | anyop | anyscript |
156      anyop anyop => compound_operator_2
157  );
158  // | term_argument postsuperarg tex_argument  => post_script
159  // | term_argument postsubscript tex_argument    => post_script
160  start!(anything);
161
162  // Also prepare the tree builder rules here (for now)
163  Ok((grammar!(), actions!(), builder!()))
164}
