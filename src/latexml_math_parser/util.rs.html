11/// Generate a textual token for each node; The parser operates on this encoded
12/// string.
13pub fn node_to_grammar_lexemes(mathnode: &Node, idx: &mut usize) -> (Vec<String>, Vec<Node>) {
14  let mut lexemes = Vec::new();
15  let mut nodes = Vec::new();
16  let top_role_opt = mathnode.get_attribute("role");
17  if let Some(ref top_role) = top_role_opt {
18    *idx += 1;
19    lexemes.push(format!("start_{top_role}:start:{idx}"));
20    nodes.push(mathnode.clone());
21  }
22  let child_nodes = filter_hints(mathnode.get_child_nodes());
23  for node in child_nodes.into_iter() {
24    if node.get_name() == "XMApp" {
25      let (mut inner_lexes, mut inner_nodes) = node_to_grammar_lexemes(&node, idx);
26      for (inner_lex, inner_node) in inner_lexes.drain(..).zip(inner_nodes.drain(..)) {
27        lexemes.push(inner_lex);
28        nodes.push(inner_node);
29      }
30    } else {
31      let role = get_grammatical_role(&node);
32      let mut text = get_token_meaning(&node);
33      if text.is_empty() {
34        text = "UNKNOWN".to_string();
35      }
36      *idx += 1;
37      let lexeme = format!("{role}:{text}:{idx}").replace(' ', "");
38      lexemes.push(lexeme);
39      nodes.push(node);
40    }
41  }
42  if let Some(top_role) = top_role_opt {
43    *idx += 1;
44    lexemes.push(format!("end_{top_role}:end:{idx}"));
45    nodes.push(mathnode.clone());
46  }
47  (lexemes, nodes)
48}

50/// Auxiliary separator for ROLE:style-lexeme into ("ROLE:style", '-', lexeme)
51pub fn distill_lexeme(name: &str) -> (&str, &str, &str) {
52  // dash separates styles, colons separate grammatical roles, and we are
53  // only trying to distill the last pure lexeme
54  // note that we are only trying to do this reasonably for letter-based names (UNKNOWN:italic-x),
55  // since some of the content symbols contain dashes themselves (e.g.
56  // OPERATOR:partial-differential)
57  if let Some(position) = name.rfind('-') {
58    let (base, trailer) = name.split_at(position);
59    let (sep, lexeme) = trailer.split_at(1);
60    (base, sep, lexeme)
61  } else if let Some(position) = name.rfind(':') {
62    let (base, trailer) = name.split_at(position);
63    let (sep, lexeme) = trailer.split_at(1);
64    (base, sep, lexeme)
65  } else {
66    ("", "", name)
67  }
68}

72/// Given a list of XML nodes (either libxml nodes, or array representations)
73/// return a list of XMRef's referring to those nodes;
74/// ensure each source node has an ID (if already instanciated as XML)
75/// or _xmkey if still in array rep. since it will get an ID later, and the connection re-made)
76/// Note that ltx:XMHint nodes are ephemeral and shouldn't be ref'd!
77/// likewise, we avoid creating XMRefs to XMRefs
78pub fn create_xmrefs(args: &mut [&mut XM], ctxt: ActionContext) -> Result<Vec<XM>, Box<dyn Error>> {
79  let nodes = ctxt.nodes;
80  let document = ctxt.document;
81  let mut refs = Vec::new();
82  for arg in args {
83    match arg {
84      XM::Token(props, _meta) => {
85        if let Some(id) = props.id.as_ref() {
86          refs.push(XM::Ref(XProps {
87            id: Some(id.clone()),
88            ..XProps::default()
89          }));
90        }
91      },
92      XM::Lexeme(lex, _) => {
93        // If arg is already XML, it's too late to get automatic ID's
94        let node = lookup_lex_node(lex, nodes).expect("lexemes should only have valid ids.");
95        // let qname   = document::get_node_qname(node, state);
96        // let nodebox     = document.get_node_box(node);
97
98        match node.get_attribute("id") {
99          //  already has id, so refer to it.
100          Some(id) => refs.push(XM::Ref(XProps {
101            id: Some(Cow::Owned(id)),
102            ..XProps::default()
103          })),
104          None => {
105            // If arg is already XML, it's too late to get automatic ID's
106            document.generate_id(&mut node.clone(), "")?;
107            refs.push(XM::Ref(XProps {
108              id: Some(Cow::Owned(
109                node
110                  .get_attribute("id")
111                  .expect("generate_id should always succeed in setting an id"),
112              )),
113              ..XProps::default()
114            }));
115          },
116        }
117      },
118      XM::Apply(_op, _args, ref mut props, _meta) => {
119        if let Some(id) = props.id.as_ref() {
120          refs.push(XM::Ref(XProps {
121            id: Some(id.clone()),
122            ..XProps::default()
123          }));
124        } else {
125          // not yet instanciated, so hasn't had chance to get auto-id; use _xmkey
126          let key = get_xmarg_id()?.to_string();
127          props.xmkey = Some(Cow::Owned(key.clone()));
128          refs.push(XM::Ref(XProps {
129            xmkey: Some(Cow::Owned(key)),
130            ..XProps::default()
131          }));
132        }
133      },
134      // clone an XMRef (w/o any attributes or id ?) rather than create an XMRef to an XMRef
135      XM::Ref(props) => {
136        refs.push(XM::Ref(props.clone()));
137      },
138      // TODO:
139      //   # XMHint's are ephemeral, they may disappear; so just clone it w/o id
140      //   if ($qname eq 'ltx:XMHint') {
141      //     my %attr = ($isarray ? %{ $$arg[1] }
142      //       : (map { $document->getNodeQName($_) => $_->getValue } $arg->attributes));
143      //     delete $attr{'xml:id'};
144      //     push(@refs, [$qname, {%attr}]); }
145      _ => todo!(),
146    }
147  }
148  Ok(refs)
149}
