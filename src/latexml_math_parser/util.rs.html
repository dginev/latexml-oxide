11/// Generate a textual token for each node; The parser operates on this encoded
12/// string.
13pub fn node_to_grammar_lexemes(mathnode: &Node, idx: &mut usize) -> (Vec<String>, Vec<Node>) {
14  let child_nodes = filter_hints(mathnode.get_child_nodes());
15  node_to_grammar_lexemes_from(mathnode, child_nodes, idx)
16}

18/// Same as `node_to_grammar_lexemes` but with pre-filtered child nodes.
19/// Used when `filter_hints` has already been called (to avoid double-filtering).
20pub fn node_to_grammar_lexemes_from(
21  mathnode: &Node,
22  child_nodes: Vec<Node>,
23  idx: &mut usize,
24) -> (Vec<String>, Vec<Node>) {
25  node_to_grammar_lexemes_ctx(mathnode, child_nodes, idx, false)

28fn node_to_grammar_lexemes_ctx(
29  mathnode: &Node,
30  child_nodes: Vec<Node>,
31  idx: &mut usize,
32  bigop_context: bool,
33) -> (Vec<String>, Vec<Node>) {
34  let mut lexemes = Vec::new();
35  let mut nodes = Vec::new();
36  let top_role_opt = mathnode.get_attribute("role");
37  if let Some(ref top_role) = top_role_opt {
38    *idx += 1;
39    // When in bigop context, emit BIGOPSUB/BIGOPSUP instead of POSTSUBSCRIPT/POSTSUPERSCRIPT
40    let mapped_role = if bigop_context {
41      match top_role.as_str() {
42        "POSTSUBSCRIPT" => "BIGOPSUB".to_string(),
43        "POSTSUPERSCRIPT" => "BIGOPSUP".to_string(),
44        _ => top_role.clone(),
45      }
46    } else {
47      top_role.clone()
48    };
49    lexemes.push(format!("start_{mapped_role}:start:{idx}"));
50    nodes.push(mathnode.clone());
51  }
52  // Track whether the last emitted token was a bigop (SUMOP/INTOP/etc.)
53  // so we can emit bigop-specific script tokens to reduce earley chart ambiguity.
54  let mut last_was_bigop = false;
55  for node in child_nodes.into_iter() {
56    // For XMRef nodes: resolve to get the target's role/meaning for lexing,
57    // but keep the ORIGINAL XMRef node in the output so the parse tree
58    // preserves XMRef indirection (matching Perl behavior).
59    // The get_grammatical_role/get_token_meaning functions already resolve XMRef
60    // internally, so we just need to handle the case where the target is a
61    // compound node (XMApp without role) — these need flattening.
62    // Note: we do NOT replace the node variable — we keep the XMRef.
63
64    if node.get_name() == "XMApp" && node.get_attribute("role").is_some() {
65      let role = node.get_attribute("role").unwrap();
66      // ARROW/METARELOP-role XMApps (decorated arrows like \xrightarrow{over},
67      // \xleftrightarrow[under]{over}) should be atomic terminals.
68      // Extract the meaning from the inner ARROW/METARELOP child token.
69      if role == "ARROW" || role == "METARELOP" {
70        let arrow_meaning = node.get_child_elements().into_iter()
71          .find(|ch| {
72            let cr = ch.get_attribute("role");
73            cr.as_deref() == Some("ARROW") || cr.as_deref() == Some("METARELOP")
74          })
75          .and_then(|ch| {
76            ch.get_attribute("meaning")
77              .or_else(|| ch.get_attribute("name"))
78              .or_else(|| Some(ch.get_content()))
79          })
80          .unwrap_or_else(|| role.to_string());
81        *idx += 1;
82        let lexeme = format!("{role}:{arrow_meaning}:{idx}").replace(' ', "");
83        lexemes.push(lexeme);
84        nodes.push(node);
85      } else if node.has_attribute("_rewrite") {
86        // Rewrite-created: treat as atomic token with the assigned role.
87        // Don't recurse — the inner structure was pre-parsed, and the role
88        // on this node overrides whatever the children contain.
89        let gram_role = get_grammatical_role(&node);
90        let mut text = get_token_meaning(&node);
91        if text.is_empty() {
92          text = "UNKNOWN".to_string();
93        }
94        *idx += 1;
95        lexemes.push(format!("{gram_role}:{text}:{idx}").replace(' ', ""));
96        nodes.push(node);
97      } else {
98        // Only recurse into XMApp nodes that have a role (scripts, etc.)
99        // Role-less XMApps (e.g. \sqrt, already-parsed structures) are atomic.
100        // Pass bigop_context for POSTSUBSCRIPT/POSTSUPERSCRIPT following a bigop
101        let is_script = matches!(role.as_str(), "POSTSUBSCRIPT" | "POSTSUPERSCRIPT");
102        let ctx = last_was_bigop && is_script;
103        let children = filter_hints(node.get_child_nodes());
104        let (mut inner_lexes, mut inner_nodes) = node_to_grammar_lexemes_ctx(&node, children, idx, ctx);
105        for (inner_lex, inner_node) in inner_lexes.drain(..).zip(inner_nodes.drain(..)) {
106          lexemes.push(inner_lex);
107          nodes.push(inner_node);
108        }
109        // Script following a bigop is still "bigop context" for the next script
110        if !is_script {
111          last_was_bigop = false;
112        }
113      }
114    } else if node.get_name() == "XMArg" {
115      // XMArg is a transparent wrapper (e.g. from \lx@post@subscript).
116      // Recurse into its children so the grammar can parse them individually.
117      // E.g. _{ij} should emit UNKNOWN:i + UNKNOWN:j, not ATOM:ij.
118      let arg_children = filter_hints(node.get_child_nodes());
119      let (mut inner_lexes, mut inner_nodes) =
120        node_to_grammar_lexemes_ctx(&node, arg_children, idx, false);
121      for (inner_lex, inner_node) in inner_lexes.drain(..).zip(inner_nodes.drain(..)) {
122        lexemes.push(inner_lex);
123        nodes.push(inner_node);
124      }
125    } else {
126      let role = get_grammatical_role(&node);
127      let mut text = get_token_meaning(&node);
128      if text.is_empty() {
129        text = "UNKNOWN".to_string();
130      }
131      *idx += 1;
132      // Track bigop tokens for bigop-specific script token emission
133      let is_bigop = matches!(role.as_str(), "SUMOP" | "INTOP" | "LIMITOP" | "DIFFOP" | "BIGOP");
134      last_was_bigop = is_bigop;
135      // Normalize langle/rangle meaning for consistent grammar matching.
136      // Do NOT remap to parentheses — langle_open/rangle_close tokens need
137      // distinct identity for QM bra-ket rules vs conditional probability.
138      let lexeme = if role == "OPEN" && text == "⟨" {
139        format!("OPEN:langle:{idx}")
140      } else if role == "CLOSE" && text == "⟩" {
141        format!("CLOSE:rangle:{idx}")
142      } else if role == "UNKNOWN" && text == "d" {
143        // M4: Emit XDIFFUNK for possible differential-d tokens.
144        // Only "d" tokens can be diffops; other unknowns skip the diffop rule.
145        format!("XDIFFUNK:{text}:{idx}")
146      } else if role == "ID" && text == "d" {
147        format!("XDIFFID:{text}:{idx}")
148      } else {
149        format!("{role}:{text}:{idx}").replace(' ', "")
150      };
151      lexemes.push(lexeme);
152      nodes.push(node);
153    }
154  }
155  if let Some(ref top_role) = top_role_opt {
156    *idx += 1;
157    let mapped_end = if bigop_context {
158      match top_role.as_str() {
159        "POSTSUBSCRIPT" => "BIGOPSUB".to_string(),
160        "POSTSUPERSCRIPT" => "BIGOPSUP".to_string(),
161        _ => top_role.clone(),
162      }
163    } else {
164      top_role.clone()
165    };
166    lexemes.push(format!("end_{mapped_end}:end:{idx}"));
167    nodes.push(mathnode.clone());
168  }
169  (lexemes, nodes)
170}

172/// Auxiliary separator for ROLE:style-lexeme into ("ROLE:style", '-', lexeme)
173pub fn distill_lexeme(name: &str) -> (&str, &str, &str) {
174  // dash separates styles, colons separate grammatical roles, and we are
175  // only trying to distill the last pure lexeme
176  // note that we are only trying to do this reasonably for letter-based names (UNKNOWN:italic-x),
177  // since some of the content symbols contain dashes themselves (e.g.
178  // OPERATOR:partial-differential)
179  if let Some(position) = name.rfind('-') {
180    let (base, trailer) = name.split_at(position);
181    let (sep, lexeme) = trailer.split_at(1);
182    (base, sep, lexeme)
183  } else if let Some(position) = name.rfind(':') {
184    let (base, trailer) = name.split_at(position);
185    let (sep, lexeme) = trailer.split_at(1);
186    (base, sep, lexeme)
187  } else {
188    ("", "", name)
189  }
190}

192/// Parse an XMHint `width` attribute string to points.
193/// Supports "3.0mu" (mu → pt by dividing by 1.8) and "1.667pt"/"0.16667em".
194/// Handles glue specs like "2.77pt plus 2.77pt" by extracting base dimension.
195fn get_xmhint_spacing(width: &str) -> f64 {
196  let width = width.trim();
197  if width.is_empty() {
198    return 0.0;
199  }
200  // Strip glue stretch/shrink: "2.77pt plus 2.77pt" → "2.77pt"
201  let base = width
202    .split_once(" plus")
203    .or_else(|| width.split_once(" minus"))
204    .map_or(width, |(base, _)| base)
205    .trim();
206  let unit_start = base.find(|c: char| c.is_alphabetic()).unwrap_or(base.len());
207  let (number_str, unit) = base.split_at(unit_start);
208  let number: f64 = number_str.trim().parse().unwrap_or(0.0);
209  match unit.trim() {
210    "mu" => number / 1.8,
211    "pt" => number,
212    "em" => number * 10.0, // assume 10pt font size
213    _ => 0.0,
214  }
215}

217/// Filter XMHint nodes from a list of child nodes, transferring their spacing
218/// info to adjacent tokens as `lpadding`/`rpadding` attributes (matching Perl).
219/// XMHints are also unlinked from the XML tree so they won't be seen again.
220/// Large spacings (≥10pt, e.g. \quad) become virtual PUNCT nodes (Perl MathParser.pm L483-490).
221pub fn filter_hints(nodes: Vec<Node>) -> Vec<Node> {
222  const HINT_PUNCT_THRESHOLD: f64 = 10.0;
223  let mut prefiltered: Vec<Node> = Vec::new();
224  let mut pending_space: f64 = 0.0;
225  // Save hint nodes that contributed to _space, for possible PUNCT reuse
226  let mut last_hint_for: Vec<Option<Node>> = Vec::new(); // parallel to prefiltered
227
228  for mut node in nodes {
229    if node.get_type() != Some(NodeType::ElementNode) {
230      continue;
231    }
232    if node.get_name() == "XMHint" {
233      if let Some(width_str) = node.get_attribute("width") {
234        let pts = get_xmhint_spacing(&width_str);
235        if pts != 0.0 {
236          let prev_role = prefiltered.last().and_then(|n| n.get_attribute("role"));
237          if prefiltered.last().is_some() && prev_role.as_deref() != Some("OPEN") {
238            let prev = prefiltered.last_mut().unwrap();
239            let s: f64 = prev
240              .get_attribute("_space")
241              .and_then(|v| v.parse().ok())
242              .unwrap_or(0.0);
243            let _ = prev.set_attribute("_space", &format!("{}", s + pts));
244            // Save this hint node for potential PUNCT reuse
245            let idx = prefiltered.len() - 1;
246            if idx < last_hint_for.len() {
247              last_hint_for[idx] = Some(node.clone());
248            }
249          } else {
250            pending_space += pts;
251          }
252        }
253      }
254      // Unlink from the XML tree; XMHints are ephemeral
255      node.unlink();
256    } else {
257      if pending_space > 0.0 {
258        let _ = node.set_attribute("lpadding", &format!("{:.1}pt", pending_space));
259        pending_space = 0.0;
260      }
261      prefiltered.push(node);
262      last_hint_for.push(None);
263    }
264  }
265
266  // Second pass: convert _space to rpadding (or PUNCT XMHint if above threshold)
267  let mut filtered: Vec<Node> = Vec::new();
268  for (i, mut node) in prefiltered.into_iter().enumerate() {
269    filtered.push(node.clone());
270    if let Some(s_str) = node.get_attribute("_space") {
271      let _ = node.remove_attribute("_space");
272      let s: f64 = s_str.parse().unwrap_or(0.0);
273      if s >= HINT_PUNCT_THRESHOLD && node.get_attribute("role").as_deref() != Some("PUNCT") {
274        // Perl MathParser.pm L487: create virtual PUNCT XMHint
275        // Reuse the saved hint node, setting role="PUNCT"
276        if let Some(Some(mut hint)) = last_hint_for.get(i).cloned() {
277          let _ = hint.set_attribute("role", "PUNCT");
278          // Clean width: round to integer if close, matching Perl format
279          let s_rounded = if (s - s.round()).abs() < 0.01 { s.round() } else { s };
280          let width = format!("{}pt", s_rounded);
281          let _ = hint.set_attribute("width", &width);
282          // Remove extraneous attributes from the reused hint node
283          let _ = hint.remove_attribute("depth");
284          let _ = hint.remove_attribute("height");
285          let quads = "q".repeat((s / 10.0) as usize);
286          let _ = hint.set_attribute("name", &format!("{quads}uad"));
287          filtered.push(hint);
288        }
289      } else {
290        let _ = node.set_attribute("rpadding", &format!("{:.1}pt", s));
291      }
292    }
293  }
294  filtered
295}

297/// Given a list of XML nodes (either libxml nodes, or array representations)
298/// return a list of XMRef's referring to those nodes;
299/// ensure each source node has an ID (if already instanciated as XML)
300/// or _xmkey if still in array rep. since it will get an ID later, and the connection re-made)
301/// Note that ltx:XMHint nodes are ephemeral and shouldn't be ref'd!
302/// likewise, we avoid creating XMRefs to XMRefs
303pub fn create_xmrefs(args: &mut [&mut XM], ctxt: ActionContext) -> Result<Vec<XM>, Box<dyn Error>> {
304  let nodes = ctxt.nodes;
305  let document = ctxt.document;
306  let mut refs = Vec::new();
307  for arg in args {
308    match arg {
309      XM::Token(ref mut props, _meta) => {
310        if let Some(id) = props.id.as_ref() {
311          refs.push(XM::Ref(XProps {
312            id: Some(id.clone()),
313            ..XProps::default()
314          }));
315        } else {
316          // Parser-created token without id — use _xmkey for deferred resolution
317          let key = get_xmarg_id()?.to_string();
318          props.xmkey = Some(Cow::Owned(key.clone()));
319          refs.push(XM::Ref(XProps {
320            xmkey: Some(Cow::Owned(key)),
321            ..XProps::default()
322          }));
323        }
324      },
325      XM::Lexeme(lex, _) => {
326        // If arg is already XML, it's too late to get automatic ID's
327        let node = lookup_lex_node(lex, nodes).expect("lexemes should only have valid ids.");
328
329        match node.get_attribute("xml:id") {
330          //  already has id, so refer to it.
331          Some(id) => refs.push(XM::Ref(XProps {
332            id: Some(Cow::Owned(id)),
333            ..XProps::default()
334          })),
335          None => {
336            // Generate xml:id for this node so we can reference it
337            document.generate_id(&mut node.clone(), "")?;
338            let generated_id = node.get_attribute("xml:id")
339              .or_else(|| node.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace"))
340              .or_else(|| node.get_attribute("id"));
341            if let Some(id) = generated_id {
342              refs.push(XM::Ref(XProps {
343                id: Some(Cow::Owned(id)),
344                ..XProps::default()
345              }));
346            }
347          },
348        }
349      },
350      XM::Apply(_op, _args, ref mut props, _meta) => {
351        if let Some(id) = props.id.as_ref() {
352          refs.push(XM::Ref(XProps {
353            id: Some(id.clone()),
354            ..XProps::default()
355          }));
356        } else {
357          // not yet instanciated, so hasn't had chance to get auto-id; use _xmkey
358          let key = get_xmarg_id()?.to_string();
359          props.xmkey = Some(Cow::Owned(key.clone()));
360          refs.push(XM::Ref(XProps {
361            xmkey: Some(Cow::Owned(key)),
362            ..XProps::default()
363          }));
364        }
365      },
366      // clone an XMRef (w/o any attributes or id ?) rather than create an XMRef to an XMRef
367      XM::Ref(props) => {
368        refs.push(XM::Ref(props.clone()));
369      },
370      XM::Dual(_, _, ref mut props, _) | XM::Wrap(_, ref mut props, _) => {
371        if let Some(id) = props.id.as_ref() {
372          refs.push(XM::Ref(XProps {
373            id: Some(id.clone()),
374            ..XProps::default()
375          }));
376        } else {
377          let key = get_xmarg_id()?.to_string();
378          props.xmkey = Some(Cow::Owned(key.clone()));
379          refs.push(XM::Ref(XProps {
380            xmkey: Some(Cow::Owned(key)),
381            ..XProps::default()
382          }));
383        }
384      },
385      _ => {
386        // XMHint's are ephemeral — clone without id
387        // Other variants: skip with warning
388      },
389    }
390  }
391  Ok(refs)
