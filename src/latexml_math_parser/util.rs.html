11/// Best-effort recovery of an xml:id (or _xmkey, as deferred id) for an
12/// `XM` value as it appears in a semantics-rule input. Returns `None` if
13/// the value isn't a single addressable node — composite Apply/Dual/etc.
14/// have IDs only via their props.id field (rarely set at this layer).
15pub fn xm_id_or_xmkey(xm: &XM, nodes: &[Node]) -> Option<String> {
16  match xm {
17    XM::Token(props, _) => {
18      props
19        .id
20        .as_deref()
21        .or(props.xmkey.as_deref())
22        .map(|s| s.to_string())
23    },
24    XM::Lexeme(lex, _) => {
25      lookup_lex_node(lex, nodes).ok().and_then(|n| {
26        n.get_attribute("xml:id")
27          .or_else(|| n.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace"))
28      })
29    },
30    XM::Apply(_, _, props, _) | XM::Ref(props) => {
31      props
32        .id
33        .as_deref()
34        .or(props.xmkey.as_deref())
35        .map(|s| s.to_string())
36    },
37    XM::Wrap(_, props, _) | XM::Dual(_, _, props, _) => {
38      props
39        .id
40        .as_deref()
41        .or(props.xmkey.as_deref())
42        .map(|s| s.to_string())
43    },
44    _ => None,
45  }
46}

48/// Record that the operator portion of `lost` was absorbed into `keep`'s
49/// existing application (the canonical Perl `ReplacedBy(lost, keep)`
50/// pattern from `infix_apply_nary`-style left-recursion). Walks the
51/// applicable id-bearing fields on each side; no-op if neither side
52/// has an extractable id.
53pub fn record_replacement_xm(lost: &XM, keep: &XM, nodes: &[Node]) {
54  if let (Some(lost_id), Some(keep_id)) = (xm_id_or_xmkey(lost, nodes), xm_id_or_xmkey(keep, nodes))
55  {
56    record_replacement(&lost_id, &keep_id);
57  }
58}

77fn node_to_grammar_lexemes_ctx(
78  mathnode: &Node,
79  child_nodes: Vec<Node>,
80  idx: &mut usize,
81  bigop_context: bool,
82) -> (Vec<String>, Vec<Node>) {
83  let mut lexemes = Vec::new();
84  let mut nodes = Vec::new();
85  let top_role_opt = mathnode.get_attribute("role");
86  if let Some(ref top_role) = top_role_opt {
87    *idx += 1;
88    // When in bigop context, emit BIGOPSUB/BIGOPSUP instead of POSTSUBSCRIPT/POSTSUPERSCRIPT
89    let mapped_role = if bigop_context {
90      match top_role.as_str() {
91        "POSTSUBSCRIPT" => "BIGOPSUB".to_string(),
92        "POSTSUPERSCRIPT" => "BIGOPSUP".to_string(),
93        _ => top_role.clone(),
94      }
95    } else {
96      top_role.clone()
97    };
98    lexemes.push(format!("start_{mapped_role}:start:{idx}"));
99    nodes.push(mathnode.clone());
100  }
101  // Track whether the last emitted token was a bigop (SUMOP/INTOP/etc.)
102  // so we can emit bigop-specific script tokens to reduce earley chart ambiguity.
103  let mut last_was_bigop = false;
104  for node in child_nodes.into_iter() {
105    // For XMRef nodes: resolve to get the target's role/meaning for lexing,
106    // but keep the ORIGINAL XMRef node in the output so the parse tree
107    // preserves XMRef indirection (matching Perl behavior).
108    // The get_grammatical_role/get_token_meaning functions already resolve XMRef
109    // internally, so we just need to handle the case where the target is a
110    // compound node (XMApp without role) — these need flattening.
111    // Note: we do NOT replace the node variable — we keep the XMRef.
112
113    if node.get_name() == "XMApp" && node.get_attribute("role").is_some() {
114      let role = node.get_attribute("role").unwrap();
115      // ARROW/METARELOP-role XMApps (decorated arrows like \xrightarrow{over},
116      // \xleftrightarrow[under]{over}) should be atomic terminals.
117      // Extract the meaning from the inner ARROW/METARELOP child token.
118      if role == "ARROW" || role == "METARELOP" {
119        let arrow_meaning = node
120          .get_child_elements()
121          .into_iter()
122          .find(|ch| {
123            let cr = ch.get_attribute("role");
124            cr.as_deref() == Some("ARROW") || cr.as_deref() == Some("METARELOP")
125          })
126          .and_then(|ch| {
127            ch.get_attribute("meaning")
128              .or_else(|| ch.get_attribute("name"))
129              .or_else(|| Some(ch.get_content()))
130          })
131          .unwrap_or_else(|| role.to_string());
132        *idx += 1;
133        let lexeme = format!("{role}:{arrow_meaning}:{idx}").replace(' ', "");
134        lexemes.push(lexeme);
135        nodes.push(node);
136      } else if node.has_attribute("_rewrite") {
137        // Rewrite-created: treat as atomic token with the assigned role.
138        // Don't recurse — the inner structure was pre-parsed, and the role
139        // on this node overrides whatever the children contain.
140        let gram_role = get_grammatical_role(&node);
141        let mut text = get_token_meaning(&node);
142        if text.is_empty() {
143          text = "UNKNOWN".to_string();
144        }
145        *idx += 1;
146        lexemes.push(format!("{gram_role}:{text}:{idx}").replace(' ', ""));
147        nodes.push(node);
148      } else {
149        // Only recurse into XMApp nodes that have a role (scripts, etc.)
150        // Role-less XMApps (e.g. \sqrt, already-parsed structures) are atomic.
151        // Pass bigop_context for POSTSUBSCRIPT/POSTSUPERSCRIPT following a bigop
152        let is_script = matches!(role.as_str(), "POSTSUBSCRIPT" | "POSTSUPERSCRIPT");
153        let ctx = last_was_bigop && is_script;
154        let children = filter_hints(node.get_child_nodes());
155        let (mut inner_lexes, mut inner_nodes) =
156          node_to_grammar_lexemes_ctx(&node, children, idx, ctx);
157        for (inner_lex, inner_node) in inner_lexes.drain(..).zip(inner_nodes.drain(..)) {
158          lexemes.push(inner_lex);
159          nodes.push(inner_node);
160        }
161        // Script following a bigop is still "bigop context" for the next script
162        if !is_script {
163          last_was_bigop = false;
164        }
165      }
166    } else if node.get_name() == "XMArg" {
167      // XMArg is a transparent wrapper (e.g. from \lx@post@subscript).
168      // Recurse into its children so the grammar can parse them individually.
169      // E.g. _{ij} should emit UNKNOWN:i + UNKNOWN:j, not ATOM:ij.
170      let arg_children = filter_hints(node.get_child_nodes());
171      let (mut inner_lexes, mut inner_nodes) =
172        node_to_grammar_lexemes_ctx(&node, arg_children, idx, false);
173      for (inner_lex, inner_node) in inner_lexes.drain(..).zip(inner_nodes.drain(..)) {
174        lexemes.push(inner_lex);
175        nodes.push(inner_node);
176      }
177    } else {
178      let role = get_grammatical_role(&node);
179      let mut text = get_token_meaning(&node);
180      if text.is_empty() {
181        text = "UNKNOWN".to_string();
182      }
183      *idx += 1;
184      // Track bigop tokens for bigop-specific script token emission
185      let is_bigop = matches!(
186        role.as_str(),
187        "SUMOP" | "INTOP" | "LIMITOP" | "DIFFOP" | "BIGOP"
188      );
189      last_was_bigop = is_bigop;
190      // Normalize langle/rangle meaning for consistent grammar matching.
191      // Do NOT remap to parentheses — langle_open/rangle_close tokens need
192      // distinct identity for QM bra-ket rules vs conditional probability.
193      //
194      // Perf: Split generic OPEN/CLOSE into OTHER_OPEN/OTHER_CLOSE for delimiters
195      // OTHER than parens/brackets/braces/langle. This prevents the grammar's
196      // generic `open ~ "OPEN"` prefix match from ALSO matching the specific
197      // `lparen = "OPEN:("`, etc. — which was producing 2× duplicate grammar
198      // derivations for every `(x)`, `[x]`, `{x}` expression.
199      let lexeme = if role == "OPEN" && (text == "⟨" || text == "langle") {
200        format!("OPEN:langle:{idx}")
201      } else if role == "CLOSE" && (text == "⟩" || text == "rangle") {
202        format!("CLOSE:rangle:{idx}")
203      } else if role == "OPEN" && !matches!(text.as_str(), "(" | "[" | "{") {
204        format!("OTHER_OPEN:{text}:{idx}").replace(' ', "")
205      } else if role == "CLOSE" && !matches!(text.as_str(), ")" | "]" | "}") {
206        format!("OTHER_CLOSE:{text}:{idx}").replace(' ', "")
207      } else if role == "UNKNOWN" && text == "d" {
208        // M4: Emit XDIFFUNK for possible differential-d tokens.
209        // Only "d" tokens can be diffops; other unknowns skip the diffop rule.
210        format!("XDIFFUNK:{text}:{idx}")
211      } else if role == "ID" && text == "d" {
212        format!("XDIFFID:{text}:{idx}")
213      } else if role == "VERTBAR" && text == "|"
214        && node.get_attribute("stretchy").as_deref() == Some("true")
215      {
216        // `\left|...\right|` produces a balanced pair of VERTBAR tokens
217        // with `stretchy="true"` (whereas bare `|x|` is `stretchy="false"`).
218        // Further distinguish by side: the `\@left` constructor tags
219        // the emitted XMTok with `role_side="left"`, `\@right` with
220        // `role_side="right"` (tex_math.rs). This refines the `role`
221        // attribute (delimiter direction) without changing it, so
222        // the grammar can use distinct LEFT_STRETCHY_VERTBAR /
223        // RIGHT_STRETCHY_VERTBAR tokens. The kerned-stack norm rules
224        // (\vertii, \vertiii, …) then don't have to enumerate every
225        // pairing of identical bars. Eliminates the VERTBAR-pairing
226        // combinatorial explosion in patterns like
227        // `\log^+ ∫ \left| f \right|^k dm(z) ≲ ...` (see
228        // docs/MATH_AMBIGUITY_AUDIT_2026-05-21.md §2) and unlocks task #263's
229        // norm-fenced grammar rules.
230        // Defensive fallback: if `role_side` is missing — legacy DOM
231        // input or a path that bypassed `\@left`/`\@right` — keep the
232        // old undirected STRETCHY_VERTBAR lexeme so legacy rules
233        // (eval_at, modulus fence) still work.
234        match node.get_attribute("role_side").as_deref() {
235          Some("left") => format!("LEFT_STRETCHY_VERTBAR:|:{idx}"),
236          Some("right") => format!("RIGHT_STRETCHY_VERTBAR:|:{idx}"),
237          _ => format!("STRETCHY_VERTBAR:|:{idx}"),
238        }
239      } else if role == "PUNCT" && punct_followed_by_wide_space(&node) {
240        // PUNCT followed by `\quad` / `\qquad` etc. carries an `rpadding`
241        // attribute. This wide spacing is a strong arXiv idiom for
242        // "formula separator (with side-condition)" rather than
243        // "list-element separator". Tag as WIDE_PUNCT so the grammar
244        // can prefer `formulae_apply` over `list_apply` for this
245        // separator unambiguously. See docs/MATH_AMBIGUITY_AUDIT_2026-05-21.md §2.
246        format!("WIDE_PUNCT:,:{idx}")
247      } else {
248        format!("{role}:{text}:{idx}").replace(' ', "")
249      };
250      lexemes.push(lexeme);
251      nodes.push(node);
252    }
253  }
254  if let Some(ref top_role) = top_role_opt {
255    *idx += 1;
256    let mapped_end = if bigop_context {
257      match top_role.as_str() {
258        "POSTSUBSCRIPT" => "BIGOPSUB".to_string(),
259        "POSTSUPERSCRIPT" => "BIGOPSUP".to_string(),
260        _ => top_role.clone(),
261      }
262    } else {
263      top_role.clone()
264    };
265    lexemes.push(format!("end_{mapped_end}:end:{idx}"));
266    nodes.push(mathnode.clone());
267  }
268  (lexemes, nodes)
269}

271/// Auxiliary separator for ROLE:style-lexeme into ("ROLE:style", '-', lexeme)
272pub fn distill_lexeme(name: &str) -> (&str, &str, &str) {
273  // dash separates styles, colons separate grammatical roles, and we are
274  // only trying to distill the last pure lexeme
275  // note that we are only trying to do this reasonably for letter-based names (UNKNOWN:italic-x),
276  // since some of the content symbols contain dashes themselves (e.g.
277  // OPERATOR:partial-differential)
278  if let Some(position) = name.rfind('-') {
279    let (base, trailer) = name.split_at(position);
280    let (sep, lexeme) = trailer.split_at(1);
281    (base, sep, lexeme)
282  } else if let Some(position) = name.rfind(':') {
283    let (base, trailer) = name.split_at(position);
284    let (sep, lexeme) = trailer.split_at(1);
285    (base, sep, lexeme)
286  } else {
287    ("", "", name)
288  }
289}

291/// Parse an XMHint `width` attribute string to points.
292/// `pub(crate)` so semantics.rs can re-use the same parser when
293/// inspecting `rpadding` (task #263 stretchy_norm_fenced guards).
294/// Supports "3.0mu" (mu → pt by dividing by 1.8) and "1.667pt"/"0.16667em".
295/// Handles glue specs like "2.77pt plus 2.77pt" by extracting base dimension.
296pub(crate) fn get_xmhint_spacing(width: &str) -> f64 {
297  let width = width.trim();
298  if width.is_empty() {
299    return 0.0;
300  }
301  // Strip glue stretch/shrink: "2.77pt plus 2.77pt" → "2.77pt"
302  let base = width
303    .split_once(" plus")
304    .or_else(|| width.split_once(" minus"))
305    .map_or(width, |(base, _)| base)
306    .trim();
307  let unit_start = base.find(|c: char| c.is_alphabetic()).unwrap_or(base.len());
308  let (number_str, unit) = base.split_at(unit_start);
309  let number: f64 = number_str.trim().parse().unwrap_or(0.0);
310  match unit.trim() {
311    "mu" => number / 1.8,
312    "pt" => number,
313    "em" => number * 10.0, // assume 10pt font size
314    _ => 0.0,
315  }
316}

318/// True iff the given PUNCT-like XMTok carries an `rpadding` attribute
319/// large enough to indicate a `\quad`-class spacing — TeX's idiom for
320/// separating a main formula from a side-condition (e.g.
321/// `A = B, \quad r \notin E`). Threshold of ≥5pt ignores `\,`/`\;`
322/// thin-space touch-ups while catching `\quad` (10pt) and `\qquad`.
323/// Used by `node_to_grammar_lexemes_ctx` to emit a `WIDE_PUNCT` token
324/// the grammar can route through `formulae_apply` unambiguously.
325fn punct_followed_by_wide_space(node: &Node) -> bool {
326  match node.get_attribute("rpadding") {
327    Some(s) => get_xmhint_spacing(&s) >= 5.0,
328    None => false,
329  }
330}

332/// Filter XMHint nodes from a list of child nodes, transferring their spacing
333/// info to adjacent tokens as `lpadding`/`rpadding` attributes (matching Perl).
334/// XMHints are also unlinked from the XML tree so they won't be seen again.
335/// Large spacings (≥10pt, e.g. \quad) become virtual PUNCT nodes (Perl MathParser.pm L483-490).
336pub fn filter_hints(nodes: Vec<Node>) -> Vec<Node> {
337  const HINT_PUNCT_THRESHOLD: f64 = 10.0;
338  let mut prefiltered: Vec<Node> = Vec::new();
339  let mut pending_space: f64 = 0.0;
340  // Save hint nodes that contributed to _space, for possible PUNCT reuse
341  let mut last_hint_for: Vec<Option<Node>> = Vec::new(); // parallel to prefiltered
342
343  for mut node in nodes {
344    if node.get_type() != Some(NodeType::ElementNode) {
345      continue;
346    }
347    if node.get_name() == "XMHint" {
348      if let Some(width_str) = node.get_attribute("width") {
349        let pts = get_xmhint_spacing(&width_str);
350        if pts != 0.0 {
351          let prev_role = prefiltered.last().and_then(|n| n.get_attribute("role"));
352          if prefiltered.last().is_some() && prev_role.as_deref() != Some("OPEN") {
353            let prev = prefiltered.last_mut().unwrap();
354            let s: f64 = prev
355              .get_attribute("_space")
356              .and_then(|v| v.parse().ok())
357              .unwrap_or(0.0);
358            let _ = prev.set_attribute("_space", &format!("{}", s + pts));
359            // Save this hint node for potential PUNCT reuse
360            let idx = prefiltered.len() - 1;
361            if idx < last_hint_for.len() {
362              last_hint_for[idx] = Some(node.clone());
363            }
364          } else {
365            pending_space += pts;
366          }
367        }
368      }
369      // Unlink from the XML tree; XMHints are ephemeral
370      node.unlink();
371    } else {
372      if pending_space > 0.0 {
373        let _ = node.set_attribute("lpadding", &format!("{:.1}pt", pending_space));
374        pending_space = 0.0;
375      }
376      prefiltered.push(node);
377      last_hint_for.push(None);
378    }
379  }
380
381  // Second pass: convert _space to rpadding (or PUNCT XMHint if above threshold)
382  let mut filtered: Vec<Node> = Vec::new();
383  for (i, mut node) in prefiltered.into_iter().enumerate() {
384    filtered.push(node.clone());
385    if let Some(s_str) = node.get_attribute("_space") {
386      let _ = node.remove_attribute("_space");
387      let s: f64 = s_str.parse().unwrap_or(0.0);
388      if s >= HINT_PUNCT_THRESHOLD && node.get_attribute("role").as_deref() != Some("PUNCT") {
389        // Perl MathParser.pm L487: create virtual PUNCT XMHint
390        // Reuse the saved hint node, setting role="PUNCT"
391        if let Some(Some(mut hint)) = last_hint_for.get(i).cloned() {
392          let _ = hint.set_attribute("role", "PUNCT");
393          // Clean width: round to integer if close, matching Perl format
394          let s_rounded = if (s - s.round()).abs() < 0.01 {
395            s.round()
396          } else {
397            s
398          };
399          let width = format!("{}pt", s_rounded);
400          let _ = hint.set_attribute("width", &width);
401          // Remove extraneous attributes from the reused hint node
402          let _ = hint.remove_attribute("depth");
403          let _ = hint.remove_attribute("height");
404          let quads = "q".repeat((s / 10.0) as usize);
405          let _ = hint.set_attribute("name", &format!("{quads}uad"));
406          filtered.push(hint);
407        }
408      } else {
409        let _ = node.set_attribute("rpadding", &format!("{:.1}pt", s));
410      }
411    }
412  }
413  filtered
414}

416/// Given a list of XML nodes (either libxml nodes, or array representations)
417/// return a list of XMRef's referring to those nodes;
418/// ensure each source node has an ID (if already instanciated as XML)
419/// or _xmkey if still in array rep. since it will get an ID later, and the connection re-made)
420/// Note that ltx:XMHint nodes are ephemeral and shouldn't be ref'd!
421/// likewise, we avoid creating XMRefs to XMRefs
422pub fn create_xmrefs(args: &mut [&mut XM], ctxt: ActionContext) -> Result<Vec<XM>, Box<dyn Error>> {
423  let nodes = ctxt.nodes;
424  let document = ctxt.document;
425  let mut refs = Vec::with_capacity(args.len());
426  for arg in args {
427    match arg {
428      XM::Token(ref mut props, _meta) => {
429        if let Some(id) = props.id.as_ref() {
430          refs.push(XM::Ref(XProps {
431            id: Some(id.clone()),
432            ..XProps::default()
433          }));
434        } else {
435          // Parser-created token without id — use _xmkey for deferred resolution
436          let key = get_xmarg_id()?.to_string();
437          props.xmkey = Some(Cow::Owned(key.clone()));
438          refs.push(XM::Ref(XProps {
439            xmkey: Some(Cow::Owned(key)),
440            ..XProps::default()
441          }));
442        }
443      },
444      XM::Lexeme(lex, _) => {
445        // If arg is already XML, it's too late to get automatic ID's.
446        // lookup_lex_node now returns Err for malformed lex strings instead
447        // of panicking; skip the arg on failure rather than abort the whole
448        // ref-building pass.
449        let node = match lookup_lex_node(lex, nodes) {
450          Ok(n) => n,
451          Err(e) => {
452            // Perl MathParser.pm:151 — Error('expected', 'id', undef,
453            //   "Cannot find a node with xml:id='$idref'", ...)
454            // We don't always have an idref string at this layer; the
455            // lookup error itself carries enough context.
456            log_math_error!(
457              "expected", "id",
458              "create_xmrefs: skipping lexeme with invalid node lookup: {}", e
459            );
460            continue;
461          },
462        };
463
464        match node.get_attribute("xml:id") {
465          //  already has id, so refer to it.
466          Some(id) => refs.push(XM::Ref(XProps {
467            id: Some(Cow::Owned(id)),
468            ..XProps::default()
469          })),
470          None => {
471            // Generate xml:id for this node so we can reference it
472            document.generate_id(&mut node.clone(), "")?;
473            let generated_id = node
474              .get_attribute("xml:id")
475              .or_else(|| node.get_attribute_ns("id", "http://www.w3.org/XML/1998/namespace"))
476              .or_else(|| node.get_attribute("id"));
477            if let Some(id) = generated_id {
478              refs.push(XM::Ref(XProps {
479                id: Some(Cow::Owned(id)),
480                ..XProps::default()
481              }));
482            }
483          },
484        }
485      },
486      XM::Apply(_op, _args, ref mut props, _meta) => {
487        if let Some(id) = props.id.as_ref() {
488          refs.push(XM::Ref(XProps {
489            id: Some(id.clone()),
490            ..XProps::default()
491          }));
492        } else {
493          // not yet instanciated, so hasn't had chance to get auto-id; use _xmkey
494          let key = get_xmarg_id()?.to_string();
495          props.xmkey = Some(Cow::Owned(key.clone()));
496          refs.push(XM::Ref(XProps {
497            xmkey: Some(Cow::Owned(key)),
498            ..XProps::default()
499          }));
500        }
501      },
502      // clone an XMRef (w/o any attributes or id ?) rather than create an XMRef to an XMRef
503      XM::Ref(props) => {
504        refs.push(XM::Ref(props.clone()));
505      },
506      XM::Dual(_, _, ref mut props, _) | XM::Wrap(_, ref mut props, _) => {
507        if let Some(id) = props.id.as_ref() {
508          refs.push(XM::Ref(XProps {
509            id: Some(id.clone()),
510            ..XProps::default()
511          }));
512        } else {
513          let key = get_xmarg_id()?.to_string();
514          props.xmkey = Some(Cow::Owned(key.clone()));
515          refs.push(XM::Ref(XProps {
516            xmkey: Some(Cow::Owned(key)),
517            ..XProps::default()
518          }));
519        }
520      },
521      _ => {
522        // XMHint's are ephemeral — clone without id
523        // Other variants: skip with warning
524      },
525    }
526  }
527  Ok(refs)
528}
