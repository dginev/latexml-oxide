290  // Recursively parse a node with some internal structure
291  // by first parsing any structured children, then it's content.
292  fn parse_rec(
293    &mut self,
294    mut node: Node,
295    rule_opt: &str,
296    document: &mut Document,
297  ) -> Result<Option<Node>> {
298    self.parse_children(&node, document)?;
299    // This will only handle 1 layer nesting (successfully?)
300    // Note that this would have been found by the top level xpath,
301    // but we've got to worry about node identity: the parent is being rebuilt
302    for nested in document.findnodes("descendant::ltx:XMath", Some(&node)) {
303      self.parse(nested, document)?;
304    }
305    let tag = get_node_qname(&node);
306    let rule = if let Some(requested_rule) = node.get_attribute("rule") {
307      requested_rule
308    } else {
309      rule_opt.to_owned()
310    };
311
312    if rule == "kludge" {
313      self.parse_kludge(&mut node, document);
314      Ok(None)
315    } else if let Some(mut result) = self.parse_single(&mut node, document, &rule)? {
316      *self.passed.entry_sym(tag).or_insert(0) += 1;
317      if tag == arena::pin_static("ltx:XMath") {
318        // Replace the content of XMath with parsed result
319        self.n_parsed += 1;
320        note_progress(&s!("[{}]", self.n_parsed));
321        for el_node in element_nodes(&node) {
322          document.unrecord_node_ids(&el_node);
323        }
324        // unbindNode followed by (append|replace)Tree (which removes ID's) should
325        // be safe
326        for mut child in node.get_child_nodes() {
327          child.unbind_node();
328        }
329        document.append_tree(&mut node, vec![result])?;
330        let mut new_element_children = element_nodes(&node);
331        result = new_element_children.remove(0);
332      } else {
333        // Replace the whole node for XMArg, XMWrap; preserve some attributes
334        //ProgressStep() if ($$self{progress}++ % $MATHPARSE_PROGRESS_QUANTUM) == 0;
335        // Copy all attributes
336        let resultid = p_get_attribute(&result, "id");
337        let mut attr = node.get_attributes();
338
339        // add to result, even allowing modification of xml node, since we're committed.
340        // [Annotate converts node to array which messes up clearing the id!]
341        let rtag = get_node_qname(&result);
342        // TODO: Is this needed in a world where `result` is always a `Node` ?
343        // // // Make sure font is "Appropriate", if we're creating a new token (yuck)
344        // // if ($isarr && $attr{_font} && ($rtag eq 'ltx:XMTok')) {
345        // // my $content = join('', @$result[2 .. $#$result]);
346        // // if ((!defined $content) || ($content eq '')) {
347        // //   delete $attr{_font}; }    # No font needed
348        // // elsif (my $font = $document->decodeFont($attr{_font})) {
349        // //   delete $attr{_font};
350        // //   $attr{font} = $font->specialize($content); } }
351        // // } else {
352        attr.remove("_font");
353        // TODO: See the namespaced attribute issue for libxml's wrapper:
354        //  https://github.com/KWARC/rust-libxml/issues/104
355        // until then, HACK ids.
356        let newid = attr.remove("id");
357        if let Some(ref nid) = newid {
358          attr.insert(String::from("xml:id"), nid.to_owned());
359        }
360        for (key, value) in attr {
361          if !(key.starts_with('_') || sym_can_have_attribute(rtag, arena::pin(&key))) {
362            continue;
363          }
364          if key == "xml:id" {
365            // Since we're moving the id...bookkeeping
366            document.unrecord_id(&value);
367            node.remove_attribute("xml:id")?;
368          }
369          // TODO: is the array/XM case still relevant?
370          // if ($isarr) { $$result[1]{$key} = $value; } else {
371          document.set_attribute(&mut result, &key, &value)?;
372          // }
373        }
374        result = document
375          .replace_tree(result, node)?
376          .expect("replacing the tree should always work.");
377        // Danger: the above code replaced the id on the parsed result with the one from XMArg,..
378        // If there are any references to `resultid`, we need to point them to `newid`!
379        if let Some(rid) = resultid {
380          if let Some(nid) = newid {
381            if rid != nid {
382              for mut tref in document.findnodes(&s!("//*[@idref='{}']", rid), None) {
383                tref.set_attribute("idref", &nid)?;
384              }
385            }
386          }
387        }
388      }
389      Ok(Some(result))
390    } else {
391      // TODO:
392      // self.parse_kludge(node, document);
393      // ProgressStep() if ($$self{progress}++ % $MATHPARSE_PROGRESS_QUANTUM) == 0;
394      // $$self{failed}{$tag}++;
395      Ok(None)
396    }
397  }

400  fn parse_children(&mut self, node: &Node, document: &mut Document) -> Result<()> {
401    for child in element_nodes(node) {
402      let tag = get_node_qname(&child);
403      if tag == arena::pin_static("ltx:XMArg") || tag == arena::pin_static("ltx:XMWrap") {
404        self.parse_rec(child, "Anything", document)?;
405      } else if tag == arena::pin_static("ltx:XMApp")
406        || tag == arena::pin_static("ltx:XMArray")
407        || tag == arena::pin_static("ltx:XMRow")
408        || tag == arena::pin_static("ltx:XMCell")
409        || tag == arena::pin_static("ltx:XMDual")
410      {
411        self.parse_children(&child, document)?;
412      }
413    }
414    Ok(())
415  }

485  pub fn parse_marpa(
486    &mut self,
487    input: &str,
488    nodes: &[Node],
489    document: &mut Document,
490  ) -> Result<XM> {
491    let parse_result = self
492      .engine
493      .run_recognizer(ByteScanner::new(Cursor::new(input)))?;
494    let mut parses = Vec::new();
495    let mut ok_trees = 0;
496    let mut pruned_trees = 0;
497    for val in parse_result {
498      match self.actions.get_tree(
499        self.builder.clone(),
500        val,
501        self.expert_pragmatics.as_slice(),
502        ActionContext { nodes, document },
503      ) {
504        Ok(tree_opt) => {
505          if let Some(tree) = tree_opt {
506            // eprintln!("-- we found a tree: {:?}", tree);
507            ok_trees += 1;
508            // ignore semantically pruned parses
509            parses.push(tree);
510          }
511        },
512        Err(_prune_err) => {
513          pruned_trees += 1;
514        }, // bookkeep the prune reasons?
515      }
516    }
517    if ok_trees + pruned_trees > 100 {
518      // let warning1 = format!(
519      //   "WARNING! too many marpa trees: {:?}, accepted as semantic trees: {:?}",
520      //   ok_trees + pruned_trees,
521      //   ok_trees
522      // );
523      // let warning2 = format!("         on input: {:?}", input);
524      // eprintln!("\n{}", Yellow.bold().paint(warning1));
525      // eprintln!("{}\n", Yellow.paint(warning2));
526    }
527
528    match parses.len() {
529      0 => Err("Failed to find any parse".into()),
530      1 => Ok(parses.into_iter().next().unwrap()),
531      2 | 3 => Ok(XM::Choices(parses)),
532      _more => {
533        // Loop over the various soft pruning algorithms available, until we are at 3 trees or less
534        let mut reduced_forest = XM::Choices(parses);
535        for pragma in self.student_pragmatics.iter() {
536          reduced_forest = reduced_forest.soft_prune_choices(*pragma);
537          match reduced_forest {
538            XM::Choices(ref trees) => match trees.len() {
539              2 | 3 => break, //reduced sufficiently, return
540              _more => {},    // keep trying to reduce
541            },
542            _ => break, // reduced sufficiently, return
543          };
544        }
545        Ok(reduced_forest)
546      },
547    }
548  }
549
550  pub fn parse_lexemes(
551    &mut self,
552    lexemes: Vec<String>,
553    nodes: &[Node],
554    document: &mut Document,
555  ) -> Result<Option<XM>> {
556    let mut input_string: String = lexemes.join(" ");
557    // Add a trailing space, in an attempt to work with
558    // a rules!() macro that has a Hard expectation of a space char following EVERY token.
559    // this - counterintuitively- allows a simple macro definition AND a simple parse tree.
560    input_string.push(' ');
561    if let Ok(parse_tree) = self.parse_marpa(&input_string, nodes, document) {
562      Ok(Some(parse_tree))
563    } else {
564      Ok(None)
565    }
566  }
567}

579// Get the Token's  meaning, else name, else content, else role
580fn get_token_meaning(node_opt: &Node, document: &Document) -> Option<String> {
581  let node = realize_xmnode(node_opt, document);
582  if let Some(x) = p_get_attribute(&node, "meaning") {
583    Some(x)
584  } else if let Some(x) = p_get_attribute(&node, "name") {
585    Some(x)
586  } else {
587    let text_content = node.get_content();
588    if !text_content.is_empty() {
589      Some(text_content)
590    } else {
591      p_get_attribute(&node, "role")
592    }
593  }
594}

596fn textrec(
597  node_opt: &Node,
598  outer_bp_opt: Option<usize>,
599  outer_name_opt: Option<&str>,
600  document: &Document,
601) -> String {
602  let node = realize_xmnode(node_opt, document);
603  let tag = get_node_qname(&node);
604  let outer_bp = outer_bp_opt.unwrap_or(0);
605  let outer_name = outer_name_opt.unwrap_or("");
606  // If node has meaning, that's the text form.
607  let meaning_opt = match p_get_attribute(&node, "meaning") {
608    Some(m) => Some(m),
609    None => p_get_attribute(&node, "name"),
610  };
611  if let Some(meaning) = meaning_opt {
612    return match PREFIX_ALIAS.get(meaning.as_str()) {
613      Some(m) => m.to_string(),
614      None => meaning,
615    };
616  }
617  if tag == arena::pin_static("ltx:XMApp") {
618    let mut args = element_nodes(&node);
619    if args.is_empty() {
620      // Error!("expected","arguments" ...);
621      todo!();
622    }
623    let arg_node = args.remove(0);
624    let op = realize_xmnode(&arg_node, document);
625    if let Some(app_role) = node.get_attribute("role") {
626      if app_role == "FLOATSUBSCRIPT" {
627        return String::from("_") + &textrec(&op, None, None, document);
628      } else if app_role == "FLOATSUPERSCRIPT" {
629        return String::from("^") + &textrec(&op, None, None, document);
630      }
631    }
632
633    let name = if with_node_qname(&op, |name| name == "ltx:XMTok") {
634      get_token_meaning(&op, document).unwrap_or_else(|| "unknown".to_owned())
635    } else {
636      String::new()
637    };
638    let (bp, string) = textrec_apply(&name, &op, args, document);
639    if (bp < outer_bp) || ((bp == outer_bp) && (name != outer_name)) {
640      format!("({string})")
641    } else {
642      string
643    }
644  } else if tag == arena::pin_static("ltx:XMDual") {
645    let children = element_nodes(&node);
646    let content = children
647      .first()
648      .expect("XMDual should always have 2 child elements.");
649    textrec(content, Some(outer_bp), Some(outer_name), document) // Just send out the
650  // semantic form
651  // Fall back to
652  // presentation, if
653  // content has poor
654  // semantics (eg. from
655  // replacement patterns)
656  // TODO
657  // return ($text =~
658  // /^\(*Unknown/ ?
659  // textrec($presentation,
660  // $outer_bp, $outer_name)
661  // : $text); }
662  } else if tag == arena::pin_static("ltx:XMTok") {
663    let name = match get_token_meaning(&node, document) {
664      Some(meaning) => meaning,
665      None => s!("Unknown"),
666    };
667    match PREFIX_ALIAS.get(name.as_str()) {
668      Some(v) => v.to_string(),
669      None => name,
670    }
671  } else if tag == arena::pin_static("ltx:XMWrap") || tag == arena::pin_static("ltx:XMCell") {
672    // ??
673    element_nodes(&node)
674      .into_iter()
675      .map(|child| textrec(&child, None, None, document))
676      .collect::<Vec<_>>()
677      .join("@")
678  } else if tag == arena::pin_static("ltx:XMArg") {
679    let args = element_nodes(&node);
680    if args.is_empty() {
681      // Error!("expected","arguments" ...);
682      todo!();
683    }
684    args
685      .iter()
686      .map(|arg| textrec(arg, None, None, document))
687      .collect::<Vec<_>>()
688      .join("")
689  } else if tag == arena::pin_static("ltx:XMArray") {
690    textrec_array(&node)
691  } else {
692    s!("[{}]", p_get_value(&node))
693  }
694}

696fn textrec_apply(name: &str, op: &Node, args: Vec<Node>, document: &Document) -> (usize, String) {
697  let role = op
698    .get_attribute("role")
699    .unwrap_or_else(|| "Unknown".to_string());
700  if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP")
701    && PRE_DIGITS_RE.is_match(&op.get_attribute("scriptpos").unwrap_or_default())
702  {
703    // Note that this will likely get parenthesized due to high bp
704    let mut inner = textrec(op, None, None, document);
705    if let Some(arg2) = args.get(1) {
706      inner.push(' ');
707      inner.push_str(&textrec(arg2, None, None, document));
708    }
709    if let Some(arg1) = args.first() {
710      inner.push(' ');
711      inner.push_str(&textrec(arg1, None, None, document));
712    }
713    (5000, inner)
714  } else if let Some(bp) = IS_INFIX.get(&role) {
715    // A sub/superscript with a meaning probably should be prefix
716    if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP") && op.has_attribute("meaning") {
717      (
718        500,
719        format!(
720          "{}@({})",
721          textrec(op, Some(10000), Some(name), document),
722          args
723            .iter()
724            .map(|a| textrec(a, None, None, document))
725            .collect::<Vec<_>>()
726            .join(", ")
727        ),
728      )
729    } else {
730      // Format as infix.
731      let textrec_op = textrec(op, None, None, document);
732      let rec_form = if args.len() == 1 {
733        // unless a single arg; then prefix.
734        textrec_op + " " + &textrec(&args[0], Some(*bp), Some(name), document)
735      } else {
736        args
737          .iter()
738          .map(|a| textrec(a, Some(*bp), Some(name), document))
739          .collect::<Vec<_>>()
740          .join(&(" ".to_string() + &textrec_op + " "))
741      };
742      (*bp, rec_form)
743    }
744  } else if role == "POSTFIX" {
745    (
746      10000,
747      textrec(&args[0], Some(10000), Some(name), document) + &textrec(op, None, None, document),
748    )
749  } else if name == "multirelation" {
750    let joined = args
751      .iter()
752      .map(|a| textrec(a, Some(2), Some(name), document))
753      .collect::<Vec<_>>()
754      .join(" ");
755    (2, joined)
756  } else {
757    (
758      500,
759      textrec(op, Some(10000), Some(name), document)
760        + "@("
761        + &args
762          .iter()
763          .map(|a| textrec(a, None, None, document))
764          .collect::<Vec<_>>()
765          .join(", ")
766        + ")",
767    )
768  }
769}

781//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
782// Cute! Were it NOT for Sub/Superscripts, the whole parsing process only
783// builds a new superstructure around the sequence of token nodes in the input.
784// Thus, any internal structure is unchanged.
785//  They get re-parented, but if the parse fails, we've only got to put them
786// BACK into the original node, to recover the original arrangment!!!
787// Thus, we don't have to clone, and deal with namespace duplication.
788// ...
789// EXCEPT, as I said, for sub/superscripts!!!!
790//
791
792//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
793// Constructors used in grammar
794// All the tree construction in the grammar should come through these
795// operations. We avoid mucking with the actual XML nodes (both to avoid
796// modifying the original tree until we have a successful parse, and to avoid
797// XML::LibXML cloning nightmares) We are converting XML nodes to array
798// representation: [$tag, {%attr},@children] This means any inspection of
799// nodes has to recognize that  * node may be in XML vs ARRAY representation
800// * node may be an XMRef to another node whose properties are the ones we
801// should use.
802//
803// Also, when we are examining a node's properties (roles, fences, script
804// positioning, etc) we should be careful to check for XMRef indirection and
805// examine the properties of the node that was referred to.
806// HOWEVER, we should construct our parse tree using (a clone of) the XMRef
807// node, rather than (a clone of) the referred to node, so as to preserve
808// identity.
809//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
810// We're currently keeping the id's on the nodes as they get cloned,
811// since they'll (maybe) replace the unparsed nodes.
812// However, if we consider multiple parses or preserving both parsed & unparsed,
813// we may have to do some adaptation and id shifting.
814// ================================================================================
815
816// ================================================================================
817// Low-level accessors
818
819// The following accessors work on both the LibXML and ARRAY representations
820// but they do NOT automatically dereference XMRef!
821pub fn p_get_value(node: &Node) -> String {
822  let node_type = node.get_type();
823  if node_type == Some(NodeType::ElementNode) {
824    let x = node.get_content();
825    if !x.is_empty() {
826      // get content, or fall back to name
827      x
828    } else {
829      node.get_attribute("name").unwrap_or_default()
830    }
831  } else {
832    node.get_content()
833  }
834}

838pub fn realize_xmnode<'a>(node: &'a Node, document: &'a Document) -> Cow<'a, Node> {
839  if with_node_qname(node, |name| name == "ltx:XMRef") {
840    if let Some(idref) = node.get_attribute("idref") {
841      // Can it happen that $realnode is, itself, an XMRef?
842      // Then we should recurse recurse!
843      if let Some(realnode) = document.lookup_id(&idref) {
844        return Cow::Borrowed(realnode);
845      } else {
846        let message = s!("Cannot find a node with xml:id='{}'", idref);
847        // TODO:
848        // LaTeXML::MathParser::IDREFS{$idref}
849        // ? "Previously bound to " .
850        // ToString($LaTeXML::MathParser::IDREFS{$idref})           : ()));
851        let err = || {
852          Error!("expected", "id", message);
853          Ok(())
854        };
855        err().ok();
856        //       return ['ltx:ERROR', {}, "Missing XMRef idref=$idref"]; } }
857      }
858    }
859  }
860  Cow::Borrowed(node)
861}
