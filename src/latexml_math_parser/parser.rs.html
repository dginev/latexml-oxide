154  /// Reset the marpa engine after a failed parse.
155  /// Creates a new engine and runs a trivial parse to advance past the
156  /// precompute step (grammar is already precomputed, can't do it again).
157  fn reset_engine(&mut self) {
158    let mut engine = Parser::with_grammar(self.grammar.clone());
159    // Run a trivial recognizer to advance state from G (precompute) through
160    // R → B → O → T → GReady. Use "NUMBER:1:1 " which is a valid single-token formula.
161    match engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 "))) {
162      Ok(_) => { self.engine = engine; },
163      Err(_e) => {
164        // Cloned grammar is in bad state — rebuild from scratch.
165        // Cache the fresh grammar for future resets to avoid repeated init_grammar() calls.
166        let (grammar, _actions, _builder) = init_grammar().unwrap();
167        let thin_grammar = grammar.unwrap();
168        self.grammar = thin_grammar.clone();
169        let mut fresh_engine = Parser::with_grammar(thin_grammar);
170        // Advance past precompute with trivial parse
171        let _ = fresh_engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 ")));
172        self.engine = fresh_engine;
173      }
174    }
175  }

177  /// Test if the recognizer accepts a given input string (for unit testing).
178  pub fn recognizes(&mut self, input: &str) -> bool {
179    let result = self.engine.run_recognizer(ByteScanner::new(Cursor::new(input)));
180    if result.is_err() {
181      self.reset_engine();
182    }
183    result.is_ok()
184  }

186  pub fn parse_math(&mut self, document: &mut Document) -> Result<()> {
187    self.clear();
188    self.cleanup_scripts(document);
189    let xmath_selector = "descendant-or-self::ltx:XMath[not(ancestor::ltx:XMath)]";
190    let xmath_nodes = document.findnodes(xmath_selector, None);
191
192    if !xmath_nodes.is_empty() {
193      note_begin("Math Parsing");
194      note_progress(&s!("{:?} formulae ...", xmath_nodes.len()));
195      // Populate the thread-local idstore for XMRef resolution during parsing.
196      // Perl uses $doc->lookupID which accesses the document's idstore directly.
197      crate::data::set_math_idstore(document.get_idstore_clone());
198      for math in xmath_nodes {
199        self.parse(math, document)?;
200      }
201      crate::data::clear_math_idstore();
202
203      // Run parse_kludge on unparsed XMath nodes with direct OPEN/CLOSE children.
204      // Collect first, then process (avoid modifying tree during XPath iteration).
205      let kludge_candidates: Vec<Node> = document.findnodes("//ltx:XMath", None);
206      // eprintln!("KLUDGE_SCAN: {} XMath candidates", kludge_candidates.len());
207      for xmath in kludge_candidates {
208        // Use get_child_nodes + manual filter (get_child_elements may miss some nodes)
209        let child_elems: Vec<Node> = xmath.get_child_nodes().into_iter()
210          .filter(|n| n.get_type() == Some(NodeType::ElementNode))
211          .collect();
212        // Skip fully-parsed XMath: single XMDual/XMApp child means parsing succeeded.
213        if child_elems.len() == 1 {
214          let name = child_elems[0].get_name();
215          if name == "XMDual" || name == "XMApp" { continue; }
216        }
217        let has_direct_open = child_elems.iter().any(|ch| {
218          let role = ch.get_attribute("role").unwrap_or_default();
219          (role == "OPEN" || role == "CLOSE") && ch.get_name() == "XMTok"
220        });
221        // Also check for XMArray with adjacent XMTok (unparsed aligned-in-fenced)
222        let has_array_with_tok = !has_direct_open
223          && child_elems.iter().any(|ch| ch.get_name() == "XMArray")
224          && child_elems.iter().any(|ch| ch.get_name() == "XMTok");
225        if has_direct_open || has_array_with_tok {
226          let mut xm = xmath;
227          self.parse_kludge(&mut xm, document);
228        }
229      }

414  // Recursively parse a node with some internal structure
415  // by first parsing any structured children, then it's content.
416  fn parse_rec(
417    &mut self,
418    mut node: Node,
419    rule_opt: &str,
420    document: &mut Document,
421  ) -> Result<Option<Node>> {
422    self.parse_children(&node, document)?;
423    // This will only handle 1 layer nesting (successfully?)
424    // Note that this would have been found by the top level xpath,
425    // but we've got to worry about node identity: the parent is being rebuilt
426    for nested in document.findnodes("descendant::ltx:XMath", Some(&node)) {
427      self.parse(nested, document)?;
428    }
429    let tag = get_node_qname(&node);
430    let rule = if let Some(requested_rule) = node.get_attribute("rule") {
431      requested_rule
432    } else {
433      rule_opt.to_owned()
434    };
435
436    if rule == "kludge" {
437      self.parse_kludge(&mut node, document);
438      Ok(None)
439    } else if let Some(mut result) = self.parse_single(&mut node, document, &rule)? {
440      *self.passed.entry_sym(tag).or_insert(0) += 1;
441      if tag == arena::pin_static("ltx:XMath") {
442        // Replace the content of XMath with parsed result
443        self.n_parsed += 1;
444        note_progress(&s!("[{}]", self.n_parsed));
445        for el_node in element_nodes(&node) {
446          document.unrecord_node_ids(&el_node);
447        }
448        // unbindNode followed by (append|replace)Tree (which removes ID's) should
449        // be safe
450        for mut child in node.get_child_nodes() {
451          child.unbind_node();
452        }
453        document.append_tree(&mut node, vec![result])?;
454        let mut new_element_children = element_nodes(&node);
455        result = new_element_children.remove(0);
456      } else {
457        // Replace the whole node for XMArg, XMWrap; preserve some attributes
458        //ProgressStep() if ($$self{progress}++ % $MATHPARSE_PROGRESS_QUANTUM) == 0;
459        // Copy all attributes
460        let resultid = p_get_attribute(&result, "id");
461        let mut attr = node.get_attributes();
462
463        // add to result, even allowing modification of xml node, since we're committed.
464        // [Annotate converts node to array which messes up clearing the id!]
465        let rtag = get_node_qname(&result);
466        // TODO: Is this needed in a world where `result` is always a `Node` ?
467        // // // Make sure font is "Appropriate", if we're creating a new token (yuck)
468        // // if ($isarr && $attr{_font} && ($rtag eq 'ltx:XMTok')) {
469        // // my $content = join('', @$result[2 .. $#$result]);
470        // // if ((!defined $content) || ($content eq '')) {
471        // //   delete $attr{_font}; }    # No font needed
472        // // elsif (my $font = $document->decodeFont($attr{_font})) {
473        // //   delete $attr{_font};
474        // //   $attr{font} = $font->specialize($content); } }
475        // // } else {
476        attr.remove("_font");
477        // TODO: See the namespaced attribute issue for libxml's wrapper:
478        //  https://github.com/KWARC/rust-libxml/issues/104
479        // until then, HACK ids.
480        let newid = attr.remove("id");
481        if let Some(ref nid) = newid {
482          attr.insert(String::from("xml:id"), nid.to_owned());
483        }
484        // Perl guard: don't overwrite _box if result already has one
485        if attr.contains_key("_box") && result.has_attribute("_box") {
486          attr.remove("_box");
487        }
488        for (key, value) in attr {
489          if !(key.starts_with('_') || sym_can_have_attribute(rtag, arena::pin(&key))) {
490            continue;
491          }
492          if key == "xml:id" {
493            // Since we're moving the id...bookkeeping
494            document.unrecord_id(&value);
495            node.remove_attribute("xml:id")?;
496          }
497          // TODO: is the array/XM case still relevant?
498          // if ($isarr) { $$result[1]{$key} = $value; } else {
499          document.set_attribute(&mut result, &key, &value)?;
500          // }
501        }
502        if let Some(r) = document.replace_tree(result.clone(), node)? {
503          result = r;
504        }
505        // If replace_tree returns None, node was already detached; keep result as-is.
506        // Danger: the above code replaced the id on the parsed result with the one from XMArg,..
507        // If there are any references to `resultid`, we need to point them to `newid`!
508        if let Some(rid) = resultid {
509          if let Some(nid) = newid {
510            if rid != nid {
511              for mut tref in document.findnodes(&s!("//*[@idref='{}']", rid), None) {
512                tref.set_attribute("idref", &nid)?;
513              }
514            }
515          }
516        }
517      }
518      Ok(Some(result))
519    } else {
520      // Parse failed — run kludge to wrap OPEN/CLOSE delimiters
521      *self.failed.entry_sym(tag).or_insert(0) += 1;
522      if tag == arena::pin_static("ltx:XMath") {
523        self.failed_xmath_ids.push(node.to_hashable());
524        // Kludge (OPEN/CLOSE wrapping) runs post-parse in core_interface.rs
525        // using failed_xmath_ids to find the failed nodes.
526      }
527      Ok(None)
528    }

531  // Depth first parsing of XMArg nodes.
532  fn parse_children(&mut self, node: &Node, document: &mut Document) -> Result<()> {
533    for child in element_nodes(node) {
534      let tag = get_node_qname(&child);
535      if tag == arena::pin_static("ltx:XMArg") {
536        self.parse_rec(child, "Anything", document)?;
537      } else if tag == arena::pin_static("ltx:XMWrap") {
538        if child.has_attribute("_rewrite") {
539          // Rewrite-created XMWrap: parse inner structure (subscripts etc.) but
540          // the XMWrap's role overrides whatever the inner parse produces.
541          // Temporarily remove role so parse_rec doesn't emit start_ROLE/end_ROLE
542          // tokens (the grammar only handles script roles).
543          let saved_role = child.get_attribute("role");
544          let mut c = child.clone();
545          if saved_role.is_some() {
546            c.remove_attribute("role").ok();
547          }
548          if let Some(mut result) = self.parse_rec(child, "Anything", document)? {
549            if let Some(ref role) = saved_role {
550              result.set_attribute("role", role).ok();
551            }
552          } else if let Some(ref role) = saved_role {
553            // Parse failed — XMWrap still in DOM, restore role
554            c.set_attribute("role", role).ok();
555          }
556        } else {
557          self.parse_rec(child, "Anything", document)?;
558        }
559      } else if tag == arena::pin_static("ltx:XMApp")
560        || tag == arena::pin_static("ltx:XMArray")
561        || tag == arena::pin_static("ltx:XMRow")
562        || tag == arena::pin_static("ltx:XMCell")
563        || tag == arena::pin_static("ltx:XMDual")
564      {
565        self.parse_children(&child, document)?;
566      }
567    }
568    Ok(())
569  }

571  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
572  // Low-Level hack parsing when "real" parsing fails;
573  // Two issues cause generated Presentation MathML to be really bad:
574  // (1) not having mrow/mfenced structures wrapping OPEN...CLOSE sequences
575  //     throws off MathML's stretchiness treatment of the fences
576  //     (they're all the same size; big)
577  // (2) un-attached sub/superscripts won't position correctly,
578  //     unless they're attached to something plausible.
579  // NOTE: we should be able to optionally switch this off.
580  // Especially, when we want to try alternative parse strategies.
581  /// Fallback parser for unparseable expressions.
582  /// NOTE: Cannot implement DOM manipulation here — any DOM changes inside
583  /// the parser module break Marpa grammar precomputation. The kludge logic
584  /// must be moved to core_interface.rs (the caller) using failed_xmath_ids.
585  /// Perl: parse_kludge (MathParser.pm L530-566)
586  /// Stack-based matching of OPEN/CLOSE delimiter pairs, wrapping in XMWrap.
587  fn parse_kludge(&self, mathnode: &mut Node, document: &mut Document) {
588    use crate::data::get_grammatical_role;
589    let children: Vec<Node> = filter_hints(mathnode.get_child_nodes());
590    if children.is_empty() { return; }
591    let child_names: Vec<String> = children.iter().map(|c| {
592      let r = c.get_attribute("role").unwrap_or_default();
593      format!("{}({})", c.get_name(), r)
594    }).collect();
595    eprintln!("parse_kludge: processing {} children: {:?}", children.len(), child_names);
596
597    // Represent content as either individual nodes or groups needing XMWrap.
598    enum Item { Node(Node), Wrap(Vec<Item>) }
599
600    // Perl: @stack = ([], []) — extra empty level handles unmatched leading CLOSEs.
601    let mut stack: Vec<Vec<Item>> = vec![vec![], vec![]];
602    let roles: Vec<String> = children.iter().map(|c| get_grammatical_role(c)).collect();
603    let mut iter = children.into_iter().zip(roles.into_iter()).peekable();
604
605    while iter.peek().is_some() || stack.len() > 1 {
606      let pair_opt = iter.next();
607      let role = pair_opt.as_ref().map(|(_, r)| r.as_str()).unwrap_or("CLOSE");
608      match role {
609        "OPEN" => {
610          stack.insert(0, vec![Item::Node(pair_opt.unwrap().0)]);
611        },
612        "CLOSE" => {
613          let mut row = stack.remove(0);
614          if let Some((node, _)) = pair_opt {
615            row.push(Item::Node(node));
616          }
617          // Wrap the row if > 1 item
618          let result = if row.len() > 1 {
619            Item::Wrap(row)
620          } else {
621            row.into_iter().next().unwrap_or(Item::Node(
622              Node::new_text("", &document.document).unwrap()))
623          };
624          if stack.is_empty() { stack.push(vec![]); }
625          stack[0].push(result);
626        },
627        _ => {
628          if let Some((node, _)) = pair_opt {
629            stack[0].push(Item::Node(node));
630          }
631        },
632      }
633    }
634
635    // Check if any wrapping happened
636    let items = stack.into_iter().next().unwrap_or_default();
637    let has_wrap = items.iter().any(|i| matches!(i, Item::Wrap(_)));
638    eprintln!("parse_kludge: items={} has_wrap={}", items.len(), has_wrap);
639    if !has_wrap { return; }
640
641    // Rebuild: clear mathnode, then reconstruct with XMWrap DOM elements.
642    document.unrecord_node_ids(mathnode);
643    for mut child in mathnode.get_child_nodes() {
644      child.unbind_node();
645    }
646    // Perl L560-563: top-level Wraps are UNWRAPPED (appendTree extracts content).
647    // But inner Wraps become actual XMWrap elements.
648    fn add_items(items: Vec<Item>, parent: &mut Node, document: &mut Document) {
649      for item in items {
650        match item {
651          Item::Node(mut n) => {
652            n.unlink_node();
653            parent.add_child(&mut n).ok();
654          },
655          Item::Wrap(inner) => {
656            eprintln!("parse_kludge: creating XMWrap with {} inner items", inner.len());
657            match document.open_element_at(parent, "ltx:XMWrap", None, None) {
658              Ok(mut wrap) => {
659                add_items(inner, &mut wrap, document);
660                document.close_element_at(&mut wrap).ok();
661                eprintln!("parse_kludge: XMWrap created, children={}", wrap.get_child_nodes().len());
662              },
663              Err(e) => {
664                eprintln!("parse_kludge: FAILED to create XMWrap: {:?}", e);
665                // Fallback: add items directly
666                add_items(inner, parent, document);
667              },
668            }
669          },
670        }
671      }
672    }
673    // Perl L560-563: at top level, unwrap single Wrap (extract XMWrap contents).
674    // Otherwise, add items directly (inner Wraps become XMWrap elements).
675    let is_single_wrap = items.len() == 1 && matches!(&items[0], Item::Wrap(_));
676    eprintln!("parse_kludge: is_single_wrap={}", is_single_wrap);
677    if is_single_wrap {
678      if let Some(Item::Wrap(inner)) = items.into_iter().next() {
679        let inner_desc: Vec<&str> = inner.iter().map(|i| match i {
680          Item::Node(_) => "Node",
681          Item::Wrap(_) => "Wrap",
682        }).collect();
683        eprintln!("parse_kludge: unwrapping single top-level, inner items: {:?}", inner_desc);
684        add_items(inner, mathnode, document);
685      }
686    } else {
687      add_items(items, mathnode, document);
688    }
689  }

692  // Low-level Parser: parse a single expression
693  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
694  // Convert to textual form for processing by MathGrammar
695  fn parse_single(
696    &mut self,
697    mathnode: &mut Node,
698    document: &mut Document,
699    rule: &str,
700  ) -> Result<Option<Node>> {
701    let mut idx = 0;
702    let mut content_nodes = filter_hints(mathnode.get_child_nodes());
703
704    // Extract trailing PUNCT/PERIOD nodes if rule ends with ',' (Perl: $rule =~ s/,$// )
705    let mut punct_nodes: Vec<Node> = Vec::new();
706    if rule.ends_with(',') {
707      while let Some(last) = content_nodes.last() {
708        let role = last.get_attribute("role").unwrap_or_default();
709        if role == "PUNCT" || role == "PERIOD" {
710          let mut p = content_nodes.pop().unwrap();
711          p.unlink(); // detach from mathnode's children
712          punct_nodes.insert(0, p);
713        } else {
714          break;
715        }
716      }
717    }
718
719    if content_nodes.is_empty() {
720      // Perl MathParser.pm L683: $result = $nodes[0] || Absent()
721      // Empty XMArg/XMWrap: create <XMTok meaning="absent"/>
722      let mut absent_tok = document.open_element_at(mathnode, "ltx:XMTok", None, None)?;
723      document.set_attribute(&mut absent_tok, "meaning", "absent")?;
724      document.close_element_at(&mut absent_tok)?;
725      absent_tok.unlink();
726      Ok(Some(absent_tok))
727    } else if content_nodes.len() == 1 && punct_nodes.is_empty() {
728      // single node, nothing to wrap
729      Ok(Some(content_nodes.remove(0)))
730    } else if content_nodes.len() == 1 {
731      // single node with trailing punct: wrap in XMDual
732      let result = content_nodes.remove(0);
733      Ok(Some(self.wrap_with_punct(result, punct_nodes, mathnode, document)?))
734    } else {
735      // Use pre-filtered content_nodes to avoid double-filtering (filter_hints already called above)
736      let (lexemes, mut nodes) = node_to_grammar_lexemes_from(mathnode, content_nodes, &mut idx);
737      if let Ok(Some(parse_tree)) = self.parse_lexemes(lexemes, &nodes, document) {
738        //START reparent: the reparenting used to be in `parse_rec` in Perl. Is this a good place?
739        // Replace the content of XMath with parsed result
740        // unbindNode followed by (append|replace)Tree (which removes ID's) should be safe
741        for child_el in element_nodes(mathnode) {
742          document.unrecord_node_ids(&child_el);
743        }
744        for mut node in mathnode.get_child_nodes() {
745          node.unlink();
746        }
747        let new_xml_tree = parse_tree.into_xmath(mathnode, &mut nodes, document)?;
748        document.append_tree(mathnode, vec![new_xml_tree])?;
749        // Resolve _xmkey references: match XMRef[@_xmkey] to elements with same _xmkey
750        resolve_xmkeys(mathnode, document)?;
751        let result = element_nodes(mathnode).remove(0);
752        //END reparent.
753        if !punct_nodes.is_empty() {
754          Ok(Some(self.wrap_with_punct(result, punct_nodes, mathnode, document)?))
755        } else {
756          Ok(Some(result))
757        }
758      } else {
759        // Parse failed; put punct nodes back in mathnode so they remain visible
760        for mut p in punct_nodes {
761          mathnode.add_child(&mut p).ok();
762        }
763        Ok(None)
764      }
765    }
766  }

768  /// Wrap `result` in an XMDual so trailing punctuation appears in the presentation
769  /// branch only, matching Perl's `if (@punct) { $result = ['ltx:XMDual', ...] }`.
770  fn wrap_with_punct(
771    &self,
772    mut result: Node,
773    punct_nodes: Vec<Node>,
774    mathnode: &mut Node,
775    document: &mut Document,
776  ) -> Result<Node> {
777    // Assign an id to result while it's still in the tree (id generation needs ancestors)
778    document.generate_id(&mut result, "")?;
779    let id = result
780      .get_attribute_ns("id", XML_NS)
781      .unwrap_or_else(|| result.get_attribute("id").unwrap_or_default());
782    // Detach result from mathnode so we can re-parent it inside XMWrap
783    result.unlink();
784    // Create XMDual in mathnode
785    let mut dual = document.open_element_at(mathnode, "ltx:XMDual", None, None)?;
786    // Content branch: XMRef pointing to result
787    let ref_attrs = {
788      let mut m = HashMap::default();
789      m.insert(s!("idref"), id);
790      Some(m)
791    };
792    let mut xmref = document.open_element_at(&mut dual, "ltx:XMRef", ref_attrs, None)?;
793    document.close_element_at(&mut xmref)?;
794    // Presentation branch: XMWrap containing result + trailing punct
795    let mut xmwrap = document.open_element_at(&mut dual, "ltx:XMWrap", None, None)?;
796    xmwrap.add_child(&mut result).ok();
797    for mut p in punct_nodes {
798      xmwrap.add_child(&mut p).ok();
799    }
800    document.close_element_at(&mut xmwrap)?;
801    document.close_element_at(&mut dual)?;
802    Ok(dual)
803  }

805  pub fn parse_marpa(
806    &mut self,
807    input: &str,
808    nodes: &[Node],
809    document: &mut Document,
810  ) -> Result<XM> {
811    let parse_result = self
812      .engine
813      .run_recognizer(ByteScanner::new(Cursor::new(input)))?;
814    let mut parses = Vec::new();
815    let mut ok_trees = 0;
816    let mut pruned_trees = 0;
817    let start = std::time::Instant::now();
818    let max_trees = 5000; // Hard limit on parse tree enumeration
819    let max_time = std::time::Duration::from_secs(30); // 30 second timeout
820    for val in parse_result {
821      // Bail out if too many trees or too much time
822      if ok_trees + pruned_trees >= max_trees || start.elapsed() > max_time {
823        return Err(format!(
824          "Parse aborted: {} trees ({} ok, {} pruned) in {:?}",
825          ok_trees + pruned_trees, ok_trees, pruned_trees, start.elapsed()
826        ).into());
827      }
828      match self.actions.get_tree(
829        self.builder.clone(),
830        val,
831        self.expert_pragmatics.as_slice(),
832        ActionContext { nodes, document },
833      ) {
834        Ok(tree_opt) => {
835          if let Some(tree) = tree_opt {
836            ok_trees += 1;
837            parses.push(tree);
838          }
839        },
840        Err(_prune_err) => {
841          pruned_trees += 1;
842        },
843      }
844    }
845    if ok_trees + pruned_trees > 100 {
846      // let warning1 = format!(
847      //   "WARNING! too many marpa trees: {:?}, accepted as semantic trees: {:?}",
848      //   ok_trees + pruned_trees,
849      //   ok_trees
850      // );
851      // let warning2 = format!("         on input: {:?}", input);
852      // eprintln!("\n{}", Yellow.bold().paint(warning1));
853      // eprintln!("{}\n", Yellow.paint(warning2));
854    }
855
856    match parses.len() {
857      0 => Err("Failed to find any parse".into()),
858      1 => Ok(parses.into_iter().next().unwrap()),
859      _more => {
860        // Loop over the various soft pruning algorithms available, until we have 1 tree
861        let mut reduced_forest = XM::Choices(parses);
862        for pragma in self.student_pragmatics.iter() {
863          reduced_forest = reduced_forest.soft_prune_choices(*pragma);
864          match reduced_forest {
865            XM::Choices(ref trees) if trees.len() <= 1 => break,
866            _ => {},
867          };
868        }
869        Ok(reduced_forest)
870      },
871    }
872  }

874  pub fn parse_lexemes(
875    &mut self,
876    lexemes: Vec<String>,
877    nodes: &[Node],
878    document: &mut Document,
879  ) -> Result<Option<XM>> {
880    let mut input_string: String = lexemes.join(" ");
881    // Add a trailing space, in an attempt to work with
882    // a rules!() macro that has a Hard expectation of a space char following EVERY token.
883    // this - counterintuitively- allows a simple macro definition AND a simple parse tree.
884    input_string.push(' ');
885
886    // Timeout guard: if the input has many tokens, it might cause exponential
887    // Marpa ambiguity (e.g. BIGOP tokens like \neg). Use a thread-based timeout.
888    let _token_count = input_string.split_whitespace().count();
889    match self.parse_marpa(&input_string, nodes, document) {
890      Ok(mut parse_tree) => {
891        self.reset_engine(); // Reset for next parse (engine is in completed state)
892        // Restructure flat formulae with \quad separators to right-recursive nesting
893        // (matching Perl's moreRHS/maybeColRHS right-recursive structure)
894        crate::semantics::restructure_formulae_right(&mut parse_tree)?;
895        // Rename `list` to `vector`/`set` when delimiters wrap the list (Perl encloseN)
896        crate::semantics::rename_fenced_lists(&mut parse_tree, nodes)?;
897        // Combine adjacent SUPOP tokens (prime+prime → prime2)
898        crate::semantics::combine_supop_post(&mut parse_tree, nodes)?;
899        Ok(Some(parse_tree))
900      },
901      Err(_e) => {
902        self.reset_engine();
903        Ok(None)
904      }
905    }
906  }
907}

936fn textrec(
937  node_opt: &Node,
938  outer_bp_opt: Option<usize>,
939  outer_name_opt: Option<&str>,
940  document: &Document,
941) -> String {
942  let node = realize_xmnode(node_opt, document);
943  let tag = get_node_qname(&node);
944  let outer_bp = outer_bp_opt.unwrap_or(0);
945  let outer_name = outer_name_opt.unwrap_or("");
946  // If node has meaning, that's the text form.
947  let meaning_opt = match p_get_attribute(&node, "meaning") {
948    Some(m) => Some(m),
949    None => p_get_attribute(&node, "name"),
950  };
951  if let Some(meaning) = meaning_opt {
952    return match PREFIX_ALIAS.get(meaning.as_str()) {
953      Some(m) => m.to_string(),
954      None => meaning,
955    };
956  }
957  if tag == arena::pin_static("ltx:XMApp") {
958    let mut args = element_nodes(&node);
959    if args.is_empty() {
960      // Error!("expected","arguments" ...);
961      todo!();
962    }
963    let arg_node = args.remove(0);
964    let op = realize_xmnode(&arg_node, document);
965    if let Some(app_role) = node.get_attribute("role") {
966      if app_role == "FLOATSUBSCRIPT" {
967        return String::from("_") + &textrec(&op, None, None, document);
968      } else if app_role == "FLOATSUPERSCRIPT" {
969        return String::from("^") + &textrec(&op, None, None, document);
970      }
971    }
972
973    let name = if with_node_qname(&op, |name| name == "ltx:XMTok") {
974      get_token_meaning(&op, document).unwrap_or_else(|| "unknown".to_owned())
975    } else {
976      String::new()
977    };
978    let (bp, string) = textrec_apply(&name, &op, args, document);
979    if (bp < outer_bp) || ((bp == outer_bp) && (name != outer_name)) {
980      format!("({string})")
981    } else {
982      string
983    }
984  } else if tag == arena::pin_static("ltx:XMDual") {
985    let children = element_nodes(&node);
986    let content = children
987      .first()
988      .expect("XMDual should always have 2 child elements.");
989    textrec(content, Some(outer_bp), Some(outer_name), document) // Just send out the
990  // semantic form
991  // Fall back to
992  // presentation, if
993  // content has poor
994  // semantics (eg. from
995  // replacement patterns)
996  // TODO
997  // return ($text =~
998  // /^\(*Unknown/ ?
999  // textrec($presentation,
1000  // $outer_bp, $outer_name)
1001  // : $text); }
1002  } else if tag == arena::pin_static("ltx:XMTok") {
1003    let name = match get_token_meaning(&node, document) {
1004      Some(meaning) => meaning,
1005      None => s!("Unknown"),
1006    };
1007    match PREFIX_ALIAS.get(name.as_str()) {
1008      Some(v) => v.to_string(),
1009      None => name,
1010    }
1011  } else if tag == arena::pin_static("ltx:XMWrap") || tag == arena::pin_static("ltx:XMCell") {
1012    // ??
1013    element_nodes(&node)
1014      .into_iter()
1015      .map(|child| textrec(&child, None, None, document))
1016      .collect::<Vec<_>>()
1017      .join("@")
1018  } else if tag == arena::pin_static("ltx:XMArg") {
1019    let args = element_nodes(&node);
1020    if args.is_empty() {
1021      return String::new();
1022    }
1023    args
1024      .iter()
1025      .map(|arg| textrec(arg, None, None, document))
1026      .collect::<Vec<_>>()
1027      .join("")
1028  } else if tag == arena::pin_static("ltx:XMArray") {
1029    textrec_array(&node, document)
1030  } else {
1031    s!("[{}]", p_get_value(&node))
1032  }
1033}

1035fn textrec_apply(name: &str, op: &Node, args: Vec<Node>, document: &Document) -> (usize, String) {
1036  let role = op
1037    .get_attribute("role")
1038    .unwrap_or_else(|| "Unknown".to_string());
1039  if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP")
1040    && PRE_DIGITS_RE.is_match(&op.get_attribute("scriptpos").unwrap_or_default())
1041  {
1042    // Note that this will likely get parenthesized due to high bp
1043    let mut inner = textrec(op, None, None, document);
1044    if let Some(arg2) = args.get(1) {
1045      inner.push(' ');
1046      inner.push_str(&textrec(arg2, None, None, document));
1047    }
1048    if let Some(arg1) = args.first() {
1049      inner.push(' ');
1050      inner.push_str(&textrec(arg1, None, None, document));
1051    }
1052    (5000, inner)
1053  } else if let Some(bp) = IS_INFIX.get(&role) {
1054    // A sub/superscript with a meaning probably should be prefix
1055    if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP") && op.has_attribute("meaning") {
1056      (
1057        500,
1058        format!(
1059          "{}@({})",
1060          textrec(op, Some(10000), Some(name), document),
1061          args
1062            .iter()
1063            .map(|a| textrec(a, None, None, document))
1064            .collect::<Vec<_>>()
1065            .join(", ")
1066        ),
1067      )
1068    } else {
1069      // Format as infix.
1070      let textrec_op = textrec(op, None, None, document);
1071      let rec_form = if args.len() == 1 {
1072        // unless a single arg; then prefix.
1073        textrec_op + " " + &textrec(&args[0], Some(*bp), Some(name), document)
1074      } else {
1075        args
1076          .iter()
1077          .map(|a| textrec(a, Some(*bp), Some(name), document))
1078          .collect::<Vec<_>>()
1079          .join(&(" ".to_string() + &textrec_op + " "))
1080      };
1081      (*bp, rec_form)
1082    }
1083  } else if role == "POSTFIX" {
1084    (
1085      10000,
1086      textrec(&args[0], Some(10000), Some(name), document) + &textrec(op, None, None, document),
1087    )
1088  } else if name == "multirelation" {
1089    let joined = args
1090      .iter()
1091      .map(|a| textrec(a, Some(2), Some(name), document))
1092      .collect::<Vec<_>>()
1093      .join(" ");
1094    (2, joined)
1095  } else {
1096    (
1097      500,
1098      textrec(op, Some(10000), Some(name), document)
1099        + "@("
1100        + &args
1101          .iter()
1102          .map(|a| textrec(a, None, None, document))
1103          .collect::<Vec<_>>()
1104          .join(", ")
1105        + ")",
1106    )
1107  }
1108}

1110fn textrec_array(node: &Node, document: &Document) -> String {
1111  let name = node
1112    .get_attribute("meaning")
1113    .or_else(|| node.get_attribute("name"))
1114    .unwrap_or_else(|| String::from("Array"));
1115  let rows: Vec<String> = element_nodes(node)
1116    .into_iter()
1117    .map(|row| {
1118      let cells: Vec<String> = element_nodes(&row)
1119        .into_iter()
1120        .map(|cell| {
1121          if let Some(first_child) = cell.get_first_child() {
1122            textrec(&first_child, None, None, document)
1123          } else {
1124            String::new()
1125          }
1126        })
1127        .collect();
1128      format!("[{}]", cells.join(", "))
1129    })
1130    .collect();
1131  format!("{}[{}]", name, rows.join(", "))
1132}

1134//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1135// Cute! Were it NOT for Sub/Superscripts, the whole parsing process only
1136// builds a new superstructure around the sequence of token nodes in the input.
1137// Thus, any internal structure is unchanged.
1138//  They get re-parented, but if the parse fails, we've only got to put them
1139// BACK into the original node, to recover the original arrangment!!!
1140// Thus, we don't have to clone, and deal with namespace duplication.
1141// ...
1142// EXCEPT, as I said, for sub/superscripts!!!!
1143//
1144
1145//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1146// Constructors used in grammar
1147// All the tree construction in the grammar should come through these
1148// operations. We avoid mucking with the actual XML nodes (both to avoid
1149// modifying the original tree until we have a successful parse, and to avoid
1150// XML::LibXML cloning nightmares) We are converting XML nodes to array
1151// representation: [$tag, {%attr},@children] This means any inspection of
1152// nodes has to recognize that  * node may be in XML vs ARRAY representation
1153// * node may be an XMRef to another node whose properties are the ones we
1154// should use.
1155//
1156// Also, when we are examining a node's properties (roles, fences, script
1157// positioning, etc) we should be careful to check for XMRef indirection and
1158// examine the properties of the node that was referred to.
1159// HOWEVER, we should construct our parse tree using (a clone of) the XMRef
1160// node, rather than (a clone of) the referred to node, so as to preserve
1161// identity.
1162//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1163// We're currently keeping the id's on the nodes as they get cloned,
1164// since they'll (maybe) replace the unparsed nodes.
1165// However, if we consider multiple parses or preserving both parsed & unparsed,
1166// we may have to do some adaptation and id shifting.
1167// ================================================================================
1168
1169// ================================================================================
1170// Low-level accessors
1171
1172// The following accessors work on both the LibXML and ARRAY representations
1173// but they do NOT automatically dereference XMRef!
1174pub fn p_get_value(node: &Node) -> String {
1175  let node_type = node.get_type();
1176  if node_type == Some(NodeType::ElementNode) {
1177    let x = node.get_content();
1178    if !x.is_empty() {
1179      // get content, or fall back to name
1180      x
1181    } else {
1182      node.get_attribute("name").unwrap_or_default()
1183    }
1184  } else {
1185    node.get_content()
1186  }
1187}

1216/// Resolve _xmkey and _pxmkey references after parse tree installation.
1217/// Matches XMRef[@_xmkey] to elements with same _xmkey, generates xml:id and sets idref.
1218/// _pxmkey is used by parser-generated XMDual (apply_delimited) to avoid
1219/// conflicting with base_xmath's \lx@dual afterConstruct resolver.
1220fn resolve_xmkeys(mathnode: &Node, document: &mut Document) -> std::result::Result<(), Box<dyn std::error::Error>> {
1221  // Resolve both _xmkey (from existing infrastructure) and _pxmkey (from parser)
1222  for attr_name in &["_xmkey", "_pxmkey"] {
1223    let refs = document.findnodes(
1224      &format!("descendant::ltx:XMRef[@{}]", attr_name), Some(mathnode));
1225    for mut ref_node in refs {
1226      let key = match ref_node.get_attribute(attr_name) {
1227        Some(k) => k,
1228        None => continue,
1229      };
1230      // Find the element with matching key (non-XMRef)
1231      let xpath = format!("descendant::*[@{}='{}'][not(self::ltx:XMRef)]", attr_name, key);
1232      let targets = document.findnodes(&xpath, Some(mathnode));
1233      if let Some(mut target) = targets.into_iter().next() {
1234        // Ensure target has xml:id
1235        let target_id = if let Some(id) = target.get_attribute("xml:id")
1236          .or_else(|| target.get_attribute("id"))
1237        {
1238          id
1239        } else {
1240          document.generate_id(&mut target, "")?;
1241          target.get_attribute("xml:id")
1242            .or_else(|| target.get_attribute("id"))
1243            .unwrap_or_default()
1244        };
1245        if !target_id.is_empty() {
1246          document.set_attribute(&mut ref_node, "idref", &target_id)?;
1247        }
1248      }
1249      let _ = ref_node.remove_attribute(attr_name);
1250    }
1251    // Clean up from non-ref elements
1252    for mut node in document.findnodes(
1253      &format!("descendant::*[@{}]", attr_name), Some(mathnode))
1254    {
1255      let _ = node.remove_attribute(attr_name);
1256    }
1257  }
1258  Ok(())
1259}
