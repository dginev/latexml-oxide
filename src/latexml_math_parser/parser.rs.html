151  /// Reset the marpa engine after a failed parse.
152  /// Creates a new engine and runs a trivial parse to advance past the
153  /// precompute step (grammar is already precomputed, can't do it again).
154  fn reset_engine(&mut self) {
155    let mut engine = Parser::with_grammar(self.grammar.clone());
156    // Run a trivial recognizer to advance state from G (precompute) through
157    // R → B → O → T → GReady. Use "NUMBER:1:1 " which is a valid single-token formula.
158    match engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 "))) {
159      Ok(_) => { self.engine = engine; },
160      Err(e) => {
161        eprintln!("RESET FAILED: {e}");
162        // If the trivial parse also fails, rebuild from scratch
163        let (grammar, _actions, _builder) = init_grammar().unwrap();
164        let thin_grammar = grammar.unwrap();
165        self.grammar = thin_grammar.clone();
166        self.engine = Parser::with_grammar(thin_grammar);
167        // Now run the trivial parse to get past precompute
168        let _ = self.engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 ")));
169      }
170    }
171  }

173  /// Test if the recognizer accepts a given input string (for unit testing).
174  pub fn recognizes(&mut self, input: &str) -> bool {
175    let result = self.engine.run_recognizer(ByteScanner::new(Cursor::new(input)));
176    if result.is_err() {
177      self.reset_engine();
178    }
179    result.is_ok()
180  }

369  // Recursively parse a node with some internal structure
370  // by first parsing any structured children, then it's content.
371  fn parse_rec(
372    &mut self,
373    mut node: Node,
374    rule_opt: &str,
375    document: &mut Document,
376  ) -> Result<Option<Node>> {
377    self.parse_children(&node, document)?;
378    // This will only handle 1 layer nesting (successfully?)
379    // Note that this would have been found by the top level xpath,
380    // but we've got to worry about node identity: the parent is being rebuilt
381    for nested in document.findnodes("descendant::ltx:XMath", Some(&node)) {
382      self.parse(nested, document)?;
383    }
384    let tag = get_node_qname(&node);
385    let rule = if let Some(requested_rule) = node.get_attribute("rule") {
386      requested_rule
387    } else {
388      rule_opt.to_owned()
389    };
390
391    if rule == "kludge" {
392      self.parse_kludge(&mut node, document);
393      Ok(None)
394    } else if let Some(mut result) = self.parse_single(&mut node, document, &rule)? {
395      *self.passed.entry_sym(tag).or_insert(0) += 1;
396      if tag == arena::pin_static("ltx:XMath") {
397        // Replace the content of XMath with parsed result
398        self.n_parsed += 1;
399        note_progress(&s!("[{}]", self.n_parsed));
400        for el_node in element_nodes(&node) {
401          document.unrecord_node_ids(&el_node);
402        }
403        // unbindNode followed by (append|replace)Tree (which removes ID's) should
404        // be safe
405        for mut child in node.get_child_nodes() {
406          child.unbind_node();
407        }
408        document.append_tree(&mut node, vec![result])?;
409        let mut new_element_children = element_nodes(&node);
410        result = new_element_children.remove(0);
411      } else {
412        // Replace the whole node for XMArg, XMWrap; preserve some attributes
413        //ProgressStep() if ($$self{progress}++ % $MATHPARSE_PROGRESS_QUANTUM) == 0;
414        // Copy all attributes
415        let resultid = p_get_attribute(&result, "id");
416        let mut attr = node.get_attributes();
417
418        // add to result, even allowing modification of xml node, since we're committed.
419        // [Annotate converts node to array which messes up clearing the id!]
420        let rtag = get_node_qname(&result);
421        // TODO: Is this needed in a world where `result` is always a `Node` ?
422        // // // Make sure font is "Appropriate", if we're creating a new token (yuck)
423        // // if ($isarr && $attr{_font} && ($rtag eq 'ltx:XMTok')) {
424        // // my $content = join('', @$result[2 .. $#$result]);
425        // // if ((!defined $content) || ($content eq '')) {
426        // //   delete $attr{_font}; }    # No font needed
427        // // elsif (my $font = $document->decodeFont($attr{_font})) {
428        // //   delete $attr{_font};
429        // //   $attr{font} = $font->specialize($content); } }
430        // // } else {
431        attr.remove("_font");
432        // TODO: See the namespaced attribute issue for libxml's wrapper:
433        //  https://github.com/KWARC/rust-libxml/issues/104
434        // until then, HACK ids.
435        let newid = attr.remove("id");
436        if let Some(ref nid) = newid {
437          attr.insert(String::from("xml:id"), nid.to_owned());
438        }
439        for (key, value) in attr {
440          if !(key.starts_with('_') || sym_can_have_attribute(rtag, arena::pin(&key))) {
441            continue;
442          }
443          if key == "xml:id" {
444            // Since we're moving the id...bookkeeping
445            document.unrecord_id(&value);
446            node.remove_attribute("xml:id")?;
447          }
448          // TODO: is the array/XM case still relevant?
449          // if ($isarr) { $$result[1]{$key} = $value; } else {
450          document.set_attribute(&mut result, &key, &value)?;
451          // }
452        }
453        result = document
454          .replace_tree(result, node)?
455          .expect("replacing the tree should always work.");
456        // Danger: the above code replaced the id on the parsed result with the one from XMArg,..
457        // If there are any references to `resultid`, we need to point them to `newid`!
458        if let Some(rid) = resultid {
459          if let Some(nid) = newid {
460            if rid != nid {
461              for mut tref in document.findnodes(&s!("//*[@idref='{}']", rid), None) {
462                tref.set_attribute("idref", &nid)?;
463              }
464            }
465          }
466        }
467      }
468      Ok(Some(result))
469    } else {
470      // TODO:
471      // self.parse_kludge(node, document);
472      // ProgressStep() if ($$self{progress}++ % $MATHPARSE_PROGRESS_QUANTUM) == 0;
473      // $$self{failed}{$tag}++;
474      Ok(None)
475    }
476  }

478  // Depth first parsing of XMArg nodes.
479  fn parse_children(&mut self, node: &Node, document: &mut Document) -> Result<()> {
480    for child in element_nodes(node) {
481      let tag = get_node_qname(&child);
482      if tag == arena::pin_static("ltx:XMArg") || tag == arena::pin_static("ltx:XMWrap") {
483        self.parse_rec(child, "Anything", document)?;

488        || tag == arena::pin_static("ltx:XMDual")

491      }
492    }
493    Ok(())
494  }

511  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
512  // Low-level Parser: parse a single expression
513  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
514  // Convert to textual form for processing by MathGrammar
515  fn parse_single(
516    &mut self,
517    mathnode: &mut Node,
518    document: &mut Document,
519    rule: &str,
520  ) -> Result<Option<Node>> {
521    let mut idx = 0;
522    let mut content_nodes = filter_hints(mathnode.get_child_nodes());
523
524    // Extract trailing PUNCT/PERIOD nodes if rule ends with ',' (Perl: $rule =~ s/,$// )
525    let mut punct_nodes: Vec<Node> = Vec::new();
526    if rule.ends_with(',') {
527      while let Some(last) = content_nodes.last() {
528        let role = last.get_attribute("role").unwrap_or_default();
529        if role == "PUNCT" || role == "PERIOD" {
530          let mut p = content_nodes.pop().unwrap();
531          p.unlink(); // detach from mathnode's children
532          punct_nodes.insert(0, p);
533        } else {
534          break;
535        }
536      }
537    }
538
539    if content_nodes.is_empty() {
540      Ok(None) // nothing to parse
541    } else if content_nodes.len() == 1 && punct_nodes.is_empty() {
542      // single node, nothing to wrap
543      Ok(Some(content_nodes.remove(0)))
544    } else if content_nodes.len() == 1 {
545      // single node with trailing punct: wrap in XMDual
546      let result = content_nodes.remove(0);
547      Ok(Some(self.wrap_with_punct(result, punct_nodes, mathnode, document)?))
548    } else {
549      let (lexemes, mut nodes) = node_to_grammar_lexemes(mathnode, &mut idx);
550      if let Ok(Some(parse_tree)) = self.parse_lexemes(lexemes, &nodes, document) {
551        //START reparent: the reparenting used to be in `parse_rec` in Perl. Is this a good place?
552        // Replace the content of XMath with parsed result
553        // unbindNode followed by (append|replace)Tree (which removes ID's) should be safe
554        for child_el in element_nodes(mathnode) {
555          document.unrecord_node_ids(&child_el);
556        }
557        for mut node in mathnode.get_child_nodes() {
558          node.unlink();
559        }
560        let new_xml_tree = parse_tree.into_xmath(mathnode, &mut nodes, document)?;
561        document.append_tree(mathnode, vec![new_xml_tree])?;
562        let result = element_nodes(mathnode).remove(0);
563        //END reparent.
564        if !punct_nodes.is_empty() {
565          Ok(Some(self.wrap_with_punct(result, punct_nodes, mathnode, document)?))
566        } else {
567          Ok(Some(result))
568        }
569      } else {
570        // Parse failed; put punct nodes back in mathnode so they remain visible
571        for mut p in punct_nodes {
572          mathnode.add_child(&mut p).ok();
573        }
574        Ok(None)
575      }
576    }
577  }

579  /// Wrap `result` in an XMDual so trailing punctuation appears in the presentation
580  /// branch only, matching Perl's `if (@punct) { $result = ['ltx:XMDual', ...] }`.
581  fn wrap_with_punct(
582    &self,
583    mut result: Node,
584    punct_nodes: Vec<Node>,
585    mathnode: &mut Node,
586    document: &mut Document,
587  ) -> Result<Node> {
588    // Assign an id to result while it's still in the tree (id generation needs ancestors)
589    document.generate_id(&mut result, "")?;
590    let id = result
591      .get_attribute_ns("id", XML_NS)
592      .unwrap_or_else(|| result.get_attribute("id").unwrap_or_default());
593    // Detach result from mathnode so we can re-parent it inside XMWrap
594    result.unlink();
595    // Create XMDual in mathnode
596    let mut dual = document.open_element_at(mathnode, "ltx:XMDual", None, None)?;
597    // Content branch: XMRef pointing to result
598    let ref_attrs = {
599      let mut m = HashMap::default();
600      m.insert(s!("idref"), id);
601      Some(m)
602    };
603    let mut xmref = document.open_element_at(&mut dual, "ltx:XMRef", ref_attrs, None)?;
604    document.close_element_at(&mut xmref)?;
605    // Presentation branch: XMWrap containing result + trailing punct
606    let mut xmwrap = document.open_element_at(&mut dual, "ltx:XMWrap", None, None)?;
607    xmwrap.add_child(&mut result).ok();
608    for mut p in punct_nodes {
609      xmwrap.add_child(&mut p).ok();
610    }
611    document.close_element_at(&mut xmwrap)?;
612    document.close_element_at(&mut dual)?;
613    Ok(dual)
614  }

616  pub fn parse_marpa(
617    &mut self,
618    input: &str,
619    nodes: &[Node],
620    document: &mut Document,
621  ) -> Result<XM> {
622    let parse_result = self
623      .engine
624      .run_recognizer(ByteScanner::new(Cursor::new(input)))?;
625    let mut parses = Vec::new();
626    let mut ok_trees = 0;
627    let mut pruned_trees = 0;
628    for val in parse_result {
629      match self.actions.get_tree(
630        self.builder.clone(),
631        val,
632        self.expert_pragmatics.as_slice(),
633        ActionContext { nodes, document },
634      ) {
635        Ok(tree_opt) => {
636          if let Some(tree) = tree_opt {
637            // eprintln!("-- we found a tree: {:?}", tree);
638            ok_trees += 1;
639            // ignore semantically pruned parses
640            parses.push(tree);
641          }
642        },
643        Err(_prune_err) => {
644          pruned_trees += 1;
645        }, // bookkeep the prune reasons?
646      }
647    }
648    if ok_trees + pruned_trees > 100 {
649      // let warning1 = format!(
650      //   "WARNING! too many marpa trees: {:?}, accepted as semantic trees: {:?}",
651      //   ok_trees + pruned_trees,
652      //   ok_trees
653      // );
654      // let warning2 = format!("         on input: {:?}", input);
655      // eprintln!("\n{}", Yellow.bold().paint(warning1));
656      // eprintln!("{}\n", Yellow.paint(warning2));
657    }
658
659    match parses.len() {
660      0 => Err("Failed to find any parse".into()),
661      1 => Ok(parses.into_iter().next().unwrap()),
662      2 | 3 => Ok(XM::Choices(parses)),
663      _more => {
664        // Loop over the various soft pruning algorithms available, until we are at 3 trees or less
665        let mut reduced_forest = XM::Choices(parses);
666        for pragma in self.student_pragmatics.iter() {
667          reduced_forest = reduced_forest.soft_prune_choices(*pragma);
668          match reduced_forest {
669            XM::Choices(ref trees) => match trees.len() {
670              2 | 3 => break, //reduced sufficiently, return
671              _more => {},    // keep trying to reduce
672            },
673            _ => break, // reduced sufficiently, return
674          };
675        }
676        Ok(reduced_forest)
677      },
678    }
679  }
680
681  pub fn parse_lexemes(
682    &mut self,
683    lexemes: Vec<String>,
684    nodes: &[Node],
685    document: &mut Document,
686  ) -> Result<Option<XM>> {
687    let mut input_string: String = lexemes.join(" ");
688    // Add a trailing space, in an attempt to work with
689    // a rules!() macro that has a Hard expectation of a space char following EVERY token.
690    // this - counterintuitively- allows a simple macro definition AND a simple parse tree.
691    input_string.push(' ');
692    match self.parse_marpa(&input_string, nodes, document) {
693      Ok(parse_tree) => Ok(Some(parse_tree)),
694      Err(_e) => {
695        // Reset the engine after a failed parse, since the marpa recognizer
696        // gets stuck in a started state and can't be reused.
697        self.reset_engine();
698        Ok(None)
699      }
700    }
701  }
702}

731fn textrec(
732  node_opt: &Node,
733  outer_bp_opt: Option<usize>,
734  outer_name_opt: Option<&str>,
735  document: &Document,
736) -> String {
737  let node = realize_xmnode(node_opt, document);
738  let tag = get_node_qname(&node);
739  let outer_bp = outer_bp_opt.unwrap_or(0);
740  let outer_name = outer_name_opt.unwrap_or("");
741  // If node has meaning, that's the text form.
742  let meaning_opt = match p_get_attribute(&node, "meaning") {
743    Some(m) => Some(m),
744    None => p_get_attribute(&node, "name"),
745  };
746  if let Some(meaning) = meaning_opt {
747    return match PREFIX_ALIAS.get(meaning.as_str()) {
748      Some(m) => m.to_string(),
749      None => meaning,
750    };
751  }
752  if tag == arena::pin_static("ltx:XMApp") {
753    let mut args = element_nodes(&node);
754    if args.is_empty() {
755      // Error!("expected","arguments" ...);
756      todo!();
757    }
758    let arg_node = args.remove(0);
759    let op = realize_xmnode(&arg_node, document);
760    if let Some(app_role) = node.get_attribute("role") {
761      if app_role == "FLOATSUBSCRIPT" {
762        return String::from("_") + &textrec(&op, None, None, document);
763      } else if app_role == "FLOATSUPERSCRIPT" {
764        return String::from("^") + &textrec(&op, None, None, document);
765      }
766    }
767
768    let name = if with_node_qname(&op, |name| name == "ltx:XMTok") {
769      get_token_meaning(&op, document).unwrap_or_else(|| "unknown".to_owned())
770    } else {
771      String::new()
772    };
773    let (bp, string) = textrec_apply(&name, &op, args, document);
774    if (bp < outer_bp) || ((bp == outer_bp) && (name != outer_name)) {
775      format!("({string})")
776    } else {
777      string
778    }
779  } else if tag == arena::pin_static("ltx:XMDual") {
780    let children = element_nodes(&node);
781    let content = children
782      .first()
783      .expect("XMDual should always have 2 child elements.");
784    textrec(content, Some(outer_bp), Some(outer_name), document) // Just send out the
785  // semantic form
786  // Fall back to
787  // presentation, if
788  // content has poor
789  // semantics (eg. from
790  // replacement patterns)
791  // TODO
792  // return ($text =~
793  // /^\(*Unknown/ ?
794  // textrec($presentation,
795  // $outer_bp, $outer_name)
796  // : $text); }
797  } else if tag == arena::pin_static("ltx:XMTok") {
798    let name = match get_token_meaning(&node, document) {
799      Some(meaning) => meaning,
800      None => s!("Unknown"),
801    };
802    match PREFIX_ALIAS.get(name.as_str()) {
803      Some(v) => v.to_string(),
804      None => name,
805    }
806  } else if tag == arena::pin_static("ltx:XMWrap") || tag == arena::pin_static("ltx:XMCell") {
807    // ??
808    element_nodes(&node)
809      .into_iter()
810      .map(|child| textrec(&child, None, None, document))
811      .collect::<Vec<_>>()
812      .join("@")
813  } else if tag == arena::pin_static("ltx:XMArg") {
814    let args = element_nodes(&node);
815    if args.is_empty() {
816      return String::new();
817    }
818    args
819      .iter()
820      .map(|arg| textrec(arg, None, None, document))
821      .collect::<Vec<_>>()
822      .join("")
823  } else if tag == arena::pin_static("ltx:XMArray") {
824    textrec_array(&node, document)
825  } else {
826    s!("[{}]", p_get_value(&node))
827  }
828}

830fn textrec_apply(name: &str, op: &Node, args: Vec<Node>, document: &Document) -> (usize, String) {
831  let role = op
832    .get_attribute("role")
833    .unwrap_or_else(|| "Unknown".to_string());
834  if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP")
835    && PRE_DIGITS_RE.is_match(&op.get_attribute("scriptpos").unwrap_or_default())
836  {
837    // Note that this will likely get parenthesized due to high bp
838    let mut inner = textrec(op, None, None, document);
839    if let Some(arg2) = args.get(1) {
840      inner.push(' ');
841      inner.push_str(&textrec(arg2, None, None, document));
842    }
843    if let Some(arg1) = args.first() {
844      inner.push(' ');
845      inner.push_str(&textrec(arg1, None, None, document));
846    }
847    (5000, inner)
848  } else if let Some(bp) = IS_INFIX.get(&role) {
849    // A sub/superscript with a meaning probably should be prefix
850    if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP") && op.has_attribute("meaning") {
851      (
852        500,
853        format!(
854          "{}@({})",
855          textrec(op, Some(10000), Some(name), document),
856          args
857            .iter()
858            .map(|a| textrec(a, None, None, document))
859            .collect::<Vec<_>>()
860            .join(", ")
861        ),
862      )
863    } else {
864      // Format as infix.
865      let textrec_op = textrec(op, None, None, document);
866      let rec_form = if args.len() == 1 {
867        // unless a single arg; then prefix.
868        textrec_op + " " + &textrec(&args[0], Some(*bp), Some(name), document)
869      } else {
870        args
871          .iter()
872          .map(|a| textrec(a, Some(*bp), Some(name), document))
873          .collect::<Vec<_>>()
874          .join(&(" ".to_string() + &textrec_op + " "))
875      };
876      (*bp, rec_form)
877    }
878  } else if role == "POSTFIX" {
879    (
880      10000,
881      textrec(&args[0], Some(10000), Some(name), document) + &textrec(op, None, None, document),
882    )
883  } else if name == "multirelation" {
884    let joined = args
885      .iter()
886      .map(|a| textrec(a, Some(2), Some(name), document))
887      .collect::<Vec<_>>()
888      .join(" ");
889    (2, joined)
890  } else {
891    (
892      500,
893      textrec(op, Some(10000), Some(name), document)
894        + "@("
895        + &args
896          .iter()
897          .map(|a| textrec(a, None, None, document))
898          .collect::<Vec<_>>()
899          .join(", ")
900        + ")",
901    )
902  }
903}

905fn textrec_array(node: &Node, document: &Document) -> String {
906  let name = node
907    .get_attribute("meaning")
908    .or_else(|| node.get_attribute("name"))
909    .unwrap_or_else(|| String::from("Array"));
910  let rows: Vec<String> = element_nodes(node)
911    .into_iter()
912    .map(|row| {
913      let cells: Vec<String> = element_nodes(&row)
914        .into_iter()
915        .map(|cell| {
916          if let Some(first_child) = cell.get_first_child() {
917            textrec(&first_child, None, None, document)
918          } else {
919            String::new()
920          }
921        })
922        .collect();
923      format!("[{}]", cells.join(", "))
924    })
925    .collect();
926  format!("{}[{}]", name, rows.join(", "))
927}

929//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
930// Cute! Were it NOT for Sub/Superscripts, the whole parsing process only
931// builds a new superstructure around the sequence of token nodes in the input.
932// Thus, any internal structure is unchanged.
933//  They get re-parented, but if the parse fails, we've only got to put them
934// BACK into the original node, to recover the original arrangment!!!
935// Thus, we don't have to clone, and deal with namespace duplication.
936// ...
937// EXCEPT, as I said, for sub/superscripts!!!!
938//
939
940//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
941// Constructors used in grammar
942// All the tree construction in the grammar should come through these
943// operations. We avoid mucking with the actual XML nodes (both to avoid
944// modifying the original tree until we have a successful parse, and to avoid
945// XML::LibXML cloning nightmares) We are converting XML nodes to array
946// representation: [$tag, {%attr},@children] This means any inspection of
947// nodes has to recognize that  * node may be in XML vs ARRAY representation
948// * node may be an XMRef to another node whose properties are the ones we
949// should use.
950//
951// Also, when we are examining a node's properties (roles, fences, script
952// positioning, etc) we should be careful to check for XMRef indirection and
953// examine the properties of the node that was referred to.
954// HOWEVER, we should construct our parse tree using (a clone of) the XMRef
955// node, rather than (a clone of) the referred to node, so as to preserve
956// identity.
957//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
958// We're currently keeping the id's on the nodes as they get cloned,
959// since they'll (maybe) replace the unparsed nodes.
960// However, if we consider multiple parses or preserving both parsed & unparsed,
961// we may have to do some adaptation and id shifting.
962// ================================================================================
963
964// ================================================================================
965// Low-level accessors
966
967// The following accessors work on both the LibXML and ARRAY representations
968// but they do NOT automatically dereference XMRef!
969pub fn p_get_value(node: &Node) -> String {
970  let node_type = node.get_type();
971  if node_type == Some(NodeType::ElementNode) {
972    let x = node.get_content();
973    if !x.is_empty() {
974      // get content, or fall back to name
975      x
976    } else {
977      node.get_attribute("name").unwrap_or_default()
978    }
979  } else {
980    node.get_content()
981  }
982}

986pub fn realize_xmnode<'a>(node: &'a Node, document: &'a Document) -> Cow<'a, Node> {
987  if with_node_qname(node, |name| name == "ltx:XMRef") {
988    if let Some(idref) = node.get_attribute("idref") {
989      // Can it happen that $realnode is, itself, an XMRef?
990      // Then we should recurse recurse!
991      if let Some(realnode) = document.lookup_id(&idref) {
992        return Cow::Borrowed(realnode);
993      } else {
994        let message = s!("Cannot find a node with xml:id='{}'", idref);
995        // TODO:
996        // LaTeXML::MathParser::IDREFS{$idref}
997        // ? "Previously bound to " .
998        // ToString($LaTeXML::MathParser::IDREFS{$idref})           : ()));
999        let err = || {
1000          Error!("expected", "id", message);
1001          Ok(())
1002        };
1003        err().ok();
1004        //       return ['ltx:ERROR', {}, "Missing XMRef idref=$idref"]; } }
1005      }
1006    }
1007  }
1008  Cow::Borrowed(node)
1009}
