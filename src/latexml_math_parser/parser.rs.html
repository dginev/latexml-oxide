111// Route `parse_marpa` through one of three paths. See
112// docs/MATH_PARSER_AND_ASF.md and marpa/docs/ASF_PERFORMANCE_FINDINGS.md.
113//
114// **HYBRID is now the default** (2026-05-17). Hybrid reads the tokens
115// once, builds the bocage once, then checks Marpa's raw
116// `ambiguity_metric()`:
117//   * metric == 1 (unambiguous) → cheap Tree-iteration via
118//     `Actions::get_tree`, same machinery the legacy path uses.
119//   * metric >= 2 (ambiguous)   → ASF traversal via `MathTraverser`,
120//     same machinery the ASF-only default used.
121//
122// Measured: Article-2025.tex (579 math-heavy formulae) wall is
123// 12.40s under HYBRID vs 17.00s under pure ASF and 12.21s under
124// pure LEGACY. Hybrid is within 1.05x LEGACY on the math-heavy
125// fixture while preserving ASF's algorithmic advantage on the
126// raw-ambiguous fraction (12.7%–40% across corpus papers).
127//
128// Escape hatches:
129//   * `LATEXML_MARPA_LEGACY=1`  → pure Tree-iteration with the 6
130//     convergence caps. Useful for engine-divergence debugging.
131//   * `LATEXML_MARPA_ASF_ONLY=1` → pure ASF (no hybrid dispatch).
132//     Useful for measuring ASF-only cost or debugging ASF behaviour
133//     in isolation.

135static PARSE_VIA_ASF_ONLY: Lazy<bool> = Lazy::new(|| {
136  !*PARSE_VIA_LEGACY && std::env::var("LATEXML_MARPA_ASF_ONLY").is_ok()
137});

141// Large-bocage fallback cap (marpa commit 5f6a19e + perf branch).
142// When the post-recognizer bocage exceeds this many total and-nodes,
143// `parse_hybrid_with_and_node_limit` routes us back through Tree
144// iteration (legacy caps) instead of ASF construction. ASF must
145// allocate the entire Rust-side glade/factoring view up front; on
146// math-bound papers with high-cardinality ambiguous forests this
147// blew through the 8 GB ulimit (19/100 OOMs on the math-bound
148// sample — see docs/PERFORMANCE.md "HYBRID at scale" addendum).
149//
150// Default 500: downstream consumers (pragmatics selection, XMath
151// builders) can't usefully process more than a handful of distinct
152// parses per formula — beyond a few hundred and-nodes the
153// `dispatch_action` Cartesian product produces alternatives that
154// will be dropped or collapsed anyway. Bigger bocages route through
155// the Tree iterator's 6 convergence caps (max_unique=10, etc.)
156// which already match what semantic selection can use. Override via
157// `LATEXML_MARPA_HYBRID_AND_NODE_LIMIT`; set `=0` (or `none`) to
158// disable the cap and force pure ASF on every ambiguous formula.

161    Ok(v) => {

164        None
165      } else {
166        trimmed.parse::<usize>().ok().or(Some(500))
167      }
168    },
169    Err(_) => Some(500),
170  }

173// Maximum number of grammar lexemes in a single formula before we skip the
174// full Marpa grammar parse and fall through to the kludge parser (the same
175// path a genuine parse failure takes). Marpa's Earley recognizer allocates
176// O(n²)+ obstack memory in the number of lexemes; on a pathologically huge
177// single formula (witness 1706.06621: an explicitly-expanded E6 fluxbrane
178// polynomial `H3 = B_1^4 B_2^8 …` of ~43k lexemes / 1 MB) it exhausts memory
179// and libmarpa's `default_out_of_memory` calls `abort()` — an UNCATCHABLE
180// SIGABRT that takes down the whole conversion (the parse runs on the worker
181// thread; there is no Rust panic to catch). Perl (Parse::RecDescent) parses
182// the same formula in bounded memory, so this is a Rust-only abort; the cap
183// degrades only the one giant formula to a kludge parse (OPEN/CLOSE matching,
184// linear), keeping the rest of the document convertible. The largest real
185// formula observed parsing cleanly is ~8.7k lexemes, so the default 12000 is
186// comfortably above genuine math while well below the explosion. Override via
187// `LATEXML_MAX_GRAMMAR_LEXEMES`; set `0`/`none` to disable (full Perl parity,
188// at the risk of the abort). See SYNC_STATUS "FATAL_134 marpa-OOM".
189static MAX_GRAMMAR_LEXEMES: Lazy<Option<usize>> = Lazy::new(|| {
190  match std::env::var("LATEXML_MAX_GRAMMAR_LEXEMES") {
191    Ok(v) => {
192      let trimmed = v.trim();
193      if trimmed == "0" || trimmed.eq_ignore_ascii_case("none") {
194        None
195      } else {
196        Some(trimmed.parse::<usize>().unwrap_or(12000))
197      }
198    },
199    Err(_) => Some(12000),
200  }

232/// Audit semantics for `LATEXML_MARPA_HYBRID_AUDIT_PARITY`.
233///
234/// The parity check answers one load-bearing question:
235/// *"If both paths accept, do they produce the same XM?"* — and
236/// only that one. An `Accepted` vs `Empty/Rejected` mismatch IS a
237/// real divergence (one path returned a tree, the other didn't),
238/// but a `Rejected("…")` vs `Empty` outcome on a formula both
239/// paths failed is shallow: both report "no parse survived",
240/// they just disagree on the failure label. The user-facing HTML
241/// is bit-identical in that case (verified empirically on
242/// Article-2025.tex's set-builder formulae — see commit
243/// 9318960974 and marpa/docs/ASF_PERFORMANCE_FINDINGS.md), so
244/// we treat any pair of non-Accepted/non-Multiple outcomes as
245/// compatible.
246pub(crate) fn parity_outcomes_compatible(
247  asf_outcome: &ParseOutcome,
248  tree_outcome: &ParseOutcome,
249) -> bool {
250  match (asf_outcome, tree_outcome) {
251    (ParseOutcome::Accepted(_), _)
252    | (ParseOutcome::Multiple(_), _)
253    | (_, ParseOutcome::Accepted(_))
254    | (_, ParseOutcome::Multiple(_)) => asf_outcome == tree_outcome,
255    // Both are some flavour of "no parse survived" (Empty or
256    // Rejected, in either order) — equivalent from the user's
257    // perspective.
258    _ => true,
259  }
260}

280fn canonicalize_xm_ids_inner(
281  tree: &mut XM,
282  ids: &mut HashMap<String, Cow<'static, str>>,
283  next_id: &mut usize,
284) {
285  match tree {
286    XM::Lexeme(..) => {},
287    XM::Token(props, _) | XM::Ref(props) => {
288      canonicalize_xprops_ids(props, ids, next_id);
289    },
290    XM::Apply(operator, args, props, _) => {
291      canonicalize_xprops_ids(props, ids, next_id);
292      canonicalize_xm_ids_inner(&mut operator.0, ids, next_id);
293      for arg in args.0.iter_mut().flatten() {
294        canonicalize_xm_ids_inner(arg, ids, next_id);
295      }
296    },
297    XM::Dual(content, presentation, props, _) => {
298      canonicalize_xprops_ids(props, ids, next_id);
299      canonicalize_xm_ids_inner(content, ids, next_id);
300      canonicalize_xm_ids_inner(presentation, ids, next_id);
301    },
302    XM::Wrap(items, props, _) => {
303      canonicalize_xprops_ids(props, ids, next_id);
304      for item in items {
305        canonicalize_xm_ids_inner(item, ids, next_id);
306      }
307    },
308    XM::Arg(items) | XM::Choices(items) => {
309      for item in items {
310        canonicalize_xm_ids_inner(item, ids, next_id);
311      }
312    },
313  }
314}

430/// Recognize a RESOURCE fatal (gullet token/pushback limit, expansion cycle
431/// guard, memory budget) that was raised inside a Marpa semantics action and
432/// flattened into a `marpa::error::Error` string on its way up. These are NOT
433/// semantic parse rejections — swallowing them produced "phantom fatals" (the
434/// `Fatal!` macro had already set the report's fatal flag, so the final
435/// summary said "1 fatal error" while no `Fatal:` line ever reached the log).
436/// The target/category structure is reconstructed from the (distinctive,
437/// engine-owned) message prefixes so the caller can `log_fatal()` + abort
438/// math parsing honestly. Witness math0402448 (amsart + xy-pic).
439fn resource_fatal_from_message(msg: &str) -> Option<latexml_core::common::error::Error> {
440  use latexml_core::common::error::{Error, ErrorCategory, ErrorTarget};
441  let category = if msg.contains("Infinite expansion loop") {
442    ErrorCategory::Recursion
443  } else if msg.contains("Token limit of") {
444    ErrorCategory::TokenLimit
445  } else if msg.contains("Pushback limit of") {
446    ErrorCategory::PushbackLimit
447  } else if msg.contains("Memory budget exceeded") {
448    ErrorCategory::MemoryBudget
449  } else {
450    return None;
451  };
452  Some(Error {
453    target: ErrorTarget::Timeout,
454    category,
455    message: msg.to_string(),
456  })
457}

459impl MathParser {
460  fn audit_hybrid_unambiguous_parity(
461    &self,
462    input: &str,
463    nodes: &[Node],
464    document: &mut Document,
465    tree_outcome: &ParseOutcome,
466  ) {
467    if !*PARSE_HYBRID_AUDIT_PARITY {
468      return;
469    }
470
471    // Explicit audit-only mode. Math actions may annotate XML nodes as
472    // they run, so this intentionally stays behind an env var and should
473    // be used on disposable benchmark/test runs, not normal conversion.
474    let mut parser = Parser::with_precomputed_grammar(self.grammar.clone());
475    let mut traverser = crate::asf_traverser::MathTraverser {
476      actions:      &self.actions,
477      pragmas:      self.expert_pragmatics.as_slice(),
478      builder:      &self.builder,
479      nodes,
480      document,
481      pruned_count: 0,
482    };
483    let mut asf_outcome = match parser.parse_and_traverse_forest(
484      ByteScanner::new(Cursor::new(input)),
485      (),
486      &mut traverser,
487    ) {
488      Ok((alts, _state)) => {
489        let mut trees: Vec<XM> = alts.iter().filter_map(|o| o.clone()).collect();
490        trees.sort_by_key(|t| t.text_summary());
491        trees.dedup();
492        match trees.len() {
493          0 => ParseOutcome::Empty,
494          1 => ParseOutcome::Accepted(trees.remove(0)),
495          _ => ParseOutcome::Multiple(trees),
496        }
497      },
498      Err(e) => ParseOutcome::Rejected(e.to_string()),
499    };
500    let mut tree_outcome = tree_outcome.clone();
501    canonicalize_parse_outcome_ids(&mut asf_outcome);
502    canonicalize_parse_outcome_ids(&mut tree_outcome);
503
504    assert!(
505      parity_outcomes_compatible(&asf_outcome, &tree_outcome),

513  /// Reset the marpa engine after a failed parse.
514  /// Creates a new engine and runs a trivial parse to advance past the
515  /// precompute step (grammar is already precomputed, can't do it again).
516  ///
517  /// Recovery ladder:
518  /// 1. Clone `self.grammar` + trivial parse — cheap, common case.
519  /// 2. Re-clone and retry once — covers transient marpa state hiccups.
520  /// 3. Full `init_grammar()` rebuild — expensive, but some grammar corruption patterns (observed
521  ///    on the `testscripts` regression fixture) only recover after a fresh precompute. The D5
522  ///    "avoid init_grammar fallback" item turned out to describe the ideal, not the floor —
523  ///    legitimate callers still hit step 3.
524  /// 4. Log + keep previous engine. Subsequent `parse_math` / `recognizes` will fail gracefully (0
525  ///    trees, no panic); better than crashing the whole conversion.
526  ///
527  /// The previous implementation called `init_grammar().unwrap()` at
528  /// step 3, which would panic on any Err. The round-17 change removes
529  /// the panic — on total failure we log and keep going.
530  fn reset_engine(&mut self) {
531    if self.try_reset_clone_path().is_ok() {
532      return;
533    }
534    if self.try_reset_clone_path().is_ok() {
535      return;
536    }
537    // Both clone attempts failed — reach for a full rebuild. Expensive,
538    // but the testscripts regression fixture demonstrates that some
539    // corruption patterns only clear after a fresh precompute.
540    match init_grammar() {
541      Ok((grammar, _actions, _builder)) => {
542        let thin_grammar = grammar.unwrap();
543        self.grammar = thin_grammar.clone();
544        let mut fresh_engine = Parser::with_grammar(thin_grammar);
545        let _ = fresh_engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 ")));
546        self.engine = fresh_engine;
547      },
548      Err(e) => {
549        // Perl MathParser.pm:56 Fatal("expected", "MathGrammar", …) —
550        // we deliberately downgrade to warn here because our caller
551        // already has a workable engine in `self.engine`; abandoning
552        // the conversion outright would be more disruptive than
553        // proceeding with the prior grammar state.
554        log_math_warn!(
555          "expected", "MathGrammar",
556          "math parser: init_grammar fallback failed ({}) — leaving engine in last known state",
557          e
558        );
559      },
560    }
561  }

563  fn try_reset_clone_path(&mut self) -> std::result::Result<(), ()> {
564    let mut engine = Parser::with_grammar(self.grammar.clone());
565    // Run a trivial recognizer to advance state from G (precompute) through
566    // R → B → O → T → GReady. Use "NUMBER:1:1 " which is a valid single-token formula.
567    match engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 "))) {
568      Ok(_) => {
569        self.engine = engine;
570        Ok(())
571      },
572      Err(_) => Err(()),
573    }
574  }

592    let parse_result = match self

594      .run_recognizer(ByteScanner::new(Cursor::new(input)))

597      Err(_) => {

600      },

822  // This is a rather peculiar cleanup that needs to be done to manage ids &
823  // idrefs Before parsing, sub/superscripts are represented by an
824  // operator-less XMApp with the script itself as the only child. Ideally,
825  // upon parsing these get merged, combined and disappear into proper XMApp of
826  // an appropriate operator on the base and scripts.  Unless there is a parse
827  // failure, in which case they remain.
828  // The problem comes from various XMDual constructs where an XMRef refers to
829  // the script XMApp. It can occur that one branch parses and the other fails:
830  // This can leave a reference to the script XMApp which no longer exists!
831  // To solve this, we find & replace all references to such script XMApps by an
832  // explicit XMApp with the XMRef refering to the script itself, not the
833  // XMApp. (make sense?)
834  pub fn cleanup_scripts(&mut self, document: &mut Document) {
835    // Perl: cleanupScripts — find script XMApp nodes that may be referenced by XMRef,
836    // and redirect those references to point at the script content (first child) instead
837    // of the XMApp wrapper. This prevents dangling idrefs when the XMApp is consumed
838    // by parsing.
839    static SCRIPT_RE: Lazy<Regex> =
840      Lazy::new(|| Regex::new(r"^(?:PRE|POST|FLOAT)(?:SUB|SUPER)SCRIPT$").unwrap());
841    let apps = document.findnodes(
842      "descendant-or-self::*[@xml:id and contains(@role,'SCRIPT')]",
843      None,
844    );
845    for mut app in apps {
846      let role = match app.get_attribute("role") {
847        Some(r) => r,
848        None => continue,
849      };
850      if !SCRIPT_RE.is_match(&role) {
851        continue;
852      }
853      let appid = match app.get_attribute("xml:id") {
854        Some(id) => id,
855        None => continue,
856      };
857      // Note: using * instead of ltx:XMRef due to XPath namespace issues in nested predicates
858      let refs_xpath = s!("descendant-or-self::*[@idref = '{}']", appid);
859      let refs = document.findnodes(&refs_xpath, None);
860      if refs.is_empty() {
861        continue;
862      }
863      // Get the script content (first child of the XMApp)
864      let mut script = match app.get_first_child() {
865        Some(child) => child,
866        None => continue,
867      };
868      // Ensure the script has an xml:id so we can create XMRef to it
869      if script.get_attribute("xml:id").is_none() {
870        let _ = document.generate_id(&mut script, "");
871      }
872      let script_id = match script.get_attribute("xml:id") {
873        Some(id) => id,
874        None => continue,
875      };
876      // Unregister the app's id and remove the attribute
877      document.unrecord_id(&appid);
878      let _ = app.remove_attribute("xml:id");
879      // Collect app attributes (except xml:id, which we already removed)
880      let attrs: Vec<(String, String)> = app
881        .get_attributes()
882        .into_iter()
883        .filter(|(name, _)| name != "xml:id")
884        .collect();
885      let ns = app.get_namespace();
886      // Replace each ref with an XMApp containing an XMRef to the script
887      for ref_node in refs {
888        // Build the replacement: ltx:XMApp{attrs}[ltx:XMRef{idref=script_id}]
889        let mut new_app = Node::new("XMApp", None, &document.document).unwrap();
890        for (name, value) in &attrs {
891          let _ = new_app.set_attribute(name, value);
892        }
893        if let Some(ref ns) = ns {
894          let _ = new_app.set_namespace(ns);
895        }
896        let mut xmref = Node::new("XMRef", None, &document.document).unwrap();
897        let _ = xmref.set_attribute("idref", &script_id);
898        if let Some(ref ns) = ns {
899          let _ = xmref.set_namespace(ns);
900        }
901        let _ = new_app.add_child(&mut xmref);
902        let _ = document.replace_tree(new_app, ref_node);
903      }
904    }
905  }

1206  // Low-Level hack parsing when "real" parsing fails;
1207  // Two issues cause generated Presentation MathML to be really bad:
1208  // (1) not having mrow/mfenced structures wrapping OPEN...CLOSE sequences
1209  //     throws off MathML's stretchiness treatment of the fences
1210  //     (they're all the same size; big)
1211  // (2) un-attached sub/superscripts won't position correctly,
1212  //     unless they're attached to something plausible.
1213  /// Perl: parse_kludge (MathParser.pm L530-566)
1214  /// Stack-based matching of OPEN/CLOSE delimiter pairs + script attachment.
1215  fn parse_kludge(&self, mathnode: &mut Node, document: &mut Document) {
1216    use crate::data::get_grammatical_role;
1217    let children: Vec<Node> = filter_hints(mathnode.get_child_nodes());
1218    if children.is_empty() {
1219      return;
1220    }
1221
1222    // Build (node, role) pairs — matching Perl's @pairs.
1223    let pairs: Vec<(Node, String)> = children
1224      .into_iter()
1225      .map(|n| {
1226        let r = get_grammatical_role(&n);
1227        (n, r)
1228      })
1229      .collect();
1230
1231    // Perl: @stack = ([], []) — extra empty level handles unmatched leading CLOSEs.
1232    let mut stack: Vec<Vec<(Node, String)>> = vec![vec![], vec![]];
1233    let mut iter = pairs.into_iter().peekable();
1234
1235    while iter.peek().is_some() || stack.len() > 1 {
1236      let pair_opt = iter.next();
1237      let role = pair_opt
1238        .as_ref()
1239        .map(|(_, r)| r.as_str())
1240        .unwrap_or("CLOSE");
1241      match role {
1242        "OPEN" => {
1243          let pair = pair_opt.unwrap();
1244          stack.insert(0, vec![pair]);
1245        },
1246        "CLOSE" => {
1247          let mut row = stack.remove(0);
1248          if let Some(pair) = pair_opt {
1249            row.push(pair);
1250          }
1251          // Perl L546: handle scripts within this fenced group
1252          let kludged = kludge_scripts_rec(row, document);
1253          // Wrap if > 1 node, give role FENCED
1254          let result = if kludged.len() > 1 {
1255            let mut wrap = Node::new("XMWrap", None, document.get_document()).unwrap();
1256            for mut n in kludged {
1257              n.unlink_node();
1258              wrap.add_child(&mut n).ok();
1259            }
1260            (wrap, "FENCED".to_string())
1261          } else {
1262            let node = kludged
1263              .into_iter()
1264              .next()
1265              .unwrap_or_else(|| Node::new_text("", &document.document).unwrap());
1266            (node, "FENCED".to_string())
1267          };
1268          if stack.is_empty() {
1269            stack.push(vec![]);
1270          }
1271          stack[0].push(result);
1272        },
1273        _ => {
1274          if let Some(pair) = pair_opt {
1275            stack[0].push(pair);
1276          }
1277        },
1278      }
1279    }
1280
1281    // Perl L555: process remaining top-level items through kludge_scripts
1282    let final_pairs = stack.into_iter().next().unwrap_or_default();
1283    let result_nodes = kludge_scripts_rec(final_pairs, document);
1284
1285    // Perl L558-563: at top level, unwrap top-level XMWraps (extract children).
1286    // Perl iterates all pairs and extracts children of any array-rep XMWrap.
1287    let mut replacements = Vec::new();
1288    for node in result_nodes {
1289      if node.get_name() == "XMWrap" {
1290        // Unwrap: extract children
1291        for child in node.get_child_nodes() {
1292          replacements.push(child);
1293        }
1294      } else {
1295        replacements.push(node);
1296      }
1297    }
1298
1299    // Rebuild: clear mathnode, then add replacement nodes.
1300    document.unrecord_node_ids(mathnode);
1301    for mut child in mathnode.get_child_nodes() {
1302      child.unbind_node();
1303    }
1304    for mut node in replacements {
1305      node.unlink_node();
1306      mathnode.add_child(&mut node).ok();
1307    }
1308    // D3b: the replacements re-parented above may carry xml:id attrs
1309    // whose idstore entries were cleared by the `unrecord_node_ids`
1310    // above (which walked mathnode's state before the replacements
1311    // were re-inserted). Re-record so `lookup_id` on any
1312    // math-generated id resolves — otherwise finalize()'s
1313    // `rebuild_idstore_from_dom` is the only recovery path and
1314    // downstream XMRef lookups can SIGSEGV via stale cache.
1315    let _ = document.record_node_ids(mathnode);
1316  }

1321  // Convert to textual form for processing by MathGrammar
1322  fn parse_single(
1323    &mut self,
1324    mathnode: &mut Node,
1325    document: &mut Document,
1326    rule: &str,
1327  ) -> Result<Option<Node>> {
1328    let mut idx = 0;
1329    let mut content_nodes = filter_hints(mathnode.get_child_nodes());
1330
1331    // Extract trailing PUNCT/PERIOD nodes if rule ends with ',' (Perl: $rule =~ s/,$// )
1332    let mut punct_nodes: Vec<Node> = Vec::new();
1333    if rule.ends_with(',') {
1334      while let Some(last) = content_nodes.last() {
1335        let role = last.get_attribute("role").unwrap_or_default();
1336        if role == "PUNCT" || role == "PERIOD" {
1337          let mut p = content_nodes.pop().unwrap();
1338          p.unlink(); // detach from mathnode's children
1339          punct_nodes.insert(0, p);
1340        } else {
1341          break;
1342        }
1343      }
1344    }
1345
1346    if content_nodes.is_empty() {
1347      // Perl MathParser.pm L683: $result = $nodes[0] || Absent()
1348      // Empty XMArg/XMWrap: create <XMTok meaning="absent"/>
1349      let mut absent_tok = document.open_element_at(mathnode, "ltx:XMTok", None, None)?;
1350      document.set_attribute(&mut absent_tok, "meaning", "absent")?;
1351      document.close_element_at(&mut absent_tok)?;
1352      absent_tok.unlink();
1353      Ok(Some(absent_tok))
1354    } else if content_nodes.len() == 1 && punct_nodes.is_empty() {
1355      // single node, nothing to wrap
1356      Ok(Some(content_nodes.remove(0)))
1357    } else if content_nodes.len() == 1 {
1358      // single node with trailing punct: wrap in XMDual
1359      let result = content_nodes.remove(0);
1360      Ok(Some(self.wrap_with_punct(
1361        result,
1362        punct_nodes,
1363        mathnode,
1364        document,
1365      )?))
1366    } else {
1367      // Use pre-filtered content_nodes to avoid double-filtering (filter_hints already called
1368      // above)
1369      let (lexemes, mut nodes) = node_to_grammar_lexemes_from(mathnode, content_nodes, &mut idx);
1370      // Skip the full grammar parse for a pathologically huge formula —
1371      // Marpa's Earley recognizer would exhaust memory and `abort()`
1372      // (uncatchable). Fall through to the kludge parser instead (the
1373      // `Ok(None)` branch). See MAX_GRAMMAR_LEXEMES (witness 1706.06621).
1374      let parse_outcome = match *MAX_GRAMMAR_LEXEMES {
1375        Some(cap) if lexemes.len() > cap => {
1376          // Emit a real Warning (not just a progress tick) so a human — or a
1377          // future Claude tasked with raising/removing this cap — has the full
1378          // context. The message explains WHAT was degraded, WHY the cap
1379          // exists, and HOW to make the formula parsable again.
1380          let warn_fn = || -> Result<()> {
1381            Warn!(
1382              "toobig",
1383              "math",
1384              s!(
1385                "Formula has {} grammar lexemes (> cap {}); skipping the Marpa \
1386                 grammar parse and falling back to the KLUDGE parser (linear \
1387                 OPEN/CLOSE matching — no real semantic/Content-MathML tree). \
1388                 WHY: Marpa's Earley recognizer allocates O(n^2)+ obstack \
1389                 memory in the lexeme count; on inputs this large it exhausts \
1390                 RAM and libmarpa's default_out_of_memory calls abort() — an \
1391                 UNCATCHABLE SIGABRT that kills the whole conversion (the parse \
1392                 runs on a worker thread; there is no Rust panic to catch). Perl \
1393                 LaTeXML (Parse::RecDescent) parses such formulae in bounded \
1394                 memory, so this cap is a Rust-only divergence. TO RAISE/REMOVE \
1395                 IT (set LATEXML_MAX_GRAMMAR_LEXEMES, 0=disable): the blow-up is \
1396                 dominated by grammar AMBIGUITY (chiefly implicit-multiplication / \
1397                 juxtaposition associativity, which makes Earley-set growth \
1398                 super-linear), so tighten those rules to reduce ambiguity, or \
1399                 give the marpa fork a bounded-memory / graceful-OOM recognizer \
1400                 path. Witness: 1706.06621-class explicitly-expanded polynomials.",
1401                lexemes.len(),
1402                cap
1403              )
1404            );
1405            Ok(())
1406          };
1407          warn_fn().ok();
1408          Ok(None)
1409        },
1410        _ => self.parse_lexemes(lexemes, &nodes, document),
1411      };
1412      // Resource fatals must propagate, not be dropped by the if-let below
1413      // (PR #249 review P1-4); other Errs keep the old treated-as-failure
1414      // behavior.
1415      let parse_outcome = match parse_outcome {
1416        Err(e) if matches!(e.target, latexml_core::common::error::ErrorTarget::Timeout) => {
1417          return Err(e);
1418        },
1419        other => other,
1420      };
1421      if let Ok(Some(parse_tree)) = parse_outcome {
1422        //START reparent: the reparenting used to be in `parse_rec` in Perl. Is this a good place?
1423        // Replace the content of XMath with parsed result
1424        // unbindNode followed by (append|replace)Tree (which removes ID's) should be safe
1425        //
1426        // Pre-snapshot the xml:ids that the old subtree exposes. Any of
1427        // these that don't reappear in the new tree below have been
1428        // "orphaned" by the parse — pre-existing XMRefs pointing at them
1429        // would go dangling. Mirrors Perl `MathParser.pm` LOSTNODES
1430        // tracking for the tree-replacement path (the analogue of
1431        // `ReplacedBy(lost, undef)` cases).
1432        let mut pre_replacement_ids: Vec<String> = Vec::new();
1433        for child_el in element_nodes(mathnode) {
1434          for descendant in document.findnodes("descendant-or-self::*[@xml:id]", Some(&child_el)) {
1435            if let Some(id) = descendant.get_attribute("xml:id") {
1436              pre_replacement_ids.push(id);
1437            }
1438          }
1439        }
1440        for child_el in element_nodes(mathnode) {
1441          document.unrecord_node_ids(&child_el);
1442        }
1443        for mut node in mathnode.get_child_nodes() {
1444          node.unlink();
1445        }
1446        let new_xml_tree = parse_tree.into_xmath(mathnode, &mut nodes, document)?;
1447        document.append_tree(mathnode, vec![new_xml_tree])?;
1448        // Resolve _xmkey references: match XMRef[@_xmkey] to elements with same _xmkey
1449        resolve_xmkeys(mathnode, document)?;
1450        // Detect orphaned IDs: any pre-snapshot ID that no longer resolves
1451        // in the document idstore. Record each as a LOSTNODE with an empty
1452        // replacement (sentinel — the top-level cleanup interprets empty
1453        // replacement as "drop the XMRef").
1454        for pre_id in &pre_replacement_ids {
1455          if document.lookup_id(pre_id).is_none() {
1456            crate::data::record_replacement(pre_id, "__LOSTNODE__");
1457          }
1458        }
1459        let result = element_nodes(mathnode).remove(0);
1460        //END reparent.
1461        if !punct_nodes.is_empty() {
1462          Ok(Some(self.wrap_with_punct(
1463            result,
1464            punct_nodes,
1465            mathnode,
1466            document,
1467          )?))
1468        } else {
1469          Ok(Some(result))
1470        }
1471      } else {
1472        // Parse failed; put punct nodes back in mathnode so they remain visible
1473        for mut p in punct_nodes {
1474          mathnode.add_child(&mut p).ok();
1475        }
1476        Ok(None)
1477      }
1478    }
1479  }

1481  /// Wrap `result` in an XMDual so trailing punctuation appears in the presentation
1482  /// branch only, matching Perl's `if (@punct) { $result = ['ltx:XMDual', ...] }`.
1483  fn wrap_with_punct(
1484    &self,
1485    mut result: Node,
1486    punct_nodes: Vec<Node>,
1487    mathnode: &mut Node,
1488    document: &mut Document,
1489  ) -> Result<Node> {
1490    // Assign an id to result while it's still in the tree (id generation needs ancestors)
1491    document.generate_id(&mut result, "")?;
1492    let id = result
1493      .get_attribute_ns("id", XML_NS)
1494      .unwrap_or_else(|| result.get_attribute("id").unwrap_or_default());
1495    // Detach result from mathnode so we can re-parent it inside XMWrap
1496    result.unlink();
1497    // Create XMDual in mathnode
1498    let mut dual = document.open_element_at(mathnode, "ltx:XMDual", None, None)?;
1499    // Content branch: XMRef pointing to result
1500    let ref_attrs = {
1501      let mut m = HashMap::default();
1502      m.insert(s!("idref"), id);
1503      Some(m)
1504    };
1505    let mut xmref = document.open_element_at(&mut dual, "ltx:XMRef", ref_attrs, None)?;
1506    document.close_element_at(&mut xmref)?;
1507    // Presentation branch: XMWrap containing result + trailing punct
1508    let mut xmwrap = document.open_element_at(&mut dual, "ltx:XMWrap", None, None)?;
1509    xmwrap.add_child(&mut result).ok();
1510    for mut p in punct_nodes {
1511      xmwrap.add_child(&mut p).ok();
1512    }
1513    document.close_element_at(&mut xmwrap)?;
1514    document.close_element_at(&mut dual)?;
1515    Ok(dual)
1516  }

1547      //

1549      // unambiguous branch counts whole-tree semantic rejection, while
1550      // the ambiguous branch reports per-combo ASF action rejections.
1551      // Do not compare this number across routing modes as a semantic
1552      // invariant.
1553      let mut traverser = crate::asf_traverser::MathTraverser {
1554        actions:      &self.actions,
1555        pragmas:      self.expert_pragmatics.as_slice(),
1556        builder:      &self.builder,
1557        nodes,
1558        document,
1559        pruned_count: 0,
1560      };
1561      let hybrid_result = self.engine.parse_hybrid_with_and_node_limit(
1562        ByteScanner::new(Cursor::new(input)),
1563        (),
1564        &mut traverser,
1565        *HYBRID_AND_NODE_LIMIT,
1566      );
1567      if *PARSE_LEXEMES_DBG {
1568        eprintln!("PARSE_LEXEMES_RECOGNIZED");
1569      }
1570      match hybrid_result {
1571        Ok(HybridParseResult::Unambiguous(mut tree_iter)) => {
1572          // Release `traverser`'s borrow of the recognizer before re-borrowing
1573          // below; `drop` is the intent even though the type has no Drop impl.
1574          #[allow(clippy::drop_non_drop)]
1575          drop(traverser);
1576          record_ambiguity_metric(1, input);
1577          let tree_outcome = if let Some(val) = tree_iter.next() {
1578            match self.actions.get_tree(
1579              self.builder.clone(),
1580              val,
1581              self.expert_pragmatics.as_slice(),
1582              ActionContext { nodes, document },
1583            ) {
1584              Ok(Some(tree)) => ParseOutcome::Accepted(tree),
1585              Ok(None) => ParseOutcome::Empty,
1586              Err(prune_err) => ParseOutcome::Rejected(prune_err.to_string()),
1587            }
1588          } else {
1589            ParseOutcome::Empty
1590          };
1591          debug_assert!(
1592            tree_iter.next().is_none(),
1593            "hybrid unambiguous branch should expose exactly one raw tree"
1594          );
1595          self.audit_hybrid_unambiguous_parity(input, nodes, document, &tree_outcome);
1596          match tree_outcome {
1597            ParseOutcome::Accepted(tree) => {
1598              ok_trees += 1;
1599              parses.push(tree);
1600            },
1601            ParseOutcome::Empty => {},
1602            ParseOutcome::Rejected(msg) => {
1603              pruned_trees += 1;
1604              if *PARSE_PRUNE_REASONS {
1605                let trimmed = msg.chars().take(140).collect::<String>();
1606                *prune_reasons.entry(trimmed).or_insert(0) += 1;
1607              }
1608            },
1609            ParseOutcome::Multiple(_) => unreachable!("tree path cannot produce multiple outcomes"),
1610          }
1611        },
1612        Ok(HybridParseResult::Ambiguous(alts, _state)) => {
1613          record_ambiguity_metric(2, input);
1614          pruned_trees = traverser.pruned_count;
1615          if std::env::var("LATEXML_MARPA_ASF_AUDIT").is_ok() {
1616            eprintln!(
1617              "ASF_AUDIT: peak returned {} alternatives ({} Some, {} None), pruned={}",
1618              alts.len(),
1619              alts.iter().filter(|a| a.is_some()).count(),
1620              alts.iter().filter(|a| a.is_none()).count(),
1621              pruned_trees,
1622            );
1623          }
1624          let alts_vec =
1625            std::rc::Rc::try_unwrap(alts).unwrap_or_else(|rc| (*rc).clone());
1626          for tree in alts_vec.into_iter().flatten() {
1627            if parses.contains(&tree) {
1628              deduped += 1;
1629            } else {
1630              ok_trees += 1;
1631              parses.push(tree);
1632            }
1633          }
1634        },
1635        Ok(HybridParseResult::AmbiguousTree(tree_iter, stats)) => {
1636          // Large-bocage fallback: codex's marpa commit 5f6a19e routes
1637          // bocages whose `and_node_count` exceeds `*HYBRID_AND_NODE_LIMIT`
1638          // through the ordinary `Tree` iterator instead of constructing
1639          // the ASF. Walk it with the same 6 convergence caps that the
1640          // legacy path uses — those caps are exactly what makes
1641          // Tree-iteration tractable on highly-ambiguous forests.
1642          // Release `traverser`'s borrow before re-borrowing below.
1643          #[allow(clippy::drop_non_drop)]
1644          drop(traverser);
1645          record_ambiguity_metric(2, input);
1646          if std::env::var("LATEXML_MARPA_ASF_AUDIT").is_ok() {
1647            eprintln!(
1648              "HYBRID_AUDIT: large-bocage fallback fired (or_nodes={}, and_nodes={}, max_per_or={}) for: {}",
1649              stats.or_node_count,
1650              stats.and_node_count,
1651              stats.max_and_nodes_per_or_node,
1652              input.chars().take(120).collect::<String>(),
1653            );
1654          }
1655          let max_trees = 5000;
1656          let max_time = std::time::Duration::from_secs(30);
1657          let max_consecutive_dupes = 16;
1658          let converge_budget = std::time::Duration::from_millis(200);
1659          let pruned_only_time_budget = std::time::Duration::from_secs(2);
1660          let pruned_only_count_threshold: usize = 200;
1661          let max_unique = 10;
1662          for val in tree_iter {
1663            if ok_trees + pruned_trees >= max_trees || start.elapsed() > max_time {
1664              break;
1665            }
1666            if consecutive_dupes >= max_consecutive_dupes && !parses.is_empty() {
1667              break;
1668            }
1669            if parses.len() >= max_unique {
1670              break;
1671            }
1672            if !parses.is_empty() && start.elapsed() > converge_budget {
1673              break;
1674            }
1675            if parses.is_empty()
1676              && pruned_trees > pruned_only_count_threshold
1677              && start.elapsed() > pruned_only_time_budget
1678            {
1679              break;
1680            }
1681            match self.actions.get_tree(
1682              self.builder.clone(),
1683              val,
1684              self.expert_pragmatics.as_slice(),
1685              ActionContext { nodes, document },
1686            ) {
1687              Ok(Some(tree)) => {
1688                ok_trees += 1;
1689                if parses.contains(&tree) {
1690                  deduped += 1;
1691                  consecutive_dupes += 1;
1692                } else {
1693                  parses.push(tree);
1694                  consecutive_dupes /= 2;
1695                }
1696              },
1697              Ok(None) => {},
1698              Err(prune_err) => {
1699                pruned_trees += 1;
1700                if *PARSE_PRUNE_REASONS {
1701                  let msg = prune_err.to_string();
1702                  let trimmed = msg.chars().take(140).collect::<String>();
1703                  *prune_reasons.entry(trimmed).or_insert(0) += 1;
1704                }
1705                if !parses.is_empty() {
1706                  consecutive_dupes += 1;
1707                }
1708              },
1709            }
1710          }
1711        },
1712        Err(e) => {
1713          // A Timeout-class error here is a RESOURCE fatal raised inside a
1714          // semantics action (e.g. the gullet cycle guard during
1715          // `create_xmrefs`→`get_xmarg_id` expansion), not a semantic parse
1716          // rejection. Swallowing it produced "phantom fatals": the `Fatal!`
1717          // macro had already counted it (final summary says "1 fatal
1718          // error") but no `Fatal:` line ever reached the log, and parsing
1719          // ground on formula-by-formula. Surface it and ABORT math parsing
1720          // (bounded + honest; witness math0402448). Non-resource errors
1721          // keep the old reject-and-continue behaviour.
1722          if let Some(err) = latexml_core::common::error::take_last_resource_fatal()
1723            .or_else(|| resource_fatal_from_message(&e.to_string()))
1724          {
1725            // No log_fatal here: the Err now PROPAGATES through
1726            // parse_lexemes/parse_single/parse_math to the converter's outer
1727            // handler, which logs the Fatal: line exactly once (P1-4).
1728            return Err(err);
1729          }
1730          if std::env::var("LATEXML_MARPA_ASF_AUDIT").is_ok() {
1731            eprintln!("HYBRID_AUDIT: parse_hybrid Err: {e}");
1732          }
1733        },
1734      }
1735    } else if *PARSE_VIA_ASF {
1736      // ASF path: one post-order memoized callback per glade.
1737      // `MathTraverser` accumulates all alternative XM parse trees
1738      // (including any pragmas-pruned ones counted as `pruned_count`)
1739      // in a single sweep; no Tree-iteration loop, no convergence
1740      // bandages. See `latexml_math_parser/src/asf_traverser.rs` and
1741      // docs/MATH_PARSER_AND_ASF.md.
1742      let mut traverser = crate::asf_traverser::MathTraverser {
1743        actions:      &self.actions,
1744        pragmas:      self.expert_pragmatics.as_slice(),
1745        builder:      &self.builder,
1746        nodes,
1747        document,
1748        pruned_count: 0,
1749      };
1750      let asf_result = self
1751        .engine
1752        .parse_and_traverse_forest(ByteScanner::new(Cursor::new(input)), (), &mut traverser);
1753      if *PARSE_LEXEMES_DBG {
1754        eprintln!("PARSE_LEXEMES_RECOGNIZED");
1755      }
1756      match asf_result {
1757        Ok((alts, _state)) => {
1758          pruned_trees = traverser.pruned_count;
1759          if std::env::var("LATEXML_MARPA_ASF_AUDIT").is_ok() {
1760            eprintln!(
1761              "ASF_AUDIT: peak returned {} alternatives ({} Some, {} None), pruned={}",
1762              alts.len(),
1763              alts.iter().filter(|a| a.is_some()).count(),
1764              alts.iter().filter(|a| a.is_none()).count(),
1765              pruned_trees,
1766            );
1767          }
1768          // The order-alignment with legacy tree-iteration is no
1769          // longer needed: the FencedLetters Dual pragma now prunes
1770          // the wrong-direction parses at validation time, and adding
1771          // `alts.reverse()` measured WORSE on the test suite
1772          // (1281 vs 1284). Leave the bocage-natural order.
1773          // End-of-traversal: take ownership of the peak Vec.
1774          // `Rc::try_unwrap` succeeds unless the marpa driver still
1775          // holds a copy in its cache (it returned its own clone via
1776          // `cache.insert(peak, output.clone())`), so fall back to
1777          // deep clone on the cold path.
1778          let alts_vec =
1779            std::rc::Rc::try_unwrap(alts).unwrap_or_else(|rc| (*rc).clone());
1780          for tree in alts_vec.into_iter().flatten() {
1781            if parses.contains(&tree) {
1782              deduped += 1;
1783            } else {
1784              ok_trees += 1;
1785              parses.push(tree);
1786            }
1787          }
1788        },
1789        Err(e) => {
1790          // Same resource-fatal surfacing as the hybrid path above — see the
1791          // comment there (witness math0402448's phantom fatal).
1792          if let Some(err) = latexml_core::common::error::take_last_resource_fatal()
1793            .or_else(|| resource_fatal_from_message(&e.to_string()))
1794          {
1795            // No log_fatal here: the Err now PROPAGATES through
1796            // parse_lexemes/parse_single/parse_math to the converter's outer
1797            // handler, which logs the Fatal: line exactly once (P1-4).
1798            return Err(err);
1799          }
1800          if std::env::var("LATEXML_MARPA_ASF_AUDIT").is_ok() {
1801            eprintln!("ASF_AUDIT: traverse_forest Err: {e}");
1802          }
1803        },
1804      }
1805    } else {
1806      // Legacy path: Tree iteration with the 6 convergence caps.
1807      let parse_result = self
1808        .engine
1809        .run_recognizer(ByteScanner::new(Cursor::new(input)))?;
1810      if *PARSE_LEXEMES_DBG {
1811        eprintln!("PARSE_LEXEMES_RECOGNIZED");
1812      }
1813      // The six caps below (max_trees, max_time, max_consecutive_dupes,
1814      // converge_budget, pruned_only_time_budget+_count_threshold,
1815      // max_unique) exist because Tree-iteration evaluates per-tree
1816      // actions O(trees × occurrences) times — defensive bandages
1817      // against the paradigm cost.
1818      //
1819      // 2026-05-18: With HYBRID as the default
1820      // (`PARSE_VIA_HYBRID`), this branch is reached only when the
1821      // user explicitly sets `LATEXML_MARPA_LEGACY=1` — an
1822      // engine-divergence debugging escape hatch. The caps stay
1823      // because their *original* protective role is exactly what
1824      // makes the escape hatch usable on real ambiguous inputs
1825      // (without them legacy would hang on grammars like the
1826      // `27-trees-from-scriptpos` family in math). Removing them
1827      // would silently break the debug path.
1828      //
1829      // For the production-path concern (long-time tree-iteration
1830      // cost), HYBRID's `metric == 2 → ASF` routing already
1831      // sidesteps these caps via per-glade memoization. See
1832      // docs/MATH_PARSER_AND_ASF.md.
1833      let max_trees = 5000; // Hard limit on parse tree enumeration
1834      let max_time = std::time::Duration::from_secs(30); // 30 second timeout
1835      // Convergence: if we've seen enough consecutive duplicates without
1836      // a new unique tree, the grammar ambiguity is purely structural
1837      // (script attachment ordering). Stop early.
1838      let max_consecutive_dupes = 16;
1839      // Time-budget convergence: once we have unique parses, stop after
1840      // this budget. For formulas where all trees are pruned (no unique
1841      // parse yet), use a longer budget before giving up.
1842      let converge_budget = std::time::Duration::from_millis(200);
1843      // Pruned-only fast-fail: if we've spent significant time and seen
1844      // many trees without finding a single semantic-acceptable parse,
1845      // the grammar is exploring a combinatorial dead end. Bail before
1846      // exhausting `max_time` (30s). Empirical: 0804.1730 case had 4536
1847      // pruned trees over 28s before timeout. After 2s with >200 prunes
1848      // the marginal probability of finding a unique drops sharply.
1849      let pruned_only_time_budget = std::time::Duration::from_secs(2);
1850      let pruned_only_count_threshold: usize = 200;
1851      // Unique-tree cap: the pragmatics/selection step only needs a handful
1852      // of distinct parses. Beyond this, additional unique trees are almost
1853      // always script-attachment ordering variants that don't improve the
1854      // final selected parse. Avoids enumerating 60+ trees for expressions
1855      // like `{}^4{}_{12}C^{5+}` where the grammar produces 27+ unique
1856      // trees from different pre/post script nesting orders.
1857      let max_unique = 10;
1858      for val in parse_result {
1859        // Truncate if too many trees or too much time
1860        if ok_trees + pruned_trees >= max_trees || start.elapsed() > max_time {
1861          break;
1862        }
1863        // Early convergence: stop if we keep seeing only duplicates.
1864        // The grammar produces 2^N duplicates from script attachment ordering.
1865        // Once we've found all unique parses, every new tree is a duplicate.
1866        if consecutive_dupes >= max_consecutive_dupes && !parses.is_empty() {
1867          break;
1868        }
1869        // Unique-tree cap: stop once we have enough distinct parses.
1870        if parses.len() >= max_unique {
1871          break;
1872        }
1873        // Time-budget convergence: if we have unique parses and have spent
1874        // >200ms, stop — the remaining trees are overwhelmingly duplicates.
1875        if !parses.is_empty() && start.elapsed() > converge_budget {
1876          break;
1877        }
1878        // Pruned-only fast-fail: bail when we have NO unique parses, have
1879        // already enumerated many trees, and have burned through the
1880        // pruned-only budget. Without this, the loop runs to `max_time`
1881        // (30s) on pathological multi-clause RELOP-list formulae where
1882        // every grammar derivation is semantically pruned (e.g.
1883        // 0804.1730 had 4536 enumerated → 0 unique → 28.29s).
1884        if parses.is_empty()
1885          && pruned_trees > pruned_only_count_threshold
1886          && start.elapsed() > pruned_only_time_budget
1887        {
1888          break;
1889        }
1890        // Note: we intentionally do NOT abort when no parse has been found
1891        // even after extended time — valid parses can appear late in the
1892        // enumeration (tree #3585 of 3713 for complex multi-equation formulas).
1893        match self.actions.get_tree(
1894          self.builder.clone(),
1895          val,
1896          self.expert_pragmatics.as_slice(),
1897          ActionContext { nodes, document },
1898        ) {
1899          Ok(tree_opt) => {
1900            if let Some(tree) = tree_opt {
1901              ok_trees += 1;
1902              // Online deduplication: check if this tree is already in our unique set
1903              if parses.contains(&tree) {
1904                deduped += 1;
1905                consecutive_dupes += 1;
1906              } else {
1907                parses.push(tree);
1908                // Half-decay (not full reset) on new unique. This lets us bail
1909                // on cases where uniques are sparse among a sea of dupes/prunes —
1910                // e.g. sin[XY] produces 10 unique parses among 1022 grammar
1911                // derivations; without decay the unique trees keep resetting the
1912                // dupe counter and we never converge. Half-decay means each new
1913                // unique halves the accumulated dupe budget instead of clearing it.
1914                consecutive_dupes /= 2;
1915              }
1916            }
1917          },
1918          Err(prune_err) => {
1919            pruned_trees += 1;
1920            if *PARSE_PRUNE_REASONS {
1921              let msg = prune_err.to_string();
1922              let trimmed = msg.chars().take(140).collect::<String>();
1923              *prune_reasons.entry(trimmed).or_insert(0) += 1;
1924            }
1925            // Pruned trees also count toward convergence if we have unique parses
1926            if !parses.is_empty() {
1927              consecutive_dupes += 1;
1928            }
1929          },
1930        }
1931      }
1932    }
1933
1934    // Diagnostic: dump parse ORDER when LATEXML_PARSE_DUMP_ORDER=1.
1935    if std::env::var("LATEXML_PARSE_DUMP_ORDER").is_ok() && parses.len() > 1 {
1936      eprintln!("PARSE_ORDER: {} unique for {}", parses.len(), input.trim());
1937      for (i, p) in parses.iter().enumerate() {
1938        eprintln!("  [{}] {}", i, p.text_summary());
1939      }
1940    }
1941    // Store count for \ltx@count@parses diagnostic macro
1942    // Use post-dedup count (distinct semantic trees), not raw grammar count
1943    self.last_parsetrees_count = parses.len();
1944
1945    if ok_trees + pruned_trees > 10 {
1946      // Diagnostic only — high ambiguity isn't a Perl-side Error in
1947      // MathParser.pm; the warn target uses a Rust-internal class.
1948      log_math_warn!(
1949        "ambiguous", "math",
1950        "Ambiguous math: {} enumerated ({} semantic, {} pruned, {} deduped→{} unique) in {:?} for: {}",
1951        ok_trees + pruned_trees + deduped,
1952        ok_trees + deduped,
1953        pruned_trees,
1954        deduped,
1955        parses.len(),
1956        start.elapsed(),
1957        input.trim()
1958      );
1959    }
1960    // Diagnostic: report parse counts when LATEXML_PARSE_AUDIT is set.
1961    // Useful for identifying grammar ambiguity hotspots across the test suite.
1962    // Usage: LATEXML_PARSE_AUDIT=1 cargo test --test 56_ams -- mathtools_test --nocapture
1963    if *PARSE_AUDIT && (ok_trees + pruned_trees > 1 || start.elapsed().as_millis() > 50) {
1964      eprintln!(
1965        "PARSE_AUDIT: {} trees ({} ok, {} pruned, {} dedup→{} unique) in {:?} | {}",
1966        ok_trees + pruned_trees + deduped,
1967        ok_trees + deduped,
1968        pruned_trees,
1969        deduped,
1970        parses.len(),
1971        start.elapsed(),
1972        input.trim().chars().take(200).collect::<String>()
1973      );
1974    }
1975    // Dump top pragma-rejection reasons when no parse survived and the
1976    // diagnostic env var is set. Helps locate over-aggressive pragmas
1977    // that reject all candidates in document context.
1978    if *PARSE_PRUNE_REASONS && parses.is_empty() && pruned_trees > 0 {
1979      let mut top: Vec<(&String, &usize)> = prune_reasons.iter().collect();
1980      top.sort_by(|a, b| b.1.cmp(a.1));
1981      eprintln!(
1982        "PARSE_PRUNE_REASONS: {} pruned, {} distinct, top-5: | input: {}",
1983        pruned_trees,
1984        prune_reasons.len(),
1985        input.trim().chars().take(120).collect::<String>()
1986      );
1987      for (msg, count) in top.iter().take(5) {
1988        eprintln!("  {count:>5}× {msg}");
1989      }
1990    }
1991
1992    match parses.len() {
1993      0 => Err("Failed to find any parse".into()),
1994      1 => Ok(parses.into_iter().next().unwrap()),
1995      _more => {
1996        // Loop over the various soft pruning algorithms available, until we have 1 tree
1997        let mut reduced_forest = XM::Choices(parses);
1998        for pragma in self.student_pragmatics.iter() {
1999          reduced_forest = reduced_forest.soft_prune_choices(*pragma);
2000          match reduced_forest {
2001            XM::Choices(ref trees) if trees.len() <= 1 => break,
2002            _ => {},
2003          };
2004        }
2005        // Multi-tree pragma: `(a, b)` and `[a, b]` (and half-open
2006        // variants) standalone should default to the named-interval
2007        // interpretation (`open-interval`, `closed-interval`, etc.)
2008        // rather than the generic `vector@(2)` or `delimited-XY@(...)`
2009        // wrapper. The math-parser grammar admits both; legacy
2010        // tree-iter happens to pick the named interval, ASF
2011        // Cartesian-product happens to pick the wrapper. This pragma
2012        // drops the wrapper parses from the forest root iff a named-
2013        // interval alternative also exists. Narrow scope: only the
2014        // math root, only 2-element, only when both alternatives are
2015        // present.
2016        reduced_forest = reduced_forest.prefer_named_interval_at_root();
2017
2018        // Multi-tree pragma: prune `set@(set@(…))`, `vector@(vector@(…))`
2019        // etc. — redundant self-wrapping at the math root — when a
2020        // non-self-wrapping alternative exists in the forest.
2021        reduced_forest = reduced_forest.prefer_non_self_wrapping_root();
2022
2023        // Multi-tree pragma: drop `multirelation@(..., absent, ...)`
2024        // chains when a non-multirelation alternative exists.
2025        // Handles `x>=0` parsed as `(x > absent = 0)` vs `>=@(x, 0)`.
2026        reduced_forest = reduced_forest.prefer_combined_relop_over_multirelation_with_absent();
2027
2028        // `prefer_zero_absent_when_available` retired 2026-05-19
2029        // (ASF item 5 Phase 2): the pragma had no dedicated test
2030        // witness. Its conceptual target (`<x|y>` inner-product
2031        // bra-ket) is already produced as `inner-product@(x, y)`
2032        // by the qm-specific pragmas + the angle-bracket grammar
2033        // rules. After the modified_term Phase 1 landing, disabling
2034        // the pragma left `cargo test --tests` = 1328/0/0 on both
2035        // HYBRID and ASF — the pragma is structurally redundant.
2036        // See commit history (the new commit) and
2037        // `docs/MATH_PARSER_ASF_TIEBREAKING.md`.
2038
2039        // Multi-tree pragma: a *specific* QM semantic
2040        // (`quantum-operator-product@(a, f, b)`, `inner-product@(a, b)`)
2041        // beats a generic `delimited-⟨⟩` wrapper for the same input.
2042        // Specific semantic recognition reflects author intent more
2043        // closely than a structural fence wrapper. Must run BEFORE
2044        // `prefer_more_delimited_wrappers`, because the latter would
2045        // otherwise filter out the qm_bracket candidate (which has
2046        // zero `delimited-` wrappers).
2047        reduced_forest = reduced_forest.prefer_qm_specific_semantics();
2048
2049        // Multi-tree pragma: prefer candidates with FEWER nested
2050        // same-meaning fences (`norm` inside `norm`, etc.). Encodes
2051        // the mathematician's greedy left-to-right bar-pairing
2052        // instinct. Resolves `||x||a||y||` → `‖x‖ · a · ‖y‖`
2053        // (siblings) over `‖x · ‖a‖ · y‖` (nested).
2054        reduced_forest = reduced_forest.prefer_fewer_nested_same_fences();
2055
2056        // Multi-tree pragma: prefer candidates that recognized
2057        // fenced sub-expressions as `delimited-X@(…)` (angle, paren,
2058        // bracket, vertbar) over flat `formulae@(…)` / multirelation
2059        // chains. Fires only when at least one candidate has a
2060        // `delimited-X` Apply AND another has fewer/zero. Resolves
2061        // `2<x,y>=z`, `0<<a,b>>1`, and bra-ket-style angle fences.
2062        // See docs/MATH_PARSER_ASF_TIEBREAKING.md § "What landed".
2063        reduced_forest = reduced_forest.prefer_more_delimited_wrappers();
2064
2065        // Multi-tree pragma: prefer candidates with MORE
2066        // `Apply(letter, [vertbar-fenced])` patterns — captures the
2067        // QM-style bra-ket reading `<a|f|b>` → `a@(|f|) * b` over
2068        // the multiplicative `a * |f| * b`. No-op outside QM/angle
2069        // contexts because K-12 readings don't admit the pattern.
2070        reduced_forest = reduced_forest.prefer_more_letter_at_vertbar();
2071
2072        // Multi-tree pragma: when at least one candidate has zero
2073        // `MODIFIEROP:conditional` Applies, drop those with more.
2074        // The conditional reading of `vertbar_modifier` recurses
2075        // through itself in the grammar, producing
2076        // `conditional@(a, conditional@(a, …))` for inputs like
2077        // `a|a|+b|b|+c|c|`. The K-12 algebra reading
2078        // `a*|a|+b*|b|+c*|c|` has no conditionals; this pragma
2079        // gives it preference. Set-builder / probability cases
2080        // where no algebraic alternative exists are unaffected.
2081        reduced_forest = reduced_forest.prefer_fewer_conditionals();
2082
2083        // Multi-tree pragma: when at least one candidate roots at
2084        // an additive operator (`+`/`-`), prefer it over candidates
2085        // with multiplicative root. K-12 reading: `a*|a|+b*|b|+c*|c|`
2086        // has `+` outermost, not `a * (one giant absolute-value)`.
2087        reduced_forest = reduced_forest.prefer_root_addition_over_outer_multiplication();
2088
2089        // Multi-tree pragma: among addition-rooted candidates,
2090        // prefer the widest n-ary `+` chain (`+@(a, b, c)` over
2091        // `+@(a, b+c)`). Resolves the inner bar-pairing of
2092        // `a|a|+b|b|+c|c|` — the 3-arg chain has every `|.|`
2093        // as a sibling absolute-value.
2094        reduced_forest = reduced_forest.prefer_wider_addition_root();
2095
2096
2097        // Multi-tree shape pragmas (`prefer_fewer_absent`,
2098        // `prefer_smaller_tree`) exist on `XM` but are
2099        // **deliberately not wired in by default**. Both proved
2100        // too coarse to apply universally — see
2101        // docs/MATH_PARSER_ASF_TIEBREAKING.md § "Empirical results":
2102        //
2103        // * `prefer_fewer_absent`: on the legacy path it is
2104        //   essentially neutral (1300/1 vs 1301 baseline), but on
2105        //   the ASF path it costs 9 tests because ASF surfaces
2106        //   absent-free parses that are semantically WRONG (e.g.
2107        //   `<a|f|b>` parsed as a multiplication chain without the
2108        //   bra-ket interpretation). The deeper issue: `absent` has
2109        //   two valid uses (structural filler vs missing-operand
2110        //   fallback) and the pragma can't tell them apart.
2111        //
2112        // * `prefer_smaller_tree`: 3 improvements vs 6 regressions
2113        //   on legacy. "Smaller is better" works for `norm@(a)` vs
2114        //   `abs(abs(a))` but not for `annotated@(x, expr)` vs
2115        //   `x@(expr)` where the larger tree is the intended one.
2116        //
2117        // The correct path forward is case-by-case semantic
2118        // pragmas (named-operator recognition, domain-typed
2119        // wrappers), not universal shape ranking. See the
2120        // research-notes doc for the design discussion.
2121        Ok(reduced_forest)
2122      },
2123    }
2124  }

2590    } else {

2592      let textrec_op = textrec(op, None, None, document);
2593      let rec_form = if args.len() == 1 {
2594        // unless a single arg; then prefix.
2595        textrec_op + " " + &textrec(&args[0], Some(*bp), Some(name), document)
2596      } else {
2597        args
2598          .iter()
2599          .map(|a| textrec(a, Some(*bp), Some(name), document))
2600          .collect::<Vec<_>>()
2601          .join(&(" ".to_string() + &textrec_op + " "))
2602      };
2603      (*bp, rec_form)
2604    }
2605  } else if role == "POSTFIX" {
2606    if args.is_empty() {
2607      (10000, textrec(op, None, None, document))
2608    } else {
2609      (
2610        10000,
2611        textrec(&args[0], Some(10000), Some(name), document) + &textrec(op, None, None, document),
2612      )
2613    }
2614  } else if name == "multirelation" {
2615    let joined = args
2616      .iter()
2617      .map(|a| textrec(a, Some(2), Some(name), document))
2618      .collect::<Vec<_>>()
2619      .join(" ");
2620    (2, joined)
2621  } else {
2622    (
2623      500,
2624      textrec(op, Some(10000), Some(name), document)
2625        + "@("
2626        + &args
2627          .iter()
2628          .map(|a| textrec(a, None, None, document))
2629          .collect::<Vec<_>>()
2630          .join(", ")
2631        + ")",
2632    )
2633  }
2634}

2660//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
2661// Cute! Were it NOT for Sub/Superscripts, the whole parsing process only
2662// builds a new superstructure around the sequence of token nodes in the input.
2663// Thus, any internal structure is unchanged.
2664//  They get re-parented, but if the parse fails, we've only got to put them
2665// BACK into the original node, to recover the original arrangment!!!
2666// Thus, we don't have to clone, and deal with namespace duplication.
2667// ...
2668// EXCEPT, as I said, for sub/superscripts!!!!
2669//
2670
2671//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
2672// Constructors used in grammar
2673// All the tree construction in the grammar should come through these
2674// operations. We avoid mucking with the actual XML nodes (both to avoid
2675// modifying the original tree until we have a successful parse, and to avoid
2676// XML::LibXML cloning nightmares) We are converting XML nodes to array
2677// representation: [$tag, {%attr},@children] This means any inspection of
2678// nodes has to recognize that  * node may be in XML vs ARRAY representation
2679// * node may be an XMRef to another node whose properties are the ones we
2680// should use.
2681//
2682// Also, when we are examining a node's properties (roles, fences, script
2683// positioning, etc) we should be careful to check for XMRef indirection and
2684// examine the properties of the node that was referred to.
2685// HOWEVER, we should construct our parse tree using (a clone of) the XMRef
2686// node, rather than (a clone of) the referred to node, so as to preserve
2687// identity.
2688//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
2689// We're currently keeping the id's on the nodes as they get cloned,
2690// since they'll (maybe) replace the unparsed nodes.
2691// However, if we consider multiple parses or preserving both parsed & unparsed,
2692// we may have to do some adaptation and id shifting.
2693// ================================================================================
2694
2695// ================================================================================
2696// Low-level accessors
2697
2698// The following accessors work on both the LibXML and ARRAY representations
2699// but they do NOT automatically dereference XMRef!
2700pub fn p_get_value(node: &Node) -> String {
2701  let node_type = node.get_type();
2702  if node_type == Some(NodeType::ElementNode) {
2703    let x = node.get_content();
2704    if !x.is_empty() {
2705      // get content, or fall back to name
2706      x
2707    } else {
2708      node.get_attribute("name").unwrap_or_default()
2709    }
2710  } else {
2711    node.get_content()
2712  }
2713}

2717pub fn realize_xmnode<'a>(node: &'a Node, document: &'a Document) -> Cow<'a, Node> {
2718  if with_node_qname(node, |name| name == "ltx:XMRef") {
2719    if let Some(idref) = node.get_attribute("idref") {
2720      // NOTE: this intentionally uses the LIVE `document.lookup_id`, NOT the
2721      // frozen `MATH_IDSTORE` snapshot path (`data::resolve_xmref`) that
2722      // `get_grammatical_role` uses. Routing this through `resolve_xmref` so
2723      // that more refs resolve mid-parse looked like it would silence the
2724      // benign transient `expected:id` warnings (refs that miss the live
2725      // idstore during a sibling element's reinstall but exist in the final
2726      // doc) — but it DUPLICATES content (`\choose` → "a + ba + b binomial
2727      // c + dc + d", regressing choose/declare/sampler): callers here rely on
2728      // an unresolved ref returning the XMRef itself. The warnings are benign
2729      // (WARN-level, targets resolve in the final tree); do not "fix" them by
2730      // swapping the resolver. See docs/EXPECTED_ID_XMREF_DESIGN_2026-06-08.md.
2731      // Can it happen that the target is itself an XMRef? Then recurse.
2732      if let Some(realnode) = document.lookup_id(&idref) {
2733        return Cow::Borrowed(realnode);
2734      } else {
2735        let message = s!("Cannot find a node with xml:id='{}'", idref);
2736        // TODO:
2737        // LaTeXML::MathParser::IDREFS{$idref}
2738        // ? "Previously bound to " .
2739        // ToString($LaTeXML::MathParser::IDREFS{$idref})           : ()));
2740        // Perl Document.pm L1553: Warn, not Error (missing XMRef targets are common)
2741        let warn_fn = || -> Result<()> {
2742          Warn!("expected", "id", message);
2743          Ok(())
2744        };
2745        warn_fn().ok();
2746        //       return ['ltx:ERROR', {}, "Missing XMRef idref=$idref"]; } }
2747      }
2748    }
2749  }
2750  Cow::Borrowed(node)
2751}

2753/// Resolve _xmkey and _pxmkey references after parse tree installation.
2754/// Matches XMRef[@_xmkey] to elements with same _xmkey, generates xml:id and sets idref.
2755/// _pxmkey is used by parser-generated XMDual (apply_delimited) to avoid
2756/// conflicting with base_xmath's \lx@dual afterConstruct resolver.
2757fn resolve_xmkeys(
2758  mathnode: &Node,
2759  document: &mut Document,
2760) -> std::result::Result<(), Box<dyn std::error::Error>> {
2761  // Resolve both _xmkey (from existing infrastructure) and _pxmkey (from parser)
2762  for attr_name in &["_xmkey", "_pxmkey"] {
2763    let refs = document.findnodes(
2764      &format!("descendant::ltx:XMRef[@{}]", attr_name),
2765      Some(mathnode),
2766    );
2767    for mut ref_node in refs {
2768      let key = match ref_node.get_attribute(attr_name) {
2769        Some(k) => k,
2770        None => continue,
2771      };
2772      // Find the element with matching key (non-XMRef)
2773      // Note: not(self::ltx:XMRef) may fail with namespace prefix in nested
2774      // predicates (XPath bug). Use local-name() check instead.
2775      let xpath = format!(
2776        "descendant::*[@{}='{}'][not(local-name()='XMRef')]",
2777        attr_name, key
2778      );
2779      let targets = document.findnodes(&xpath, Some(mathnode));
2780      if let Some(mut target) = targets.into_iter().next() {
2781        // Ensure target has xml:id
2782        let target_id = if let Some(id) = target
2783          .get_attribute("xml:id")
2784          .or_else(|| target.get_attribute("id"))
2785        {
2786          id
2787        } else {
2788          document.generate_id(&mut target, "")?;
2789          target
2790            .get_attribute("xml:id")
2791            .or_else(|| target.get_attribute("id"))
2792            .unwrap_or_default()
2793        };
2794        if !target_id.is_empty() {
2795          document.set_attribute(&mut ref_node, "idref", &target_id)?;
2796        }
2797      }
2798      let _ = ref_node.remove_attribute(attr_name);
2799    }
2800    // Clean up from non-ref elements
2801    for mut node in document.findnodes(&format!("descendant::*[@{}]", attr_name), Some(mathnode)) {
2802      let _ = node.remove_attribute(attr_name);
2803    }
2804  }
2805  Ok(())
2806}
