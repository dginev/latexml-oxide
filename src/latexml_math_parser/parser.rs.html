166  /// Reset the marpa engine after a failed parse.
167  /// Creates a new engine and runs a trivial parse to advance past the
168  /// precompute step (grammar is already precomputed, can't do it again).
169  ///
170  /// Recovery ladder:
171  /// 1. Clone `self.grammar` + trivial parse — cheap, common case.
172  /// 2. Re-clone and retry once — covers transient marpa state hiccups.
173  /// 3. Full `init_grammar()` rebuild — expensive, but some grammar corruption patterns (observed
174  ///    on the `testscripts` regression fixture) only recover after a fresh precompute. The D5
175  ///    "avoid init_grammar fallback" item turned out to describe the ideal, not the floor —
176  ///    legitimate callers still hit step 3.
177  /// 4. Log + keep previous engine. Subsequent `parse_math` / `recognizes` will fail gracefully (0
178  ///    trees, no panic); better than crashing the whole conversion.
179  ///
180  /// The previous implementation called `init_grammar().unwrap()` at
181  /// step 3, which would panic on any Err. The round-17 change removes
182  /// the panic — on total failure we log and keep going.
183  fn reset_engine(&mut self) {
184    if self.try_reset_clone_path().is_ok() {
185      return;
186    }
187    if self.try_reset_clone_path().is_ok() {
188      return;
189    }
190    // Both clone attempts failed — reach for a full rebuild. Expensive,
191    // but the testscripts regression fixture demonstrates that some
192    // corruption patterns only clear after a fresh precompute.
193    match init_grammar() {
194      Ok((grammar, _actions, _builder)) => {
195        let thin_grammar = grammar.unwrap();
196        self.grammar = thin_grammar.clone();
197        let mut fresh_engine = Parser::with_grammar(thin_grammar);
198        let _ = fresh_engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 ")));
199        self.engine = fresh_engine;
200      },
201      Err(e) => {
202        log::warn!(
203          "math parser: init_grammar fallback failed ({}) — leaving engine in last known state",
204          e
205        );
206      },
207    }
208  }

210  fn try_reset_clone_path(&mut self) -> std::result::Result<(), ()> {
211    let mut engine = Parser::with_grammar(self.grammar.clone());
212    // Run a trivial recognizer to advance state from G (precompute) through
213    // R → B → O → T → GReady. Use "NUMBER:1:1 " which is a valid single-token formula.
214    match engine.run_recognizer(ByteScanner::new(Cursor::new("NUMBER:1:1 "))) {
215      Ok(_) => {
216        self.engine = engine;
217        Ok(())
218      },
219      Err(_) => Err(()),
220    }
221  }

224  pub fn recognizes(&mut self, input: &str) -> bool {
225    let result = self
226      .engine
227      .run_recognizer(ByteScanner::new(Cursor::new(input)));
228    if result.is_err() {
229      self.reset_engine();
230    }
231    result.is_ok()
232  }

234  /// Count the number of raw Marpa grammar trees for a lexeme string (for unit testing).
235  /// This counts ALL derivation trees the grammar produces, before semantic pruning
236  /// and deduplication. Useful for detecting grammar ambiguity.
237  /// Returns the count, or None if recognition fails.
238  pub fn count_raw_trees(&mut self, input: &str) -> Option<usize> {
239    let parse_result = match self
240      .engine
241      .run_recognizer(ByteScanner::new(Cursor::new(input)))
242    {
243      Ok(r) => r,
244      Err(_) => {
245        self.reset_engine();
246        return None;
247      },
248    };
249    let mut count = 0usize;
250    let max = 5000;
251    for _val in parse_result {
252      count += 1;
253      if count >= max {
254        break;
255      }
256    }
257    self.reset_engine();
258    Some(count)
259  }

264    let xmath_selector = "descendant-or-self::ltx:XMath[not(ancestor::ltx:XMath)]";

319          && child_elems.iter().any(|ch| ch.get_name() == "XMArray")
320          && child_elems.iter().any(|ch| ch.get_name() == "XMTok");
321        if has_direct_open || has_array_with_tok {
322          let mut xm = xmath;
323          self.parse_kludge(&mut xm, document);
324        }
325      }

390  // This is a rather peculiar cleanup that needs to be done to manage ids &
391  // idrefs Before parsing, sub/superscripts are represented by an
392  // operator-less XMApp with the script itself as the only child. Ideally,
393  // upon parsing these get merged, combined and disappear into proper XMApp of
394  // an appropriate operator on the base and scripts.  Unless there is a parse
395  // failure, in which case they remain.
396  // The problem comes from various XMDual constructs where an XMRef refers to
397  // the script XMApp. It can occur that one branch parses and the other fails:
398  // This can leave a reference to the script XMApp which no longer exists!
399  // To solve this, we find & replace all references to such script XMApps by an
400  // explicit XMApp with the XMRef refering to the script itself, not the
401  // XMApp. (make sense?)
402  pub fn cleanup_scripts(&mut self, document: &mut Document) {
403    // Perl: cleanupScripts — find script XMApp nodes that may be referenced by XMRef,
404    // and redirect those references to point at the script content (first child) instead
405    // of the XMApp wrapper. This prevents dangling idrefs when the XMApp is consumed
406    // by parsing.
407    static SCRIPT_RE: Lazy<Regex> =
408      Lazy::new(|| Regex::new(r"^(?:PRE|POST|FLOAT)(?:SUB|SUPER)SCRIPT$").unwrap());
409    let apps = document.findnodes(
410      "descendant-or-self::*[@xml:id and contains(@role,'SCRIPT')]",
411      None,
412    );
413    for mut app in apps {
414      let role = match app.get_attribute("role") {
415        Some(r) => r,
416        None => continue,
417      };
418      if !SCRIPT_RE.is_match(&role) {
419        continue;
420      }
421      let appid = match app.get_attribute("xml:id") {
422        Some(id) => id,
423        None => continue,
424      };
425      // Note: using * instead of ltx:XMRef due to XPath namespace issues in nested predicates
426      let refs_xpath = s!("descendant-or-self::*[@idref = '{}']", appid);
427      let refs = document.findnodes(&refs_xpath, None);
428      if refs.is_empty() {
429        continue;
430      }
431      // Get the script content (first child of the XMApp)
432      let mut script = match app.get_first_child() {
433        Some(child) => child,
434        None => continue,
435      };
436      // Ensure the script has an xml:id so we can create XMRef to it
437      if script.get_attribute("xml:id").is_none() {
438        let _ = document.generate_id(&mut script, "");
439      }
440      let script_id = match script.get_attribute("xml:id") {
441        Some(id) => id,
442        None => continue,
443      };
444      // Unregister the app's id and remove the attribute
445      document.unrecord_id(&appid);
446      let _ = app.remove_attribute("xml:id");
447      // Collect app attributes (except xml:id, which we already removed)
448      let attrs: Vec<(String, String)> = app
449        .get_attributes()
450        .into_iter()
451        .filter(|(name, _)| name != "xml:id")
452        .collect();
453      let ns = app.get_namespace();
454      // Replace each ref with an XMApp containing an XMRef to the script
455      for ref_node in refs {
456        // Build the replacement: ltx:XMApp{attrs}[ltx:XMRef{idref=script_id}]
457        let mut new_app = Node::new("XMApp", None, &document.document).unwrap();
458        for (name, value) in &attrs {
459          let _ = new_app.set_attribute(name, value);
460        }
461        if let Some(ref ns) = ns {
462          let _ = new_app.set_namespace(ns);
463        }
464        let mut xmref = Node::new("XMRef", None, &document.document).unwrap();
465        let _ = xmref.set_attribute("idref", &script_id);
466        if let Some(ref ns) = ns {
467          let _ = xmref.set_namespace(ns);
468        }
469        let _ = new_app.add_child(&mut xmref);
470        let _ = document.replace_tree(new_app, ref_node);
471      }
472    }
473  }

617  // by first parsing any structured children, then it's content.
618  fn parse_rec(
619    &mut self,
620    mut node: Node,
621    rule_opt: &str,
622    document: &mut Document,
623  ) -> Result<Option<Node>> {
624    self.parse_children(&node, document)?;
625    // This will only handle 1 layer nesting (successfully?)
626    // Note that this would have been found by the top level xpath,
627    // but we've got to worry about node identity: the parent is being rebuilt
628    for nested in document.findnodes("descendant::ltx:XMath", Some(&node)) {
629      self.parse(nested, document)?;
630    }
631    let tag = get_node_qname(&node);
632    let rule = if let Some(requested_rule) = node.get_attribute("rule") {
633      requested_rule
634    } else {
635      rule_opt.to_owned()
636    };
637
638    if rule == "kludge" {
639      self.parse_kludge(&mut node, document);
640      Ok(None)
641    } else if let Some(mut result) = self.parse_single(&mut node, document, &rule)? {
642      *self.passed.entry_sym(tag).or_insert(0) += 1;
643      if tag == pin!("ltx:XMath") {
644        // Replace the content of XMath with parsed result
645        self.n_parsed += 1;
646        note_progress(&s!("[{}]", self.n_parsed));
647        for el_node in element_nodes(&node) {
648          document.unrecord_node_ids(&el_node);
649        }

651        // be safe
652        for mut child in node.get_child_nodes() {
653          child.unbind_node();
654        }
655        document.append_tree(&mut node, vec![result])?;
656        let mut new_element_children = element_nodes(&node);

665        // add to result, even allowing modification of xml node, since we're committed.
666        // [Annotate converts node to array which messes up clearing the id!]

668        // TODO: Is this needed in a world where `result` is always a `Node` ?

675        // //   delete $attr{_font};
676        // //   $attr{font} = $font->specialize($content); } }
677        // // } else {
678        attr.remove("_font");
679        // TODO: See the namespaced attribute issue for libxml's wrapper:
680        //  https://github.com/KWARC/rust-libxml/issues/104
681        // until then, HACK ids.
682        let newid = attr.remove("id");

700          // if ($isarr) { $$result[1]{$key} = $value; } else {
701          document.set_attribute(&mut result, &key, &value)?;
702          // }

706        }

708        // Danger: the above code replaced the id on the parsed result with the one from XMArg,..

733  // Depth first parsing of XMArg nodes.
734  fn parse_children(&mut self, node: &Node, document: &mut Document) -> Result<()> {
735    for child in element_nodes(node) {
736      let tag = get_node_qname(&child);
737      if tag == pin!("ltx:XMArg") {
738        self.parse_rec(child, "Anything", document)?;
739      } else if tag == pin!("ltx:XMWrap") {
740        if child.has_attribute("_rewrite") {
741          // Rewrite-created XMWrap: parse inner structure (subscripts etc.) but
742          // the XMWrap's role overrides whatever the inner parse produces.
743          // Temporarily remove role so parse_rec doesn't emit start_ROLE/end_ROLE
744          // tokens (the grammar only handles script roles).
745          let saved_role = child.get_attribute("role");
746          let mut c = child.clone();
747          if saved_role.is_some() {
748            c.remove_attribute("role").ok();
749          }
750          if let Some(mut result) = self.parse_rec(child, "Anything", document)? {
751            if let Some(ref role) = saved_role {
752              result.set_attribute("role", role).ok();
753            }
754          } else if let Some(ref role) = saved_role {
755            // Parse failed — XMWrap still in DOM, restore role
756            c.set_attribute("role", role).ok();
757          }
758        } else {
759          self.parse_rec(child, "Anything", document)?;
760        }
761      } else if tag == pin!("ltx:XMApp")
762        || tag == pin!("ltx:XMArray")
763        || tag == pin!("ltx:XMRow")
764        || tag == pin!("ltx:XMCell")
765        || tag == pin!("ltx:XMDual")
766      {
767        self.parse_children(&child, document)?;
768      }
769    }
770    Ok(())
771  }

774  // Low-Level hack parsing when "real" parsing fails;
775  // Two issues cause generated Presentation MathML to be really bad:
776  // (1) not having mrow/mfenced structures wrapping OPEN...CLOSE sequences
777  //     throws off MathML's stretchiness treatment of the fences
778  //     (they're all the same size; big)
779  // (2) un-attached sub/superscripts won't position correctly,
780  //     unless they're attached to something plausible.
781  /// Perl: parse_kludge (MathParser.pm L530-566)
782  /// Stack-based matching of OPEN/CLOSE delimiter pairs + script attachment.
783  fn parse_kludge(&self, mathnode: &mut Node, document: &mut Document) {
784    use crate::data::get_grammatical_role;
785    let children: Vec<Node> = filter_hints(mathnode.get_child_nodes());
786    if children.is_empty() {
787      return;
788    }
789
790    // Build (node, role) pairs — matching Perl's @pairs.
791    let pairs: Vec<(Node, String)> = children
792      .into_iter()
793      .map(|n| {
794        let r = get_grammatical_role(&n);
795        (n, r)
796      })
797      .collect();
798
799    // Perl: @stack = ([], []) — extra empty level handles unmatched leading CLOSEs.
800    let mut stack: Vec<Vec<(Node, String)>> = vec![vec![], vec![]];
801    let mut iter = pairs.into_iter().peekable();
802
803    while iter.peek().is_some() || stack.len() > 1 {
804      let pair_opt = iter.next();
805      let role = pair_opt
806        .as_ref()
807        .map(|(_, r)| r.as_str())
808        .unwrap_or("CLOSE");
809      match role {
810        "OPEN" => {
811          let pair = pair_opt.unwrap();
812          stack.insert(0, vec![pair]);
813        },
814        "CLOSE" => {
815          let mut row = stack.remove(0);
816          if let Some(pair) = pair_opt {
817            row.push(pair);
818          }
819          // Perl L546: handle scripts within this fenced group
820          let kludged = kludge_scripts_rec(row, document);
821          // Wrap if > 1 node, give role FENCED
822          let result = if kludged.len() > 1 {
823            let mut wrap = Node::new("XMWrap", None, document.get_document()).unwrap();
824            for mut n in kludged {
825              n.unlink_node();
826              wrap.add_child(&mut n).ok();
827            }
828            (wrap, "FENCED".to_string())
829          } else {
830            let node = kludged
831              .into_iter()
832              .next()
833              .unwrap_or_else(|| Node::new_text("", &document.document).unwrap());
834            (node, "FENCED".to_string())
835          };
836          if stack.is_empty() {
837            stack.push(vec![]);
838          }
839          stack[0].push(result);
840        },
841        _ => {
842          if let Some(pair) = pair_opt {
843            stack[0].push(pair);
844          }
845        },
846      }
847    }
848
849    // Perl L555: process remaining top-level items through kludge_scripts
850    let final_pairs = stack.into_iter().next().unwrap_or_default();
851    let result_nodes = kludge_scripts_rec(final_pairs, document);
852
853    // Perl L558-563: at top level, unwrap top-level XMWraps (extract children).
854    // Perl iterates all pairs and extracts children of any array-rep XMWrap.
855    let mut replacements = Vec::new();
856    for node in result_nodes {
857      if node.get_name() == "XMWrap" {
858        // Unwrap: extract children
859        for child in node.get_child_nodes() {
860          replacements.push(child);
861        }
862      } else {
863        replacements.push(node);
864      }
865    }
866
867    // Rebuild: clear mathnode, then add replacement nodes.
868    document.unrecord_node_ids(mathnode);
869    for mut child in mathnode.get_child_nodes() {
870      child.unbind_node();
871    }
872    for mut node in replacements {
873      node.unlink_node();
874      mathnode.add_child(&mut node).ok();
875    }
876    // D3b: the replacements re-parented above may carry xml:id attrs
877    // whose idstore entries were cleared by the `unrecord_node_ids`
878    // above (which walked mathnode's state before the replacements
879    // were re-inserted). Re-record so `lookup_id` on any
880    // math-generated id resolves — otherwise finalize()'s
881    // `rebuild_idstore_from_dom` is the only recovery path and
882    // downstream XMRef lookups can SIGSEGV via stale cache.
883    let _ = document.record_node_ids(mathnode);
884  }

889  // Convert to textual form for processing by MathGrammar
890  fn parse_single(
891    &mut self,
892    mathnode: &mut Node,
893    document: &mut Document,
894    rule: &str,
895  ) -> Result<Option<Node>> {
896    let mut idx = 0;
897    let mut content_nodes = filter_hints(mathnode.get_child_nodes());
898
899    // Extract trailing PUNCT/PERIOD nodes if rule ends with ',' (Perl: $rule =~ s/,$// )
900    let mut punct_nodes: Vec<Node> = Vec::new();
901    if rule.ends_with(',') {
902      while let Some(last) = content_nodes.last() {
903        let role = last.get_attribute("role").unwrap_or_default();
904        if role == "PUNCT" || role == "PERIOD" {
905          let mut p = content_nodes.pop().unwrap();
906          p.unlink(); // detach from mathnode's children
907          punct_nodes.insert(0, p);
908        } else {
909          break;
910        }
911      }
912    }
913
914    if content_nodes.is_empty() {
915      // Perl MathParser.pm L683: $result = $nodes[0] || Absent()
916      // Empty XMArg/XMWrap: create <XMTok meaning="absent"/>
917      let mut absent_tok = document.open_element_at(mathnode, "ltx:XMTok", None, None)?;
918      document.set_attribute(&mut absent_tok, "meaning", "absent")?;
919      document.close_element_at(&mut absent_tok)?;
920      absent_tok.unlink();
921      Ok(Some(absent_tok))
922    } else if content_nodes.len() == 1 && punct_nodes.is_empty() {
923      // single node, nothing to wrap
924      Ok(Some(content_nodes.remove(0)))
925    } else if content_nodes.len() == 1 {
926      // single node with trailing punct: wrap in XMDual
927      let result = content_nodes.remove(0);
928      Ok(Some(self.wrap_with_punct(
929        result,
930        punct_nodes,
931        mathnode,
932        document,
933      )?))
934    } else {
935      // Use pre-filtered content_nodes to avoid double-filtering (filter_hints already called
936      // above)
937      let (lexemes, mut nodes) = node_to_grammar_lexemes_from(mathnode, content_nodes, &mut idx);
938      if let Ok(Some(parse_tree)) = self.parse_lexemes(lexemes, &nodes, document) {
939        //START reparent: the reparenting used to be in `parse_rec` in Perl. Is this a good place?
940        // Replace the content of XMath with parsed result
941        // unbindNode followed by (append|replace)Tree (which removes ID's) should be safe
942        for child_el in element_nodes(mathnode) {
943          document.unrecord_node_ids(&child_el);
944        }
945        for mut node in mathnode.get_child_nodes() {
946          node.unlink();
947        }
948        let new_xml_tree = parse_tree.into_xmath(mathnode, &mut nodes, document)?;
949        document.append_tree(mathnode, vec![new_xml_tree])?;
950        // Resolve _xmkey references: match XMRef[@_xmkey] to elements with same _xmkey
951        resolve_xmkeys(mathnode, document)?;
952        let result = element_nodes(mathnode).remove(0);
953        //END reparent.
954        if !punct_nodes.is_empty() {
955          Ok(Some(self.wrap_with_punct(
956            result,
957            punct_nodes,
958            mathnode,
959            document,
960          )?))
961        } else {
962          Ok(Some(result))
963        }
964      } else {
965        // Parse failed; put punct nodes back in mathnode so they remain visible
966        for mut p in punct_nodes {
967          mathnode.add_child(&mut p).ok();
968        }
969        Ok(None)
970      }
971    }
972  }

974  /// Wrap `result` in an XMDual so trailing punctuation appears in the presentation
975  /// branch only, matching Perl's `if (@punct) { $result = ['ltx:XMDual', ...] }`.
976  fn wrap_with_punct(
977    &self,
978    mut result: Node,
979    punct_nodes: Vec<Node>,
980    mathnode: &mut Node,
981    document: &mut Document,
982  ) -> Result<Node> {
983    // Assign an id to result while it's still in the tree (id generation needs ancestors)
984    document.generate_id(&mut result, "")?;
985    let id = result
986      .get_attribute_ns("id", XML_NS)
987      .unwrap_or_else(|| result.get_attribute("id").unwrap_or_default());
988    // Detach result from mathnode so we can re-parent it inside XMWrap
989    result.unlink();
990    // Create XMDual in mathnode
991    let mut dual = document.open_element_at(mathnode, "ltx:XMDual", None, None)?;
992    // Content branch: XMRef pointing to result
993    let ref_attrs = {
994      let mut m = HashMap::default();
995      m.insert(s!("idref"), id);
996      Some(m)
997    };
998    let mut xmref = document.open_element_at(&mut dual, "ltx:XMRef", ref_attrs, None)?;
999    document.close_element_at(&mut xmref)?;
1000    // Presentation branch: XMWrap containing result + trailing punct
1001    let mut xmwrap = document.open_element_at(&mut dual, "ltx:XMWrap", None, None)?;
1002    xmwrap.add_child(&mut result).ok();
1003    for mut p in punct_nodes {
1004      xmwrap.add_child(&mut p).ok();
1005    }
1006    document.close_element_at(&mut xmwrap)?;
1007    document.close_element_at(&mut dual)?;
1008    Ok(dual)
1009  }

1040    // a new unique tree, the grammar ambiguity is purely structural
1041    // (script attachment ordering). Stop early.
1042    let max_consecutive_dupes = 16;
1043    // Time-budget convergence: once we have unique parses, stop after
1044    // this budget. For formulas where all trees are pruned (no unique
1045    // parse yet), use a longer budget before giving up.
1046    let converge_budget = std::time::Duration::from_millis(200);
1047    // Pruned-only fast-fail: if we've spent significant time and seen
1048    // many trees without finding a single semantic-acceptable parse,
1049    // the grammar is exploring a combinatorial dead end. Bail before
1050    // exhausting `max_time` (30s). Empirical: 0804.1730 case had 4536
1051    // pruned trees over 28s before timeout. After 2s with >200 prunes
1052    // the marginal probability of finding a unique drops sharply.
1053    let pruned_only_time_budget = std::time::Duration::from_secs(2);
1054    let pruned_only_count_threshold: usize = 200;
1055    // Unique-tree cap: the pragmatics/selection step only needs a handful
1056    // of distinct parses. Beyond this, additional unique trees are almost
1057    // always script-attachment ordering variants that don't improve the
1058    // final selected parse. Avoids enumerating 60+ trees for expressions
1059    // like `{}^4{}_{12}C^{5+}` where the grammar produces 27+ unique
1060    // trees from different pre/post script nesting orders.
1061    let max_unique = 10;
1062    for val in parse_result {
1063      // Truncate if too many trees or too much time
1064      if ok_trees + pruned_trees >= max_trees || start.elapsed() > max_time {
1065        break;
1066      }
1067      // Early convergence: stop if we keep seeing only duplicates.
1068      // The grammar produces 2^N duplicates from script attachment ordering.
1069      // Once we've found all unique parses, every new tree is a duplicate.
1070      if consecutive_dupes >= max_consecutive_dupes && !parses.is_empty() {
1071        break;
1072      }
1073      // Unique-tree cap: stop once we have enough distinct parses.
1074      if parses.len() >= max_unique {
1075        break;
1076      }
1077      // Time-budget convergence: if we have unique parses and have spent
1078      // >200ms, stop — the remaining trees are overwhelmingly duplicates.
1079      if !parses.is_empty() && start.elapsed() > converge_budget {
1080        break;
1081      }
1082      // Pruned-only fast-fail: bail when we have NO unique parses, have
1083      // already enumerated many trees, and have burned through the
1084      // pruned-only budget. Without this, the loop runs to `max_time`
1085      // (30s) on pathological multi-clause RELOP-list formulae where
1086      // every grammar derivation is semantically pruned (e.g.
1087      // 0804.1730 had 4536 enumerated → 0 unique → 28.29s).
1088      if parses.is_empty()
1089        && pruned_trees > pruned_only_count_threshold
1090        && start.elapsed() > pruned_only_time_budget
1091      {
1092        break;
1093      }
1094      // Note: we intentionally do NOT abort when no parse has been found
1095      // even after extended time — valid parses can appear late in the
1096      // enumeration (tree #3585 of 3713 for complex multi-equation formulas).
1097      match self.actions.get_tree(
1098        self.builder.clone(),
1099        val,
1100        self.expert_pragmatics.as_slice(),
1101        ActionContext { nodes, document },
1102      ) {
1103        Ok(tree_opt) => {
1104          if let Some(tree) = tree_opt {
1105            ok_trees += 1;
1106            // Online deduplication: check if this tree is already in our unique set
1107            if parses.contains(&tree) {
1108              deduped += 1;
1109              consecutive_dupes += 1;
1110            } else {
1111              parses.push(tree);
1112              // Half-decay (not full reset) on new unique. This lets us bail
1113              // on cases where uniques are sparse among a sea of dupes/prunes —
1114              // e.g. sin[XY] produces 10 unique parses among 1022 grammar
1115              // derivations; without decay the unique trees keep resetting the
1116              // dupe counter and we never converge. Half-decay means each new
1117              // unique halves the accumulated dupe budget instead of clearing it.
1118              consecutive_dupes /= 2;
1119            }
1120          }
1121        },
1122        Err(_prune_err) => {
1123          pruned_trees += 1;
1124          // Pruned trees also count toward convergence if we have unique parses
1125          if !parses.is_empty() {
1126            consecutive_dupes += 1;
1127          }
1128        },
1129      }
1130    }
1131
1132    // Store count for \ltx@count@parses diagnostic macro
1133    // Use post-dedup count (distinct semantic trees), not raw grammar count
1134    self.last_parsetrees_count = parses.len();
1135
1136    if ok_trees + pruned_trees > 10 {
1137      log::warn!(
1138        "Ambiguous math: {} enumerated ({} semantic, {} pruned, {} deduped→{} unique) in {:?} for: {}",
1139        ok_trees + pruned_trees + deduped,
1140        ok_trees + deduped,
1141        pruned_trees,
1142        deduped,
1143        parses.len(),
1144        start.elapsed(),
1145        input.trim()
1146      );
1147    }
1148    // Diagnostic: report parse counts when LATEXML_PARSE_AUDIT is set.
1149    // Useful for identifying grammar ambiguity hotspots across the test suite.
1150    // Usage: LATEXML_PARSE_AUDIT=1 cargo test --test 56_ams -- mathtools_test --nocapture
1151    if *PARSE_AUDIT && (ok_trees + pruned_trees > 1 || start.elapsed().as_millis() > 50) {
1152      eprintln!(
1153        "PARSE_AUDIT: {} trees ({} ok, {} pruned, {} dedup→{} unique) in {:?} | {}",
1154        ok_trees + pruned_trees + deduped,
1155        ok_trees + deduped,
1156        pruned_trees,
1157        deduped,
1158        parses.len(),
1159        start.elapsed(),
1160        input.trim().chars().take(200).collect::<String>()
1161      );
1162    }
1163
1164    match parses.len() {
1165      0 => Err("Failed to find any parse".into()),
1166      1 => Ok(parses.into_iter().next().unwrap()),
1167      _more => {
1168        // Loop over the various soft pruning algorithms available, until we have 1 tree
1169        let mut reduced_forest = XM::Choices(parses);
1170        for pragma in self.student_pragmatics.iter() {
1171          reduced_forest = reduced_forest.soft_prune_choices(*pragma);
1172          match reduced_forest {
1173            XM::Choices(ref trees) if trees.len() <= 1 => break,
1174            _ => {},
1175          };
1176        }
1177        Ok(reduced_forest)
1178      },
1179    }
1180  }

1182  pub fn parse_lexemes(
1183    &mut self,
1184    lexemes: Vec<String>,
1185    nodes: &[Node],
1186    document: &mut Document,
1187  ) -> Result<Option<XM>> {
1188    let mut input_string: String = lexemes.join(" ");
1189    // Add a trailing space, in an attempt to work with
1190    // a rules!() macro that has a Hard expectation of a space char following EVERY token.
1191    // this - counterintuitively- allows a simple macro definition AND a simple parse tree.
1192    input_string.push(' ');
1193
1194    match self.parse_marpa(&input_string, nodes, document) {
1195      Ok(mut parse_tree) => {
1196        // Perf: after successful parse, Marpa engine is in state T. The next
1197        // run_recognizer call will naturally advance T → GReady → R (fresh
1198        // Recognizer) without triggering precompute. Avoiding reset_engine
1199        // here saves the ~8% CPU time that precompute consumed per formula.
1200        // Restructure flat formulae with \quad separators to right-recursive nesting
1201        // (matching Perl's moreRHS/maybeColRHS right-recursive structure)
1202        crate::semantics::restructure_formulae_right(&mut parse_tree)?;
1203        // Rename `list` to `vector`/`set` when delimiters wrap the list (Perl encloseN)
1204        crate::semantics::rename_fenced_lists(&mut parse_tree, nodes)?;
1205        // Combine adjacent SUPOP tokens (prime+prime → prime2)
1206        crate::semantics::combine_supop_post(&mut parse_tree, nodes)?;
1207        Ok(Some(parse_tree))
1208      },
1209      Err(_e) => {
1210        self.reset_engine();
1211        Ok(None)
1212      },
1213    }
1214  }

1222/// Perl: parse_kludgeScripts_rec (MathParser.pm L568-589)
1223/// Takes a list of (Node, role) pairs and attaches scripts to their bases:
1224/// - POSTSUPERSCRIPT/POSTSUBSCRIPT → attach to preceding node
1225/// - FLOATSUPERSCRIPT/FLOATSUBSCRIPT → pre-script on following node
1226fn kludge_scripts_rec(pairs: Vec<(Node, String)>, document: &mut Document) -> Vec<Node> {
1227  use crate::data::get_grammatical_role;
1228  if pairs.is_empty() {
1229    return vec![];
1230  }
1231  if pairs.len() == 1 {
1232    return vec![pairs.into_iter().next().unwrap().0];
1233  }
1234
1235  let mut acc: Vec<Node> = Vec::new();
1236  let mut iter = pairs.into_iter();
1237  let mut x = iter.next().unwrap(); // (node, role)
1238
1239  loop {
1240    let y_opt = iter.next();
1241    let y = match y_opt {
1242      Some(y) => y,
1243      None => {
1244        // Base case: only x remains
1245        acc.push(x.0);
1246        break;
1247      },
1248    };
1249    let more: Vec<(Node, String)> = iter.collect();
1250
1251    if is_float_script(&x.1) {
1252      if is_post_script(&y.1) {
1253        if !more.is_empty() {
1254          // Perl L575-576: FLOAT + POST + more → combined pre-sub+super
1255          let mut rest = kludge_scripts_rec(more, document);
1256          if let Some(base) = rest.first_mut() {
1257            let inner = new_script_node(base.clone(), &y, document);
1258            let outer = new_script_node(inner, &x, document);
1259            rest[0] = outer;
1260          }
1261          acc.extend(rest);
1262          break;
1263        } else {
1264          // Perl L578: FLOAT + POST, no more → floating sub+super on Absent
1265          let absent = new_absent_node(document);
1266          let inner = new_script_node(absent, &y, document);
1267          let outer = new_script_node(inner, &x, document);
1268          acc.push(outer);
1269          break;
1270        }
1271      } else {
1272        // Perl L580-581: FLOAT + non-script → prescript on whatever follows
1273        let mut rest_pairs = vec![y];
1274        rest_pairs.extend(more);
1275        let mut rest = kludge_scripts_rec(rest_pairs, document);
1276        if let Some(base) = rest.first_mut() {
1277          let scripted = new_script_node(base.clone(), &x, document);
1278          rest[0] = scripted;
1279        }
1280        acc.extend(rest);
1281        break;
1282      }
1283    } else if is_post_script(&y.1) {
1284      // Perl L583-584: POST script → attach to preceding, recurse with result
1285      let scripted = new_script_node(x.0.clone(), &y, document);
1286      let role = get_grammatical_role(&scripted);
1287      let mut rest_pairs = vec![(scripted, role)];
1288      rest_pairs.extend(more);
1289      let rest = kludge_scripts_rec(rest_pairs, document);
1290      acc.extend(rest);
1291      break;
1292    } else {
1293      // Perl L585-586: neither is a script → accumulate x, advance
1294      acc.push(x.0);
1295      x = y;
1296      iter = more.into_iter();
1297    }
1298  }
1299  acc
1300}

1306/// Perl: NewScript (MathParser.pm L1597-1644)
1307/// Creates XMApp(XMTok[role=SCRIPTOP, scriptpos=...], base, script_content).
1308fn new_script_node(base: Node, script_pair: &(Node, String), document: &mut Document) -> Node {
1309  let (script_node, script_role) = script_pair;
1310
1311  // Determine SUPER vs SUB from role
1312  let is_super = script_role.contains("SUPER");
1313  let is_float = script_role.starts_with("FLOAT");
1314  let op_role = if is_super {
1315    "SUPERSCRIPTOP"
1316  } else {
1317    "SUBSCRIPTOP"
1318  };
1319
1320  // Extract scriptpos from base and script (Perl L1613-1635)
1321  // For XMApp bases, check the inner operator's scriptpos too
1322  let rbase_sp = {
1323    let mut sp = base.get_attribute("scriptpos").unwrap_or_default();
1324    if sp.is_empty() && base.get_name() == "XMApp" {
1325      // Check inner operator (first child) for scriptpos
1326      if let Some(op) = base.get_child_elements().first() {
1327        sp = op.get_attribute("scriptpos").unwrap_or_default();
1328      }
1329    }
1330    sp
1331  };
1332  let script_sp = script_node.get_attribute("scriptpos").unwrap_or_default();
1333
1334  let (bx, bl) = parse_scriptpos_str(&rbase_sp);
1335  let (sx, sl) = parse_scriptpos_str(&script_sp);
1336  let bl = if bl > 0 {
1337    bl
1338  } else if sl > 0 {
1339    sl
1340  } else {
1341    1
1342  };
1343  let sl = if sl > 0 { sl } else { bl };
1344  let mut l = if sl > 0 {
1345    sl
1346  } else if bl > 0 {
1347    bl
1348  } else {
1349    1
1350  };
1351
1352  // Perl: if base was a float script, bump level
1353  if base.get_attribute("_wasfloat").is_some() {
1354    l += 1;
1355  } else if let Some(bump_str) = base.get_attribute("_bumplevel") {
1356    if let Ok(bump) = bump_str.parse::<u32>() {
1357      l = bump;
1358    }
1359  }
1360
1361  // Perl L1632: position from base or script, defaulting to "post"
1362  let x = if is_float {
1363    "pre"
1364  } else if bl == sl {
1365    if !bx.is_empty() { bx } else { "post" }
1366  } else {
1367    if !sx.is_empty() { sx } else { "post" }
1368  };
1369  let scriptpos = format!("{x}{l}");
1370
1371  // Create XMTok operator: <XMTok role="SUPERSCRIPTOP" scriptpos="post1"/>
1372  let mut op_node = Node::new("XMTok", None, document.get_document()).unwrap();
1373  op_node.set_attribute("role", op_role).ok();
1374  op_node.set_attribute("scriptpos", &scriptpos).ok();
1375
1376  // Extract script content: first child element of the script XMApp
1377  // Perl: Arg($script, 0) — gets first element child
1378  let script_content = script_node.get_child_elements().into_iter().next();
1379
1380  // Create XMApp: <XMApp> op, base, script_content </XMApp>
1381  let mut app_node = Node::new("XMApp", None, document.get_document()).unwrap();
1382
1383  // Propagate _font from operator if present
1384  if let Some(font) = op_node.get_attribute("_font") {
1385    app_node.set_attribute("_font", &font).ok();
1386  }
1387  // Mark as float if applicable
1388  if is_float {
1389    app_node.set_attribute("_wasfloat", "1").ok();
1390  }
1391
1392  // Add children: op, base, script_content
1393  op_node.unlink_node();
1394  app_node.add_child(&mut op_node).ok();
1395  let mut base_clone = base;
1396  base_clone.unlink_node();
1397  app_node.add_child(&mut base_clone).ok();
1398  if let Some(mut content) = script_content {
1399    content.unlink_node();
1400    app_node.add_child(&mut content).ok();
1401  }
1402  app_node
1403}

1464fn textrec(
1465  node_opt: &Node,
1466  outer_bp_opt: Option<usize>,
1467  outer_name_opt: Option<&str>,
1468  document: &Document,
1469) -> String {
1470  // Depth guard — see TEXTREC_MAX_DEPTH comment. Increment + defer a
1471  // decrement so early returns still balance.
1472  let entered = TEXTREC_DEPTH.with(|d| {
1473    let v = d.get();
1474    if v >= TEXTREC_MAX_DEPTH {
1475      false
1476    } else {
1477      d.set(v + 1);
1478      true
1479    }
1480  });
1481  if !entered {
1482    return String::new();
1483  }
1484  struct _Dec;
1485  impl Drop for _Dec {
1486    fn drop(&mut self) { TEXTREC_DEPTH.with(|d| d.set(d.get().saturating_sub(1))); }
1487  }
1488  let _dec = _Dec;
1489
1490  let node = realize_xmnode(node_opt, document);
1491  let tag = get_node_qname(&node);
1492  let outer_bp = outer_bp_opt.unwrap_or(0);
1493  let outer_name = outer_name_opt.unwrap_or("");
1494  // If node has meaning, that's the text form.
1495  let meaning_opt = match p_get_attribute(&node, "meaning") {
1496    Some(m) => Some(m),
1497    None => p_get_attribute(&node, "name"),
1498  };
1499  if let Some(meaning) = meaning_opt {
1500    return match PREFIX_ALIAS.get(meaning.as_str()) {
1501      Some(m) => m.to_string(),
1502      None => meaning,
1503    };
1504  }
1505  if tag == pin!("ltx:XMApp") {
1506    let mut args = element_nodes(&node);
1507    if args.is_empty() {
1508      // Perl: Error('expected','arguments',...) then returns empty
1509      log::warn!("XMApp element has no child arguments");
1510      return String::new();
1511    }
1512    let arg_node = args.remove(0);
1513    let op = realize_xmnode(&arg_node, document);
1514    if let Some(app_role) = node.get_attribute("role") {
1515      if app_role == "FLOATSUBSCRIPT" {
1516        return String::from("_") + &textrec(&op, None, None, document);
1517      } else if app_role == "FLOATSUPERSCRIPT" {
1518        return String::from("^") + &textrec(&op, None, None, document);
1519      }
1520    }
1521
1522    let name = if with_node_qname(&op, |name| name == "ltx:XMTok") {
1523      get_token_meaning(&op, document).unwrap_or_else(|| "unknown".to_owned())
1524    } else {
1525      String::new()
1526    };
1527    let (bp, string) = textrec_apply(&name, &op, args, document);
1528    if (bp < outer_bp) || ((bp == outer_bp) && (name != outer_name)) {
1529      format!("({string})")
1530    } else {
1531      string
1532    }
1533  } else if tag == pin!("ltx:XMDual") {
1534    // XMDual normally has exactly 2 children (content-branch, presentation-
1535    // branch). A malformed dual can show up in practice — e.g. after a
1536    // partial rewrite that detached one side, or from a replacement pattern
1537    // whose semantic arm is empty. Fall back to the empty string so the
1538    // tex-attribute serialization stays intact rather than aborting the
1539    // whole conversion.
1540    let children = element_nodes(&node);
1541    let Some(content) = children.first() else {
1542      log::warn!("XMDual element has no child arguments");
1543      return String::new();
1544    };
1545    textrec(content, Some(outer_bp), Some(outer_name), document) // Just send out the
1546  // semantic form
1547  // Fall back to
1548  // presentation, if
1549  // content has poor
1550  // semantics (eg. from
1551  // replacement patterns)
1552  // TODO
1553  // return ($text =~
1554  // /^\(*Unknown/ ?
1555  // textrec($presentation,
1556  // $outer_bp, $outer_name)
1557  // : $text); }
1558  } else if tag == pin!("ltx:XMTok") {
1559    let name = match get_token_meaning(&node, document) {
1560      Some(meaning) => meaning,
1561      None => s!("Unknown"),
1562    };
1563    match PREFIX_ALIAS.get(name.as_str()) {
1564      Some(v) => v.to_string(),
1565      None => name,
1566    }
1567  } else if tag == pin!("ltx:XMWrap") || tag == pin!("ltx:XMCell") {
1568    // ??
1569    element_nodes(&node)
1570      .into_iter()
1571      .map(|child| textrec(&child, None, None, document))
1572      .collect::<Vec<_>>()
1573      .join("@")
1574  } else if tag == pin!("ltx:XMArg") {
1575    let args = element_nodes(&node);
1576    if args.is_empty() {
1577      return String::new();
1578    }
1579    args
1580      .iter()
1581      .map(|arg| textrec(arg, None, None, document))
1582      .collect::<Vec<_>>()
1583      .join("")
1584  } else if tag == pin!("ltx:XMArray") {
1585    textrec_array(&node, document)
1586  } else {
1587    s!("[{}]", p_get_value(&node))
1588  }
1589}

1591fn textrec_apply(name: &str, op: &Node, args: Vec<Node>, document: &Document) -> (usize, String) {
1592  let role = op
1593    .get_attribute("role")
1594    .unwrap_or_else(|| "Unknown".to_string());
1595  if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP")
1596    && PRE_DIGITS_RE.is_match(&op.get_attribute("scriptpos").unwrap_or_default())
1597  {
1598    // Note that this will likely get parenthesized due to high bp
1599    let mut inner = textrec(op, None, None, document);
1600    if let Some(arg2) = args.get(1) {
1601      inner.push(' ');
1602      inner.push_str(&textrec(arg2, None, None, document));
1603    }
1604    if let Some(arg1) = args.first() {
1605      inner.push(' ');
1606      inner.push_str(&textrec(arg1, None, None, document));
1607    }
1608    (5000, inner)
1609  } else if let Some(bp) = IS_INFIX.get(&role) {
1610    // A sub/superscript with a meaning probably should be prefix
1611    if (role == "SUBSCRIPTOP" || role == "SUPERSCRIPTOP") && op.has_attribute("meaning") {
1612      (
1613        500,
1614        format!(
1615          "{}@({})",
1616          textrec(op, Some(10000), Some(name), document),
1617          args
1618            .iter()
1619            .map(|a| textrec(a, None, None, document))
1620            .collect::<Vec<_>>()
1621            .join(", ")
1622        ),
1623      )
1624    } else {
1625      // Format as infix.
1626      let textrec_op = textrec(op, None, None, document);
1627      let rec_form = if args.len() == 1 {
1628        // unless a single arg; then prefix.
1629        textrec_op + " " + &textrec(&args[0], Some(*bp), Some(name), document)
1630      } else {
1631        args
1632          .iter()
1633          .map(|a| textrec(a, Some(*bp), Some(name), document))
1634          .collect::<Vec<_>>()
1635          .join(&(" ".to_string() + &textrec_op + " "))
1636      };
1637      (*bp, rec_form)
1638    }
1639  } else if role == "POSTFIX" {
1640    if args.is_empty() {
1641      (10000, textrec(op, None, None, document))
1642    } else {
1643      (
1644        10000,
1645        textrec(&args[0], Some(10000), Some(name), document) + &textrec(op, None, None, document),
1646      )
1647    }
1648  } else if name == "multirelation" {
1649    let joined = args
1650      .iter()
1651      .map(|a| textrec(a, Some(2), Some(name), document))
1652      .collect::<Vec<_>>()
1653      .join(" ");
1654    (2, joined)
1655  } else {
1656    (
1657      500,
1658      textrec(op, Some(10000), Some(name), document)
1659        + "@("
1660        + &args
1661          .iter()
1662          .map(|a| textrec(a, None, None, document))
1663          .collect::<Vec<_>>()
1664          .join(", ")
1665        + ")",
1666    )
1667  }
1668}

1670fn textrec_array(node: &Node, document: &Document) -> String {
1671  let name = node
1672    .get_attribute("meaning")
1673    .or_else(|| node.get_attribute("name"))
1674    .unwrap_or_else(|| String::from("Array"));
1675  let rows: Vec<String> = element_nodes(node)
1676    .into_iter()
1677    .map(|row| {
1678      let cells: Vec<String> = element_nodes(&row)
1679        .into_iter()
1680        .map(|cell| {
1681          if let Some(first_child) = cell.get_first_child() {
1682            textrec(&first_child, None, None, document)
1683          } else {
1684            String::new()
1685          }
1686        })
1687        .collect();
1688      format!("[{}]", cells.join(", "))
1689    })
1690    .collect();
1691  format!("{}[{}]", name, rows.join(", "))
1692}

1694//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1695// Cute! Were it NOT for Sub/Superscripts, the whole parsing process only
1696// builds a new superstructure around the sequence of token nodes in the input.
1697// Thus, any internal structure is unchanged.
1698//  They get re-parented, but if the parse fails, we've only got to put them
1699// BACK into the original node, to recover the original arrangment!!!
1700// Thus, we don't have to clone, and deal with namespace duplication.
1701// ...
1702// EXCEPT, as I said, for sub/superscripts!!!!
1703//
1704
1705//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1706// Constructors used in grammar
1707// All the tree construction in the grammar should come through these
1708// operations. We avoid mucking with the actual XML nodes (both to avoid
1709// modifying the original tree until we have a successful parse, and to avoid
1710// XML::LibXML cloning nightmares) We are converting XML nodes to array
1711// representation: [$tag, {%attr},@children] This means any inspection of
1712// nodes has to recognize that  * node may be in XML vs ARRAY representation
1713// * node may be an XMRef to another node whose properties are the ones we
1714// should use.
1715//
1716// Also, when we are examining a node's properties (roles, fences, script
1717// positioning, etc) we should be careful to check for XMRef indirection and
1718// examine the properties of the node that was referred to.
1719// HOWEVER, we should construct our parse tree using (a clone of) the XMRef
1720// node, rather than (a clone of) the referred to node, so as to preserve
1721// identity.
1722//%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
1723// We're currently keeping the id's on the nodes as they get cloned,
1724// since they'll (maybe) replace the unparsed nodes.
1725// However, if we consider multiple parses or preserving both parsed & unparsed,
1726// we may have to do some adaptation and id shifting.
1727// ================================================================================
1728
1729// ================================================================================
1730// Low-level accessors
1731
1732// The following accessors work on both the LibXML and ARRAY representations
1733// but they do NOT automatically dereference XMRef!
1734pub fn p_get_value(node: &Node) -> String {
1735  let node_type = node.get_type();
1736  if node_type == Some(NodeType::ElementNode) {
1737    let x = node.get_content();
1738    if !x.is_empty() {
1739      // get content, or fall back to name
1740      x
1741    } else {
1742      node.get_attribute("name").unwrap_or_default()
1743    }
1744  } else {
1745    node.get_content()
1746  }
1747}

1777/// Resolve _xmkey and _pxmkey references after parse tree installation.
1778/// Matches XMRef[@_xmkey] to elements with same _xmkey, generates xml:id and sets idref.
1779/// _pxmkey is used by parser-generated XMDual (apply_delimited) to avoid
1780/// conflicting with base_xmath's \lx@dual afterConstruct resolver.
1781fn resolve_xmkeys(
1782  mathnode: &Node,
1783  document: &mut Document,
1784) -> std::result::Result<(), Box<dyn std::error::Error>> {
1785  // Resolve both _xmkey (from existing infrastructure) and _pxmkey (from parser)
1786  for attr_name in &["_xmkey", "_pxmkey"] {
1787    let refs = document.findnodes(
1788      &format!("descendant::ltx:XMRef[@{}]", attr_name),
1789      Some(mathnode),
1790    );
1791    for mut ref_node in refs {
1792      let key = match ref_node.get_attribute(attr_name) {
1793        Some(k) => k,
1794        None => continue,
1795      };
1796      // Find the element with matching key (non-XMRef)
1797      // Note: not(self::ltx:XMRef) may fail with namespace prefix in nested
1798      // predicates (XPath bug). Use local-name() check instead.
1799      let xpath = format!(
1800        "descendant::*[@{}='{}'][not(local-name()='XMRef')]",
1801        attr_name, key
1802      );
1803      let targets = document.findnodes(&xpath, Some(mathnode));
1804      if let Some(mut target) = targets.into_iter().next() {
1805        // Ensure target has xml:id
1806        let target_id = if let Some(id) = target
1807          .get_attribute("xml:id")
1808          .or_else(|| target.get_attribute("id"))
1809        {
1810          id
1811        } else {
1812          document.generate_id(&mut target, "")?;
1813          target
1814            .get_attribute("xml:id")
1815            .or_else(|| target.get_attribute("id"))
1816            .unwrap_or_default()
1817        };
1818        if !target_id.is_empty() {
1819          document.set_attribute(&mut ref_node, "idref", &target_id)?;
1820        }
1821      }
1822      let _ = ref_node.remove_attribute(attr_name);
1823    }
1824    // Clean up from non-ref elements
1825    for mut node in document.findnodes(&format!("descendant::*[@{}]", attr_name), Some(mathnode)) {
1826      let _ = node.remove_attribute(attr_name);
1827    }
1828  }
1829  Ok(())
1830}
