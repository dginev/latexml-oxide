68  pub fn can_specialize(&self) -> bool {
69    // does the meta object have any fields that are meaningful during specialization? Namely:
70    self.fenced.is_some()
71      || self.curry_level.is_some()
72      || !self.syntax_trace.is_empty()
73      || !self.curry_constraints.is_empty()
74  }

76  /// Check if no type fields are set, as would be the default case
77  pub fn is_empty(&self) -> bool {
78    self.fenced.is_none() && self.curry_level.is_none() && self.curry_constraints.is_empty()
79  }

81  /// Override all fields of meta with the nonempty fields of the incoming data
82  /// Special features:
83  /// - fenced:* can be used to invalidate `self` if it has no fenced attribute set
84  /// - fenced.overaccent will rename a single Var curry level to an embellished variant
85  pub fn with(self, mut other: Meta) -> Result<Self, Box<dyn Error>> {
86    // EMBELLISHED specialization
87    // overaccents lead to renaming of all pieces
88    let specialize = if other.specialize.is_some() {
89      other.specialize
90    } else {
91      self.specialize
92    };
93
94    // TODO: We would need a smart tracking algorithm to extend the full trace
95    // for now keep the last step.
96    //self.syntax_trace.extend(other.syntax_trace.into_iter());
97    let syntax_trace = if other.syntax_trace.is_empty() {
98      self.syntax_trace
99    } else {
100      other.syntax_trace
101    };
102
103    let fenced = if let Some(ref fenced_kind) = other.fenced {
104      // special case, the star requires some existing value, or we prune
105      if fenced_kind == "*" {
106        if self.fenced.is_none() {
107          return Err("fenced.* requirement failed".into());
108        } else {
109          self.fenced
110        }
111      } else {
112        other.fenced
113      }
114    } else {
115      self.fenced
116    };
117    let mut curry_constraints = self.curry_constraints;
118    for other_constraint in other.curry_constraints.drain() {
119      curry_constraints.insert(other_constraint);
120    }
121
122    // Incoming levels are to be seen as constraints on the current level.
123    // When no current level has been set, the incoming level is used as current with a >= 1
124    // constraint.
125    let curry_level = if let Some(current_level) = self.curry_level {
126      if let Some(new_level) = other.curry_level {
127        if current_level != new_level {
128          curry_constraints.insert(CurryConstraint((
129            current_level,
130            Ordering::Equal,
131            new_level.clone(),
132          )));
133        }
134        Some(new_level)
135      } else {
136        Some(current_level)
137      }
138    } else {
139      other.curry_level
140    };
141
142    Ok(Meta {
143      syntax_trace,
144      fenced,
145      specialize,
146      curry_level,
147      curry_constraints,
148    })
149  }

163  /// Specializing with a metadata object over a curry atom, implies we can
164  /// substitute the curry level with the name of the leaf node, and
165  /// instead create a constraint
166  pub fn with_curry_atom(self, mut into: Meta, name: &str) -> Result<Self, Box<dyn Error>> {
167    // massage the name so that cbc solver can handle it without weird glitches.
168    let lex_parts = distill_lexeme(name);
169    let curry_var = CurryTerm::Var(format!(":{}", lex_parts.2));
170    // We don't have to build this in, since by default we will enforce all variables are in a fixed
171    // curry range of [1,10]. into.curry_constraints.insert(CurryConstraint((
172    //   curry_var.clone(),
173    //   Ordering::Greater,
174    //   CurryTerm::Literal(0),
175    // )));
176
177    if let Some(ref level) = into.curry_level {
178      if &curry_var != level {
179        into.curry_constraints.insert(CurryConstraint((
180          curry_var.clone(),
181          Ordering::Equal,
182          level.clone(),
183        )));
184      }
185    }
186    into.curry_level = Some(curry_var);
187    self.with(into)
188  }
