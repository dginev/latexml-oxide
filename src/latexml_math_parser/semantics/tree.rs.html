64/// Custom PartialEq: ignores `xmkey`, `id`, `idref`, and `scriptpos` which are
65/// bookkeeping/layout fields. Structurally identical parse trees that differ only
66/// in internal reference keys or script-position labels (pre1 vs pre2 vs post1)
67/// should be considered equal for deduplication purposes.
68/// `scriptpos` is excluded because different grammar paths produce different
69/// pre/post level assignments for the same mathematical expression. E.g.,
70/// `{}^4{}_{12}C^{5+}` can produce 27 structurally distinct trees that differ
71/// only in scriptpos values — all represent the same expression.
72impl PartialEq for XProps {
73  fn eq(&self, other: &Self) -> bool {
74    self.content == other.content
75      && self.role == other.role
76      && self.meaning == other.meaning
77      && self.name == other.name
78      // Skip: scriptpos — layout hint, not semantic distinction
79      // Skip: id, idref, xmkey — bookkeeping for Dual/Ref resolution
80      && self.font == other.font
81      && self.fontref == other.fontref
82      && self.stretchy == other.stretchy
83      && self.possible_function == other.possible_function
84      && self.mathstyle == other.mathstyle
85      && self.thickness == other.thickness
86      && self.decl_id == other.decl_id
87      && self.lpadding == other.lpadding
88      && self.rpadding == other.rpadding
89  }
90}

160/// The math parsing process can manipulate a variety of trees,
161/// finally serialized via the XMath schema of LaTeXML.
162///
163/// The main structural variants are associated with
164/// a "parsing state::, via an attached `Meta` object.
165#[derive(Debug, Clone, PartialEq, Eq)]
166pub enum XM {
167  /// Token-name lexeme (e.g. "RELOP:less-than:3", "letter:a",
168  /// "delimited-open-paren"). The name is stored as `Rc<str>` so:
169  ///
170  /// 1. **Byte-glade hot path** (asf_traverser case 1): each ASCII byte resolves to a cached
171  ///    `Rc<str>` via `byte_lexeme_rc(b)`. Clones are refcount bumps — no allocation per byte
172  ///    glade.
173  /// 2. **Marpa ASF cache clones** (`asf.rs:156, 208`): cloning a `ParseTree = Vec<Option<XM>>` no
174  ///    longer deep-clones the lexeme name. Refcount-bump per Lexeme in the Vec.
175  /// 3. **Read sites** (`name.starts_with(...)`, `name == "..."`, `name.split(...)`,
176  ///    `name.contains(...)`): unchanged — `Rc<str>` derefs to `&str`.
177  ///
178  /// Construction at runtime: `Rc::from(string)` / `Rc::from("…")`.
179  Lexeme(Rc<str>, Meta),
180  Token(XProps, Meta), // does this need Meta?
181  Apply(Operator, Args, XProps, Meta),
182  Dual(Box<XM>, Box<XM>, XProps, Meta),
183  Ref(XProps),
184  Wrap(Vec<XM>, XProps, Meta),
185  Arg(Vec<XM>),
186  Choices(Vec<XM>),
187}
188impl From<XProps> for XM {
189  fn from(t: XProps) -> Self { XM::Token(t, Meta::default()) }
190}

246impl Args {
247  fn fmt_indented(&self, level: &[bool], f: &mut fmt::Formatter<'_>) -> fmt::Result {
248    let mut peekable = self.0.iter().peekable();
249    while let Some(maybe_arg) = peekable.next() {
250      if peekable.peek().is_some() {
251        maybe_arg
252          .as_ref()
253          .unwrap_or(&XM::Lexeme(Rc::from("missing_argument"), Meta::default()))
254          .fmt_indented(level, f)?;
255      } else {
256        let mut last_level: Vec<bool> = level.to_vec();
257        if !last_level.is_empty() {
258          last_level.pop();
259          last_level.push(false);
260        }
261        maybe_arg
262          .as_ref()
263          .unwrap_or(&XM::Lexeme(Rc::from("missing_argument"), Meta::default()))
264          .fmt_indented(&last_level, f)?;
265      };
266    }
267    Ok(())
268  }

343  pub fn get_value(&self, nodes: &[Node]) -> Result<Cow<'_, str>, Box<dyn Error>> {
344    Ok(match self {
345      XM::Lexeme(lex, _) => Cow::Owned(p_get_value(lookup_lex_node(lex, nodes)?)),
346      XM::Token(props, _) => match props.content {
347        None => props.name.clone().unwrap_or(Cow::Borrowed("")),
348        Some(ref v) => {
349          if v.is_empty() {
350            props.name.clone().unwrap_or(Cow::Borrowed(""))
351          } else {
352            v.clone()
353          }
354        },
355      },
356      // Propagate a bad-lexeme-id lookup failure with `?` instead of
357      // `.expect()`-panicking — these recursive arms must match the top-level
358      // arm's graceful Err (a `lookup_lex_node` miss on adversarial input is
359      // recoverable; the caller prunes the parse).
360      XM::Apply(op, args, ..) => {
361        let head = op.0.get_value(nodes)?;
362        let parts = args
363          .trees()
364          .iter()
365          .map(|t| t.get_value(nodes))
366          .collect::<Result<Vec<_>, _>>()?;
367        Cow::Owned(format!("{head}{}", parts.join("")))
368      },
369      // Choices/Arg don't carry a serialized value — return empty for safety;
370      // callers treat Ref similarly (see the XM::Ref arm below).
371      XM::Choices(_) | XM::Arg(_) => Cow::Borrowed(""),
372      XM::Dual(content, pres, ..) => Cow::Owned(format!(
373        "{}{}",
374        content.get_value(nodes)?,
375        pres.get_value(nodes)?
376      )),
377      XM::Wrap(args, ..) => Cow::Owned(
378        args
379          .iter()
380          .map(|a| a.get_value(nodes))
381          .collect::<Result<Vec<_>, _>>()?
382          .join(""),
383      ),
384      XM::Ref(_) => Cow::Borrowed(""),
385    })
386  }

388  /// Specialize a tree with the given Meta object,
389  /// also verifying the tree's inner consistency.
390  ///
391  /// Whenever a contradiction/inconsistency is detected, we return an error
392  /// This method should always be called on tree construction, as it also manages the various curry
393  /// constraints, keeping the resolution local / fast.
394  pub fn specialize(
395    self,
396    mut into: Meta,
397    pragmas: &[ValidationPragmatics],
398  ) -> Result<Self, Box<dyn Error>> {
399    match self {
400      XM::Lexeme(name, meta) => {
401        let new_meta = meta.with_curry_atom(into, &name)?;
402        Ok(XM::Lexeme(name, new_meta))
403      },
404      XM::Token(t, meta) => {
405        // Specialization of bare Token variants isn't exercised by current
406        // grammar rules. Return the tree unchanged so a future rule that
407        // invokes this path fails at a higher layer (validation) rather
408        // than panicking here.
409        Ok(XM::Token(t, meta))
410      },
411      XM::Ref(_) => Ok(self),
412      XM::Apply(mut op, mut args, props, meta) => {
413        // First, if we have a specialize directive, execute it:
414        match into.specialize {
415          Some(ref directive)
416            if directive == "embellish"
417            // Atoms with embellishments should get their curry levels renamed
418            // to avoid conflicts with the same atoms *without* the embellishments
419            // as often this technique is used to generate new unique names.
420            && args.0.len() <= 2 =>
421          {
422            if let Some(XM::Lexeme(_, arg_meta)) = &mut args.0[0] {
423              if let Some(CurryTerm::Var(ref mut curry_var)) = arg_meta.curry_level {
424                let mut base_op = op.0.base_operator_name();
425                // fish out a local name to use as an embellishment
426                if let Some(last_colon_idx) = base_op.rfind(':') {
427                  base_op.replace_range(..=last_colon_idx, "");
428                } else if let Some(last_dot_idx) = base_op.rfind('.') {
429                  base_op.replace_range(..=last_dot_idx, "");
430                }
431                if base_op.is_empty() {
432                  base_op = String::from("embellished");
433                }
434                curry_var.push('-');
435                curry_var.push_str(&base_op);
436              }
437              into.curry_level.clone_from(&arg_meta.curry_level)
438            }
439          },
440          _ => {},
441        }
442        // Next, we validate the constraints.
443        let initial_constraint_count = meta.curry_constraints.len();
444        let mut new_meta = meta;
445        // SPECIAL CASE (of course): if we are using an "unconstrained" directive, no need to do any
446        // of this
447        if into.specialize.as_deref() == Some("unconstrained") {
448          op.unconstrain_recursive();
449          args.unconstrain_recursive();
450          new_meta = new_meta.with(into)?;
451          new_meta.curry_constraints = CurryConstraints::new();
452          new_meta.specialize = None;
453        } else {
454          new_meta = new_meta.with(into)?;
455          // COPY constraints from inner subtrees to the new apply root
456          for constraint in op.get_constraints().into_iter() {
457            new_meta.curry_constraints.insert(constraint.clone());
458          }
459          for constraint in args.get_constraints().into_iter() {
460            new_meta.curry_constraints.insert(constraint.clone());
461          }
462          // and require the current curry level to be >= 1 if compound
463          // (otherwise, it's already built in)
464          if let Some(ref expr) = new_meta.curry_level {
465            match expr {
466              CurryTerm::Add(..) | CurryTerm::Sub(..) => {
467                new_meta.curry_constraints.insert(CurryConstraint((
468                  expr.clone(),
469                  Ordering::Greater,
470                  CurryTerm::Literal(0),
471                )));
472              },
473              _ => {},
474            }
475          }
476        }
477        if initial_constraint_count < new_meta.curry_constraints.len() {
478          // whenever we add a constraint, re-valdiate the expression, and prune it if needed
479          // println!("Tree: \n{}", XM::Apply(op.clone(), args.clone(), new_meta.clone()));
480          new_meta.validate()?;
481        }
482        let new_tree = XM::Apply(op, args, props, new_meta);
483        for pragma in pragmas {
484          // expert pragmatics get to validate each new tree,
485          // in order to prune wrong interpretations as early as possible
486          pragma.validate(&new_tree)?;
487        }
488        Ok(new_tree)
489      },
490      // Dual/Wrap/Arg variants aren't specialized by current grammar rules
491      // — return unchanged rather than panic. Future grammar extensions can
492      // add real specialization rules if needed.
493      dual @ XM::Dual(..) => Ok(dual),
494      wrap @ XM::Wrap(..) => Ok(wrap),
495      arg @ XM::Arg(_) => Ok(arg),
496      XM::Choices(_) => Err("can not specialize choices".into()),
497    }
498  }

500  /// Prunes choices based on a validation pass leveraging a choice of pragmatics
501  /// if the pruning arrives at no viable trees at all, the original tree is returned,
502  /// hence the "soft" function name.
503  /// These are executed at the end of the program, so need to be invoked recursively on each
504  /// subtree
505  pub fn soft_prune_choices(self, pragmatics: ValidationPragmatics) -> Self {
506    match self {
507      XM::Choices(trees) => {
508        let (consistent_trees, inconsistent_trees): (Vec<XM>, Vec<XM>) = trees
509          .into_iter()
510          .partition(|tree| pragmatics.validate_recursive(tree).is_ok());
511        match consistent_trees.len() {
512          0 => XM::Choices(inconsistent_trees),
513          1 => consistent_trees.into_iter().next().unwrap(),
514          _more => XM::Choices(consistent_trees),
515        }
516      },
517      other => other,
518    }
519  }

521  /// Count the number of `absent` markers in this tree.
522  ///
523  /// `absent` appears as a sentinel `XM::Token` (with
524  /// `meaning="absent"`) when a parse has to fill in a missing
525  /// operand position — e.g. `< x >` parsed as the relational
526  /// chain `absent < x > absent`. Fewer `absent` markers signals
527  /// a parse that satisfies the grammar without needing fillers.
528  pub fn count_absent(&self) -> usize {
529    match self {
530      XM::Token(props, _) => {
531        if props.meaning.as_deref() == Some("absent") {
532          1
533        } else {
534          0
535        }
536      },
537      XM::Apply(op, args, ..) => {
538        op.0.count_absent() + args.trees().iter().map(|a| a.count_absent()).sum::<usize>()
539      },
540      XM::Dual(c, p, ..) => c.count_absent() + p.count_absent(),
541      XM::Wrap(items, ..) => items.iter().map(|i| i.count_absent()).sum(),
542      XM::Choices(trees) => trees.iter().map(|t| t.count_absent()).sum(),
543      XM::Arg(items) => items.iter().map(|i| i.count_absent()).sum(),
544      XM::Lexeme(..) | XM::Ref(_) => 0,
545    }
546  }

548  /// Semantic-node count used by parse-ranking pragmas to prefer
549  /// compact interpretations over verbose ones. Not a general
550  /// "tree size" function — it deliberately ignores presentation
551  /// duplication so structurally-equivalent parses get a fair
552  /// comparison.
553  ///
554  /// Smaller wins: `norm@(a)` (2 nodes) beats
555  /// `absolute-value@(absolute-value@(a))` (3 nodes);
556  /// `differential-d@(x)` (2 nodes) beats `d*x` (3 nodes).
557  ///
558  /// **Counting conventions**:
559  /// * Each `XM::Apply` contributes ONE node — the operator is part of the Apply's identity, not a
560  ///   separate child. So `f@(x)` is 2 nodes (the Apply + x), not 3.
561  /// * `XM::Dual(content, presentation)` counts **only the content tree** — the presentation branch
562  ///   is a parallel rendering of the same semantics and contributes the same count, so
563  ///   double-counting would inflate purely-cosmetic siblings.
564  /// * `XM::Ref(props)` is resolved to its target via the presentation-branch index built at the
565  ///   Dual boundary — so a Ref pointing to a deep sub-tree contributes its full target's node
566  ///   count, not just 1. This keeps the ranking honest when one parse uses a single Ref to a
567  ///   complex node versus another that lays out multiple Refs to leaves.
568  pub fn count_nodes_for_parse_ranking(&self) -> usize {
569    let mut index: HashMap<String, &XM> = HashMap::default();
570    self.build_ref_index(&mut index);
571    self.count_nodes_with_index(&index, &mut Vec::new())
572  }

574  /// Walk this tree and register every `(id, &XM)` pair that a
575  /// `Ref` might lookup. Refs key on `props.id` or `props.xmkey`.
576  fn build_ref_index<'a>(&'a self, out: &mut HashMap<String, &'a XM>) {
577    let take = |p: &'a XProps| -> Option<String> {
578      p.id
579        .as_ref()
580        .map(|c| c.to_string())
581        .or_else(|| p.xmkey.as_ref().map(|c| c.to_string()))
582    };
583    match self {
584      XM::Token(props, _) => {
585        if let Some(k) = take(props) {
586          out.entry(k).or_insert(self);
587        }
588      },
589      XM::Apply(_op, args, props, _) => {
590        if let Some(k) = take(props) {
591          out.entry(k).or_insert(self);
592        }
593        for arg in args.trees() {
594          arg.build_ref_index(out);
595        }
596      },
597      XM::Dual(c, p, props, _) => {
598        if let Some(k) = take(props) {
599          out.entry(k).or_insert(self);
600        }
601        c.build_ref_index(out);
602        p.build_ref_index(out);
603      },
604      XM::Wrap(items, props, _) => {
605        if let Some(k) = take(props) {
606          out.entry(k).or_insert(self);
607        }
608        for it in items {
609          it.build_ref_index(out);
610        }
611      },
612      XM::Arg(items) | XM::Choices(items) => {
613        for it in items {
614          it.build_ref_index(out);
615        }
616      },
617      XM::Lexeme(..) | XM::Ref(_) => {},
618    }
619  }

621  /// Counting worker with the Ref index. `visited` is a stack of
622  /// idref/xmkey strings currently being resolved — guards against
623  /// cyclic references (defensive; idref graphs in valid XMath
624  /// are acyclic).
625  fn count_nodes_with_index(
626    &self,
627    index: &HashMap<String, &XM>,
628    visited: &mut Vec<String>,
629  ) -> usize {
630    match self {
631      XM::Token(..) | XM::Lexeme(..) => 1,
632      XM::Apply(_op, args, ..) => {
633        1 + args
634          .trees()
635          .iter()
636          .map(|a| a.count_nodes_with_index(index, visited))
637          .sum::<usize>()
638      },
639      XM::Dual(c, _p, ..) => c.count_nodes_with_index(index, visited),
640      XM::Wrap(items, ..) => {
641        1 + items
642          .iter()
643          .map(|i| i.count_nodes_with_index(index, visited))
644          .sum::<usize>()
645      },
646      XM::Choices(trees) => trees
647        .iter()
648        .map(|t| t.count_nodes_with_index(index, visited))
649        .sum(),
650      XM::Arg(items) => {
651        1 + items
652          .iter()
653          .map(|i| i.count_nodes_with_index(index, visited))
654          .sum::<usize>()
655      },
656      XM::Ref(props) => {
657        let key = props
658          .idref
659          .as_ref()
660          .map(|c| c.to_string())
661          .or_else(|| props.xmkey.as_ref().map(|c| c.to_string()));
662        match key {
663          Some(k) if !visited.contains(&k) => match index.get(&k) {
664            Some(target) => {
665              visited.push(k);
666              let n = target.count_nodes_with_index(index, visited);
667              visited.pop();
668              n
669            },
670            None => 1,
671          },
672          // Already visiting this ref (cycle) or no key at all —
673          // count as 1 so we don't double-count or infinite-loop.
674          _ => 1,
675        }
676      },
677    }
678  }

820  /// Does the root Dual's **presentation** Wrap contain exactly one
821  /// non-delimiter child that is itself a `Dual` whose content
822  /// shares the same operator meaning as the outer? This is the
823  /// shape that produces `set@(set@(…))` / `vector@(vector@(…))`
824  /// when rendered — the outer content's Ref resolves to the inner
825  /// Dual instead of to flat items.
826  fn root_dual_has_redundant_inner_wrap(&self) -> bool {
827    let XM::Dual(content, presentation, ..) = self else {
828      return false;
829    };
830    let outer_meaning = match &**content {
831      XM::Apply(Operator(op), ..) => match &**op {
832        XM::Token(props, _) => props.meaning.as_deref().map(String::from),
833        XM::Lexeme(name, _) => Some(name.to_string()),
834        _ => None,
835      },
836      _ => return false,
837    };
838    let Some(outer_m) = outer_meaning else {
839      return false;
840    };
841    let XM::Wrap(items, ..) = &**presentation else {
842      return false;
843    };
844    let is_delim = |x: &XM| match x {
845      XM::Token(p, _) => matches!(p.role.as_deref(), Some("OPEN") | Some("CLOSE")),
846      XM::Lexeme(n, _) => n.starts_with("OPEN:") || n.starts_with("CLOSE:"),
847      _ => false,
848    };
849    let inner: Vec<&XM> = items.iter().filter(|x| !is_delim(x)).collect();
850    if inner.len() != 1 {
851      return false;
852    }
853    let XM::Dual(inner_content, ..) = inner[0] else {
854      return false;
855    };
856    // Inner Dual's content should be an Apply whose meaning matches
857    // outer (or is a closely-related "list"/"vector"/"formulae"
858    // meaning — common when `interpret_delimited` lifts the list
859    // wrapper above an inner Dual).
860    if let XM::Apply(Operator(inner_op), ..) = &**inner_content {
861      let inner_meaning = match &**inner_op {
862        XM::Token(props, _) => props.meaning.as_deref(),
863        XM::Lexeme(name, _) => Some(&**name),
864        _ => None,
865      };
866      // Outer-set wrapping inner-{set/list/vector/formulae} is the
867      // shape we want to prune. The legacy never picks this.
868      let inner_str = inner_meaning.unwrap_or("");
869      if outer_m == inner_str
870        || (matches!(outer_m.as_str(), "set" | "vector")
871          && matches!(inner_str, "set" | "vector" | "list" | "formulae"))
872      {
873        return true;
874      }
875    }
876    false
877  }

880  /// the Apply's first child (after following any `XM::Ref` through
881  /// the presentation branch's idref index) is ALSO an `Apply` with
882  /// the same `op_meaning`? Used to detect redundant
883  /// `set@(set@(...))` and similar self-wrapping shapes.
884  fn root_dual_is_redundant_self_wrap(&self) -> bool {
885    let XM::Dual(content, ..) = self else {
886      return false;
887    };
888    let outer_meaning = match &**content {
889      XM::Apply(Operator(op), args, ..) if args.trees().len() == 1 => match &**op {
890        XM::Token(props, _) => props.meaning.as_deref().map(String::from),
891        XM::Lexeme(name, _) => Some(name.to_string()),
892        _ => None,
893      },
894      _ => return false,
895    };
896    let Some(outer_m) = outer_meaning else {
897      return false;
898    };
899    // Resolve the inner: dereference Apply's single child. If it's
900    // a Ref, follow it through the idref index built across the
901    // whole Dual.
902    let mut index: HashMap<String, &XM> = HashMap::default();
903    self.build_ref_index(&mut index);
904    if let XM::Apply(_, args, ..) = &**content {
905      let first_arg = args.trees().first().copied();
906      let resolved = match first_arg {
907        Some(XM::Ref(props)) => {
908          let key = props
909            .idref
910            .as_ref()
911            .map(|c| c.to_string())
912            .or_else(|| props.xmkey.as_ref().map(|c| c.to_string()));
913          key.and_then(|k| index.get(&k).copied())
914        },
915        other => other,
916      };
917      if let Some(XM::Apply(Operator(inner_op), ..)) = resolved {
918        let inner_meaning = match &**inner_op {
919          XM::Token(props, _) => props.meaning.as_deref(),
920          XM::Lexeme(name, _) => Some(&**name),
921          _ => None,
922        };
923        return inner_meaning == Some(outer_m.as_str());
924      }
925    }
926    false
927  }

960  /// Does the root match `Apply(multirelation, [..., absent, ...])`
961  /// or `Apply(formulae, [..., absent, ...])` where `absent`
962  /// appears in the **interior** of the args list (NOT at the
963  /// first or last positions)?
964  ///
965  /// The distinction matters: an `absent` at the boundary is
966  /// legitimate (e.g. `<a|f|b>` parses as
967  /// `multirelation(absent, <, a, |, f, |, b, >, absent)` — the
968  /// outer `<` and `>` need left/right operands, which are absent
969  /// because the expression IS the whole math). An `absent` in the
970  /// middle (e.g. `x >= 0` parsed as `multirelation(x, >, absent,
971  /// =, 0)`) signals a failed `two_part_relop` combination — the
972  /// alternative parse using a combined `>=` operator is strictly
973  /// better.
974  fn root_is_multirelation_with_interior_absent(&self) -> bool {
975    let apply = match self {
976      XM::Apply(..) => self,
977      XM::Dual(content, ..) => &**content,
978      _ => return false,
979    };
980    let XM::Apply(Operator(op), args, ..) = apply else {
981      return false;
982    };
983    let meaning = match &**op {
984      XM::Token(props, _) => props.meaning.as_deref(),
985      XM::Lexeme(name, _) => Some(&**name),
986      _ => None,
987    };
988    // Both `multirelation` (explicit relation chain) and `formulae`
989    // (comma-separated relational chain) can fall into this pattern.
990    if !matches!(meaning, Some("multirelation") | Some("formulae")) {
991      return false;
992    }
993    let trees = args.trees();
994    if trees.len() < 3 {
995      return false;
996    }
997    let is_absent =
998      |a: &&XM| matches!(a, XM::Token(p, _) if p.meaning.as_deref() == Some("absent"));
999    // Skip the first and last positions; only inspect interior.
1000    trees[1..trees.len() - 1].iter().any(is_absent)
1001  }

1162  /// Count Apply nodes whose meaning is a **fence operator**
1163  /// (`norm`, `absolute-value`, `floor`, `ceiling`, etc.) AND that
1164  /// have an ancestor Apply with the **same** fence meaning. This
1165  /// is the "nested same-fence" count: e.g. `‖x · ‖a‖ · y‖` has one
1166  /// `norm` inside another `norm`, while `‖x‖ · a · ‖y‖` has none.
1167  ///
1168  /// Mathematicians read consecutive bar fences with left-to-right
1169  /// greedy pairing: `||x||a||y||` → `‖x‖ · a · ‖y‖`, not
1170  /// `‖x · ‖a‖ · y‖`. Nested same-meaning fences are unusual without
1171  /// explicit size cues (`\bigl\|`, parens). The forest pragma below
1172  /// prefers candidates with FEWER such nested same-fences.
1173  pub fn count_nested_same_fence(&self) -> usize {
1174    fn is_fence_meaning(m: &str) -> bool {
1175      matches!(
1176        m,
1177        "absolute-value"
1178          | "norm"
1179          | "floor"
1180          | "ceiling"
1181          | "inner-product"
1182          | "quantum-operator-product"
1183      )
1184    }
1185    // The ancestor fence is threaded as a borrowed `Option<&str>` (Copy):
1186    // the per-node `Option<String>` clones + the per-Apply `String::from`
1187    // this walk used to pay showed up in the clippy redundant-clone sweep
1188    // (2026-07-02 perf audit) — pure scoring, output-identical.
1189    fn walk<'a>(node: &'a XM, ancestor_fence: Option<&'a str>) -> usize {
1190      let meaning_of = |op: &'a XM| -> Option<&'a str> {
1191        match op {
1192          XM::Token(p, _) => p.meaning.as_deref(),
1193          _ => None,
1194        }
1195      };
1196      match node {
1197        XM::Apply(op, args, ..) => {
1198          let my_meaning = meaning_of(&op.0);
1199          let here = match (my_meaning, ancestor_fence) {
1200            (Some(m), Some(a)) if m == a && is_fence_meaning(m) => 1,
1201            _ => 0,
1202          };
1203          // Pass current meaning as ancestor if it's a fence; otherwise
1204          // keep the existing ancestor (so we detect nesting across
1205          // intermediate non-fence Applies like `times`).
1206          let new_anc = match my_meaning {
1207            Some(m) if is_fence_meaning(m) => Some(m),
1208            _ => ancestor_fence,
1209          };
1210          here + args.trees().iter().map(|a| walk(a, new_anc)).sum::<usize>()
1211        },
1212        XM::Dual(c, p, ..) => {
1213          // If the Dual's content is a fence-Apply, propagate that
1214          // meaning when walking the presentation Wrap — because the
1215          // actual nested expression lives inside the Wrap, not
1216          // inside the Ref-pointing content.
1217          let dual_fence = match &**c {
1218            XM::Apply(op_inner, ..) => match &*op_inner.0 {
1219              XM::Token(p_inner, _) => p_inner.meaning.as_deref().filter(|m| is_fence_meaning(m)),
1220              _ => None,
1221            },
1222            _ => None,
1223          };
1224          let pres_anc = dual_fence.or(ancestor_fence);
1225          walk(c, ancestor_fence) + walk(p, pres_anc)
1226        },
1227        XM::Wrap(items, ..) => items.iter().map(|i| walk(i, ancestor_fence)).sum(),
1228        XM::Choices(trees) => trees.iter().map(|t| walk(t, ancestor_fence)).sum(),
1229        XM::Arg(items) => items.iter().map(|i| walk(i, ancestor_fence)).sum(),
1230        XM::Token(..) | XM::Lexeme(..) | XM::Ref(_) => 0,
1231      }
1232    }
1233    walk(self, None)
1234  }

1344  /// Count "letter applies to vertbar-fenced argument" patterns
1345  /// that occur **inside** a `delimited-⟨⟩` (angle-fenced) Apply.
1346  /// Outside the QM/bra-ket context, the K-12 convention prevails
1347  /// (`a|b|` = `a * |b|`); this pragma must NOT bias those.
1348  ///
1349  /// The recursion descends through arbitrary parents, but only
1350  /// **counts** an `Apply(letter, [vertbar-fenced])` hit when the
1351  /// path from the tree root passed through an `Apply` whose
1352  /// operator has meaning `delimited-⟨⟩`. The flag is tracked via
1353  /// a closure parameter to avoid recomputing the ancestry.
1354  pub fn count_letter_at_vertbar(&self) -> usize {
1355    fn op_is_letter(op: &XM) -> bool {
1356      match op {
1357        XM::Token(p, _) => p.role.as_deref() == Some("UNKNOWN") || p.role.as_deref() == Some("ID"),
1358        // Lexeme-form letters: name like "UNKNOWN:..." or "ID:..."
1359        XM::Lexeme(name, _) => {
1360          let head = name.split(':').next().unwrap_or("");
1361          head == "UNKNOWN" || head == "ID"
1362        },
1363        _ => false,
1364      }
1365    }
1366    fn arg_is_vertbar_fenced(arg: &XM) -> bool {
1367      let XM::Dual(_, presentation, ..) = arg else {
1368        return false;
1369      };
1370      let XM::Wrap(ref items, ..) = **presentation else {
1371        return false;
1372      };
1373      let is_vertbar = |x: Option<&XM>| -> bool {
1374        match x {
1375          Some(XM::Token(p, _)) => {
1376            p.content.as_deref() == Some("|") || p.content.as_deref() == Some("‖")
1377          },
1378          Some(XM::Lexeme(name, _)) => {
1379            name.starts_with("OPEN:|:")
1380              || name.starts_with("CLOSE:|:")
1381              || name.starts_with("OPEN:‖:")
1382              || name.starts_with("CLOSE:‖:")
1383              || &**name == "VERTBAR"
1384              || name.starts_with("VERTBAR:")
1385          },
1386          _ => false,
1387        }
1388      };
1389      is_vertbar(items.first()) && is_vertbar(items.last())
1390    }
1391    fn is_delim_angle_op(op: &XM) -> bool {
1392      match op {
1393        XM::Token(p, _) => p.meaning.as_deref() == Some("delimited-⟨⟩"),
1394        _ => false,
1395      }
1396    }
1397    // Recognise "bra-ket-by-context": a multirelation Apply whose
1398    // **boundary** arguments (first and last) are both `absent`,
1399    // e.g. `absent < a |f| b > absent` produced by plain `<a|f|b>`.
1400    // In that surrounding shape, the same QM convention applies as
1401    // for explicit `\langle...\rangle`.
1402    fn is_qm_multirelation_apply(op: &XM, args: &Args) -> bool {
1403      let meaning_is_multirelation = match op {
1404        XM::Token(p, _) => p.meaning.as_deref() == Some("multirelation"),
1405        _ => false,
1406      };
1407      if !meaning_is_multirelation {
1408        return false;
1409      }
1410      let trees = args.trees();
1411      let is_absent = |x: Option<&&XM>| -> bool {
1412        matches!(x,
1413          Some(XM::Token(p, _)) if p.meaning.as_deref() == Some("absent"))
1414      };
1415      is_absent(trees.first()) && is_absent(trees.last())
1416    }
1417    fn walk(
1418      node: &XM,
1419      inside_angle: bool,
1420      op_is_letter: &impl Fn(&XM) -> bool,
1421      arg_is_vertbar_fenced: &impl Fn(&XM) -> bool,
1422    ) -> usize {
1423      match node {
1424        XM::Token(..) | XM::Lexeme(..) | XM::Ref(_) => 0,
1425        XM::Apply(op, args, ..) => {
1426          let trees = args.trees();
1427          let here = if inside_angle
1428            && op_is_letter(&op.0)
1429            && trees.len() == 1
1430            && trees.first().is_some_and(|a| arg_is_vertbar_fenced(a))
1431          {
1432            1
1433          } else {
1434            0
1435          };
1436          // Recurse with `inside_angle` set true if THIS Apply is
1437          // a delimited-⟨⟩ OR a QM-style multirelation (absent < … >
1438          // absent), both of which signal bra-ket context.
1439          let child_inside =
1440            inside_angle || is_delim_angle_op(&op.0) || is_qm_multirelation_apply(&op.0, args);
1441          here
1442            + args
1443              .trees()
1444              .iter()
1445              .map(|a| walk(a, child_inside, op_is_letter, arg_is_vertbar_fenced))
1446              .sum::<usize>()
1447        },
1448        // Dual content/presentation: the Dual's content typically
1449        // holds the operator Apply (e.g. `Apply(delim-⟨⟩, [Ref])`)
1450        // while the presentation Wrap holds the actual body. So if
1451        // the content marks a delim-⟨⟩ wrapper, the presentation
1452        // body should be walked with `inside_angle=true`.
1453        XM::Dual(c, p, ..) => {
1454          let this_is_angle = matches!(&**c,
1455            XM::Apply(op, _, _, _) if is_delim_angle_op(&op.0));
1456          let pres_inside = inside_angle || this_is_angle;
1457          walk(c, inside_angle, op_is_letter, arg_is_vertbar_fenced)
1458            + walk(p, pres_inside, op_is_letter, arg_is_vertbar_fenced)
1459        },
1460        XM::Wrap(items, ..) => items
1461          .iter()
1462          .map(|i| walk(i, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1463          .sum(),
1464        XM::Choices(trees) => trees
1465          .iter()
1466          .map(|t| walk(t, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1467          .sum(),
1468        XM::Arg(items) => items
1469          .iter()
1470          .map(|i| walk(i, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1471          .sum(),
1472      }
1473    }
1474    walk(self, false, &op_is_letter, &arg_is_vertbar_fenced)
1475  }

1634  /// claimed as one big absolute-value enclosing the rest.
1635  pub fn prefer_wider_addition_root(self) -> Self {
1636    fn args_count_if_addition_root(t: &XM) -> Option<usize> {
1637      match t {
1638        XM::Apply(op, args, ..) => match &*op.0 {
1639          XM::Token(p, _) => match p.meaning.as_deref() {
1640            Some("plus") | Some("minus") | Some("plus-or-minus") | Some("minus-or-plus") => {
1641              Some(args.trees().len())
1642            },
1643            _ => None,
1644          },
1645          XM::Lexeme(lex, _) => {
1646            let head = lex.split(':').next().unwrap_or("");
1647            if head == "ADDOP" {
1648              Some(args.trees().len())
1649            } else {
1650              None
1651            }
1652          },
1653          _ => None,
1654        },
1655        XM::Dual(c, ..) => args_count_if_addition_root(c),
1656        _ => None,
1657      }
1658    }
1659    match self {
1660      XM::Choices(trees) if trees.len() > 1 => {
1661        let max = trees.iter().filter_map(args_count_if_addition_root).max();
1662        let Some(max) = max else {
1663          return XM::Choices(trees);
1664        };
1665        if max <= 2 {
1666          return XM::Choices(trees);
1667        }
1668        let kept: Vec<XM> = trees
1669          .into_iter()
1670          .filter(|t| {
1671            args_count_if_addition_root(t)
1672              .map(|n| n == max)
1673              .unwrap_or(false)
1674              || args_count_if_addition_root(t).is_none()
1675          })
1676          .collect();
1677        match kept.len() {
1678          0 => XM::Choices(Vec::new()),
1679          1 => kept.into_iter().next().unwrap(),
1680          _ => XM::Choices(kept),
1681        }
1682      },
1683      other => other,
1684    }
1685  }

1687  /// Forest pragma: when at least one candidate roots at an
1688  /// additive operator (`+` / `-`) and another roots at a
1689  /// multiplicative operator (`*` / invisible-times), prefer the
1690  /// additive root. K-12 algebra: `a*|a| + b*|b| + c*|c|` parses
1691  /// with `+` outermost, not as `a * (one giant absolute-value)`.
1692  ///
1693  /// Fires only when the discriminator is unambiguous (at least
1694  /// one of each shape exists); otherwise inert.
1695  pub fn prefer_root_addition_over_outer_multiplication(self) -> Self {
1696    match self {
1697      XM::Choices(trees) if trees.len() > 1 => {
1698        let has_addition_root = trees.iter().any(|t| t.root_is_addition());
1699        if !has_addition_root {
1700          return XM::Choices(trees);
1701        }
1702        let kept: Vec<XM> = trees.into_iter().filter(|t| t.root_is_addition()).collect();
1703        match kept.len() {
1704          0 => XM::Choices(Vec::new()),
1705          1 => kept.into_iter().next().unwrap(),
1706          _ => XM::Choices(kept),
1707        }
1708      },
1709      other => other,
1710    }
1711  }

1929  /// Rebuild a marpa-derived parse tree into an XMath XML tree
1930  pub fn into_xmath(
1931    self,
1932    owner: &mut Node,
1933    nodes: &mut [Node],
1934    document: &mut Document,
1935  ) -> Result<Node, Box<dyn Error + Send + Sync>> {
1936    // Grow the call stack on demand if we are running low. Heavily
1937    // nested math trees (XMApp(op, [XMApp(...)]) chains) recurse
1938    // through this function at depth proportional to expression
1939    // nesting; on Rust's 8 MB main-thread default stack that overflows
1940    // on grammar-ambiguous papers (sandbox 0711.4787, 0903.3289,
1941    // hep-th0101151, math0505371, math9204211, hep-ph9210253,
1942    // hep-ph9512208, astro-ph0612758 — 8 papers, SIGABRT with
1943    // `thread 'main' has overflowed its stack`). The growth guard
1944    // allocates a new stack chunk when the current frame's remaining
1945    // space drops below the red zone; its params are configurable in
1946    // `latexml_core::stack_guard`.
1947    latexml_core::stack_guard::maybe_grow(|| self.into_xmath_inner(owner, nodes, document))
1948  }
1949
1950  fn into_xmath_inner(
1951    self,
1952    owner: &mut Node,
1953    nodes: &mut [Node],
1954    document: &mut Document,
1955  ) -> Result<Node, Box<dyn Error + Send + Sync>> {
1956    // Drain nested XM::Choices iteratively before the match. Each
1957    // Choices wrapper is a tail call into the first remaining choice
1958    // — preserving the per-layer Info log so the discard count
1959    // is visible identically to the prior recursive form.
1960    let mut tree = self;
1961    loop {
1962      match tree {
1963        XM::Choices(mut choices) => {
1964          Debug!(
1965            "math_parser",
1966            "choices",
1967            format!(
1968              "to_xmath handler discarded {} parse choices.",
1969              choices.len() - 1
1970            )
1971          );
1972          tree = choices.remove(0);
1973        },
1974        other => {
1975          tree = other;
1976          break;
1977        },
1978      }
1979    }
1980    match tree {
1981      XM::Lexeme(content, _meta) => {
1982        let id = content
1983          .split(':')
1984          .next_back()
1985          .unwrap()
1986          .parse::<usize>()
1987          .unwrap()
1988          - 1;
1989        let atom_node = &mut nodes[id];
1990        atom_node.unbind();
1991        Ok(atom_node.clone())
1992      },
1993      XM::Token(props, _meta) => {
1994        // Transition the {font} property to the "_font" attribute.
1995        let has_explicit_font = props.font.is_some();
1996        let (content_opt, font, attrs) = props.into_attributes();
1997        let mut xmtok = document.open_element_at(owner, "ltx:XMTok", attrs, font)?;
1998        if let Some(ref content) = content_opt
1999          && !content.is_empty()
2000        {
2001          xmtok.set_content(content)?;
2002        }
2003        // Perl: Font->specialize($content) — for parser-created tokens without
2004        // explicit font, the ambient _font is specialized based on content.
2005        // Operators get default font (no italic), letters get italic.
2006        if !has_explicit_font && let Some(font_hash) = xmtok.get_attribute("_font") {
2007          let content = content_opt.as_deref().unwrap_or("");
2008          if content.is_empty() {
2009            // Empty content: no font needed
2010            let _ = xmtok.remove_attribute("_font");
2011          } else if let Some(font) = document.decode_font(&font_hash) {
2012            let specialized = font.specialize(content);
2013            // Re-encode: store the specialized font and update _font hash
2014            document.set_node_font(&mut xmtok, &specialized)?;
2015          }
2016        }
2017        document.close_element_at(&mut xmtok)?;
2018        Ok(xmtok)
2019      },
2020      XM::Apply(op, args, props, _meta) => {
2021        // let mut apply_node = Node::new("XMApp", None, document.get_document()).unwrap();
2022        // props.into_xmath(&mut apply_node,document)?;
2023        let (_, font, attrs) = props.into_attributes();
2024        let mut apply_node = document.open_element_at(owner, "ltx:XMApp", attrs, font)?;
2025        let mut op_node = op.0.into_xmath(&mut apply_node, nodes, document)?;
2026
2027        add_child_guard_xmarg(&mut apply_node, &mut op_node)?;
2028        for arg in args.0.into_iter().flatten() {
2029          let mut arg_node = arg.into_xmath(&mut apply_node, nodes, document)?;
2030          add_child_guard_xmarg(&mut apply_node, &mut arg_node)?;
2031        }
2032        document.close_element_at(&mut apply_node)?;
2033        Ok(apply_node)
2034      },
2035      XM::Dual(content, pres, props, _meta) => {
2036        let (_, font, attrs) = props.into_attributes();
2037        let mut dual_node = document.open_element_at(owner, "ltx:XMDual", attrs, font)?;
2038        // Content branch first, then presentation (Perl convention)
2039        let mut content_node = content.into_xmath(&mut dual_node, nodes, document)?;
2040        add_child_guard_xmarg(&mut dual_node, &mut content_node)?;
2041        let mut pres_node = pres.into_xmath(&mut dual_node, nodes, document)?;
2042        add_child_guard_xmarg(&mut dual_node, &mut pres_node)?;
2043        document.close_element_at(&mut dual_node)?;
2044        Ok(dual_node)
2045      },
2046      XM::Wrap(content, props, _meta) => {
2047        let (_, font, attrs) = props.into_attributes();
2048        let mut wrap_node = document.open_element_at(owner, "ltx:XMWrap", attrs, font)?;
2049        for c in content.into_iter() {
2050          let mut content_node = c.into_xmath(&mut wrap_node, nodes, document)?;
2051          add_child_guard_xmarg(&mut wrap_node, &mut content_node)?;
2052        }
2053        document.close_element_at(&mut wrap_node)?;
2054        Ok(wrap_node)
2055      },
2056      XM::Ref(refprops) => {
2057        let mut ref_node = Node::new("XMRef", None, document.get_document()).unwrap();
2058        if let Some(id) = refprops.id {
2059          document.set_attribute(&mut ref_node, "idref", &id)?;
2060        }
2061        if let Some(xmkey) = refprops.xmkey {
2062          // Use _pxmkey for parser-generated keys (pxm prefix) to avoid
2063          // conflicting with base_xmath's \lx@dual resolver.
2064          // Regular _xmkey for all other refs.
2065          let attr_name = if xmkey.starts_with("pxm") {
2066            "_pxmkey"
2067          } else {
2068            "_xmkey"
2069          };
2070          document.set_attribute(&mut ref_node, attr_name, &xmkey)?;
2071        }
2072        Ok(ref_node)
2073      },
2074      XM::Arg(inner_list) => {
2075        let mut arg_node = Node::new("XMArg", None, document.get_document()).unwrap();
2076        for inner_item in inner_list {
2077          let mut inner_node = inner_item.into_xmath(&mut arg_node, nodes, document)?;
2078          add_child_guard_xmarg(&mut arg_node, &mut inner_node)?;
2079        }
2080        Ok(arg_node)
2081      },
2082      // XM::Choices was already drained iteratively above the match.
2083      XM::Choices(_) => unreachable!("Choices drained before match"),
2084    }
2085  }

2190  // Lex strings are produced internally by the grammar (`{name}:{value}:{idx}`
2191  // shape), so the suffix parses cleanly under normal operation. If we ever
2192  // hit a malformed lex — e.g. because a custom rule emitted a short form —
2193  // return an Err rather than panicking, matching the degrade-not-panic
2194  // policy used elsewhere on user-reachable paths.
2195  let idx_str = lex.split(':').next_back().unwrap_or("");
2196  let node_idx = idx_str
2197    .parse::<usize>()
2198    .map_err(|e| format!("malformed lex {lex:?}: {e}"))?
2199    .checked_sub(1)
2200    .ok_or_else(|| format!("lex idx 0 (expected 1-based) in {lex:?}"))?;
2201  nodes.get(node_idx).ok_or_else(|| {
2202    format!(
2203      "lex idx {node_idx} out of range (nodes.len={}) in {lex:?}",
2204      nodes.len()
2205    )
2206    .into()
2207  })
2208}
