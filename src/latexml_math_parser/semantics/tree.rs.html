63/// Custom PartialEq: ignores `xmkey`, `id`, `idref`, and `scriptpos` which are
64/// bookkeeping/layout fields. Structurally identical parse trees that differ only
65/// in internal reference keys or script-position labels (pre1 vs pre2 vs post1)
66/// should be considered equal for deduplication purposes.
67/// `scriptpos` is excluded because different grammar paths produce different
68/// pre/post level assignments for the same mathematical expression. E.g.,
69/// `{}^4{}_{12}C^{5+}` can produce 27 structurally distinct trees that differ
70/// only in scriptpos values — all represent the same expression.
71impl PartialEq for XProps {
72  fn eq(&self, other: &Self) -> bool {
73    self.content == other.content
74      && self.role == other.role
75      && self.meaning == other.meaning
76      && self.name == other.name
77      // Skip: scriptpos — layout hint, not semantic distinction
78      // Skip: id, idref, xmkey — bookkeeping for Dual/Ref resolution
79      && self.font == other.font
80      && self.fontref == other.fontref
81      && self.stretchy == other.stretchy
82      && self.possible_function == other.possible_function
83      && self.mathstyle == other.mathstyle
84      && self.thickness == other.thickness
85      && self.decl_id == other.decl_id
86      && self.lpadding == other.lpadding
87      && self.rpadding == other.rpadding
88  }
89}

159/// The math parsing process can manipulate a variety of trees,
160/// finally serialized via the XMath schema of LaTeXML.
161///
162/// The main structural variants are associated with
163/// a "parsing state::, via an attached `Meta` object.
164#[derive(Debug, Clone, PartialEq, Eq)]
165pub enum XM {
166  /// Token-name lexeme (e.g. "RELOP:less-than:3", "letter:a",
167  /// "delimited-open-paren"). The name is stored as `Rc<str>` so:
168  ///
169  /// 1. **Byte-glade hot path** (asf_traverser case 1): each ASCII
170  ///    byte resolves to a cached `Rc<str>` via `byte_lexeme_rc(b)`.
171  ///    Clones are refcount bumps — no allocation per byte glade.
172  /// 2. **Marpa ASF cache clones** (`asf.rs:156, 208`): cloning a
173  ///    `ParseTree = Vec<Option<XM>>` no longer deep-clones the
174  ///    lexeme name. Refcount-bump per Lexeme in the Vec.
175  /// 3. **Read sites** (`name.starts_with(...)`, `name == "..."`,
176  ///    `name.split(...)`, `name.contains(...)`): unchanged — `Rc<str>`
177  ///    derefs to `&str`.
178  ///
179  /// Construction at runtime: `Rc::from(string)` / `Rc::from("…")`.
180  Lexeme(Rc<str>, Meta),
181  Token(XProps, Meta), // does this need Meta?
182  Apply(Operator, Args, XProps, Meta),
183  Dual(Box<XM>, Box<XM>, XProps, Meta),
184  Ref(XProps),
185  Wrap(Vec<XM>, XProps, Meta),
186  Arg(Vec<XM>),
187  Choices(Vec<XM>),
188}
189impl From<XProps> for XM {
190  fn from(t: XProps) -> Self { XM::Token(t, Meta::default()) }
191}

248  fn fmt_indented(&self, level: &[bool], f: &mut fmt::Formatter<'_>) -> fmt::Result {
249    let mut peekable = self.0.iter().peekable();
250    while let Some(maybe_arg) = peekable.next() {
251      if peekable.peek().is_some() {
252        maybe_arg
253          .as_ref()
254          .unwrap_or(&XM::Lexeme(
255            Rc::from("missing_argument"),
256            Meta::default(),
257          ))
258          .fmt_indented(level, f)?;
259      } else {
260        let mut last_level: Vec<bool> = level.to_vec();
261        if !last_level.is_empty() {
262          last_level.pop();
263          last_level.push(false);
264        }
265        maybe_arg
266          .as_ref()
267          .unwrap_or(&XM::Lexeme(
268            Rc::from("missing_argument"),
269            Meta::default(),
270          ))
271          .fmt_indented(&last_level, f)?;
272      };
273    }
274    Ok(())
275  }

392  /// Specialize a tree with the given Meta object,
393  /// also verifying the tree's inner consistency.
394  ///
395  /// Whenever a contradiction/inconsistency is detected, we return an error
396  /// This method should always be called on tree construction, as it also manages the various curry
397  /// constraints, keeping the resolution local / fast.
398  pub fn specialize(
399    self,
400    mut into: Meta,
401    pragmas: &[ValidationPragmatics],
402  ) -> Result<Self, Box<dyn Error>> {
403    match self {
404      XM::Lexeme(name, meta) => {
405        let new_meta = meta.with_curry_atom(into, &name)?;
406        Ok(XM::Lexeme(name, new_meta))
407      },
408      XM::Token(t, meta) => {
409        // Specialization of bare Token variants isn't exercised by current
410        // grammar rules. Return the tree unchanged so a future rule that
411        // invokes this path fails at a higher layer (validation) rather
412        // than panicking here.
413        Ok(XM::Token(t, meta))
414      },
415      XM::Ref(_) => Ok(self),
416      XM::Apply(mut op, mut args, props, meta) => {
417        // First, if we have a specialize directive, execute it:
418        match into.specialize {
419          Some(ref directive)
420            if directive == "embellish"
421            // Atoms with embellishments should get their curry levels renamed
422            // to avoid conflicts with the same atoms *without* the embellishments
423            // as often this technique is used to generate new unique names.
424            && args.0.len() <= 2 =>
425          {
426            if let Some(XM::Lexeme(_, arg_meta)) = &mut args.0[0] {
427              if let Some(CurryTerm::Var(ref mut curry_var)) = arg_meta.curry_level {
428                let mut base_op = op.0.base_operator_name();
429                // fish out a local name to use as an embellishment
430                if let Some(last_colon_idx) = base_op.rfind(':') {
431                  base_op.replace_range(..=last_colon_idx, "");
432                } else if let Some(last_dot_idx) = base_op.rfind('.') {
433                  base_op.replace_range(..=last_dot_idx, "");
434                }
435                if base_op.is_empty() {
436                  base_op = String::from("embellished");
437                }
438                curry_var.push('-');
439                curry_var.push_str(&base_op);
440              }
441              into.curry_level.clone_from(&arg_meta.curry_level)
442            }
443          },
444          _ => {},
445        }
446        // Next, we validate the constraints.
447        let initial_constraint_count = meta.curry_constraints.len();
448        let mut new_meta = meta;
449        // SPECIAL CASE (of course): if we are using an "unconstrained" directive, no need to do any
450        // of this
451        if into.specialize.as_deref() == Some("unconstrained") {
452          op.unconstrain_recursive();
453          args.unconstrain_recursive();
454          new_meta = new_meta.with(into)?;
455          new_meta.curry_constraints = CurryConstraints::new();
456          new_meta.specialize = None;
457        } else {
458          new_meta = new_meta.with(into)?;
459          // COPY constraints from inner subtrees to the new apply root
460          for constraint in op.get_constraints().into_iter() {
461            new_meta.curry_constraints.insert(constraint.clone());
462          }
463          for constraint in args.get_constraints().into_iter() {
464            new_meta.curry_constraints.insert(constraint.clone());
465          }
466          // and require the current curry level to be >= 1 if compound
467          // (otherwise, it's already built in)
468          if let Some(ref expr) = new_meta.curry_level {
469            match expr {
470              CurryTerm::Add(..) | CurryTerm::Sub(..) => {
471                new_meta.curry_constraints.insert(CurryConstraint((
472                  expr.clone(),
473                  Ordering::Greater,
474                  CurryTerm::Literal(0),
475                )));
476              },
477              _ => {},
478            }
479          }
480        }
481        if initial_constraint_count < new_meta.curry_constraints.len() {
482          // whenever we add a constraint, re-valdiate the expression, and prune it if needed
483          // println!("Tree: \n{}", XM::Apply(op.clone(), args.clone(), new_meta.clone()));
484          new_meta.validate()?;
485        }
486        let new_tree = XM::Apply(op, args, props, new_meta);
487        for pragma in pragmas {
488          // expert pragmatics get to validate each new tree,
489          // in order to prune wrong interpretations as early as possible
490          pragma.validate(&new_tree)?;
491        }
492        Ok(new_tree)
493      },
494      // Dual/Wrap/Arg variants aren't specialized by current grammar rules
495      // — return unchanged rather than panic. Future grammar extensions can
496      // add real specialization rules if needed.
497      dual @ XM::Dual(..) => Ok(dual),
498      wrap @ XM::Wrap(..) => Ok(wrap),
499      arg @ XM::Arg(_) => Ok(arg),
500      XM::Choices(_) => Err("can not specialize choices".into()),
501    }
502  }

504  /// Prunes choices based on a validation pass leveraging a choice of pragmatics
505  /// if the pruning arrives at no viable trees at all, the original tree is returned,
506  /// hence the "soft" function name.
507  /// These are executed at the end of the program, so need to be invoked recursively on each
508  /// subtree
509  pub fn soft_prune_choices(self, pragmatics: ValidationPragmatics) -> Self {
510    match self {
511      XM::Choices(trees) => {
512        let (consistent_trees, inconsistent_trees): (Vec<XM>, Vec<XM>) = trees
513          .into_iter()
514          .partition(|tree| pragmatics.validate_recursive(tree).is_ok());
515        match consistent_trees.len() {
516          0 => XM::Choices(inconsistent_trees),
517          1 => consistent_trees.into_iter().next().unwrap(),
518          _more => XM::Choices(consistent_trees),
519        }
520      },
521      other => other,
522    }
523  }

525  /// Count the number of `absent` markers in this tree.
526  ///
527  /// `absent` appears as a sentinel `XM::Token` (with
528  /// `meaning="absent"`) when a parse has to fill in a missing
529  /// operand position — e.g. `< x >` parsed as the relational
530  /// chain `absent < x > absent`. Fewer `absent` markers signals
531  /// a parse that satisfies the grammar without needing fillers.
532  pub fn count_absent(&self) -> usize {
533    match self {
534      XM::Token(props, _) => {
535        if props.meaning.as_deref() == Some("absent") {
536          1
537        } else {
538          0
539        }
540      },
541      XM::Apply(op, args, ..) => op.0.count_absent() + args.trees().iter().map(|a| a.count_absent()).sum::<usize>(),
542      XM::Dual(c, p, ..) => c.count_absent() + p.count_absent(),
543      XM::Wrap(items, ..) => items.iter().map(|i| i.count_absent()).sum(),
544      XM::Choices(trees) => trees.iter().map(|t| t.count_absent()).sum(),
545      XM::Arg(items) => items.iter().map(|i| i.count_absent()).sum(),
546      XM::Lexeme(_, _) | XM::Ref(_) => 0,
547    }
548  }

550  /// Semantic-node count used by parse-ranking pragmas to prefer
551  /// compact interpretations over verbose ones. Not a general
552  /// "tree size" function — it deliberately ignores presentation
553  /// duplication so structurally-equivalent parses get a fair
554  /// comparison.
555  ///
556  /// Smaller wins: `norm@(a)` (2 nodes) beats
557  /// `absolute-value@(absolute-value@(a))` (3 nodes);
558  /// `differential-d@(x)` (2 nodes) beats `d*x` (3 nodes).
559  ///
560  /// **Counting conventions**:
561  /// * Each `XM::Apply` contributes ONE node — the operator is part
562  ///   of the Apply's identity, not a separate child. So `f@(x)`
563  ///   is 2 nodes (the Apply + x), not 3.
564  /// * `XM::Dual(content, presentation)` counts **only the content
565  ///   tree** — the presentation branch is a parallel rendering of
566  ///   the same semantics and contributes the same count, so
567  ///   double-counting would inflate purely-cosmetic siblings.
568  /// * `XM::Ref(props)` is resolved to its target via the
569  ///   presentation-branch index built at the Dual boundary —
570  ///   so a Ref pointing to a deep sub-tree contributes its full
571  ///   target's node count, not just 1. This keeps the ranking
572  ///   honest when one parse uses a single Ref to a complex node
573  ///   versus another that lays out multiple Refs to leaves.
574  pub fn count_nodes_for_parse_ranking(&self) -> usize {
575    let mut index: HashMap<String, &XM> = HashMap::default();
576    self.build_ref_index(&mut index);
577    self.count_nodes_with_index(&index, &mut Vec::new())
578  }

580  /// Walk this tree and register every `(id, &XM)` pair that a
581  /// `Ref` might lookup. Refs key on `props.id` or `props.xmkey`.
582  fn build_ref_index<'a>(&'a self, out: &mut HashMap<String, &'a XM>) {
583    let take = |p: &'a XProps| -> Option<String> {
584      p.id.as_ref().map(|c| c.to_string()).or_else(|| p.xmkey.as_ref().map(|c| c.to_string()))
585    };
586    match self {
587      XM::Token(props, _) => {
588        if let Some(k) = take(props) {
589          out.entry(k).or_insert(self);
590        }
591      },
592      XM::Apply(_op, args, props, _) => {
593        if let Some(k) = take(props) {
594          out.entry(k).or_insert(self);
595        }
596        for arg in args.trees() {
597          arg.build_ref_index(out);
598        }
599      },
600      XM::Dual(c, p, props, _) => {
601        if let Some(k) = take(props) {
602          out.entry(k).or_insert(self);
603        }
604        c.build_ref_index(out);
605        p.build_ref_index(out);
606      },
607      XM::Wrap(items, props, _) => {
608        if let Some(k) = take(props) {
609          out.entry(k).or_insert(self);
610        }
611        for it in items {
612          it.build_ref_index(out);
613        }
614      },
615      XM::Arg(items) | XM::Choices(items) => {
616        for it in items {
617          it.build_ref_index(out);
618        }
619      },
620      XM::Lexeme(_, _) | XM::Ref(_) => {},
621    }
622  }

624  /// Counting worker with the Ref index. `visited` is a stack of
625  /// idref/xmkey strings currently being resolved — guards against
626  /// cyclic references (defensive; idref graphs in valid XMath
627  /// are acyclic).
628  fn count_nodes_with_index(&self, index: &HashMap<String, &XM>, visited: &mut Vec<String>) -> usize {
629    match self {
630      XM::Token(_, _) | XM::Lexeme(_, _) => 1,
631      XM::Apply(_op, args, ..) => 1 + args.trees().iter().map(|a| a.count_nodes_with_index(index, visited)).sum::<usize>(),
632      XM::Dual(c, _p, ..) => c.count_nodes_with_index(index, visited),
633      XM::Wrap(items, ..) => 1 + items.iter().map(|i| i.count_nodes_with_index(index, visited)).sum::<usize>(),
634      XM::Choices(trees) => trees.iter().map(|t| t.count_nodes_with_index(index, visited)).sum(),
635      XM::Arg(items) => 1 + items.iter().map(|i| i.count_nodes_with_index(index, visited)).sum::<usize>(),
636      XM::Ref(props) => {
637        let key = props.idref.as_ref().map(|c| c.to_string()).or_else(|| props.xmkey.as_ref().map(|c| c.to_string()));
638        match key {
639          Some(k) if !visited.contains(&k) => match index.get(&k) {
640            Some(target) => {
641              visited.push(k);
642              let n = target.count_nodes_with_index(index, visited);
643              visited.pop();
644              n
645            },
646            None => 1,
647          },
648          // Already visiting this ref (cycle) or no key at all —
649          // count as 1 so we don't double-count or infinite-loop.
650          _ => 1,
651        }
652      },
653    }
654  }

779  /// Does the root Dual's **presentation** Wrap contain exactly one
780  /// non-delimiter child that is itself a `Dual` whose content
781  /// shares the same operator meaning as the outer? This is the
782  /// shape that produces `set@(set@(…))` / `vector@(vector@(…))`
783  /// when rendered — the outer content's Ref resolves to the inner
784  /// Dual instead of to flat items.
785  fn root_dual_has_redundant_inner_wrap(&self) -> bool {
786    let XM::Dual(content, presentation, _, _) = self else {
787      return false;
788    };
789    let outer_meaning = match &**content {
790      XM::Apply(Operator(op), ..) => match &**op {
791        XM::Token(props, _) => props.meaning.as_deref().map(String::from),
792        XM::Lexeme(name, _) => Some(name.to_string()),
793        _ => None,
794      },
795      _ => return false,
796    };
797    let Some(outer_m) = outer_meaning else { return false };
798    let XM::Wrap(items, _, _) = &**presentation else {
799      return false;
800    };
801    let is_delim = |x: &XM| match x {
802      XM::Token(p, _) => matches!(p.role.as_deref(), Some("OPEN") | Some("CLOSE")),
803      XM::Lexeme(n, _) => n.starts_with("OPEN:") || n.starts_with("CLOSE:"),
804      _ => false,
805    };
806    let inner: Vec<&XM> = items.iter().filter(|x| !is_delim(x)).collect();
807    if inner.len() != 1 {
808      return false;
809    }
810    let XM::Dual(inner_content, ..) = inner[0] else {
811      return false;
812    };
813    // Inner Dual's content should be an Apply whose meaning matches
814    // outer (or is a closely-related "list"/"vector"/"formulae"
815    // meaning — common when `interpret_delimited` lifts the list
816    // wrapper above an inner Dual).
817    if let XM::Apply(Operator(inner_op), ..) = &**inner_content {
818      let inner_meaning = match &**inner_op {
819        XM::Token(props, _) => props.meaning.as_deref(),
820        XM::Lexeme(name, _) => Some(&**name),
821        _ => None,
822      };
823      // Outer-set wrapping inner-{set/list/vector/formulae} is the
824      // shape we want to prune. The legacy never picks this.
825      let inner_str = inner_meaning.unwrap_or("");
826      if outer_m == inner_str
827        || (matches!(outer_m.as_str(), "set" | "vector")
828          && matches!(inner_str, "set" | "vector" | "list" | "formulae"))
829      {
830        return true;
831      }
832    }
833    false
834  }

837  /// the Apply's first child (after following any `XM::Ref` through
838  /// the presentation branch's idref index) is ALSO an `Apply` with
839  /// the same `op_meaning`? Used to detect redundant
840  /// `set@(set@(...))` and similar self-wrapping shapes.
841  fn root_dual_is_redundant_self_wrap(&self) -> bool {
842    let XM::Dual(content, _, _, _) = self else {
843      return false;
844    };
845    let outer_meaning = match &**content {
846      XM::Apply(Operator(op), args, ..) if args.trees().len() == 1 => {
847        let m = match &**op {
848          XM::Token(props, _) => props.meaning.as_deref().map(String::from),
849          XM::Lexeme(name, _) => Some(name.to_string()),
850          _ => None,
851        };
852        m
853      },
854      _ => return false,
855    };
856    let Some(outer_m) = outer_meaning else {
857      return false;
858    };
859    // Resolve the inner: dereference Apply's single child. If it's
860    // a Ref, follow it through the idref index built across the
861    // whole Dual.
862    let mut index: HashMap<String, &XM> = HashMap::default();
863    self.build_ref_index(&mut index);
864    if let XM::Apply(_, args, _, _) = &**content {
865      let first_arg = args.trees().first().copied();

867        Some(XM::Ref(props)) => {
868          let key = props
869            .idref
870            .as_ref()
871            .map(|c| c.to_string())
872            .or_else(|| props.xmkey.as_ref().map(|c| c.to_string()));

874        },
875        other => other,
876      };

878        let inner_meaning = match &**inner_op {
879          XM::Token(props, _) => props.meaning.as_deref(),
880          XM::Lexeme(name, _) => Some(&**name),
881          _ => None,
882        };
883        return inner_meaning == Some(outer_m.as_str());
884      }
885    }
886    false
887  }

919  /// Does the root match `Apply(multirelation, [..., absent, ...])`
920  /// or `Apply(formulae, [..., absent, ...])` where `absent`
921  /// appears in the **interior** of the args list (NOT at the
922  /// first or last positions)?
923  ///
924  /// The distinction matters: an `absent` at the boundary is
925  /// legitimate (e.g. `<a|f|b>` parses as
926  /// `multirelation(absent, <, a, |, f, |, b, >, absent)` — the
927  /// outer `<` and `>` need left/right operands, which are absent
928  /// because the expression IS the whole math). An `absent` in the
929  /// middle (e.g. `x >= 0` parsed as `multirelation(x, >, absent,
930  /// =, 0)`) signals a failed `two_part_relop` combination — the
931  /// alternative parse using a combined `>=` operator is strictly
932  /// better.
933  fn root_is_multirelation_with_interior_absent(&self) -> bool {
934    let apply = match self {
935      XM::Apply(..) => self,
936      XM::Dual(content, _, _, _) => &**content,
937      _ => return false,
938    };
939    let XM::Apply(Operator(op), args, ..) = apply else {
940      return false;
941    };
942    let meaning = match &**op {
943      XM::Token(props, _) => props.meaning.as_deref(),
944      XM::Lexeme(name, _) => Some(&**name),
945      _ => None,
946    };
947    // Both `multirelation` (explicit relation chain) and `formulae`
948    // (comma-separated relational chain) can fall into this pattern.
949    if !matches!(meaning, Some("multirelation") | Some("formulae")) {
950      return false;
951    }
952    let trees = args.trees();
953    if trees.len() < 3 {
954      return false;
955    }
956    let is_absent = |a: &&XM| matches!(a, XM::Token(p, _) if p.meaning.as_deref() == Some("absent"));
957    // Skip the first and last positions; only inspect interior.
958    trees[1..trees.len() - 1].iter().any(is_absent)
959  }

1069  /// Forest pragma: among surviving candidates, prefer those with
1070  /// MORE `delimited-X` Apply nodes. Fires only when the maximum
1071  /// is strictly positive (i.e., at least one candidate recognized
1072  /// a fence) AND another candidate has a strictly smaller count.
1073  ///
1074  /// **Resolves**:
1075  /// * `2<x,y>=z` → `2 * delimited-<>@(list(x,y)) = z` instead of
1076  ///   `formulae@(2 < x, y >= z)` (ambiguous_relations).
1077  /// * `0<<a,b>>1` → multirelation around `delimited-<>` instead of
1078  ///   `formulae@(0 << a, b >> 1)`.
1079  /// * `<a|f|b>` style bra-kets — angle-fence reading wins over
1080  ///   flat formula chain.
1081  ///
1082  /// **Why this is sound**: the grammar admits `delimited-X` only
1083  /// when there's a balanced pair of fence tokens AND structured
1084  /// content inside (e.g. `term_list`). When this rule fires, the
1085  /// fence pair was *intentional* in the input; preferring the
1086  /// fenced reading respects the author's notation.

1108  /// Count Apply nodes whose meaning is a **fence operator**
1109  /// (`norm`, `absolute-value`, `floor`, `ceiling`, etc.) AND that
1110  /// have an ancestor Apply with the **same** fence meaning. This
1111  /// is the "nested same-fence" count: e.g. `‖x · ‖a‖ · y‖` has one
1112  /// `norm` inside another `norm`, while `‖x‖ · a · ‖y‖` has none.
1113  ///
1114  /// Mathematicians read consecutive bar fences with left-to-right
1115  /// greedy pairing: `||x||a||y||` → `‖x‖ · a · ‖y‖`, not
1116  /// `‖x · ‖a‖ · y‖`. Nested same-meaning fences are unusual without
1117  /// explicit size cues (`\bigl\|`, parens). The forest pragma below
1118  /// prefers candidates with FEWER such nested same-fences.
1119  pub fn count_nested_same_fence(&self) -> usize {
1120    fn is_fence_meaning(m: &str) -> bool {
1121      matches!(
1122        m,
1123        "absolute-value"
1124          | "norm"
1125          | "floor"
1126          | "ceiling"
1127          | "inner-product"
1128          | "quantum-operator-product"
1129      )
1130    }
1131    fn walk(node: &XM, ancestor_fence: Option<String>) -> usize {
1132      let meaning_of = |op: &XM| -> Option<String> {
1133        match op {
1134          XM::Token(p, _) => p.meaning.as_deref().map(String::from),
1135          _ => None,
1136        }
1137      };
1138      match node {
1139        XM::Apply(op, args, _, _) => {
1140          let my_meaning = meaning_of(&op.0);
1141          let here = match (&my_meaning, &ancestor_fence) {
1142            (Some(m), Some(a)) if m == a && is_fence_meaning(m) => 1,
1143            _ => 0,
1144          };
1145          // Pass current meaning as ancestor if it's a fence; otherwise
1146          // keep the existing ancestor (so we detect nesting across
1147          // intermediate non-fence Applies like `times`).
1148          let new_anc = match &my_meaning {
1149            Some(m) if is_fence_meaning(m) => Some(m.clone()),
1150            _ => ancestor_fence.clone(),
1151          };
1152          here + args.trees().iter().map(|a| walk(a, new_anc.clone())).sum::<usize>()
1153        },
1154        XM::Dual(c, p, _, _) => {
1155          // If the Dual's content is a fence-Apply, propagate that
1156          // meaning when walking the presentation Wrap — because the
1157          // actual nested expression lives inside the Wrap, not
1158          // inside the Ref-pointing content.
1159          let dual_fence = match &**c {
1160            XM::Apply(op_inner, _, _, _) => match &*op_inner.0 {
1161              XM::Token(p_inner, _) => p_inner
1162                .meaning
1163                .as_deref()
1164                .filter(|m| is_fence_meaning(m))
1165                .map(String::from),
1166              _ => None,
1167            },
1168            _ => None,
1169          };
1170          let pres_anc = dual_fence.clone().or_else(|| ancestor_fence.clone());
1171          walk(c, ancestor_fence.clone()) + walk(p, pres_anc)
1172        },
1173        XM::Wrap(items, _, _) => items.iter().map(|i| walk(i, ancestor_fence.clone())).sum(),
1174        XM::Choices(trees) => trees.iter().map(|t| walk(t, ancestor_fence.clone())).sum(),
1175        XM::Arg(items) => items.iter().map(|i| walk(i, ancestor_fence.clone())).sum(),
1176        XM::Token(_, _) | XM::Lexeme(_, _) | XM::Ref(_) => 0,
1177      }
1178    }
1179    walk(self, None)
1180  }

1271  /// Count "letter applies to vertbar-fenced argument" patterns
1272  /// that occur **inside** a `delimited-⟨⟩` (angle-fenced) Apply.
1273  /// Outside the QM/bra-ket context, the K-12 convention prevails
1274  /// (`a|b|` = `a * |b|`); this pragma must NOT bias those.
1275  ///
1276  /// The recursion descends through arbitrary parents, but only
1277  /// **counts** an `Apply(letter, [vertbar-fenced])` hit when the
1278  /// path from the tree root passed through an `Apply` whose
1279  /// operator has meaning `delimited-⟨⟩`. The flag is tracked via
1280  /// a closure parameter to avoid recomputing the ancestry.
1281  pub fn count_letter_at_vertbar(&self) -> usize {
1282    fn op_is_letter(op: &XM) -> bool {
1283      match op {
1284        XM::Token(p, _) => p.role.as_deref() == Some("UNKNOWN") || p.role.as_deref() == Some("ID"),
1285        // Lexeme-form letters: name like "UNKNOWN:..." or "ID:..."
1286        XM::Lexeme(name, _) => {
1287          let head = name.split(':').next().unwrap_or("");
1288          head == "UNKNOWN" || head == "ID"
1289        },
1290        _ => false,
1291      }
1292    }
1293    fn arg_is_vertbar_fenced(arg: &XM) -> bool {
1294      let XM::Dual(_, ref presentation, _, _) = arg else {
1295        return false;
1296      };
1297      let XM::Wrap(ref items, _, _) = **presentation else {
1298        return false;
1299      };
1300      let is_vertbar = |x: Option<&XM>| -> bool {
1301        match x {
1302          Some(XM::Token(p, _)) => p.content.as_deref() == Some("|") || p.content.as_deref() == Some("‖"),
1303          Some(XM::Lexeme(name, _)) => name.starts_with("OPEN:|:") || name.starts_with("CLOSE:|:")
1304            || name.starts_with("OPEN:‖:") || name.starts_with("CLOSE:‖:")
1305            || &**name == "VERTBAR" || name.starts_with("VERTBAR:"),
1306          _ => false,
1307        }
1308      };
1309      is_vertbar(items.first()) && is_vertbar(items.last())
1310    }
1311    fn is_delim_angle_op(op: &XM) -> bool {
1312      match op {
1313        XM::Token(p, _) => p.meaning.as_deref() == Some("delimited-⟨⟩"),
1314        _ => false,
1315      }
1316    }
1317    // Recognise "bra-ket-by-context": a multirelation Apply whose
1318    // **boundary** arguments (first and last) are both `absent`,
1319    // e.g. `absent < a |f| b > absent` produced by plain `<a|f|b>`.
1320    // In that surrounding shape, the same QM convention applies as
1321    // for explicit `\langle...\rangle`.
1322    fn is_qm_multirelation_apply(op: &XM, args: &Args) -> bool {
1323      let meaning_is_multirelation = match op {
1324        XM::Token(p, _) => p.meaning.as_deref() == Some("multirelation"),
1325        _ => false,
1326      };
1327      if !meaning_is_multirelation {
1328        return false;
1329      }
1330      let trees = args.trees();
1331      let is_absent = |x: Option<&&XM>| -> bool {
1332        matches!(x,
1333          Some(XM::Token(p, _)) if p.meaning.as_deref() == Some("absent"))
1334      };
1335      is_absent(trees.first()) && is_absent(trees.last())
1336    }
1337    fn walk(node: &XM, inside_angle: bool, op_is_letter: &impl Fn(&XM) -> bool, arg_is_vertbar_fenced: &impl Fn(&XM) -> bool) -> usize {
1338      match node {
1339        XM::Token(_, _) | XM::Lexeme(_, _) | XM::Ref(_) => 0,
1340        XM::Apply(op, args, ..) => {
1341          let trees = args.trees();
1342          let here = if inside_angle
1343            && op_is_letter(&op.0)
1344            && trees.len() == 1
1345            && trees.first().is_some_and(|a| arg_is_vertbar_fenced(a))
1346          {
1347            1
1348          } else {
1349            0
1350          };
1351          // Recurse with `inside_angle` set true if THIS Apply is
1352          // a delimited-⟨⟩ OR a QM-style multirelation (absent < … >
1353          // absent), both of which signal bra-ket context.
1354          let child_inside = inside_angle
1355            || is_delim_angle_op(&op.0)
1356            || is_qm_multirelation_apply(&op.0, args);
1357          here + args
1358            .trees()
1359            .iter()
1360            .map(|a| walk(a, child_inside, op_is_letter, arg_is_vertbar_fenced))
1361            .sum::<usize>()
1362        },
1363        // Dual content/presentation: the Dual's content typically
1364        // holds the operator Apply (e.g. `Apply(delim-⟨⟩, [Ref])`)
1365        // while the presentation Wrap holds the actual body. So if
1366        // the content marks a delim-⟨⟩ wrapper, the presentation
1367        // body should be walked with `inside_angle=true`.
1368        XM::Dual(c, p, ..) => {
1369          let this_is_angle = matches!(&**c,
1370            XM::Apply(op, _, _, _) if is_delim_angle_op(&op.0));
1371          let pres_inside = inside_angle || this_is_angle;
1372          walk(c, inside_angle, op_is_letter, arg_is_vertbar_fenced)
1373            + walk(p, pres_inside, op_is_letter, arg_is_vertbar_fenced)
1374        },
1375        XM::Wrap(items, ..) => items
1376          .iter()
1377          .map(|i| walk(i, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1378          .sum(),
1379        XM::Choices(trees) => trees
1380          .iter()
1381          .map(|t| walk(t, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1382          .sum(),
1383        XM::Arg(items) => items
1384          .iter()
1385          .map(|i| walk(i, inside_angle, op_is_letter, arg_is_vertbar_fenced))
1386          .sum(),
1387      }
1388    }
1389    walk(self, false, &op_is_letter, &arg_is_vertbar_fenced)
1390  }

1585  /// additive operator (`+` / `-`) and another roots at a
1586  /// multiplicative operator (`*` / invisible-times), prefer the
1587  /// additive root. K-12 algebra: `a*|a| + b*|b| + c*|c|` parses
1588  /// with `+` outermost, not as `a * (one giant absolute-value)`.
1589  ///
1590  /// Fires only when the discriminator is unambiguous (at least
1591  /// one of each shape exists); otherwise inert.
1592  pub fn prefer_root_addition_over_outer_multiplication(self) -> Self {
1593    match self {
1594      XM::Choices(trees) if trees.len() > 1 => {
1595        let has_addition_root = trees.iter().any(|t| t.root_is_addition());
1596        if !has_addition_root {
1597          return XM::Choices(trees);
1598        }
1599        let kept: Vec<XM> = trees.into_iter().filter(|t| t.root_is_addition()).collect();
1600        match kept.len() {
1601          0 => XM::Choices(Vec::new()),
1602          1 => kept.into_iter().next().unwrap(),
1603          _ => XM::Choices(kept),
1604        }
1605      },
1606      other => other,
1607    }
1608  }

1610  /// given a tree, return the base operator name, if any
1611  /// Simple text summary for debug logging (no DOM access needed)
1612  pub fn text_summary(&self) -> String {
1613    match self {
1614      XM::Lexeme(name, meta) => {
1615        let fenced = meta.fenced.as_deref().unwrap_or("");
1616        if fenced.is_empty() {
1617          name.to_string()
1618        } else {
1619          format!("{name}[{fenced}]")
1620        }
1621      },
1622      XM::Token(props, _) => {
1623        let role = props.role.as_deref().unwrap_or("?");
1624        let meaning = props.meaning.as_deref().unwrap_or("");
1625        if meaning.is_empty() {
1626          role.to_string()
1627        } else {
1628          format!("{role}:{meaning}")
1629        }
1630      },
1631      XM::Apply(op, args, ..) => {
1632        let op_str = op.0.text_summary();
1633        let args_str: Vec<String> = args.trees().iter().map(|a| a.text_summary()).collect();
1634        format!("{}@({})", op_str, args_str.join(", "))
1635      },
1636      XM::Dual(c, p, ..) => format!("Dual({}, {})", c.text_summary(), p.text_summary()),
1637      XM::Wrap(items, ..) => {
1638        let items_str: Vec<String> = items.iter().map(|i| i.text_summary()).collect();
1639        format!("Wrap({})", items_str.join(", "))
1640      },
1641      XM::Choices(ref trees) => format!("Choices({})", trees.len()),
1642      XM::Ref(idx) => format!("Ref({})", idx),
1643      XM::Arg(items) => {
1644        let items_str: Vec<String> = items.iter().map(|i| i.text_summary()).collect();
1645        format!("Arg({})", items_str.join(", "))
1646      },
1647    }
1648  }

1651    match self {
1652      XM::Lexeme(ref name, _) => name.to_string(),
1653      XM::Apply(ref op, ref args, ..) => {
1654        match &*op.0 {
1655          XM::Lexeme(ref name, _) if &**name == "unknown.subscript" => {
1656            let arg_base = args.0.first().unwrap().as_ref().unwrap().clone();
1657            format!("sub__{}", arg_base.base_operator_name())
1658          },
1659          XM::Lexeme(ref name, _) if &**name == "unknown.superscript" => {
1660            // TODO: Too much datastructure boilerplate with the unwrap incantation
1661            //       might be better to create some getter methods to explain the intent better
1662            //       this is meant to do "give me a clone of the first argument to this XM::Apply"
1663            //       which happens to be a base of a sub or super-script.
1664            let arg_base = args.0.first().unwrap().as_ref().unwrap();
1665            arg_base.base_operator_name()
1666          },
1667          XM::Lexeme(other, _) => other.to_string(),
1668          XM::Apply(sub_other, ..) => format!("reduced__{}", sub_other.0.base_operator_name()),
1669          _ => String::new(),
1670        }
1671      },
1672      _ => String::new(),
1673    }

1676  pub fn get_baseline(&self) -> &Self {
1677    match self {
1678      XM::Lexeme(..) => self,
1679      XM::Token(..) => self,
1680      XM::Ref(_) => self,
1681      XM::Apply(ref op, ref args, ..) => {
1682        if let XM::Lexeme(name, _) = &*op.0 {
1683          if &**name == "unknown.subscript" || &**name == "unknown.superscript" {
1684            args.trees().first().unwrap().get_baseline()
1685          } else {
1686            self
1687          }
1688        } else {
1689          self
1690        }
1691      },
1692      // Dual/Wrap/Arg: fall back to treating self as the baseline. These
1693      // variants aren't typically exercised in get_baseline contexts today,
1694      // but returning self is safe — callers use the baseline to attach
1695      // scripts to, and these shapes don't break that attachment.
1696      XM::Dual(..) | XM::Wrap(..) | XM::Arg(_) => self,
1697      XM::Choices(args) => args.first().unwrap().get_baseline(),
1698    }

1709  pub fn unconstrain_recursive(&mut self) {
1710    match self {
1711      XM::Lexeme(_, meta) => {
1712        meta.curry_constraints.drain();
1713      },
1714      XM::Token(_, meta) => {
1715        meta.curry_constraints.drain();
1716      },
1717      XM::Apply(Operator(op), args, _, meta) => {
1718        meta.curry_constraints.drain();
1719        op.unconstrain_recursive();
1720        args.unconstrain_recursive();
1721      },
1722      XM::Dual(content, pres, _props, meta) => {
1723        meta.curry_constraints.drain();
1724        content.unconstrain_recursive();
1725        pres.unconstrain_recursive();

1727      XM::Wrap(content, _props, meta) => {
1728        meta.curry_constraints.drain();

1734        for tree in args {
1735          tree.unconstrain_recursive();

1738      XM::Ref(_) => {},

1742  /// level-indented formatter akin to the std::fmt Display trait
1743  pub fn fmt_indented(&self, level: &[bool], f: &mut fmt::Formatter<'_>) -> fmt::Result {
1744    let indent = aux_generate_indent(level, false);
1745    match self {
1746      XM::Lexeme(name, meta) => writeln!(f, "{indent}{name} {meta}"),
1747      XM::Ref(idref) => writeln!(f, "{indent}Ref[{idref}]"),
1748      XM::Token(t, meta) => writeln!(f, "{indent}{t} {meta}"),
1749      XM::Apply(op, args, _, meta) => {
1750        if !meta.syntax_trace.is_empty() {
1751          let indent_base = aux_generate_indent(level, true);
1752          writeln!(f, "{indent_base}\n{indent_base}{meta}")?;
1753        }
1754        let mut arg_level: Vec<bool> = level.to_vec();
1755        arg_level.push(true);
1756        op.fmt_indented(level, f)?;
1757        args.fmt_indented(&arg_level, f)
1758      },
1759      XM::Dual(content, pres, ..) => {
1760        writeln!(f, "\n{indent}Dual")?;

1763        let mut peekable = vec![content, pres].into_iter().peekable();
1764        while let Some(arg) = peekable.next() {
1765          if peekable.peek().is_none() {
1766            arg_level.pop();
1767            arg_level.push(false);
1768            arg.fmt_indented(&arg_level, f)?
1769          } else {
1770            arg.fmt_indented(&arg_level, f)?
1771          }
1772        }
1773        writeln!(f)
1774      },
1775      XM::Wrap(content, ..) => {
1776        writeln!(f, "\n{indent}Wrap")?;
1777        let mut arg_level: Vec<bool> = level.to_vec();
1778        arg_level.push(true);
1779        let mut peekable = content.iter().peekable();
1780        while let Some(arg) = peekable.next() {
1781          if peekable.peek().is_none() {
1782            arg_level.pop();
1783            arg_level.push(false);
1784            arg.fmt_indented(&arg_level, f)?
1785          } else {
1786            arg.fmt_indented(&arg_level, f)?
1787          }
1788        }
1789        writeln!(f)
1790      },
1791      XM::Arg(args) => {
1792        writeln!(f, "\n{indent}Arg")?;
1793        let mut arg_level: Vec<bool> = level.to_vec();
1794        arg_level.push(true);
1795        let mut peekable = args.iter().peekable();
1796        while let Some(arg) = peekable.next() {
1797          if peekable.peek().is_none() {
1798            arg_level.pop();
1799            arg_level.push(false);
1800            arg.fmt_indented(&arg_level, f)?
1801          } else {
1802            arg.fmt_indented(&arg_level, f)?
1803          }
1804        }
1805        writeln!(f)
1806      },
1807      XM::Choices(args) => {
1808        writeln!(f, "\n{indent}Choices")?;
1809        let mut arg_level: Vec<bool> = level.to_vec();
1810        arg_level.push(true);
1811        let mut peekable = args.iter().peekable();
1812        while let Some(arg) = peekable.next() {
1813          if peekable.peek().is_none() {
1814            arg_level.pop();
1815            arg_level.push(false);
1816            arg.fmt_indented(&arg_level, f)?
1817          } else {
1818            arg.fmt_indented(&arg_level, f)?
1819          }
1820        }
1821        writeln!(f)
1822      },

1826  /// Rebuild a marpa-derived parse tree into an XMath XML tree
1827  pub fn into_xmath(
1828    self,
1829    owner: &mut Node,
1830    nodes: &mut [Node],
1831    document: &mut Document,
1832  ) -> Result<Node, Box<dyn Error + Send + Sync>> {
1833    // Grow the call stack on demand if we are running low. Heavily
1834    // nested math trees (XMApp(op, [XMApp(...)]) chains) recurse
1835    // through this function at depth proportional to expression
1836    // nesting; on Rust's 8 MB main-thread default stack that overflows
1837    // on grammar-ambiguous papers (sandbox 0711.4787, 0903.3289,
1838    // hep-th0101151, math0505371, math9204211, hep-ph9210253,
1839    // hep-ph9512208, astro-ph0612758 — 8 papers, SIGABRT with
1840    // `thread 'main' has overflowed its stack`). stacker::maybe_grow
1841    // allocates a new stack chunk on a separate thread when the
1842    // current frame's remaining space drops below the red zone.
1843    stacker::maybe_grow(64 * 1024, 4 * 1024 * 1024, || {
1844      self.into_xmath_inner(owner, nodes, document)
1845    })
1846  }

1848  fn into_xmath_inner(
1849    self,
1850    owner: &mut Node,
1851    nodes: &mut [Node],
1852    document: &mut Document,
1853  ) -> Result<Node, Box<dyn Error + Send + Sync>> {
1854    // Drain nested XM::Choices iteratively before the match. Each
1855    // Choices wrapper is a tail call into the first remaining choice
1856    // — preserving the per-layer Info log so the discard count
1857    // is visible identically to the prior recursive form.
1858    let mut tree = self;
1859    loop {
1860      match tree {
1861        XM::Choices(mut choices) => {
1862          Info!(
1863            "math_parser",
1864            "choices",
1865            format!(
1866              "to_xmath handler discarded {} parse choices.",
1867              choices.len() - 1
1868            )
1869          );
1870          tree = choices.remove(0);
1871        },
1872        other => {
1873          tree = other;
1874          break;
1875        },
1876      }
1877    }
1878    match tree {
1879      XM::Lexeme(content, _meta) => {
1880        let id = content
1881          .split(':')
1882          .next_back()
1883          .unwrap()
1884          .parse::<usize>()
1885          .unwrap()
1886          - 1;
1887        let atom_node = &mut nodes[id];
1888        atom_node.unbind();
1889        Ok(atom_node.clone())
1890      },
1891      XM::Token(props, _meta) => {
1892        // Transition the {font} property to the "_font" attribute.
1893        let has_explicit_font = props.font.is_some();
1894        let (content_opt, font, attrs) = props.into_attributes();
1895        let mut xmtok = document.open_element_at(owner, "ltx:XMTok", attrs, font)?;
1896        if let Some(ref content) = content_opt {
1897          if !content.is_empty() {
1898            xmtok.set_content(content)?;
1899          }
1900        }
1901        // Perl: Font->specialize($content) — for parser-created tokens without
1902        // explicit font, the ambient _font is specialized based on content.
1903        // Operators get default font (no italic), letters get italic.
1904        if !has_explicit_font {
1905          if let Some(font_hash) = xmtok.get_attribute("_font") {
1906            let content = content_opt.as_deref().unwrap_or("");
1907            if content.is_empty() {
1908              // Empty content: no font needed
1909              let _ = xmtok.remove_attribute("_font");
1910            } else if let Some(font) = document.decode_font(&font_hash) {
1911              let specialized = font.specialize(content);
1912              // Re-encode: store the specialized font and update _font hash
1913              document.set_node_font(&mut xmtok, &specialized)?;
1914            }
1915          }
1916        }
1917        document.close_element_at(&mut xmtok)?;
1918        Ok(xmtok)
1919      },
1920      XM::Apply(op, args, props, _meta) => {
1921        // let mut apply_node = Node::new("XMApp", None, document.get_document()).unwrap();
1922        // props.into_xmath(&mut apply_node,document)?;
1923        let (_, font, attrs) = props.into_attributes();
1924        let mut apply_node = document.open_element_at(owner, "ltx:XMApp", attrs, font)?;
1925        let mut op_node = op.0.into_xmath(&mut apply_node, nodes, document)?;
1926
1927        add_child_guard_xmarg(&mut apply_node, &mut op_node)?;
1928        for arg in args.0.into_iter().flatten() {
1929          let mut arg_node = arg.into_xmath(&mut apply_node, nodes, document)?;
1930          add_child_guard_xmarg(&mut apply_node, &mut arg_node)?;
1931        }
1932        document.close_element_at(&mut apply_node)?;
1933        Ok(apply_node)
1934      },
1935      XM::Dual(content, pres, props, _meta) => {
1936        let (_, font, attrs) = props.into_attributes();
1937        let mut dual_node = document.open_element_at(owner, "ltx:XMDual", attrs, font)?;
1938        // Content branch first, then presentation (Perl convention)
1939        let mut content_node = content.into_xmath(&mut dual_node, nodes, document)?;
1940        add_child_guard_xmarg(&mut dual_node, &mut content_node)?;
1941        let mut pres_node = pres.into_xmath(&mut dual_node, nodes, document)?;
1942        add_child_guard_xmarg(&mut dual_node, &mut pres_node)?;
1943        document.close_element_at(&mut dual_node)?;
1944        Ok(dual_node)
1945      },
1946      XM::Wrap(content, props, _meta) => {
1947        let (_, font, attrs) = props.into_attributes();
1948        let mut wrap_node = document.open_element_at(owner, "ltx:XMWrap", attrs, font)?;
1949        for c in content.into_iter() {
1950          let mut content_node = c.into_xmath(&mut wrap_node, nodes, document)?;
1951          add_child_guard_xmarg(&mut wrap_node, &mut content_node)?;
1952        }
1953        document.close_element_at(&mut wrap_node)?;
1954        Ok(wrap_node)
1955      },
1956      XM::Ref(refprops) => {
1957        let mut ref_node = Node::new("XMRef", None, document.get_document()).unwrap();
1958        if let Some(id) = refprops.id {
1959          document.set_attribute(&mut ref_node, "idref", &id)?;
1960        }
1961        if let Some(xmkey) = refprops.xmkey {
1962          // Use _pxmkey for parser-generated keys (pxm prefix) to avoid
1963          // conflicting with base_xmath's \lx@dual resolver.
1964          // Regular _xmkey for all other refs.
1965          let attr_name = if xmkey.starts_with("pxm") {
1966            "_pxmkey"
1967          } else {
1968            "_xmkey"
1969          };
1970          document.set_attribute(&mut ref_node, attr_name, &xmkey)?;
1971        }
1972        Ok(ref_node)
1973      },
1974      XM::Arg(inner_list) => {
1975        let mut arg_node = Node::new("XMArg", None, document.get_document()).unwrap();
1976        for inner_item in inner_list {
1977          let mut inner_node = inner_item.into_xmath(&mut arg_node, nodes, document)?;
1978          add_child_guard_xmarg(&mut arg_node, &mut inner_node)?;
1979        }
1980        Ok(arg_node)
1981      },
1982      // XM::Choices was already drained iteratively above the match.
1983      XM::Choices(_) => unreachable!("Choices drained before match"),
1984    }
1985  }

2003      Some(ref v) if !v.is_empty() => Some(Cow::Borrowed(v)),

2005        Some(ref v) if !v.is_empty() => Some(Cow::Borrowed(v)),

2007          Some(ref v) if !v.is_empty() => Some(Cow::Borrowed(v)),

2011          },
2012        },
2013      },

2084/// Looks up the node associated with a given lexeme,
2085/// via the node index held in the third colon-separated lexeme piece.
2086pub(crate) fn lookup_lex_node<'a>(
2087  lex: &'a str,
2088  nodes: &'a [Node],
2089) -> Result<&'a Node, Box<dyn Error>> {
2090  // Lex strings are produced internally by the grammar (`{name}:{value}:{idx}`
2091  // shape), so the suffix parses cleanly under normal operation. If we ever
2092  // hit a malformed lex — e.g. because a custom rule emitted a short form —
2093  // return an Err rather than panicking, matching the degrade-not-panic
2094  // policy used elsewhere on user-reachable paths.
2095  let idx_str = lex.split(':').next_back().unwrap_or("");
2096  let node_idx = idx_str
2097    .parse::<usize>()
2098    .map_err(|e| format!("malformed lex {lex:?}: {e}"))?
2099    .checked_sub(1)
2100    .ok_or_else(|| format!("lex idx 0 (expected 1-based) in {lex:?}"))?;
2101  nodes.get(node_idx).ok_or_else(|| {
2102    format!(
2103      "lex idx {node_idx} out of range (nodes.len={}) in {lex:?}",
2104      nodes.len()
2105    )
2106    .into()
2107  })
2108}

2111  let mut indent = String::new();
2112  let mut peekable = level.iter().peekable();
2113  while let Some(is_inked) = peekable.next() {
2114    indent += if peekable.peek().is_none() {
2115      if is_base {
2116        "   │  "
2117      } else if *is_inked {
2118        "   ├── "
2119      } else {
2120        "   └── "
2121      }
2122    } else if *is_inked {
2123      "   │"
2124    } else {
2125      "    "
2126    }
2127  }
2128  indent
2129}
