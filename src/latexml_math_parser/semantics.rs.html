29/// A runtime context for a semantic math parser action
30/// Ideally, these are all immutable borrows of various `Core` data.
31pub struct ActionContext<'a> {
32  /// The original XML nodes involved in this parse request
33  pub nodes:    &'a [XMLNode],
34  /// The owner document of the parsed nodes
35  pub document: &'a mut Document,
36}
37pub type ActionClosure = Rc<
38  dyn Fn(
39    i32,
40    Vec<Option<XM>>,
41    &[ValidationPragmatics],
42    ActionContext,
43  ) -> Result<Option<XM>, Box<dyn Error>>,
44>;
45
46#[derive(Default)]
47pub struct Actions {
48  dispatch: HashMap<i32, ActionClosure>,
49}
50
51impl Actions {
52  pub fn register(&mut self, id: i32, closure: ActionClosure) { self.dispatch.insert(id, closure); }
53  pub fn action_on(
54    &self,
55    id: i32,
56    mut args: Vec<Option<XM>>,
57    pragmas: &[ValidationPragmatics],
58    ctxt: ActionContext,
59  ) -> Result<Option<XM>, Box<dyn Error>> {
60    if let Some(action) = self.dispatch.get(&id) {
61      action(id, args, pragmas, ctxt)
62    } else {
63      match args.len() {
64        0 => Ok(None),
65        1 => Ok(args.remove(0)),
66        more => {
67          eprintln!(
68            "Only returning first of {more:?} elements at rule id {id:?} content: {args:?}"
69          );
70          Ok(args.remove(0))
71        },
72      }
73    }
74  }
75
76  pub fn get_tree(
77    &self,
78    b: TreeBuilder,
79    v: Value,
80    pragmas: &[ValidationPragmatics],
81    ctxt: ActionContext,
82  ) -> Result<Option<XM>, Box<dyn Error>> {
83    let handle = proc_value(b, v);
84    self.translate_node(&handle, pragmas, ctxt)
85  }
86
87  pub fn translate_node<T: Token>(
88    &self,
89    n: &Handle<T>,
90    pragmas: &[ValidationPragmatics],
91    ctxt: ActionContext,
92  ) -> Result<Option<XM>, Box<dyn Error>> {
93    match *n.borrow() {
94      Node::Tree(ref rule, ref children) => {
95        let mut translated_children = Vec::new();
96        for child in children.iter() {
97          let translated = self.translate_node(child, pragmas, ActionContext {
98            nodes:    ctxt.nodes,
99            document: ctxt.document,
100          })?;
101          translated_children.push(translated);
102        }
103        self.action_on(*rule, translated_children, pragmas, ctxt)
104      },
105      Node::Rule(ref rule, ref children) => {
106        let mut translated_children = Vec::new();
107        for child in children.iter() {
108          translated_children.push(self.translate_node(child, pragmas, ActionContext {
109            nodes:    ctxt.nodes,
110            document: ctxt.document,
111          })?);
112        }
113        self.action_on(*rule, translated_children, pragmas, ctxt)
114      },
115      Node::Token(_ty, ref val) => {
116        let token_str = ::std::str::from_utf8(val).unwrap_or("malformed-utf8");
117        Ok(Some(
118          XM::Lexeme(token_str.to_owned(), Meta::default()).specialize(Meta::default(), pragmas)?,
119        ))
120      },
121      Node::Leaf(ref tok) => Ok(Some(XM::Lexeme(tok.to_string(), Meta::default()))),
122      Node::Null(_) => {
123        // e.g.* argument failed nothing, just skip.
124        Ok(None)
125        // XM::Lexeme("null".into())
126      },
127    }
128  }
129}

148/// application with trailing elision, as in `x \cdot y \cdot\cdot\cdot`
149pub fn infix_apply_and_elide(
150  rule_id: i32,
151  mut args: Vec<Option<XM>>,
152  p: &[ValidationPragmatics],
153  ctxt: ActionContext,
154) -> Result<Option<XM>, Box<dyn Error>> {
155  unp!(args => arg1, infixop, arg2, elision);
156  // check if "left" is already an application of infix op, in which case we can do n-ary apply.
157  if let Some(XM::Apply(new_op, mut new_args, props, meta)) =
158    infix_apply_nary(rule_id, vec![arg1, infixop, arg2], p, ctxt)?
159  {
160    new_args.0.push(elision);
161    Ok(Some(XM::Apply(new_op, new_args, props, meta)))
162  } else {
163    Ok(None)
164  }
165}

167// infix_apply in the base case,
168// but when chained, using the flat "multirelation" behavior of latexml
169pub fn infix_relation(
170  _rule_id: i32,
171  mut args: Vec<Option<XM>>,
172  _: &[ValidationPragmatics],
173  _: ActionContext,
174) -> Result<Option<XM>, Box<dyn Error>> {
175  unp!(args => left, infixop, right);
176  // if left has a "multirelation" already, add right in.
177  // if left applies a relation, flatten it out to infix form.
178  // base case - build a simple infix apply
179  let mut left = left;
180  match left {
181    Some(XM::Apply(ref op, ref mut left_args, _, ref _left_meta)) => {
182      if let XM::Token(ref tok, _) = *op.0 {
183        if tok.meaning == Some(Cow::Borrowed("multirelation")) {
184          left_args.0.push(infixop);
185          left_args.0.push(right);
186          Ok(left)
187        } else {
188          Ok(Some(XM::Apply(
189            infixop.into(),
190            Args(vec![left, right]),
191            XProps::default(),
192            Meta::default(),
193          )))
194        }
195      } else if let XM::Lexeme(ref lex, ref _left_meta) = *op.0 {
196        if lex.split(':').next().unwrap().contains("RELOP") {
197          // first multirelation need is here.
198          let multirel_tok = XProps {
199            meaning: Some(Cow::Borrowed("multirelation")),
200            ..XProps::default()
201          };
202          let mut drained_left_args = left_args.0.drain(..);
203          let left_1 = drained_left_args.next().unwrap();
204          let left_2 = drained_left_args.next().unwrap();
205          let moved_op = (*op.0).clone();
206          Ok(Some(XM::Apply(
207            multirel_tok.into(),
208            Args(vec![left_1, Some(moved_op), left_2, infixop, right]),
209            XProps::default(),
210            Meta::default(),
211          )))
212        } else {
213          Ok(Some(XM::Apply(
214            infixop.into(),
215            Args(vec![left, right]),
216            XProps::default(),
217            Meta::default(),
218          )))
219        }
220      } else {
221        Ok(Some(XM::Apply(
222          infixop.into(),
223          Args(vec![left, right]),
224          XProps::default(),
225          Meta::default(),
226        )))
227      }
228    },
229    _ => Ok(Some(XM::Apply(
230      infixop.into(),
231      Args(vec![left, right]),
232      XProps::default(),
233      Meta::default(),
234    ))),
235  }
236}

238pub fn infix_apply_nary(
239  _rule_id: i32,
240  mut args: Vec<Option<XM>>,
241  _: &[ValidationPragmatics],
242  _: ActionContext,
243) -> Result<Option<XM>, Box<dyn Error>> {
244  unp!(args => left, infixop, right);
245  let mut left = left;
246  // left-to-right associative:
247  // 1. if "left" is already an application of "infixop",
248  // 2. then tuck "right" inside it.
249  if let Some(XM::Apply(ref left_op, ref mut left_args, _, ref _m)) = left {
250    if let XM::Lexeme(left_op_lex, _xmeta) = &*left_op.0 {
251      if let Some(XM::Lexeme(ref infix_op_lex, _)) = infixop {
252        let left_op_pieces: Vec<_> = left_op_lex.split(':').collect();
253        let infix_op_pieces: Vec<_> = infix_op_lex.split(':').collect();
254        if left_op_pieces.len() == 3
255          && infix_op_pieces.len() == 3
256          && left_op_pieces[0] == infix_op_pieces[0]
257          && left_op_pieces[1] == infix_op_pieces[1]
258        {
259          left_args.0.push(right);
260          return Ok(left);
261        }
262      }
263    }
264  }
265  // base case: new apply tree
266  let apply_tree = XM::Apply(
267    infixop.into(),
268    Args(vec![left, right]),
269    XProps::default(),
270    Meta::default(),
271  );
272  Ok(Some(apply_tree))
273}

346/// This is similar, but "interprets" a delimited list as being the
347/// application of some operator to the items in the list.
348fn interpret_delimited(
349  op: XM,
350  mut stuff: Vec<XM>,
351  ctxt: ActionContext,
352) -> Result<XM, Box<dyn Error>> {
353  let upto = stuff.len() - 1;
354  let (_seps, mut args) = extract_separators(&mut stuff[1..upto]);
355  let ref_args = create_xmrefs(&mut args, ctxt)?;
356  Ok(XM::Dual(
357    Box::new(XM::Apply(
358      op.into(),
359      ref_args.into(),
360      XProps::default(),
361      Meta::default(),
362    )),
363    Box::new(XM::Wrap(stuff, XProps::default(), Meta::default())),
364    XProps::default(),
365    Meta::default(),
366  ))
367}

435/// This is loosely in the lines of MathParser::NewScript, but taking into account
436/// the realities of our new data structures.
437pub fn new_script(
438  base: Option<XM>,
439  script: XM,
440  ctxt: ActionContext,
441) -> Result<Option<XM>, Box<dyn Error>> {
442  if let XM::Lexeme(ref lex, _) = script {
443    // TODO: continue porting...
444    // let rbase = base.as_ref().map(|b| b.realize_xmnode(&ctxt).expect("if a script is to be built
445    // with a base, we expect it to have a Node.")).flatten(); let rscript =
446    // script.realize_xmnode(&ctxt)?;
447    let script_wrap = lookup_lex_node(lex.as_str(), ctxt.nodes)?;
448    let node_role = script_wrap.get_attribute("role").unwrap();
449    let is_float = node_role.starts_with("FLOAT");
450    let is_super = node_role.ends_with("SUPERSCRIPT");
451    let role = Cow::Borrowed(if is_super {
452      "SUPERSCRIPTOP"
453    } else {
454      "SUBSCRIPTOP"
455    });
456    let scriptpos = Cow::Borrowed(if is_float { "pre1" } else { "post1" });
457    // TODO: scriptpos => "$x$l"
458    let op = new_props(
459      None,
460      None,
461      Some(raw_map!("role"=>role, "scriptpos"=>scriptpos)),
462    );
463    let script_arg = obtain_arg(script, 0, ctxt)?;
464    Ok(Some(XM::Apply(
465      op.into(),
466      Args(vec![base, script_arg]),
467      XProps::default(),
468      Meta::default(),
469    )))
470  } else {
471    panic!(
472      "new_script is meant to be called on script terminals (e.g. POSTSUBSCRIPT/POSTSUPERSCRIPT), \
473got {:?}",
474      script
475    );
476  }
477}

587pub fn new_list(mut pieces: Vec<XM>, ctxt: ActionContext) -> Result<Option<XM>, Box<dyn Error>> {
588  // drop placeholder token for missing trailing punct, if any
589  if pieces.len() > 1 {
590    let last_meaning_opt = pieces.last().unwrap().get_token_meaning(ctxt.nodes)?;
591    if let Some(last_meaning) = last_meaning_opt {
592      if last_meaning == "absent" {
593        pieces.pop();
594      }
595    }
596  }
597  if pieces.len() == 1 {
598    Ok(pieces.pop())
599  } else {
600    let (_seps, mut items) = extract_separators(&mut pieces);
601    Ok(Some(XM::Dual(
602      Box::new(XM::Apply(
603        new_props(Some(Cow::Borrowed("list")), None, None).into(),
604        create_xmrefs(&mut items, ctxt)?.into(),
605        XProps::default(),
606        Meta::default(),
607      )),
608      Box::new(XM::Wrap(pieces, XProps::default(), Meta::default())),
609      XProps::default(),
610      Meta::default(),
611    )))
612  }
613}

615/// Given  alternating expressions & separators (punctuation,...)
616/// extract the separators as a concatenated string,
617/// returning (separators, args...)
618/// But note that the separators are never used for anything!?
619fn extract_separators(items: &mut [XM]) -> (Vec<&mut XM>, Vec<&mut XM>) {
620  // TODO: consider using the separators at some point, but not for now
621  let punct = Vec::new();
622  let mut args = Vec::new();
623  let mut items_iter = items.iter_mut();
624  while let Some(arg) = items_iter.next() {
625    args.push(arg);
626    let _discard_punct = items_iter.next();
627  }
628  (punct, args)
629}
