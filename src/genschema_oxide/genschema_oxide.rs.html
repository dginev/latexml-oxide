1//! `genschema_oxide` — RelaxNG → schema.tex generator.
2//!
3//! Native Rust replacement for the Perl `tools/genschema` driver.
4//! Walks an RNG file with [`latexml_core::common::relaxng`] (port of
5//! `LaTeXML::Common::Model::RelaxNG.pm`) and prints the resulting
6//! `\schemamodule{}` / `\patterndef{}` / `\elementdef{}` /
7//! `\attrdef{}` LaTeX manual.tex consumed by `latexmlman.sty`.
8//!
9//! Usage:
10//!
11//! ```text
12//! genschema_oxide <RNG> [--output FILE] [--path DIR ...] [--module-abstract]
13//!                       [--ns prefix=URI ...] [--no-latexml-defaults]
14//!                       [--no-skip-svg] [--no-skip-aria] [--no-skip-xhtml]
15//! ```
16

34#[derive(Parser, Debug)]
35#[command(about = "Generate LaTeX manual.tex documentation from a RelaxNG schema.")]
36struct Cli {
37  /// Path to the .rng entry-point. Resolved against `--path` if not
38  /// found at the bare path.
39  schema: PathBuf,
40
41  /// Output file. Defaults to stdout.
42  #[arg(short, long, value_name = "FILE")]
43  output: Option<PathBuf>,
44
45  /// Search paths for `<include href="..."/>` resolution. Repeatable.
46  #[arg(long = "path", value_name = "DIR")]
47  paths: Vec<PathBuf>,
48
49  /// Don't skip the SVG schema branch (default: skip).
50  #[arg(long)]
51  no_skip_svg:   bool,
52  /// Don't skip the ARIA schema branch (default: skip).
53  #[arg(long)]
54  no_skip_aria:  bool,
55  /// Don't skip the XHTML schema branch (default: skip).
56  #[arg(long)]
57  no_skip_xhtml: bool,
58
59  /// Lift the first `\patterndef` doc-arg per `\schemamodule` block
60  /// into a top-level `\moduleabstract{...}` macro. RNC `## comments`
61  /// flow into doc annotations on the first `<define>` of each module
62  /// (per the trang convention); this lift promotes them to module-
63  /// level so the schema-doc renderer can show them as a per-module
64  /// narrative aside rather than inline with one specific pattern.
65  #[arg(long)]
66  module_abstract: bool,
67
68  /// Pre-register a namespace prefix → URI binding before scanning.
69  /// Repeatable. Useful when a `.rnc` declares `default namespace =
70  /// "..."` without a prefix: trang strips the binding to the default
71  /// `<grammar ns="..."/>`, leaving no `xmlns:` for the scanner to
72  /// pick up, so element names render as the synthesized fallback
73  /// `namespaceN:foo`. `--ns tei=http://www.tei-c.org/ns/1.0` makes
74  /// them render as `tei:foo` instead.
75  #[arg(long = "ns", value_name = "PREFIX=URI", value_parser = parse_ns_arg)]
76  ns: Vec<(String, String)>,
77
78  /// Skip the built-in LaTeXML namespace defaults (`xml`, `ltx`,
79  /// `svg`, `xlink`, `m`, `xhtml`). Use when documenting a non-LaTeXML
80  /// schema; pair with `--ns` to declare your own conventions.
81  #[arg(long)]
82  no_latexml_defaults: bool,
83}

85fn parse_ns_arg(arg: &str) -> Result<(String, String), String> {
86  let (prefix, uri) = arg
87    .split_once('=')
88    .ok_or_else(|| format!("expected `prefix=URI`, got `{}`", arg))?;
89  if prefix.is_empty() {
90    return Err("prefix may not be empty".into());
91  }
92  if uri.is_empty() {
93    return Err("URI may not be empty".into());
94  }
95  Ok((prefix.to_string(), uri.to_string()))
96}

98fn main() {
99  let cli = Cli::parse();
100
101  let schema_path = match resolve_schema_path(&cli.schema, &cli.paths) {
102    Some(p) => p,
103    None => {
104      eprintln!(
105        "genschema_oxide: schema file not found: {}",
106        cli.schema.display()
107      );
108      process::exit(2);
109    },
110  };
111
112  let mut search_paths: Vec<&Path> = Vec::with_capacity(cli.paths.len() + 1);
113  let dir = schema_path.parent().unwrap_or_else(|| Path::new("."));
114  search_paths.push(dir);
115  for p in &cli.paths {
116    search_paths.push(p);
117  }
118
119  let schema_name = schema_path
120    .file_stem()
121    .and_then(|s| s.to_str())
122    .unwrap_or("schema")
123    .to_string();
124  let mut rng = Relaxng::new(schema_name);
125  if !cli.no_latexml_defaults {
126    rng.with_latexml_defaults();
127  }
128  for (prefix, uri) in &cli.ns {
129    rng.register_namespace(prefix, uri);
130  }
131
132  let raw = match scan_external(
133    &mut rng,
134    schema_path
135      .file_name()
136      .and_then(|f| f.to_str())
137      .unwrap_or(""),
138    None,
139    &search_paths,
140  ) {
141    Ok(p) => p,
142    Err(e) => {
143      eprintln!("genschema_oxide: scan failed: {}", e);
144      process::exit(1);
145    },
146  };
147  let _ = simplify_top(&mut rng, raw);
148
149  // Auto-elide the schema's primary namespace prefix from rendered
150  // display names — `default namespace = "http://…"` in the master
151  // RNG resolves to a prefix in `document_namespaces`; that prefix
152  // becomes contextually obvious for every element / attribute name
153  // on the site, so we drop it (`xhtml:div` → `div`, `ltx:para` →
154  // `para`, `m:math` → `math`).
155  rng.auto_strip_primary_namespace();
156
157  let tex_opts = TexOptions {
158    skip_svg:   !cli.no_skip_svg,
159    skip_aria:  !cli.no_skip_aria,
160    skip_xhtml: !cli.no_skip_xhtml,
161  };
162  let mut docs = document_modules(&rng, tex_opts);
163  if cli.module_abstract {
164    docs = hoist_document_abstract(&docs);
165    docs = lift_module_abstract(&docs);
166  }
167
168  match cli.output {
169    Some(out) => {
170      if let Err(e) = std::fs::write(&out, &docs) {
171        eprintln!("genschema_oxide: write {}: {}", out.display(), e);
172        process::exit(1);
173      }
174    },
175    None => print!("{}", docs),
176  }
177}

179fn resolve_schema_path(schema: &Path, paths: &[PathBuf]) -> Option<PathBuf> {
180  if schema.is_file() {
181    return Some(schema.to_path_buf());
182  }
183  for dir in paths {
184    let cand = dir.join(schema);
185    if cand.is_file() {
186      return Some(cand);
187    }
188  }
189  None
190}

192/// Promote the first `\patterndef`'s doc-arg of each `\schemamodule`
193/// to a top-level `\moduleabstract{...}` macro.
194///
195/// Scan a TeX group starting at the byte offset of its opening brace;
196/// return the contents and the offset just past the closing brace.
197/// Needed because doc-args routinely carry nested `\code{...}` groups,
198/// which a `[^}]*` regex would truncate at the first inner brace.
199fn read_tex_group(tex: &str, open: usize) -> Option<(&str, usize)> {
200  let bytes = tex.as_bytes();
201  if bytes.get(open) != Some(&b'{') {
202    return None;
203  }
204  let mut depth = 0usize;
205  for (i, &b) in bytes.iter().enumerate().skip(open) {
206    match b {
207      b'{' => depth += 1,
208      b'}' => {
209        depth -= 1;
210        if depth == 0 {
211          return Some((&tex[open + 1..i], i + 1));
212        }
213      },
214      _ => {},
215    }
216  }
217  None
218}

220/// The driver module is the first `\begin{schemamodule}` in the
221/// emitted TeX, and the `## comments` at the head of the driver file
222/// are the schema's own front-page overview. Left in place they would
223/// be sectioned away with the rest of the module when the postprocessor
224/// splits at `\section` — landing on the module page while the
225/// document's `index.html` shows only the title and table of contents.
226/// Hoist them: when the driver's first patterndef carries a
227/// multi-paragraph doc-arg, move the whole narrative to document level
228/// (before the first `\begin{schemamodule}`) as `\documentabstract`,
229/// leaving the patterndef's doc empty. Single-paragraph docs stay put —
230/// those are per-pattern commentary (e.g. `Inline.model` in
231/// LaTeXML.rnc), not a front-page narrative.
232fn hoist_document_abstract(tex: &str) -> String {
233  let Some(module_start) = tex.find("\\begin{schemamodule}") else {
234    return tex.to_owned();
235  };
236  let Some(pdef_rel) = tex[module_start..].find("\\patterndef{") else {
237    return tex.to_owned();
238  };
239  let pdef_start = module_start + pdef_rel;
240  // Don't reach into a later module: the patterndef must belong to
241  // the driver module block.
242  if let Some(next_module_rel) = tex[module_start + 1..].find("\\begin{schemamodule}")
243    && module_start + 1 + next_module_rel < pdef_start
244  {
245    return tex.to_owned();
246  }
247  let name_open = pdef_start + "\\patterndef".len();
248  let Some((_name, doc_open)) = read_tex_group(tex, name_open) else {
249    return tex.to_owned();
250  };
251  let Some((doc, doc_end)) = read_tex_group(tex, doc_open) else {
252    return tex.to_owned();
253  };
254  let paragraphs: Vec<&str> = doc
255    .split("\n\n")
256    .map(str::trim)
257    .filter(|p| !p.is_empty())
258    .collect();
259  if paragraphs.len() < 2 {
260    return tex.to_owned();
261  }
262  // All paragraphs but the last form the document narrative; the
263  // final paragraph is the first define's own documentation and
264  // stays on the patterndef (mirroring the module-lift convention,
265  // where trailing paragraphs document the define they precede).
266  let (narrative, pattern_doc) = paragraphs.split_at(paragraphs.len() - 1);
267  let mut out = String::with_capacity(tex.len() + 32);
268  out.push_str(&tex[..module_start]);
269  out.push_str("\\documentabstract{");
270  out.push_str(&narrative.join("\n\n"));
271  out.push_str("}\n");
272  out.push_str(&tex[module_start..doc_open]);
273  out.push('{');
274  out.push_str(pattern_doc[0]);
275  out.push('}');
276  out.push_str(&tex[doc_end..]);
277  out
278}

280/// Mirrors the Perl regex in `tools/genschema`'s post-processing step:
281/// the `## comments` at the head of each RNC file land — via trang's
282/// `<a:documentation>` and `RelaxNG.pm`'s `doc` op — as the doc-arg of
283/// whichever `<define>` happens to come first in the module. This lift
284/// rewrites the emitted block so the doc reads as a module-level
285/// narrative rather than as documentation attached to one specific
286/// pattern.
287fn lift_module_abstract(tex: &str) -> String {
288  // Unified per-module rendering — no kind subsections; defs flow
289  // into a single `description` env after the optional preamble.
290  // Structure:
291  //
292  //   \begin{schemamodule}{NAME}
293  //   \par\noindent\textit{Includes:} …            (optional preamble)
294  //   \begin{description}
295  //   \patterndef{…}{DOC}{…}        ← only this is liftable
296  //
297  // Promote the first patterndef's DOC up to the module-section
298  // level (above the description-list opener) so it renders as a
299  // per-module narrative aside, not as part of one specific pattern.
300  // The patterndef's doc-arg then becomes empty.
301  //
302  // **Why only patterndef.** A `## doc` placed *before* a define in
303  // the .rnc lands inside `<define>` (sibling of `<element>`) — that
304  // shape is what produces a `\patterndef` with a doc-arg, and the
305  // author's intent is typically a module-level narrative.
306  // A `## doc` placed *inside* `define = ## … element …` lands
307  // inside `<element>` instead, the simplify singleton-shortcut
308  // hoists the element out of the define, and `\elementdef` ends up
309  // carrying that doc. Those docs are element-specific (e.g. "The
310  // document root." on `\elementdef{document}` in LaTeXML-structure)
311  // — lifting them would silently steal the per-element commentary.
312  let re = regex::Regex::new(concat!(
313    r"(\\begin\{schemamodule\}\{[^}]+\}\n)",
314    r"((?:\\par\\noindent[^\n]*\n)*)",
315    r"(\\begin\{description\}\n)",
316    r"\\patterndef\{([^}]+)\}\{([^}]*)\}\{",
317  ))
318  .expect("static regex compiles");
319  re.replace_all(tex, |caps: &regex::Captures| {
320    let module_head = &caps[1];
321    let preamble = &caps[2];
322    let desc_open = &caps[3];
323    let pname = &caps[4];
324    // Match Perl's `if ($doc =~ /\S/)` — promote whenever any
325    // non-whitespace exists, but DON'T trim the doc itself: the
326    // trailing newline emitted by `Pattern::Doc` is part of the
327    // canonical schema.tex shape.
328    let doc = &caps[5];
329    // Split the doc-arg into paragraphs (`extract_docs` separates
330    // adjacent `<a:documentation>` blocks with `\n\n`). The lift
331    // rule:
332    //
333    // * 0 paragraphs: nothing to do.
334    // * 1 paragraph: leave it on the patterndef. A single `## doc` on a `<define>` is per-pattern
335    //   documentation (e.g. "Combined model for inline content." annotates `Inline.model` in
336    //   LaTeXML.rnc) — lifting it would steal the per-pattern commentary and present it as a module
337    //   narrative.
338    // * ≥ 2 paragraphs: lift the FIRST as the module narrative, keep the rest on the patterndef.
339    //   The convention: a `## comment` block at the top of a file (separated from the next block by
340    //   a blank line) is the module narrative; subsequent blocks document the first define.
341    let paragraphs: Vec<&str> = doc
342      .split("\n\n")
343      .map(str::trim)
344      .filter(|p| !p.is_empty())
345      .collect();
346    if paragraphs.len() >= 2 {
347      let module_para = paragraphs[0];
348      let pattern_doc = paragraphs[1..].join("\n\n");
349      format!(
350        "{}{}\\moduleabstract{{{}}}\n{}\\patterndef{{{}}}{{{}\n}}{{",
351        module_head, preamble, module_para, desc_open, pname, pattern_doc
352      )
353    } else {
354      format!(
355        "{}{}{}\\patterndef{{{}}}{{{}}}{{",
356        module_head, preamble, desc_open, pname, doc
357      )
358    }
359  })
360  .into_owned()
