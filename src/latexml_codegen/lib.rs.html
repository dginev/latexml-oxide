13/// Extract the string value of a `#[name = "..."]` attribute (syn 2 idiom).
14///
15/// Replaces the syn-1 `attr.parse_meta()` + `Meta::NameValue { lit: Lit::Str(_), .. }`
16/// pattern, which was removed in syn 2 (the NameValue arm now stores `value: Expr`
17/// instead of `lit: Lit`).
18pub(crate) fn attr_name_value_str(attr: &syn::Attribute, expected: &str) -> String {
19  if let syn::Meta::NameValue(nv) = &attr.meta {
20    if let syn::Expr::Lit(syn::ExprLit {
21      lit: syn::Lit::Str(s),
22      ..
23    }) = &nv.value
24    {
25      return s.value();
26    }
27  }
28  panic!("only accepts #[{expected} = \"value\"] attribute syntax, mandatory double-quotes")
29}
