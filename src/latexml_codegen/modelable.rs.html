1use latexml_core::{common::error::*, fatal, s};
2use once_cell::sync::Lazy;
3use proc_macro::TokenStream;
4use quote::quote;
5use regex::Regex;
6use syn::DeriveInput;
7
8static TAG_MODEL_LINE: Lazy<Regex> =
9  Lazy::new(|| Regex::new(r"^([^\{]+)\{(.*?)\}\((.*?)\)$").unwrap());
10// Mirrors Perl Model.pm L149: `m/^([^:=]+):=\(?([^)]*?)\)?$/` — the
11// `\(?…\)?` pair strips the surrounding parens from
12// `classname:=(elt1,elt2,...)` so the elements split cleanly.
13static CLASS_MODEL_LINE: Lazy<Regex> =
14  Lazy::new(|| Regex::new(r"^([^:=]+):=\(?([^)]*?)\)?$").unwrap());
15static NAMESPACE_MODEL_LINE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^([^=]+)=(.*?)$").unwrap());
16
17pub fn load_model(input: DeriveInput) -> Result<TokenStream> {
18  let name = crate::attr_name_value_str(&input.attrs[0], "name");
19
20  // Read from `latexml_core`'s embedded RelaxNG table (this proc-macro crate links
21  // latexml_core, whose build.rs `include_str!`s the tree), NOT `pathname::find` —
22  // that resolves cwd-relative, so it works only inside our checkout and panics
23  // `Model "LaTeXML" not found` from a crates.io install. See CRATES_IO_PUBLISH.md B3b.
24  let embed_key = s!("{}.model", name);
25  let Some(source) = latexml_core::common::relaxng::embedded::lookup(&embed_key) else {
26    panic!(
27      "Model {name:?} not found in the embedded RelaxNG table, required to load a compiled \
28       model! (expected key {embed_key:?} from latexml_core/resources/RelaxNG/)"
29    )
30  };
31  // Only ever used for diagnostics below — there is no file to name anymore.
32  let path = s!("<embedded>/{}", embed_key);
33
34  let mut operations = Vec::new();
35  // NOTE: Do something automatic about this too!?!
36  // We'll need to generate namespace prefixes for all namespaces found in the doc!
37  operations.push(quote!(
38    model::register_document_namespace("", Some("http://dlmf.nist.gov/LaTeXML"));
39    model::set_schema(Relaxng{ name: #name.to_owned(), ..Relaxng::default()});
40  ));
41
42  // note_begin(&(s!("Compiling .model file: {}", path)));
43  for line in source.lines() {
44    if let Some(caps) = TAG_MODEL_LINE.captures(line) {
45      let tag = caps.get(1).map_or("", |m| m.as_str()).to_string();
46      let attr = caps.get(2).map_or("", |m| m.as_str()).to_string();
47      let children = caps.get(3).map_or("", |m| m.as_str()).to_string();
48
49      let attr_vec: Vec<String> = attr.split(',').map(ToString::to_string).collect();
50      let child_vec: Vec<String> = children.split(',').map(ToString::to_string).collect();
51
52      operations.push(quote!(
53        model::add_tag_attribute(#tag, vec![#(#attr_vec),*]);
54        model::add_tag_content(#tag, vec![#(#child_vec),*]);
55      ));
56    } else if let Some(caps) = CLASS_MODEL_LINE.captures(line) {
57      let classname = caps.get(1).map_or("", |m| m.as_str()).to_string();
58      let elements = caps.get(2).map_or("", |m| m.as_str()).to_string();
59      let elements_vec = elements
60        .split(',')
61        .map(ToString::to_string)
62        .collect::<Vec<String>>();
63
64      operations.push(quote!(
65        model::set_schema_class(#classname,
66          rustc_hash::FxHashSet::from_iter(vec![#(#elements_vec),*].into_iter()
67          .map(latexml_core::common::arena::pin_static)));
68      ));
69    } else if let Some(caps) = NAMESPACE_MODEL_LINE.captures(line) {
70      let prefix = caps.get(1).map_or("", |m| m.as_str());
71      let namespace = caps.get(2).map_or("", |m| m.as_str());
72      operations.push(quote!(
73        model::register_document_namespace(#prefix, Some(#namespace));
74      ));
75    } else {
76      fatal!(
77        Codegen,
78        Malformed,
79        s!(" Loaded model '{}' is malformatted at \"{}\"", path, line)
80      );
81    }
82  }
83
84  // note_end(&(s!("Compiling .model file: {}", path)));
85
86  Ok(TokenStream::from(quote!(
87    impl _ModelLoader {
88      fn build_model() {
89        #(#operations)*
90      }
91    }
92  )))
93}
