1use glob::glob;
2use proc_macro::TokenStream;
3use quote::{format_ident, quote};
4use syn::{DeriveInput, Lit, Meta};
5
6/// The only purpose of doing the glob for "*.tex" tests at compile-time is to
7/// make sure each TeX entry gets a dedicated #[test] header, and respectively
8/// increments the counter for the number of tests which have been run.
9/// In addition, this allows running more tests in parallel.
10pub fn compile_tests_at(input: DeriveInput) -> TokenStream {
11  let directory: String = match input.attrs[0].parse_meta().unwrap() {
12    Meta::NameValue(v) => match v.lit {
13      Lit::Str(v) => v.value(),
14      _ => {
15        panic!("only accepts #[directory = \"value\"] attribute, mandatory double-quotes (Lit)")
16      },
17    },
18    _ => panic!(
19      "only accepts #[directory = \"value\"] attribute, mandatory double-quotes (parse_meta)"
20    ),
21  };
22  // TODO: How do we best manage the relative directories changing from compile-time to test-time?
23  let test_functions: Vec<_> = glob(&format!("latexml_oxide/{directory}/*.tex"))
24    .unwrap()
25    .flatten()
26    .filter(|pb| {
27      // ensure there is an XML target
28      let mut xml_pb = pb.to_owned();
29      xml_pb.set_extension("xml");
30      xml_pb.is_file()
31    })
32    .map(|pb| {
33      let filebase = pb.file_stem().unwrap().to_string_lossy();
34      let file = pb.strip_prefix("latexml_oxide/").unwrap().to_string_lossy();
35      let fn_filename = filebase.replace(['-', ' ', '.'], "_");
36      // Rust identifiers cannot start with a digit; prepend `t_` for
37      // tex filenames like `3d-cone.tex` that would otherwise produce
38      // `3d_cone_test` (an invalid ident).
39      let fn_filename = if fn_filename
40        .chars()
41        .next()
42        .is_some_and(|c| c.is_ascii_digit())
43      {
44        format!("t_{fn_filename}")
45      } else {
46        fn_filename
47      };
48      let fn_name = format_ident!("{fn_filename}_test");
49      let attrs = quote!(#[test]);
50      quote!(
51        #attrs
52        fn #fn_name() {
53          latexml::util::test::latexml_test_single(#file, #filebase, #directory,
54            this_test_requires!(), this_test_dispatch!())
55        }
56      )
57    })
58    .collect();
59
60  quote!(
61    #(#test_functions)*
62  )
63  .into()
64}
