1//! TeX Hyphenation
2//!
3//! Core TeX Implementation for LaTeXML
4use crate::prelude::*;
5
6LoadDefinitions!({
7  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
8  // Hyphenation Family of primitive control sequences
9  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
10
11  //======================================================================
12  // explicit hyphenation
13  //----------------------------------------------------------------------
14  // - (discretionary hyphen)        d       inserts a discretionary hyphen.
15  // \discretionary    c  specifies a discretionary break in a paragraph.
16  def_primitive_noop("\\-")?;
17  DefMacro!("\\discretionary{}{}{}", "#3"); // No hyphenation here!
18
19  //======================================================================
20  // hyphenation tables
21  //----------------------------------------------------------------------
22  // \hyphenation      c  adds words to the hyphenation exception dictionary for the current
23  // language. \patterns         c  is used in INITEX to add patterns to the pattern dictionary
24  // for the current language.
25  DefMacro!("\\hyphenation GeneralText", None);
26  DefMacro!("\\patterns{}", None);
27
28  //======================================================================
29  // language choice
30  //----------------------------------------------------------------------
31  // \setlanguage      c  inserts a language whatsit in restricted horizontal mode.
32  // \language         pi selects a language to use with hyphenation and \patterns.
33  DefRegister!("\\language", Number!(0));
34  def_primitive_noop("\\setlanguage Number")?;
35
36  // \languagename — current language name (text). Perl's latex.dump captures
37  // `\def\languagename{nohyphenation}` as the format-time default
38  // (latex_dump.pool.ltxml L16522). Subsequent \selectlanguage{...} calls
39  // overwrite. Mirror that default exactly: apacite.sty L1422-1423 explicitly
40  // tests `\ifx\languagename<nohyphenation>` and skips its language-aware
41  // file lookup when the test passes, which is the Perl-clean path.
42  // Witness: 0906.3507 — Rust's prior "english" default caused apacite to
43  // load the system english.apc (newer than local apacite.sty), triggering
44  // an undefined `\if@APAC@natbib@apa` cascade.
45  DefMacro!("\\languagename", "nohyphenation");
46
47  //======================================================================
48  // codepoints used for hyphenation
49  //----------------------------------------------------------------------
50  // \hyphenchar       iq holds the current hyphen character used with hyphenation.
51  // \defaulthyphenchar pi is the \hyphenchar value to use when a new font is loaded.
52  // \lefthyphenmin    pi is the minimum number of characters that must appear before the first
53  // hyphen in an hyphenated word. \righthyphenmin   pi is the minimum number of characters that
54  // must appear after the last hyphen in an hyphenated word. \uchyph           pi prevents
55  // hyphenation of uppercase words unless this is positive.
56
57  // Perl: getter looks up $$fontinfo{hyphenchar}, setter stores in fontinfo
58  // Uses shared font key so fonts with same name+size share hyphenchar
59  // Perl: $$fontinfo{hyphenchar} = $value (modifies shared hash directly, unscoped)
60  DefRegister!("\\hyphenchar FontToken", Number::new(b'-' as i64),
61    getter => sub[args] {
62      let font_token = args.remove(0).expected_token();
63      let cs_str = font_token.to_string();
64      // Resolve to canonical font identity via Primitive.font_id so
65      // `\let`-aliased fonts share hyphenchar storage. Mirrors the
66      // fontdimen indirection in tex_fonts.rs.
67      let canonical_cs = lookup_meaning(&font_token)
68        .and_then(|m| if let Stored::Primitive(p) = m { p.font_id }
69                      else { None })
70        .map(|fid| {
71          let s = with(fid, |x| x.to_string());
72          s.strip_prefix("fontinfo_").unwrap_or(&s).to_string()
73        })
74        .unwrap_or_else(|| cs_str.clone());
75      let hc_key = with_value(&s!("font_shared_key_{canonical_cs}"), |v| match v {
76        Some(Stored::String(s)) => with(*s, |sk| s!("hyphenchar_{sk}")),
77        _ => s!("hyphenchar_{canonical_cs}"),
78      });
79      with_value(&hc_key, |v| match v {
80        Some(Stored::Number(n)) => *n,
81        _ => Number::new(b'-' as i64),
82      })
83    },
84    setter => sub[value, _scope, args] {
85      let font_token = args.remove(0).expected_token();
86      let cs_str = font_token.to_string();
87      let canonical_cs = lookup_meaning(&font_token)
88        .and_then(|m| if let Stored::Primitive(p) = m { p.font_id }
89                      else { None })
90        .map(|fid| {
91          let s = with(fid, |x| x.to_string());
92          s.strip_prefix("fontinfo_").unwrap_or(&s).to_string()
93        })
94        .unwrap_or_else(|| cs_str.clone());
95      let hc_key = with_value(&s!("font_shared_key_{canonical_cs}"), |v| match v {
96        Some(Stored::String(s)) => with(*s, |sk| s!("hyphenchar_{sk}")),
97        _ => s!("hyphenchar_{canonical_cs}"),
98      });
99      assign_value(
100        &hc_key,
101        Stored::Number(value.into()),
102        Some(Scope::Global),
103      );
104    }
105  );
106  // \defaulthyphenchar lives in tex_fonts.rs (Perl: TeX_Fonts.pool.ltxml L78);
107  // referenced as a comment here in the Perl source (L53) but not defined.
108  DefRegister!("\\lefthyphenmin", Number!(0));
109  DefRegister!("\\righthyphenmin", Number!(0));
110  DefRegister!("\\uchyph", Number!(1));
111});
