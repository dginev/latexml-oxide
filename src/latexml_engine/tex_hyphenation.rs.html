1//! TeX Hyphenation
2//!
3//! Core TeX Implementation for LaTeXML
4use crate::prelude::*;
5LoadDefinitions!({
6  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
7  // Hyphenation Family of primitive control sequences
8  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
9
10  //======================================================================
11  // explicit hyphenation
12  //----------------------------------------------------------------------
13  // - (discretionary hyphen)        d       inserts a discretionary hyphen.
14  // \discretionary    c  specifies a discretionary break in a paragraph.
15  DefPrimitive!("\\-", None);
16  DefMacro!("\\discretionary{}{}{}", "#3"); // No hyphenation here!
17
18  //======================================================================
19  // hyphenation tables
20  //----------------------------------------------------------------------
21  // \hyphenation      c  adds words to the hyphenation exception dictionary for the current
22  // language. \patterns         c  is used in INITEX to add patterns to the pattern dictionary
23  // for the current language.
24  DefMacro!("\\hyphenation GeneralText", None);
25  DefMacro!("\\patterns{}", None);
26
27  //======================================================================
28  // language choice
29  //----------------------------------------------------------------------
30  // \setlanguage      c  inserts a language whatsit in restricted horizontal mode.
31  // \language         pi selects a language to use with hyphenation and \patterns.
32  DefRegister!("\\language", Number!(0));
33  DefPrimitive!("\\setlanguage Number", None);
34
35  // \languagename — current language name (text). Perl's latex.dump captures
36  // `\def\languagename{nohyphenation}` as the format-time default
37  // (latex_dump.pool.ltxml L16522). Subsequent \selectlanguage{...} calls
38  // overwrite. Mirror that default exactly: apacite.sty L1422-1423 explicitly
39  // tests `\ifx\languagename<nohyphenation>` and skips its language-aware
40  // file lookup when the test passes, which is the Perl-clean path.
41  // Witness: 0906.3507 — Rust's prior "english" default caused apacite to
42  // load the system english.apc (newer than local apacite.sty), triggering
43  // an undefined `\if@APAC@natbib@apa` cascade.
44  DefMacro!("\\languagename", "nohyphenation");
45
46  //======================================================================
47  // codepoints used for hyphenation
48  //----------------------------------------------------------------------
49  // \hyphenchar       iq holds the current hyphen character used with hyphenation.
50  // \defaulthyphenchar pi is the \hyphenchar value to use when a new font is loaded.
51  // \lefthyphenmin    pi is the minimum number of characters that must appear before the first
52  // hyphen in an hyphenated word. \righthyphenmin   pi is the minimum number of characters that
53  // must appear after the last hyphen in an hyphenated word. \uchyph           pi prevents
54  // hyphenation of uppercase words unless this is positive.
55
56  // Perl: getter looks up $$fontinfo{hyphenchar}, setter stores in fontinfo
57  // Uses shared font key so fonts with same name+size share hyphenchar
58  // Perl: $$fontinfo{hyphenchar} = $value (modifies shared hash directly, unscoped)
59  DefRegister!("\\hyphenchar FontToken", Number::new(b'-' as i64),
60    getter => sub[args] {
61      let font_token = args.remove(0).expected_token();
62      let cs_str = font_token.to_string();
63      // Look up the shared font key via with_value to avoid cloning the
64      // Stored envelope on the String arm (just carrying a Copy SymStr).
65      let hc_key = state::with_value(&s!("font_shared_key_{cs_str}"), |v| match v {
66        Some(Stored::String(s)) => arena::with(*s, |sk| s!("hyphenchar_{sk}")),
67        _ => s!("hyphenchar_{cs_str}"),
68      });
69      state::with_value(&hc_key, |v| match v {
70        Some(Stored::Number(n)) => *n,
71        _ => Number::new(b'-' as i64),
72      })
73    },
74    setter => sub[value, _scope, args] {
75      let font_token = args.remove(0).expected_token();
76      let cs_str = font_token.to_string();
77      // Look up the shared font key via with_value — same reasoning as above.
78      let hc_key = state::with_value(&s!("font_shared_key_{cs_str}"), |v| match v {
79        Some(Stored::String(s)) => arena::with(*s, |sk| s!("hyphenchar_{sk}")),
80        _ => s!("hyphenchar_{cs_str}"),
81      });
82      // Perl stores directly in fontinfo hash (unscoped/global)
83      state::assign_value(
84        &hc_key,
85        Stored::Number(value.into()),
86        Some(Scope::Global),
87      );
88    }
89  );
90  // \defaulthyphenchar lives in tex_fonts.rs (Perl: TeX_Fonts.pool.ltxml L78);
91  // referenced as a comment here in the Perl source (L53) but not defined.
92  DefRegister!("\\lefthyphenmin", Number!(0));
93  DefRegister!("\\righthyphenmin", Number!(0));
94  DefRegister!("\\uchyph", Number!(1));
95});
