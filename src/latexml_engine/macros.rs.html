2/// Macro for compiling string construction replacements into closures
3/// that execute the needed ops in libxml.
4/// Approach borrowed from diesel-codegen
5macro_rules! compile_replacement {
6  ($var:ident, $replacement:literal) => {{
7    use latexml_core::definition::ReplacementClosure;
8    #[derive(CompileReplacement)]
9    #[replacement=$replacement]
10    struct _Dummy;
11    let tmp: Option<ReplacementClosure> = this_replacement!();
12    $var = tmp;
13  }};
14}
