5use crate::prelude::*;
6LoadDefinitions!({
7  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
8  // Macro Family of primitive control sequences
9  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
10  //
11  //======================================================================
12  // Basics
13  //----------------------------------------------------------------------
14  // \begingroup       c  starts a group that must be ended by \endgroup.
15  // \endgroup         c  ends a group that was begun by \begingroup.
16  // \relax            c  is a control sequence which typesets nothing.
17  // \afterassignment  c  saves a token and inserts it after the next assignment.
18  // \aftergroup       c  saves a token and inserts it after the current group is complete.
19  DefPrimitive!("\\begingroup", {
20    begingroup();
21  });
22  DefPrimitive!("\\endgroup", {
23    endgroup()?;
24  });
25  // This makes \relax disappear completely after digestion
26  // (which seems most TeX like).
27  DefPrimitive!(T_CS!("\\relax"), None, {});
28  // \protect is used in LaTeX for robust commands; acts like \relax in the base engine.
29  Let!("\\protect", "\\relax");
30  //## However, this keeps a box, so it can appear in UnTeX
31  //## DefPrimitive('\relax',undef);
32  //# But if you do that, you've got to watch out since it usually
33  //## shouldn't be a box; See the isRelax code in handleScripts, below
34
35  // NON-STANDARD: Internal token produced by Gullet in response to \dont_expand;
36  // Acts like \relax, but isn't equal to it.
37  DefPrimitive!(T_CS!("\\special_relax"), None, {});
38
39  // \afterassignment saves ONE token (globally!) to execute after the next assignment
40  DefPrimitive!("\\afterassignment Token", sub[(t)] {
41    assign_value("afterAssignment", t, Some(Scope::Global));
42  });
43  // \aftergroup saves ALL tokens (from repeated calls) to be executed IN ORDER after the next
44  // egroup or }
45  DefPrimitive!("\\aftergroup Token", sub[(t)] { push_value("afterGroup", t) });
46
47  //======================================================================
48  // CSName
49  //----------------------------------------------------------------------
50  // \csname           c  forms a control sequence name from the characters making up a collection
51  // of tokens. \endcsname        c  is used with \csname to make a control sequence name.
52  DefParameterType!(CSName, reader => reader!( _inner, _extra, { read_cs_name() }));
53  // Quiet version: used by \ifcsname — no errors for non-expandable CS tokens
54  DefParameterType!(CSNameQuiet, reader => reader!( _inner, _extra, { read_cs_name_quiet() }));
55  DefMacro!("\\csname CSName", sub[(token)] {
56    if !has_meaning(&token) {
57      let relax_meaning = lookup_meaning(&TOKEN_RELAX).unwrap();
58      assign_meaning(&token, relax_meaning, None);
59    }
60    token
61  });
62
63  DefPrimitive!("\\endcsname", {
64    if !state::lookup_bool("SUPPRESS_UNEXPECTED_ERRORS") {
65      Error!("unexpected", "\\endcsname", "Extra \\endcsname");
66    }
67  });
68
69  //======================================================================
70  // Definition flags
71  //----------------------------------------------------------------------
72  // \global        c  is an assignment prefix which makes the assignment transcend its group.
73  // \long          c  is a prefix for definitions which require multi-paragraph arguments.
74  // \outer         c  is a prefix for a definition which restricts where the definition may be
75  // used. \globaldefs    pi if positive, all assignments are global; if negative, \global is
76  // ignored.
77
78  DefPrimitive!("\\global",{ state::set_prefix("global"); }, is_prefix => true);
79  DefPrimitive!("\\long",  { state::set_prefix("long");   }, is_prefix => true);
80  DefPrimitive!("\\outer", { state::set_prefix("outer");  }, is_prefix => true);
81
82  DefRegister!("\\globaldefs", Number!(0));
83  //======================================================================
84  // Definitions
85  //----------------------------------------------------------------------
86  // \def          c  defines a macro.
87  // \edef         c  is similar to \def, except control sequences in the replacement
88  //                  text are expanded when the definition is made.
89  // \gdef         d  is equivalent to `\global\def'.
90  // \xdef         d  is equivalent to `\global\edef'.
91  DefPrimitive!("\\def SkipSpaces Token UntilBrace DefPlain",
92    sub[(cs,params,body)] {
93      do_def(false,cs,params,body)?;
94    },
95    locked => true);
96  DefPrimitive!("\\gdef SkipSpaces Token UntilBrace DefPlain",
97    sub[(cs,params,body)] {
98      do_def(true,cs,params,body)?;
99    },
100    locked => true);
101  DefPrimitive!("\\edef SkipSpaces Token UntilBrace DefExpanded",
102    sub[(cs,params,body)] {
103      do_def(false,cs,params,body)?;
104    },
105    locked => true);
106  DefPrimitive!("\\xdef SkipSpaces Token UntilBrace DefExpanded",
107    sub[(cs,params,body)] {
108      do_def(true,cs,params,body)?;
109    },
110    locked => true);
111  //======================================================================
112  // Copying definitions
113  //----------------------------------------------------------------------
114  //  \let       c  gives a control sequence a token's current meaning.
115  // \futurelet  c  `<cs> <token1> <token2>' is equivalent to `\let <cs> = <token2> <token1>
116  // <token2>'.
117  // `\let` — TeXbook Ch 24:
118  //   `\let <cs> <equals> <one optional space> <token>`
119  // where `<equals> = <optional spaces> | <optional spaces> =`,
120  // and `<one optional space>` is exactly one space if present.
121  //
122  // Perl uses the parameter spec
123  //   `\let SkipSpaces Token SkipSpaces SkipMatch:= Skip1Space Token`
124  // which in our engine routed the optional-`=` consumption through
125  // the generic Match parameter type's unread-on-no-match recovery.
126  // Implementing the read sequence explicitly here is Perl-faithful
127  // and TeX-faithful, and avoids a subtle interaction with the
128  // generic recovery path: `<one optional space>` is always tried,
129  // independent of whether `=` was consumed.
130  DefPrimitive!("\\let SkipSpaces Token", sub[(token1)] {
131    // <equals> = optional-spaces ['=']
132    gullet::skip_spaces()?;
133    if let Some(t) = gullet::read_token()? {
134      let is_eq =
135        t.get_catcode() == latexml_core::token::Catcode::OTHER && t.text == pin!("=");
136      if !is_eq {
137        gullet::unread_one(t);
138      }
139    }
140    // <one optional space>
141    gullet::skip_one_space(false)?;
142    // <token>
143    let token2 = gullet::read_token()?.unwrap_or_else(|| T_CS!("\\relax"));
144    Let!(token1, token2);
145  });
146  DefPrimitive!("\\futurelet Token Token Token", sub[(cs, token1, token2)] {
147    // NOT expandable, but puts tokens back
148    gullet::unread(Tokens!(token1,token2));
149    Let!(cs, token2);
150  });
151  //======================================================================
152  // Expansion control
153  //----------------------------------------------------------------------
154  // \expandafter      c  `<token1><token2>' is equivalent to `<token1> expansion of <token2>'.
155  // \noexpand         c  prevents the expansion of the following token.
156  DefMacro!("\\expandafter Token Token", sub[(tok, xtok)] {
157    let mut xtok = xtok;
158    let mut skipped : Vec<Token> = vec![tok];
159    while xtok.defined_as(&TOKEN_EXPANDAFTER) {
160      if let Some(ntok) = gullet::read_token()? {
161        skipped.push(ntok);
162        if let Some(nxtok) = gullet::read_token()? {
163          xtok = nxtok;
164        } else {
165          Error!("expected","expandafter", "\\expandafter wrongly used without 2 arguments.");
166        }
167      } else {
168        Error!("expected", "expandafter", "\\expandafter wrongly used without 2 arguments.");
169      }
170    }
171    if let Some(defn) = lookup_expandable(&xtok, None)? {
172      state::local_current_token(xtok);
173      let invoked = defn.invoke(true)?;
174      if !invoked.is_empty() {
175        skipped.extend(invoked.unlist()); // Expand `xtok` ONCE ONLY!
176      }
177      state::expire_current_token();
178    } else if !has_meaning(&xtok) {
179      // Undefined token is an error, as expansion is expected.
180      // BUT The unknown token is NOT consumed, (see TeX B book, item 367)
181      // since probably in a real TeX run it would have been defined.
182      state::generate_error_stub(&xtok)?;
183      skipped.push(xtok);
184    } else {
185      skipped.push(xtok);
186    };
187    Ok(Tokens::new(skipped))
188  });
189  // If next token is expandable, prefix it with the internal marker \dont_expand
190  // That token is never defined, explicitly handled in Gullet & should never escape the Gullet.
191  //
192  // Mirrors Perl `TeX_Macro.pool.ltxml:228-235`: uses `isDontExpandable`
193  // (not `lookupExpandable`). The predicate is true for CS/Active that are
194  // expandable OR undefined — so `\noexpand` smuggles `\dont_expand` for
195  // both, which is what makes `\noexpand\undef` ≠ `\undef` under `\ifx`
196  // (Gullet rewrites `\dont_expand X` into `\special_relax` with `X`
197  // smuggled in slot[2]; that asymmetry is the whole point).
198  DefMacro!(T_CS!("\\noexpand"), None, {
199    if let Some(token) = gullet::read_token()? {
200      let cc = token.get_catcode();
201      if matches!(cc, Catcode::CS | Catcode::ACTIVE) && state::is_dont_expandable(&token) {
202        vec![T_CS!("\\dont_expand"), token]
203      } else {
204        vec![token]
205      }
206    } else {
207      // Missing token likely the result of "{\noexpand}" for which TeX would be unperturbed
208      Vec::new()
209    }
210  });
211  // NON-STANDARD:
212  DefPrimitive!(T_CS!("\\dont_expand"), None, {
213    Error!(
214      "misdefined",
215      "\\dont_expand",
216      "The token \\dont_expand should never reach Stomach!"
217    );
218  });
219
220  //======================================================================
221  // \the
222  //----------------------------------------------------------------------
223  // \the              c  returns character tokens for an internal quantity's or parameter's current
224  // value. The argument to \the is a variety of "Internal Quantities", being parameters,
225  // registers, internal registers, codenames, etc. See TeX Book, pp.214--215.
226  // [Since \the is expandable, perhaps should just be built into \the's code? Never need to revert]
227  DefMacro!("\\the", {
228    if let Some(token) = read_x_token(None, false, None)? {
229      // Follow `\let`-alias chain so that `\the \tex_count:D` (= `\count`)
230      // and `\the \__int_eval:w` (= `\numexpr`) — and any other expl3
231      // primitive-alias to a register-style CS — find the underlying
232      // register definition. Without this, `lookup_definition` returns
233      // None for `Stored::Token` let-aliases and `\the` falls through
234      // to the "undefined" branch, emitting a 0 plus a stub-installation.
235      // Driver: 2406.14142 expl3 regex VM (`\the \__int_eval:w ... \__int_eval_end:`).
236      let mut effective_token = token;
237      for _ in 0..16 {
238        match state::with_meaning(&effective_token, |m| match m {
239          Some(Stored::Token(t)) => Some(*t),
240          _ => None,
241        }) {
242          Some(t) if t != effective_token => effective_token = t,
243          _ => break,
244        }
245      }
246      if let Some(defn) = lookup_definition(&effective_token)? {
247        if defn.is_register() {
248          // SOME kind of register is acceptable
249          let args = if let Some(params) = defn.get_parameters() {
250            params.read_arguments(None)?
251          } else {
252            Vec::new()
253          };
254          return match defn.value_of(args) {
255            Some(RegisterValue::Token(t)) => Ok(Tokens!(t)),
256            Some(RegisterValue::Tokens(ts)) => Ok(ts),
257            Some(other) => Ok(Tokens!(Explode!(other.to_string()))),
258            None => Ok(Tokens!()),
259          };
260        } else if defn.get_cs_name() == "\\font" {
261          // HACK to get the \fontcmd that would have selected the current font (see FontDef)
262          if let Some(Stored::Token(t)) = state::lookup_value("current_FontDef") {
263            return Ok(Tokens!(t));
264          } else {
265            return Ok(Tokens!(T_CS!("\\lx@default@font")));
266          }
267        } else {
268          // Perl: elsif ($defn->isFontDef) { return $defn->getCS; }
269          // Check if this is a font CS defined by \font (has fontinfo in state)
270          let cs_str = token.to_string();
271          if let Some(Stored::Font(_)) = state::lookup_value(&s!("fontinfo_{cs_str}")) {
272            return Ok(Tokens!(token));
273          }
274        }
275      } else {
276        // the token is Undefined
277        if token.get_catcode() == Catcode::CS {
278          // but IS a cs \something
279          Error!(
280            "expected",
281            "<register>",
282            "A <register> was supposed to be here",
283            s!("Got {} Defining it now.", token)
284          );
285          // Hackery: to avoid potential repeated errors, define it now as a number register
286          def_register(token, None, Number!(0), None)?; // Dimension, or what?
287          return Ok(Tokens!(T_OTHER!("0")));
288        }
289      }
290      // If we fall through to here, whatever $token is shouldn't have been used with \the
291      let (the_t, msg) =
292        token.with_str(|tstr| (s!("\\the{tstr}"), s!("You can't use {tstr} after \\the")));
293      Error!("unexpected", the_t, msg);
294      T_OTHER!("0")
295    } else {
296      Error!(
297        "expected",
298        "<register>",
299        "A <register> was supposed to be here. Got nothing."
300      );
301      T_OTHER!("0")
302    }
303  });
304});
