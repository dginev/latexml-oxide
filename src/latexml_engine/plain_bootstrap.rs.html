1// plain_bootstrap — Bootstrap code for reading plain.tex for LaTeXML.
2// Corresponds to Perl Engine/plain_bootstrap.pool.ltxml.
3//
4// Loaded BEFORE the plain dump. Contains stubs that override plain.tex's
5// own allocation mechanisms with LaTeXML's versions.
6use crate::prelude::*;
7
8#[rustfmt::skip]
9LoadDefinitions!({
10  // INITEX letter/digit char-code initialization. Per plain.tex L112-113
11  // INITEX (the initial TeX engine) sets `\uccode\x=\X`, `\uccode\X=\X`,
12  // `\lccode\x=\x`, `\lccode\X=\x` for every letter; mathcodes for
13  // digits & letters per plain.tex Table 17.2; sfcode 999 for letters.
14  // Previously this loop lived only in `plain_base.rs` (NODUMP path).
15  // With `plain_dump` loaded instead, the initialization was missing —
16  // the dump-build snapshot is taken BEFORE plain.tex's raw load, and
17  // plain.tex itself doesn't (re)assign these codes (it assumes INITEX
18  // has). So with the dump active, `\uccode` for letters returned 0 and
19  // `\MakeUppercase` produced lowercase output. Mirror INITEX here in
20  // `plain_bootstrap` (which runs in BOTH paths AND before the
21  // dump-build snapshot) — the resulting codes land in the snapshot
22  // baseline and don't need to round-trip through the dump diff.
23  // NOTE: mathcodes (digits and letters) are deliberately NOT set here.
24  // plain_base.rs sets them via 0x7100+letter / 0x7030+digit (class 7
25  // variable family). With the dump path active these mathcodes route
26  // input through `decode_math_char`, which preempts the `DefMath`-
27  // based per-letter handlers in base_xmath that produce the expected
28  // unicode-substituted output (e.g. `\mathbb{A}` → 𝔸, `\mathbb{0}` →
29  // 𝟘). Without a full Stored::Font roundtrip in the dump these paths
30  // diverge — 22_fonts::bbold_test regressed when letter/digit
31  // mathcodes were eagerly set in this bootstrap.
32  // lccode/uccode/sfcode for letters DO need this initialization so
33  // `\MakeUppercase` (which consults uccode, not mathcode) works
34  // outside math mode in the dump path.
35  for letter in b'A'..=b'Z' {
36    assign_lccode(letter, letter + 32, Some(Scope::Global));
37    assign_uccode(letter, letter, Some(Scope::Global));
38    assign_sfcode(letter as char, 999u16, Some(Scope::Global));
39
40    assign_lccode(letter + 32, letter + 32, Some(Scope::Global));
41    assign_uccode(letter + 32, letter, Some(Scope::Global));
42  }
43
44  // Perl: plain_bootstrap.pool.ltxml L19-27 — CSS-based TeX logo
45  DefConstructor!("\\TeX", "<ltx:text class='ltx_TeX_logo' cssstyle='letter-spacing:-0.2em; margin-right:0.2em'
46  >T<ltx:text cssstyle='font-variant:small-caps;font-size:120%;' yoffset='-0.2ex'
47  >e</ltx:text>X</ltx:text>",
48    locked => true,
49    enter_horizontal => true,
50    sizer => sub[_whatsit] { Ok((Dimension!("1.9em"), Dimension!("1.6ex"), Dimension!("0.5ex"))) });
51
52  //======================================================================
53  // Perl: plain_bootstrap.pool.ltxml L32-33
54  // Use LaTeXML's register allocation to avoid allocating same slot twice
55  DefMacro!("\\alloc@{}{}{}{}{}", r"\lx@alloc@{#2}{\count1#1}{#3}{#5}", locked => true);
56  DefMacro!("\\ch@ck{}{}{}", None, locked => true);
57
58  //======================================================================
59  // Perl: plain_bootstrap.pool.ltxml L37-40
60  //   DefPrimitive('\newif DefToken', sub { DefConditionalI($cs, undef); }, locked => 1);
61  // Use LaTeXML's conditional machinery. State side-effect → DefPrimitive
62  // (stomach level), not DefMacro (expansion level).
63  DefPrimitive!("\\newif DefToken", sub[(cs)] {
64    def_conditional(cs, None, None, ConditionalOptions::default())?;
65  }, locked => true);
66
67  //======================================================================
68  // Perl: plain_bootstrap.pool.ltxml L43
69  DefPrimitive!("\\leavevmode", { enter_horizontal(); });
70});
