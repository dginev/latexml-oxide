1//! TeX Logic
2//!
3//! Core TeX Implementation for LaTeXML

5use crate::prelude::*;
6LoadDefinitions!({
7  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
8  // Logic Family of primitive control sequences
9  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
10  //======================================================================
11  // Basic logic
12  //----------------------------------------------------------------------
13  // \iftrue           c  is a conditional which is always true.
14  // \iffalse          c  is a conditional which is always false.
15  // \else             c  begins the false part of a conditional.
16  // \fi               c  is the concluding command of a conditional.
17  // \or               c  separates cases in an \ifcase conditional.
18  DefConditional!("\\iftrue", { true });
19  DefConditional!("\\iffalse", { false });
20  DefConditional!("\\else"); // BUILT-IN to Definition
21  DefConditional!("\\or"); // BUILT-IN to Definition
22  DefConditional!("\\fi"); // BUILT-IN to Definition
23
24  //======================================================================
25  // Token testing
26  //----------------------------------------------------------------------
27  // \if               c  tests if two tokens have the same character codes (i.e., values 0-256).
28  // \ifx              c  tests if two tokens are the same.
29  // \ifcat            c  tests if two tokens have the same category codes (i.e., values 0-16).
30  DefParameterType!(ExpandedIfToken, sub[_inner, _extra] {
31    let token_opt = gullet::read_x_token(Some(false), true, None)?;
32    match token_opt {
33      Some(t) => t,
34      None => {
35        Error!("expected", "ExpandedIfToken",
36          "conditional expected a token argument, came back empty. Falling back to \\lx@empty");
37        T_CS!("\\lx@empty")
38      }}
39  });
40
41  DefConditional!("\\if ExpandedIfToken ExpandedIfToken", sub[(left,right)] {
42    left.get_charcode() == right.get_charcode()
43  });
44  DefConditional!("\\ifx Token Token", sub[(left,right)] { x_equals(&left, &right) });
45  DefConditional!("\\ifcat ExpandedIfToken ExpandedIfToken", sub[(left,right)] {
46    left.get_catcode() == right.get_catcode()
47  });
48
49  //======================================================================
50  // Number testing
51  //----------------------------------------------------------------------
52  // \ifnum            c  compares two integers.
53  // \ifodd            c  tests for an odd integer.
54  // \ifcase           c  begins a multi-case conditional.
55  // Perl (2026-03-18): Relation parameter type = skip spaces + readXToken (for <, =, >)
56  DefConditional!("\\ifnum Number Relation Number", sub[(u,rel,v)] {
57    compare(u.value_of(), rel, v.value_of())
58  });
59  DefConditional!("\\ifodd Number", sub[(u)] {
60    u.value_of() % 2 == 1
61  });
62  DefConditional!("\\ifcase Number");
63
64  //======================================================================
65  // Dimension testing
66  //----------------------------------------------------------------------
67  // \ifdim            c  compares two dimensions.
68  DefConditional!("\\ifdim Dimension Relation Dimension", sub[(u,rel,v)] {
69    compare(u.value_of(), rel, v.value_of())
70  });
71
72  //======================================================================
73  // Box testing
74  //----------------------------------------------------------------------
75  // \ifhbox           c  is true if a box register contains an \hbox.
76  // \ifvbox           c  is true if a box register contains a \vbox.
77  // \ifvoid           c  is true if a box register is void.
78  //
79  // Perl TeX_Logic.pool.ltxml L111-113: \ifvoid / \ifhbox / \ifvbox.
80  DefConditional!("\\ifvoid Number", sub[(arg)] { classify_box(arg)?.is_empty() });
81  DefConditional!("\\ifhbox Number", sub[(arg)] { classify_box(arg)? == "hbox" });
82  DefConditional!("\\ifvbox Number", sub[(arg)] { classify_box(arg)? == "vbox" });
83
84  //======================================================================
85  // Mode testing
86  //----------------------------------------------------------------------
87  // \ifhmode          c  is true if TeX is in horizontal or restricted horizontal mode.
88  // \ifinner          c  is true if TeX is in internal vertical, restricted horizontal, or
89  // nondisplay math mode. \ifmmode          c  is true if TeX is in math or display math mode.
90  // \ifvmode          c  is true if TeX is in vertical or internal vertical mode.
91
92  DefConditional!("\\ifvmode", {
93    let mode = state::lookup_string_from_sym(pin!("MODE"));
94    mode.ends_with("vertical")
95  });
96  DefConditional!("\\ifhmode", {
97    let mode = state::lookup_string_from_sym(pin!("MODE"));
98    mode.ends_with("horizontal")
99  });
100  DefConditional!("\\ifinner", {
101    let mode = state::lookup_string_from_sym(pin!("MODE"));
102    matches!(
103      mode.as_str(),
104      "restricted_horizontal" | "internal_vertical" | "math"
105    )
106  });
107  // Perl: LookupValue('MODE') =~ /math$/
108  DefConditional!("\\ifmmode", {
109    let mode = state::lookup_string_from_sym(pin!("MODE"));
110    mode.ends_with("math")
111  });
112
113  //======================================================================
114  // I/O testing
115  //----------------------------------------------------------------------
116  // \ifeof c tests for the end of a file .
117  DefConditional!("\\ifeof Number", sub[(port)] {
118    with_value(&s!("input_file:{}", port), |val_opt|
119      if let Some(Stored::Mouth(mouth)) = val_opt {
120        mouth.borrow().at_eof()
121      } else {
122        true
123      })
124  });
125});

127fn compare(u: i64, rel: Token, v: i64) -> bool {
128  // NOTE: One would expect this to be best written as an advanced match state::ent
129  // however, due to the shallow comparison of Cow<str> the Cow::Borrowed("<") and
130  // Cow::Owned("<") variants will NOT be equal via a destructuring match.
131  // However, since we've defined our own PartialEq trait over Token, an equality comparison
132  // will produce the right behavior
133  if rel == T_OTHER!("<") || rel == T_CS!("\\@@<") {
134    u < v
135  } else if rel == T_OTHER!("=") {
136    u == v
137  } else if rel == T_OTHER!(">") || rel == T_CS!("\\@@>") {
138    u > v
139  } else {
140    let message = s!(
141      "Expected a relational token for comparision. Got {:?} (cc {:?})",
142      rel,
143      rel.get_catcode()
144    );
145    let err = || {
146      Error!("expected", "<relationaltoken>", message);
147      Ok(())
148    };
149    err().ok();
150    false
151  }
