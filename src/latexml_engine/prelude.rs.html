104// Functions callable from constructor templates via &GetKeyVal(#1,key) syntax.
105// Returns Option<Digested> to be compatible with both attribute (to_attribute/to_string)
106// and body absorption (Into<Option<Digested>>) contexts in constructor templates.
107#[allow(non_snake_case)]
108pub fn GetKeyVal(keyval_opt: &Option<Digested>, key: &str) -> Option<Digested> {
109  match keyval_opt {
110    Some(digested) => match digested.data() {
111      DigestedData::KeyVals(keyval) => keyval.get_value_digested(key).cloned(),
112      _ => None,
113    },
114    _ => None,
115  }
