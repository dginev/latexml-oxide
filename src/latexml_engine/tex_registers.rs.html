5use crate::prelude::*;
6LoadDefinitions!({
7  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
8  // Registers Family of primitive control sequences

11  //======================================================================
12  // Accessing Registers
13  //----------------------------------------------------------------------
14  // \count            iq assigns an integer to a \count register.
15  // \dimen            iq assigns a <dimen> to a \dimen register.
16  // \skip             iq assigns <glue> to a \skip register.
17  // \muskip           iq assigns <muglue> to a \muskip register.
18  // \toks             iq assigns <replacement text> to a \toks register.
19
20  DefRegister!("\\count Number"  => Number::new(0));
21  DefRegister!("\\dimen Number"  => Dimension::new(0));
22  DefRegister!("\\skip Number"   => Glue::new(0));
23  DefRegister!("\\muskip Number" => MuGlue::new(0));
24  DefRegister!("\\toks Number"   => Tokens!());
25
26  //======================================================================
27  // Defining Registers, shorthands
28  //----------------------------------------------------------------------
29  // \countdef         c  creates a symbolic name for a \count register.
30  // \dimendef         c  creates a symbolic name for a \dimen register.
31  // \skipdef          c  creates a symbolic name for a \skip register.
32  // \muskipdef        c  creates a symbolic name for a \muskip register.
33  // \toksdef          c  creates a symbolic name for a \toks register.
34  DefPrimitive!("\\countdef SkipSpaces Token SkipSpaces SkipMatch:=", sub[(cs)] {
35    shorthand_def(cs, "\\count", Number::new(0).into())
36  });
37  DefPrimitive!("\\dimendef SkipSpaces Token SkipSpaces SkipMatch:=", sub[(cs)] {
38    shorthand_def(cs, "\\dimen", Dimension::new(0).into())
39  });
40  DefPrimitive!("\\skipdef SkipSpaces Token SkipSpaces SkipMatch:=", sub[(cs)] {
41    shorthand_def(cs, "\\skip", Glue::new(0).into())
42  });
43  DefPrimitive!("\\muskipdef SkipSpaces Token SkipSpaces SkipMatch:=", sub[(cs)] {
44    shorthand_def(cs, "\\muskip", MuGlue::new(0).into())
45  });
46  DefPrimitive!("\\toksdef SkipSpaces Token SkipSpaces SkipMatch:=", sub[(cs)] {
47    shorthand_def(cs, "\\toks", Tokens!().into())
48  });
49
50  // Candidate for use defining plain's \alloc@ and latex's \e@alloc
51  DefMacro!("\\lx@alloc@ DefToken {} {} DefToken", sub[(typ,tracker,allocator,cs)] {
52    use latexml_core::binding::def::dialect::allocate_register;
53    let type_str = typ.to_string();
54    let tracker_str = tracker.to_string();
55    let allocator_str = allocator.to_string();
56    // Check if allocator is a register-shorthand type (\countdef, \dimendef, etc.)
57    let is_stored = matches!(allocator_str.as_str(),
58      "\\countdef" | "\\dimendef" | "\\skipdef" | "\\muskipdef" | "\\toksdef");
59    let next = if is_stored {
60      // Use the register allocation system
61      if let Some(addr) = allocate_register(&type_str, &cs.to_string())? {
62        // Extract the number part from e.g. "\\count256"
63        addr.trim_start_matches(&type_str).to_string()
64      } else {
65        s!("0")
66      }
67    } else {
68      let xnext = lookup_number(&tracker_str).unwrap_or_else(|| Number::new(0));
69      let next = xnext.value_of() + 1;
70      next.to_string()
71    };
72    assign_value(&tracker_str,
73      Number::new(next.parse::<i64>().unwrap_or(0)), Some(Scope::Global));
74    let mut result: Vec<Token> = vec![T_CS!("\\allocationnumber")];
75    result.extend(Explode!(next));
76    result.push(T_CS!("\\relax"));
77    result.push(T_CS!("\\global"));
78    result.extend(allocator.unlist());
79    result.push(cs);
80    result.push(T_OTHER!("="));
81    result.push(T_CS!("\\allocationnumber"));
82    result
83  });
84
85  // Out of place, but utility for LaTeX-style \the<ctr>; used by Package's NewCounter
86  DefMacro!("\\lx@counter@arabic{}", sub[(ctr)] {
87    let ctr_name = Expand!(ctr).to_string();
88    ExplodeText!(CounterValue!(&ctr_name).value_of())
89  });
90
91  //======================================================================
92  // Numeric Registers
93  //----------------------------------------------------------------------
94  // \advance          c  increases or decreases a numeric variable.
95  // \multiply         c  multiplies a register by an integer.
96  // \divide           c  divides a register by an integer.
97
98  DefPrimitive!("\\advance Variable SkipKeyword:by", sub[(var)] {
99    if let ArgWrap::RegisterDefinition(dbox) = var {
100      let (defn_token, inner) = *dbox;
101      let defn_token_str = defn_token.to_string();
102      if !defn_token_str.is_empty() && defn_token_str != "missing" {
103        let defn_opt = lookup_register_definition(&defn_token);
104        local_current_token(defn_token);
105        if let Some(defn) = defn_opt {
106          let summand = read_value(defn.register_type().unwrap())?;
107          let defn_args : Vec<ArgWrap> = inner.clone();
108          let defn_value = defn.value_of(inner).unwrap_or_default();
109          defn.set_value(defn_value.add(summand), None, defn_args);
110        } else {
111          let message = s!("\\advance expected a defined variable for {:?}, found no definition",
112          defn_token_str);
113          Error!("expected","definition", message);
114        }
115        expire_current_token();
116      }
117    }
118  });
119
120  DefPrimitive!("\\multiply Variable SkipKeyword:by Number", sub[(var,scale)] {
121    if let ArgWrap::RegisterDefinition(dbox) = var {
122      let (varname, inner) = *dbox;
123      // Upgrade: Why are the arguments used twice here? Is there a way to avoid cloning them?
124      match lookup_register_definition(&varname) { Some(defn) => {
125        let defn_args : Vec<ArgWrap> = inner.clone();
126        let defn_value = defn.value_of(inner).unwrap_or_default();
127        defn.set_value(defn_value.multiply(scale), None, defn_args);
128      } _ => {
129        let message =
130          s!("\\multiply expected a defined variable for {:?}, found no definition", varname);
131        Error!("expected","definition", message);
132      }}
133    } else {
134      let message = s!("\\multiply expected a Variable argument, but got nothing.");
135      Error!("expected","variable", message);
136    }
137  });
138
139  DefPrimitive!("\\divide Variable SkipKeyword:by Number", sub[(var,scale)] {
140    if let ArgWrap::RegisterDefinition(dbox) = var {
141      let (varname, inner) = *dbox;
142      // Upgrade: Why are the arguments used twice here? Is there a way to avoid cloning them?
143      let defn_args : Vec<ArgWrap> = inner.clone();
144      match lookup_register_definition(&varname) { Some(defn) => {
145        let defn_value = defn.value_of(inner).unwrap_or_default();
146        let mut denominator = scale.value_f64();
147        if denominator == 0.0 {
148          Error!("misdefined", scale, "Illegal \\divide by 0; assuming 1");
149          denominator = 1.0;
150        }
151        defn.set_value(defn_value.divide(Float::new_f64(denominator)), None, defn_args);
152      } _ => {
153        let message =
154          s!("\\divide expected a defined variable for {:?}, found no definition", varname);
155        Error!("expected","definition", message);
156      }}
157    } else {
158      let message = s!("\\divide expected a Variable argument, but got nothing.");
159      Error!("expected","variable", message);
160    }
161  });
162});

164/// Note that these define a "shorthand" for eg. \count123, but are NOT macros!
165pub fn shorthand_def(cs: Token, address_type: &str, init: RegisterValue) -> Result<()> {
166  // Let w/o AfterAssign
167  let relax_meaning = lookup_meaning(&TOKEN_RELAX).unwrap();
168  assign_meaning(&cs, relax_meaning, None);
169  // define
170  let num = read_number()?;
171  let address = s!("{address_type}{}", num.value_of());
172  let options = Some(RegisterOptions {
173    address: Some(address),
174    ..RegisterOptions::default()
175  });
176  def_register(cs, None, init, options)?;
177  after_assignment();
178  Ok(())
179}
