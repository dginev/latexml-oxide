132  // \pdfximage [ image attr spec ] general text (h, v, m)
133  // Real pdfTeX reads optional `[image attr spec]` then a balanced text
134  // (the file path). Stub: drop a leading `[...]` if present, then
135  // consume one balanced general-text arg. No PDF emission. Driver:
136  // 2406.14142 (`\pdfximage{...}` in graphics-bbox-precompute path).
137  DefPrimitive!("\\pdfximage", sub[_args] {
138    gullet::skip_spaces()?;
139    if gullet::if_next(T_OTHER!("["))? {
140      // discard up to matching `]`
141      while let Some(t) = gullet::read_token()? {
142        if matches!(t.get_catcode(), Catcode::OTHER) && t.to_string() == "]" {
143          break;
144        }
145      }
146    }
147    gullet::skip_spaces()?;
148    let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
149    Ok(vec![])
150  });

175  // Perl: DefParameterType('OpenAnnotSpecification', sub { ... }, optional, undigested).
176  // Reads and discards the pdfTeX annotation-spec prefix:
177  //   reserveobjnum  | useobjnum <n>  | stream [attr <text>]
178  // then consumes the trailing general-text spec.
179  DefParameterType!(OpenAnnotSpecification, reader => reader!(_args, _extra, {
180    if read_keyword(&["reserveobjnum"])?.is_some() {
181      return Ok(ArgWrap::None);
182    } else if read_keyword(&["useobjnum"])?.is_some() {
183      let _ = gullet::read_number()?;
184    } else if read_keyword(&["stream"])?.is_some()
185      && read_keyword(&["attr"])?.is_some() {
186        gullet::skip_spaces()?;
187        let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
188      }
189    gullet::skip_spaces()?;
190    let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
191  }), optional => true);

234  // \pdfcolorstack stack_num {set|push|pop|current} [general_text]
235  //
236  // Perl pdfTeX.pool L210: reads stack-number + action keyword, then
237  // consumes a trailing general-text spec UNLESS the action was `pop`
238  // (which has no spec, just pops the top of the stack). All values
239  // are discarded — our engine doesn't emit PDF colorstack operations.
240  //
241  // Using OptionalMatch for each keyword matches the Perl signature.
242  // GeneralText is the balanced-group reader.
243  DefPrimitive!(
244    "\\pdfcolorstack Number OptionalMatch:set OptionalMatch:push OptionalMatch:pop OptionalMatch:current",
245    sub[(_number, _set, _push, pop, _current)] {
246      // If action was `pop`, there's no trailing general-text spec.
247      // Otherwise read and discard the general-text argument.
248      if pop.is_none() {

250        let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
251      }
252    }
