143  // \pdfximage [ image attr spec ] general text (h, v, m)
144  // Real pdfTeX reads optional `[image attr spec]` then a balanced text
145  // (the file path). Stub: drop a leading `[...]` if present, then
146  // consume one balanced general-text arg. No PDF emission. Driver:
147  // 2406.14142 (`\pdfximage{...}` in graphics-bbox-precompute path).
148  DefPrimitive!("\\pdfximage", sub[_args] {
149    gullet::skip_spaces()?;
150    if gullet::if_next(T_OTHER!("["))? {
151      // discard up to matching `]`
152      while let Some(t) = gullet::read_token()? {
153        if matches!(t.get_catcode(), Catcode::OTHER) && t.to_string() == "]" {
154          break;
155        }
156      }
157    }
158    gullet::skip_spaces()?;
159    let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
160    Ok(vec![])
161  });

196  DefParameterType!(OpenAnnotSpecification, reader => reader!(_args, _extra, {
197    if read_keyword(&["reserveobjnum"])?.is_some() {
198      return Ok(ArgWrap::None);
199    } else if read_keyword(&["useobjnum"])?.is_some() {
200      let _ = gullet::read_number()?;
201    } else if read_keyword(&["stream"])?.is_some()
202      && read_keyword(&["attr"])?.is_some() {
203        gullet::skip_spaces()?;
204        let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
205      }
206    gullet::skip_spaces()?;
207    let _ = gullet::read_balanced(ExpansionLevel::Off, false, true)?;
208  }), optional => true);
