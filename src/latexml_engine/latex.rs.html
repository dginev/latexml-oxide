17// Process-once cached env vars (see WISDOM #56 — getenv hot-path race).
18static DUMP_PATH: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_PATH").ok());
19static DUMP_DIR: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_DIR").ok());
20static INI_MODE: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_INI_MODE").is_some());
21static NODUMP: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_NODUMP").is_some());
22
23const DEV_DUMPS_DIR: &str = concat!(env!("CARGO_MANIFEST_DIR"), "/../resources/dumps");

25/// Perl `FindFile($format._dump, ...)` parity for the latex dump.
26/// Returns true when any `latex.YYYY.dump.txt` is reachable through env
27/// overrides, the exe-relative installed layout, the dev-tree path, or
28/// the embedded fallback. Used by `LoadFormat('latex')` to decide
29/// between the dump branch and the base branch.
30fn latex_dump_available() -> bool {
31  if *NODUMP {
32    return false;
33  }
34  if let Some(p) = DUMP_PATH.as_deref()
35    && Path::new(p).is_file()
36  {
37    return true;
38  }
39  let prefer = crate::dump_paths::detect_ambient_texlive_year();
40  if let Some(dir) = DUMP_DIR.as_deref()
41    && crate::dump_paths::resolve_versioned_in_dir(Path::new(dir), "latex", prefer).is_some()
42  {
43    return true;
44  }
45  if let Ok(exe) = std::env::current_exe()
46    && let Some(exe_dir) = exe.parent()
47  {
48    let installed = exe_dir.join("../resources/dumps");
49    if crate::dump_paths::resolve_versioned_in_dir(&installed, "latex", prefer).is_some() {
50      return true;
51    }
52    if crate::dump_paths::resolve_versioned_in_dir(exe_dir, "latex", prefer).is_some() {
53      return true;
54    }
55  }
56  let dev = Path::new(DEV_DUMPS_DIR);
57  if dev.is_dir() && crate::dump_paths::resolve_versioned_in_dir(dev, "latex", prefer).is_some() {
58    return true;
59  }
60  crate::embedded_dumps::embedded_latex_dump(prefer).is_some()
61}

63LoadDefinitions!({
64  //**********************************************************************
65  // Organized following
66  //  "LaTeX: A Document Preparation System"
67  //   by Leslie Lamport
68  //   2nd edition
69  // Addison Wesley, 1994
70  // Appendix C. Reference Manual
71  //**********************************************************************
72  // NOTE: This will be loaded after TeX.pool.ltxml, so it inherits.
73  //**********************************************************************
74
75  // Perl: LaTeX.pool.ltxml — LoadPool('TeX'); LoadFormat('latex');
76  LoadPool!("TeX");
77
78  InnerPool!(latex_bootstrap);
79
80  // In `--init=latex.ltx` (dump-build) mode, stop after latex_bootstrap.
81  // The same reasoning as in `tex.rs`'s plain branch: pre-loading
82  // latex_dump / latex_base / latex_constructs pollutes the snapshot
83  // and silences the diff for everything raw latex.ltx defines.
84  // `LATEXML_INI_MODE=1` is set by `bin/latexml_oxide.rs` BEFORE
85  // `prepare_session`, so this branch fires before latex.rs runs.
86  if *INI_MODE {
87    return Ok(());
88  }
89
90  // Perl `LoadFormat('latex')` strict split:
91  //   if dump available: bootstrap → dump → constructs (NO base)
92  //   else:              bootstrap → base → constructs (NO dump)
93  if !*NODUMP && latex_dump_available() {
94    if let Err(e) = crate::latex_dump::load_definitions() {
95      Warn!("latex_dump", "load", s!("{}", e));
96    }
97  } else {
98    // DEGRADED raw-load path: no precompiled LaTeX kernel dump is
99    // available. This silently fires `latex.ltx` + `expl3-code.tex` raw,
100    // which in turn hits known raw-load-only cascades (the expl3-code
101    // L33075 codepoint dangling-group, and the `\@expl@pop@filename@@`
102    // expl-status desync) that the dump avoids. Running the large canvas
103    // here inflates per-paper error counts by ~1000× on expl3-heavy
104    // articles (e.g. 2112.11932: 1 → 1003), masquerading as a Rust parity
105    // gap when it is really a missing-kernel-dump setup error.
106    //
107    // The dump is REQUIRED for canvas/parity work. Surface its absence
108    // loudly, once per process, unless `LATEXML_NODUMP=1` made it
109    // intentional. Not an `Error:`/`Fatal:` (those are reserved for the
110    // conversion log and would corrupt canvas error counts) — a plain
111    // one-shot stderr banner the operator cannot miss.
112    if !*NODUMP {
113      crate::dump_paths::warn_degraded_no_dump();
114    }
115    InnerPool!(latex_base);
116  }
117
118  // Format-layering rule: real LaTeX (INITEX-based) never defines plain.tex's
119  // tabbing shorthand `\+` (= `\tabalign`), but our latex format is layered
120  // on the plain layer (dump record or raw plain.tex), which does. A stray
121  // `\+` in a LaTeX document (author typo, e.g. `\!+\+` for `\!+\!`) then
122  // expanded into \tabalign's \halign and detonated the mode-mismatch runaway
123  // (102-error TooManyErrors fatal; witness cond-mat0001412 in the 2026-07
124  // full-arXiv run) — where pdflatex gives a single undefined-CS error
125  // (latex209.def compat: `\let\+\@empty`) and Perl LaTeXML, which never
126  // defines `\+` in its TeX.pool, likewise reports one undefined macro.
127  // Retract the inherited definition here, at the "latex kernel layer
128  // complete" seam (covers both the dump and raw-base branches; skipped in
129  // INI_MODE by the early return above, so dump generation is unaffected).
130  // Guarded on the body still being plain's bare `\tabalign`, so a user's
131  // own pre-\documentclass `\def\+` survives the lazy pool load.
132  if let Ok(Some(defn)) = lookup_definition(&T_CS!("\\+"))
133    && let Some(ExpansionBody::Tokens(body)) = defn.get_expansion()
134    && body.len() == 1
135    && body
136      .unlist_ref()
137      .first()
138      .is_some_and(|t| t.text == pin!("\\tabalign"))
139  {
140    remove_meaning_global(&T_CS!("\\+"));
141  }
142
143  InnerPool!(latex_constructs);
144
145  // Rust-only overrides — loaded LAST so they can patch CSes set up by
146  // any earlier pool (dump, bootstrap, base, constructs). Without this
147  // call, definitions in `latex_constructs_rust_only.rs` (e.g.
148  // `\UseRawInputEncoding`, `\ltx@hard@MessageBreak`) never register.
149  InnerPool!(latex_constructs_rust_only);
150
151  // Retry any PA/MPA let-aliases whose target wasn't defined at
152  // dump-load time (they were queued rather than silently dropped).
153  // Classic example: `\let\a=\@tabacckludge` — `\@tabacckludge`
154  // itself is defined in latex_constructs (which loads after the
155  // dump), so the alias has to wait until now.
156  let (applied, skipped) = dump_reader::flush_deferred_aliases();
157  if applied + skipped > 0 {
158    Info!(
159      "latex_dump",
160      "deferred",
161      s!("deferred aliases: {} applied, {} skipped", applied, skipped)
162    );
163  }
164});
