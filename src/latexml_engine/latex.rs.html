15// Process-once cached env vars (see WISDOM #56 — getenv hot-path race).
16static DUMP_PATH: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_PATH").ok());
17static DUMP_DIR: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_DIR").ok());
18static INI_MODE: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_INI_MODE").is_some());
19static NODUMP: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_NODUMP").is_some());
20
21const DEV_DUMPS_DIR: &str = concat!(env!("CARGO_MANIFEST_DIR"), "/../resources/dumps");
22
23/// Perl `FindFile($format._dump, ...)` parity for the latex dump.
24/// Returns true when any `latex.YYYY.dump.txt` is reachable through env
25/// overrides, the exe-relative installed layout, the dev-tree path, or
26/// the embedded fallback. Used by `LoadFormat('latex')` to decide
27/// between the dump branch and the base branch.
28fn latex_dump_available() -> bool {
29  if *NODUMP {
30    return false;
31  }
32  if let Some(p) = DUMP_PATH.as_deref() {
33    if Path::new(p).is_file() {
34      return true;
35    }
36  }
37  let prefer = crate::dump_paths::detect_ambient_texlive_year();
38  if let Some(dir) = DUMP_DIR.as_deref() {
39    if crate::dump_paths::resolve_versioned_in_dir(Path::new(dir), "latex", prefer).is_some() {
40      return true;
41    }
42  }
43  if let Ok(exe) = std::env::current_exe() {
44    if let Some(exe_dir) = exe.parent() {
45      let installed = exe_dir.join("../resources/dumps");
46      if crate::dump_paths::resolve_versioned_in_dir(&installed, "latex", prefer).is_some() {
47        return true;
48      }
49      if crate::dump_paths::resolve_versioned_in_dir(exe_dir, "latex", prefer).is_some() {
50        return true;
51      }
52    }
53  }
54  let dev = Path::new(DEV_DUMPS_DIR);
55  if dev.is_dir()
56    && crate::dump_paths::resolve_versioned_in_dir(dev, "latex", prefer).is_some()
57  {
58    return true;
59  }
60  crate::embedded_dumps::embedded_latex_dump(prefer).is_some()
61}

63LoadDefinitions!({
64  //**********************************************************************
65  // Organized following
66  //  "LaTeX: A Document Preparation System"
67  //   by Leslie Lamport
68  //   2nd edition
69  // Addison Wesley, 1994
70  // Appendix C. Reference Manual
71  //**********************************************************************
72  // NOTE: This will be loaded after TeX.pool.ltxml, so it inherits.
73  //**********************************************************************
74
75  // Perl: LaTeX.pool.ltxml — LoadPool('TeX'); LoadFormat('latex');
76  LoadPool!("TeX");
77
78  InnerPool!(latex_bootstrap);
79
80  // In `--init=latex.ltx` (dump-build) mode, stop after latex_bootstrap.
81  // The same reasoning as in `tex.rs`'s plain branch: pre-loading
82  // latex_dump / latex_base / latex_constructs pollutes the snapshot
83  // and silences the diff for everything raw latex.ltx defines.
84  // `LATEXML_INI_MODE=1` is set by `bin/latexml_oxide.rs` BEFORE
85  // `prepare_session`, so this branch fires before latex.rs runs.
86  if *INI_MODE {
87    return Ok(());
88  }
89
90  // Perl `LoadFormat('latex')` strict split:
91  //   if dump available: bootstrap → dump → constructs (NO base)
92  //   else:              bootstrap → base → constructs (NO dump)
93  if !*NODUMP && latex_dump_available() {
94    if let Err(e) = crate::latex_dump::load_definitions() {
95      Warn!("latex_dump", "load", s!("{}", e));
96    }
97  } else {
98    InnerPool!(latex_base);
99  }
100
101  InnerPool!(latex_constructs);
102
103  // Rust-only overrides — loaded LAST so they can patch CSes set up by
104  // any earlier pool (dump, bootstrap, base, constructs). Without this
105  // call, definitions in `latex_constructs_rust_only.rs` (e.g.
106  // `\UseRawInputEncoding`, `\ltx@hard@MessageBreak`) never register.
107  InnerPool!(latex_constructs_rust_only);
108
109  // Retry any PA/MPA let-aliases whose target wasn't defined at
110  // dump-load time (they were queued rather than silently dropped).
111  // Classic example: `\let\a=\@tabacckludge` — `\@tabacckludge`
112  // itself is defined in latex_constructs (which loads after the
113  // dump), so the alias has to wait until now.
114  let (applied, skipped) = latexml_core::dump_reader::flush_deferred_aliases();
115  if applied + skipped > 0 {
116    Info!(
117      "latex_dump", "deferred",
118      s!("deferred aliases: {} applied, {} skipped", applied, skipped)
119    );
120  }
121});
