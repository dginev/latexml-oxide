11/// DEP-17d helper for the dominant 3-arg `DefMath!("\\cs", None, "char"…)`
12/// shape (195+ entries here). `paramlist = None` matches the macro's
13/// `(text, paramlist, present)` arm exactly — distinct from
14/// `def_math_sym`'s 2-arg form which goes through `parse_prototype` and
15/// produces `Some(empty Parameters)`. Both yield the same `nargs = 0`
16/// downstream, but keeping the None form preserves struct-level parity
17/// for any consumer that distinguishes.
18fn def_math_atom(cs: &str, present: &str, role: Option<&str>, meaning: Option<&str>) -> Result<()> {
19  let cs_tok = T_CS!(cs);
20  let mut opts = MathPrimitiveOptions::default();
21  if let Some(r) = role {
22    opts.role = Some(r.to_string());
23  }
24  if let Some(m) = meaning {
25    opts.meaning = Some(m.to_string());
26  }
27  def_math(cs_tok, None, present.to_string(), opts)?;
28  Ok(())
29}

114/// Set role and augment delimiter properties on the last child element.
115/// Mirrors Perl's augmentDelimiterProperties (TeX_Math.pool.ltxml).
116/// Perl: augmentDelimiterProperties($doc, $whatsit, $role, $stretchy)
117/// Look up delimiter character in DELIM_CHAR_MAP and set name/meaning/role.
118/// When role is empty, don't change the role (Perl: $role=undef).
119fn augment_delimiter_properties(document: &mut Document, role: &str) -> Result<()> {
120  let current = document.get_node().clone();
121  let delim_opt = current
122    .get_child_nodes()
123    .into_iter()
124    .filter(|n| n.get_type() == Some(NodeType::ElementNode))
125    .last();
126  if let Some(mut delim) = delim_opt {
127    let char_content = delim.get_content();
128    let first_char = char_content.chars().next();
129    // Look up delimiter properties by char content
130    if let Some(entry) = first_char.and_then(|c| DELIM_CHAR_MAP.get(&c)) {
131      // Role: only change if explicitly requested (non-empty role)
132      if !role.is_empty() {
133        let new_role = if role == "OPEN" {
134          entry.left_role
135        } else {
136          entry.right_role
137        };
138        let current_role = delim.get_attribute("role");
139        match current_role.as_deref() {
140          None | Some("OPEN") | Some("MIDDLE") | Some("CLOSE") | Some("VERTBAR") => {
141            document.set_attribute(&mut delim, "role", new_role)?;
142          },
143          _ => {},
144        }
145      }
146      // Set name if entry has one
147      if let Some(name) = entry.name {
148        document.set_attribute(&mut delim, "name", name)?;
149      }
150      // Handle meaning
151      if entry.remove_meaning {
152        let _ = delim.remove_attribute("meaning");
153      }
154      // Handle char replacement (e.g. < → ⟨)
155      if let Some(replacement) = entry.replace_char
156        && let Some(mut first_child) = delim.get_first_child()
157      {
158        let _ = first_child.set_content(&replacement.to_string());
159      }
160    } else if !role.is_empty() {
161      // No map entry — just set role if explicitly requested
162      let current_role = delim.get_attribute("role");
163      match current_role.as_deref() {
164        None | Some("OPEN") | Some("MIDDLE") | Some("CLOSE") | Some("VERTBAR") => {
165          document.set_attribute(&mut delim, "role", role)?;
166        },
167        _ => {},
168      }
169    }
170  }
171  Ok(())
172}

227  // \accent <number> <optional assignments> <character>; See TeX Book p.287
228  // Reads a number (font position), then optional assignments, then a character.
229  // Decodes the font position to a glyph, looks up accent data, applies accent.
230  DefPrimitive!("\\accent Number", sub[(num)] {
231    use crate::tex_character;
232    // 1. Decode the accent glyph from font position (BEFORE processing assignments)
233    let n = num.value_of() as i32;
234    let (glyph_opt, _font) = font_decode(n, None, None);

236    // 2. Process optional assignments (Perl lines 117-123)
237    //    <assignments>: (<prefix>) simple assignment or macro assignment
238    //    <character> : letter, other, \char, \chardef token, \noboundary
239    let mut assignments: Vec<Digested> = Vec::new();
240    let mut last_token: Option<Token> = None;
241    let mut last_defn: Option<Rc<dyn Definition>> = None;
242    loop {
243      let token_opt = read_x_non_space()?;
244      let token = match token_opt {
245        Some(t) => t,
246        None => { break; }
247      };
248      let defn = if token.get_catcode().is_active_or_cs() {
249        lookup_definition(&token)?.map(|d| d as Rc<dyn Definition>)
250      } else {
251        None
252      };
253      // Perl: isPrefix || isFontDef || (isRegister && !isCharDef)
254      //       || token matches \def|\edef|\gdef|\xdef
255      let is_assignment = if let Some(ref d) = defn {
256        if d.is_prefix() || (
257          d.is_register() && !matches!(d.register_type(), Some(RegisterType::CharDef))) {
258          true
259        } else {
260          // Check isFontDef: lookupValue("fontinfo_<cs>")
261          let cs_str = token.to_string();
262          let fontinfo_key = s!("fontinfo_{}", cs_str);
263          if with_value(&fontinfo_key, |v| v.is_some()) {
264            true
265          } else {
266            // Check \def, \edef, \gdef, \xdef
267            matches!(cs_str.as_str(), "\\def" | "\\edef" | "\\gdef" | "\\xdef")
268          }
269        }
270      } else {
271        false
272      };
273      if !is_assignment {
274        last_token = Some(token);
275        last_defn = defn;
276        break;
277      }
278      // Process the assignment: invoke the token
279      let digested = invoke_token(&token)?;
280      assignments.extend(digested);
281    }
282
283    // 3. Read the base character token (Perl lines 126-134)
284    let letter = if let Some(t) = last_token {
285      let cc = t.get_catcode();

287        || last_defn.as_ref().is_some_and(|d|

291        Tokens!(t, ExplodeText!(&read_number()?.to_string()))

293        Tokens!() // Treat as empty
294      } else {
295        unread_one(t);
296        Tokens!()
297      }
298    } else {
299      Tokens!()
300    };
301
302    // 4. Enter horizontal mode
303    enter_horizontal();
304
305    // 5. Apply accent (Perl lines 137-141)
306    let accent_result: Vec<Digested> = if let Some(glyph_char) = glyph_opt {
307      let glyph_str = glyph_char.to_string();
308      if let Some(entry) = tex_character::unicode_accent(&glyph_str) {

315          letter, entry.combiner, entry.standalone, Some(reversion))?;

317      } else {
