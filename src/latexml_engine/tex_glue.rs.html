5static UNICODE_EM_SPACES: [(f64, char); 7] = [
6  // Spaces to fake spacing, with width in ems
7  (0.100, '\u{200A}'), // Hair space (thinner than thin space)
8  (0.167, '\u{2006}'), // six-per-em
9  (0.200, '\u{2009}'), // five-per-em, thin space
10  (0.250, '\u{2005}'), // four-per-em, mid space
11  (0.333, '\u{2004}'), // three-per-em, thick space
12  (0.500, '\u{2002}'), // en-quad, "nut"
13  (1.000, '\u{2003}'), // em-quad, "mutton"
14];
15
16// Spacing widths with their canonical TeX command (Perl @spaces in
17// TeX_Glue.pool.ltxml L31-40). Used by `revert_skip` to recover the
18// original macro name from a digested `\hskip` length, so reversions
19// like `\quad` are preserved instead of decaying to `\hskip 10.0pt`.
20static REVERT_SKIPS: [(f64, &str); 6] = [
21  (0.200, "\\thinspace"), // five-per-em
22  (0.250, "\\>"),         // four-per-em
23  (0.333, "\\;"),         // three-per-em
24  (0.500, "\\enskip"),    // en-quad
25  (1.000, "\\quad"),      // em-quad
26  (2.000, "\\qquad"),     // double em
27];
28
29/// String of spacing chars with width roughly equivalent to $dimen
30pub(crate) fn dimension_to_spaces(dimen: Dimension) -> String {
31  let fs = lookup_font().unwrap().get_size().unwrap_or(1.0); // 1 em
32  let mut ems = dimen.pt_value(None) / fs;
33  let mut s = String::default();
34  for (w, space_char) in UNICODE_EM_SPACES.into_iter().rev() {
35    if ems <= 0.0 {
36      break;
37    }
38    if ems + 0.01 > w {
39      let n = ((ems + 0.01) / w).floor() as usize;
40      ems -= n as f64 * w;
41      for _ in 0..n {
42        s.push(space_char);
43      }
44    }
45  }
46  s
47}

49/// Revert a skip length to its canonical command (Perl
50/// `revertSkip` in TeX_Glue.pool.ltxml L57-65). If `dimen` matches one
51/// of the well-known em-multiple widths to within 0.01em, returns the
52/// corresponding `\quad`/`\qquad`/`\enskip`/etc CS token. Otherwise
53/// returns `<command> <dimen>` (e.g. `\hskip 10.0pt`).
54pub(crate) fn revert_skip(command: Token, dimen: Dimension) -> Tokens {
55  let fs = lookup_font().and_then(|f| f.get_size()).unwrap_or(10.0);
56  let ems = dimen.pt_value(None) / fs;
57  for (w, cs) in REVERT_SKIPS.iter() {
58    if ems > w + 0.01 {
59      continue;
60    }
61    if ems < w + 0.01 && ems > w - 0.01 {
62      return Tokens!(T_CS!(*cs));
63    }
64  }
65  let dim_str = dimen.to_string();
66  let mut out: Vec<Token> = vec![command];
67  for ch in dim_str.chars() {
68    out.push(T_OTHER!(ch.to_string()));
69  }
70  Tokens::new(out)
71}

86  // \hskip handled similarly to \kern
87  // \hskip can be ignored in certain situations...
88  DefConstructor!("\\hskip Glue", sub[document, args, props] {
89    unref!(args => length_digested);
90    let length = match  length_digested.data() {
91      DigestedData::RegisterValue(v) => v.into(),
92      _ => Dimension::default()
93    };
94    let parent = document.get_node();
95    if document::with_node_qname(parent, |name| name == "svg:g") {
96      // Perl: TeX_Glue.pool.ltxml L71-76
97      // SVG translate handling — append translate to parent's transform attribute
98      let x = length.px_value(None);
99      if x != 0.0 {
100        let parent = document.get_node_mut();
101        let transform = parent.get_attribute("transform").unwrap_or_default();
102        let new_transform = if transform.is_empty() {
103          s!("translate({x},0)")
104        } else {
105          s!("{transform} translate({x},0)")
106        };
107        parent.set_attribute("transform", &new_transform)?;
108      }
109    } else if in_svg(document) {
110      Warn!("unexpected", "kern", s!("Lost hskip in SVG {length}"));
111    } else if props.get("isMath") == Some(&Stored::Bool(true)) {
112      // Perl: TeX_Glue.pool.ltxml L80 — in math, emit XMHint so the math
113      // parser can absorb the spacing into the adjacent token's
114      // lpadding/rpadding (or promote it to a virtual PUNCT for big skips).
115      // Without this branch, `\qquad` after `,` lost its `rpadding="20.0pt"`.
116      document.insert_element("ltx:XMHint", Vec::new(), Some(map!("width" => length.to_attribute())))?;
117    } else {
118      let spaces = dimension_to_spaces(length);
119      document.absorb_string(&spaces, &SymHashMap::default())?;
120    }
121  },
122  // Perl: enterHorizontal => 1
123  enter_horizontal => true,
124  // Perl `reversion => sub { revertSkip(T_CS('\hskip'), $_[1]); }`
125  // (TeX_Glue.pool.ltxml L78). Recovers `\quad`/`\qquad`/`\enskip`/etc
126  // when the skip width matches an em-multiple within 0.01em tolerance.
127  reversion => sub[_whatsit, args] {
128    let length = match args.first().and_then(|o| o.as_ref()).map(|d| d.data()) {
129      Some(DigestedData::RegisterValue(v)) => v.into(),
130      _ => Dimension::default(),
131    };
132    Ok(revert_skip(T_CS!("\\hskip"), length))
133  },
134  properties => sub[args] {
135    unref!(args => length_digested);
136    let width: Dimension = match length_digested.data() {
137      DigestedData::RegisterValue(v) => v.into(),
138      _ => Dimension::default()
139    };
140    Ok(stored_map!(
141      "width" => width, "isSpace" => true, "isSkip" => true))
142  });
143
144  //======================================================================
145  // If this is the right solution...
146  // then we also should put the desired spacing on a style attribute?!?!?!
147  DefConstructor!("\\vskip Glue", sub[document, args, _props] {
148    unref!(args => length);
149    let pt = length.pt_value(None);
150    if pt <= 0.0 {
151      // Negative or zero skip: do nothing
152    } else if pt < 4.0 && document.is_closeable("ltx:p").is_some() {
153      document.close_element("ltx:p")?;
154    } else if document.is_closeable("ltx:para").is_some() {
155      document.close_element("ltx:para")?;
156    }},
157    // Perl: leaveHorizontal => 1
158    before_digest => { leave_horizontal()?; },
159    // Perl: height => $_[1] — stores glue value as height property
160    // so getSize() returns it, making \noalign{\vskip X} contribute to row spacing
161    properties => sub[args] {
162      // The Glue argument arrives as a RegisterValue wrapping the glue.
163      // Extract its base dimension for the height property.
164      let height: Stored = match args[0].as_ref() {
165        Some(d) => {
166          if let Some(dim) = d.get_dimension() {
167            Stored::Dimension(dim)
168          } else {
169            // Fallback: try converting via Stored
170            Stored::from(d.clone())
171          }
172        },
173        None => Stored::None,
174      };
175      Ok(stored_map!("isSpace" => true, "isSkip" => true,
176        "isVerticalSpace" => true, "isBreak" => true,
177        "height" => height))
178    }
179  );
180  // Remove skip, if last on LIST
181  DefPrimitive!("\\unskip", {
182    let mut comments = Vec::new();
183    while let Some(last_box) = pop_box_list() {
184      // Scan past any Comment boxes
185      if matches!(last_box.data(), DigestedData::Comment(_)) {
186        comments.push(last_box);
187      } else if last_box.get_property_bool("isSkip") {
188        break;
189      } else {
190        // return a non-skip box to the list.
191        push_box_list(last_box);
192        break;
193      }
194    }
195    // Restore any comment boxes that were scanned past
196    for comment in comments.into_iter().rev() {
197      push_box_list(comment);
198    }
199  });

203  //----------------------------------------------------------------------
204  // \hfil             d  inserts first order infinitely stretchable horizontal glue in a horizontal

216      None,
217      None,

227      None,
228      Tokens!(T_CS!("\\hfill")),
229      stored_map!("isSpace" => true, "isFill" => true),

231  });

237  // list. \vfill            d  inserts second order infinitely stretchable vertical glue in a
238  // vertical list. \vfilneg          d  cancels the stretchability of \vfil.
239  // \vss              d  insert infinitely stretchable and shrinkable vertical glue in a vertical
240  // list.
241
242  // Perl: all have leaveHorizontal => 1
243  DefPrimitive!("\\vfil", None, leave_horizontal => true);
244  DefPrimitive!("\\vfill", None, leave_horizontal => true);
245  DefPrimitive!("\\vss", None, leave_horizontal => true);
246  DefPrimitive!("\\vfilneg", None, leave_horizontal => true);

250  //----------------------------------------------------------------------

253  DefRegister!("\\lastskip", Dimension::new(0), readonly => true, getter => {
254    stomach::with_box_list(|stomach_box_list| {
255      let box_iter = stomach_box_list.iter().rev();
256      for box_in_list in box_iter {
257        if !matches!(box_in_list.data(), DigestedData::Comment(_)) {
258          if box_in_list.get_property_bool("isSkip") {
259            let Some(width_stored) = box_in_list.get_property("width") else {
260              break;
261            };
262            if let Stored::Dimension(ref width_d) = *width_stored {
263              return *width_d;
264            } else {
265              log::warn!("Unexpected type of \"width\" value in State: {width_stored:?}");
266              break;
267            }
268          } else {
269            break;
270          }
271        }
272      }
273      Dimension::new(0)
274    })
