5use crate::prelude::*;
6use chrono::prelude::*;
7use once_cell::sync::Lazy;

9// Process-once cached env var (see WISDOM #56 — getenv hot-path race).
10static SOURCE_DATE_EPOCH: Lazy<Option<String>> =
11  Lazy::new(|| std::env::var("SOURCE_DATE_EPOCH").ok());
12
13LoadDefinitions!({
14  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
15  // Job Family of primitive control sequences
16  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
17
18  //======================================================================
19  // The current Job
20  //----------------------------------------------------------------------
21  // \jobname          c  is the underlying file name for a job.
22  // \time             pi holds the current time in minutes after midnight (0-1439).
23  // \day              pi holds the current day of the month (1-31).
24  // \month            pi holds the current month of the year (1-12).
25  // \year             pi holds the current year (e.g., 2000).
26  // \mag              pi holds the magnification ratio times 1000.
27  DefMacro!(T_CS!("\\jobname"), None, Tokens!()); // Set to the filename by initialization
28  DefRegister!("\\time", Number!(0));
29  DefRegister!("\\day", Number!(0));
30  DefRegister!("\\month", Number!(0));
31  DefRegister!("\\year", Number!(0));
32  DefRegister!("\\mag", Number!(1000));
33
34  // TODO: This may mess up Daemon state? Reinit when setting jobname?
35  // Respect SOURCE_DATE_EPOCH env var for reproducible builds (like Perl)
36  let dt: DateTime<Local> = if let Some(epoch_str) = SOURCE_DATE_EPOCH.as_deref() {
37    if let Ok(epoch) = epoch_str.trim().parse::<i64>() {
38      DateTime::from_timestamp(epoch, 0)
39        .map(|utc| utc.with_timezone(&Local))
40        .unwrap_or_else(Local::now)
41    } else {
42      Local::now()
43    }
44  } else {
45    Local::now()
46  };
47  AssignValue!("\\day", Number!(dt.day()), Scope::Global);
48  AssignValue!("\\month", Number!(dt.month()), Scope::Global);
49  AssignValue!("\\year", Number!(dt.year()), Scope::Global);
50  AssignValue!(
51    "\\time",
52    Number!(60 * dt.hour() + dt.minute()),
53    Scope::Global
54  );
55
56  //======================================================================
57  // Random Job related things
58  //----------------------------------------------------------------------
59  // \end              c  terminates the current job.
60  // \everyjob         pt holds tokens which are inserted at the start of every job.
61  // \deadcycles       iq is the number of times \output was called since the last \shipout.
62  // \maxdeadcycles    pi is the maximum allowed value of \deadcycles before an error is generated.
63  // Perl: $stomach->leaveHorizontal; $stomach->getGullet->flush;
64  DefPrimitive!("\\lx@end@document", {
65    // When called during package/definition loading (e.g., expl3's error handler
66    // calls \tex_end:D via \msg_fatal), ignore it. Package errors should not
67    // terminate the entire document processing.
68    if !state::lookup_bool_sym(pin!("INTERPRETING_DEFINITIONS")) {
69      leave_horizontal()?;
70      gullet::flush();
71    }
72    // else: silently ignore during definition loading
73  });
74  Let!("\\end", "\\lx@end@document");
75  // Save the TeX primitive \end under \@@end so that expl3's primitive rename
76  // (\__kernel_primitive:NN \end \tex_end:D) gets the real primitive, not
77  // LaTeX's \end{environment} handler which consumes {} arguments.
78  // In Perl, @@end is the saved TeX \end (latex.ltx saves it).
79  Let!("\\@@end", "\\lx@end@document");
80
81  DefRegister!("\\everyjob", Tokens!());
82  DefRegister!("\\deadcycles", Number!(0));
83  DefRegister!("\\maxdeadcycles", Number!(0));
84
85  //======================================================================
86  // Dumping
87  //----------------------------------------------------------------------
88  // \dump             c  outputs a format file in INITEX; otherwise it is equivalent to \end.
89
90  DefMacro!("\\dump", {
91    Warn!("unexpected", "dump", "Do not know how to \\dump yet, sorry");
92  });
93
94  // TODO: load_dump
95  // TODO: load_latex
96
97  //======================================================================
98  // LaTeX 2.09 compatibility
99  //----------------------------------------------------------------------
100  // Perl latex_constructs.pool.ltxml:97-129 (`\documentstyle` afterDigest).
101  // LaTeX 2.09 compat shim. Three branches mirroring Perl strictly:
102  //
103  //   1. <class>.sty exists  → input_definitions("article", cls, handleoptions=true, options=opts)
104  //                          + require_package(class, as_class=true, after=\compat@loadpackages)
105  //   2. <class>.cls exists  → load_class(class, opts, after=\compat@loadpackages)
106  //   3. neither             → input_definitions("OmniBus", cls, handleoptions=true, options=opts,
107  //      after=\compat@loadpackages)
108  //                          + require_package(class, as_class=true)
109  //
110  // Critical Perl semantics (matching latex_constructs.pool.ltxml:97-129):
111  //   * `handleoptions => 1` makes the cls's `\DeclareOption`/`\ProcessOptions` consume the
112  //     `<opts>` and route leftovers onto `@unusedoptionlist`.
113  //   * `after => \compat@loadpackages` runs *after* the cls finishes its option-processing — at
114  //     that point unmet options sit on the unused list and `\compat@loadpackages`
115  //     (`latex_constructs.rs:2502`) walks them, RequirePackage's any that resolve, and triggers
116  //     OmniBus when anything went unmet. That is what lets `\documentstyle[paspconf] {article}`
117  //     transitively load `aas_macros.sty.ltxml` to define `\affil` / `\altaffilmark` /
118  //     `\acknowledgments` etc.
119  // The latex_dump unconditionally redefines `\documentstyle` with the
120  // kernel-style `\input{latex209.def}\documentclass` form (Perl
121  // latex_dump.pool.ltxml entry for `\documentstyle`). In Perl that
122  // version is itself overridden by latex_constructs.pool.ltxml's
123  // DefConstructor; in our Rust port `latex_constructs.rs` doesn't
124  // redefine `\documentstyle`, so without intervention any autoload-of-
125  // LaTeX-pool path (e.g. `\newcommand` before `\documentstyle`)
126  // replaces our impl with the dump's, breaking the
127  // `\compat@loadpackages` after-hook that dispatches `[epsf]` etc.
128  // from `@unusedoptionlist`. Witness: hep-th9912229
129  // (`\newcommand` before `\documentstyle[12pt,epsf]`).
130  //
131  // Workaround: register the impl under a stable backup name
132  // `\lx@documentstyle@impl` and `\let \documentstyle = \lx@documentstyle@impl`
133  // at the end of `\@load@latex@pool` so we restore our impl after
134  // every LaTeX pool load.
135  DefMacro!("\\lx@documentstyle@impl[]{}", sub[(options_opt, class_tks)] {
136    use latexml_core::binding::content::{find_file, find_file_fallback_exists, FindFileOptions, load_class};
137    // Perl latex_constructs.pool.ltxml:79 — `$class =~ s/\s+//g;`. Strip ALL
138    // whitespace (not just leading/trailing as a bare `.trim()` would), so a
139    // multi-line `\documentstyle[...]{<newline>article}` — whose post-`{`
140    // newline becomes an embedded space token in the Semiverbatim arg — still
141    // matches its binding registration. Mirrors the sibling `\documentclass`
142    // fix (latex_constructs.rs); a whitespace-bearing name otherwise misses
143    // the binding and falls through to OmniBus.
144    let class: String = class_tks
145      .to_string()
146      .chars()
147      .filter(|c| !c.is_ascii_whitespace())
148      .collect();
149
150    let pool = if class == "amsppt" { "AmSTeX" } else { "LaTeX" };
151    input_definitions(pool, InputDefinitionOptions {
152      extension: Some(Cow::Borrowed("pool")),
153      ..InputDefinitionOptions::default()
154    })?;
155
156    state::assign_value("2.09_COMPATIBILITY", true, Some(Scope::Global));
157
158    // Perl TeX.pool.ltxml:60-65 — when the class triggers the AmSTeX pool
159    // (only `amsppt` today), Perl LoadPool's AmSTeX and *re-emits*
160    // `\documentstyle{class}` so the AmSTeX-pool-defined `\documentstyle`
161    // (amstex.rs L40) takes over. Critically, Perl never loads the LaTeX
162    // pool (`latex_constructs.pool.ltxml`) on the amsppt path, so its
163    // `Let('\magnification', '\@undefined')` (L56) never fires and plain
164    // TeX's `\magnification = \magstep N` keeps working. The Rust
165    // implementation previously fell through to the article.cls +
166    // RequirePackage chain unconditionally, which loads latex.ltx and
167    // re-binds `\magnification` to `\@undefined` — an amsppt-only
168    // regression. Mirror Perl: for amsppt, just RequirePackage(amsppt,
169    // as_class=true) and return. amstex.rs's `\documentstyle` does the
170    // same thing in the same way.
171    if pool == "AmSTeX" {
172      let _ = require_package(&class, RequireOptions {
173        notex: Some(true),
174        as_class: true,
175        ..RequireOptions::default()
176      });
177      return Ok(Tokens!());
178    }
179
180    // Perl L132-135 `compatDefinitions` — pre-bind LaTeX 2.09 dimensions.
181    // Perl helper redefines `\@maxsep` and `\@dblmaxsep`; if these come
182    // from another file and are already defined, redef is harmless.
183    let zero_dim = Stored::Number(latexml_core::common::number::Number::new(0));
184    state::assign_value("\\@maxsep", zero_dim.clone(), Some(Scope::Global));
185    state::assign_value("\\@dblmaxsep", zero_dim, Some(Scope::Global));
186
187    // Now that LaTeX pool is loaded, undefine `\magnification` so babel
188    // detects "running under LaTeX" correctly. The latex_constructs
189    // pool-load no longer does this unconditionally; deferring to
190    // \documentclass/\documentstyle keeps plain TeX papers (whose first
191    // line is often `\magnification=\magstep1`) functional under the
192    // cortex_worker's eager LaTeX.pool preload. See sibling change in
193    // latex_constructs.rs `\documentclass` after_digest.
194    if let Some(undef_meaning) = state::lookup_meaning(&T_CS!("\\@undefined")) {
195      state::assign_meaning(&T_CS!("\\magnification"), undef_meaning, Some(Scope::Global));
196    }
197
198    // Comma-list to Vec<String>. Whitespace-strip per Perl
199    // TrimmedCommaList. Empty entries dropped.
200    let opts_vec: Vec<String> = options_opt.as_ref()
201      .map(|t| t.to_string())
202      .unwrap_or_default()
203      .split(',')
204      .map(|s| s.trim().to_string())
205      .filter(|s| !s.is_empty())
206      .collect();
207
208    // Perl uses `notex => !LookupValue('INCLUDE_CLASSES')` which defaults to
209    // `notex = true` — FindFile consults the @ltxml_paths binding registry
210    // as well as the filesystem.
211    let notex = !state::lookup_bool("INCLUDE_CLASSES");
212    // Probe `.sty` then `.cls`, AND fall back to version-stripping
213    // (`find_file_fallback`) so e.g. `\documentstyle{aipproc2}` resolves
214    // to aipproc.sty.ltxml — matching Perl's `FindFile` which consults
215    // the binding-name registry plus the `\d+`-suffix fallback. Without
216    // this, versioned class names go to OmniBus → aipproc.sty.ltxml is
217    // never loaded → `\epsfsize` etc. that aipproc.sty's
218    // `RequirePackage('psfig'→'epsfig')` chain provides stay undefined.
219    // Check `<class>.sty` via binding registry (notex), version-strip
220    // fallback, AND disk-only probe (`forbid_ltxml: true`) so paper-local
221    // `<class>.sty` files (e.g. `jinstpub.sty`, with no .cls counterpart)
222    // can be loaded as a class. Disk probe is gated on the absence of
223    // ANY `<class>.cls` binding (exact OR via version-strip fallback) so
224    // that `\documentstyle{mn1}` falls back to `mn.cls.ltxml` (Perl-faithful
225    // priority) instead of the local `mn1.sty` whose raw load is suppressed
226    // by the default `INCLUDE_STYLES=false` gate. Witness: astro-ph0002213
227    // (\documentstyle[epsfig]{mn1} + local mn1.sty + \begin{keywords}/\psfig).
228    let class_sty_binding = find_file(
229      &format!("{}.sty", class),
230      Some(FindFileOptions { notex, ..Default::default() }),
231    ).is_some();
232    // Pre-flight existence checks must NOT eagerly load the fallback
233    // binding (its body would run \LoadClass / \RequirePackage with no
234    // option-pass-through, contaminating the actual `\documentstyle`
235    // load that happens further below). Fix for the `\psfig` cluster
236    // (12 papers) — astro-ph0002213 R=1→0 with this gate change.
237    let class_sty_fallback = find_file_fallback_exists(&class, "sty");
238    let class_cls_binding_exact = find_file(
239      &format!("{}.cls", class),
240      Some(FindFileOptions { notex, ..Default::default() }),
241    ).is_some();
242    let class_cls_via_fallback = find_file_fallback_exists(&class, "cls");
243    let class_sty_via_disk = !class_sty_binding
244      && !class_sty_fallback
245      && !class_cls_binding_exact
246      && !class_cls_via_fallback
247      && find_file(
248        &class,
249        Some(FindFileOptions { ext_type: Some(Cow::Borrowed("sty")), forbid_ltxml: true, ..Default::default() }),
250      ).is_some();
251    let class_sty_found = class_sty_binding || class_sty_via_disk || class_sty_fallback;
252    let class_cls_found = !class_sty_found && (class_cls_binding_exact
253      || class_cls_via_fallback);
254
255    let after = Tokens!(T_CS!("\\compat@loadpackages"));
256
257    if class_sty_found {
258      // Branch 1 — class is actually a `.sty` (e.g. spackap, aipproc,
259      // kluwer): load article.cls under it, then RequirePackage(class).
260      // Pin extension to "sty" so the version-stripping fallback (e.g.
261      // aipproc2 → aipproc) prefers the binding registry's `.sty` entry
262      // instead of also probing for a same-name `.cls` (which has a
263      // different DefMacro signature for `\author` etc. that conflicts
264      // with the `.sty` body — root cause for nucl-th0010030).
265      // Choose article.cls vs OmniBus.cls as the underlying class. When
266      // the `.sty` was found via binding registry or version-strip
267      // fallback, the user's intent is "load this binding as the
268      // document class" — article.cls is the right base. When the
269      // `.sty` was found ONLY via paper-local disk-probe (no binding /
270      // no fallback), the file is an arbitrary user style that doesn't
271      // know about LaTeXML's frontmatter conventions; OmniBus is the
272      // right base because it provides broad fallback coverage
273      // (`\citeauthoryear` autoload trigger, generic `\affil`, etc.).
274      // Mirrors Perl's flow for paper-local-only `\documentstyle`
275      // classes (verified via `--verbose` on astro-ph0510540 and
276      // astro-ph0008100 — Perl loads OmniBus for both).
277      let underlying = if class_sty_via_disk { "OmniBus" } else { "article" };
278      input_definitions(underlying, InputDefinitionOptions {
279        extension: Some(Cow::Borrowed("cls")),
280        options: opts_vec.clone(),
281        handleoptions: true,
282        noerror: true,
283        ..InputDefinitionOptions::default()
284      })?;
285      // When the .sty was found ONLY via paper-local disk-probe (no
286      // binding, no fallback), pass `notex=false` to allow the raw
287      // load. Otherwise the default `INCLUDE_STYLES=false` gate inside
288      // require_package would force `notex=true` and suppress the
289      // load. Mirrors the `\compat@loadpackages` fix from commit
290      // bb4cf2e17. Witness: astro-ph0008100
291      // (\documentstyle[PASJadd]{PASJ95} + uppercase-named PASJ95.STY).
292      let notex_for_require = if class_sty_via_disk { Some(false) } else { None };
293      require_package(&class, RequireOptions {
294        options: opts_vec,
295        extension: Some(Cow::Borrowed("sty")),
296        notex: notex_for_require,
297        after,
298        ..RequireOptions::default()
299      })?;
300    } else if class_cls_found {
301      // Branch 2 — `<class>.cls` exists: load it as the document class.
302      load_class(&class, opts_vec, after)?;
303    } else {
304      // Branch 3 — neither sty nor cls found. Load OmniBus to provide the
305      // wide AAS/elsevier/etc. coverage, then attempt the user-named class
306      // as a package (will likely no-op via missing_file warn).
307      input_definitions("OmniBus", InputDefinitionOptions {
308        extension: Some(Cow::Borrowed("cls")),
309        options: opts_vec.clone(),
310        handleoptions: true,
311        noerror: true,
312        after,
313        ..InputDefinitionOptions::default()
314      })?;
315      require_package(&class, RequireOptions {
316        options: opts_vec,
317        as_class: true,
318        ..RequireOptions::default()
319      })?;
320    }
321
322    Ok(Tokens!())
323  });
324
325  // Initial alias `\documentstyle = \lx@documentstyle@impl`. The latex_dump
326  // may overwrite this on autoload; `\@load@latex@pool` (tex.rs) re-applies
327  // the Let after every pool load so our impl wins regardless.
328  Let!("\\documentstyle", "\\lx@documentstyle@impl");
329});
