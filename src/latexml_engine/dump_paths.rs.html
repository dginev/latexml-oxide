1//! Versioned dump filename conventions and ambient-TeXLive detection.
2//!
3//! All kernel dumps generated by `--init=plain.tex` and `--init=latex.ltx`
4//! are tagged with the TeXLive release year of the ambient install:
5//!
6//! ```text
7//! resources/dumps/plain.2025.dump.txt
8//! resources/dumps/latex.2025.dump.txt
9//! resources/dumps/texlive.2025.version
10//! ```
11//!
12//! At runtime we detect the ambient year (via the `kpsewhich --version` /
13//! `kpsewhich -var-value=SELFAUTOPARENT` banners, with `pdflatex --version`
14//! as fallback — see [`detect_ambient_texlive_year`]) and pick the matching
15//! dump. If no exact match exists — or no TeXLive is installed at all — we
16//! fall back to the most-recent year present.
17//!
18//! Why versioned: `latex.dump.txt` content reflects whatever raw `latex.ltx`
19//! / `expl3-code.tex` shipped with the ambient TeXLive at `--init` time.
20//! Cross-loading a dump from a different TL into a different TL silently
21//! diverges. Tagging by year makes the relationship explicit and lets us
22//! bundle multiple TL years into a single binary (see
23//! [`embedded_dumps`](crate::embedded_dumps)).
24
25use std::{
26  path::{Path, PathBuf},
27  sync::OnceLock,
28};

30/// Detect the ambient TeXLive release year. Returns `None` if no TeXLive
31/// is installed (or detection fails for some other reason).
32///
33/// Strategy (ordered for one-subprocess-per-distro; see the body for the
34/// spawn-cost rationale):
35///   1. `kpsewhich --version` → MiKTeX prints `MiKTeX YY.MM` → `20YY` (TeX Live
36///      has no year here and falls through).
37///   2. `kpsewhich -var-value=SELFAUTOPARENT` → e.g. `/usr/local/texlive/2025`. Last path
38///      component, if it parses as a 4-digit year (2000..=2099), wins.
39///   3. `pdflatex --version` → first line typically contains `(TeX Live YYYY)`.
40pub fn detect_ambient_texlive_year() -> Option<u32> {
41  // Process-global memo. The ambient TeX year is a constant property of the
42  // installed distribution — it cannot change mid-process — but detection is
43  // called from several sites per conversion (plain/latex dump load, ini), and
44  // each site spawns a probe subprocess. On MiKTeX those subprocesses are
45  // ~340ms EACH (~10-25x TeX Live's), so an unmemoized detect cost ~0.7s PER
46  // call site — the dominant Windows/MiKTeX slowdown (the "load" before the
47  // first dump_reader log). Detect once. Thread-safe: the year is
48  // process-level, not per-thread State.
49  static AMBIENT_TEXLIVE_YEAR: OnceLock<Option<u32>> = OnceLock::new();
50  *AMBIENT_TEXLIVE_YEAR.get_or_init(|| {
51    // Probe order is chosen so EACH distribution resolves in a single
52    // subprocess (the spawn is the whole cost — ~340ms on MiKTeX):
53    //   1. `kpsewhich --version` banner. MiKTeX prints `MiKTeX YY.MM` on stdout
54    //      → `20YY`, and this is MiKTeX's ONLY single-spawn year signal
55    //      (SELFAUTOPARENT is empty there, so step 2 can't help it). Resolving
56    //      here saves MiKTeX the extra `pdflatex --version` spawn. TeX Live's
57    //      banner is `kpathsea version 6.4.x` (no year) so it returns None — but
58    //      only costs ~28ms on TeX Live, so the extra probe is effectively free.
59    //   2. `kpsewhich -var-value=SELFAUTOPARENT` → `.../texlive/2026`: TeX Live's
60    //      fast path (MiKTeX already exited at step 1).
61    //   3. `pdflatex --version`: last-resort fallback for exotic setups where
62    //      neither kpsewhich signal carries a year.
63    // Answer-identical to a SELFAUTOPARENT-first order on both distros; just
64    // fewer spawns (MiKTeX: 2 → 1).
65    if let Some(year) = year_from_kpsewhich_version_banner() {
66      return Some(year);
67    }
68    if let Some(year) = year_from_kpsewhich_selfautoparent() {
69      return Some(year);
70    }
71    year_from_pdflatex_version()
72  })
73}

75/// The ambient `kpsewhich --version` banner, memoized ONCE per process in
76/// `latexml_core` — the lowest crate that spawns kpsewhich, shared with the
77/// kpathsea backend selection (`select_kpaths`), so the whole process spawns
78/// `kpsewhich --version` at most once across backend-choice, year detection,
79/// and the stamp check. Thin re-export for this crate's consumers. Returns
80/// `None` if kpsewhich is absent or the call fails.
81pub fn ambient_kpsewhich_version() -> Option<&'static str> {
82  latexml_core::util::pathname::ambient_kpsewhich_version()
83}

85/// Year from the memoized `kpsewhich --version` banner. Only MiKTeX carries a
86/// year here (`MiKTeX YY.MM` on stdout, e.g. `MiKTeX 25.12` → 2025); TeX Live
87/// prints `kpathsea version 6.4.x` (no year) and so returns `None`, deferring
88/// to SELFAUTOPARENT. Reuses [`parse_year_from_version_banner`] — the same
89/// extractor as the `pdflatex --version` fallback, which also handles a
90/// leading MiKTeX "check for updates" warning line without a false match.
91fn year_from_kpsewhich_version_banner() -> Option<u32> {
92  parse_year_from_version_banner(ambient_kpsewhich_version()?)
93}

95fn year_from_kpsewhich_selfautoparent() -> Option<u32> {
96  let out = std::process::Command::new("kpsewhich")
97    .arg("-var-value=SELFAUTOPARENT")
98    .output()
99    .ok()?;
100  if !out.status.success() {
101    return None;
102  }
103  let s = String::from_utf8(out.stdout).ok()?;
104  let trimmed = s.trim();
105  let last = Path::new(trimmed).file_name()?.to_str()?;
106  parse_year_str(last)
107}

109fn year_from_pdflatex_version() -> Option<u32> {
110  let out = std::process::Command::new("pdflatex")
111    .arg("--version")
112    .output()
113    .ok()?;
114  if !out.status.success() {
115    return None;
116  }
117  let s = String::from_utf8(out.stdout).ok()?;
118  parse_year_from_version_banner(&s)
119}

121/// Extract a TL-equivalent year from a `pdflatex --version` banner.
122///
123///   - TeX Live: first lines contain `(TeX Live YYYY)`.
124///   - MiKTeX: no TL year exists anywhere (and `SELFAUTOPARENT` is EMPTY on
125///     MiKTeX, so detection lands here). The banner reads
126///     `MiKTeX-pdfTeX 4.23 (MiKTeX 25.12)`; MiKTeX is a rolling release
127///     whose `YY.MM` package snapshot tracks the same-year TeX Live, so map
128///     `MiKTeX YY.*` → `20YY` and let dump resolution do its usual
129///     nearest/most-recent fallback if that exact year isn't bundled.
130fn parse_year_from_version_banner(s: &str) -> Option<u32> {
131  for line in s.lines().take(3) {
132    if let Some(idx) = line.find("TeX Live ") {
133      let tail = &line[idx + "TeX Live ".len()..];
134      let candidate: String = tail.chars().take(4).collect();
135      if let Some(y) = parse_year_str(&candidate) {
136        return Some(y);
137      }
138    }
139    // Match the parenthesized distro stamp, not the product prefix: the
140    // banner starts `MiKTeX-pdfTeX 4.23 (MiKTeX 25.12)` and only the
141    // second occurrence carries the version.
142    if let Some(idx) = line.find("MiKTeX ") {
143      let tail = &line[idx + "MiKTeX ".len()..];
144      let digits: String = tail.chars().take_while(char::is_ascii_digit).collect();
145      if digits.len() == 2
146        && let Ok(yy) = digits.parse::<u32>()
147        && (20..=99).contains(&yy)
148      {
149        return Some(2000 + yy);
150      }
151    }
152  }
153  None
154}

156fn parse_year_str(s: &str) -> Option<u32> {
157  // Accept a trailing non-digit suffix: MacTeX/BasicTeX name their TL root
158  // with one (e.g. `/usr/local/texlive/2026basic`), so SELFAUTOPARENT's
159  // final component carries the year + "basic". Parse the leading digit
160  // run; a string that doesn't START with the year (e.g. "texlive") still
161  // fails over to the `pdflatex --version` detection.
162  let digits = &s[..s.find(|c: char| !c.is_ascii_digit()).unwrap_or(s.len())];
163  let y: u32 = digits.parse().ok()?;
164  if (2000..=2099).contains(&y) {
165    Some(y)
166  } else {
167    None
168  }
169}

171/// Versioned filename for a kernel dump, e.g.
172/// `dump_filename("plain", 2025) == "plain.2025.dump.txt"`.
173pub fn dump_filename(kind: &str, year: u32) -> String { format!("{}.{}.dump.txt", kind, year) }
174
175/// One-shot stderr banner emitted by the `LoadFormat` branches in
176/// `tex.rs` / `latex.rs` when NO precompiled kernel dump is available
177/// and the engine falls back to the raw `latex.ltx` + `expl3-code.tex`
178/// load path.
179///
180/// Why this exists: the raw-load path hits known raw-load-ONLY cascades
181/// (the `expl3-code.tex` L33075 codepoint dangling-group and the
182/// `\@expl@pop@filename@@` expl-status desync) that the dump avoids. On
183/// expl3-heavy arXiv articles this inflates per-paper error counts by
184/// ~1000× (2112.11932: 1 → 1003; 2110.10227: 4 → 102), which previously
185/// masqueraded as a Rust parity gap when it was really a missing-dump
186/// SETUP error (`resources/dumps/` not populated — run
187/// `tools/make_formats.sh`). The dump is a required kernel piece for
188/// canvas/parity work, so its absence must never be silent.
189///
190/// Fires at most once per process (shared `Once`), and is a plain stderr
191/// line — NOT an `Error:`/`Fatal:` (those are reserved for the per-paper
192/// conversion log and would corrupt canvas error counts).
193pub fn warn_degraded_no_dump() {
194  static WARNED: std::sync::Once = std::sync::Once::new();
195  WARNED.call_once(|| {
196    let year = detect_ambient_texlive_year()
197      .map(|y| y.to_string())
198      .unwrap_or_else(|| "unknown".into());
199    eprintln!(
200      "[latexml-oxide] WARNING: no precompiled kernel dump found (TL{year}); \
201       running in DEGRADED raw-load mode. Conversions will show many spurious \
202       errors (expl3 catcode/group cascades). Build the dump with \
203       `tools/make_formats.sh` before canvas/parity runs. (Set \
204       LATEXML_NODUMP=1 to silence this if the raw path is intentional.)"
205    );
206  });
207}

209/// Versioned filename for the texlive stamp, e.g.
210/// `version_filename(2025) == "texlive.2025.version"`.
211pub fn version_filename(year: u32) -> String { format!("texlive.{}.version", year) }

213/// Reverse of [`dump_filename`]: extract the year tag from a versioned
214/// dump filename. Returns `None` for un-versioned legacy names.
215pub fn parse_year_from_dump_filename(name: &str, kind: &str) -> Option<u32> {
216  let prefix = format!("{}.", kind);
217  let rest = name.strip_prefix(&prefix)?;
218  let year_str = rest.strip_suffix(".dump.txt")?;
219  parse_year_str(year_str)
220}

222/// All TL years for which a `<kind>.YYYY.dump.txt` file exists in `dir`,
223/// sorted descending (most-recent first).
224pub fn available_years_in_dir(dir: &Path, kind: &str) -> Vec<u32> {
225  let mut years = Vec::new();
226  let Ok(entries) = std::fs::read_dir(dir) else {
227    return years;
228  };
229  for entry in entries.flatten() {
230    let name = entry.file_name();
231    let Some(name_str) = name.to_str() else {
232      continue;
233    };
234    if let Some(y) = parse_year_from_dump_filename(name_str, kind) {
235      years.push(y);
236    }
237  }
238  years.sort_unstable_by(|a, b| b.cmp(a));
239  years.dedup();
240  years
241}

243/// Pick the best dump in `dir` for the given `kind` (`"plain"` /
244/// `"latex"`):
245///   - if `prefer` is `Some(y)` and `<kind>.y.dump.txt` exists, use it;
246///   - else use the most-recent year present;
247///   - else `None`.
248pub fn resolve_versioned_in_dir(
249  dir: &Path,
250  kind: &str,
251  prefer: Option<u32>,
252) -> Option<(PathBuf, u32)> {
253  if let Some(y) = prefer {
254    let p = dir.join(dump_filename(kind, y));
255    if p.is_file() {
256      return Some((p, y));
257    }
258  }
259  let years = available_years_in_dir(dir, kind);
260  let chosen = years.first().copied()?;
261  Some((dir.join(dump_filename(kind, chosen)), chosen))
262}

264/// Sibling stamp file for a dump path — e.g. `texlive.2025.version`
265/// next to `latex.2025.dump.txt`.
266pub fn stamp_path_for_dump(dump_path: &Path, year: u32) -> Option<PathBuf> {
267  Some(dump_path.parent()?.join(version_filename(year)))
268}

271mod year_parse_tests {
272  use super::{parse_year_from_version_banner, parse_year_str};

274  #[test]
275  fn texlive_banner() {
276    assert_eq!(
277      parse_year_from_version_banner("pdfTeX 3.141592653-2.6-1.40.29 (TeX Live 2026)"),
278      Some(2026)
279    );
280    // Homebrew stamp variant.
281    assert_eq!(
282      parse_year_from_version_banner("pdfTeX 3.14 (TeX Live 2026/Homebrew)"),
283      Some(2026)
284    );
285  }

287  #[test]
288  fn miktex_banner_maps_rolling_version_to_year() {
289    // Verbatim from MiKTeX 25.12 on Windows (probe 2026-07-12); note
290    // SELFAUTOPARENT is empty on MiKTeX, so this arm is the only signal.
291    assert_eq!(
292      parse_year_from_version_banner("MiKTeX-pdfTeX 4.23 (MiKTeX 25.12)"),
293      Some(2025)
294    );
295  }
296
297  #[test]
298  fn unknown_banner_rejected() {
299    assert_eq!(
300      parse_year_from_version_banner("pdfTeX 3.14 (Web2C 2026)"),
301      None
302    );
303    // A hypothetical 3-digit MiKTeX version must not parse as a year.
304    assert_eq!(parse_year_from_version_banner("(MiKTeX 125.1)"), None);
305  }

307  #[test]

317  }

319  #[test]

323    assert_eq!(parse_year_str(""), None);
324    assert_eq!(parse_year_str("1999"), None); // outside the sane range
325    assert_eq!(parse_year_str("20269"), None); // 5-digit run is not a year
