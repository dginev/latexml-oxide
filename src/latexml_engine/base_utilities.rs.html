33  // LaTeX has a very particular, but useful, notion of "Undefined",
34  //    so let's get that squared away at the outset; it's useful for TeX, too!
35  //
36  // Naturally, it uses \csname to check, which ends up DEFINING the possibly undefined macro as
37  // \relax
38  // Perl Base_Utility.pool.ltxml L23-31
39  DefMacro!("\\lx@ifundefined{}{}{}", sub[(name, if_token, else_token)] {
40    let cs = T_CS!(s!("\\{}", Expand!(name).to_string()));
41    // Autoload triggers (declared via `def_autoload` at tex.rs:238-247)
42    // install a closure under the trigger CS so the package auto-loads on
43    // first invocation. Perl scopes the equivalent `DefAutoload` entries
44    // to `OmniBus.cls.ltxml`, so for non-OmniBus papers Perl sees these
45    // CSes as truly undefined. Mirror Perl by treating an unfired
46    // autoload trigger as "undefined" in `\@ifundefined`. We must NOT
47    // overwrite the trigger CS with `\relax` in this case (the kernel
48    // `\csname X\endcsname \ifx \relax` idiom DOES overwrite, but doing
49    // so here would destroy the autoload — subsequent use of the trigger
50    // CS would no-op instead of loading its package). Driver:
51    // arXiv:2507.23241v1 (smfart.cls) — line 373's
52    // `\@ifundefined{numberwithin}` branches wrong when our preloaded
53    // `\numberwithin` autoload makes the CS "look" defined, then the
54    // `\@gobbletwo` branch eats `\ifx \relax` and orphans `\else` / `\fi`.
55    // An autoload TRIGGER counts as "undefined" only while UNFIRED — i.e. its
56    // target package has not yet loaded. `def_autoload` stores the package name
57    // (`.sty`) as a String; once `<pkg>.sty_loaded`/`_raw_loaded` is set the
58    // trigger CS holds the package's real definition and must read as DEFINED
59    // (fixes `\@ifundefined{align}` after `\usepackage{amsmath}`). `.pool`
60    // triggers keep the legacy Bool form and stay "undefined until used".
61    let is_autoload = cs.with_cs_name(|cs_name| {
62      match lookup_value(&s!("{cs_name}:autoload")) {
63        Some(Stored::String(pkg_sym)) => {
64          let pkg = with(pkg_sym, |s| s.to_string());
65          !lookup_bool(&s!("{pkg}.sty_loaded"))
66            && !lookup_bool(&s!("{pkg}.sty_raw_loaded"))
67        },
68        Some(Stored::Bool(b)) => b,
69        _ => false,
70      }
71    });
72    if IsDefined!(&cs) && !is_autoload {
73      Ok(else_token)
74    } else {
75      if !is_autoload {
76        assign_meaning(&cs, lookup_meaning(&TOKEN_RELAX), None);  // Let w/o AfterAssign
77      }
78      Ok(if_token)
79    }

232        while let Some(peeked) = first_text_iter.peek() {
233          if peeked.is_whitespace() || matches!(peeked, ',' | '!' | ';' | '.' | ':' | '?') {
234            first_text_iter.next();
235          } else {
236            break;
237          }
238        }
239        let new_text = first_text_iter.collect::<String>();
240        if first_text != new_text {
241          first.set_content(&new_text)?;
242        }
243      }
244      if let Some(mut last) = node.get_last_child()
245        && last.get_type() == Some(NodeType::TextNode) {
246          let last_text = last.get_content();
247          let mut last_text_iter  = last_text.chars().rev().peekable();
248          while let Some(peeked) = last_text_iter.peek() {

514    match get_frontmatter_name(options.get("name"), &tag, &role)? {
515      Some(name) => { options.insert("name".to_string(), name); },
516      None => { options.remove("name"); },
517    }
518    options.insert("_label".to_string(), labels.join(","));
519
520    // Snapshot the parent entries (those not role=pending), inherit labels
521    // from an enclosing pending entry, and push a tentative stub entry —
522    // in case digestion changes labels!
523    let (parent_indices, stub_idx) = with_value_mut("frontmatter", |val_opt| {
524      if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt {
525        let list = frnt.entry(parenttag.clone()).or_insert_with(Vec::new);
526        let parent_indices: Vec<usize> = list.iter().enumerate()
527          .filter(|(_, e)| e.attr.get("role").map(String::as_str).unwrap_or("") != "pending")
528          .map(|(i, _)| i)
529          .collect();
530        // IF this item encountered WITHIN another frontmatter
531        // we inherit that frontmatter's labels (even override!)
532        if let Some(last) = list.last()
533          && last.attr.get("role").map(String::as_str) == Some("pending")
534            && matches!(last.content.first(), Some(TagContent::PlaceKeeper))
535          {
536            let inherited = last.attr.get("_annotations").cloned().unwrap_or_default();
537            options.insert("_label".to_string(), inherited);
538          }
539        let mut stub_attr = TagAttrs::default();
540        stub_attr.insert("role".to_string(), "pending".to_string());
541        stub_attr.insert("_annotations".to_string(),
542          options.get("_label").cloned().unwrap_or_default());
543        let stub = TagData {
544          tag: parenttag.clone(),
545          attr: stub_attr,
546          content: vec![TagContent::PlaceKeeper],
547        };
548        DebugFeature!("frontmatter", "FRONT Add stub {}\n  for annotation {} [{}]",
549          show_frontmatter(&stub), tag, options.get("_label").map(String::as_str).unwrap_or(""));
550        list.push(stub);
551        (parent_indices, list.len() - 1)
552      } else {
553        (Vec::new(), 0)
554      }
555    });
556    let nparents = parent_indices.len();
557    let xcontent = digest_frontmatter_item(&tag, content)?;
558    // Reset if changed (eg. by \lx@set@frontmatter@label during digestion)!
559    let stub_label = with_value("frontmatter", |v| {
560      if let Some(Stored::HashTagData(frnt)) = v {
561        frnt.get(&parenttag)
562          .and_then(|l| l.get(stub_idx))
563          .and_then(|e| e.attr.get("_annotations"))
564          .cloned()
565      } else {
566        None
567      }
568    }).unwrap_or_default();
569    options.insert("_label".to_string(), stub_label.clone());
570    let datum = if preformatted {
571      TagContent::Box(xcontent)
572    } else {
573      TagContent::Entry(TagData {
574        tag: tag.clone(),
575        attr: options.clone(),
576        content: vec![TagContent::Box(xcontent)],
577      })
578    };
579    if !preformatted && (!stub_label.is_empty() || nparents == 0) {
580      // deferred until we can compare labels
581      DebugFeature!("frontmatter", "... deferring to label={stub_label}");
582      frontmatter_set_first_content(&parenttag, stub_idx, datum);
583    } else {
584      let annotate = options.get("annotate").cloned().unwrap_or_default();
585      let (mut nprev, newonly): (i64, bool) = if annotate.is_empty() {
586        (1, false)
587      } else if annotate.chars().all(|c| c.is_ascii_digit()) {
588        (annotate.parse().unwrap_or(1), false)
589      } else if annotate == "all" {
590        (nparents as i64, false)
591      } else if annotate == "new" {
592        (nparents as i64, true)
593      } else {
594        Info!("unexpected", &tag, s!("Frontmatter annotate '{annotate}' unrecognized"));
595        (1, false)
596      };
597      DebugFeature!("frontmatter", "...adding to {nprev} previous{}",
598        if newonly { " new" } else { "" });
599      with_value_mut("frontmatter", |val_opt| {
600        if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt
601          && let Some(list) = frnt.get_mut(&parenttag) {
602            // Remove unneeded (stub) entry.
603            if list.get(stub_idx)
604              .map(|e| e.attr.get("role").map(String::as_str) == Some("pending"))
605              .unwrap_or(false)
606            {
607              list.remove(stub_idx);
608            }
609            let has_role_key = s!("_has{role}");
610            let mut indices = parent_indices.clone();
611            while let Some(pi) = indices.pop() {
612              if nprev <= 0 {
613                break;
614              }
615              nprev -= 1;
616              if let Some(parent) = list.get_mut(pi) {
617                parent.content.push(datum.clone());
618                if !role.is_empty() {
619                  parent.attr.insert(has_role_key.clone(), "1".to_string());
620                  if newonly
621                    && let Some(&next_pi) = indices.last()
622                      && list.get(next_pi)
623                        .map(|p| p.attr.contains_key(&has_role_key))
624                        .unwrap_or(false)
625                      {
626                        break;
627                      }
628                }
629              }
630            }
631          }
632      });
633    }

832  DefMacro!("\\lx@add@authors{}", sub[(stuff)] {
833    let mut calls: Vec<Token> = Vec::new();
834    dequeue_front_matter("ltx:creator", &[("role", "author")]);
835    // If too much formatting, fall back to unstructured author content
836    let stuff_string = stuff.to_string();
837    if stuff_string.contains("{tabular}")
838      || stuff_string.contains("{minipage}")
839      || stuff_string.contains("\\halign")
840    {
841      calls.extend(Invocation!(T_CS!("\\lx@add@author"), vec![None, Some(stuff)]).unlist());
842    } else if position_of(&stuff, &authorsup_markers()).is_some() {
843      let lines = split_tokens(stuff, author_affil_splits());
844      // entries of (is_author, line)
845      let mut entries: Vec<(bool, Tokens)> = Vec::new();
846      for line in lines {
847        if line.is_empty() {
848          continue;
849        }
850        match position_of(&line, &authorsup_markers()) {
851          None => {
852            // No marker?
853            if let Some(last) = entries.last_mut() {
854              // continues previous entry; Append
855              let mut appended = last.1.clone().unlist();
856              appended.extend(line.unlist());
857              last.1 = Tokens::new(appended);
858            } else {
859              entries.push((true, line)); // safest to assume author?
860            }
861          },
862          Some(p) if p < 8 => {
863            // Close to front? assume affiliation
864            entries.push((false, line));
865          },
866          Some(_) => {
867            // Marker sits far from the front: an author line. Split it into the
868            // individual creators it names (see split_author_line).
869            for author in split_author_line(line) {
870              entries.push((true, author));
871            }
872          },
873        }
874      }
875      for (is_author, line) in entries {
876        if is_author {
877          let withsup = Invocation!(T_CS!("\\lx@author@withsup"), vec![Some(line)]);
878          calls.extend(
879            Invocation!(T_CS!("\\lx@add@author"), vec![None, Some(withsup)]).unlist());
880        } else {
881          let withsup = Invocation!(T_CS!("\\lx@affiliation@withsup"), vec![Some(line)]);
882          calls.extend(
883            Invocation!(T_CS!("\\lx@add@affiliation"), vec![None, Some(withsup)]).unlist());
884        }
885      }
886    } else {
887      for block in split_tokens(stuff, author_splits()) {
888        if block.is_empty() {
889          continue;
890        }
891        let mut pieces = split_tokens(block, vec![SplitDelim::Token(T_CS!("\\\\"))]).into_iter();
892        let author = pieces.next().unwrap_or_default();
893        let mut body: Vec<Token> = author.unlist();
894        for line in pieces {
895          if !line.is_empty() {
896            body.extend(
897              Invocation!(T_CS!("\\lx@add@affiliation"), vec![None, Some(line)]).unlist());
898          }
899        }
900        calls.extend(
901          Invocation!(T_CS!("\\lx@add@author"), vec![None, Some(Tokens::new(body))]).unlist());
902      }
903    }
904    Ok(Tokens::new(calls))

1659/// Digest FrontMatter (if not already?)
1660/// Perl: digestFrontMatter().
1661pub fn digest_front_matter() -> Result<()> {
1662  bgroup();
1663  // INTENTIONAL DIVERGENCE from Perl PR #2767 (KNOWN_PERL_ERRORS #30,
1664  // OXIDIZED_DESIGN): clear the queue BEFORE digesting. Perl digests
1665  // from the live queue and wipes it after the loop — but when a queued
1666  // entry's own content re-triggers digestFrontMatter (witness aa.cls
1667  // 0907.0384: `\abstract{...}{}` dispatches the 5-arg \abstract@new,
1668  // whose greedy params swallow the document's `\maketitle` into arg #5,
1669  // so the queued abstract CONTAINS \maketitle → \lx@frontmatterhere →
1670  // afterDigest → re-entry), Perl re-digests the same queue unboundedly:
1671  // PR-head Perl dies `Fatal:perl:deep_recursion ... invokeToken`, zero
1672  // output (verified 2026-06-04). Pre-clearing makes the nested
1673  // invocation see an empty queue and terminate; the same paper then
1674  // converts with zero errors (real LaTeX also compiles it). Everything
1675  // else — deferred timing, digestion order, late re-let/\def fidelity —
1676  // is exactly the PR's. Newly-queued entries during this digest are
1677  // processed by the next invocation (or the end-of-document fallback).
1678  let commands: Vec<RawFrontmatter> = match remove_value("frontmatter_raw") {
1679    Some(Stored::FrontmatterRaw(commands)) => commands,
1680    _ => Vec::new(),
1681  };
1682  if !commands.is_empty() {
1683    let_i(
1684      &T_CS!("\\lx@add@frontmatter"),
1685      &T_CS!("\\lx@add@frontmatter@now"),
1686      None,
1687    );
1688    let_i(
1689      &T_CS!("\\lx@annotate@frontmatter"),
1690      &T_CS!("\\lx@annotate@frontmatter@now"),
1691      None,
1692    );
1693    for (tag, attr, command) in commands {
1694      DebugFeature!(
1695        "frontmatter",
1696        "FRONT Digesting {tag} [{:?}] {command}",
1697        attr
1698      );
1699      // Perl parity (review 2026-06-04, replacing the master-era
1700      // fatal-swallow from witness arXiv:1903.01633): a Fatal raised
1701      // while digesting deferred frontmatter propagates and aborts
1702      // the conversion with a proper `Fatal:` log line, exactly like
1703      // Perl's un-eval'd `$stomach->digest($command)`. The 1903.01633
1704      // bug was the *silent* swallow (`let _ = digest(...)` left
1705      // `report.fatal=true` with no log line); propagation fixes the
1706      // silence without diverging from Perl. Non-fatal Error!s inside
1707      // the digest log-and-continue in both engines as before.
1708      digest(command)?;
1709    }
1710  }
1711  // Add punctuation to all ltx:creators, now that we know how many of each role.
1712  let mut updates: Vec<(usize, bool)> = Vec::new(); // (index, use_conjunction)
1713  with_value("frontmatter", |v| {
1714    if let Some(Stored::HashTagData(frnt)) = v
1715      && let Some(list) = frnt.get("ltx:creator")
1716    {
1717      for (i, item) in list.iter().enumerate() {
1718        let role = item.attr.get("role").map(String::as_str).unwrap_or("");
1719        let num: i64 = item
1720          .attr
1721          .get("_num")
1722          .and_then(|n| n.parse().ok())
1723          .unwrap_or(0);
1724        if !role.is_empty() && num > 1 {
1725          let n = lookup_mapping_int("num_ltx:creator", role);
1726          updates.push((i, num >= n));
1727        }
1728      }
1729    }
1730  });
1731  for (index, use_conjunction) in updates {
1732    let separator = DigestText!(Tokens!(if use_conjunction {
1733      T_CS!("\\lx@author@conj")
1734    } else {
1735      T_CS!("\\lx@author@sep")
1736    }))?;
1737    // `\lx@author@conj` may digest to \hskip Whatsits (\qquad) — extract
1738    // text-or-spaces (see \lx@author@prefix note in the previous port).
1739    let text = digested_to_text(&separator)?;
1740    with_value_mut("frontmatter", |val_opt| {
1741      if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt
1742        && let Some(entry) = frnt.get_mut("ltx:creator").and_then(|l| l.get_mut(index))
1743      {
1744        entry.attr.insert("before".to_string(), text.clone());
1745      }
1746    });
1747  }
1748  egroup()?;
1749  Ok(())
1750}

1798/// Beyond-Perl heuristic: recover a document title from a hand-formatted leading
1799/// block when the author never used `\title`/`\maketitle`.
1800///
1801/// Many papers set their title as a plain `\begin{center}{\Large ...}\end{center}`
1802/// block instead of the frontmatter machinery, then declare only an abstract (e.g.
1803/// arXiv 1609.07638). With no `\title` there is no `ltx:title` frontmatter to
1804/// flush, so the document renders titleless. This promotes that first block — gated
1805/// conservatively so it does not fire on epigraphs, dedications, centered figures or
1806/// "draft" notices — into a real `<ltx:title>` at the document top.
1807///
1808/// Called only from the ABSTRACT-ONLY `\lx@frontmatter@fallback` branch: reaching it
1809/// already means no `\title`/`\author`/`\maketitle` were used yet the paper carries
1810/// frontmatter (an abstract) — the strongest signal a leading display block is the
1811/// title. It fires only when no `<ltx:title>` exists yet, the first non-resource body
1812/// element holds a leading centered `<ltx:p>` set in a larger-than-body font, and
1813/// that paragraph has non-whitespace text. On a match the paragraph's inline children
1814/// are MOVED into a fresh `<ltx:title>` after the resources, pruning empty wrappers.
1815fn maybe_promote_leading_title(document: &mut Document) -> Result<()> {
1816  // Never override an existing (real) title.
1817  if document.findnode("/ltx:document/ltx:title", None).is_some() {
1818    return Ok(());
1819  }
1820  // The first non-resource body element (the candidate title block). Fetch it
1821  // with an ABSOLUTE query: nodes returned from a RELATIVE-context findnode are
1822  // detached for child traversal (a rust-libxml shared-node artifact —
1823  // get_content works but get_child_nodes is empty), so every step below walks
1824  // the live DOM by hand from this anchor.
1825  let Some(first_body) = document.findnode("/ltx:document/*[not(self::ltx:resource)][1]", None)
1826  else {
1827    return Ok(());
1828  };
1829  // Never DESCEND INTO SECTIONAL content: on the second fallback pass (fired
1830  // at the next \section) the first body element can be the COMPLETED first
1831  // section, and an unconstrained DFS would steal a centered display-font
1832  // paragraph from inside it (e.g. a "Dedicated to ..." block) as the paper
1833  // title. A hand-formatted title block is a plain para/quote at document
1834  // level — reject sectional anchors outright (PR_READINESS must-fix 3).
1835  let anchor_q = document::get_node_qname(&first_body);
1836  let is_sectional = with(anchor_q, |q| {
1837    matches!(
1838      q,
1839      "ltx:section"
1840        | "ltx:subsection"
1841        | "ltx:subsubsection"
1842        | "ltx:paragraph"
1843        | "ltx:subparagraph"
1844        | "ltx:chapter"
1845        | "ltx:part"
1846        | "ltx:appendix"
1847        | "ltx:bibliography"
1848        | "ltx:index"
1849        | "ltx:glossary"
1850    )
1851  });
1852  if is_sectional {
1853    return Ok(());
1854  }
1855  // First centered <ltx:p> in document order within the block; it must be set in
1856  // a larger-than-body font (the display-title signal) and hold real text.
1857  let Some(title_p) = first_centered_paragraph(&first_body) else {
1858    return Ok(());
1859  };
1860  let nominal = {
1861    let v = lookup_int("NOMINAL_FONT_SIZE");
1862    if v > 0 { v as f64 } else { 10.0 }
1863  };
1864  if title_p.get_content().trim().is_empty()
1865    || !descendant_has_display_font(document, &title_p, nominal)
1866  {
1867    return Ok(());
1868  }
1869  // Create <ltx:title> at the document top (after the last resource), mirroring
1870  // the frontmatter fallback's placement.
1871  let mut point = document.findnode("/ltx:document/ltx:resource[last()]", None);
1872  if let Some(p) = point.take() {
1873    point = p.get_next_sibling();
1874  }
1875  // No resources (or resource is the last child): insert before the first
1876  // body element instead of silently declining (review corner-case).
1877  let point = point.unwrap_or_else(|| first_body.clone());
1878  // Create with a BARE tag + bind the root (default LaTeXML) namespace, so it
1879  // serializes as <title> like a real frontmatter title — a prefixed "ltx:title"
1880  // qname would emit a stray <ltx:title>. Mirrors open_element_internal's
1881  // default-namespace path.
1882  let mut title = document.insert_element_before(&point, "title", None)?;
1883  if let Some(rns) = document
1884    .get_document()
1885    .get_root_element()
1886    .and_then(|r| r.get_namespace())
1887  {
1888    let _ = title.set_namespace(&rns);
1889  }
1890  // MOVE the paragraph's inline children into the title (a true move preserves
1891  // xml:ids, so any footnotes/marks carry over rather than being cloned+orphaned).
1892  for mut child in title_p.get_child_nodes() {
1893    child.unbind();
1894    title.add_child(&mut child)?;
1895  }
1896  // Drop the now-empty paragraph, and any wrapper ancestor it leaves empty
1897  // (para → logical-block), so we don't strand empty blocks above the abstract.
1898  let mut victim = title_p;
1899  loop {
1900    let parent = victim.get_parent();
1901    document.remove_node(victim);
1902    match parent {
1903      Some(p) if !has_element_child(&p) => {
1904        let is_wrapper = with(document::get_node_qname(&p), |name| {
1905          name == "ltx:para" || name == "ltx:logical-block"
1906        });
1907        if is_wrapper {
1908          victim = p;
1909          continue;
1910        }
1911      },
1912      _ => {},
1913    }
1914    break;
1915  }
1916  Ok(())
1917}

1919/// Insert FrontMatter into document, if not already added
1920/// Perl: insertFrontMatter($document).
1921pub fn insert_frontmatter(document: &mut Document) -> Result<()> {

1923    return Ok(());
1924  }
1925  digest_front_matter()?; // If needed
1926  let frontmatter_elements_set: HashSet<String> = FRONTMATTER_ELEMENTS
1927    .iter()
1928    .map(ToString::to_string)
1929    .collect();
1930
1931  // Collect the frontmatter hash keys via with_value — we only need the
1932  // key set here; the full HashTagData clone previously happened just
1933  // to call .keys().cloned().collect(). The hash itself is consumed a
1934  // few lines below via remove_value, so no iteration on the borrow
1935  // survives past this closure.
1936  let set_keys: Vec<String> = with_value("frontmatter", |v| match v {
1937    Some(Stored::HashTagData(frnt)) => frnt.keys().cloned().collect(),
1938    _ => Vec::new(),
1939  });
1940  if set_keys.is_empty() {
1941    return Ok(());
1942  }
1943
1944  // If doc ONLY has abstract as frontmatter, defer until abstract's document location
1945  if set_keys.len() == 1 && set_keys[0] == "ltx:abstract" && !lookup_bool("frontmatter_deferred") {
1946    assign_value("frontmatter_deferred", true, Some(Scope::Global));
1947    return Ok(());
1948  }
1949
1950  // OK, we're placing FrontMatter here, now.
1951  assign_value("frontmatter_done", true, Some(Scope::Global));
1952
1953  // Remove frontmatter and replace with empty
1954  let mut frontmatter = match remove_value("frontmatter") {
1955    Some(Stored::HashTagData(frnt)) => frnt,
1956    _ => return Ok(()),
1957  };
1958  assign_value(
1959    "frontmatter",
1960    Stored::HashTagData(HashMap::default()),
1961    Some(Scope::Global),
1962  );
1963
1964  // Order: first go through frontmatter_elements order, then any custom keys
1965  let custom_keys: Vec<String> = frontmatter
1966    .keys()
1967    .filter(|key| !frontmatter_elements_set.contains(key.as_str()))
1968    .map(ToString::to_string)
1969    .collect();
1970  let mut all_keys: Vec<String> = FRONTMATTER_ELEMENTS
1971    .iter()
1972    .map(ToString::to_string)
1973    .collect();
1974  all_keys.extend(custom_keys);
1975
1976  for key in &all_keys {
1977    if let Some(list) = frontmatter.remove(key) {
1978      // Dubious, but assures that frontmatter appears in text mode...
1979      document.set_box_to_absorb(
1980        Tbox::new(
1981          pin!(""),
1982          lookup_font(),
1983          None,

1985          SymHashMap::default(),
1986        )
1987        .into(),
1988      );
1989      for item in list {
1990        insert_frontmatter_entry(document, &item)?;
1991      }
1992      document.expire_box_to_absorb();
1993    }
1994  }
1995  relocate_annotations(document)?;
1996  Ok(())
1997}

1999/// Insert one frontmatter entry `[tag, {attr}, @content]`.
2000/// Perl: insertFrontMatter_rec($document, $item) for the ARRAY case.
2001fn insert_frontmatter_entry(document: &mut Document, entry: &TagData) -> Result<()> {
2002  let TagData { tag, attr, content } = entry;
2003  DebugFeature!("frontmatter", "FRONT Inserting {}", show_frontmatter(entry));
2004  // token-locators: frontmatter elements (e.g. <ltx:title> from `\title{…}`)
2005  // are opened here, far from their source, around content that was digested
2006  // and stored back at `\lx@add@frontmatter` time. open_element would otherwise
2007  // stamp them with no/last locator (→ the whole-document fallback in clients).
2008  // Recover the deferred content's span and stamp the element with it.
2009  #[cfg(feature = "token-locators")]
2010  if let Some(TagContent::Box(stuff)) = content.iter().find(|c| matches!(c, TagContent::Box(_))) {
2011    document.set_current_box_locator(latexml_core::definition::constructor::child_span(stuff));
2012  }
2013  // Perl: font => $stuff[0]->getFont, _force_font => 'true' when the tag
2014  // can have a font attribute and there is content.
2015  let mut attributes: HashMap<String, String> = attr.clone();
2016  let mut font: Option<Font> = None;
2017  if !content.is_empty()
2018    && document::can_have_attribute(tag, "font")
2019    && let Some(TagContent::Box(first)) = content.first()
2020    && let Ok(Some(f)) = first.get_font()
2021  {
2022    font = Some(f.into_owned());
2023    attributes.insert("_force_font".to_string(), "true".to_string());
2024  }
2025  document.open_element(tag, Some(attributes), font.as_ref())?;
2026  for item in content {
2027    insert_frontmatter_rec(document, item)?;
2028  }
2029  document.close_element(tag)?;
2030  // At this time, the frontmatter element should really carry the actual literal values intended.
2031  // (Perl PR #2767 disables the former empty-element pruning here.)
2032  Ok(())
2033}

2118              .iter()

2214pub fn parse_def_parameters(cs: &Token, params_in: Tokens) -> Result<Option<Parameters>> {
2215  let mut tokens: VecDeque<Token> = VecDeque::from(params_in.pack_parameters()?.unlist());
2216  // Now, recognize parameters and delimiters.
2217  let mut params = Vec::new();
2218  let mut n = 0;
2219  while let Some(mut t) = tokens.pop_front() {
2220    let cc = t.get_catcode();
2221    if cc == Catcode::PARAM || cc == Catcode::ARG {
2222      if cc == Catcode::PARAM {
2223        if tokens.is_empty() {
2224          // Special case: lone # NOT following a numbered parameter
2225          // Note that we require a { to appear next, but do NOT read it!
2226          params.push(Parameter::new(
2227            Cow::Borrowed("RequireBrace"),
2228            Cow::Borrowed("RequireBrace"),
2229            None,
2230          )?);
2231          break;
2232        } else {
2233          n += 1;
2234          if let Some(t_next) = tokens.pop_front() {
2235            t = t_next;
2236          } else {
2237            unreachable!("tokens.is_empty() was false, so pop_front must return Some");
2238          }
2239        }
2240      } else {
2241        // CC_ARG case, keep looking at this token
2242        n += 1;
2243      }
2244      if n > 0 {
2245        let t_num = t.with_str(|ts| ts.parse::<i8>()).unwrap_or(-1);
2246        if t_num != n {
2247          fatal!(
2248            ParamSpec,
2249            Expected,
2250            s!(
2251              "Parameters for {:?} not in order. Got {:?}, expected {:?}. in {:?}",
2252              cs,
2253              t,
2254              n,
2255              params
2256            )
2257          );
2258        }
2259      }
2260      // Check for delimiting text following the parameter #n
2261      let mut delim = Vec::new();
2262      let mut pc = Catcode::MARKER; // throwaway initial val
2263      while !tokens.is_empty() {
2264        let inner_cc = tokens.front().unwrap().get_catcode();
2265        if inner_cc == Catcode::PARAM || inner_cc == Catcode::ARG {
2266          break;
2267        }
2268        let d = tokens.pop_front().unwrap();
2269        if !(pc == Catcode::SPACE && inner_cc == Catcode::SPACE) {
2270          // BUT collapse whitespace!
2271          delim.push(d);
2272        }
2273        pc = inner_cc;
2274      }

2276      if !delim.is_empty() {

2280            name: pin!("Until"),
2281            spec: pin(format!("Until:{extra}")),
2282            extra: vec![extra],
2283            ..Parameter::default()
2284          }
2285          .init()?,
2286        );
2287      } else if tokens.len() == 1 && tokens.front().unwrap().get_catcode() == Catcode::PARAM {
2288        // Special case: trailing sole # => delimited by next opening brace.
2289        tokens.pop_front();
2290        params.push(Parameter::new("UntilBrace", "UntilBrace", None)?);
2291      } else {
2292        // Nothing? Just a plain parameter.
2293        params.push(Parameter::new("Plain", "{}", None)?);
2294      }
2295    } else {
2296      // Initial delimiting text is required.
2297      let mut lit: Vec<Token> = vec![t];
2298      while !tokens.is_empty() {
2299        let lit_cc = tokens.front().unwrap().get_catcode();
2300        if lit_cc == Catcode::PARAM || lit_cc == Catcode::ARG {
2301          break;
2302        }
2303        lit.push(tokens.pop_front().unwrap());
2304      }
2305      let expected = Tokens::new(lit);
2306      params.push(
2307        Parameter {
2308          name: pin!("Match"),
2309          spec: pin(s!("Match:{expected}")),
2310          extra: vec![expected],
2311          novalue: true,
2312          ..Parameter::default()
2313        }
2314        .init()?,
2315      );
2316    }
2317  }
2318  // return (@params ? LaTeXML::Core::Parameters->new(@params) : undef);
2319  if params.is_empty() {
2320    Ok(None)
2321  } else {
2322    Ok(Some(Parameters::new(params)))
2323  }
2324}

2383/// Perl-faithful body-digest for VBoxContents / HBoxContents parameters.
2384/// Mirrors `readBoxContents($gullet, $everybox, $mode)` exactly:
2385/// pushes a fresh frame for `$mode`, digests tokens until matching T_END,
2386/// pops the frame.
2387///
2388/// The mode-aware variant matters when `\vtop` is invoked inside a `p{}`
2389/// column body (alignment is in horizontal mode, but the `\vtop`'s VBox
2390/// content must be digested in internal_vertical mode so `\@startpbox`'s
2391/// `\vtop\bgroup ...` doesn't try to close groups in the wrong mode).
2392/// Witness: 2210.13325 `\begin{tabular}{|p{1cm}|...}` — pre-fix Rust
2393/// inherited the surrounding horizontal mode and emitted `\vtop`
2394/// errors; Perl uses 'internal_vertical' here regardless of where the
2395/// `\vtop` was invoked.
2396pub fn predigest_box_contents_in_mode(_tokens: ArgWrap, mode: &str) -> Result<Option<Digested>> {
2397  // Perl: readBoxContents calls beginMode($mode) / endMode($mode) around the body reading.
2398  // This creates a scoped frame where enterHorizontal can change MODE inplace.
2399  // When endMode is called, leaveHorizontal_internal detects MODE='horizontal' with
2400  // BOUND_MODE ending in 'vertical', triggers repackHorizontal, then pops the frame.
2401  //
2402  // NOTE: read_box_contents already consumed the opening { or \bgroup via defined_as(T_BEGIN).
2403  // invoke_token(T_BEGIN) pushes a synthetic group frame. The matching } or \egroup
2404  // in the content will pop this frame, since \egroup is \let to T_END and
2405  // invoke_token handles it via the standard group-closing mechanism.
2406  // Perl: $stomach->beginMode($mode) — push a new frame for this box content scope
2407  // VERTICAL box modes (`\vbox`/`\vtop`/p{}-cell VBoxContents): faithful port of
2408  // Perl `readBoxContents` (TeX_Box.pool.ltxml L139-160). ONE mode frame as the
2409  // group; a loop that STOPS at the matching `}` WITHOUT processing it, so
2410  // `end_mode` runs `leave_horizontal_internal` (repack — capturing the *inner*
2411  // `\hsize`) BEFORE `pop_stack_frame` restores it. The `invoke_token(&T_BEGIN!())`
2412  // path used for other modes lets the body's `}` egroup-restore `\hsize` before
2413  // the post-hoc repack, so a `\vbox{\hsize=W …para…}` would wrap at the OUTER
2414  // `\hsize` (the Cluster G `\narrow`-hang; rotated p{}-cell mis-sizing —
2415  // SYNC_STATUS 1610.00974 step-3). hbox/math don't paragraph-wrap → the simpler
2416  // `invoke_token(T_BEGIN)` path below. (A tabular inside a `\vbox`/`\vtop` is
2417  // correctly SKIPPED by the repack because `\@@tabular`/`\halign` mark the result
2418  // box `internal_vertical`.)
2419  if mode.ends_with("vertical") {
2420    begin_mode(mode)?;
2421    let level = get_frame_depth(); // depth AFTER the mode frame (Perl: $level)
2422    new_local_box_list(); // Perl: local @LaTeXML::LIST = ()
2423    loop {
2424      let next = match get_pending_comment() {
2425        Some(comment) => Some(comment),
2426        None => read_x_token(Some(true), false, None)?,
2427      };
2428      let Some(token) = next else { break };
2429      // Perl: last if T_END && (level >= getFrameDepth). The box's own `}` (no
2430      // nested group open → depth == level) ends the loop and is closed by
2431      // `end_mode`, NOT egroup; a nested group's `}` (depth > level) is processed.
2432      if token.defined_as(&T_END!()) && level >= get_frame_depth() {
2433        break;
2434      }
2435      check_timeout()?; // runaway guard (mirrors the canonical group-digest loop)
2436      extend_box_list(invoke_token(&token)?);
2437    }
2438    // Perl: $stomach->endMode($mode) — leave_horizontal_internal (repack with the
2439    // still-in-scope inner `\hsize`) THEN pop_stack_frame (restores `\hsize`).
2440    end_mode(mode)?;
2441    let mode_tex = if lookup_bool_sym(pin!("IN_MATH")) {
2442      TexMode::Math
2443    } else {
2444      TexMode::Text
2445    };
2446    let mut digested_list = List::new(expire_local_box_list());
2447    digested_list.mode = Some(mode_tex);
2448    let mut item: Digested = digested_list.into();
2449    // Perl: List(@LaTeXML::LIST, mode => $mode)
2450    item.set_property("mode", Stored::String(pin(mode)));
2451    return Ok(Some(simplify_vertical_list(item)));
2452  }
2453  if mode.ends_with("vertical") || mode.ends_with("horizontal") {
2454    begin_mode(mode)?;
2455  }
2456  let mut contents = invoke_token(&T_BEGIN!())?;
2457  if contents.is_empty() {
2458    // Perl: $stomach->endMode($mode)
2459    if mode.ends_with("vertical") || mode.ends_with("horizontal") {
2460      end_mode(mode)?;
2461    }
2462    Ok(None)
2463  } else {
2464    let mut item = contents.remove(0);
2465    // Perl's endMode triggers leaveHorizontal_internal → repackHorizontal
2466    // when enterHorizontal changed MODE to 'horizontal' inplace within this frame.
2467    // Check the condition BEFORE endMode pops the frame.
2468    let post_mode = lookup_string_from_sym(pin!("MODE"));
2469    let bound_mode = lookup_string_from_sym(pin!("BOUND_MODE"));
2470    if post_mode == "horizontal"
2471      && bound_mode.ends_with("vertical")
2472      && has_only_simple_horizontal_content(&item)
2473    {
2474      repack_horizontal_in_list(&mut item);
2475      // Restore MODE like leave_horizontal_internal does
2476      assign_value_inplace_sym(pin!("MODE"), pin(&bound_mode));
2477    }
2478    // Perl: $stomach->endMode($mode) — pop the frame
2479    if mode.ends_with("vertical") || mode.ends_with("horizontal") {
2480      end_mode(mode)?;
2481    }
2482    // Set the mode property on the resulting item (matching Perl's List(@boxes, mode => $mode))
2483    if !mode.is_empty() {
2484      item.set_property("mode", Stored::String(pin(mode)));
2485    }
2486    // Apply Perl's List() single-item simplification for vertical modes.
2487    // In Perl, List(@boxes, mode=>'internal_vertical') returns the single box
2488    // directly when @boxes has 1 element and the box's mode is also vertical.
2489    // This is critical for nested \vbox/\vtop: the inner box's `is_vbox` property
2490    // must be visible to the outer box's constructor.
2491    Ok(Some(simplify_vertical_list(item)))
2492  }
2493}

2495/// Check if a List contains only simple horizontal content (TBoxes,
2496/// Comments, or sub-Lists whose mode is horizontal/restricted_horizontal/
2497/// math). This guards against repack being triggered for cases like
2498/// `\vtop{\begin{tabular}...}` where the tabular processing leaks
2499/// MODE='horizontal' but the content (an Alignment/Whatsit) should NOT
2500/// be paragraph-wrapped.

2502/// Inline brace-groups `{...}` inside running text produce sub-Lists
2503/// (one per group) whose mode is `restricted_horizontal`. Without
2504/// accepting those, e.g. `\vbox{\small\bfseries hello {,} world}` would
2505/// fail the repack gate and be measured as 3 separate vertical lines.
2506/// Witness: aistats2026.sty's `\def\And{\unskip{,}\enspace}` in the
2507/// `\@runningauthor` body — every author separator emits a `{,}` group
2508/// that fragments the vbox into many short rows, making `\ht\autrun`
2509/// far exceed 10pt and triggering the class's `\PackageError{Document}
2510/// {Running heading author exceeds size limitations}` (driver paper:
2511/// arXiv:2602.11863).

2513  if let DigestedData::List(l) = item.data() {

2521      DigestedData::TBox(_) | DigestedData::Comment(_) => true,
2522      DigestedData::List(sub) => {

2524        // a sub-List with no `mode` property (implicit hbox). Accept
2525        // those plus sub-Lists explicitly tagged as horizontal-flavour.
2526        // Reject `mode=vertical|internal_vertical` (structural
2527        // sub-vboxes) and any other tagged mode.
2528        let sub_mode = sub
2529          .borrow()
2530          .properties
2531          .get("mode")
2532          .map(|v| v.to_string())
2533          .unwrap_or_default();

2535          sub_mode.as_str(),
2536          "" | "horizontal" | "restricted_horizontal" | "math"
2537        )
2538      },
2539      _ => false,
2540    })
2541  } else {

2543  }

2546/// Replicates Perl's repackHorizontal() within a List's children.
2547///
2548/// Perl (Stomach.pm lines 442-456): In readBoxContents, after digesting box content
2549/// in vertical mode, repackHorizontal groups consecutive horizontal-mode items
2550/// from @LaTeXML::LIST into a single List(@para, mode => 'horizontal') with
2551/// width set to \hsize. This enables compute_boxes_size to do paragraph wrapping.
2552///
2553/// Without this, \vbox{hop} measures each character individually (width=5.55pt)
2554/// instead of wrapping as a paragraph at \hsize (width=469.75pt).
2555fn repack_horizontal_in_list(item: &mut Digested) {
2556  if let DigestedData::List(l) = item.data() {
2557    let mut list = l.borrow_mut();
2558    let children = std::mem::take(&mut list.boxes);
2559    let mut result: Vec<Digested> = Vec::new();
2560    let mut para: Vec<Digested> = Vec::new();
2561    let mut keep = false;
2562
2563    for child in children {
2564      let child_mode = child
2565        .get_property("mode")
2566        .map(|v| v.to_string())
2567        .unwrap_or_else(|| "horizontal".to_string());
2568      // Empty-string mode means "implicit hbox" — produced by inline
2569      // brace-groups `{...}` in horizontal context. Treat as horizontal
2570      // so the surrounding running text doesn't get fragmented.
2571      let effective_mode = if child_mode.is_empty() {
2572        "horizontal"
2573      } else {
2574        child_mode.as_str()
2575      };
2576      if effective_mode == "horizontal"
2577        || effective_mode == "restricted_horizontal"
2578        || effective_mode == "math"
2579      {
2580        // Perl: $keep = 1 if ($mode ne 'horizontal') || !$item->getProperty('isSpace');
2581        if effective_mode != "horizontal" || !child.get_property_bool("isSpace") {
2582          keep = true;
2583        }
2584        para.push(child);
2585      } else {
2586        // Flush accumulated horizontal items as a horizontal List
2587        if keep {
2588          let horiz_list = make_horizontal_list(std::mem::take(&mut para));
2589          result.push(Digested::from(horiz_list));
2590        } else {
2591          result.extend(std::mem::take(&mut para));
2592        }
2593        keep = false;
2594        result.push(child);
2595      }
2596    }
2597    // Flush remaining horizontal items
2598    if keep {
2599      let horiz_list = make_horizontal_list(para);
2600      result.push(Digested::from(horiz_list));
2601    } else {
2602      result.extend(para);
2603    }
2604    list.boxes = result;
2605  }
2606}

2608/// Create a horizontal List with mode='horizontal' and width=\hsize.
2609/// Perl: push(@LaTeXML::LIST, List(@para, mode => 'horizontal')) if $keep;

3159          .collect();
3160        for m in cell_kids {

3163            //  — Math wraps an XMath, XMath wraps the actual tokens. Pull

3166              .get_child_nodes()
3167              .into_iter()
3168              .flat_map(|x| x.get_child_nodes())

3171          } else {

3174        }
3175      }
3176    }

3181    }
3182    // Unwrap the XMText (its only child is now `table`).
3183    document.unwrap_nodes(text_node)?;

3186      // Re-record the id on the renamed table (and any nested ids).

3216/// Does the remaining `stream` contain a `)` that balances the `(` just
3217/// popped? Used by `split_tokens` to protect delimiters inside *balanced*
3218/// parens only — an unbalanced `(` must NOT trigger paren-protection (else it
3219/// would greedily swallow the rest of the author block, including `\\`
3220/// name/affiliation separators).
3221fn paren_closes_ahead(stream: &VecDeque<Token>) -> bool {
3222  let mut level = 1usize;
3223  for t in stream {
3224    if *t == T_OTHER!("(") {
3225      level += 1;
3226    } else if *t == T_OTHER!(")") {
3227      level -= 1;
3228      if level == 0 {
3229        return true;
3230      }
3231    }
3232  }
3233  false
3234}

3236/// Perl: SplitTokens($tokens, @delims) — Base_Utility.pool.ltxml (PR #2767).
3237/// Each of `delims` is a Token, OR Tokens to match a sequence.
3238/// Returns a list of Tokens for the sub-sequences, with leading/trailing
3239/// spaces trimmed from each piece, and any empty trailing piece dropped.
3240///
3241/// Like Perl, delimiters are not matched inside `{…}` braces or `$…$` math.
3242/// As an INTENTIONAL DIVERGENCE FROM PERL, they are also not matched inside
3243/// balanced `(…)` parentheses (see the paren branch below + OXIDIZED_DESIGN).
3244pub fn split_tokens(tokens: Tokens, delims: Vec<SplitDelim>) -> Vec<Tokens> {
3245  let mut items: Vec<Tokens> = Vec::new();
3246  let mut toks: Vec<Token> = Vec::new();
3247  let trim_spaces = |toks: &mut Vec<Token>| {
3248    while toks.first().is_some_and(|x| *x == T_SPACE!()) {
3249      toks.remove(0);
3250    }
3251    while toks.last().is_some_and(|x| *x == T_SPACE!()) {
3252      toks.pop();
3253    }
3254  };
3255  if !tokens.is_empty() {
3256    let mut stream: VecDeque<Token> = VecDeque::from(tokens.unlist());
3257    while let Some(t) = stream.pop_front() {
3258      let mut matched = false;
3259      for delim in &delims {
3260        match delim {
3261          SplitDelim::Token(d) => {
3262            // Perl: Equals($t, $delim); the Rust port additionally matches by
3263            // meaning, so \AND (let to \and) matches \and as delimiter.
3264            if *d == t
3265              || (t.get_catcode() == Catcode::CS && d.get_catcode() == Catcode::CS && {
3266                let meaning_t = lookup_definition(&t).ok().flatten();
3267                let meaning_d = lookup_definition(d).ok().flatten();
3268                meaning_t.is_some() && meaning_t == meaning_d
3269              })
3270            {
3271              matched = true;
3272              break;
3273            }
3274          },
3275          SplitDelim::Tokens(seq) => {
3276            let mut tomatch: &[Token] = seq.unlist_ref();
3277            let mut peeked: Vec<Token> = Vec::new(); // tokens consumed beyond `t`
3278            let mut cur: Option<Token> = Some(t);
3279            while let (Some(c), Some(m)) = (cur, tomatch.first()) {
3280              if c == *m || (*m == T_SPACE!() && c == T_CS!("\\ ")) {
3281                // HACK space!
3282                tomatch = &tomatch[1..];
3283                if !tomatch.is_empty() {
3284                  cur = stream.pop_front();
3285                  if let Some(p) = cur {
3286                    peeked.push(p);
3287                  }
3288                }
3289              } else {
3290                break;
3291              }
3292            }
3293            if tomatch.is_empty() {
3294              matched = true;
3295              break;
3296            } else {
3297              // failed to match all: put back the peeked tokens
3298              for p in peeked.into_iter().rev() {
3299                stream.push_front(p);
3300              }
3301            }
3302          },
3303        }
3304      }
3305      if matched {
3306        trim_spaces(&mut toks);
3307        items.push(Tokens::new(std::mem::take(&mut toks)));
3308      } else if t.defined_as(&T_BEGIN!()) {
3309        toks.push(t);
3310        let mut level = 1;
3311        while let Some(t) = stream.pop_front() {
3312          match t.get_catcode() {
3313            Catcode::BEGIN => level += 1,
3314            Catcode::END => level -= 1,
3315            _ => {},
3316          }
3317          toks.push(t);
3318          if level < 1 {
3319            // done if balanced.
3320            break;
3321          }
3322        }
3323      } else if t.defined_as(&T_MATH!()) {
3324        toks.push(t);
3325        while let Some(t) = stream.pop_front() {
3326          let is_math = t.get_catcode() == Catcode::MATH;
3327          toks.push(t);
3328          if is_math {
3329            break;
3330          }
3331        }
3332      } else if t == T_OTHER!("(") && paren_closes_ahead(&stream) {
3333        // INTENTIONAL DIVERGENCE FROM PERL: also protect delimiters inside
3334        // BALANCED parentheses (mirroring the brace/math skipping above), so a
3335        // parenthesized affiliation like "(Scuola Normale Superiore, Pisa)" is
3336        // NOT split at its internal comma into a spurious second author. Perl's
3337        // SplitTokens has no paren-awareness and makes exactly this mistake
3338        // (witness arXiv 0804.0870 — "Pisa)" became a second <personname>).
3339        // The `paren_closes_ahead` guard means an UNBALANCED `(` (rare/
3340        // malformed) is treated as an ordinary token, so it never swallows a
3341        // later `\\` name/affiliation separator. See OXIDIZED_DESIGN
3342        // "Intentional divergences". NOTE: bare (unparenthesized) commas in an
3343        // affiliation ("MIT, Cambridge") remain genuinely ambiguous — the same
3344        // tokens read as either one comma-affiliation or two authors — so we
3345        // match Perl's recall-oriented over-split there rather than guess.
3346        toks.push(t);
3347        let mut level = 1;
3348        while let Some(t) = stream.pop_front() {
3349          if t == T_OTHER!("(") {
3350            level += 1;
3351          } else if t == T_OTHER!(")") {
3352            level -= 1;
3353          }
3354          toks.push(t);
3355          if level < 1 {
3356            break;
3357          }
3358        }
3359      } else {
3360        toks.push(t);
3361      }
3362    }
3363  }
3364  trim_spaces(&mut toks);
3365  if !toks.is_empty() {
3366    items.push(Tokens::new(toks));
3367  }
3368  items
3369}

3371/// If `tokens` — ignoring leading/trailing spaces — is exactly a single control
3372/// sequence applied to one brace group that spans to the end (`\cmd{ … }`),
3373/// return `(\cmd, inner)`. Used to see through a whole-line font wrapper when
3374/// splitting an author line.
3375fn whole_line_cs_wrapper(tokens: &Tokens) -> Option<(Token, Tokens)> {
3376  let v = tokens.unlist_ref();
3377  let mut start = 0;
3378  let mut end = v.len();
3379  while start < end && v[start] == T_SPACE!() {
3380    start += 1;
3381  }
3382  while end > start && v[end - 1] == T_SPACE!() {
3383    end -= 1;
3384  }
3385  let s = &v[start..end];
3386  // Need at least `\cmd { }` (the group may be empty, but then there is nothing
3387  // to split, so the caller's `len > 1` guard makes it a no-op anyway).
3388  if s.len() < 3 || s[0].get_catcode() != Catcode::CS || s[1].get_catcode() != Catcode::BEGIN {
3389    return None;
3390  }
3391  let mut level = 0;
3392  for (i, t) in s.iter().enumerate().skip(1) {
3393    match t.get_catcode() {
3394      Catcode::BEGIN => level += 1,
3395      Catcode::END => {
3396        level -= 1;
3397        if level == 0 {
3398          // The group must close exactly on the last token for this to be a
3399          // whole-line wrapper (no trailing content after `}`).
3400          return (i == s.len() - 1).then(|| (s[0], Tokens::new(s[2..i].to_vec())));
3401        }
3402      },
3403      _ => {},
3404    }
3405  }
3406  None
3407}

3409/// Second-level split of the author heuristic: divide ONE line that has already
3410/// been classified as an *author* line (its superscript marker sits far enough
3411/// from the front — `Some(p)` with `p >= 8` in `\lx@add@authors`) into the
3412/// individual creators it names.
3413///
3414/// This is guarded for affiliations by that upstream classification: an
3415/// *affiliation* line (marker near the front, `p < 8`) or a continuation line
3416/// (no marker) never reaches here — they take the other arms — so name-level
3417/// separators are only ever applied to author text, never to institution names.
3418/// That is why the literal word " and " is a separator here (it splits "Alice
3419/// and Bob") yet is deliberately absent from the line-level `author_affil_splits`
3420/// (where it would shred "Language and Intelligence").
3421///
3422/// The split proceeds by hierarchy:
3423///  1. the name separators ("," and " and ") at top level; if that already
3424///     yields more than one name we are done;
3425///  2. otherwise, if the whole line is a single font wrapper `\cmd{ a, b, c }`
3426///     (Perl's `SplitTokens` can't see the brace-hidden separators, so the
3427///     wrapper collapses into ONE creator that then hoards every `$^n$` marker
3428///     as a duplicate affiliation — arXiv 2605.00347), descend through the
3429///     wrapper, split its inner name list, and re-apply `\cmd` to each name so
3430///     every author becomes its own creator with the correct affiliation.
3431fn split_author_line(line: Tokens) -> Vec<Tokens> {
3432  // Name-level separators: comma and the literal word " and " ("Alice and Bob").
3433  let name_seps = || vec![SplitDelim::Token(T_OTHER!(",")), literal_and()];
3434  let top = split_tokens(line.clone(), name_seps());
3435  if top.len() > 1 {
3436    return top;
3437  }
3438  if let Some((cmd, inner)) = whole_line_cs_wrapper(&line) {
3439    let inner_split = split_tokens(inner, name_seps());
3440    if inner_split.len() > 1 {
3441      return inner_split
3442        .into_iter()
3443        .map(|piece| {
3444          let mut v = vec![cmd, T_BEGIN!()];
3445          v.extend(piece.unlist());
3446          v.push(T_END!());
3447          Tokens::new(v)
3448        })
3449        .collect();
3450    }
3451  }
3452  top
3453}

3584/// Support for Key / Value arguments.
3585// The very basic form is
3586//   RequiredKeyVals: $keyset
3587//   OptionalKeyVals: $keyset
3588// to parse Key-Value pairs from a given keyset (see the 'keyval' package
3589// documentation for more information). These types of KeyVal
3590// parameters will return a LaTeXML::Core::KeyVals object, which can then be
3591// used to access the values of the individual items.
3592// The difference between the two forms is that RequiredKeyVals expects a set of
3593// key-value pairs wrapped in T_BEGIN T_END, where as OptionalKeyVals optionally
3594// expects a set of KeyValue pairs wrapped in T_OTHER('[') T_OTHER(']')
3595//
3596// Several extension of the keyval package exist, the most common one we support
3597// is the xkeyval package. This introduces further variations on the keyval
3598// arguments parsing, in particular it allows to read keys from more than one
3599// keyset at once. These can be specified by giving comma-separated values in
3600// the keyset argument. By default, a key will only be set in the **first**
3601// keyset it occurs in. By using
3602//   RequiredKeyVals+: $keysets
3603//   OptionalKeyVals+: $keysets
3604// the key will be set in all keysets instead.
3605//
3606// All keys to be parsed with these arguments should be declared using
3607// DefKeyVal in LaTeXML::Package. By default, an error is thrown if an unknown
3608// key is encountered. To surpress this behaviour, and instead store all
3609// undefined keys, use
3610//   RequiredKeyVals*: $keysets
3611//   OptionalKeyVals*: $keysets
3612// instead. The '*' and '+' modifiers can be combined by using:
3613//   RequiredKeyVals*+: $keysets
3614//   OptionalKeyVals*+: $keysets
3615//
3616// Furthermore, the xkeyval package supports giving prefixes to keys,
3617//   RequiredKeyVals[*][+]: $prefix|$keysets
3618//   OptionalKeyVals[*][+]: $prefix|$keysets
3619//
3620// Finally, it is possible to specify specific keys to skip when digesting the
3621// object. This can be achieved using comma-separated key values in
3622//   RequiredKeyVals[*][+]: $prefix|$keysets|$skip
3623//   OptionalKeyVals[*][+]: $prefix|$keysets|$skip
3624
3625// function to handle all the
3626#[derive(Default)]
3627pub struct KVSpec {
3628  pub star:    bool,
3629  pub plus:    bool,
3630  pub prefix:  Option<String>,
3631  pub keysets: Vec<String>,
3632  pub skip:    Vec<String>,
3633}

3635  let KVSpec {
3636    mut star,

3640    skip,
3641  } = spec;
3642  // support both "keysets" and "prefix|keysets"
3643  if keysets.is_empty() {

3775pub fn make_generic_message(cmd: &str, args: Vec<Tokens>, kind: &str) -> Result<()> {
3776  // Faithful port of Perl latex_constructs.pool.ltxml `make_message`
3777  // (L5569-5588). The first arg is the message PREFIX: its ToString,
3778  // stripped of `\@spaces`/`\space` runs, becomes the diagnostic `what`
3779  // ($type), falling back to the command name when the prefix is empty
3780  // (e.g. `\@latex@error`'s all-`\space` prefix). The REMAINING args form
3781  // the body — Perl `join(" ", map { ToString(Expand($_, \@spaces,
3782  // \@spaces)) } @args)`. Earlier the Rust port discarded the prefix and
3783  // pinned `what` to the command name, so package/class diagnostics lost
3784  // their `(pkgname)` $type and `\GenericWarning`/`\GenericInfo` even
3785  // leaked the prefix into the body. Witness: hep-th-style `\PackageError`
3786  // /`\GenericWarning` now match Perl byte-for-byte (`(mypkg)` $type, no
3787  // prefix in body).
3788  let mut args = args.into_iter();
3789  let lead = args.next().unwrap_or_default();
3790  // $type = ToString(lead) with `\@?spaces?` runs removed (Perl regex
3791  // `s/(?:\\\@?spaces?)+//g`); fall back to the command name when empty.
3792  let type_str = strip_message_spaces(&lead.to_string());
3793  let type_str: &str = if type_str.is_empty() { cmd } else { &type_str };
3794
3795  bgroup();
3796  let_i(&T_CS!("\\protect"), &T_CS!("\\string"), None);
3797  let_i(
3798    &T_CS!("\\MessageBreak"),
3799    &T_CS!("\\ltx@hard@MessageBreak"),
3800    None,
3801  ); // tricky, we need Expand() to execute it
3802  let mut parts: Vec<String> = Vec::new();
3803  for arg in args {
3804    let mut arg_toks = arg.unlist();
3805    // Perl `make_message` pads each Expand input with two `\@spaces` tokens:
3806    //   ToString(Expand($_, T_CS('\@spaces'), T_CS('\@spaces')))
3807    // Comment in Perl: "expand padded with two \@spaces, to avoid pointless
3808    // errors. e.g. a trailing `\csq@noline` from csquotes.sty (let to
3809    // `\@gobble`) will produce a `gobble has no argument` error on every
3810    // message." Same fix needed here for etoc.sty's
3811    // `\PackageWarning{etoc}{...!\@gobbletwo}` — `\@gobbletwo` would otherwise
3812    // consume the do_expand-inserted T_END closer, leaving readBalanced
3813    // unbalanced.
3814    arg_toks.push(T_CS!("\\@spaces"));
3815    arg_toks.push(T_CS!("\\@spaces"));
3816    parts.push(Expand!(arg_toks).to_string());
3817  }
3818  // Perl joins the body args with a single space (the `\MessageBreak`s
3819  // *within* an arg already became hard newlines via the let above).
3820  let message = parts.join(" ");
3821
3822  egroup()?;
3823  // Downgrade vendor-class typesetting-only errors to Info. Publisher
3824  // classes routinely guard line widths, header heights, and other
3825  // PDF-layout concerns with `\PackageError`/`\GenericError`. We
3826  // produce XML/HTML, not PDF — these guards have no semantic value
3827  // in our output. See WISDOM #50 and
3828  // memory/feedback_size_layout_errors_moot.md.
3829  let effective_kind = if kind == "error" && is_typesetting_only_message(&message) {
3830    "info"
3831  } else {
3832    kind
3833  };
3834  //   return ('latex', $type, $stomach, $message);
3835  match effective_kind {
3836    "error" => {
3837      Error!("latex", type_str, message);
3838    },
3839    "warn" => {
3840      Warn!("latex", type_str, message);
3841    },
3842    "info" => {
3843      Info!("latex", type_str, message);
3844    },
3845    _other => panic!("Only call make_generic_message with error|warn|info message kinds."),
3846  };
3847  Ok(())
3848}

3850/// Heuristic classifier for vendor `\PackageError`/`\GenericError`
3851/// messages whose only concern is PDF typesetting (size, layout,
3852/// position, page-fit). These have no signal in XML/HTML output and
3853/// are downgraded to `Info:` per WISDOM #50.
3854fn is_typesetting_only_message(message: &str) -> bool {
3855  let lower = message.to_ascii_lowercase();
3856  // Phrase set tuned against the stage-1 sweep of the 100k warning
3857  // corpus. Conservative — every phrase here is purely about visual
3858  // layout or vendor-deprecation chatter, never about semantic
3859  // correctness. Examples:
3860  //   "Running heading author exceeds size limitations" (AISTATS)
3861  //   "Running heading title exceeds size limitations" (AISTATS)
3862  //   "Caption too wide for page" (various)
3863  //   "Heading breaks the line" (revtex, IEEEtran)
3864  //   "You are loading directly a language style" (babel: czech.sty,
3865  //     francais.sty, etc. unconditionally fire `\PackageError` to nag
3866  //     the user toward `\usepackage[<lang>]{babel}`; pdflatex shows
3867  //     the message but continues, and the document typesets normally
3868  //     — the message is informational, not a real failure)
3869  const PHRASES: &[&str] = &[
3870    "exceeds size limitations",
3871    "exceeds size limitation",
3872    "running heading",
3873    "running title",
3874    "running author",
3875    "breaks the line",
3876    "too wide for",
3877    "too tall for",
3878    "too long for",
3879    "too narrow for",
3880    "doesn't fit",
3881    "does not fit",
3882    "page overflow",
3883    "column overflow",
3884    "exceeds the page",
3885    "exceeds the column",
3886    "exceeds the line",
3887    "exceeds the textwidth",
3888    "exceeds \\textwidth",
3889    "exceeds \\columnwidth",
3890    "exceeds \\linewidth",
3891    "loading directly a language style",
3892    "syntax is deprecated",
3893    // babel TL2025: legacy `<lang>.ldf` files have been retired in
3894    // favour of `locale/<iso>/babel-<lang>.tex` (the ini-file
3895    // system). For papers that load `\usepackage[<lang>]{babel}`
3896    // without `provide=*`, babel fires:
3897    //   Package babel Error: Unknown option '<lang>'.
3898    //   Either you misspelled it or the language definition file
3899    //   <lang>.ldf was not found
3900    // The .ldf-missing case is benign: pdflatex shows the message
3901    // and proceeds — the document still typesets, just without
3902    // the language's captions/shorthands loaded. Same effective
3903    // outcome on our side: downgrade to Info, conversion continues.
3904    // Surpass-Perl: Perl raw-loads babel.sty and errors identically
3905    // on the same 58 papers in Round-27 Cluster D. This downgrade
3906    // closes the cluster (cannot reach 0-error AND load the proper
3907    // ini file without redesigning babel option processing).
3908    "either you misspelled it",
3909    // catoptions.sty (loaded transitively by many class/sty bundles)
3910    // calls `\@latex@error{Command \protect\\special_relax already
3911    // defined...}` because it `\def\special_relax{...}` and our engine
3912    // pre-registers `\special_relax` as an internal Gullet helper.
3913    // pdflatex shows the message and proceeds (cat's def takes over);
3914    // surpass-Perl: Perl also raw-loads catoptions and errors. The
3915    // resulting "Command ... already defined" cascade currently
3916    // produces 100+ errors per paper on 14 wp5 papers. Downgrade so
3917    // conversion continues. Same applies to "Command \end... illegal,
3918    // see p.192 of the manual" tail that LaTeX appends to the same
3919    // message ("\@latex@error" generic-error template).
3920    "already defined. or name",
3921    "command \\end... illegal",
3922    // amsfonts-not-installed: aims-class L: `\@latex@error{Package
3923    // `amsfonts' not installed, or version too old?}`. The class ships
3924    // its own font tables and only loads amsfonts as an enhancement;
3925    // pdflatex/Perl LaTeXML both proceed to typeset successfully when
3926    // amsfonts is missing. Our raw-load reports the message verbatim
3927    // but conversion is fine without the AMS msam/msbm fonts. Witness
3928    // 2202.13120.
3929    "amsfonts package will not be loaded",
3930    "package `amsfonts'",
3931    "package 'amsfonts'",
3932    // hrefhide-format-too-old: hrefhide.sty requires LaTeX format
3933    // 2022-11-01+; older formats trigger an error. The package is
3934    // strictly visual (hides hyperref colors on print), so the error
3935    // is moot for XML/HTML output. Witness 2202.03936.
3936    "newer latex format needed",
3937    "older hrefhide package",
3938  ];
3939  PHRASES.iter().any(|p| lower.contains(p))
3940}
