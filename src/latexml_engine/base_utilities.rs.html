26  //    so let's get that squared away at the outset; it's useful for TeX, too!

28  // Naturally, it uses \csname to check, which ends up DEFINING the possibly undefined macro as

31  DefMacro!("\\lx@ifundefined{}{}{}", sub[(name, if_token, else_token)] {
32    let cs = T_CS!(s!("\\{}", Expand!(name).to_string()));
33    // Autoload triggers (declared via `def_autoload` at tex.rs:238-247)
34    // install a closure under the trigger CS so the package auto-loads on
35    // first invocation. Perl scopes the equivalent `DefAutoload` entries
36    // to `OmniBus.cls.ltxml`, so for non-OmniBus papers Perl sees these
37    // CSes as truly undefined. Mirror Perl by treating an unfired
38    // autoload trigger as "undefined" in `\@ifundefined`. We must NOT
39    // overwrite the trigger CS with `\relax` in this case (the kernel
40    // `\csname X\endcsname \ifx \relax` idiom DOES overwrite, but doing
41    // so here would destroy the autoload — subsequent use of the trigger
42    // CS would no-op instead of loading its package). Driver:
43    // arXiv:2507.23241v1 (smfart.cls) — line 373's
44    // `\@ifundefined{numberwithin}` branches wrong when our preloaded
45    // `\numberwithin` autoload makes the CS "look" defined, then the
46    // `\@gobbletwo` branch eats `\ifx \relax` and orphans `\else` / `\fi`.
47    let is_autoload = cs.with_cs_name(|cs_name| {
48      state::lookup_bool(&s!("{cs_name}:autoload"))
49    });
50    if IsDefined!(&cs) && !is_autoload {
51      Ok(else_token)
52    } else {
53      if !is_autoload {
54        state::assign_meaning(&cs, state::lookup_meaning(&TOKEN_RELAX), None);  // Let w/o AfterAssign
55      }
56      Ok(if_token)
57    }
58  }, locked=>true);

178      inv_tokens.push(T_OTHER!("]"));

185      inv_tokens.push(T_OTHER!("["));

190    inv_tokens.extend(tokens.unlist());

196  // keys can have: replace (to replace the current entry, if any),
197  //                ifnew (only add if no previous entry)
198  DefPrimitive!("\\@add@frontmatter@now OptionalKeyVals {} OptionalKeyVals {}",
199    sub[(keys_opt,tag_tks,attrs_opt,tokens)] {
200    // Digest this as if we're already in the document body!
201    let inpreamble = state::lookup_bool_sym(pin!("inPreamble"));
202    assign_value("inPreamble", false, None);
203    // Be careful since the contents may also want to add frontmatter
204    // (which should be inside or after this one!)
205    // So, we append this entry before digesting (Perl comment from Base_Utility.pool.ltxml)
206    let tag = tag_tks.to_string();
207    if let Some(keys) = keys_opt {
208      let known_key = state::with_value("frontmatter", |v| {
209        matches!(v, Some(Stored::HashTagData(frnt)) if frnt.contains_key(&tag))
210      });
211      if known_key && keys.has_key("replace") {
212        state::with_value_mut("frontmatter", |val_opt| {
213          if let Some(&mut Stored::HashTagData(ref mut frnt)) = val_opt {
214            frnt.insert(tag.clone(), Vec::new());
215          }
216        });
217      }
218      if known_key && keys.has_key("ifnew") {
219        return Ok(Vec::new());
220      }
221    }
222
223    let attrs_digested = if let Some(attr_kvs) = attrs_opt {
224      if let DigestedData::KeyVals(digested) = attr_kvs.be_digested()?.data() {
225        Some(digested.get_hash_digested())
226      } else {
227        None
228      }
229    } else {
230      None
231    };
232    // Perl: Create entry FIRST, then digest content (so nested @add@to@ appends to this entry)
233    let placeholder = Digested::from(List::new(Vec::new()));
234    let entry = (tag.clone(), attrs_digested, placeholder);
235    state::with_value_mut("frontmatter", |val_opt| {
236      let frontmatter = match val_opt {
237        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
238        _ => fatal!(TexPool, Expected, "Global TeX Frontmatter hash was not available, should never happen"),
239      };
240      let f_entry = frontmatter.entry(tag.clone()).or_insert_with(Vec::new);
241      f_entry.push(entry);
242      Ok(())
243    })?;
244    // NOW digest the content — nested @add@to@frontmatter calls will see this entry as "last"
245    // Perl Base_Utility.pool.ltxml L204: `DigestText(Tokens($tokens))` — text-mode
246    // digest forces math content (e.g. `$$Id…$$` CVS markers in `\date{}`) to
247    // flatten into plain text instead of producing `<ltx:equation>` whatsits
248    // that wouldn't fit the schema slot (e.g. `<ltx:date>` rejects equations).
249    //
250    // Was: wrapped tokens in T_BEGIN/T_END, opening an extra group inside
251    // DigestText. The extra group can interact with mode-changing tokens
252    // in the body (e.g. \itshape via DefPrimitive setting font, or
253    // \@@@affiliation in IEEEtran multi-author setups) such that the
254    // DigestText's end_mode("text") sees BOUND_MODE shifted by the inner
255    // group's font/mode side effects. Mirror Perl's `DigestText(Tokens($tokens))`
256    // — no wrap. Driver: 2403.14274 IEEEconf abstract+keywords+IEEEpeerreviewmaketitle.
257    let digested_tokens = DigestText!(tokens)?;
258    // Fill in the placeholder
259    state::with_value_mut("frontmatter", |val_opt| {
260      let frontmatter = match val_opt {
261        Some(&mut Stored::HashTagData(ref mut frnt)) => frnt,
262        _ => return Ok::<(), latexml_core::Error>(()),
263      };
264      if let Some(list) = frontmatter.get_mut(&tag) {
265        if let Some(last) = list.last_mut() {
266          last.2 = digested_tokens;
267        }
268      }
269      Ok(())
270    })?;
271    AssignValue!("inPreamble", inpreamble);
272  }, bounded => true);

713  }

717/// Insert FrontMatter into document, if not already added
718pub fn insert_frontmatter(document: &mut Document) -> Result<()> {
719  if lookup_bool("frontmatter_done") {
720    return Ok(());
721  }
722  let frontmatter_elements_set: HashSet<String> = FRONTMATTER_ELEMENTS
723    .iter()
724    .map(ToString::to_string)
725    .collect();
726
727  // Collect the frontmatter hash keys via with_value — we only need the
728  // key set here; the full HashTagData clone previously happened just
729  // to call .keys().cloned().collect(). The hash itself is consumed a
730  // few lines below via remove_value, so no iteration on the borrow
731  // survives past this closure.
732  let set_keys: Vec<String> = state::with_value("frontmatter", |v| match v {
733    Some(Stored::HashTagData(frnt)) => frnt.keys().cloned().collect(),
734    _ => Vec::new(),
735  });
736  if set_keys.is_empty() {
737    return Ok(());
738  }
739
740  // If doc ONLY has abstract as frontmatter, defer until abstract's document location
741  if set_keys.len() == 1 && set_keys[0] == "ltx:abstract" && !lookup_bool("frontmatter_deferred") {
742    state::assign_value("frontmatter_deferred", true, Some(Scope::Global));
743    return Ok(());
744  }
745
746  state::assign_value("frontmatter_done", true, Some(Scope::Global));
747
748  // Remove frontmatter and replace with empty
749  let mut frontmatter = match state::remove_value("frontmatter") {
750    Some(Stored::HashTagData(frnt)) => frnt,
751    _ => return Ok(()),
752  };
753  state::assign_value(
754    "frontmatter",
755    Stored::HashTagData(HashMap::default()),
756    Some(Scope::Global),
757  );
758
759  // Order: first go through frontmatter_elements order, then any custom keys
760  let custom_keys: Vec<String> = frontmatter
761    .keys()
762    .filter(|key| !frontmatter_elements_set.contains(key.as_str()))
763    .map(ToString::to_string)
764    .collect();
765  let mut all_keys: Vec<String> = FRONTMATTER_ELEMENTS
766    .iter()
767    .map(ToString::to_string)
768    .collect();
769  all_keys.extend(custom_keys);
770
771  for key in &all_keys {
772    if let Some(list) = frontmatter.remove(key) {
773      // Dubious, but assures that frontmatter appears in text mode...
774      document.set_box_to_absorb(
775        Tbox::new(
776          pin!(""),
777          lookup_font(),
778          None,
779          Tokens!(T_SPACE!()),
780          SymHashMap::default(),
781        )
782        .into(),
783      );
784      for (tag, attr, stuff) in list {
785        // Add a dedicated class for frontmatter notes
786        let attr = if tag == "ltx:note" {
787          let mut a = attr.unwrap_or_default();
788          let existing = a.get("class").cloned().unwrap_or_default();
789          let new_class = if existing.is_empty() {
790            "ltx_note_frontmatter".to_string()
791          } else {
792            s!("{existing} ltx_note_frontmatter")
793          };
794          a.insert("class".to_string(), new_class);
795          Some(a)
796        } else {
797          attr
798        };
799        document.open_element(&tag, attr, None)?;
800        document.absorb(&stuff, None)?;
801        let completed_node = document.close_element(&tag)?;
802        // Prune empty frontmatter elements (except ltx:rdf)
803        if tag != "ltx:rdf" {
804          if let Some(ref node) = completed_node {
805            if node.get_child_nodes().is_empty() {
806              document.remove_node(node.clone());
807            }
808          }
809        }
810      }
811      document.expire_box_to_absorb();
812    }
813  }
814  Ok(())
815}

885  // conversion alive when something has clobbered the value table.
886  let read = |key: &str, default: i32| -> i32 {
887    match state::lookup_value(key) {
888      Some(Stored::Int(n)) => n as i32,
889      Some(Stored::Number(n)) => n.value_of() as i32,
890      _ => default,
891    }
892  };
893  let m = read("\\month", 1).clamp(1, 12) as usize;
894  let month = month_names[m - 1];
895  let day = read("\\day", 1);
896  let year = read("\\year", 1900);
897  Ok(s!("{} {}, {}", month, day, year))
898}

900pub fn parse_def_parameters(cs: &Token, params_in: Tokens) -> Result<Option<Parameters>> {
901  let mut tokens: VecDeque<Token> = VecDeque::from(params_in.pack_parameters()?.unlist());
902  // Now, recognize parameters and delimiters.
903  let mut params = Vec::new();
904  let mut n = 0;
905  while let Some(mut t) = tokens.pop_front() {
906    let cc = t.get_catcode();
907    if cc == Catcode::PARAM || cc == Catcode::ARG {
908      if cc == Catcode::PARAM {
909        if tokens.is_empty() {
910          // Special case: lone # NOT following a numbered parameter
911          // Note that we require a { to appear next, but do NOT read it!
912          params.push(Parameter::new(
913            Cow::Borrowed("RequireBrace"),
914            Cow::Borrowed("RequireBrace"),
915            None,
916          )?);
917          break;
918        } else {
919          n += 1;
920          if let Some(t_next) = tokens.pop_front() {
921            t = t_next;
922          } else {
923            unreachable!("tokens.is_empty() was false, so pop_front must return Some");
924          }
925        }
926      } else {
927        // CC_ARG case, keep looking at this token
928        n += 1;
929      }
930      if n > 0 {
931        let t_num = t.with_str(|ts| ts.parse::<i8>()).unwrap_or(-1);
932        if t_num != n {
933          fatal!(
934            ParamSpec,
935            Expected,
936            s!(
937              "Parameters for {:?} not in order. Got {:?}, expected {:?}. in {:?}",
938              cs,
939              t,
940              n,
941              params
942            )
943          );
944        }
945      }
946      // Check for delimiting text following the parameter #n
947      let mut delim = Vec::new();
948      let mut pc = Catcode::MARKER; // throwaway initial val
949      while !tokens.is_empty() {
950        let inner_cc = tokens.front().unwrap().get_catcode();
951        if inner_cc == Catcode::PARAM || inner_cc == Catcode::ARG {
952          break;
953        }
954        let d = tokens.pop_front().unwrap();
955        if !(pc == Catcode::SPACE && inner_cc == Catcode::SPACE) {
956          // BUT collapse whitespace!
957          delim.push(d);
958        }
959        pc = inner_cc;
960      }
961      // Found text that marks the end of the parameter
962      if !delim.is_empty() {
963        let extra = Tokens::new(delim);
964        params.push(
965          Parameter {
966            name: arena::pin_static("Until"),
967            spec: arena::pin(format!("Until:{extra}")),
968            extra: vec![extra],
969            ..Parameter::default()
970          }
971          .init()?,
972        );
973      } else if tokens.len() == 1 && tokens.front().unwrap().get_catcode() == Catcode::PARAM {
974        // Special case: trailing sole # => delimited by next opening brace.
975        tokens.pop_front();
976        params.push(Parameter::new("UntilBrace", "UntilBrace", None)?);
977      } else {
978        // Nothing? Just a plain parameter.
979        params.push(Parameter::new("Plain", "{}", None)?);
980      }
981    } else {
982      // Initial delimiting text is required.
983      let mut lit: Vec<Token> = vec![t];
984      while !tokens.is_empty() {
985        let lit_cc = tokens.front().unwrap().get_catcode();
986        if lit_cc == Catcode::PARAM || lit_cc == Catcode::ARG {
987          break;
988        }
989        lit.push(tokens.pop_front().unwrap());
990      }
991      let expected = Tokens::new(lit);
992      params.push(
993        Parameter {
994          name: arena::pin_static("Match"),
995          spec: arena::pin(s!("Match:{expected}")),
996          extra: vec![expected],
997          novalue: true,
998          ..Parameter::default()
999        }
1000        .init()?,
1001      );
1002    }
1003  }
1004  // return (@params ? LaTeXML::Core::Parameters->new(@params) : undef);
1005  if params.is_empty() {
1006    Ok(None)
1007  } else {
1008    Ok(Some(Parameters::new(params)))
1009  }
1010}

1069/// Perl-faithful body-digest for VBoxContents / HBoxContents parameters.
1070/// Mirrors `readBoxContents($gullet, $everybox, $mode)` exactly:
1071/// pushes a fresh frame for `$mode`, digests tokens until matching T_END,
1072/// pops the frame.
1073///
1074/// The mode-aware variant matters when `\vtop` is invoked inside a `p{}`
1075/// column body (alignment is in horizontal mode, but the `\vtop`'s VBox
1076/// content must be digested in internal_vertical mode so `\@startpbox`'s
1077/// `\vtop\bgroup ...` doesn't try to close groups in the wrong mode).
1078/// Witness: 2210.13325 `\begin{tabular}{|p{1cm}|...}` — pre-fix Rust
1079/// inherited the surrounding horizontal mode and emitted `\vtop`
1080/// errors; Perl uses 'internal_vertical' here regardless of where the
1081/// `\vtop` was invoked.
1082pub fn predigest_box_contents_in_mode(
1083  _tokens: ArgWrap,
1084  mode: &str,
1085) -> Result<Option<Digested>> {
1086  // Perl: readBoxContents calls beginMode($mode) / endMode($mode) around the body reading.
1087  // This creates a scoped frame where enterHorizontal can change MODE inplace.
1088  // When endMode is called, leaveHorizontal_internal detects MODE='horizontal' with
1089  // BOUND_MODE ending in 'vertical', triggers repackHorizontal, then pops the frame.
1090  //
1091  // NOTE: read_box_contents already consumed the opening { or \bgroup via defined_as(T_BEGIN).
1092  // invoke_token(T_BEGIN) pushes a synthetic group frame. The matching } or \egroup
1093  // in the content will pop this frame, since \egroup is \let to T_END and
1094  // invoke_token handles it via the standard group-closing mechanism.
1095  // Perl: $stomach->beginMode($mode) — push a new frame for this box content scope
1096  if mode.ends_with("vertical") || mode.ends_with("horizontal") {
1097    stomach::begin_mode(mode)?;
1098  }
1099  let mut contents = stomach::invoke_token(&T_BEGIN!())?;
1100  if contents.is_empty() {
1101    // Perl: $stomach->endMode($mode)
1102    if mode.ends_with("vertical") || mode.ends_with("horizontal") {
1103      stomach::end_mode(mode)?;
1104    }
1105    Ok(None)
1106  } else {
1107    let mut item = contents.remove(0);
1108    // Perl's endMode triggers leaveHorizontal_internal → repackHorizontal
1109    // when enterHorizontal changed MODE to 'horizontal' inplace within this frame.
1110    // Check the condition BEFORE endMode pops the frame.
1111    let post_mode = state::lookup_string_from_sym(pin!("MODE"));
1112    let bound_mode = state::lookup_string_from_sym(pin!("BOUND_MODE"));
1113    if post_mode == "horizontal"
1114      && bound_mode.ends_with("vertical")
1115      && has_only_simple_horizontal_content(&item)
1116    {
1117      repack_horizontal_in_list(&mut item);
1118      // Restore MODE like leave_horizontal_internal does
1119      state::assign_value_inplace_sym(pin!("MODE"), arena::pin(&bound_mode));
1120    }
1121    // Perl: $stomach->endMode($mode) — pop the frame
1122    if mode.ends_with("vertical") || mode.ends_with("horizontal") {
1123      stomach::end_mode(mode)?;
1124    }
1125    // Set the mode property on the resulting item (matching Perl's List(@boxes, mode => $mode))
1126    if !mode.is_empty() {
1127      item.set_property("mode", Stored::String(arena::pin(mode)));
1128    }
1129    // Apply Perl's List() single-item simplification for vertical modes.
1130    // In Perl, List(@boxes, mode=>'internal_vertical') returns the single box
1131    // directly when @boxes has 1 element and the box's mode is also vertical.
1132    // This is critical for nested \vbox/\vtop: the inner box's `is_vbox` property
1133    // must be visible to the outer box's constructor.
1134    Ok(Some(simplify_vertical_list(item)))
1135  }
1136}

1138/// Check if a List contains only simple horizontal content (TBoxes,
1139/// Comments, or sub-Lists whose mode is horizontal/restricted_horizontal/
1140/// math). This guards against repack being triggered for cases like
1141/// `\vtop{\begin{tabular}...}` where the tabular processing leaks
1142/// MODE='horizontal' but the content (an Alignment/Whatsit) should NOT
1143/// be paragraph-wrapped.
1144///
1145/// Inline brace-groups `{...}` inside running text produce sub-Lists
1146/// (one per group) whose mode is `restricted_horizontal`. Without
1147/// accepting those, e.g. `\vbox{\small\bfseries hello {,} world}` would
1148/// fail the repack gate and be measured as 3 separate vertical lines.
1149/// Witness: aistats2026.sty's `\def\And{\unskip{,}\enspace}` in the
1150/// `\@runningauthor` body — every author separator emits a `{,}` group
1151/// that fragments the vbox into many short rows, making `\ht\autrun`
1152/// far exceed 10pt and triggering the class's `\PackageError{Document}
1153/// {Running heading author exceeds size limitations}` (driver paper:
1154/// arXiv:2602.11863).
1155fn has_only_simple_horizontal_content(item: &Digested) -> bool {
1156  if let DigestedData::List(l) = item.data() {
1157    let list = l.borrow();
1158    let non_empty: Vec<_> = list
1159      .boxes
1160      .iter()
1161      .filter(|b| !b.get_property_bool("isEmpty"))
1162      .collect();
1163    non_empty.iter().all(|b| match b.data() {
1164      DigestedData::TBox(_) | DigestedData::Comment(_) => true,
1165      DigestedData::List(sub) => {
1166        // Inline brace-groups `{...}` in horizontal context digest to
1167        // a sub-List with no `mode` property (implicit hbox). Accept
1168        // those plus sub-Lists explicitly tagged as horizontal-flavour.
1169        // Reject `mode=vertical|internal_vertical` (structural
1170        // sub-vboxes) and any other tagged mode.
1171        let sub_mode = sub
1172          .borrow()
1173          .properties
1174          .get("mode")
1175          .map(|v| v.to_string())
1176          .unwrap_or_default();
1177        matches!(
1178          sub_mode.as_str(),
1179          "" | "horizontal" | "restricted_horizontal" | "math"
1180        )
1181      },
1182      _ => false,
1183    })
1184  } else {
1185    false
1186  }
1187}

1189/// Replicates Perl's repackHorizontal() within a List's children.
1190///
1191/// Perl (Stomach.pm lines 442-456): In readBoxContents, after digesting box content
1192/// in vertical mode, repackHorizontal groups consecutive horizontal-mode items
1193/// from @LaTeXML::LIST into a single List(@para, mode => 'horizontal') with
1194/// width set to \hsize. This enables compute_boxes_size to do paragraph wrapping.
1195///
1196/// Without this, \vbox{hop} measures each character individually (width=5.55pt)
1197/// instead of wrapping as a paragraph at \hsize (width=469.75pt).
1198fn repack_horizontal_in_list(item: &mut Digested) {
1199  if let DigestedData::List(l) = item.data() {
1200    let mut list = l.borrow_mut();
1201    let children = std::mem::take(&mut list.boxes);
1202    let mut result: Vec<Digested> = Vec::new();
1203    let mut para: Vec<Digested> = Vec::new();
1204    let mut keep = false;
1205
1206    for child in children {
1207      let child_mode = child
1208        .get_property("mode")
1209        .map(|v| v.to_string())
1210        .unwrap_or_else(|| "horizontal".to_string());
1211      // Empty-string mode means "implicit hbox" — produced by inline
1212      // brace-groups `{...}` in horizontal context. Treat as horizontal
1213      // so the surrounding running text doesn't get fragmented.
1214      let effective_mode = if child_mode.is_empty() { "horizontal" } else { child_mode.as_str() };
1215      if effective_mode == "horizontal" || effective_mode == "restricted_horizontal" || effective_mode == "math"
1216      {
1217        // Perl: $keep = 1 if ($mode ne 'horizontal') || !$item->getProperty('isSpace');
1218        if effective_mode != "horizontal" || !child.get_property_bool("isSpace") {
1219          keep = true;
1220        }
1221        para.push(child);
1222      } else {
1223        // Flush accumulated horizontal items as a horizontal List
1224        if keep {
1225          let horiz_list = make_horizontal_list(std::mem::take(&mut para));
1226          result.push(Digested::from(horiz_list));
1227        } else {
1228          result.extend(std::mem::take(&mut para));
1229        }
1230        keep = false;
1231        result.push(child);
1232      }
1233    }
1234    // Flush remaining horizontal items
1235    if keep {
1236      let horiz_list = make_horizontal_list(para);
1237      result.push(Digested::from(horiz_list));
1238    } else {
1239      result.extend(para);
1240    }
1241    list.boxes = result;
1242  }
1243}

1246/// Perl: push(@LaTeXML::LIST, List(@para, mode => 'horizontal')) if $keep;
1247/// Perl: $list->setProperty(width => LookupRegister('\hsize')) if $mode eq 'horizontal';
1248fn make_horizontal_list(para: Vec<Digested>) -> List {
1249  let mut list = List::new(para);
1250  list.mode = Some(TexMode::Text);
1251  list.set_property("mode", Stored::String(arena::pin_static("horizontal")));

1258/// Perl's List() single-item simplification for vertical modes.
1259///
1260/// Perl (List.pm line 41-44):
1261/// ```perl
1262/// if ((scalar(@boxes) == 1)
1263///     && (!$mode || ($mode !~ /vertical$/)
1264///         || (($boxes[0]->getProperty('mode')||'') =~ /vertical$/))) {
1265///     return $boxes[0]; }   # Simplify!
1266/// ```
1267///
1268/// When a List in vertical mode contains a single non-empty item whose mode is also
1269/// vertical, return that item directly. This is critical for nested \vbox/\vtop:
1270/// the inner \vbox Whatsit has `is_vbox = true` set by after_digest, and the outer
1271/// \vtop constructor needs to see this property to skip double insertBlock wrapping.
1272fn simplify_vertical_list(item: Digested) -> Digested {
1273  // Only simplify if the item is a List
1274  let is_vertical_list = match item.data() {
1275    DigestedData::List(l) => {
1276      let list = l.borrow();
1277      // Check if the List's mode property indicates vertical
1278      list
1279        .properties
1280        .get("mode")
1281        .map(|m| m.ends_with_text("vertical"))
1282        .unwrap_or(false)
1283    },
1284    _ => false,
1285  };
1286  if !is_vertical_list {
1287    return item;
1288  }
1289
1290  // Extract the List's boxes, filtering out empty marker items (isEmpty property)
1291  let non_empty: Vec<Digested> = match item.data() {
1292    DigestedData::List(l) => {
1293      let list = l.borrow();
1294      list
1295        .boxes
1296        .iter()
1297        .filter(|b| !b.get_property_bool("isEmpty"))
1298        .cloned()
1299        .collect()
1300    },
1301    _ => unreachable!(),
1302  };
1303
1304  // Perl simplification: single non-empty item whose mode is also vertical
1305  if non_empty.len() == 1 {
1306    let single = &non_empty[0];
1307    let child_is_vertical = match single.data() {
1308      DigestedData::List(l) => l
1309        .borrow()
1310        .properties
1311        .get("mode")
1312        .map(|m| m.ends_with_text("vertical"))
1313        .unwrap_or(false),
1314      DigestedData::Whatsit(w) => {
1315        // Check whatsit's mode property (set by DefConstructor mode => "internal_vertical")
1316        w.borrow()
1317          .get_property("mode")
1318          .map(|m| m.ends_with_text("vertical"))
1319          .unwrap_or(false)
1320      },
1321      _ => false,
1322    };
1323    if child_is_vertical {
1324      return non_empty.into_iter().next().unwrap();
1325    }
1326  }
1327  item
1328}

1371/// This attempts to be a generalize vbox construction;
1372///
1373/// The idea is to receeive block-like material, possibly wrapped in appropriate
1374/// container which gets attributes.
1375///
1376/// The contents are constructed in an ltx:_CaptureBlock_ element,
1377/// designed to accept all reasonable block material from several levels,
1378/// and then determine which container element is most apprpriate for both the conent & context
1379/// from block, logical-block or sectional-block, or the inline- variants.
1380/// Perl: isVAttached — checks if node or any single-child descendant has 'vattach'
1381fn is_v_attached(node: &Node) -> bool {
1382  let mut current = node.clone();
1383  loop {
1384    if current.get_attribute("vattach").is_some() {
1385      return true;
1386    }
1387    let children: Vec<_> = current
1388      .get_child_nodes()
1389      .into_iter()
1390      .filter(|n| matches!(n.get_type(), Some(NodeType::ElementNode)))
1391      .collect();
1392    if children.len() != 1 {
1393      return false;
1394    }
1395    current = children[0].clone();
1396  }
1397}

1399pub fn insert_block(
1400  document: &mut Document,
1401  contents: &Digested,
1402  block_attr: HashMap<String, String>,
1403) -> Result<Vec<Node>> {
1404  // Create something like:
1405  // "<ltx:inline-block vattach='$vattach' height='#height'>#2</ltx:inline-block>"
1406  let context_opt = document.get_element(); // Where we originally start inserting.
1407  if context_opt.is_none() {
1408    // edge case: if we start the doc with a block, the context is empty
1409    document.absorb(contents, None)?;
1410    return Ok(Vec::new());
1411  }
1412  let mut context = context_opt.unwrap();
1413  let mut context_tag = document::get_node_qname(&context);
1414  // svg is slightly tricky
1415  let (is_svg, is_xmath, is_xmtext) = arena::with(context_tag, |tag| {
1416    (
1417      tag.starts_with("svg:"),
1418      tag.starts_with("ltx:XM"),
1419      tag == "ltx:XMText",
1420    )
1421  });
1422  // Perl L420-421: in SVG context, convert width from Dimension (pt) to em units
1423  let mut block_attr = block_attr;
1424  if is_svg {
1425    if let Some(width_str) = block_attr.get("width").cloned() {
1426      if let Some(pt_val) = width_str
1427        .strip_suffix("pt")
1428        .and_then(|s| s.parse::<f64>().ok())
1429      {
1430        // Convert pt to em using content's font em width
1431        let em_width = contents
1432          .get_font()
1433          .ok()
1434          .flatten()
1435          .map(|f| f.get_em_width())
1436          .unwrap_or((10.0 * 65536.0) as i64);
1437        let em_val = (pt_val * 65536.0) / em_width as f64;
1438        let em_rounded = latexml_core::common::numeric_ops::round_to(em_val, None);
1439        block_attr.insert("width".to_string(), format!("{}em", em_rounded));
1440      }
1441    }
1442  }
1443  let ignorable_attr = is_svg || block_attr.is_empty(); // if we do not REQUIRE the attributes
1444  if is_xmath && !is_xmtext {
1445    // but math always needs this
1446    context = document.open_element("ltx:XMText", None, None)?;
1447    context_tag = document::get_node_qname(&context);
1448  }
1449  let is_inline = is_svg || document::can_contain(&context, "#PCDATA");
1450  let container_attr = block_attr.clone();
1451  let mut container = document.open_element("ltx:_CaptureBlock_", Some(container_attr), None)?;
1452  document.absorb(contents, None)?;
1453
1454  let mut nodes = content_nodes(&container);
1455  let node_tags = nodes
1456    .iter()
1457    .map(document::get_node_qname)
1458    .collect::<Vec<_>>();
1459  let nnodes = nodes.len();
1460  document.close_to_node(&container, true)?;
1461  document.close_node(&container)?;
1462  document.close_to_node(&context, true)?;
1463
1464  // Perl: Hack: apparently TeX doesn't shift (vattach) a single node in a vbox/vtop/...
1465  #[allow(clippy::redundant_locals)]
1466  let mut block_attr = block_attr;
1467  let mut ignorable_attr = ignorable_attr;
1468  if nnodes == 1 && block_attr.contains_key("vattach") && is_v_attached(&nodes[0]) {
1469    container.remove_attribute("vattach")?;
1470    block_attr.remove("vattach");
1471    ignorable_attr = is_svg || block_attr.is_empty();
1472  }
1473
1474  if nnodes < 1 {
1475    // Insertion came up empty?
1476    document.remove_node(container); // then remove the new block entirely
1477    return Ok(nodes);
1478  } else if ignorable_attr
1479    && node_tags
1480      .iter()
1481      .all(|tag| document::can_contain_qsym(context_tag, *tag))
1482  {
1483    // No attributes, contents allowed in context?
1484    document.unwrap_nodes(container)?; // No container needed, at all.
1485    return Ok(nodes);
1486  } else if nnodes == 1 {
1487    if document::can_contain_qsym(context_tag, node_tags[0])
1488      && (ignorable_attr
1489        || block_attr
1490          .keys()
1491          .all(|key| document::sym_can_have_attribute(node_tags[0], arena::pin(key))))
1492    {
1493      // IF: Single node, allowed in context & accepts attributes
1494      // THEN: Add attributes and unwrap the single node
1495      for (k, v) in block_attr.iter() {
1496        document.set_attribute(&mut nodes[0], k, v)?;
1497      }
1498      document.unwrap_nodes(container)?;
1499      return Ok(nodes);
1500    } else if let Some(newcontainer) = document::sym_can_contain_somehow(context_tag, node_tags[0])
1501    {
1502      if ignorable_attr
1503        || block_attr.keys().all(|key| {
1504          newcontainer
1505            .map(|nc| document::sym_can_have_attribute(nc, arena::pin(key)))
1506            .unwrap_or(false)
1507        })
1508      {
1509        if let Some(nc) = newcontainer {
1510          // rename the capture to that container
1511          document.rename_node_qsym(container, nc, true)?;
1512          return Ok(nodes);
1513        }
1514      }
1515    }
1516  }
1517  // This jagged conditional is a "code smell", due to the difficulty of refactoring
1518  // the in-conditional-assignments from Perl.
1519
1520  // Otherwise, rename the capture
1521  // MAY need foreignObject wrapper
1522  if is_svg
1523    && node_tags
1524      .iter()
1525      .any(|tag| arena::with(*tag, |tag_str| tag_str.starts_with("ltx:")))
1526  {
1527    context = document
1528      .wrap_nodes("svg:foreignObject", vec![container.clone()])?
1529      .expect("foreign object wrap should always succeed in SVG");
1530    context_tag = document::get_node_qname(&context);
1531  }
1532  let candidates = if is_inline {
1533    [
1534      "ltx:inline-block",
1535      "ltx:inline-logical-block",
1536      "ltx:inline-sectional-block",
1537    ]
1538    .map(arena::pin_static)
1539    .to_vec()
1540  } else {
1541    [
1542      "ltx:block",
1543      "ltx:logical-block",
1544      "ltx:sectional-block",
1545      "ltx:figure",
1546    ]
1547    .map(arena::pin_static)
1548    .to_vec()
1549  };
1550  let filtered_candidates = candidates
1551    .into_iter()
1552    .filter(|candidate| {
1553      node_tags
1554        .iter()
1555        .all(|tag| document::sym_can_contain_somehow(*candidate, *tag).is_some())
1556    })
1557    .collect::<Vec<_>>();
1558  // and are allowed in the context
1559  let allowed_candidates = filtered_candidates
1560    .iter()
1561    .filter(|candidate| document::can_contain_qsym(context_tag, **candidate))
1562    .copied()
1563    .collect::<Vec<_>>();
1564  if let Some(final_tag) = allowed_candidates
1565    .first()
1566    .map_or_else(|| filtered_candidates.first(), Some)
1567  {
1568    // Rename the capture to the correct container
1569    // TODO: There is an arena code smell here. The `Model` interface needs to become lock-free
1570    // where Symbol tickets and &str are equally intuitive to use without runtime panics from
1571    // arena mutability exceptions.
1572    document.rename_node(container, &arena::to_string(*final_tag), true)?;
1573  } else {
1574    // we didn't know what to do?
1575    let message = arena::with(context_tag, |ctxt_str| {
1576      s!(
1577        "Did not find a block-like candidate in {} (with attributes ({})",
1578        ctxt_str,
1579        block_attr
1580          .iter()
1581          .map(|(k, v)| s!("{k}={v}"))
1582          .collect::<Vec<_>>()
1583          .join(";")
1584      )
1585    });
1586    Warn!("malformed", "_CaptureBlock_", message);
1587    document.rename_node(container, "ltx:block", true)?;
1588  }
1589  Ok(nodes)
1590}

1592pub fn cleanup_math(document: &mut Document, mathnode: Node) -> Result<()> {
1593  // Cleanup ltx:Math elements; particularly if they aren't "really" math.
1594  // But record the oddity with class=ltx_markedasmath
1595
1596  // If the Math ONLY contains XMath/XMText and XMHint, it apparently isn't math at all!?!
1597  // Single token PUNCTs can also be taken out of math.
1598  let xpath = concat!(
1599    "ltx:XMath/ltx:*[local-name() != 'XMText' and local-name() != 'XMHint'",
1600    " and not(",
1601    "local-name() = 'XMTok' and (@role='PUNCT' or @role='PERIOD')",
1602    " and not(preceding-sibling::*) and not(following-sibling::*) )]"
1603  );
1604  if document.findnodes(xpath, Some(&mathnode)).is_empty() {
1605    // So unwrap down to the contents of the XMText's.
1606    let xmath_children: Vec<_> = mathnode
1607      .get_child_nodes()
1608      .into_iter()
1609      .flat_map(|child| child.get_child_nodes())
1610      .collect();
1611    let mut texts: Vec<Node> = vec![];
1612    for xmnode in xmath_children {
1613      let is_hint = document::with_node_qname(&xmnode, |qname| qname == "ltx:XMHint");
1614      if is_hint {
1615        // Convert XMHint width to spacing characters
1616        if let Some(width_str) = xmnode.get_attribute("width") {
1617          // Width may be a full glue spec like "2.22217pt plus 1.11108pt minus 2.22217pt"
1618          // Extract just the base dimension (before "plus" or "minus")
1619          let base_dim_str = width_str
1620            .split_once(" plus")
1621            .or_else(|| width_str.split_once(" minus"))
1622            .map_or(width_str.as_str(), |(base, _)| base);
1623          // Try parsing as Dimension (pt). If that fails, handle mu units
1624          // by converting mu→pt (1mu = font_size/18).
1625          let dim_opt = Dimension::from_str(base_dim_str).ok().or_else(|| {
1626            if base_dim_str.ends_with("mu") {
1627              let mu_str = base_dim_str.trim_end_matches("mu").trim();
1628              mu_str.parse::<f64>().ok().map(|mu_val| {
1629                let fs = state::lookup_font()
1630                  .and_then(|f| f.get_size())
1631                  .unwrap_or(10.0);
1632                Dimension::from_str(&format!("{}pt", mu_val * fs / 18.0)).unwrap_or_default()
1633              })
1634            } else {
1635              None
1636            }
1637          });
1638          if let Some(dim) = dim_opt {
1639            let spaces = super::tex_glue::dimension_to_spaces(dim);
1640            if !spaces.is_empty() {
1641              if let Ok(text_node) = Node::new_text(&spaces, &document.document) {
1642                texts.push(text_node);
1643              }

1827// Since this is used for textual tokens, typically to split author lists,
1828// we don't split within braces or math
1829pub fn split_tokens(tokens: Tokens, delims: Vec<Token>) -> Vec<Tokens> {
1830  let mut items = Vec::new();
1831  let mut toks = Vec::new();
1832  if !tokens.is_empty() {
1833    let tokens = tokens.unlist();
1834    let mut tokens_iter = tokens.into_iter();
1835    while let Some(t) = tokens_iter.next() {
1836      // Perl: Equals($t, $delim) checks meaning via lookupMeaning.
1837      // So \And (let to \and) matches \and as delimiter.
1838      if delims.iter().any(|d| {
1839        d == &t
1840          || (t.get_catcode() == Catcode::CS && d.get_catcode() == Catcode::CS && {
1841            let meaning_t = state::lookup_definition(&t).ok().flatten();
1842            let meaning_d = state::lookup_definition(d).ok().flatten();
1843            meaning_t.is_some() && meaning_t == meaning_d
1844          })
1845      }) {
1846        items.push(Tokens::new(std::mem::take(&mut toks)));
1847      } else if t == T_BEGIN!() {
1848        toks.push(t);
1849        let mut level = 1;
1850        for t in tokens_iter.by_ref() {
1851          match t.get_catcode() {
1852            Catcode::BEGIN => level += 1,
1853            Catcode::END => level -= 1,
1854            _ => {},
1855          }
1856          toks.push(t);
1857          if level < 1 {
1858            // done if balanced.
1859            break;
1860          }
1861        }
1862      } else if t == T_MATH!() {
1863        toks.push(t);
1864        for t in tokens_iter.by_ref() {
1865          let is_math = t.get_catcode() == Catcode::MATH;
1866          toks.push(t);
1867          if is_math {
1868            break;
1869          }
1870        }
1871      } else {
1872        toks.push(t);
1873      }
1874    }
1875    // last author is in toks, add to items
1876    items.push(Tokens::new(toks));
1877  }
1878  items
1879}

1883  // Only split on \and. The meaning-based check in split_tokens also matches

1898/// Converts tokens to a string in the fashion of \message and others
1899///
1900/// doubles #, converts to string; optionally adds spaces after control sequences
1901/// in the spirit of the B Book, "show_token_list" routine, in 292.
1902/// [This could be a $tokens->unpackParameters, but for the curious space treatment]
1903pub fn writable_tokens(tokens: &Tokens) -> String {
1904  let mut wv = Vec::new();
1905  for t in tokens.unlist_ref().iter() {
1906    match t.code {
1907      Catcode::CS => {
1908        wv.push(*t);
1909        // Perl: add space after CS unless it's a single non-alpha char CS (like \{, \\, \#)
1910        // i.e. skip space only for "\X" where X is exactly one non-[a-zA-Z] character
1911        let is_single_nonalpha_cs = arena::with(t.text, |s| {
1912          s.starts_with('\\') && {
1913            let rest = &s[1..];
1914            rest.chars().count() == 1 && !rest.chars().next().unwrap_or(' ').is_ascii_alphabetic()
1915          }
1916        });
1917        if !is_single_nonalpha_cs {
1918          wv.push(T_SPACE!());
1919        }
1920      },
1921      Catcode::SPACE => {
1922        wv.push(T_SPACE!());
1923      },
1924      Catcode::PARAM => {
1925        wv.push(*t);
1926        wv.push(*t);
1927      },
1928      Catcode::ARG => {
1929        // B Book, 294. Reduce to param+integer
1930        wv.push(T_PARAM!());
1931        wv.push(t.as_other());
1932      },
1933      _ => {
1934        wv.push(*t);
1935      },
1936    }
1937  }

1939}

1941/// Support for Key / Value arguments.
1942// The very basic form is
1943//   RequiredKeyVals: $keyset
1944//   OptionalKeyVals: $keyset
1945// to parse Key-Value pairs from a given keyset (see the 'keyval' package
1946// documentation for more information). These types of KeyVal
1947// parameters will return a LaTeXML::Core::KeyVals object, which can then be
1948// used to access the values of the individual items.
1949// The difference between the two forms is that RequiredKeyVals expects a set of
1950// key-value pairs wrapped in T_BEGIN T_END, where as OptionalKeyVals optionally
1951// expects a set of KeyValue pairs wrapped in T_OTHER('[') T_OTHER(']')
1952//
1953// Several extension of the keyval package exist, the most common one we support
1954// is the xkeyval package. This introduces further variations on the keyval
1955// arguments parsing, in particular it allows to read keys from more than one
1956// keyset at once. These can be specified by giving comma-separated values in
1957// the keyset argument. By default, a key will only be set in the **first**
1958// keyset it occurs in. By using
1959//   RequiredKeyVals+: $keysets
1960//   OptionalKeyVals+: $keysets
1961// the key will be set in all keysets instead.
1962//
1963// All keys to be parsed with these arguments should be declared using
1964// DefKeyVal in LaTeXML::Package. By default, an error is thrown if an unknown
1965// key is encountered. To surpress this behaviour, and instead store all
1966// undefined keys, use
1967//   RequiredKeyVals*: $keysets
1968//   OptionalKeyVals*: $keysets
1969// instead. The '*' and '+' modifiers can be combined by using:
1970//   RequiredKeyVals*+: $keysets
1971//   OptionalKeyVals*+: $keysets
1972//
1973// Furthermore, the xkeyval package supports giving prefixes to keys,
1974//   RequiredKeyVals[*][+]: $prefix|$keysets
1975//   OptionalKeyVals[*][+]: $prefix|$keysets
1976//
1977// Finally, it is possible to specify specific keys to skip when digesting the
1978// object. This can be achieved using comma-separated key values in
1979//   RequiredKeyVals[*][+]: $prefix|$keysets|$skip
1980//   OptionalKeyVals[*][+]: $prefix|$keysets|$skip
1981
1982// function to handle all the
1983#[derive(Default)]
1984pub struct KVSpec {
1985  pub star:    bool,
1986  pub plus:    bool,
1987  pub prefix:  Option<String>,
1988  pub keysets: Vec<String>,
1989  pub skip:    Vec<String>,
1990}
1991pub fn keyvals_aux(until: Option<Token>, spec: KVSpec) -> Result<KeyVals> {
1992  let KVSpec {
1993    mut star,
1994    plus,
1995    mut prefix,
1996    mut keysets,
1997    skip,
1998  } = spec;
1999  // support both "keysets" and "prefix|keysets"
2000  if keysets.is_empty() {
2001    if let Some(pfx) = prefix.take() {
2002      keysets = vec![pfx];
2003    }
2004
2005    // to emulate old behaviour, throw no errors
2006    // when we have a single keyset and no prefix (or no keyset at all)
2007    if keysets.is_empty() {
2008      star = true;
2009    }
2010  }
2011
2012  // create a new set of Key-Value arguments
2013  let mut keyvals = KeyVals::new(KeyvalsConfig {
2014    prefix,
2015    keysets,
2016    set_all: plus,
2017    set_internals: true,
2018    skip,
2019    skip_missing: if star {
2020      keyvals::SkipMissing::All
2021    } else {
2022      keyvals::SkipMissing::None
2023    },

2025  });
2026  // and read it from the gullet
2027  if let Some(until_token) = until {
2028    keyvals.read_from(until_token, false)?;
2029  }
2030  // we still want to make use of the hash
2031  Ok(keyvals)
2032}

2176/// Heuristic classifier for vendor `\PackageError`/`\GenericError`
2177/// messages whose only concern is PDF typesetting (size, layout,

2179/// are downgraded to `Info:` per WISDOM #50.
2180fn is_typesetting_only_message(message: &str) -> bool {
2181  let lower = message.to_ascii_lowercase();
2182  // Phrase set tuned against the stage-1 sweep of the 100k warning
2183  // corpus. Conservative — every phrase here is purely about visual
2184  // layout or vendor-deprecation chatter, never about semantic
2185  // correctness. Examples:
2186  //   "Running heading author exceeds size limitations" (AISTATS)
2187  //   "Running heading title exceeds size limitations" (AISTATS)
2188  //   "Caption too wide for page" (various)
2189  //   "Heading breaks the line" (revtex, IEEEtran)
