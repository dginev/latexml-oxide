10  DefParameterType!(Plain, sub[inner, _extra] {
11    let mut value = ArgWrap::Tokens(gullet::read_arg(ExpansionLevel::Off)?);
12    if let Some(inner_ps) = inner {
13      // TODO: How many arguments can we expect back? One? Many?
14      //       Currently only passing through the first
15      value = inner_ps.reparse_argument(value)?.remove(0);
16    }
17    Ok(value)
18  },
19  reversion => sub[arg, inner, _extra] {
20    // let mut reverted_inner;
21    let mut read_tokens: Vec<Token> = vec![T_BEGIN!()];
22    read_tokens.extend(if let Some(inner_ps) = inner {
23      inner_ps.revert_arguments(vec![Some(Tokens::new(arg))])?
24    } else {
25      arg.iter().map(|t| t.revert()).collect()
26    });
27    read_tokens.push(T_END!());
28    Ok(Tokens::new(read_tokens))
29  });

32    let mut value = ArgWrap::Tokens(gullet::read_balanced(ExpansionLevel::Off, true, true)?);
33    if let Some(inner_ps) = inner {
34      value = inner_ps.reparse_argument( value)?.remove(0);
35    }

50  DefParameterType!(Optional, sub[inner, default] {
51    let value = gullet::read_optional(None)?;
52    if value.is_none() && !default.is_empty() {
53      // TODO: Is the default really multiple Vec<Tokens> ? Or just a single Tokens?
54      //       the default[0] is suspicious, compared to the original perl "$default"
55      ArgWrap::Tokens(default[0].clone())
56    } else if let Some(inner_ps) = inner {
57      let mut reparsed = inner_ps.reparse_argument( value.into())?;
58      if !reparsed.is_empty() {
59        reparsed.remove(0)
60      } else {
61        ArgWrap::None
62      }
63    } else {
64      value.into()
65    }
66  },
67  optional => true,
68  reversion => sub[arg, inner, _extra] {
69    // TODO: Same question for the type of "arg" as the one above "default" above:
70    //  should this be a single `Tokens` rather than a `Vec<Token>`?
71    if !arg.is_empty() {
72      let mut read_tokens: Vec<Token> = vec![T_OTHER!("[")];
73      read_tokens.extend(match inner {
74        None => arg.into_iter().map(Token::revert).collect(),
75        Some(inner_ps) => inner_ps.revert_arguments(vec![Some(Tokens::new(arg))])?,
76      });
77      read_tokens.push(T_OTHER!("]"));
78      Ok(Tokens::new(read_tokens))
79    } else {
80      Ok(Tokens!())
81    }

124  DefParameterType!(Token, sub[_inner, _extra] {
125    // Perl Base_ParameterTypes.pool.ltxml L91:
126    //   DefParameterType('Token', sub { $_[0]->readToken; });
127    // No error raised on EOF — silently returns undef. Our prior impl
128    // raised an error, which manifested as a cascade after upstream
129    // recovery (e.g. \^ with no argument in math mode). EOF fallback
130    // returns `\relax` (a no-op token) so downstream `try_to_token`
131    // succeeds — using empty Tokens here would cascade
132    // "Error:expected:argument: try_to_token: empty Tokens" at the
133    // primitive's arg-coerce step (witness: 0910.2125 sub/superscript
134    // cascade where 13 errors match Perl + 1 extra Rust empty-Tokens).
135    match gullet::read_token()? {
136      Some(t) => Ok(ArgWrap::Token(t)),
137      None => Ok(ArgWrap::Token(T_CS!("\\relax"))),
138    }
139  });

194  // Read a Pair (x,y) — parenthesized, comma-separated pair of Float values.
195  // Perl: ReadPair in latex_constructs.pool.ltxml
196  // Returns ArgWrap::Pair if ( is found, ArgWrap::None otherwise (for Optional).
197  //
198  // Helper: read a float that may be wrapped in braces (e.g. `{36.5}`).
199  // Some authors brace pair coordinates to disambiguate negative-number
200  // tokenization or to keep \multiput / \put pair args together.
201  // Witness: hep-th/9610147 — `\multiput(-89,{36.5})(-6,-1){6}{…}`.
202  DefParameterType!(Pair, sub[_inner, _extra] {
203    use latexml_core::common::pair::Pair;
204    use latexml_core::token::Catcode;
205    fn read_pair_float() -> Result<latexml_core::common::float::Float> {
206      let _ = gullet::skip_spaces();
207      // If next is BEGIN brace, consume to matching END brace and read float inside.
208      if let Some(tok) = gullet::read_token()? {
209        if tok.get_catcode() == Catcode::BEGIN {
210          let _ = gullet::skip_spaces();
211          let f = gullet::read_float()?;
212          let _ = gullet::skip_spaces();
213          // Consume matching close brace
214          if let Some(close) = gullet::read_token()? {
215            if close.get_catcode() != Catcode::END {
216              gullet::unread_one(close);
217            }
218          }
219          return Ok(f);
220        }
221        gullet::unread_one(tok);
222      }
223      gullet::read_float()
224    }
225    let _ = gullet::skip_spaces();
226    if gullet::if_next(T_OTHER!("("))? {
227      gullet::read_token()?; // consume (
228      let _ = gullet::skip_spaces();
229      let x = read_pair_float()?;
230      // Perl latex_constructs.pool.ltxml:ReadPair L4910-4912 uses
231      //   $gullet->skipSpaces; $gullet->readUntil(T_OTHER(',')); $gullet->skipSpaces;
232      //   my $y = ...;
233      //   $gullet->skipSpaces; $gullet->readUntil(T_OTHER(')')); $gullet->skipSpaces;
234      // — `readUntil` is tolerant of extra junk between the float and the
235      // separator. Witness: physics/9709007 line 1594
236      //   \multiput(3.2,3,8)(.3,0){2}{\circle*{.1}}
237      // The user typoed `3,8` for `3.8`; Perl reads x=3.2, swallows nothing
238      // up to the comma, reads y=3, then `readUntil(')')` consumes the
239      // extra `,8` silently. The earlier Rust port read one token after y
240      // and only consumed it if literally `)` — bailing the second pair.
241      let _ = gullet::skip_spaces();
242      let _ = gullet::read_until(&Tokens!(T_OTHER!(",")));
243      let _ = gullet::skip_spaces();
244      let y = read_pair_float()?;
245      let _ = gullet::skip_spaces();
246      let _ = gullet::read_until(&Tokens!(T_OTHER!(")")));
247      let _ = gullet::skip_spaces();
248      Ok(ArgWrap::Pair(Pair::new(x, y)))
249    } else {
250      Ok(ArgWrap::None)
251    }
252  });

272  DefParameterType!(XUntil, sub[_inner, untils] {
273    // XUntil requires a delimiter token in its Extras slot. A malformed
274    // parameter spec (bare `XUntil` with no `:delim` attached) would panic
275    // with the prior .expect(); degrade to reading nothing so the caller
276    // sees an empty Tokens result instead of an abort.
277    let Some(until_tks) = untils.first() else {
278      Warn!("expected", "token", "XUntil parameter missing delimiter token");
279      return Ok(ArgWrap::Tokens(Tokens!()));
280    };
281    let until : Token = until_tks.into();
282    let mut tokens : Vec<Token> = Vec::new();
283    while let Some(token) = gullet::read_x_token(Some(false), false, None)? {
284      if token == until {
285        break;
286      } else if token.get_catcode() == Catcode::BEGIN {
287        tokens.push(token);
288        tokens.extend(gullet::read_balanced(ExpansionLevel::Off,false,false)?.unlist());
289        tokens.push(T_END!());
290      } else {
291        // After read_x_token, an expandable macro should already have been
292        // expanded — what remains here is non-expandable (Primitive,
293        // Constructor, Conditional, Register, MathPrimitive) or a Let to
294        // a literal token. Eagerly calling `read_arguments()` on those
295        // was a bug: e.g. for `\hspace*{-4mm} $^*\,$` inside an XUntil
296        // body, `\hspace`'s primitive `Dimension` reader would
297        // over-consume past `}`, eating the following `$` token and
298        // causing the math frame to leak (witness: astro-ph9903386
299        // \institute math-mode leak). Only re-Invocation-emit for
300        // GENUINE Expandable defs that for some reason escaped
301        // `read_x_token`'s expansion (e.g. \protected macros). Inspect
302        // the raw `Stored` variant rather than `lookup_definition_stored`
303        // (which synthesizes a no-op Expandable around `Stored::Token`
304        // entries like `\sb`-Let-to-T_SUB; rebuilding such a synthesized
305        // invocation calls `build_invocation`, which rejects Token
306        // meanings and erroneously fires "Can't invoke; it is undefined"
307        // — witness: math0610119 `\sb` inside amsppt
308        // `\@bibfield XUntil:\@end@bibfield`).
309        let is_real_expandable =
310          matches!(state::lookup_meaning(&token), Some(Stored::Expandable(_)));
311        // Perl XUntil L144-146: ALWAYS calls `readArguments` for any defined
312        // token, wrapping in `Invocation`. Without this, a Constructor
313        // body like `\href{u}{t}` → `\lx@hyper@url@\href{}{}{u}{t}` lets
314        // XUntil's outer `read_x_token` re-expand the `\href` token (which
315        // the Constructor is supposed to consume as Undigested arg #1),
316        // producing infinite recursion. Witness: 1902.01143 elsarticle
317        // `\begin{keyword} ... \href{...}{...}` hangs at 100M token limit.
318        // Targeted to Constructors only — Primitives have side-effecting
319        // arg readers (e.g. `\hspace`'s Dimension reader over-consumed
320        // past `}` in astro-ph9903386 — the recorded regression that
321        // motivated the original "bare-token push" else-branch).
322        let is_constructor =
323          matches!(state::lookup_meaning(&token), Some(Stored::Constructor(_)));
324        // Definitional primitives (\def/\edef/\gdef/\xdef/\let/\futurelet) consume
325        // their target token from the input AT EXECUTION TIME — but XUntil's outer
326        // `read_x_token` would expand the target away if it's `\let`-bound to
327        // something expandable (e.g. `\@date` Let'd to `\@empty`). Witness:
328        // 0805.1712 elsart `\date{X}` inside `\begin{keyword}` body, where
329        // `\date` expands to `\def\@date{X}\@add@frontmatter{ltx:date}[...]{X}`,
330        // and inside the keyword's XUntil read, `\@date` (Let'd to `\@empty`)
331        // gets consumed by `read_x_token` BEFORE `\def` ever runs, leaving the
332        // captured tokens malformed (`\def {X} ...` with no def-target).
333        // Re-Invoke these primitives to read their parameters from the gullet
334        // here, matching Perl's XUntil L144-146 behavior. Targeted to the def
335        // family to avoid the `\hspace`/dimension-reader over-read issue
336        // recorded in the comment above (astro-ph9903386 leak).
337        let is_def_family = token.with_cs_name(|cs| {
338          matches!(cs, "\\def" | "\\gdef"
339            | "\\futurelet" | "\\global" | "\\protected" | "\\long" | "\\outer")
340        });
341        // \edef, \xdef AND \let need their target/body captured RAW — not
342        // through read_arguments→Invocation. Reasons:
343        //   - \edef/\xdef: DefExpanded body parameter expands eagerly
344        //     (`\itdefault` → `it`); Invocation revert drops the target
345        //     Token and the {…}-braces, leaving `\edef i t \selectfont`
346        //     stream that re-reads as malformed at EOF.
347        //     Driver: 2403.14274 IEEEconf `\itshape …`.
348        //   - \let: DefToken/Token reversion in some test paths emits the
349        //     `\let` token alone without the cs1/cs2 args, then the body
350        //     re-reads `\let \let \let …` recursively as `\let {…}` with
351        //     the next `\let`'s target = `{` — triggering 100-deep
352        //     recursion through `\lx@acronym`. Driver: 2103.11356 elsart
353        //     `\begin{keyword}\ac{CNNs} …`.
354        // Capture all three RAW: `<token> <skipped-spaces> <Token>
355        // <until-brace> {<balanced-raw>}`. Preserves original input
356        // exactly so re-emission re-reads cleanly.
357        let is_def_raw_capture =
358          token.with_cs_name(|cs| matches!(cs, "\\edef" | "\\xdef" | "\\let"));
359        if is_def_raw_capture {
360          tokens.push(token);
361          gullet::skip_spaces()?;
362          if let Some(target) = gullet::read_token()? {
363            tokens.push(target);
364          }
365          // \let has no UntilBrace + body — just `<target> <value>`.
366          // \edef/\xdef have UntilBrace + balanced-body.
367          let is_let = token.with_cs_name(|cs| cs == "\\let");
368          if is_let {
369            // Read second token (the value to alias to).
370            // Optional `=` and one space are also valid in real TeX
371            // — DefToken handles those, but Token doesn't. Peek for `=`.
372            gullet::skip_spaces()?;
373            if let Some(value) = gullet::read_token()? {
374              tokens.push(value);
375            }
376          } else {
377            if let Some(prebrace) = gullet::read_until_brace()? {
378              tokens.extend(prebrace.unlist());
379            }
380            tokens.push(T_BEGIN!());
381            tokens.extend(gullet::read_balanced(ExpansionLevel::Off, false, true)?.unlist());
382            tokens.push(T_END!());
383          }
384        } else if is_real_expandable || is_def_family || is_constructor {
385          if let Some(defn) = lookup_definition_stored(&token)? {
386            let args = defn.read_arguments()?;
387            tokens.extend(Invocation!(token, args).unlist());
388          } else {
389            tokens.push(token);
390          }
391        } else {
392          tokens.push(token);
393        }
394      }
395    }
396    Ok(Tokens::new(tokens))
397  });

1080  // CommaList expects something like {item1,item2,...}; items may be
1081  // `{balanced}` groups or plain token runs. Perl returns a
1082  // `LaTeXML::Core::Array`; Rust has no Array type and follows the
1083  // `DirectoryList` convention — emit a token stream where each item
1084  // is wrapped in its own `{...}`.
1085  //
1086  // Perl supports a parameterised form `CommaList:Type` (e.g.
1087  // `CommaList:Number`) where each item is re-parsed through `Type`'s
1088  // own reader (`$typedef->reparseArgument`). When the parameter is
1089  // declared as `CommaList:Number`, `inner` carries the parsed
1090  // `Number` Parameter; we route each item through
1091  // `inner.reparse_argument(...)` and emit the reverted form so the
1092  // result is canonical-shape tokens of the typed value. Untyped
1093  // `CommaList` (no `inner`) keeps the original brace-delimit form.
1094  DefParameterType!(CommaList, sub[inner, _extra] {
1095    gullet::skip_spaces()?;
1096    // Phase 1: gather items as raw Tokens (one Tokens per comma-
1097    // separated piece). Done in a single pass to mirror the Perl
1098    // gullet-reader logic; the typed-reparse step happens after.
1099    let mut items: Vec<Tokens> = Vec::new();
1100    if gullet::if_next(T_BEGIN!())? {
1101      gullet::read_token()?; // consume outer `{`
1102      let mut current: Vec<Token> = Vec::new();
1103      let comma = T_OTHER!(",");
1104      while let Some(token) = gullet::read_token()? {
1105        let cc = token.get_catcode();
1106        if cc == Catcode::END {
1107          items.push(Tokens::new(std::mem::take(&mut current)));
1108          break;
1109        } else if token == comma {
1110          items.push(Tokens::new(std::mem::take(&mut current)));
1111        } else if cc == Catcode::BEGIN {
1112          // Nested `{balanced}` — preserve brace wrapping.
1113          current.push(token);
1114          let balanced = gullet::read_balanced(ExpansionLevel::Off, false, false)?;
1115          current.extend(balanced.unlist());
1116          current.push(T_END!());
1117        } else {
1118          current.push(token);
1119        }
1120      }
1121    } else if let Some(token) = gullet::read_token()? {
1122      // No outer brace — read a single token as the sole item.
1123      items.push(Tokens::new(vec![token]));
1124    }
1125    // Phase 2: if a typed form was declared, reparse each item via
1126    // the inner type's reader (`Perl reparseArgument` parity). The
1127    // reverted form is emitted so the downstream callsite sees
1128    // canonical tokens of the typed value.
1129    let mut collected: Vec<Token> = Vec::new();
1130    if let Some(inner_ps) = inner {
1131      for item in items {
1132        let reparsed = inner_ps.reparse_argument(ArgWrap::Tokens(item))?;
1133        collected.push(T_BEGIN!());
1134        for wrap in reparsed {
1135          collected.extend(wrap.revert()?.unlist());
1136        }
1137        collected.push(T_END!());
1138      }
1139    } else {
1140      for item in items {
1141        collected.push(T_BEGIN!());
1142        collected.extend(item.unlist());
1143        collected.push(T_END!());
1144      }
1145    }
1146    Tokens::new(collected)
1147  });

1149  // Support for Key / Value arguments.
1150  // The very basic form is
1151  //   RequiredKeyVals: $keyset
1152  //   OptionalKeyVals: $keyset
1153  // to parse Key-Value pairs from a given keyset (see the 'keyval' package
1154  // documentation for more information). These types of KeyVal
1155  // parameters will return a LaTeXML::Core::KeyVals object, which can then be
1156  // used to access the values of the individual items.
1157  // The difference between the two forms is that RequiredKeyVals expects a set of
1158  // key-value pairs wrapped in T_BEGIN T_END, where as OptionalKeyVals optionally
1159  // expects a set of KeyValue pairs wrapped in T_OTHER('[') T_OTHER(']')
1160  //
1161  // Several extension of the keyval package exist, the most common one we support
1162  // is the xkeyval package. This introduces further variations on the keyval
1163  // arguments parsing, in particular it allows to read keys from more than one
1164  // keyset at once. These can be specified by giving comma-separated values in
1165  // the keyset argument. By default, a key will only be set in the **first**
1166  // keyset it occurs in. By using
1167  //   RequiredKeyVals+: $keysets
1168  //   OptionalKeyVals+: $keysets
1169  // the key will be set in all keysets instead.
1170  //
1171  // All keys to be parsed with these arguments should be declared using
1172  // DefKeyVal in LaTeXML::Package. By default, an error is thrown if an unknown
1173  // key is encountered. To surpress this behaviour, and instead store all
1174  // undefined keys, use
1175  //   RequiredKeyVals*: $keysets
1176  //   OptionalKeyVals*: $keysets
1177  // instead. The '*' and '+' modifiers can be combined by using:
1178  //   RequiredKeyVals*+: $keysets
1179  //   OptionalKeyVals*+: $keysets
1180  //
1181  // Furthermore, the xkeyval package supports giving prefixes to keys,
1182  //   RequiredKeyVals[*][+]: $prefix|$keysets
1183  //   OptionalKeyVals[*][+]: $prefix|$keysets
1184  //
1185  // Finally, it is possible to specify specific keys to skip when digesting the
1186  // object. This can be achieved using comma-separated key values in
1187  //   RequiredKeyVals[*][+]: $prefix|$keysets|$skip
1188  //   OptionalKeyVals[*][+]: $prefix|$keysets|$skip
1189
1190  pub fn required_key_vals(
1191    star: bool,
1192    plus: bool,
1193    _inner: Option<&Parameters>,
1194    extra: &[Tokens],
1195  ) -> Result<KeyVals> {
1196    // Skip whitespace between this arg and the previous one. TeX-style
1197    // parameter matching `#1#2{...}` skips spaces before each `{`-delimited
1198    // group; our `RequiredKeyVals` parameter type checks `if_next(T_BEGIN)`
1199    // directly, so without an explicit skip it errors on user input like
1200    //   \newglossaryentry{RIS}\n{ name={...}, ... }
1201    // (\n + indentation between args[0] and args[1]). Driver: 2203.11854
1202    // R=1 → R=0 — Perl raw-loads glossaries.sty so `\newglossaryentry` is
1203    // a 2-arg `\newcommand` whose TeX matching handles this natively.
1204    gullet::skip_spaces()?;

1206      let mut extra_iter = extra.iter();

1208      let prefix = extra_iter.next().map(ToString::to_string);
1209      // TODO: is the last extra field actually a "skip" ? Example?
1210      let keysets = extra_iter.map(ToString::to_string).collect();

1212        star,
1213        plus,
1214        prefix,
1215        keysets,

1217      })

1221      // engine convention is lowercase categories throughout.
