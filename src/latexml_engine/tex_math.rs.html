10/// Perl's mergeLimits (TeX_Math.pool.ltxml): walks backward through the
11/// digest list, extracts any existing script level from the previous
12/// `scriptpos` value, and sets `scriptpos` to `pos` + level.
13fn merge_limits(pos: &str) {
14  // is_script is now defined in this file (moved from tex_scripts.rs)
15  // Compute script level before borrowing the box list mutably,
16  // since get_script_level() also borrows the stomach.
17  let default_level = get_script_level().to_string();
18  stomach::with_box_list_mut(|list| {
19    for b in list.iter_mut().rev() {
20      // Extract trailing level digits from existing scriptpos. Find the
21      // first non-digit from the end, slice off the tail — skips the
22      // chars.rev.take_while.collect.into_iter.rev.collect two-Vec
23      // round-trip the previous code did.
24      let prev = b
25        .get_property("scriptpos")
26        .map(|v| v.to_string())
27        .unwrap_or_default();
28      let digit_start = prev
29        .bytes()
30        .rposition(|c| !c.is_ascii_digit())
31        .map(|i| i + 1)
32        .unwrap_or(0);
33      let level_str: &str = &prev[digit_start..];
34      let level = if level_str.is_empty() {
35        &default_level
36      } else {
37        level_str
38      };
39      b.set_property("scriptpos", format!("{pos}{level}"));
40      // Perl: last unless IsEmpty($box) || IsScript($box)
41      // Continue past empty boxes AND script boxes
42      if !b.is_empty().unwrap_or(false) && is_script(b).is_none() {
43        break;
44      }
45    }
46  });
47}

53static SCRIPT_NAME_RE: Lazy<Regex> =
54  Lazy::new(|| Regex::new(r"^\\lx@(floating|post)@(subscript|superscript)$").unwrap());

56/// Remember a "safe" way to test a script Whatsit.
57/// Returns [ (FLOATING|POST) , (SUBSCRIPT|SUPERSCRIPT) ] or nothing
58pub fn is_script(object: &Digested) -> Option<(String, Catcode)> {
59  let box_opt = match object.data() {
60    DigestedData::List(obj) => obj.borrow().boxes.last().map(|v| Cow::Owned(v.clone())),
61    _ => Some(Cow::Borrowed(object)),
62  };
63  if let Some(boxobj) = box_opt {
64    if let DigestedData::Whatsit(ref obj) = boxobj.data() {
65      obj.borrow().get_definition().get_cs().with_cs_name(|name| {
66        SCRIPT_NAME_RE.captures(name).map(|cap| {
67          (
68            cap.get(1).map_or("", |m| m.as_str()).to_uppercase(),
69            if cap.get(2).map_or("", |m| m.as_str()) == "subscript" {
70              Catcode::SUB
71            } else {
72              Catcode::SUPER
73            },
74          )
75        })
76      })
77    } else {
78      None
79    }
80  } else {
81    None
82  }
83}

85fn script_handler(cc: Catcode) -> Result<Vec<Digested>> {
86  let font = lookup_font().unwrap();
87  if font.get_mathstyle().is_some() {
88    let mut putback = VecDeque::new();
89    let mut nscripts = 0;
90
91    let mut cs = if cc == Catcode::SUPER {
92      "\\lx@floating@superscript"
93    } else {
94      "\\lx@floating@subscript"
95    };
96    let mut prevscript = None;
97    let mut prevspace = false;
98    let mut base = None;
99    while let Some(prev) = { pop_box_list() } {
100      if prev.get_property_bool("isSpace") || prev.get_property_bool("isEmpty") {
101        prevspace = true;
102        putback.push_front(prev);
103        continue;
104      } else if prev.is_empty()? {
105        // Perl `TeX_Math.pool.ltxml:378-379`: "If empty, the script floats,
106        // can't conflict, but don't put back". `cs` stays at `\lx@floating@*`
107        // (initial value); the empty `{}` is consumed.
108        break;
109      } else if let Some(prevop) = is_script(&prev) {
110        if prevop.1 == cc {
111          putback.push_front(prev);
112          let lcode = if prevop.1 == Catcode::SUPER {
113            "superscript"
114          } else {
115            "subscript"
116          };
117          if !prevspace {
118            Error!("unexpected", s!("double-{lcode}"), s!("Double {lcode}"));
119          }
120          cs = if cc == Catcode::SUPER {
121            "\\lx@floating@superscript"
122          } else {
123            "\\lx@floating@subscript"
124          };
125          break;
126        } else {
127          prevscript = Some(prev.clone());
128          putback.push_front(prev);
129          cs = if cc == Catcode::SUPER {
130            "\\lx@post@superscript"
131          } else {
132            "\\lx@post@subscript"
133          };
134        }
135        if prevop.0 == "FLOATING" {
136          break;
137        }
138        nscripts += 1;
139        if nscripts > 1 {
140          break;
141        }
142      } else {
143        base = Some(prev.clone());
144        putback.push_front(prev);
145        cs = if cc == Catcode::SUPER {
146          "\\lx@post@superscript"
147        } else {
148          "\\lx@post@subscript"
149        };
150        break;
151      }
152    }
153    extend_box_list(putback);
154    MergeFont!(scripted => true);
155    let mut stuff = Vec::new();
156    while let Some(tok) = gullet::read_x_token(Some(false), false, None)? {
157      stuff = stomach::invoke_token(&tok)?;
158      if !stuff.is_empty() {
159        break;
160      }
161    }
162    if stuff.is_empty() {
163      Error!("expected", "{", "Missing sub/superscript argument");
164      stuff.push(Digested::default());
165    }
166    let script = stuff.remove(0);
167
168    if !script.is_empty()? {
169      let mut properties = {
170        stored_map!(
171          "isMath" => true,
172          "base"        => if let Some(b) = base { Stored::Digested(b) }
173            else { Stored::None },
174          "scriptlevel" => get_script_level(),
175          "level"       => get_boxing_level()
176        )
177      };
178      if let Some(pvs) = prevscript {
179        properties.insert("prevscript", pvs.into());
180      }
181      if let Some(Stored::Digested(ref b)) = properties.get("base") {
182        if let Some(bsp) = b.get_property("scriptpos") {
183          let bsp_str = bsp.to_string();
184          if !bsp_str.is_empty() {
185            let base_prefix: String = bsp_str
186              .chars()
187              .take_while(|c| !c.is_ascii_digit())
188              .collect();
189            let sl = get_script_level();
190            properties.insert("scriptpos", Stored::from(format!("{base_prefix}{sl}")));
191          }
192        }
193      }
194      if let Some(font) = script.get_font()? {
195        properties.insert("font", font.into());
196      }
197      let mut with_script = vec![Digested::from(Whatsit {
198        definition: lookup_definition(&T_CS!(cs))?.unwrap(),
199        args: vec![Some(script)],
200        properties,
201        ..Whatsit::default()
202      })];
203      with_script.extend(stuff);
204      stuff = with_script;
205    }
206    assign_font(font, Some(Scope::Local)); // revert
207    Ok(stuff)
208  } else {
209    let c = if cc == Catcode::SUPER { '^' } else { '_' };
210    // Perl-faithful lowercase category — matches TeX_Math.pool.ltxml:424
211    // `Error('unexpected', $c, $stomach, "Script $c can only appear in math
212    // mode")` and the rest of the engine's lowercase categories. Was a typo.
213    Error!(
214      "unexpected",
215      c,
216      format!("Script {c} can only appear in math mode")
217    );
218    let placeholder = if cc == Catcode::SUPER {
219      T_SUPER!()
220    } else {
221      T_SUB!()
222    };
223    Ok(vec![Digested::from(Tbox::new(
224      arena::pin_char(c),
225      None,
226      None,
227      Tokens!(placeholder),
228      SymHashMap::default(),
229    ))])
230  }
231}

233pub fn revert_script(script: &Digested) -> Result<Vec<Token>> {
234  let tokens = script.revert()?;
235  let mut ts = tokens.unlist();
236  if ts.len() > 1
237    && ts.first().unwrap().code == Catcode::BEGIN
238    && ts.last().unwrap().code == Catcode::END
239  {
240    Ok(ts)
241  } else {
242    let mut wrapped = vec![T_BEGIN!()];
243    wrapped.append(&mut ts);
244    wrapped.push(T_END!());
245    Ok(wrapped)
246  }
247}

249/// Fraction sizer — TeX-style width/height/depth for a fraction whose
250/// numerator and denominator are `top` and `bottom` digested boxes.
251///
252/// Perl: `fracSizer` in TeX_Math.pool.ltxml L1054-1059:
253///   w = max(numerator.width, denominator.width)
254///   d = denominator.total_height * 0.5
255///   h = numerator.total_height + d
256///
257/// Used by `\lx@generalized@over`, `\over`, `\atop`, `\above*` etc.
258pub fn frac_sizer(top: &Digested, bottom: &Digested) -> Result<(Dimension, Dimension, Dimension)> {
259  let (tw, th, td, ..) = top.clone().get_size(None)?;
260  let (bw, bh, bd, ..) = bottom.clone().get_size(None)?;
261  // width: max of top and bottom widths
262  let w = Dimension(tw.value_of().max(bw.value_of()));
263  // depth: half of denominator's total height (height + depth)
264  let bot_total = bh.value_of() + bd.value_of();
265  let d = Dimension(bot_total / 2);
266  // height: numerator total height + depth
267  let top_total = th.value_of() + td.value_of();
268  let h = Dimension(top_total + d.value_of());
269  Ok((w, h, d))
270}

272fn script_sizer(
273  script: &Digested,
274  base_opt: Option<&Stored>,
275  prev_opt: Option<&Stored>,
276  op: &str,
277  _pos: &str,
278) -> Result<(Dimension, Dimension, Dimension)> {
279  let script_size = script.clone().get_size(None)?;
280  let (mut ws, hs, ds) = (
281    script_size.0.value_of() as f64,
282    script_size.1.value_of() as f64,
283    script_size.2.value_of() as f64,
284  );
285  let (base_font_size, mathstyle) = if let Some(Stored::Digested(ref base)) = base_opt {
286    let bfont = base.get_font()?.map(|f| f.into_owned());
287    let fs = bfont.as_ref().and_then(|f| f.get_size()).unwrap_or(10.0);
288    let ms = bfont
289      .as_ref()
290      .and_then(|f| f.mathstyle.as_deref().map(|s| s.to_string()))
291      .unwrap_or_else(|| "text".to_string());
292    (fs, ms)
293  } else {
294    let f = lookup_font().unwrap();
295    let fs = f.get_size().unwrap_or(10.0);
296    let ms = f
297      .mathstyle
298      .as_deref()
299      .map(|s| s.to_string())
300      .unwrap_or_else(|| "text".to_string());
301    (fs, ms)
302  };
303  let (_wb, hb, db) = if let Some(Stored::Digested(ref base)) = base_opt {
304    let base_size = base.clone().get_size(None)?;
305    (
306      base_size.0.value_of() as f64,
307      base_size.1.value_of() as f64,
308      base_size.2.value_of() as f64,
309    )
310  } else {
311    let nominal_size = lookup_font().unwrap().get_nominal_size();
312    (
313      nominal_size.0.value_of() as f64,
314      nominal_size.1.value_of() as f64,
315      nominal_size.2.value_of() as f64,
316    )
317  };
318  let w;
319  let (mut h, mut d) = (0.0, 0.0);
320  let cmsy_size = base_font_size as i64;
321  let cmsy_name = format!("cmsy{}", cmsy_size);
322  let get_font_dimen = |param: usize| -> f64 {
323    let lookup = |name: &str| -> Option<f64> {
324      STDMETRICS.get(name).and_then(|m| {
325        if param > 0 && param <= m.parameters.len() {
326          Some(m.parameters[param - 1])
327        } else {
328          None
329        }
330      })
331    };
332    lookup(&cmsy_name).or_else(|| lookup("cmsy")).unwrap_or(0.0) * base_font_size
333  };
334  let xheight = get_font_dimen(5);
335  let inferred_pos = if let Some(Stored::Digested(ref base)) = base_opt {
336    let base_pos = base
337      .get_property("scriptpos")
338      .map(|s| s.to_string())
339      .unwrap_or_default();
340    if base_pos.is_empty() {
341      Cow::Borrowed("post")
342    } else {
343      let stripped: String = base_pos
344        .chars()
345        .take_while(|c| !c.is_ascii_digit())
346        .collect();
347      Cow::Owned(if stripped.is_empty() {
348        base_pos
349      } else {
350        stripped
351      })
352    }
353  } else {
354    Cow::Borrowed("post")
355  };
356  if inferred_pos == "mid" {
357    w = (ws - _wb).max(0.0);
358    if op == "SUPERSCRIPT" {
359      h = hb + ds + hs;
360    } else {
361      d = db + hs + ds;
362    }
363  } else {
364    let wp = if let Some(Stored::Digested(ref prev)) = prev_opt {
365      prev.get_width(None)?.unwrap_or_default().value_of() as f64
366    } else {
367      0.0
368    };
369    let scriptspace = state::lookup_register("\\scriptspace", Vec::new())
370      .ok()
371      .flatten()
372      .map(|rv| match rv {
373        RegisterValue::Dimension(d) => d.value_of() as f64,
374        _ => 32768.0,
375      })
376      .unwrap_or(32768.0);
377    ws += scriptspace;
378    w = (ws - wp).max(0.0);
379    if op == "SUPERSCRIPT" {
380      let supshift = get_font_dimen(match mathstyle.as_str() {
381        "display" => 13,
382        "scriptscript" => 15,
383        _ => 14,
384      });
385      h = hb.max(hs + (ds + xheight / 4.0).max(supshift));
386    } else {
387      let subshift = get_font_dimen(16);
388      d = db.max(ds + (hs - xheight * 0.8).max(subshift));
389    }
390  }
391  Ok((
392    Dimension::new_f64(w),
393    Dimension::new_f64(h),
394    Dimension::new_f64(d),
395  ))
396}

2015/// Adjust a TBox's font mathstyle using Font::merge to trigger size recalculation
2016fn adjust_mathstyle_internal(outerstyle: &str, tbox: &mut Tbox) {
2017  let origstyle_owned = tbox
2018    .font
2019    .mathstyle
2020    .as_deref()
2021    .unwrap_or("display")
2022    .to_string();
2023  let newstyle = mathstyle_adjust(outerstyle, &origstyle_owned);
2024  if newstyle != origstyle_owned {
2025    // Use Font::merge to trigger size recalculation via STYLE_SIZE mapping
2026    let merge_font = Font {
2027      mathstyle: Some(Cow::Borrowed(newstyle)),
2028      ..Font::default()
2029    };
2030    let merged = tbox.font.merge(merge_font);
2031    tbox.font = Rc::new(merged);
2032  }
2033}

2035/// Adjust a Whatsit's font mathstyle, returning the new style if it had a mathstyle property
2036fn adjust_mathstyle_internal_whatsit(outerstyle: &str, whatsit: &mut Whatsit) -> Option<String> {
2037  if let Some(Stored::Font(ref font)) = whatsit.properties.get("font") {
2038    let origstyle = font.mathstyle.as_deref().unwrap_or("display").to_string();
2039    let newstyle = mathstyle_adjust(outerstyle, &origstyle);
2040    if newstyle != origstyle {
2041      let merge_font = Font {
2042        mathstyle: Some(Cow::Borrowed(newstyle)),
2043        ..Font::default()
2044      };
2045      let merged = font.merge(merge_font);
2046      whatsit
2047        .properties
2048        .insert("font", Stored::Font(Rc::new(merged)));
2049    }
2050  }
2051  // If whatsit has a recorded mathstyle property, adjust it too
2052  if let Some(Stored::String(ms)) = whatsit.properties.get("mathstyle") {
2053    // mathstyle_adjust takes &str and returns &'static str, so we can
2054    // compute the adjusted style inside arena::with without allocating
2055    // an owned String for the interned mathstyle. The &'static str
2056    // escapes the closure cleanly.
2057    let newstyle: &'static str = arena::with(*ms, |ms_str| mathstyle_adjust(outerstyle, ms_str));
2058    whatsit
2059      .properties
2060      .insert("mathstyle", Stored::String(arena::pin(newstyle)));
2061    Some(newstyle.to_string())
2062  } else {
2063    None
2064  }
2065}
