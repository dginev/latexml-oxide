9/// Perl's mergeLimits (TeX_Math.pool.ltxml): walks backward through the
10/// digest list, extracts any existing script level from the previous
11/// `scriptpos` value, and sets `scriptpos` to `pos` + level.
12fn merge_limits(pos: &str) {
13  // is_script is now defined in this file (moved from tex_scripts.rs)
14  // Compute script level before borrowing the box list mutably,
15  // since get_script_level() also borrows the stomach.
16  let default_level = get_script_level().to_string();
17  with_box_list_mut(|list| {
18    for b in list.iter_mut().rev() {
19      // Extract trailing level digits from existing scriptpos. Find the
20      // first non-digit from the end, slice off the tail — skips the
21      // chars.rev.take_while.collect.into_iter.rev.collect two-Vec
22      // round-trip the previous code did.
23      let prev = b
24        .get_property("scriptpos")
25        .map(|v| v.to_string())
26        .unwrap_or_default();
27      let digit_start = prev
28        .bytes()
29        .rposition(|c| !c.is_ascii_digit())
30        .map(|i| i + 1)
31        .unwrap_or(0);
32      let level_str: &str = &prev[digit_start..];
33      let level = if level_str.is_empty() {
34        &default_level
35      } else {
36        level_str
37      };
38      b.set_property("scriptpos", format!("{pos}{level}"));
39      // Perl: last unless IsEmpty($box) || IsScript($box)
40      // Continue past empty boxes AND script boxes
41      if !b.is_empty().unwrap_or(false) && is_script(b).is_none() {
42        break;
43      }
44    }
45  });
46}

52static SCRIPT_NAME_RE: Lazy<Regex> =
53  Lazy::new(|| Regex::new(r"^\\lx@(floating|post)@(subscript|superscript)$").unwrap());

55/// Remember a "safe" way to test a script Whatsit.
56/// Returns [ (FLOATING|POST) , (SUBSCRIPT|SUPERSCRIPT) ] or nothing
57pub fn is_script(object: &Digested) -> Option<(String, Catcode)> {
58  let box_opt = match object.data() {
59    DigestedData::List(obj) => obj.borrow().boxes.last().map(|v| Cow::Owned(v.clone())),
60    _ => Some(Cow::Borrowed(object)),
61  };
62  if let Some(boxobj) = box_opt {
63    if let DigestedData::Whatsit(obj) = boxobj.data() {
64      obj.borrow().get_definition().get_cs().with_cs_name(|name| {
65        SCRIPT_NAME_RE.captures(name).map(|cap| {
66          (
67            cap.get(1).map_or("", |m| m.as_str()).to_uppercase(),
68            if cap.get(2).map_or("", |m| m.as_str()) == "subscript" {
69              Catcode::SUB
70            } else {
71              Catcode::SUPER
72            },
73          )
74        })
75      })
76    } else {
77      None
78    }
79  } else {
80    None
81  }
82}

84/// True if the script body is a space-like box/whatsit (or a list
85/// containing one). Mirrors Perl's `IsEmpty` (Package.pm:1029), which —
86/// unlike our `Digested::is_empty`/`Whatsit::is_empty` — does NOT count an
87/// `isSpace` whatsit as empty: a `Whatsit` with no `content_box` property
88/// returns `0` (not empty) there regardless of `isSpace`. So a math space
89/// (`\,` / `\ `, a TBox) AND a strut/hint (`\mathstrut` / `\vphantom`, a
90/// `DefConstructor` whatsit carrying `isSpace=true`) inside `^{...}` /
91/// `_{...}` are both valid, non-empty script bodies. Without the whatsit
92/// arm, `x_e{}^{\mathstrut}_{t}` dropped the empty `^{\mathstrut}` floating
93/// superscript, consumed the `{}` separator, and then `_{t}` re-attacked
94/// the earlier `_e` for a spurious "Double subscript" (witness 1803.08859,
95/// `\vort_e{}^{\mathstrut}_{t}`; cluster across 1904.07182 etc.).
96fn script_has_space_content(d: &Digested) -> bool {
97  use latexml_core::digested::DigestedData;
98  match d.data() {
99    DigestedData::TBox(b) => b.borrow().get_property_bool("isSpace"),
100    DigestedData::Whatsit(w) => w.borrow().get_property_bool("isSpace"),
101    DigestedData::List(l) => l.borrow().boxes.iter().any(script_has_space_content),
102    _ => false,
103  }
104}

106fn script_handler(cc: Catcode) -> Result<Vec<Digested>> {
107  let font = lookup_font().unwrap();
108  if font.get_mathstyle().is_some() {
109    let mut putback = VecDeque::new();
110    let mut nscripts = 0;
111
112    let mut cs = if cc == Catcode::SUPER {
113      "\\lx@floating@superscript"
114    } else {
115      "\\lx@floating@subscript"
116    };
117    let mut prevscript = None;
118    let mut prevspace = false;
119    let mut base = None;
120    while let Some(prev) = { pop_box_list() } {
121      if prev.get_property_bool("isSpace") || prev.get_property_bool("isEmpty") {
122        prevspace = true;
123        putback.push_front(prev);
124        continue;
125      } else if prev.is_empty()? {
126        // Perl `TeX_Math.pool.ltxml:378-379`: "If empty, the script floats,
127        // can't conflict, but don't put back". `cs` stays at `\lx@floating@*`
128        // (initial value); the empty `{}` is consumed.
129        //
130        // BUT a box that is `is_empty()` only because its sole content is a
131        // phantom/strut (an isSpace XMHint wrapped in a group, e.g.
132        // `{\vphantom{x}}`) must be treated like a space — put back, set
133        // `prevspace` — NOT discarded. In Perl that group reports
134        // `getProperty('isSpace')` (caught above); our List wrapper carries
135        // the isSpace on the inner whatsit only, so detect it recursively.
136        // Discarding it lets a following `_x` re-attack the real base and
137        // fire a spurious "Double subscript": witness 1904.07182
138        // (`\ibraket`/`\mprescript` → `… {\vphantom{x}}^{}_{i}`), the cluster
139        // already noted on `script_has_space_content` above.
140        if script_has_space_content(&prev) {
141          prevspace = true;
142          putback.push_front(prev);
143          continue;
144        }
145        break;
146      } else if let Some(prevop) = is_script(&prev) {
147        if prevop.1 == cc {
148          putback.push_front(prev);
149          let lcode = if prevop.1 == Catcode::SUPER {
150            "superscript"
151          } else {
152            "subscript"
153          };
154          if !prevspace {
155            Error!("unexpected", s!("double-{lcode}"), s!("Double {lcode}"));
156          }
157          cs = if cc == Catcode::SUPER {
158            "\\lx@floating@superscript"
159          } else {
160            "\\lx@floating@subscript"
161          };
162          break;
163        } else {
164          prevscript = Some(prev.clone());
165          putback.push_front(prev);
166          cs = if cc == Catcode::SUPER {
167            "\\lx@post@superscript"
168          } else {
169            "\\lx@post@subscript"
170          };
171        }
172        if prevop.0 == "FLOATING" {
173          break;
174        }
175        nscripts += 1;
176        if nscripts > 1 {
177          break;
178        }
179      } else {
180        base = Some(prev.clone());
181        putback.push_front(prev);
182        cs = if cc == Catcode::SUPER {
183          "\\lx@post@superscript"
184        } else {
185          "\\lx@post@subscript"
186        };
187        break;
188      }
189    }
190    extend_box_list(putback);
191    MergeFont!(scripted => true);
192    let mut stuff = Vec::new();
193    while let Some(tok) = read_x_token(Some(false), false, None)? {
194      stuff = invoke_token(&tok)?;
195      if !stuff.is_empty() {
196        break;
197      }
198    }
199    if stuff.is_empty() {
200      Error!("expected", "{", "Missing sub/superscript argument");
201      stuff.push(Digested::default());
202    }
203    let script = stuff.remove(0);
204
205    // Perl L416: `unless IsEmpty($script)`. Our `Digested::is_empty()`
206    // considers any TBox with `isSpace=true` empty — but in math mode
207    // `\,` / `\ ` etc. produce isSpace TBoxes that ARE meaningful
208    // script bodies (visible width, valid placeholder). Skipping the
209    // whatsit there lets a subsequent `^{...}_x` or `_x` re-attack the
210    // base and fire a spurious double-script error. Treat isSpace=true
211    // scripts as non-empty so they consume the script slot.
212    let script_is_truly_empty = script.is_empty()? && !script_has_space_content(&script);
213    if !script_is_truly_empty {
214      let mut properties = {
215        stored_map!(
216          "isMath" => true,
217          "base"        => if let Some(b) = base { Stored::Digested(b) }
218            else { Stored::None },
219          "scriptlevel" => get_script_level(),
220          "level"       => get_boxing_level()
221        )
222      };
223      if let Some(pvs) = prevscript {
224        properties.insert("prevscript", pvs.into());
225      }
226      if let Some(Stored::Digested(b)) = properties.get("base")
227        && let Some(bsp) = b.get_property("scriptpos")
228      {
229        let bsp_str = bsp.to_string();
230        if !bsp_str.is_empty() {
231          let base_prefix: String = bsp_str
232            .chars()
233            .take_while(|c| !c.is_ascii_digit())
234            .collect();
235          let sl = get_script_level();
236          properties.insert("scriptpos", Stored::from(format!("{base_prefix}{sl}")));
237        }
238      }
239      if let Some(font) = script.get_font()? {
240        properties.insert("font", font.into());
241      }
242      let mut with_script = vec![Digested::from(Whatsit {
243        definition: lookup_definition(&T_CS!(cs))?.unwrap(),
244        args: vec![Some(script)],
245        properties,
246        ..Whatsit::default()
247      })];
248      with_script.extend(stuff);
249      stuff = with_script;
250    }
251    assign_font(font, Some(Scope::Local)); // revert
252    Ok(stuff)
253  } else {
254    let c = if cc == Catcode::SUPER { '^' } else { '_' };
255    // Perl-faithful lowercase category — matches TeX_Math.pool.ltxml:424
256    // `Error('unexpected', $c, $stomach, "Script $c can only appear in math
257    // mode")` and the rest of the engine's lowercase categories. Was a typo.
258    Error!(
259      "unexpected",
260      c,
261      format!("Script {c} can only appear in math mode")
262    );
263    let placeholder = if cc == Catcode::SUPER {
264      T_SUPER!()
265    } else {
266      T_SUB!()
267    };
268    Ok(vec![Digested::from(Tbox::new(
269      pin_char(c),
270      None,
271      None,
272      Tokens!(placeholder),
273      SymHashMap::default(),
274    ))])
275  }
276}

278pub fn revert_script(script: &Digested) -> Result<Vec<Token>> {
279  let tokens = script.revert()?;
280  let mut ts = tokens.unlist();
281  if ts.len() > 1
282    && ts.first().unwrap().code == Catcode::BEGIN
283    && ts.last().unwrap().code == Catcode::END
284  {
285    Ok(ts)
286  } else {
287    let mut wrapped = vec![T_BEGIN!()];
288    wrapped.append(&mut ts);
289    wrapped.push(T_END!());
290    Ok(wrapped)
291  }
292}

294/// Fraction sizer — TeX-style width/height/depth for a fraction whose
295/// numerator and denominator are `top` and `bottom` digested boxes.
296///
297/// Perl: `fracSizer` in TeX_Math.pool.ltxml L1054-1059:
298///   w = max(numerator.width, denominator.width)
299///   d = denominator.total_height * 0.5
300///   h = numerator.total_height + d
301///
302/// Used by `\lx@generalized@over`, `\over`, `\atop`, `\above*` etc.
303pub fn frac_sizer(top: &Digested, bottom: &Digested) -> Result<(Dimension, Dimension, Dimension)> {
304  let (tw, th, td, ..) = top.clone().get_size(None)?;
305  let (bw, bh, bd, ..) = bottom.clone().get_size(None)?;
306  // width: max of top and bottom widths
307  let w = Dimension(tw.value_of().max(bw.value_of()));
308  // depth: half of denominator's total height (height + depth)
309  let bot_total = bh.value_of() + bd.value_of();
310  let d = Dimension(bot_total / 2);
311  // height: numerator total height + depth
312  let top_total = th.value_of() + td.value_of();
313  let h = Dimension(top_total + d.value_of());
314  Ok((w, h, d))
315}

317fn script_sizer(
318  script: &Digested,
319  base_opt: Option<&Stored>,
320  prev_opt: Option<&Stored>,
321  op: &str,
322  _pos: &str,
323) -> Result<(Dimension, Dimension, Dimension)> {
324  let script_size = script.clone().get_size(None)?;
325  let (mut ws, hs, ds) = (
326    script_size.0.value_of() as f64,
327    script_size.1.value_of() as f64,
328    script_size.2.value_of() as f64,
329  );
330  let (base_font_size, mathstyle) = if let Some(Stored::Digested(base)) = base_opt {
331    let bfont = base.get_font()?.map(|f| f.into_owned());
332    let fs = bfont.as_ref().and_then(|f| f.get_size()).unwrap_or(10.0);
333    let ms = bfont
334      .as_ref()
335      .and_then(|f| f.mathstyle.as_deref().map(|s| s.to_string()))
336      .unwrap_or_else(|| "text".to_string());
337    (fs, ms)
338  } else {
339    let f = lookup_font().unwrap();
340    let fs = f.get_size().unwrap_or(10.0);
341    let ms = f
342      .mathstyle
343      .as_deref()
344      .map(|s| s.to_string())
345      .unwrap_or_else(|| "text".to_string());
346    (fs, ms)
347  };
348  let (_wb, hb, db) = if let Some(Stored::Digested(base)) = base_opt {
349    let base_size = base.clone().get_size(None)?;
350    (
351      base_size.0.value_of() as f64,
352      base_size.1.value_of() as f64,
353      base_size.2.value_of() as f64,
354    )
355  } else {
356    let nominal_size = lookup_font().unwrap().get_nominal_size();
357    (
358      nominal_size.0.value_of() as f64,
359      nominal_size.1.value_of() as f64,
360      nominal_size.2.value_of() as f64,
361    )
362  };
363  let w;
364  let (mut h, mut d) = (0.0, 0.0);
365  let cmsy_size = base_font_size as i64;
366  let cmsy_name = format!("cmsy{}", cmsy_size);
367  let get_font_dimen = |param: usize| -> f64 {
368    let lookup = |name: &str| -> Option<f64> {
369      STDMETRICS.get(name).and_then(|m| {
370        if param > 0 && param <= m.parameters.len() {
371          Some(m.parameters[param - 1])
372        } else {
373          None
374        }
375      })
376    };
377    lookup(&cmsy_name).or_else(|| lookup("cmsy")).unwrap_or(0.0) * base_font_size
378  };
379  let xheight = get_font_dimen(5);
380  let inferred_pos = if let Some(Stored::Digested(base)) = base_opt {
381    let base_pos = base
382      .get_property("scriptpos")
383      .map(|s| s.to_string())
384      .unwrap_or_default();
385    if base_pos.is_empty() {
386      Cow::Borrowed("post")
387    } else {
388      let stripped: String = base_pos
389        .chars()
390        .take_while(|c| !c.is_ascii_digit())
391        .collect();
392      Cow::Owned(if stripped.is_empty() {
393        base_pos
394      } else {
395        stripped
396      })
397    }
398  } else {
399    Cow::Borrowed("post")
400  };
401  if inferred_pos == "mid" {
402    w = (ws - _wb).max(0.0);
403    if op == "SUPERSCRIPT" {
404      h = hb + ds + hs;
405    } else {
406      d = db + hs + ds;
407    }
408  } else {
409    let wp = if let Some(Stored::Digested(prev)) = prev_opt {
410      prev.get_width(None)?.unwrap_or_default().value_of() as f64
411    } else {
412      0.0
413    };
414    let scriptspace = lookup_register("\\scriptspace", Vec::new())
415      .ok()
416      .flatten()
417      .map(|rv| match rv {
418        RegisterValue::Dimension(d) => d.value_of() as f64,
419        _ => 32768.0,
420      })
421      .unwrap_or(32768.0);
422    ws += scriptspace;
423    w = (ws - wp).max(0.0);
424    if op == "SUPERSCRIPT" {
425      let supshift = get_font_dimen(match mathstyle.as_str() {
426        "display" => 13,
427        "scriptscript" => 15,
428        _ => 14,
429      });
430      h = hb.max(hs + (ds + xheight / 4.0).max(supshift));
431    } else {
432      let subshift = get_font_dimen(16);
433      d = db.max(ds + (hs - xheight * 0.8).max(subshift));
434    }
435  }
436  Ok((
437    Dimension::new_f64(w),
438    Dimension::new_f64(h),
439    Dimension::new_f64(d),
440  ))
441}

443LoadDefinitions!({
444  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
445  // Math Family of primitive control sequences
446  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
447
448  //======================================================================
449  // NOT YET IMPLEMENTED !?!?!
450  //----------------------------------------------------------------------
451  // \radical                c  makes a radical atom from the delimiter (27-bit number) and the math
452  // field. \muskipdef              c  creates a symbolic name for a \muskip register.
453  // \muskip                 iq assigns <muglue> to a \muskip register.
454  // \nonscript              c  ignores immediately following glue or kern in script and
455  // scriptscript styles.
456  // Should discard following skip/glue; for now a no-op stub.
457  DefPrimitive!("\\nonscript", None);
458
459  //======================================================================
460  // The next two sections are the basic LaTeXML Infrastructure for math.
461  // There are several internal control sequences which need to be renamed!
462  //======================================================================
463
464  // Decide whether we're going into or out of math, inline or display.
465  Tag!("ltx:XMText", auto_open => true, auto_close => true);
466  // This really should be T_MATH
467  // and it should (or not) check for a second $ only if not in restricted horizontal mode!
468  // (and then all the \lx@dollar@in@(text|math|normal)mode defns would not be needed.
469  DefPrimitive!(T_CS!("\\lx@dollar@default"), None, {
470    let mut op = "\\lx@begin@inline@math";
471    {
472      let mode = lookup_string_from_sym(pin!("MODE"));
473      if mode == "display_math" {
474        if if_next(T_MATH!())? {
475          read_token()?;
476          op = "\\lx@end@display@math";
477        } else {
478          Error!(
479            "expected",
480            "$",
481            "Missing $ closing display math.\nIgnoring; expect to be in wrong math/text mode."
482          );
483          op = "";
484        }
485      } else if mode == "math" {
486        // Perl L62: `elsif ($mode eq 'math')` — because beginMode("inline_math")
487        // maps to MODE="math" via bindable_mode. NOT "inline_math".
488        op = "\\lx@end@inline@math";
489      } else {
490        // Perl: only check for $$ when within a vertical bound mode
491        let bound = lookup_string_from_sym(pin!("BOUND_MODE"));
492        if bound.ends_with("vertical") && if_next(T_MATH!())? {
493          read_token()?;
494          op = "\\lx@begin@display@math";
495        }
496      }
497    }
498    if !op.is_empty() {
499      Ok(invoke_token(&T_CS!(op))?)
500    } else {
501      Ok(Vec::new())
502    }
503  });
504  // Let this be the default, conventional $
505  Let!(T_MATH!(), T_CS!("\\lx@dollar@default"));
506  // Obsolete aliases
507  Let!("\\lx@dollar@in@normalmode", "\\lx@dollar@default");
508
509  //======================================================================
510  // Math mode in special cases: math alignments, or perverse equations for ...$text$...
511  // In Perl, \lx@dollar@in@textmode is now aliased to \lx@dollar@default.
512  Let!("\\lx@dollar@in@textmode", "\\lx@dollar@default");
513  // Note that $ within a math alignment (eg array environment),
514  // switches to text mode! There's no $$ for display math.
515
516  // This one is for $ appearing within an alignment that's already math.
517  // This should switch to text mode (because it's balancing the hidden $
518  // wrapping each alignment cell!!!!!!)
519  // However, it should be like a normal $ if it's inside something like \mbox
520  // that itself makes a text box!!!!!!
521  // Thus, we need to know at what boxing level we started the last math or text.
522  // This is all complicated by the need to know _how_ we got into or out of math mode!
523  // Gawd, this is awful!
524  // NOTE: Probably the most "Right" thing to do would be to process
525  // alignments in text mode only (like TeX), sneaking $'s in where needed,
526  // but then afterwards, morph them into math arrays?
527  // This would be complicated by the need to hide these $ from untex.
528  DefPrimitive!(T_CS!("\\lx@dollar@in@mathmode"), None, {
529    let level = get_boxing_level();
530    if lookup_int("MATH_ALIGN_$_BEGUN") == (level as i64) {
531      // If we're begun making _something_ with $.
532      let l = if lookup_bool_sym(pin!("IN_MATH")) {
533        // But we're somehow in math?
534        invoke_token(&T_CS!("\\lx@end@inline@math"))
535      } else {
536        invoke_token(&T_CS!("\\lx@end@inmath@text"))
537      };
538      assign_value("MATH_ALIGN_$_BEGUN", 0, None); // Reset this AFTER finishing the something
539      l
540    } else {
541      assign_value("MATH_ALIGN_$_BEGUN", level + 1, None); // Note that we've begun something
542      if lookup_bool_sym(pin!("IN_MATH")) {
543        // If we're "still" in math
544        invoke_token(&T_CS!("\\lx@begin@inmath@text"))
545      } else {
546        invoke_token(&T_CS!("\\lx@begin@inline@math"))
547      }
548    }
549  });
550  //======================================================================
551  // For inserting (non-trivial?) text while in math mode
552  DefConstructor!("\\lx@begin@inmath@text",
553    "<ltx:XMText>#body</ltx:XMText>",
554    // alias => T_MATH ? do we support that ?
555    alias => "$",
556    // Perl: beginMode('restricted_horizontal') — NOT 'text'
557    before_digest => sub { begin_mode("restricted_horizontal")?; },
558    capture_body => true
559  );
560  DefConstructor!("\\lx@end@inmath@text", "", alias => "$",
561    before_digest => sub { end_mode("restricted_horizontal")?; });
562  //======================================================================
563  // Effectively these are the math hooks, redefine these to do what you want with math?
564  // Perl TeX_Math.pool.ltxml L124-137 — DefConstructorI baseline. NOTE: this is the
565  // TeX-pool version without xml:id; latex_constructs.rs:4953 redefines it with
566  // xml:id='#id' and RefStepID('equation') for LaTeX numbering.
567  DefConstructor!("\\lx@begin@display@math",
568    "<ltx:equation><ltx:Math mode='display'><ltx:XMath>#body</ltx:XMath></ltx:Math></ltx:equation>",
569    reversion    => Tokens!(T_MATH!(), T_MATH!()),
570    before_digest => {
571      // Perl: $_[0]->enterHorizontal; (TeX_Math.pool.ltxml line 134)
572      enter_horizontal();
573      // begin_mode handles \everydisplay injection (Stomach.pm lines 504-507)
574      begin_mode("display_math")?;

1779  // Rewrite: a floating super/subscript that is really text — a plain-text
1780  // frontmatter mark, or a text-mode `\mbox{…}` of rendered markup (e.g. an
1781  // ORCID logo) — into an HTML ltx:sup/ltx:sub rather than empty-base math.
1782  //
1783  // Performance note (R35.D, 2026-05-22). The original XPath included
1784  // a `not(./*/*[not(self::ltx:XMTok)])` predicate (every XMApp
1785  // grandchild must be an XMTok). libxml2 evaluates this with a
1786  // walk over each candidate's grandchildren PER candidate Math
1787  // element, which on gr-qc0209055 (witness paper for R35.D) ran
1788  // past the 120s wall-clock budget.
1789  //
1790  // Keep the parent-chain + sibling-absence predicates in XPath
1791  // (those are cheap: structural axis checks at the candidate node),
1792  // but lift the only-XMTok-grandchildren check into the Rust
1793  // replace closure. The closure already walks the same tree to
1794  // extract content; doing the check there costs O(grandchildren)
1795  // ONCE per actual match, instead of `O(N×grandchildren)` walking
1796  // the entire document's Math tree for each candidate.
1797  //
1798  // On the witness paper, this dropped rule-#0 wall time from >80s
1799  // to ~150ms — the rule's structural check is now lazy.
1800  DefRewrite!(xpath =>
1801    concat!(
1802      "descendant::ltx:Math[descendant::ltx:XMApp[",
1803      "@role='FLOATSUPERSCRIPT' or @role='FLOATSUBSCRIPT']]"
1804    ),
1805    replace => sub[document, nodes] {
1806      let math = nodes.pop().unwrap();
1807      let mut replaced = false;
1808      // Structural guard previously inlined in the XPath:
1809      // 1. Math contains exactly one XMath child
1810      // 2. XMath contains exactly one XMApp child (this is the FLOATSUPER...)
1811      // 3. XMApp has no preceding/following siblings (covered by #2)
1812      // 4. XMApp contains exactly one XMArg, whose grandchildren are all XMTok
1813      // If any check fails, re-attach the original `math` so the rewrite
1814      // engine's Replace removal is undone — this rule is opt-in by shape.
1815      let restore = |document: &mut Document,
1816                     math: &mut Node|
1817       -> Result<()> {
1818        // The active insertion point is the parent (set via
1819        // `document.set_node(&parent)` before the closure runs).
1820        // Re-attach the original math node as a child so the rewrite
1821        // engine's `inserted` collection picks it up cleanly.
1822        document.get_node_mut().add_child(math).map(|_| ())?;
1823        Ok(())
1824      };
1825      let xmath_children: Vec<Node> = math.get_child_nodes().into_iter()
1826        .filter(|n| n.get_type() == Some(NodeType::ElementNode)).collect();
1827      if xmath_children.len() != 1 { return restore(document, math); }
1828      if let Some(xmath) = xmath_children.first() {
1829        let xmapp_children: Vec<Node> = xmath.get_child_nodes().into_iter()
1830          .filter(|n| n.get_type() == Some(NodeType::ElementNode)).collect();
1831        if xmapp_children.len() != 1 { return restore(document, math); }
1832        if let Some(xmapp) = xmapp_children.first() {
1833          let role = xmapp.get_attribute("role").unwrap_or_default();
1834          if role != "FLOATSUPERSCRIPT" && role != "FLOATSUBSCRIPT" {
1835            return restore(document, math);
1836          }
1837          let xmarg_children: Vec<Node> = xmapp.get_child_nodes().into_iter()
1838            .filter(|n| n.get_type() == Some(NodeType::ElementNode)).collect();
1839          if xmarg_children.len() != 1 { return restore(document, math); }
1840          let qname = if role == "FLOATSUPERSCRIPT" { "ltx:sup" } else { "ltx:sub" };
1841          let content_box = xmarg_children.first().unwrap();
1842          // Locate a text-mode box (`ltx:XMText`, from `\mbox{…}`): either the
1843          // float's argument itself, or — the argument is still wrapped at
1844          // rewrite time — the sole XMText inside its XMArg wrapper, ignoring
1845          // any spacing hints (the `\,` in `\orcidID` becomes an XMHint sibling).
1846          let text_box = if document::with_node_qname(content_box, |q| q == "ltx:XMText") {
1847            Some(content_box.clone())
1848          } else {
1849            let inner: Vec<Node> = content_box.get_child_nodes().into_iter()
1850              .filter(|n| n.get_type() == Some(NodeType::ElementNode))
1851              .filter(|n| !document::with_node_qname(n, |q| q == "ltx:XMHint"))
1852              .collect();
1853            match inner.first() {
1854              Some(n) if inner.len() == 1
1855                && document::with_node_qname(n, |q| q == "ltx:XMText") => Some(n.clone()),
1856              _ => None,
1857            }
1858          };
1859          // A text-mode box carrying rendered elements rather than math tokens
1860          // is not math at all. LNCS `\orcidID` expands to
1861          // `$^{\,\mbox{\scriptsize\orcidlink{…}}}$` — an ORCID logo `<ref>`/`<svg>`;
1862          // the plain-text path below (`get_content()`) would drop that markup
1863          // and leave an empty script, so move the box's children verbatim into
1864          // the new sup/sub instead. Witness 2605.16562.
1865          let markup_box = text_box.filter(|tb| {
1866            tb.get_child_nodes().into_iter()
1867              .filter(|n| n.get_type() == Some(NodeType::ElementNode))
1868              .any(|gc| !document::with_node_qname(&gc, |q| q == "ltx:XMTok"))
1869          });
1870          if let Some(tb) = markup_box {
1871            document.open_element(qname, None, None)?;
1872            for mut child in tb.get_child_nodes() {
1873              // unlink BEFORE add_child: libxml2 < 2.13 does NOT unlink
1874              // internally — moving a still-linked node corrupts both child
1875              // chains (double-visit/double-free on legacy distros). Every
1876              // other move site in the codebase unlinks first
1877              // (PR_READINESS must-fix 7).
1878              child.unlink_node();
1879              document.get_node_mut().add_child(&mut child)?;
1880            }
1881            document.close_element(qname)?;
1882            replaced = true;
1883          } else {
1884            // Original frontmatter case: a plain-text float (all XMTok), e.g. an
1885            // author affiliation mark. Extract its text into the sup/sub.
1886            let only_xmtok_grandchildren = xmarg_children.iter().all(|c| {
1887              c.get_child_nodes().into_iter()
1888                .filter(|n| n.get_type() == Some(NodeType::ElementNode))
1889                .all(|gc| document::with_node_qname(&gc, |q| q == "ltx:XMTok"))
1890            });
1891            if !only_xmtok_grandchildren { return restore(document, math); }
1892            if let Some(xmarg) = xmarg_children.first() {
1893              let text = xmarg.get_content();
1894              let font_attr = {
1895                let from_attr = xmarg.get_child_nodes().into_iter()
1896                  .filter(|n| n.get_type() == Some(NodeType::ElementNode))
1897                  .find_map(|n| {
1898                    let attr = n.get_attribute("font");
1899                    if attr.is_some() { return attr; }
1900                    let node_font = document.get_node_font(&n);
1901                    node_font.get_shape().and_then(|s|
1902                      if s.as_ref() == "italic" { Some("italic".to_string()) } else { None }
1903                    )
1904                  });
1905                if from_attr.is_some() {
1906                  from_attr
1907                } else {
1908                  document.get_node_box(xmarg).and_then(|tbox| {
1909                    tbox.get_font().ok().flatten().and_then(|font| {
1910                      if font.get_family().map(|f| f.as_ref() == "math").unwrap_or(false) {
1911                        Some("italic".to_string())
1912                      } else {
1913                        None
1914                      }
1915                    })
1916                  })
1917                }
1918              };
1919              document.open_element(qname, None, None)?;
1920              if let Some(ref font) = font_attr {
1921                let mut text_node = document.open_element("ltx:text", None, None)?;
1922                document.set_attribute(&mut text_node, "font", font)?;
1923                document.get_node_mut().append_text(&text)?;
1924                document.close_element("ltx:text")?;
1925              } else {
1926                document.get_node_mut().append_text(&text)?;
1927              }
1928              document.close_element(qname)?;
1929              replaced = true;
1930            }
1931          }
1932        }
1933      }
1934      if !replaced {
1935        document.get_node_mut().add_child(math)?;
1936      }
1937    }
1938  );
