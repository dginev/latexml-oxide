11  //======================================================================
12  // Basic kerning
13  //----------------------------------------------------------------------
14  // \kern             c  adds a kern item to the current list.
15  // \unkern           c  removes a kern from the current list.
16  // \lastkern         iq is 0.0 pt or the last kern on the current list.
17
18  // \kern is heavily used by xy.
19  // Note that \kern should add vertical spacing in vertical modes!
20  // Perl: TeX_Kern.pool.ltxml L31-52
21  DefConstructor!("\\kern Dimension", sub[document,args, props] {
22    let length : Dimension = if let DigestedData::RegisterValue(RegisterValue::Dimension(d)) =
23      args[0].as_ref().unwrap().data() {
24        *d
25      } else { Dimension::default() };
26    let is_svg_g = document::with_node_qname(document.get_node(),
27      |qname| qname == "svg:g");
28    if is_svg_g {
29      let x = length.px_value(None);
30      if x != 0.0 {
31        let shift = s!("translate({x},0)");
32        let parent = document.get_node_mut();
33        let has_children = parent.get_first_child().is_some();
34        if has_children {
35          // Perl L37-38: If already have positioned children, open new svg:g
36          let attrs = string_map!("_autoclose" => "1", "transform" => shift);
37          document.open_element("svg:g", Some(attrs), None)?;
38        } else {
39          // Perl L40-41: No children yet — append to parent's transform attribute
40          let prev = parent.get_attribute("transform").unwrap_or_default();
41          let new_transform = if prev.is_empty() { shift } else { s!("{prev} {shift}") };
42          parent.set_attribute("transform", &new_transform)?;
43        }
44      }
45    } else if in_svg(document) {
46      Warn!("unexpected", "kern", s!("Lost kern in SVG {length}"));
47    } else if props.get("isMath") == Some(&Stored::Bool(true)) {
48      document.insert_element("ltx:XMHint", Vec::new(), Some(map!("width" => length.to_attribute())))?;
49    } else {
50      // Add space to document
51      // Use the precise Unicode space mapping from tex_glue (matching Perl's TeX_Glue algorithm),
52      // not the simple threshold version from base_functions.
53      let spaces = super::tex_glue::dimension_to_spaces(length);
54      document.absorb_string(&spaces, &SymHashMap::default())?;
55    }
56  },
57  enter_horizontal => true,
58  properties => sub[args] {
59    unref!(args => length);
60    Ok(stored_map!("width" => length, "isSpace" => true, "isKern" => true))
61  });
62
63  // Remove kern, if last on LIST
64  DefPrimitive!("\\unkern", {
65    let mut comments = Vec::new();
66    // Scan past any Comment boxes
67    while let Some(last_box) = pop_box_list() {
68      if matches!(last_box.data(), DigestedData::Comment(_)) {
69        comments.push(last_box);
70      } else {
71        if !last_box.get_property_bool("isKern") {
72          push_box_list(last_box);
73        }
74        break;
75      }
76    }
77    let comments_rev_iter = comments.into_iter().rev();
78    for comment in comments_rev_iter {
79      push_box_list(comment);
80    }
81  });
82  // Get kern, if last on LIST
83  DefRegister!("\\lastkern" => Dimension::new(0), readonly => true,
84  getter => {
85    with_box_list(|stomach_box_list| {
86      let box_iter = stomach_box_list.iter().rev();
87      for box_in_list in box_iter {
88        if !matches!(box_in_list.data(), DigestedData::Comment(_)) {
89          if box_in_list.get_property_bool("isKern") {
90            let width_stored = box_in_list.get_property("width").unwrap();
91            match &*width_stored {
92              Stored::Dimension(width_d) => return *width_d,
93              Stored::Digested(d) => {
94                if let DigestedData::RegisterValue(RegisterValue::Dimension(dim)) = d.data() {
95                  return *dim;
96                }
97                return Dimension::new(0);
98              }
99              _ => return Dimension::new(0),
100            }
101          } else {
102            break;
103          }
104        }
105      }
106      Dimension::new(0)
107    })
108  });

110  //======================================================================
111  // Moving Vertically
112  //----------------------------------------------------------------------
113  // \raise            c  shifts a box up and appends it to the current horizontal or math list.
114  // \lower            c  shifts a box down and appends it to the current horizontal or math list.
115  // \lower <dimen> <box>
116  // \raise <dimen> <box>
117  // But <box> apparently must really explicitly be an \hbox, \vbox or \vtop (?)
118  // OR something that expands into one!!
119  // Perl: TeX_Kern.pool.ltxml L94-103
120  // Template: ?&inSVG()(<svg:g transform='#transform' _noautoclose='1'>#2</svg:g>)
121  //                    (<ltx:text yoffset='#y'  _noautoclose='1'>#2</ltx:text>)
122  DefConstructor!("\\lower Dimension MoveableBox",
123    sub[document, args, props] {
124      let svg = in_svg(document);
125      // Move the initial attrs map unconditionally into whichever branch
126      // runs — only one executes per call, so no clone is needed.
127      let mut attrs = string_map!("_noautoclose" => "1");
128      let node = if svg {
129        let transform = match props.get("transform") {
130          Some(Stored::String(s)) => to_string(*s), _ => String::new()
131        };
132        if !transform.is_empty() { attrs.insert(String::from("transform"), transform); }
133        document.open_element("svg:g", Some(attrs), None)?
134      } else {
135        let y_attr = match props.get("y") {
136          Some(Stored::Dimension(d)) => d.to_attribute(), _ => String::new()
137        };
138        if !y_attr.is_empty() { attrs.insert(String::from("yoffset"), y_attr); }
139        document.open_element("ltx:text", Some(attrs), None)?
140      };
141      if let Some(Some(content)) = args.get(1) {
142        document.absorb(content, None)?;
143      }
144      document.maybe_close_node(&node)?;
145    },
146    // Perl: sizer => sub { raisedSizer($_[0]->getArg(2), $_[0]->getArg(1)->negate); }
147    sizer => sub[whatsit] {
148      let y_val = -(whatsit.get_arg(1).map(|a| a.value_of()).unwrap_or(0));
149      if let Some(content) = whatsit.get_arg(2) {
150        let (w, h, d) = content.compute_size(Default::default())?;
151        let new_h = Dimension::new((h.value_of() + y_val).max(0));
152        let new_d = Dimension::new((d.value_of() - y_val).max(0));
153        Ok((w, new_h, new_d))
154      } else {
155        Ok((Dimension::new(0), Dimension::new(0), Dimension::new(0)))
156      }
157    },
158    enter_horizontal => true,
159    after_digest => sub[whatsit] {
160      let y         = Dimension(-whatsit.get_arg(1).unwrap().value_of());
161      let ypx       = y.px_value(None);
162      let transform = if ypx != 0.0 { s!("translate(0,{ypx})") } else { String::new() };
163      whatsit.set_property("y", y);
164      whatsit.set_property("transform", transform);
165    }
166  );

168  // Perl: TeX_Kern.pool.ltxml L105-114
169  // Template: ?&inSVG()(<svg:g transform='#transform' _noautoclose='1'>#2</svg:g>)
170  //                    (<ltx:text yoffset='#y'  _noautoclose='1'>#2</ltx:text>)
171  DefConstructor!("\\raise Dimension MoveableBox",
172  sub[document, args, props] {
173    let svg = in_svg(document);
174    let mut attrs = string_map!("_noautoclose" => "1");
175    let node = if svg {
176      let transform = match props.get("transform") {
177        Some(Stored::String(s)) => to_string(*s), _ => String::new()
178      };
179      if !transform.is_empty() { attrs.insert(String::from("transform"), transform); }
180      document.open_element("svg:g", Some(attrs), None)?
181    } else {
182      let y_attr = match props.get("y") {
183        Some(Stored::Dimension(d)) => d.to_attribute(), _ => String::new()
184      };
185      if !y_attr.is_empty() { attrs.insert(String::from("yoffset"), y_attr); }
186      document.open_element("ltx:text", Some(attrs), None)?
187    };
188    if let Some(Some(content)) = args.get(1) {
189      document.absorb(content, None)?;
190    }
191    document.maybe_close_node(&node)?;
192  },
193  // Perl: sizer => sub { raisedSizer($_[0]->getArg(2), $_[0]->getArg(1)); }
194  // Adjusts reported height/depth by the raise amount so \ht/\dp reflect the shift.
195  sizer => sub[whatsit] {
196    let y_val = whatsit.get_arg(1).map(|a| a.value_of()).unwrap_or(0);
197    if let Some(content) = whatsit.get_arg(2) {
198      let (w, h, d) = content.compute_size(Default::default())?;
199      let new_h = Dimension::new((h.value_of() + y_val).max(0));
200      let new_d = Dimension::new((d.value_of() - y_val).max(0));
201      Ok((w, new_h, new_d))
202    } else {
203      Ok((Dimension::new(0), Dimension::new(0), Dimension::new(0)))
204    }
205  },
206  enter_horizontal => true,
207  after_digest => sub[whatsit] {
208    let y         = Dimension(whatsit.get_arg(1).unwrap().value_of());
209    let ypx       = y.px_value(None);
210    let transform = if ypx != 0.0 { s!("translate(0,{ypx})") } else { String::new() };
211    whatsit.set_property("y", y);
212    whatsit.set_property("transform", transform);
213  });

225  enter_horizontal => true,
226  after_digest => sub[whatsit] {
227    if let DigestedData::RegisterValue(d) = whatsit.get_arg(1).unwrap().data() {
228      whatsit.set_property("x", d.clone().multiply(Number::new(-1)));
229    }
230  });
231  // Perl: enterHorizontal => 1
232  DefConstructor!("\\moveright Dimension MoveableBox",
233  "<ltx:text xoffset='#x' _noautoclose='true'>#2</ltx:text>",
234  enter_horizontal => true,
235  after_digest => sub[whatsit] {
236    if let Some(dimension) = whatsit.get_arg(1) {
237      whatsit.set_property("x", dimension.clone());
238    }
239  });
