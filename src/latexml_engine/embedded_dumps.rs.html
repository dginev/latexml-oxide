1//! Compile-time embedded TeXLive kernel dumps, indexed by TeXLive year.
2//!
3//! `latexml_engine/build.rs` scans `resources/dumps/` for versioned dumps
4//! (`plain.YYYY.dump.txt`, `latex.YYYY.dump.txt`, `texlive.YYYY.version`),
5//! **gzips each `*.dump.txt`** into `$OUT_DIR` (DEP-12, 2026-05-18), and
6//! emits `embedded_dumps_manifest.rs` listing every bundled year as
7//! `&'static [u8]` gzip blobs. This module `include!`s that manifest,
8//! exposing year-aware accessors that decompress on demand.
9//!
10//! Why compress: ~4.7× gzip ratio on the record-shaped dump text
11//! (3.8 MB → 805 KB per year). Both years combined drop from
12//! ~7.6 MB to ~870 KB in `.rodata` — a ~6.7 MB binary-size win for a
13//! one-time ~6.5 ms decompression cost at engine bootstrap (latex)
14//! plus ~0.1 ms (plain), measured at 560 MB/s on this machine.
15//!
16//! Two-layer cache:
17//!
18//! 1. **Per-thread** `thread_local!(RefCell<HashMap<(year, kind), &'static str>>)` populated via
19//!    `Box::leak`. Avoids re-decompressing or even re-reading the disk file when the same
20//!    `embedded_*_dump` is called twice (e.g. probe-then-load: `*_embedded_available()` followed by
21//!    the actual loader). Matches the project's thread-local-state idiom — no `Mutex` on the hot
22//!    path.
23//! 2. **Cross-process** disk cache at `$TMPDIR/latexml-oxide-dumps-<hash>/`. First process per
24//!    machine-boot decompresses and atomically writes the plain-text dump; every subsequent process
25//!    reads the file directly (~1 ms on tmpfs) instead of gunzipping again.
26//!
27//! Skipped entirely under `LATEXML_NO_EMBEDDED_DUMP=1` (no embedded
28//! fallback fires) or when an ambient-year disk dump is found earlier
29//! in the resolution chain.
30//!
31//! Selection at runtime:
32//!
33//! 1. If [`crate::dump_paths::detect_ambient_texlive_year`] returns a year that is bundled, use
34//!    that exact year.
35//! 2. Otherwise (no ambient TeXLive, or ambient year not bundled), fall back to the most-recent
36//!    bundled year.
37//! 3. If nothing is bundled at all, return `None`.
38//!
39//! Opt out of the embedded fallback altogether with
40//! `LATEXML_NO_EMBEDDED_DUMP=1` — useful when iterating locally and you
41//! want the binary to surface "no dump available" instead of silently
42//! using a stale embedded snapshot.
43

55static NO_EMBEDDED: Lazy<bool> =
56  Lazy::new(|| std::env::var_os("LATEXML_NO_EMBEDDED_DUMP").is_some());
57
58thread_local! {
59  /// In-thread dedupe so the same `(year, kind)` is decompressed (or
60  /// disk-read) at most once per thread. Keyed by `(year, kind)` where
61  /// `kind` is the static string `"plain"` or `"latex"`. Matches the
62  /// project's thread-local-state convention (see CLAUDE.md "State is
63  /// a thread-local, global, mutable singleton").
64  static IN_THREAD_CACHE: RefCell<HashMap<(u32, &'static str), &'static str>> =
65    RefCell::new(HashMap::default());
66}
67
68/// Per-machine cache dir for decompressed embedded dumps, keyed by the
69/// build-time content hash. Under the system temp dir
70/// (`std::env::temp_dir()` — cross-platform: `$TMPDIR`/`/tmp` on Unix,
71/// `$TMPDIR` on macOS, `%TEMP%`/`%TMP%` on Windows). Properties:
72/// * Always writable (no `$HOME` / CI / read-only-mount issues),
73/// * Usually tmpfs on Linux — RAM-speed reads, faster than gunzip,
74/// * Cleared at the OS's discretion (reboot on Linux/macOS, possibly never on Windows) — natural
75///   cache invalidation when stale, with the per-build content hash giving stable cross-process
76///   reuse for the same binary version.
77static CACHE_DIR: Lazy<PathBuf> = Lazy::new(|| {
78  std::env::temp_dir().join(format!(
79    "latexml-oxide-dumps-{}",
80    EMBEDDED_DUMPS_CONTENT_HASH
81  ))
82});

84/// Pick the embedded entry that best matches `prefer` (typically the
85/// ambient TL year). Returns `(entry, exact_match)`.
86pub(crate) fn select_embedded(prefer: Option<u32>) -> Option<(&'static EmbeddedDumpYear, bool)> {
87  if *NO_EMBEDDED {
88    return None;
89  }
90  let entries = non_empty_entries();
91  if entries.is_empty() {
92    return None;
93  }
94  if let Some(year) = prefer
95    && let Some(e) = entries.iter().find(|e| e.year == year)
96  {
97    return Some((e, true));
98  }
99  // EMBEDDED_DUMPS is sorted descending by build.rs, so first non-empty is
100  // the most-recent year.
101  entries.first().copied().map(|e| (e, false))
102}
103
104fn non_empty_entries() -> Vec<&'static EmbeddedDumpYear> {
105  EMBEDDED_DUMPS
106    .iter()
107    .filter(|e| !e.plain_gz.is_empty() && !e.latex_gz.is_empty())
108    .collect()
109}

111/// Resolve the cached-decompressed dump for `(year, kind)`. Tiered lookup:
112///
113/// 0. **In-process cache** — `Mutex<HashMap>` keyed by `(year, kind)`. Avoids re-allocating on
114///    repeated calls within one process (e.g. `*_embedded_available()` probe followed by the actual
115///    loader).
116/// 1. **Disk cache** at `$TMPDIR/latexml-oxide-dumps-<hash>/<kind>.<year>.dump.txt` (set up on a
117///    prior invocation in this boot cycle). Plain `read_to_string` — typically <1 ms on tmpfs.
118/// 2. **Decompress + persist**: gunzip the bundled blob (~6.5 ms for latex, ~0.1 ms for plain),
119///    then atomically write the result to the cache dir so the next process can take path #1.
120///    Best-effort write — disk-full / EROFS does not fail the conversion.
121///
122/// Returns `None` for empty embedded blobs (build.rs stub when no
123/// dumps were available at compile time) or for unrecoverable gunzip
124/// failures.
125fn decompressed_dump(year: u32, kind: &'static str, gz: &[u8]) -> Option<&'static str> {
126  if gz.is_empty() {
127    return None;
128  }
129
130  // Tier 0: per-thread cache.
131  if let Some(cached) = IN_THREAD_CACHE.with(|c| c.borrow().get(&(year, kind)).copied()) {
132    return Some(cached);
133  }
134
135  let leaked: &'static str = {
136    let cache_path = CACHE_DIR.join(format!("{kind}.{year}.dump.txt"));
137
138    // Tier 1: try disk cache.
139    if let Ok(disk_text) = std::fs::read_to_string(&cache_path) {
140      log::debug!(
141        "[embedded_dumps] {kind} TL{year} loaded from disk cache {}",
142        cache_path.display()
143      );
144      Box::leak(disk_text.into_boxed_str())
145    } else {
146      // Tier 2: gunzip the bundled blob.
147      let mut decoder = flate2::read::GzDecoder::new(gz);
148      let mut buf = String::with_capacity(gz.len() * 5); // ~4.7× ratio + slack
149      if decoder.read_to_string(&mut buf).is_err() {
150        log::warn!(
151          "[embedded_dumps] gunzip failed for {kind} TL{year} ({} bytes); \
152           falling back to no embedded dump",
153          gz.len()
154        );
155        return None;
156      }
157
158      // Tier 3: best-effort atomic persist for the next process.
159      // Atomic = write-to-sibling-temp + rename; on POSIX rename is atomic
160      // within a filesystem, so concurrent first-time writers can't tear.
161      if let Err(e) = write_cache_atomic(&cache_path, &buf) {
162        log::debug!(
163          "[embedded_dumps] disk cache write skipped ({}): {} — falling back to in-process only",
164          cache_path.display(),
165          e
166        );
167      }
168
169      Box::leak(buf.into_boxed_str())
170    }
171  };
172
173  // Publish into the per-thread cache so subsequent calls on this
174  // thread (probe-then-load) skip even the disk read.
175  IN_THREAD_CACHE.with(|c| c.borrow_mut().insert((year, kind), leaked));
176
177  Some(leaked)
178}

180/// Best-effort atomic write of the decompressed dump to the cache
181/// dir. Returns errors silently — the caller continues with the
182/// in-memory copy regardless. Race-safe across platforms: two
183/// parallel first-time writers each write to a unique
184/// pid-suffixed tempfile and then `std::fs::rename` into place.
185///
186/// `std::fs::rename` is atomic for files within one filesystem on
187/// Unix (POSIX rename(2)) and on Windows (MoveFileExW with
188/// REPLACE_EXISTING). The second rename simply overwrites the first
189/// — file content is identical so either outcome is correct.
190fn write_cache_atomic(target: &std::path::Path, content: &str) -> std::io::Result<()> {
191  std::fs::create_dir_all(&*CACHE_DIR)?;
192  let temp_path = target.with_file_name(format!(
193    "{}.tmp.{}",
194    target
195      .file_name()
196      .and_then(|s| s.to_str())
197      .unwrap_or("dump"),
198    std::process::id()
199  ));
200  {
201    let mut fh = std::fs::File::create(&temp_path)?;
202    fh.write_all(content.as_bytes())?;
203    fh.sync_data().ok(); // best-effort durability; not critical
204  }
205  std::fs::rename(&temp_path, target)
206}

208/// Bundled `plain.YYYY.dump.txt` content for the year that best matches
209/// `prefer`. `None` when nothing is bundled or `LATEXML_NO_EMBEDDED_DUMP`
210/// is set. First call per machine-boot decompresses + caches to
211/// `/tmp`; subsequent calls in this AND future processes load the
212/// decompressed file directly from the disk cache.
213pub fn embedded_plain_dump(prefer: Option<u32>) -> Option<&'static str> {
214  let (entry, _) = select_embedded(prefer)?;
215  decompressed_dump(entry.year, "plain", entry.plain_gz)
216}
217
218/// Bundled `latex.YYYY.dump.txt` content for the year that best matches
219/// `prefer`. See [`embedded_plain_dump`] for cache semantics. First
220/// call per machine-boot pays ~6.5 ms gunzip + ~5 ms write; every
221/// subsequent process across the boot reads the cached file in ~1 ms.
222pub fn embedded_latex_dump(prefer: Option<u32>) -> Option<&'static str> {
223  let (entry, _) = select_embedded(prefer)?;
224  decompressed_dump(entry.year, "latex", entry.latex_gz)
225}

227/// First line of the bundled `texlive.YYYY.version` stamp for the chosen
228/// year (used by the staleness check). `None` if no embedded dump applies.
229pub fn embedded_texlive_version_first_line(prefer: Option<u32>) -> Option<&'static str> {
230  let (entry, _) = select_embedded(prefer)?;
231  entry.stamp.lines().next()
232}

234/// Year-tag for the chosen embedded dump (used in log messages).
235/// Returns `None` if no embedded dump applies.
236pub fn embedded_year(prefer: Option<u32>) -> Option<u32> {
237  let (entry, _) = select_embedded(prefer)?;
238  Some(entry.year)
239}

241/// Whether at least one embedded plain+latex pair is bundled and the
242/// opt-out env var isn't set.
243pub fn plain_embedded_available() -> bool { embedded_plain_dump(None).is_some() }

245pub fn latex_embedded_available() -> bool { embedded_latex_dump(None).is_some() }
