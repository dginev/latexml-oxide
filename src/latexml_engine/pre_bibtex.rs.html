1//! Pre-BibTeX: low-level `.bib` file parser.
2//!
3//! Perl: `LaTeXML/blib/lib/LaTeXML/Pre/BibTeX.pm` (~350 LOC) +
4//! `LaTeXML/blib/lib/LaTeXML/Pre/BibTeX/Entry.pm` (~70 LOC).
5//!
6//! Responsible for the lowest-level parsing of BibTeX files: scans
7//! `.bib` source, identifies `@preamble`, `@string`, `@comment`, and
8//! entry blocks, performs `@string`-macro substitution, and stores
9//! parsed entries. It does NOT analyze fields semantically — that is
10//! `BibTeX.pool.ltxml` (Rust: `bibtex.rs`).
11//!
12//! Per Perl `Pre/BibTeX.pm` L23-32, the result `toTeX()` is wrapped in
13//! a `\begin{bibtex@bibliography}…\end{bibtex@bibliography}` block
14//! whose body is a sequence of `\ProcessBibTeXEntry{<key>}` macro
15//! calls. The bibtex.rs orchestration then drives field dispatch +
16//! XML construction from the registered `BibEntry`.
17//!
18//! ## Faithful-port notes
19//!
20//! - `parse_entry_type` / `parse_entry_name` / `parse_field_name` character classes are
21//!   byte-for-byte the Perl regex `[BIBNAME_re ++ BIBNOISE_re]` (Perl L221-222). Entry-name
22//!   additionally allows `"#%&'()={` (Perl L233); field-name additionally allows `&` (Perl L238).
23//!   Entry-type and field-name are lower-cased; entry-name preserves case at the parse layer (the
24//!   registry then normalises via `register_entry`).
25//!
26//! - `parse_string` reads `"…"` (allowing balanced `{…}` inside that defeat the closing quote) OR
27//!   `{…}` (balanced braces, outer braces stripped) — exact match for Perl L252-274.
28//!
29//! - `parse_value` concatenates simple values separated by `#`. A simple value is either a
30//!   delimited string or a name; a name matching `^\d+$` is taken literally, otherwise looked up in
31//!   the macro table.
32//!
33//! - `skip_junk` is greedy: everything up to the next `@` is discarded (Perl L335-346: "anything
34//!   until @ as an implied comment").

90/// Match Perl's `%CLOSE = ("{" => "}", "(" => ")")` (L154).
91fn close_for(open: char) -> char {
92  match open {
93    '{' => '}',
94    '(' => ')',
95    _ => open,
96  }

104/// `Pre::BibTeX` — low-level .bib parser state.
105///
106/// Perl object hash keys → Rust fields:
107/// * `source` → `source`
108/// * `file`   → `file_label` (display only)
109/// * `lines`  → `lines` (remaining tail; current line lives in `line`)
110/// * `line`   → `line` (current scan buffer)
111/// * `lineno` → `lineno`
112/// * `preamble`  → `preamble`
113/// * `entries`   → `entries` as `(type, key, fields, rawfields)` quadruples
114/// * `macros`    → `macros`
115/// * `parsed` flag → `parsed`
116pub struct PreBibTeX {
117  pub source:     Option<String>,
118  pub file_label: String,
119  lines:          Vec<String>,
120  line:           String,
121  lineno:         usize,
122  pub preamble:   Vec<String>,
123  pub entries:    Vec<ParsedEntry>,
124  macros:         HashMap<String, String>,
125  parsed:         bool,
126}

128/// One parsed entry. Mirrors Perl `LaTeXML::Pre::BibTeX::Entry` (only
129/// the data is kept here; processed‐Tokens fields live in
130/// `bibtex::BibEntry` and are populated by the pool orchestration).
131#[derive(Debug, Clone)]
132pub struct ParsedEntry {
133  pub entry_type: String,
134  pub key:        String,
135  /// (field-name, macro-expanded-value) — Perl `fieldlist`.
136  pub fields:     Vec<(String, String)>,

138  /// `rawfieldlist`.
139  pub raw_fields: Vec<(String, String)>,

142/// Parse errors. The Perl side calls `Error(...)` (recoverable) for
143/// most issues; we surface them as `Err` instead so the caller can
144/// decide whether to keep parsing.
145#[derive(Debug, Clone, PartialEq, Eq)]
146pub enum BibParseError {
147  /// `parse_match` couldn't find any of `delims` at the head of the
148  /// current line.
149  Expected { delims: String, lineno: usize },
150  /// `parse_value` saw neither a delimited string nor a name.
151  ExpectedValue { lineno: usize },
152  /// `parse_string` saw something that wasn't `"` or `{`.
153  ExpectedDelimitedString { lineno: usize },
154  /// `extend_line` hit EOF while scanning a string.
155  UnexpectedEof { lineno: usize },
156  /// `parse_value` encountered a name that isn't a defined macro.
157  /// (Perl emits Error but defaults the value to the name itself; we
158  /// follow that recovery — this variant is for tests / diagnostics.)
159  UndefinedMacro { name: String, lineno: usize },
160}

163  /// Perl `newFromString` (L77-86).
164  pub fn new_from_string(s: &str) -> Self {
165    let mut lines: Vec<String> = s.split('\n').map(String::from).collect();
166    let first = if lines.is_empty() {
167      String::new()
168    } else {
169      lines.remove(0)
170    };
171    Self {
172      source: None,
173      file_label: s!("<anonymous>"),
174      lines,
175      line: first,
176      lineno: 1,
177      preamble: Vec::new(),
178      entries: Vec::new(),
179      macros: default_macros(),
180      parsed: false,
181    }
182  }

184  /// Perl `newFromFile` (L60-75). Reads the file at `path` directly;
185  /// search-path resolution is the caller's responsibility (we do not
186  /// reach into `state` here so the parser stays unit-testable).
187  pub fn new_from_file(path: &str) -> std::io::Result<Self> {
188    let content = std::fs::read_to_string(path)?;
189    let mut me = Self::new_from_string(&content);
190    me.source = Some(path.to_string());

192    Ok(me)

195  /// Perl `newFromGullet` (L89-104). Drains *raw* lines from the
196  /// gullet mouths (with `close_mouth` between exhausted mouths) and
197  /// builds a `PreBibTeX` from the concatenated body. Mirrors the
198  /// integration point in `Core.pm` L160-162: after `InputContent`
199  /// has opened a mouth onto the `.bib` file, drain that mouth here.
200  ///
201  /// Requires that the calling thread has a live `State`/gullet
202  /// (i.e. is being run as part of a `digest_file` flow). Not
203  /// available from `#[cfg(test)]` unit tests.
204  pub fn new_from_gullet(name: &str) -> Self {
205    use latexml_core::gullet;
206    let mut lines: Vec<String> = Vec::new();
207    while gullet::has_more_input() {
208      while let Some(line) = gullet::read_raw_line() {
209        lines.push(s!("{line}\n"));
210      }
211      // `close_mouth` may pop a stacked mouth; we then re-check
212      // `has_more_input` on the next iteration. Forced=false is fine
213      // because we've drained the mouth to EOF.
214      let _ = gullet::close_mouth(false);
215    }
216    let joined = lines.join("");
217    let mut me = Self::new_from_string(&joined);
218    me.source = Some(name.to_string());
219    me.file_label = name.to_string();
220    me
221  }

246  /// Perl `parseTopLevel` (L138-151).
247  pub fn parse_top_level(&mut self) -> Result<(), BibParseError> {
248    while self.skip_junk() {
249      let typ = match self.parse_entry_type() {
250        Some(t) => t,
251        None => continue,
252      };
253      match typ.as_str() {
254        "preamble" => self.parse_preamble()?,
255        "string" => self.parse_macro()?,
256        "comment" => self.parse_comment()?,
257        _ => self.parse_entry(&typ)?,
258      }
259    }
260    self.parsed = true;
261    Ok(())
262  }

264  /// Perl `parsePreamble` (L158-164).
265  fn parse_preamble(&mut self) -> Result<(), BibParseError> {
266    let open = self.parse_match("({")?;
267    let (value, _raw) = self.parse_value()?;
268    self.preamble.push(value);
269    self.parse_match(&close_for(open).to_string())?;
270    Ok(())
271  }

273  /// Perl `parseMacro` (L168-174).
274  fn parse_macro(&mut self) -> Result<(), BibParseError> {
275    let open = self.parse_match("({")?;
276    let (fields, _raw) = self.parse_fields("@string", open)?;
277    for (name, value) in fields {
278      self.macros.insert(name, value);

290  /// Perl `parseEntry` (L185-193).
291  fn parse_entry(&mut self, entry_type: &str) -> Result<(), BibParseError> {
292    let open = self.parse_match("({")?;
293    let key = self.parse_entry_name().unwrap_or_default();
294    self.parse_match(",")?;
295    let (fields, raw_fields) = self.parse_fields("@entry", open)?;
296    self.entries.push(ParsedEntry {
297      entry_type: entry_type.to_string(),
298      key,
299      fields,
300      raw_fields,
301    });
302    Ok(())
303  }

305  /// Perl `parseFields` (L195-209).
306  fn parse_fields(
307    &mut self,
308    _for: &str,
309    open: char,
310  ) -> Result<(BibFieldList, BibFieldList), BibParseError> {
311    let mut fields = Vec::new();
312    let mut raw_fields = Vec::new();
313    let close = close_for(open);
314    loop {
315      let name = match self.parse_field_name() {
316        Some(n) if !n.is_empty() => n,
317        _ => break,
318      };
319      self.parse_match("=")?;
320      let (value, raw) = self.parse_value()?;
321      fields.push((name.clone(), value));
322      raw_fields.push((name, raw));
323      self.skip_white();
324      // Perl: `($$self{line} =~ s/^,//) && skipWhite && ($$self{line} !~ /^\Q$CLOSE{$open}\E/)`
325      if !self.consume_prefix(",") {
326        break;
327      }
328      self.skip_white();
329      if self.line.starts_with(close) {
330        break;
331      }
332    }
333    self.parse_match(&close.to_string())?;
334    Ok((fields, raw_fields))
335  }

339  /// Perl `parseEntryType` (L224-227). Lowercased.

341    self.skip_white();
342    let s = self.consume_while(is_bib_name_or_noise);

350  /// Perl `parseEntryName` (L229-233). Case preserved.
351  fn parse_entry_name(&mut self) -> Option<String> {
352    self.skip_white();
353    // Allows the BibName+BibNoise classes *plus* the literal set:
354    //   "  #  %  &  '  (  )  =  {
355    let s = self.consume_while(|c| {
356      is_bib_name_or_noise(c) || matches!(c, '"' | '#' | '%' | '&' | '\'' | '(' | ')' | '=' | '{')
357    });
358    if s.is_empty() { None } else { Some(s) }
359  }

373  /// Perl `parseMatch` (L240-249). On success consumes exactly one
374  /// of `delims` and returns it. On miss returns
375  /// `BibParseError::Expected`.
376  fn parse_match(&mut self, delims: &str) -> Result<char, BibParseError> {
377    self.skip_white();
378    if delims.is_empty() {
379      return Err(BibParseError::Expected {
380        delims: String::new(),
381        lineno: self.lineno,
382      });
383    }
384    let head = self.line.chars().next();
385    if let Some(c) = head
386      && delims.contains(c)
387    {
388      // pop the matched character (always ASCII single-byte here)
389      let mut new_line = String::with_capacity(self.line.len() - c.len_utf8());
390      let mut it = self.line.chars();
391      it.next();
392      for ch in it {
393        new_line.push(ch);
394      }
395      self.line = new_line;
396      return Ok(c);
397    }
398    Err(BibParseError::Expected {
399      delims: delims.to_string(),
400      lineno: self.lineno,
401    })
402  }

404  /// Perl `parseString` (L252-274).
405  fn parse_string(&mut self) -> Result<String, BibParseError> {
406    self.skip_white();
407    let mut out = String::new();
408    if self.line.starts_with('"') {
409      // opening "; consume it
410      self.line.replace_range(0..1, "");
411      // loop until we find the closing "
412      loop {
413        if self.line.starts_with('"') {
414          self.line.replace_range(0..1, "");
415          break;
416        }
417        if self.line.is_empty() {
418          if !self.extend_line()? {
419            return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
420          }
421        } else if self.line.starts_with('{') {
422          out.push_str(&self.parse_balanced_braces()?);
423        } else {
424          // pull off everything except a brace or "
425          let idx = self.line.find(['"', '{']).unwrap_or(self.line.len());
426          out.push_str(&self.line[..idx]);
427          self.line.replace_range(..idx, "");
428        }
429      }
430    } else if self.line.starts_with('{') {
431      let mut s = self.parse_balanced_braces()?;
432      // strip the surrounding delimiters (Perl L267-268)
433      if s.starts_with('{') {
434        s.remove(0);
435      }
436      if s.ends_with('}') {
437        s.pop();
438      }
439      out = s;
440    } else {
441      return Err(BibParseError::ExpectedDelimitedString { lineno: self.lineno });
442    }
443    // trim leading/trailing whitespace (Perl L272-273)
444    Ok(out.trim().to_string())
445  }

447  /// Perl `parseBalancedBraces` (L276-283). Uses our own balanced-
448  /// brace extractor (no Perl `Text::Balanced` dependency). Pulls
449  /// in further lines via `extend_line` until a balanced pair is
450  /// found. Brace counting follows `Text::Balanced::extract_bracketed`
451  /// semantics for `'{}'`: nested braces count, and a `\{` / `\}`
452  /// pair is treated as a single escaped brace (Perl's
453  /// `extract_bracketed` honours backslash-escaped delimiters when
454  /// the delimiter class is just the pair).
455  fn parse_balanced_braces(&mut self) -> Result<String, BibParseError> {
456    // Make sure there's at least one `}` to *potentially* close.
457    while !self.line.contains('}') {
458      if !self.extend_line()? {
459        return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
460      }
461    }
462    loop {
463      if let Some(end) = find_balanced_brace_end(&self.line) {
464        let s = self.line[..end].to_string();
465        self.line.replace_range(..end, "");
466        return Ok(s);
467      }
468      if !self.extend_line()? {
469        return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
470      }
471    }
472  }

474  /// Perl `extendLine` (L285-295). Returns Ok(true) if a line was
475  /// appended, Ok(false) on EOF (Perl's `Error('unexpected', ...)`).
476  fn extend_line(&mut self) -> Result<bool, BibParseError> {
477    if self.lines.is_empty() {
478      return Ok(false);
479    }
480    let next = self.lines.remove(0);
481    // Perl `<$BIB>` lines keep the trailing newline; `split /\n/` on a
482    // string does not. Re-insert one to preserve line semantics so
483    // that a `%`-comment inside a continued value still terminates on
484    // the next physical line (BibTeX itself doesn't treat `%` as a
485    // comment but other downstream digesters might).
486    self.line.push('\n');
487    self.line.push_str(&next);
488    self.lineno += 1;
489    Ok(true)
490  }

492  /// Perl `parseValue` (L299-318). Returns `(macro_expanded, raw)`.
493  /// `raw` is the byte-for-byte concatenation of the simple values'
494  /// source — Perl actually stores the same `raw` per element of
495  /// `rawfieldlist`, but never re-emits it as a single string. We
496  /// preserve the convention here so `BibEntry.add_raw_field` gets the
497  /// pre-macro-expansion text.
498  fn parse_value(&mut self) -> Result<(String, String), BibParseError> {
499    let mut value = String::new();
500    let mut raw = String::new();
501    loop {
502      self.skip_white();
503      let head = self.line.chars().next();
504      if matches!(head, Some('"') | Some('{')) {
505        // Pull the raw form *before* macro expansion (parse_string
506        // strips delimiters; for the raw side we want the actual
507        // source bytes, delimiters and all).
508        let raw_start_line = self.line.clone();
509        let s = self.parse_string()?;
510        let raw_consumed = consumed_diff(&raw_start_line, &self.line);
511        raw.push_str(&raw_consumed);
512        value.push_str(&s);
513      } else if let Some(name) = self.parse_field_name() {
514        let is_digits = !name.is_empty() && name.chars().all(|c| c.is_ascii_digit());
515        let resolved = if is_digits {
516          name.clone()
517        } else {
518          self.macros.get(&name).cloned().unwrap_or_else(|| {
519            // Perl recovery: leave the name as-is.
520            name.clone()
521          })
522        };
523        raw.push_str(&name);
524        value.push_str(&resolved);
525      } else {
526        return Err(BibParseError::ExpectedValue { lineno: self.lineno });
527      }
528      self.skip_white();
529      if !self.consume_prefix("#") {
530        break;
531      }
532      // Perl spelling: a `#`-separated value re-emits as a `#` in raw.
533      raw.push('#');
534    }
535    Ok((value, raw))
536  }

538  /// Perl `skipWhite` (L320-330). Returns true if the cursor is now
539  /// on a non-empty line, false on EOF.
540  fn skip_white(&mut self) -> bool {
541    loop {
542      // Trim leading whitespace (any unicode whitespace, like Perl `\s`).
543      let trimmed = self.line.trim_start_matches(char::is_whitespace);
544      if trimmed.len() < self.line.len() {
545        self.line = trimmed.to_string();
546      }
547      if !self.line.is_empty() {
548        return true;
549      }
550      if self.lines.is_empty() {
551        return false;
552      }
553      self.line = self.lines.remove(0);
554      self.lineno += 1;
555    }
556  }

558  /// Perl `skipJunk` (L335-346). Discards everything up to the next
559  /// `@` (or `%`, which Perl treats as the start of a one-line
560  /// comment though BibTeX itself doesn't). Returns true when an `@`
561  /// was found and consumed; false on EOF.
562  fn skip_junk(&mut self) -> bool {
563    loop {
564      if let Some(idx) = self.line.find(['@', '%']) {
565        // Drop everything before the marker.
566        self.line.replace_range(..idx, "");
567        // Perl L340: `return '@' if $$self{line} =~ s/^@//;`. If we
568        // landed on `%`, the line is treated as a comment — drop the
569        // rest of it and fetch the next line.
570        if self.line.starts_with('@') {
571          self.line.replace_range(0..1, "");
572          return true;
573        }
574        // %-comment: drop the rest of this line and continue.
575        self.line.clear();
576      } else {
577        self.line.clear();
578      }
579      if self.lines.is_empty() {
580        return false;
581      }
582      self.line = self.lines.remove(0);
583      self.lineno += 1;
584    }
585  }

589  fn consume_while(&mut self, predicate: impl Fn(char) -> bool) -> String {
590    let end = self
591      .line
592      .char_indices()
593      .find(|(_, c)| !predicate(*c))
594      .map(|(i, _)| i)
595      .unwrap_or(self.line.len());
596    let head = self.line[..end].to_string();
597    self.line.replace_range(..end, "");
598    head
599  }

601  fn consume_prefix(&mut self, p: &str) -> bool {
602    if self.line.starts_with(p) {
603      self.line.replace_range(..p.len(), "");
604      true
605    } else {
606      false
607    }
608  }

612  /// Perl `toTeX` (L110-122). Registers every parsed entry into the
613  /// `bibtex.rs` thread-local registry (Perl: `assignValue
614  /// 'BIBENTRY@<lc-key>'`), then returns the wrapper TeX that the
615  /// digester will read back.
616  pub fn to_tex(&mut self) -> Result<String, BibParseError> {
617    if !self.parsed {
618      self.parse_top_level()?;
619    }
620    for parsed in &self.entries {
621      let mut entry = BibEntry::new(parsed.key.clone(), parsed.entry_type.clone());
622      // Perl `Pre::BibTeX::Entry::new` stores both processed `fields`
623      // (post-macro-expansion, delimiter-stripped) and `rawfields`
624      // (verbatim source). The processed form is what the field
625      // dispatcher consumes; the verbatim form survives only for
626      // `\bib@@origbibentry`'s source-style re-rendering. The Rust
627      // BibEntry's `raw_fields: Vec<(String,String)>` slot is the
628      // dispatcher input and `pretty_print` source. Feed the
629      // *processed* value so dispatched handlers see macro-expanded
630      // text (e.g. `tcs` → "Theoretical Computer Science").
631      for (name, value) in &parsed.fields {
632        entry.add_raw_field(name.clone(), value.clone());
633      }
634      register_entry(&parsed.key, entry);
635    }
636    let mut out = String::new();
637    for pre in &self.preamble {
638      out.push_str(pre);
639      out.push('\n');
640    }
641    out.push_str("\\begin{bibtex@bibliography}\n");
642    for entry in &self.entries {
643      out.push_str("\\ProcessBibTeXEntry{");
644      out.push_str(&entry.key);
645      out.push_str("}\n");
646    }
647    out.push_str("\\end{bibtex@bibliography}");
648    Ok(out)
649  }
650}

655fn is_bib_name_or_noise(c: char) -> bool {
656  // BIBNAME: [a-zA-Z0-9]
657  if c.is_ascii_alphanumeric() {
658    return true;
659  }
660  // BIBNOISE: . + - * / ^ _ : ; @ ` ? ! ~ | < > $ [ ]
661  matches!(
662    c,
663    '.'
664      | '+'
665      | '-'
666      | '*'
667      | '/'
668      | '^'
669      | '_'
670      | ':'
671      | ';'
672      | '@'
673      | '`'
674      | '?'
675      | '!'
676      | '~'
677      | '|'
678      | '<'
679      | '>'
680      | '$'
681      | '['
682      | ']'
683  )
684}

686/// Compute the byte-slice that was consumed from `before` to produce
687/// `after`. Used by `parse_value` to recover the raw source of a
688/// just-parsed string without re-parsing.
689fn consumed_diff(before: &str, after: &str) -> String {
690  if before.len() < after.len() {
691    return String::new();
692  }
693  before[..before.len() - after.len()].to_string()
694}
695
696/// Find the byte index immediately after the closing brace of a
697/// `{`-prefixed balanced-brace group in `s`. Returns `None` if no
698/// balanced close exists in `s`.
699///
700/// Honors backslash-escaped `\{` and `\}` as a single token, matching
701/// the Perl `Text::Balanced::extract_bracketed($s, '{}')` semantics
702/// for the simple pair specification.
703fn find_balanced_brace_end(s: &str) -> Option<usize> {
704  let bytes = s.as_bytes();
705  if bytes.first() != Some(&b'{') {
706    return None;
707  }
708  let mut depth: i32 = 0;
709  let mut i = 0;
710  while i < bytes.len() {
711    match bytes[i] {
712      b'\\' if i + 1 < bytes.len() => {
713        // Skip the escaped char
714        i += 2;
715        continue;
716      },
717      b'{' => depth += 1,
718      b'}' => {
719        depth -= 1;
720        if depth == 0 {
721          return Some(i + 1);
722        }
723      },
724      _ => {},
725    }
726    i += 1;
727  }
728  None
729}
