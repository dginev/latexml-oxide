1//! Pre-BibTeX: low-level `.bib` file parser.
2//!
3//! Perl: `LaTeXML/blib/lib/LaTeXML/Pre/BibTeX.pm` (~350 LOC) +
4//! `LaTeXML/blib/lib/LaTeXML/Pre/BibTeX/Entry.pm` (~70 LOC).
5//!
6//! Responsible for the lowest-level parsing of BibTeX files: scans
7//! `.bib` source, identifies `@preamble`, `@string`, `@comment`, and
8//! entry blocks, performs `@string`-macro substitution, and stores
9//! parsed entries. It does NOT analyze fields semantically — that is
10//! `BibTeX.pool.ltxml` (Rust: `bibtex.rs`).
11//!
12//! Per Perl `Pre/BibTeX.pm` L23-32, the result `toTeX()` is wrapped in
13//! a `\begin{bibtex@bibliography}…\end{bibtex@bibliography}` block
14//! whose body is a sequence of `\ProcessBibTeXEntry{<key>}` macro
15//! calls. The bibtex.rs orchestration then drives field dispatch +
16//! XML construction from the registered `BibEntry`.
17//!
18//! ## Faithful-port notes
19//!
20//! - `parse_entry_type` / `parse_entry_name` / `parse_field_name`
21//!   character classes are byte-for-byte the Perl regex
22//!   `[BIBNAME_re ++ BIBNOISE_re]` (Perl L221-222). Entry-name
23//!   additionally allows `"#%&'()={` (Perl L233); field-name
24//!   additionally allows `&` (Perl L238). Entry-type and field-name
25//!   are lower-cased; entry-name preserves case at the parse layer
26//!   (the registry then normalises via `register_entry`).
27//!
28//! - `parse_string` reads `"…"` (allowing balanced `{…}` inside that
29//!   defeat the closing quote) OR `{…}` (balanced braces, outer
30//!   braces stripped) — exact match for Perl L252-274.
31//!
32//! - `parse_value` concatenates simple values separated by `#`. A
33//!   simple value is either a delimited string or a name; a name
34//!   matching `^\d+$` is taken literally, otherwise looked up in the
35//!   macro table.
36//!
37//! - `skip_junk` is greedy: everything up to the next `@` is
38//!   discarded (Perl L335-346: "anything until @ as an implied
39//!   comment").

90/// Match Perl's `%CLOSE = ("{" => "}", "(" => ")")` (L154).
91fn close_for(open: char) -> char {
92  match open {
93    '{' => '}',
94    '(' => ')',
95    _ => open,
96  }

99/// List of `(field-name, value)` pairs returned by `parse_fields`.

104/// `Pre::BibTeX` — low-level .bib parser state.
105///
106/// Perl object hash keys → Rust fields:
107/// * `source` → `source`
108/// * `file`   → `file_label` (display only)
109/// * `lines`  → `lines` (remaining tail; current line lives in `line`)
110/// * `line`   → `line` (current scan buffer)
111/// * `lineno` → `lineno`
112/// * `preamble`  → `preamble`
113/// * `entries`   → `entries` as `(type, key, fields, rawfields)` quadruples
114/// * `macros`    → `macros`
115/// * `parsed` flag → `parsed`
116pub struct PreBibTeX {
117  pub source:    Option<String>,
118  pub file_label: String,
119  lines:         Vec<String>,
120  line:          String,
121  lineno:        usize,
122  pub preamble:  Vec<String>,
123  pub entries:   Vec<ParsedEntry>,
124  macros:        HashMap<String, String>,
125  parsed:        bool,
126}

128/// One parsed entry. Mirrors Perl `LaTeXML::Pre::BibTeX::Entry` (only
129/// the data is kept here; processed‐Tokens fields live in
130/// `bibtex::BibEntry` and are populated by the pool orchestration).
131#[derive(Debug, Clone)]
132pub struct ParsedEntry {
133  pub entry_type: String,
134  pub key:        String,
135  /// (field-name, macro-expanded-value) — Perl `fieldlist`.
136  pub fields:     Vec<(String, String)>,
137  /// (field-name, raw-source-value, *before* macro expansion) — Perl
138  /// `rawfieldlist`.
139  pub raw_fields: Vec<(String, String)>,
140}

142/// Parse errors. The Perl side calls `Error(...)` (recoverable) for
143/// most issues; we surface them as `Err` instead so the caller can
144/// decide whether to keep parsing.
145#[derive(Debug, Clone, PartialEq, Eq)]
146pub enum BibParseError {
147  /// `parse_match` couldn't find any of `delims` at the head of the
148  /// current line.
149  Expected { delims: String, lineno: usize },
150  /// `parse_value` saw neither a delimited string nor a name.
151  ExpectedValue { lineno: usize },
152  /// `parse_string` saw something that wasn't `"` or `{`.
153  ExpectedDelimitedString { lineno: usize },
154  /// `extend_line` hit EOF while scanning a string.
155  UnexpectedEof { lineno: usize },
156  /// `parse_value` encountered a name that isn't a defined macro.
157  /// (Perl emits Error but defaults the value to the name itself; we
158  /// follow that recovery — this variant is for tests / diagnostics.)
159  UndefinedMacro { name: String, lineno: usize },
160}

163  /// Perl `newFromString` (L77-86).
164  pub fn new_from_string(s: &str) -> Self {
165    let mut lines: Vec<String> = s.split('\n').map(String::from).collect();
166    let first = if lines.is_empty() {
167      String::new()
168    } else {
169      lines.remove(0)
170    };
171    Self {
172      source:      None,
173      file_label:  s!("<anonymous>"),
174      lines,
175      line:        first,
176      lineno:      1,
177      preamble:    Vec::new(),
178      entries:     Vec::new(),
179      macros:      default_macros(),
180      parsed:      false,
181    }
182  }

184  /// Perl `newFromFile` (L60-75). Reads the file at `path` directly;
185  /// search-path resolution is the caller's responsibility (we do not
186  /// reach into `state` here so the parser stays unit-testable).
187  pub fn new_from_file(path: &str) -> std::io::Result<Self> {
188    let content = std::fs::read_to_string(path)?;
189    let mut me = Self::new_from_string(&content);
190    me.source = Some(path.to_string());
191    me.file_label = path.to_string();
192    Ok(me)
193  }

195  /// Perl `newFromGullet` (L89-104). Drains *raw* lines from the
196  /// gullet mouths (with `close_mouth` between exhausted mouths) and
197  /// builds a `PreBibTeX` from the concatenated body. Mirrors the
198  /// integration point in `Core.pm` L160-162: after `InputContent`
199  /// has opened a mouth onto the `.bib` file, drain that mouth here.
200  ///
201  /// Requires that the calling thread has a live `State`/gullet
202  /// (i.e. is being run as part of a `digest_file` flow). Not
203  /// available from `#[cfg(test)]` unit tests.
204  pub fn new_from_gullet(name: &str) -> Self {
205    use latexml_core::gullet;
206    let mut lines: Vec<String> = Vec::new();
207    while gullet::has_more_input() {
208      while let Some(line) = gullet::read_raw_line() {
209        lines.push(s!("{line}\n"));
210      }
211      // `close_mouth` may pop a stacked mouth; we then re-check
212      // `has_more_input` on the next iteration. Forced=false is fine
213      // because we've drained the mouth to EOF.
214      let _ = gullet::close_mouth(false);
215    }
216    let joined = lines.join("");
217    let mut me = Self::new_from_string(&joined);
218    me.source = Some(name.to_string());
219    me.file_label = name.to_string();
220    me
221  }

245  /// Perl `parseTopLevel` (L138-151).
246  pub fn parse_top_level(&mut self) -> Result<(), BibParseError> {
247    while self.skip_junk() {
248      let typ = match self.parse_entry_type() {
249        Some(t) => t,
250        None => continue,
251      };
252      match typ.as_str() {
253        "preamble" => self.parse_preamble()?,
254        "string" => self.parse_macro()?,
255        "comment" => self.parse_comment()?,
256        _ => self.parse_entry(&typ)?,
257      }
258    }
259    self.parsed = true;
260    Ok(())
261  }

263  /// Perl `parsePreamble` (L158-164).
264  fn parse_preamble(&mut self) -> Result<(), BibParseError> {
265    let open = self.parse_match("({")?;
266    let (value, _raw) = self.parse_value()?;
267    self.preamble.push(value);
268    self.parse_match(&close_for(open).to_string())?;
269    Ok(())
270  }

272  /// Perl `parseMacro` (L168-174).
273  fn parse_macro(&mut self) -> Result<(), BibParseError> {
274    let open = self.parse_match("({")?;
275    let (fields, _raw) = self.parse_fields("@string", open)?;
276    for (name, value) in fields {
277      self.macros.insert(name, value);
278    }
279    Ok(())
280  }

289  /// Perl `parseEntry` (L185-193).
290  fn parse_entry(&mut self, entry_type: &str) -> Result<(), BibParseError> {
291    let open = self.parse_match("({")?;
292    let key = self.parse_entry_name().unwrap_or_default();
293    self.parse_match(",")?;
294    let (fields, raw_fields) = self.parse_fields("@entry", open)?;
295    self.entries.push(ParsedEntry {
296      entry_type: entry_type.to_string(),
297      key,
298      fields,
299      raw_fields,
300    });
301    Ok(())
302  }

304  /// Perl `parseFields` (L195-209).
305  fn parse_fields(
306    &mut self,
307    _for: &str,
308    open: char,
309  ) -> Result<(BibFieldList, BibFieldList), BibParseError> {
310    let mut fields = Vec::new();
311    let mut raw_fields = Vec::new();
312    let close = close_for(open);
313    loop {
314      let name = match self.parse_field_name() {
315        Some(n) if !n.is_empty() => n,
316        _ => break,
317      };
318      self.parse_match("=")?;
319      let (value, raw) = self.parse_value()?;
320      fields.push((name.clone(), value));
321      raw_fields.push((name, raw));
322      self.skip_white();
323      // Perl: `($$self{line} =~ s/^,//) && skipWhite && ($$self{line} !~ /^\Q$CLOSE{$open}\E/)`
324      if !self.consume_prefix(",") {
325        break;
326      }
327      self.skip_white();
328      if self.line.starts_with(close) {
329        break;
330      }
331    }
332    self.parse_match(&close.to_string())?;
333    Ok((fields, raw_fields))
334  }

338  /// Perl `parseEntryType` (L224-227). Lowercased.
339  fn parse_entry_type(&mut self) -> Option<String> {
340    self.skip_white();
341    let s = self.consume_while(is_bib_name_or_noise);
342    if s.is_empty() {
343      None
344    } else {
345      Some(s.to_ascii_lowercase())
346    }

349  /// Perl `parseEntryName` (L229-233). Case preserved.
350  fn parse_entry_name(&mut self) -> Option<String> {
351    self.skip_white();
352    // Allows the BibName+BibNoise classes *plus* the literal set:
353    //   "  #  %  &  '  (  )  =  {
354    let s = self.consume_while(|c| {
355      is_bib_name_or_noise(c) || matches!(c, '"' | '#' | '%' | '&' | '\'' | '(' | ')' | '=' | '{')
356    });
357    if s.is_empty() { None } else { Some(s) }
358  }

360  /// Perl `parseFieldName` (L235-238). Lowercased; allows `&` in
361  /// addition to the BibName+BibNoise classes.
362  fn parse_field_name(&mut self) -> Option<String> {
363    self.skip_white();
364    let s = self.consume_while(|c| is_bib_name_or_noise(c) || c == '&');

372  /// Perl `parseMatch` (L240-249). On success consumes exactly one
373  /// of `delims` and returns it. On miss returns
374  /// `BibParseError::Expected`.
375  fn parse_match(&mut self, delims: &str) -> Result<char, BibParseError> {
376    self.skip_white();
377    if delims.is_empty() {
378      return Err(BibParseError::Expected {
379        delims: String::new(),
380        lineno: self.lineno,
381      });
382    }
383    let head = self.line.chars().next();
384    if let Some(c) = head {
385      if delims.contains(c) {
386        // pop the matched character (always ASCII single-byte here)
387        let mut new_line = String::with_capacity(self.line.len() - c.len_utf8());
388        let mut it = self.line.chars();
389        it.next();
390        for ch in it {
391          new_line.push(ch);
392        }
393        self.line = new_line;
394        return Ok(c);
395      }
396    }
397    Err(BibParseError::Expected {
398      delims: delims.to_string(),
399      lineno: self.lineno,
400    })
401  }

403  /// Perl `parseString` (L252-274).
404  fn parse_string(&mut self) -> Result<String, BibParseError> {
405    self.skip_white();
406    let mut out = String::new();
407    if self.line.starts_with('"') {
408      // opening "; consume it
409      self.line.replace_range(0..1, "");
410      // loop until we find the closing "
411      loop {
412        if self.line.starts_with('"') {
413          self.line.replace_range(0..1, "");
414          break;
415        }
416        if self.line.is_empty() {
417          if !self.extend_line()? {
418            return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
419          }
420        } else if self.line.starts_with('{') {
421          out.push_str(&self.parse_balanced_braces()?);
422        } else {
423          // pull off everything except a brace or "
424          let idx = self.line.find(['"', '{']).unwrap_or(self.line.len());
425          out.push_str(&self.line[..idx]);
426          self.line.replace_range(..idx, "");
427        }
428      }
429    } else if self.line.starts_with('{') {
430      let mut s = self.parse_balanced_braces()?;
431      // strip the surrounding delimiters (Perl L267-268)
432      if s.starts_with('{') {
433        s.remove(0);
434      }
435      if s.ends_with('}') {
436        s.pop();
437      }
438      out = s;
439    } else {
440      return Err(BibParseError::ExpectedDelimitedString { lineno: self.lineno });
441    }
442    // trim leading/trailing whitespace (Perl L272-273)
443    Ok(out.trim().to_string())
444  }

446  /// Perl `parseBalancedBraces` (L276-283). Uses our own balanced-
447  /// brace extractor (no Perl `Text::Balanced` dependency). Pulls
448  /// in further lines via `extend_line` until a balanced pair is
449  /// found. Brace counting follows `Text::Balanced::extract_bracketed`
450  /// semantics for `'{}'`: nested braces count, and a `\{` / `\}`
451  /// pair is treated as a single escaped brace (Perl's
452  /// `extract_bracketed` honours backslash-escaped delimiters when
453  /// the delimiter class is just the pair).
454  fn parse_balanced_braces(&mut self) -> Result<String, BibParseError> {
455    // Make sure there's at least one `}` to *potentially* close.
456    while !self.line.contains('}') {
457      if !self.extend_line()? {
458        return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
459      }
460    }
461    loop {
462      if let Some(end) = find_balanced_brace_end(&self.line) {
463        let s = self.line[..end].to_string();
464        self.line.replace_range(..end, "");
465        return Ok(s);
466      }
467      if !self.extend_line()? {
468        return Err(BibParseError::UnexpectedEof { lineno: self.lineno });
469      }
470    }
471  }

473  /// Perl `extendLine` (L285-295). Returns Ok(true) if a line was
474  /// appended, Ok(false) on EOF (Perl's `Error('unexpected', ...)`).
475  fn extend_line(&mut self) -> Result<bool, BibParseError> {
476    if self.lines.is_empty() {
477      return Ok(false);
478    }
479    let next = self.lines.remove(0);
480    // Perl `<$BIB>` lines keep the trailing newline; `split /\n/` on a
481    // string does not. Re-insert one to preserve line semantics so
482    // that a `%`-comment inside a continued value still terminates on
483    // the next physical line (BibTeX itself doesn't treat `%` as a
484    // comment but other downstream digesters might).
485    self.line.push('\n');
486    self.line.push_str(&next);
487    self.lineno += 1;
488    Ok(true)
489  }

491  /// Perl `parseValue` (L299-318). Returns `(macro_expanded, raw)`.
492  /// `raw` is the byte-for-byte concatenation of the simple values'
493  /// source — Perl actually stores the same `raw` per element of
494  /// `rawfieldlist`, but never re-emits it as a single string. We
495  /// preserve the convention here so `BibEntry.add_raw_field` gets the
496  /// pre-macro-expansion text.
497  fn parse_value(&mut self) -> Result<(String, String), BibParseError> {
498    let mut value = String::new();
499    let mut raw = String::new();
500    loop {
501      self.skip_white();
502      let head = self.line.chars().next();
503      if matches!(head, Some('"') | Some('{')) {
504        // Pull the raw form *before* macro expansion (parse_string
505        // strips delimiters; for the raw side we want the actual
506        // source bytes, delimiters and all).
507        let raw_start_line = self.line.clone();
508        let s = self.parse_string()?;
509        let raw_consumed = consumed_diff(&raw_start_line, &self.line);
510        raw.push_str(&raw_consumed);
511        value.push_str(&s);
512      } else if let Some(name) = self.parse_field_name() {
513        let is_digits = !name.is_empty() && name.chars().all(|c| c.is_ascii_digit());
514        let resolved = if is_digits {
515          name.clone()
516        } else {
517          self
518            .macros
519            .get(&name)
520            .cloned()
521            .unwrap_or_else(|| {
522              // Perl recovery: leave the name as-is.
523              name.clone()
524            })
525        };
526        raw.push_str(&name);
527        value.push_str(&resolved);
528      } else {
529        return Err(BibParseError::ExpectedValue { lineno: self.lineno });
530      }
531      self.skip_white();
532      if !self.consume_prefix("#") {
533        break;
534      }
535      // Perl spelling: a `#`-separated value re-emits as a `#` in raw.
536      raw.push('#');
537    }
538    Ok((value, raw))
539  }

541  /// Perl `skipWhite` (L320-330). Returns true if the cursor is now
542  /// on a non-empty line, false on EOF.
543  fn skip_white(&mut self) -> bool {
544    loop {
545      // Trim leading whitespace (any unicode whitespace, like Perl `\s`).
546      let trimmed = self.line.trim_start_matches(char::is_whitespace);
547      if trimmed.len() < self.line.len() {
548        self.line = trimmed.to_string();
549      }
550      if !self.line.is_empty() {
551        return true;
552      }
553      if self.lines.is_empty() {
554        return false;
555      }
556      self.line = self.lines.remove(0);
557      self.lineno += 1;
558    }
559  }

561  /// Perl `skipJunk` (L335-346). Discards everything up to the next
562  /// `@` (or `%`, which Perl treats as the start of a one-line
563  /// comment though BibTeX itself doesn't). Returns true when an `@`
564  /// was found and consumed; false on EOF.
565  fn skip_junk(&mut self) -> bool {
566    loop {
567      if let Some(idx) = self.line.find(['@', '%']) {
568        // Drop everything before the marker.
569        self.line.replace_range(..idx, "");
570        // Perl L340: `return '@' if $$self{line} =~ s/^@//;`. If we
571        // landed on `%`, the line is treated as a comment — drop the
572        // rest of it and fetch the next line.
573        if self.line.starts_with('@') {
574          self.line.replace_range(0..1, "");
575          return true;
576        }
577        // %-comment: drop the rest of this line and continue.
578        self.line.clear();
579      } else {
580        self.line.clear();
581      }
582      if self.lines.is_empty() {
583        return false;
584      }
585      self.line = self.lines.remove(0);
586      self.lineno += 1;
587    }
588  }

592  fn consume_while(&mut self, predicate: impl Fn(char) -> bool) -> String {
593    let end = self
594      .line
595      .char_indices()
596      .find(|(_, c)| !predicate(*c))
597      .map(|(i, _)| i)
598      .unwrap_or(self.line.len());
599    let head = self.line[..end].to_string();
600    self.line.replace_range(..end, "");
601    head
602  }

604  fn consume_prefix(&mut self, p: &str) -> bool {
605    if self.line.starts_with(p) {
606      self.line.replace_range(..p.len(), "");
607      true
608    } else {
609      false
610    }
611  }

615  /// Perl `toTeX` (L110-122). Registers every parsed entry into the
616  /// `bibtex.rs` thread-local registry (Perl: `assignValue
617  /// 'BIBENTRY@<lc-key>'`), then returns the wrapper TeX that the
618  /// digester will read back.
619  pub fn to_tex(&mut self) -> Result<String, BibParseError> {
620    if !self.parsed {
621      self.parse_top_level()?;
622    }
623    for parsed in &self.entries {
624      let mut entry = BibEntry::new(parsed.key.clone(), parsed.entry_type.clone());
625      // Perl `Pre::BibTeX::Entry::new` stores both processed `fields`
626      // (post-macro-expansion, delimiter-stripped) and `rawfields`
627      // (verbatim source). The processed form is what the field
628      // dispatcher consumes; the verbatim form survives only for
629      // `\bib@@origbibentry`'s source-style re-rendering. The Rust
630      // BibEntry's `raw_fields: Vec<(String,String)>` slot is the
631      // dispatcher input and `pretty_print` source. Feed the
632      // *processed* value so dispatched handlers see macro-expanded
633      // text (e.g. `tcs` → "Theoretical Computer Science").
634      for (name, value) in &parsed.fields {
635        entry.add_raw_field(name.clone(), value.clone());
636      }
637      register_entry(&parsed.key, entry);
638    }
639    let mut out = String::new();
640    for pre in &self.preamble {
641      out.push_str(pre);
642      out.push('\n');
643    }
644    out.push_str("\\begin{bibtex@bibliography}\n");
645    for entry in &self.entries {
646      out.push_str("\\ProcessBibTeXEntry{");
647      out.push_str(&entry.key);
648      out.push_str("}\n");
649    }
650    out.push_str("\\end{bibtex@bibliography}");
651    Ok(out)
652  }
653}

658fn is_bib_name_or_noise(c: char) -> bool {
659  // BIBNAME: [a-zA-Z0-9]
660  if c.is_ascii_alphanumeric() {
661    return true;
662  }
663  // BIBNOISE: . + - * / ^ _ : ; @ ` ? ! ~ | < > $ [ ]
664  matches!(
665    c,
666    '.' | '+' | '-' | '*' | '/' | '^' | '_' | ':' | ';' | '@' | '`' | '?' | '!' | '~' | '|' |
667    '<' | '>' | '$' | '[' | ']'
668  )
669}

671/// Compute the byte-slice that was consumed from `before` to produce
672/// `after`. Used by `parse_value` to recover the raw source of a
673/// just-parsed string without re-parsing.
674fn consumed_diff(before: &str, after: &str) -> String {
675  if before.len() < after.len() {
676    return String::new();
677  }
678  before[..before.len() - after.len()].to_string()
679}
680
681/// Find the byte index immediately after the closing brace of a
682/// `{`-prefixed balanced-brace group in `s`. Returns `None` if no
683/// balanced close exists in `s`.
684///
685/// Honors backslash-escaped `\{` and `\}` as a single token, matching
686/// the Perl `Text::Balanced::extract_bracketed($s, '{}')` semantics
687/// for the simple pair specification.
688fn find_balanced_brace_end(s: &str) -> Option<usize> {
689  let bytes = s.as_bytes();
690  if bytes.first() != Some(&b'{') {
691    return None;
692  }
693  let mut depth: i32 = 0;
694  let mut i = 0;
695  while i < bytes.len() {
696    match bytes[i] {
697      b'\\' if i + 1 < bytes.len() => {
698        // Skip the escaped char
699        i += 2;
700        continue;
701      },
702      b'{' => depth += 1,
703      b'}' => {
704        depth -= 1;
705        if depth == 0 {
706          return Some(i + 1);
707        }
708      },
709      _ => {},
710    }
711    i += 1;
712  }
713  None
714}
