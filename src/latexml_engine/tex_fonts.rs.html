12  //======================================================================
13  // Font declaration
14  //----------------------------------------------------------------------
15  // \font             iq loads information about a font into TeX's memory.
16  // \fontname         c  returns the system file name for a font.
17  // \fontdimen        iq holds font parameters.
18  // \nullfont         iq is a predefined font with no characters.
19
20  // # 2nd arg is <font> = <fontdef token> | \font | <family member>
21  // #  <family member> = <font range><4bit number>
22  // #  <font range> = \textfont | \scriptfont | \scriptscriptfont
23  // Perl: FontDef param type — for \textfont/\scriptfont/\scriptscriptfont, reads family
24  // number and looks up the stored font CS token.  For \font, returns current_FontDef.
25  DefParameterType!(FontToken, sub[_inner, _extra] {
26    let token = gullet::read_token()?.unwrap();
27    if let Some(font_type) = token.with_str(|ts| {
28      if ts.starts_with("\\textfont") && ts == "\\textfont" { Some("textfont") }
29      else if ts.starts_with("\\scriptscriptfont") && ts == "\\scriptscriptfont" { Some("scriptscriptfont") }
30      else if ts.starts_with("\\scriptfont") && ts == "\\scriptfont" { Some("scriptfont") }
31      else { None }
32    }) {
33      // Perl: $token = LookupValue($type . 'font_' . $fam->valueOf).
34      // with_value avoids the Stored envelope clone; Token is Copy.
35      let fam = gullet::read_number()?.value_of();
36      let key = s!("{font_type}_{fam}");
37      state::with_value(&key, |v| match v {
38        Some(Stored::Token(t)) => *t,
39        _ => token,
40      })
41    } else if token.with_str(|ts| ts == "\\font") {
42      // Perl: $token = LookupValue('current_FontDef') || T_CS('\lx@default@font')
43      state::with_value("current_FontDef", |v| match v {
44        Some(Stored::Token(t)) => *t,
45        _ => T_CS!("\\lx@default@font"),
46      })
47    } else {
48      token
49    }
50  });
51
52  // Perl TeX_Fonts.pool.ltxml L77-78: default skew/hyphen registers
53  // before \font.
54  DefRegister!("\\defaultskewchar", Number!(-1));
55  DefRegister!("\\defaulthyphenchar", Number!(45)); // ord('-') = 45
56
57  // Perl: \font SkipSpaces Token SkipSpaces SkipMatch:= SkipSpaces TeXFileName
58  // (TeX_Fonts.pool.ltxml:82). Token does NOT auto-skip spaces in Rust gullet
59  // (parity with Perl readToken), so the leading SkipSpaces is the
60  // Perl-faithful guard for inputs like `\font  \foo  =  cmr10`.
61  DefPrimitive!("\\font SkipSpaces Token SkipSpaces SkipMatch:= SkipSpaces TeXFileName",
62  sub[(cs, name_arg)] {
63    let name = name_arg.to_string();
64    // Read optional "at <dimen>" or "scaled <number>" — Perl: TeX_Fonts.pool.ltxml L88-94
65    //   if    ($gullet->readKeyword('at'))     { $at = $gullet->readDimension; }
66    //   elsif ($gullet->readKeyword('scaled')) { $scaled = $gullet->readNumber/1000; }
67    // The `elsif` is load-bearing: only one branch fires. Without it, an
68    // `at <dim> scaled <n>` input would have BOTH consumed instead of leaving
69    // `scaled <n>` in the stream as Perl does.
70    let mut at_pt = None;
71    let mut scaled = None;
72    if gullet::read_keyword(&["at"])?.is_some() {
73      at_pt = Some(gullet::read_dimension()?.pt_value(None));
74    } else if gullet::read_keyword(&["scaled"])?.is_some() {
75      scaled = Some(gullet::read_number()?.value_of() as f64 / 1000.0);
76    }
77    let props_opt = if let Some(mut props) = font::decode_fontname(&name, at_pt, scaled) {
78      props.name = Some(Cow::Owned(name.clone()));
79      Some(props)
80    } else { // Failed?
81      let message = s!("Unrecognized font name {:?} Font switch macro {:?}
82      will have no effect", name, cs.stringify());
83      Info!("unexpected", name, message);
84      None
85    };
86    gullet::skip_spaces()?;
87    let cs_str = cs.to_string();
88    if let Some(ref props) = props_opt {
89      AssignValue!(&s!("fontinfo_{cs_str}"), props.clone());
90    }
91    // Perl: $key = 'fontinfo_' . $name; $key .= " at " . ToString($at) if $at;
92    // Shared font key: fonts with same name+size share hyphenchar/skewchar
93    let at_str_opt = if let Some(at_val) = at_pt {
94      Some(s!("{at_val:.1}pt"))
95    } else if let Some(_sc) = scaled {
96      props_opt.as_ref().and_then(|p| p.size).map(|sz| s!("{sz:.1}pt"))
97    } else {
98      None
99    };
100    let shared_key = if let Some(ref at_str) = at_str_opt {
101      s!("fontinfo_{name} at {at_str}")
102    } else {
103      s!("fontinfo_{name}")
104    };
105    // Store CS → shared key mapping
106    state::assign_value(
107      &s!("font_shared_key_{cs_str}"),
108      Stored::String(arena::pin(&shared_key)),
109      None,
110    );
111    // Store explicit "at" value for \meaning (Perl: $$fontinfo{at} = ToString($at))
112    if let Some(ref at_str) = at_str_opt {
113      state::assign_value(
114        &s!("fontinfo_at_{cs_str}"),
115        Stored::String(arena::pin(at_str)),
116        None,
117      );
118    }
119    // Perl: only initialize hyphenchar/skewchar if this font key hasn't been seen before
120    // (shared fontinfo means second \font with same name+size reuses existing values)
121    let hc_key = s!("hyphenchar_{shared_key}");
122    if !state::has_value(&hc_key) {
123      let default_hyphen = lookup_int("\\defaulthyphenchar");
124      state::assign_value(
125        &hc_key,
126        Stored::Number(Number::new(default_hyphen as i64)),
127        Some(Scope::Global),
128      );
129      let default_skew = lookup_int("\\defaultskewchar");
130      state::assign_value(
131        &s!("skewchar_{shared_key}"),
132        Stored::Number(Number::new(default_skew as i64)),
133        Some(Scope::Global),
134      );
135    }
136    // Perl: installDefinition(FontDef->new($cs, $key))
137    //   FontDef.pm L42: assignValue(current_FontDef => $$self{cs}, 'local')
138    // Perl's State::installDefinition with $scope undef → assign_internal
139    // defaults to local-with-\global-prefix-promotion (State.pm L152).
140    // Rust DefPrimitive! defaults match. TODO (SYNC_STATUS): rewrite this
141    // primitive to a strict Perl-faithful translation — see Work Plan.
142    let is_global = state::get_prefix("global");
143    let cs_for_fontdef = cs;
144    let font_id_key = arena::pin(s!("fontinfo_{cs_str}").as_str());
145    DefPrimitive!(cs, None, None, font => props_opt,
146      before_digest => sub {
147        AssignValue!("current_FontDef", cs_for_fontdef, None);
148      }
149    );
150    // Tag the just-installed primitive with its fontinfo lookup key — this
151    // is the Rust `font_id` (Perl `LaTeXML::Core::Definition::FontDef::fontID`)
152    // that lets dump_writer emit `FD\t<font_id>` (Perl `dump_primitive` path
153    // L383-389) instead of the generic Primitive serialization. Without the
154    // tag, the writer falls into the `PA\t<self_cs>` self-alias path which
155    // dump_reader skips, leaving the CS undefined post-dump.
156    if let Some(Stored::Primitive(p)) = state::lookup_meaning(&cs) {
157      let mut p_owned: latexml_core::definition::primitive::Primitive = (*p).clone();
158      p_owned.font_id = Some(font_id_key);
159      state::assign_meaning(&cs, Stored::Primitive(std::rc::Rc::new(p_owned)),
160        if is_global { Some(Scope::Global) } else { None });
161    }
162    if is_global {
163      if let Some(meaning) = state::lookup_meaning(&cs) {
164        state::assign_meaning(&cs, meaning, Some(Scope::Global));
165      }
166    }
167  });

169  // Perl: DefMacro('\fontname FontDef', sub { Explode($fontinfo && $$fontinfo{name}
170  //       || "fontname not available") })
171  // Perl's FontDef param type: for \font, looks up current_FontDef, falls back to \lx@default@font
172  DefMacro!("\\fontname FontToken", sub[args] {
173    let token = args.into_iter().next().unwrap().expected_token();
174    let cs_str = token.to_string();
175    // Determine which font CS to look up
176    let lookup_cs = if cs_str == "\\font" {
177      // Current font — look up current_FontDef, fallback to \lx@default@font.
178      // with_value avoids the Stored envelope clone; Token is Copy.
179      state::with_value("current_FontDef", |v| match v {
180        Some(Stored::Token(t)) => t.to_string(),
181        _ => s!("\\lx@default@font"),
182      })
183    } else {
184      cs_str
185    };
186    let key = s!("fontinfo_{}", lookup_cs);
187    // Same pattern for the font-name read — we only need the owned name
188    // String, never the Rc<Font>.
189    let name = state::with_value(&key, |v| match v {
190      Some(Stored::Font(f)) => f.name.as_ref().map(|n| n.to_string()),
191      _ => None,
192    });
193    Tokens::new(Explode!(name.unwrap_or_else(|| s!("fontname not available"))))
194  });
195  DefRegister!("\\fontdimen Number FontToken", Dimension::new(0),

482fn non_typewriter(font: &Font) -> bool {
483  font.get_family().unwrap_or(&Cow::Borrowed("")) != "typewriter"
484}
485
486fn non_typewriter_t1(font: &Font) -> bool {
487  non_typewriter(font)
488    && matches!(
489      font
490        .get_encoding()
491        .unwrap_or(&Cow::Borrowed("OT1"))
492        .as_ref(),
493      "OT1" | "T1"
494    )
