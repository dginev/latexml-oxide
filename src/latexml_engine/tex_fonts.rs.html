13  // Font declaration
14  //----------------------------------------------------------------------
15  // \font             iq loads information about a font into TeX's memory.
16  // \fontname         c  returns the system file name for a font.
17  // \fontdimen        iq holds font parameters.
18  // \nullfont         iq is a predefined font with no characters.
19
20  // # 2nd arg is <font> = <fontdef token> | \font | <family member>
21  // #  <family member> = <font range><4bit number>
22  // #  <font range> = \textfont | \scriptfont | \scriptscriptfont
23  // Perl: FontDef param type — for \textfont/\scriptfont/\scriptscriptfont, reads family
24  // number and looks up the stored font CS token.  For \font, returns current_FontDef.
25  DefParameterType!(FontToken, sub[_inner, _extra] {
26    // \textfont/\scriptfont/\scriptscriptfont/\font require a following token; on
27    // input-exhaustion emit the parity "file ended" error and fall back to the
28    // default font CS (which flows through the `else` arm below) rather than panic.
29    let token = read_token_required("\\font")?.unwrap_or_else(|| T_CS!("\\lx@default@font"));
30    if let Some(font_type) = token.with_str(|ts| {
31      if ts.starts_with("\\textfont") && ts == "\\textfont" { Some("textfont") }
32      else if ts.starts_with("\\scriptscriptfont") && ts == "\\scriptscriptfont" { Some("scriptscriptfont") }
33      else if ts.starts_with("\\scriptfont") && ts == "\\scriptfont" { Some("scriptfont") }
34      else { None }
35    }) {
36      // Perl: $token = LookupValue($type . 'font_' . $fam->valueOf).
37      // with_value avoids the Stored envelope clone; Token is Copy.
38      let fam = read_number()?.value_of();
39      let key = s!("{font_type}_{fam}");
40      with_value(&key, |v| match v {
41        Some(Stored::Token(t)) => *t,
42        _ => token,
43      })
44    } else if token.with_str(|ts| ts == "\\font") {
45      // Perl: $token = LookupValue('current_FontDef') || T_CS('\lx@default@font')
46      with_value("current_FontDef", |v| match v {
47        Some(Stored::Token(t)) => *t,
48        _ => T_CS!("\\lx@default@font"),
49      })
50    } else {
51      token
52    }
53  });

55  // Perl TeX_Fonts.pool.ltxml L77-78: default skew/hyphen registers
56  // before \font.
57  DefRegister!("\\defaultskewchar", Number!(-1));
58  DefRegister!("\\defaulthyphenchar", Number!(45)); // ord('-') = 45

60  // Perl: \font SkipSpaces Token SkipSpaces SkipMatch:= SkipSpaces TeXFileName
61  // (TeX_Fonts.pool.ltxml:82). Token does NOT auto-skip spaces in Rust gullet
62  // (parity with Perl readToken), so the leading SkipSpaces is the
63  // Perl-faithful guard for inputs like `\font  \foo  =  cmr10`.
64  DefPrimitive!("\\font SkipSpaces Token SkipSpaces SkipMatch:= SkipSpaces TeXFileName",
65  sub[(cs, name_arg)] {
66    let name = name_arg.to_string();
67    // Read optional "at <dimen>" or "scaled <number>" — Perl: TeX_Fonts.pool.ltxml L88-94
68    //   if    ($gullet->readKeyword('at'))     { $at = $gullet->readDimension; }
69    //   elsif ($gullet->readKeyword('scaled')) { $scaled = $gullet->readNumber/1000; }
70    // The `elsif` is load-bearing: only one branch fires. Without it, an
71    // `at <dim> scaled <n>` input would have BOTH consumed instead of leaving
72    // `scaled <n>` in the stream as Perl does.
73    let mut at_pt = None;
74    let mut at_sp = None;  // raw sp value, needed for shared-key precision
75    let mut scaled = None;
76    if read_keyword(&["at"])?.is_some() {
77      let d = read_dimension()?;
78      at_sp = Some(d.value_of());            // exact sp count, lossless
79      at_pt = Some(d.pt_value(None));        // ≈2-decimal pt for display
80    } else if read_keyword(&["scaled"])?.is_some() {
81      scaled = Some(read_number()?.value_of() as f64 / 1000.0);
82    }
83    let props_opt = if let Some(mut props) = font::decode_fontname(&name, at_pt, scaled) {
84      props.name = Some(Cow::Owned(name.clone()));
85      Some(props)
86    } else { // Failed?
87      let message = s!("Unrecognized font name {:?} Font switch macro {:?}
88      will have no effect", name, cs.stringify());
89      Info!("unexpected", name, message);
90      None
91    };
92    skip_spaces()?;
93    let cs_str = cs.to_string();
94    if let Some(ref props) = props_opt {
95      AssignValue!(&s!("fontinfo_{cs_str}"), props.clone());
96    }
97    // When `scaled <N>` was used, derive `at_sp` from the resolved size
98    // so two `\font` declarations resolving to the same effective size
99    // (e.g. `at 5pt` vs `scaled 500` of cmr10) share the same shared
100    // key — without this, hyphenchar/fontdimen writes through one are
101    // invisible from the other. plainfonts_test L42 catches this.
102    if at_sp.is_none()
103      && let Some(sz_pt) = props_opt.as_ref().and_then(|p| p.size) {
104        at_sp = Some((sz_pt * 65536.0).round() as i64);
105      }
106    // Perl: $key = 'fontinfo_' . $name; $key .= " at " . ToString($at) if $at;
107    // Shared font key: fonts with same name+size share hyphenchar/skewchar.
108    // Precision matters here — expl3's intarray font-hack creates a
109    // distinct `\font \foo = cmr10 at <N> sp` for each intarray, where
110    // <N> is a unique integer (the intarray table index). Rounding to
111    // `.1pt` (or any pt-decimal precision) collapses all these to
112    // `0.0pt` and breaks intarray storage because all "different"
113    // intarrays then share the same fontdimen backing. Use the raw
114    // sp integer (lossless) in the key so e.g. `at 23 sp` and `at 24 sp`
115    // are guaranteed distinct. Keep the pt-formatted form for `\meaning`
116    // display, but separate the two concerns.
117    // `at_str_opt` is used only for the `\meaning` display ("at 5.0pt").
118    // The shared key uses `at_sp` (lossless sp count) for uniqueness.
119    let at_str_opt = if let Some(at_val) = at_pt {
120      Some(s!("{at_val:.1}pt"))
121    } else if let Some(_sc) = scaled {
122      props_opt.as_ref().and_then(|p| p.size).map(|sz| s!("{sz:.1}pt"))
123    } else {
124      None
125    };
126    // Compose the shared key using exact sp value (or `at_str_opt` for
127    // the `scaled` branch which has no sp). Two `\font` calls with
128    // different sp sizes produce different shared keys, so their
129    // fontdimen/hyphenchar/skewchar state stays independent.
130    let shared_key = if let Some(sp) = at_sp {
131      s!("fontinfo_{name} at {sp}sp")
132    } else if let Some(ref at_str) = at_str_opt {
133      s!("fontinfo_{name} at {at_str}")
134    } else {
135      s!("fontinfo_{name}")
136    };
137    // Store CS → shared key mapping
138    assign_value(
139      &s!("font_shared_key_{cs_str}"),
140      Stored::String(pin(&shared_key)),
141      None,
142    );
143    // Store explicit "at" value for \meaning (Perl: $$fontinfo{at} = ToString($at))
144    if let Some(ref at_str) = at_str_opt {
145      assign_value(
146        &s!("fontinfo_at_{cs_str}"),
147        Stored::String(pin(at_str)),
148        None,
149      );
150    }
151    // Perl: only initialize hyphenchar/skewchar if this font key hasn't been seen before
152    // (shared fontinfo means second \font with same name+size reuses existing values)
153    let hc_key = s!("hyphenchar_{shared_key}");
154    if !has_value(&hc_key) {
155      let default_hyphen = lookup_int("\\defaulthyphenchar");
156      assign_value(
157        &hc_key,
158        Stored::Number(Number::new(default_hyphen as i64)),
159        Some(Scope::Global),
160      );
161      let default_skew = lookup_int("\\defaultskewchar");
162      assign_value(
163        &s!("skewchar_{shared_key}"),
164        Stored::Number(Number::new(default_skew as i64)),
165        Some(Scope::Global),
166      );
167    }
168    // Perl: installDefinition(FontDef->new($cs, $key))
169    //   FontDef.pm L42: assignValue(current_FontDef => $$self{cs}, 'local')
170    // Perl's State::installDefinition with $scope undef → assign_internal
171    // defaults to local-with-\global-prefix-promotion (State.pm L152).
172    // Rust DefPrimitive! defaults match. TODO (SYNC_STATUS): rewrite this
173    // primitive to a strict Perl-faithful translation — see Work Plan.
174    let is_global = get_prefix("global");
175    let cs_for_fontdef = cs;
176    let font_id_key = pin(s!("fontinfo_{cs_str}").as_str());
177    // SYNC_STATUS Cluster C: do NOT bypass the lock here. If the target CS
178    // is locked (e.g. `\abstract`, `\title`), `\font\<cs>=<file>` is a
179    // documented no-op — state::install_definition emits an
180    // Info!("ignore", "<cs>:locked", ...) and skips. Witnesses are ~46
181    // pre-2000 plain-TeX-style papers that misuse `\font\abstract=cmr8`
182    // expecting a font-switch group; both Perl and Rust LaTeXML treat
183    // `\font` on a locked primitive as a no-op (SHARED-FAILURE on the
184    // downstream `{\abstract ...}` mode-switch). The author's source
185    // violates a LaTeX convention (shadowing a class-provided macro;
186    // should have used `\newfont` which does an `\@ifundefined` check).
187    // User directive 2026-05-19: "\font on a locked primitive shouldn't
188    // work" — do not surpass Perl here.
189    DefPrimitive!(cs, None, None, font => props_opt,
190      before_digest => sub {
191        AssignValue!("current_FontDef", cs_for_fontdef, None);
192      }
193    );
194    // Tag the just-installed primitive with its fontinfo lookup key — this
195    // is the Rust `font_id` (Perl `LaTeXML::Core::Definition::FontDef::fontID`)
196    // that lets dump_writer emit `FD\t<font_id>` (Perl `dump_primitive` path
197    // L383-389) instead of the generic Primitive serialization. Without the
198    // tag, the writer falls into the `PA\t<self_cs>` self-alias path which
199    // dump_reader skips, leaving the CS undefined post-dump.
200    if let Some(Stored::Primitive(p)) = lookup_meaning(&cs) {
201      let mut p_owned: Primitive = (*p).clone();
202      p_owned.font_id = Some(font_id_key);
203      assign_meaning(&cs, Stored::Primitive(Rc::new(p_owned)),
204        if is_global { Some(Scope::Global) } else { None });
205    }
206    if is_global
207      && let Some(meaning) = lookup_meaning(&cs) {
208        assign_meaning(&cs, meaning, Some(Scope::Global));
209      }
210  });

212  // Perl: DefMacro('\fontname FontDef', sub { Explode($fontinfo && $$fontinfo{name}
213  //       || "fontname not available") })
214  // Perl's FontDef param type: for \font, looks up current_FontDef, falls back to \lx@default@font
215  DefMacro!("\\fontname FontToken", sub[args] {
216    let token = args.into_iter().next().unwrap().expected_token();
217    let cs_str = token.to_string();
218    // Determine which font CS to look up
219    let lookup_cs = if cs_str == "\\font" {
220      // Current font — look up current_FontDef, fallback to \lx@default@font.
221      // with_value avoids the Stored envelope clone; Token is Copy.
222      with_value("current_FontDef", |v| match v {
223        Some(Stored::Token(t)) => t.to_string(),
224        _ => s!("\\lx@default@font"),
225      })
226    } else {
227      cs_str
228    };
229    let key = s!("fontinfo_{}", lookup_cs);
230    // Same pattern for the font-name read — we only need the owned name
231    // String, never the Rc<Font>.
232    let name = with_value(&key, |v| match v {
233      Some(Stored::Font(f)) => f.name.as_ref().map(|n| n.to_string()),
234      _ => None,
235    });
236    Tokens::new(Explode!(name.unwrap_or_else(|| s!("fontname not available"))))
237  });
238  DefRegister!("\\fontdimen Number FontToken", Dimension::new(0),
239    getter => sub[args] {
240      // `\fontdimen` is a parametrized register (Number FontToken). A bare
241      // register lookup with no args — e.g. pgfmath's `pgfmath_register_lookup`
242      // (tikz-cd 1908.10358, 1910.04182) — supplies none. Degrade to the 0pt
243      // default instead of panicking on `args.remove(0)` (pgfmath maps both
244      // None and 0pt to 0.0 anyway).
245      if args.len() < 2 {
246        return Some(Dimension::new(0).into());
247      }
248      let p = args.remove(0).expect_number().value_of();
249      let font_token = args.remove(0).expected_token();
250      let cs_str = font_token.to_string();
251      // Per-font fontdimen<p> override. Resolve to the canonical
252      // font identity via the token's Primitive `font_id` (which is
253      // shared across `\let` aliases — mirrors Perl's FontDef object
254      // sharing on `\let`). Without this indirection, `\let \fb = \fa`
255      // followed by `\fontdimen 1 \fb = 42pt` would store under
256      // `font_shared_key_\fb` while `\the\fontdimen 1 \fb` reads
257      // `font_shared_key_\fa`, producing 0pt.
258      //
259      // Witness: tests/structure/glossary.tex `\Gls{cabbage}` → "Cabbage"
260      // needs expl3's `c__codepoint_uppercase_index_intarray` populated,
261      // and the c__ alias is created via `\cs_gset_eq:cc { c__... }
262      // { g__... }` (=`\let`).
263      let canonical_cs = lookup_meaning(&font_token)
264        .and_then(|m| if let Stored::Primitive(p) = m { p.font_id }
265                      else { None })
266        .map(|fid| {
267          let s = with(fid, |x| x.to_string());
268          s.strip_prefix("fontinfo_").unwrap_or(&s).to_string()
269        })
270        .unwrap_or_else(|| cs_str.clone());
271      let fd_key = with_value(&s!("font_shared_key_{canonical_cs}"), |v| match v {
272        Some(Stored::String(s)) => with(*s, |sk| s!("fontdimen_{sk}_{p}")),
273        _ => s!("fontdimen_{canonical_cs}_{p}"),
274      });
275      let stored_val = with_value(&fd_key, |v| match v {
276        Some(Stored::Dimension(d)) => Some(*d),
277        Some(Stored::Number(n)) => Some(Dimension::new(n.value_of())),
278        _ => None,
279      });
280      if let Some(d) = stored_val { return Some(d.into()); }
281      // Fall-through: hard-coded cmr10-like defaults for indices that
282      // user code commonly reads (\fontdimen2..22) when no explicit write
283      // has happened. Preserves the prior behaviour for math layout code.
284      match p {
285        2 => Dimension::from_str("0.5em").ok()?,    // interword space
286        5 => Dimension::from_str("1ex").ok()?,      // x-height
287        6 => Dimension::from_str("1em").ok()?,      // quad width
288        8 => Dimension::from_str("0.677em").ok()?,  // num1: numerator shift (display)
289        9 => Dimension::from_str("0.394em").ok()?,  // num2: numerator shift (text)
290        10 => Dimension::from_str("0.444em").ok()?, // num3
291        11 => Dimension::from_str("0.686em").ok()?, // denom1: denominator shift (display)
292        12 => Dimension::from_str("0.345em").ok()?, // denom2: denominator shift (text)
293        22 => Dimension::from_str("0.25em").ok()?,  // math axis height (cmsy10: 2.5pt at 10pt)
294        _ => Dimension::new(0)
295      }
296    },
297    setter => sub[value, _scope, args] {
298      // Same parametrized-register guard as the getter: skip a no-arg write
299      // rather than panic on `args.remove(0)`.
300      if args.len() < 2 {
301        return;
302      }
303      let p = args.remove(0).expect_number().value_of();
304      let font_token = args.remove(0).expected_token();
305      let cs_str = font_token.to_string();
306      // Resolve to canonical font identity via Primitive.font_id —
307      // matches the getter so `\let`-aliased fonts share storage.
308      let canonical_cs = lookup_meaning(&font_token)
309        .and_then(|m| if let Stored::Primitive(p) = m { p.font_id }
310                      else { None })
311        .map(|fid| {
312          let s = with(fid, |x| x.to_string());
313          s.strip_prefix("fontinfo_").unwrap_or(&s).to_string()
314        })
315        .unwrap_or_else(|| cs_str.clone());
316      let fd_key = with_value(&s!("font_shared_key_{canonical_cs}"), |v| match v {
317        Some(Stored::String(s)) => with(*s, |sk| s!("fontdimen_{sk}_{p}")),
318        _ => s!("fontdimen_{canonical_cs}_{p}"),
319      });
320      assign_value(
321        &fd_key,
322        Stored::Dimension(value.into()),
323        Some(Scope::Global),
324      );
325    }
326  );

596// Perl: TeX_Fonts.pool.ltxml L338-344
597fn non_typewriter(font: &Font) -> bool {
598  font.get_family().unwrap_or(&Cow::Borrowed("")) != "typewriter"
599}
600
601fn non_typewriter_t1(font: &Font) -> bool {
602  non_typewriter(font)
603    && matches!(
604      font
605        .get_encoding()
606        .unwrap_or(&Cow::Borrowed("OT1"))
607        .as_ref(),
608      "OT1" | "T1"
609    )
