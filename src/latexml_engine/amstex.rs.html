37  // \documentstyle{<style>} — pretend the style is a class for
38  // RequirePackage purposes; fall back to amsppt (Perl L65-73).
39  //
40  // Perl uses `RequirePackage($style, ...) || RequirePackage('amsppt')`,
41  // where `||` checks the BOOLEAN success of InputDefinitions. The
42  // Rust port previously used `is_ok()` on `require_package`, but
43  // `require_package` returns `Ok(())` even when nothing loaded
44  // (the miss-handler fires `Warning:missing_file` and returns OK), so
45  // the fallback to `amsppt` never ran. Mirror Perl's semantics by
46  // probing the `<style>.sty_loaded` flag (or the raw equivalent)
47  // AFTER the call. Witness: arXiv:gr-qc/9507042 — paper has
48  // `\documentstyle{fic}` (fic.sty is a paper-local AmSTeX style
49  // derived from amsppt). Without the fallback, `\leftheadtext` and
50  // 23 other amsppt CSes stayed undefined.
51  DefConstructor!("\\documentstyle Semiverbatim",
52    "<?latexml class='#1' amstex='true'?>",
53    after_digest => sub[whatsit] {
54      let style = whatsit.get_arg(1).map(|d| d.to_string()).unwrap_or_default();
55      let style = style.trim().to_string();
56      let primary_ok = if style.is_empty() { false } else {
57        require_package(&style, RequireOptions {
58          extension: Some(Cow::Borrowed("sty")),
59          notex: Some(true),
60          as_class: true,
61          ..RequireOptions::default()
62        })?;
63        let filename = s!("{}.sty", style);
64        lookup_bool(&s!("{}_loaded", filename))
65          || lookup_bool(&s!("{}_raw_loaded", filename))
66      };
67      if !primary_ok {
68        require_package("amsppt", RequireOptions::default())?;
69      }
70    });
