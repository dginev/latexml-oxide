8//! ## Status (2026-05-15): Phase 1 foundation
9//!
10//! Per [`docs/archive/BIBTEX_PORT_PLAN_2026-06-20.md`](../../../docs/archive/BIBTEX_PORT_PLAN_2026-06-20.md),

42/// One BibTeX-style bibliography entry. Mirrors Perl's per-entry hash
43/// object (`LaTeXML/Engine/BibTeX.pool.ltxml` uses
44/// `$entry->getField(...)`, `$entry->addField(...)` etc.).
45///
46/// `fields` are *processed* token values (the post-digestion form).
47/// `raw_fields` are the *raw* verbatim source strings as they
48/// appeared in the input (used by BibTeX-flavoured re-rendering in
49/// `\bib@@title`'s case-folding and the `\bib@@origbibentry`
50/// roundtrip).
51#[derive(Debug, Clone, Default)]
52pub struct BibEntry {
53  pub key:        String,
54  pub entry_type: String,
55  fields:         Vec<(String, Tokens)>,
56  raw_fields:     Vec<(String, String)>,
57}
58
59impl BibEntry {
60  pub fn new(key: impl Into<String>, entry_type: impl Into<String>) -> Self {
61    Self {
62      key: key.into(),
63      entry_type: entry_type.into(),
64      ..Self::default()
65    }
66  }
67
68  /// Perl: `$entry->getField($name)` — first match wins (Perl uses
69  /// the same first-set-wins semantics; multiple `addField` calls
70  /// for the same name are stored in order, and `getField` returns
71  /// the first).
72  pub fn get_field(&self, name: &str) -> Option<&Tokens> {
73    self.fields.iter().find(|(k, _)| k == name).map(|(_, v)| v)
74  }
75
76  /// Perl: `$entry->addField($name, $value)`. If the field already
77  /// exists, the new value is *appended* — Perl's behaviour for
78  /// authors/editors that name-merge across entries. For typical
79  /// single-value fields the caller adds at most once.
80  pub fn add_field(&mut self, name: impl Into<String>, value: Tokens) {
81    self.fields.push((name.into(), value));
82  }
83
84  /// Perl: `$entry->getRawField($name)` — raw source string.
85  pub fn get_raw_field(&self, name: &str) -> Option<&str> {
86    self
87      .raw_fields
88      .iter()
89      .find(|(k, _)| k == name)
90      .map(|(_, v)| v.as_str())
91  }
92
93  pub fn add_raw_field(&mut self, name: impl Into<String>, value: impl Into<String>) {
94    self.raw_fields.push((name.into(), value.into()));
95  }
96
97  /// Iterate all field names (in insertion order). Used by
98  /// crossref-copy and entry-completion logic.
99  pub fn field_names(&self) -> impl Iterator<Item = &str> {
100    self.fields.iter().map(|(k, _)| k.as_str())
101  }
102
103  /// Render the entry in BibTeX source format. Perl
104  /// `LaTeXML::Util::BibTeX::Entry::prettyPrint`. Output shape:
105  /// ```text
106  /// @article{Smith2020,
107  ///   author = {John Smith},
108  ///   title = {On Examples},
109  ///   year = {2020}
110  /// }
111  /// ```
112  /// Uses raw fields (not digested), since the goal is to capture
113  /// the original BibTeX-flavoured source for the
114  /// `\bib@@origbibentry` round-trip.
115  pub fn pretty_print(&self) -> String {
116    let mut out = format!("@{}{{{}", self.entry_type, self.key);
117    if self.raw_fields.is_empty() {
118      out.push('}');
119      return out;
120    }
121    for (k, v) in &self.raw_fields {
122      // Skip internal `_*` raw fields (e.g. `_raw_keyvals` added by
123      // the amsrefs Phase-2 stub).
124      if k.starts_with('_') {
125        continue;
126      }
127      // Match Perl's source reconstruction: each field on its own line,
128      // the name right-justified so the `=` aligns — `max(1, 10 - len)`
129      // leading spaces (≥1 space even for names ≥10 chars). The entry's
130      // closing `}` follows the last value directly (no newline before it).
131      let lead = 10usize.saturating_sub(k.len()).max(1);
132      out.push_str(",\n");
133      out.push_str(&" ".repeat(lead));
134      out.push_str(k);
135      out.push_str(" = {");
136      out.push_str(v);
137      out.push('}');
138    }
139    out.push('}');
140    out
141  }
142}

157/// Register a fresh `BibEntry` under its normalized key, and set it
158/// as the current entry. Returns an `Rc<RefCell<...>>` so callers
159/// can hold a handle while the registry retains its copy.
160///
161/// Perl roughly: `AssignValue('BIBENTRY@'.NormalizeBibKey($key),
162/// $entry, 'global'); AssignValue('current_bib_entry', $entry,...)`.
163pub fn register_entry(key: &str, entry: BibEntry) -> Rc<RefCell<BibEntry>> {
164  use latexml_core::common::cleaners::normalize_bib_key;
165  let normkey = normalize_bib_key(key);
166  let cell = Rc::new(RefCell::new(entry));
167  BIB_ENTRIES.with(|m| {
168    m.borrow_mut().insert(normkey.clone(), cell.clone());
169  });
170  CURRENT_ENTRY_KEY.with(|k| *k.borrow_mut() = Some(normkey));
171  cell
172}

178  let normkey = normalize_bib_key(key);

240/// Perl: `copyCrossrefFields(@fields)` — for each named field, if
241/// the current entry doesn't already have it but its crossref'd
242/// parent does, copy the value over (processed and raw paths). The
243/// crossref field itself is read from `current_entry().get_field`.
244pub fn copy_crossref_fields(fields: &[&str]) {
245  let Some(current) = current_entry() else {
246    return;
247  };
248  // Get the crossref target's raw key; if no crossref, nothing to do.
249  let xref_key = current
250    .borrow()
251    .get_raw_field("crossref")
252    .map(str::to_string);
253  let Some(xref_key) = xref_key else {
254    return;
255  };
256  let Some(parent) = lookup_entry(&xref_key) else {
257    return;
258  };
259  // Self-crossref is a paper bug; skip silently to avoid infinite
260  // looping if a user writes `crossref={selfkey}`.
261  if Rc::ptr_eq(&current, &parent) {
262    return;
263  }
264  let parent_b = parent.borrow();
265  let mut current_b = current.borrow_mut();
266  for field in fields {
267    if current_b.get_field(field).is_none()
268      && let Some(v) = parent_b.get_field(field)
269    {
270      current_b.add_field(*field, v.clone());
271    }
272    if current_b.get_raw_field(field).is_none()
273      && let Some(v) = parent_b.get_raw_field(field)
274    {
275      current_b.add_raw_field(*field, v.to_string());
276    }
277  }
278}

288/// FxHashMap alias used by `bib_add_to_container` — matches

293/// Build the find-or-create XPath used by `bib_add_to_container`.
294/// Perl `BibTeX.pool.ltxml:248-250`: `$tag[@k1='v1' and @k2='v2']`
295/// with the attribute keys sorted so identical attr sets always
296/// produce identical xpaths (cache hit on the second call).
297fn bib_container_xpath(tag: &str, attrs: &FxAttrMap) -> String {
298  if attrs.is_empty() {
299    return tag.to_string();
300  }
301  let mut keys: Vec<&String> = attrs.keys().collect();
302  keys.sort();
303  let preds: Vec<String> = keys
304    .iter()
305    .map(|k| format!("@{}='{}'", k, attrs.get(*k).unwrap()))
306    .collect();
307  format!("{tag}[{}]", preds.join(" and "))
308}

310/// Perl `bibAddToContainer($document, $tag, $data, %attr)` —
311/// `BibTeX.pool.ltxml:242-257`. Find-or-create a child of the
312/// current `<ltx:bibentry>` / `<ltx:bib-related>` ancestor with
313/// matching tag+attrs. If found, absorb `data` into it; if not,
314/// insert a fresh element with `data` as content.
315///
316/// Used by `\bib@addto@related` (deduplicates same-(type,role)
317/// bib-related groups, e.g. all authors collected in one
318/// `<ltx:bib-related role='authors'>`).
319pub fn bib_add_to_container(
320  doc: &mut Document,
321  tag: &str,
322  data: Option<&Digested>,
323  attrs: FxAttrMap,
324) -> Result<()> {
325  let current = doc.get_node().clone();
326  let entry = doc.findnode(
327    "ancestor-or-self::ltx:bibentry | ancestor-or-self::ltx:bib-related",
328    Some(&current),
329  );
330  let xpath = bib_container_xpath(tag, &attrs);
331  match doc.findnode(&xpath, entry.as_ref()) {
332    Some(rel) => {
333      doc.set_node(&rel);
334      if let Some(d) = data {
335        doc.absorb(d, None)?;
336      }
337      doc.set_node(&current);
338    },
339    _ => {
340      let content: Vec<&Digested> = match data {
341        Some(d) => vec![d],
342        None => vec![],
343      };
344      doc.insert_element(tag, content, Some(attrs))?;
345    },
346  }
347  Ok(())
348}

373/// Perl `splitWords` (`BibTeX.pool.ltxml` L921-949). Split a name
374/// string into words on whitespace / comma / tilde, but treat
375/// `{balanced}` groups as a single word. Commas survive as their
376/// own tokens. `\~` is preserved (Perl protects it via the
377/// `####` placeholder trick — we use a sentinel `\x00` byte
378/// which can't appear in TeX source).
379fn split_words(input: &str) -> Vec<String> {
380  // 1. Protect `\~`, normalise leading whitespace + `%\n` line continuations.
381  const PLACEHOLDER: &str = "\x00";
382  let s = input.replace("\\~", PLACEHOLDER);
383  let s = s.trim_start_matches(|c: char| c.is_whitespace() || c == '~');
384  let s = s.replace("%\n", "");
385
386  let mut words: Vec<String> = Vec::new();
387  let mut word = String::new();
388  let bytes = s.as_bytes();
389  let mut i = 0;
390  while i < bytes.len() {
391    let b = bytes[i];
392    // Check for `(comma?) whitespace+` separators. Perl regex:
393    // s/^(,?)[\s~]+//
394    if b == b',' || b == b' ' || b == b'\t' || b == b'\n' || b == b'\r' || b == b'~' {
395      let had_comma = b == b',';
396      let mut j = i + 1;
397      // Either a leading comma we just consumed, or the whole run is
398      // pure whitespace/tilde — collect the trailing whitespace.
399      if had_comma {
400        while j < bytes.len() && matches!(bytes[j], b' ' | b'\t' | b'\n' | b'\r' | b'~') {
401          j += 1;
402        }
403      } else {
404        // Pure whitespace run; skip them.
405        while j < bytes.len() && matches!(bytes[j], b' ' | b'\t' | b'\n' | b'\r' | b'~') {
406          j += 1;
407        }
408        // If we didn't actually see any whitespace beyond this one
409        // char, we still advance (the `[\s~]+` pattern matches `+`
410        // which requires ≥1; we have 1 = the current char).
411      }
412      // Flush accumulated word
413      if !word.is_empty() {
414        words.push(std::mem::take(&mut word));
415      }
416      if had_comma {
417        words.push(",".to_string());
418      }
419      i = j;
420    } else if b == b'{' {
421      // Extract balanced group; include the braces.
422      let start = i;
423      let mut depth = 0i32;
424      while i < bytes.len() {
425        match bytes[i] {
426          b'{' => depth += 1,
427          b'}' => {
428            depth -= 1;
429            if depth == 0 {
430              i += 1;
431              break;
432            }
433          },
434          _ => {},
435        }
436        i += 1;
437      }
438      // Append the entire braced chunk (including the braces) to the
439      // current word so it stays atomic across word splits — matches
440      // Perl `$word .= $t`.
441      word.push_str(&s[start..i]);
442    } else {
443      // Greedily accumulate until the next separator / `{`.
444      let start = i;
445      while i < bytes.len()
446        && !matches!(bytes[i], b' ' | b'\t' | b'\n' | b'\r' | b'~' | b',' | b'{')
447      {
448        i += 1;
449      }
450      word.push_str(&s[start..i]);
451    }
452  }
453  if !word.is_empty() {
454    words.push(word);
455  }
456  // 6. Restore `\~`.
457  for w in words.iter_mut() {
458    if w.contains(PLACEHOLDER) {
459      *w = w.replace(PLACEHOLDER, "\\~");
460    }
461  }
462  words
463}

465/// Perl `processBibNameList` (`BibTeX.pool.ltxml` L872-918): parse a
466/// BibTeX-style author/editor list into structured names. The input
467/// is a raw string (Perl uses `UnTeX($names, 1)` to flatten Tokens
468/// first). Returns the parsed names plus an `etal` flag.
469///
470/// Name shapes recognised (BibTeX `bibtex.web` convention):
471///   - `First Last` — given before surname; the surname starts at the first lowercase word (e.g.
472///     `von`, `de la`) or, in the no-lowercase case, at the LAST word.
473///   - `Last, First` — surname first; one comma.
474///   - `Last, Jr., First` — surname + lineage + given; two commas.
475///
476/// Multiple names separated by ` and ` (case-insensitive). A
477/// trailing `and others` / `and et al.` sets `etal = true`.
478pub fn process_bib_name_list(input: &str) -> BibNameList {
479  let mut words = split_words(input);
480  let mut etal = false;
481  // Detect trailing `and others` / `and et al(.)?` and strip the
482  // last two words if present. Perl L878.
483  if words.len() >= 2 {
484    let last = &words[words.len() - 1];
485    let prev = &words[words.len() - 2];
486    if prev.eq_ignore_ascii_case("and") {
487      let last_lc = last.to_ascii_lowercase();
488      let last_lc = last_lc.trim_end_matches('.');
489      if matches!(last_lc, "others" | "et al" | "etal") {
490        words.pop();
491        words.pop();
492        etal = true;
493      }
494    }
495  }
496  let mut names: Vec<BibName> = Vec::new();
497  while !words.is_empty() {
498    // Collect words for one name, splitting comma-delimited phrases.
499    let mut phrases: Vec<Vec<String>> = Vec::new();
500    let mut phrase: Vec<String> = Vec::new();
501    while !words.is_empty() {
502      let word = words.remove(0);
503      if word.eq_ignore_ascii_case("and") {
504        break;
505      }
506      if word == "," {
507        phrases.push(std::mem::take(&mut phrase));
508      } else {
509        phrase.push(word);
510      }
511    }
512    if phrase.is_empty() && phrases.is_empty() {
513      // Empty name (consecutive `and`s, or stray comma+and). Perl
514      // emits Warn; we silently skip — the caller's Tokens output
515      // would be a no-op anyway.
516      continue;
517    }
518    if !phrase.is_empty() {
519      phrases.push(phrase);
520    }
521
522    let (given, surname, lineage) = match phrases.len() {
523      3 => {
524        // "von Last, Jr, First"
525        let surname = phrases[0].join(" ");
526        let lineage = phrases[1].join(" ");
527        let given = phrases[2].join(" ");
528        (given, surname, lineage)
529      },
530      2 => {
531        // "von Last, First"
532        let surname = phrases[0].join(" ");
533        let given = phrases[1].join(" ");
534        (given, surname, String::new())
535      },
536      _ => {
537        // "First [von] Last" — words before the FIRST lowercase
538        // word are given; the rest are surname. If no lowercase
539        // word, the LAST word is surname and the rest are given.
540        let pwords = &phrases[0];
541        let mut first: Vec<String> = Vec::new();
542        let mut rest: Vec<String> = pwords.clone();
543        while !rest.is_empty() && !starts_lowercase(&rest[0]) {
544          first.push(rest.remove(0));
545        }
546        if rest.is_empty() && !first.is_empty() {
547          // No lowercase word — move the last `first` word into rest.
548          rest.push(first.pop().unwrap());
549        }
550        let given = first.join(" ");
551        let surname = rest.join(" ");
552        (given, surname, String::new())
553      },
554    };
555    names.push(BibName { given, surname, lineage });
556  }
557  BibNameList { names, etal }
558}

560/// Helper for `process_bib_name_list`: does a word start with a
561/// lowercase letter? Perl test: `$word !~ /^[a-z]/` (NOT lowercase).
562/// We invert that. Word may have a leading `{...}` group — peek past
563/// it to find the first actual letter. Words starting with a control
564/// sequence like `\von` are treated as lowercase by inspecting the
565/// CS name's first letter (matches Perl behaviour for `\von Last`
566/// where the regex sees `\` which IS lowercase via the
567/// `[a-z]` heuristic — Perl's comment "This case test is not
568/// correct!!!" acknowledges this is approximate).
569fn starts_lowercase(word: &str) -> bool {
570  // Strip a single leading `{...}` envelope (BibTeX convention for
571  // forcing capitalisation: `{Smith}` should be opaque).
572  let stripped = if let Some(rest) = word.strip_prefix('{') {
573    rest.trim_end_matches('}')
574  } else {
575    word
576  };
577  // Skip a leading `\` (control sequence) — peek at the next char.
578  let stripped = stripped.strip_prefix('\\').unwrap_or(stripped);
579  stripped.chars().next().is_some_and(|c| c.is_lowercase())
580}

613/// Perl `\bib@@title` body (`BibTeX.pool.ltxml:293-332`): re-case a
614/// title string while preserving brace-grouped and `$math$`-delimited
615/// regions verbatim. Words are runs of `\w` chars or `\<word>`/
616/// `\<single-char>` control-sequence escapes; whitespace separates
617/// words and may modify the next word's "first" status.
618///
619/// `wb` (word-beginning) is 1 at the start of a word run; `wc` (word
620/// counter) increments at each word start. Together they let the
621/// `Capitalize1` mode capitalise the FIRST word and lowercase the
622/// rest, while `Capitalize` uppercases every word.
623pub fn recase_title(title: &str, mode: TitleCaseMode) -> String {
624  let bytes = title.as_bytes();
625  let mut out = String::with_capacity(title.len());
626  let mut wb: bool = true;
627  let mut wc: u32 = 0;
628  let mut i = 0;
629  while i < bytes.len() {
630    let b = bytes[i];
631    // Whitespace run — copy verbatim, set word-beginning.
632    if b == b' ' || b == b'\t' || b == b'\n' || b == b'\r' {
633      let start = i;
634      while i < bytes.len() && matches!(bytes[i], b' ' | b'\t' | b'\n' | b'\r') {
635        i += 1;
636      }
637      out.push_str(&title[start..i]);
638      wb = true;
639      continue;
640    }
641    // Balanced `{...}` — copy verbatim, atomic word.
642    if b == b'{' {
643      let start = i;
644      let mut depth = 0i32;
645      while i < bytes.len() {
646        match bytes[i] {
647          b'{' => depth += 1,
648          b'}' => {
649            depth -= 1;
650            if depth == 0 {
651              i += 1;
652              break;
653            }
654          },
655          _ => {},
656        }
657        i += 1;
658      }
659      if wb {
660        wc += 1;
661      }
662      out.push_str(&title[start..i]);
663      wb = false;
664      continue;
665    }
666    // Balanced `$...$` — copy verbatim, no word-counter bump.
667    if b == b'$' {
668      let start = i;
669      i += 1;
670      while i < bytes.len() && bytes[i] != b'$' {
671        i += 1;
672      }
673      if i < bytes.len() {
674        i += 1;
675      }
676      out.push_str(&title[start..i]);
677      wb = false;
678      continue;
679    }
680    // Word: ASCII alphanumeric/underscore OR `\<word>` / `\<char>` escape.
681    let word_start = i;
682    let mut consumed_word = false;
683    loop {
684      if i >= bytes.len() {
685        break;
686      }
687      let c = bytes[i];
688      let is_wordchar = c.is_ascii_alphanumeric() || c == b'_';
689      if is_wordchar {
690        i += 1;
691        consumed_word = true;
692        continue;
693      }
694      if c == b'\\' && i + 1 < bytes.len() {
695        // \<word> or \<single-char>
696        i += 1;
697        if bytes[i].is_ascii_alphabetic() {
698          while i < bytes.len() && bytes[i].is_ascii_alphabetic() {
699            i += 1;
700          }
701        } else {
702          i += 1;
703        }
704        consumed_word = true;
705        continue;
706      }
707      break;
708    }
709    if consumed_word {
710      let word = &title[word_start..i];
711      let recased = match mode {
712        TitleCaseMode::AsIs => word.to_string(),
713        TitleCaseMode::Uppercase => word.to_uppercase(),
714        _ if !wb
715          || (mode == TitleCaseMode::Capitalize1 && wc > 0)
716          || mode == TitleCaseMode::Lowercase =>
717        {
718          word.to_lowercase()
719        },
720        TitleCaseMode::Capitalize | TitleCaseMode::Capitalize1 => ucfirst(word),
721        _ => word.to_string(),
722      };
723      out.push_str(&recased);
724      if wb {
725        wc += 1;
726      }
727      wb = false;
728      continue;
729    }
730    // Fallback single char (e.g. punctuation).
731    let ch_start = i;
732    let mut chars = title[i..].chars();
733    if let Some(c) = chars.next() {
734      i += c.len_utf8();
735    } else {
736      i += 1;
737    }
738    out.push_str(&title[ch_start..i]);
739    wb = true;
740  }
741  out
742}

750/// Parse an amsrefs-style keyval string `"key = {value}, key = value, ..."`
751/// into a list of `(name, value)` pairs. Names are lowercased; values
752/// have their outer `{...}` envelope stripped (if any) and surrounding
753/// whitespace trimmed.
754///
755/// Used by the amsrefs `\bib{key}{type}{keyvals}` stub to populate
756/// `BibEntry::raw_fields` for downstream `\ProcessBibTeXEntry`
757/// dispatch. Mirrors Perl `$keyvals->getPairs` + `lc()` + `UnTeX()`
758/// from `amsrefs.sty.ltxml:42-50`.
759pub fn parse_amsrefs_keyvals(s: &str) -> Vec<(String, String)> {
760  let bytes = s.as_bytes();
761  let mut out: Vec<(String, String)> = Vec::new();
762  let mut i = 0;
763  while i < bytes.len() {
764    // Skip whitespace + commas.
765    while i < bytes.len() && matches!(bytes[i], b' ' | b'\t' | b'\n' | b'\r' | b',') {
766      i += 1;
767    }
768    if i >= bytes.len() {
769      break;
770    }
771    // Read key up to `=` (or end / `,`).
772    let key_start = i;
773    while i < bytes.len() && !matches!(bytes[i], b'=' | b',') {
774      i += 1;
775    }
776    let key = s[key_start..i].trim().to_ascii_lowercase();
777    if key.is_empty() {
778      // Stray comma or trailing garbage; skip the separator and continue.
779      if i < bytes.len() {
780        i += 1;
781      }
782      continue;
783    }
784    if i >= bytes.len() || bytes[i] != b'=' {
785      // No `=` — key without value; record empty value.
786      out.push((key, String::new()));
787      continue;
788    }
789    i += 1; // skip `=`
790    // Skip whitespace before value.
791    while i < bytes.len() && matches!(bytes[i], b' ' | b'\t' | b'\n' | b'\r') {
792      i += 1;
793    }
794    // Value: balanced `{...}` group OR until next top-level `,`.
795    let value: String;
796    if i < bytes.len() && bytes[i] == b'{' {
797      let start = i + 1;
798      let mut depth = 1i32;
799      i += 1;
800      while i < bytes.len() && depth > 0 {
801        match bytes[i] {
802          b'{' => depth += 1,
803          b'}' => depth -= 1,
804          _ => {},
805        }
806        if depth == 0 {
807          break;
808        }
809        i += 1;
810      }
811      value = s[start..i].to_string();
812      if i < bytes.len() {
813        i += 1;
814      } // skip closing `}`
815    } else {
816      let start = i;
817      while i < bytes.len() && bytes[i] != b',' {
818        i += 1;
819      }
820      value = s[start..i].trim().to_string();
821    }
822    out.push((key, value));
823  }
824  out
825}

827/// Perl `ucfirst($s)` — uppercase the first char, leave the rest.
828fn ucfirst(s: &str) -> String {
829  let mut chars = s.chars();
830  match chars.next() {
831    Some(c) => {
832      let mut out = String::with_capacity(s.len());
833      for upper in c.to_uppercase() {
834        out.push(upper);
835      }
836      out.push_str(chars.as_str());
837      out
838    },
839    None => String::new(),
840  }
841}

1358  // Take the raw year, optionally append `-<month>` (mapped from
1359  // English/abbrev names to digit) and `-<day>`. Then re-emit as a
1360  // `\bib@field@default@date{<date>}` invocation. If a `date` field
1361  // is already set, ignore the year (per Perl `currentBibEntryField('date')`).
1362  DefMacro!("\\bib@field@default@year {}", sub[args] {
1363    if current_entry_field("date").is_some() {
1364      return Ok(Tokens!());
1365    }
1366    let mut date = if args[0].is_some() { args[0].to_string() } else {
1367      current_entry_field("year").map(|t| t.to_string())
1368        .or_else(|| current_entry_raw_field("year"))
1369        .unwrap_or_default()
1370    };
1371    let month_lookup = |s: &str| -> Option<&'static str> {
1372      match s.to_lowercase().as_str() {
1373        "jan" | "january" => Some("1"),
1374        "feb" | "february" => Some("2"),
1375        "mar" | "march" => Some("3"),
1376        "apr" | "april" => Some("4"),
1377        "may" => Some("5"),
1378        "jun" | "june" => Some("6"),
1379        "jul" | "july" => Some("7"),
1380        "aug" | "august" => Some("8"),
1381        "sep" | "september" => Some("9"),
1382        "oct" | "october" => Some("10"),
1383        "nov" | "november" => Some("11"),
1384        "dec" | "december" => Some("12"),
1385        _ => None,
1386      }
1387    };
1388    let month_str: Option<String> = current_entry_raw_field("month")
1389      .as_deref()
1390      .and_then(month_lookup)
1391      .map(str::to_string)
1392      .or_else(|| {
1393        current_entry_field("month").and_then(|t| {
1394          let s = t.to_string();
1395          month_lookup(&s).map(str::to_string).or(Some(s))
1396        })
1397      });
1398    if let Some(mstr) = month_str {
1399      // Pad single digit month with leading 0.
1400      let mpad = if mstr.len() == 1 && mstr.chars().all(|c| c.is_ascii_digit()) {
1401        format!("0{mstr}")
1402      } else { mstr };
1403      date.push('-');
1404      date.push_str(&mpad);
1405      if let Some(day) = current_entry_field("day") {
1406        let day_s = day.to_string();
1407        let dpad = if day_s.len() == 1 && day_s.chars().all(|c| c.is_ascii_digit()) {
1408          format!("0{day_s}")
1409        } else { day_s };
1410        date.push('-');
1411        date.push_str(&dpad);
1412      }
1413    }

1415    // would strip the arg here since `\bib@field@default@date` is a
1416    // parameter-less DefMacro whose expansion expects the value to be
1417    // sitting in the input stream (so `\bib@@field`'s Digested arg

1419    // `(T_CS('\bib@field@default@date'), T_BEGIN, Tokenize($date)->unlist, T_END)`.

1423    out_toks.extend(Tokenize!(date.as_str()).unlist());
1424    out_toks.push(T_END!());
1425    Ok(Tokens::new(out_toks))
1426  });

1726  // -------- Phase 6: orchestration (Perl L111-190) --------
1727  // `\ProcessBibTeXEntry{key}` drives the per-entry pipeline:
1728  //   1. resolve type alias chain
1729  //   2. dispatch prepare macros (type → alias → default)
1730  //   3. open `<ltx:bibentry>` (via the `{bib@entry}` environment)
1731  //   4. dispatch each field via the most specific handler
1732  //   5. dispatch complete macros (type → alias → default)
1733  //   6. close `<ltx:bibentry>`
1734  //
1735  // The Perl pool splits this across two DefPrimitives
1736  // (`\bibentry@prepare` + `\bibentry@create`) with a `\stomach->bgroup;
1737  // AssignValue('CURRENT@BIBKEY' => $key); ...; egroup` dance that
1738  // gives the per-entry state automatic group-scope restore. The
1739  // Rust port does it all in one DefMacro returning a Tokens stream,
1740  // since our gullet handles the tokens-back path natively (no
1741  // openMouth needed). Divergence is documented under audit B1
1742  // (`current_bib_key` rustdoc + `docs/archive/BIBTEX_PORT_PLAN_2026-06-20.md`).
1743  DefMacro!("\\ProcessBibTeXEntry Semiverbatim", sub[args] {
1744    let key = if args[0].is_some() { args[0].to_string() } else {
1745      return Ok(Tokens!());
1746    };
1747    let entry_rc = match lookup_entry(&key) {
1748      Some(e) => e,
1749      None => return Ok(Tokens!()),
1750    };
1751    let entry = entry_rc.borrow();
1752    let origtype = entry.entry_type.clone();
1753
1754    // Alias resolution: if `\bib@entry@<origtype>@alias` is defined,
1755    // its expansion is the resolved type. Otherwise resolved == orig.
1756    let alias_cs_name = format!("\\bib@entry@{}@alias", origtype);
1757    let alias_tok = T_CS!(alias_cs_name.as_str());
1758    let resolved_type = match lookup_definition(&alias_tok)? {
1759      Some(_) => {
1760        // Expand the alias CS via the gullet to get the target type
1761        // name. Most aliases are pure-text DefMacros (e.g. "thesis"),
1762        // so a single do_expand is enough.
1763        match do_expand(alias_tok) {
1764          Ok(toks) => toks.to_string(),
1765          Err(_) => origtype.clone(),
1766        }
1767      },
1768      None => origtype.clone(),
1769    };
1770
1771    // Set the current bib key so the per-field handlers (which call
1772    // `current_entry_field` etc.) see the right entry.
1773    set_current_entry(&key);
1774
1775    let mut out: Vec<Token> = Vec::new();
1776    // `\begin{bib@entry}{<type>}{<key>}`
1777    out.push(T_CS!("\\begin"));
1778    out.push(T_BEGIN!());
1779    out.extend(Explode!("bib@entry"));
1780    out.push(T_END!());
1781    out.push(T_BEGIN!());
1782    out.extend(Explode!(resolved_type.as_str()));
1783    out.push(T_END!());
1784    out.push(T_BEGIN!());
1785    out.extend(Explode!(&key));
1786    out.push(T_END!());
1787
1788    // Dispatch prepare macros. Perl L128-131: prepare for the
1789    // resolved type, then the orig type if different, then default.
1790    let prepare_csnames = [
1791      format!("\\bib@entry@{}@prepare", resolved_type),
1792      if origtype != resolved_type { format!("\\bib@entry@{}@prepare", origtype) } else { String::new() },
1793      "\\bib@entry@default@prepare".to_string(),
1794    ];
1795    for cs_name in &prepare_csnames {
1796      if cs_name.is_empty() { continue; }
1797      let tok = T_CS!(cs_name.as_str());
1798      if lookup_definition(&tok)?.is_some() {
1799        out.push(tok);
1800      }
1801    }
1802
1803    // Dispatch each field via the most specific handler. Perl L147-157.
1804    for (field, value) in entry.raw_fields.iter() {
1805      if field.starts_with('_') { continue; }  // internal fields
1806      let candidates = [
1807        format!("\\bib@field@{}@{}", resolved_type, field),
1808        if origtype != resolved_type { format!("\\bib@field@{}@{}", origtype, field) } else { String::new() },
1809        format!("\\bib@field@default@{}", field),
1810      ];
1811      let mut handler: Option<&str> = None;
1812      for c in candidates.iter() {
1813        if c.is_empty() { continue; }
1814        let tok = T_CS!(c.as_str());
1815        if lookup_definition(&tok)?.is_some() {
1816          handler = Some(c.as_str());
1817          break;
1818        }
1819      }
1820      match handler {
1821        Some(h) => {
1822          // `\csname <h>\endcsname{value}`
1823          out.push(T_CS!(h));
1824          out.push(T_BEGIN!());
1825          out.extend(Explode!(value));
1826          out.push(T_END!());
1827        },
1828        None => {
1829          // Fallback per Perl L157: `\bib@field@default@default{field}{value}`.
1830          out.push(T_CS!("\\bib@field@default@default"));
1831          out.push(T_BEGIN!());
1832          out.extend(Explode!(field));
1833          out.push(T_END!());
1834          out.push(T_BEGIN!());
1835          out.extend(Explode!(value));
1836          out.push(T_END!());
1837        },
1838      }
1839    }
1840
1841    // Dispatch complete macros (Perl L158-164).
1842    let complete_csnames = [
1843      format!("\\bib@entry@{}@complete", resolved_type),
1844      if origtype != resolved_type { format!("\\bib@entry@{}@complete", origtype) } else { String::new() },
1845      "\\bib@entry@default@complete".to_string(),
1846    ];
1847    for cs_name in &complete_csnames {
1848      if cs_name.is_empty() { continue; }
1849      let tok = T_CS!(cs_name.as_str());
1850      if lookup_definition(&tok)?.is_some() {
1851        out.push(tok);
1852      }
1853    }
1854
1855    // `\end{bib@entry}`
1856    out.push(T_CS!("\\end"));
1857    out.push(T_BEGIN!());
1858    out.extend(Explode!("bib@entry"));
1859    out.push(T_END!());
1860
1861    Ok(Tokens::new(out))
1862  });
