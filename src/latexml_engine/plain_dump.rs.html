1//! Runtime loader for `resources/dumps/plain.dump.txt` — Perl-parity for
2//! `LoadFormat('plain')`. Mirrors `latex_dump.rs` (which pulls a generated
3//! loader from `OUT_DIR/latex_dump_loader.rs`) but is hand-written here to
4//! keep the dump pipeline surgical: build.rs emits exactly one loader for
5//! the LaTeX dump, and this file is the parallel one for the Plain dump.
6//!
7//! Load order (mirrors Perl `TeX.pool.ltxml: LoadFormat('plain')`):
8//!   plain_bootstrap → plain_base → **plain_dump (this)** → plain_constructs
9//!
10//! Resolution order matches `latex_dump_loader.rs`:
11//!   0. `$LATEXML_NODUMP` → skip (Perl `Package.pm` `LoadFormat` parity)
12//!   1. `$LATEXML_PLAIN_DUMP_PATH` (explicit override, full path)
13//!   2. `$LATEXML_DUMP_DIR/plain.dump.txt` (directory override)
14//!   3. `<exe_dir>/../resources/dumps/plain.dump.txt` (installed layout)
15//!   4. dev-tree path (compile-time `CARGO_MANIFEST_DIR/../resources/dumps/plain.dump.txt`)
16//!
17//! If no dump is found we log at info and return `Ok(())` — the engine
18//! falls back to whatever stubs `plain_base.rs` already installed. This
19//! matches Perl's `LoadFormat` graceful-fallback behavior.
20
21use once_cell::sync::Lazy;
22use std::path::{Path, PathBuf};
23
24const DEV_PATH: &str = concat!(
25  env!("CARGO_MANIFEST_DIR"),
26  "/../resources/dumps/plain.dump.txt"
27);
28
29// Process-once cached env vars. Hoisted out of the runtime path because
30// `getenv` is unsafe under high-volume concurrent reads (test harness
31// threads) — see WISDOM #56 in user memory and `gullet::TRACE_GROUP_END`
32// for the underlying cause.
33static NODUMP: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_NODUMP").is_some());
34static PLAIN_DUMP_PATH: Lazy<Option<String>> =
35  Lazy::new(|| std::env::var("LATEXML_PLAIN_DUMP_PATH").ok());
36static DUMP_DIR: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_DIR").ok());
37
38pub fn load_definitions() -> latexml_core::common::error::Result<()> {
39  if *NODUMP {
40    log::info!(
41      "[plain_dump] LATEXML_NODUMP set — skipping dump, engine will reconstruct \
42       plain.tex state from _base pool (slower, Perl-parity)"
43    );
44    return Ok(());
45  }
46  let Some(path) = resolve_dump_path() else {
47    log::info!(
48      "[plain_dump] no dump found (checked $LATEXML_PLAIN_DUMP_PATH, $LATEXML_DUMP_DIR, \
49       exe-relative, and dev-tree path); run `latexml_oxide --init=plain.tex` to generate"
50    );
51    return Ok(());
52  };
53  let content = match std::fs::read_to_string(&path) {
54    Ok(c) => c,
55    Err(e) => {
56      log::warn!("[plain_dump] failed to read {}: {}", path.display(), e);
57      return Ok(());
58    },
59  };
60  let count = latexml_core::dump_reader::load_from_str_plain(&content)
61    .map_err(|e: String| -> latexml_core::common::error::Error { e.into() })?;
62  log::info!(
63    "[plain_dump] loaded {} entries from {}",
64    count,
65    path.display()
66  );
67  Ok(())
68}
69
70fn resolve_dump_path() -> Option<PathBuf> {
71  if let Some(p) = PLAIN_DUMP_PATH.as_deref() {
72    let pb = PathBuf::from(p);
73    if pb.is_file() {
74      return Some(pb);
75    }
76  }
77  if let Some(dir) = DUMP_DIR.as_deref() {
78    let pb = Path::new(dir).join("plain.dump.txt");
79    if pb.is_file() {
80      return Some(pb);
81    }
82  }
83  if let Ok(exe) = std::env::current_exe() {
84    if let Some(exe_dir) = exe.parent() {
85      let installed = exe_dir.join("../resources/dumps/plain.dump.txt");
86      if installed.is_file() {
87        return Some(installed);
88      }
89      let sibling = exe_dir.join("plain.dump.txt");
90      if sibling.is_file() {
91        return Some(sibling);
92      }
93    }
94  }
95  let dev = PathBuf::from(DEV_PATH);
96  if dev.is_file() {
97    return Some(dev);
98  }
99  None
100}
