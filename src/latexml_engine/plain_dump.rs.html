1//! Runtime loader for `plain.YYYY.dump.txt` — Perl-parity for
2//! `LoadFormat('plain')`.
3//!
4//! Mirrors the generated `latex_dump_loader.rs` but is hand-written to
5//! keep the dump pipeline surgical: build.rs emits exactly one loader
6//! (for the LaTeX dump) and this file is the parallel one for Plain.
7//!
8//! Load order (mirrors Perl `TeX.pool.ltxml: LoadFormat('plain')`):
9//!   plain_bootstrap → plain_base → **plain_dump (this)** → plain_constructs
10//!
11//! Resolution order (matches `latex_dump_loader.rs`):
12//!   0. `$LATEXML_NODUMP` → skip (Perl `Package.pm` `LoadFormat` parity)
13//!   1. `$LATEXML_PLAIN_DUMP_PATH` (explicit full path)
14//!   2. `$LATEXML_DUMP_DIR/plain.YYYY.dump.txt` (best year-match in dir; else most-recent)
15//!   3. `<exe_dir>/../resources/dumps/plain.YYYY.dump.txt` (installed layout)
16//!   4. Sibling-of-exe `plain.YYYY.dump.txt`
17//!   5. Dev-tree `$CARGO_MANIFEST_DIR/../resources/dumps/plain.YYYY.dump.txt`
18//!   6. **Embedded dump** (compile-time-bundled snapshot)
19//!
20//! Year selection: prefer ambient TeXLive year (via
21//! [`crate::dump_paths::detect_ambient_texlive_year`]); fall back to the
22//! most-recent year present at each step.
23

30const DEV_DUMPS_DIR: &str = concat!(env!("CARGO_MANIFEST_DIR"), "/../resources/dumps");

32// Process-once cached env vars. Hoisted out of the runtime path because
33// `getenv` is unsafe under high-volume concurrent reads (test harness
34// threads) — see WISDOM #56 in user memory and `gullet::TRACE_GROUP_END`
35// for the underlying cause.
36static NODUMP: Lazy<bool> = Lazy::new(|| std::env::var_os("LATEXML_NODUMP").is_some());
37static PLAIN_DUMP_PATH: Lazy<Option<String>> =
38  Lazy::new(|| std::env::var("LATEXML_PLAIN_DUMP_PATH").ok());
39static DUMP_DIR: Lazy<Option<String>> = Lazy::new(|| std::env::var("LATEXML_DUMP_DIR").ok());

41pub fn load_definitions() -> Result<()> {
42  if *NODUMP {
43    Info!(
44      "plain_dump",
45      "nodump",
46      "LATEXML_NODUMP set — skipping dump, engine will reconstruct \
47       plain.tex state from _base pool (slower, Perl-parity)"
48    );
49    return Ok(());
50  }
51  let prefer = crate::dump_paths::detect_ambient_texlive_year();
52  let (content, source_label) = if let Some((path, _year)) = resolve_dump_path(prefer) {
53    match std::fs::read_to_string(&path) {
54      Ok(c) => (c, path.display().to_string()),
55      Err(e) => {
56        Warn!(
57          "plain_dump",
58          "read",
59          s!("failed to read {}: {}", path.display(), e)
60        );
61        return Ok(());
62      },
63    }
64  } else if let Some(embedded) = crate::embedded_dumps::embedded_plain_dump(prefer) {
65    let year = crate::embedded_dumps::embedded_year(prefer).unwrap_or(0);
66    Info!(
67      "plain_dump",
68      "embedded",
69      s!("using embedded TL{} dump — no on-disk dump found", year)
70    );
71    (embedded.to_string(), format!("<embedded TL{}>", year))
72  } else {
73    Info!(
74      "plain_dump",
75      "missing",
76      "no dump found (checked $LATEXML_PLAIN_DUMP_PATH, $LATEXML_DUMP_DIR, \
77       exe-relative, dev-tree path, and embedded fallback); \
78       run `latexml_oxide --init=plain.tex` to generate"
79    );
80    return Ok(());
81  };
82  // Single load message: `dump_reader:loaded` now names the real `source_label`
83  // (disk path in dev, `<embedded TLyyyy>` in the shipped binary) and carries the
84  // skipped/errors detail — so we no longer emit a second, redundant
85  // `plain_dump:loaded` line for the same load.
86  let _count = dump_reader::load_from_str_labeled(&content, &source_label)
87    .map_err(|e: String| -> Error { e.into() })?;
88  Ok(())
89}

91fn resolve_dump_path(prefer: Option<u32>) -> Option<(PathBuf, u32)> {
92  // 1. Explicit full path.
93  if let Some(p) = PLAIN_DUMP_PATH.as_deref() {
94    let pb = PathBuf::from(p);
95    if pb.is_file() {
96      let year = pb
97        .file_name()
98        .and_then(|n| n.to_str())
99        .and_then(|n| crate::dump_paths::parse_year_from_dump_filename(n, "plain"))
100        .unwrap_or(0);
101      return Some((pb, year));
102    }
103  }
104  // 2. Directory override — pick versioned dump in that directory.
105  if let Some(dir) = DUMP_DIR.as_deref()
106    && let Some(found) =
107      crate::dump_paths::resolve_versioned_in_dir(Path::new(dir), "plain", prefer)
108  {
109    return Some(found);
110  }
111  // 3. Installed layout: <exe_dir>/../resources/dumps/.
112  if let Ok(exe) = std::env::current_exe()
113    && let Some(exe_dir) = exe.parent()
114  {
115    let installed = exe_dir.join("../resources/dumps");
116    if let Some(found) = crate::dump_paths::resolve_versioned_in_dir(&installed, "plain", prefer) {
117      return Some(found);
118    }
119    // Sibling-of-exe (test binaries at target/<profile>/deps/).
120    if let Some(found) = crate::dump_paths::resolve_versioned_in_dir(exe_dir, "plain", prefer) {
121      return Some(found);
122    }
123  }
124  // 4. Dev-tree path baked in at compile time.
125  let dev = Path::new(DEV_DUMPS_DIR);
126  if dev.is_dir()
127    && let Some(found) = crate::dump_paths::resolve_versioned_in_dir(dev, "plain", prefer)
128  {
129    return Some(found);
130  }
131  None
132}

134/// True when any plain.YYYY.dump.txt is reachable through any of the
135/// runtime-resolution paths or the embedded fallback. Used by `tex.rs`
136/// to decide whether `LoadFormat('plain')` takes the dump branch or the
137/// _base reconstruction branch.
138pub fn plain_dump_available() -> bool {
139  if *NODUMP {
140    return false;
141  }
142  let prefer = crate::dump_paths::detect_ambient_texlive_year();
143  resolve_dump_path(prefer).is_some()
144    || crate::embedded_dumps::embedded_plain_dump(prefer).is_some()
145}
