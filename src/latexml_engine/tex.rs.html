8/// Perl: DefAutoload — define a macro that auto-loads a package on first use.
9/// When the command is first invoked, it loads the specified package (via
10/// RequirePackage), then re-emits the original CS so it gets re-executed
11/// with the proper definition.
12///
13/// Mirrors Perl `Package.pm` `DefAutoload` (L1081-1100) + `ClearAutoLoad`
14/// (L1106-1111): clear the trigger's meaning GLOBALLY before loading, then
15/// run the package, then re-emit the original CS. Without the global clear,
16/// if the autoload fires from inside a `\begin{X}` group whose env-CS is
17/// itself the trigger, the package's bindings install at LOCAL scope on
18/// the env's frame and get popped on `\end{X}` — reverting the trigger CS
19/// to the autoload closure and re-firing on every subsequent invocation
20/// (1.6M+ infinite locked-redefinition loop, observed in 1711.11576 etc).
21/// Clearing globally short-circuits the loop: after first fire, the
22/// trigger is `Stored::None` globally, so subsequent invocations with no
23/// in-scope local binding hit a clean undefined-CS error rather than
24/// re-entering the autoload.
25///
26/// Sandbox math0004154 + amsppt class-route hang at amsfonts.sty
27/// (see project_amsppt_cls_dispatcher.md memory) — the original re-entry
28/// guard was a `Tokens(vec![])` no-op installed at LOCAL scope before the
29/// package load. The Perl-faithful global clear (Stored::None at Global)
30/// covers the same re-entry-during-package-load case AND the
31/// autoload-from-inside-a-group case.
32fn def_autoload(cs_name: &str, package: &str) -> Result<()> {
33  use latexml_core::common::store::Stored;
34  use latexml_core::definition::ExpansionBody;
35  let cs_tok = T_CS!(cs_name);
36  // Don't overwrite if already defined
37  if IsDefined!(&cs_tok) {
38    return Ok(());
39  }
40  let pkg_name = package.to_string();
41  let cs_for_closure = cs_tok;
42  def_macro(
43    cs_tok,
44    None,
45    ExpansionBody::Closure(Rc::new(move |_args| {
46      // If the target package is *already* loaded, this closure is firing
47      // because a separate CS was \let to the autoload trigger BEFORE the
48      // package later loaded under a different name (e.g. `\let\varmathbb
49      // =\mathbb` while \mathbb is still the autoload trigger; then
50      // amssymb → amsfonts loads and \mathbb gets a real def, but
51      // \varmathbb still holds the closure). In that case, clearing
52      // cs_for_closure (\mathbb) would erase its real definition, and
53      // re-emitting \mathbb would land on an undefined CS — producing the
54      // spurious `Error:undefined:\mathbb` reported during `\varmathbb{D}`
55      // expansion. Witness 2310.13684. Skip the clear+load and just
56      // re-emit cs_for_closure, which by now has the real meaning the
57      // already-loaded package installed.
58      let pkg_loaded_key = s!("{}.sty_loaded", pkg_name);
59      let pkg_raw_key = s!("{}.sty_raw_loaded", pkg_name);
60      if latexml_core::state::lookup_bool(&pkg_loaded_key)
61        || latexml_core::state::lookup_bool(&pkg_raw_key) {
62        return Ok(Tokens::new(vec![cs_for_closure]));
63      }
64      // Perl `ClearAutoLoad` — assign_internal('meaning', $trigger => undef,
65      // 'global'). Removes the autoload trigger globally.
66      state::assign_meaning(&cs_for_closure, Stored::None, Some(Scope::Global));
67      // Snapshot the calling frame's MEANING keys before the package load.
68      // Anything new the package installs at the calling frame is then
69      // hoisted to GLOBAL so it survives `\end{X}`'s pop_frame.
70      //
71      // Without this, autoload-triggered package loads from inside a body
72      // group (e.g. amsmath autoload via `\subequations` inside an
73      // environment group at depth>=1) install ALL of the package's CSes
74      // at the calling frame; those CSes vanish when the group exits, so
75      // a sibling autoload trigger (e.g. `\align` later in the body)
76      // fires, finds no real binding, and clears+re-emits → undefined CS
77      // → 10000-error MaxLimit cascade. Observed in 1711.11576 + cluster.
78      //
79      // The single-trigger `let_i(\trigger, \trigger, Global)` from the
80      // earlier round-17 fix is subsumed by the delta-hoist (since the
81      // trigger CS is itself a meaning-key newly-installed by the package
82      // load). (1711.11576: cleanly converts post-fix.)
83      let pre_keys = latexml_core::state::snapshot_top_frame_meaning_keys();
84      require_package(&pkg_name, RequireOptions::default())?;
85      latexml_core::state::hoist_top_frame_meaning_delta(&pre_keys);
86      Ok(Tokens::new(vec![cs_for_closure]))
87    })),
88    None,
89  )?;
90  // Mark this CS as an autoload trigger so `isDefinableLaTeX` treats
91  // `\newcommand{\cs}{…}` as a redefinition of an undefined CS (matching
92  // Perl, where the equivalent `DefAutoload` entries live in
93  // `OmniBus.cls.ltxml` and only fire when OmniBus is actually loaded —
94  // i.e. they don't block user `\newcommand` in normal LaTeX papers).
95  // Witness: nucl-th9902037 redefines `\Bbb` via `\newcommand` to a
96  // paper-local symbol; without this flag the autoload trigger silently
97  // wins, then expands `\Bbb $arg…$` as `\mathbb{$}` and cascades into
98  // 62 mode-switch errors. Perl on the same input: 0 errors.
99  state::assign_value(
100    &s!("{cs_name}:autoload"),
101    Stored::Bool(true),
102    Some(Scope::Global),
103  );
104  Ok(())
105}

107/// Variant of `def_autoload` that loads a `.pool` (engine-level
108/// definitions file) instead of a `.sty` package. Mirrors Perl's
109/// `DefAutoload($trigger, '<Pool>.pool.ltxml')` form used by TeX.pool
110/// to lazy-load AmSTeX.pool on amstex-style triggers.
111fn def_autoload_pool(cs_name: &str, pool: &str) -> Result<()> {
112  use latexml_core::common::store::Stored;
113  use latexml_core::definition::ExpansionBody;
114  let cs_tok = T_CS!(cs_name);
115  if IsDefined!(&cs_tok) {
116    return Ok(());
117  }
118  let pool_name = pool.to_string();
119  let cs_for_closure = cs_tok;
120  def_macro(
121    cs_tok,
122    None,
123    ExpansionBody::Closure(Rc::new(move |_args| {
124      state::assign_meaning(&cs_for_closure, Stored::None, Some(Scope::Global));
125      let pre_keys = latexml_core::state::snapshot_top_frame_meaning_keys();
126      input_definitions(&pool_name, InputDefinitionOptions {
127        extension: Some(Cow::Borrowed("pool")),
128        ..InputDefinitionOptions::default()
129      })?;
130      latexml_core::state::hoist_top_frame_meaning_delta(&pre_keys);
131      Ok(Tokens::new(vec![cs_for_closure]))
132    })),
133    None,
134  )?;
135  state::assign_value(
136    &s!("{cs_name}:autoload"),
137    Stored::Bool(true),
138    Some(Scope::Global),
139  );
140  Ok(())
141}
