11/// Perl: DefAutoload — define a macro that auto-loads a package on first use.
12/// When the command is first invoked, it loads the specified package (via
13/// RequirePackage), then re-emits the original CS so it gets re-executed
14/// with the proper definition.
15///
16/// Mirrors Perl `Package.pm` `DefAutoload` (L1081-1100) + `ClearAutoLoad`
17/// (L1106-1111): clear the trigger's meaning GLOBALLY before loading, then
18/// run the package, then re-emit the original CS. Without the global clear,
19/// if the autoload fires from inside a `\begin{X}` group whose env-CS is
20/// itself the trigger, the package's bindings install at LOCAL scope on
21/// the env's frame and get popped on `\end{X}` — reverting the trigger CS
22/// to the autoload closure and re-firing on every subsequent invocation
23/// (1.6M+ infinite locked-redefinition loop, observed in 1711.11576 etc).
24/// Clearing globally short-circuits the loop: after first fire, the
25/// trigger is `Stored::None` globally, so subsequent invocations with no
26/// in-scope local binding hit a clean undefined-CS error rather than
27/// re-entering the autoload.
28///
29/// Sandbox math0004154 + amsppt class-route hang at amsfonts.sty
30/// (see project_amsppt_cls_dispatcher.md memory) — the original re-entry
31/// guard was a `Tokens(vec![])` no-op installed at LOCAL scope before the
32/// package load. The Perl-faithful global clear (Stored::None at Global)
33/// covers the same re-entry-during-package-load case AND the
34/// autoload-from-inside-a-group case.
35fn def_autoload(cs_name: &str, package: &str) -> Result<()> {
36  use latexml_core::common::store::Stored;
37  use latexml_core::definition::ExpansionBody;
38  let cs_tok = T_CS!(cs_name);
39  // Don't overwrite if already defined
40  if IsDefined!(&cs_tok) {
41    return Ok(());
42  }
43  let pkg_name = package.to_string();
44  let cs_for_closure = cs_tok;
45  def_macro(
46    cs_tok,
47    None,
48    ExpansionBody::Closure(Rc::new(move |_args| {
49      // Perl `ClearAutoLoad` — assign_internal('meaning', $trigger => undef,
50      // 'global'). Removes the autoload trigger globally.
51      state::assign_meaning(&cs_for_closure, Stored::None, Some(Scope::Global));
52      // Snapshot the calling frame's MEANING keys before the package load.
53      // Anything new the package installs at the calling frame is then
54      // hoisted to GLOBAL so it survives `\end{X}`'s pop_frame.
55      //
56      // Without this, autoload-triggered package loads from inside a body
57      // group (e.g. amsmath autoload via `\subequations` inside an
58      // environment group at depth>=1) install ALL of the package's CSes
59      // at the calling frame; those CSes vanish when the group exits, so
60      // a sibling autoload trigger (e.g. `\align` later in the body)
61      // fires, finds no real binding, and clears+re-emits → undefined CS
62      // → 10000-error MaxLimit cascade. Observed in 1711.11576 + cluster.
63      //
64      // The single-trigger `let_i(\trigger, \trigger, Global)` from the
65      // earlier round-17 fix is subsumed by the delta-hoist (since the
66      // trigger CS is itself a meaning-key newly-installed by the package
67      // load). (1711.11576: cleanly converts post-fix.)
68      let pre_keys = latexml_core::state::snapshot_top_frame_meaning_keys();
69      require_package(&pkg_name, RequireOptions::default())?;
70      latexml_core::state::hoist_top_frame_meaning_delta(&pre_keys);
71      Ok(Tokens::new(vec![cs_for_closure]))
72    })),
73    None,
74  )?;
75  Ok(())
76}

78/// Perl `FindFile($format._dump, ...)` parity for the plain dump.
79/// Returns `true` if `plain.dump.txt` is reachable through any of the
80/// runtime resolution paths (env overrides, exe-relative install layout,
81/// dev-tree). Mirrors `plain_dump::resolve_dump_path` exactly so the
82/// branch decision in `LoadFormat('plain')` doesn't drift.
83fn plain_dump_available() -> bool {
84  if let Some(p) = PLAIN_DUMP_PATH.as_deref() {
85    if std::path::Path::new(p).is_file() {
86      return true;
87    }
88  }
89  if let Some(dir) = DUMP_DIR.as_deref() {
90    if std::path::Path::new(dir).join("plain.dump.txt").is_file() {
91      return true;
92    }
93  }
94  if let Ok(exe) = std::env::current_exe() {
95    if let Some(exe_dir) = exe.parent() {
96      if exe_dir.join("../resources/dumps/plain.dump.txt").is_file() {
97        return true;
98      }
99      if exe_dir.join("plain.dump.txt").is_file() {
100        return true;
101      }
102    }
103  }
104  let dev = concat!(
105    env!("CARGO_MANIFEST_DIR"),
106    "/../resources/dumps/plain.dump.txt"
107  );
108  std::path::Path::new(dev).is_file()
109}
