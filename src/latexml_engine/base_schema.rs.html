1//! Base Schema                                                         | #
2//!
3//! Core TeX Implementation for LaTeXML                                 | #
4use crate::prelude::*;
5
6LoadDefinitions!({
7  // NOTE that these define the namespaces we'll (probably) use
8  // along with the prefixes to be used in "code"
9  // The generated XML will use the prefixes defined by RegisterDocumentNamespace(...) (if ever)
10  // or those prefixes defined by the Schema (typically RelaxNGSchema(..)
11  RegisterNamespace!("ltx", "http://dlmf.nist.gov/LaTeXML");
12  RegisterNamespace!("svg", "http://www.w3.org/2000/svg");
13  // Needed for SVG
14  RegisterNamespace!("xlink", "http://www.w3.org/1999/xlink");
15  // Not directly used, but let's stake out the ground
16  RegisterNamespace!("m", "http://www.w3.org/1998/Math/MathML");
17  RegisterNamespace!("xhtml", "http://www.w3.org/1999/xhtml");
18  // Namespace for arbitrary data attributes (mapped to data-xxx in html5)
19  RegisterNamespace!("data" => "http://dlmf.nist.gov/LaTeXML/data");
20  // Needed for ARIA accessibility attributes
21  RegisterNamespace!("aria" => "http://www.w3.org/ns/wai-aria");
22  // Document-level prefixes for the same two: the schema declares both
23  // (LaTeXML-common.rnc: ARIA.attributes + the data/foreign wildcard, and
24  // the precompiled model carries the aria:* attribute tables), and Perl
25  // derives the output prefixes from the schema. Rust registers them
26  // explicitly so any binding can emit aria:*/data:* attributes without
27  // a per-binding RegisterDocumentNamespace (arXiv-fork a11y capability;
28  // acmart's \Description was the first consumer).
29  RegisterDocumentNamespace!("data", "http://dlmf.nist.gov/LaTeXML/data");
30  RegisterDocumentNamespace!("aria", "http://www.w3.org/ns/wai-aria");
31
32  // This is used for plain TeX, but needs to be undone for LaTeX (or...)!
33  RelaxNGSchema!("LaTeXML");
34
35  Tag!("ltx:section", auto_close => true);
36  Tag!("ltx:document", auto_close => true, auto_open => true);
37  Tag!("ltx:document", after_open => sub[document,root] {
38    let mut bg_to_set = None;
39    if let Some(bg) = document.get_node_font(root).get_background()
40      && *bg != common::color::WHITE {
41        bg_to_set = Some(bg.to_attribute());
42      }
43    if let Some(bg) = bg_to_set {
44      document.set_attribute(root, "backgroundcolor", &bg)?;
45    }
46    // Apply font language as xml:lang on document root.
47    // Also update the document element's font language to match, so the
48    // font delta serializer doesn't override xml:lang with a stale value
49    // (e.g., from class options processed before babel determines the main language).
50    if let Some(lang) = lookup_value("DOCUMENT_LANGUAGE") {
51      let lang_str = lang.to_string();
52      if !lang_str.is_empty() {
53        document.set_attribute(root, "xml:lang", &lang_str)?;
54        let mut font = document.get_node_font(root).clone();
55        font.language = Some(Cow::Owned(lang_str));
56        document.set_node_font(root, &font)?;
57      }
58    }
59  });
60
61  //======================================================================
62  // Core ID functionality.
63  //======================================================================
64  DefMacro!("\\lx@empty", None);
65
66  // DOCUMENTID is the ID of the document
67  // AND prefixes IDs on all other elements.
68  let doc_id = lookup_string("DOCUMENTID");
69  if !doc_id.is_empty() {
70    // Wrap in T_OTHER so funny chars don't screw up (no space!)
71    let doc_id_token = T_OTHER!(doc_id);
72    DefMacro!(T_CS!("\\thedocument@ID"), None, doc_id_token);
73  } else {
74    Let!("\\thedocument@ID", "\\lx@empty");
75  }
76
77  NewCounter!("@lx@xmarg", "document", idprefix => "XM");
78
79  //======================================================================
80  Tag!("ltx:document",
81  after_open => sub[document, _node] {
82    document.process_pending_resources()?;
83  });
84  RequireResource!("LaTeXML.css");
85  //======================================================================
86  // The default "initial context" for XML+RDFa specifies some default
87  // terms and prefixes, but no default vocabulary.
88  // Ought to have a default for @vocab, but settable?
89  // can we detect use of simple "term"s in attributes so we know whether we need @vocab?
90  // Ought to have a default set of prefixes from RDFa Core,
91  // but allow prefixes to be added.
92  // Probably ought to scan rdf attributes for all uses of prefixes,
93  // and include them in @prefix
94  // The following prefixes are listed in http://www.w3.org/2011/rdfa-context/rdfa-1.1
95  let rdf_prefixes = map!(
96    "cc"      => "http://creativecommons.org/ns#",
97    "ctag"    => "http://commontag.org/ns#",
98    "dc"      => "http://purl.org/dc/terms/",
99    "dcterms" => "http://purl.org/dc/terms/",
100    "ical"    => "http://www.w3.org/2002/12/cal/icaltzd#",
101    "foaf"    => "http://xmlns.com/foaf/0.1/",
102    "gr"      => "http://purl.org/goodrelations/v1#",
103    "grddl"   => "http://www.w3.org/2003/g/data-view#",
104    "ma"      => "http://www.w3.org/ns/ma-ont#",
105    "og"      => "http://ogp.me/ns#",
106    "owl"     => "http://www.w3.org/2002/07/owl#",
107    "rdf"     => "http://www.w3.org/1999/02/22-rdf-syntax-ns#",
108    "rdfa"    => "http://www.w3.org/ns/rdfa#",
109    "rdfs"    => "http://www.w3.org/2000/01/rdf-schema#",
110    "rev"     => "http://purl.org/stuff/rev#",
111    "rif"     => "http://www.w3.org/2007/rif#",
112    "rr"      => "http://www.w3.org/ns/r2rml#",
113    "schema"  => "http://schema.org/",
114    "sioc"    => "http://rdfs.org/sioc/ns#",
115    "skos"    => "http://www.w3.org/2004/02/skos/core#",
116    "skosxl"  => "http://www.w3.org/2008/05/skos-xl#",
117    "v"       => "http://rdf.data-vocabulary.org/#",
118    "vcard"   => "http://www.w3.org/2006/vcard/ns#",
119    "void"    => "http://rdfs.org/ns/void#",
120    "xhv"     => "http://www.w3.org/1999/xhtml/vocab#",
121    "xml"     => "http://www.w3.org/XML/1998/namespace",
122    "xsd"     => "http://www.w3.org/2001/XMLSchema#",
123    "wdr"     => "http://www.w3.org/2007/05/powder#",
124    "wdrs"    => "http://www.w3.org/2007/05/powder-s#"
125  );
126
127  for (k, v) in rdf_prefixes.iter() {
128    AssignMapping!("RDFa_prefixes", k => *v);
129  }
130});
