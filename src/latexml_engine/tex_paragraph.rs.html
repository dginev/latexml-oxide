7/// Helper used by `\leftline`/`\rightline`/`\centerline` and friends.
8/// Perl `TeX_Paragraph.pool.ltxml:75` `sub alignLine`.
9pub fn align_line(
10  document: &mut Document,
11  line: &[Option<Digested>],
12  alignment: &str,
13) -> Result<()> {
14  if document.is_openable("ltx:p") {
15    let line_content = line.iter().filter_map(|c| c.as_ref()).collect();
16    document.insert_element(
17      "ltx:p",
18      line_content,
19      Some(string_map!("class" => s!("ltx_align_{alignment}"))),
20    )?;
21  } else if document.is_openable("ltx:text") {
22    let line_content = line.iter().filter_map(|c| c.as_ref()).collect();
23    document.insert_element(
24      "ltx:text",
25      line_content,
26      Some(string_map!("class" => s!("ltx_align_{alignment}"))),
27    )?;
28    document.insert_element("ltx:break", Vec::new(), None)?;
29  } else if let Some(Some(line_content)) = line.first() {
30    document.absorb(line_content, None)?;
31  }
32  Ok(())
33}

36LoadDefinitions!({
37  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
38  // Paragraph Family of primitive control sequences
39  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
40
41  //======================================================================
42  // Spacing tweaks
43  //----------------------------------------------------------------------
44  // \ignorespaces           c  makes TeX read and expand tokens but do nothing until a nonspace
45  // token is reached. \noboundary             c  if present, breaks ligatures and kerns.
46  // \vadjust                c  inserts a vertical list between two lines in a paragraph.
47
48  def_primitive_noop("\\ignorespaces SkipSpaces")?;
49  def_primitive_noop("\\noboundary")?;
50  // \vadjust<filler>{<vertical mode material>}
51  // Note: \vadjust ignores in vertical mode...
52  DefPrimitive!("\\vadjust {}", sub[(arg)] { push_tokens("vAdjust", arg); });
53
54  //======================================================================
55  // Basic Paragraph
56  //----------------------------------------------------------------------
57  // \everypar               pt holds tokens added at the beginning of every paragraph.
58  // \indent                 c  begins a new paragraph indented by \parindent.
59  // \noindent               c  begins a new paragraph that is not indented.
60  // \par                    c  is an explicit command to end a paragraph.
61  DefRegister!("\\everypar", Tokens!());
62  // These determine whether the _next_ paragraph gets indented!
63  // thus it needs \par to check whether such indentation has been set.
64  DefConstructor!("\\indent", sub[document] {
65    if let Some(mut node) = document.get_element() {
66      let tag = document::get_node_qname(&node);
67      let para_tag = arena::pin_static("ltx:para");
68      if tag == para_tag {
69        node.set_attribute("class","ltx_indent")?;
70      } else if document::sym_can_contain_somehow(tag, para_tag).is_some() {
71        // Perversely ignore indent on 1st para after sectioning titles (Perl parity)
72        let prev_is_title = node.get_last_child().map(|prev| {
73          let prev_qname = document::get_node_qname(&prev);
74          arena::with(prev_qname, |s| s == "ltx:title" || s == "ltx:toctitle")
75        }).unwrap_or(false);
76        if prev_is_title {
77          document.open_element("ltx:para", None, None)?;
78        } else {
79          document.open_element("ltx:para", Some(string_map!("class"=>"ltx_indent")), None)?;
80        }
81      }
82      // Otherwise ignore.
83    }
84  },
85  properties => { stored_map!("isSpace" => true) },
86  // Perl: enterHorizontal => 1
87  before_digest => { enter_horizontal(); });
88  DefConstructor!("\\noindent", sub[document] {
89    if let Some(mut node) = document.get_element() {
90      let tag = document::get_node_qname(&node);
91      let para_tag = arena::pin_static("ltx:para");
92      if tag == para_tag {
93        node.set_attribute("class","ltx_noindent")?;
94      } else if document::sym_can_contain_somehow(tag, para_tag ).is_some() {
95        // Used in a position where a paragraph can be started, start
96        document.open_element("ltx:para", Some(string_map!("class"=>"ltx_noindent")), None)?;
97      }
98      // Otherwise ignore.
99    }
100  },
101  properties => { stored_map!("isSpace" => true) },
102  // Perl: enterHorizontal => 1
103  before_digest => { enter_horizontal(); });
104
105  // <ltx:para> represents a Logical Paragraph, whereas <ltx:p> is a `physical paragraph'.
106  // A para can contain both p and displayed equations and such.
107
108  // Remember; \par _closes_, not opens, paragraphs!
109  // Here, we want to close both an open p and para (if either are open).
110  // NOTE Also that the whole inPreamble bit is, I think, overused.
111  // For example, \par should be a NOOP in vertical mode, and that would generally make it
112  // ignored in the preamble.
113  let mut skippable_props: SymHashMap<Stored> = SymHashMap::default();
114  skippable_props.insert("alignmentSkippable", Stored::Bool(true));
115
116  DefConstructor!("\\lx@normal@par",
117    sub[document, _args, props] {
118      if !prop_bool!(props, "inPreamble") {
119        document.maybe_close_element("ltx:p")?;
120        let element = document.get_element();
121        if let Some(mut node) = element {
122          let qname = document::get_node_qname(&node);
123          // Only set on the para about to close, if unknown!
124          if qname == arena::pin_static("ltx:para") && node.get_attribute("class").is_none() {
125            let class_sym = prop_str!(props,"class");
126            if class_sym != pin!("") {
127              let class_s = arena::with(class_sym, |s| s.to_string());
128              document.set_attribute(&mut node, "class", &class_s)?;
129            }
130          } else if qname == arena::pin_static("ltx:figure") {
131            // insert breaks in figures, for vertically separating subfigures
132            document.insert_element("ltx:break",Vec::new(), None)?;
133          }
134        }
135        if !prop_bool!(props, "internal_par") {
136          document.maybe_close_element("ltx:para")?;
137        }
138      }
139    },
140    before_digest => {
141      // Perl: combine any digested horizontal material into a horizontal List
142      let mode = state::lookup_string_from_sym(pin!("MODE"));
143      let bound = state::lookup_string_from_sym(pin!("BOUND_MODE"));
144      if mode == "horizontal" && bound.ends_with("vertical") {
145        // Perl: $stomach->repackHorizontal;
146        repack_horizontal();
147        assign_value_inplace_sym(pin!("MODE"), bound); // Resume vertical/internal_vertical
148      }
149      state::assign_value("parshape", Stored::None, None);
150      state::assign_value("interlinepenalties", Stored::None, None);
151    },
152    after_digest => sub[whatsit] {
153      whatsit.set_property("mode", state::lookup_string_from_sym(pin!("MODE")));
154      // When invoked by leave_horizontal: no reversion, don't close ltx:para
155      if LookupBool!("INTERNAL_PAR") {
156        whatsit.set_property("internal_par", true);
157        whatsit.set_property("reversion", Tokens!());
158      }
159      let in_preamble = LookupBool!("inPreamble");
160      if in_preamble {
161        whatsit.set_property("inPreamble", true);
162        Ok(Vec::new())
163      } else {
164        if let Some(c) = lookup_value("next_para_class") {
165          // Check if flags were set by prior \par:
166          whatsit.set_property("class", c);
167          { state::assign_value("next_para_class", Stored::None, None); }
168        }
169        // Per eTeX spec, \interlinepenalties (like \parshape) is reset after each paragraph.
170        { state::assign_value("interlinepenalties", Stored::None, None); }
171        // Fish out flags for next ltx:para, to be used when the next \par closes:
172        if state::lookup_register("\\parindent",Vec::new())?.unwrap().value_of() == 0 {
173          // respect \parindent if no overrides are given
174          { state::assign_value("next_para_class", "ltx_noindent", None); }
175        }
176        // Vertical adjustments
177        if let Some(Stored::Tokens(vadj)) = { state::remove_value("vAdjust") } {
178          state::assign_value("vAdjust", Tokens!(), Some(Scope::Global));
179          Ok(vec![ Digest!(vadj)? ])
180        } else {
181          Ok(Vec::new())
182        }
183      }
184    },
185    properties => skippable_props,
186    alias => "\\par"
187  );
188  Let!("\\par", "\\lx@normal@par");
189  Tag!("ltx:para", auto_close => true, auto_open => true,
190    after_close => sub[document, node] {
191      prune_empty_para(document, node)?;
192  });
193  Tag!("ltx:p", auto_close => true, auto_open => true,
194    after_close => sub[document, node] {
195      document.trim_node_whitespace(node)?;
196  });
197
198  //======================================================================
199  // Paragraph Shape
200  //----------------------------------------------------------------------
201  // \parshape               iq specifies an arbitrary paragraph shape.
202  // Acts like a Number register (returns count of lines when read).
203  // Setter reads n pairs of dimensions and stores them in state.
204  DefRegister!("\\parshape", Number::new(0),
205    getter => sub[_args] {
206      with_value("parshape", |val_opt| {
207        if let Some(Stored::VecDequeStored(shape)) = val_opt {
208          Some(RegisterValue::Number(Number::new((shape.len() / 2) as i64)))
209        } else {
210          Some(RegisterValue::Number(Number::new(0)))
211        }
212      })
213    },
214    setter => sub[value, _scope, _args] {
215      let n_val = value.value_of();
216      let n = if n_val < 0 { 0 } else { n_val } as usize;
217      let mut shape = VecDeque::new();
218      for _ in 0..n {
219        let indent = gullet::read_dimension().unwrap_or_default();
220        let length = gullet::read_dimension().unwrap_or_default();
221        shape.push_back(Stored::Dimension(indent));
222        shape.push_back(Stored::Dimension(length));
223      }
224      assign_value(
225        "parshape",
226        if n > 0 { Stored::VecDequeStored(shape) } else { Stored::None },
227        Some(Scope::Global),
228      );
229    }
230  );
231  //======================================================================
232  // Paragraph Shape
233  //----------------------------------------------------------------------
234  // \prevgraf               iq is the number of lines in the paragraph most recently completed or
235  // partially completed. \spacefactor            iq controls interword spacing.
236  // \emergencystretch       pd is glue used in the third pass made for bad paragraphs.
237  // \hangindent             pd is the amount of hanging indentation.
238  // \hsize                  pd is the width of normal lines in a paragraph.
239  // \lineskiplimit          pd is the cutoff used to select between \baselineskip and \lineskip.
240  // \parindent              pd is the width of indentation at the beginning of a paragraph.
241  // \baselineskip           pg is glue added between lines to keep their baselines consistently
242  // spaced. \leftskip               pg is glue added at the left of every line in a paragraph.
243  // \rightskip              pg is glue added at the right of every line in a paragraph.
244  // \lineskip               pg is alternate interline glue used if the \baselineskip glue is not
245  // feasible   . \parskip                pg is extra glue put between paragraphs.
246  // \parfillskip            pg is glue which finishs the last line of a paragraph.
247  // \spaceskip              pg is alternate interword glue.
248  // \xspaceskip             pg is alternate intersentence glue.
249  // \adjdemerits            pi holds the demerits for visually incompatible adjacent lines.
250  // \doublehyphendemerits   pi holds the demerits added if two consecutive lines end with
251  // discretionary breaks. \finalhyphendemerits    pi holds the demerits added if the penultimate
252  // line in a paragraph ends with a discretionary break. \hangafter              pi is the number
253  // of lines before hanging indentation changes. \looseness              pi tells TeX to try and
254  // increase or decrease the number of lines in a paragraph.
255
256  // \tolerance              pi is the acceptable \badness of lines after hyphenation.
257  // \pretolerance           pi is the acceptable \badness of lines in a paragraph before
258  // hyphenation is attempted.
259  DefRegister!("\\spacefactor", Number!(0));
260  DefRegister!("\\prevgraf", Number!(0));
261  DefRegister!("\\emergencystretch", Dimension!("0"));
262  DefRegister!("\\hangindent", Dimension!("0"));
263  DefRegister!("\\hsize", Dimension!("6.5in"));
264  DefRegister!("\\lineskiplimit", Dimension!("0"));
265  DefRegister!("\\parindent", Dimension!("20pt"));
266  DefRegister!("\\baselineskip", Glue!("12pt"));
267  DefRegister!("\\leftskip", Glue!("0"));
268  DefRegister!("\\rightskip", Glue!("0"));
269  DefRegister!("\\lineskip", Glue!("1pt"));
270  DefRegister!("\\parskip", Glue!("0pt plus 1pt"));
271  DefRegister!("\\parfillskip", Glue!("0pt plus 1fil"));
272  DefRegister!("\\spaceskip", Glue!("0"));
273  DefRegister!("\\xspaceskip", Glue!("0"));
274  DefRegister!("\\adjdemerits", Number!(10000));
275  DefRegister!("\\doublehyphendemerits", Number!(10000));
276  DefRegister!("\\finalhyphendemerits", Number!(5000));
277  DefRegister!("\\hangafter", Number!(0));
278  DefRegister!("\\looseness", Number!(0));
279  DefRegister!("\\tolerance", Number!(200));
280  DefRegister!("\\pretolerance", Number!(100));
