4static EXCEPTION_MACRO_NAMES_FOR_MEANING: Lazy<Regex> =
5  Lazy::new(|| Regex::new("^\\\\(?:(?:un)?expanded|detokenize)$").unwrap());
6static LEAD_W_COLON_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\w+):").unwrap());
7static UNTIL_SPEC: Lazy<Regex> = Lazy::new(|| Regex::new("^\\w?Until(\\w*):").unwrap());

11static TRACE_COMMANDS: u8 = 0x2;
12static _TRACE_ALL: u8 = 0x3; // MACROS | COMMANDS

66  //======================================================================
67  // meaning
68  //----------------------------------------------------------------------
69  // \meaning          c  adds characters describing a token to the output stream.
70  // Not sure about this yet...
71  // NOTE: Lots of back-and-forth mangle with definition vs cs; don't do that!
72  DefMacro!("\\meaning Token", sub[(token)] {
73    let mut meaning = String::from("undefined");
74    if let Some(definition) = if token == T_ALIGN!() {
75      Some(Stored::Token(token))
76    } else {
77      lookup_meaning(&token)
78    } {
79      // First, if this definition is a primitive|conditional|constructor,
80      // check to see if it has an alias, which would allow us to work with a token
81      // Check for font-defined primitives: \meaning\fiverm => "select font cmr5"
82      // Perl: only shows "at Xpt" when explicit "at" or "scaled" was used in \font
83      if let Stored::Primitive(_) = &definition {
84        let cs_str = token.to_string();
85        let key = s!("fontinfo_{}", cs_str);
86        // with_value avoids the Stored envelope clone on the Font arm;
87        // we only need the font's name string out.
88        let name_opt = state::with_value(&key, |v| match v {
89          Some(Stored::Font(f)) => f.name.as_ref().map(|n| n.to_string()),
90          _ => None,
91        });
92        if let Some(name) = name_opt {
93          let at_key = s!("fontinfo_at_{}", cs_str);
94          let at_info = state::with_value(&at_key, |v| match v {
95            Some(Stored::String(s)) => arena::with(*s, |at| format!(" at {at}")),
96            _ => String::new(),
97          });
98          meaning = format!("select font {}{}", name, at_info);
99          return Ok(Tokens::new(Explode!(meaning)));
100        }
101      }
102      let definition : Stored = match definition {
103        Stored::Primitive(primitive) =>
104          Stored::Token(primitive.get_cs_or_alias().into_owned()),
105        Stored::Constructor(constructor) =>
106          Stored::Token(constructor.get_cs_or_alias().into_owned()),
107        Stored::Conditional(cond) =>
108          Stored::Token(cond.get_cs_or_alias().into_owned()),
109        other => other
110      };
111
112      // Now that we've tried to obtain an expandable definition, do the TeX dance:
113      match definition {
114        Stored::Token(t) => {
115          let cc = t.get_catcode();
116          let text = if cc == Catcode::SPACE {
117            String::from(" ")
118          } else {
119            t.to_string()
120          };
121          meaning = String::from(cc.meaning());
122          if !meaning.is_empty() {
123            meaning.push(' ');
124          }
125          meaning.push_str(&text);
126        },
127        Stored::Register(register) => {
128          meaning = register.get_address().to_string();
129        },
130        Stored::Expandable(expandable) => {
131          // short-circuit some troublesome discrepancies with TeX, which end up macros on our end,
132          // but \meaning expects as primitives in the CTAN ecosystem.
133          let cs = expandable.get_cs_or_alias().to_string();
134          // These exceptions could be extended further, as we add more .sty/.cls support
135          if EXCEPTION_MACRO_NAMES_FOR_MEANING.is_match(&cs) {
136            return Ok(Tokens::new(Explode!(cs)));
137          }
138          let params = match expandable.get_parameters() {
139            Some(ps) => ps.get_parameters(),
140            None => Vec::new()
141          };
142          let mut spec_parts : Vec<SymStr> = Vec::new();
143          let mut p_trailer = "";
144          // params.iter().map(|param| LEAD_W_COLON_RE.replace(&param.spec,"") ).collect();
145          let mut arg_index = 0;
146          for param in params.iter() {
147            let mut p_spec = pin!("");
148            let mut continue_flag = false;
149            // TODO: avoiding the allocation is quite painful here, since arena gets into mutability
150            // locking
151            let spec = arena::to_string(param.spec);
152            match spec.as_str() {
153              "RequireBrace" => {
154                // tex's \meaning prints out the required braces for "\def\a#{}" variants
155                p_trailer = "{";
156                p_spec    = arena::pin_static("{");
157              },
158              "UntilBrace" => {
159                p_trailer = "{";
160                arg_index+=1;
161                p_spec = arena::pin(
162                  arena::with(p_spec, |p_str| format!("#{arg_index}{p_str}")));
163              }
164              other if other.starts_with("Match:") => {
165                // just match, don't increment arg index
166                p_spec = arena::pin(LEAD_W_COLON_RE.replace(other,""));
167              },
168              other if UNTIL_SPEC.is_match(other) => {
169                // implied argument at this slot
170                p_spec = arena::pin(LEAD_W_COLON_RE.replace(other,""));
171                arg_index +=1 ;
172                p_spec = arena::pin(
173                  arena::with(p_spec, |p_str| s!("#{arg_index}{p_str}")));
174              },
175              _other => { // regular parameter, increment
176              // skip the latexml-only requirement params, but only here,
177              // since Match also have "novalue" set.
178                if param.novalue {
179                  continue_flag = true;
180                } else {
181                  arg_index+=1;
182                  p_spec = arena::pin(s!("#{arg_index}"));
183                }
184              }
185            }
186            if !continue_flag {
187              spec_parts.push(p_spec);
188            }
189          }
190          let mut spec : String = arena::join(&spec_parts,"");
191          spec = spec.replace("{}","");
192          spec = spec.replace("Token","");
193
194          let mut prefixes = String::new();
195          if expandable.is_protected {
196            prefixes.push_str("\\protected");
197          }
198          if expandable.is_long {
199            prefixes.push_str("\\long");
200          }
201          if expandable.is_outer {
202            prefixes.push_str("\\outer");
203          }
204          if !prefixes.is_empty() {
205            prefixes.push(' ');
206          }
207          let expansion = match expandable.get_expansion() {
208            None => String::new(),
209            // TODO: How to print closures? This follows Perl's raw pointer format
210            Some(ExpansionBody::Closure(exp)) => format!("CODE({:p})", Rc::as_ptr(exp)),
211            Some(ExpansionBody::Tokens(tks)) => writable_tokens(tks)
212          };
213          meaning = format!("{prefixes}macro:{spec}->{expansion}{p_trailer}");
214        },
215        e => {
216          // Handle other Stored variants gracefully (e.g., Register, Constructor, etc.)
217          meaning = format!("{e}");
218        }
219      }
220    }
221    ExplodeChars!(meaning)
222  });

297  });
