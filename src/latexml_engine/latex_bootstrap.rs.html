1// latex_bootstrap — Bootstrap code for reading latex.ltx for LaTeXML.
2// Corresponds to Perl Engine/latex_bootstrap.pool.ltxml.
3//
4// Loaded BEFORE the LaTeX dump. Contains stubs that override latex.ltx's
5// own mechanisms with LaTeXML's versions, plus CSS-based logos.
6use crate::prelude::*;
7
8#[rustfmt::skip]
9LoadDefinitions!({
10  // Perl: latex_bootstrap.pool.ltxml L18
11  InnerPool!(plain_bootstrap);
12
13  //======================================================================
14  // Perl: latex_bootstrap.pool.ltxml L22-44 — CSS-based LaTeX/LaTeXe logos
15  DefConstructor!("\\LaTeX","<ltx:text class='ltx_LaTeX_logo' cssstyle='letter-spacing:-0.2em; margin-right:0.1em'
16  >L<ltx:text cssstyle='font-variant:small-caps;' yoffset='0.4ex'
17  >a</ltx:text
18  >T<ltx:text cssstyle='font-variant:small-caps;font-size:120%' yoffset='-0.2ex'
19  >e</ltx:text
20  >X</ltx:text>",
21  enter_horizontal => true, locked => true,
22  sizer => { Ok((Dimension!("2.6em"), Dimension!("1.6ex"), Dimension!("0.5ex"))) });
23
24  DefConstructor!("\\LaTeXe","<ltx:text class='ltx_LaTeX_logo' cssstyle='letter-spacing:-0.2em; margin-right:0.1em'
25  >L<ltx:text cssstyle='font-variant:small-caps;' yoffset='0.4ex'
26  >a</ltx:text
27  >T<ltx:text cssstyle='font-variant:small-caps;font-size:120%' yoffset='-0.2ex'

29  >X\u{2002}2<ltx:text cssstyle='font-style:italic' yoffset='-0.3ex'
30  >\u{03B5}</ltx:text></ltx:text>",
31  enter_horizontal => true, locked => true,
32  sizer => { Ok((Dimension!("3.7em"), Dimension!("1.6ex"), Dimension!("0.5ex"))) });
33
34  //======================================================================
35  // Perl: latex_bootstrap.pool.ltxml L49 — register allocation override
36  DefMacro!("\\e@alloc{}{}{}{}{}{}", r"\lx@alloc@{#1}{#3}{#2}{#6}", locked => true);
37  DefMacro!("\\e@ch@ck{}{}{}{}", "", locked => true);
38
39  // Perl: latex_bootstrap.pool.ltxml L51-54 — counter/font stubs
40  DefPrimitive!("\\newcounter{}[]", sub[(cs, default_opt)] {
41    let default = if let Some(tks) = default_opt {
42      if !tks.is_empty() { Expand!(tks) } else { Tokens!() }
43    } else {
44      Tokens!()
45    };
46    let cs_expanded = &Expand!(cs).to_string();
47    NewCounter!(cs_expanded, &default.to_string());
48  }, locked => true);
49  // Perl uses `DefMacro` here, but the dump overwrites `\newcounter` with
50  // the raw latex.ltx Expandable body that expands to `\@definecounter`.
51  // If `\@definecounter` is a macro that re-expands to `\newcounter`, we
52  // get an infinite loop after dump load. Perl's `DefMacro` is fine
53  // because Perl's Token-list expansion of `\newcounter` is a Token, and
54  // Perl's `installDefinition` would skip the dump-overwrite if our
55  // `\newcounter` were locked (it's not — Perl bypasses lock too). The
56  // working semantics in Perl actually rely on `\@definecounter`
57  // resolving to the bootstrap Primitive at substitution time: by the
58  // time `\@definecounter` is invoked (inside the dump-loaded
59  // `\newcounter` body), the active `\newcounter` is the dump
60  // Expandable — same loop. The Perl loop test shows it doesn't
61  // actually loop in user code because... TODO investigate. For now,
62  // use `Let!` to snapshot the Primitive at bootstrap time, breaking
63  // the cycle. Token-stream-equivalent for downstream callers that
64  // simply invoke `\@definecounter{...}`.
65  Let!("\\@definecounter", "\\newcounter");
66  DefMacro!("\\try@load@fontshape", "", locked => true);
67  DefMacro!("\\define@newfont", "", locked => true);
68
69  //======================================================================
70  // Perl: latex_bootstrap.pool.ltxml L58
71  Let!("\\@@input", "\\input"); // Save TeX's version.
72});
