1//! latexml_engine — TeX/LaTeX engine pool bindings (1:1 match with Perl
2//! `LaTeXML/Engine/*.pool.ltxml`). Split out of `latexml_package` to
3//! reduce CI compile-time RAM peaks and improve incremental rebuild
4//! locality. See `docs/SYNC_STATUS.md` "latexml_engine extraction" for
5//! the full motivation.
6//!
7//! Loading hierarchy (mirrors Perl):
8//!   LaTeX.pool → TeX.pool → Base.pool → Base_*, TeX_*, eTeX, pdfTeX, Base_Deprecated
9//!     LoadFormat('plain') → plain_bootstrap → plain_base → plain_dump → plain_constructs →
10//! math_common     LoadFormat('latex') → latex_bootstrap → latex_base → latex_dump →
11//! latex_constructs
12

16// Auxiliary compile-time macros (compile_*: prototype/replacement/expansion
17// helpers used by setup_binding_language). #[macro_export] makes them
18// crate-public; #[macro_use] here makes them visible to engine modules.
19#[macro_use]
20pub mod macros;

22// Main macros (DefMacro!, DefConstructor!, LoadDefinitions!, ...) live in
23// setup_binding_language. They use `#[macro_export]` for cross-crate
24// reach and `#[macro_use]` here to also be visible to engine's own
25// modules without `crate::name!` path prefixes.
26#[macro_use]
