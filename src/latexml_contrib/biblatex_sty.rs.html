64fn bib_as_thebibliography() -> Tokens {
65  let variant: Vec<Token> = match lookup_value("rebuilt_bibtex_variant") {
66    Some(Stored::Tokens(t)) => t.unlist(),
67    _ => return Tokens::default(),
68  };
69  if variant.is_empty() {
70    return Tokens::default();
71  }
72  // Reset variant and entry-count so re-invocation is idempotent (matches
73  // Perl L113-115).
74  assign_value(
75    "rebuilt_bibtex_variant",
76    Stored::Tokens(Tokens::default()),
77    Some(Scope::Global),
78  );
79  let count = bib_state_int("biblatex_entry_count");
80  bib_state_set_int("biblatex_entry_count", 0);
81  let preamble: Vec<Token> = match lookup_value("biblatex_preamble") {
82    Some(Stored::Tokens(t)) => t.unlist(),
83    _ => Vec::new(),
84  };
85  let mut result: Vec<Token> = Vec::with_capacity(variant.len() + 16);
86  result.push(T_CS!("\\thebibliography"));
87  result.push(T_BEGIN!());
88  result.extend(preamble);
89  result.extend(ExplodeText!(&count.to_string()));
90  result.push(T_END!());
91  result.extend(variant);
92  result.push(T_CS!("\\endthebibliography"));
93  Tokens::new(result)
94}

96/// Perl `$fullname =~ s/\\\w+|[}{]//g` (ar5iv biblatex.sty.ltxml L324):
97/// strip leftover control sequences (`\bibinitperiod`, …) and braces from a
98/// name fragment, then trim. `\w` is Perl-ASCII (`[A-Za-z0-9_]`) — a backslash
99/// NOT followed by a word char (e.g. the `\"` of an accent) is preserved, as
100/// in Perl.
101fn bib_clean_name(s: &str) -> String {
102  let mut out = String::with_capacity(s.len());
103  let mut chars = s.chars().peekable();
104  while let Some(ch) = chars.next() {
105    if ch == '\\' {
106      if chars
107        .peek()
108        .is_some_and(|c| c.is_ascii_alphanumeric() || *c == '_')
109      {
110        while chars
111          .peek()
112          .is_some_and(|c| c.is_ascii_alphanumeric() || *c == '_')
113        {
114          chars.next();
115        }
116      } else {
117        out.push(ch);
118      }
119    } else if ch != '{' && ch != '}' {
120      out.push(ch);
121    }
122  }
123  out.trim().to_string()
124}

126/// Parse a biblatex keyval name block — the inner sub-group of a modern
127/// (biber, bbl format ≥ 3.x) `\name` author record, e.g.
128/// `family={Turtayev},familyi={T\bibinitperiod},given={Rustem},giveni=…,givenun=0`.
129/// Splits on depth-0 commas (commas inside `{…}` don't split), then for each
130/// `key=value` strips one layer of surrounding braces off the value. Mirrors
131/// the outcome of Perl's `LaTeXML::Core::KeyVals->readFrom` (L302-306) without
132/// the full KeyVals machinery — we only need `given`/`family` (and their
133/// `i`-initial fallbacks).
134fn parse_name_keyvals(s: &str) -> Vec<(String, String)> {
135  let mut pairs: Vec<(String, String)> = Vec::new();
136  let mut depth = 0i32;
137  let mut cur = String::new();
138  let flush = |seg: &str, pairs: &mut Vec<(String, String)>| {
139    let seg = seg.trim();
140    if seg.is_empty() {
141      return;
142    }
143    if let Some(eq) = seg.find('=') {
144      let key = seg[..eq].trim().to_string();
145      let mut val = seg[eq + 1..].trim();
146      if val.starts_with('{') && val.ends_with('}') && val.len() >= 2 {
147        val = &val[1..val.len() - 1];
148      }
149      pairs.push((key, val.to_string()));
150    }
151  };
152  for ch in s.chars() {
153    match ch {
154      '{' => {
155        depth += 1;
156        cur.push(ch);
157      },
158      '}' => {
159        depth -= 1;
160        cur.push(ch);
161      },
162      ',' if depth == 0 => {
163        flush(&cur, &mut pairs);
164        cur.clear();
165      },
166      _ => cur.push(ch),
167    }
168  }
169  flush(&cur, &mut pairs);
170  pairs
171}

173/// Perl L232-237 / L253-258: strip leading SPACE/`{` and trailing SPACE/`}`
174/// tokens from a captured DOI/eprint field value before splicing it into an
175/// `\href{…}` target, so a `\field{doi}{ {10.x/y} }`-style value yields a
176/// clean URI.
177fn bib_trim_url_tokens(toks: Tokens) -> Vec<Token> {
178  let mut v = toks.unlist();
179  let mut start = 0usize;
180  while start < v.len() {
181    let t = &v[start];
182    if t.code == Catcode::SPACE || t.with_str(|s| s == "{") {
183      start += 1;
184    } else {
185      break;
186    }
187  }
188  let mut end = v.len();
189  while end > start {
190    let t = &v[end - 1];
191    if t.code == Catcode::SPACE || t.with_str(|s| s == "}") {
192      end -= 1;
193    } else {
194      break;
195    }
196  }
197  v.truncate(end);
198  v.drain(..start);
199  v
200}

255/// Perl convention (biblatex/natbib): one optional arg is a postnote, two
256/// are prenote + postnote. Faithful nuance (arxiv-readability#10 /
257/// ar5iv-bindings#4): `\parencite[see][]{key}` has a PRESENT-but-EMPTY
258/// second optional — that must NOT swap ("see" stays the prenote); only an
259/// ABSENT second optional makes the first one a postnote. Empties are
260/// dropped after the swap decision.
261fn blx_swap_pre_post(
262  pre: Option<Tokens>,
263  post: Option<Tokens>,
264) -> (Option<Tokens>, Option<Tokens>) {
265  let (pre, post) = if post.is_none() {
266    (None, pre)
267  } else {
268    (pre, post)
269  };
270  (
271    pre.filter(|t| !t.is_empty()),
272    post.filter(|t| !t.is_empty()),
273  )
274}

276/// Perl `_cite_fallback`: delegate to the saved core \cite for
277/// non-author-year styles. Built by hand rather than via Invocation!: the
278/// core \cite is robust, so its top-level binding is a parameterless
279/// protect-wrapper and Invocation would revert ZERO arguments, silently
280/// dropping both the postnote and the keys.
281fn blx_cite_fallback(post: Option<Tokens>, keys: Tokens) -> Tokens {
282  let mut toks = vec![T_CS!("\\blx@saved@cite")];
283  if let Some(p) = post.filter(|t| !t.is_empty()) {
284    toks.push(T_OTHER!("["));
285    toks.extend(p.unlist());
286    toks.push(T_OTHER!("]"));
287  }
288  toks.push(T_BEGIN!());
289  toks.extend(keys.unlist());
290  toks.push(T_END!());
291  Tokens::new(toks)
292}

419/// One `[pre][post]{keys}` group of a biblatex multicite command.
420type BlxCiteGroup = (Option<Tokens>, Option<Tokens>, Tokens);
421
422/// Perl `_read_multicite_groups`: greedily read repeated
423/// `[pre][post]{keys}` groups from the gullet; stop at the first token
424/// that starts neither an optional nor a mandatory group.
425fn blx_read_multicite_groups() -> Result<Vec<BlxCiteGroup>> {
426  let mut groups: Vec<BlxCiteGroup> = Vec::new();
427  loop {
428    skip_spaces()?;
429    let Some(next) = read_token()? else { break };
430    if next.get_catcode() == Catcode::BEGIN {
431      // `{keys}` with no optional args.
432      unread_one(next);
433      let keys = read_arg(ExpansionLevel::Off)?;
434      groups.push((None, None, keys));
435    } else if next.get_catcode() == Catcode::OTHER && next.with_str(|s| s == "[") {
436      unread_one(next);
437      let opt1 = read_optional(None)?;
438      let opt2 = read_optional(None)?;
439      let keys = read_arg(ExpansionLevel::Off)?;
440      let (pre, post) = if opt2.is_some() {
441        (opt1, opt2)
442      } else {
443        (None, opt1)
444      };
445      groups.push((
446        pre.filter(|t| !t.is_empty()),
447        post.filter(|t| !t.is_empty()),
448        keys,
449      ));
450    } else {
451      unread_one(next); // not ours — put it back
452      break;
453    }
454  }
455  Ok(groups)
456}

458/// Perl `_joined_keys`: comma-join all groups' keys for delegation to a
459/// single \cite in the non-author-year fallback.
460fn blx_joined_keys(groups: &[BlxCiteGroup]) -> Tokens {
461  let mut toks: Vec<Token> = Vec::new();
462  for (_, _, keys) in groups {
463    if !toks.is_empty() {
464      toks.push(T_OTHER!(","));
465    }
466    toks.extend(keys.clone().unlist());
467  }
468  Tokens::new(toks)
469}

471/// Join per-group token runs with "; " (Perl's multicite group separator).
472fn blx_join_groups(parts: Vec<Vec<Token>>) -> Vec<Token> {
473  let mut body: Vec<Token> = Vec::new();
474  for (i, part) in parts.into_iter().enumerate() {
475    if i > 0 {
476      body.push(T_OTHER!(";"));
477      body.push(T_SPACE!());
478    }
479    body.extend(part);
480  }
481  body
482}

624/// Perl label split `/^(.+),\s*(\d{4}\w*)$/` — greedy `.+` means the LAST
625/// comma whose tail is a 4-digit year plus an optional disambiguation
626/// suffix wins. Returns (author_part, year_with_suffix).
627fn blx_split_ay_label(label: &str) -> Option<(String, String)> {
628  for (idx, _) in label.char_indices().rev().filter(|(_, c)| *c == ',') {
629    if idx == 0 {
630      continue;
631    }
632    let tail = label[idx + 1..].trim_start();
633    if tail.len() >= 4
634      && tail.chars().take(4).all(|c| c.is_ascii_digit())
635      && tail
636        .chars()
637        .skip(4)
638        .all(|c| c.is_ascii_alphanumeric() || c == '_')
639    {
640      return Some((label[..idx].to_string(), tail.to_string()));
641    }
642  }
643  None
644}
