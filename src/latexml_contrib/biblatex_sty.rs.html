3// === biblatex .bbl entry-pipeline helpers ===
4// Mirror Perl ar5iv-bindings/biblatex.sty.ltxml entry/endentry/name/list/field
5// closures (L127-340). The .bbl file emits \entry{key}{type}{} … \endentry per
6// reference; between them \field/\strng/\list/\name records metadata into the
7// `biblatex_entry` HashStored, and \endentry flushes a `\bibitem[label]{key}
8// authors. \newblock title. \newblock In: journal. year, pages.` token stream
9// onto `rebuilt_bibtex_variant`. The list-end macros (\enddatalist etc.) wrap
10// the accumulated variant in `\thebibliography{count}…\endthebibliography`.
11
12fn bib_entry_get() -> SymHashMap<Stored> {
13  match state::lookup_value("biblatex_entry") {
14    Some(Stored::HashStored(map)) => map.clone(),
15    _ => SymHashMap::default(),
16  }
17}
18
19fn bib_entry_save(map: SymHashMap<Stored>) {
20  state::assign_value(
21    "biblatex_entry",
22    Stored::HashStored(map),
23    Some(Scope::Global),
24  );
25}
26
27fn bib_entry_set_tokens(name: &str, val: Tokens) {
28  let mut entry = bib_entry_get();
29  entry.insert(name, Stored::Tokens(val));
30  bib_entry_save(entry);
31}
32
33fn bib_entry_get_tokens(map: &SymHashMap<Stored>, name: &str) -> Option<Tokens> {
34  map.get(name).and_then(|s| match s {
35    Stored::Tokens(t) => Some(t.clone()),
36    _ => None,
37  })
38}
39
40fn bib_state_int(key: &str) -> i64 {
41  match state::lookup_value(key) {
42    Some(Stored::Int(n)) => n,
43    _ => 0,
44  }
45}
46
47fn bib_state_set_int(key: &str, value: i64) {
48  state::assign_value(key, Stored::Int(value), Some(Scope::Global));
49}

51fn bib_variant_push(toks: Vec<Token>) {
52  let mut acc: Vec<Token> = match state::lookup_value("rebuilt_bibtex_variant") {
53    Some(Stored::Tokens(t)) => t.clone().unlist(),
54    _ => Vec::new(),
55  };
56  acc.extend(toks);
57  state::assign_value(
58    "rebuilt_bibtex_variant",
59    Stored::Tokens(Tokens::new(acc)),
60    Some(Scope::Global),
61  );
62}

64fn bib_as_thebibliography() -> Tokens {
65  let variant: Vec<Token> = match state::lookup_value("rebuilt_bibtex_variant") {
66    Some(Stored::Tokens(t)) => t.clone().unlist(),
67    _ => return Tokens::default(),
68  };
69  if variant.is_empty() {
70    return Tokens::default();
71  }
72  // Reset variant and entry-count so re-invocation is idempotent (matches
73  // Perl L113-115).
74  state::assign_value(
75    "rebuilt_bibtex_variant",
76    Stored::Tokens(Tokens::default()),
77    Some(Scope::Global),
78  );
79  let count = bib_state_int("biblatex_entry_count");
80  bib_state_set_int("biblatex_entry_count", 0);
81  let preamble: Vec<Token> = match state::lookup_value("biblatex_preamble") {
82    Some(Stored::Tokens(t)) => t.clone().unlist(),
83    _ => Vec::new(),
84  };
85  let mut result: Vec<Token> = Vec::with_capacity(variant.len() + 16);
86  result.push(T_CS!("\\thebibliography"));
87  result.push(T_BEGIN!());
88  result.extend(preamble);
89  result.extend(ExplodeText!(&count.to_string()));
90  result.push(T_END!());
91  result.extend(variant);
92  result.push(T_CS!("\\endthebibliography"));
93  Tokens::new(result)
94}

468  // Perl L270-346: \name{type}{count}{maybe-content} — biblatex's author
469  // record. The TeX-2.5+ .bbl shape is `\name{author}{N}{}{ {{}{Family}…} }`
470  // where the 3rd arg is empty and the 4th is the author body. Older variants
471  // pass 3 args. Simplified port: declare 4 mandatory args; the 4th-arg
472  // capture covers the modern shape used by the vast majority of arxiv .bbl
473  // files. The body holds N inner-author groups, each of the form
474  // `{hash}{family}{familyi}{given}{giveni}{}{}{}{}` — we extract family +
475  // given pairs into "Given Family" strings (no Perl-faithful keyval/hash
476  // ordering yet) and stash them comma-joined under `authors_str` /
477  // `editors_str` in the entry hash for `\endentry` to emit verbatim.
478  DefMacro!("\\name{}{}{}{}", sub[(ty, _count, _maybe, body)] {
479    let type_str = ty.to_string();
480    // The body's tokens start with `{` `{}` (empty hash) or `{hash=…}` then
481    // `{family}{familyi}{given}{giveni}{}{}{}{}` repeated, separated by
482    // optional whitespace. Walk top-level groups.
483    let body_toks: Vec<Token> = body.unlist();
484    // Helper: scan one balanced {...} group, advancing index.
485    fn read_group(tokens: &[Token], i: &mut usize) -> Option<Vec<Token>> {
486      while *i < tokens.len() {
487        let cc = tokens[*i].code;
488        if cc == Catcode::SPACE { *i += 1; continue; }
489        if cc == Catcode::BEGIN { break; }
490        return None; // not a group
491      }
492      if *i >= tokens.len() { return None; }
493      *i += 1; // consume BEGIN
494      let mut depth = 1usize;
495      let mut out = Vec::new();
496      while *i < tokens.len() {
497        let cc = tokens[*i].code;
498        if cc == Catcode::BEGIN {
499          depth += 1;
500          out.push(tokens[*i]);
501        } else if cc == Catcode::END {
502          depth -= 1;
503          if depth == 0 { *i += 1; return Some(out); }
504          out.push(tokens[*i]);
505        } else {
506          out.push(tokens[*i]);
507        }
508        *i += 1;
509      }
510      Some(out) // unterminated: best effort
511    }
512    let mut names: Vec<String> = Vec::new();
513    let mut idx = 0usize;
514    while idx < body_toks.len() {
