64fn bib_as_thebibliography() -> Tokens {
65  let variant: Vec<Token> = match state::lookup_value("rebuilt_bibtex_variant") {
66    Some(Stored::Tokens(t)) => t.unlist(),
67    _ => return Tokens::default(),
68  };
69  if variant.is_empty() {
70    return Tokens::default();
71  }
72  // Reset variant and entry-count so re-invocation is idempotent (matches
73  // Perl L113-115).
74  state::assign_value(
75    "rebuilt_bibtex_variant",
76    Stored::Tokens(Tokens::default()),
77    Some(Scope::Global),
78  );
79  let count = bib_state_int("biblatex_entry_count");
80  bib_state_set_int("biblatex_entry_count", 0);
81  let preamble: Vec<Token> = match state::lookup_value("biblatex_preamble") {
82    Some(Stored::Tokens(t)) => t.unlist(),
83    _ => Vec::new(),
84  };
85  let mut result: Vec<Token> = Vec::with_capacity(variant.len() + 16);
86  result.push(T_CS!("\\thebibliography"));
87  result.push(T_BEGIN!());
88  result.extend(preamble);
89  result.extend(ExplodeText!(&count.to_string()));
90  result.push(T_END!());
91  result.extend(variant);
92  result.push(T_CS!("\\endthebibliography"));
93  Tokens::new(result)
94}

96/// Perl `$fullname =~ s/\\\w+|[}{]//g` (ar5iv biblatex.sty.ltxml L324):
97/// strip leftover control sequences (`\bibinitperiod`, …) and braces from a
98/// name fragment, then trim. `\w` is Perl-ASCII (`[A-Za-z0-9_]`) — a backslash
99/// NOT followed by a word char (e.g. the `\"` of an accent) is preserved, as
100/// in Perl.
101fn bib_clean_name(s: &str) -> String {
102  let mut out = String::with_capacity(s.len());
103  let mut chars = s.chars().peekable();
104  while let Some(ch) = chars.next() {
105    if ch == '\\' {
106      if chars.peek().is_some_and(|c| c.is_ascii_alphanumeric() || *c == '_') {
107        while chars.peek().is_some_and(|c| c.is_ascii_alphanumeric() || *c == '_') {
108          chars.next();
109        }
110      } else {
111        out.push(ch);
112      }
113    } else if ch != '{' && ch != '}' {
114      out.push(ch);
115    }
116  }
117  out.trim().to_string()
118}

120/// Parse a biblatex keyval name block — the inner sub-group of a modern
121/// (biber, bbl format ≥ 3.x) `\name` author record, e.g.
122/// `family={Turtayev},familyi={T\bibinitperiod},given={Rustem},giveni=…,givenun=0`.
123/// Splits on depth-0 commas (commas inside `{…}` don't split), then for each
124/// `key=value` strips one layer of surrounding braces off the value. Mirrors
125/// the outcome of Perl's `LaTeXML::Core::KeyVals->readFrom` (L302-306) without
126/// the full KeyVals machinery — we only need `given`/`family` (and their
127/// `i`-initial fallbacks).
128fn parse_name_keyvals(s: &str) -> Vec<(String, String)> {
129  let mut pairs: Vec<(String, String)> = Vec::new();
130  let mut depth = 0i32;
131  let mut cur = String::new();
132  let flush = |seg: &str, pairs: &mut Vec<(String, String)>| {
133    let seg = seg.trim();
134    if seg.is_empty() {
135      return;
136    }
137    if let Some(eq) = seg.find('=') {
138      let key = seg[..eq].trim().to_string();
139      let mut val = seg[eq + 1..].trim();
140      if val.starts_with('{') && val.ends_with('}') && val.len() >= 2 {
141        val = &val[1..val.len() - 1];
142      }
143      pairs.push((key, val.to_string()));
144    }
145  };
146  for ch in s.chars() {
147    match ch {
148      '{' => { depth += 1; cur.push(ch); }
149      '}' => { depth -= 1; cur.push(ch); }
150      ',' if depth == 0 => { flush(&cur, &mut pairs); cur.clear(); }
151      _ => cur.push(ch),
152    }
153  }
154  flush(&cur, &mut pairs);
155  pairs
156}

158/// Perl L232-237 / L253-258: strip leading SPACE/`{` and trailing SPACE/`}`
159/// tokens from a captured DOI/eprint field value before splicing it into an
160/// `\href{…}` target, so a `\field{doi}{ {10.x/y} }`-style value yields a
161/// clean URI.
162fn bib_trim_url_tokens(toks: Tokens) -> Vec<Token> {
163  let mut v = toks.unlist();
164  let mut start = 0usize;
165  while start < v.len() {
166    let t = &v[start];
167    if t.code == Catcode::SPACE || t.with_str(|s| s == "{") { start += 1; } else { break; }
168  }
169  let mut end = v.len();
170  while end > start {
171    let t = &v[end - 1];
172    if t.code == Catcode::SPACE || t.with_str(|s| s == "}") { end -= 1; } else { break; }
173  }
174  v.truncate(end);
175  v.drain(..start);
176  v
177}
