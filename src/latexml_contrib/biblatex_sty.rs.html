3// === biblatex .bbl entry-pipeline helpers ===
4// Mirror Perl ar5iv-bindings/biblatex.sty.ltxml entry/endentry/name/list/field
5// closures (L127-340). The .bbl file emits \entry{key}{type}{} … \endentry per
6// reference; between them \field/\strng/\list/\name records metadata into the
7// `biblatex_entry` HashStored, and \endentry flushes a `\bibitem[label]{key}
8// authors. \newblock title. \newblock In: journal. year, pages.` token stream
9// onto `rebuilt_bibtex_variant`. The list-end macros (\enddatalist etc.) wrap
10// the accumulated variant in `\thebibliography{count}…\endthebibliography`.
11
12fn bib_entry_get() -> SymHashMap<Stored> {
13  match state::lookup_value("biblatex_entry") {
14    Some(Stored::HashStored(map)) => map,
15    _ => SymHashMap::default(),
16  }
17}
18
19fn bib_entry_save(map: SymHashMap<Stored>) {
20  state::assign_value(
21    "biblatex_entry",
22    Stored::HashStored(map),
23    Some(Scope::Global),
24  );
25}
26
27fn bib_entry_set_tokens(name: &str, val: Tokens) {
28  let mut entry = bib_entry_get();
29  entry.insert(name, Stored::Tokens(val));
30  bib_entry_save(entry);
31}
32
33fn bib_entry_get_tokens(map: &SymHashMap<Stored>, name: &str) -> Option<Tokens> {
34  map.get(name).and_then(|s| match s {
35    Stored::Tokens(t) => Some(t.clone()),
36    _ => None,
37  })
38}
39
40fn bib_state_int(key: &str) -> i64 {
41  match state::lookup_value(key) {
42    Some(Stored::Int(n)) => n,
43    _ => 0,
44  }
45}
46
47fn bib_state_set_int(key: &str, value: i64) {
48  state::assign_value(key, Stored::Int(value), Some(Scope::Global));
49}
50
51fn bib_variant_push(toks: Vec<Token>) {
52  let mut acc: Vec<Token> = match state::lookup_value("rebuilt_bibtex_variant") {
53    Some(Stored::Tokens(t)) => t.unlist(),
54    _ => Vec::new(),
55  };
56  acc.extend(toks);
57  state::assign_value(
58    "rebuilt_bibtex_variant",
59    Stored::Tokens(Tokens::new(acc)),
60    Some(Scope::Global),
61  );
62}
63
64fn bib_as_thebibliography() -> Tokens {
65  let variant: Vec<Token> = match state::lookup_value("rebuilt_bibtex_variant") {
66    Some(Stored::Tokens(t)) => t.unlist(),
67    _ => return Tokens::default(),
68  };
69  if variant.is_empty() {
70    return Tokens::default();
71  }
72  // Reset variant and entry-count so re-invocation is idempotent (matches
73  // Perl L113-115).
74  state::assign_value(
75    "rebuilt_bibtex_variant",
76    Stored::Tokens(Tokens::default()),
77    Some(Scope::Global),
78  );
79  let count = bib_state_int("biblatex_entry_count");
80  bib_state_set_int("biblatex_entry_count", 0);
81  let preamble: Vec<Token> = match state::lookup_value("biblatex_preamble") {
82    Some(Stored::Tokens(t)) => t.unlist(),
83    _ => Vec::new(),
84  };
85  let mut result: Vec<Token> = Vec::with_capacity(variant.len() + 16);
86  result.push(T_CS!("\\thebibliography"));
87  result.push(T_BEGIN!());
88  result.extend(preamble);
89  result.extend(ExplodeText!(&count.to_string()));
90  result.push(T_END!());
91  result.extend(variant);
92  result.push(T_CS!("\\endthebibliography"));
93  Tokens::new(result)
94}

289  // Perl L132-263: \endentry — flush the accumulated entry hash as a
290  // `\bibitem[label]{key} authors. \newblock title. \newblock In: journal.
291  // year, pages.` token stream onto rebuilt_bibtex_variant.
292  // Simplified port: handles label-or-auto-label, key, authors (if collected
293  // by \name as plain "fullname" tokens), title, journal/booktitle, year,
294  // pages, doi/url/eprint. Pre-typeset name strings (the `{names}` array
295  // form) are emitted comma-joined.
296  DefMacro!("\\endentry", sub[_args] {
297    let entry = bib_entry_get();
298    state::assign_value("biblatex_entry", Stored::None, Some(Scope::Global));
299
300    // label: Perl uses labelalpha if present, else label, else auto-counter.
301    let label_toks = bib_entry_get_tokens(&entry, "labelalpha")
302      .or_else(|| bib_entry_get_tokens(&entry, "label"));
303    let label_str: String = match label_toks {
304      Some(t) if !t.is_empty() => {
305        // Perl: strip \word and braces from label for safety.
306        let s = t.to_string();
307        let mut cleaned = String::with_capacity(s.len());
308        let mut chars = s.chars();
309        while let Some(ch) = chars.next() {
310          if ch == '\\' {
311            // skip \word
312            for c in chars.by_ref() { if !c.is_ascii_alphabetic() { break; } }
313          } else if ch != '{' && ch != '}' {
314            cleaned.push(ch);
315          }
316        }
317        let cleaned = cleaned.trim().to_string();
318        if cleaned.is_empty() {
319          let n = bib_state_int("biblatex_auto_label") + 1;
320          bib_state_set_int("biblatex_auto_label", n);
321          n.to_string()
322        } else {
323          cleaned
324        }
325      },
326      _ => {
327        let n = bib_state_int("biblatex_auto_label") + 1;
328        bib_state_set_int("biblatex_auto_label", n);
329        n.to_string()
330      },
331    };
332
333    // Bump entry count for thebibliography wrapper.
334    bib_state_set_int("biblatex_entry_count", bib_state_int("biblatex_entry_count") + 1);
335
336    let mut variant: Vec<Token> = Vec::with_capacity(64);
337    let key_toks = bib_entry_get_tokens(&entry, "key").unwrap_or_default();
338    variant.push(T_CS!("\\bibitem"));
339    variant.push(T_OTHER!("["));
340    variant.extend(ExplodeText!(&label_str));
341    variant.push(T_OTHER!("]"));
342    variant.push(T_BEGIN!());
343    variant.extend(key_toks.unlist());
344    variant.push(T_END!());
345
346    // Authors: if \name stashed a comma-joined string under "authors_str",
347    // emit it. Defer the et-al / per-author re-tokenization for now — most
348    // .bbl files give us pre-formatted author tokens.
349    let authors_toks = bib_entry_get_tokens(&entry, "authors_str");
350    let mut have_authors = false;
351    if let Some(toks) = authors_toks {
352      if !toks.is_empty() {
353        variant.extend(toks.unlist());
354        have_authors = true;
355      }
356    }
357
358    // Title
359    if let Some(title) = bib_entry_get_tokens(&entry, "title") {
360      if !title.is_empty() {
361        if have_authors {
362          variant.push(T_CS!("\\newblock"));
363        }
364        variant.push(T_OTHER!("`"));
365        variant.push(T_OTHER!("`"));
366        variant.extend(title.unlist());
367        variant.push(T_OTHER!("'"));
368        variant.push(T_OTHER!("'"));
369      }
370    }
371    // Note
372    if let Some(note) = bib_entry_get_tokens(&entry, "note") {
373      if !note.is_empty() {
374        variant.push(T_SPACE!());
375        variant.extend(note.unlist());
376      }
377    }
378    // Journal / booktitle
379    let journal = bib_entry_get_tokens(&entry, "booktitle")
380      .or_else(|| bib_entry_get_tokens(&entry, "journaltitle"))
381      .or_else(|| bib_entry_get_tokens(&entry, "journal"));
382    if let Some(j) = journal.as_ref() {
383      if !j.is_empty() {
384        variant.push(T_CS!("\\newblock"));
385        variant.extend(ExplodeText!("In "));
386        variant.push(T_CS!("\\emph"));
387        variant.push(T_BEGIN!());
388        variant.extend(j.clone().unlist());
389        variant.push(T_END!());
390      }
391    }
392    // Volume + (number)
393    if let Some(volume) = bib_entry_get_tokens(&entry, "volume") {
394      if journal.is_some() && !volume.is_empty() {
395        variant.push(T_SPACE!());
396        variant.push(T_CS!("\\textbf"));
397        variant.push(T_BEGIN!());
398        variant.extend(volume.unlist());
399        if let Some(num) = bib_entry_get_tokens(&entry, "number") {
400          if !num.is_empty() {
401            variant.push(T_OTHER!("."));
402            variant.extend(num.unlist());
403          }
404        }
405        variant.push(T_END!());
406      }
407    }
408    // Publisher / location
409    if let Some(publisher) = bib_entry_get_tokens(&entry, "publisher") {
410      if !publisher.is_empty() {
411        variant.push(T_CS!("\\newblock"));
412        if let Some(loc) = bib_entry_get_tokens(&entry, "location") {
413          if !loc.is_empty() {
414            variant.extend(loc.unlist());
415            variant.push(T_OTHER!(":"));
416            variant.push(T_SPACE!());
417          }
418        }
419        variant.extend(publisher.unlist());
420      }
421    }
422    // Year
423    if let Some(year) = bib_entry_get_tokens(&entry, "year") {
424      if !year.is_empty() {
425        variant.push(T_OTHER!(","));
426        variant.push(T_SPACE!());
427        variant.extend(year.unlist());
428      }
429    }
430    // Pages
431    if let Some(pages) = bib_entry_get_tokens(&entry, "pages") {
432      if !pages.is_empty() {
433        variant.push(T_OTHER!(","));
434        variant.push(T_SPACE!());
435        variant.extend(ExplodeText!("pp. "));
436        variant.extend(pages.unlist());
437      }
438    }
439    // DOI / URL / eprint (URL-style)
440    if let Some(doi) = bib_entry_get_tokens(&entry, "doi") {
441      if !doi.is_empty() {
442        variant.push(T_CS!("\\newblock"));
443        variant.extend(ExplodeText!("DOI: "));
444        variant.push(T_CS!("\\href"));
445        variant.push(T_BEGIN!());
446        variant.extend(ExplodeText!("https://dx.doi.org/"));
447        variant.extend(doi.clone().unlist());
448        variant.push(T_END!());
449        variant.push(T_BEGIN!());
450        variant.extend(doi.unlist());
451        variant.push(T_END!());
452      }
453    } else if let Some(url) = bib_entry_get_tokens(&entry, "url") {
454      if !url.is_empty() {
455        variant.push(T_CS!("\\newblock"));
456        variant.extend(ExplodeText!("URL: "));
457        variant.push(T_CS!("\\url"));
458        variant.push(T_BEGIN!());
459        variant.extend(url.unlist());
460        variant.push(T_END!());
461      }
462    } else if let Some(eprint) = bib_entry_get_tokens(&entry, "eprint") {
463      if !eprint.is_empty() {
464        variant.push(T_CS!("\\newblock"));
465        let etype = bib_entry_get_tokens(&entry, "eprinttype")
466          .map(|t| t.to_string().to_uppercase()).unwrap_or_default();
467        let etype = if etype == "ARXIV" { "arXiv".to_string() }
468          else if etype.is_empty() { "eprint".to_string() }
469          else { etype };
470        variant.extend(ExplodeText!(&format!("{etype}:")));
471        variant.push(T_SPACE!());
472        if etype == "arXiv" {
473          variant.push(T_CS!("\\href"));
474          variant.push(T_BEGIN!());
475          variant.extend(ExplodeText!("https://arxiv.org/abs/"));
476          variant.extend(eprint.clone().unlist());
477          variant.push(T_END!());
478          variant.push(T_BEGIN!());
479          variant.extend(eprint.unlist());
480          variant.push(T_END!());
481        } else {
482          variant.extend(eprint.unlist());
483        }
484      }
485    }
486
487    bib_variant_push(variant);
488    Ok(Tokens::default())
489  }, locked => true);

503  // `{hash}{family}{familyi}{given}{giveni}{}{}{}{}` — we extract family +
504  // given pairs into "Given Family" strings (no Perl-faithful keyval/hash
505  // ordering yet) and stash them comma-joined under `authors_str` /
506  // `editors_str` in the entry hash for `\endentry` to emit verbatim.
507  DefMacro!("\\name{}{}{}{}", sub[(ty, _count, _maybe, body)] {
508    let type_str = ty.to_string();
509    // The body's tokens start with `{` `{}` (empty hash) or `{hash=…}` then
510    // `{family}{familyi}{given}{giveni}{}{}{}{}` repeated, separated by
511    // optional whitespace. Walk top-level groups.
512    let body_toks: Vec<Token> = body.unlist();
513    // Helper: scan one balanced {...} group, advancing index.
514    fn read_group(tokens: &[Token], i: &mut usize) -> Option<Vec<Token>> {
515      while *i < tokens.len() {
516        let cc = tokens[*i].code;
517        if cc == Catcode::SPACE { *i += 1; continue; }
518        if cc == Catcode::BEGIN { break; }
519        return None; // not a group
520      }
521      if *i >= tokens.len() { return None; }
522      *i += 1; // consume BEGIN
523      let mut depth = 1usize;
524      let mut out = Vec::new();
525      while *i < tokens.len() {
526        let cc = tokens[*i].code;
527        if cc == Catcode::BEGIN {
528          depth += 1;
529          out.push(tokens[*i]);
530        } else if cc == Catcode::END {
531          depth -= 1;
532          if depth == 0 { *i += 1; return Some(out); }
533          out.push(tokens[*i]);
534        } else {
535          out.push(tokens[*i]);
536        }
537        *i += 1;
538      }
539      Some(out) // unterminated: best effort
540    }
541    let mut names: Vec<String> = Vec::new();
542    let mut idx = 0usize;
543    while idx < body_toks.len() {
544      // Skip space tokens between author groups.
545      while idx < body_toks.len() &&
546            body_toks[idx].code == Catcode::SPACE {
547        idx += 1;
548      }
549      if idx >= body_toks.len() { break; }
550      // Read the per-author group.
551      let author_grp = match read_group(&body_toks, &mut idx) {
552        Some(g) => g,
553        None => break,
554      };
555      // Inside, read up to 5 sub-groups: hash, family, familyi, given, giveni
556      let mut j = 0usize;
557      let _hash = read_group(&author_grp, &mut j);
558      let family = read_group(&author_grp, &mut j)
559        .map(|g| Tokens::new(g).to_string()).unwrap_or_default();
560      let _familyi = read_group(&author_grp, &mut j);
561      let given = read_group(&author_grp, &mut j)
562        .map(|g| Tokens::new(g).to_string()).unwrap_or_default();
563      // Trim whitespace.
564      let family = family.trim().to_string();
565      let given = given.trim().to_string();
566      let fullname = if !given.is_empty() && !family.is_empty() {
567        format!("{given} {family}")
568      } else if !family.is_empty() {
569        family
570      } else if !given.is_empty() {
571        given
572      } else {
573        continue;
574      };
575      names.push(fullname);
576    }
577    // Format with et-al limit (default 4 per Perl L192).
578    let etal_limit = bib_state_int("biblatex_maxbibnames");
579    let etal_limit = if etal_limit > 0 { etal_limit as usize } else { 4 };
580    let joined = if names.len() > etal_limit {
581      format!("{} et al.", names[0])
582    } else {
583      let mut acc = String::new();
584      let n = names.len();
585      for (k, name) in names.iter().enumerate() {
586        if k > 0 {
587          if k + 1 == n {
588            acc.push_str(" and ");
589          } else {
590            acc.push_str(", ");
591          }
592        }
593        acc.push_str(name);
594      }
595      acc
596    };
597    // Stash under "authors_str" or "editors_str" depending on type.
598    let key = if type_str.trim() == "editor" { "editors_str" } else { "authors_str" };
599    if !joined.is_empty() {
600      let toks = Tokens::new(ExplodeText!(&joined));
601      bib_entry_set_tokens(key, toks);
602    }
603    Ok(Tokens::default())
604  }, locked => true);

663  //     \verb{key}

666  // The first `\verb` is `\let`'d to `\biblatex@verb` and reads `{key}`;
667  // the second `\verb` then reads VALUE as a raw line; `\endverb` stores

670  // `{key}` and "Until:\\endverb" — the captured body is everything between
671  // the first `\verb{key}` line and `\endverb`, including the second
672  // `\verb` token plus the URL chars. We strip the inner `\verb` token and
673  // surrounding whitespace before storing.
674  // Without this, \verb LEAKS the URL into body text — and consumes the
675  // first character (`h` of `http`) as a `{}` arg, producing the
676  // characteristic `ttp://…` corruption on egpaper_final.tex.

678    let key_str = key.to_string();
679    let body_toks = body.unlist();
680    // Skip leading whitespace + the inner `\verb` token + one space.
681    let mut start = 0usize;
682    while start < body_toks.len() {
683      let cc = body_toks[start].code;
684      if cc == Catcode::SPACE || cc == Catcode::EOL { start += 1; continue; }
685      break;
686    }
687    if start < body_toks.len() && body_toks[start].code == Catcode::CS {
688      // Skip the `\verb` CS token (or whatever CS leads — should be \verb).
689      start += 1;
690      // Skip following space tokens.
691      while start < body_toks.len() {
692        let cc = body_toks[start].code;
693        if cc == Catcode::SPACE || cc == Catcode::EOL { start += 1; continue; }
694        break;
695      }
696    }
697    // Strip trailing whitespace.
698    let mut end = body_toks.len();
699    while end > start {
700      let cc = body_toks[end - 1].code;
701      if cc == Catcode::SPACE || cc == Catcode::EOL { end -= 1; continue; }
702      break;
703    }
704    // Sanitize: biblatex `\verb` is a verbatim primitive — its body is a
705    // literal string. The mouth tokenized chars with their normal catcodes
706    // (so `_` is SUB, `^` is SUPER, `#` is PARAM, etc.). When `\endentry`
707    // later splices these tokens into `\href{URL}{text}` the SUB chars
708    // trigger `Script _ can only appear in math mode` during horizontal
709    // digestion. Reset structural catcodes to OTHER so the captured string
710    // round-trips through the bibitem variant safely.
711    //
712    // Also detokenize CS tokens (e.g. `\href`, an inner `\verb`) and
713    // brace tokens to literal OTHER characters: biblatex .bbl files
714    // occasionally include `\verb \href{url}{label}` inside a `\verb`
715    // field (witness arXiv:1004.4538 entry 17, `\verb \href{...}{...}`
716    // wrapped across two `\verb` lines). Without detokenization the
717    // captured tokens later expand inside `\url{<body>}` and `\href`
718    // / `\verb` execute, pushing back tens of thousands of tokens and
719    // tripping the 650K PushbackLimit safety net.
720    //
721    // Witness cluster: ~29 papers/stage in next_warning_papers (Stages
722    // 15-20 v3) hit this on biblatex bbl `\verb 10.1162/EVCO_a_00133`.
723    let mut value_vec: Vec<Token> = Vec::with_capacity(end - start);
724    for tok in &body_toks[start..end] {
725      match tok.code {
726        Catcode::CS => {
727          // Spell out the CS name as OTHER chars, dropping the
728          // backslash: `\href` → `href`. (Matches the visible
729          // rendering of a verbatim URL.)
730          tok.with_str(|s| {
731            let name = s.strip_prefix('\\').unwrap_or(s);
732            for ch in name.chars() {
733              value_vec.push(T_OTHER!(&ch.to_string()));
734            }
735          });
736        },
737        Catcode::BEGIN | Catcode::END |
738        Catcode::SUB | Catcode::SUPER | Catcode::PARAM |
739        Catcode::ALIGN | Catcode::MATH | Catcode::ACTIVE => {
740          value_vec.push(Token {
741            text: tok.text,
742            code: Catcode::OTHER,
743            #[cfg(feature = "token-locators")]
744            loc: 0,
745          });
746        },
747        _ => value_vec.push(*tok),
748      }
749    }
750    let value = Tokens::new(value_vec);
751    bib_entry_set_tokens(key_str.trim(), value);
752    Ok(Tokens::default())
