1//! Stub for lmcs.cls (Logical Methods in Computer Science journal class).
2//!
3//! lmcs.cls extends amsart for the LMCS journal. The raw cls relies on
4//! pgfmath + tikz + XeTeXLinkBox for ORCID-logo rendering, which fails
5//! mid-load in our system; consequently `\lmcsdoi`, `\lmcsheading`,
6//! `\lmcsorcid` (the publication-metadata macros every LMCS paper uses
7//! in the preamble) end up undefined. Provide content-preserving stubs.
8//! Witness 2305.14448, 2305.19985.
9use latexml_package::prelude::*;
10
11
12LoadDefinitions!({
13  LoadClass!("amsart");
14  RequirePackage!("amsmath");
15  RequirePackage!("amssymb");
16  RequirePackage!("amsthm");
17  RequirePackage!("hyperref");
18  RequirePackage!("xcolor");
19  RequirePackage!("enumitem");
20  RequirePackage!("etoolbox");
21
22  // LMCS publication metadata. Real macros assign internal counters
23  // and set up running headers; for HTML rendering we just preserve
24  // the args as named notes.
25  DefMacro!("\\lmcsdoi{}{}{}",
26    "\\@add@frontmatter{ltx:note}[role=lmcs-doi]{Volume #1, Issue #2, Paper #3}");
27  // \lmcsheading{vol}{issue}{year}{pages}{subm}{publ}{rev}{spec_iss}{title}
28  // The raw cls signature is 7-args but with optional/positional variations.
29  // We don't reproduce the running-header layout — just discard, since the
30  // metadata is already captured by \lmcsdoi.
31  def_macro_noop("\\lmcsheading{}{}{}{}")?;
32  // \lmcsorcid{orcid-id} — render as a plain link rather than the
33  // tikz/XeTeXLinkBox logo construction.
34  DefMacro!("\\lmcsorcid{}",
35    "\\href{https://orcid.org/#1}{ORCID:#1}");
36
37  // Section-numbering and shortauthors/shorttitle helpers used by raw
38  // cls header layout. Stub as no-op or pass-through.
39  def_macro_noop("\\shorttitle{}")?;
40  def_macro_noop("\\shortauthors{}")?;
41
42  // `\dOi` placeholder produced by the raw cls when no \lmcsdoi was
43  // declared. Stub to empty so it doesn't appear as red error text.
44  def_macro_noop("\\dOi")?;
45});
