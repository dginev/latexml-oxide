1//! Stub for IMS (Institute of Mathematical Statistics) `imsart` class.
2//!
3//! imsart.cls loads `article` + requires `imsart.sty` (support file with
4//! \startlocaldefs, \endlocaldefs, etc.). We fall back to OmniBus and raw-
5//! load imsart.sty so most user macros become available.
6use latexml_package::prelude::*;
7
8
9LoadDefinitions!({
10  LoadClass!("OmniBus");
11  RequirePackage!("amsmath");
12  // NOTE: do NOT eagerly `RequirePackage!("amsthm")` here. OmniBus already
13  // provides lazy amsthm autoload (theorem-env stubs), and pre-loading it
14  // broke the common `\let\proof\relax` + `\usepackage{amsthm}` idiom: the
15  // paper's explicit \usepackage{amsthm} would no-op (already loaded), so
16  // amsthm's `\let\proof\@proof` never re-ran after the paper cleared
17  // `\proof` → `Error:undefined:{proof}`. Letting the paper's
18  // \usepackage{amsthm} be the first real load matches Perl (clean).
19  // Witness 1612.03054 (`\let\proof\relax` L5 + amsthm L22).
20  // imsart.cls L149: \RequirePackage{imsart}.
21  InputDefinitions!("imsart", noltxml => true, extension => Some(Cow::Borrowed("sty")));
22
23  // Frontmatter primitives commonly used in imsart papers but not
24  // always defined by imsart.sty (some are journal-driver dependent).
25  // \startlocaldefs / \endlocaldefs are defined in imsart.sty L657-660;
26  // these are belt-and-suspenders in case the raw load is short-circuited.
27  DefMacro!("\\startlocaldefs", "\\makeatletter");
28  DefMacro!("\\endlocaldefs", "\\makeatother");
29  // imsart.sty L2268, L2360: \let\kwd@sep\relax inside conditionals
30  // we may not fully replay. Define defensively. Witness 2406.17390.
31  Let!("\\kwd@sep", "\\relax");
32
33  // {funding} env — IMS journal funding-statement frontmatter.
34  // Preserve as ltx:note (content-preservation directive). Witness
35  // 2406.15844 (+5 imsart papers). Use internal_vertical mode so the
36  // body can contain paragraphs / lists without tripping
37  // mode-mismatch errors.
38  DefEnvironment!("{funding}",
39    "<ltx:note role='funding'>#body</ltx:note>",
40    mode => "internal_vertical");
41  // {acknowledgement} / {acknowledgements} aliases for spelling variants.
42  DefEnvironment!("{acknowledgement}",
43    "<ltx:acknowledgements>#body</ltx:acknowledgements>",
44    mode => "internal_vertical");
45  // {acks} env — IMS-specific acknowledgements ("acks" shorthand).
46  // Witness 2406.15844, 2406.04191, 2406.02840 (3 imsart papers).
47  DefEnvironment!("{acks}",
48    "<ltx:acknowledgements>#body</ltx:acknowledgements>",
49    mode => "internal_vertical");
50  // IMS authors use \orcid for ORCID identifier. Preserve as ltx:note.
51  DefMacro!("\\orcid{}",
52    "\\@add@frontmatter{ltx:note}[role=orcid]{#1}");
53  // IMS journal bibliography entry types — imsart.sty defines these as
54  // \def commands but they're used as environments in some .bbl files.
55  // Provide as no-op envs (the actual bibliography rendering is handled
56  // by biblatex/natbib elsewhere). Witness 2406.15844 (+4 imsart papers).
57  DefEnvironment!("{barticle}", "#body");
58  DefEnvironment!("{bbook}",    "#body");
59  DefEnvironment!("{bbooklet}", "#body");
60  DefEnvironment!("{binbook}",  "#body");
61  DefEnvironment!("{bincollection}", "#body");
62  DefEnvironment!("{bunpublished}",  "#body");
63  DefEnvironment!("{bmisc}",         "#body");
64  DefEnvironment!("{bproceedings}",  "#body");
65  DefEnvironment!("{bphdthesis}",    "#body");
66  DefEnvironment!("{bmastersthesis}", "#body");
67  DefEnvironment!("{btechreport}",   "#body");
68
69  // IMS bibliography field-tagging macros — imsart.sty defines these
70  // as NLM/JATS-style 1-arg setters inside `{barticle}` etc. envs:
71  // \bauthor{name}, \binits{initials}, \bfnm{first}, \bsnm{surname},
72  // \byear{2024}, \bvolume{42}, \bissue{3}, \bpages{1-20},
73  // \bjournal{Annals}, \bpublisher{Springer}, \bseries{Lecture Notes},
74  // \btitle{...}, \bmrnumber{...}, etc. Raw imsart.sty defines them,
75  // but its preamble has complex catcode/group state that sometimes
76  // fails mid-load, leaving these undefined. Provide content-
77  // preserving stubs that emit args inline so the substantive
78  // bibliography text survives. Witness 2305.13037, 2306.02821.
79  def_macro_identity("\\bauthor{}")?;
80  def_macro_identity("\\binits{}")?;
81  def_macro_identity("\\bfnm{}")?;
82  def_macro_identity("\\bsnm{}")?;
83  def_macro_identity("\\byear{}")?;
84  def_macro_identity("\\bvolume{}")?;
85  def_macro_identity("\\bissue{}")?;
86  def_macro_identity("\\bpages{}")?;
87  def_macro_identity("\\bjournal{}")?;
88  def_macro_identity("\\bpublisher{}")?;
89  def_macro_identity("\\bseries{}")?;
90  def_macro_identity("\\btitle{}")?;
91  def_macro_identity("\\bmrnumber{}")?;
92  def_macro_identity("\\bedition{}")?;
93  def_macro_identity("\\beditor{}")?;
94  def_macro_identity("\\beditortype{}")?;
95  def_macro_identity("\\baddress{}")?;
96  def_macro_identity("\\borganization{}")?;
97  def_macro_identity("\\bcollaboration{}")?;
98  def_macro_identity("\\bdoi{}")?;
99  def_macro_identity("\\burl{}")?;
100  def_macro_identity("\\bothertype{}")?;
101  def_macro_identity("\\bparticle{}")?;
102  def_macro_identity("\\bnote{}")?;
103  def_macro_identity("\\btype{}")?;
104  // imsart.sty `\common@pub@types` also `\let`s these to `\@firstofone`
105  // (identity), but they were missing from the list above — so an imsart
106  // `.bbl` using `\begin{barticle}…\betal{…}` (bold-"et al." separator) or
107  // `\banumber{…}` saw them undefined. Witness 1912.11583 (`\betal`, 1 error
108  // → 0). Mirror `\common@pub@types`.
109  def_macro_identity("\\betal{}")?;
110  def_macro_identity("\\banumber{}")?;
111  // Additional imsart bibliography field macros. The bundled imsart.cls/sty
112  // `\let`s each of these to `\@firstofone` (identity) inside its bib setup
113  // (or applies a style via `\set@bibl@cmd`, e.g. `\bbooktitle` → \itshape —
114  // we keep content-preserving identity, matching the sibling stubs above).
115  // They were missing, so an imsart `.bbl` using `\bbooktitle{…}` (book title
116  // in an `In …` reference), `\bchapter`, `\bschool` (theses), etc. saw them
117  // undefined. Witness 2006.02044 (`\bbooktitle`, 1 error → 0; Perl errors on
118  // ALL 28 imsart `\b*`/`{b*}` constructs, so this also surpasses Perl). NB:
119  // `\bmisc` is intentionally NOT added as a macro — it would clobber the
120  // `{bmisc}` environment defined above.
121  def_macro_identity("\\bbooktitle{}")?;
122  def_macro_identity("\\bchapter{}")?;
123  def_macro_identity("\\bhowpublished{}")?;
124  def_macro_identity("\\binstitution{}")?;
125  def_macro_identity("\\bisbn{}")?;
126  def_macro_identity("\\blocation{}")?;
127  def_macro_identity("\\bnumber{}")?;
128  def_macro_identity("\\bschool{}")?;
129  def_macro_identity("\\bsuffix{}")?;
130});
