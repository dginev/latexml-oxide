1//! Stub for IMS (Institute of Mathematical Statistics) `imsart` class.
2//!
3//! imsart.cls loads `article` + requires `imsart.sty` (support file with
4//! \startlocaldefs, \endlocaldefs, etc.). We fall back to OmniBus and raw-
5//! load imsart.sty so most user macros become available.
6use latexml_package::prelude::*;
7
8
9LoadDefinitions!({
10  LoadClass!("OmniBus");
11  RequirePackage!("amsmath");
12  RequirePackage!("amsthm");
13  // imsart.cls L149: \RequirePackage{imsart}.
14  InputDefinitions!("imsart", noltxml => true, extension => Some(Cow::Borrowed("sty")));
15
16  // Frontmatter primitives commonly used in imsart papers but not
17  // always defined by imsart.sty (some are journal-driver dependent).
18  // \startlocaldefs / \endlocaldefs are defined in imsart.sty L657-660;
19  // these are belt-and-suspenders in case the raw load is short-circuited.
20  DefMacro!("\\startlocaldefs", "\\makeatletter");
21  DefMacro!("\\endlocaldefs", "\\makeatother");
22  // imsart.sty L2268, L2360: \let\kwd@sep\relax inside conditionals
23  // we may not fully replay. Define defensively. Witness 2406.17390.
24  Let!("\\kwd@sep", "\\relax");
25
26  // {funding} env — IMS journal funding-statement frontmatter.
27  // Preserve as ltx:note (content-preservation directive). Witness
28  // 2406.15844 (+5 imsart papers). Use internal_vertical mode so the
29  // body can contain paragraphs / lists without tripping
30  // mode-mismatch errors.
31  DefEnvironment!("{funding}",
32    "<ltx:note role='funding'>#body</ltx:note>",
33    mode => "internal_vertical");
34  // {acknowledgement} / {acknowledgements} aliases for spelling variants.
35  DefEnvironment!("{acknowledgement}",
36    "<ltx:acknowledgements>#body</ltx:acknowledgements>",
37    mode => "internal_vertical");
38  // {acks} env — IMS-specific acknowledgements ("acks" shorthand).
39  // Witness 2406.15844, 2406.04191, 2406.02840 (3 imsart papers).
40  DefEnvironment!("{acks}",
41    "<ltx:acknowledgements>#body</ltx:acknowledgements>",
42    mode => "internal_vertical");
43  // IMS authors use \orcid for ORCID identifier. Preserve as ltx:note.
44  DefMacro!("\\orcid{}",
45    "\\@add@frontmatter{ltx:note}[role=orcid]{#1}");
46  // IMS journal bibliography entry types — imsart.sty defines these as
47  // \def commands but they're used as environments in some .bbl files.
48  // Provide as no-op envs (the actual bibliography rendering is handled
49  // by biblatex/natbib elsewhere). Witness 2406.15844 (+4 imsart papers).
50  DefEnvironment!("{barticle}", "#body");
51  DefEnvironment!("{bbook}",    "#body");
52  DefEnvironment!("{bbooklet}", "#body");
53  DefEnvironment!("{binbook}",  "#body");
54  DefEnvironment!("{bincollection}", "#body");
55  DefEnvironment!("{bunpublished}",  "#body");
56  DefEnvironment!("{bmisc}",         "#body");
57  DefEnvironment!("{bproceedings}",  "#body");
58  DefEnvironment!("{bphdthesis}",    "#body");
59  DefEnvironment!("{bmastersthesis}", "#body");
60  DefEnvironment!("{btechreport}",   "#body");
61
62  // IMS bibliography field-tagging macros — imsart.sty defines these
63  // as NLM/JATS-style 1-arg setters inside `{barticle}` etc. envs:
64  // \bauthor{name}, \binits{initials}, \bfnm{first}, \bsnm{surname},
65  // \byear{2024}, \bvolume{42}, \bissue{3}, \bpages{1-20},
66  // \bjournal{Annals}, \bpublisher{Springer}, \bseries{Lecture Notes},
67  // \btitle{...}, \bmrnumber{...}, etc. Raw imsart.sty defines them,
68  // but its preamble has complex catcode/group state that sometimes
69  // fails mid-load, leaving these undefined. Provide content-
70  // preserving stubs that emit args inline so the substantive
71  // bibliography text survives. Witness 2305.13037, 2306.02821.
72  def_macro_identity("\\bauthor{}")?;
73  def_macro_identity("\\binits{}")?;
74  def_macro_identity("\\bfnm{}")?;
75  def_macro_identity("\\bsnm{}")?;
76  def_macro_identity("\\byear{}")?;
77  def_macro_identity("\\bvolume{}")?;
78  def_macro_identity("\\bissue{}")?;
79  def_macro_identity("\\bpages{}")?;
80  def_macro_identity("\\bjournal{}")?;
81  def_macro_identity("\\bpublisher{}")?;
82  def_macro_identity("\\bseries{}")?;
83  def_macro_identity("\\btitle{}")?;
84  def_macro_identity("\\bmrnumber{}")?;
85  def_macro_identity("\\bedition{}")?;
86  def_macro_identity("\\beditor{}")?;
87  def_macro_identity("\\beditortype{}")?;
88  def_macro_identity("\\baddress{}")?;
89  def_macro_identity("\\borganization{}")?;
90  def_macro_identity("\\bcollaboration{}")?;
91  def_macro_identity("\\bdoi{}")?;
92  def_macro_identity("\\burl{}")?;
93  def_macro_identity("\\bothertype{}")?;
94  def_macro_identity("\\bparticle{}")?;
95  def_macro_identity("\\bnote{}")?;
96  def_macro_identity("\\btype{}")?;
97});
