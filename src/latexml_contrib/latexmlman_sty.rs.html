1//! Port of LaTeXML's `latexmlman.sty` schema-doc support.
2//!
3//! Used by the scholarly-schema docs pipeline to render
4//! `\begin{schemamodule}...\end{schemamodule}` blocks containing
5//! `\elementdef` / `\attrdef` / `\patterndef` / `\moduleref` /
6//! `\patternref` / `\elementref` / `\attrval` / `\typename` etc.
7//!
8//! Mirrors `doc/sty/latexmlman.sty` from Perl LaTeXML for the
9//! schema-doc-relevant subset; the title-page customizations and the
10//! `\pod` / `\ltxpod` family are intentionally omitted (they target
11//! Perl module documentation, not schema docs).
12//!
13//! `\cleanhypername` is intentionally NOT ported. Identifiers in the
14//! emitted schema docs may contain `:` (e.g. `schema.namespace1:div`);
15//! HTML5 permits that and cross-references resolve as long as
16//! `\hypertarget` and `\hyperlink` use the same form, which they do.
17
18use latexml_package::prelude::*;
19
20
21LoadDefinitions!({
22  // hyperref provides \hypertarget / \hyperlink / \hyperref used below.
23  RequirePackage!("hyperref");
24
25  //--- Font-switching aliases (mirror upstream \def commands) ---------
26  DefMacro!("\\perlfont",   "\\ttfamily");
27  DefMacro!("\\shellfont",  "\\ttfamily");
28  DefMacro!("\\latexfont",  "\\ttfamily");
29  DefMacro!("\\schemafont", "\\sffamily");
30  DefMacro!("\\patternfont","\\sffamily\\slshape");
31
32  //--- Phrase-level inline markup -------------------------------------
33  DefMacro!("\\code{}",     "{\\ttfamily #1}");
34  DefMacro!("\\codevar{}",  "{\\ttfamily\\itshape #1}");
35  DefMacro!("\\method{}",   "{\\ttfamily ->#1}");
36  DefMacro!("\\attr{}",     "{\\sffamily #1}");
37  DefMacro!("\\attrval{}",  "{\\ttfamily #1}");
38  DefMacro!("\\cmd{}",      "{\\ttfamily #1}");
39  DefMacro!("\\cs{}",       "{\\ttfamily $\\backslash$#1}");
40  DefMacro!("\\typename{}", "\\textit{#1}");
41
42  //--- Schema module section ------------------------------------------
43  // `\schemamodule` is a labeled section. Definitions inside it are
44  // \newtheorem-defined named statements, NOT subsections — so they
45  // appear as `<ltx:theorem>` blocks at module scope, side-by-side
46  // with the module preamble paragraphs.
47  DefMacro!("\\schemamodule{}",
48    "\\section{Module \\texttt{#1}}\\label{schema.#1}");
49  def_macro_noop("\\endschemamodule")?;
50
51  //--- Schema-doc list environments -----------------------------------
52  // Each kind bucket (`\subsection{Patterns}` / `\subsection{Elements}`)
53  // wraps its defs in a description list. Defs are list items; their
54  // bodies open further description lists for facts (Attributes /
55  // Content / Used by). Mirrors upstream `latexmlman.sty` exactly —
56  // nested `<ltx:item>`s under another item are valid (an inline
57  // `\elementdef` inside a pattern's content becomes a nested `<dl>`),
58  // which avoids the structural conflict that bare-`\subsubsection`
59  // defs hit when nested inside another def's content model.
60  DefMacro!("\\elementdescription",      "\\begin{description}");
61  DefMacro!("\\endelementdescription",   "\\end{description}");
62  DefMacro!("\\patterndescription",      "\\begin{description}");
63  DefMacro!("\\endpatterndescription",   "\\end{description}");
64
65  //--- Schema definition macros ---------------------------------------
66  // Definitions are description-list items, matching the upstream
67  // `RelaxNG.pm` shape (Perl `\elementdef`/`\patterndef` always
68  // emitted `\item[Element/Pattern …]`, not a section command). The
69  // anchor is `\hypertarget{schema.<name>}{…}` (paired with
70  // `\hyperlink` in the cross-references below).
71  DefMacro!("\\elementdef{}{}{}",
72    "\\item[\\textit{Element }{\\bfseries\\schemafont #1}]\
73     \\hypertarget{schema.#1}{#2}\
74     \\begin{elementdescription}#3\\end{elementdescription}");
75
76  // \attrdef stays an `\item` — it only ever appears inside an
77  // \elementdef / \patterndef body's description list.
78  DefMacro!("\\attrdef{}{}{}",
79    "\\item[\\textit{Attribute }{\\bfseries\\schemafont #1}] = #3\
80     \\par\\noindent #2");
81
82  DefMacro!("\\patterndef{}{}{}",
83    "\\item[\\textit{Pattern }{\\bfseries\\patternfont #1}]\
84     \\hypertarget{schema.#1}{#2}\
85     \\begin{patterndescription}#3\\end{patterndescription}");
86
87  DefMacro!("\\patternadd{}{}{}",
88    "\\item[\\textit{Add to }{\\bfseries\\patternfont #1}] \\hspace{1em} #2\
89     \\begin{patterndescription}#3\\end{patterndescription}");
90
91  DefMacro!("\\patterndefadd{}{}{}",
92    "\\item[\\textit{Add to }{\\bfseries\\patternfont #1}]\
93     \\hypertarget{schema.#1}{#2}\
94     \\begin{patterndescription}#3\\end{patterndescription}");
95
96  //--- Module-level prose abstract -------------------------------------
97  // Synthesized by `tools/genschema` from the `## comments` at the
98  // head of each RNC module file (which trang preserves as
99  // `<a:documentation>` annotations). Emits a paragraph carrying the
100  // `schema_module_narrative` class token; CSS in
101  // `resources/CSS/relaxng-schema-rustdoc-theme.css` styles that
102  // class as the left-bordered narrative aside above each module's
103  // definitions.
104  // `<ltx:para>` (a paragraph-block, not a single paragraph) so
105  // multi-paragraph abstracts — produced when trang emits more than
106  // one `<a:documentation>` per define — keep all their `<p>`s
107  // wrapped under one element with the `schema_module_narrative`
108  // marker class. With `<ltx:p>`, LaTeXML closes the paragraph at
109  // the first `\par` and starts new `<p>`s without the class, which
110  // leaves the trailing prose stranded outside the post-pass aside.
111  DefConstructor!(
112    "\\moduleabstract{}",
113    "<ltx:para class='schema_module_narrative'>#1</ltx:para>"
114  );
115  // Document-level narrative: the driver module's head comments,
116  // hoisted by genschema_oxide before the first `\section` so the
117  // overview survives the section split and renders on `index.html`.
118  // Mirrors \moduleabstract's shape; multi-paragraph content splits
119  // into sibling paras during digestion, so styling hooks should
120  // target the container region rather than rely on this class
121  // reaching every paragraph.
122  DefConstructor!(
123    "\\documentabstract{}",
124    "<ltx:para class='schema_document_narrative'>#1</ltx:para>"
125  );
126
127  //--- Cross-references -----------------------------------------------
128  DefMacro!("\\moduleref{}",  "\\hyperref[schema.#1]{{\\ttfamily #1}}");
129  DefMacro!("\\patternref{}", "\\hyperlink{schema.#1}{{\\sffamily\\slshape #1}}");
130  DefMacro!("\\elementref{}", "\\hyperlink{schema.#1}{{\\sffamily #1}}");
131
132  //--- Source-link footer for the cover page --------------------------
133  // `\schemasource{label}{url}` renders a small "Source:" line that
134  // links back to the upstream schema file. The orchestration shell
135  // (`tools/generate-scholarly-schema-docs`) invokes it with
136  // git-derived values when the schema lives in a checkout — typically
137  // a SHA-pinned GitHub/GitLab `blob/SHA/path` URL — and skips the
138  // call entirely when the schema isn't in any git repo.
139  //
140  // Emitted as a `<ltx:para class='schema_source'>` so the post-pass
141  // / theme stylesheet can style it (small, muted) without touching
142  // the document body. Uses `\href` so the link target is the URL
143  // and the label remains a friendly path/SHA fragment.
144  DefConstructor!(
145    "\\schemasource{}{}",
146    "<ltx:para class='schema_source'><ltx:text>Source: </ltx:text><ltx:ref href='#2' class='ltx_url'><ltx:text class='ltx_font_typewriter'>#1</ltx:text></ltx:ref></ltx:para>"
147  );
148});
