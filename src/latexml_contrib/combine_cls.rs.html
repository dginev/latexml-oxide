1//! Stub for combine.cls (multi-document combiner).
2//!
3//! combine.cls (TL `latex/combine/combine.cls`) is a class for
4//! merging multiple LaTeX documents into a single PDF. It defines
5//! `{papers}`, `{document*}`, etc. envs that wrap sub-document
6//! content. Our HTML/XML output is single-document; just render the
7//! body transparently. Witness 2310.14794.
8use latexml_package::prelude::*;
9
10LoadDefinitions!({
11  LoadClass!("article");
12  RequirePackage!("amsmath");
13  RequirePackage!("amssymb");
14  // Eager xcolor preload removed for Perl parity: it makes a later document
15  // xcolor[table] load a no-op, so colortbl/array never load and array m{}/b{}
16  // columns break (Unrecognized tabular template -> Extra alignment tab). The
17  // document loads xcolor itself; color/definecolor stay via hyperref->color.
18  // See ifacconf_cls.rs and SYNC_STATUS (eager-xcolor cluster).
19  RequirePackage!("hyperref");
20
21  // combine.cls L760 redefines `\import` to take ONE arg (the
22  // basename of a sub-document) and `\input`s `<arg>.tex`. This
23  // conflicts with the `import` PACKAGE's `\import{<dir>/}{<file>}`
24  // (two-arg form) so we must NOT RequirePackage{import} here —
25  // instead define `\import` as combine.cls does. Witness 2310.14794.
26  DefMacro!("\\import{}", "\\input{#1}");
27
28  // Sub-documents inside `{papers}` have their own preambles —
29  // `\documentclass` + `\usepackage` calls. Real combine.cls L194-209
30  // saves and re-routes these via `\c@lbdocumentclass` / `\c@lbusepackage`
31  // when wrapped under `\begin{papers}`. For HTML output we simply
32  // discard them inside `{papers}` so the body content survives.
33  // Use a flag toggled at env begin/end; redefine the kernel CS
34  // to a conditional no-op only while `\ifc@lpapers` is true.
35  RawTeX!(r"\newif\ifc@lpapers");
36  DefMacro!(
37    T_CS!("\\begin{papers}"),
38    None,
39    "\\c@lpaperstrue\\let\\c@l@orig@docclass\\documentclass\
40     \\let\\c@l@orig@usepackage\\usepackage\
41     \\def\\documentclass{\\c@lpapers@gobble@docclass}%
42     \\def\\usepackage{\\c@lpapers@gobble@usepackage}"
43  );
44  DefMacro!(
45    T_CS!("\\end{papers}"),
46    None,
47    "\\let\\documentclass\\c@l@orig@docclass\
48     \\let\\usepackage\\c@l@orig@usepackage\
49     \\c@lpapersfalse"
50  );
51  // Helpers gobble optional + mandatory args.
52  DefMacro!("\\c@lpapers@gobble@docclass [] {}", None);
53  DefMacro!("\\c@lpapers@gobble@usepackage [] {}", None);
54
55  // combine.cls L942: `\newenvironment{papers}[1][\cleardoublepage]{...}{...}`
56  // The body just contains `\subimport`'d sub-documents (or their
57  // verbatim content). Render transparently. (Definitions added
58  // below to also gobble inner \documentclass / \usepackage.)
59  // \pretitle / \posttitle / \preauthor / \postauthor / \predate /
60  // \postdate — combine.cls hooks for sub-document title pages.
61  DefMacro!("\\pretitle{}", None);
62  DefMacro!("\\posttitle{}", None);
63  DefMacro!("\\preauthor{}", None);
64  DefMacro!("\\postauthor{}", None);
65  DefMacro!("\\predate{}", None);
66  DefMacro!("\\postdate{}", None);
67  // `\c@l...` family — combine.cls internal hooks. No-ops since we
68  // don't combine multiple sub-documents.
69  DefMacro!("\\c@lbusepackage[]{}", None);
70  DefMacro!("\\c@lbLoadClass", None);
71  DefMacro!("\\c@lbdocumentclass[]{}", None);
72});
