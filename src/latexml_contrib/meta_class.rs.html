1//! Shared helper: the frontmatter interface common to the "addtolist ML
2//! meta-class" family (`fairmeta` / `selfevolagent` / `openmoss` and future
3//! siblings) — pre-print classes whose `\author`/`\affiliation`/`\contribution`/
4//! `\correspondence`/`\abstract`/`\beginappendix` commands are defined on an
5//! `\addtolist` accumulator in the class BODY. An unknown `.cls` body is not
6//! raw-loaded (OmniBus extracts dependencies only), so every one of those is
7//! `Error:undefined` without a binding. Mirrors the shared-helper pattern of
8//! [`crate::discard_env`].
9use latexml_package::prelude::*;
10
11/// Install the identical, order-independent frontmatter routing shared by every
12/// sibling class: route `\author` to the creator sink and the rest to
13/// `\@add@frontmatter`/`\lx@add@abstract` so they reach `<ltx:document>`
14/// frontmatter. Call once from inside the class's `LoadDefinitions!` block.
15///
16/// The per-class parts stay in each `*_cls.rs`: the (order-sensitive) dependency
17/// list, the colour palette, and the class-specific labeled field — `\metadata`
18/// vs `\checkdata`, and whether its label routes to a `role` attribute
19/// (attribute-safe labels, `fairmeta`) or to note content (arbitrary-markup
20/// labels, `selfevolagent`/`openmoss`).
21pub fn install_meta_class_frontmatter() -> Result<()> {
22  // Accumulator lists → no-ops; the `\@add@frontmatter` sink accumulates.
23  def_macro_noop("\\authorlist")?;
24  def_macro_noop("\\affiliationlist")?;
25  def_macro_noop("\\contributionlist")?;
26  // `\author[mark]{name}` → the creator sink (NOT `\author` — that re-matches
27  // this macro and recurses); the leading affiliation mark #1 is dropped.
28  DefMacro!("\\author[]{}", "\\lx@add@author{#2}");
29  DefMacro!(
30    "\\affiliation[]{}",
31    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}"
32  );
33  DefMacro!(
34    "\\contribution[]{}",
35    "\\@add@frontmatter{ltx:note}[role=contribution]{#2}"
36  );
37  DefMacro!(
38    "\\correspondence{}",
39    "\\@add@frontmatter{ltx:note}[role=correspondence]{#1}"
40  );
41  DefMacro!("\\abstract{}", "\\lx@add@abstract{#1}");
42  DefMacro!("\\email{}", "\\href{mailto:#1}{\\texttt{#1}}");
43  DefMacro!("\\beginappendix", "\\appendix");
44  Ok(())
45}
