1//! Shared helper: discard an environment body by reading tokens up to
2//! matching `\end{kind}`. Mirrors the Perl `discard_env_body` closures
3//! used in ar5iv-bindings/nicematrix / forest / diagrams / pb-diagram
4//! stub packages.
5//!
6//! Perl reference (identical across those stubs):
7//! ```perl
8//! sub discard_env_body {
9//!   my ($stomach, $kind) = @_;
10//!   my $gullet = $stomach->getGullet;
11//!   $stomach->bgroup;
12//!   if (!$reported{$kind}) {
13//!     $reported{$kind} = 1;
14//!     Error('undefined', "{$kind}", $gullet,
15//!       "$kind has no support, this is a stub binding."); }
16//!   while (my $_ = $gullet->readUntil(T_CS('\end'))) {
17//!     my $_drop_open = $gullet->readToken;
18//!     my $env        = $gullet->readBalanced;
19//!     last if ToString($env) eq $kind; }
20//!   $stomach->egroup;
21//!   return; }
22//! ```
23
24use std::cell::RefCell;
25
26use latexml_package::prelude::*;
27use rustc_hash::FxHashSet as HashSet;
28
29thread_local! {
30  /// One error per kind per conversion run, matching Perl's
31  /// `our $reported{$kind}` cache inside each stub package.
32  static REPORTED: RefCell<HashSet<String>> = RefCell::new(HashSet::default());
33}
34
35/// Read and discard tokens up to and including a matching `\end{kind}`.
36/// Emits a one-time `Error("undefined", "{kind}", ...)` on the first
37/// invocation per `kind`.
38pub fn discard_env_body(kind: &str, source: &str) -> Result<()> {
39  bgroup();
40  let first_time = REPORTED.with(|cell| {
41    let mut set = cell.borrow_mut();
42    if set.contains(kind) {
43      false
44    } else {
45      set.insert(kind.to_string());
46      true
47    }
48  });
49  if first_time {
50    let obj = format!("{{{}}}", kind);
51    let msg = format!(
52      "{} has no support in {}, this is a stub binding.",
53      kind, source
54    );
55    Error!("undefined", &obj, msg);
56  }
57  let end_delim = Tokens!(T_CS!("\\end"));
58  loop {
59    let _upto_end = read_until(&end_delim)?;
60    let _drop_open = read_token()?;
61    // require_open=false because `_drop_open` just consumed the `{` —
62    // read_balanced should read the inside, not a second `{`. Mirrors
63    // Perl's argless `$gullet->readBalanced` which assumes the `{` is
64    // already open. Driver: 2402.09676 + nicematrix stub cascaded
65    // "Expected opening '{'" because of the spurious require_open.
66    let env = read_balanced(ExpansionLevel::Off, false, false)?;
67    if env.to_string() == kind {
68      break;
69    }
70  }
71  egroup()?;
72  Ok(())
73}
