1//! Stub for Elsevier cas-dc.cls / cas-sc.cls (CAS journals double-column).
2//!
3//! The cas-* classes load cas-common.sty which uses xparse/expl3
4//! NewDocumentCommand to define many frontmatter helpers. Our raw load
5//! may not invoke them; provide gobble stubs for the most common
6//! frontmatter macros.
7use latexml_package::prelude::*;
8
9LoadDefinitions!({
10  LoadClass!("OmniBus");
11  RequirePackage!("amsmath");
12  RequirePackage!("amsthm");
13  // amssymb supplies \lesssim, \gtrsim, \nleq, \ngeq, \square, \blacksquare
14  // and other binary-relation/blackboard-bold symbols that CAS authors
15  // routinely use without explicit \usepackage{amssymb}. Witness
16  // 2312.12523 (\lesssim undefined).
17  RequirePackage!("amssymb");
18  RequirePackage!("xcolor");
19  RequirePackage!("hyperref");
20  // cas-dc.cls L63: \RequirePackage{booktabs,makecell,multirow,array,colortbl,dcolumn,stfloats}.
21  RequirePackage!("booktabs");
22  RequirePackage!("multirow");
23  RequirePackage!("array");
24  RequirePackage!("colortbl");
25  RequirePackage!("makecell");
26  // cas-dc.cls L: `\RequirePackage{etoolbox,balance}` — pulls in the
27  // balance package which defines `\balance` (column-balancing in
28  // 2-col docs). Without it, papers using `\balance` near the end
29  // trip Error:undefined. Witness 2303.04712.
30  RequirePackage!("balance");
31  // cas-dc.cls (Elsevier CAS) aliases \newdefinition and \newproof to \newtheorem
32  // (elsarticle convention), used to declare definition/proof-like theorem envs.
33  // Were undefined (Perl defines them). Witness 2306.04212.
34  Let!("\\newdefinition", "\\newtheorem");
35  Let!("\\newproof", "\\newtheorem");
36
37  // cas-common.sty dynamically defines `\tblwidth` via
38  // `\csgdef{tblwidth}{\dim_use:N \l_tbl_width_dim}` only inside its
39  // own table-float wrapper. Authors use `\tblwidth` directly as the
40  // width argument of `\begin{tabular*}{\tblwidth}{...}` even outside
41  // that wrapper. Provide a \linewidth fallback so the tabular still
42  // renders with reasonable width. Witness 2209.06932 (cas-dc table
43  // cascade — 136 errors).
44  DefMacro!("\\tblwidth", "\\linewidth");
45  // cas-common.sty L2070-2072: \newcolumntype{L|R|C}{...}. Authors use
46  // L/R/C column types in `\begin{tabular*}{...}{@{} LLLLLL @{}}`.
47  // Without these, the unrecognized column letters trigger one
48  // "Extra alignment tab '&'" per & cell. Define matching short-hand
49  // raggedright/etc. column types.
50  RawTeX!(
51    r"\newcolumntype{L}{@{\extracolsep{\fill}}l}\newcolumntype{R}{@{\extracolsep{\fill}}r}\newcolumntype{C}{@{\extracolsep{\fill}}c}"
52  );
53
54  // cas-common frontmatter — gobble cleanly.
55  // Title-note text — author prose. Preserve as ltx:note frontmatter.
56  DefMacro!(
57    "\\tnotetext[]{}",
58    "\\@add@frontmatter{ltx:note}[role=titlenote]{#2}"
59  );
60  def_macro_noop("\\tnotemark[]")?; // mark only, no body
61  DefMacro!(
62    "\\tnoteref[]{}",
63    "\\@add@frontmatter{ltx:note}[role=titlenote-ref]{#2}"
64  );
65  def_macro_noop("\\fnmark[]")?;
66  DefMacro!(
67    "\\fnref[]{}",
68    "\\@add@frontmatter{ltx:note}[role=footnote-ref]{#2}"
69  );
70  // Footnote / author-note text — preserve as ltx:note rather than
71  // gobble (content-preserving). `\fntext` / `\nonumnote` carry the
72  // actual note prose, `\cortext` is the corresponding-author byline.
73  DefMacro!(
74    "\\fntext[]{}",
75    "\\@add@frontmatter{ltx:note}[role=footnote]{#2}"
76  );
77  DefMacro!(
78    "\\nonumnote{}",
79    "\\@add@frontmatter{ltx:note}[role=note]{#1}"
80  );
81  DefMacro!(
82    "\\nonumtnotetext{}",
83    "\\@add@frontmatter{ltx:note}[role=note]{#1}"
84  );
85  DefMacro!(
86    "\\cortext[]{}",
87    "\\@add@frontmatter{ltx:note}[role=corresponding]{#2}"
88  );
89  def_macro_noop("\\cormark[]")?; // mark only, no body
90  def_macro_noop("\\corref[]")?;
91  // \affiliation[id]{text} — affiliation string author typed.
92  DefMacro!(
93    "\\affiliation[]{}",
94    "\\@add@to@frontmatter{ltx:creator}{\\@@@affiliation{#2}}"
95  );
96  // \ead[type]{address} — author email/url, preserve as contact.
97  DefMacro!(
98    "\\ead[]{}",
99    "\\@add@to@frontmatter{ltx:creator}{\\@@@email{#1}{#2}}"
100  );
101  // ltx:contact stubs (mirror elsart_support_core@@@affiliation form)
102  DefConstructor!(
103    "\\@@@affiliation{}",
104    "^ <ltx:contact role='affiliation'>#1</ltx:contact>"
105  );
106  DefConstructor!(
107    "\\@@@email{}{}",
108    "^ <ltx:contact role='#1'>#2</ltx:contact>"
109  );
110
111  // \sep — author/affil separator that cas-common defines.
112  DefMacro!("\\sep", ",");
113
114  // cas-common credit-tagging macros (CRediT taxonomy). \credit{role}
115  // attaches an author contribution (Conceptualization, Investigation,
116  // etc.); \printcredits emits the credit list. Round-34 surpass-Perl:
117  // preserve the role text as a frontmatter note so the CRediT
118  // taxonomy is retained for downstream JATS conversion. Witness
119  // 2405.20972.
120  DefMacro!(
121    "\\credit{}",
122    "\\@add@frontmatter{ltx:note}[role=credit]{#1}"
123  );
124  def_macro_noop("\\printcredits")?;
125
126  // Elsevier highlights / biography environments (cas-common.sty).
127  // `{highlights}` collects bullet points for the dedicated highlights
128  // section; `{bio}` / `\bio` defines an author biography. Render as
129  // ltx:note (frontmatter) so author content is preserved.
130  // Witness 2503.16816, 2502.18516 (cas-dc).
131  // Use internal_vertical mode so the body can contain paragraphs,
132  // lists, and other block-level content. Without the explicit mode,
133  // the env infers restricted_horizontal and trips
134  // `Error:unexpected:\endbio Attempt to end mode restricted_horizontal
135  // in internal_vertical` when the bio contains \par-separated prose.
136  // Witness 2503.16816.
137  DefEnvironment!("{highlights}",
138    "<ltx:note role='highlights'>#body</ltx:note>",
139    mode => "internal_vertical");
140  DefEnvironment!("{bio}{}",
141    "<ltx:note role='biography' name='#1'>#body</ltx:note>",
142    mode => "internal_vertical");
143  // \newproof{env}{display-name} — cas-common's environment factory
144  // for proof-like environments. Define a basic environment that
145  // wraps content in ltx:proof.
146  DefMacro!(
147    "\\newproof{}{}",
148    "\\newenvironment{#1}{\\par\\noindent\\textbf{#2.}\\hspace{0.5em}}{\\par}"
149  );
150});
