1//! czipreprint.cls — Chan Zuckerberg Initiative preprint class (author-bundled;
2//! not raw-loaded). It uses acmart-style per-author frontmatter — `\author[n]{}`
3//! / `\author*[n]{}` (starred = corresponding), followed by `\affiliation`,
4//! `\orcid`, `\email` — which OmniBus leaves undefined, so they leak as literal
5//! text (witness 2508.00826 → `\affiliation \orcid \thanks …`). Bind them to the
6//! same structured creator/contact markup acmart uses. `\author[n]{}` already
7//! works via OmniBus; add the starred (corresponding-author) form and the
8//! contact macros.
9use latexml_package::prelude::*;
10
11LoadDefinitions!({
12  LoadClass!("OmniBus");
13
14  // czipreprint's `\author[n]{Name}` carries a NUMERIC affiliation index in the
15  // optional arg (not a running-head short name), and `\author*[n]{Name}` marks
16  // the corresponding author. Peek for the star via `\@ifstar` — writing
17  // `\author*[]{}` in a signature would instead redefine `\author` to REQUIRE a
18  // literal `*`, breaking the plain `\author[1]{}` (leaked `]Name`). Both forms
19  // map to a creator; the numeric index is dropped (the `\affiliation` blocks
20  // below carry the institution text).
21  DefMacro!("\\author", "\\@ifstar\\lxczi@author\\lxczi@author");
22  DefMacro!("\\lxczi@author[]{}", "\\lx@add@creator[role=author]{#2}");
23  // czipreprint `\newcommand\affiliation[2][0]`: `\affiliation[n]{text}` — an
24  // optional numeric index + the institution text. Drop the index (the number
25  // links to the author's `[n]`; the text is what we render).
26  DefMacro!(
27    "\\affiliation[]{}",
28    "\\lx@add@contact[role=affiliation,annotate=new]{#2}"
29  );
30  DefMacro!(
31    "\\orcid{}",
32    "\\lx@add@contact[role=orcid, name={OrcID: }]{#1}"
33  );
34  DefMacro!(
35    "\\email [] Semiverbatim",
36    "\\lx@add@contact[role=email,name={email: }]{#2}"
37  );
38});
