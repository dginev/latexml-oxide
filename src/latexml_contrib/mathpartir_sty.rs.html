1//! mathpartir.sty — math paragraphs for typesetting inference rules
2//! by Didier Remy.
3//!
4//! Provides `mathpar`/`mathparpagebreakable` environments and the
5//! `\inferrule[label]{premises}{conclusion}` command for
6//! horizontally-laid-out inference rules.
7//!
8//! Perl LaTeXML has no mathpartir binding. With its default
9//! `INCLUDE_STYLES=false` Perl skips the raw .sty file, emits a
10//! single "missing binding" warning, and lets the user document
11//! continue — any `\inferrule`/`mathpar` use surfaces as one
12//! "undefined" error per use site.
13//!
14//! Our `INCLUDE_STYLES=true` default raw-loads mathpartir.sty,
15//! which hits `\halign \bgroup \hfil $##$\hfil\cr` patterns in
16//! `\mathvbox@` — the `##` PARAM tokens cascade as `# should
17//! never reach Stomach` errors and the `\halign` outside an
18//! alignment template trips `Missing \halign box` (witness
19//! arXiv:1310.8644: amsart + mathpartir → 100+ errors + fatal,
20//! while Perl converts with the single missing-binding warning).
21//!
22//! Stub the public API as faithful enough wrappers — inference
23//! rules render as a vertical `numerator / line / denominator`
24//! frac-style — so the raw file is skipped and the document
25//! continues. Lost fidelity: precise label placement, multi-rule
26//! `\and` separation. Gained: error-free conversion + readable
27//! inference rules.
28use latexml_package::prelude::*;
29
30LoadDefinitions!({
31  Warn!(
32    "missing_file",
33    "mathpartir.sty",
34    "mathpartir.sty is minimally stubbed — \\inferrule renders as a frac; mathpar / mathparpagebreakable wrap their body in a display equation."
35  );
36
37  // \inferrule[label]{premises}{conclusion}
38  //   → \ensuremath{\frac{premises}{conclusion}} with the label
39  //     appended parenthetically when present.
40  // The `\frac` is math-mode-only, but mathpartir's `\inferrule` is
41  // routinely used in *text* mode (e.g. bare inside `\begin{tabular}{c}`
42  // — witness arXiv:1404.0085 §3 Fig.3, the π-calculus reduction rules).
43  // Emitting a bare `\frac` there drops the math `XMApp` straight into
44  // the `<ltx:td>` with no `<ltx:Math>` wrapper → "ltx:XMApp isn't
45  // allowed in <ltx:td>" (Perl, which raw-loads the real mathpartir,
46  // wraps it). `\ensuremath` enters math mode only when not already in
47  // it, so this is correct in BOTH text-mode (tabular) and math-mode
48  // (`mathpar` = display equation) use sites.
49  // Use OptionalMatch:* to consume the starred form `\inferrule*`
50  // (mathpartir's \mpr@inferstar branch). Optional `[label]`,
51  // then two required {} args.
52  DefMacro!("\\inferrule OptionalMatch:* [] {} {}", sub[(_star, label, prem, conc)] {
53    let mut out: Vec<Token> = Vec::new();
54    out.push(T_CS!("\\ensuremath"));
55    out.push(T_BEGIN!());
56    out.push(T_CS!("\\frac"));
57    out.push(T_BEGIN!());
58    // mathpartir separates premises with `\\`, laying them out SIDE BY SIDE
59    // above the rule line. We render via `\frac`, so the `\\` must become a
60    // horizontal separator (`\quad`) — emitting it raw inside `\frac{…}` is a
61    // hard error in BOTH Perl and Rust ("\\ in \frac"), and inside a display
62    // alignment (`gather*`/`align`) the leaked `\\` starts a spurious row,
63    // desyncing math mode → `\lx@end@inline@math … end mode math in math`
64    // cascade (witness 1702.02972: llncs + `\begin{gather*}\inferrule{A \\ B}{C}`,
65    // Perl 0, raw-mathpartir lays the premises out without leaking `\\`).
66    // Skip a `\\[dim]` optional spacing arg so it doesn't render literally.
67    let prem_toks = prem.unlist();
68    let mut i = 0;
69    while i < prem_toks.len() {
70      if prem_toks[i] == T_CS!("\\\\") {
71        out.push(T_CS!("\\quad"));
72        i += 1;
73        // Skip an optional `[dim]` spacing argument following `\\`.
74        if i < prem_toks.len() && prem_toks[i].get_catcode() == Catcode::OTHER
75          && prem_toks[i].to_string() == "["
76        {
77          while i < prem_toks.len()
78            && !(prem_toks[i].get_catcode() == Catcode::OTHER && prem_toks[i].to_string() == "]")
79          {
80            i += 1;
81          }
82          if i < prem_toks.len() {
83            i += 1; // consume the `]`
84          }
85        }
86      } else {
87        out.push(prem_toks[i]);
88        i += 1;
89      }
90    }
91    out.push(T_END!());
92    out.push(T_BEGIN!());
93    out.extend(conc.unlist());
94    out.push(T_END!());
95    out.push(T_END!());
96    if let Some(lab) = label
97      && !lab.is_empty() {
98        out.push(T_CS!("\\quad"));
99        out.push(T_CS!("\\textsc"));
100        out.push(T_BEGIN!());
101        out.extend(lab.unlist());
102        out.push(T_END!());
103      }
104    Ok(Tokens::new(out))
105  });
106  // `\infer` is a deprecated alias — install it ONLY when `\infer` is not
107  // already defined. The real mathpartir.sty does not define `\infer` at all
108  // (it is the `proof` package's command, whose premise syntax splits on `&`:
109  // `\infer{concl}{prem1 & prem2}`). An UNCONDITIONAL `\let\infer\inferrule`
110  // clobbered proof's `\infer` whenever `proof` was loaded BEFORE `mathpartir`
111  // — then `\infer{C}{A & B}` ran `\inferrule` (premises split on `\\`, not `&`)
112  // and every premise `&` errored "Stray alignment &" during Building (witness
113  // 1801.08114: `\usepackage{proof}` then `\usepackage{mathpartir}` → 15 errors,
114  // Perl 0). Guarding with `\@ifundefined` preserves proof's `\infer` and still
115  // offers the alias to mathpartir-only documents.
116  RawTeX!(r"\@ifundefined{infer}{\let\infer\inferrule}{}");
117
118  // `mathpar` environment: wrap body in a display equation.
119  // Optional `[keys]` consumed and discarded — we don't honour
120  // mathpartir's lineskip / column-width keys, those are
121  // typesetting hints with no HTML equivalent.
122  DefMacro!("\\mathpar []", "\\begin{equation*}");
123  DefMacro!("\\endmathpar", "\\end{equation*}");
124  // mathparpagebreakable is identical for our rendering purposes —
125  // it differs from `mathpar` only in page-break behaviour, which
126  // is irrelevant in HTML.
127  DefMacro!("\\mathparpagebreakable []", "\\begin{equation*}");
128  DefMacro!("\\endmathparpagebreakable", "\\end{equation*}");
129
130  // Configuration setters: silently consume their keyval argument.
131  DefMacro!("\\mprset {}", "");
132  DefMacro!("\\MathparLineskip", "");
133  DefMacro!("\\MathparBindings", "");
134  // Label-style hooks: pass-through `#1` so the label still renders.
135  DefMacro!("\\TirName {}", "\\textsc{#1}");
136  DefMacro!("\\LeftTirName {}", "\\textsc{#1}");
137  DefMacro!("\\RightTirName {}", "\\textsc{#1}");
138  DefMacro!("\\TirNameStyle {}", "\\textsc{#1}");
139  DefMacro!("\\LeftTirNameStyle {}", "\\textsc{#1}");
140  DefMacro!("\\RightTirNameStyle {}", "\\textsc{#1}");
141});
