1//! ieeetj.cls — IEEE Transactions journal template (author-bundled, ~4900 lines;
2//! not raw-loaded). OmniBus renders the authors (`\author`, `\affil` via
3//! inst_support) fine, but the class's late-defined journal-metadata frontmatter
4//! macros are undefined and leak as literal text (witness 2405.01673, 2603.04284
5//! → `\corresp \authornote \receiveddate …`). Bind just those frontmatter
6//! macros; everything else falls through to OmniBus.
7use latexml_package::prelude::*;
8
9LoadDefinitions!({
10  LoadClass!("OmniBus");
11
12  // Corresponding-author byline and author/funding note — preserve as notes.
13  DefMacro!(
14    "\\corresp{}",
15    "\\lx@add@frontmatter{ltx:note}[role=corresponding]{#1}"
16  );
17  DefMacro!(
18    "\\authornote{}",
19    "\\lx@add@frontmatter{ltx:note}[role=note]{#1}"
20  );
21  // `\affil{…}` — a numbered affiliation block. inst_support already defines
22  // `\affil`; keep that. (Listed here only for the record.)
23
24  // Editorial dates and identifiers. IEEE templates ship these as unfilled
25  // "XX Month, XXXX" / "XXXX.2022.1234567" placeholders; rendering them would
26  // surface template noise, so gobble them (they leaked as raw text before).
27  def_macro_noop("\\receiveddate{}")?;
28  def_macro_noop("\\reviseddate{}")?;
29  def_macro_noop("\\accepteddate{}")?;
30  def_macro_noop("\\publisheddate{}")?;
31  def_macro_noop("\\currentdate{}")?;
32  def_macro_noop("\\doiinfo{}")?;
33  def_macro_noop("\\history{}")?;
34  def_macro_noop("\\articletype{}")?;
35  // Author-supplied funding acknowledgement — preserve rather than gobble.
36  DefMacro!(
37    "\\fundingtext{}",
38    "\\lx@add@frontmatter{ltx:note}[role=funding]{#1}"
39  );
40});
