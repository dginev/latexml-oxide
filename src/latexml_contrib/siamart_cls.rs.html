1//! Stub binding for SIAM siamart-family classes (siamart, siamart220329, ...).
2//!
3//! Activated via prefix-match: any class name starting with "siamart" routes
4//! here. Defines the high-level macros papers use (\newsiamthm, \newsiamremark,
5//! \headers, \dedicatory) on top of OmniBus's article-like behaviour.
6use latexml_package::prelude::*;
7
8LoadDefinitions!({
9  LoadClass!("OmniBus");
10  // siamart220329.cls L58: \RequirePackage[leqno]{amsmath}.
11  RequirePackage!("amsmath");
12  RequirePackage!("amsthm");
13  // Many siamart papers pre-define colors in their macros.tex before
14  // their own `\usepackage{xcolor}`. Defensive xcolor load matches Perl
15  // behaviour. Witness 2405.17955 (EPCID).
16  // Eager xcolor preload removed for Perl parity: it makes a later document
17  // xcolor[table] load a no-op, so colortbl/array never load and array m{}/b{}
18  // columns break (Unrecognized tabular template -> Extra alignment tab). The
19  // document loads xcolor itself; color/definecolor stay via hyperref->color.
20  // See ifacconf_cls.rs and SYNC_STATUS (eager-xcolor cluster).
21  // siamart220329 L1361: \RequirePackage{algorithm}.
22  RequirePackage!("algorithm");
23  RequirePackage!("url");
24  // siamart220329 L1285: \RequirePackage{hyperref}[6.83] (unconditional).
25  // Mirror so papers using \hidelinks/\href/\hypersetup don't error.
26  // hyperref MUST come before cleveref (cleveref errors out otherwise).
27  // Witness 2407.00765 (siamart220329 with `[hidelinks,…]` class option).
28  RequirePackage!("hyperref");
29  // siamart220329 L1371: \RequirePackage[capitalize,nameinlink]{cleveref}.
30  // Loaded AFTER hyperref to satisfy cleveref's ordering check.
31  // Witness 2501.11060 (Error:latex:cleveref must be loaded after hyperref!).
32  RequirePackage!("cleveref");
33  // siamonline220329 L1676: \RequirePackage[mathlines]{lineno}.
34  RequirePackage!("lineno");
35  // ifpdf is auto-loaded inside epstopdf; our binding triggers
36  // \ifpdf usage during epstopdf raw-load, so preload it here.
37  RequirePackage!("ifpdf");
38  RawTeX!(
39    r"\newtheorem{theorem}{Theorem}
40\newtheorem{lemma}[theorem]{Lemma}
41\newtheorem{corollary}[theorem]{Corollary}
42\newtheorem{proposition}[theorem]{Proposition}
43\newtheorem{definition}[theorem]{Definition}"
44  );
45
46  // \newsiamthm{name}{title}, \newsiamremark{name}{title}: siamart220329 L1452,
47  // L1469. Both reduce to `\newtheorem{name}[theorem]{title}`; the theoremstyle
48  // tweaks are visual and don't affect XML.
49  DefMacro!("\\newsiamthm{}{}", r"\newtheorem{#1}[theorem]{#2}");
50  DefMacro!("\\newsiamremark{}{}", r"\newtheorem{#1}[theorem]{#2}");
51
52  // siamart frontmatter primitives. Round-34 surpass-Perl: preserve
53  // \dedicatory, \fundingsource, and \funding as author-typed
54  // frontmatter notes (the funding text is often a real funding
55  // statement worth keeping). \headers{left}{right} → running-head
56  // text, preserve as ltx:note.
57  DefMacro!(
58    "\\headers{}{}",
59    "\\@add@frontmatter{ltx:note}[role=runningheads]{#1 / #2}"
60  );
61  DefMacro!(
62    "\\dedicatory{}",
63    "\\@add@frontmatter{ltx:note}[role=dedicatory]{#1}"
64  );
65  DefMacro!(
66    "\\fundingsource{}",
67    "\\@add@frontmatter{ltx:note}[role=funding-source]{#1}"
68  );
69  // siamart papers often \externaldocument supplement/article before
70  // loading xr — pre-stub. siamart loads xr-hyper (not xr), which
71  // supports `\externaldocument[prefix][nocite]{file}` (two optional
72  // args). Semiverbatim on the file arg neutralizes `_` so paper-
73  // bundled filenames like `ex_supplement` don't trip text-mode `_`
74  // errors. Witness 2402.12241.
75  def_macro_noop("\\externaldocument[][] Semiverbatim")?;
76  def_macro_noop("\\externalcitedocument[][] Semiverbatim")?;
77  // siamart220329 L1130: \funding{...} writes a marked line in the
78  // titlepage. Preserve as a frontmatter ltx:acknowledgements via
79  // \@add@frontmatter so the element lands at top-level no matter
80  // where the macro is invoked. Papers commonly nest \funding{...}
81  // inside \thanks{...} (which is rendered as ltx:note); an inline
82  // DefConstructor of ltx:acknowledgements there fires
83  // Error:malformed:ltx:acknowledgements isn't allowed in <ltx:note>.
84  // Witness 2311.08549.
85  DefMacro!(
86    "\\funding{}",
87    "\\@add@frontmatter{ltx:acknowledgements}[name=Funding]{#1}"
88  );
89  // {MSCcodes} env — siamart220329 L743 wraps content in an "@abssec"
90  // (frontmatter section). Mirror as keywords-like classification block.
91  DefEnvironment!(
92    "{MSCcodes}",
93    "<ltx:classification scheme='MSC'>#body</ltx:classification>"
94  );
95
96  // {AMS} env — siamart190516 L742 same pattern (AMS subject
97  // classification). Some older SIAM templates use {AMS} instead of
98  // {MSCcodes}. Witness 2306.11286, 2306.13351 (siamart190516).
99  DefEnvironment!(
100    "{AMS}",
101    "<ltx:classification scheme='AMS'>#body</ltx:classification>"
102  );
103
104  // {@abssec}{title} — siamonline190516.cls L810 (`\newenvironment{@abssec}[1]`):
105  // a titled frontmatter paragraph (`#1.` bold heading, then body). The
106  // public abstract/keywords/AMS/keyword envs wrap it, but papers also
107  // use `\begin{@abssec}{Author Information…}…\end{@abssec}` directly.
108  // `\begin{@abssec}` resolves via `\csname @abssec\endcsname`, so it
109  // works regardless of `@` catcode. siamonline190516.cls is reported
110  // missing-file in Perl (24 undefined-macro errors there); our siamart
111  // binding covers everything else, so defining this env makes the
112  // conversion clean. `{@doisec}` (L820) is the same shape with a
113  // trailing rule. Witness 2005.11911 (`\documentclass{siamonline190516}`).
114  // Use `inline-logical-block` (Misc.class) rather than `ltx:para`:
115  // papers place `\begin{@abssec}` inside frontmatter note contexts
116  // (e.g. an author/acknowledgements block already wrapped in
117  // `ltx:note`), where a block `ltx:para` is rejected
118  // (`malformed:ltx:para isn't allowed in <ltx:note>`). Misc.class is
119  // accepted in both inline and block positions.
120  DefEnvironment!(
121    "{@abssec}{}",
122    "<ltx:inline-logical-block class='ltx_abssec'><ltx:text font='bold'>#1. </ltx:text>#body</ltx:inline-logical-block>"
123  );
124  DefEnvironment!(
125    "{@doisec}{}",
126    "<ltx:inline-logical-block class='ltx_doisec'><ltx:text font='bold'>#1. </ltx:text>#body</ltx:inline-logical-block>"
127  );
128});
