1//! Stub for cvpr.sty / iccv.sty / iccvw.sty (computer vision conference style).
2//!
3//! cvpr.sty redefines \title to save the argument in \thetitle so it can
4//! be reused (typically by \maketitlesupplementary). Our raw load of
5//! cvpr.sty appears not to wire this up reliably; bind cvpr defensively
6//! to define \thetitle as a no-op default, plus stub the rebuttal/
7//! supplementary frontmatter.
8use latexml_package::prelude::*;
9
10
11LoadDefinitions!({
12  // Eager dependency loads — cvpr2025.sty L30-37 lists these as
13  // RequirePackage. The raw-load of cvpr*.sty doesn't always execute
14  // them via our system, so load them here so user macros like
15  // \toprule/\midrule/\bottomrule (booktabs), \includegraphics
16  // (graphicx) resolve. Witness 2503.24026 (cvpr2025, \toprule
17  // undefined).
18  //
19  // Pre-load xcolor WITH [dvipsnames, table] options: CVPR papers
20  // overwhelmingly use \color{Maroon}/{ForestGreen}/{MidnightBlue}
21  // (dvipsnames named palette) AND \cellcolor (colortbl, via the
22  // `table` option). If we pre-load xcolor WITHOUT options, the
23  // user's later `\usepackage[dvipsnames, table]{xcolor}` becomes
24  // an option-clash no-op and dvipsnam.def never gets loaded →
25  // 60+ undefined-color errors per CVPR-style paper. Witness 2305.13500.
26  RequirePackage!("xcolor", options => vec!["dvipsnames".to_string(), "table".to_string()]);
27  RequirePackage!("graphicx");
28  RequirePackage!("amsmath");
29  RequirePackage!("amssymb");
30  RequirePackage!("booktabs");
31  RequirePackage!("natbib");
32  RequirePackage!("etoolbox");
33  RequirePackage!("hyperref");
34  // caption.sty for \captionof — many CVPR templates use this for
35  // figure/table sub-captions. Witness 2503.24026 (cvpr2025).
36  RequirePackage!("caption");
37  // subcaption — many CVPR papers use \begin{subfigure}{...} for
38  // multi-panel figures without explicit \usepackage{subcaption}.
39  // cvpr2025.sty L30-37 doesn't \RequirePackage{subcaption}, but the
40  // template's example file does — so authors copy the example and
41  // omit the use-package. Witness 2312.03526.
42  RequirePackage!("subcaption");
43
44  // \thetitle: default-empty, gets overridden when user calls \title{...}.
45  def_macro_noop("\\thetitle")?;
46  def_macro_noop("\\maketitlesupplementary")?;
47
48  // cvpr.sty supplies these toggles via etoolbox — provide as fallback.
49  DefConditional!("\\ifcvprfinal");
50  DefConditional!("\\ifcvprrebuttal");
51  DefConditional!("\\ifcvprpagenumbers");
52
53  // CV-conference convention abbreviations (cvpr.sty L640-645, iccv.sty
54  // similar). Real cvpr.sty defines these via `\def\etal{\emph{et al}\onedot}`
55  // and friends. Our path-stripping fallback (`iccv_template/cvpr.sty` →
56  // `cvpr.sty.ltxml` registry → our binding) means the InputDefinitions
57  // raw-load below is asked to find plain `cvpr.sty` on disk — which is
58  // NOT there (the paper bundles it under `iccv_template/`). So \etal etc
59  // stay undefined unless we stub them here. Witness 2305.13460: was 5
60  // errors (\etal/\ie/\etc/\wrt/\vs), now 0.
61  DefMacro!("\\onedot",  "\\@onedot");
62  DefMacro!("\\@onedot", ".\\@");  // approximates the spacing trick
63  DefMacro!("\\etal",    "\\emph{et al}\\onedot");
64  DefMacro!("\\ie",      "\\emph{i.e}\\onedot");
65  DefMacro!("\\eg",      "\\emph{e.g}\\onedot");
66  DefMacro!("\\cf",      "\\emph{c.f}\\onedot");
67  DefMacro!("\\etc",     "\\emph{etc}\\onedot");
68  DefMacro!("\\vs",      "\\emph{vs}\\onedot");
69  DefMacro!("\\wrt",     "\\emph{w.r.t}\\onedot");
70  DefMacro!("\\dof",     "d.o.f\\onedot");
71
72  // Raw-load whichever cvpr*.sty the user actually `\usepackage`d. The
73  // binding registry routes cvpr / cvpr2023 / cvpr2024 / cvpr2025 ALL to
74  // this same load_definitions; each paper's bundled .sty file differs in
75  // small ways (e.g. cvpr2024_conference.sty configures cleveref via
76  // \AtEndPreamble — which our stub doesn't reproduce). Falling back to a
77  // hard-coded "cvpr" name means the paper-bundled `cvpr2024_conference.sty`
78  // is never raw-loaded — its `\AtEndPreamble{\usepackage{cleveref}...}` is
79  // skipped, leaving `\cref` undefined and triggering "_ in math mode"
80  // cascades on every `\cref{fig:foo_bar}`. Witness 2312.06720.
81  //
82  // \@currname expands to the actual package name being processed (e.g.
83  // "cvpr2024_conference"). Use it so the raw-load targets the user's
84  // own file. Fall back to "cvpr" if \@currname is somehow empty.
85  let currname = gullet::do_expand(T_CS!("\\@currname"))?.to_string();
86  let raw_name = if currname.is_empty() { "cvpr".to_string() } else { currname };
87  let _ = input_definitions(&raw_name, InputDefinitionOptions {
88    extension: Some(Cow::Borrowed("sty")),
89    noltxml: true,
90    noerror: true,
91    ..Default::default()
92  });
93});
