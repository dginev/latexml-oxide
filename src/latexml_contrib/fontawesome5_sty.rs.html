3/// Runtime equivalent of `DefMacro!("\\faXxx[]", "\\faIcon[#1]{kebab}")`
4/// for the 1373 trivial FontAwesome 5 icon convenience macros below.
5///
6/// Each `DefMacro!` literal-pair arm expands at compile time into
7/// `parse_prototype!` + `compile_expansion!` calls that *inline* the
8/// resulting `Tokens` constants into the generated function body —
9/// adding up to ~2.0 MiB of `.text` for 1373 nearly-identical calls
10/// (audit DEP-15). Routing through a single runtime helper drops the
11/// `load_definitions` function to ~100 KiB at the cost of one extra
12/// `format!` + `parse_prototype` + `tokenize_internal` per icon
13/// (~7 ms aggregate at engine bootstrap, never on the hot path).
14///
15/// Behaviour matches the macro arm: parameter list `[]` (one optional
16/// bracketed arg), expansion body packed via `Expandable::new`'s
17/// default pack_parameters path.
18fn def_fa5_icon(suffix: &str, kebab: &str) -> Result<()> {
19  let proto = format!("\\fa{suffix}[]");
20  let (cs, params) = parse_prototype(&proto, true)?;
21  let body = mouth::tokenize_internal(&format!("\\faIcon[#1]{{{kebab}}}"));
22  def_macro(cs, params, ExpansionBody::Tokens(body), None)?;
23  Ok(())
24}
