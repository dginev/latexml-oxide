1//! Stub for jmlr2e.sty / jmlr2e_preprint.sty (JMLR / PMLR author-block macros).
2//!
3//! These styles are used by Journal of Machine Learning Research and
4//! related Proceedings of Machine Learning Research papers. Define the
5//! \name / \addr / \email author-block font switches as identity so
6//! `\author{\name Foo \email a@b.c \\ \addr Place}` parses cleanly.
7use latexml_package::prelude::*;
8
9
10LoadDefinitions!({
11  RequirePackage!("natbib");
12  RequirePackage!("amsthm");
13  // jmlr2e.sty L57-63 pulls in epsfig, amssymb, graphicx, hyperref.
14  // Mirror that so user code that calls \hypersetup / \href / \blacklozenge
15  // (from amssymb) at preamble time doesn't error. Witness 2406.03260.
16  RequirePackage!("amssymb");
17  RequirePackage!("graphicx");
18  RequirePackage!("hyperref");
19
20  // Author-block font switches: no-op (identity).
21  def_macro_noop("\\name")?;
22  def_macro_noop("\\addr")?;
23  def_macro_noop("\\email")?;
24  DefMacro!("\\And", " ");
25
26  // Frontmatter / pagination ceremony. Round-34 surpass-Perl:
27  // preserve the author-typed text content (volume/page/etc. tuples
28  // and running-head author+title pair) as ltx:note rather than
29  // dropping silently.
30  DefMacro!("\\jmlrheading{}{}{}{}{}{}",
31    "\\@add@frontmatter{ltx:note}[role=heading]{#1 #2 #3 #4 #5 #6}");
32  DefMacro!("\\ShortHeadings{}{}",
33    "\\@add@frontmatter{ltx:note}[role=shortheadings]{#1 / #2}");
34  DefMacro!("\\firstpageno{}",
35    "\\@add@frontmatter{ltx:note}[role=firstpage]{#1}");
36  // \editor / \editors carry author-supplied editor names — preserve as
37  // ltx:note rather than dropping. JMLR papers cite the editor in the
38  // header; this keeps the credit visible.
39  DefMacro!("\\editor{}",
40    "\\@add@frontmatter{ltx:note}[role=editor]{#1}");
41  DefMacro!("\\editors{}",
42    "\\@add@frontmatter{ltx:note}[role=editor]{#1}");
43
44  // jmlr2e.sty L372: \acks{text} — acknowledgments section. Emit as
45  // structural ltx:acknowledgements with the funding-disclosure label
46  // (post-processors map to canonical role/styling).
47  DefConstructor!("\\acks{}",
48    "<ltx:acknowledgements name='acknowledgments-disclosure-of-funding'>#1</ltx:acknowledgements>");
49
50  // {keywords} env — frontmatter list, render as classification block.
51  DefEnvironment!(
52    "{keywords}",
53    "<ltx:classification scheme='keywords'>#body</ltx:classification>"
54  );
55
56  // Theorem-likes JMLR papers commonly use.
57  RawTeX!(
58    r"\newtheorem{theorem}{Theorem}
59\newtheorem{lemma}[theorem]{Lemma}
60\newtheorem{corollary}[theorem]{Corollary}
61\newtheorem{proposition}[theorem]{Proposition}
62\newtheorem{definition}[theorem]{Definition}
63\newtheorem{example}[theorem]{Example}
64\newtheorem{remark}[theorem]{Remark}"
65  );
66});
