1//! Stub for jmlr2e.sty / jmlr2e_preprint.sty (JMLR / PMLR author-block macros).
2//!
3//! These styles are used by Journal of Machine Learning Research and
4//! related Proceedings of Machine Learning Research papers. Define the
5//! \name / \addr / \email author-block font switches as identity so
6//! `\author{\name Foo \email a@b.c \\ \addr Place}` parses cleanly.
7use latexml_package::prelude::*;
8
9LoadDefinitions!({
10  RequirePackage!("natbib");
11  RequirePackage!("amsthm");
12  // jmlr2e.sty L57-63 pulls in epsfig, amssymb, graphicx, hyperref.
13  // Mirror that so user code that calls \hypersetup / \href / \blacklozenge
14  // (from amssymb) at preamble time doesn't error. Witness 2406.03260.
15  RequirePackage!("amssymb");
16  RequirePackage!("graphicx");
17  RequirePackage!("hyperref");
18
19  // Author-block font switches: no-op (identity).
20  def_macro_noop("\\name")?;
21  def_macro_noop("\\addr")?;
22  def_macro_noop("\\email")?;
23  DefMacro!("\\And", " ");
24
25  // jmlr2e.sty: `\newcommand{\BlackBox}{\rule{1.5ex}{1.5ex}}` — the
26  // end-of-proof QED box. Papers use `\hfill\BlackBox` or
27  // `\def\qed{\hfill\BlackBox}`. The stub previously omitted it, so a
28  // JMLR paper that loads jmlr2e and ends proofs with `\BlackBox` saw it
29  // undefined (witness 2001.10284: the only residual error after the
30  // stub handled \editor/{keywords}/\ShortHeadings/\firstpageno). Mirror
31  // the real definition.
32  DefMacro!("\\BlackBox", "\\rule{1.5ex}{1.5ex}");
33
34  // Frontmatter / pagination ceremony. Round-34 surpass-Perl:
35  // preserve the author-typed text content (volume/page/etc. tuples
36  // and running-head author+title pair) as ltx:note rather than
37  // dropping silently.
38  DefMacro!(
39    "\\jmlrheading{}{}{}{}{}{}",
40    "\\@add@frontmatter{ltx:note}[role=heading]{#1 #2 #3 #4 #5 #6}"
41  );
42  // jmlr2e.sty L256: `\def\ewrlheading#1#2#3#4{…\def\ps@jmlrtps{…}…}` — the
43  // EWRL-proceedings variant of `\jmlrheading`, setting the running-head page
44  // style from {volume}{year}{date/location}{authors}. Our binding intercepts
45  // jmlr2e.sty (so the raw def never runs); Perl raw-loads it and is clean.
46  // Preserve the proceedings metadata as an ltx:note, same as `\jmlrheading`
47  // (HTML drops the running head, so the raw def would lose this content).
48  // Witness 1802.03976 (`\ewrlheading{14}{2018}{October 2018, Lille, France}
49  // {…authors…}`).
50  DefMacro!(
51    "\\ewrlheading{}{}{}{}",
52    "\\@add@frontmatter{ltx:note}[role=heading]{#1 #2 #3 #4}"
53  );
54  DefMacro!(
55    "\\ShortHeadings{}{}",
56    "\\@add@frontmatter{ltx:note}[role=shortheadings]{#1 / #2}"
57  );
58  DefMacro!(
59    "\\firstpageno{}",
60    "\\@add@frontmatter{ltx:note}[role=firstpage]{#1}"
61  );
62  // \editor / \editors carry author-supplied editor names — preserve as
63  // ltx:note rather than dropping. JMLR papers cite the editor in the
64  // header; this keeps the credit visible.
65  DefMacro!(
66    "\\editor{}",
67    "\\@add@frontmatter{ltx:note}[role=editor]{#1}"
68  );
69  DefMacro!(
70    "\\editors{}",
71    "\\@add@frontmatter{ltx:note}[role=editor]{#1}"
72  );
73  // jmlr2e.sty L194: `\def\address#1{\gdef\@address{#1}}` — the author's
74  // institutional address, rendered in the title block (`\@name \\ \@address`,
75  // L187). Our binding intercepts jmlr2e.sty (so the raw def never runs), and
76  // OmniBus only autoloads `\address` for revtex/OmniBus contexts — so a
77  // jmlr2e paper using `\address{…}` left it undefined where Perl (raw-loads
78  // jmlr2e) is clean. Preserve the address content as an ltx:note, consistent
79  // with the other jmlr2e frontmatter macros above. Witness 1711.01660.
80  DefMacro!(
81    "\\address{}",
82    "\\@add@frontmatter{ltx:note}[role=address]{#1}"
83  );
84
85  // jmlr2e.sty L372: \acks{text} — acknowledgments section. Emit as
86  // structural ltx:acknowledgements with the funding-disclosure label
87  // (post-processors map to canonical role/styling).
88  DefConstructor!(
89    "\\acks{}",
90    "<ltx:acknowledgements name='acknowledgments-disclosure-of-funding'>#1</ltx:acknowledgements>"
91  );
92
93  // jmlr2e.sty L33-36: the camera-ready toggle.
94  //   \newif\ifnipsfinal \nipsfinalfalse
95  //   \def\nipsfinalcopy{\nipsfinaltrue}
96  // Papers uncomment `\nipsfinalcopy` for the camera-ready build (it gates
97  // submission-notice / page-number formatting in the raw .sty). Our binding
98  // intercepts jmlr2e.sty so the raw def never runs; Perl raw-loads it and is
99  // clean. Mirror the real definitions — the gated formatting is layout-only
100  // (moot in our XML paradigm), so the conditional just needs to exist.
101  // Witness 1604.06057 (`\nipsfinalcopy` in main.tex).
102  RawTeX!(r"\newif\ifnipsfinal\nipsfinalfalse\def\nipsfinalcopy{\nipsfinaltrue}");
103
104  // {keywords} env — frontmatter list, render as classification block.
105  DefEnvironment!(
106    "{keywords}",
107    "<ltx:classification scheme='keywords'>#body</ltx:classification>"
108  );
109
110  // Theorem-likes JMLR papers commonly use.
111  RawTeX!(
112    r"\newtheorem{theorem}{Theorem}
113\newtheorem{lemma}[theorem]{Lemma}
114\newtheorem{corollary}[theorem]{Corollary}
115\newtheorem{proposition}[theorem]{Proposition}
116\newtheorem{definition}[theorem]{Definition}
117\newtheorem{example}[theorem]{Example}
118\newtheorem{remark}[theorem]{Remark}"
119  );
120});
