1use latexml_package::prelude::*;

3LoadDefinitions!({
4  InputDefinitions!("harvmac", noltxml => true, extension => Some(Cow::Borrowed("tex")));
5  // ar5iv-bindings/bindings/harvmac.tex.ltxml L16-22: override \eqn{\label}{body}
6  // to (a) capture `(\secsym\the\meqno)` into the given label macro at invocation
7  // time (mirrors the `\xdef #1{(\secsym\the\meqno)}` in raw harvmac.tex L130)
8  // and (b) emit the body either directly (when IN_MATH) or wrapped in
9  // \lx@begin@display@math ... \lx@end@display@math — the Perl port's XML-mode
10  // substitute for the raw `$$#2\eqno#1\eqlabeL#1$$` plain-TeX wrapping.
11  DefMacro!("\\eqn{}{}", sub[args] {
12    let mut it = args.into_iter();
13    let label_toks: Tokens = it.next().unwrap().into();
14    let content: Tokens = it.next().unwrap().into();
15    let label_cs = label_toks.unlist().into_iter().next().unwrap_or_else(|| T_CS!("\\relax"));
16    let expanded = do_expand(Tokenize!(r"(\secsym\the\meqno)"))?;
17    def_macro(label_cs, None, expanded, None)?;
18    if lookup_bool_sym(pin!("IN_MATH")) {
19      Ok(content)
20    } else {
21      let mut out = vec![T_CS!("\\lx@begin@display@math")];
22      out.extend(content.unlist());
23      out.push(T_CS!("\\lx@end@display@math"));
24      Ok(Tokens::new(out))
25    }
26  });
27  // ar5iv-bindings/bindings/harvmac.tex.ltxml L30-34: override \listtoc and
28  // \writetoc to no-ops. The raw harvmac.tex version of \listtoc runs
29  // `\openin\ch@ckfile=\jobname.toc` + `\input\jobname.toc`, producing
30  // `Can't find TeX file <jobname>.toc` when the .toc hasn't been written
31  // (single-pass processing). arxiv 0711.4787 used to fail here.
32  // `locked=>1` matches Perl — prevents the raw harvmac.tex \def\listtoc /
33  // \def\writetoc definitions from overriding these.
34  DefMacro!("\\listtoc", "", locked => true);
35  DefMacro!("\\writetoc", sub[_args] {
36    Warn!(
37      "expected",
38      "TOC",
39      "harvmac.tex.ltxml has not yet implemented Table-of-contents"
40    );
41    Ok(Tokens!())
42  }, locked => true);
43});
