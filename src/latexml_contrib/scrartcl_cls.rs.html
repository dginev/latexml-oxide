1//! Stub for KOMA-Script `scrartcl` class.
2//!
3//! Article variant of scrbook (KOMA's typographically refined article).
4//! We don't replay KOMA's typographic engine — fall back to OmniBus and
5//! stub the most common configuration macros so author preamble doesn't
6//! trip undefined-macro errors. Same pattern as scrbook_cls.
7use latexml_package::prelude::*;
8
9LoadDefinitions!({
10  Warn!(
11    "missing_file",
12    "scrartcl.cls",
13    "scrartcl.cls is only minimally stubbed and will not be interpreted raw."
14  );
15  LoadClass!("OmniBus");
16  // Real scrartcl.cls pulls in the KOMA dependency chain (scrkbase, tocbasic,
17  // scrlayer-scrpage, bookmark, typearea, xpatch, scrlogo, auxhook), which
18  // transitively loads `iftex` — so `\ifpdf`/`\ifpdftex`/`\ifluatex`/… are
19  // defined for author preamble doing engine/driver detection
20  // (`\ifpdf \DeclareGraphicsExtensions{.eps,.pdf,…} \else …`). Perl ships no
21  // scrartcl binding and raw-loads the .cls, picking up iftex that way (its
22  // dependency-scan loads iftex.sty.ltxml). Our OmniBus stub intercepts the
23  // class, so without this `\ifpdf` is undefined where Perl is clean. Mirror
24  // the real class's dependency. Witness 1802.07175.
25  RequirePackage!("iftex");
26  // KOMA section-font hooks. scrartcl.cls L170-201 defines `\sectfont` (the
27  // heading font = `\normalcolor\maybesffamily\bfseries`) plus an empty
28  // `\size@<unit>` selector family. tocloft keys on these whenever it detects a
29  // KOMA class: `\cfttoctitlefont` becomes `\size@chapter\sectfont` (chapter
30  // classes) or `\size@section\sectfont` (article), tocloft.sty L169/L172 — so
31  // `\tableofcontents` under scrartcl+tocloft expands them. Our OmniBus stub
32  // intercepts scrartcl and omitted them, so they were undefined where Perl
33  // (raw-loads scrartcl) is clean. Mirror koma's definitions. `\maybesffamily`
34  // is empty (scrartcl's `\if@sfdefaults` is false by default → serif). We also
35  // provide the empty `\size@chapter`: OmniBus defines `\chapter`, so tocloft's
36  // `\if@cfthaschapter` branch fires and references the chapter form — the
37  // spacing it selects is layout-only and invisible in our output.
38  // Witness 1702.04336 (scrartcl + tocloft + \tableofcontents).
39  def_macro_noop("\\maybesffamily")?;
40  DefMacro!("\\sectfont", "\\normalcolor\\maybesffamily\\bfseries");
41  def_macro_noop("\\size@part")?;
42  def_macro_noop("\\size@partnumber")?;
43  def_macro_noop("\\size@chapter")?;
44  def_macro_noop("\\size@section")?;
45  def_macro_noop("\\size@subsection")?;
46  def_macro_noop("\\size@subsubsection")?;
47  def_macro_noop("\\size@paragraph")?;
48  def_macro_noop("\\size@subparagraph")?;
49  // KOMA configuration knobs — layout/typography only, no body content.
50  def_macro_noop("\\setkomafont{}{}")?;
51  def_macro_noop("\\addtokomafont{}{}")?;
52  def_macro_noop("\\setcapindent{}")?;
53  def_macro_noop("\\deffootnote[]{}{}{}")?;
54  def_macro_noop("\\deffootnotemark{}")?;
55  // \KOMAoptions{key=val,...} — runtime option setter, no body content.
56  def_macro_noop("\\KOMAoptions{}")?;
57  // KOMA page-style commands `\headmark` / `\pagemark`. Defined by
58  // KOMA classes (or scrlayer-scrpage) for use in custom page-style
59  // declarations as the current chapter/section mark / page number.
60  // Since we OmniBus-fall back for scrartcl rather than reading the
61  // .cls, these stay undefined and any author preamble using
62  // `\usepackage{scrlayer-scrpage}` or scrpage2 page-style hooks
63  // triggers `Error:undefined`. Stub as no-ops — running heads/feet
64  // are typesetting-only concerns; our HTML output doesn't render
65  // them. Witness: 11 papers in R-stages for each.
66  def_macro_noop("\\headmark")?;
67  def_macro_noop("\\pagemark")?;
68  // \subject{}, \dictum{}, \uppertitleback{}, \lowertitleback{},
69  // \publishers{} — KOMA frontmatter pieces that DO carry author
70  // content. Preserve as ltx:note frontmatter so the text reaches the
71  // XML (rather than silently gobbling).
72  DefMacro!(
73    "\\subject{}",
74    "\\@add@frontmatter{ltx:note}[role=subject]{#1}"
75  );
76  DefMacro!(
77    "\\dictum[]{}",
78    "\\@add@frontmatter{ltx:note}[role=dictum]{#2}"
79  );
80  DefMacro!(
81    "\\publishers{}",
82    "\\@add@frontmatter{ltx:note}[role=publishers]{#1}"
83  );
84});
