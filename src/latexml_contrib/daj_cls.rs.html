1//! Stub for daj.cls (Discrete Analysis journal class).
2//!
3//! daj.cls (2016, v1.03) extends tocbase to typeset "Discrete Analysis"
4//! articles. The raw cls defines `\dajAUTHORdetails`, `\dajEDITORdetails`,
5//! `\dajdetails`, and `{dajauthors}` env via `\setkeys` machinery. The raw
6//! cls loads but its body fails to define them at runtime (likely a
7//! `\setkeys`/key-handler interaction). Provide content-preserving stubs
8//! so the substantive author/editor information is captured. Witness
9//! 2305.10828, 2305.11062.
10use latexml_package::prelude::*;
11
12LoadDefinitions!({
13  LoadClass!("article");
14  RequirePackage!("amsmath");
15  RequirePackage!("amsthm");
16  RequirePackage!("amssymb");
17  RequirePackage!("amsfonts");
18  RequirePackage!("hyperref");
19
20  // \dajdetails / \dajAUTHORdetails / \dajEDITORdetails — argument
21  // is a comma-separated key=val list (publication details, ORCID,
22  // affiliation, etc.). Preserve as frontmatter note.
23  DefMacro!("\\dajdetails{}",
24    "\\@add@frontmatter{ltx:note}[role=daj-details]{#1}");
25  DefMacro!("\\dajAUTHORdetails{}",
26    "\\@add@frontmatter{ltx:note}[role=daj-author-details]{#1}");
27  DefMacro!("\\dajEDITORdetails{}",
28    "\\@add@frontmatter{ltx:note}[role=daj-editor-details]{#1}");
29
30  // {dajauthors} environment wraps repeated `{authorinfo}[id]{body}` blocks.
31  // The raw cls nests `\newenvironment{authorinfo}` inside `{dajauthors}`,
32  // which fails to register at runtime. Define both so the substantive
33  // author prose is preserved as semantically-named notes.
34  DefEnvironment!("{dajauthors}",
35    "<ltx:note role='daj-authors'>#body</ltx:note>",
36    mode => "internal_vertical");
37  DefEnvironment!("{authorinfo}[]",
38    "<ltx:note role='daj-authorinfo'>#body</ltx:note>",
39    mode => "internal_vertical");
40
41  // daj.cls L54-55: `\imageat` and `\imagedot` are graphic stand-ins
42  // for `@` and `.` in email addresses (anti-bot scraping). The raw
43  // cls defines them as `\tocat`/`\tocdot` (image macros). Render as
44  // plain characters — preserves email readability in HTML.
45  DefMacro!("\\imageat", "@");
46  DefMacro!("\\imagedot", ".");
47
48  // {frontmatter}[options] — wraps title/authors/abstract block. The
49  // paper at 2305.10828 uses `\begin{frontmatter}[classification=text]`
50  // and closes with `\end{frontmatter}` around all the meta. Treat as
51  // pass-through (children retain their own \title/\author handlers).
52  DefEnvironment!("{frontmatter}[]", "#body",
53    mode => "internal_vertical");
54});
