1//! Stub for daj.cls (Discrete Analysis journal class).
2//!
3//! daj.cls (2016, v1.03) extends tocbase to typeset "Discrete Analysis"
4//! articles. The raw cls defines `\dajAUTHORdetails`, `\dajEDITORdetails`,
5//! `\dajdetails`, and `{dajauthors}` env via `\setkeys` machinery. The raw
6//! cls loads but its body fails to define them at runtime (likely a
7//! `\setkeys`/key-handler interaction). Provide content-preserving stubs
8//! so the substantive author/editor information is captured. Witness
9//! 2305.10828, 2305.11062.
10use latexml_package::prelude::*;
11
12LoadDefinitions!({
13  LoadClass!("article");
14  // daj.cls does `\LoadClass{tocbase}`; we use article-base + a curated
15  // emulation of tocbase's `\RequirePackage` set (tocbase.cls), so daj papers
16  // get the packages tocbase would have loaded. Previously only ams*/hyperref
17  // were here, so tocbase's graphicx was missing → `\includegraphics` undefined
18  // (Perl raw-loads daj→tocbase→graphicx; witness 2208.01327, Discrete Analysis).
19  RequirePackage!("amsmath");
20  RequirePackage!("amsthm");
21  RequirePackage!("amssymb");
22  RequirePackage!("amsfonts");
23  RequirePackage!("hyperref");
24  // tocbase.cls core requires (the safe, commonly-used ones with bindings):
25  RequirePackage!("graphicx"); // \includegraphics — the witnessed gap
26  RequirePackage!("xcolor");
27  RequirePackage!("etoolbox");
28  RequirePackage!("eucal"); // math journal: \mathcal/\mathscr script font
29  RequirePackage!("mathptmx"); // tocbase's text/math font choice
30  RequirePackage!("microtype");
31  RequirePackage!("lastpage");
32  RequirePackage!("textcomp");
33
34  // \dajdetails / \dajAUTHORdetails / \dajEDITORdetails — argument
35  // is a comma-separated key=val list (publication details, ORCID,
36  // affiliation, etc.). Preserve as frontmatter note.
37  DefMacro!(
38    "\\dajdetails{}",
39    "\\@add@frontmatter{ltx:note}[role=daj-details]{#1}"
40  );
41  DefMacro!(
42    "\\dajAUTHORdetails{}",
43    "\\@add@frontmatter{ltx:note}[role=daj-author-details]{#1}"
44  );
45  DefMacro!(
46    "\\dajEDITORdetails{}",
47    "\\@add@frontmatter{ltx:note}[role=daj-editor-details]{#1}"
48  );
49
50  // {dajauthors} environment wraps repeated `{authorinfo}[id]{body}` blocks.
51  // The raw cls nests `\newenvironment{authorinfo}` inside `{dajauthors}`,
52  // which fails to register at runtime. Define both so the substantive
53  // author prose is preserved as semantically-named notes.
54  DefEnvironment!("{dajauthors}",
55    "<ltx:note role='daj-authors'>#body</ltx:note>",
56    mode => "internal_vertical");
57  DefEnvironment!("{authorinfo}[]",
58    "<ltx:note role='daj-authorinfo'>#body</ltx:note>",
59    mode => "internal_vertical");
60
61  // daj.cls L54-55: `\imageat` and `\imagedot` are graphic stand-ins
62  // for `@` and `.` in email addresses (anti-bot scraping). The raw
63  // cls defines them as `\tocat`/`\tocdot` (image macros). Render as
64  // plain characters — preserves email readability in HTML.
65  DefMacro!("\\imageat", "@");
66  DefMacro!("\\imagedot", ".");
67
68  // {frontmatter}[options] — wraps title/authors/abstract block. The
69  // paper at 2305.10828 uses `\begin{frontmatter}[classification=text]`
70  // and closes with `\end{frontmatter}` around all the meta. Treat as
71  // pass-through (children retain their own \title/\author handlers).
72  DefEnvironment!("{frontmatter}[]", "#body",
73    mode => "internal_vertical");
74});
