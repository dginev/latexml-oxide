1//! Stub for sagej.cls (SAGE journals).
2use latexml_package::prelude::*;
3
4LoadDefinitions!({
5  LoadClass!("OmniBus");
6  // sagej.cls L108-L109 load these math-symbol/font packages:
7  //   L108: \RequirePackage{latexsym,ifthen,rotating,calc,textcase,...}
8  //   L109: \RequirePackage{amsfonts,amssymb,amsbsy,amsmath,amsthm}
9  // Perl ships NO sagej binding → raw-loads the bundled sagej.cls → all of
10  // these load, so e.g. `\leqslant` (amssymb), `\mathbb` (amsfonts) and the
11  // latexsym glyphs are defined. Our OmniBus binding intercepts the class,
12  // so the `\RequirePackage` line never runs — replicate the AMS cluster
13  // here (in cls order: amsfonts before amssymb, which depends on it).
14  // Pure math-symbol/font packages — no layout side effects. Witness
15  // 1802.07225 (`\leqslant` in a paper that loads sagej but not amssymb).
16  RequirePackage!("latexsym");
17  RequirePackage!("amsfonts");
18  RequirePackage!("amssymb");
19  RequirePackage!("amsbsy");
20  RequirePackage!("amsmath");
21  RequirePackage!("amsthm");
22  // Do NOT eager-load xcolor (Perl ships no sagej binding → OmniBus, no
23  // preload). A preloaded xcolor makes a later `\usepackage[table]{xcolor}`
24  // a no-op → colortbl/array never load → array `m{}`/`b{}` columns are
25  // "Unrecognized tabular template" → "Extra alignment tab". The document
26  // loads xcolor with its own options; `\color`/`\definecolor` stay
27  // available via hyperref→color. See ifacconf_cls.rs / SYNC_STATUS.
28  RequirePackage!("hyperref");
29  // sagej templates use \toprule / \midrule / \bottomrule from booktabs.
30  // The raw cls relies on the user `\usepackage{booktabs}` but many
31  // papers don't load it explicitly. Eager-load. Witness 2309.01813.
32  RequirePackage!("booktabs");
33  // sagej.cls L108: `\RequirePackage{latexsym,ifthen,rotating,calc,textcase,
34  // booktabs,color,endnotes}`. Perl ships no sagej binding → raw-loads the
35  // bundled sagej.cls → endnotes loaded → `\endnote`/`\theendnotes` defined.
36  // Our binding intercepts sagej.cls (so that `\RequirePackage` never runs),
37  // leaving `\endnote` undefined where Perl is clean. Load endnotes to match
38  // the real .cls (SAGE papers use `\endnote{…}` for footnote-style notes).
39  // Witness 1901.10968 (`\endnote`, `\theendnotes`).
40  RequirePackage!("endnotes");
41
42  // sagej frontmatter — preserve author content.
43  DefMacro!("\\corrauth{}",
44    "\\@add@frontmatter{ltx:note}[role=corresponding]{#1}");
45  DefMacro!("\\affiliation{}",
46    "\\@add@frontmatter{ltx:note}[role=affiliation]{#1}");
47  DefMacro!("\\runninghead{}",
48    "\\@add@frontmatter{ltx:note}[role=runninghead]{#1}");
49
50  // {acks}, {funding}, {dci} envs (sagej L456-470). Use
51  // internal_vertical mode so block-level body (paragraphs, lists,
52  // funding-statement prose) is accepted — restricted_horizontal
53  // default tripped `Attempt to end mode restricted_horizontal in
54  // internal_vertical` on multi-paragraph bodies.
55  DefEnvironment!("{acks}", "<ltx:acknowledgements>#body</ltx:acknowledgements>",
56    mode => "internal_vertical");
57  DefEnvironment!("{funding}", "<ltx:acknowledgements name='funding'>#body</ltx:acknowledgements>",
58    mode => "internal_vertical");
59  DefEnvironment!("{dci}", "<ltx:acknowledgements name='dci'>#body</ltx:acknowledgements>",
60    mode => "internal_vertical");
61});
