1//! Stub for sagej.cls (SAGE journals).
2use latexml_package::prelude::*;
3
4LoadDefinitions!({
5  LoadClass!("OmniBus");
6  RequirePackage!("amsmath");
7  RequirePackage!("amsthm");
8  RequirePackage!("xcolor");
9  RequirePackage!("hyperref");
10  // sagej templates use \toprule / \midrule / \bottomrule from booktabs.
11  // The raw cls relies on the user `\usepackage{booktabs}` but many
12  // papers don't load it explicitly. Eager-load. Witness 2309.01813.
13  RequirePackage!("booktabs");
14
15  // sagej frontmatter — preserve author content.
16  DefMacro!("\\corrauth{}",
17    "\\@add@frontmatter{ltx:note}[role=corresponding]{#1}");
18  DefMacro!("\\affiliation{}",
19    "\\@add@frontmatter{ltx:note}[role=affiliation]{#1}");
20  DefMacro!("\\runninghead{}",
21    "\\@add@frontmatter{ltx:note}[role=runninghead]{#1}");
22
23  // {acks}, {funding}, {dci} envs (sagej L456-470). Use
24  // internal_vertical mode so block-level body (paragraphs, lists,
25  // funding-statement prose) is accepted — restricted_horizontal
26  // default tripped `Attempt to end mode restricted_horizontal in
27  // internal_vertical` on multi-paragraph bodies.
28  DefEnvironment!("{acks}", "<ltx:acknowledgements>#body</ltx:acknowledgements>",
29    mode => "internal_vertical");
30  DefEnvironment!("{funding}", "<ltx:acknowledgements name='funding'>#body</ltx:acknowledgements>",
31    mode => "internal_vertical");
32  DefEnvironment!("{dci}", "<ltx:acknowledgements name='dci'>#body</ltx:acknowledgements>",
33    mode => "internal_vertical");
34});
