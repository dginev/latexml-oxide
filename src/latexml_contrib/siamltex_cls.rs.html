3/// Push an `{AMS}`/`{AM}`/`{PII}` classification environment's digested body
4/// into the document `frontmatter` as `<ltx:classification scheme=…>`, instead
5/// of constructing it inline where the env was invoked. Mirrors Perl
6/// siamltex.cls.ltxml (`classification_tokens_for_env` → `\@add@frontmatter`)
7/// and OmniBus's `push_keyword_body_to_frontmatter`, just scheme-parameterized.
8/// Floating is REQUIRED because these envs are routinely used INSIDE
9/// `\begin{abstract}…\end{abstract}` (SIAM house style), where an inline
10/// `<ltx:classification>` is a content-model violation
11/// ("ltx:classification isn't allowed in ltx:abstract"). Witness 2009.00379.
12fn push_classification_to_frontmatter(
13  whatsit: &mut latexml_core::whatsit::Whatsit,
14  scheme: &str,
15) -> latexml_core::Result<Vec<latexml_core::digested::Digested>> {
16  use latexml_core::BoxOps;
17  use latexml_core::common::store::Stored;
18  if let Some(body) = whatsit.get_body()? {
19    let mut attrs: rustc_hash::FxHashMap<String, String> = rustc_hash::FxHashMap::default();
20    attrs.insert("scheme".to_string(), scheme.to_string());
21    let entry = latexml_core::document::tag::TagData {
22      tag: "ltx:classification".to_string(),
23      attr: attrs,
24      content: vec![latexml_core::document::tag::TagContent::Box(body)],
25    };
26    latexml_core::state::with_value_mut("frontmatter", |val_opt| {
27      if let Some(Stored::HashTagData(frnt)) = val_opt {
28        frnt
29          .entry("ltx:classification".to_string())
30          .or_insert_with(Vec::new)
31          .push(entry);
32      }
33    });
34  }
35  Ok(Vec::new())
36}
