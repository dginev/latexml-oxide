1use latexml_package::prelude::*;
2
3// apxproof.sty — "Automatic proofs in appendix" (Senellart, CTAN).
4//
5// No Perl LaTeXML binding exists (neither upstream nor ar5iv-bindings),
6// so Perl relies on raw-loading apxproof.sty under --includestyles and
7// *fails*: apxproof L58 `\ProcessLocalKeyvalOptions*` trips Perl's
8// kvoptions handling ("intended for packages only") → the whole
9// bibliography is dropped (0 rendered entries). See
10// docs/parity/KNOWN_PERL_ERRORS.md.
11//
12// This binding's single action is to force the raw .sty to load in EVERY
13// config (bare, --includestyles, ar5iv), not only under raw-style loading.
14// That is required because apxproof, when merely skipped (missing_file in
15// bare mode), still leaves a `<?latexml package="apxproof"?>` footprint
16// while its biblatex citation wiring never runs — so `\cite`s render
17// unlinked and every entry warns `expected:ids Missing Entry`. Forcing the
18// raw load runs apxproof's real setup (it internally `\RequirePackage`s
19// biblatex, kvoptions, etc.), so proofs keep LaTeXML's usual amsthm
20// `ltx_proof` markup and citations resolve against the bibliography.
21//
22// Rust's kvoptions raw-load handles `\ProcessLocalKeyvalOptions*`, so the
23// raw apxproof.sty loads cleanly where Perl aborts — a faithful
24// surpass-Perl improvement.
25LoadDefinitions!({
26  InputDefinitions!("apxproof", noltxml => true, extension => Some(Cow::Borrowed("sty")));
27});
