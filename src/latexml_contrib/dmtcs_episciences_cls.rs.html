1//! Stub for dmtcs-episciences.cls (Discrete Mathematics & Theoretical
2//! Computer Science journal class, Episciences platform).
3//!
4//! The raw cls preamble has font/page-layout machinery and conditional
5//! `\@firsthead` formatting (real `\publicationdata` builds running
6//! headers via tikz). Mid-load fails before reaching the publication
7//! macros, leaving `\publicationdata`, `\papertype`, `\fundinginfo`,
8//! `\accepted` undefined. Witness 2309.05874, 2309.12265.
9use latexml_package::prelude::*;
10
11
12LoadDefinitions!({
13  // Base on OmniBus, NOT article: Perl ships no dmtcs binding and falls back to
14  // OmniBus, which provides the generic journal frontmatter (`\received`,
15  // `\revised`, `\acknowledgements`/`\acknowledgments`, …). Basing on plain
16  // `article` left those undefined where Perl (OmniBus) is clean. The
17  // dmtcs-specific setters below are defined AFTER and override OmniBus.
18  // Witness 1904.12329 (`\received{}`/`\revised{}`/`\acknowledgements`).
19  LoadClass!("OmniBus");
20  RequirePackage!("amsmath");
21  RequirePackage!("amsthm");
22  RequirePackage!("amssymb");
23  RequirePackage!("graphicx");
24  RequirePackage!("hyperref");
25
26  // \publicationdata{volume}{year}{issue}{doi}{dates}{accepted} —
27  // 6-arg setter for running headers. Preserve as frontmatter note.
28  DefMacro!("\\publicationdata{}{}{}{}{}{}",
29    "\\@add@frontmatter{ltx:note}[role=dmtcs-publicationdata]{Volume #1 (#2), \\##3, doi:#4}");
30  // \papertype{T} — single-arg paper-type marker (Article/Editorial/…).
31  DefMacro!("\\papertype{}",
32    "\\@add@frontmatter{ltx:note}[role=papertype]{#1}");
33  // \fundinginfo{text} — funding statement.
34  DefMacro!("\\fundinginfo{}",
35    "\\@add@frontmatter{ltx:note}[role=funding]{#1}");
36  // \accepted{date} — accepted-date setter.
37  DefMacro!("\\accepted{}",
38    "\\@add@frontmatter{ltx:note}[role=accepted]{Accepted: #1}");
39  // \processdates / \endprocessdates — internal parser hooks; no-op.
40  def_macro_noop("\\processdates")?;
41  def_macro_noop("\\endprocessdates")?;
42  def_macro_noop("\\getinfo")?;
43
44  // \affiliation{text} — dmtcs-episciences.cls L251: stores affil text
45  // (`\gdef\@affil{#1}`). Preserve as ltx:contact frontmatter.
46  // Witness 2403.03614.
47  DefMacro!("\\affiliation{}",
48    "\\@add@to@frontmatter{ltx:creator}{\\@@@affiliation{#1}}");
49  DefConstructor!("\\@@@affiliation{}",
50    "^ <ltx:contact role='affiliation'>#1</ltx:contact>");
51  // \affiliationmark{tag} — L229: superscript marker on author/affil.
52  def_macro_noop("\\affiliationmark{}")?;
53  // \keywords{text} — L311: stores keywords (`\gdef\@keywords{#1}`).
54  // Preserve as frontmatter keywords.
55  DefMacro!("\\keywords{}",
56    "\\@add@frontmatter{ltx:keywords}{#1}");
57});
