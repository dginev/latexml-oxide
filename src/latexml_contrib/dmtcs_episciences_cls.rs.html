1//! Stub for dmtcs-episciences.cls (Discrete Mathematics & Theoretical
2//! Computer Science journal class, Episciences platform).
3//!
4//! The raw cls preamble has font/page-layout machinery and conditional
5//! `\@firsthead` formatting (real `\publicationdata` builds running
6//! headers via tikz). Mid-load fails before reaching the publication
7//! macros, leaving `\publicationdata`, `\papertype`, `\fundinginfo`,
8//! `\accepted` undefined. Witness 2309.05874, 2309.12265.
9use latexml_package::prelude::*;
10
11LoadDefinitions!({
12  // Base on OmniBus, NOT article: Perl ships no dmtcs binding and falls back to
13  // OmniBus, which provides the generic journal frontmatter (`\received`,
14  // `\revised`, `\acknowledgements`/`\acknowledgments`, ŌĆ”). Basing on plain
15  // `article` left those undefined where Perl (OmniBus) is clean. The
16  // dmtcs-specific setters below are defined AFTER and override OmniBus.
17  // Witness 1904.12329 (`\received{}`/`\revised{}`/`\acknowledgements`).
18  LoadClass!("OmniBus");
19  RequirePackage!("amsmath");
20  RequirePackage!("amsthm");
21  RequirePackage!("amssymb");
22  RequirePackage!("graphicx");
23  RequirePackage!("hyperref");
24
25  // \publicationdata{volume}{year}{issue}{doi}{dates}{accepted} ŌĆö
26  // 6-arg setter for running headers. Preserve as frontmatter note.
27  DefMacro!(
28    "\\publicationdata{}{}{}{}{}{}",
29    "\\@add@frontmatter{ltx:note}[role=dmtcs-publicationdata]{Volume #1 (#2), \\##3, doi:#4}"
30  );
31  // \papertype{T} ŌĆö single-arg paper-type marker (Article/Editorial/ŌĆ”).
32  DefMacro!(
33    "\\papertype{}",
34    "\\@add@frontmatter{ltx:note}[role=papertype]{#1}"
35  );
36  // \fundinginfo{text} ŌĆö funding statement.
37  DefMacro!(
38    "\\fundinginfo{}",
39    "\\@add@frontmatter{ltx:note}[role=funding]{#1}"
40  );
41  // \accepted{date} ŌĆö accepted-date setter.
42  DefMacro!(
43    "\\accepted{}",
44    "\\@add@frontmatter{ltx:note}[role=accepted]{Accepted: #1}"
45  );
46  // \processdates / \endprocessdates ŌĆö internal parser hooks; no-op.
47  def_macro_noop("\\processdates")?;
48  def_macro_noop("\\endprocessdates")?;
49  def_macro_noop("\\getinfo")?;
50
51  // \affiliation{text} ŌĆö dmtcs-episciences.cls L251: stores affil text
52  // (`\gdef\@affil{#1}`). Preserve as ltx:contact frontmatter.
53  // Witness 2403.03614.
54  DefMacro!(
55    "\\affiliation{}",
56    "\\@add@to@frontmatter{ltx:creator}{\\@@@affiliation{#1}}"
57  );
58  DefConstructor!(
59    "\\@@@affiliation{}",
60    "^ <ltx:contact role='affiliation'>#1</ltx:contact>"
61  );
62  // \affiliationmark{tag} ŌĆö L229: superscript marker on author/affil.
63  def_macro_noop("\\affiliationmark{}")?;
64  // \keywords{text} ŌĆö L311: stores keywords (`\gdef\@keywords{#1}`).
65  // Preserve as frontmatter keywords.
66  DefMacro!("\\keywords{}", "\\@add@frontmatter{ltx:keywords}{#1}");
67});
