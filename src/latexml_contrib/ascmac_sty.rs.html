1//! Stub for ascmac.sty — a Japanese LaTeX2e add-on (part of `jsclasses`
2//! family) that provides boxed environments such as `{itembox}`,
3//! `{screen}`, `{shadebox}`, `{boxnote}`. Not in standard TeX Live.
4//!
5//! Witness: 2601.09339 (`\usepackage{ascmac, fancybox}` +
6//! `\begin{itembox}[l]{title}` body `\end{itembox}`). We treat the
7//! boxed envs as transparent paragraphs and preserve the title as
8//! ltx:note, so author content is not dropped.
9use latexml_package::prelude::*;
10
11LoadDefinitions!({
12  // {itembox}[align]{title} ... \end{itembox}
13  // Render as a transparent block; preserve title via ltx:note.
14  DefEnvironment!(
15    "{itembox}[]{}",
16    "<ltx:para><ltx:p><ltx:note role='itembox-title'>#2</ltx:note>#body</ltx:p></ltx:para>",
17    mode => "internal_vertical"
18  );
19  // {screen}, {boxnote}, {shadebox} take no args.
20  DefEnvironment!(
21    "{screen}",
22    "<ltx:para><ltx:p>#body</ltx:p></ltx:para>",
23    mode => "internal_vertical"
24  );
25  DefEnvironment!(
26    "{boxnote}",
27    "<ltx:para><ltx:p>#body</ltx:p></ltx:para>",
28    mode => "internal_vertical"
29  );
30  DefEnvironment!(
31    "{shadebox}",
32    "<ltx:para><ltx:p>#body</ltx:p></ltx:para>",
33    mode => "internal_vertical"
34  );
35});
