1//! refstyle.sty — flexible cross-reference styling (`\newref{type}{...}`,
2//! `\vref`, `\Vref`, `\Ref`, ...).
3//!
4//! refstyle.sty's refstyle.cfg has `\newref{eq}{...}` which tries
5//! to redefine `\eqref` after `\RS@removedef{eqref}`. The
6//! `\RS@removedef` does `\let\eqref\@undefined` — in TeX this
7//! makes `\eqref` effectively undefined, but our `\@ifundefined`
8//! test (`\ifx\csname...\endcsname\relax`) doesn't recognize
9//! `\@undefined` as `\relax`, so the test sees `\eqref` as still
10//! defined and refstyle's `\RS@notdefinable` fires
11//! `\PackageError{refstyle}{Command \eqref already defined}`.
12//!
13//! Witnesses: arXiv:2009.10518, arXiv:1804.06350 — both load
14//! refstyle (and additionally `cleveref` which provides
15//! `\eqref` anyway). Perl LaTeXML has no refstyle binding
16//! (INCLUDE_STYLES=false skips), zero errors.
17//!
18//! Stub: no-op binding. The user-facing `\newref` / `\vref` /
19//! `\Vref` / `\Ref` are stubbed to passthrough `\ref`.
20use latexml_package::prelude::*;
21
22LoadDefinitions!({
23  Warn!(
24    "missing_file",
25    "refstyle.sty",
26    "refstyle.sty is minimally stubbed — \\newref/\\Vref/\\vref pass through to \\ref."
27  );
28  // refstyle's main public API. We don't honor the per-type
29  // templating; just dispatch to plain \ref so the labels still
30  // render correctly (just without the type-prefix prose).
31  DefMacro!("\\newref [] {} []", "");
32  DefMacro!("\\Vref {}", "\\ref{#1}");
33  DefMacro!("\\vref {}", "\\ref{#1}");
34  DefMacro!("\\Ref {}", "\\ref{#1}");
35  DefMacro!("\\refstyle {}", "");
36  DefMacro!("\\rangeref {} {}", "\\ref{#1}~--~\\ref{#2}");
37  DefMacro!("\\Rangeref {} {}", "\\ref{#1}~--~\\ref{#2}");
38  DefMacro!("\\nrefrange [] {} {}", "\\ref{#2}~--~\\ref{#3}");
39  DefMacro!("\\extdef {} {}", "");
40  // \refstylefirst — toggles "use long-form name on first ref"; no-op.
41  def_macro_noop("\\refstylefirst")?;
42  // RSfooform — formatter templates; ignored.
43});
