1//! Stub for apacite.sty (APA citation/bibliography style).
2//!
3//! Apacite defines a large family of APAref* macros driven by its .bbl
4//! generator. We don't implement APA formatting; pass-through the
5//! content-bearing args (authors, titles) and gobble the rest.
6use latexml_package::prelude::*;
7
8
9LoadDefinitions!({
10  RequirePackage!("natbib");
11  // apacite-generated .bbl entries routinely contain `\url{...}` and
12  // `\doi{10.x/...}` even when the user's main .tex doesn't load url /
13  // doi. Pre-load url (defines \url) and stub \doi as a no-op-wrapped
14  // href so the bbl pass doesn't error out. Witness 2205.09172 (cogsci
15  // article with apacite-formatted main.bbl).
16  RequirePackage!("url");
17  DefMacro!("\\doi Semiverbatim", "doi:#1");
18
19  // Core APAref* set (apacite.sty L1257-2243). Render as the
20  // content-bearing argument so titles / authors survive in the XML.
21  def_macro_noop("\\APACinsertmetastar{}")?;
22  DefMacro!("\\APACrefatitle{}{}", "#2");
23  DefMacro!("\\APACrefbtitle{}{}", "#2");
24  DefMacro!("\\APACrefYear{}", "(#1)");
25  DefMacro!("\\APACrefYearMonthDay{}{}{}", "(#1)");
26  DefMacro!("\\APACjournalVolNumPages{}{}{}{}", "#1 #2 #3 #4");
27  def_macro_identity("\\APAChowpublished{}")?;
28  DefMacro!("\\APACaddressPublisher{}{}", "#1: #2");
29  DefMacro!("\\APACaddressInstitution{}{}", "#1: #2");
30  DefMacro!("\\APACexlab{}{}", "#2");
31  // \APACmonth{name} — month text (was gobbled). Pass through inline.
32  def_macro_identity("\\APACmonth{}")?;
33  def_macro_identity("\\APACrefnote{}")?;
34  DefMacro!("\\APAhyperref{}{}", "#2");
35  def_macro_noop("\\PrintBackRefs{}")?;
36  def_macro_noop("\\CurrentBib")?;
37  def_macro_noop("\\bibcomputersoftwaremanual{}{}{}")?;
38
39  // APAref* environments
40  DefEnvironment!("{APACrefauthors}", "#body");
41  DefEnvironment!("{APACrefURL}", "#body");
42  DefEnvironment!("{APACrefDOI}", "#body");
43
44  // Additional APAC* macros (apacite ships many).
45  DefMacro!("\\APACyear{}", "(#1)");
46  DefMacro!("\\APACciteatitle{}{}", "#2");
47  DefMacro!("\\APACcitebtitle{}{}", "#2");
48  DefMacro!("\\APACrefaetitle{}{}", "#2");
49  DefMacro!("\\APACrefbetitle{}{}", "#2");
50  DefMacro!("\\APACbVolEdTR{}{}", "#2");
51  DefMacro!("\\APACbVolEdTRpgs{}{}{}", "#3");
52  DefMacro!("\\APACaddressInstitutionEqAuth{}{}", "#1: #2");
53  DefMacro!("\\APACaddressPublisherEqAuth{}{}", "#1: #2");
54  DefMacro!("\\APACaddressSchool{}{}", "#1: #2");
55  DefMacro!("\\APACtypeAddressSchool{}{}{}", "#3");
56  def_macro_noop("\\APACmetastar")?;
57  DefMacro!("\\APACorigyearnote{}", "(#1)");
58  def_macro_identity("\\APACorigjournalnote{}")?;
59  def_macro_identity("\\APACorigbooknote{}")?;
60  DefMacro!("\\APACorigED", "Ed.");
61  DefMacro!("\\APACorigEDS", "Eds.");
62  def_macro_identity("\\APACstd{}")?;
63  def_macro_noop("\\APACSortNoop{}")?;
64  def_macro_noop("\\APACmetaprenote")?;
65  def_macro_noop("\\APACrefauthstyle{}")?;
66  def_macro_noop("\\APACbibcite{}")?;
67
68  // apacite citation forms (apacite.sty L328+). Delegate to natbib's
69  // \cite which we wrapped in natbib_sty.rs. Forms:
70  //   \citeA[pre][post]{key} — author-only ("Smith")
71  //   \citeauthor[pre][post]{key} — author-only (alternate spelling)
72  //   \citeNP[pre][post]{key} — citation without parens
73  //   \citeyearNP[pre][post]{key} — year-only without parens
74  // Witness 2407.14158, 2407.18402, 2407.16770 (apacite-using papers).
75  DefMacro!("\\citeA[][] Semiverbatim", "\\citet[#1][#2]{#3}");
76  DefMacro!("\\citeNP[][] Semiverbatim", "\\citealp[#1][#2]{#3}");
77  DefMacro!("\\citeyearNP[][] Semiverbatim", "\\citeyear[#1][#2]{#3}");
78  def_macro_noop("\\APACrestorebibitem")?;
79  def_macro_noop("\\APACemindex{}")?;
80  def_macro_noop("\\APACltxemindex{}")?;
81  def_macro_noop("\\APACtocindex{}")?;
82  def_macro_noop("\\APACstdindex{}")?;
83  def_macro_noop("\\APACurlBreaks")?;
84
85  // Short-form helpers (apacite L1300+: \BBA, \BCnt, \BPGS, etc.)
86  // \BBA = `\BBAA` = `\&` (escaped ampersand, NOT alignment `&`).
87  // apacite.sty L2123: `\newcommand{\BBAA}{\&}`. Using plain `&` would
88  // emit a catcode-ALIGN cell separator inside the .bbl, triggering
89  // 19+ "Stray alignment \"&\"" errors (witness 2205.09172).
90  DefMacro!("\\BBA", "\\&");
91  // \BBAA: same as \BBA — the underlying glyph macro.
92  DefMacro!("\\BBAA", "\\&");
93  def_macro_noop("\\BBCQ")?;
94  def_macro_noop("\\BBOQ")?;
95  DefMacro!("\\BPBI", ".");
96  DefMacro!("\\BHBI", "-");
97  def_macro_noop("\\BDBL")?;
98  def_macro_noop("\\BCBT")?;
99  def_macro_noop("\\BCBL")?;
100  def_macro_identity("\\BCnt{}")?;
101  def_macro_identity("\\BPGS{}")?;
102  def_macro_identity("\\BVOL{}")?;
103  DefMacro!("\\BOthers{}", "et al.");
104  DefMacro!("\\BEDS", "Eds.");
105  DefMacro!("\\BED", "Ed.");
106  DefMacro!("\\BIn", "In");
107  // apacite.sty technical-report citation short forms. Witness 2205.05718.
108  DefMacro!("\\BTR",     "Tech.\\ Rep.");
109  DefMacro!("\\BNUM",    "No.");
110  DefMacro!("\\BNUMS",   "Nos.");
111  DefMacro!("\\BTRANS",  "Trans.");
112  DefMacro!("\\BTRANSS", "Trans.");
113  DefMacro!("\\BTRANSL", "trans.");
114  // \BPG = singular page (vs \BPGS = pages). Witness 2205.09172 (cogsci
115  // article + apacite .bbl): "\BPGS\ 1173--1182" plural already worked,
116  // but single-page entries use "\BPG\ N". apacite.sty L1336.
117  DefMacro!("\\BPG{}", "p.\\ #1");
118  // \shortciteA[pre][post]{key} — author-only short cite (apacite
119  // citation form). Delegate to natbib's \citet (author-name cite).
120  DefMacro!("\\shortciteA[][] Semiverbatim", "\\citet[#1][#2]{#3}");
121  // \shortciteauthor[pre][post]{key} — short form of \citeauthor.
122  DefMacro!("\\shortciteauthor[][] Semiverbatim", "\\citeauthor[#1][#2]{#3}");
123  // apacite.sty L1448-1452 allocates \bibleftmargin et al via
124  // `\newskip{\bibleftmargin}` and friends, then main.tex bodies use
125  // `\setlength{\bibleftmargin}{...}` and `\setlength{\bibindent}{-\bibleftmargin}`.
126  // We don't raw-load apacite.sty, so these must be defined as glue
127  // registers here so \setlength's Variable parameter resolves.
128  // Witness 2205.09172 (cogsci paper, "Error:expected:<variable>").
129  DefRegister!("\\bibleftmargin" => Glue!("2.5em"));
130  DefRegister!("\\bibindent"     => Glue::new(0));
131  DefRegister!("\\bibparsep"     => Glue::new(0));
132  DefRegister!("\\bibitemsep"    => Glue::new(0));
133  DefRegister!("\\biblabelsep"   => Glue::new(0));
134});
