67  // apacite citation forms (apacite.sty L328+). Delegate to natbib's \cite
68  // which we wrapped in natbib_sty.rs. Forms:
69  //   \citeA<pre>[post]{key}      — author-only ("Smith")
70  //   \citeNP<pre>[post]{key}     — citation without parens
71  //   \citeyearNP<pre>[post]{key} — year-only without parens
72  // Witness 2407.14158, 2407.18402, 2407.16770 (apacite-using papers).
73  //
74  // apacite's ANGLE-BRACKET pre-note. apacite.sty L259-311 gives every classic
75  // cite form the dispatch `\@ifnextchar< {\@cite} {\@cite<>}`, and L313-327
76  // `\def\@cite<#1>{... \@ifnextchar[ {\@@cite<#1>} {\@@cite<#1>[]}}` — so the
77  // full apacite citation syntax is
78  //   \cite<pre-note>[post-note]{key-list}
79  // Without the `<...>` form the kernel/natbib `\cite` takes the single token
80  // `<` as its key list: the citation renders as a dangling `[<]`, the REAL
81  // keys are never cited (so they are silently absent from the References) and
82  // `see>` leaks into the body text. Witness 2605.10951
83  // (`\cite<see>{Gangopadhyay02,Ferris25}`, agujournal2019), 2606.16518,
84  // 2606.19048, 2606.21531, 2606.24563.
85  //
86  // `OptionalAngled` (NOT `OptionalMatch:< OptionalUntil:>`): `Until` never
87  // checks for the OPENING delimiter, so when no `<` is present it scans to the
88  // next `>` anywhere downstream — `\citeA{Smith} and $a > b$` swallowed the
89  // cite and the math and reported the key as `b`.
90  // The signature `[pre1] OptionalAngled [post]` is UNAMBIGUOUS and so serves
91  // both spellings with one prototype: natbib's `[pre][post]` can only ever
92  // match `[pre1]…[post]` (the angled slot peeks, sees `[`, and yields nothing),
93  // and apacite's `<pre>[post]` can only ever match `OptionalAngled [post]`
94  // (the `[pre1]` slot peeks, sees `<`, and yields nothing). The two pre-note
95  // slots are therefore mutually exclusive, so `[#1#2]` simply concatenates
96  // whichever one fired — no conditional needed. `\citeA[a][b]{k}` /
97  // `\citeA[a]{k}` keep emitting exactly what they did before this change.
98  DefMacro!(
99    "\\citeA [] OptionalAngled [] Semiverbatim",
100    "\\citet[#1#2][#3]{#4}"
101  );
102  DefMacro!(
103    "\\citeNP [] OptionalAngled [] Semiverbatim",
104    "\\citealp[#1#2][#3]{#4}"
105  );
106  DefMacro!(
107    "\\citeyearNP [] OptionalAngled [] Semiverbatim",
108    "\\citeyear[#1#2][#3]{#4}"
109  );

111  // `\cite` keeps natbib's meaning on the no-`<` path (apacite's `\cite` is
112  // parenthetical, but re-pointing it at `\citep` would change rendering for
113  // every apacite paper — out of scope here); it only gains the pre-note form.
114  // Re-emitting `[#1#2][#3]` is behaviour-preserving for the bracket spellings:
115  // natbib's `swap_pre_post` keys off a NON-EMPTY post, so `\cite[a]{k}` →
116  // `[a][]` still swaps to post=a exactly as the bare `\cite[a]{k}` does.
117  Let!("\\lx@apac@core@cite", "\\cite");
118  DefMacro!(
119    "\\cite [] OptionalAngled [] Semiverbatim",
120    "\\lx@apac@core@cite[#1#2][#3]{#4}"

210  // \PrintOrdinal{N} → "Nth" ordinal (1st/2nd/3rd/4th…11th/12th/13th/Nth).
211  // Ported verbatim from apacite.sty L2098-2138 (pure TeX; uses \count@,
212  // \afterassignment, \ifcase, last-digit recursion). Used in `.bbl` edition
213  // fields, e.g. `\PrintOrdinal{3}` → "3rd".
214  RawTeX!(
215    r"%
216\let\@xp\expandafter
217\newcommand{\PrintOrdinal}[1]{%
218    \afterassignment\print@ordinal
219    \count@ 0#1\relax\@nil}
220\def\print@ordinal#1#2\@nil{%
221    \ifx\relax#1\relax
222        \ifnum\count@>\z@ \CardinalNumeric\count@ \else ??th\fi
223    \else
224        \ifnum \count@>\z@ \number\count@ \fi #1#2\relax
225    \fi}
226\newcommand{\CardinalNumeric}[1]{%
227    \number#1\relax
228    \if \ifnum#1<14 \ifnum#1>\thr@@ T\else F\fi \else F\fi Tth%
229    \else \@xp\keep@last@digit\@xp#1\number#1\relax
230        \ifcase#1th\or st\or nd\or rd\else th\fi
231    \fi}
232\def\keep@last@digit#1#2{%
233    \ifx\relax#2\@xp\@gobbletwo \else #1=#2\relax \fi \keep@last@digit#1}"
234  );
