1//! Binding for fairmeta.cls (FAIR / Meta pre-print class).
2//!
3//! fairmeta.cls builds its frontmatter interface
4//! (\author/\affiliation/\contribution/\metadata/\correspondence/\abstract) on
5//! an \addtolist[5] accumulator in the class BODY. Since an unknown .cls body is
6//! NOT raw-loaded (OmniBus extracts dependencies only), every one of those
7//! commands is Error:undefined without a binding — and the class's
8//! \RequirePackage{nicematrix} etc. never take effect either. Route the
9//! user-facing frontmatter through \@add@frontmatter so title/authors/
10//! affiliations/metadata/abstract reach the XML, and pull in the real
11//! dependency packages the class relies on.
12//!
13//! Witnesses: 2412.06264 (ar5iv #520), 2509.24704 (#567), 2511.16624 (#576).
14//! (selfevolagent.cls, 2508.07407/#556, is a near-identical sibling class.)
15use latexml_package::prelude::*;
16
17LoadDefinitions!({
18  LoadClass!("OmniBus");
19  // Dependency packages the class \RequirePackage's and the paper body relies
20  // on (those with bindings / user-facing commands). Purely-cosmetic layout
21  // packages (geometry, microtype, setspace, parskip, titlesec, …) are omitted
22  // — they are visual no-ops in LaTeXML and OmniBus degrades gracefully.
23  RequirePackage!("amsmath");
24  RequirePackage!("amssymb");
25  RequirePackage!("graphicx");
26  RequirePackage!("xcolor");
27  RequirePackage!("booktabs");
28  RequirePackage!("multirow");
29  RequirePackage!("bm");
30  RequirePackage!("etoolbox");
31  RequirePackage!("caption");
32  RequirePackage!("hyperref");
33  RequirePackage!("cleveref");
34  RequirePackage!("natbib");
35  RequirePackage!("nicematrix");
36  // Faithful to the class's `\RequirePackage[most]{tcolorbox}` (fairmeta.cls
37  // L42). PassOptions BEFORE the require (Perl idiom, mirrors ar5iv.sty.ltxml)
38  // so tcolorbox.sty's own \ProcessOptions sees `most` at raw-load time and
39  // \tcbuselibrary{most} loads the enhanced/breakable/skins keys.
40  pass_options("tcolorbox", "sty", vec![s!("most")])?;
41  RequirePackage!("tcolorbox");
42
43  // \geometry{...} — page-geometry hint; visual-only, gobble the arg.
44  def_macro_noop("\\geometry{}")?;
45
46  // Class palette (\color{metafg} is used by the abstract box).
47  Digest!("\\definecolor{metablue}{HTML}{E2EFEF}")?;
48  Digest!("\\definecolor{metafg}{HTML}{1C2B33}")?;
49  Digest!("\\definecolor{metabg}{HTML}{EFF6F6}")?;
50
51  // Shared "addtolist meta-class" frontmatter routing (\author/\affiliation/
52  // \contribution/\correspondence/\abstract/\email/\beginappendix + the list
53  // no-ops) — see `meta_class::install_meta_class_frontmatter`.
54  crate::meta_class::install_meta_class_frontmatter()?;
55
56  // Class-specific labeled fields (kept per-class):
57  def_macro_noop("\\metadatalist")?;
58  // \metadata[label]{value} — fairmeta's labels are simple words, so the label
59  // goes to the note `role` attribute. (The siblings whose labels can be
60  // arbitrary markup — selfevolagent/openmoss — render "label: value" as
61  // content instead; keep fairmeta's labels attribute-safe.)
62  DefMacro!(
63    "\\metadata[]{}",
64    "\\@add@frontmatter{ltx:note}[role=#1]{#2}"
65  );
66  // \date{text} (class: \metadata[Date]{...}; the \faCalendar icon is dropped).
67  DefMacro!("\\date{}", "\\@add@frontmatter{ltx:note}[role=date]{#1}");
68  // \nm{...} — no-op text wrapper (\newcommand{\nm}[1]{#1}).
69  DefMacro!("\\nm{}", "#1");
70});
