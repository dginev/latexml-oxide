18  // Author block. jmlr.cls L259/374-449:
19  //   \author[short]{ \Name{N1} \Email{E1} \\ \Name{N2} \Email{E2}
20  //                    \\ \addr Affiliation }
21  // The `\Name`/`\Email`/`\addr` sub-macros carry the structure. The generic
22  // \and/comma author splitter mis-reads it — it crams every \Name into a
23  // single <personname> and splits the affiliation's commas into phantom
24  // authors ("Oxford", "UK"). So when the body uses \Name, digest it directly
25  // into structured creators (each \Name → ltx:creator/ltx:personname, \Email →
26  // ltx:contact[role=email], the trailing \addr block → ltx:contact
27  // [role=affiliation]); otherwise fall back to the standard \author splitter.
28  // Beyond-Perl (Perl ships no jmlr binding). Witness 2410.16138, 2409.07012.
29  let_i(&T_CS!("\\lx@jmlr@core@author"), &T_CS!("\\author"), None);
30  DefMacro!("\\author[]{}", sub[(short, body)] {
31    if body.to_string().contains("\\Name") {
32      Ok(Invocation!(T_CS!("\\lx@jmlr@structauthor"), vec![Some(body)]))
33    } else {
34      Ok(Invocation!(T_CS!("\\lx@jmlr@core@author"), vec![short, Some(body)]))
35    }
36  });
37  // Digest the structured body in a group so the row separators (\\, \and, \AND)
38  // collapse to spaces; a sentinel is appended so a trailing \addr can capture
39  // the (shared) affiliation up to it.
40  DefMacro!(
41    "\\lx@jmlr@structauthor{}",
42    "\\begingroup\\def\\\\{ }\\def\\and{ }\\def\\AND{ }#1\\lx@jmlr@endaddr\\endgroup"
43  );
44  DefMacro!("\\Name[]{}", "\\lx@add@creator[role=author]{#2}");
45  DefMacro!("\\Email{}", "\\lx@add@email{#1}");
46  DefMacro!("\\addr Until:\\lx@jmlr@endaddr", "\\lx@add@affiliation{#1}");
47  def_macro_noop("\\lx@jmlr@endaddr")?;
48  // jmlr.cls \nametag{tag} appends a marker (typically a \thanks) next to an
