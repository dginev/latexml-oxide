1//! Stub for Elsevier's autart.cls (Automatica journal).
2//!
3//! autart.cls is an article-derivative. The native binding fallback to
4//! OmniBus misses class-defined helpers like {ack} environment and
5//! several elsart-style frontmatter macros.
6use latexml_package::prelude::*;
7
8LoadDefinitions!({
9  LoadClass!("OmniBus");
10  RequirePackage!("amsmath");
11  // Do NOT eager-load amsthm here. Perl ships no autart binding (→ OmniBus,
12  // no amsthm preload), and OmniBus already installs LAZY theorem-env
13  // autoload stubs (`\begin{theorem}`/`\begin{proof}`/… each `require`
14  // amsthm on first use). Preloading amsthm eagerly breaks the common
15  // pattern of a document that clears a class-defined `\proof` and then
16  // (re)loads amsthm to get amsthm's version:
17  //     \let\proof\relax        % drop autart/class \proof
18  //     \usepackage{amsthm}     % expect amsthm to (re)define \proof
19  // With amsthm pre-loaded, the `\usepackage{amsthm}` is a no-op (already
20  // loaded), so amsthm's `\let\proof\@proof` never re-runs and `\proof`
21  // stays `\relax` → `\begin{proof}` → "{proof} environment not defined".
22  // Witness arXiv:2009.00150 (autart + `\let\proof\relax` + amsthm). The
23  // lazy OmniBus stub still covers papers that DON'T load amsthm themselves.
24
25  // autart.cls L317-323: \def\ack{\section*{Acknowledgements}}
26  // with \let\endack\par. Bind {ack}/{ack*} as structural
27  // ltx:acknowledgements (post-processors map to canonical
28  // role/styling).
29  DefEnvironment!("{ack}",  "<ltx:acknowledgements>#body</ltx:acknowledgements>",
30    mode => "internal_vertical");
31  DefEnvironment!("{ack*}", "<ltx:acknowledgements>#body</ltx:acknowledgements>",
32    mode => "internal_vertical");
33
34  // Common elsart frontmatter macros (autart inherits elsart style) —
35  // preserve author-supplied content as ltx:note frontmatter.
36  DefMacro!("\\address[]{}",
37    "\\@add@frontmatter{ltx:note}[role=address]{#2}");
38  DefMacro!("\\thanksref{}", "\\textsuperscript{#1}");
39  DefMacro!("\\corauthref{}", "\\textsuperscript{*#1}");
40  DefMacro!("\\corauth{}",
41    "\\@add@frontmatter{ltx:note}[role=corresponding]{#1}");
42  DefMacro!("\\thanks{}",
43    "\\@add@frontmatter{ltx:note}[role=thanks]{#1}");
44
45  // autart.cls L516 defines `\qed` directly at class level (a `\Box` at the
46  // end of a proof). Perl — which OmniBus-fallbacks autart and dep-scans
47  // autart.cls's `\if@amsthm \RequirePackage{amsthm}` (the regex dep-scan
48  // ignores the `\if@amsthm` guard) — ends up loading amsthm and so produces
49  // amsthm's `\qed` (∎). We do NOT eager-load amsthm here (see the amsthm note
50  // above re: 2009.00150), so mirror amsthm's `\qed`/`\ltx@qed` (∎) directly to
51  // match Perl's ground-truth output. A paper that later `\usepackage{amsthm}`
52  // simply re-installs the identical definitions. Used via the common
53  // `\def\epf{\hfill\mbox{\qed}}` idiom OUTSIDE any proof env, so OmniBus's
54  // lazy theorem-env autoload never fires. Witness 1703.03101.
55  DefMacro!("\\qed", "\\ltx@qed");
56  DefConstructor!("\\ltx@qed",
57    "?#isMath(<ltx:XMTok role='PUNCT'>\u{220E}</ltx:XMTok>)(\u{220E})",
58    enter_horizontal => true,
59    reversion => "\\qed"
60  );
61  Let!("\\mathqed",    "\\qed");
62  Let!("\\textsquare", "\\qed");
63  Let!("\\qedsymbol",  "\\qed");
64  Let!("\\openbox",    "\\qed");
65});
