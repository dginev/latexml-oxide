24  // contents in \begin{minted}{language}...\end{minted}. The Rust
25  // \begin{minted} short-circuits to \begin{lstlisting} (via the

29  use latexml_core::binding::content::find_file;
30  DefMacro!("\\inputminted[]{}{}", sub[(_opts, _lang, file_arg)] {
31    let file_str = file_arg.to_string();
32    let mut tokens: Vec<Token> = Vec::new();
33    tokens.push(T_CS!("\\begin{lstlisting}"));
34    if let Some(path) = find_file(&file_str, None) {
35      if let Ok(contents) = std::fs::read_to_string(&path) {
36        tokens.extend(Explode!(&contents));
37      }
38    }
39    tokens.push(T_CS!("\\end{lstlisting}"));
40    Ok(Tokens::new(tokens))
41  });
