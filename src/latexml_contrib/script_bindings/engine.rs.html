1//! The sandboxed Rhai engine: every `register_fn` of the binding-language
2//! surface (state, tokens, counters, definitions, package/class machinery,
3//! document/whatsit proxies). Split from `mod.rs` for navigability; each
4//! registration lowers to the same native function its compile-time macro
5//! does (`setup_binding_language.rs`).
6
7use super::*;
8
9/// Mini-DSL to expose a `document.<rhai>(qname)` → `Document::<rust>(qname)`
10/// method whose result is discarded — the common side-effect-on-element shape.
11/// Adding a new such method is then one line; see the registrations in
12/// `make_engine`.
13macro_rules! doc_qname_method {
14  ($engine:ident, $rhai:literal, $rust:ident) => {
15    $engine.register_fn(
16      $rhai,
17      |_d: &mut DocProxy, qname: &str| -> std::result::Result<(), Box<EvalAltResult>> {
18        with_doc(|doc, _props| {
19          doc
20            .$rust(qname)
21            .map_err(rhai_err)?;
22          Ok(())
23        })
24      },
25    );
26  };
27}

29/// Build a sandboxed Rhai engine with the binding API registered.
30pub(super) fn make_engine() -> Engine {
31  let mut engine = Engine::new();
32  engine.set_max_operations(MAX_OPERATIONS);
33  engine.set_max_call_levels(MAX_CALL_LEVELS);
34  engine.set_max_string_size(MAX_STRING_SIZE);
35
36  // ── registration API (wired to native defs IMMEDIATELY, in script order) ──
37  engine.register_fn(
38    "DefMacro",
39    |proto: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
40      wire_now(|e, a| wire_macro(e, a, proto, body))
41    },
42  );
43  // Option-bag form (Perl's trailing `key => value`s): scalars onto
44  // `ExpandableOptions` via the shared mapper.
45  engine.register_fn(
46    "DefMacro",
47    |proto: &str, body: FnPtr, opts: Map| -> std::result::Result<(), Box<EvalAltResult>> {
48      wire_now(|e, a| wire_macro_opts(e, a, proto, body, opts))
49    },
50  );
51  engine.register_fn(
52    "DefPrimitive",
53    |proto: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
54      wire_now(|e, a| wire_primitive(e, a, proto, body, PrimitiveOptions::default()))
55    },
56  );
57  engine.register_fn(
58    "DefPrimitive",
59    |proto: &str, body: FnPtr, opts: Map| -> std::result::Result<(), Box<EvalAltResult>> {
60      wire_now(|e, a| wire_primitive(e, a, proto, body, primitive_options_from_map(opts)))
61    },
62  );
63  // Class/package option. Mirrors the `DeclareOption!` macro's lowering
64  // (setup_binding_language.rs): push the name onto `@declaredoptions` and
65  // define a `\ds@<opt>` primitive carrying the body. Installed immediately,
66  // so a following `ProcessOptions()` in the same script sees it.
67  engine.register_fn(
68    "DeclareOption",
69    |opt: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
70      wire_now(|e, a| wire_option(e, a, opt, body))
71    },
72  );
73
74  // ── free-function helpers (prelude names) over core datatypes ──
75  // `Tokens` is registered as an opaque pass-through handle (Tier-2): a script
76  // produces it with Tokenize and hands it to Digest without inspecting it.
77  engine.register_type_with_name::<Tokens>("Tokens");
78  engine.register_fn("Tokenize", |s: &str| -> Tokens { mouth::tokenize(s) });
79  engine.register_fn("TokenizeInternal", |s: &str| -> Tokens {
80    mouth::tokenize_internal(s)
81  });
82  engine.register_fn(
83    "Digest",
84    |t: Tokens| -> std::result::Result<(), Box<EvalAltResult>> {
85      latexml_core::stomach::digest(t).map_err(rhai_err)?;
86      Ok(())
87    },
88  );
89
90  // ── state API (for primitive side-effects / value-reading macros) ──
91  // `assign_value` is group-local (TeX default); `assign_global` persists past
92  // the enclosing group. (Untrusted-script key-namespacing is a documented
93  // follow-up — see the plan's critical re-eval §5.)
94  engine.register_fn("assign_value", |key: &str, val: &str| {
95    latexml_core::state::assign_value(key, val.to_string(), Some(Scope::Local));
96  });
97  engine.register_fn("assign_global", |key: &str, val: &str| {
98    latexml_core::state::assign_value(key, val.to_string(), Some(Scope::Global));
99  });
100  // Curated pool helpers, registered 1:1 under their Perl/Rust names so binding
101  // bodies read like the originals (e.g. `beforeDigest: || neutralize_font()`).
102  engine.register_fn(
103    "neutralize_font",
104    latexml_engine::base_utilities::neutralize_font,
105  );
106  engine.register_fn("lookup_value", |key: &str| -> String {
107    match latexml_core::state::lookup_value(key) {
108      Some(Stored::String(s)) => arena::to_string(s),
109      Some(other) => other.to_string(),
110      None => String::new(),
111    }
112  });
113
114  // ── the binding-language surface, 1:1 under its macro names ──
115  // Each registration lowers to the SAME native function its compile-time
116  // macro does (setup_binding_language.rs), so a Rhai binding reads like the
117  // `.pool`/`_sty.rs` original. String values in, strings/handles out.
118  engine.register_fn("AssignValue", |k: &str, v: &str| {
119    latexml_core::state::assign_value(k, v.to_string(), None);
120  });
121  engine.register_fn("AssignValue", |k: &str, v: &str, scope: &str| {
122    latexml_core::state::assign_value(k, v.to_string(), scope_of(scope));
123  });
124  engine.register_fn("LookupString", |k: &str| -> String {
125    latexml_core::state::lookup_string(k)
126  });
127  engine.register_fn("LookupNumber", |k: &str| -> i64 {
128    latexml_core::state::lookup_number(k)
129      .map(|n| n.0)
130      .unwrap_or(0)
131  });
132  engine.register_fn("LookupBool", |k: &str| -> bool {
133    latexml_core::state::lookup_bool(k)
134  });
135  engine.register_fn("LookupTokens", |k: &str| -> Tokens {
136    latexml_core::state::lookup_tokens(k).unwrap_or_default()
137  });
138  // Catcode access: the char as a 1-char string, the catcode as its TeX int.
139  engine.register_fn("LookupCatcode", |c: &str| -> i64 {
140    c.chars()
141      .next()
142      .and_then(latexml_core::state::lookup_catcode)
143      .map(|cc| cc as i64)
144      .unwrap_or(12)
145  });
146  engine.register_fn("AssignCatcode", |c: &str, code: i64| {
147    if let Some(ch) = c.chars().next() {
148      latexml_core::state::assign_catcode(ch, Catcode::from(code as u8), None);
149    }
150  });
151  // LookupMeaning: the meaning of a CS as its display string ("" if none).
152  engine.register_fn("LookupMeaning", |cs: &str| -> String {
153    match latexml_core::state::lookup_meaning(&latexml_core::T_CS!(cs)) {
154      Some(m) => m.to_string(),
155      None => String::new(),
156    }
157  });
158  engine.register_fn("Let", |a: &str, b: &str| {
159    latexml_core::state::let_i(&latexml_core::T_CS!(a), &latexml_core::T_CS!(b), None);
160  });
161  engine.register_fn("XEquals", |a: &str, b: &str| -> bool {
162    latexml_core::state::x_equals(&latexml_core::T_CS!(a), &latexml_core::T_CS!(b))
163  });
164  engine.register_fn("IsDefined", |cs: &str| -> bool {
165    latexml_core::binding::def::dialect::is_defined_token(&latexml_core::T_CS!(cs))
166  });
167  // RawTeX: process literal TeX as definitions input (the raw-`\def` escape
168  // hatch every nontrivial binding uses). TeX: tokenize-internal + digest.
169  engine.register_fn(
170    "RawTeX",
171    |text: &str| -> std::result::Result<(), Box<EvalAltResult>> {
172      latexml_core::stomach::raw_tex(text).map_err(rhai_err)
173    },
174  );
175  engine.register_fn(
176    "TeX",
177    |text: &str| -> std::result::Result<(), Box<EvalAltResult>> {
178      latexml_core::stomach::digest(mouth::tokenize_internal(text)).map_err(rhai_err)?;
179      Ok(())
180    },
181  );
182  engine.register_fn(
183    "Expand",
184    |t: Tokens| -> std::result::Result<Tokens, Box<EvalAltResult>> {
185      latexml_core::gullet::do_expand(t).map_err(rhai_err)
186    },
187  );
188  engine.register_fn(
189    "ExpandPartially",
190    |t: Tokens| -> std::result::Result<Tokens, Box<EvalAltResult>> {
191      latexml_core::gullet::do_expand_partially(t).map_err(rhai_err)
192    },
193  );
194  engine.register_fn("UnTeX", |t: Tokens| -> String { t.untex() });
195  // DigestText: digest a TeX string into a `Digested` handle — the workhorse
196  // of `properties` closures that precompute content (e.g. IEEEproof's title).
197  engine.register_fn(
198    "DigestText",
199    |s: &str| -> std::result::Result<Digested, Box<EvalAltResult>> {
200      latexml_core::binding::content::digest_text(mouth::tokenize_internal(s)).map_err(rhai_err)
201    },
202  );
203  // ToString/ToAttribute/Revert on a Digested handle (Perl ToString/Revert).
204  engine.register_fn("ToString", |d: Digested| -> String { d.to_string() });
205  engine.register_fn("ToAttribute", |d: Digested| -> String { d.to_attribute() });
206  engine.register_fn(
207    "Revert",
208    |d: Digested| -> std::result::Result<Tokens, Box<EvalAltResult>> {
209      d.revert().map_err(rhai_err)
210    },
211  );
212  engine.register_fn(
213    "Today",
214    || -> std::result::Result<String, Box<EvalAltResult>> {
215      latexml_engine::base_utilities::today().map_err(rhai_err)
216    },
217  );
218  engine.register_fn("Warn", |cat: &str, obj: &str, msg: &str| {
219    latexml_core::Warn!(cat, obj, msg.to_string());
220  });
221  engine.register_fn(
222    "Error",
223    |cat: &str, obj: &str, msg: &str| -> std::result::Result<(), Box<EvalAltResult>> {
224      // Error! escalates to Fatal past MAX_ERRORS (a latexml `Err`) — surface
225      // that to the script as a Rhai error so conversion aborts cleanly.
226      let res: Result<()> = (|| {
227        latexml_core::Error!(cat, obj, msg.to_string());
228        Ok(())
229      })();
230      res.map_err(rhai_err)
231    },
232  );

261    "AddToCounter",

306  // GetKeyVal / GetKeyVals over the keyval dict's TeX-source form (how an
307  // `OptionalKeyVals:` argument reaches a script body): "k=v,k2={v2}" → value
308  // by key / a Rhai map of all pairs. Brace-aware at depth 0, like keyval.
309  engine.register_fn("GetKeyVal", |kv: &str, key: &str| -> String {
310    latexml_core::keyval::split_keyval_source(kv)
311      .into_iter()
312      .find(|(k, _)| k == key)
313      .map(|(_, v)| v)
314      .unwrap_or_default()
315  });
316  engine.register_fn("GetKeyVals", |kv: &str| -> Map {
317    let mut m = Map::new();
318    for (k, v) in latexml_core::keyval::split_keyval_source(kv) {
319      m.insert(k.into(), Dynamic::from(v));
320    }
321    m
322  });
