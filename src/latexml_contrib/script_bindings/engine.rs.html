1//! The sandboxed Rhai engine: every `register_fn` of the binding-language
2//! surface (state, tokens, counters, definitions, package/class machinery,
3//! document/whatsit proxies). Split from `mod.rs` for navigability; each
4//! registration lowers to the same native function its compile-time macro
5//! does (`setup_binding_language.rs`).

9/// Mini-DSL to expose a `document.<rhai>(qname)` → `Document::<rust>(qname)`
10/// method whose result is discarded — the common side-effect-on-element shape.
11/// Adding a new such method is then one line; see the registrations in
12/// `make_engine`.
13macro_rules! doc_qname_method {
14  ($engine:ident, $rhai:literal, $rust:ident) => {
15    $engine.register_fn(
16      $rhai,
17      |_d: &mut DocProxy, qname: &str| -> std::result::Result<(), Box<EvalAltResult>> {
18        with_doc(|doc, _props| {
19          doc.$rust(qname).map_err(rhai_err)?;
20          Ok(())
21        })
22      },
23    );
24  };
25}

27/// Build a sandboxed Rhai engine with the binding API registered.
28pub(super) fn make_engine() -> Engine {
29  let mut engine = Engine::new();
30  engine.set_max_operations(MAX_OPERATIONS);
31  engine.set_max_call_levels(MAX_CALL_LEVELS);
32  engine.set_max_string_size(MAX_STRING_SIZE);

34  // ── registration API (wired to native defs IMMEDIATELY, in script order) ──
35  engine.register_fn(
36    "DefMacro",
37    |proto: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
38      wire_now(|e, a| wire_macro(e, a, proto, body))
39    },
40  );
41  // String-body form — Perl's most common `DefMacro('\foo', 'bar')`, where the
42  // body is TeX source that becomes the expansion directly (no Rhai closure).
43  // Wires the same native expandable as the compile-time `DefMacro!` string
44  // form, so a `.rhai` binding reads like the `.ltxml` original. The wiring
45  // ignores the script handles, hence the throwaway `|_, _|`.
46  engine.register_fn(
47    "DefMacro",
48    |proto: &str, body: &str| -> std::result::Result<(), Box<EvalAltResult>> {
49      wire_now(|_, _| wire_macro_string(proto, body))
50    },
51  );
52  // Option-bag form (Perl's trailing `key => value`s): scalars onto
53  // `ExpandableOptions` via the shared mapper. Both the closure-body and the
54  // string-body variants accept it.
55  engine.register_fn(
56    "DefMacro",
57    |proto: &str, body: FnPtr, opts: Map| -> std::result::Result<(), Box<EvalAltResult>> {
58      wire_now(|e, a| wire_macro_opts(e, a, proto, body, opts))
59    },
60  );
61  engine.register_fn(
62    "DefMacro",
63    |proto: &str, body: &str, opts: Map| -> std::result::Result<(), Box<EvalAltResult>> {
64      wire_now(|_, _| wire_macro_string_opts(proto, body, opts))
65    },
66  );
67  engine.register_fn(
68    "DefPrimitive",
69    |proto: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
70      wire_now(|e, a| wire_primitive(e, a, proto, body, PrimitiveOptions::default()))
71    },
72  );
73  engine.register_fn(
74    "DefPrimitive",
75    |proto: &str, body: FnPtr, opts: Map| -> std::result::Result<(), Box<EvalAltResult>> {
76      wire_now(|e, a| wire_primitive(e, a, proto, body, primitive_options_from_map(opts, e, a)))
77    },
78  );
79  // Class/package option. Mirrors the `DeclareOption!` macro's lowering
80  // (setup_binding_language.rs): push the name onto `@declaredoptions` and
81  // define a `\ds@<opt>` primitive carrying the body. Installed immediately,
82  // so a following `ProcessOptions()` in the same script sees it.
83  engine.register_fn(
84    "DeclareOption",
85    |opt: &str, body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
86      wire_now(|e, a| wire_option(e, a, opt, body))
87    },
88  );
89  // String-body form: `DeclareOption('opt', '\xdef...')` — the body is TeX,
90  // installed as a token-expansion macro (Perl's `DeclareOption($o, $string)`).
91  engine.register_fn(
92    "DeclareOption",
93    |opt: &str, body: &str| -> std::result::Result<(), Box<EvalAltResult>> {
94      wire_now(|_, _| wire_option_string(opt, body))
95    },
96  );
97  // Default-option handler: `DeclareOption(sub {...})` with no name — Perl's
98  // `DeclareOption(undef, sub {...})`, defining `\default@ds` for any option not
99  // otherwise declared.
100  engine.register_fn(
101    "DeclareOption",
102    |body: FnPtr| -> std::result::Result<(), Box<EvalAltResult>> {
103      wire_now(|e, a| wire_option_default(e, a, body))
104    },
105  );

107  // ── free-function helpers (prelude names) over core datatypes ──
108  // `Tokens` is registered as an opaque pass-through handle (Tier-2): a script
109  // produces it with Tokenize and hands it to Digest without inspecting it.
110  engine.register_type_with_name::<Tokens>("Tokens");
111  engine.register_fn("Tokenize", |s: &str| -> Tokens { mouth::tokenize(s) });
112  engine.register_fn("TokenizeInternal", |s: &str| -> Tokens {
113    mouth::tokenize_internal(s)
114  });
115  // `Digest` returns the digested handle (Perl returns the digested box), so a
116  // script can `ToString(Digest(...))` — e.g. a default-option handler reading
117  // `\CurrentOption`. Callers that ignore the return (the common side-effect
118  // use) are unaffected.
119  engine.register_fn(
120    "Digest",
121    |t: Tokens| -> std::result::Result<Digested, Box<EvalAltResult>> {
122      latexml_core::stomach::digest(t).map_err(rhai_err)
123    },
124  );
125  // `T_CS('\foo')` — a single control-sequence token (wrapped as `Tokens` so it
126  // composes with `Digest`/`Expand`). The dominant token-constructor use.
127  engine.register_fn("T_CS", |name: &str| -> Tokens {
128    Tokens::new(vec![latexml_core::T_CS!(name)])
129  });

132  // `assign_value` is group-local (TeX default); `assign_global` persists past
133  // the enclosing group. (Untrusted-script key-namespacing is a documented

137  });
138  engine.register_fn("assign_global", |key: &str, val: &str| {
139    latexml_core::state::assign_value(key, val.to_string(), Some(Scope::Global));
140  });

144    "neutralize_font",
145    latexml_engine::base_utilities::neutralize_font,
146  );

148    match latexml_core::state::lookup_value(key) {
149      Some(Stored::String(s)) => arena::to_string(s),

156  // Each registration lowers to the SAME native function its compile-time

158  // `.pool`/`_sty.rs` original. String values in, strings/handles out.

168  // Perl-style `LookupValue`: returns the stored value in its natural shape. A
169  // list value (e.g. `class_options`, a pushed queue) becomes a Rhai array —
170  // mirroring Perl's `LookupValue` returning the arrayref (#315) — rather than
171  // the leaked Debug repr `LookupString` used to emit. Scalars stringify as
172  // before; a missing key is unit (Perl `undef`).

174    match latexml_core::state::lookup_value(k) {
175      None => Dynamic::UNIT,
176      Some(v) if v.is_list() => {
177        let arr: rhai::Array = v
178          .list_items()
179          .unwrap_or_default()
180          .into_iter()
181          .map(Dynamic::from)
182          .collect();
183        Dynamic::from_array(arr)
184      },
185      Some(Stored::String(s)) => arena::to_string(s).into(),
186      Some(other) => other.to_string().into(),
187    }
188  });

190    latexml_core::state::lookup_number(k)
191      .map(|n| n.0)
192      .unwrap_or(0)
193  });

195    latexml_core::state::lookup_bool(k)
196  });

204  });

211      .next()
212      .and_then(latexml_core::state::lookup_catcode)
213      .map(|cc| cc as i64)

340  // ── external commands: a thin mirror of `std::process::Command` (#318). Lets
341  // a TRUSTED binding shell out (BookML's `latexmk`/`dvisvgm` during digestion),
342  // as Perl `.ltxml` does with `system()`. Method names mirror std exactly, so a
343  // user reads the `std::process::Command` docs. SAFETY.md: the runtime-bindings
344  // feature is the trust boundary — untrusted deployments drop it. Runs to
345  // completion like std/`system()` (no hidden timeout); the overall conversion
346  // timeout is the outer bound. ──
347  engine.register_type_with_name::<RhaiCommand>("Command");
348  engine.register_fn("Command", |program: &str| RhaiCommand {
349    program: program.to_string(),
350    ..RhaiCommand::default()
351  });
352  engine.register_fn("arg", |c: &mut RhaiCommand, a: &str| {

354  });

357  });

359    c.envs.push((k.to_string(), v.to_string()));
360  });
361  engine.register_fn("current_dir", |c: &mut RhaiCommand, dir: &str| {

363  });
364  // `output()` mirrors std's: runs to completion, capturing stdout/stderr.
365  // Returns `std::process::Output` flattened for Rhai: `ExitStatus` → `status`
366  // (exit code, -1 on signal) + `success`; `Vec<u8>` → lossy `String`. A spawn
367  // failure (program not found) surfaces as a Rhai error, mirroring std's
368  // `io::Result::Err` (catchable with `try`/`catch`).
369  engine.register_fn(
370    "output",
371    |c: &mut RhaiCommand| -> std::result::Result<Dynamic, Box<EvalAltResult>> {
372      // Allowed by default (Perl `.ltxml` runs `system()` freely), but BLOCKABLE
373      // via `LATEXML_DISABLE_SHELL_ESCAPE` — the opt-out an untrusted-input
374      // deployment (e.g. `cortex_worker`) sets, since a `.rhai` beside the
375      // source auto-loads (converter.rs `rhai_dispatch`) and could otherwise
376      // shell out. See SAFETY.md.
377      if std::env::var_os("LATEXML_DISABLE_SHELL_ESCAPE").is_some() {
378        return Err(rhai_err(format!(
379          "Command '{}': external commands from bindings are disabled \
380           (LATEXML_DISABLE_SHELL_ESCAPE is set)",
381          c.program
382        )));
383      }
384      let mut cmd = std::process::Command::new(&c.program);
385      cmd.args(&c.args);
386      for (k, v) in &c.envs {
387        cmd.env(k, v);
388      }
389      if let Some(dir) = &c.cwd {
390        cmd.current_dir(dir);
391      }
392      let out = cmd
393        .output()
394        .map_err(|e| rhai_err(format!("Command '{}': {e}", c.program)))?;
395      let mut m = Map::new();
396      m.insert(
397        "status".into(),
398        (out.status.code().unwrap_or(-1) as i64).into(),
399      );
400      m.insert("success".into(), out.status.success().into());
401      m.insert(
402        "stdout".into(),
403        String::from_utf8_lossy(&out.stdout).into_owned().into(),
404      );
405      m.insert(
406        "stderr".into(),
407        String::from_utf8_lossy(&out.stderr).into_owned().into(),
408      );
409      Ok(m.into())
410    },
411  );

485  // GetKeyVal / GetKeyVals over a keyval argument. A `KeyVals`/`OptionalKeyVals`
486  // argument reaches a constructor *closure* body as a digested `KeyVals` handle
487  // (or unit `()` when an optional set was omitted); it reaches other contexts
488  // as its TeX-source string ("k=v,k2={v2}"). `GetKeyVal` accepts both — plus
489  // unit — so a script body can `GetKeyVal(#1, "width")` uniformly. Missing key
490  // (or unit) → "".
491  engine.register_fn("GetKeyVal", |kv: Dynamic, key: &str| -> String {
492    if let Some(d) = kv.clone().try_cast::<Digested>() {
493      return match d.data() {
494        latexml_core::digested::DigestedData::KeyVals(keyval) => keyval
495          .get_value_digested(key)
496          .map(|v| v.to_string())
497          .unwrap_or_default(),
498        _ => String::new(),
499      };

1462  // ParseXML(markup) -> [Node] — the parser exposed as its own primitive (#350),
1463  // so a script can inspect or edit markup BEFORE it reaches the document. Each
1464  // returned node OWNS the throwaway document it was parsed into (see
1465  // `NodeProxy`), so it is safe to hold, walk and mutate for as long as the
1466  // script keeps it — unlike an in-tree node from a rewrite callback. Returns an
1467  // array because a chunk may legitimately be a fragment of several siblings.
1468  engine.register_fn(
1469    "ParseXML",
1470    |markup: &str| -> std::result::Result<rhai::Array, Box<EvalAltResult>> {
1471      // Malformed markup is reported as a latexml `Error:` and yields an EMPTY
1472      // array — deliberately NOT a Rhai runtime error. A script error raised here
1473      // would propagate out of the constructor and ABORT THE WHOLE CONVERSION
1474      // (`wire.rs` maps it to a hard `Error::from`), so a single bad snippet in one
1475      // contrib binding would cost the entire document. That contradicts the
1476      // failure-isolation contract (`script_bindings_plan.md`: a bad binding
1477      // "degrades only that package"), and it would also make these two siblings
1478      // behave oppositely on the same bad input — `insertXML(markup)` already
1479      // degrades this way. Loud, but contained: the author still gets the `Error:`.
1480      match latexml_core::common::xml::parse_fragment(markup) {
1481        Ok(parsed) => {
1482          let doc = parsed.document().clone();
1483          Ok(
1484            parsed
1485              .nodes()
1486              .into_iter()
1487              .map(|n| Dynamic::from(NodeProxy::owning(n, doc.clone())))
1488              .collect(),
1489          )
1490        },
1491        Err(e) => {
1492          // The one thing that MAY still abort is `Error!` escalating to Fatal
1493          // past MAX_ERRORS — same shape as the `Error` registration above.
1494          let res: Result<()> = (|| {
1495            latexml_core::Error!(
1496              "malformed",
1497              "ParseXML",
1498              format!("could not parse XML markup: {e}")
1499            );
1500            Ok(())
1501          })();
1502          res.map_err(rhai_err)?;
1503          Ok(rhai::Array::new())
1504        },
1505      }
1506    },
