1//! Runtime script bindings via embedded **Rhai** (see `docs/parity/script_bindings_plan.md`).
2//!
3//! Feature-gated behind `runtime-bindings` (ON by default and in the distribution
4//! build ŌĆö downstream single-binary consumers customize without recompiling; the
5//! module keeps its original name, the feature does not). The ONLY module in the
6//! workspace that embeds Rhai; core/engine/package are untouched. Pure-Rust, no
7//! FFI, no ABI ŌĆö the reliable successor to the abandoned libperl approach.
8//!
9//! ## Model (the "fine seam", Model A)
10//!
11//! A binding is a Rhai script that calls registration functions (`DefMacro`,
12//! `DefConstructor`, ŌĆ”) with the binding body as a Rhai function value. Each
13//! registration installs a **native** latexml-oxide `Definition` whose body
14//! closure trampolines into the Rhai engine. Prototype parsing, argument
15//! reading, the gullet/stomach/document machinery all stay native; Rhai runs
16//! only the body.
17//!
18//! ## Seams implemented
19//!
20//! * **`DefMacro`** (expandable) ŌĆö body receives args as strings, returns a string of TeX,
21//!   re-tokenized via `mouth::tokenize_internal` so an expansion to `\textit{x}` faithfully yields
22//!   a control-sequence token.
23//! * **`DefConstructor`** (construction) ŌĆö two forms. A string XML template, or an imperative body
24//!   that reads like the Perl original: it receives a `Document` **proxy** as its first argument
25//!   (Perl's `$_[0]`) and each digested argument as an opaque handle, e.g. `|document, x| {
26//!   document.openElement("ltx:emph"); document.absorb(x); document.closeElement("ltx:emph"); }`.
27//!   The proxy's methods resolve a thread-local **active context** (the live `&mut Document` +
28//!   props, published for the duration of the call; never borrowed across a re-entrant call).
29//!
30//! ## Lifecycle
31//!
32//! `load_script` compiles the script to an `AST`, runs it once to collect the
33//! registrations, wraps engine + AST in `Rc`, and installs one native definition
34//! per registration ŌĆö each capturing `Rc<Engine>`/`Rc<AST>` so a body stays
35//! callable for the whole conversion (a deferred `FnPtr::call` needs both alive).
36

136/// Generate an RAII push/pop guard for one of the active-context stacks.
137///
138/// Why RAII and not manual `push`ŌĆ”`pop()`: Rhai does NOT `catch_unwind` its
139/// native calls, so a panic in a script body (or a deep `RefCell` borrow panic,
140/// incl. the B1 re-entrancy guard) skips a trailing manual `pop()` ŌĆö leaving a
141/// **stale** entry on the stack. `reset_thread_state` doesn't clear these
142/// `latexml_contrib` thread-locals, so the dangling entry survives the cortex
143/// per-paper `catch_unwind` and leaks into the next paper. A `Drop` guard pops
144/// on EVERY scope exit ŌĆö normal return, `?` early-return, or unwind. Review M1.
145macro_rules! ctx_stack_guard {
146  ($guard:ident, $stack:ident, $elem:ty) => {
147    #[must_use = "the guard pops on drop; binding it to `_` would pop immediately"]
148    struct $guard(());
149    impl $guard {
150      fn new(entry: $elem) -> Self {
151        $stack.with(|c| c.borrow_mut().push(entry));
152        Self(())
153      }
154    }
155    impl Drop for $guard {
156      fn drop(&mut self) {
157        $stack.with(|c| {
158          c.borrow_mut().pop();
159        });
160      }
161    }
162  };
163}
164
165ctx_stack_guard!(CtorCtxGuard, CTOR_CTX, CtorCtx);
166ctx_stack_guard!(WhatsitCtxGuard, WHATSIT_CTX, (*mut Whatsit, bool));
167ctx_stack_guard!(ScriptCtxGuard, CURRENT_SCRIPT, (Rc<Engine>, Rc<AST>));

169/// The engine/AST of the innermost running script (for immediate wiring).
170fn current_script() -> std::result::Result<(Rc<Engine>, Rc<AST>), Box<EvalAltResult>> {
171  CURRENT_SCRIPT.with(|c| {
172    c.borrow()
173      .last()
174      .cloned()
175      .ok_or_else(|| Box::<EvalAltResult>::from("registration called outside a script load"))
176  })
177}

179/// Count one installed definition (the `load_script` return value).
180fn note_wired() { WIRED_COUNT.with(|c| *c.borrow_mut() += 1); }

182/// Wire immediately from inside a registration closure: resolve the current
183/// script handles, run the wiring fn, count, map errors to Rhai.
184fn wire_now(
185  wire: impl FnOnce(&Rc<Engine>, &Rc<AST>) -> Result<()>,
186) -> std::result::Result<(), Box<EvalAltResult>> {
187  let (engine, ast) = current_script()?;
188  wire(&engine, &ast).map_err(rhai_err)?;
189  note_wired();
190  Ok(())
191}

193/// Call a DEFERRED Rhai body (a primitive/macro/constructor/ŌĆ” closure) during
194/// digestion, with its `(engine, AST)` pushed onto `CURRENT_SCRIPT` for the
195/// duration. This lets the body itself call a registration (`DefPrimitive`,
196/// `DefMacro`, ŌĆ”): the nested `wire_now` ŌåÆ `current_script()` resolves this
197/// pushed context instead of failing "registration called outside a script
198/// load" (#316). Mirrors Perl, where a `def*` sub is callable from anywhere ŌĆö
199/// script load OR digestion. The RAII guard nests correctly for a body that
200/// runs another body.
201pub(super) fn call_deferred_body<A: rhai::FuncArgs>(
202  engine: &Rc<Engine>,
203  ast: &Rc<AST>,
204  body: &FnPtr,
205  args: A,
206) -> std::result::Result<Dynamic, Box<EvalAltResult>> {
207  let _script_guard = ScriptCtxGuard::new((engine.clone(), ast.clone()));
208  body.call::<Dynamic>(engine, ast, args)
209}

222/// Run `f` with the active constructor's live document and props ŌĆö the SINGLE
223/// audited site that re-mints the `&mut Document` the core published for the
224/// in-flight body (review B1: consolidated here so the `unsafe` has one
225/// documented home instead of being scattered across every proxy op).
226///
227/// RE-ENTRANCY SOUNDNESS (B1 ŌĆö RESOLVED 2026-06-27, verified, not a band-aid):
228/// for a NESTED script construct ŌĆö `\wrap{\myemph{..}}`, where the outer body's
229/// `Document::absorb` re-enters the bridge while its own `&mut self` is still
230/// live ŌĆö this re-mints a *second* `&mut` from a raw pointer while the outer one
231/// is parked on the stack. The earlier B1 review feared this was aliasing UB; a
232/// careful reborrow analysis (and a Miri model ŌĆö see below) shows it is **sound**:
233/// the nested pointer is a reborrow **descendant** of the outer one, not an alias.
234/// The chain is linear: the outer body's `with_doc` reborrows the top pointer ŌåÆ
235/// `Document::absorb(&mut self)` reborrows that ŌåÆ the core threads a reborrow of
236/// `&mut self` down to the nested constructor (`absorb` ŌåÆ `be_absorbed(self)` ŌåÆ
237/// the nested trampoline's `&mut Document`) ŌåÆ that body's `with_doc` reborrows
238/// *it*. Because `CTOR_CTX` is a STACK and `current_ctx()` always returns the
239/// **innermost** published pointer, every `with_doc` re-mints from a genuine
240/// descendant of all parked outer `&mut`s; a descendant reborrow never
241/// invalidates its ancestors. The native path is sound for the same reason
242/// (reborrow down the call chain); the Rhai closure boundary erases the
243/// lifetime, so the round-trip through `*mut` is unavoidable here, but it
244/// preserves the reborrow lineage.
245///
246/// VERIFIED: the exact pattern (thread-local `*mut` stack + RAII guard +
247/// `with_doc` re-mint + nested `absorb` reborrowing down) is modeled over a
248/// libxml2-free `Doc` in `latexml_core::runtime_bindings_reentrancy_model` and
249/// passes Miri under **both Stacked and Tree Borrows, 0 UB** (the real path can't
250/// be Miri-checked directly because `Document` is libxml2/FFI). The earlier
251/// checked-guard "fix" was correctly REJECTED ŌĆö there is no UB to guard against,
252/// and failing the re-entrant op would deadlock `Document::absorb`'s loop, which
253/// requires the nested construction to SUCCEED. Tracked in `docs/SYNC_STATUS.md`
254/// "Open tasks #3 / PR #248 B1".
255fn with_doc<R>(
256  f: impl FnOnce(&mut Document, &SymHashMap<Stored>) -> std::result::Result<R, Box<EvalAltResult>>,
257) -> std::result::Result<R, Box<EvalAltResult>> {
258  let ctx = current_ctx()?;
259  // SAFETY: `ctx.document` is the `&mut Document` the core published for this
260  // body (valid for its duration). For the non-nested case this is the unique
261  // live `&mut`; for the nested case see the CAVEAT above. `ctx.props` is read
262  // only ŌĆö never mutated through this `*const`.
263  let doc = unsafe { &mut *ctx.document };
264  let props = unsafe { &*ctx.props };
265  f(doc, props)
266}

268/// [`with_doc`] for a query that cannot fail, so the registration returns a plain
269/// value instead of a `Result`. Outside a constructor body there is no document
270/// to consult, and a script has nothing useful to do with that fact, so the
271/// caller's `Default` (an empty array, a `()`) is the answer.
272fn with_doc_infallible<R: Default>(f: impl FnOnce(&mut Document) -> R) -> R {
273  with_doc(|doc, _props| Ok(f(doc))).unwrap_or_default()
274}

285/// Unwrap a script-held node for a document operation that SPLICES it, refusing
286/// one that came from `ParseXML`.
287///
288/// A parsed node is still owned by the throwaway document it was parsed into, so
289/// moving it into ours would leave our tree pointing at memory freed the moment
290/// the script drops its handle ŌĆö a use-after-free across FFI, the failure mode
291/// that has already cost this project SIGSEGVs and that `catch_unwind` cannot
292/// contain. `insertXML` is the supported route for parsed markup: it RE-CREATES
293/// the subtree through `append_tree` rather than moving it.
294fn in_tree(
295  node: NodeProxy,
296  method: &str,
297) -> std::result::Result<libxml::tree::Node, Box<EvalAltResult>> {
298  if node.1.is_some() {
299    return Err(Box::<EvalAltResult>::from(format!(
300      "{method}: this node came from ParseXML and belongs to its own parsed \
301       document; insert it with insertXML (which re-creates it) instead of \
302       splicing it in directly"
303    )));
304  }
305  Ok(node.0)
306}

308/// [`in_tree`] over an array argument, rejecting non-node entries by position ŌĆö
309/// never silently skipping one, which would drop part of what a script asked for.
310fn in_tree_array(
311  nodes: rhai::Array,
312  method: &str,
313) -> std::result::Result<Vec<libxml::tree::Node>, Box<EvalAltResult>> {
314  let mut out = Vec::with_capacity(nodes.len());
315  for (i, entry) in nodes.into_iter().enumerate() {
316    let kind = entry.type_name();
317    let proxy = entry.try_cast::<NodeProxy>().ok_or_else(|| {
318      Box::<EvalAltResult>::from(format!(
319        "{method}: entry {i} of the array is a {kind}, not a Node"
320      ))
321    })?;
322    out.push(in_tree(proxy, method)?);
323  }
324  Ok(out)
325}

327/// `Document::rename_node` `expect`s a parent ŌĆö renaming an ORPHAN panics, and a
328/// panic from an untrusted script aborts the conversion (the process, under
329/// `panic = "abort"`). Check first and report it as an ordinary script error.
330fn rename_node_checked(
331  node: NodeProxy,
332  newname: &str,
333  reinsert: bool,
334) -> std::result::Result<Dynamic, Box<EvalAltResult>> {
335  let node = in_tree(node, "renameNode")?;
336  if node.get_parent().is_none() {
337    return Err(Box::<EvalAltResult>::from(
338      "renameNode: the node has no parent (already removed, or a root); \
339       renaming it is not possible",
340    ));
341  }
342  with_doc(|doc, _props| {
343    let renamed = doc.rename_node(node, newname, reinsert).map_err(rhai_err)?;
344    Ok(Dynamic::from(NodeProxy::borrowed(renamed)))
345  })
346}

372/// Rhai proxy for a document-tree node: wraps the clonable libxml handle
373/// directly ŌĆö no lifetimes, no active context. Handed to closure-form matcher
374/// bodies (`DefMathLigature(|node| ŌĆ”)`, read-only) and rewrite-`replace` bodies
375/// (mutable ŌĆö `setAttribute`/`setContent`/`unlink`).
376///
377/// ŌÜĀ LIFETIME FOOTGUN (review m5): the wrapped handle aliases a live C `xmlNode`
378/// owned by the conversion's document tree, and is valid ONLY for the duration
379/// of the body that received it. A script that stows a `NodeProxy` in a variable
380/// and dereferences it AFTER `unlink()` has detached the node, or after the
381/// conversion ends and the tree is freed (`reset_thread_engine`), is a C-level
382/// use-after-free (cf. WISDOM #58). Unlike `DocProxy`/`WhatsitProxy` ŌĆö which
383/// resolve their target through the active-context stack on each call and carry
384/// no pointer ŌĆö `NodeProxy` is the only proxy holding a raw tree handle. Use it
385/// within the body; never retain it across calls or past `unlink()`.
386/// A node may instead OWN its document, which lifts that restriction for parsed
387/// markup: `ParseXML` builds its nodes in a throwaway document, and since libxml's
388/// `Document` is refcounted, holding a clone here keeps the node valid for exactly
389/// as long as the script keeps the handle. One type serves both provenances on
390/// purpose ŌĆö every node method then works on in-tree and parsed nodes alike, and a
391/// new one is a single registration rather than a parallel API.
392#[derive(Clone)]
393pub(crate) struct NodeProxy(
394  pub(crate) libxml::tree::Node,
395  /// `Some` when this handle keeps its own (parsed) document alive; `None` for a
396  /// node inside the conversion's document, whose lifetime the caller guarantees.
397  pub(crate) Option<libxml::tree::Document>,
398);
399
400impl NodeProxy {
401  /// A node inside the live conversion document ŌĆö valid only for the body that
402  /// received it (see the caveat above).
403  pub(crate) fn borrowed(node: libxml::tree::Node) -> Self { Self(node, None) }
404
405  /// A node whose owning document this handle keeps alive.
406  pub(crate) fn owning(node: libxml::tree::Node, doc: libxml::tree::Document) -> Self {
407    Self(node, Some(doc))
408  }
409
410  /// A node reached FROM this one (parent, sibling, child): it lives in the same
411  /// document, so it must inherit the same provenance ŌĆö otherwise walking a
412  /// parsed tree would hand back nodes that no longer keep their owner alive.
413  pub(crate) fn derived(&self, node: libxml::tree::Node) -> Self { Self(node, self.1.clone()) }
414}

477/// Load a binding script: compile once (cached by source), then RUN the script
478/// against the current State. Every `DefŌĆ”` registration wires its native
479/// definition IMMEDIATELY as the script executes (sequential, Perl `.ltxml`
480/// semantics ŌĆö so `DeclareOption` precedes a later `ProcessOptions()`), and
481/// top-level side-effect calls (`RawTeX`, `Let`, `NewCounter`, `AssignValue`,
482/// `DefRegister`, ŌĆ”) re-apply on every load ŌĆö i.e. per conversion, not just
483/// the first one that happened to compile the script. Returns the number of
484/// definitions installed.
485pub fn load_script(src: &str) -> Result<usize> {
486  // Fast path: a previously-compiled script with the same source.
487  let cached = SCRIPT_CACHE.with(|c| c.borrow().get(src).cloned());
488  let cached = match cached {
489    Some(c) => c,
490    None => {
491      let engine = make_engine();
492      let ast = engine
493        .compile(src)
494        .map_err(|e| Error::from(format!("script-binding compile error: {e}")))?;
495      let cs = CachedScript {
496        engine: Rc::new(engine),
497        ast:    Rc::new(ast),
498      };
499      SCRIPT_CACHE.with(|c| {
500        c.borrow_mut().insert(src.to_string(), cs.clone());
501      });
502      cs
503    },
504  };
505
506  // Publish the script handles for the registration closures, run, unpublish
507  // (a stack, so a script loading another script nests correctly). RAII pop on
508  // every exit incl. a panic out of `run_ast` (review M1).
509  let before = WIRED_COUNT.with(|c| *c.borrow());
510  let run = {
511    let _script_guard = ScriptCtxGuard::new((cached.engine.clone(), cached.ast.clone()));
512    cached
513      .engine
514      .run_ast(&cached.ast)
515      .map_err(|e| Error::from(format!("script-binding run error: {e}")))
516  };
517  run?;
518  Ok(WIRED_COUNT.with(|c| *c.borrow()) - before)
519}

521/// Build a partial `Font` from a Rhai map (family/series/shape/ŌĆ” keys ŌĆö the
522/// `fontmap!` analog, shared by `MergeFont` and the `font =>` option).
523fn font_from_rhai_map(opts: Map) -> latexml_core::common::font::Font {
524  let mut font = latexml_core::common::font::Font::default();
525  for (key, val) in opts {
526    let v = dynamic_to_string(val.clone());
527    match key.as_str() {
528      "family" => font.family = Some(v.into()),
529      "series" => font.series = Some(v.into()),
530      "shape" => font.shape = Some(v.into()),
531      "encoding" => font.encoding = Some(v.into()),
532      "language" => font.language = Some(v.into()),
533      "mathstyle" => font.mathstyle = Some(v.into()),
534      "size" => {
535        font.size = val
536          .as_float()
537          .ok()
538          .or_else(|| val.as_int().ok().map(|i| i as f64))
539      },
540      _ => {},
541    }
542  }
543  font
544}

546/// Convert a Rhai object map into a whatsit property map. `Digested` handles
547/// (from `DigestText`, `RefStepCounter`, ŌĆ”) pass through as `Stored::Digested`;
548/// a `Digested` under the `font` key contributes its FONT (the IEEEproof
549/// "titlefont" idiom); everything else lands as its string form.
550fn rhai_map_to_props(map: Map) -> SymHashMap<Stored> {
551  let mut props: SymHashMap<Stored> = SymHashMap::default();
552  for (k, v) in map {
553    match v.clone().try_cast::<Digested>() {
554      Some(d) => {
555        if k.as_str() == "font"
556          && let Ok(Some(f)) = d.get_font()
557        {
558          props.insert("font", Stored::Font(Rc::new(f.into_owned())));
559          continue;
560        }
561        props.insert(k.as_str(), Stored::Digested(d));
562      },
563      _ => {
564        props.insert(k.as_str(), dynamic_to_string(v).into());
565      },
566    }
567  }
568  props
569}

571/// Marshal a counter/whatsit property map back into a Rhai object map (the
572/// inverse of [`rhai_map_to_props`]). The single source shared by
573/// `RefStepCounter`/`RefStepID`/`RefCurrentID`, which inlined this 3├Ś before
574/// (review m4).
575fn props_to_map(props: SymHashMap<Stored>) -> Map {
576  let mut m = Map::new();
577  for (k, v) in props {
578    m.insert(arena::to_string(k).into(), stored_to_dynamic(v));
579  }
580  m
581}

600/// Marshal a `Stored` out to a Rhai value: `Digested` as a live handle,
601/// scalars as scalars, the rest via `Display`.
602fn stored_to_dynamic(v: Stored) -> Dynamic {
603  match v {
604    Stored::Digested(d) => Dynamic::from(d),
605    Stored::String(s) => Dynamic::from(arena::to_string(s)),
606    Stored::Bool(b) => Dynamic::from(b),
607    other => Dynamic::from(other.to_string()),
608  }
609}

611/// Marshal a digested macro argument into a Rhai value. Every argument is passed
612/// as its TeX-source string (authors parse it in-script as needed).
613fn arg_to_dynamic(arg: ArgWrap) -> Dynamic { Dynamic::from(arg.to_string()) }

615/// Read a Rhai return value back as a string (the TeX-source of the expansion).
616fn dynamic_to_string(d: Dynamic) -> String {
617  if d.is_string() {
618    d.into_string().unwrap_or_default()
619  } else {
620    d.to_string()
621  }
622}
