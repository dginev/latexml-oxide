1//! Runtime script bindings via embedded **Rhai** (see `docs/parity/script_bindings_plan.md`).
2//!
3//! Feature-gated behind `runtime-bindings` (ON by default and in the distribution
4//! build — downstream single-binary consumers customize without recompiling; the
5//! module keeps its original name, the feature does not). The ONLY module in the
6//! workspace that embeds Rhai; core/engine/package are untouched. Pure-Rust, no
7//! FFI, no ABI — the reliable successor to the abandoned libperl approach.
8//!
9//! ## Model (the "fine seam", Model A)
10//!
11//! A binding is a Rhai script that calls registration functions (`DefMacro`,
12//! `DefConstructor`, …) with the binding body as a Rhai function value. Each
13//! registration installs a **native** latexml-oxide `Definition` whose body
14//! closure trampolines into the Rhai engine. Prototype parsing, argument
15//! reading, the gullet/stomach/document machinery all stay native; Rhai runs
16//! only the body.
17//!
18//! ## Seams implemented
19//!
20//! * **`DefMacro`** (expandable) — body receives args as strings, returns a string of TeX,
21//!   re-tokenized via `mouth::tokenize_internal` so an expansion to `\textit{x}` faithfully yields
22//!   a control-sequence token.
23//! * **`DefConstructor`** (construction) — two forms. A string XML template, or an imperative body
24//!   that reads like the Perl original: it receives a `Document` **proxy** as its first argument
25//!   (Perl's `$_[0]`) and each digested argument as an opaque handle, e.g. `|document, x| {
26//!   document.openElement("ltx:emph"); document.absorb(x); document.closeElement("ltx:emph"); }`.
27//!   The proxy's methods resolve a thread-local **active context** (the live `&mut Document` +
28//!   props, published for the duration of the call; never borrowed across a re-entrant call).
29//!
30//! ## Lifecycle
31//!
32//! `load_script` compiles the script to an `AST`, runs it once to collect the
33//! registrations, wraps engine + AST in `Rc`, and installs one native definition
34//! per registration — each capturing `Rc<Engine>`/`Rc<AST>` so a body stays
35//! callable for the whole conversion (a deferred `FnPtr::call` needs both alive).

89/// A compiled script, cached by source so the (relatively expensive) Rhai
90/// compile happens once per unique binding. The RUN is per-load (per
91/// conversion) — Perl semantics: each `Def…`/side-effect call installs into
92/// the current State sequentially as the script executes.
93#[derive(Clone)]
94struct CachedScript {
95  engine: Rc<Engine>,
96  ast:    Rc<AST>,
97}

99thread_local! {
100  /// The script currently being RUN (a stack — a script can trigger loading
101  /// another script). `Def…` registration closures grab these handles to wire
102  /// their trampolines immediately, in script order — so e.g. a
103  /// `DeclareOption` is installed before a following `ProcessOptions()` runs.
104  static CURRENT_SCRIPT: RefCell<Vec<(Rc<Engine>, Rc<AST>)>> = const { RefCell::new(Vec::new()) };
105
106  /// Definitions installed by the innermost running script (the load_script
107  /// return value).
108  static WIRED_COUNT: RefCell<usize> = const { RefCell::new(0) };
109
110  static SCRIPT_CACHE: RefCell<HashMap<String, CachedScript>> =
111    RefCell::new(HashMap::new());
112
113  /// Active-context stack for constructor bodies. Each entry publishes the live
114  /// `&mut Document` + props of an in-flight `DefConstructor` call as raw
115  /// pointers, so the `Document`-proxy methods can reach them. A stack (not a
116  /// single slot) so a constructor body that triggers nested construction works.
117  /// (Digested args are passed to the body directly as handles, not via here.)
118  static CTOR_CTX: RefCell<Vec<CtorCtx>> = const { RefCell::new(Vec::new()) };
119
120  /// Active-context stack for whatsit-receiving hook closures (`afterDigest`, …).
121  /// Lets a parameterless hook body reach the in-flight whatsit via `whatsit()` —
122  /// referenced only when needed (the "omit as implied" model).
123  /// Entries are (whatsit pointer, mutable?). Digestion hooks (`afterDigest`,
124  /// `properties`-adjacent) publish mutable whatsits; construction hooks
125  /// (`before/afterConstruct`, `reversion`, `sizer`) publish READ-ONLY ones —
126  /// `setProperty` errors there instead of mutating through a shared ref.
127  static WHATSIT_CTX: RefCell<Vec<(*mut Whatsit, bool)>> = const { RefCell::new(Vec::new()) };
128}

136/// Generate an RAII push/pop guard for one of the active-context stacks.
137///
138/// Why RAII and not manual `push`…`pop()`: Rhai does NOT `catch_unwind` its
139/// native calls, so a panic in a script body (or a deep `RefCell` borrow panic,
140/// incl. the B1 re-entrancy guard) skips a trailing manual `pop()` — leaving a
141/// **stale** entry on the stack. `reset_thread_state` doesn't clear these
142/// `latexml_contrib` thread-locals, so the dangling entry survives the cortex
143/// per-paper `catch_unwind` and leaks into the next paper. A `Drop` guard pops
144/// on EVERY scope exit — normal return, `?` early-return, or unwind. Review M1.
145macro_rules! ctx_stack_guard {
146  ($guard:ident, $stack:ident, $elem:ty) => {
147    #[must_use = "the guard pops on drop; binding it to `_` would pop immediately"]
148    struct $guard(());
149    impl $guard {
150      fn new(entry: $elem) -> Self {
151        $stack.with(|c| c.borrow_mut().push(entry));
152        Self(())
153      }
154    }
155    impl Drop for $guard {
156      fn drop(&mut self) {
157        $stack.with(|c| {
158          c.borrow_mut().pop();
159        });
160      }
161    }
162  };
163}
164
165ctx_stack_guard!(CtorCtxGuard, CTOR_CTX, CtorCtx);
166ctx_stack_guard!(WhatsitCtxGuard, WHATSIT_CTX, (*mut Whatsit, bool));
167ctx_stack_guard!(ScriptCtxGuard, CURRENT_SCRIPT, (Rc<Engine>, Rc<AST>));

169/// The engine/AST of the innermost running script (for immediate wiring).
170fn current_script() -> std::result::Result<(Rc<Engine>, Rc<AST>), Box<EvalAltResult>> {
171  CURRENT_SCRIPT.with(|c| {
172    c.borrow()
173      .last()
174      .cloned()
175      .ok_or_else(|| Box::<EvalAltResult>::from("registration called outside a script load"))
176  })
177}

179/// Count one installed definition (the `load_script` return value).
180fn note_wired() { WIRED_COUNT.with(|c| *c.borrow_mut() += 1); }
181
182/// Wire immediately from inside a registration closure: resolve the current
183/// script handles, run the wiring fn, count, map errors to Rhai.
184fn wire_now(
185  wire: impl FnOnce(&Rc<Engine>, &Rc<AST>) -> Result<()>,
186) -> std::result::Result<(), Box<EvalAltResult>> {
187  let (engine, ast) = current_script()?;
188  wire(&engine, &ast).map_err(rhai_err)?;
189  note_wired();
190  Ok(())
191}

193/// Copy the top active-context out (so we never hold the `CTOR_CTX` borrow
194/// across a Document call that might re-enter the bridge).
195fn current_ctx() -> std::result::Result<CtorCtx, Box<EvalAltResult>> {
196  CTOR_CTX.with(|c| {
197    c.borrow()
198      .last()
199      .copied()
200      .ok_or_else(|| Box::<EvalAltResult>::from("document op called outside a constructor body"))
201  })

204/// Run `f` with the active constructor's live document and props — the SINGLE
205/// audited site that re-mints the `&mut Document` the core published for the
206/// in-flight body (review B1: consolidated here so the `unsafe` has one
207/// documented home instead of being scattered across every proxy op).
208///
209/// RE-ENTRANCY SOUNDNESS (B1 — RESOLVED 2026-06-27, verified, not a band-aid):
210/// for a NESTED script construct — `\wrap{\myemph{..}}`, where the outer body's
211/// `Document::absorb` re-enters the bridge while its own `&mut self` is still
212/// live — this re-mints a *second* `&mut` from a raw pointer while the outer one
213/// is parked on the stack. The earlier B1 review feared this was aliasing UB; a
214/// careful reborrow analysis (and a Miri model — see below) shows it is **sound**:
215/// the nested pointer is a reborrow **descendant** of the outer one, not an alias.
216/// The chain is linear: the outer body's `with_doc` reborrows the top pointer →
217/// `Document::absorb(&mut self)` reborrows that → the core threads a reborrow of
218/// `&mut self` down to the nested constructor (`absorb` → `be_absorbed(self)` →
219/// the nested trampoline's `&mut Document`) → that body's `with_doc` reborrows
220/// *it*. Because `CTOR_CTX` is a STACK and `current_ctx()` always returns the
221/// **innermost** published pointer, every `with_doc` re-mints from a genuine
222/// descendant of all parked outer `&mut`s; a descendant reborrow never
223/// invalidates its ancestors. The native path is sound for the same reason
224/// (reborrow down the call chain); the Rhai closure boundary erases the
225/// lifetime, so the round-trip through `*mut` is unavoidable here, but it
226/// preserves the reborrow lineage.
227///
228/// VERIFIED: the exact pattern (thread-local `*mut` stack + RAII guard +
229/// `with_doc` re-mint + nested `absorb` reborrowing down) is modeled over a
230/// libxml2-free `Doc` in `latexml_core::runtime_bindings_reentrancy_model` and
231/// passes Miri under **both Stacked and Tree Borrows, 0 UB** (the real path can't
232/// be Miri-checked directly because `Document` is libxml2/FFI). The earlier
233/// checked-guard "fix" was correctly REJECTED — there is no UB to guard against,
234/// and failing the re-entrant op would deadlock `Document::absorb`'s loop, which
235/// requires the nested construction to SUCCEED. Tracked in `docs/SYNC_STATUS.md`
236/// "Open tasks #3 / PR #248 B1".
237fn with_doc<R>(
238  f: impl FnOnce(&mut Document, &SymHashMap<Stored>) -> std::result::Result<R, Box<EvalAltResult>>,
239) -> std::result::Result<R, Box<EvalAltResult>> {
240  let ctx = current_ctx()?;
241  // SAFETY: `ctx.document` is the `&mut Document` the core published for this
242  // body (valid for its duration). For the non-nested case this is the unique
243  // live `&mut`; for the nested case see the CAVEAT above. `ctx.props` is read
244  // only — never mutated through this `*const`.
245  let doc = unsafe { &mut *ctx.document };
246  let props = unsafe { &*ctx.props };
247  f(doc, props)
248}

258/// Rhai proxy for a document-tree node: wraps the clonable libxml handle
259/// directly — no lifetimes, no active context. Handed to closure-form matcher
260/// bodies (`DefMathLigature(|node| …)`, read-only) and rewrite-`replace` bodies
261/// (mutable — `setAttribute`/`setContent`/`unlink`).
262///
263/// ⚠ LIFETIME FOOTGUN (review m5): the wrapped handle aliases a live C `xmlNode`
264/// owned by the conversion's document tree, and is valid ONLY for the duration
265/// of the body that received it. A script that stows a `NodeProxy` in a variable
266/// and dereferences it AFTER `unlink()` has detached the node, or after the
267/// conversion ends and the tree is freed (`reset_thread_engine`), is a C-level
268/// use-after-free (cf. WISDOM #58). Unlike `DocProxy`/`WhatsitProxy` — which
269/// resolve their target through the active-context stack on each call and carry
270/// no pointer — `NodeProxy` is the only proxy holding a raw tree handle. Use it
271/// within the body; never retain it across calls or past `unlink()`.
272#[derive(Clone)]
273pub(crate) struct NodeProxy(pub(crate) libxml::tree::Node);

304/// path is in scope (like reading `.sty` from texmf).

310/// Load a binding script: compile once (cached by source), then RUN the script
311/// against the current State. Every `Def…` registration wires its native
312/// definition IMMEDIATELY as the script executes (sequential, Perl `.ltxml`
313/// semantics — so `DeclareOption` precedes a later `ProcessOptions()`), and
314/// top-level side-effect calls (`RawTeX`, `Let`, `NewCounter`, `AssignValue`,
315/// `DefRegister`, …) re-apply on every load — i.e. per conversion, not just
316/// the first one that happened to compile the script. Returns the number of
317/// definitions installed.
318pub fn load_script(src: &str) -> Result<usize> {
319  // Fast path: a previously-compiled script with the same source.
320  let cached = SCRIPT_CACHE.with(|c| c.borrow().get(src).cloned());
321  let cached = match cached {
322    Some(c) => c,
323    None => {
324      let engine = make_engine();
325      let ast = engine
326        .compile(src)
327        .map_err(|e| Error::from(format!("script-binding compile error: {e}")))?;
328      let cs = CachedScript {
329        engine: Rc::new(engine),
330        ast:    Rc::new(ast),
331      };
332      SCRIPT_CACHE.with(|c| {
333        c.borrow_mut().insert(src.to_string(), cs.clone());
334      });
335      cs
336    },
337  };
338
339  // Publish the script handles for the registration closures, run, unpublish
340  // (a stack, so a script loading another script nests correctly). RAII pop on
341  // every exit incl. a panic out of `run_ast` (review M1).
342  let before = WIRED_COUNT.with(|c| *c.borrow());
343  let run = {
344    let _script_guard = ScriptCtxGuard::new((cached.engine.clone(), cached.ast.clone()));
345    cached
346      .engine
347      .run_ast(&cached.ast)
348      .map_err(|e| Error::from(format!("script-binding run error: {e}")))
349  };
350  run?;
351  Ok(WIRED_COUNT.with(|c| *c.borrow()) - before)
352}

354/// Build a partial `Font` from a Rhai map (family/series/shape/… keys — the
355/// `fontmap!` analog, shared by `MergeFont` and the `font =>` option).
356fn font_from_rhai_map(opts: Map) -> latexml_core::common::font::Font {
357  let mut font = latexml_core::common::font::Font::default();
358  for (key, val) in opts {
359    let v = dynamic_to_string(val.clone());
360    match key.as_str() {
361      "family" => font.family = Some(v.into()),
362      "series" => font.series = Some(v.into()),
363      "shape" => font.shape = Some(v.into()),
364      "encoding" => font.encoding = Some(v.into()),
365      "language" => font.language = Some(v.into()),
366      "mathstyle" => font.mathstyle = Some(v.into()),
367      "size" => {
368        font.size = val
369          .as_float()
370          .ok()
371          .or_else(|| val.as_int().ok().map(|i| i as f64))
372      },
373      _ => {},
374    }
375  }
376  font
377}

379/// Convert a Rhai object map into a whatsit property map. `Digested` handles
380/// (from `DigestText`, `RefStepCounter`, …) pass through as `Stored::Digested`;
381/// a `Digested` under the `font` key contributes its FONT (the IEEEproof
382/// "titlefont" idiom); everything else lands as its string form.
383fn rhai_map_to_props(map: Map) -> SymHashMap<Stored> {
384  let mut props: SymHashMap<Stored> = SymHashMap::default();
385  for (k, v) in map {
386    match v.clone().try_cast::<Digested>() {
387      Some(d) => {
388        if k.as_str() == "font"
389          && let Ok(Some(f)) = d.get_font()
390        {
391          props.insert("font", Stored::Font(Rc::new(f.into_owned())));
392          continue;
393        }
394        props.insert(k.as_str(), Stored::Digested(d));
395      },
396      _ => {
397        props.insert(k.as_str(), dynamic_to_string(v).into());
398      },
399    }
400  }
401  props
402}

404/// Marshal a counter/whatsit property map back into a Rhai object map (the
405/// inverse of [`rhai_map_to_props`]). The single source shared by
406/// `RefStepCounter`/`RefStepID`/`RefCurrentID`, which inlined this 3× before
407/// (review m4).
408fn props_to_map(props: SymHashMap<Stored>) -> Map {
409  let mut m = Map::new();
410  for (k, v) in props {
411    m.insert(arena::to_string(k).into(), stored_to_dynamic(v));
412  }
413  m
414}

433/// Marshal a `Stored` out to a Rhai value: `Digested` as a live handle,
434/// scalars as scalars, the rest via `Display`.
435fn stored_to_dynamic(v: Stored) -> Dynamic {
436  match v {
437    Stored::Digested(d) => Dynamic::from(d),
438    Stored::String(s) => Dynamic::from(arena::to_string(s)),
439    Stored::Bool(b) => Dynamic::from(b),
440    other => Dynamic::from(other.to_string()),
441  }
442}

448/// Read a Rhai return value back as a string (the TeX-source of the expansion).
449fn dynamic_to_string(d: Dynamic) -> String {
450  if d.is_string() {
451    d.into_string().unwrap_or_default()
452  } else {
453    d.to_string()
454  }
