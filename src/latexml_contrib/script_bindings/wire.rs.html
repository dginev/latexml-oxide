12/// Install one `DefMacro` registration as a native expandable definition.
13pub(super) fn wire_macro(
14  engine: &Rc<Engine>,
15  ast: &Rc<AST>,
16  proto: &str,
17  body: FnPtr,
18) -> Result<()> {
19  let (cs, paramlist) = parse_prototype(proto, true)?;
20  let cs_name = cs.to_string();
21  let engine = engine.clone();
22  let ast = ast.clone();
23
24  let closure: ExpansionClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<Tokens> {
25    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
26    let ret: Dynamic = call_deferred_body(&engine, &ast, &body, dyn_args)
27      .map_err(|e| Error::from(format!("script macro {cs_name}: {e}")))?;
28    Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
29  });
30
31  def_macro(cs, paramlist, ExpansionBody::Closure(closure), None)?;
32  Ok(())
33}

35/// Install one string-body `DefMacro('\proto', 'replacement')`. The body is TeX
36/// source that becomes the macro's token expansion directly — Perl's most
37/// common `DefMacro($proto, $string)` form, where `$string` is `TokenizeInternal`d
38/// into the replacement (NOT a Rhai closure, so no per-expansion engine call).
39pub(super) fn wire_macro_string(proto: &str, body: &str) -> Result<()> {
40  let (cs, paramlist) = parse_prototype(proto, true)?;
41  def_macro(
42    cs,
43    paramlist,
44    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
45    None,
46  )?;
47  Ok(())
48}

50/// String-body `DefMacro` with a trailing option bag (`DefMacro($proto, $string,
51/// %opts)`). Same token expansion as [`wire_macro_string`], plus the scalar
52/// `ExpandableOptions` from the map.
53pub(super) fn wire_macro_string_opts(proto: &str, body: &str, opts: Map) -> Result<()> {
54  let (cs, paramlist) = parse_prototype(proto, true)?;
55  def_macro(
56    cs,
57    paramlist,
58    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
59    Some(expandable_options_from_map(opts)),
60  )?;
61  Ok(())
62}

64/// The imperative-body replacement closure: publishes the active context, calls
65/// the body as `|document, arg1, …|`, pops the context. Shared by every
66/// constructor form that uses a closure body.
67pub(super) fn closure_replacement(
68  body: FnPtr,
69  engine: Rc<Engine>,
70  ast: Rc<AST>,
71  cs_name: String,
72) -> ReplacementClosure {
73  Rc::new(
74    move |document: &mut Document, args: &Vec<Option<Digested>>, props| -> Result<()> {
75      // RAII: the entry is popped on EVERY exit of this closure, including a
76      // panic out of `body.call` (Rhai doesn't catch_unwind native calls) — so
77      // a script-body panic can't leave a stale CTOR_CTX entry. Review M1.
78      let _ctx_guard = CtorCtxGuard::new(CtorCtx { document, props });
79      // `document` first (Perl's `$_[0]`), then each digested arg as a handle.
80      let mut call_args: Vec<Dynamic> = Vec::with_capacity(args.len() + 1);
81      call_args.push(Dynamic::from(DocProxy));
82      for a in args {
83        call_args.push(match a {
84          Some(d) => Dynamic::from(d.clone()),
85          None => Dynamic::UNIT,
86        });
87      }
88      call_deferred_body(&engine, &ast, &body, call_args)
89        .map(|_| ())
90        .map_err(|e| Error::from(format!("script constructor {cs_name}: {e}")))
91    },
92  )
93}

95/// The string-template replacement closure. The template is parsed **once** here
96/// (at wire time) into the shared `ReplacementOp` AST — the same parser the
97/// compile-time codegen uses (#171) — and the cached AST is interpreted per
98/// invocation. This eliminates the former per-invocation byte-scan and removes
99/// the second, divergent template implementation.
100pub(super) fn template_replacement(template: &str) -> Result<ReplacementClosure> {
101  let ops = Rc::new(replacement::parse_replacement(template)?);
102  Ok(Rc::new(move |document: &mut Document, args, props| {
103    replacement::apply_ops(&ops, document, args, props)
104  }))
105}

107/// Install one `DefConstructor` (imperative body, no options) via the shared
108/// `ConstructorBuilder`.
109pub(super) fn wire_constructor(
110  engine: &Rc<Engine>,
111  ast: &Rc<AST>,
112  proto: &str,
113  body: FnPtr,
114) -> Result<()> {
115  let repl = closure_replacement(body, engine.clone(), ast.clone(), proto.to_string());
116  ConstructorBuilder::new(proto)?.replacement(repl).install()
117}

119/// Install a `DefConstructor` with an option bag (`#{ mode, afterDigest, … }`)
120/// through the shared `ConstructorBuilder` — the *same* builder the
121/// `DefConstructor!` macro targets, so the two front-ends cannot drift. Scalar
122/// options route through the builder's single-source `set_option`; closure
123/// options through its typed setters (here, `after_digest`).
124pub(super) fn wire_constructor_opts(
125  engine: &Rc<Engine>,
126  ast: &Rc<AST>,
127  proto: &str,
128  repl: ConstructorRepl,
129  opts: Map,
130) -> Result<()> {
131  let replacement = match repl {
132    ConstructorRepl::Template(t) => template_replacement(&t)?,
133    ConstructorRepl::Closure(b) => {
134      closure_replacement(b, engine.clone(), ast.clone(), proto.to_string())
135    },
136  };
137  let builder = ConstructorBuilder::new(proto)?.replacement(replacement);
138  apply_opts(builder, opts, engine, ast)?.install()
139}

141/// Install one `DefEnvironment` registration (template or imperative body, with
142/// an option bag — possibly empty) through the shared `EnvironmentBuilder`.
143/// The imperative body reaches `#body` via `document.absorbProperty("body")`.
144pub(super) fn wire_environment(
145  engine: &Rc<Engine>,
146  ast: &Rc<AST>,
147  proto: &str,
148  repl: ConstructorRepl,
149  opts: Map,
150) -> Result<()> {
151  let replacement = match repl {
152    ConstructorRepl::Template(t) => template_replacement(&t)?,
153    ConstructorRepl::Closure(b) => {

156  };

161/// The builder surface the option-bag loop needs — implemented for both core
162/// builders so [`apply_opts`] is written once. (A local trait over the foreign
163/// builder types; new closure options are added in `apply_opts` + one impl line
164/// per builder.)
165pub(super) trait BindingBuilder: Sized {
166  fn set_option(self, key: &str, value: OptionValue) -> Result<Self>;
167  fn after_digest(self, hook: DigestionClosure) -> Self;
168  fn after_digest_begin(self, hook: DigestionClosure) -> Self;
169  fn after_digest_body(self, hook: DigestionClosure) -> Self;
170  fn before_digest(self, hook: BeforeDigestClosure) -> Self;
171  fn before_digest_end(self, hook: BeforeDigestClosure) -> Self;
172  fn before_construct(self, hook: ConstructionClosure) -> Self;
173  fn after_construct(self, hook: ConstructionClosure) -> Self;
174  fn properties(self, props: PropertiesClosure) -> Self;
175  fn reversion(self, rev: Reversion) -> Self;
176  fn font(self, font: FontDirective) -> Self;
177  fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self;
178  // (`install` stays inherent on each builder: call sites get the concrete
179  // type back from `apply_opts`, so a trait method would be dead code.)
180}
181
182macro_rules! impl_binding_builder {
183  ($t:ty) => {
184    impl BindingBuilder for $t {
185      fn set_option(self, key: &str, value: OptionValue) -> Result<Self> {
186        <$t>::set_option(self, key, value)
187      }
188      fn after_digest(self, hook: DigestionClosure) -> Self { <$t>::after_digest(self, hook) }
189      fn after_digest_begin(self, hook: DigestionClosure) -> Self {
190        <$t>::after_digest_begin(self, hook)
191      }
192      fn after_digest_body(self, hook: DigestionClosure) -> Self {
193        <$t>::after_digest_body(self, hook)
194      }
195      fn before_digest(self, hook: BeforeDigestClosure) -> Self { <$t>::before_digest(self, hook) }
196      fn before_digest_end(self, hook: BeforeDigestClosure) -> Self {
197        <$t>::before_digest_end(self, hook)
198      }
199      fn before_construct(self, hook: ConstructionClosure) -> Self {
200        <$t>::before_construct(self, hook)
201      }
202      fn after_construct(self, hook: ConstructionClosure) -> Self {
203        <$t>::after_construct(self, hook)
204      }
205      fn properties(self, props: PropertiesClosure) -> Self { <$t>::properties(self, props) }
206      fn reversion(self, rev: Reversion) -> Self { <$t>::reversion(self, rev) }
207      fn font(self, font: FontDirective) -> Self { <$t>::font(self, font) }
208      fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self {
209        <$t>::sizer(self, sizer)
210      }
211    }
212  };
213}

217/// Apply a Rhai option bag onto a builder: closure options become trampolines
218/// (typed setters), `properties` also accepts a static map, scalars route
219/// through the builder's single-source `set_option`. Shared by the constructor
220/// and environment front-ends.
221pub(super) fn apply_opts<B: BindingBuilder>(
222  mut builder: B,
223  opts: Map,
224  engine: &Rc<Engine>,
225  ast: &Rc<AST>,
226) -> Result<B> {
227  for (key, val) in opts {
228    match val.clone().try_cast::<FnPtr>() {
229      Some(fp) => {
230        // Closure option → typed builder setter (front-end builds the closure).
231        match key.as_str() {
232          "afterDigest" => {
233            builder =
234              builder.after_digest(after_digest_trampoline(fp, engine.clone(), ast.clone()));
235          },
236          "afterDigestBegin" => {
237            builder =
238              builder.after_digest_begin(after_digest_trampoline(fp, engine.clone(), ast.clone()));
239          },
240          "afterDigestBody" => {
241            builder =
242              builder.after_digest_body(after_digest_trampoline(fp, engine.clone(), ast.clone()));
243          },
244          "properties" => {
245            builder = builder.properties(properties_trampoline(fp, engine.clone(), ast.clone()));
246          },
247          "beforeDigest" => {
248            builder =
249              builder.before_digest(before_digest_trampoline(fp, engine.clone(), ast.clone()));
250          },
251          "beforeDigestEnd" => {
252            builder =
253              builder.before_digest_end(before_digest_trampoline(fp, engine.clone(), ast.clone()));
254          },
255          "beforeConstruct" => {
256            builder =
257              builder.before_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
258          },
259          "reversion" => {
260            builder = builder.reversion(Reversion::Closure(reversion_trampoline(
261              fp,
262              engine.clone(),
263              ast.clone(),
264            )));
265          },
266          "sizer" => {
267            builder = builder.sizer(sizer_trampoline(fp, engine.clone(), ast.clone()));
268          },
269          "afterConstruct" => {
270            builder =
271              builder.after_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
272          },
273          // Unknown closure options are silently ignored (forgiving, like Perl
274          // %options; the builder's set_option does the same for scalars).
275          _ => {},
276        }
277      },
278      _ => {
279        if key.as_str() == "properties" && val.is_map() {
280          // Static property map (Perl's `properties => { key => value, … }`).
281          let map = val.cast::<Map>();
282          builder = builder.properties(Rc::new(move |_args| Ok(rhai_map_to_props(map.clone()))));
283        } else if key.as_str() == "reversion" {
284          // String reversion (`reversion => "\\begin{x}#1\\end{x}"`, "" disables).
285          builder = builder.reversion(Reversion::Tokens(mouth::tokenize_internal(
286            &dynamic_to_string(val),
287          )));
288        } else if key.as_str() == "font" && val.is_map() {
289          // Partial-font directive (`font => { family => 'typewriter', … }`).
290          let font = font_from_rhai_map(val.cast::<Map>());
291          builder = builder.font(FontDirective::Asset(Rc::new(font)));
292        } else if let Some(ov) = dynamic_to_option_value(&val) {
293          // Scalar option → the builder's generic, single-source `set_option`.
294          builder = builder.set_option(key.as_str(), ov)?;
295        }
296      },
297    }
298  }
299  Ok(builder)
300}

302/// Build a `before/afterConstruct` trampoline: the body runs with the live
303/// document published as active context (so `document.*` proxy calls work);
304/// the whatsit itself is not yet exposed there (read-only marshaling TBD).
305pub(super) fn construction_trampoline(
306  fp: FnPtr,
307  engine: Rc<Engine>,
308  ast: Rc<AST>,
309) -> ConstructionClosure {
310  Rc::new(
311    move |document: &mut Document, whatsit: &Whatsit| -> Result<()> {
312      // RAII pop on every exit incl. panic (review M1).
313      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
314        document,
315        props: whatsit.get_properties(),
316      });
317      let _: Dynamic = call_deferred_body(&engine, &ast, &fp, (Dynamic::from(DocProxy),))
318        .map_err(|e| Error::from(format!("script afterConstruct: {e}")))?;
319      Ok(())
320    },
321  )
322}

324/// Build an `afterDigest` trampoline: publish the whatsit so a parameterless body
325/// can reach it via `whatsit()`, call the Rhai closure, pop.
326pub(super) fn after_digest_trampoline(
327  fp: FnPtr,
328  engine: Rc<Engine>,
329  ast: Rc<AST>,
330) -> DigestionClosure {
331  Rc::new(move |w: &mut Whatsit| -> Result<Vec<Digested>> {
332    // RAII pop on every exit incl. panic (review M1).
333    let _whatsit_guard = WhatsitCtxGuard::new((w as *mut Whatsit, true));
334    let _: Dynamic = call_deferred_body(&engine, &ast, &fp, ())
335      .map_err(|e| Error::from(format!("script afterDigest: {e}")))?;
336    Ok(Vec::new())
337  })
338}

340/// Build a `beforeDigest` trampoline (Perl `beforeDigest => sub {…}`): runs the
341/// parameterless Rhai closure before the constructor's arguments are digested
342/// (state/font side-effects, e.g. `neutralize_font()`); contributes no boxes.
343pub(super) fn before_digest_trampoline(
344  fp: FnPtr,
345  engine: Rc<Engine>,
346  ast: Rc<AST>,
347) -> BeforeDigestClosure {
348  Rc::new(move || -> Result<Vec<Digested>> {
349    let _: Dynamic = call_deferred_body(&engine, &ast, &fp, ())
350      .map_err(|e| Error::from(format!("script beforeDigest: {e}")))?;
351    Ok(Vec::new())
352  })
353}

355/// Which hook list a runtime `LookupDefinition(cs).push*/unshift*` targets (#321).
356/// Names mirror the `DefConstructor` option-bag hooks 1:1. The construct and body
357/// families exist only on `Constructor` (Perl `DefConstructor`): `Primitive` and
358/// `MathPrimitive` run only the digest pair, so a construct/body push onto them is
359/// a script error, not a silent no-op.
360#[derive(Clone, Copy)]
361pub(super) enum HookFamily {
362  BeforeDigest,
363  AfterDigest,
364  BeforeConstruct,
365  AfterConstruct,
366  AfterDigestBody,
367}
368
369impl HookFamily {
370  fn name(self) -> &'static str {
371    match self {
372      HookFamily::BeforeDigest => "beforeDigest",
373      HookFamily::AfterDigest => "afterDigest",
374      HookFamily::BeforeConstruct => "beforeConstruct",
375      HookFamily::AfterConstruct => "afterConstruct",
376      HookFamily::AfterDigestBody => "afterDigestBody",
377    }
378  }
379}

381/// Append (`at_front = false`, Perl `push`) or prepend (`at_front = true`, Perl
382/// `unshift`) a hook onto a definition's hook list.
383fn insert_hook<T>(list: &mut Vec<T>, at_front: bool, hook: T) {
384  if at_front {
385    list.insert(0, hook);
386  } else {
387    list.push(hook);
388  }
389}

391fn def_kind_name(stored: &Stored) -> &'static str {
392  match stored {
393    Stored::Constructor(_) => "a Constructor",
394    Stored::Primitive(_) => "a Primitive",
395    Stored::MathPrimitive(_) => "a MathPrimitive",
396    Stored::Expandable(_) => "a Macro",
397    Stored::Conditional(_) => "a Conditional",
398    Stored::Register(_) => "a Register",
399    _ => "a non-definition",
400  }
401}

403fn wrong_kind_err(cs: &str, family: HookFamily, stored: &Stored) -> Box<EvalAltResult> {
404  Box::<EvalAltResult>::from(format!(
405    "LookupDefinition({cs}): cannot push a {} hook onto {} — only DefConstructor \
406     carries construct/body hooks; digest hooks need a DefConstructor/DefPrimitive/DefMath",
407    family.name(),
408    def_kind_name(stored),
409  ))
410}

412/// Runtime hook-push behind `LookupDefinition(cs).push<H>(fn)` / `.unshift<H>(fn)`
413/// (#321 — the BookML `push(@{ $$def{afterConstruct} }, sub{…})` shape). Perl
414/// mutates the shared blessed def-hash in place; our installed defs have no
415/// interior mutability, so we clone the CURRENT front definition, splice the
416/// trampolined hook into the matching list, and re-install at GLOBAL scope. Global
417/// install `push_front`s the patched def and clears its lower-frame undo entries
418/// (`state::assign_internal`), so it is the active meaning immediately (sequential
419/// pushes accumulate by re-looking-up) and persists across group exits — a
420/// faithful match to Perl's in-place, globally-visible mutation.
421pub(super) fn push_definition_hook(
422  cs: &str,
423  family: HookFamily,
424  at_front: bool,
425  fp: FnPtr,
426) -> std::result::Result<(), Box<EvalAltResult>> {
427  let (engine, ast) = current_script()?;
428  let token = latexml_core::T_CS!(cs);
429  let stored = lookup_definition_stored(&token)
430    .map_err(rhai_err)?
431    .ok_or_else(|| {
432      Box::<EvalAltResult>::from(format!(
433        "LookupDefinition({cs}).push{}: {cs} is not defined",
434        family.name()
435      ))
436    })?;
437
438  // Build the correctly-typed trampoline for THIS family once, then splice it into
439  // the matching list of whichever concrete definition kind is installed (only one
440  // match arm runs, so the trampoline is moved into it).
441  match family {
442    HookFamily::BeforeDigest => {
443      let hook = before_digest_trampoline(fp, engine, ast);
444      match stored {
445        Stored::Constructor(rc) => {
446          let mut d = (*rc).clone();
447          insert_hook(&mut d.before_digest, at_front, hook);
448          install_definition(d, Some(Scope::Global));
449        },
450        Stored::Primitive(rc) => {
451          let mut d = (*rc).clone();
452          insert_hook(&mut d.before_digest, at_front, hook);
453          install_definition(d, Some(Scope::Global));
454        },
455        Stored::MathPrimitive(rc) => {
456          let mut d = (*rc).clone();
457          insert_hook(&mut d.options.before_digest, at_front, hook);
458          install_definition(d, Some(Scope::Global));
459        },
460        other => return Err(wrong_kind_err(cs, family, &other)),
461      }
462    },
463    HookFamily::AfterDigest => {
464      let hook = after_digest_trampoline(fp, engine, ast);
465      match stored {
466        Stored::Constructor(rc) => {
467          let mut d = (*rc).clone();
468          insert_hook(&mut d.after_digest, at_front, hook);
469          install_definition(d, Some(Scope::Global));
470        },
471        Stored::Primitive(rc) => {
472          let mut d = (*rc).clone();
473          insert_hook(&mut d.after_digest, at_front, hook);
474          install_definition(d, Some(Scope::Global));
475        },
476        Stored::MathPrimitive(rc) => {
477          let mut d = (*rc).clone();
478          insert_hook(&mut d.options.after_digest, at_front, hook);
479          install_definition(d, Some(Scope::Global));
480        },
481        other => return Err(wrong_kind_err(cs, family, &other)),
482      }
483    },
484    HookFamily::BeforeConstruct => {
485      let hook = construction_trampoline(fp, engine, ast);
486      match stored {
487        Stored::Constructor(rc) => {
488          let mut d = (*rc).clone();
489          insert_hook(&mut d.before_construct, at_front, hook);
490          install_definition(d, Some(Scope::Global));
491        },
492        other => return Err(wrong_kind_err(cs, family, &other)),
493      }
494    },
495    HookFamily::AfterConstruct => {
496      let hook = construction_trampoline(fp, engine, ast);
497      match stored {
498        Stored::Constructor(rc) => {
499          let mut d = (*rc).clone();
500          insert_hook(&mut d.after_construct, at_front, hook);
501          install_definition(d, Some(Scope::Global));
502        },
503        other => return Err(wrong_kind_err(cs, family, &other)),
504      }
505    },
506    HookFamily::AfterDigestBody => {
507      let hook = after_digest_trampoline(fp, engine, ast);
508      match stored {
509        Stored::Constructor(rc) => {
510          let mut d = (*rc).clone();
511          insert_hook(&mut d.after_digest_body, at_front, hook);
512          install_definition(d, Some(Scope::Global));
513        },
514        other => return Err(wrong_kind_err(cs, family, &other)),
515      }
516    },
517  }
518  Ok(())
519}

521/// Build a `properties` trampoline (Perl `properties => sub {…}`): the Rhai
522/// closure receives each digested argument as its TeX-source string (`()` for an
523/// omitted optional) and returns a map; its entries become the whatsit's
524/// properties as string `Stored`s, ready for the template's `#name` holes.
525pub(super) fn properties_trampoline(
526  fp: FnPtr,
527  engine: Rc<Engine>,
528  ast: Rc<AST>,
529) -> PropertiesClosure {
530  Rc::new(
531    move |args: &Vec<Option<Digested>>| -> Result<SymHashMap<Stored>> {
532      let dyn_args: Vec<Dynamic> = args
533        .iter()
534        .map(|a| match a {
535          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
536          None => Dynamic::UNIT,
537        })
538        .collect();
539      let ret: Dynamic = call_deferred_body(&engine, &ast, &fp, dyn_args)
540        .map_err(|e| Error::from(format!("script properties: {e}")))?;
541      if ret.is_unit() {
542        Ok(SymHashMap::default())
543      } else if ret.is_map() {
544        Ok(rhai_map_to_props(ret.cast::<Map>()))
545      } else {
546        Err(Error::from(
547          "script properties: body must return a map (or unit)",
548        ))
549      }
550    },
551  )
552}

554/// Map a Rhai scalar `Dynamic` to a builder `OptionValue` (string/bool/int).
555pub(super) fn dynamic_to_option_value(v: &Dynamic) -> Option<OptionValue> {
556  if v.is_string() {
557    Some(OptionValue::Str(
558      v.clone().into_string().unwrap_or_default(),
559    ))
560  } else if v.is_bool() {
561    v.as_bool().ok().map(OptionValue::Bool)
562  } else {
563    v.as_int().ok().map(OptionValue::Int)
564  }
565}

567/// Coerce a Rhai scalar to a bool with the SAME policy as the builder's
568/// `OptionValue::into_bool` (bool as-is, int ≠ 0, non-empty string = true) —
569/// the single shared coercion. Review m4: the `*_options_from_map` loops below
570/// previously used `as_bool().unwrap_or(false)`, which silently coerced an int
571/// (`bounded: 1`) or string (`protected: "yes"`) option to `false`, diverging
572/// from the `OptionValue`/`set_option` path that the same bindings reach via
573/// scalar options.
574fn dynamic_to_bool(v: &Dynamic) -> bool {
575  dynamic_to_option_value(v)
576    .and_then(|ov| ov.into_bool().ok())
577    .unwrap_or(false)
578}

894/// `DefKeyVal(keyset, key, vtype, default, #{prefix, kind, macroprefix,
895/// choices})` — the option-bag form (xkeyval-style keys with a prefix and a
896/// kind). The 3-/4-arg forms (plain `KV`-prefixed keys) are registered inline in
897/// `engine.rs`; this covers the richer xkeyval declarations (`myxkeyval`'s
898/// `prefix`/`kind`/`choices`). An empty `default` maps to Perl `undef`.
899/// `mismatch` (a warning closure) is intentionally unsupported — the fixtures
900/// that set it never exercise it, matching the prior compiled binding.
901pub(super) fn keyval_define_from_map(
902  keyset: &str,
903  key: &str,
904  vtype: &str,
905  default: &str,
906  opts: Map,
907) -> Result<()> {
908  use latexml_core::keyval::{self, KeyvalConfig};
909  let mut prefix = String::from("KV");
910  let mut kind: Option<String> = None;
911  let mut macroprefix: Option<String> = None;
912  // `choices` requires `Vec<&'static str>`; key definitions persist for the
913  // program lifetime, so leaking the split entries is acceptable (mirrors the
914  // prior `xkvview_sty`/`myxkeyval_sty` compiled bindings).
915  let mut choices: Vec<&'static str> = Vec::new();
916  for (k, v) in opts {
917    match k.as_str() {
918      "prefix" => prefix = dynamic_to_string(v),

922        if let Some(arr) = v.try_cast::<rhai::Array>() {
923          choices = arr
924            .into_iter()
925            .map(|d| &*Box::leak(dynamic_to_string(d).into_boxed_str()))
926            .collect();
927        }
928      },
929      _ => {},
930    }
931  }
932  keyval::define(KeyvalConfig {
933    prefix: &prefix,
934    keyset,
935    key,
936    vtype,
937    default: if default.is_empty() {
938      None
939    } else {
940      Some(default)
941    },
942    kind: kind.as_deref(),
943    macroprefix: macroprefix.as_deref(),
944    choices,
945    ..KeyvalConfig::default()
946  })
947}

958/// Build a closure-form `reversion` trampoline: the body runs with the whatsit
959/// published READ-ONLY (argString/propertyString available) and each digested
960/// arg as its TeX-source string; its returned string is the reversion TeX.
961pub(super) fn reversion_trampoline(
962  fp: FnPtr,
963  engine: Rc<Engine>,
964  ast: Rc<AST>,
965) -> latexml_core::definition::DigestedReversionClosure {
966  Rc::new(
967    move |w: &Whatsit, args: &Vec<Option<Digested>>| -> Result<Tokens> {
968      // RAII pop on every exit incl. panic (review M1). Whatsit is READ-ONLY here.
969      let _whatsit_guard = WhatsitCtxGuard::new((w as *const Whatsit as *mut Whatsit, false));
970      let dyn_args: Vec<Dynamic> = args
971        .iter()
972        .map(|a| match a {
973          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
974          None => Dynamic::UNIT,
975        })
976        .collect();
977      let ret = call_deferred_body(&engine, &ast, &fp, dyn_args)
978        .map_err(|e| Error::from(format!("script reversion: {e}")))?;
979      Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
980    },
981  )
982}

1056/// The `DefMathLigature!` data-form lowering: walk `ntomatch` preceding
1057/// `ltx:XMTok` siblings matching the pattern (reversed), then rewrite to
1058/// `replacement` with the given role/name/meaning attributes.
1059pub(super) fn def_math_ligature_impl(pattern: &str, replacement: &str, opts: Map) {
1060  use latexml_core::ligature::{Ligature, MathLigatureOptions};
1061  let mut attr = MathLigatureOptions::default();
1062  for (key, val) in opts {
1063    let v = dynamic_to_string(val);
1064    match key.as_str() {
1065      "role" => attr.role = Some(v),
1066      "name" => attr.name = Some(v),
1067      "meaning" => attr.meaning = Some(v),
1068      _ => {},
1069    }
1070  }
1071  let chars: Vec<char> = pattern.chars().rev().collect();
1072  let ntomatch = chars.len();
1073  let replacement = replacement.to_string();
1074  let matcher: Option<latexml_core::ligature::LigatureMatcher> = Some(Rc::new(
1075    move |_document: &mut Document, node_opt: &mut libxml::tree::Node| {
1076      let mut node: libxml::tree::Node;
1077      let mut node_mut = node_opt;
1078      for c in chars.iter() {
1079        if latexml_core::common::model::with_node_qname(node_mut, |qname| qname != "ltx:XMTok")
1080          || node_mut.get_content() != c.to_string()
1081        {
1082          return Ok(None);
1083        }
1084        match node_mut.get_prev_sibling() {
1085          Some(sibling) => {
1086            node = sibling;
1087            node_mut = &mut node;
1088          },
1089          _ => {
1090            return Ok(None);
1091          },
1092        }
1093      }
1094      if ntomatch > 0 {
1095        Ok(Some((ntomatch, replacement.clone(), attr.clone())))
1096      } else {
1097        Ok(None)
1098      }
1099    },
1100  ));
1101  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
1102    id: latexml_core::state::generate_ligature_id(),
1103    matcher,
1104    code: None,
1105    font_test: None,
1106    regex: None,
1107  }]);

1119  let mut o = RewriteOptions {
1120    is_math: kind == "math",
1121    ..RewriteOptions::default()
1122  };
1123  for (key, val) in opts {
1124    match key.as_str() {
1125      "label" => o.label = Some(dynamic_to_string(val)),
1126      "xpath" => o.xpath = Some(dynamic_to_string(val)),
1127      "select" => o.select = Some(dynamic_to_string(val)),
1128      "select_count" => o.select_count = val.as_int().ok().map(|i| i as usize),
1129      "attributes" => {
1130        if val.is_map() {
1131          let mut m = rustc_hash::FxHashMap::default();
1132          for (k, v) in val.cast::<Map>() {
1133            m.insert(k.to_string(), dynamic_to_string(v));
1134          }
1135          o.attributes_map = Some(m);
1136        } else {
1137          o.attributes = Some(dynamic_to_string(val));
1138        }
1139      },
1140      "regexp" => o.regexp = Some(dynamic_to_string(val)),
1141      "match" => o.on_match = Some(mouth::tokenize_internal(&dynamic_to_string(val))),
1142      "scope" => o.scope = scope_of(&dynamic_to_string(val)),
1143      _ => {},
1144    }
1145  }
1146  o
1147}

1160/// `DefRewrite(opts, |nodes| …)` — the `replace` closure form. NB the core
1161/// Replace clause DETACHES the matched nodes and splices in whatever the
1162/// closure inserts (replace-by-reinsertion, Perl Rewrite.pm L122): the Rhai
1163/// body receives the detached nodes as proxies; document-insertion context
1164/// for the body is the pending follow-up (e2e lands with it).
1165pub(super) fn wire_rewrite_replace(
1166  kind: &str,
1167  engine: &Rc<Engine>,
1168  ast: &Rc<AST>,
1169  opts: Map,
1170  replace: FnPtr,
1171) -> Result<()> {
1172  use latexml_core::rewrite::Rewrite;
1173  // Same superset parser as the data form (review M3) — so the replace form now
1174  // honors `attributes` too, and neither form silently drops the other's key.
1175  let mut o = parse_rewrite_options(kind, opts);
1176  // The Replace clause consumes `select_count` matched nodes (native
1177  // replace-rewrites always set it); default to 1 when the bag didn't.
1178  o.select_count.get_or_insert(1);
1179  let engine = engine.clone();
1180  let ast = ast.clone();
1181  // The core Replace clause positions the document at the splice parent before
1182  // calling us; publish it so the body's `document.*` proxy inserts land there.
1183  // The props map carries a default text font — at finalization there is no
1184  // box_to_absorb for absorb_string to fall back on. Held by an `Rc` MOVED into
1185  // the replace closure (NOT `Box::leak`ed): the map lives exactly as long as
1186  // the rule and is freed when the rule is dropped — `load_script` runs per
1187  // conversion, so the old leak grew one map per replace-rule PER PAPER in a
1188  // long-lived worker (review M2). `CtorCtx.props` is `*const`, and the captured
1189  // `Rc` keeps the target alive for the whole call.
1190  let rewrite_props: Rc<SymHashMap<Stored>> = {
1191    let mut m = SymHashMap::default();
1192    m.insert(
1193      "font",
1194      Stored::Font(Rc::new(latexml_core::common::font::Font::text_default())),
1195    );
1196    Rc::new(m)
1197  };
1198  o.replace = Some(Rc::new(
1199    move |document: &mut Document, nodes: Vec<&mut libxml::tree::Node>| -> Result<()> {
1200      let arr: rhai::Array = nodes
1201        .into_iter()
1202        .map(|n| Dynamic::from(NodeProxy(n.clone())))
1203        .collect();
1204      // RAII pop on every exit incl. panic (review M1).
1205      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
1206        document,
1207        props: Rc::as_ptr(&rewrite_props),
1208      });
1209      let _: Dynamic = call_deferred_body(&engine, &ast, &replace, (Dynamic::from(DocProxy), arr))
1210        .map_err(|e| Error::from(format!("script rewrite replace: {e}")))?;
1211      Ok(())
1212    },
1213  ));
1214  latexml_core::state::push_value("DOCUMENT_REWRITE_RULES", Rewrite::new(kind, o))?;
1215  Ok(())
1216}

1218/// Closure-form `DefMathLigature` (`matcher => sub[document,node]`): the Rhai
1219/// body receives a read-only `Node` proxy; UNIT means no match, a map
1220/// `#{ n, replacement, role?, name?, meaning? }` reports one.
1221pub(super) fn wire_math_ligature_matcher(
1222  engine: &Rc<Engine>,
1223  ast: &Rc<AST>,
1224  matcher: FnPtr,
1225) -> Result<()> {
1226  use latexml_core::ligature::{Ligature, MathLigatureOptions};
1227  let engine = engine.clone();
1228  let ast = ast.clone();
1229  let rust_matcher: latexml_core::ligature::LigatureMatcher = Rc::new(
1230    move |_document: &mut Document, node: &mut libxml::tree::Node| {
1231      let ret: Dynamic = call_deferred_body(
1232        &engine,
1233        &ast,
1234        &matcher,
1235        (Dynamic::from(NodeProxy(node.clone())),),
1236      )
1237      .map_err(|e| Error::from(format!("script math-ligature matcher: {e}")))?;
1238      if !ret.is_map() {
1239        return Ok(None);
1240      }
1241      let m = ret.cast::<Map>();
1242      let n = m.get("n").and_then(|v| v.as_int().ok()).unwrap_or(0) as usize;
1243      let replacement = m
1244        .get("replacement")
1245        .map(|v| dynamic_to_string(v.clone()))
1246        .unwrap_or_default();
1247      if n == 0 {
1248        return Ok(None);
1249      }
1250      let attr = MathLigatureOptions {
1251        role:    m.get("role").map(|v| dynamic_to_string(v.clone())),
1252        name:    m.get("name").map(|v| dynamic_to_string(v.clone())),
1253        meaning: m.get("meaning").map(|v| dynamic_to_string(v.clone())),
1254      };
1255      Ok(Some((n, replacement, attr)))
1256    },
1257  );
1258  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
1259    id:        latexml_core::state::generate_ligature_id(),
1260    matcher:   Some(rust_matcher),
1261    code:      None,
1262    font_test: None,
1263    regex:     None,
1264  }]);
1265  Ok(())
1266}
