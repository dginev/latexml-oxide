9/// Install one `DefMacro` registration as a native expandable definition.
10pub(super) fn wire_macro(
11  engine: &Rc<Engine>,
12  ast: &Rc<AST>,
13  proto: &str,
14  body: FnPtr,
15) -> Result<()> {
16  let (cs, paramlist) = parse_prototype(proto, true)?;
17  let cs_name = cs.to_string();
18  let engine = engine.clone();
19  let ast = ast.clone();
20
21  let closure: ExpansionClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<Tokens> {
22    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
23    let ret: Dynamic = body
24      .call::<Dynamic>(&engine, &ast, dyn_args)
25      .map_err(|e| Error::from(format!("script macro {cs_name}: {e}")))?;
26    Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
27  });
28
29  def_macro(cs, paramlist, ExpansionBody::Closure(closure), None)?;
30  Ok(())
31}

33/// Install one string-body `DefMacro('\proto', 'replacement')`. The body is TeX
34/// source that becomes the macro's token expansion directly — Perl's most
35/// common `DefMacro($proto, $string)` form, where `$string` is `TokenizeInternal`d
36/// into the replacement (NOT a Rhai closure, so no per-expansion engine call).
37pub(super) fn wire_macro_string(proto: &str, body: &str) -> Result<()> {
38  let (cs, paramlist) = parse_prototype(proto, true)?;
39  def_macro(
40    cs,
41    paramlist,
42    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
43    None,
44  )?;
45  Ok(())
46}

48/// String-body `DefMacro` with a trailing option bag (`DefMacro($proto, $string,
49/// %opts)`). Same token expansion as [`wire_macro_string`], plus the scalar
50/// `ExpandableOptions` from the map.
51pub(super) fn wire_macro_string_opts(proto: &str, body: &str, opts: Map) -> Result<()> {
52  let (cs, paramlist) = parse_prototype(proto, true)?;
53  def_macro(
54    cs,
55    paramlist,
56    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
57    Some(expandable_options_from_map(opts)),
58  )?;
59  Ok(())
60}

62/// The imperative-body replacement closure: publishes the active context, calls
63/// the body as `|document, arg1, …|`, pops the context. Shared by every
64/// constructor form that uses a closure body.
65pub(super) fn closure_replacement(
66  body: FnPtr,
67  engine: Rc<Engine>,
68  ast: Rc<AST>,
69  cs_name: String,
70) -> ReplacementClosure {
71  Rc::new(
72    move |document: &mut Document, args: &Vec<Option<Digested>>, props| -> Result<()> {
73      // RAII: the entry is popped on EVERY exit of this closure, including a
74      // panic out of `body.call` (Rhai doesn't catch_unwind native calls) — so
75      // a script-body panic can't leave a stale CTOR_CTX entry. Review M1.
76      let _ctx_guard = CtorCtxGuard::new(CtorCtx { document, props });
77      // `document` first (Perl's `$_[0]`), then each digested arg as a handle.
78      let mut call_args: Vec<Dynamic> = Vec::with_capacity(args.len() + 1);
79      call_args.push(Dynamic::from(DocProxy));
80      for a in args {
81        call_args.push(match a {
82          Some(d) => Dynamic::from(d.clone()),
83          None => Dynamic::UNIT,
84        });
85      }
86      body
87        .call::<Dynamic>(&engine, &ast, call_args)
88        .map(|_| ())
89        .map_err(|e| Error::from(format!("script constructor {cs_name}: {e}")))
90    },
91  )
92}

94/// The string-template replacement closure. The template is parsed **once** here
95/// (at wire time) into the shared `ReplacementOp` AST — the same parser the
96/// compile-time codegen uses (#171) — and the cached AST is interpreted per
97/// invocation. This eliminates the former per-invocation byte-scan and removes
98/// the second, divergent template implementation.
99pub(super) fn template_replacement(template: &str) -> Result<ReplacementClosure> {
100  let ops = Rc::new(replacement::parse_replacement(template)?);
101  Ok(Rc::new(move |document: &mut Document, args, props| {
102    replacement::apply_ops(&ops, document, args, props)
103  }))
104}

106/// Install one `DefConstructor` (imperative body, no options) via the shared
107/// `ConstructorBuilder`.
108pub(super) fn wire_constructor(
109  engine: &Rc<Engine>,
110  ast: &Rc<AST>,
111  proto: &str,
112  body: FnPtr,
113) -> Result<()> {
114  let repl = closure_replacement(body, engine.clone(), ast.clone(), proto.to_string());
115  ConstructorBuilder::new(proto)?.replacement(repl).install()
116}

118/// Install a `DefConstructor` with an option bag (`#{ mode, afterDigest, … }`)
119/// through the shared `ConstructorBuilder` — the *same* builder the
120/// `DefConstructor!` macro targets, so the two front-ends cannot drift. Scalar
121/// options route through the builder's single-source `set_option`; closure
122/// options through its typed setters (here, `after_digest`).
123pub(super) fn wire_constructor_opts(
124  engine: &Rc<Engine>,
125  ast: &Rc<AST>,
126  proto: &str,
127  repl: ConstructorRepl,
128  opts: Map,
129) -> Result<()> {
130  let replacement = match repl {
131    ConstructorRepl::Template(t) => template_replacement(&t)?,
132    ConstructorRepl::Closure(b) => {
133      closure_replacement(b, engine.clone(), ast.clone(), proto.to_string())
134    },
135  };
136  let builder = ConstructorBuilder::new(proto)?.replacement(replacement);
137  apply_opts(builder, opts, engine, ast)?.install()
138}

143pub(super) fn wire_environment(
144  engine: &Rc<Engine>,
145  ast: &Rc<AST>,
146  proto: &str,
147  repl: ConstructorRepl,

149) -> Result<()> {

160/// The builder surface the option-bag loop needs — implemented for both core
161/// builders so [`apply_opts`] is written once. (A local trait over the foreign
162/// builder types; new closure options are added in `apply_opts` + one impl line
163/// per builder.)
164pub(super) trait BindingBuilder: Sized {
165  fn set_option(self, key: &str, value: OptionValue) -> Result<Self>;
166  fn after_digest(self, hook: DigestionClosure) -> Self;
167  fn after_digest_begin(self, hook: DigestionClosure) -> Self;
168  fn before_digest(self, hook: BeforeDigestClosure) -> Self;
169  fn before_digest_end(self, hook: BeforeDigestClosure) -> Self;
170  fn before_construct(self, hook: ConstructionClosure) -> Self;
171  fn after_construct(self, hook: ConstructionClosure) -> Self;
172  fn properties(self, props: PropertiesClosure) -> Self;
173  fn reversion(self, rev: Reversion) -> Self;
174  fn font(self, font: FontDirective) -> Self;
175  fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self;
176  // (`install` stays inherent on each builder: call sites get the concrete
177  // type back from `apply_opts`, so a trait method would be dead code.)
178}
179
180macro_rules! impl_binding_builder {
181  ($t:ty) => {
182    impl BindingBuilder for $t {
183      fn set_option(self, key: &str, value: OptionValue) -> Result<Self> {
184        <$t>::set_option(self, key, value)
185      }
186      fn after_digest(self, hook: DigestionClosure) -> Self { <$t>::after_digest(self, hook) }
187      fn after_digest_begin(self, hook: DigestionClosure) -> Self {
188        <$t>::after_digest_begin(self, hook)
189      }
190      fn before_digest(self, hook: BeforeDigestClosure) -> Self { <$t>::before_digest(self, hook) }
191      fn before_digest_end(self, hook: BeforeDigestClosure) -> Self {
192        <$t>::before_digest_end(self, hook)
193      }
194      fn before_construct(self, hook: ConstructionClosure) -> Self {
195        <$t>::before_construct(self, hook)
196      }
197      fn after_construct(self, hook: ConstructionClosure) -> Self {
198        <$t>::after_construct(self, hook)
199      }
200      fn properties(self, props: PropertiesClosure) -> Self { <$t>::properties(self, props) }
201      fn reversion(self, rev: Reversion) -> Self { <$t>::reversion(self, rev) }
202      fn font(self, font: FontDirective) -> Self { <$t>::font(self, font) }
203      fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self {
204        <$t>::sizer(self, sizer)
205      }
206    }
207  };
208}

212/// Apply a Rhai option bag onto a builder: closure options become trampolines
213/// (typed setters), `properties` also accepts a static map, scalars route
214/// through the builder's single-source `set_option`. Shared by the constructor
215/// and environment front-ends.
216pub(super) fn apply_opts<B: BindingBuilder>(
217  mut builder: B,
218  opts: Map,
219  engine: &Rc<Engine>,
220  ast: &Rc<AST>,
221) -> Result<B> {
222  for (key, val) in opts {
223    match val.clone().try_cast::<FnPtr>() {
224      Some(fp) => {
225        // Closure option → typed builder setter (front-end builds the closure).
226        match key.as_str() {
227          "afterDigest" => {
228            builder =
229              builder.after_digest(after_digest_trampoline(fp, engine.clone(), ast.clone()));
230          },
231          "afterDigestBegin" => {
232            builder =
233              builder.after_digest_begin(after_digest_trampoline(fp, engine.clone(), ast.clone()));
234          },
235          "properties" => {
236            builder = builder.properties(properties_trampoline(fp, engine.clone(), ast.clone()));
237          },
238          "beforeDigest" => {
239            builder =
240              builder.before_digest(before_digest_trampoline(fp, engine.clone(), ast.clone()));
241          },
242          "beforeDigestEnd" => {
243            builder =
244              builder.before_digest_end(before_digest_trampoline(fp, engine.clone(), ast.clone()));
245          },
246          "beforeConstruct" => {
247            builder =
248              builder.before_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
249          },
250          "reversion" => {
251            builder = builder.reversion(Reversion::Closure(reversion_trampoline(
252              fp,
253              engine.clone(),
254              ast.clone(),
255            )));
256          },
257          "sizer" => {
258            builder = builder.sizer(sizer_trampoline(fp, engine.clone(), ast.clone()));
259          },
260          "afterConstruct" => {
261            builder =
262              builder.after_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
263          },
264          // Unknown closure options are silently ignored (forgiving, like Perl
265          // %options; the builder's set_option does the same for scalars).
266          _ => {},
267        }
268      },
269      _ => {
270        if key.as_str() == "properties" && val.is_map() {
271          // Static property map (Perl's `properties => { key => value, … }`).
272          let map = val.cast::<Map>();
273          builder = builder.properties(Rc::new(move |_args| Ok(rhai_map_to_props(map.clone()))));
274        } else if key.as_str() == "reversion" {
275          // String reversion (`reversion => "\\begin{x}#1\\end{x}"`, "" disables).
276          builder = builder.reversion(Reversion::Tokens(mouth::tokenize_internal(
277            &dynamic_to_string(val),
278          )));
279        } else if key.as_str() == "font" && val.is_map() {
280          // Partial-font directive (`font => { family => 'typewriter', … }`).
281          let font = font_from_rhai_map(val.cast::<Map>());
282          builder = builder.font(FontDirective::Asset(Rc::new(font)));
283        } else if let Some(ov) = dynamic_to_option_value(&val) {
284          // Scalar option → the builder's generic, single-source `set_option`.
285          builder = builder.set_option(key.as_str(), ov)?;
286        }
287      },
288    }
289  }
290  Ok(builder)
291}

293/// Build a `before/afterConstruct` trampoline: the body runs with the live
294/// document published as active context (so `document.*` proxy calls work);
295/// the whatsit itself is not yet exposed there (read-only marshaling TBD).
296pub(super) fn construction_trampoline(
297  fp: FnPtr,
298  engine: Rc<Engine>,
299  ast: Rc<AST>,
300) -> ConstructionClosure {
301  Rc::new(
302    move |document: &mut Document, whatsit: &Whatsit| -> Result<()> {
303      // RAII pop on every exit incl. panic (review M1).
304      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
305        document,
306        props: whatsit.get_properties(),
307      });
308      let _: Dynamic = fp
309        .call::<Dynamic>(&engine, &ast, (Dynamic::from(DocProxy),))
310        .map_err(|e| Error::from(format!("script afterConstruct: {e}")))?;
311      Ok(())
312    },
313  )
314}

316/// Build an `afterDigest` trampoline: publish the whatsit so a parameterless body
317/// can reach it via `whatsit()`, call the Rhai closure, pop.
318pub(super) fn after_digest_trampoline(
319  fp: FnPtr,
320  engine: Rc<Engine>,
321  ast: Rc<AST>,
322) -> DigestionClosure {
323  Rc::new(move |w: &mut Whatsit| -> Result<Vec<Digested>> {
324    // RAII pop on every exit incl. panic (review M1).
325    let _whatsit_guard = WhatsitCtxGuard::new((w as *mut Whatsit, true));
326    let _: Dynamic = fp
327      .call::<Dynamic>(&engine, &ast, ())
328      .map_err(|e| Error::from(format!("script afterDigest: {e}")))?;
329    Ok(Vec::new())
330  })
331}

333/// Build a `beforeDigest` trampoline (Perl `beforeDigest => sub {…}`): runs the
334/// parameterless Rhai closure before the constructor's arguments are digested
335/// (state/font side-effects, e.g. `neutralize_font()`); contributes no boxes.
336pub(super) fn before_digest_trampoline(
337  fp: FnPtr,
338  engine: Rc<Engine>,
339  ast: Rc<AST>,
340) -> BeforeDigestClosure {
341  Rc::new(move || -> Result<Vec<Digested>> {
342    let _: Dynamic = fp
343      .call::<Dynamic>(&engine, &ast, ())
344      .map_err(|e| Error::from(format!("script beforeDigest: {e}")))?;
345    Ok(Vec::new())
346  })
347}

349/// Build a `properties` trampoline (Perl `properties => sub {…}`): the Rhai
350/// closure receives each digested argument as its TeX-source string (`()` for an
351/// omitted optional) and returns a map; its entries become the whatsit's
352/// properties as string `Stored`s, ready for the template's `#name` holes.
353pub(super) fn properties_trampoline(
354  fp: FnPtr,
355  engine: Rc<Engine>,
356  ast: Rc<AST>,
357) -> PropertiesClosure {
358  Rc::new(
359    move |args: &Vec<Option<Digested>>| -> Result<SymHashMap<Stored>> {
360      let dyn_args: Vec<Dynamic> = args
361        .iter()
362        .map(|a| match a {
363          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
364          None => Dynamic::UNIT,
365        })
366        .collect();
367      let ret: Dynamic = fp
368        .call::<Dynamic>(&engine, &ast, dyn_args)
369        .map_err(|e| Error::from(format!("script properties: {e}")))?;
370      if ret.is_unit() {
371        Ok(SymHashMap::default())
372      } else if ret.is_map() {
373        Ok(rhai_map_to_props(ret.cast::<Map>()))
374      } else {
375        Err(Error::from(
376          "script properties: body must return a map (or unit)",
377        ))
378      }
379    },
380  )
381}

383/// Map a Rhai scalar `Dynamic` to a builder `OptionValue` (string/bool/int).
384pub(super) fn dynamic_to_option_value(v: &Dynamic) -> Option<OptionValue> {
385  if v.is_string() {
386    Some(OptionValue::Str(
387      v.clone().into_string().unwrap_or_default(),
388    ))
389  } else if v.is_bool() {
390    v.as_bool().ok().map(OptionValue::Bool)
391  } else {
392    v.as_int().ok().map(OptionValue::Int)
393  }
394}

396/// Coerce a Rhai scalar to a bool with the SAME policy as the builder's
397/// `OptionValue::into_bool` (bool as-is, int ≠ 0, non-empty string = true) —
398/// the single shared coercion. Review m4: the `*_options_from_map` loops below
399/// previously used `as_bool().unwrap_or(false)`, which silently coerced an int
400/// (`bounded: 1`) or string (`protected: "yes"`) option to `false`, diverging
401/// from the `OptionValue`/`set_option` path that the same bindings reach via
402/// scalar options.
403fn dynamic_to_bool(v: &Dynamic) -> bool {
404  dynamic_to_option_value(v)
405    .and_then(|ov| ov.into_bool().ok())
406    .unwrap_or(false)
407}

470/// Map a Rhai option bag onto `ExpandableOptions` (the `DefMacro!` scalar set).
471pub(super) fn expandable_options_from_map(opts: Map) -> ExpandableOptions {
472  let mut o = ExpandableOptions::default();
473  for (key, val) in opts {
474    let b = dynamic_to_bool(&val);
475    match key.as_str() {
476      "locked" => o.locked = b,
477      "protected" => o.protected = b,
478      "outer" => o.outer = b,
479      "long" => o.long = b,
480      "mathactive" => o.mathactive = b,
481      "robust" => o.robust = b,
482      "scope" => o.scope = scope_of(&dynamic_to_string(val.clone())),
483      "alias" => o.alias = Some(dynamic_to_string(val.clone())),
484      _ => {},
485    }
486  }
487  o
488}

513/// Install one `DefConditional` registration: the Rhai test closure receives
514/// each argument as its TeX-source string and must return a bool (mirrors the
515/// `DefConditional!` macro's `sub` test → `def_conditional`).
516pub(super) fn wire_conditional(
517  engine: &Rc<Engine>,
518  ast: &Rc<AST>,
519  proto: &str,
520  test: FnPtr,
521) -> Result<()> {
522  let (cs, paramlist) = parse_prototype(proto, true)?;
523  let cs_name = cs.to_string();
524  let engine = engine.clone();
525  let ast = ast.clone();
526  let closure: ConditionalClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<bool> {
527    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
528    let ret: Dynamic = test
529      .call::<Dynamic>(&engine, &ast, dyn_args)
530      .map_err(|e| Error::from(format!("script conditional {cs_name}: {e}")))?;
531    Ok(ret.as_bool().unwrap_or(false))
532  });
533  def_conditional(cs, paramlist, Some(closure), ConditionalOptions::default())
534}

536/// Install one `DefColumnType` registration: `"<char><params>"` →
537/// `\NC@rewrite@<char>` expandable whose body trampolines into Rhai (mirrors
538/// the `DefColumnType!` macro's lowering).
539pub(super) fn wire_columntype(
540  engine: &Rc<Engine>,
541  ast: &Rc<AST>,
542  proto: &str,
543  body: FnPtr,
544) -> Result<()> {
545  let mut chars = proto.chars();
546  let Some(first_c) = chars.next() else {
547    return Err(Error::from(
548      "DefColumnType: expected a column-specifier character",
549    ));
550  };
551  let rest: String = chars.collect::<String>().trim_start().to_string();
552  let cs = latexml_core::T_CS!(latexml_core::s!("\\NC@rewrite@{first_c}"));
553  let paramlist = if rest.is_empty() {
554    None
555  } else {
556    latexml_core::common::def_parser::parse_parameters(&rest, &cs, true)?
557  };
558  let cs_name = cs.to_string();
559  let engine = engine.clone();
560  let ast = ast.clone();
561  let closure: ExpansionClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<Tokens> {
562    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
563    let ret: Dynamic = body
564      .call::<Dynamic>(&engine, &ast, dyn_args)
565      .map_err(|e| Error::from(format!("script column type {cs_name}: {e}")))?;
566    Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
567  });
568  def_macro(cs, paramlist, ExpansionBody::Closure(closure), None)?;

687/// `DefKeyVal(keyset, key, vtype, default, #{prefix, kind, macroprefix,
688/// choices})` — the option-bag form (xkeyval-style keys with a prefix and a
689/// kind). The 3-/4-arg forms (plain `KV`-prefixed keys) are registered inline in
690/// `engine.rs`; this covers the richer xkeyval declarations (`myxkeyval`'s
691/// `prefix`/`kind`/`choices`). An empty `default` maps to Perl `undef`.
692/// `mismatch` (a warning closure) is intentionally unsupported — the fixtures
693/// that set it never exercise it, matching the prior compiled binding.
694pub(super) fn keyval_define_from_map(
695  keyset: &str,
696  key: &str,
697  vtype: &str,
698  default: &str,
699  opts: Map,
700) -> Result<()> {
701  use latexml_core::keyval::{self, KeyvalConfig};
702  let mut prefix = String::from("KV");
703  let mut kind: Option<String> = None;
704  let mut macroprefix: Option<String> = None;
705  // `choices` requires `Vec<&'static str>`; key definitions persist for the
706  // program lifetime, so leaking the split entries is acceptable (mirrors the
707  // prior `xkvview_sty`/`myxkeyval_sty` compiled bindings).
708  let mut choices: Vec<&'static str> = Vec::new();
709  for (k, v) in opts {
710    match k.as_str() {
711      "prefix" => prefix = dynamic_to_string(v),
712      "kind" => kind = Some(dynamic_to_string(v)),
713      "macroprefix" => macroprefix = Some(dynamic_to_string(v)),
714      "choices" => {
715        if let Some(arr) = v.try_cast::<rhai::Array>() {
716          choices = arr
717            .into_iter()
718            .map(|d| &*Box::leak(dynamic_to_string(d).into_boxed_str()))
719            .collect();
720        }
721      },
722      _ => {},
723    }
724  }
725  keyval::define(KeyvalConfig {
726    prefix: &prefix,
727    keyset,
728    key,
729    vtype,
730    default: if default.is_empty() {
731      None
732    } else {
733      Some(default)
734    },
735    kind: kind.as_deref(),
736    macroprefix: macroprefix.as_deref(),
737    choices,
738    ..KeyvalConfig::default()
739  })
740}

751/// Build a closure-form `reversion` trampoline: the body runs with the whatsit
752/// published READ-ONLY (argString/propertyString available) and each digested
753/// arg as its TeX-source string; its returned string is the reversion TeX.
754pub(super) fn reversion_trampoline(
755  fp: FnPtr,
756  engine: Rc<Engine>,
757  ast: Rc<AST>,
758) -> latexml_core::definition::DigestedReversionClosure {
759  Rc::new(
760    move |w: &Whatsit, args: &Vec<Option<Digested>>| -> Result<Tokens> {
761      // RAII pop on every exit incl. panic (review M1). Whatsit is READ-ONLY here.
762      let _whatsit_guard = WhatsitCtxGuard::new((w as *const Whatsit as *mut Whatsit, false));
763      let dyn_args: Vec<Dynamic> = args
764        .iter()
765        .map(|a| match a {
766          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
767          None => Dynamic::UNIT,
768        })
769        .collect();
770      let ret = fp
771        .call::<Dynamic>(&engine, &ast, dyn_args)
772        .map_err(|e| Error::from(format!("script reversion: {e}")))?;
773      Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
774    },
775  )
776}

851/// The `DefMathLigature!` data-form lowering: walk `ntomatch` preceding
852/// `ltx:XMTok` siblings matching the pattern (reversed), then rewrite to
853/// `replacement` with the given role/name/meaning attributes.
854pub(super) fn def_math_ligature_impl(pattern: &str, replacement: &str, opts: Map) {
855  use latexml_core::ligature::{Ligature, MathLigatureOptions};
856  let mut attr = MathLigatureOptions::default();
857  for (key, val) in opts {
858    let v = dynamic_to_string(val);
859    match key.as_str() {
860      "role" => attr.role = Some(v),
861      "name" => attr.name = Some(v),
862      "meaning" => attr.meaning = Some(v),
863      _ => {},
864    }
865  }
866  let chars: Vec<char> = pattern.chars().rev().collect();
867  let ntomatch = chars.len();
868  let replacement = replacement.to_string();
869  let matcher: Option<latexml_core::ligature::LigatureMatcher> = Some(Rc::new(
870    move |_document: &mut Document, node_opt: &mut libxml::tree::Node| {
871      let mut node: libxml::tree::Node;
872      let mut node_mut = node_opt;
873      for c in chars.iter() {
874        if latexml_core::common::model::with_node_qname(node_mut, |qname| qname != "ltx:XMTok")
875          || node_mut.get_content() != c.to_string()
876        {
877          return Ok(None);
878        }
879        match node_mut.get_prev_sibling() {
880          Some(sibling) => {
881            node = sibling;
882            node_mut = &mut node;
883          },
884          _ => {
885            return Ok(None);
886          },
887        }
888      }
889      if ntomatch > 0 {
890        Ok(Some((ntomatch, replacement.clone(), attr.clone())))
891      } else {
892        Ok(None)
893      }
894    },
895  ));
896  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
897    id: latexml_core::state::generate_ligature_id(),
898    matcher,
899    code: None,
900    font_test: None,
901    regex: None,
902  }]);
903}

905/// Parse a rewrite option bag into the FULL `RewriteOptions` superset — the
906/// single source shared by the data form (`def_rewrite_impl`) and the
907/// replace-closure form (`wire_rewrite_replace`), so the two cannot drift on
908/// which keys they honor. (Review M3: the data form silently dropped
909/// `select_count` and the replace form silently dropped `attributes`; each
910/// `_ => {}`-ignored the other's key.) The replace form just sets its closure
911/// and a `select_count` default afterward.
912fn parse_rewrite_options(kind: &str, opts: Map) -> latexml_core::rewrite::RewriteOptions {
913  use latexml_core::rewrite::RewriteOptions;
914  let mut o = RewriteOptions {
915    is_math: kind == "math",
916    ..RewriteOptions::default()
917  };
918  for (key, val) in opts {
919    match key.as_str() {
920      "label" => o.label = Some(dynamic_to_string(val)),
921      "xpath" => o.xpath = Some(dynamic_to_string(val)),
922      "select" => o.select = Some(dynamic_to_string(val)),
923      "select_count" => o.select_count = val.as_int().ok().map(|i| i as usize),
924      "attributes" => {
925        if val.is_map() {
926          let mut m = rustc_hash::FxHashMap::default();
927          for (k, v) in val.cast::<Map>() {
928            m.insert(k.to_string(), dynamic_to_string(v));
929          }
930          o.attributes_map = Some(m);
931        } else {
932          o.attributes = Some(dynamic_to_string(val));
933        }
934      },
935      "regexp" => o.regexp = Some(dynamic_to_string(val)),
936      "match" => o.on_match = Some(mouth::tokenize_internal(&dynamic_to_string(val))),
937      "scope" => o.scope = scope_of(&dynamic_to_string(val)),
938      _ => {},
939    }
940  }
941  o
942}

955/// `DefRewrite(opts, |nodes| …)` — the `replace` closure form. NB the core
956/// Replace clause DETACHES the matched nodes and splices in whatever the
957/// closure inserts (replace-by-reinsertion, Perl Rewrite.pm L122): the Rhai
958/// body receives the detached nodes as proxies; document-insertion context
959/// for the body is the pending follow-up (e2e lands with it).
960pub(super) fn wire_rewrite_replace(
961  kind: &str,
962  engine: &Rc<Engine>,
963  ast: &Rc<AST>,
964  opts: Map,
965  replace: FnPtr,
966) -> Result<()> {
967  use latexml_core::rewrite::Rewrite;
968  // Same superset parser as the data form (review M3) — so the replace form now
969  // honors `attributes` too, and neither form silently drops the other's key.
970  let mut o = parse_rewrite_options(kind, opts);
971  // The Replace clause consumes `select_count` matched nodes (native
972  // replace-rewrites always set it); default to 1 when the bag didn't.
973  o.select_count.get_or_insert(1);
974  let engine = engine.clone();
975  let ast = ast.clone();
976  // The core Replace clause positions the document at the splice parent before
977  // calling us; publish it so the body's `document.*` proxy inserts land there.
978  // The props map carries a default text font — at finalization there is no
979  // box_to_absorb for absorb_string to fall back on. Held by an `Rc` MOVED into
980  // the replace closure (NOT `Box::leak`ed): the map lives exactly as long as
981  // the rule and is freed when the rule is dropped — `load_script` runs per
982  // conversion, so the old leak grew one map per replace-rule PER PAPER in a
983  // long-lived worker (review M2). `CtorCtx.props` is `*const`, and the captured
984  // `Rc` keeps the target alive for the whole call.
985  let rewrite_props: Rc<SymHashMap<Stored>> = {
986    let mut m = SymHashMap::default();
987    m.insert(
988      "font",
989      Stored::Font(Rc::new(latexml_core::common::font::Font::text_default())),
990    );
991    Rc::new(m)
992  };
993  o.replace = Some(Rc::new(
994    move |document: &mut Document, nodes: Vec<&mut libxml::tree::Node>| -> Result<()> {
995      let arr: rhai::Array = nodes
996        .into_iter()
997        .map(|n| Dynamic::from(NodeProxy(n.clone())))
998        .collect();
999      // RAII pop on every exit incl. panic (review M1).
1000      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
1001        document,
1002        props: Rc::as_ptr(&rewrite_props),
1003      });
1004      let _: Dynamic = replace
1005        .call::<Dynamic>(&engine, &ast, (Dynamic::from(DocProxy), arr))
1006        .map_err(|e| Error::from(format!("script rewrite replace: {e}")))?;
1007      Ok(())
1008    },
1009  ));
1010  latexml_core::state::push_value("DOCUMENT_REWRITE_RULES", Rewrite::new(kind, o))?;
1011  Ok(())
1012}

1014/// Closure-form `DefMathLigature` (`matcher => sub[document,node]`): the Rhai
1015/// body receives a read-only `Node` proxy; UNIT means no match, a map
1016/// `#{ n, replacement, role?, name?, meaning? }` reports one.
1017pub(super) fn wire_math_ligature_matcher(
1018  engine: &Rc<Engine>,
1019  ast: &Rc<AST>,
1020  matcher: FnPtr,
1021) -> Result<()> {
1022  use latexml_core::ligature::{Ligature, MathLigatureOptions};
1023  let engine = engine.clone();
1024  let ast = ast.clone();
1025  let rust_matcher: latexml_core::ligature::LigatureMatcher = Rc::new(
1026    move |_document: &mut Document, node: &mut libxml::tree::Node| {
1027      let ret: Dynamic = matcher
1028        .call::<Dynamic>(&engine, &ast, (Dynamic::from(NodeProxy(node.clone())),))
1029        .map_err(|e| Error::from(format!("script math-ligature matcher: {e}")))?;
1030      if !ret.is_map() {
1031        return Ok(None);
1032      }
1033      let m = ret.cast::<Map>();
1034      let n = m.get("n").and_then(|v| v.as_int().ok()).unwrap_or(0) as usize;
1035      let replacement = m
1036        .get("replacement")
1037        .map(|v| dynamic_to_string(v.clone()))
1038        .unwrap_or_default();
1039      if n == 0 {
1040        return Ok(None);
1041      }
1042      let attr = MathLigatureOptions {
1043        role:    m.get("role").map(|v| dynamic_to_string(v.clone())),
1044        name:    m.get("name").map(|v| dynamic_to_string(v.clone())),
1045        meaning: m.get("meaning").map(|v| dynamic_to_string(v.clone())),
1046      };
1047      Ok(Some((n, replacement, attr)))
1048    },
1049  );
1050  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
1051    id:        latexml_core::state::generate_ligature_id(),
1052    matcher:   Some(rust_matcher),
1053    code:      None,
1054    font_test: None,
1055    regex:     None,
1056  }]);
1057  Ok(())
