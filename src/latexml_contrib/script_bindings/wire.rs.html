9/// Install one `DefMacro` registration as a native expandable definition.
10pub(super) fn wire_macro(
11  engine: &Rc<Engine>,
12  ast: &Rc<AST>,
13  proto: &str,
14  body: FnPtr,
15) -> Result<()> {
16  let (cs, paramlist) = parse_prototype(proto, true)?;
17  let cs_name = cs.to_string();
18  let engine = engine.clone();
19  let ast = ast.clone();
20
21  let closure: ExpansionClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<Tokens> {
22    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
23    let ret: Dynamic = call_deferred_body(&engine, &ast, &body, dyn_args)
24      .map_err(|e| Error::from(format!("script macro {cs_name}: {e}")))?;
25    Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
26  });
27
28  def_macro(cs, paramlist, ExpansionBody::Closure(closure), None)?;
29  Ok(())
30}

32/// Install one string-body `DefMacro('\proto', 'replacement')`. The body is TeX
33/// source that becomes the macro's token expansion directly — Perl's most
34/// common `DefMacro($proto, $string)` form, where `$string` is `TokenizeInternal`d
35/// into the replacement (NOT a Rhai closure, so no per-expansion engine call).
36pub(super) fn wire_macro_string(proto: &str, body: &str) -> Result<()> {
37  let (cs, paramlist) = parse_prototype(proto, true)?;
38  def_macro(
39    cs,
40    paramlist,
41    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
42    None,
43  )?;
44  Ok(())
45}

47/// String-body `DefMacro` with a trailing option bag (`DefMacro($proto, $string,
48/// %opts)`). Same token expansion as [`wire_macro_string`], plus the scalar
49/// `ExpandableOptions` from the map.
50pub(super) fn wire_macro_string_opts(proto: &str, body: &str, opts: Map) -> Result<()> {
51  let (cs, paramlist) = parse_prototype(proto, true)?;
52  def_macro(
53    cs,
54    paramlist,
55    ExpansionBody::Tokens(mouth::tokenize_internal(body)),
56    Some(expandable_options_from_map(opts)),
57  )?;
58  Ok(())
59}

61/// The imperative-body replacement closure: publishes the active context, calls
62/// the body as `|document, arg1, …|`, pops the context. Shared by every
63/// constructor form that uses a closure body.
64pub(super) fn closure_replacement(
65  body: FnPtr,
66  engine: Rc<Engine>,
67  ast: Rc<AST>,
68  cs_name: String,
69) -> ReplacementClosure {
70  Rc::new(
71    move |document: &mut Document, args: &Vec<Option<Digested>>, props| -> Result<()> {
72      // RAII: the entry is popped on EVERY exit of this closure, including a
73      // panic out of `body.call` (Rhai doesn't catch_unwind native calls) — so
74      // a script-body panic can't leave a stale CTOR_CTX entry. Review M1.
75      let _ctx_guard = CtorCtxGuard::new(CtorCtx { document, props });
76      // `document` first (Perl's `$_[0]`), then each digested arg as a handle.
77      let mut call_args: Vec<Dynamic> = Vec::with_capacity(args.len() + 1);
78      call_args.push(Dynamic::from(DocProxy));
79      for a in args {
80        call_args.push(match a {
81          Some(d) => Dynamic::from(d.clone()),
82          None => Dynamic::UNIT,
83        });
84      }
85      call_deferred_body(&engine, &ast, &body, call_args)
86        .map(|_| ())
87        .map_err(|e| Error::from(format!("script constructor {cs_name}: {e}")))
88    },
89  )
90}

92/// The string-template replacement closure. The template is parsed **once** here
93/// (at wire time) into the shared `ReplacementOp` AST — the same parser the
94/// compile-time codegen uses (#171) — and the cached AST is interpreted per
95/// invocation. This eliminates the former per-invocation byte-scan and removes
96/// the second, divergent template implementation.
97pub(super) fn template_replacement(template: &str) -> Result<ReplacementClosure> {
98  let ops = Rc::new(replacement::parse_replacement(template)?);
99  Ok(Rc::new(move |document: &mut Document, args, props| {
100    replacement::apply_ops(&ops, document, args, props)
101  }))
102}

104/// Install one `DefConstructor` (imperative body, no options) via the shared
105/// `ConstructorBuilder`.
106pub(super) fn wire_constructor(
107  engine: &Rc<Engine>,
108  ast: &Rc<AST>,
109  proto: &str,
110  body: FnPtr,
111) -> Result<()> {
112  let repl = closure_replacement(body, engine.clone(), ast.clone(), proto.to_string());
113  ConstructorBuilder::new(proto)?.replacement(repl).install()
114}

116/// Install a `DefConstructor` with an option bag (`#{ mode, afterDigest, … }`)
117/// through the shared `ConstructorBuilder` — the *same* builder the
118/// `DefConstructor!` macro targets, so the two front-ends cannot drift. Scalar
119/// options route through the builder's single-source `set_option`; closure
120/// options through its typed setters (here, `after_digest`).
121pub(super) fn wire_constructor_opts(
122  engine: &Rc<Engine>,
123  ast: &Rc<AST>,
124  proto: &str,
125  repl: ConstructorRepl,
126  opts: Map,
127) -> Result<()> {
128  let replacement = match repl {
129    ConstructorRepl::Template(t) => template_replacement(&t)?,
130    ConstructorRepl::Closure(b) => {
131      closure_replacement(b, engine.clone(), ast.clone(), proto.to_string())
132    },
133  };
134  let builder = ConstructorBuilder::new(proto)?.replacement(replacement);
135  apply_opts(builder, opts, engine, ast)?.install()
136}

138/// Install one `DefEnvironment` registration (template or imperative body, with
139/// an option bag — possibly empty) through the shared `EnvironmentBuilder`.
140/// The imperative body reaches `#body` via `document.absorbProperty("body")`.
141pub(super) fn wire_environment(
142  engine: &Rc<Engine>,
143  ast: &Rc<AST>,
144  proto: &str,
145  repl: ConstructorRepl,
146  opts: Map,
147) -> Result<()> {
148  let replacement = match repl {
149    ConstructorRepl::Template(t) => template_replacement(&t)?,
150    ConstructorRepl::Closure(b) => {

158/// The builder surface the option-bag loop needs — implemented for both core
159/// builders so [`apply_opts`] is written once. (A local trait over the foreign
160/// builder types; new closure options are added in `apply_opts` + one impl line
161/// per builder.)
162pub(super) trait BindingBuilder: Sized {
163  fn set_option(self, key: &str, value: OptionValue) -> Result<Self>;
164  fn after_digest(self, hook: DigestionClosure) -> Self;
165  fn after_digest_begin(self, hook: DigestionClosure) -> Self;
166  fn before_digest(self, hook: BeforeDigestClosure) -> Self;
167  fn before_digest_end(self, hook: BeforeDigestClosure) -> Self;
168  fn before_construct(self, hook: ConstructionClosure) -> Self;
169  fn after_construct(self, hook: ConstructionClosure) -> Self;
170  fn properties(self, props: PropertiesClosure) -> Self;
171  fn reversion(self, rev: Reversion) -> Self;
172  fn font(self, font: FontDirective) -> Self;
173  fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self;
174  // (`install` stays inherent on each builder: call sites get the concrete
175  // type back from `apply_opts`, so a trait method would be dead code.)
176}
177
178macro_rules! impl_binding_builder {
179  ($t:ty) => {
180    impl BindingBuilder for $t {
181      fn set_option(self, key: &str, value: OptionValue) -> Result<Self> {
182        <$t>::set_option(self, key, value)
183      }
184      fn after_digest(self, hook: DigestionClosure) -> Self { <$t>::after_digest(self, hook) }
185      fn after_digest_begin(self, hook: DigestionClosure) -> Self {
186        <$t>::after_digest_begin(self, hook)
187      }
188      fn before_digest(self, hook: BeforeDigestClosure) -> Self { <$t>::before_digest(self, hook) }
189      fn before_digest_end(self, hook: BeforeDigestClosure) -> Self {
190        <$t>::before_digest_end(self, hook)
191      }
192      fn before_construct(self, hook: ConstructionClosure) -> Self {
193        <$t>::before_construct(self, hook)
194      }
195      fn after_construct(self, hook: ConstructionClosure) -> Self {
196        <$t>::after_construct(self, hook)
197      }
198      fn properties(self, props: PropertiesClosure) -> Self { <$t>::properties(self, props) }
199      fn reversion(self, rev: Reversion) -> Self { <$t>::reversion(self, rev) }
200      fn font(self, font: FontDirective) -> Self { <$t>::font(self, font) }
201      fn sizer(self, sizer: latexml_core::definition::SizingClosure) -> Self {
202        <$t>::sizer(self, sizer)
203      }
204    }
205  };
206}
207impl_binding_builder!(ConstructorBuilder);
208impl_binding_builder!(EnvironmentBuilder);

210/// Apply a Rhai option bag onto a builder: closure options become trampolines
211/// (typed setters), `properties` also accepts a static map, scalars route
212/// through the builder's single-source `set_option`. Shared by the constructor
213/// and environment front-ends.
214pub(super) fn apply_opts<B: BindingBuilder>(
215  mut builder: B,
216  opts: Map,
217  engine: &Rc<Engine>,
218  ast: &Rc<AST>,
219) -> Result<B> {
220  for (key, val) in opts {
221    match val.clone().try_cast::<FnPtr>() {
222      Some(fp) => {
223        // Closure option → typed builder setter (front-end builds the closure).
224        match key.as_str() {
225          "afterDigest" => {
226            builder =
227              builder.after_digest(after_digest_trampoline(fp, engine.clone(), ast.clone()));
228          },
229          "afterDigestBegin" => {
230            builder =
231              builder.after_digest_begin(after_digest_trampoline(fp, engine.clone(), ast.clone()));
232          },
233          "properties" => {
234            builder = builder.properties(properties_trampoline(fp, engine.clone(), ast.clone()));
235          },
236          "beforeDigest" => {
237            builder =
238              builder.before_digest(before_digest_trampoline(fp, engine.clone(), ast.clone()));
239          },
240          "beforeDigestEnd" => {
241            builder =
242              builder.before_digest_end(before_digest_trampoline(fp, engine.clone(), ast.clone()));
243          },
244          "beforeConstruct" => {
245            builder =
246              builder.before_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
247          },
248          "reversion" => {
249            builder = builder.reversion(Reversion::Closure(reversion_trampoline(
250              fp,
251              engine.clone(),
252              ast.clone(),
253            )));
254          },
255          "sizer" => {
256            builder = builder.sizer(sizer_trampoline(fp, engine.clone(), ast.clone()));
257          },
258          "afterConstruct" => {
259            builder =
260              builder.after_construct(construction_trampoline(fp, engine.clone(), ast.clone()));
261          },
262          // Unknown closure options are silently ignored (forgiving, like Perl
263          // %options; the builder's set_option does the same for scalars).
264          _ => {},
265        }
266      },
267      _ => {
268        if key.as_str() == "properties" && val.is_map() {
269          // Static property map (Perl's `properties => { key => value, … }`).
270          let map = val.cast::<Map>();
271          builder = builder.properties(Rc::new(move |_args| Ok(rhai_map_to_props(map.clone()))));
272        } else if key.as_str() == "reversion" {
273          // String reversion (`reversion => "\\begin{x}#1\\end{x}"`, "" disables).
274          builder = builder.reversion(Reversion::Tokens(mouth::tokenize_internal(
275            &dynamic_to_string(val),
276          )));
277        } else if key.as_str() == "font" && val.is_map() {
278          // Partial-font directive (`font => { family => 'typewriter', … }`).
279          let font = font_from_rhai_map(val.cast::<Map>());
280          builder = builder.font(FontDirective::Asset(Rc::new(font)));
281        } else if let Some(ov) = dynamic_to_option_value(&val) {
282          // Scalar option → the builder's generic, single-source `set_option`.
283          builder = builder.set_option(key.as_str(), ov)?;
284        }
285      },
286    }
287  }
288  Ok(builder)
289}

291/// Build a `before/afterConstruct` trampoline: the body runs with the live
292/// document published as active context (so `document.*` proxy calls work);
293/// the whatsit itself is not yet exposed there (read-only marshaling TBD).
294pub(super) fn construction_trampoline(
295  fp: FnPtr,
296  engine: Rc<Engine>,
297  ast: Rc<AST>,
298) -> ConstructionClosure {
299  Rc::new(
300    move |document: &mut Document, whatsit: &Whatsit| -> Result<()> {
301      // RAII pop on every exit incl. panic (review M1).
302      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
303        document,
304        props: whatsit.get_properties(),
305      });
306      let _: Dynamic = call_deferred_body(&engine, &ast, &fp, (Dynamic::from(DocProxy),))
307        .map_err(|e| Error::from(format!("script afterConstruct: {e}")))?;
308      Ok(())
309    },
310  )
311}

313/// Build an `afterDigest` trampoline: publish the whatsit so a parameterless body
314/// can reach it via `whatsit()`, call the Rhai closure, pop.
315pub(super) fn after_digest_trampoline(
316  fp: FnPtr,
317  engine: Rc<Engine>,
318  ast: Rc<AST>,
319) -> DigestionClosure {
320  Rc::new(move |w: &mut Whatsit| -> Result<Vec<Digested>> {
321    // RAII pop on every exit incl. panic (review M1).
322    let _whatsit_guard = WhatsitCtxGuard::new((w as *mut Whatsit, true));
323    let _: Dynamic = call_deferred_body(&engine, &ast, &fp, ())
324      .map_err(|e| Error::from(format!("script afterDigest: {e}")))?;
325    Ok(Vec::new())
326  })
327}

329/// Build a `beforeDigest` trampoline (Perl `beforeDigest => sub {…}`): runs the
330/// parameterless Rhai closure before the constructor's arguments are digested
331/// (state/font side-effects, e.g. `neutralize_font()`); contributes no boxes.
332pub(super) fn before_digest_trampoline(
333  fp: FnPtr,
334  engine: Rc<Engine>,
335  ast: Rc<AST>,
336) -> BeforeDigestClosure {
337  Rc::new(move || -> Result<Vec<Digested>> {
338    let _: Dynamic = call_deferred_body(&engine, &ast, &fp, ())
339      .map_err(|e| Error::from(format!("script beforeDigest: {e}")))?;
340    Ok(Vec::new())
341  })
342}

344/// Build a `properties` trampoline (Perl `properties => sub {…}`): the Rhai
345/// closure receives each digested argument as its TeX-source string (`()` for an
346/// omitted optional) and returns a map; its entries become the whatsit's
347/// properties as string `Stored`s, ready for the template's `#name` holes.
348pub(super) fn properties_trampoline(
349  fp: FnPtr,
350  engine: Rc<Engine>,
351  ast: Rc<AST>,
352) -> PropertiesClosure {
353  Rc::new(
354    move |args: &Vec<Option<Digested>>| -> Result<SymHashMap<Stored>> {
355      let dyn_args: Vec<Dynamic> = args
356        .iter()
357        .map(|a| match a {
358          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
359          None => Dynamic::UNIT,
360        })
361        .collect();
362      let ret: Dynamic = call_deferred_body(&engine, &ast, &fp, dyn_args)
363        .map_err(|e| Error::from(format!("script properties: {e}")))?;
364      if ret.is_unit() {
365        Ok(SymHashMap::default())
366      } else if ret.is_map() {
367        Ok(rhai_map_to_props(ret.cast::<Map>()))
368      } else {
369        Err(Error::from(
370          "script properties: body must return a map (or unit)",
371        ))
372      }
373    },
374  )
375}

377/// Map a Rhai scalar `Dynamic` to a builder `OptionValue` (string/bool/int).
378pub(super) fn dynamic_to_option_value(v: &Dynamic) -> Option<OptionValue> {
379  if v.is_string() {
380    Some(OptionValue::Str(
381      v.clone().into_string().unwrap_or_default(),
382    ))
383  } else if v.is_bool() {
384    v.as_bool().ok().map(OptionValue::Bool)
385  } else {
386    v.as_int().ok().map(OptionValue::Int)
387  }
388}
389
390/// Coerce a Rhai scalar to a bool with the SAME policy as the builder's
391/// `OptionValue::into_bool` (bool as-is, int ≠ 0, non-empty string = true) —
392/// the single shared coercion. Review m4: the `*_options_from_map` loops below
393/// previously used `as_bool().unwrap_or(false)`, which silently coerced an int
394/// (`bounded: 1`) or string (`protected: "yes"`) option to `false`, diverging
395/// from the `OptionValue`/`set_option` path that the same bindings reach via
396/// scalar options.
397fn dynamic_to_bool(v: &Dynamic) -> bool {
398  dynamic_to_option_value(v)
399    .and_then(|ov| ov.into_bool().ok())
400    .unwrap_or(false)
401}

462/// Map a Rhai option bag onto `ExpandableOptions` (the `DefMacro!` scalar set).
463pub(super) fn expandable_options_from_map(opts: Map) -> ExpandableOptions {
464  let mut o = ExpandableOptions::default();
465  for (key, val) in opts {
466    let b = dynamic_to_bool(&val);
467    match key.as_str() {
468      "locked" => o.locked = b,
469      "protected" => o.protected = b,
470      "outer" => o.outer = b,
471      "long" => o.long = b,
472      "mathactive" => o.mathactive = b,
473      "robust" => o.robust = b,
474      "scope" => o.scope = scope_of(&dynamic_to_string(val.clone())),
475      "alias" => o.alias = Some(dynamic_to_string(val.clone())),
476      _ => {},
477    }
478  }
479  o
480}

505/// Install one `DefConditional` registration: the Rhai test closure receives
506/// each argument as its TeX-source string and must return a bool (mirrors the
507/// `DefConditional!` macro's `sub` test → `def_conditional`).
508pub(super) fn wire_conditional(
509  engine: &Rc<Engine>,
510  ast: &Rc<AST>,
511  proto: &str,
512  test: FnPtr,
513) -> Result<()> {
514  let (cs, paramlist) = parse_prototype(proto, true)?;
515  let cs_name = cs.to_string();
516  let engine = engine.clone();
517  let ast = ast.clone();
518  let closure: ConditionalClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<bool> {
519    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
520    let ret: Dynamic = call_deferred_body(&engine, &ast, &test, dyn_args)
521      .map_err(|e| Error::from(format!("script conditional {cs_name}: {e}")))?;
522    Ok(ret.as_bool().unwrap_or(false))
523  });
524  def_conditional(cs, paramlist, Some(closure), ConditionalOptions::default())
525}

527/// Install one `DefColumnType` registration: `"<char><params>"` →
528/// `\NC@rewrite@<char>` expandable whose body trampolines into Rhai (mirrors
529/// the `DefColumnType!` macro's lowering).
530pub(super) fn wire_columntype(
531  engine: &Rc<Engine>,
532  ast: &Rc<AST>,
533  proto: &str,
534  body: FnPtr,
535) -> Result<()> {
536  let mut chars = proto.chars();
537  let Some(first_c) = chars.next() else {
538    return Err(Error::from(
539      "DefColumnType: expected a column-specifier character",
540    ));
541  };
542  let rest: String = chars.collect::<String>().trim_start().to_string();
543  let cs = latexml_core::T_CS!(latexml_core::s!("\\NC@rewrite@{first_c}"));
544  let paramlist = if rest.is_empty() {
545    None
546  } else {
547    latexml_core::common::def_parser::parse_parameters(&rest, &cs, true)?
548  };
549  let cs_name = cs.to_string();
550  let engine = engine.clone();
551  let ast = ast.clone();
552  let closure: ExpansionClosure = Rc::new(move |args: Vec<ArgWrap>| -> Result<Tokens> {
553    let dyn_args: Vec<Dynamic> = args.into_iter().map(arg_to_dynamic).collect();
554    let ret: Dynamic = call_deferred_body(&engine, &ast, &body, dyn_args)
555      .map_err(|e| Error::from(format!("script column type {cs_name}: {e}")))?;
556    Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
557  });
558  def_macro(cs, paramlist, ExpansionBody::Closure(closure), None)?;
559  Ok(())

677/// `DefKeyVal(keyset, key, vtype, default, #{prefix, kind, macroprefix,
678/// choices})` — the option-bag form (xkeyval-style keys with a prefix and a
679/// kind). The 3-/4-arg forms (plain `KV`-prefixed keys) are registered inline in
680/// `engine.rs`; this covers the richer xkeyval declarations (`myxkeyval`'s
681/// `prefix`/`kind`/`choices`). An empty `default` maps to Perl `undef`.
682/// `mismatch` (a warning closure) is intentionally unsupported — the fixtures
683/// that set it never exercise it, matching the prior compiled binding.
684pub(super) fn keyval_define_from_map(
685  keyset: &str,
686  key: &str,
687  vtype: &str,
688  default: &str,
689  opts: Map,
690) -> Result<()> {
691  use latexml_core::keyval::{self, KeyvalConfig};
692  let mut prefix = String::from("KV");
693  let mut kind: Option<String> = None;
694  let mut macroprefix: Option<String> = None;
695  // `choices` requires `Vec<&'static str>`; key definitions persist for the
696  // program lifetime, so leaking the split entries is acceptable (mirrors the
697  // prior `xkvview_sty`/`myxkeyval_sty` compiled bindings).
698  let mut choices: Vec<&'static str> = Vec::new();
699  for (k, v) in opts {
700    match k.as_str() {
701      "prefix" => prefix = dynamic_to_string(v),
702      "kind" => kind = Some(dynamic_to_string(v)),
703      "macroprefix" => macroprefix = Some(dynamic_to_string(v)),
704      "choices" => {
705        if let Some(arr) = v.try_cast::<rhai::Array>() {
706          choices = arr
707            .into_iter()
708            .map(|d| &*Box::leak(dynamic_to_string(d).into_boxed_str()))
709            .collect();
710        }
711      },
712      _ => {},
713    }
714  }
715  keyval::define(KeyvalConfig {
716    prefix: &prefix,
717    keyset,
718    key,
719    vtype,
720    default: if default.is_empty() {
721      None
722    } else {
723      Some(default)
724    },
725    kind: kind.as_deref(),
726    macroprefix: macroprefix.as_deref(),
727    choices,
728    ..KeyvalConfig::default()
729  })
730}

741/// Build a closure-form `reversion` trampoline: the body runs with the whatsit
742/// published READ-ONLY (argString/propertyString available) and each digested
743/// arg as its TeX-source string; its returned string is the reversion TeX.
744pub(super) fn reversion_trampoline(
745  fp: FnPtr,
746  engine: Rc<Engine>,
747  ast: Rc<AST>,
748) -> latexml_core::definition::DigestedReversionClosure {
749  Rc::new(
750    move |w: &Whatsit, args: &Vec<Option<Digested>>| -> Result<Tokens> {
751      // RAII pop on every exit incl. panic (review M1). Whatsit is READ-ONLY here.
752      let _whatsit_guard = WhatsitCtxGuard::new((w as *const Whatsit as *mut Whatsit, false));
753      let dyn_args: Vec<Dynamic> = args
754        .iter()
755        .map(|a| match a {
756          Some(d) => Dynamic::from(d.untex().unwrap_or_default()),
757          None => Dynamic::UNIT,
758        })
759        .collect();
760      let ret = call_deferred_body(&engine, &ast, &fp, dyn_args)
761        .map_err(|e| Error::from(format!("script reversion: {e}")))?;
762      Ok(mouth::tokenize_internal(&dynamic_to_string(ret)))
763    },
764  )
765}

770  fp: FnPtr,

839/// The `DefMathLigature!` data-form lowering: walk `ntomatch` preceding
840/// `ltx:XMTok` siblings matching the pattern (reversed), then rewrite to
841/// `replacement` with the given role/name/meaning attributes.
842pub(super) fn def_math_ligature_impl(pattern: &str, replacement: &str, opts: Map) {
843  use latexml_core::ligature::{Ligature, MathLigatureOptions};
844  let mut attr = MathLigatureOptions::default();
845  for (key, val) in opts {
846    let v = dynamic_to_string(val);
847    match key.as_str() {
848      "role" => attr.role = Some(v),
849      "name" => attr.name = Some(v),
850      "meaning" => attr.meaning = Some(v),
851      _ => {},
852    }
853  }
854  let chars: Vec<char> = pattern.chars().rev().collect();
855  let ntomatch = chars.len();
856  let replacement = replacement.to_string();
857  let matcher: Option<latexml_core::ligature::LigatureMatcher> = Some(Rc::new(
858    move |_document: &mut Document, node_opt: &mut libxml::tree::Node| {
859      let mut node: libxml::tree::Node;
860      let mut node_mut = node_opt;
861      for c in chars.iter() {
862        if latexml_core::common::model::with_node_qname(node_mut, |qname| qname != "ltx:XMTok")
863          || node_mut.get_content() != c.to_string()
864        {
865          return Ok(None);
866        }
867        match node_mut.get_prev_sibling() {
868          Some(sibling) => {
869            node = sibling;
870            node_mut = &mut node;
871          },
872          _ => {
873            return Ok(None);
874          },
875        }
876      }
877      if ntomatch > 0 {
878        Ok(Some((ntomatch, replacement.clone(), attr.clone())))
879      } else {
880        Ok(None)
881      }
882    },
883  ));
884  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
885    id: latexml_core::state::generate_ligature_id(),
886    matcher,
887    code: None,
888    font_test: None,
889    regex: None,
890  }]);
891}

893/// Parse a rewrite option bag into the FULL `RewriteOptions` superset — the
894/// single source shared by the data form (`def_rewrite_impl`) and the
895/// replace-closure form (`wire_rewrite_replace`), so the two cannot drift on
896/// which keys they honor. (Review M3: the data form silently dropped
897/// `select_count` and the replace form silently dropped `attributes`; each
898/// `_ => {}`-ignored the other's key.) The replace form just sets its closure
899/// and a `select_count` default afterward.
900fn parse_rewrite_options(kind: &str, opts: Map) -> latexml_core::rewrite::RewriteOptions {
901  use latexml_core::rewrite::RewriteOptions;
902  let mut o = RewriteOptions {
903    is_math: kind == "math",
904    ..RewriteOptions::default()
905  };
906  for (key, val) in opts {
907    match key.as_str() {
908      "label" => o.label = Some(dynamic_to_string(val)),
909      "xpath" => o.xpath = Some(dynamic_to_string(val)),
910      "select" => o.select = Some(dynamic_to_string(val)),
911      "select_count" => o.select_count = val.as_int().ok().map(|i| i as usize),
912      "attributes" => {
913        if val.is_map() {
914          let mut m = rustc_hash::FxHashMap::default();
915          for (k, v) in val.cast::<Map>() {
916            m.insert(k.to_string(), dynamic_to_string(v));
917          }
918          o.attributes_map = Some(m);
919        } else {
920          o.attributes = Some(dynamic_to_string(val));
921        }
922      },
923      "regexp" => o.regexp = Some(dynamic_to_string(val)),
924      "match" => o.on_match = Some(mouth::tokenize_internal(&dynamic_to_string(val))),
925      "scope" => o.scope = scope_of(&dynamic_to_string(val)),
926      _ => {},
927    }
928  }
929  o
930}

943/// `DefRewrite(opts, |nodes| …)` — the `replace` closure form. NB the core
944/// Replace clause DETACHES the matched nodes and splices in whatever the
945/// closure inserts (replace-by-reinsertion, Perl Rewrite.pm L122): the Rhai
946/// body receives the detached nodes as proxies; document-insertion context
947/// for the body is the pending follow-up (e2e lands with it).
948pub(super) fn wire_rewrite_replace(
949  kind: &str,
950  engine: &Rc<Engine>,
951  ast: &Rc<AST>,
952  opts: Map,
953  replace: FnPtr,
954) -> Result<()> {
955  use latexml_core::rewrite::Rewrite;
956  // Same superset parser as the data form (review M3) — so the replace form now
957  // honors `attributes` too, and neither form silently drops the other's key.
958  let mut o = parse_rewrite_options(kind, opts);
959  // The Replace clause consumes `select_count` matched nodes (native
960  // replace-rewrites always set it); default to 1 when the bag didn't.
961  o.select_count.get_or_insert(1);
962  let engine = engine.clone();
963  let ast = ast.clone();
964  // The core Replace clause positions the document at the splice parent before
965  // calling us; publish it so the body's `document.*` proxy inserts land there.
966  // The props map carries a default text font — at finalization there is no
967  // box_to_absorb for absorb_string to fall back on. Held by an `Rc` MOVED into
968  // the replace closure (NOT `Box::leak`ed): the map lives exactly as long as
969  // the rule and is freed when the rule is dropped — `load_script` runs per
970  // conversion, so the old leak grew one map per replace-rule PER PAPER in a
971  // long-lived worker (review M2). `CtorCtx.props` is `*const`, and the captured
972  // `Rc` keeps the target alive for the whole call.
973  let rewrite_props: Rc<SymHashMap<Stored>> = {
974    let mut m = SymHashMap::default();
975    m.insert(
976      "font",
977      Stored::Font(Rc::new(latexml_core::common::font::Font::text_default())),
978    );
979    Rc::new(m)
980  };
981  o.replace = Some(Rc::new(
982    move |document: &mut Document, nodes: Vec<&mut libxml::tree::Node>| -> Result<()> {
983      let arr: rhai::Array = nodes
984        .into_iter()
985        .map(|n| Dynamic::from(NodeProxy(n.clone())))
986        .collect();
987      // RAII pop on every exit incl. panic (review M1).
988      let _ctx_guard = CtorCtxGuard::new(CtorCtx {
989        document,
990        props: Rc::as_ptr(&rewrite_props),
991      });
992      let _: Dynamic = call_deferred_body(&engine, &ast, &replace, (Dynamic::from(DocProxy), arr))
993        .map_err(|e| Error::from(format!("script rewrite replace: {e}")))?;
994      Ok(())
995    },
996  ));
997  latexml_core::state::push_value("DOCUMENT_REWRITE_RULES", Rewrite::new(kind, o))?;
998  Ok(())
999}

1001/// Closure-form `DefMathLigature` (`matcher => sub[document,node]`): the Rhai
1002/// body receives a read-only `Node` proxy; UNIT means no match, a map
1003/// `#{ n, replacement, role?, name?, meaning? }` reports one.
1004pub(super) fn wire_math_ligature_matcher(
1005  engine: &Rc<Engine>,
1006  ast: &Rc<AST>,
1007  matcher: FnPtr,
1008) -> Result<()> {
1009  use latexml_core::ligature::{Ligature, MathLigatureOptions};
1010  let engine = engine.clone();
1011  let ast = ast.clone();
1012  let rust_matcher: latexml_core::ligature::LigatureMatcher = Rc::new(
1013    move |_document: &mut Document, node: &mut libxml::tree::Node| {
1014      let ret: Dynamic = call_deferred_body(
1015        &engine,
1016        &ast,
1017        &matcher,
1018        (Dynamic::from(NodeProxy(node.clone())),),
1019      )
1020      .map_err(|e| Error::from(format!("script math-ligature matcher: {e}")))?;
1021      if !ret.is_map() {
1022        return Ok(None);
1023      }
1024      let m = ret.cast::<Map>();
1025      let n = m.get("n").and_then(|v| v.as_int().ok()).unwrap_or(0) as usize;
1026      let replacement = m
1027        .get("replacement")
1028        .map(|v| dynamic_to_string(v.clone()))
1029        .unwrap_or_default();
1030      if n == 0 {
1031        return Ok(None);
1032      }
1033      let attr = MathLigatureOptions {
1034        role:    m.get("role").map(|v| dynamic_to_string(v.clone())),
1035        name:    m.get("name").map(|v| dynamic_to_string(v.clone())),
1036        meaning: m.get("meaning").map(|v| dynamic_to_string(v.clone())),
1037      };
1038      Ok(Some((n, replacement, attr)))
1039    },
1040  );
1041  latexml_core::state::unshift_value("MATH_LIGATURES", vec![Ligature {
1042    id:        latexml_core::state::generate_ligature_id(),
1043    matcher:   Some(rust_matcher),
1044    code:      None,
1045    font_test: None,
1046    regex:     None,
1047  }]);
1048  Ok(())
