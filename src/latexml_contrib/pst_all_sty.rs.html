1//! pst-all.sty — convenience loader for the full pstricks suite.
2//!
3//! pst-all chain-loads pst-tree / pst-coil / pst-text / pst-3d /
4//! pst-eps / pst-fill / pstricks-add. Raw-loading these hits
5//! PostScript-specific primitives we don't model (`\psk@rot`,
6//! PSTricks parameter cascade) and floods the 100-error cap.
7//!
8//! Witness arXiv:1402.6510 — loads pst-all but never invokes a
9//! single PSTricks command; Perl converts with 367 warnings.
10//!
11//! Match Perl: stub as a no-op so the raw-load is skipped.
12//! pst-plot already has this treatment (see pst_plot_sty.rs).
13use latexml_package::prelude::*;
14
15LoadDefinitions!({
16  Warn!(
17    "missing_file",
18    "pst-all.sty",
19    "pst-all.sty is not implemented; loading pstricks core only."
20  );
21  // pst-all loads pst-tree/pst-coil/pst-text/pst-3d/pst-eps/pst-fill/
22  // pstricks-add. We provide pstricks core so the common `\rput`,
23  // `\cnode`, `\psset` etc. macros resolve. The fancier sub-packages
24  // remain unstubbed for now (papers that use them get undefined-CS
25  // warnings but not fatals).
26  RequirePackage!("pstricks");
27  // pst-all.sty L23 does `\RequirePackage{pst-node}`. Load our pst-node
28  // binding (a clean stub — NOT the raw .sty) so node macros
29  // (`\Cnode`/`\cnode`/`\rnode`/nc-*/…) resolve with correct
30  // `(coord)`-consuming signatures. Without it, `\Cnode(1,1){000}` →
31  // `undefined:\Cnode` (witness 1509.04932/1604.02906/1809.03593/…).
32  RequirePackage!("pst-node");
33  // A few connection macros below are kept as belt-and-suspenders for
34  // the rare paper that loads pst-all but somehow not pst-node; pst-node
35  // (loaded above) already defines these, so these are redundant
36  // overrides with the same no-op behavior.
37  DefMacro!("\\ncline OptionalMatch:* [] {} {}", "");
38  DefMacro!("\\nccurve OptionalMatch:* [] {} {}", "");
39  DefMacro!("\\ncarc OptionalMatch:* [] {} {}", "");
40  DefMacro!("\\ncarcbox OptionalMatch:* [] {} {} {} {} {}", "");
41  DefMacro!("\\pnode OptionalMatch:* () {}", "");
42  DefMacro!("\\nput OptionalMatch:* {{}} {} {}", "");
43  DefMacro!("\\aput OptionalMatch:* {{}} [] {}", "#3");
44  DefMacro!("\\bput OptionalMatch:* {{}} [] {}", "#3");
45  DefMacro!("\\ncangle OptionalMatch:* [] {} {}", "");
46  DefMacro!("\\ncangles OptionalMatch:* [] {} {}", "");
47  DefMacro!("\\ncloop OptionalMatch:* [] {} {}", "");
48  DefMacro!("\\ncdiag OptionalMatch:* [] {} {}", "");
49  DefMacro!("\\ncbar OptionalMatch:* [] {} {}", "");
50  DefMacro!("\\ncput OptionalMatch:* {{}} [] {}", "#3");
51  DefMacro!("\\cnodeput OptionalMatch:* [] () {} {}", "#4");
52});
