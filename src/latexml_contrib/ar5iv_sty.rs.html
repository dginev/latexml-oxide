1use latexml_package::prelude::*;
2
3LoadDefinitions!({
4  // Perl: PassOptions('latexml', 'sty', ...) + RequirePackage('latexml')
5  // Mirror Perl ar5iv.sty.ltxml: pass `rawstyles` (INCLUDE_STYLES => true,
6  // kpsewhich enabled, system-wide texmf reachable). Earlier the Rust
7  // port passed `localrawstyles` (kpsewhich suppressed, only paper-local
8  // SEARCHPATHS) per past user direction (commit 9869267eb), but that
9  // diverged from the Perl ar5iv profile and caused real parity gaps:
10  // papers using system-installed-but-unbound .sty packages
11  // (colonequals, comment, gnuplot, …) loaded fine in Perl but errored
12  // in Rust with `\<missing-cs> undefined`. Switch back to `rawstyles`
13  // for Perl-baseline parity (cf. feedback_sandbox_perl_baseline.md).
14  // Perl ar5iv.sty.ltxml ships `pushbacklimit=599999, iflimit=3999999`
15  // but the Rust port hits both ceilings on real ar5iv-profile papers
16  // (witness arXiv:2605.16752v1). Empirical bisect (2026-05-22) on that
17  // witness pinned the actual minima at: pushback ≈ 630000, iflimit
18  // ≈ 8000000.
19  //
20  // ROOT-CAUSED 2026-06-30: the iflimit gap to Perl is NOT a bug to
21  // "tighten away" — it is *deliberate, more-comprehensive runaway
22  // counting*. Rust defines `\ifx`/`\ifcsname` (and the other low-level
23  // TeX conditionals) as real `DefConditional`s that increment the global
24  // `if_count` runaway guard; Perl does NOT count `\ifx`/`\ifcsname` toward
25  // its `if_count` at all. On pgfkeys-driven tikz/pgfplots input (both
26  // engines raw-load the real `pgfkeys.code.tex` — Perl's native pgfkeys
27  // override is `__END__`-disabled), the gap is enormous: a controlled
28  // 2-plot pgfplots figure that BOTH engines render identically (≈86 vs 87
29  // graphic nodes) counts **148,078** conditionals in Rust vs **<200** in
30  // Perl (≈740×), dominated by `\ifx` (63%) + `\ifcsname` (15%) inside the
31  // pgfkeys key-dispatch. Counting these is *correct* — it is exactly what
32  // the guard is for (a `\ifx`/`\ifcsname` runaway is invisible to Perl's
33  // counter). So the right response is to raise the limit, not to count
34  // less. Real *finite* heavy docs (multi-figure pgfplots papers) measure
35  // ≈10–15M conditionals and complete in ≈24–43 s; a genuine runaway is
36  // still caught well before the worker wall-clock lease (≈350k cond/s ⇒
37  // 16M in ≈46 s ≪ the 180 s lease) and by the RSS fuse. **iflimit raised
38  // 8M → 16M (4× Perl's 3,999,999)** to recover coverage on the ≈17-paper
39  // `\tikz@dashphase` Timeout cluster that Perl cannot convert at all
40  // (Perl chokes on these papers' expl3 first). Pre-approved 2026-06-30.
41  //
42  // tokenlimit RECALIBRATED 2026-06-10 (PR #249 review P1-2): the gullet
43  // read checkpoints now count in all three reader loops (was: read_token
44  // only), so the old 249999999 — calibrated under the old accounting —
45  // silently shrank by the multi-counting factor. Measured heaviest
46  // known-good ar5iv-profile paper under the new accounting: math0402448 at
47  // 80.2M (`Info:gullet:progress`). 999999999 keeps the canvas profile's
48  // generous backstop posture (runaways are cut much earlier by the cycle
49  // guards / pushbacklimit / byte budget; the tokenlimit only bounds
50  // aperiodic grind).
51  pass_options("latexml", "sty", vec![
52    s!("ids"),
53    s!("rawstyles"),
54    s!("bibconfig=bbl,bib"),
55    s!("nobreakuntex"),
56    s!("magnify=1.2"),
57    s!("zoomout=1.2"),
58    s!("tokenlimit=999999999"),
59    s!("iflimit=16000000"),
60    s!("absorblimit=1299999"),
61    s!("pushbacklimit=650000"),
62  ])?;
63  RequirePackage!("latexml");
64
65  // Practical maximum for warnings
66  AssignValue!("MAX_WARNINGS" => 10000i64, Scope::Global);
67
68  // No \today in archival conversions. Perl L23-25:
69  //   AtBeginDocument(sub {
70  //     DefMacroI('\today', undef, '\relax', locked => 1, scope => 'global');
71  //   });
72  // We bind at load time with `locked => true, Scope::Global` instead of
73  // wrapping in `\AtBeginDocument{\def\today{\relax}}` (which loses both
74  // flags — `\def` is plain-TeX, with no LaTeXML lock). The lock makes
75  // timing irrelevant: any preamble package that tries to (re)define
76  // \today after this point is silently rejected, matching the intent of
77  // Perl's AtBeginDocument hook (defer until all packages have loaded).
78  DefMacro!("\\today", "\\relax", locked => true, scope => Some(Scope::Global));
79
80  // Perl L30-35: drop all non-remote <ltx:resource> nodes (keep only `http*`
81  // src so the archival run doesn't embed default local CSS / JS).
82  //   DefRewrite(xpath => 'descendant-or-self::ltx:resource', replace => sub {
83  //     my ($self, $node) = @_;
84  //     my $src = $node->getAttribute('src') || '';
85  //     return if $src !~ /^http/;      # non-remote → silently drop
86  //     $self->getNode->appendChild($node); });   # remote → re-attach
87  DefRewrite!(xpath => "descendant-or-self::ltx:resource",
88  replace => sub[document, nodes] {
89    let node = nodes.pop().unwrap();
90    let src = node.get_attribute("src").unwrap_or_default();
91    if src.starts_with("http") {
92      document.get_node_mut().add_child(node)?;
93    }
94  });
95
96  // NOTE: Perl additionally monkey-patches LaTeXML::Post::MathML::outerWrapper
97  // to set intent=':literal' on the top-level math element. That is a
98  // post-processing hook (not a compile-time binding), tracked separately —
99  // we do not emulate it here. See Perl source L45-73 for context.
100});
