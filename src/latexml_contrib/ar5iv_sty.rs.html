1use latexml_package::prelude::*;
2
3LoadDefinitions!({
4  // Perl: PassOptions('latexml', 'sty', ...) + RequirePackage('latexml')
5  // Mirror Perl ar5iv.sty.ltxml: pass `rawstyles` (INCLUDE_STYLES => true,
6  // kpsewhich enabled, system-wide texmf reachable). Earlier the Rust
7  // port passed `localrawstyles` (kpsewhich suppressed, only paper-local
8  // SEARCHPATHS) per past user direction (commit 9869267eb), but that
9  // diverged from the Perl ar5iv profile and caused real parity gaps:
10  // papers using system-installed-but-unbound .sty packages
11  // (colonequals, comment, gnuplot, …) loaded fine in Perl but errored
12  // in Rust with `\<missing-cs> undefined`. Switch back to `rawstyles`
13  // for Perl-baseline parity (cf. feedback_sandbox_perl_baseline.md).
14  // Perl ar5iv.sty.ltxml ships `pushbacklimit=599999, iflimit=3999999`
15  // but the Rust port hits both ceilings on real ar5iv-profile papers
16  // (witness arXiv:2605.16752v1). Empirical bisect (2026-05-22) on that
17  // witness pinned the actual minima at: pushback ≈ 630000, iflimit
18  // ≈ 8000000. We pick 650000 / 8000000 with a small safety margin over
19  // the minimum-pass bracket. The gap to Perl's defaults suggests our
20  // pushback / conditional accounting is roughly 1.1× Perl's (pushback)
21  // and 2× Perl's (iflimit) on heavily macro-driven IEEEtran + tikz +
22  // pgfplots input; root-causing those is tracked as a follow-up but
23  // would let us tighten the constants back toward the Perl defaults.
24  latexml_core::binding::content::pass_options("latexml", "sty", vec![
25    s!("ids"),
26    s!("rawstyles"),
27    s!("bibconfig=bbl,bib"),
28    s!("nobreakuntex"),
29    s!("magnify=1.2"),
30    s!("zoomout=1.2"),
31    s!("tokenlimit=249999999"),
32    s!("iflimit=8000000"),
33    s!("absorblimit=1299999"),
34    s!("pushbacklimit=650000"),
35  ])?;
36  RequirePackage!("latexml");
37
38  // Practical maximum for warnings
39  AssignValue!("MAX_WARNINGS" => 10000i64, Scope::Global);
40
41  // No \today in archival conversions. Perl L23-25:
42  //   AtBeginDocument(sub {
43  //     DefMacroI('\today', undef, '\relax', locked => 1, scope => 'global');
44  //   });
45  // We bind at load time with `locked => true, Scope::Global` instead of
46  // wrapping in `\AtBeginDocument{\def\today{\relax}}` (which loses both
47  // flags — `\def` is plain-TeX, with no LaTeXML lock). The lock makes
48  // timing irrelevant: any preamble package that tries to (re)define
49  // \today after this point is silently rejected, matching the intent of
50  // Perl's AtBeginDocument hook (defer until all packages have loaded).
51  DefMacro!("\\today", "\\relax", locked => true, scope => Some(Scope::Global));
52
53  // Perl L30-35: drop all non-remote <ltx:resource> nodes (keep only `http*`
54  // src so the archival run doesn't embed default local CSS / JS).
55  //   DefRewrite(xpath => 'descendant-or-self::ltx:resource', replace => sub {
56  //     my ($self, $node) = @_;
57  //     my $src = $node->getAttribute('src') || '';
58  //     return if $src !~ /^http/;      # non-remote → silently drop
59  //     $self->getNode->appendChild($node); });   # remote → re-attach
60  DefRewrite!(xpath => "descendant-or-self::ltx:resource",
61  replace => sub[document, nodes] {
62    let node = nodes.pop().unwrap();
63    let src = node.get_attribute("src").unwrap_or_default();
64    if src.starts_with("http") {
65      document.get_node_mut().add_child(node)?;
66    }
67  });
68
69  // NOTE: Perl additionally monkey-patches LaTeXML::Post::MathML::outerWrapper
70  // to set intent=':literal' on the top-level math element. That is a
71  // post-processing hook (not a compile-time binding), tracked separately —
72  // we do not emulate it here. See Perl source L45-73 for context.
73});
