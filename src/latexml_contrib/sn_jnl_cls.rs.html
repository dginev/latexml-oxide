26  DefMacro!("\\bmhead{}", "\\subsubsection*{#1}");
27  DefMacro!("\\bmsection{}", "\\section*{#1}");

40  // Author-block — preserve author-supplied affiliation / equalcont /
41  // presentaddress content as ltx:note frontmatter.
42  //
43  // sn-jnl.cls defines `\author` to dispatch on `\@ifstar` to either
44  // `\@@corrauthor` (corresponding author) or `\@@author` (regular).
45  // We collapse both to the core `\author{#name}` semantics — we don't
46  // distinguish corresponding from regular in the output.
47  //
48  // CAVEAT (root-cause for 2306.11901): our DefMacro prototype parser
49  // treats the `*` immediately after `\author` as a Token parameter
50  // (literal star), NOT as a CS suffix. So a naive
51  //   `DefMacro!("\\author*[]{}", "\\author{#2}")`
52  // overrides `\author` to *require* a literal star and then the body
53  // `\author{#2}` calls itself with the empty optional `#2`, recursing
54  // forever on `\author{X}` (which has no star and no [opt]). Use
55  // `OptionalMatch:*` so the star is optional, save the core `\author`
56  // first, and forward to it with the full `[opt]{name}` shape so the
57  // body never re-enters this stub.
58  state::let_i(&T_CS!("\\lx@sn@core@author"), &T_CS!("\\author"), None);
59  DefMacro!("\\author OptionalMatch:* []{}",
60    "\\lx@sn@core@author[#2]{#3}");
61  DefMacro!("\\affil OptionalMatch:* []{}",
62    "\\@add@frontmatter{ltx:note}[role=affiliation]{#3}");
63  DefMacro!("\\equalcont{}",
64    "\\@add@frontmatter{ltx:note}[role=equal-contributors]{#1}");
65  DefMacro!("\\presentaddress{}",
66    "\\@add@frontmatter{ltx:note}[role=present-address]{#1}");
67  // Name part helpers (first-name, surname) — emit inline.
68  DefMacro!("\\fnm{}", "#1");
69  DefMacro!("\\sur{}", "#1");
70  // sn-jnl.cls L599-606: \orgdiv / \orgname / \orgaddress / \street /
71  // \postcode / \city / \state / \country — affiliation-element helpers
72  // that pass through their argument as inline text. The paper-bundled
73  // class file defines all of them as `\newcommand{\foo}[1]{#1}` (plain
74  // pass-through). Our raw-load path doesn't always invoke them, so
75  // bind explicitly. Without these stubs, papers using the standard
76  // sn-jnl `\affil*[1]{\orgdiv{...}, \orgname{...}, \orgaddress{...}}`
77  // pattern report undefined CS cascade. Witness 2311.09249, 2311.08387.
78  DefMacro!("\\orgdiv{}",     "#1");
79  DefMacro!("\\orgname{}",    "#1");
80  DefMacro!("\\orgaddress{}", "#1");
81  DefMacro!("\\street{}",     "#1");
82  DefMacro!("\\postcode{}",   "#1");
83  DefMacro!("\\city{}",       "#1");
84  DefMacro!("\\state{}",      "#1");
85  DefMacro!("\\country{}",    "#1");
86  // sn-jnl.cls defines \botrule as a bottom-rule table separator
87  // (similar shape to \toprule / \midrule from booktabs). Authors use
88  // it inside \begin{tabular}...\end{tabular} for Springer-Nature
89  // bottom rules. Map to \hline so the table still renders.
90  // Witness 2402.17342.
91  Let!("\\botrule", "\\hline");

93  // Frontmatter envs — internal_vertical mode for multi-paragraph
94  // bodies (declarations especially carries author prose with \par
95  // separators). Without explicit mode, restricted_horizontal default
96  // trips Endgroup mismatch on \par-containing bodies.
97  DefEnvironment!("{abstract}", "<ltx:abstract>#body</ltx:abstract>",
98    mode => "internal_vertical");
99  // Real sn-jnl.cls defines `\abstract{...}` as a *macro* form (not the
100  // standard env). Without this stub, our environment binding's
101  // auto-created `\abstract` token (which expects a matching
102  // `\endabstract`) eats every subsequent `\title`/`\author`/`\maketitle`
103  // /`\section` into the still-open `<ltx:abstract>`, producing a
104  // cascade of `Error:malformed:ltx:* isn't allowed in <ltx:abstract>`.
105  // Forward to the env so the body is wrapped *and* properly closed.
106  // Witness 2306.11901.
107  DefMacro!("\\abstract{}",
108    "\\begin{abstract}#1\\end{abstract}");
109  DefEnvironment!("{declarations}", "<ltx:acknowledgements name='declarations'>#body</ltx:acknowledgements>",
110    mode => "internal_vertical");
111  DefEnvironment!("{appendices}", "<ltx:appendix>#body</ltx:appendix>",
112    mode => "internal_vertical");
