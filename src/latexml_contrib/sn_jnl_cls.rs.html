10  // preload). A preloaded xcolor makes a later `\usepackage[table]{xcolor}`
11  // a no-op → colortbl/array never load → array `m{}`/`b{}` columns are
12  // "Unrecognized tabular template" → "Extra alignment tab". The document
13  // loads xcolor with its own options; `\color`/`\definecolor` stay
14  // available via hyperref→color. See ifacconf_cls.rs / SYNC_STATUS.

16  RequirePackage!("graphicx");

18  // call \\geometry{margin=2cm} without an explicit usepackage.
19  // Witness 2503.06846.

21  // sn-jnl.cls L615-618 raw-loads algorithm + algorithmicx +
22  // algpseudocode so papers can use `\begin{algorithm}` and
23  // `\begin{algorithmic}` (along with `\State`/`\If`/`\For`/`\While`)
24  // without an explicit `\usepackage{...}`. Witness 2201.08889.
25  RequirePackage!("algorithm");
26  RequirePackage!("algorithmicx");
27  RequirePackage!("algpseudocode");
28  // Real sn-jnl.cls L298/301/302 raw-loads multirow, mathrsfs and
29  // `[figuresright]{rotating}`. Because this binding short-circuits the
30  // unbound-class dependency scan (a real `.cls` binding is responsible for
31  // its own `\RequirePackage`s), those deps would otherwise stay unloaded:
32  // a paper using `\begin{sidewaystable}` (rotating) then hits
33  // `undefined:{sidewaystable}` + `\caption outside any known float`, where
34  // Perl — which ships NO sn-jnl binding and so OmniBus-dep-scans the raw
35  // .cls — loads rotating and is clean. `rotating`'s `figuresright` option
36  // is pure figure-orientation (no XML signal). Add the three benign,
37  // commonly-used deps (NOT xcolor — see the option-conflict note above).
38  // Witness 2101.02753 (`\begin{sidewaystable}` ×2).
39  RequirePackage!("multirow");
40  RequirePackage!("mathrsfs");
41  RequirePackage!("rotating");

43  // Real sn-jnl.cls loads natbib for EVERY reference style (L1649/1652/1662/
44  // 1669/1677: `\usepackage[numbers,sort&compress]{natbib}` for the numeric
45  // styles, `\usepackage[authoryear]{natbib}` for the author-year ones).
46  // OmniBus only `def_autoload`s natbib off `\citet`/`\citep`/`\citeyear`/…
47  // — deliberately NOT off `\cite`, which the kernel already defines. So a
48  // paper that cites solely via natbib's TWO-optional `\cite[pre][post]{keys}`
49  // never triggers the autoload, and the kernel's single-optional
50  // `\cite[] Semiverbatim` reads `[` as the whole key list: the real keys are
51  // dropped (never cited, silently absent from the References) and `]{keys}`
52  // leaks as body text. Requiring natbib here mirrors the real class and also
53  // spares these papers the autoload dance. Options are left at natbib's
54  // default (numbers) — same as what the autoload path already produced, so
55  // no rendering change for papers that did trigger it; the per-refstyle
56  // numbers-vs-authoryear mapping needs class-option processing this stub
57  // doesn't do yet. Witness 2605.23484 (sn-mathphys-num), 2606.10002
58  // (sn-basic), 2606.10215, 2606.11534.
59  RequirePackage!("natbib");

61  // sn-jnl frontmatter — gobble layout-only / preserve author text.
62  DefMacro!("\\bmhead{}", "\\subsubsection*{#1}");
63  DefMacro!("\\bmsection{}", "\\section*{#1}");
64  // \sectiontitle{text} carries an author-typed section title used in
65  // sn-jnl's TOC/running-head pipeline. Preserve as ltx:note rather
66  // than silently dropping the words. Content-preserving.
67  DefMacro!(
68    "\\sectiontitle{}",
69    "\\@add@frontmatter{ltx:note}[role=sectiontitle]{#1}"
70  );

78  // Author-block — preserve author-supplied affiliation / equalcont /
79  // presentaddress content as ltx:note frontmatter.
80  //
81  // sn-jnl.cls defines `\author` to dispatch on `\@ifstar` to either
82  // `\@@corrauthor` (corresponding author) or `\@@author` (regular).
83  // We collapse both to the core `\author{#name}` semantics — we don't
84  // distinguish corresponding from regular in the output.
85  //
86  // CAVEAT (root-cause for 2306.11901): our DefMacro prototype parser
87  // treats the `*` immediately after `\author` as a Token parameter
88  // (literal star), NOT as a CS suffix. So a naive
89  //   `DefMacro!("\\author*[]{}", "\\author{#2}")`
90  // overrides `\author` to *require* a literal star and then the body
91  // `\author{#2}` calls itself with the empty optional `#2`, recursing
92  // forever on `\author{X}` (which has no star and no [opt]). Use
93  // `OptionalMatch:*` so the star is optional, save the core `\author`
94  // first, and forward to it with the full `[opt]{name}` shape so the
95  // body never re-enters this stub.
96  let_i(&T_CS!("\\lx@sn@core@author"), &T_CS!("\\author"), None);
97  DefMacro!(
98    "\\author OptionalMatch:* []{}",
99    "\\lx@sn@core@author[#2]{#3}"
100  );
101  DefMacro!(
102    "\\affil OptionalMatch:* []{}",
103    "\\@add@frontmatter{ltx:note}[role=affiliation]{#3}"
104  );
105  DefMacro!(
106    "\\equalcont{}",
107    "\\@add@frontmatter{ltx:note}[role=equal-contributors]{#1}"
108  );
109  DefMacro!(
110    "\\presentaddress{}",
111    "\\@add@frontmatter{ltx:note}[role=present-address]{#1}"
112  );
113  // sn-jnl.cls L1788: \gdef\orcid#1{\href{#1}{\orcidlogo}} (ORCID-logo
114  // hyperlink, used INLINE inside \author{...\orcid{id}}). Render as an inline
115  // ORCID link showing the id (content-preserving: the id stays in the output,
116  // both as visible text and the proper orcid.org href; the real cls's logo
117  // image Orcidlogo.eps isn't shipped). A frontmatter note does NOT work here —
118  // it fires inside the author-name digestion and is dropped. Was undefined →
119  // error (Perl defines it). Witness 2211.09693.
120  DefMacro!("\\orcid{}", "\\href{https://orcid.org/#1}{#1}");
121  // \sep — elsarticle-family keyword separator (`\def\sep{\unskip, }`). sn-jnl
122  // papers use it in keyword lists (`kw1 \sep kw2 \sep ...`); the sn-jnl cls
123  // doesn't define it but Perl provides it (elsarticle convention). Render as a
124  // comma separator. Was undefined → error. Witness 2309.06763.
125  DefMacro!("\\sep", "\\unskip, ");
126  // Name part helpers (first-name, surname) — emit inline.
127  DefMacro!("\\fnm{}", "#1");
128  DefMacro!("\\sur{}", "#1");
129  // sn-jnl.cls L599-606: \orgdiv / \orgname / \orgaddress / \street /
130  // \postcode / \city / \state / \country — affiliation-element helpers
131  // that pass through their argument as inline text. The paper-bundled
132  // class file defines all of them as `\newcommand{\foo}[1]{#1}` (plain
133  // pass-through). Our raw-load path doesn't always invoke them, so
134  // bind explicitly. Without these stubs, papers using the standard
135  // sn-jnl `\affil*[1]{\orgdiv{...}, \orgname{...}, \orgaddress{...}}`
136  // pattern report undefined CS cascade. Witness 2311.09249, 2311.08387.

138  DefMacro!("\\orgname{}", "#1");
139  DefMacro!("\\orgaddress{}", "#1");
140  DefMacro!("\\street{}", "#1");
141  DefMacro!("\\postcode{}", "#1");
142  DefMacro!("\\city{}", "#1");
143  DefMacro!("\\state{}", "#1");
144  DefMacro!("\\country{}", "#1");
145  // sn-jnl.cls defines \botrule as a bottom-rule table separator
146  // (similar shape to \toprule / \midrule from booktabs). Authors use
147  // it inside \begin{tabular}...\end{tabular} for Springer-Nature
148  // bottom rules. Map to \hline so the table still renders.
149  // Witness 2402.17342.
150  Let!("\\botrule", "\\hline");

152  // Frontmatter envs — internal_vertical mode for multi-paragraph
153  // bodies (declarations especially carries author prose with \par
154  // separators). Without explicit mode, restricted_horizontal default
155  // trips Endgroup mismatch on \par-containing bodies.
156  DefEnvironment!("{abstract}", "<ltx:abstract>#body</ltx:abstract>",
157    mode => "internal_vertical");
158  // Real sn-jnl.cls defines `\abstract{...}` as a *macro* form (not the
159  // standard env). Without this stub, our environment binding's
160  // auto-created `\abstract` token (which expects a matching
161  // `\endabstract`) eats every subsequent `\title`/`\author`/`\maketitle`
162  // /`\section` into the still-open `<ltx:abstract>`, producing a
163  // cascade of `Error:malformed:ltx:* isn't allowed in <ltx:abstract>`.
164  // Forward to the env so the body is wrapped *and* properly closed.
165  // Witness 2306.11901.
166  DefMacro!("\\abstract{}", "\\begin{abstract}#1\\end{abstract}");
167  DefEnvironment!("{declarations}", "<ltx:acknowledgements name='declarations'>#body</ltx:acknowledgements>",
168    mode => "internal_vertical");
169  DefEnvironment!("{appendices}", "<ltx:appendix>#body</ltx:appendix>",
170    mode => "internal_vertical");
