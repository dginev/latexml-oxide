9  // Do NOT eager-load xcolor (Perl ships no sn-jnl binding → OmniBus, no
10  // preload). A preloaded xcolor makes a later `\usepackage[table]{xcolor}`
11  // a no-op → colortbl/array never load → array `m{}`/`b{}` columns are
12  // "Unrecognized tabular template" → "Extra alignment tab". The document
13  // loads xcolor with its own options; `\color`/`\definecolor` stay
14  // available via hyperref→color. See ifacconf_cls.rs / SYNC_STATUS.
15  RequirePackage!("hyperref");
16  RequirePackage!("graphicx");
17  // Real sn-jnl.cls loads geometry for page setup — papers commonly
18  // call \\geometry{margin=2cm} without an explicit usepackage.
19  // Witness 2503.06846.
20  RequirePackage!("geometry");
21  // sn-jnl.cls L615-618 raw-loads algorithm + algorithmicx +
22  // algpseudocode so papers can use `\begin{algorithm}` and
23  // `\begin{algorithmic}` (along with `\State`/`\If`/`\For`/`\While`)
24  // without an explicit `\usepackage{...}`. Witness 2201.08889.
25  RequirePackage!("algorithm");
26  RequirePackage!("algorithmicx");
27  RequirePackage!("algpseudocode");
28  // Real sn-jnl.cls L298/301/302 raw-loads multirow, mathrsfs and
29  // `[figuresright]{rotating}`. Because this binding short-circuits the
30  // unbound-class dependency scan (a real `.cls` binding is responsible for
31  // its own `\RequirePackage`s), those deps would otherwise stay unloaded:
32  // a paper using `\begin{sidewaystable}` (rotating) then hits
33  // `undefined:{sidewaystable}` + `\caption outside any known float`, where
34  // Perl — which ships NO sn-jnl binding and so OmniBus-dep-scans the raw
35  // .cls — loads rotating and is clean. `rotating`'s `figuresright` option
36  // is pure figure-orientation (no XML signal). Add the three benign,
37  // commonly-used deps (NOT xcolor — see the option-conflict note above).
38  // Witness 2101.02753 (`\begin{sidewaystable}` ×2).
39  RequirePackage!("multirow");
40  RequirePackage!("mathrsfs");
41  RequirePackage!("rotating");

43  // sn-jnl frontmatter — gobble layout-only / preserve author text.
44  DefMacro!("\\bmhead{}", "\\subsubsection*{#1}");
45  DefMacro!("\\bmsection{}", "\\section*{#1}");
46  // \sectiontitle{text} carries an author-typed section title used in
47  // sn-jnl's TOC/running-head pipeline. Preserve as ltx:note rather
48  // than silently dropping the words. Content-preserving.
49  DefMacro!(
50    "\\sectiontitle{}",
51    "\\@add@frontmatter{ltx:note}[role=sectiontitle]{#1}"
52  );
53  // \headtype{...} / \extralength{...} are layout knobs (no author body).
54  def_macro_noop("\\headtype{}")?;
55  def_macro_noop("\\extralength{}")?;
56  // \theHfigure / \theHtable are hyperref H-counter overrides (no body).
57  def_macro_noop("\\theHfigure{}")?;
58  def_macro_noop("\\theHtable{}")?;

60  // Author-block — preserve author-supplied affiliation / equalcont /
61  // presentaddress content as ltx:note frontmatter.
62  //
63  // sn-jnl.cls defines `\author` to dispatch on `\@ifstar` to either
64  // `\@@corrauthor` (corresponding author) or `\@@author` (regular).
65  // We collapse both to the core `\author{#name}` semantics — we don't
66  // distinguish corresponding from regular in the output.
67  //
68  // CAVEAT (root-cause for 2306.11901): our DefMacro prototype parser
69  // treats the `*` immediately after `\author` as a Token parameter
70  // (literal star), NOT as a CS suffix. So a naive
71  //   `DefMacro!("\\author*[]{}", "\\author{#2}")`
72  // overrides `\author` to *require* a literal star and then the body
73  // `\author{#2}` calls itself with the empty optional `#2`, recursing
74  // forever on `\author{X}` (which has no star and no [opt]). Use
75  // `OptionalMatch:*` so the star is optional, save the core `\author`
76  // first, and forward to it with the full `[opt]{name}` shape so the
77  // body never re-enters this stub.
78  let_i(&T_CS!("\\lx@sn@core@author"), &T_CS!("\\author"), None);
79  DefMacro!(
80    "\\author OptionalMatch:* []{}",
81    "\\lx@sn@core@author[#2]{#3}"
82  );
83  DefMacro!(
84    "\\affil OptionalMatch:* []{}",
85    "\\@add@frontmatter{ltx:note}[role=affiliation]{#3}"
86  );
87  DefMacro!(
88    "\\equalcont{}",
89    "\\@add@frontmatter{ltx:note}[role=equal-contributors]{#1}"
90  );
91  DefMacro!(
92    "\\presentaddress{}",
93    "\\@add@frontmatter{ltx:note}[role=present-address]{#1}"
94  );
95  // sn-jnl.cls L1788: \gdef\orcid#1{\href{#1}{\orcidlogo}} (ORCID-logo
96  // hyperlink, used INLINE inside \author{...\orcid{id}}). Render as an inline
97  // ORCID link showing the id (content-preserving: the id stays in the output,
98  // both as visible text and the proper orcid.org href; the real cls's logo
99  // image Orcidlogo.eps isn't shipped). A frontmatter note does NOT work here —
100  // it fires inside the author-name digestion and is dropped. Was undefined →
101  // error (Perl defines it). Witness 2211.09693.
102  DefMacro!("\\orcid{}", "\\href{https://orcid.org/#1}{#1}");
103  // \sep — elsarticle-family keyword separator (`\def\sep{\unskip, }`). sn-jnl
104  // papers use it in keyword lists (`kw1 \sep kw2 \sep ...`); the sn-jnl cls
105  // doesn't define it but Perl provides it (elsarticle convention). Render as a
106  // comma separator. Was undefined → error. Witness 2309.06763.
107  DefMacro!("\\sep", "\\unskip, ");
108  // Name part helpers (first-name, surname) — emit inline.
109  DefMacro!("\\fnm{}", "#1");
110  DefMacro!("\\sur{}", "#1");
111  // sn-jnl.cls L599-606: \orgdiv / \orgname / \orgaddress / \street /
112  // \postcode / \city / \state / \country — affiliation-element helpers
113  // that pass through their argument as inline text. The paper-bundled
114  // class file defines all of them as `\newcommand{\foo}[1]{#1}` (plain
115  // pass-through). Our raw-load path doesn't always invoke them, so
116  // bind explicitly. Without these stubs, papers using the standard
117  // sn-jnl `\affil*[1]{\orgdiv{...}, \orgname{...}, \orgaddress{...}}`
118  // pattern report undefined CS cascade. Witness 2311.09249, 2311.08387.
119  DefMacro!("\\orgdiv{}", "#1");
120  DefMacro!("\\orgname{}", "#1");
121  DefMacro!("\\orgaddress{}", "#1");
122  DefMacro!("\\street{}", "#1");
123  DefMacro!("\\postcode{}", "#1");
124  DefMacro!("\\city{}", "#1");
125  DefMacro!("\\state{}", "#1");
126  DefMacro!("\\country{}", "#1");
127  // sn-jnl.cls defines \botrule as a bottom-rule table separator
128  // (similar shape to \toprule / \midrule from booktabs). Authors use
129  // it inside \begin{tabular}...\end{tabular} for Springer-Nature
130  // bottom rules. Map to \hline so the table still renders.
131  // Witness 2402.17342.
132  Let!("\\botrule", "\\hline");

134  // Frontmatter envs — internal_vertical mode for multi-paragraph
135  // bodies (declarations especially carries author prose with \par
136  // separators). Without explicit mode, restricted_horizontal default
137  // trips Endgroup mismatch on \par-containing bodies.
138  DefEnvironment!("{abstract}", "<ltx:abstract>#body</ltx:abstract>",
139    mode => "internal_vertical");
140  // Real sn-jnl.cls defines `\abstract{...}` as a *macro* form (not the
141  // standard env). Without this stub, our environment binding's
142  // auto-created `\abstract` token (which expects a matching
143  // `\endabstract`) eats every subsequent `\title`/`\author`/`\maketitle`
144  // /`\section` into the still-open `<ltx:abstract>`, producing a
145  // cascade of `Error:malformed:ltx:* isn't allowed in <ltx:abstract>`.
146  // Forward to the env so the body is wrapped *and* properly closed.
147  // Witness 2306.11901.
148  DefMacro!("\\abstract{}", "\\begin{abstract}#1\\end{abstract}");
149  DefEnvironment!("{declarations}", "<ltx:acknowledgements name='declarations'>#body</ltx:acknowledgements>",
150    mode => "internal_vertical");
151  DefEnvironment!("{appendices}", "<ltx:appendix>#body</ltx:appendix>",
152    mode => "internal_vertical");
