1//! Stub for jgcc.cls (JGCC journal class).
2//!
3//! jgcc.cls L429-450 defines `\jgccdoi{vol}{issue}{paper}{paperid}`
4//! and `\jgccheading{vol}{issue}{year}{pages}{subm}{publ}{rev}`. Same
5//! pattern as LMCS — raw cls preamble (font/page-layout machinery)
6//! fails mid-load, leaving these undefined. Witness 2308.10812,
7//! 2308.02610, 2309.06144.
8use latexml_package::prelude::*;
9
10LoadDefinitions!({
11  LoadClass!("article");
12  RequirePackage!("amsmath");
13  RequirePackage!("amsthm");
14  RequirePackage!("amssymb");
15  RequirePackage!("hyperref");
16
17  // JGCC publication metadata.
18  DefMacro!(
19    "\\jgccdoi{}{}{}{}",
20    "\\@add@frontmatter{ltx:note}[role=jgcc-doi]{Volume #1, Issue #2, Paper #3 (#4)}"
21  );
22  // \jgccheading uses 7-9 args depending on variant; capture the
23  // common 7 explicitly (vol/issue/year/pages/subm/publ/rev).
24  def_macro_noop("\\jgccheading{}{}{}{}{}{}{}")?;
25
26  DefMacro!("\\jgccorcid{}", "\\href{https://orcid.org/#1}{ORCID:#1}");
27
28  DefConditional!("\\ifjgccheadingcalled");
29  def_macro_noop("\\jgccheadingcalledtrue")?;
30  def_macro_noop("\\jgccheadingcalledfalse")?;
31});
