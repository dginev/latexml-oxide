1use latexml_package::prelude::*;
2
3LoadDefinitions!({
4  RequirePackage!("keyval");
5
6  DefKeyVal!("foo", "path", "Semiverbatim");
7
8  DefConstructor!(
9    "\\KVsimple OptionalKeyVals:foo",
10    "<ltx:graphics graphic='none' options='#1'/>"
11  );
12
13  DefConstructor!(
14    "\\KVcomplex OptionalKeyVals:foo",
15    "<ltx:graphics graphic='none' imagewidth='&GetKeyVal(#1,width)' imageheight='&GetKeyVal(#1,height)'/>"
16  );
17
18  DefEnvironment!(
19    "{KVenv} OptionalKeyVals:foo",
20    "<ltx:text width='&GetKeyVal(#1,width)' height='&GetKeyVal(#1,height)'>#body</ltx:text>"
21  );
22
23  // KVstruct: hand-written replacement closure because opening two consecutive
24  // ltx:text child elements inside an environment body triggers a double-free
25  // in the libxml node management. Instead, we use absorb_string for inline text.
26  {
27    let replacement: ReplacementClosure = Rc::new(
28      |document: &mut Document, args: &Vec<Option<Digested>>, props: &arena::SymHashMap<Stored>| {
29        document.open_element("ltx:text", None, None)?;
30        // ?&GetKeyVal(#1,width)(Width: &GetKeyVal(#1,width))
31        if let Some(width) = GetKeyVal(&args[0], "width") {
32          let w_str = width.to_string();
33          if !w_str.is_empty() && w_str != "false" {
34            document.absorb_string(&format!("Width: {}", w_str), props)?;
35          }
36        }
37        // &amp; (literal ampersand)
38        document.absorb_string("&", props)?;
39        // ?&GetKeyVal(#1,height)(Height: &GetKeyVal(#1,height))
40        if let Some(height) = GetKeyVal(&args[0], "height") {
41          let h_str = height.to_string();
42          if !h_str.is_empty() && h_str != "false" {
43            document.absorb_string(&format!("Height: {}", h_str), props)?;
44          }
45        }
46        // #body
47        if let Some(body_stored) = props.get("body") {
48          let digested_opt: Option<Digested> = body_stored.into();
49          if let Some(ref digested) = digested_opt {
50            document.absorb(digested, None)?;
51          }
52        }
53        document.close_element("ltx:text")?;
54        Ok(())
55      },
56    );
57    let options = ConstructorOptions::default();
58    let name = "KVstruct".to_string();
59    let cs = T_CS!(s!("\\{}", &name));
60    let paramlist = parse_parameters("OptionalKeyVals:foo", &cs, true)?;
61    def_environment(name, paramlist, Some(replacement), options);
62  }
63
64  DefConstructor!("\\KVauto RequiredKeyVals:foo", "<ltx:text>#1</ltx:text>");
65});
