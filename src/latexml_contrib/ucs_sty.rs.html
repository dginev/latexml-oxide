12  DefMacro!("\\unichar Expanded", sub[(hexcode)] {
13    let char_str = hexcode.to_string();
14    if char_str.chars().all(|c| c.is_ascii_hexdigit()) {
15      if let Ok(cp) = u32::from_str_radix(&char_str, 16) {
16        if cp <= 0x10FFFF
17          && let Some(ch) = char::from_u32(cp) {

19          }
20        Error!("unexpected", &char_str,
21          format!("{} too large for Unicode. Values between 0 and 10FFFF are permitted.",
22                  char_str));
23      }
24    } else {
