1//! scrreprt.cls — the KOMA-Script report class (chapter-based, like `report` /
2//! `scrbook`).
3//!
4//! Raw-loading a real `.cls` is not yet reliable in latexml-oxide (the class
5//! bootstrap infrastructure is a long-term goal), so — exactly like
6//! `scrbook_cls` / `scrartcl_cls` — this binding maps scrreprt onto the
7//! `OmniBus` fallback class and stubs the KOMA-specific commands a typical
8//! document touches. Perl LaTeXML has no `scrreprt.cls.ltxml` and (lacking the
9//! same raw-`.cls` support) behaves equivalently.
10use latexml_package::prelude::*;
11
12LoadDefinitions!({
13  Warn!(
14    "missing_file",
15    "scrreprt.cls",
16    "scrreprt.cls is only minimally stubbed and will not be interpreted raw."
17  );
18  LoadClass!("OmniBus");
19  // Mirror scrbook_cls.rs: the real KOMA chain transitively loads iftex, which
20  // OmniBus does not, so pull it in for `\ifpdf` / engine-detection authors.
21  RequirePackage!("iftex");
22
23  // KOMA section-font hooks — see scrbook_cls.rs for the full rationale (tocloft
24  // expands `\sectfont` / `\size@chapter` when a KOMA class is detected; as a
25  // chapter class scrreprt uses the `\size@chapter` form).
26  def_macro_noop("\\maybesffamily")?;
27  DefMacro!("\\sectfont", "\\normalcolor\\maybesffamily\\bfseries");
28  def_macro_noop("\\size@part")?;
29  def_macro_noop("\\size@partnumber")?;
30  def_macro_noop("\\size@chapter")?;
31  def_macro_noop("\\size@section")?;
32  def_macro_noop("\\size@subsection")?;
33  def_macro_noop("\\size@subsubsection")?;
34  def_macro_noop("\\size@paragraph")?;
35  def_macro_noop("\\size@subparagraph")?;
36  def_macro_noop("\\setkomafont{}{}")?;
37  def_macro_noop("\\addtokomafont{}{}")?;
38  def_macro_noop("\\setcapindent{}")?;
39  def_macro_noop("\\deffootnote[]{}{}{}")?;
40  def_macro_noop("\\deffootnotemark{}")?;
41  // KOMA page-style marks — see scrartcl_cls.rs.
42  def_macro_noop("\\headmark")?;
43  def_macro_noop("\\pagemark")?;
44
45  // KOMA `\minisec{title}`: an unnumbered, un-TOC'd run-in-ish heading below
46  // `\subparagraph`. Map to the closest standard structural heading so it lands
47  // as `ltx:paragraph` rather than erroring.
48  DefMacro!("\\minisec{}", "\\paragraph*{#1}");
49
50  // KOMA `addmargin` environment: `\begin{addmargin}[innermargin]{outermargin}`
51  // (or `{both}`) indents a block. The margin is a visual-layout concern with no
52  // structural meaning in the XML tree, so render the body transparently.
53  RawTeX!(r"\newenvironment{addmargin}[2][]{}{}");
54  RawTeX!(r"\newenvironment{addmargin*}[2][]{}{}");
55});
