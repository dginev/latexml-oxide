1//! chemnum.sty — comprehensive numbering of chemical compounds
2//! (`\cmpd`, `\refcmpd`, `\initcmpd`, `\setchemnum`, ...).
3//!
4//! Heavy expl3/xparse-based code: 20+ `\NewDocumentCommand`s and
5//! hundreds of `\cs_new_protected:Npn` declarations. Raw-loading
6//! into our engine triggers cascading errors (101 errors + fatal
7//! on the witness paper) because our expl3 support doesn't model
8//! enough of the `\tl_*` / `\msg_*` / `\keys_*` plumbing.
9//!
10//! Witness arXiv:2103.03138 — loads `chemfig + chemnum` but
11//! never invokes a single chemnum CS. Perl LaTeXML has no
12//! chemnum binding (INCLUDE_STYLES=false skips), single
13//! "missing binding" warning, conversion completes.
14//!
15//! Match Perl: stub the public API as no-ops so the raw .sty is
16//! never loaded. We lose actual compound numbering; for papers
17//! that USE `\cmpd{name}` the printed labels are gone, but the
18//! document body still renders.
19use latexml_package::prelude::*;
20
21LoadDefinitions!({
22  Warn!(
23    "missing_file",
24    "chemnum.sty",
25    "chemnum.sty is minimally stubbed — chemical-compound numbering API is no-op'd."
26  );
27  // Public chemnum API surface — consume args silently.
28  DefMacro!("\\cmpd OptionalMatch:* [] {}", "");
29  DefMacro!("\\refcmpd [] {}", "");
30  DefMacro!("\\labelcmpd [] {}", "");
31  DefMacro!("\\initcmpd [] {}", "");
32  DefMacro!("\\resetcmpd []", "");
33  DefMacro!("\\setchemnum {}", "");
34  DefMacro!("\\setcmpdproperty {} {} {}", "");
35  DefMacro!("\\setcmpdlabel {} {}", "");
36  DefMacro!("\\newcmpdcounterformat {} {}", "");
37  DefMacro!("\\replacecmpd OptionalMatch:* {}", "");
38  DefMacro!("\\cmpdprintlabelid {}", "");
39  DefMacro!("\\cmpdshowlabelmargin {}", "");
40  DefMacro!("\\cmpdshowlabelinline {}", "");
41  DefMacro!("\\chemnumshowdef {}", "");
42  DefMacro!("\\chemnumshowref {}", "");
43  DefMacro!("\\cmpdshowdef {}", "");
44  DefMacro!("\\cmpdshowref {}", "");
45  DefMacro!("\\subcmpdshowdef {} {}", "");
46  DefMacro!("\\subcmpdshowref {} {}", "");
47});
