1//! spconf.sty / INTERSPEECH2021.sty — the ICASSP/older-Interspeech frontmatter
2//! package (loaded on top of `\documentclass{article}`).
3//!
4//! Both define the single-argument conference convention
5//!   `\name{Author1$^1$, Author2$^2$, …}`  `\address{$^1$Inst … $^2$ …}`
6//!   `\email{…}`
7//! (spconf.sty L170-172, INTERSPEECH2021.sty L171-173). With no binding the
8//! raw `.sty` `\def\name#1{\gdef\@name{…}}` merely stashes the names and
9//! article's structural `\maketitle` never emits them → zero creators (witness
10//! 2309.14838, 2405.13379, 2605.10272). Route `\name` through the standard
11//! author machinery so the comma/superscript-marked list becomes structured
12//! creators, and keep `\address`/`\email` as frontmatter.
13use latexml_package::prelude::*;
14
15LoadDefinitions!({
16  // `\name{names}` — the whole author list in one argument. Hand it to `\author`
17  // (→ `\lx@add@authors`), which splits the comma / superscript-marked list into
18  // individual creators and links their affiliation superscripts.
19  DefMacro!("\\name{}", "\\author{#1}");
20  // `\address{affils}` — the (superscript-numbered) affiliation block. Preserve
21  // as a frontmatter note so the numbered institutions are kept.
22  DefMacro!(
23    "\\address{}",
24    "\\lx@add@frontmatter{ltx:note}[role=address]{#1}"
25  );
26  DefMacro!(
27    "\\email{}",
28    "\\lx@add@frontmatter{ltx:note}[role=email]{#1}"
29  );
30  // spconf uppercases the title; keep LaTeX's `\title` semantics (no forced
31  // uppercase — casing is presentational and belongs in CSS).
32  DefMacro!("\\ninept", "");
33  def_macro_noop("\\copyrightnotice{}")?;
34  def_macro_noop("\\toappear{}")?;
35});
