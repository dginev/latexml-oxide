1//! scicite.sty — Science (the journal) citation style
2//!
3//! `scicite.sty` is a slightly-modified `cite.sty` (D. Arseneau, 1989-2003)
4//! tailored for the journal Science. The raw file is 513 lines of
5//! `\edef`/`\catcode` manipulation; without a Rust binding we attempt to
6//! tokenize the whole thing and hang on its catcode dance.
7//!
8//! Like `cite.sty.ltxml` / `cite_sty.rs` (Perl + Rust), we short-circuit
9//! the raw load by defining only the public-API macros that downstream
10//! documents actually invoke (`\citeleft`, `\citeright`, `\citepunct`,
11//! …). Citation list compression / sorting / styling has no equivalent
12//! in XML output anyway — formatting is handled at the post-XSLT layer.
13//!
14//! Recovers the 7-paper hang cluster in the 10k_errors v4 sandbox
15//! (1010.2781, 1011.5494, 1102.0562, 1210.1294, 1303.2601, 1704.07345,
16//! 1706.03851 + similarly-named files) seen in v4 partial logs.
17
18use latexml_package::prelude::*;
19
20LoadDefinitions!({
21  // Mirror `cite_sty.rs` — defaults for citation formatting macros.
22  // scicite.sty has slightly different defaults (Science-journal
23  // bracketing/punctuation) but the XML-side semantics are the same:
24  // emit a parenthesized comma-separated list. Override at the post
25  // layer if a per-journal style is desired.
26  DefMacro!("\\citeleft",  "[");
27  DefMacro!("\\citeright", "]");
28  DefMacro!("\\citedash",  "--");
29  DefMacro!("\\citemid",   ", ");
30  DefMacro!("\\citepunct", ", ");
31  DefMacro!("\\citeform{}", "#1");
32
33  // scicite-specific options that papers often set in the preamble.
34  // No-op them so `\usepackage[option]{scicite}` doesn't error.
35  DefMacro!("\\nocitepunct", "");
36  DefMacro!("\\citen", "\\cite");
37});
