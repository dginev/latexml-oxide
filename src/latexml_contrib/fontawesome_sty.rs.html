3/// Runtime equivalent of `DefMacro!("\\faXxx", "\\faIconFromMacro{faXxx}")`
4/// for the 719 trivial FontAwesome 4 icon convenience macros below.
5/// See `def_fa5_icon` in `fontawesome5_sty.rs` for the FA5 sibling.
6/// Audit DEP-15: collapses each call site from a ~1.5 KiB inlined
7/// `Tokens` constant to a single function-call instruction + 1-2
8/// `&str` pointers.
9fn def_fa4_icon(suffix: &str) -> Result<()> {
10  let proto = format!("\\fa{suffix}");
11  let (cs, params) = parse_prototype(&proto, true)?;
12  let body = mouth::tokenize_internal(&format!("\\faIconFromMacro{{fa{suffix}}}"));
13  def_macro(cs, params, ExpansionBody::Tokens(body), None)?;
14  Ok(())
15}

26  // Perl L18: \faicon[opts]{name} — inline icon by name.
27  DefConstructor!("\\faicon[]{}",
28    "<ltx:inline-block aria:hidden='true' class='fa fa-#2'></ltx:inline-block>");

30  // Perl L19-27: \faIconFromMacro{name} — lowercase + kebab-ify the CamelCase
31  // macro suffix. e.g. faAlignLeft → fa-align-left. Used as the shared target
32  // of all the \faXxx convenience macros.
33  DefConstructor!("\\faIconFromMacro{}",
34    "<ltx:inline-block aria:hidden='true' class='fa #name'></ltx:inline-block>",
35    properties => sub[args] {
36      let name = args[0].as_ref().map(|a| a.to_string()).unwrap_or_default();
37      // Perl L24: s/([A-Z])/-$1/g then lc. Hyphenate before each uppercase
38      // letter, then lowercase.
39      let mut kebab = String::with_capacity(name.len() + 8);
40      for c in name.chars() {
41        if c.is_ascii_uppercase() {
42          kebab.push('-');
43          kebab.push(c.to_ascii_lowercase());
44        } else {
45          kebab.push(c);
46        }
47      }
48      Ok(stored_map!("name" => kebab))
49    });
