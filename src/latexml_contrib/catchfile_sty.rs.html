1use latexml_core::binding::content::find_file;
2use latexml_package::prelude::*;
3
4LoadDefinitions!({
5  // Perl: ar5iv-bindings/catchfile.sty.ltxml — DefMacro with Perl closure
6  // that Input()s a file and DefMacroI()s its contents into the target CS.
7  //
8  // Rust port: locate the file via find_file and read it as raw bytes,
9  // then def_macro the target CS to the slurped contents (lossy UTF-8).
10  // This lets `\CatchFileDef\paramtable{tables/main_table.tex}{}`
11  // actually populate \paramtable so downstream `\paramtable` use
12  // doesn't trip undefined. Witness 2210.08043 (mnras + CatchFileDef).
13  DefPrimitive!("\\CatchFileDef DefToken {}{}", sub[(target, path, _opts)] {
14    let path_str = path.to_string();
15    // Perl catchfile.sty.ltxml: `my $contents = Input($_[2]); DefMacroI($_[1],
16    // undef, $contents)`. DefMacroI ALWAYS defines the target CS — even when
17    // the file is missing (Input of a non-existent file yields empty content).
18    // The earlier Rust port only def_macro'd the target inside the
19    // `if let Some(disk) … if let Ok(bytes) …` guards, so a missing/unreadable
20    // file left the target UNDEFINED, unlike Perl. That breaks the common
21    // pattern of reading a previous-run aux file absent from the arXiv source
22    // — e.g. makron.sty L61 `\CatchFileEdef\tmp{\jobname.runs}{…}` then
23    // `\setcounter{…}{\tmp}`: `\tmp` was undefined where Perl is clean. Always
24    // define the target; empty body when the file can't be read. Witness
25    // 1611.01359.
26    let body = find_file(&path_str, None)
27      .and_then(|disk| std::fs::read(&disk).ok())
28      .map(|bytes| String::from_utf8_lossy(&bytes).into_owned())
29      .unwrap_or_default();
30    let tokens = mouth::tokenize_internal(&body);
31    def_macro(target, None, tokens, None)?;
32    Ok(())
33  });
34  // \CatchFileEdef variant — same shape, but in Perl edef-expands
35  // the contents before storing. We slurp + tokenize identically;
36  // expansion happens lazily when target CS is used.
37  Let!("\\CatchFileEdef", "\\CatchFileDef");
38});
