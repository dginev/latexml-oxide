11  def_macro_noop("\\pmemlabel{}")?;

18  // Mode must be internal_vertical so `$$`-display-math inside the env
19  // is recognised by the dollar handler (tex_math.rs's
20  // `BOUND_MODE.ends_with("vertical")` gate). Without this, witness
21  // 2305.09826: `$$\log_2(x)$$` inside `\begin{adjustwidth}{...}{...}`
22  // emitted Error:unexpected:_ "Script _ can only appear in math mode"
23  // for every `_` in the formula because the wrapping env's default
24  // restricted_horizontal mode kept the `$` handler from entering
25  // display math. See sibling fix in
26  // latexml_package/src/package/changepage_sty.rs.
27  DefEnvironment!("{adjustwidth} OptionalMatch:* []{}{}", "#body",
28    mode => "internal_vertical");
29  // `adjustwidth*` is a SEPARATE environment in changepage.sty (L122
30  // `\newenvironment{adjustwidth*}[2]`) — the `*` is part of the env NAME,
31  // not a `*`-argument, so the `OptionalMatch:*` on `{adjustwidth}` above
32  // never matches it. Perl has no changepage binding and raw-loads the real
33  // .sty, which defines both; our stub previously defined only the unstarred
34  // form, so `\begin{adjustwidth*}{..}{..}` (odd/even-page margin variant)
35  // raised "The environment {adjustwidth*} is not defined" (witness
36  // 2006.09676). Mirror the unstarred env (the odd/even-page distinction in
37  // the real def is moot for our paradigm — both branches just set list
38  // margins, which we ignore). Same `internal_vertical` mode fix for
39  // `$$`-display-math inside the env.
40  DefEnvironment!("{adjustwidth*} []{}{}", "#body",
41    mode => "internal_vertical");
42  def_macro_noop("\\changetext{}{}{}{}{}")?;
43  def_macro_noop("\\changepage{}{}{}{}{}{}{}{}{}")?;
44});
