1//! Stub for semantic.sty (proof tree / inference rules + ligatures).
2//!
3//! semantic.sty L?: \TestForConflict{\@ifnext, ...} checks if any of
4//! those CSes are already defined, then errors via \@ifdefinable with
5//! the "redefined" message: `Package Semantic Error: The N command(s)
6//! listed above have been redefined.` Pre-undefine them so semantic's
7//! TestForConflict check passes silently. Witness 2403.04708.
8//! NOTE: upstream #2833 removed the kernel `Let('\@ifnext','\@ifnextchar')`
9//! alias, so `\@ifnext` is no longer kernel-bound — its pre-undefine below
10//! is now a defensive no-op (kept for robustness alongside the others).
11use latexml_package::prelude::*;
12
13LoadDefinitions!({
14  // Pre-undefine our LaTeX-kernel \@ifnext etc so semantic.sty's
15  // TestForConflict check in L?:
16  //   \TestForConflict{\@dropnext,\@ifnext,\@ifn,\@ifNextMacro,\@ifnMacro}
17  // passes silently. After raw-load, semantic.sty will define its own
18  // \@ifnext (which is the same shape as \@ifnextchar anyway).
19  assign_meaning(&T_CS!("\\@ifnext"), Stored::None, Some(Scope::Global));
20  assign_meaning(&T_CS!("\\@dropnext"), Stored::None, Some(Scope::Global));
21  assign_meaning(&T_CS!("\\@ifn"), Stored::None, Some(Scope::Global));
22  assign_meaning(&T_CS!("\\@ifNextMacro"), Stored::None, Some(Scope::Global));
23  assign_meaning(&T_CS!("\\@ifnMacro"), Stored::None, Some(Scope::Global));
24
25  // Now let semantic.sty load raw.
26  InputDefinitions!("semantic", extension => Some(Cow::Borrowed("sty")), noltxml => true);
27});
