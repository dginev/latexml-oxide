1//! Stub for semantic.sty (proof tree / inference rules + ligatures).
2//!
3//! semantic.sty L?: \TestForConflict{\@ifnext, ...} checks if our LaTeX
4//! kernel-defined `\@ifnext` is already defined, then errors via
5//! \@ifdefinable with the "redefined" message. Our LaTeXML kernel
6//! pre-binds \@ifnext (latex_constructs.rs: Let!("\\@ifnext",
7//! "\\@ifnextchar")), so semantic.sty bails with
8//! `Package Semantic Error: The 1 command(s) listed above have been
9//! redefined.` Pre-undefine those CSes so semantic's TestForConflict
10//! check passes silently. Witness 2403.04708.
11use latexml_package::prelude::*;
12
13LoadDefinitions!({
14  // Pre-undefine our LaTeX-kernel \@ifnext etc so semantic.sty's
15  // TestForConflict check in L?:
16  //   \TestForConflict{\@dropnext,\@ifnext,\@ifn,\@ifNextMacro,\@ifnMacro}
17  // passes silently. After raw-load, semantic.sty will define its own
18  // \@ifnext (which is the same shape as \@ifnextchar anyway).
19  state::assign_meaning(&T_CS!("\\@ifnext"), latexml_core::common::store::Stored::None, Some(Scope::Global));
20  state::assign_meaning(&T_CS!("\\@dropnext"), latexml_core::common::store::Stored::None, Some(Scope::Global));
21  state::assign_meaning(&T_CS!("\\@ifn"), latexml_core::common::store::Stored::None, Some(Scope::Global));
22  state::assign_meaning(&T_CS!("\\@ifNextMacro"), latexml_core::common::store::Stored::None, Some(Scope::Global));
23  state::assign_meaning(&T_CS!("\\@ifnMacro"), latexml_core::common::store::Stored::None, Some(Scope::Global));
24
25  // Now let semantic.sty load raw.
26  InputDefinitions!("semantic", extension => Some(Cow::Borrowed("sty")), noltxml => true);
27});
