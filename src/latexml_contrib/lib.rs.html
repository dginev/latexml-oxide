252/// Runtime lookup: route `filename` (e.g. `"MnSymbol.sty"`,
253/// `"hobby.code.tex"`) through its compiled `load_definitions` fn, or return
254/// `None` when the filename has no registered binding. Splits on the *first*
255/// `.` so `("hobby", "code.tex", …)` matches correctly — mirrors
256/// `latexml_package::dispatch`.
257pub fn dispatch(filename: &str) -> Option<Result<()>> {
258  let (base, ext) = filename.split_once('.')?;
259  // Perl pathname_find L383-389: strict-case first, case-insensitive fallback
260  // (matches `\documentclass{jhep}` against `JHEP.cls.ltxml`-style entries).
261  // See latexml_package::dispatch for the parallel comment.
262  BINDINGS
263    .iter()
264    .find(|(name, extension, _)| *name == base && *extension == ext)
265    .or_else(|| {
266      BINDINGS.iter().find(|(name, extension, _)| {
267        name.eq_ignore_ascii_case(base) && extension.eq_ignore_ascii_case(ext)
268      })
269    })
270    .map(|(_, _, loader)| loader())
271}

273/// All registered (name, extension) pairs for this crate's BINDINGS.
274/// Mirror of `latexml_package::binding_names`. Consumed by
275/// `find_file(notex=true)` to detect compiled-binding existence across
276/// all registered extensions (cls/sty/def/pool/code.tex/...). The class
277/// names (entries with `ext == "cls"`) flow into `load_class`'s
278/// Perl-parity prefix-match fallback via the
279/// `state::get_class_binding_names()` filtered view.
280pub fn binding_names() -> &'static [(&'static str, &'static str)] {
281  use std::sync::OnceLock;
282  static NAMES: OnceLock<Vec<(&'static str, &'static str)>> = OnceLock::new();
283  NAMES
284    .get_or_init(|| {
285      BINDINGS
286        .iter()
287        .map(|(name, ext, _)| (*name, *ext))
288        .collect()
289    })
290    .as_slice()
291}
