452/// Runtime lookup: route `filename` (e.g. `"MnSymbol.sty"`,
453/// `"hobby.code.tex"`) through its compiled `load_definitions` fn, or return
454/// `None` when the filename has no registered binding. Splits on the *first*
455/// `.` so `("hobby", "code.tex", …)` matches correctly — mirrors
456/// `latexml_package::dispatch`.
457pub fn dispatch(filename: &str) -> Option<Result<()>> {
458  let (base, ext) = filename.split_once('.')?;
459  // Strip directory prefix: `\documentclass{Definitions/mdpi}` →
460  // basename `mdpi`. Perl Package.pm L2167-2170 (FindFile_fallback)
461  // does the same. Without this, paper-bundled classes like
462  // `Definitions/mdpi.cls` miss the registered binding and cascade.
463  // See latexml_package::dispatch.
464  let base_only = base.rsplit_once(['/', '\\']).map_or(base, |(_, b)| b);
465  // Perl pathname_find L383-389: strict-case first, case-insensitive fallback
466  // (matches `\documentclass{jhep}` against `JHEP.cls.ltxml`-style entries).
467  // See latexml_package::dispatch for the parallel comment.
468  BINDINGS
469    .iter()
470    .find(|(name, extension, _)| *name == base && *extension == ext)
471    .or_else(|| {
472      BINDINGS
473        .iter()
474        .find(|(name, extension, _)| *name == base_only && *extension == ext)
475    })
476    .or_else(|| {
477      BINDINGS.iter().find(|(name, extension, _)| {
478        name.eq_ignore_ascii_case(base) && extension.eq_ignore_ascii_case(ext)
479      })
480    })
481    .or_else(|| {
482      BINDINGS.iter().find(|(name, extension, _)| {
483        name.eq_ignore_ascii_case(base_only) && extension.eq_ignore_ascii_case(ext)
484      })
485    })
486    .map(|(_, _, loader)| loader())
487}

489/// All registered (name, extension) pairs for this crate's BINDINGS.
490/// Mirror of `latexml_package::binding_names`. Consumed by
491/// `find_file(notex=true)` to detect compiled-binding existence across
492/// all registered extensions (cls/sty/def/pool/code.tex/...). The class
493/// names (entries with `ext == "cls"`) flow into `load_class`'s
494/// Perl-parity prefix-match fallback via the
495/// `state::get_class_binding_names()` filtered view.
496pub fn binding_names() -> &'static [(&'static str, &'static str)] {
497  use std::sync::OnceLock;
498  static NAMES: OnceLock<Vec<(&'static str, &'static str)>> = OnceLock::new();
499  NAMES
500    .get_or_init(|| {
501      BINDINGS
502        .iter()
503        .map(|(name, ext, _)| (*name, *ext))
504        .collect()
505    })
506    .as_slice()
507}
