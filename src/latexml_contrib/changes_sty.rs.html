1use latexml_package::prelude::*;

3LoadDefinitions!({
4  // Preserve change-marked content (author body) as ltx:text with a
5  // semantic class. The pre-content-preserving stub gobbled `\deleted`
6  // arg #2 entirely; even when authors mark text for deletion in a
7  // review-tracking context, that text is article material and the
8  // semantic information ("this was marked deleted by author X") is
9  // typeset-relevant. Match `\added` / `\replaced` / `\highlight` /
10  // `\comment` semantics with named ltx:text classes so HTML/JATS
11  // post-processors get the same role information real changes.sty
12  // emits in pdf output.
13  RequirePackage!("xcolor");
14  RequirePackage!("ulem");
15  RequirePackage!("todonotes");
16  RequirePackage!("xstring");
17  // \added[author]{text} — pass-through (#2). No container element in
18  // the LaTeXML schema accepts ltx:section, so wrapping in <ltx:text>,
19  // <ltx:inline-block>, or <ltx:note> all break on appendix-wide
20  // `\added{\section{...}...}` patterns when the wrapper auto-closes
21  // and the }-token tries to close it. Pass-through preserves content;
22  // semantic class is lost in HTML output. Witness 2404.13783,
23  // 2110.12098 (aastex63 with `\added{\section{Model Limitations}...}`).
24  DefMacro!("\\added[]{}", "#2");
25  // `\deleted[author]{text}` → gobble to nothing, matching Perl's
26  // changes.sty.ltxml L25 `DefMacro('\deleted[]{}',Tokens())`. The
27  // `changes` package's `final` option (and Perl's always-final stub)
28  // ACCEPTS deletions — the text is removed from the typeset output. The
29  // `{}` parameter reads the text as an unexpanded balanced group, so any
30  // command inside (even an undefined one) is discarded as raw tokens
31  // rather than executed. Keeping the text (`#2`, the old behavior)
32  // diverged from Perl, rendered author-deleted text as if present, AND
33  // expanded fragile inner commands: witness 1901.02252 (`\usepackage
34  // [final]{changes}`, `\deleted{… \citep{…} …}` with paclic32's ACL-style
35  // citations that lack natbib's `\citep`) → `undefined:\citep` + a
36  // `malformed:ltx:para` cascade where Perl is clean. `\added`/`\replaced`
37  // stay content-preserving (`#2`), exactly as Perl does. RUST 2 → 0.
38  DefMacro!("\\deleted[]{}", "");
39  DefMacro!("\\replaced[]{}{}", "#2");
40  DefMacro!("\\highlight[]{}", "#2");
41  DefMacro!("\\comment[]{}", "#2");
42  def_macro_noop("\\ChangesListline{}{}{}{}")?;
43  DefMacro!("\\listofchangesname", "List of changes");
44  DefMacro!("\\summaryofchangesname", "Changes");
45  DefMacro!("\\compactsummaryofchangesname", "Changes (compact)");
46  DefMacro!("\\changesaddedname", "Added");
47  DefMacro!("\\changesdeletedname", "Deleted");
48  DefMacro!("\\changesreplacedname", "Replaced");
49  DefMacro!("\\changeshighlightname", "Highlighted");
50  DefMacro!("\\changescommentname", "Commented");
51  DefMacro!("\\changesauthorname", "Author");
52  DefMacro!("\\changesanonymousname", "anonymous");
53  DefMacro!("\\changesnochanges", "No changes.");
54  DefMacro!(
55    "\\changesnoloc",
56    "List of changes is available after the next \\LaTeX\\ run."
57  );
58  DefMacro!(
59    "\\changesnosoc",
60    "Summary of changes is available after the next \\LaTeX\\ run."
61  );
62  Let!("\\cleaders", "\\leaders");
63  def_macro_noop("\\definechangesauthor[]{}")?;
64  def_macro_noop("\\listofchanges[]")?;
65  def_macro_noop("\\origcontentsline")?;
