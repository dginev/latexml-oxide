1//! Stub for iccv.sty / iccvw.sty (ICCV conference style).
2//!
3//! Same \thetitle pattern as cvpr.sty.
4use latexml_package::prelude::*;
5
6LoadDefinitions!({
7  def_macro_noop("\\thetitle")?;
8  def_macro_noop("\\maketitlesupplementary")?;
9  DefConditional!("\\ificcvfinal");
10  DefConditional!("\\ificcvrebuttal");
11  DefConditional!("\\ificcvpagenumbers");
12  // \iccvfinalcopy / \iccvPaperID — page-numbering toggles in ICCV
13  // templates. Affect print layout only; HTML rendering is invariant.
14  // Witness 2 stage-2 papers.
15  def_macro_noop("\\iccvfinalcopy")?;
16  def_macro_noop("\\iccvPaperID{}")?;
17
18  // CV-conference convention abbreviations (iccv.sty L254+). Same
19  // rationale as in contrib/cvpr_sty.rs — when the path-stripping
20  // fallback (`iccv2023AuthorKit/iccv.sty` → our binding) takes over,
21  // the InputDefinitions raw-load below fails to find plain `iccv.sty`
22  // on disk (bundled under a subdir), so \etal/\ie/\eg/\etc/\vs/\wrt/\dof
23  // stay undefined. Stub them directly. Witness 2305.20091 (iccv2023
24  // AuthorKit paper): 3 errors (\ie/\eg/\etal) → 0.
25  DefMacro!("\\onedot", "\\@onedot");
26  DefMacro!("\\@onedot", ".\\@");
27  DefMacro!("\\etal", "\\emph{et al}\\onedot");
28  DefMacro!("\\ie", "\\emph{i.e}\\onedot");
29  DefMacro!("\\eg", "\\emph{e.g}\\onedot");
30  DefMacro!("\\cf", "\\emph{c.f}\\onedot");
31  DefMacro!("\\etc", "\\emph{etc}\\onedot");
32  DefMacro!("\\vs", "\\emph{vs}\\onedot");
33  DefMacro!("\\wrt", "\\emph{w.r.t}\\onedot");
34  DefMacro!("\\dof", "d.o.f\\onedot");
35
36  InputDefinitions!("iccv", noltxml => true, extension => Some(Cow::Borrowed("sty")));
37});
