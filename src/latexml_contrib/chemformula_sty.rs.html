1//! Stub for chemformula.sty (chemical formulas).
2//!
3//! Maps \ch{...} to mhchem's \ce since both render similarly for our
4//! HTML/XML output where the chemistry notation isn't fully styled.
5use latexml_package::prelude::*;
6
7
8LoadDefinitions!({
9  RequirePackage!("mhchem");
10  // chemformula 4.x is an expl3 LaTeX3 package; the INCLUDE_STYLES
11  // post-binding raw load calls \ProcessKeysPackageOptions at
12  // chemformula.sty L481. Without l3keys2e + xparse loaded first,
13  // the raw load errors with \ProcessKeysPackageOptions undefined.
14  // Driver cluster: stage11_v3 2504.13749 (chemformula raw-load).
15  RequirePackage!("l3keys2e");
16  RequirePackage!("xparse");
17  // Mirror chemformula.sty L29 `\RequirePackage{tikz,amsmath,xfrac,nicefrac}`
18  // faithfully — Perl has no chemformula binding, so it raw-loads the real
19  // chemformula.sty and pulls in ALL of these transitively. `\sfrac` (xfrac)
20  // becomes available to the document (witness 2006.07679: `\sfrac{\theta}{2}`
21  // in plain math, 1 error → 0). `tikz` was previously omitted "to keep the
22  // stub light" (the stub renders `\ch` via mhchem `\ce`, not chemformula's
23  // tikz-drawn arrows), but that omission is a DIVERGENCE: the real chemformula
24  // requires tikz, and tikz → pgf → pgfsys-latexml loads `xcolor`. A paper that
25  // does `\PassOptionsToPackage{table}{xcolor}` then relies on chemformula to
26  // pull in xcolor (with the table option → `\rowcolors`/colortbl) saw
27  // `\rowcolors` undefined where Perl had it. Witness 1809.04023 (revtex4-1 +
28  // `\PassOptionsToPackage{table}{xcolor}` + chemformula + `\rowcolors`):
29  // 1 error → 0. So require tikz too, matching chemformula.sty L29 exactly.
30  RequirePackage!("tikz");
31  RequirePackage!("xfrac");
32  RequirePackage!("nicefrac");
33  Let!("\\ch", "\\ce");
34  Let!("\\chcpd", "\\ce");
35  def_macro_noop("\\chsetup{}")?;
36  def_macro_noop("\\setchemformula{}")?;
37});
