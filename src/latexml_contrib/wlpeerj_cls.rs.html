1//! Stub for wlpeerj.cls (Wiley PeerJ template).
2use latexml_package::prelude::*;
3
4LoadDefinitions!({
5  LoadClass!("OmniBus");
6  // Real wlpeerj.cls: `\RequirePackage{amsmath,amsfonts,amssymb}`.
7  // We omitted amsfonts + amssymb before, leaving common math glyphs
8  // like \gtrsim, \lesssim, \mathbb undefined for PeerJ-using papers.
9  // Witness 2305.10817 (Trieste causality paper).
10  RequirePackage!("amsmath");
11  RequirePackage!("amsfonts");
12  RequirePackage!("amssymb");
13  RequirePackage!("amsthm");
14  // Eager xcolor preload removed for Perl parity: it makes a later document
15  // xcolor[table] load a no-op, so colortbl/array never load and array m{}/b{}
16  // columns break (Unrecognized tabular template -> Extra alignment tab). The
17  // document loads xcolor itself; color/definecolor stay via hyperref->color.
18  // See ifacconf_cls.rs and SYNC_STATUS (eager-xcolor cluster).
19  RequirePackage!("hyperref");
20  // wlpeerj.cls L23: `\RequirePackage{lineno}` unconditional.
21  RequirePackage!("lineno");
22  // wlpeerj.cls L55: `\RequirePackage{fancyhdr}` (custom headers/footers).
23  // PeerJ-template papers set `\pagestyle{fancy}` and use the legacy fancyhdr
24  // interface (`\lhead`/`\chead`/`\rhead`/`\cfoot`/`\fancyplain`) directly in
25  // the preamble. Our binding intercepts wlpeerj (OmniBus) and mirrored most of
26  // its RequirePackage list but omitted fancyhdr, so those macros were
27  // undefined where Perl — which raw-dep-scans the .cls and loads fancyhdr — is
28  // clean. Witness 1507.06496.
29  RequirePackage!("fancyhdr");
30
31  // Many PeerJ-template papers (witness 2305.10817) use
32  // `\printbibliography[…]` without loading biblatex, expecting the
33  // class to wire it up. Provide a `\bibliography`-style fallback so
34  // we don't fire `Error:undefined:\printbibliography`. If the user
35  // actually loads biblatex, our biblatex_sty binding redefines this
36  // with the real semantics.
37  DefMacro!("\\printbibliography[]", None);
38
39  // Frontmatter — preserve author content.
40  // \corrauthor[mark]{name}{email} — emit name as author + email note.
41  DefMacro!(
42    "\\corrauthor[]{}{}",
43    "\\author{#2}\\@add@frontmatter{ltx:note}[role=email]{#3}"
44  );
45  // \authoraffiliation[mark]{name}{affil} — emit name + affil note.
46  DefMacro!(
47    "\\authoraffiliation[]{}{}",
48    "\\author{#2}\\@add@frontmatter{ltx:note}[role=affiliation]{#3}"
49  );
50  DefMacro!(
51    "\\affil[]{}",
52    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}"
53  );
54});
