1//! Stub for wlpeerj.cls (Wiley PeerJ template).
2use latexml_package::prelude::*;
3
4LoadDefinitions!({
5  LoadClass!("OmniBus");
6  // Real wlpeerj.cls: `\RequirePackage{amsmath,amsfonts,amssymb}`.
7  // We omitted amsfonts + amssymb before, leaving common math glyphs
8  // like \gtrsim, \lesssim, \mathbb undefined for PeerJ-using papers.
9  // Witness 2305.10817 (Trieste causality paper).
10  RequirePackage!("amsmath");
11  RequirePackage!("amsfonts");
12  RequirePackage!("amssymb");
13  RequirePackage!("amsthm");
14  RequirePackage!("xcolor");
15  RequirePackage!("hyperref");
16  // wlpeerj.cls L23: `\RequirePackage{lineno}` unconditional.
17  RequirePackage!("lineno");
18
19  // Many PeerJ-template papers (witness 2305.10817) use
20  // `\printbibliography[…]` without loading biblatex, expecting the
21  // class to wire it up. Provide a `\bibliography`-style fallback so
22  // we don't fire `Error:undefined:\printbibliography`. If the user
23  // actually loads biblatex, our biblatex_sty binding redefines this
24  // with the real semantics.
25  DefMacro!("\\printbibliography[]", None);
26
27  // Frontmatter — preserve author content.
28  // \corrauthor[mark]{name}{email} — emit name as author + email note.
29  DefMacro!("\\corrauthor[]{}{}",
30    "\\author{#2}\\@add@frontmatter{ltx:note}[role=email]{#3}");
31  // \authoraffiliation[mark]{name}{affil} — emit name + affil note.
32  DefMacro!("\\authoraffiliation[]{}{}",
33    "\\author{#2}\\@add@frontmatter{ltx:note}[role=affiliation]{#3}");
34  DefMacro!("\\affil[]{}",
35    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}");
36});
