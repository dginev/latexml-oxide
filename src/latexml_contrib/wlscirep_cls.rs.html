2use latexml_package::prelude::*;

4LoadDefinitions!({
5  LoadClass!("OmniBus");
6  RequirePackage!("amsmath");
7  RequirePackage!("amssymb");
8  RequirePackage!("amsthm");
9  RequirePackage!("xcolor");
10  RequirePackage!("hyperref");
11  // wlscirep.cls L53: `\RequirePackage[superscript,biblabel,nomove]{cite}`.
12  // Our cite binding Lets `\citeonline` -> `\cite`, so authors writing
13  // `\citeonline{key}` (witness 2306.04599) need cite loaded.
14  RequirePackage!("cite");
15  // wlscirep.cls L52 has a commented-out `\RequirePackage[numbers]{natbib}`
16  // — Perl's dep-scanner picks it up anyway (it scans .cls source-text
17  // not the LaTeX execution path). Authors writing
18  // `\bibliographystyle{agsm}` get a .bbl with `\harvarditem` /
19  // `\harvardand` / `\harvardyearleft`, which natbib provides as
20  // backwards-compat aliases for cite. Witness 2308.08350.
21  RequirePackage!("natbib");
22  // wlscirep.cls L29: \RequirePackage{booktabs} unconditionally.
23  // Witness 2408.07161 (\toprule/\midrule/\bottomrule used without
24  // explicit \usepackage{booktabs}).
25  RequirePackage!("booktabs");
26  // wlscirep also configures caption layout — pull caption.sty so
27  // \captionsetup is available. Witness 2411.06447, 2411.10607.
28  RequirePackage!("caption");
29  // wlscirep.cls L60: \RequirePackage{fancyhdr} for custom headers/footers.
30  // Without it, \fancyhf / \fancyfoot used in the cls body (and in user
31  // documents that override the page style) stay undefined. Witness
32  // 2310.16477.
33  RequirePackage!("fancyhdr");
34
35  // wlscirep frontmatter / bibliography helpers — preserve author content.
36  DefMacro!("\\JournalTitle{}", "\\emph{#1}");
37  DefMacro!("\\affiliation{}",
38    "\\@add@frontmatter{ltx:note}[role=affiliation]{#1}");
39  DefMacro!("\\corres{}",
40    "\\@add@frontmatter{ltx:note}[role=corresponding]{#1}");
41  DefMacro!("\\presentadd[]{}",
42    "\\@add@frontmatter{ltx:note}[role=present-address]{#2}");
43});
