1//! mhchem.sty — chemical formula typesetting.
2//!
3//! TODO(strict-perl-parity): once `latexml_engine` can faithfully
4//! handle the expl3 / xparse / chemgreek raw-load chain (currently
5//! the gaps are around `\group_begin:` non-boxing-frame handling
6//! and l3regex/l3tl-analysis register access during the chemgreek
7//! load triggered by the first `\ce{...}`), DELETE this binding so
8//! that `\usepackage{mhchem}` raw-loads the actual TL `mhchem.sty`,
9//! matching Perl LaTeXML's behavior (Perl has no `mhchem.sty.ltxml`).
10//! Driver paper: arXiv:1806.06448 (3 errors → 0 errors with this
11//! stub; full chemistry rendering needs the engine fix).
12//!
13//! Perl LaTeXML has no `mhchem.sty.ltxml` and raw-loads the actual
14//! TL `mhchem.sty` (which `\RequirePackage{chemgreek}` →
15//! `\RequirePackage{xparse}` → heavy expl3 machinery). Perl's expl3
16//! emulation is mature enough that this works.
17//!
18//! Rust's expl3 emulation has gaps (e.g. `\group_begin:` non-boxing
19//! frame handling, `\l__tl_analysis_*_int` register access in
20//! l3regex/l3tl-analysis), so the chemgreek raw-load triggered by the
21//! first `\ce{...}` invocation leaves the gullet in an unbalanced state
22//! (open `\iffalse`, unmatched `{` at end-of-input).
23//!
24//! Until the expl3 cluster is fixed, this binding intercepts the
25//! mhchem load and provides a minimal stub: `\ce{...}` typesets its
26//! argument as roman text, no chemistry layout. This is a documented
27//! divergence from Perl LaTeXML — the full chemistry rendering needs
28//! a real port. Driver paper: 1806.06448 (3 errors → 0 errors).
29//!
30//! Stubs cover the public mhchem v3/v4 surface most papers actually
31//! use: `\ce`, `\cee`, `\cf`, plus `\mhchemoptions` (no-op).
32
33use latexml_package::prelude::*;
34
35#[rustfmt::skip]
36LoadDefinitions!({
37  // Accept both v3 and v4: the package option is `version=N` — handled
38  // at \usepackage time but irrelevant to our stub.
39  DefMacro!("\\mhchemoptions RequiredKeyVals", "");
40
41  // \ce{<formula>} — chemistry mode. Real mhchem renders subscripts,
42  // charges, arrows, etc. Papers invoke \ce{H_2O} / \ce{N_2} both in
43  // math context (equation*) AND in text context (paragraphs).
44  // \ensuremath wraps body in math mode if not already in math, so
45  // `_`/`^` parse as scripts in both contexts. Loses roman-text
46  // rendering of plain text chemistry, but avoids cascading errors.
47  //
48  // Strip embedded `$` toggles from the body before re-entering math:
49  // mhchem v3 papers commonly write `\ce{Cs$_x$MA$_{1-x}$PbI3}` where
50  // the `$` pairs are mhchem's own subscript-grouping hint, NOT real
51  // math toggles. Without stripping, `\ensuremath{...$_x$...}` re-toggles
52  // out of math at the first `$`, leaving `_x` in text mode — which
53  // errors with "Script _ can only appear in math mode".
54  // Witnesses: 1908.05236 (\ce{MAPb(I_{1-x}Br_x)3}), 0907.1390 (\ce{N_2}).
55  fn strip_math_toggles(arg: &Tokens) -> Tokens {
56    let stripped: Vec<Token> = arg.unlist_ref().iter().copied()
57      .filter(|t| t.get_catcode() != Catcode::MATH)
58      .collect();
59    Tokens::new(stripped)
60  }
61  DefMacro!("\\ce{}", sub[(body)] {
62    let stripped = strip_math_toggles(&body);
63    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
64    result.extend(stripped.unlist());
65    result.push(T_END!());
66    Ok(Tokens::new(result))
67  });
68  DefMacro!("\\cee{}", sub[(body)] {
69    let stripped = strip_math_toggles(&body);
70    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
71    result.extend(stripped.unlist());
72    result.push(T_END!());
73    Ok(Tokens::new(result))
74  });
75  DefMacro!("\\cf{}", sub[(body)] {
76    let stripped = strip_math_toggles(&body);
77    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
78    result.extend(stripped.unlist());
79    result.push(T_END!());
80    Ok(Tokens::new(result))
81  });
82
83  // \arrow / \chemarrow — used inside \ce arguments. Stub as small text
84  // arrow so a `\ce{A \arrow B}` doesn't error if it leaks out.
85  DefMacro!("\\chemarrow", "\\rightarrow");
86});
