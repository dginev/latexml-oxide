1//! mhchem.sty — chemical formula typesetting.
2//!
3//! TODO(strict-perl-parity): once `latexml_engine` can faithfully
4//! handle the expl3 / xparse / chemgreek raw-load chain (currently
5//! the gaps are around `\group_begin:` non-boxing-frame handling
6//! and l3regex/l3tl-analysis register access during the chemgreek
7//! load triggered by the first `\ce{...}`), DELETE this binding so
8//! that `\usepackage{mhchem}` raw-loads the actual TL `mhchem.sty`,
9//! matching Perl LaTeXML's behavior (Perl has no `mhchem.sty.ltxml`).
10//! Driver paper: arXiv:1806.06448 (3 errors → 0 errors with this
11//! stub; full chemistry rendering needs the engine fix).
12//!
13//! Perl LaTeXML has no `mhchem.sty.ltxml` and raw-loads the actual
14//! TL `mhchem.sty` (which `\RequirePackage{chemgreek}` →
15//! `\RequirePackage{xparse}` → heavy expl3 machinery). Perl's expl3
16//! emulation is mature enough that this works.
17//!
18//! Rust's expl3 emulation has gaps (e.g. `\group_begin:` non-boxing
19//! frame handling, `\l__tl_analysis_*_int` register access in
20//! l3regex/l3tl-analysis), so the chemgreek raw-load triggered by the
21//! first `\ce{...}` invocation leaves the gullet in an unbalanced state
22//! (open `\iffalse`, unmatched `{` at end-of-input).
23//!
24//! Until the expl3 cluster is fixed, this binding intercepts the
25//! mhchem load and provides a minimal stub: `\ce{...}` typesets its
26//! argument as roman text, no chemistry layout. This is a documented
27//! divergence from Perl LaTeXML — the full chemistry rendering needs
28//! a real port. Driver paper: 1806.06448 (3 errors → 0 errors).
29//!
30//! Stubs cover the public mhchem v3/v4 surface most papers actually
31//! use: `\ce`, `\cee`, `\cf`, plus `\mhchemoptions` (no-op).
32
33use latexml_package::prelude::*;
34
35#[rustfmt::skip]
36LoadDefinitions!({
37  // [mhchem retirement probe, 2026-05-19] When env var
38  // LATEXML_MHCHEM_NOLTXML is set, bypass this stub and force a
39  // raw load of the actual TL mhchem.sty (mirroring Perl
40  // LaTeXML's behaviour — Perl has no mhchem.sty.ltxml). Lets us
41  // measure the engine gap (expected:<relationaltoken>,
42  // unexpected:\fi, etc. — see SYNC_STATUS Cluster E / Task #22).
43  // No-op when unset — production users keep the stub.
44  if std::env::var("LATEXML_MHCHEM_NOLTXML").is_ok() {
45    InputDefinitions!("mhchem", noltxml => true, extension => Some(Cow::Borrowed("sty")));
46    return Ok(());
47  }
48  // Perl LaTeXML auto-scans mhchem.sty for `\RequirePackage` calls
49  // and brings in ifthen, calc, twoopt, amsmath, keyval, graphics, pgf,
50  // tikz as transitive deps. Since this Rust stub intercepts the load
51  // (so the raw RequirePackage chain never fires), papers that rely on
52  // those deps via mhchem alone hit undefined-CS errors. Pull in the
53  // ones most commonly needed: amsmath (for \boldsymbol, \eqref,
54  // \text, align*, etc.) and graphicx (for figure handling). Witness:
55  // 1311.6762 (stage 15 RUST-REGRESSION) — paper loads mhchem but
56  // not amsmath, then uses `\boldsymbol` / `\eqref`. Perl's auto-dep
57  // scan loads amsmath → 0 errors; Rust stub didn't → 2 errors.
58  RequirePackage!("amsmath");
59  RequirePackage!("graphicx");
60
61  // Accept both v3 and v4: the package option is `version=N` — handled
62  // at \usepackage time but irrelevant to our stub.
63  def_macro_noop("\\mhchemoptions RequiredKeyVals")?;
64
65  // \ce{<formula>} — chemistry mode. Real mhchem renders subscripts,
66  // charges, arrows, etc. Papers invoke \ce{H_2O} / \ce{N_2} both in
67  // math context (equation*) AND in text context (paragraphs).
68  // \ensuremath wraps body in math mode if not already in math, so
69  // `_`/`^` parse as scripts in both contexts. Loses roman-text
70  // rendering of plain text chemistry, but avoids cascading errors.
71  //
72  // Strip embedded `$` toggles from the body before re-entering math:
73  // mhchem v3 papers commonly write `\ce{Cs$_x$MA$_{1-x}$PbI3}` where
74  // the `$` pairs are mhchem's own subscript-grouping hint, NOT real
75  // math toggles. Without stripping, `\ensuremath{...$_x$...}` re-toggles
76  // out of math at the first `$`, leaving `_x` in text mode — which
77  // errors with "Script _ can only appear in math mode".
78  // Witnesses: 1908.05236 (\ce{MAPb(I_{1-x}Br_x)3}), 0907.1390 (\ce{N_2}).
79  //
80  // Also convert `#` (PARAM-catcode) tokens to `\equiv` CS: mhchem v3
81  // uses `#` for triple bond (e.g. `\ce{-C#C-}` renders as `-C≡C-`).
82  // Without conversion, the bare `#` reaches the Stomach as a PARAM
83  // token and triggers "should never reach Stomach!". Witness:
84  // arXiv:2508.11040 (`\ce{-C#C-}`).
85  fn strip_math_toggles(arg: &Tokens) -> Tokens {
86    let mut out: Vec<Token> = Vec::with_capacity(arg.unlist_ref().len());
87    for tok in arg.unlist_ref().iter().copied() {
88      match tok.get_catcode() {
89        Catcode::MATH => continue,
90        Catcode::PARAM => out.push(T_CS!("\\equiv")),
91        _ => out.push(tok),
92      }
93    }
94    Tokens::new(out)
95  }
96  DefMacro!("\\ce{}", sub[(body)] {
97    let stripped = strip_math_toggles(&body);
98    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
99    result.extend(stripped.unlist());
100    result.push(T_END!());
101    Ok(Tokens::new(result))
102  });
103  DefMacro!("\\cee{}", sub[(body)] {
104    let stripped = strip_math_toggles(&body);
105    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
106    result.extend(stripped.unlist());
107    result.push(T_END!());
108    Ok(Tokens::new(result))
109  });
110  DefMacro!("\\cf{}", sub[(body)] {
111    let stripped = strip_math_toggles(&body);
112    let mut result = vec![T_CS!("\\ensuremath"), T_BEGIN!()];
113    result.extend(stripped.unlist());
114    result.push(T_END!());
115    Ok(Tokens::new(result))
116  });
117
118  // \arrow / \chemarrow — used inside \ce arguments. Stub as small text
119  // arrow so a `\ce{A \arrow B}` doesn't error if it leaks out.
120  DefMacro!("\\chemarrow", "\\rightarrow");
121});
