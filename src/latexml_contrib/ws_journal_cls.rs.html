1//! Stub for World Scientific journal classes (ws-ijgmmp, ws-ijmpd, etc.).
2//!
3//! These classes share frontmatter macros like `\catchline{vol}{no}{year}{pgs}{pgsE}`,
4//! `\title{...}`, `\Author{name}`. Raw load of the cls may fail before
5//! defining these (cls uses complex font/page-layout machinery that
6//! doesn't transfer to HTML conversion). Stubs preserve substantive
7//! content as frontmatter notes.
8//!
9//! Witness: 2306.12455 (ws-ijgmmp.cls), 2306.15982 (ws-ijmpd.cls).
10use latexml_package::prelude::*;
11
12
13LoadDefinitions!({
14  LoadClass!("OmniBus");
15  RequirePackage!("amsmath");
16  RequirePackage!("amsthm");
17  RequirePackage!("amssymb");
18  RequirePackage!("graphicx");
19  RequirePackage!("xcolor");
20  RequirePackage!("hyperref");
21
22  // \catchline{vol}{issue}{year}{pageStart}{pageEnd} — bibliographic
23  // running-header metadata. Preserve as frontmatter note.
24  DefMacro!("\\catchline{}{}{}{}{}",
25    "\\@add@frontmatter{ltx:note}[role=catchline]{Vol. #1, No. #2 (#3) #4--#5}");
26
27  // \Journal{...} {...}{...}{...} — citation format helper (4-arg per WS conv).
28  // Render the args inline as a generic citation string.
29  DefMacro!("\\Journal{}{}{}{}", "{#1} {\\bf #2}, #3 (#4)");
30
31  // No-op layout helpers.
32  def_macro_noop("\\copyrightheading{}")?;
33  def_macro_noop("\\paperBack")?;
34  DefMacro!("\\catchlinefont", "\\footnotesize");
35
36  // \ccode{...} — WS classification-codes block (PACS-style codes in
37  // ws-ijmpe and friends). Real def is styled paragraph; preserve as
38  // a classification note. Witness 2307.12748, 2307.16467.
39  DefMacro!("\\ccode{}",
40    "\\@add@frontmatter{ltx:classification}[scheme=PACS]{#1}");
41  // \category{code}{name} — ws classification 2-arg variant.
42  DefMacro!("\\category{}{}",
43    "\\@add@frontmatter{ltx:classification}[scheme=#1]{#2}");
44
45  // {history} env — publication-history wrapper (received/revised/
46  // accepted dates). Preserve as a frontmatter note. Witness
47  // 2307.12748, 2307.16467 + 2 stage-3/4 ws papers.
48  DefEnvironment!("{history}",
49    "<ltx:note role='history'>#body</ltx:note>",
50    mode => "internal_vertical");
51  // \received{date}, \revised{date}, \accepted{date}, \comby{name}
52  // — used inside {history}. Preserve as inline notes.
53  DefMacro!("\\received{}", "Received #1\\par");
54  DefMacro!("\\revised{}", "Revised #1\\par");
55  DefMacro!("\\accepted{}", "Accepted #1\\par");
56  DefMacro!("\\comby{}", "Communicated by #1\\par");
57  // \email / \http / \uurl — render contact info inline.
58  DefMacro!("\\email{}", "\\textit{#1}\\par");
59  DefMacro!("\\http{}", "\\textit{http://#1}\\par");
60  DefMacro!("\\uurl{}", "\\textit{#1}\\par");
61
62  // ws-mpla.cls and other ws classes: \tbl{caption}{body} — table
63  // caption + tabular body wrapper. Real def is essentially
64  // \caption{#1}<body #2 in tabular>. Bridge to \caption{caption}
65  // followed by the body content. Witness 2404.18954 (ws-mpla).
66  DefMacro!("\\tbl{}{}", "\\caption{#1}#2");
67  // \ttbl{label}{caption}{body} — 3-arg variant with explicit label.
68  DefMacro!("\\ttbl{}{}{}", "\\caption{#2}\\label{#1}#3");
69  // \tcaption{caption} — single-arg variant in some ws families.
70  DefMacro!("\\tcaption{}", "\\caption{#1}");
71});
