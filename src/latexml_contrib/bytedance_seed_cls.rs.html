1//! Stub for bytedance_seed.cls (ByteDance Seed paper template).
2//!
3//! User-bundled template with author/affiliation/contribution-list
4//! helpers built on \addtolist. Our raw-load fails on the
5//! \addtolist[5] (5-arg, optional-arg-first) signature; provide
6//! content-preserving stubs so the metadata reaches XML output.
7//!
8//! Witness: 2503.04598 (Seed-1.5 thinking paper).
9use latexml_package::prelude::*;
10
11
12LoadDefinitions!({
13  LoadClass!("OmniBus");
14  RequirePackage!("amsmath");
15  RequirePackage!("amssymb");
16  RequirePackage!("xcolor");
17  RequirePackage!("graphicx");
18  RequirePackage!("hyperref");
19  RequirePackage!("booktabs");
20  RequirePackage!("etoolbox");
21
22  // Author/affiliation/contribution lists — preserve as ltx:note.
23  def_macro_noop("\\authorlist")?;
24  def_macro_noop("\\affiliationlist")?;
25  def_macro_noop("\\contributionlist")?;
26  def_macro_noop("\\checkdatalist")?;
27  // \author[mark]{name} — emit name as author.
28  DefMacro!("\\author[]{}", "\\author{#2}");
29  // \affiliation[mark]{text} — emit affiliation note.
30  DefMacro!("\\affiliation[]{}",
31    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}");
32  // \contribution[mark]{text} — emit contribution note.
33  DefMacro!("\\contribution[]{}",
34    "\\@add@frontmatter{ltx:note}[role=contribution]{#2}");
35  // \checkdata[label]{value} — emit as keyed note (label: value).
36  DefMacro!("\\checkdata[]{}",
37    "\\@add@frontmatter{ltx:note}[role=#1]{#2}");
38  // \correspondence{text} — emit corresponding-author note.
39  DefMacro!("\\correspondence{}",
40    "\\@add@frontmatter{ltx:note}[role=correspondence]{#1}");
41  // \beginappendix — bytedance uses this in place of \appendix.
42  DefMacro!("\\beginappendix", "\\appendix");
43  DefMacro!("\\seedblue{}", "{\\color[HTML]{2E5AA8}\\textbf{#1}}");
44  DefMacro!("\\nm{}", "#1");
45  DefMacro!("\\citeas{}", "\\cite{#1}");
46});
