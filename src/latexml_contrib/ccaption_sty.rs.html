1//! ccaption.sty — extended captioning by Peter Wilson (Herries Press).
2//!
3//! ccaption provides:
4//!  * Non-numbered/`*` captions (`\bottomcaption`, `\namedlegend`)
5//!  * Continued captions (`\contcaption`)
6//!  * Bilingual captions (`\bicaption`)
7//!  * Sidecap-style captions, custom caption fonts/labels
8//!  * `[subfigure]` compatibility: redefines `\caption`,
9//!    `\@subcaption`, `\@float`, `\@dbflt` to track main-vs-sub state.
10//!
11//! Raw-loading ccaption.sty into our engine is fragile: with the
12//! `[subfigure]` option, ccaption redefines `\@subcaption` to use
13//! `\xdef\@subfigcaptionlist{... {\protect\numberline {\@currentlabel}
14//! ...}}`. Our `\@currentlabel` expansion produces tokens like
15//! `\the\c@figure` that the `\xdef` mis-handles, surfacing the
16//! "You can't use `}` after `\the`" cascade at every figure +
17//! `\caption{...}\label{a}` pair (driver: 1105.3285).
18//!
19//! Perl LaTeXML has no ccaption binding either; with default
20//! `INCLUDE_STYLES=false`, raw ccaption.sty is not loaded. Match
21//! Perl: stub the user-facing extension macros as thin wrappers
22//! over `\caption`, leaving the kernel `\caption`/`\label`/
23//! `\@subcaption` untouched.
24use latexml_package::prelude::*;
25
26LoadDefinitions!({
27  Warn!(
28    "missing_file",
29    "ccaption.sty",
30    "ccaption.sty is minimally stubbed — extended caption features (\\bicaption, \\contcaption, \\bottomcaption, sidecap-style) fall back to the base \\caption."
31  );
32  // Continued / split / bilingual captions — treat as regular captions.
33  DefMacro!("\\contcaption{}", "\\caption{#1}");
34  DefMacro!("\\bottomcaption{}", "\\caption{#1}");
35  DefMacro!("\\legend{}", "\\caption{#1}");
36  DefMacro!("\\namedlegend{}", "\\caption{#1}");
37  // Bilingual caption: 4 args (label1, caption1, label2, caption2);
38  // render the first language only (Perl's effective behavior).
39  DefMacro!("\\bicaption{}{}{}{}", "\\caption{#2}");
40  // Sub-captions (when ccaption's `[subfigure]` compat would be
41  // active): forward to the standalone subfigure/subcaption packages
42  // when available; otherwise emit as italic text. Both `\subcaption`
43  // and `\subtop`/`\subbottom` follow this pattern.
44  DefMacro!("\\subcaption{}", "{\\itshape #1}");
45  DefMacro!("\\subtop{}", "{\\itshape #1}");
46  DefMacro!("\\subbottom{}", "{\\itshape #1}");
47  // Caption-formatting / -setup commands ccaption provides — accept
48  // and discard.
49  DefMacro!("\\captionnamefont{}", "");
50  DefMacro!("\\captiontitlefont{}", "");
51  DefMacro!("\\captiondelim{}", "");
52  DefMacro!("\\captionstyle{}", "");
53  DefMacro!("\\hangcaption", "");
54  DefMacro!("\\indentcaption{}", "");
55  DefMacro!("\\normalcaption", "");
56  // `\newfixedcaption{cs}{type}` defines `cs` as a `\caption` for a
57  // specific float type. Define `cs` to forward.
58  DefMacro!("\\newfixedcaption[]{}{}", "\\providecommand#2{\\caption}");
59  DefMacro!("\\renewfixedcaption[]{}{}", "\\renewcommand#2{\\caption}");
60  DefMacro!("\\providefixedcaption[]{}{}", "\\providecommand#2{\\caption}");
61  // Continued-figure subitem counter (ccaption tracks via
62  // `\@captype`); user code may `\thesubfigure` etc. We provide a
63  // benign `\thesubfigure` placeholder if not defined.
64  def_macro_noop("\\contlabel{}")?;
65  def_macro_noop("\\bicaptionbox{}{}")?;
66});
