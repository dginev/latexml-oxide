1//! Stub for pnas-new.cls (PNAS journal class).
2//!
3//! pnas-new.cls is the modern PNAS submission class. It is not in TeXLive;
4//! authors bundle it with their paper. The class defines several
5//! frontmatter helpers (\leadauthor, \correspondingauthor, \authorcontributions,
6//! \significancestatement, \authordeclaration) and a \templatetype{...}
7//! command that loads a per-article-type style (pnasresearcharticle.sty,
8//! pnasperspective.sty etc.). Since our binding registry routes
9//! pnas-new.cls → this binding, we provide gobble + frontmatter-preserve
10//! stubs for the author-facing helpers and a \templatetype that defers to
11//! \RequirePackage like the upstream does.
12//!
13//! Witness: 2305.01604 (significant 7-error gap on canvas, all from
14//! undefined pnas-new frontmatter macros).
15use latexml_package::prelude::*;
16
17LoadDefinitions!({
18  LoadClass!("OmniBus");
19  RequirePackage!("amsmath");
20  RequirePackage!("amssymb");
21  RequirePackage!("amsthm");
22  RequirePackage!("graphicx");
23  // Eager xcolor preload removed for Perl parity: it makes a later document
24  // xcolor[table] load a no-op, so colortbl/array never load and array m{}/b{}
25  // columns break (Unrecognized tabular template -> Extra alignment tab). The
26  // document loads xcolor itself; color/definecolor stay via hyperref->color.
27  // See ifacconf_cls.rs and SYNC_STATUS (eager-xcolor cluster).
28  RequirePackage!("hyperref");
29  RequirePackage!("authblk");
30  RequirePackage!("natbib");
31  // pnas-new.cls L17+ requires extarticle, lmodern, helvet, fontenc,
32  // lettrine, ifpdf, ifxetex, tikz, mdframed, draftwatermark, textcomp,
33  // colortbl, booktabs, algorithm, algpseudocode, changepage, geometry,
34  // caption, etoolbox, fancyhdr, lastpage, titlesec, lineno, footmisc,
35  // enumitem, sidecap, float, stfloats, marginnote.
36  // Most of those are noop-loaded by OmniBus already; bring in the
37  // critical ones explicitly so author content (booktabs rules, etc.) works.
38  RequirePackage!("booktabs");
39  RequirePackage!("fancyhdr");
40  RequirePackage!("etoolbox");
41  RequirePackage!("titlesec");
42  RequirePackage!("caption");
43  RequirePackage!("enumitem");
44  // pnasresearcharticle.sty uses \afterpage and \mdfdefinestyle.
45  RequirePackage!("afterpage");
46  RequirePackage!("mdframed");
47
48  // pnas-new.cls L272: \newcommand{\leadauthor}[1]{\def\@leadauthor{#1}}
49  // Preserve as frontmatter author-list note.
50  DefMacro!(
51    "\\leadauthor{}",
52    "\\@add@frontmatter{ltx:note}[role=lead-author]{#1}"
53  );
54  // pnas-new.cls L275: \newcommand{\authorcontributions}[1]{\def\@authorcontributions{#1}}
55  DefMacro!(
56    "\\authorcontributions{}",
57    "\\@add@frontmatter{ltx:note}[role=author-contributions]{#1}"
58  );
59  // pnas-new.cls L276: \newcommand{\authordeclaration}[1]{\def\@authordeclaration{#1}}
60  DefMacro!(
61    "\\authordeclaration{}",
62    "\\@add@frontmatter{ltx:note}[role=author-declaration]{#1}"
63  );
64  // pnas-new.cls L278: \newcommand{\correspondingauthor}[1]{\def\@correspondingauthor{#1}}
65  // Email content may contain `_` — Semiverbatim neutralizes catcode-8
66  // chars so the note renders cleanly in horizontal mode.
67  DefMacro!(
68    "\\correspondingauthor Semiverbatim",
69    "\\@add@frontmatter{ltx:note}[role=corresponding-author]{#1}"
70  );
71  // pnas-new.cls L279: \newcommand{\significancestatement}[1]{\def\@significancestatement{#1}}
72  DefMacro!(
73    "\\significancestatement{}",
74    "\\@add@frontmatter{ltx:abstract}[role=significance]{#1}"
75  );
76  // pnas-new.cls: \templatetype{pkg} → \RequirePackage{pkg}. Defer to
77  // \RequirePackage so per-article style packages (pnasresearcharticle.sty,
78  // pnasperspective.sty etc.) load correctly.
79  DefMacro!("\\templatetype{}", "\\RequirePackage{#1}");
80  // pnas-new.cls uses \newif\ifshortarticle for layout switching. Provide
81  // the conditional so authors can call \shortarticletrue/\shortarticlefalse.
82  DefConditional!("\\ifshortarticle");
83  // pnas-new.cls L362: \newif\ifsinglecolumn — single/double column layout.
84  DefConditional!("\\ifsinglecolumn");
85  // pnas-new.cls L269: \newcommand{\additionalelement}[1]{\def\@additionalelement{#1}}
86  // Used by pnasresearcharticle.sty (which loads BEFORE pnas-new.cls
87  // completes via \templatetype). Gobble cleanly.
88  def_macro_noop("\\additionalelement{}")?;
89  // pnas-new.cls L299: \newcommand{\abscontent}{...} — abstract-content
90  // renderer that pnasresearcharticle.sty patches via \patchcmd. Provide
91  // a noop so the patch silently fails without error.
92  def_macro_noop("\\abscontent")?;
93  def_macro_noop("\\abscontentformatted")?;
94
95  // Other commonly used pnas-new frontmatter that this stub should preserve:
96  DefMacro!("\\keywords{}", "\\@add@frontmatter{ltx:keywords}{#1}");
97  // \equalauthors / \etalauthors — joint-authorship markers.
98  def_macro_noop("\\equalauthors{}")?;
99  def_macro_noop("\\etalauthors{}")?;
100  // \pnasbreak — content-level command, no rendering effect.
101  def_macro_noop("\\pnasbreak")?;
102});
