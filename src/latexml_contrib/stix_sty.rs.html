1use latexml_package::prelude::*;
2
3LoadDefinitions!({
4  // Perl `ar5iv-bindings/bindings/stix.sty.ltxml` is an empty stub —
5  // it intercepts the FindFile lookup so the heavy raw-load of
6  // TL stix.sty doesn't fire, but doesn't provide any symbols. Papers
7  // that import `\usepackage[notextcomp]{stix}` and use the package's
8  // math symbols (e.g. `\blacktriangle`, `\mdblksquare`) without
9  // separately loading `amssymb` then hit undefined-CS errors.
10  //
11  // Extend the stub with the most-witnessed STIX-only math symbols.
12  // The full TL `stix.sty` defines 1478 symbols via `\stix@MathSymbol`;
13  // we cover just the ones papers in the arxmliv corpus actually use,
14  // mapped to their canonical Unicode codepoints from the STIX font
15  // tables (cf. TL `stix.sty` L1213, L1593-1647).
16  // Witness: arXiv:2509.13186 — `\blacktriangle` + `\mdblksquare`.
17  DefMath!("\\blacktriangle",          "\u{25B2}", role => "ID"); // ▲
18  DefMath!("\\blacktriangledown",      "\u{25BC}", role => "ID"); // ▼
19  DefMath!("\\blacktriangleleft",      "\u{25C0}", role => "RELOP"); // ◀
20  DefMath!("\\blacktriangleright",     "\u{25B6}", role => "RELOP"); // ▶
21  DefMath!("\\bigblacktriangleup",     "\u{25B2}", role => "ID"); // ▲ (large)
22  DefMath!("\\bigblacktriangledown",   "\u{25BC}", role => "ID"); // ▼ (large)
23  DefMath!("\\smallblacktriangleright","\u{25B8}", role => "RELOP"); // ▸
24  DefMath!("\\smallblacktriangleleft", "\u{25C2}", role => "RELOP"); // ◂
25  DefMath!("\\lrblacktriangle",        "\u{25E2}", role => "ID"); // ◢
26  DefMath!("\\llblacktriangle",        "\u{25E3}", role => "ID"); // ◣
27  DefMath!("\\ulblacktriangle",        "\u{25E4}", role => "ID"); // ◤
28  DefMath!("\\urblacktriangle",        "\u{25E5}", role => "ID"); // ◥
29  DefMath!("\\mdblksquare",            "\u{25FC}", role => "ID"); // ◼ medium black square
30});
