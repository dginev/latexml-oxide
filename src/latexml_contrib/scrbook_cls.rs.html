1use latexml_package::prelude::*;

4LoadDefinitions!({
5  Warn!(
6    "missing_file",
7    "scrbook.cls",
8    "scrbook.cls is only minimally stubbed and will not be interpreted raw."
9  );
10  LoadClass!("OmniBus");
11  // Real scrbook.cls loads the KOMA dependency chain (scrkbase, tocbasic, …),
12  // which transitively loads `iftex`, defining `\ifpdf`/`\ifpdftex`/… for
13  // author engine/driver detection. Perl raw-loads the .cls and gets iftex
14  // that way; our OmniBus stub intercepts it. Mirror the dependency (same
15  // rationale as scrartcl_cls.rs, witness 1802.07175).
16  RequirePackage!("iftex");
17  // KOMA section-font hooks (`\sectfont` + empty `\size@<unit>` family) — see
18  // scrartcl_cls.rs for the full rationale. tocloft expands these in
19  // `\cfttoctitlefont` when a KOMA class is detected; as a chapter class
20  // scrbook genuinely uses the `\size@chapter` form (tocloft.sty L169). Without
21  // them, scrbook + tocloft + `\tableofcontents` hits undefined `\sectfont` /
22  // `\size@chapter` where Perl (raw scrbook) is clean.
23  def_macro_noop("\\maybesffamily")?;
24  DefMacro!("\\sectfont", "\\normalcolor\\maybesffamily\\bfseries");
25  def_macro_noop("\\size@part")?;
26  def_macro_noop("\\size@partnumber")?;
27  def_macro_noop("\\size@chapter")?;
28  def_macro_noop("\\size@section")?;
29  def_macro_noop("\\size@subsection")?;
30  def_macro_noop("\\size@subsubsection")?;
31  def_macro_noop("\\size@paragraph")?;
32  def_macro_noop("\\size@subparagraph")?;
33  def_macro_noop("\\setkomafont{}{}")?;
34  def_macro_noop("\\setcapindent{}")?;
35  def_macro_noop("\\deffootnote[]{}{}{}")?;
36  def_macro_noop("\\deffootnotemark{}")?;
37  // KOMA page-style marks — see scrartcl_cls.rs for rationale.
38  def_macro_noop("\\headmark")?;
39  def_macro_noop("\\pagemark")?;
40});
