1//! Stub for IEEE Access journal class (ieeeaccess.cls).
2//!
3//! ieeeaccess.cls is a derivative of IEEEtran. Route to the IEEEtran
4//! binding and stub the IEEE Access-specific frontmatter macros.
5use latexml_package::prelude::*;
6
7LoadDefinitions!({
8  LoadClass!("IEEEtran");
9  RequirePackage!("ifpdf");
10
11  // IEEE Access frontmatter (ieeeaccess.cls L270, L448-453, L839) —
12  // preserve author-supplied content as ltx:note frontmatter.
13  DefMacro!(
14    "\\history{}",
15    "\\@add@frontmatter{ltx:note}[role=history]{#1}"
16  );
17  DefMacro!(
18    "\\corresp{}",

20  );

24  // when called outside expected float context.

26  DefRegister!("\\titlepgskip" => Dimension!("0pt"));
27  // Metadata — preserve author values.
28  DefMacro!("\\doi{}", "\\@add@frontmatter{ltx:note}[role=doi]{#1}");
29  DefMacro!(
