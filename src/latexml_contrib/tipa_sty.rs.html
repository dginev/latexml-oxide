1use latexml_package::prelude::*;

3LoadDefinitions!({
4  // load raw for now.
5  InputDefinitions!("tipa", noltxml => true, extension => Some(Cow::Borrowed("sty")));
6  // tipa.sty internally calls `\RequirePackage[T3,\f@encoding]{fontenc}` to
7  // pull in the T3 encoding definitions (`t3enc.def`), which is what
8  // defines `\textrhookrevepsilon` / `\textbaru` / etc IPA symbols. Our
9  // fontenc binding's `LoadDefinitions!` body only runs once per package,
10  // so the already-loaded fontenc (with [T1] options) doesn't re-process
11  // options when tipa's `\RequirePackage` re-arrives. Compensate by
12  // directly reading `t3enc.def` here. Driver: arXiv:1802.05444.
13  InputDefinitions!("t3enc", extension => Some(Cow::Borrowed("def")));
14});
