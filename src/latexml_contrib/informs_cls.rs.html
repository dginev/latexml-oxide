1//! Stub for INFORMS journal classes (informs, informs3).
2//!
3//! The informs* classes are used by Operations Research, Management Science,
4//! and other INFORMS journals. They define a large frontmatter API
5//! (\TITLE, \ARTICLEAUTHORS, \ABSTRACT, \KEYWORDS, etc.) inside the cls,
6//! which we never raw-load. Provide gobble stubs so papers using this
7//! class convert without "undefined" cascades.
8use latexml_package::prelude::*;
9
10LoadDefinitions!({
11  LoadClass!("OmniBus");
12  RequirePackage!("amsmath");
13  RequirePackage!("amsthm");
14
15  // Frontmatter / paper metadata — preserve author content.
16  // \TITLE / \ARTICLETITLE → \title to populate the document title.
17  DefMacro!("\\TITLE{}", "\\title{#1}");
18  DefMacro!("\\ARTICLETITLE{}", "\\title{#1}");
19  // Running header variants — short title for header; preserve as note.
20  DefMacro!(
21    "\\RUNAUTHOR{}",
22    "\\@add@frontmatter{ltx:note}[role=runningauthor]{#1}"
23  );
24  DefMacro!(
25    "\\RUNTITLE{}",
26    "\\@add@frontmatter{ltx:note}[role=runningtitle]{#1}"
27  );
28  DefMacro!(
29    "\\ECRUNAUTHOR{}",
30    "\\@add@frontmatter{ltx:note}[role=ec-runningauthor]{#1}"
31  );
32  DefMacro!(
33    "\\ECRUNTITLE{}",
34    "\\@add@frontmatter{ltx:note}[role=ec-runningtitle]{#1}"
35  );
36  // \AUTHOR{name}{affiliation} — emit name as author, affiliation as note.
37  DefMacro!(
38    "\\AUTHOR{}{}",
39    "\\author{#1}\\@add@frontmatter{ltx:note}[role=affiliation]{#2}"
40  );
41  DefMacro!(
42    "\\AFF[]{}",
43    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}"
44  );
45  // \ABSTRACT → abstract env so the text is preserved as document abstract.
46  DefMacro!("\\ABSTRACT{}", "\\begin{abstract}#1\\end{abstract}");
47  DefMacro!("\\ARTICLEABSTRACT{}", "\\begin{abstract}#1\\end{abstract}");
48  DefMacro!(
49    "\\KEYWORDS{}",
50    "\\@add@frontmatter{ltx:classification}[scheme=keywords]{#1}"
51  );
52  DefMacro!(
53    "\\MANUSCRIPTNO{}",
54    "\\@add@frontmatter{ltx:note}[role=manuscriptno]{#1}"
55  );
56  DefMacro!(
57    "\\HISTORY{}",
58    "\\@add@frontmatter{ltx:note}[role=history]{#1}"
59  );
60  DefMacro!(
61    "\\ARTICLEAUTHORS{}",
62    "\\@add@frontmatter{ltx:note}[role=authors]{#1}"
63  );
64  DefMacro!(
65    "\\authorinfo{}",
66    "\\@add@frontmatter{ltx:note}[role=authorinfo]{#1}"
67  );
68  def_macro_noop("\\thetitle")?;
69
70  // Layout / structure switches — no visual effect in XML.
71  def_macro_noop("\\TheoremsNumberedThrough")?;
72  def_macro_noop("\\TheoremsNumberedBySection")?;
73  def_macro_noop("\\EquationsNumberedThrough")?;
74  def_macro_noop("\\EquationsNumberedBySection")?;
75  def_macro_noop("\\ECRepeatTheorems")?;
76  def_macro_noop("\\OneAndAHalfSpacedXI")?;
77  def_macro_noop("\\OneAndAHalfSpacedXII")?;
78  def_macro_noop("\\DoubleSpacedXI")?;
79  def_macro_noop("\\DoubleSpacedXII")?;
80  def_macro_noop("\\SingleSpacedXI")?;
81
82  // {APPENDICES} env — render contents as appendix section.
83  DefMacro!(T_CS!("\\begin{APPENDICES}"), None, "\\appendix");
84  DefMacro!(T_CS!("\\end{APPENDICES}"), None, "");
85
86  // informs3.cls L932: `\def\Halmos{\mbox{\quad$\square$}}` — proof-end
87  // QED box. Render the square in math mode.
88  DefMacro!("\\Halmos", "\\ensuremath{\\square}");
89  // informs3.cls L1231: `\def\EMAIL#1{#1}` — used within \AFF; plain
90  // passthrough of the email text.
91  DefMacro!("\\EMAIL{}", "#1");
92  // informs3.cls L1273: `\long\def\ACKNOWLEDGMENT#1{\section*{\bf
93  // \theACKname.}{#1}}` (\theACKname defaults to "Acknowledgments").
94  // Route the body to a structural acknowledgements block (see
95  // feedback: prefer ltx:acknowledgements over a flattened \section*).
96  DefConstructor!(
97    "\\ACKNOWLEDGMENT{}",
98    "<ltx:acknowledgements name='Acknowledgments'>#1</ltx:acknowledgements>"
99  );
100  // \ACKname{name} sets the acknowledgements heading name.
101  def_macro_noop("\\ACKname{}")?;
102  // informs3.cls L2642: `\newenvironment{APPENDIX}[1]{…appendix with
103  // title #1…}`. The singular env wraps a single titled appendix.
104  // Begin enters appendix mode + emits the title as a section; the arg
105  // is read by the helper. Mirrors the {APPENDICES} handling.
106  DefMacro!(
107    T_CS!("\\begin{APPENDIX}"),
108    None,
109    "\\appendix\\lx@informs@appendixhead"
110  );
111  DefMacro!("\\lx@informs@appendixhead{}", "\\section{#1}");
112  DefMacro!(T_CS!("\\end{APPENDIX}"), None, "");
113});
