1//! Stub for INFORMS journal classes (informs, informs3).
2//!
3//! The informs* classes are used by Operations Research, Management Science,
4//! and other INFORMS journals. They define a large frontmatter API
5//! (\TITLE, \ARTICLEAUTHORS, \ABSTRACT, \KEYWORDS, etc.) inside the cls,
6//! which we never raw-load. Provide gobble stubs so papers using this
7//! class convert without "undefined" cascades.
8use latexml_package::prelude::*;
9
10
11LoadDefinitions!({
12  LoadClass!("OmniBus");
13  RequirePackage!("amsmath");
14  RequirePackage!("amsthm");
15
16  // Frontmatter / paper metadata — preserve author content.
17  // \TITLE / \ARTICLETITLE → \title to populate the document title.
18  DefMacro!("\\TITLE{}", "\\title{#1}");
19  DefMacro!("\\ARTICLETITLE{}", "\\title{#1}");
20  // Running header variants — short title for header; preserve as note.
21  DefMacro!("\\RUNAUTHOR{}",
22    "\\@add@frontmatter{ltx:note}[role=runningauthor]{#1}");
23  DefMacro!("\\RUNTITLE{}",
24    "\\@add@frontmatter{ltx:note}[role=runningtitle]{#1}");
25  DefMacro!("\\ECRUNAUTHOR{}",
26    "\\@add@frontmatter{ltx:note}[role=ec-runningauthor]{#1}");
27  DefMacro!("\\ECRUNTITLE{}",
28    "\\@add@frontmatter{ltx:note}[role=ec-runningtitle]{#1}");
29  // \AUTHOR{name}{affiliation} — emit name as author, affiliation as note.
30  DefMacro!("\\AUTHOR{}{}",
31    "\\author{#1}\\@add@frontmatter{ltx:note}[role=affiliation]{#2}");
32  DefMacro!("\\AFF[]{}",
33    "\\@add@frontmatter{ltx:note}[role=affiliation]{#2}");
34  // \ABSTRACT → abstract env so the text is preserved as document abstract.
35  DefMacro!("\\ABSTRACT{}",
36    "\\begin{abstract}#1\\end{abstract}");
37  DefMacro!("\\ARTICLEABSTRACT{}",
38    "\\begin{abstract}#1\\end{abstract}");
39  DefMacro!("\\KEYWORDS{}",
40    "\\@add@frontmatter{ltx:classification}[scheme=keywords]{#1}");
41  DefMacro!("\\MANUSCRIPTNO{}",
42    "\\@add@frontmatter{ltx:note}[role=manuscriptno]{#1}");
43  DefMacro!("\\HISTORY{}",
44    "\\@add@frontmatter{ltx:note}[role=history]{#1}");
45  DefMacro!("\\ARTICLEAUTHORS{}",
46    "\\@add@frontmatter{ltx:note}[role=authors]{#1}");
47  DefMacro!("\\authorinfo{}",
48    "\\@add@frontmatter{ltx:note}[role=authorinfo]{#1}");
49  def_macro_noop("\\thetitle")?;
50
51  // Layout / structure switches — no visual effect in XML.
52  def_macro_noop("\\TheoremsNumberedThrough")?;
53  def_macro_noop("\\TheoremsNumberedBySection")?;
54  def_macro_noop("\\EquationsNumberedThrough")?;
55  def_macro_noop("\\EquationsNumberedBySection")?;
56  def_macro_noop("\\ECRepeatTheorems")?;
57  def_macro_noop("\\OneAndAHalfSpacedXI")?;
58  def_macro_noop("\\OneAndAHalfSpacedXII")?;
59  def_macro_noop("\\DoubleSpacedXI")?;
60  def_macro_noop("\\DoubleSpacedXII")?;
61  def_macro_noop("\\SingleSpacedXI")?;
62
63  // {APPENDICES} env — render contents as appendix section.
64  DefMacro!(T_CS!("\\begin{APPENDICES}"), None, "\\appendix");
65  DefMacro!(T_CS!("\\end{APPENDICES}"), None, "");
66
67  // informs3.cls L932: `\def\Halmos{\mbox{\quad$\square$}}` — proof-end
68  // QED box. Render the square in math mode.
69  DefMacro!("\\Halmos", "\\ensuremath{\\square}");
70  // informs3.cls L1231: `\def\EMAIL#1{#1}` — used within \AFF; plain
71  // passthrough of the email text.
72  DefMacro!("\\EMAIL{}", "#1");
73  // informs3.cls L1273: `\long\def\ACKNOWLEDGMENT#1{\section*{\bf
74  // \theACKname.}{#1}}` (\theACKname defaults to "Acknowledgments").
75  // Route the body to a structural acknowledgements block (see
76  // feedback: prefer ltx:acknowledgements over a flattened \section*).
77  DefConstructor!("\\ACKNOWLEDGMENT{}",
78    "<ltx:acknowledgements name='Acknowledgments'>#1</ltx:acknowledgements>");
79  // \ACKname{name} sets the acknowledgements heading name.
80  def_macro_noop("\\ACKname{}")?;
81  // informs3.cls L2642: `\newenvironment{APPENDIX}[1]{…appendix with
82  // title #1…}`. The singular env wraps a single titled appendix.
83  // Begin enters appendix mode + emits the title as a section; the arg
84  // is read by the helper. Mirrors the {APPENDICES} handling.
85  DefMacro!(T_CS!("\\begin{APPENDIX}"), None, "\\appendix\\lx@informs@appendixhead");
86  DefMacro!("\\lx@informs@appendixhead{}", "\\section{#1}");
87  DefMacro!(T_CS!("\\end{APPENDIX}"), None, "");
88});
