14  def_macro_noop("\\newmdtheoremenv[]{}{}[]")?;
15  def_macro_noop("\\newmdenv[]{}")?;
16  def_macro_noop("\\renewmdenv[]{}")?;
17  def_macro_noop("\\surroundwithmdframed[]{}")?;
18  def_macro_noop("\\mdfsubtitle[]{}")?;
19  def_macro_noop("\\mdfapptodefinestyle{}{}")?;
20  def_macro_noop("\\mdfsetup{}")?;

22  DefRegister!("\\mdflength" => Dimension::new(0));
23  // Wrap body in `inline-logical-block` (Misc.class container that
24  // accepts Para.model body).
25  //
26  // Rust-only surpass-Perl divergence: Perl ar5iv-bindings/mdframed.sty.ltxml
27  // L31-34 uses `inline-block` (Block.model only), which the schema rejects
28  // when an `mdframed` body contains a `\begin{theorem}` (theorem lives in
29  // Para.class, not Block.class). `inline-logical-block` is the strictly
30  // safer swap:
31  //   * Same `Misc.class` membership as `inline-block` — accepted in every
32  //     parent context where Perl's choice fits (inline AND block). The
33  //     alternative `logical-block` is in `Para.class` and would BREAK
34  //     inline-context uses of mdframed (`\fbox{\begin{mdframed}…}` etc.).
35  //   * Same Backgroundable.attributes surface (`framed`, `framecolor`,
36  //     `backgroundcolor`).
37  //   * Same `display: inline-block` CSS in LaTeXML.css (no visual change).
38  //   * `Para.model` body — accepts theorem/proof/para inside.
39  //
40  // The template emits `framecolor=` only when the #framecolor property is
41  // set (via the `?#framecolor(...)` guard), so an unset color correctly
42  // omits the attribute rather than emitting `framecolor=''`. Driver:
43  // arXiv:2506.03074v1 (ICML 2025 paper with
44  // `\begin{mdframed}\begin{theorem}…\end{theorem}\end{mdframed}`).
45  DefEnvironment!(
46    "{mdframed}[]",
47    "<ltx:inline-logical-block framed='rectangle' ?#framecolor(framecolor='#framecolor') _noautoclose='1'>#body</ltx:inline-logical-block>",
48    properties => sub[_args] {
49      let mut props = arena::SymHashMap::default();
50      if let Some(font) = latexml_core::state::lookup_font() {
51        if let Some(color) = font.get_color() {
52          props.insert("framecolor", Stored::from(color.to_attribute()));
53        }
54      }
55      Ok(props)
56    },
57    // mdframed bodies routinely contain multi-paragraph content
58    // (theorems, displayed equations, multiple `$$..$$` blocks). The
59    // DefEnvironment default of restricted_horizontal makes
60    // BOUND_MODE never end with "vertical", so tex_math.rs:467's
61    // `$$` → display-math check stays false: each `$$` is parsed as
62    // open + immediate close, leaving body content in text mode and
63    // cascading "Script _/^ can only appear in math mode" on subscripts.
64    // Witness 2402.07712 (eqnarray + multiple `$$..$$` in mdframed).
65    mode => "internal_vertical");
66});
