12  // ar5iv-bindings/bindings/arydshln.sty.ltxml L21-24: ':' column type adds a
13  // dashed right-border marker via the between-column slot. The \vrule gets
14  // decorated by \@ADDCLASS{ltx_border_r_dashed}\relax so CSS renders a
15  // dashed vertical rule.
16  DefColumnType!(":", {
17    with_current_build_template(|template_opt| {
18      template_opt.unwrap().add_between_column(vec![
19        T_CS!("\\vrule"),
20        T_CS!("\\@ADDCLASS"),
21        T_BEGIN!(),
22        T_OTHER!("ltx_border_r_dashed"),
23        T_END!(),
24        T_CS!("\\relax"),
25      ]);
26    });
27  });
28  Let!("\\firsthdashline", "\\firsthline");
29  Let!("\\lasthdashline", "\\lasthline");
30  DefRegister!("\\dashlinedash" => Dimension!("4pt"));
31  DefRegister!("\\dashlinegap" => Dimension!("4pt"));
32  Let!("\\hdashlinewidth", "\\dashlinedash");
33  Let!("\\hdashlinegap", "\\dashlinegap");
34  def_macro_noop("\\ADLactivate")?;
35  def_macro_noop("\\ADLdrawingmode")?;
36  def_macro_noop("\\ADLinactivate")?;
37  def_macro_noop("\\ADLnoshorthanded")?;
38  def_macro_noop("\\ADLnullwide")?;
39  def_macro_noop("\\ADLnullwidehline")?;
40  def_macro_noop("\\ADLsomewide")?;
41  def_macro_noop("\\ADLsomewidehline")?;
42  // `\arrayrulecolor[model]{color}` and `\doublerulesepcolor[model]{color}` are
43  // colortbl's commands (colortbl.sty.ltxml L85/L88: `[]{}`). The ar5iv arydshln
44  // binding declares them 0-arg (`DefMacro('\arrayrulecolor', Tokens())`) — which
45  // is order-fragile: it only behaves when colortbl loads AFTER arydshln (so
46  // colortbl's `[]{}` overrides). When colortbl loads FIRST (e.g. pulled in
47  // ahead of arydshln via the package dep graph — witness 1909.02323, mnras +
48  // arydshln + colortbl), arydshln's 0-arg form is the final meaning and
49  // `\arrayrulecolor{gray}\hline` leaves `{gray}` behind as a cell, so the
50  // following `\hline` (= `\noalign{…}`) fires "\noalign cannot be used here".
51  // Declare them with colortbl's `[]{}` arity so the color argument is consumed
52  // regardless of load order — matching Perl's *effective* (colortbl-wins) result.
53  DefMacro!("\\arrayrulecolor[]{}", None);
54  def_macro_noop("\\dashgapcolor{}")?;
55  DefMacro!("\\doublerulesepcolor[]{}", None);
56  // NOTE: do NOT noop `\endlongtable`. The ar5iv Perl binding
57  // (arydshln.sty.ltxml L45) does `DefMacro('\endlongtable', Tokens())`, but
58  // that diverges from the REAL arydshln.sty, which SAVES and RESTORES
59  // longtable's original `\endlongtable` (`\let\endlongtable\adl@org@endlongtable`,
60  // arydshln.sty L796) rather than neutralizing it. Our longtable binding
61  // relies on `\endlongtable` = `\lx@end@alignment\@end@tabular` to close the
62  // alignment's boxing group; noop'ing it leaks that `{`-group so the
63  // environment's `\endgroup` mismatches → mode cascade → `pop last locked
64  // stack frame` FATAL (1510.04473: any `arydshln` + `longtable` with `p{}`
65  // columns). Perl recovers from the same mismatch with 9 errors; our engine
66  // aborts. Keeping longtable's `\endlongtable` functional matches the real
67  // package and produces clean output (0 errors).
68  def_macro_noop("\\nodashgapcolor")?;
69  def_macro_noop("\\xleaders")?;
70});
