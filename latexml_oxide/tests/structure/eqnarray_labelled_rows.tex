% dead code in Perl and the rows collapse onto one number. We honor the intent
% (matching pdfTeX): the three constraint rows below each carry their own
% \label with no \nonumber, so they must remain FOUR separately numbered
% equations (a, b, c, d) -- not collapse to (a) and (d).

& & b(x) \leq 0 \label{eq:cb}\\
& & c(x) = 0 \label{eq:cc}
\end{eqnarray}
\end{subequations}
